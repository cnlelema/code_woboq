140bool LLVMTableGenMain(raw_ostream &OS, RecordKeeper &Records) {
141 switch (Action) {
142 case PrintRecords:
143 OS << Records; // No argument, dump all contents
144 break;
145 case DumpJSON:
146 EmitJSON(Records, OS);
147 break;
148 case GenEmitter:
149 EmitCodeEmitter(Records, OS);
150 break;
151 case GenRegisterInfo:
152 EmitRegisterInfo(Records, OS);
153 break;
154 case GenInstrInfo:
155 EmitInstrInfo(Records, OS);
156 break;
157 case GenInstrDocs:
158 EmitInstrDocs(Records, OS);
159 break;
160 case GenCallingConv:
161 EmitCallingConv(Records, OS);
162 break;
163 case GenAsmWriter:
164 EmitAsmWriter(Records, OS);
165 break;
166 case GenAsmMatcher:
167 EmitAsmMatcher(Records, OS);
168 break;
169 case GenDisassembler:
170 EmitDisassembler(Records, OS);
171 break;
172 case GenPseudoLowering:
173 EmitPseudoLowering(Records, OS);
174 break;
175 case GenCompressInst:
176 EmitCompressInst(Records, OS);
177 break;
178 case GenDAGISel:
179 EmitDAGISel(Records, OS);
180 break;
181 case GenDFAPacketizer:
182 EmitDFAPacketizer(Records, OS);
183 break;
184 case GenFastISel:
185 EmitFastISel(Records, OS);
186 break;
187 case GenSubtarget:
188 EmitSubtarget(Records, OS);
189 break;
190 case GenIntrinsicEnums:
191 EmitIntrinsicEnums(Records, OS);
192 break;
193 case GenIntrinsicImpl:
194 EmitIntrinsicImpl(Records, OS);
195 break;
196 case GenTgtIntrinsicEnums:
197 EmitIntrinsicEnums(Records, OS, true);
198 break;
199 case GenTgtIntrinsicImpl:
200 EmitIntrinsicImpl(Records, OS, true);
201 break;
202 case GenOptParserDefs:
203 EmitOptParser(Records, OS);
204 break;
205 case PrintEnums:
206 {
207 for (Record *Rec : Records.getAllDerivedDefinitions(Class))
208 OS << Rec->getName() << ", ";
209 OS << "\n";
210 break;
211 }
212 case PrintSets:
213 {
214 SetTheory Sets;
215 Sets.addFieldExpander("Set", "Elements");
216 for (Record *Rec : Records.getAllDerivedDefinitions("Set")) {
217 OS << Rec->getName() << " = [";
218 const std::vector<Record*> *Elts = Sets.expand(Rec);
219 assert(Elts && "Couldn't expand Set instance");
220 for (Record *Elt : *Elts)
221 OS << ' ' << Elt->getName();
222 OS << " ]\n";
223 }
224 break;
225 }
226 case GenCTags:
227 EmitCTags(Records, OS);
228 break;
229 case GenAttributes:
230 EmitAttributes(Records, OS);
231 break;
232 case GenSearchableTables:
233 EmitSearchableTables(Records, OS);
234 break;
235 case GenGlobalISel:
236 EmitGlobalISel(Records, OS);
237 break;
238 case GenRegisterBank:
239 EmitRegisterBank(Records, OS);
240 break;
241 case GenX86EVEX2VEXTables:
242 EmitX86EVEX2VEXTables(Records, OS);
243 break;
244 case GenX86FoldTables:
245 EmitX86FoldTables(Records, OS);
246 break;
247 case GenExegesis:
248 EmitExegesis(Records, OS);
249 break;
250 }
251
252 return false;
253}
254}

256int main(int argc, char **argv) {
257 sys::PrintStackTraceOnErrorSignal(argv[0]);
258 PrettyStackTraceProgram X(argc, argv);
259 cl::ParseCommandLineOptions(argc, argv);
260
261 llvm_shutdown_obj Y;
262
263 return TableGenMain(argv[0], &LLVMTableGenMain);
264}
