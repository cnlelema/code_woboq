304std::vector<ValueTypeByHwMode> CodeGenTarget::getRegisterVTs(Record *R)
305 const {
306 const CodeGenRegister *Reg = getRegBank().getReg(R);
307 std::vector<ValueTypeByHwMode> Result;
308 for (const auto &RC : getRegBank().getRegClasses()) {
309 if (RC.contains(Reg)) {
310 ArrayRef<ValueTypeByHwMode> InVTs = RC.getValueTypes();
311 Result.insert(Result.end(), InVTs.begin(), InVTs.end());
312 }
313 }
314
315 // Remove duplicates.
316 llvm::sort(Result);
317 Result.erase(std::unique(Result.begin(), Result.end()), Result.end());
318 return Result;
319}

339void CodeGenTarget::ReadInstructions() const {
340 NamedRegionTimer T("Read Instructions", "Time spent reading instructions",
341 "CodeGenTarget", "CodeGenTarget", TimeRegions);
342 std::vector<Record*> Insts = Records.getAllDerivedDefinitions("Instruction");
343 if (Insts.size() <= 2)
344 PrintFatalError("No 'Instruction' subclasses defined!");
345
346 // Parse the instructions defined in the .td file.
347 for (unsigned i = 0, e = Insts.size(); i != e; ++i)
348 Instructions[Insts[i]] = llvm::make_unique<CodeGenInstruction>(Insts[i]);
349}

351static const CodeGenInstruction *
352GetInstByName(const char *Name,
353 const DenseMap<const Record*,
354 std::unique_ptr<CodeGenInstruction>> &Insts,
355 RecordKeeper &Records) {
356 const Record *Rec = Records.getDef(Name);
357
358 const auto I = Insts.find(Rec);
359 if (!Rec || I == Insts.end())
360 PrintFatalError(Twine("Could not find '") + Name + "' instruction!");
361 return I->second.get();
362}

373/// Return all of the instructions defined by the target, ordered by
374/// their enum value.
375void CodeGenTarget::ComputeInstrsByEnum() const {
376 const auto &Insts = getInstructions();
377 for (const char *const *p = FixedInstrs; *p; ++p) {
378 const CodeGenInstruction *Instr = GetInstByName(*p, Insts, Records);
379 assert(Instr && "Missing target independent instruction");
380 assert(Instr->Namespace == "TargetOpcode" && "Bad namespace");
381 InstrsByEnum.push_back(Instr);
382 }
383 unsigned EndOfPredefines = InstrsByEnum.size();
384 assert(EndOfPredefines == getNumFixedInstructions() &&
385 "Missing generic opcode");
386
387 for (const auto &I : Insts) {
388 const CodeGenInstruction *CGI = I.second.get();
389 if (CGI->Namespace != "TargetOpcode") {
390 InstrsByEnum.push_back(CGI);
391 if (CGI->TheDef->getValueAsBit("isPseudo"))
392 ++NumPseudoInstructions;
393 }
394 }
395
396 assert(InstrsByEnum.size() == Insts.size() && "Missing predefined instr");
397
398 // All of the instructions are now in random order based on the map iteration.
399 llvm::sort(
400 InstrsByEnum.begin() + EndOfPredefines, InstrsByEnum.end(),
401 [](const CodeGenInstruction *Rec1, const CodeGenInstruction *Rec2) {
402 const auto &D1 = *Rec1->TheDef;
403 const auto &D2 = *Rec2->TheDef;
404 return std::make_tuple(!D1.getValueAsBit("isPseudo"), D1.getName()) <
405 std::make_tuple(!D2.getValueAsBit("isPseudo"), D2.getName());
406 });
407}

417/// reverseBitsForLittleEndianEncoding - For little-endian instruction bit
418/// encodings, reverse the bit order of all instructions.
419void CodeGenTarget::reverseBitsForLittleEndianEncoding() {
420 if (!isLittleEndianEncoding())
421 return;
422
423 std::vector<Record*> Insts = Records.getAllDerivedDefinitions("Instruction");
424 for (Record *R : Insts) {
425 if (R->getValueAsString("Namespace") == "TargetOpcode" ||
426 R->getValueAsBit("isPseudo"))
427 continue;
428
429 BitsInit *BI = R->getValueAsBitsInit("Inst");
430
431 unsigned numBits = BI->getNumBits();
432
433 SmallVector<Init *, 16> NewBits(numBits);
434
435 for (unsigned bit = 0, end = numBits / 2; bit != end; ++bit) {
436 unsigned bitSwapIdx = numBits - bit - 1;
437 Init *OrigBit = BI->getBit(bit);
438 Init *BitSwap = BI->getBit(bitSwapIdx);
439 NewBits[bit] = BitSwap;
440 NewBits[bitSwapIdx] = OrigBit;
441 }
442 if (numBits % 2) {
443 unsigned middle = (numBits + 1) / 2;
444 NewBits[middle] = BI->getBit(middle);
445 }
446
447 BitsInit *NewBI = BitsInit::get(NewBits);
448
449 // Update the bits in reversed order so that emitInstrOpBits will get the
450 // correct endianness.
451 R->getValue("Inst")->setValue(NewBI);
452 }
453}

465// ComplexPattern implementation
466//
467ComplexPattern::ComplexPattern(Record *R) {
468 Ty = ::getValueType(R->getValueAsDef("Ty"));
469 NumOperands = R->getValueAsInt("NumOperands");
470 SelectFunc = R->getValueAsString("SelectFunc");
471 RootNodes = R->getValueAsListOfDefs("RootNodes");
472
473 // FIXME: This is a hack to statically increase the priority of patterns which
474 // maps a sub-dag to a complex pattern. e.g. favors LEA over ADD. To get best
475 // possible pattern match we'll need to dynamically calculate the complexity
476 // of all patterns a dag can potentially map to.
477 int64_t RawComplexity = R->getValueAsInt("Complexity");
478 if (RawComplexity == -1)
479 Complexity = NumOperands * 3;
480 else
481 Complexity = RawComplexity;
482
483 // FIXME: Why is this different from parseSDPatternOperatorProperties?
484 // Parse the properties.
485 Properties = 0;
486 std::vector<Record*> PropList = R->getValueAsListOfDefs("Properties");
487 for (unsigned i = 0, e = PropList.size(); i != e; ++i)
488 if (PropList[i]->getName() == "SDNPHasChain") {
489 Properties |= 1 << SDNPHasChain;
490 } else if (PropList[i]->getName() == "SDNPOptInGlue") {
491 Properties |= 1 << SDNPOptInGlue;
492 } else if (PropList[i]->getName() == "SDNPMayStore") {
493 Properties |= 1 << SDNPMayStore;
494 } else if (PropList[i]->getName() == "SDNPMayLoad") {
495 Properties |= 1 << SDNPMayLoad;
496 } else if (PropList[i]->getName() == "SDNPSideEffect") {
497 Properties |= 1 << SDNPSideEffect;
498 } else if (PropList[i]->getName() == "SDNPMemOperand") {
499 Properties |= 1 << SDNPMemOperand;
500 } else if (PropList[i]->getName() == "SDNPVariadic") {
501 Properties |= 1 << SDNPVariadic;
502 } else if (PropList[i]->getName() == "SDNPWantRoot") {
503 Properties |= 1 << SDNPWantRoot;
504 } else if (PropList[i]->getName() == "SDNPWantParent") {
505 Properties |= 1 << SDNPWantParent;
506 } else {
507 PrintFatalError(R->getLoc(), "Unsupported SD Node property '" +
508 PropList[i]->getName() +
509 "' on ComplexPattern '" + R->getName() +
510 "'!");
511 }
512}

517
518CodeGenIntrinsicTable::CodeGenIntrinsicTable(const RecordKeeper &RC,
519 bool TargetOnly) {
520 std::vector<Record*> Defs = RC.getAllDerivedDefinitions("Intrinsic");
521
522 Intrinsics.reserve(Defs.size());
523
524 for (unsigned I = 0, e = Defs.size(); I != e; ++I) {
525 bool isTarget = Defs[I]->getValueAsBit("isTarget");
526 if (isTarget == TargetOnly)
527 Intrinsics.push_back(CodeGenIntrinsic(Defs[I]));
528 }
529 llvm::sort(Intrinsics,
530 [](const CodeGenIntrinsic &LHS, const CodeGenIntrinsic &RHS) {
531 return std::tie(LHS.TargetPrefix, LHS.Name) <
532 std::tie(RHS.TargetPrefix, RHS.Name);
533 });
534 Targets.push_back({"", 0, 0});
535 for (size_t I = 0, E = Intrinsics.size(); I < E; ++I)
536 if (Intrinsics[I].TargetPrefix != Targets.back().Name) {
537 Targets.back().Count = I - Targets.back().Offset;
538 Targets.push_back({Intrinsics[I].TargetPrefix, I, 0});
539 }
540 Targets.back().Count = Intrinsics.size() - Targets.back().Offset;
541}

543CodeGenIntrinsic::CodeGenIntrinsic(Record *R) {
544 TheDef = R;
545 std::string DefName = R->getName();
546 ArrayRef<SMLoc> DefLoc = R->getLoc();
547 ModRef = ReadWriteMem;
548 Properties = 0;
549 isOverloaded = false;
550 isCommutative = false;
551 canThrow = false;
552 isNoReturn = false;
553 isCold = false;
554 isNoDuplicate = false;
555 isConvergent = false;
556 isSpeculatable = false;
557 hasSideEffects = false;
558
559 if (DefName.size() <= 4 ||
560 std::string(DefName.begin(), DefName.begin() + 4) != "int_")
561 PrintFatalError(DefLoc,
562 "Intrinsic '" + DefName + "' does not start with 'int_'!");
563
564 EnumName = std::string(DefName.begin()+4, DefName.end());
565
566 if (R->getValue("GCCBuiltinName")) // Ignore a missing GCCBuiltinName field.
567 GCCBuiltinName = R->getValueAsString("GCCBuiltinName");
568 if (R->getValue("GCCBuiltinAliasName")) // Ignore a missing GCCBuiltinAliasName field.
569 GCCBuiltinAliasName = R->getValueAsString("GCCBuiltinAliasName");
570 if (R->getValue("MSBuiltinName")) // Ignore a missing MSBuiltinName field.
571 MSBuiltinName = R->getValueAsString("MSBuiltinName");
572
573 TargetPrefix = R->getValueAsString("TargetPrefix");
574 Name = R->getValueAsString("LLVMName");
575
576 if (Name == "") {
577 // If an explicit name isn't specified, derive one from the DefName.
578 Name = "llvm.";
579
580 for (unsigned i = 0, e = EnumName.size(); i != e; ++i)
581 Name += (EnumName[i] == '_') ? '.' : EnumName[i];
582 } else {
583 // Verify it starts with "llvm.".
584 if (Name.size() <= 5 ||
585 std::string(Name.begin(), Name.begin() + 5) != "llvm.")
586 PrintFatalError(DefLoc, "Intrinsic '" + DefName +
587 "'s name does not start with 'llvm.'!");
588 }
589
590 // If TargetPrefix is specified, make sure that Name starts with
591 // "llvm.<targetprefix>.".
592 if (!TargetPrefix.empty()) {
593 if (Name.size() < 6+TargetPrefix.size() ||
594 std::string(Name.begin() + 5, Name.begin() + 6 + TargetPrefix.size())
595 != (TargetPrefix + "."))
596 PrintFatalError(DefLoc, "Intrinsic '" + DefName +
597 "' does not start with 'llvm." +
598 TargetPrefix + ".'!");
599 }
600
601 ListInit *RetTypes = R->getValueAsListInit("RetTypes");
602 ListInit *ParamTypes = R->getValueAsListInit("ParamTypes");
603
604 // First collate a list of overloaded types.
605 std::vector<MVT::SimpleValueType> OverloadedVTs;
606 for (ListInit *TypeList : {RetTypes, ParamTypes}) {
607 for (unsigned i = 0, e = TypeList->size(); i != e; ++i) {
608 Record *TyEl = TypeList->getElementAsRecord(i);
609 assert(TyEl->isSubClassOf("LLVMType") && "Expected a type!");
610
611 if (TyEl->isSubClassOf("LLVMMatchType"))
612 continue;
613
614 MVT::SimpleValueType VT = getValueType(TyEl->getValueAsDef("VT"));
615 // iFATPTRAny is overloaded from the perspective of the back end (it
616 // becomes one of the fixed-sized iFATPTR types), but it is not overloaded
617 // from the perspective of the IR, where it is (currently, at least) always
618 // an address-space-200 pointer.
619 if (MVT(VT).isOverloaded() && (MVT(VT) != MVT::iFATPTRAny)) {
620 OverloadedVTs.push_back(VT);
621 isOverloaded = true;
622 }
623 }
624 }
625
626 // Parse the list of return types.
627 ListInit *TypeList = RetTypes;
628 for (unsigned i = 0, e = TypeList->size(); i != e; ++i) {
629 Record *TyEl = TypeList->getElementAsRecord(i);
630 assert(TyEl->isSubClassOf("LLVMType") && "Expected a type!");
631 MVT::SimpleValueType VT;
632 if (TyEl->isSubClassOf("LLVMMatchType")) {
633 unsigned MatchTy = TyEl->getValueAsInt("Number");
634 assert(MatchTy < OverloadedVTs.size() &&
635 "Invalid matching number!");
636 VT = OverloadedVTs[MatchTy];
637 // It only makes sense to use the extended and truncated vector element
638 // variants with iAny types; otherwise, if the intrinsic is not
639 // overloaded, all the types can be specified directly.
640 assert(((!TyEl->isSubClassOf("LLVMExtendedType") &&
641 !TyEl->isSubClassOf("LLVMTruncatedType")) ||
642 VT == MVT::iAny || VT == MVT::vAny) &&
643 "Expected iAny or vAny type");
644 } else {
645 VT = getValueType(TyEl->getValueAsDef("VT"));
646 }
647
648 // Reject invalid types.
649 if (VT == MVT::isVoid)
650 PrintFatalError(DefLoc, "Intrinsic '" + DefName +
651 " has void in result type list!");
652
653 IS.RetVTs.push_back(VT);
654 IS.RetTypeDefs.push_back(TyEl);
655 }
656
657 // Parse the list of parameter types.
658 TypeList = ParamTypes;
659 for (unsigned i = 0, e = TypeList->size(); i != e; ++i) {
660 Record *TyEl = TypeList->getElementAsRecord(i);
661 assert(TyEl->isSubClassOf("LLVMType") && "Expected a type!");
662 MVT::SimpleValueType VT;
663 if (TyEl->isSubClassOf("LLVMMatchType")) {
664 unsigned MatchTy = TyEl->getValueAsInt("Number");
665 if (MatchTy >= OverloadedVTs.size()) {
666 PrintError(R->getLoc(),
667 "Parameter #" + Twine(i) + " has out of bounds matching "
668 "number " + Twine(MatchTy));
669 PrintFatalError(DefLoc,
670 Twine("ParamTypes is ") + TypeList->getAsString());
671 }
672 VT = OverloadedVTs[MatchTy];
673 // It only makes sense to use the extended and truncated vector element
674 // variants with iAny types; otherwise, if the intrinsic is not
675 // overloaded, all the types can be specified directly.
676 assert(((!TyEl->isSubClassOf("LLVMExtendedType") &&
677 !TyEl->isSubClassOf("LLVMTruncatedType") &&
678 !TyEl->isSubClassOf("LLVMScalarOrSameVectorWidth")) ||
679 VT == MVT::iAny || VT == MVT::vAny) &&
680 "Expected iAny or vAny type");
681 } else
682 VT = getValueType(TyEl->getValueAsDef("VT"));
683
684 // Reject invalid types.
685 if (VT == MVT::isVoid && i != e-1 /*void at end means varargs*/)
686 PrintFatalError(DefLoc, "Intrinsic '" + DefName +
687 " has void in result type list!");
688
689 IS.ParamVTs.push_back(VT);
690 IS.ParamTypeDefs.push_back(TyEl);
691 }
692
693 // Parse the intrinsic properties.
694 ListInit *PropList = R->getValueAsListInit("IntrProperties");
695 for (unsigned i = 0, e = PropList->size(); i != e; ++i) {
696 Record *Property = PropList->getElementAsRecord(i);
697 assert(Property->isSubClassOf("IntrinsicProperty") &&
698 "Expected a property!");
699
700 if (Property->getName() == "IntrNoMem")
701 ModRef = NoMem;
702 else if (Property->getName() == "IntrReadMem")
703 ModRef = ModRefBehavior(ModRef & ~MR_Mod);
704 else if (Property->getName() == "IntrWriteMem")
705 ModRef = ModRefBehavior(ModRef & ~MR_Ref);
706 else if (Property->getName() == "IntrArgMemOnly")
707 ModRef = ModRefBehavior((ModRef & ~MR_Anywhere) | MR_ArgMem);
708 else if (Property->getName() == "IntrInaccessibleMemOnly")
709 ModRef = ModRefBehavior((ModRef & ~MR_Anywhere) | MR_InaccessibleMem);
710 else if (Property->getName() == "IntrInaccessibleMemOrArgMemOnly")
711 ModRef = ModRefBehavior((ModRef & ~MR_Anywhere) | MR_ArgMem |
712 MR_InaccessibleMem);
713 else if (Property->getName() == "Commutative")
714 isCommutative = true;
715 else if (Property->getName() == "Throws")
716 canThrow = true;
717 else if (Property->getName() == "IntrNoDuplicate")
718 isNoDuplicate = true;
719 else if (Property->getName() == "IntrConvergent")
720 isConvergent = true;
721 else if (Property->getName() == "IntrNoReturn")
722 isNoReturn = true;
723 else if (Property->getName() == "IntrCold")
724 isCold = true;
725 else if (Property->getName() == "IntrSpeculatable")
726 isSpeculatable = true;
727 else if (Property->getName() == "IntrHasSideEffects")
728 hasSideEffects = true;
729 else if (Property->isSubClassOf("NoCapture")) {
730 unsigned ArgNo = Property->getValueAsInt("ArgNo");
731 ArgumentAttributes.push_back(std::make_pair(ArgNo, NoCapture));
732 } else if (Property->isSubClassOf("Returned")) {
733 unsigned ArgNo = Property->getValueAsInt("ArgNo");
734 ArgumentAttributes.push_back(std::make_pair(ArgNo, Returned));
735 } else if (Property->isSubClassOf("ReadOnly")) {
736 unsigned ArgNo = Property->getValueAsInt("ArgNo");
737 ArgumentAttributes.push_back(std::make_pair(ArgNo, ReadOnly));
738 } else if (Property->isSubClassOf("WriteOnly")) {
739 unsigned ArgNo = Property->getValueAsInt("ArgNo");
740 ArgumentAttributes.push_back(std::make_pair(ArgNo, WriteOnly));
741 } else if (Property->isSubClassOf("ReadNone")) {
742 unsigned ArgNo = Property->getValueAsInt("ArgNo");
743 ArgumentAttributes.push_back(std::make_pair(ArgNo, ReadNone));
744 } else if (Property->isSubClassOf("ImmArg")) {
745 unsigned ArgNo = Property->getValueAsInt("ArgNo");
746 ArgumentAttributes.push_back(std::make_pair(ArgNo, ImmArg));
747 } else
748 llvm_unreachable("Unknown property!");
749 }
750
751 // Also record the SDPatternOperator Properties.
752 Properties = parseSDPatternOperatorProperties(R);
753
754 // Sort the argument attributes for later benefit.
755 llvm::sort(ArgumentAttributes);
756}
