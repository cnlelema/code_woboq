58void IntrinsicEmitter::run(raw_ostream &OS, bool Enums) {
59 emitSourceFileHeader("Intrinsic Function Source Fragment", OS);
60
61 CodeGenIntrinsicTable Ints(Records, TargetOnly);
62
63 if (TargetOnly && !Ints.empty())
64 TargetPrefix = Ints[0].TargetPrefix;
65
66 EmitPrefix(OS);
67
68 if (Enums) {
69 // Emit the enum information.
70 EmitEnumInfo(Ints, OS);
71 } else {
72 // Emit the target metadata.
73 EmitTargetInfo(Ints, OS);
74
75 // Emit the intrinsic ID -> name table.
76 EmitIntrinsicToNameTable(Ints, OS);
77
78 // Emit the intrinsic ID -> overload table.
79 EmitIntrinsicToOverloadTable(Ints, OS);
80
81 // Emit the intrinsic declaration generator.
82 EmitGenerator(Ints, OS);
83
84 // Emit the intrinsic parameter attributes.
85 EmitAttributes(Ints, OS);
86
87 // Emit code to translate GCC builtins into LLVM intrinsics.
88 EmitIntrinsicToBuiltinMap(Ints, true, OS);
89
90 // Emit code to translate MS builtins into LLVM intrinsics.
91 EmitIntrinsicToBuiltinMap(Ints, false, OS);
92 }
93
94 EmitSuffix(OS);
95}

97void IntrinsicEmitter::EmitPrefix(raw_ostream &OS) {
98 OS << "// VisualStudio defines setjmp as _setjmp\n"
99 "#if defined(_MSC_VER) && defined(setjmp) && \\\n"
100 " !defined(setjmp_undefined_for_msvc)\n"
101 "# pragma push_macro(\"setjmp\")\n"
102 "# undef setjmp\n"
103 "# define setjmp_undefined_for_msvc\n"
104 "#endif\n\n";
105}

107void IntrinsicEmitter::EmitSuffix(raw_ostream &OS) {
108 OS << "#if defined(_MSC_VER) && defined(setjmp_undefined_for_msvc)\n"
109 "// let's return it to _setjmp state\n"
110 "# pragma pop_macro(\"setjmp\")\n"
111 "# undef setjmp_undefined_for_msvc\n"
112 "#endif\n\n";
113}

115void IntrinsicEmitter::EmitEnumInfo(const CodeGenIntrinsicTable &Ints,
116 raw_ostream &OS) {
117 OS << "// Enum values for Intrinsics.h\n";
118 OS << "#ifdef GET_INTRINSIC_ENUM_VALUES\n";
119 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
120 OS << " " << Ints[i].EnumName;
121 OS << ((i != e-1) ? ", " : " ");
122 if (Ints[i].EnumName.size() < 40)
123 OS << std::string(40-Ints[i].EnumName.size(), ' ');
124 OS << " // " << Ints[i].Name << "\n";
125 }
126 OS << "#endif\n\n";
127}

129void IntrinsicEmitter::EmitTargetInfo(const CodeGenIntrinsicTable &Ints,
130 raw_ostream &OS) {
131 OS << "// Target mapping\n";
132 OS << "#ifdef GET_INTRINSIC_TARGET_DATA\n";
133 OS << "struct IntrinsicTargetInfo {\n"
134 << " llvm::StringLiteral Name;\n"
135 << " size_t Offset;\n"
136 << " size_t Count;\n"
137 << "};\n";
138 OS << "static constexpr IntrinsicTargetInfo TargetInfos[] = {\n";
139 for (auto Target : Ints.Targets)
140 OS << " {llvm::StringLiteral(\"" << Target.Name << "\"), " << Target.Offset
141 << ", " << Target.Count << "},\n";
142 OS << "};\n";
143 OS << "#endif\n\n";
144}

146void IntrinsicEmitter::EmitIntrinsicToNameTable(
147 const CodeGenIntrinsicTable &Ints, raw_ostream &OS) {
148 OS << "// Intrinsic ID to name table\n";
149 OS << "#ifdef GET_INTRINSIC_NAME_TABLE\n";
150 OS << " // Note that entry #0 is the invalid intrinsic!\n";
151 for (unsigned i = 0, e = Ints.size(); i != e; ++i)
152 OS << " \"" << Ints[i].Name << "\",\n";
153 OS << "#endif\n\n";
154}

156void IntrinsicEmitter::EmitIntrinsicToOverloadTable(
157 const CodeGenIntrinsicTable &Ints, raw_ostream &OS) {
158 OS << "// Intrinsic ID to overload bitset\n";
159 OS << "#ifdef GET_INTRINSIC_OVERLOAD_TABLE\n";
160 OS << "static const uint8_t OTable[] = {\n";
161 OS << " 0";
162 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
163 // Add one to the index so we emit a null bit for the invalid #0 intrinsic.
164 if ((i+1)%8 == 0)
165 OS << ",\n 0";
166 if (Ints[i].isOverloaded)
167 OS << " | (1<<" << (i+1)%8 << ')';
168 }
169 OS << "\n};\n\n";
170 // OTable contains a true bit at the position if the intrinsic is overloaded.
171 OS << "return (OTable[id/8] & (1 << (id%8))) != 0;\n";
172 OS << "#endif\n\n";
173}

231static void EncodeFixedValueType(MVT::SimpleValueType VT,
232 std::vector<unsigned char> &Sig) {
233 if (MVT(VT).isInteger()) {
234 unsigned BitWidth = MVT(VT).getSizeInBits();
235 switch (BitWidth) {
236 default: PrintFatalError("unhandled integer type width in intrinsic!");
237 case 1: return Sig.push_back(IIT_I1);
238 case 8: return Sig.push_back(IIT_I8);
239 case 16: return Sig.push_back(IIT_I16);
240 case 32: return Sig.push_back(IIT_I32);
241 case 64: return Sig.push_back(IIT_I64);
242 case 128: return Sig.push_back(IIT_I128);
243 }
244 }
245
246 switch (VT) {
247 default: PrintFatalError("unhandled MVT in intrinsic!");
248 case MVT::iFATPTR64: return Sig.push_back(IIT_IFATPTR64);
249 case MVT::iFATPTR128: return Sig.push_back(IIT_IFATPTR128);
250 case MVT::iFATPTR256: return Sig.push_back(IIT_IFATPTR256);
251 case MVT::iFATPTR512: return Sig.push_back(IIT_IFATPTR512);
252 case MVT::iFATPTRAny: return Sig.push_back(IIT_IFATPTRAny);
253 case MVT::f16: return Sig.push_back(IIT_F16);
254 case MVT::f32: return Sig.push_back(IIT_F32);
255 case MVT::f64: return Sig.push_back(IIT_F64);
256 case MVT::f128: return Sig.push_back(IIT_F128);
257 case MVT::token: return Sig.push_back(IIT_TOKEN);
258 case MVT::Metadata: return Sig.push_back(IIT_METADATA);
259 case MVT::x86mmx: return Sig.push_back(IIT_MMX);
260 // MVT::OtherVT is used to mean the empty struct type here.
261 case MVT::Other: return Sig.push_back(IIT_EMPTYSTRUCT);
262 // MVT::isVoid is used to represent varargs here.
263 case MVT::isVoid: return Sig.push_back(IIT_VARARG);
264 }
265}

271static void EncodeFixedType(Record *R, std::vector<unsigned char> &ArgCodes,
272 unsigned &NextArgCode,
273 std::vector<unsigned char> &Sig,
274 ArrayRef<unsigned char> Mapping) {
275
276 if (R->isSubClassOf("LLVMMatchType")) {
277 unsigned Number = Mapping[R->getValueAsInt("Number")];
278 assert(Number < ArgCodes.size() && "Invalid matching number!");
279 if (R->isSubClassOf("LLVMExtendedType"))
280 Sig.push_back(IIT_EXTEND_ARG);
281 else if (R->isSubClassOf("LLVMTruncatedType"))
282 Sig.push_back(IIT_TRUNC_ARG);
283 else if (R->isSubClassOf("LLVMHalfElementsVectorType"))
284 Sig.push_back(IIT_HALF_VEC_ARG);
285 else if (R->isSubClassOf("LLVMScalarOrSameVectorWidth")) {
286 Sig.push_back(IIT_SAME_VEC_WIDTH_ARG);
287 Sig.push_back((Number << 3) | ArgCodes[Number]);
288 MVT::SimpleValueType VT = getValueType(R->getValueAsDef("ElTy"));
289 EncodeFixedValueType(VT, Sig);
290 return;
291 }
292 else if (R->isSubClassOf("LLVMPointerTo"))
293 Sig.push_back(IIT_PTR_TO_ARG);
294 else if (R->isSubClassOf("LLVMVectorOfAnyPointersToElt")) {
295 Sig.push_back(IIT_VEC_OF_ANYPTRS_TO_ELT);
296 // Encode overloaded ArgNo
297 Sig.push_back(NextArgCode++);
298 // Encode LLVMMatchType<Number> ArgNo
299 Sig.push_back(Number);
300 return;
301 } else if (R->isSubClassOf("LLVMPointerToElt"))
302 Sig.push_back(IIT_PTR_TO_ELT);
303 else if (R->isSubClassOf("LLVMVectorElementType"))
304 Sig.push_back(IIT_VEC_ELEMENT);
305 else
306 Sig.push_back(IIT_ARG);
307 return Sig.push_back((Number << 3) | 7 /*IITDescriptor::AK_MatchType*/);
308 }
309
310 MVT::SimpleValueType VT = getValueType(R->getValueAsDef("VT"));
311
312 unsigned Tmp = 0;
313 switch (VT) {
314 default: break;
315 case MVT::iPTRAny: ++Tmp; LLVM_FALLTHROUGH;
316 case MVT::vAny: ++Tmp; LLVM_FALLTHROUGH;
317 case MVT::fAny: ++Tmp; LLVM_FALLTHROUGH;
318 case MVT::iAny: ++Tmp; LLVM_FALLTHROUGH;
319 case MVT::Any: {
320 // If this is an "any" valuetype, then the type is the type of the next
321 // type in the list specified to getIntrinsic().
322 Sig.push_back(IIT_ARG);
323
324 // Figure out what arg # this is consuming, and remember what kind it was.
325 assert(NextArgCode < ArgCodes.size() && ArgCodes[NextArgCode] == Tmp &&
326 "Invalid or no ArgCode associated with overloaded VT!");
327 unsigned ArgNo = NextArgCode++;
328
329 // Encode what sort of argument it must be in the low 3 bits of the ArgNo.
330 return Sig.push_back((ArgNo << 3) | Tmp);
331 }
332
333 case MVT::iPTR: {
334 unsigned AddrSpace = 0;
335 if (R->isSubClassOf("LLVMQualPointerType")) {
336 AddrSpace = R->getValueAsInt("AddrSpace");
337 assert(AddrSpace < 256 && "Address space exceeds 255");
338 }
339 if (AddrSpace) {
340 Sig.push_back(IIT_ANYPTR);
341 Sig.push_back(AddrSpace);
342 } else {
343 Sig.push_back(IIT_PTR);
344 }
345 return EncodeFixedType(R->getValueAsDef("ElTy"), ArgCodes, NextArgCode, Sig,
346 Mapping);
347 }
348 case MVT::iFATPTR64:
349 case MVT::iFATPTR128:
350 case MVT::iFATPTR256:
351 case MVT::iFATPTR512:
352 case MVT::iFATPTRAny: {
353 switch (VT) {
354 default: llvm_unreachable("VT already checked!");
355 case MVT::iFATPTR64: Sig.push_back(IIT_IFATPTR64); break;
356 case MVT::iFATPTR128: Sig.push_back(IIT_IFATPTR128); break;
357 case MVT::iFATPTR256: Sig.push_back(IIT_IFATPTR256); break;
358 case MVT::iFATPTR512: Sig.push_back(IIT_IFATPTR512); break;
359 case MVT::iFATPTRAny: Sig.push_back(IIT_IFATPTRAny); break;
360 }
361 Sig.push_back(200);
362 return EncodeFixedType(R->getValueAsDef("ElTy"), ArgCodes, NextArgCode, Sig,
363 Mapping);
364 }
365 }
366
367 if (MVT(VT).isVector()) {
368 MVT VVT = VT;
369 switch (VVT.getVectorNumElements()) {
370 default: PrintFatalError("unhandled vector type width in intrinsic!");
371 case 1: Sig.push_back(IIT_V1); break;
372 case 2: Sig.push_back(IIT_V2); break;
373 case 4: Sig.push_back(IIT_V4); break;
374 case 8: Sig.push_back(IIT_V8); break;
375 case 16: Sig.push_back(IIT_V16); break;
376 case 32: Sig.push_back(IIT_V32); break;
377 case 64: Sig.push_back(IIT_V64); break;
378 case 512: Sig.push_back(IIT_V512); break;
379 case 1024: Sig.push_back(IIT_V1024); break;
380 }
381
382 return EncodeFixedValueType(VVT.getVectorElementType().SimpleTy, Sig);
383 }
384
385 EncodeFixedValueType(VT, Sig);
386}

388static void UpdateArgCodes(Record *R, std::vector<unsigned char> &ArgCodes,
389 unsigned int &NumInserted,
390 SmallVectorImpl<unsigned char> &Mapping) {
391 if (R->isSubClassOf("LLVMMatchType")) {
392 if (R->isSubClassOf("LLVMVectorOfAnyPointersToElt")) {
393 ArgCodes.push_back(3 /*vAny*/);
394 ++NumInserted;
395 }
396 return;
397 }
398
399 unsigned Tmp = 0;
400 switch (getValueType(R->getValueAsDef("VT"))) {
401 default: break;
402 case MVT::iPTRAny:
403 ++Tmp;
404 LLVM_FALLTHROUGH;
405 case MVT::vAny:
406 ++Tmp;
407 LLVM_FALLTHROUGH;
408 case MVT::fAny:
409 ++Tmp;
410 LLVM_FALLTHROUGH;
411 case MVT::iAny:
412 ++Tmp;
413 LLVM_FALLTHROUGH;
414 case MVT::Any:
415 unsigned OriginalIdx = ArgCodes.size() - NumInserted;
416 assert(OriginalIdx >= Mapping.size());
417 Mapping.resize(OriginalIdx+1);
418 Mapping[OriginalIdx] = ArgCodes.size();
419 ArgCodes.push_back(Tmp);
420 break;
421 }
422}

428/// ComputeFixedEncoding - If we can encode the type signature for this
429/// intrinsic into 32 bits, return it. If not, return ~0U.
430static void ComputeFixedEncoding(const CodeGenIntrinsic &Int,
431 std::vector<unsigned char> &TypeSig) {
432 std::vector<unsigned char> ArgCodes;
433
434 // Add codes for any overloaded result VTs.
435 unsigned int NumInserted = 0;
436 SmallVector<unsigned char, 8> ArgMapping;
437 for (unsigned i = 0, e = Int.IS.RetVTs.size(); i != e; ++i)
438 UpdateArgCodes(Int.IS.RetTypeDefs[i], ArgCodes, NumInserted, ArgMapping);
439
440 // Add codes for any overloaded operand VTs.
441 for (unsigned i = 0, e = Int.IS.ParamTypeDefs.size(); i != e; ++i)
442 UpdateArgCodes(Int.IS.ParamTypeDefs[i], ArgCodes, NumInserted, ArgMapping);
443
444 unsigned NextArgCode = 0;
445 if (Int.IS.RetVTs.empty())
446 TypeSig.push_back(IIT_Done);
447 else if (Int.IS.RetVTs.size() == 1 &&
448 Int.IS.RetVTs[0] == MVT::isVoid)
449 TypeSig.push_back(IIT_Done);
450 else {
451 switch (Int.IS.RetVTs.size()) {
452 case 1: break;
453 case 2: TypeSig.push_back(IIT_STRUCT2); break;
454 case 3: TypeSig.push_back(IIT_STRUCT3); break;
455 case 4: TypeSig.push_back(IIT_STRUCT4); break;
456 case 5: TypeSig.push_back(IIT_STRUCT5); break;
457 case 6: TypeSig.push_back(IIT_STRUCT6); break;
458 case 7: TypeSig.push_back(IIT_STRUCT7); break;
459 case 8: TypeSig.push_back(IIT_STRUCT8); break;
460 default: llvm_unreachable("Unhandled case in struct");
461 }
462
463 for (unsigned i = 0, e = Int.IS.RetVTs.size(); i != e; ++i)
464 EncodeFixedType(Int.IS.RetTypeDefs[i], ArgCodes, NextArgCode, TypeSig,
465 ArgMapping);
466 }
467
468 for (unsigned i = 0, e = Int.IS.ParamTypeDefs.size(); i != e; ++i)
469 EncodeFixedType(Int.IS.ParamTypeDefs[i], ArgCodes, NextArgCode, TypeSig,
470 ArgMapping);
471}

477void IntrinsicEmitter::EmitGenerator(const CodeGenIntrinsicTable &Ints,
478 raw_ostream &OS) {
479 // If we can compute a 32-bit fixed encoding for this intrinsic, do so and
480 // capture it in this vector, otherwise store a ~0U.
481 std::vector<unsigned> FixedEncodings;
482
483 SequenceToOffsetTable<std::vector<unsigned char> > LongEncodingTable;
484
485 std::vector<unsigned char> TypeSig;
486
487 // Compute the unique argument type info.
488 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
489 // Get the signature for the intrinsic.
490 TypeSig.clear();
491 ComputeFixedEncoding(Ints[i], TypeSig);
492
493 // Check to see if we can encode it into a 32-bit word. We can only encode
494 // 8 nibbles into a 32-bit word.
495 if (TypeSig.size() <= 8) {
496 bool Failed = false;
497 unsigned Result = 0;
498 for (unsigned i = 0, e = TypeSig.size(); i != e; ++i) {
499 // If we had an unencodable argument, bail out.
500 if (TypeSig[i] > 15) {
501 Failed = true;
502 break;
503 }
504 Result = (Result << 4) | TypeSig[e-i-1];
505 }
506
507 // If this could be encoded into a 31-bit word, return it.
508 if (!Failed && (Result >> 31) == 0) {
509 FixedEncodings.push_back(Result);
510 continue;
511 }
512 }
513
514 // Otherwise, we're going to unique the sequence into the
515 // LongEncodingTable, and use its offset in the 32-bit table instead.
516 LongEncodingTable.add(TypeSig);
517
518 // This is a placehold that we'll replace after the table is laid out.
519 FixedEncodings.push_back(~0U);
520 }
521
522 LongEncodingTable.layout();
523
524 OS << "// Global intrinsic function declaration type table.\n";
525 OS << "#ifdef GET_INTRINSIC_GENERATOR_GLOBAL\n";
526
527 OS << "static const unsigned IIT_Table[] = {\n ";
528
529 for (unsigned i = 0, e = FixedEncodings.size(); i != e; ++i) {
530 if ((i & 7) == 7)
531 OS << "\n ";
532
533 // If the entry fit in the table, just emit it.
534 if (FixedEncodings[i] != ~0U) {
535 OS << "0x" << Twine::utohexstr(FixedEncodings[i]) << ", ";
536 continue;
537 }
538
539 TypeSig.clear();
540 ComputeFixedEncoding(Ints[i], TypeSig);
541
542
543 // Otherwise, emit the offset into the long encoding table. We emit it this
544 // way so that it is easier to read the offset in the .def file.
545 OS << "(1U<<31) | " << LongEncodingTable.get(TypeSig) << ", ";
546 }
547
548 OS << "0\n};\n\n";
549
550 // Emit the shared table of register lists.
551 OS << "static const unsigned char IIT_LongEncodingTable[] = {\n";
552 if (!LongEncodingTable.empty())
553 LongEncodingTable.emit(OS, printIITEntry);
554 OS << " 255\n};\n\n";
555
556 OS << "#endif\n\n"; // End of GET_INTRINSIC_GENERATOR_GLOBAL
557}

559namespace {
560struct AttributeComparator {
561 bool operator()(const CodeGenIntrinsic *L, const CodeGenIntrinsic *R) const {
562 // Sort throwing intrinsics after non-throwing intrinsics.
563 if (L->canThrow != R->canThrow)
564 return R->canThrow;
565
566 if (L->isNoDuplicate != R->isNoDuplicate)
567 return R->isNoDuplicate;
568
569 if (L->isNoReturn != R->isNoReturn)
570 return R->isNoReturn;
571
572 if (L->isCold != R->isCold)
573 return R->isCold;
574
575 if (L->isConvergent != R->isConvergent)
576 return R->isConvergent;
577
578 if (L->isSpeculatable != R->isSpeculatable)
579 return R->isSpeculatable;
580
581 if (L->hasSideEffects != R->hasSideEffects)
582 return R->hasSideEffects;
583
584 // Try to order by readonly/readnone attribute.
585 CodeGenIntrinsic::ModRefBehavior LK = L->ModRef;
586 CodeGenIntrinsic::ModRefBehavior RK = R->ModRef;
587 if (LK != RK) return (LK > RK);
588 // Order by argument attributes.
589 // This is reliable because each side is already sorted internally.
590 return (L->ArgumentAttributes < R->ArgumentAttributes);
591 }
592};
593} // End anonymous namespace

595/// EmitAttributes - This emits the Intrinsic::getAttributes method.
596void IntrinsicEmitter::EmitAttributes(const CodeGenIntrinsicTable &Ints,
597 raw_ostream &OS) {
598 OS << "// Add parameter attributes that are not common to all intrinsics.\n";
599 OS << "#ifdef GET_INTRINSIC_ATTRIBUTES\n";
600 if (TargetOnly)
601 OS << "static AttributeList getAttributes(LLVMContext &C, " << TargetPrefix
602 << "Intrinsic::ID id) {\n";
603 else
604 OS << "AttributeList Intrinsic::getAttributes(LLVMContext &C, ID id) {\n";
605
606 // Compute the maximum number of attribute arguments and the map
607 typedef std::map<const CodeGenIntrinsic*, unsigned,
608 AttributeComparator> UniqAttrMapTy;
609 UniqAttrMapTy UniqAttributes;
610 unsigned maxArgAttrs = 0;
611 unsigned AttrNum = 0;
612 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
613 const CodeGenIntrinsic &intrinsic = Ints[i];
614 maxArgAttrs =
615 std::max(maxArgAttrs, unsigned(intrinsic.ArgumentAttributes.size()));
616 unsigned &N = UniqAttributes[&intrinsic];
617 if (N) continue;
618 assert(AttrNum < 256 && "Too many unique attributes for table!");
619 N = ++AttrNum;
620 }
621
622 // Emit an array of AttributeList. Most intrinsics will have at least one
623 // entry, for the function itself (index ~1), which is usually nounwind.
624 OS << " static const uint8_t IntrinsicsToAttributesMap[] = {\n";
625
626 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
627 const CodeGenIntrinsic &intrinsic = Ints[i];
628
629 OS << " " << UniqAttributes[&intrinsic] << ", // "
630 << intrinsic.Name << "\n";
631 }
632 OS << " };\n\n";
633
634 OS << " AttributeList AS[" << maxArgAttrs + 1 << "];\n";
635 OS << " unsigned NumAttrs = 0;\n";
636 OS << " if (id != 0) {\n";
637 OS << " switch(IntrinsicsToAttributesMap[id - ";
638 if (TargetOnly)
639 OS << "Intrinsic::num_intrinsics";
640 else
641 OS << "1";
642 OS << "]) {\n";
643 OS << " default: llvm_unreachable(\"Invalid attribute number\");\n";
644 for (UniqAttrMapTy::const_iterator I = UniqAttributes.begin(),
645 E = UniqAttributes.end(); I != E; ++I) {
646 OS << " case " << I->second << ": {\n";
647
648 const CodeGenIntrinsic &intrinsic = *(I->first);
649
650 // Keep track of the number of attributes we're writing out.
651 unsigned numAttrs = 0;
652
653 // The argument attributes are alreadys sorted by argument index.
654 unsigned ai = 0, ae = intrinsic.ArgumentAttributes.size();
655 if (ae) {
656 while (ai != ae) {
657 unsigned argNo = intrinsic.ArgumentAttributes[ai].first;
658 unsigned attrIdx = argNo + 1; // Must match AttributeList::FirstArgIndex
659
660 OS << " const Attribute::AttrKind AttrParam" << attrIdx << "[]= {";
661 bool addComma = false;
662
663 do {
664 switch (intrinsic.ArgumentAttributes[ai].second) {
665 case CodeGenIntrinsic::NoCapture:
666 if (addComma)
667 OS << ",";
668 OS << "Attribute::NoCapture";
669 addComma = true;
670 break;
671 case CodeGenIntrinsic::Returned:
672 if (addComma)
673 OS << ",";
674 OS << "Attribute::Returned";
675 addComma = true;
676 break;
677 case CodeGenIntrinsic::ReadOnly:
678 if (addComma)
679 OS << ",";
680 OS << "Attribute::ReadOnly";
681 addComma = true;
682 break;
683 case CodeGenIntrinsic::WriteOnly:
684 if (addComma)
685 OS << ",";
686 OS << "Attribute::WriteOnly";
687 addComma = true;
688 break;
689 case CodeGenIntrinsic::ReadNone:
690 if (addComma)
691 OS << ",";
692 OS << "Attribute::ReadNone";
693 addComma = true;
694 break;
695 case CodeGenIntrinsic::ImmArg:
696 if (addComma)
697 OS << ',';
698 OS << "Attribute::ImmArg";
699 addComma = true;
700 break;
701 }
702
703 ++ai;
704 } while (ai != ae && intrinsic.ArgumentAttributes[ai].first == argNo);
705 OS << "};\n";
706 OS << " AS[" << numAttrs++ << "] = AttributeList::get(C, "
707 << attrIdx << ", AttrParam" << attrIdx << ");\n";
708 }
709 }
710
711 if (!intrinsic.canThrow ||
712 intrinsic.ModRef != CodeGenIntrinsic::ReadWriteMem ||
713 intrinsic.isNoReturn || intrinsic.isCold || intrinsic.isNoDuplicate ||
714 intrinsic.isConvergent || intrinsic.isSpeculatable ||
715 intrinsic.hasSideEffects) {
716 OS << " const Attribute::AttrKind Atts[] = {";
717 bool addComma = false;
718 if (!intrinsic.canThrow) {
719 OS << "Attribute::NoUnwind";
720 addComma = true;
721 }
722 if (intrinsic.isNoReturn) {
723 if (addComma)
724 OS << ",";
725 OS << "Attribute::NoReturn";
726 addComma = true;
727 }
728 if (intrinsic.isCold) {
729 if (addComma)
730 OS << ",";
731 OS << "Attribute::Cold";
732 addComma = true;
733 }
734 if (intrinsic.isNoDuplicate) {
735 if (addComma)
736 OS << ",";
737 OS << "Attribute::NoDuplicate";
738 addComma = true;
739 }
740 if (intrinsic.isConvergent) {
741 if (addComma)
742 OS << ",";
743 OS << "Attribute::Convergent";
744 addComma = true;
745 }
746 if (intrinsic.isSpeculatable) {
747 if (addComma)
748 OS << ",";
749 OS << "Attribute::Speculatable";
750 addComma = true;
751 }
752 if (intrinsic.hasSideEffects) {
753 if (addComma)
754 OS << ",";
755 OS << "Attribute::HasSideEffects";
756 addComma = true;
757 }
758
759 switch (intrinsic.ModRef) {
760 case CodeGenIntrinsic::NoMem:
761 if (addComma)
762 OS << ",";
763 OS << "Attribute::ReadNone";
764 break;
765 case CodeGenIntrinsic::ReadArgMem:
766 if (addComma)
767 OS << ",";
768 OS << "Attribute::ReadOnly,";
769 OS << "Attribute::ArgMemOnly";
770 break;
771 case CodeGenIntrinsic::ReadMem:
772 if (addComma)
773 OS << ",";
774 OS << "Attribute::ReadOnly";
775 break;
776 case CodeGenIntrinsic::ReadInaccessibleMem:
777 if (addComma)
778 OS << ",";
779 OS << "Attribute::ReadOnly,";
780 OS << "Attribute::InaccessibleMemOnly";
781 break;
782 case CodeGenIntrinsic::ReadInaccessibleMemOrArgMem:
783 if (addComma)
784 OS << ",";
785 OS << "Attribute::ReadOnly,";
786 OS << "Attribute::InaccessibleMemOrArgMemOnly";
787 break;
788 case CodeGenIntrinsic::WriteArgMem:
789 if (addComma)
790 OS << ",";
791 OS << "Attribute::WriteOnly,";
792 OS << "Attribute::ArgMemOnly";
793 break;
794 case CodeGenIntrinsic::WriteMem:
795 if (addComma)
796 OS << ",";
797 OS << "Attribute::WriteOnly";
798 break;
799 case CodeGenIntrinsic::WriteInaccessibleMem:
800 if (addComma)
801 OS << ",";
802 OS << "Attribute::WriteOnly,";
803 OS << "Attribute::InaccessibleMemOnly";
804 break;
805 case CodeGenIntrinsic::WriteInaccessibleMemOrArgMem:
806 if (addComma)
807 OS << ",";
808 OS << "Attribute::WriteOnly,";
809 OS << "Attribute::InaccessibleMemOrArgMemOnly";
810 break;
811 case CodeGenIntrinsic::ReadWriteArgMem:
812 if (addComma)
813 OS << ",";
814 OS << "Attribute::ArgMemOnly";
815 break;
816 case CodeGenIntrinsic::ReadWriteInaccessibleMem:
817 if (addComma)
818 OS << ",";
819 OS << "Attribute::InaccessibleMemOnly";
820 break;
821 case CodeGenIntrinsic::ReadWriteInaccessibleMemOrArgMem:
822 if (addComma)
823 OS << ",";
824 OS << "Attribute::InaccessibleMemOrArgMemOnly";
825 break;
826 case CodeGenIntrinsic::ReadWriteMem:
827 break;
828 }
829 OS << "};\n";
830 OS << " AS[" << numAttrs++ << "] = AttributeList::get(C, "
831 << "AttributeList::FunctionIndex, Atts);\n";
832 }
833
834 if (numAttrs) {
835 OS << " NumAttrs = " << numAttrs << ";\n";
836 OS << " break;\n";
837 OS << " }\n";
838 } else {
839 OS << " return AttributeList();\n";
840 OS << " }\n";
841 }
842 }
843
844 OS << " }\n";
845 OS << " }\n";
846 OS << " return AttributeList::get(C, makeArrayRef(AS, NumAttrs));\n";
847 OS << "}\n";
848 OS << "#endif // GET_INTRINSIC_ATTRIBUTES\n\n";
849}

851void IntrinsicEmitter::EmitIntrinsicToBuiltinMap(
852 const CodeGenIntrinsicTable &Ints, bool IsGCC, raw_ostream &OS) {
853 StringRef CompilerName = (IsGCC ? "GCC" : "MS");
854 typedef std::map<std::string, std::map<std::string, std::string>> BIMTy;
855 BIMTy BuiltinMap;
856 StringToOffsetTable Table;
857 for (unsigned i = 0, e = Ints.size(); i != e; ++i) {
858 const std::string &BuiltinName =
859 IsGCC ? Ints[i].GCCBuiltinName : Ints[i].MSBuiltinName;
860 if (!BuiltinName.empty()) {
861 // Get the map for this target prefix.
862 std::map<std::string, std::string> &BIM =
863 BuiltinMap[Ints[i].TargetPrefix];
864
865 if (!BIM.insert(std::make_pair(BuiltinName, Ints[i].EnumName)).second)
866 PrintFatalError(Ints[i].TheDef->getLoc(),
867 "Intrinsic '" + Ints[i].TheDef->getName() +
868 "': duplicate " + CompilerName + " builtin name!");
869 Table.GetOrAddStringOffset(BuiltinName);
870 }
871 if (!Ints[i].GCCBuiltinAliasName.empty()) {
872 // Get the map for this target prefix.
873 std::map<std::string, std::string> &BIM =BuiltinMap[Ints[i].TargetPrefix];
874
875 if (!BIM.insert(std::make_pair(Ints[i].GCCBuiltinAliasName,
876 Ints[i].EnumName)).second)
877 PrintFatalError("Intrinsic '" + Ints[i].TheDef->getName() +
878 "': duplicate GCC builtin name!");
879 Table.GetOrAddStringOffset(Ints[i].GCCBuiltinAliasName);
880 }
881 }
882
883 OS << "// Get the LLVM intrinsic that corresponds to a builtin.\n";
884 OS << "// This is used by the C front-end. The builtin name is passed\n";
885 OS << "// in as BuiltinName, and a target prefix (e.g. 'ppc') is passed\n";
886 OS << "// in as TargetPrefix. The result is assigned to 'IntrinsicID'.\n";
887 OS << "#ifdef GET_LLVM_INTRINSIC_FOR_" << CompilerName << "_BUILTIN\n";
888
889 if (TargetOnly) {
890 OS << "static " << TargetPrefix << "Intrinsic::ID "
891 << "getIntrinsicFor" << CompilerName << "Builtin(const char "
892 << "*TargetPrefixStr, StringRef BuiltinNameStr) {\n";
893 } else {
894 OS << "Intrinsic::ID Intrinsic::getIntrinsicFor" << CompilerName
895 << "Builtin(const char "
896 << "*TargetPrefixStr, StringRef BuiltinNameStr) {\n";
897 }
898
899 if (Table.Empty()) {
900 OS << " return ";
901 if (!TargetPrefix.empty())
902 OS << "(" << TargetPrefix << "Intrinsic::ID)";
903 OS << "Intrinsic::not_intrinsic;\n";
904 OS << "}\n";
905 OS << "#endif\n\n";
906 return;
907 }
908
909 OS << " static const char BuiltinNames[] = {\n";
910 Table.EmitCharArray(OS);
911 OS << " };\n\n";
912
913 OS << " struct BuiltinEntry {\n";
914 OS << " Intrinsic::ID IntrinID;\n";
915 OS << " unsigned StrTabOffset;\n";
916 OS << " const char *getName() const {\n";
917 OS << " return &BuiltinNames[StrTabOffset];\n";
918 OS << " }\n";
919 OS << " bool operator<(StringRef RHS) const {\n";
920 OS << " return strncmp(getName(), RHS.data(), RHS.size()) < 0;\n";
921 OS << " }\n";
922 OS << " };\n";
923
924 OS << " StringRef TargetPrefix(TargetPrefixStr);\n\n";
925
926 // Note: this could emit significantly better code if we cared.
927 for (BIMTy::iterator I = BuiltinMap.begin(), E = BuiltinMap.end();I != E;++I){
928 OS << " ";
929 if (!I->first.empty())
930 OS << "if (TargetPrefix == \"" << I->first << "\") ";
931 else
932 OS << "/* Target Independent Builtins */ ";
933 OS << "{\n";
934
935 // Emit the comparisons for this target prefix.
936 OS << " static const BuiltinEntry " << I->first << "Names[] = {\n";
937 for (const auto &P : I->second) {
938 OS << " {Intrinsic::" << P.second << ", "
939 << Table.GetOrAddStringOffset(P.first) << "}, // " << P.first << "\n";
940 }
941 OS << " };\n";
942 OS << " auto I = std::lower_bound(std::begin(" << I->first << "Names),\n";
943 OS << " std::end(" << I->first << "Names),\n";
944 OS << " BuiltinNameStr);\n";
945 OS << " if (I != std::end(" << I->first << "Names) &&\n";
946 OS << " I->getName() == BuiltinNameStr)\n";
947 OS << " return I->IntrinID;\n";
948 OS << " }\n";
949 }
950 OS << " return ";
951 if (!TargetPrefix.empty())
952 OS << "(" << TargetPrefix << "Intrinsic::ID)";
953 OS << "Intrinsic::not_intrinsic;\n";
954 OS << "}\n";
955 OS << "#endif\n\n";
956}
