147// The set (BIT_TRUE, BIT_FALSE, BIT_UNSET) represents a ternary logic system
148// for a bit value.
149//
150// BIT_UNFILTERED is used as the init value for a filter position. It is used
151// only for filter processings.
152typedef enum {
153 BIT_TRUE, // '1'
154 BIT_FALSE, // '0'
155 BIT_UNSET, // '?'
156 BIT_UNFILTERED // unfiltered
157} bit_value_t;
158
159static bool ValueSet(bit_value_t V) {
160 return (V == BIT_TRUE || V == BIT_FALSE);
161}
162
163static bool ValueNotSet(bit_value_t V) {
164 return (V == BIT_UNSET);
165}
166
167static int Value(bit_value_t V) {
168 return ValueNotSet(V) ? -1 : (V == BIT_FALSE ? 0 : 1);
169}
170
171static bit_value_t bitFromBits(const BitsInit &bits, unsigned index) {
172 if (BitInit *bit = dyn_cast<BitInit>(bits.getBit(index)))
173 return bit->getValue() ? BIT_TRUE : BIT_FALSE;
174
175 // The bit is uninitialized.
176 return BIT_UNSET;
177}
178
179// Prints the bit value for each position.
180static void dumpBits(raw_ostream &o, const BitsInit &bits) {
181 for (unsigned index = bits.getNumBits(); index > 0; --index) {
182 switch (bitFromBits(bits, index - 1)) {
183 case BIT_TRUE:
184 o << "1";
185 break;
186 case BIT_FALSE:
187 o << "0";
188 break;
189 case BIT_UNSET:
190 o << "_";
191 break;
192 default:
193 llvm_unreachable("unexpected return value from bitFromBits");
194 }
195 }
196}

210/// Filter - Filter works with FilterChooser to produce the decoding tree for
211/// the ISA.
212///
213/// It is useful to think of a Filter as governing the switch stmts of the
214/// decoding tree in a certain level. Each case stmt delegates to an inferior
215/// FilterChooser to decide what further decoding logic to employ, or in another
216/// words, what other remaining bits to look at. The FilterChooser eventually
217/// chooses a best Filter to do its job.
218///
219/// This recursive scheme ends when the number of Opcodes assigned to the
220/// FilterChooser becomes 1 or if there is a conflict. A conflict happens when
221/// the Filter/FilterChooser combo does not know how to distinguish among the
222/// Opcodes assigned.
223///
224/// An example of a conflict is
225///
226/// Conflict:
227/// 111101000.00........00010000....
228/// 111101000.00........0001........
229/// 1111010...00........0001........
230/// 1111010...00....................
231/// 1111010.........................
232/// 1111............................
233/// ................................
234/// VST4q8a 111101000_00________00010000____
235/// VST4q8b 111101000_00________00010000____
236///
237/// The Debug output shows the path that the decoding tree follows to reach the
238/// the conclusion that there is a conflict. VST4q8a is a vst4 to double-spaced
239/// even registers, while VST4q8b is a vst4 to double-spaced odd registers.
240///
241/// The encoding info in the .td files does not specify this meta information,
242/// which could have been used by the decoder to resolve the conflict. The
243/// decoder could try to decode the even/odd register numbering and assign to
244/// VST4q8a or VST4q8b, but for the time being, the decoder chooses the "a"
245/// version and return the Opcode since the two have the same Asm format string.
246class Filter {
247protected:
248 const FilterChooser *Owner;// points to the FilterChooser who owns this filter
249 unsigned StartBit; // the starting bit position
250 unsigned NumBits; // number of bits to filter
251 bool Mixed; // a mixed region contains both set and unset bits
252
253 // Map of well-known segment value to the set of uid's with that value.
254 std::map<uint64_t, std::vector<unsigned>> FilteredInstructions;
255
256 // Set of uid's with non-constant segment values.
257 std::vector<unsigned> VariableInstructions;
258
259 // Map of well-known segment value to its delegate.
260 std::map<unsigned, std::unique_ptr<const FilterChooser>> FilterChooserMap;
261
262 // Number of instructions which fall under FilteredInstructions category.
263 unsigned NumFiltered;
264
265 // Keeps track of the last opcode in the filtered bucket.
266 unsigned LastOpcFiltered;
267
268public:
269 Filter(Filter &&f);
270 Filter(FilterChooser &owner, unsigned startBit, unsigned numBits, bool mixed);
271
272 ~Filter() = default;
273
274 unsigned getNumFiltered() const { return NumFiltered; }
275
276 unsigned getSingletonOpc() const {
277 assert(NumFiltered == 1);
278 return LastOpcFiltered;
279 }
280
281 // Return the filter chooser for the group of instructions without constant
282 // segment values.
283 const FilterChooser &getVariableFC() const {
284 assert(NumFiltered == 1);
285 assert(FilterChooserMap.size() == 1);
286 return *(FilterChooserMap.find((unsigned)-1)->second);

318/// FilterChooser - FilterChooser chooses the best filter among a set of Filters
319/// in order to perform the decoding of instructions at the current level.
320///
321/// Decoding proceeds from the top down. Based on the well-known encoding bits
322/// of instructions available, FilterChooser builds up the possible Filters that
323/// can further the task of decoding by distinguishing among the remaining
324/// candidate instructions.
325///
326/// Once a filter has been chosen, it is called upon to divide the decoding task
327/// into sub-tasks and delegates them to its inferior FilterChoosers for further
328/// processings.
329///
330/// It is useful to think of a Filter as governing the switch stmts of the
331/// decoding tree. And each case is delegated to an inferior FilterChooser to

395 // Populates the insn given the uid.
396 void insnWithID(insn_t &Insn, unsigned Opcode) const {
397 BitsInit &Bits = getBitsField(*AllInstructions[Opcode].EncodingDef, "Inst");
398
399 // We may have a SoftFail bitmask, which specifies a mask where an encoding
400 // may differ from the value in "Inst" and yet still be valid, but the
401 // disassembler should return SoftFail instead of Success.
402 //
403 // This is used for marking UNPREDICTABLE instructions in the ARM world.
404 BitsInit *SFBits =
405 AllInstructions[Opcode].EncodingDef->getValueAsBitsInit("SoftFail");

408 if (SFBits && bitFromBits(*SFBits, i) == BIT_TRUE)
409 Insn.push_back(BIT_UNSET);
410 else
411 Insn.push_back(bitFromBits(Bits, i));
412 }
413 }

513 : Owner(f.Owner), StartBit(f.StartBit), NumBits(f.NumBits), Mixed(f.Mixed),

520Filter::Filter(FilterChooser &owner, unsigned startBit, unsigned numBits,
521 bool mixed)
522 : Owner(&owner), StartBit(startBit), NumBits(numBits), Mixed(mixed) {
523 assert(StartBit + NumBits - 1 < Owner->BitWidth);
524
525 NumFiltered = 0;
526 LastOpcFiltered = 0;
527
528 for (unsigned i = 0, e = Owner->Opcodes.size(); i != e; ++i) {
529 insn_t Insn;
530
531 // Populates the insn given the uid.
532 Owner->insnWithID(Insn, Owner->Opcodes[i]);
533
534 uint64_t Field;
535 // Scans the segment for possibly well-specified encoding bits.
536 bool ok = Owner->fieldFromInsn(Field, Insn, StartBit, NumBits);
537
538 if (ok) {
539 // The encoding bits are well-known. Lets add the uid of the
540 // instruction into the bucket keyed off the constant field value.
541 LastOpcFiltered = Owner->Opcodes[i];
542 FilteredInstructions[Field].push_back(LastOpcFiltered);
543 ++NumFiltered;
544 } else {
545 // Some of the encoding bit(s) are unspecified. This contributes to
546 // one additional member of "Variable" instructions.
547 VariableInstructions.push_back(Owner->Opcodes[i]);
548 }
549 }
550
551 assert((FilteredInstructions.size() + VariableInstructions.size() > 0)
552 && "Filter returns no instruction categories");
553}
554
555// Divides the decoding task into sub tasks and delegates them to the
556// inferior FilterChooser's.
557//
558// A special case arises when there's only one entry in the filtered
559// instructions. In order to unambiguously decode the singleton, we need to
560// match the remaining undecoded encoding bits against the singleton.
561void Filter::recurse() {
562 // Starts by inheriting our parent filter chooser's filter bit values.
563 std::vector<bit_value_t> BitValueArray(Owner->FilterBitValues);
564
565 if (!VariableInstructions.empty()) {
566 // Conservatively marks each segment position as BIT_UNSET.
567 for (unsigned bitIndex = 0; bitIndex < NumBits; ++bitIndex)
568 BitValueArray[StartBit + bitIndex] = BIT_UNSET;
569
570 // Delegates to an inferior filter chooser for further processing on this
571 // group of instructions whose segment values are variable.
572 FilterChooserMap.insert(
573 std::make_pair(-1U, llvm::make_unique<FilterChooser>(
574 Owner->AllInstructions, VariableInstructions,
575 Owner->Operands, BitValueArray, *Owner)));
576 }
577
578 // No need to recurse for a singleton filtered instruction.
579 // See also Filter::emit*().
580 if (getNumFiltered() == 1) {
581 assert(FilterChooserMap.size() == 1);
582 return;
583 }
584
585 // Otherwise, create sub choosers.
586 for (const auto &Inst : FilteredInstructions) {
587
588 // Marks all the segment positions with either BIT_TRUE or BIT_FALSE.
589 for (unsigned bitIndex = 0; bitIndex < NumBits; ++bitIndex) {
590 if (Inst.first & (1ULL << bitIndex))
591 BitValueArray[StartBit + bitIndex] = BIT_TRUE;
592 else
593 BitValueArray[StartBit + bitIndex] = BIT_FALSE;
594 }
595
596 // Delegates to an inferior filter chooser for further processing on this
597 // category of instructions.
598 FilterChooserMap.insert(std::make_pair(
599 Inst.first, llvm::make_unique<FilterChooser>(
600 Owner->AllInstructions, Inst.second,
601 Owner->Operands, BitValueArray, *Owner)));
602 }
603}

605static void resolveTableFixups(DecoderTable &Table, const FixupList &Fixups,
606 uint32_t DestIdx) {
607 // Any NumToSkip fixups in the current scope can resolve to the
608 // current location.
609 for (FixupList::const_reverse_iterator I = Fixups.rbegin(),
610 E = Fixups.rend();
611 I != E; ++I) {
612 // Calculate the distance from the byte following the fixup entry byte
613 // to the destination. The Target is calculated from after the 16-bit
614 // NumToSkip entry itself, so subtract two from the displacement here
615 // to account for that.
616 uint32_t FixupIdx = *I;
617 uint32_t Delta = DestIdx - FixupIdx - 3;
618 // Our NumToSkip entries are 24-bits. Make sure our table isn't too
619 // big.
620 assert(Delta < (1u << 24));
621 Table[FixupIdx] = (uint8_t)Delta;
622 Table[FixupIdx + 1] = (uint8_t)(Delta >> 8);
623 Table[FixupIdx + 2] = (uint8_t)(Delta >> 16);
624 }
625}

627// Emit table entries to decode instructions given a segment or segments
628// of bits.
629void Filter::emitTableEntry(DecoderTableInfo &TableInfo) const {
630 TableInfo.Table.push_back(MCD::OPC_ExtractField);
631 TableInfo.Table.push_back(StartBit);
632 TableInfo.Table.push_back(NumBits);
633
634 // A new filter entry begins a new scope for fixup resolution.
635 TableInfo.FixupStack.emplace_back();
636
637 DecoderTable &Table = TableInfo.Table;
638
639 size_t PrevFilter = 0;
640 bool HasFallthrough = false;
641 for (auto &Filter : FilterChooserMap) {
642 // Field value -1 implies a non-empty set of variable instructions.
643 // See also recurse().
644 if (Filter.first == (unsigned)-1) {
645 HasFallthrough = true;
646
647 // Each scope should always have at least one filter value to check
648 // for.
649 assert(PrevFilter != 0 && "empty filter set!");
650 FixupList &CurScope = TableInfo.FixupStack.back();
651 // Resolve any NumToSkip fixups in the current scope.
652 resolveTableFixups(Table, CurScope, Table.size());
653 CurScope.clear();
654 PrevFilter = 0; // Don't re-process the filter's fallthrough.
655 } else {
656 Table.push_back(MCD::OPC_FilterValue);
657 // Encode and emit the value to filter against.
658 uint8_t Buffer[16];
659 unsigned Len = encodeULEB128(Filter.first, Buffer);
660 Table.insert(Table.end(), Buffer, Buffer + Len);
661 // Reserve space for the NumToSkip entry. We'll backpatch the value
662 // later.
663 PrevFilter = Table.size();
664 Table.push_back(0);
665 Table.push_back(0);
666 Table.push_back(0);
667 }
668
669 // We arrive at a category of instructions with the same segment value.
670 // Now delegate to the sub filter chooser for further decodings.
671 // The case may fallthrough, which happens if the remaining well-known
672 // encoding bits do not match exactly.
673 Filter.second->emitTableEntries(TableInfo);
674
675 // Now that we've emitted the body of the handler, update the NumToSkip
676 // of the filter itself to be able to skip forward when false. Subtract
677 // two as to account for the width of the NumToSkip field itself.
678 if (PrevFilter) {
679 uint32_t NumToSkip = Table.size() - PrevFilter - 3;
680 assert(NumToSkip < (1u << 24) && "disassembler decoding table too large!");
681 Table[PrevFilter] = (uint8_t)NumToSkip;
682 Table[PrevFilter + 1] = (uint8_t)(NumToSkip >> 8);
683 Table[PrevFilter + 2] = (uint8_t)(NumToSkip >> 16);
684 }
685 }
686
687 // Any remaining unresolved fixups bubble up to the parent fixup scope.
688 assert(TableInfo.FixupStack.size() > 1 && "fixup stack underflow!");
689 FixupScopeList::iterator Source = TableInfo.FixupStack.end() - 1;
690 FixupScopeList::iterator Dest = Source - 1;
691 Dest->insert(Dest->end(), Source->begin(), Source->end());
692 TableInfo.FixupStack.pop_back();
693
694 // If there is no fallthrough, then the final filter should get fixed
695 // up according to the enclosing scope rather than the current position.
696 if (!HasFallthrough)
697 TableInfo.FixupStack.back().push_back(PrevFilter);
698}

715// Emit the decoder state machine table.
716void FixedLenDecoderEmitter::emitTable(formatted_raw_ostream &OS,
717 DecoderTable &Table,
718 unsigned Indentation,
719 unsigned BitWidth,
720 StringRef Namespace) const {
721 OS.indent(Indentation) << "static const uint8_t DecoderTable" << Namespace
722 << BitWidth << "[] = {\n";
723
724 Indentation += 2;
725
726 // FIXME: We may be able to use the NumToSkip values to recover
727 // appropriate indentation levels.
728 DecoderTable::const_iterator I = Table.begin();
729 DecoderTable::const_iterator E = Table.end();
730 while (I != E) {
731 assert (I < E && "incomplete decode table entry!");
732
733 uint64_t Pos = I - Table.begin();
734 OS << "/* " << Pos << " */";
735 OS.PadToColumn(12);
736
737 switch (*I) {
738 default:
739 PrintFatalError("invalid decode table opcode");
740 case MCD::OPC_ExtractField: {
741 ++I;
742 unsigned Start = *I++;
743 unsigned Len = *I++;
744 OS.indent(Indentation) << "MCD::OPC_ExtractField, " << Start << ", "
745 << Len << ", // Inst{";
746 if (Len > 1)
747 OS << (Start + Len - 1) << "-";
748 OS << Start << "} ...\n";
749 break;
750 }
751 case MCD::OPC_FilterValue: {
752 ++I;
753 OS.indent(Indentation) << "MCD::OPC_FilterValue, ";
754 // The filter value is ULEB128 encoded.
755 while (*I >= 128)
756 OS << (unsigned)*I++ << ", ";
757 OS << (unsigned)*I++ << ", ";
758
759 // 24-bit numtoskip value.
760 uint8_t Byte = *I++;
761 uint32_t NumToSkip = Byte;
762 OS << (unsigned)Byte << ", ";
763 Byte = *I++;
764 OS << (unsigned)Byte << ", ";
765 NumToSkip |= Byte << 8;
766 Byte = *I++;
767 OS << utostr(Byte) << ", ";
768 NumToSkip |= Byte << 16;
769 OS << "// Skip to: " << ((I - Table.begin()) + NumToSkip) << "\n";
770 break;
771 }
772 case MCD::OPC_CheckField: {
773 ++I;
774 unsigned Start = *I++;
775 unsigned Len = *I++;
776 OS.indent(Indentation) << "MCD::OPC_CheckField, " << Start << ", "
777 << Len << ", ";// << Val << ", " << NumToSkip << ",\n";
778 // ULEB128 encoded field value.
779 for (; *I >= 128; ++I)
780 OS << (unsigned)*I << ", ";
781 OS << (unsigned)*I++ << ", ";
782 // 24-bit numtoskip value.
783 uint8_t Byte = *I++;
784 uint32_t NumToSkip = Byte;
785 OS << (unsigned)Byte << ", ";
786 Byte = *I++;
787 OS << (unsigned)Byte << ", ";
788 NumToSkip |= Byte << 8;
789 Byte = *I++;
790 OS << utostr(Byte) << ", ";
791 NumToSkip |= Byte << 16;
792 OS << "// Skip to: " << ((I - Table.begin()) + NumToSkip) << "\n";
793 break;
794 }
795 case MCD::OPC_CheckPredicate: {
796 ++I;
797 OS.indent(Indentation) << "MCD::OPC_CheckPredicate, ";
798 for (; *I >= 128; ++I)
799 OS << (unsigned)*I << ", ";
800 OS << (unsigned)*I++ << ", ";
801
802 // 24-bit numtoskip value.
803 uint8_t Byte = *I++;
804 uint32_t NumToSkip = Byte;
805 OS << (unsigned)Byte << ", ";
806 Byte = *I++;
807 OS << (unsigned)Byte << ", ";
808 NumToSkip |= Byte << 8;
809 Byte = *I++;
810 OS << utostr(Byte) << ", ";
811 NumToSkip |= Byte << 16;
812 OS << "// Skip to: " << ((I - Table.begin()) + NumToSkip) << "\n";
813 break;
814 }
815 case MCD::OPC_Decode:
816 case MCD::OPC_TryDecode: {
817 bool IsTry = *I == MCD::OPC_TryDecode;
818 ++I;
819 // Extract the ULEB128 encoded Opcode to a buffer.
820 uint8_t Buffer[16], *p = Buffer;
821 while ((*p++ = *I++) >= 128)
822 assert((p - Buffer) <= (ptrdiff_t)sizeof(Buffer)
823 && "ULEB128 value too large!");
824 // Decode the Opcode value.
825 unsigned Opc = decodeULEB128(Buffer);
826 OS.indent(Indentation) << "MCD::OPC_" << (IsTry ? "Try" : "")
827 << "Decode, ";
828 for (p = Buffer; *p >= 128; ++p)
829 OS << (unsigned)*p << ", ";
830 OS << (unsigned)*p << ", ";
831
832 // Decoder index.
833 for (; *I >= 128; ++I)
834 OS << (unsigned)*I << ", ";
835 OS << (unsigned)*I++ << ", ";
836
837 if (!IsTry) {
838 OS << "// Opcode: " << NumberedEncodings[Opc] << "\n";
839 break;
840 }
841
842 // Fallthrough for OPC_TryDecode.
843
844 // 24-bit numtoskip value.
845 uint8_t Byte = *I++;
846 uint32_t NumToSkip = Byte;
847 OS << (unsigned)Byte << ", ";
848 Byte = *I++;
849 OS << (unsigned)Byte << ", ";
850 NumToSkip |= Byte << 8;
851 Byte = *I++;
852 OS << utostr(Byte) << ", ";
853 NumToSkip |= Byte << 16;
854
855 OS << "// Opcode: " << NumberedEncodings[Opc]
856 << ", skip to: " << ((I - Table.begin()) + NumToSkip) << "\n";
857 break;
858 }
859 case MCD::OPC_SoftFail: {
860 ++I;
861 OS.indent(Indentation) << "MCD::OPC_SoftFail";
862 // Positive mask
863 uint64_t Value = 0;
864 unsigned Shift = 0;
865 do {
866 OS << ", " << (unsigned)*I;
867 Value += (*I & 0x7f) << Shift;
868 Shift += 7;
869 } while (*I++ >= 128);
870 if (Value > 127) {
871 OS << " /* 0x";
872 OS.write_hex(Value);
873 OS << " */";
874 }
875 // Negative mask
876 Value = 0;
877 Shift = 0;
878 do {
879 OS << ", " << (unsigned)*I;
880 Value += (*I & 0x7f) << Shift;
881 Shift += 7;
882 } while (*I++ >= 128);
883 if (Value > 127) {
884 OS << " /* 0x";
885 OS.write_hex(Value);
886 OS << " */";
887 }
888 OS << ",\n";
889 break;
890 }
891 case MCD::OPC_Fail: {
892 ++I;
893 OS.indent(Indentation) << "MCD::OPC_Fail,\n";
894 break;
895 }
896 }
897 }
898 OS.indent(Indentation) << "0\n";
899
900 Indentation -= 2;
901
902 OS.indent(Indentation) << "};\n\n";
903}

956// Populates the field of the insn given the start position and the number of
957// consecutive bits to scan for.
958//
959// Returns false if and on the first uninitialized bit value encountered.
960// Returns true, otherwise.

962 unsigned StartBit, unsigned NumBits) const {
963 Field = 0;

969 if (Insn[StartBit + i] == BIT_TRUE)

972
973 return true;

1011// Calculates the island(s) needed to decode the instruction.
1012// This returns a list of undecoded bits of an instructions, for example,
1013// Inst{20} = 1 && Inst{3-0} == 0b1111 represents two islands of yet-to-be
1014// decoded bits in order to verify that the instruction matches the Opcode.
1015unsigned FilterChooser::getIslands(std::vector<unsigned> &StartBits,
1016 std::vector<unsigned> &EndBits,
1017 std::vector<uint64_t> &FieldVals,
1018 const insn_t &Insn) const {
1019 unsigned Num, BitNo;
1020 Num = BitNo = 0;
1021
1022 uint64_t FieldVal = 0;
1023
1024 // 0: Init
1025 // 1: Water (the bit value does not affect decoding)
1026 // 2: Island (well-known bit value needed for decoding)
1027 int State = 0;
1028 int Val = -1;
1029
1030 for (unsigned i = 0; i < BitWidth; ++i) {
1031 Val = Value(Insn[i]);
1032 bool Filtered = PositionFiltered(i);
1033 switch (State) {
1034 default: llvm_unreachable("Unreachable code!");
1035 case 0:
1036 case 1:
1037 if (Filtered || Val == -1)
1038 State = 1; // Still in Water
1039 else {
1040 State = 2; // Into the Island
1041 BitNo = 0;
1042 StartBits.push_back(i);
1043 FieldVal = Val;
1044 }
1045 break;
1046 case 2:
1047 if (Filtered || Val == -1) {
1048 State = 1; // Into the Water
1049 EndBits.push_back(i - 1);
1050 FieldVals.push_back(FieldVal);
1051 ++Num;
1052 } else {
1053 State = 2; // Still in Island
1054 ++BitNo;
1055 FieldVal = FieldVal | Val << BitNo;
1056 }
1057 break;
1058 }
1059 }
1060 // If we are still in Island after the loop, do some housekeeping.
1061 if (State == 2) {
1062 EndBits.push_back(BitWidth - 1);
1063 FieldVals.push_back(FieldVal);
1064 ++Num;
1065 }
1066
1067 assert(StartBits.size() == Num && EndBits.size() == Num &&
1068 FieldVals.size() == Num);
1069 return Num;
1070}

1253void FilterChooser::emitSoftFailTableEntry(DecoderTableInfo &TableInfo,
1254 unsigned Opc) const {
1255 BitsInit *SFBits =
1256 AllInstructions[Opc].EncodingDef->getValueAsBitsInit("SoftFail");
1257 if (!SFBits) return;
1258 BitsInit *InstBits =
1259 AllInstructions[Opc].EncodingDef->getValueAsBitsInit("Inst");
1260
1261 APInt PositiveMask(BitWidth, 0ULL);
1262 APInt NegativeMask(BitWidth, 0ULL);
1263 for (unsigned i = 0; i < BitWidth; ++i) {
1264 bit_value_t B = bitFromBits(*SFBits, i);
1265 bit_value_t IB = bitFromBits(*InstBits, i);
1266
1267 if (B != BIT_TRUE) continue;
1268
1269 switch (IB) {
1270 case BIT_FALSE:
1271 // The bit is meant to be false, so emit a check to see if it is true.
1272 PositiveMask.setBit(i);
1273 break;
1274 case BIT_TRUE:
1275 // The bit is meant to be true, so emit a check to see if it is false.
1276 NegativeMask.setBit(i);
1277 break;
1278 default:
1279 // The bit is not set; this must be an error!
1280 errs() << "SoftFail Conflict: bit SoftFail{" << i << "} in "
1281 << AllInstructions[Opc] << " is set but Inst{" << i
1282 << "} is unset!\n"
1283 << " - You can only mark a bit as SoftFail if it is fully defined"
1284 << " (1/0 - not '?') in Inst\n";
1285 return;
1286 }
1287 }
1288
1289 bool NeedPositiveMask = PositiveMask.getBoolValue();
1290 bool NeedNegativeMask = NegativeMask.getBoolValue();
1291
1292 if (!NeedPositiveMask && !NeedNegativeMask)
1293 return;
1294
1295 TableInfo.Table.push_back(MCD::OPC_SoftFail);
1296
1297 SmallString<16> MaskBytes;
1298 raw_svector_ostream S(MaskBytes);
1299 if (NeedPositiveMask) {
1300 encodeULEB128(PositiveMask.getZExtValue(), S);
1301 for (unsigned i = 0, e = MaskBytes.size(); i != e; ++i)
1302 TableInfo.Table.push_back(MaskBytes[i]);
1303 } else
1304 TableInfo.Table.push_back(0);
1305 if (NeedNegativeMask) {
1306 MaskBytes.clear();
1307 encodeULEB128(NegativeMask.getZExtValue(), S);
1308 for (unsigned i = 0, e = MaskBytes.size(); i != e; ++i)
1309 TableInfo.Table.push_back(MaskBytes[i]);
1310 } else
1311 TableInfo.Table.push_back(0);
1312}

1314// Emits table entries to decode the singleton.
1315void FilterChooser::emitSingletonTableEntry(DecoderTableInfo &TableInfo,
1316 unsigned Opc) const {
1317 std::vector<unsigned> StartBits;
1318 std::vector<unsigned> EndBits;
1319 std::vector<uint64_t> FieldVals;
1320 insn_t Insn;
1321 insnWithID(Insn, Opc);
1322
1323 // Look for islands of undecoded bits of the singleton.
1324 getIslands(StartBits, EndBits, FieldVals, Insn);
1325
1326 unsigned Size = StartBits.size();
1327
1328 // Emit the predicate table entry if one is needed.
1329 emitPredicateTableEntry(TableInfo, Opc);
1330
1331 // Check any additional encoding fields needed.
1332 for (unsigned I = Size; I != 0; --I) {
1333 unsigned NumBits = EndBits[I-1] - StartBits[I-1] + 1;
1334 TableInfo.Table.push_back(MCD::OPC_CheckField);
1335 TableInfo.Table.push_back(StartBits[I-1]);
1336 TableInfo.Table.push_back(NumBits);
1337 uint8_t Buffer[16], *p;
1338 encodeULEB128(FieldVals[I-1], Buffer);
1339 for (p = Buffer; *p >= 128 ; ++p)
1340 TableInfo.Table.push_back(*p);
1341 TableInfo.Table.push_back(*p);
1342 // Push location for NumToSkip backpatching.
1343 TableInfo.FixupStack.back().push_back(TableInfo.Table.size());
1344 // The fixup is always 24-bits, so go ahead and allocate the space
1345 // in the table so all our relative position calculations work OK even
1346 // before we fully resolve the real value here.
1347 TableInfo.Table.push_back(0);
1348 TableInfo.Table.push_back(0);
1349 TableInfo.Table.push_back(0);
1350 }
1351
1352 // Check for soft failure of the match.
1353 emitSoftFailTableEntry(TableInfo, Opc);
1354
1355 bool HasCompleteDecoder;
1356 unsigned DIdx = getDecoderIndex(TableInfo.Decoders, Opc, HasCompleteDecoder);
1357
1358 // Produce OPC_Decode or OPC_TryDecode opcode based on the information
1359 // whether the instruction decoder is complete or not. If it is complete
1360 // then it handles all possible values of remaining variable/unfiltered bits
1361 // and for any value can determine if the bitpattern is a valid instruction
1362 // or not. This means OPC_Decode will be the final step in the decoding
1363 // process. If it is not complete, then the Fail return code from the
1364 // decoder method indicates that additional processing should be done to see
1365 // if there is any other instruction that also matches the bitpattern and
1366 // can decode it.
1367 TableInfo.Table.push_back(HasCompleteDecoder ? MCD::OPC_Decode :
1368 MCD::OPC_TryDecode);
1369 uint8_t Buffer[16], *p;
1370 encodeULEB128(Opc, Buffer);
1371 for (p = Buffer; *p >= 128 ; ++p)
1372 TableInfo.Table.push_back(*p);
1373 TableInfo.Table.push_back(*p);
1374
1375 SmallString<16> Bytes;
1376 raw_svector_ostream S(Bytes);
1377 encodeULEB128(DIdx, S);
1378
1379 // Decoder index
1380 for (unsigned i = 0, e = Bytes.size(); i != e; ++i)
1381 TableInfo.Table.push_back(Bytes[i]);
1382
1383 if (!HasCompleteDecoder) {
1384 // Push location for NumToSkip backpatching.
1385 TableInfo.FixupStack.back().push_back(TableInfo.Table.size());
1386 // Allocate the space for the fixup.
1387 TableInfo.Table.push_back(0);
1388 TableInfo.Table.push_back(0);
1389 TableInfo.Table.push_back(0);
1390 }
1391}

1394void FilterChooser::emitSingletonTableEntry(DecoderTableInfo &TableInfo,
1395 const Filter &Best) const {

1397
1398 // complex singletons need predicate checks from the first singleton
1399 // to refer forward to the variable filterchooser that follows.

1401
1402 emitSingletonTableEntry(TableInfo, Opc);
1403

1408 Best.getVariableFC().emitTableEntries(TableInfo);

1431// FilterProcessor scans the well-known encoding bits of the instructions and
1432// builds up a list of candidate filters. It chooses the best filter and
1433// recursively descends down the decoding tree.
1434bool FilterChooser::filterProcessor(bool AllowMixed, bool Greedy) {
1435 Filters.clear();
1436 BestIndex = -1;
1437 unsigned numInstructions = Opcodes.size();
1438
1439 assert(numInstructions && "Filter created with no instructions");
1440
1441 // No further filtering is necessary.
1442 if (numInstructions == 1)
1443 return true;
1444
1445 // Heuristics. See also doFilter()'s "Heuristics" comment when num of
1446 // instructions is 3.
1447 if (AllowMixed && !Greedy) {
1448 assert(numInstructions == 3);
1449
1450 for (unsigned i = 0; i < Opcodes.size(); ++i) {
1451 std::vector<unsigned> StartBits;
1452 std::vector<unsigned> EndBits;
1453 std::vector<uint64_t> FieldVals;
1454 insn_t Insn;
1455
1456 insnWithID(Insn, Opcodes[i]);
1457
1458 // Look for islands of undecoded bits of any instruction.
1459 if (getIslands(StartBits, EndBits, FieldVals, Insn) > 0) {
1460 // Found an instruction with island(s). Now just assign a filter.
1461 runSingleFilter(StartBits[0], EndBits[0] - StartBits[0] + 1, true);
1462 return true;
1463 }
1464 }
1465 }
1466
1467 unsigned BitIndex;
1468
1469 // We maintain BIT_WIDTH copies of the bitAttrs automaton.
1470 // The automaton consumes the corresponding bit from each
1471 // instruction.
1472 //
1473 // Input symbols: 0, 1, and _ (unset).
1474 // States: NONE, FILTERED, ALL_SET, ALL_UNSET, and MIXED.
1475 // Initial state: NONE.
1476 //
1477 // (NONE) ------- [01] -> (ALL_SET)
1478 // (NONE) ------- _ ----> (ALL_UNSET)
1479 // (ALL_SET) ---- [01] -> (ALL_SET)
1480 // (ALL_SET) ---- _ ----> (MIXED)
1481 // (ALL_UNSET) -- [01] -> (MIXED)
1482 // (ALL_UNSET) -- _ ----> (ALL_UNSET)
1483 // (MIXED) ------ . ----> (MIXED)
1484 // (FILTERED)---- . ----> (FILTERED)
1485
1486 std::vector<bitAttr_t> bitAttrs;
1487
1488 // FILTERED bit positions provide no entropy and are not worthy of pursuing.
1489 // Filter::recurse() set either BIT_TRUE or BIT_FALSE for each position.
1490 for (BitIndex = 0; BitIndex < BitWidth; ++BitIndex)
1491 if (FilterBitValues[BitIndex] == BIT_TRUE ||
1492 FilterBitValues[BitIndex] == BIT_FALSE)
1493 bitAttrs.push_back(ATTR_FILTERED);
1494 else
1495 bitAttrs.push_back(ATTR_NONE);
1496
1497 for (unsigned InsnIndex = 0; InsnIndex < numInstructions; ++InsnIndex) {
1498 insn_t insn;
1499
1500 insnWithID(insn, Opcodes[InsnIndex]);
1501
1502 for (BitIndex = 0; BitIndex < BitWidth; ++BitIndex) {
1503 switch (bitAttrs[BitIndex]) {
1504 case ATTR_NONE:
1505 if (insn[BitIndex] == BIT_UNSET)
1506 bitAttrs[BitIndex] = ATTR_ALL_UNSET;
1507 else
1508 bitAttrs[BitIndex] = ATTR_ALL_SET;
1509 break;
1510 case ATTR_ALL_SET:
1511 if (insn[BitIndex] == BIT_UNSET)
1512 bitAttrs[BitIndex] = ATTR_MIXED;
1513 break;
1514 case ATTR_ALL_UNSET:
1515 if (insn[BitIndex] != BIT_UNSET)
1516 bitAttrs[BitIndex] = ATTR_MIXED;
1517 break;
1518 case ATTR_MIXED:
1519 case ATTR_FILTERED:
1520 break;
1521 }
1522 }
1523 }
1524
1525 // The regionAttr automaton consumes the bitAttrs automatons' state,
1526 // lowest-to-highest.
1527 //
1528 // Input symbols: F(iltered), (all_)S(et), (all_)U(nset), M(ixed)
1529 // States: NONE, ALL_SET, MIXED
1530 // Initial state: NONE
1531 //
1532 // (NONE) ----- F --> (NONE)
1533 // (NONE) ----- S --> (ALL_SET) ; and set region start
1534 // (NONE) ----- U --> (NONE)
1535 // (NONE) ----- M --> (MIXED) ; and set region start
1536 // (ALL_SET) -- F --> (NONE) ; and report an ALL_SET region
1537 // (ALL_SET) -- S --> (ALL_SET)
1538 // (ALL_SET) -- U --> (NONE) ; and report an ALL_SET region
1539 // (ALL_SET) -- M --> (MIXED) ; and report an ALL_SET region
1540 // (MIXED) ---- F --> (NONE) ; and report a MIXED region
1541 // (MIXED) ---- S --> (ALL_SET) ; and report a MIXED region
1542 // (MIXED) ---- U --> (NONE) ; and report a MIXED region
1543 // (MIXED) ---- M --> (MIXED)
1544
1545 bitAttr_t RA = ATTR_NONE;
1546 unsigned StartBit = 0;
1547
1548 for (BitIndex = 0; BitIndex < BitWidth; ++BitIndex) {
1549 bitAttr_t bitAttr = bitAttrs[BitIndex];
1550
1551 assert(bitAttr != ATTR_NONE && "Bit without attributes");
1552
1553 switch (RA) {
1554 case ATTR_NONE:
1555 switch (bitAttr) {
1556 case ATTR_FILTERED:
1557 break;
1558 case ATTR_ALL_SET:
1559 StartBit = BitIndex;
1560 RA = ATTR_ALL_SET;
1561 break;
1562 case ATTR_ALL_UNSET:
1563 break;
1564 case ATTR_MIXED:
1565 StartBit = BitIndex;
1566 RA = ATTR_MIXED;
1567 break;
1568 default:
1569 llvm_unreachable("Unexpected bitAttr!");
1570 }
1571 break;
1572 case ATTR_ALL_SET:
1573 switch (bitAttr) {
1574 case ATTR_FILTERED:
1575 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1576 RA = ATTR_NONE;
1577 break;
1578 case ATTR_ALL_SET:
1579 break;
1580 case ATTR_ALL_UNSET:
1581 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1582 RA = ATTR_NONE;
1583 break;
1584 case ATTR_MIXED:
1585 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1586 StartBit = BitIndex;
1587 RA = ATTR_MIXED;
1588 break;
1589 default:
1590 llvm_unreachable("Unexpected bitAttr!");
1591 }
1592 break;
1593 case ATTR_MIXED:
1594 switch (bitAttr) {
1595 case ATTR_FILTERED:
1596 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1597 StartBit = BitIndex;
1598 RA = ATTR_NONE;
1599 break;
1600 case ATTR_ALL_SET:
1601 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1602 StartBit = BitIndex;
1603 RA = ATTR_ALL_SET;
1604 break;
1605 case ATTR_ALL_UNSET:
1606 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1607 RA = ATTR_NONE;
1608 break;
1609 case ATTR_MIXED:
1610 break;
1611 default:
1612 llvm_unreachable("Unexpected bitAttr!");
1613 }
1614 break;
1615 case ATTR_ALL_UNSET:
1616 llvm_unreachable("regionAttr state machine has no ATTR_UNSET state");
1617 case ATTR_FILTERED:
1618 llvm_unreachable("regionAttr state machine has no ATTR_FILTERED state");
1619 }
1620 }
1621
1622 // At the end, if we're still in ALL_SET or MIXED states, report a region
1623 switch (RA) {
1624 case ATTR_NONE:
1625 break;
1626 case ATTR_FILTERED:
1627 break;
1628 case ATTR_ALL_SET:
1629 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1630 break;
1631 case ATTR_ALL_UNSET:
1632 break;
1633 case ATTR_MIXED:
1634 reportRegion(RA, StartBit, BitIndex, AllowMixed);
1635 break;
1636 }
1637
1638 // We have finished with the filter processings. Now it's time to choose
1639 // the best performing filter.
1640 BestIndex = 0;
1641 bool AllUseless = true;
1642 unsigned BestScore = 0;
1643
1644 for (unsigned i = 0, e = Filters.size(); i != e; ++i) {
1645 unsigned Usefulness = Filters[i].usefulness();
1646
1647 if (Usefulness)
1648 AllUseless = false;
1649
1650 if (Usefulness > BestScore) {
1651 BestIndex = i;
1652 BestScore = Usefulness;
1653 }
1654 }
1655
1656 if (!AllUseless)
1657 bestFilter().recurse();
1658
1659 return !AllUseless;
1660} // end of FilterChooser::filterProcessor(bool)

1662// Decides on the best configuration of filter(s) to use in order to decode
1663// the instructions. A conflict of instructions may occur, in which case we
1664// dump the conflict set to the standard error.
1665void FilterChooser::doFilter() {
1666 unsigned Num = Opcodes.size();
1667 assert(Num && "FilterChooser created with no instructions");
1668
1669 // Try regions of consecutive known bit values first.
1670 if (filterProcessor(false))
1671 return;
1672
1673 // Then regions of mixed bits (both known and unitialized bit values allowed).
1674 if (filterProcessor(true))
1675 return;
1676
1677 // Heuristics to cope with conflict set {t2CMPrs, t2SUBSrr, t2SUBSrs} where
1678 // no single instruction for the maximum ATTR_MIXED region Inst{14-4} has a
1679 // well-known encoding pattern. In such case, we backtrack and scan for the
1680 // the very first consecutive ATTR_ALL_SET region and assign a filter to it.
1681 if (Num == 3 && filterProcessor(true, false))
1682 return;
1683
1684 // If we come to here, the instruction decoding has failed.
1685 // Set the BestIndex to -1 to indicate so.
1686 BestIndex = -1;
1687}

1689// emitTableEntries - Emit state machine entries to decode our share of
1690// instructions.
1691void FilterChooser::emitTableEntries(DecoderTableInfo &TableInfo) const {
1692 if (Opcodes.size() == 1) {
1693 // There is only one instruction in the set, which is great!
1694 // Call emitSingletonDecoder() to see whether there are any remaining
1695 // encodings bits.
1696 emitSingletonTableEntry(TableInfo, Opcodes[0]);
1697 return;
1698 }
1699
1700 // Choose the best filter to do the decodings!
1701 if (BestIndex != -1) {
1702 const Filter &Best = Filters[BestIndex];
1703 if (Best.getNumFiltered() == 1)
1704 emitSingletonTableEntry(TableInfo, Best);
1705 else
1706 Best.emitTableEntry(TableInfo);
1707 return;
1708 }
1709
1710 // We don't know how to decode these instructions! Dump the
1711 // conflict set and bail.
1712
1713 // Print out useful conflict information for postmortem analysis.
1714 errs() << "Decoding Conflict:\n";
1715
1716 dumpStack(errs(), "\t\t");
1717
1718 for (unsigned i = 0; i < Opcodes.size(); ++i) {
1719 errs() << '\t' << AllInstructions[Opcodes[i]] << " ";
1720 dumpBits(errs(),
1721 getBitsField(*AllInstructions[Opcodes[i]].EncodingDef, "Inst"));
1722 errs() << '\n';
1723 }
1724}

1753static bool populateInstruction(CodeGenTarget &Target,
1754 const CodeGenInstruction &CGI, unsigned Opc,
1755 std::map<unsigned, std::vector<OperandInfo>> &Operands){
1756 const Record &Def = *CGI.TheDef;
1757 // If all the bit positions are not specified; do not decode this instruction.
1758 // We are bound to fail! For proper disassembly, the well-known encoding bits
1759 // of the instruction must be fully specified.
1760
1761 BitsInit &Bits = getBitsField(Def, "Inst");
1762 if (Bits.allInComplete()) return false;
1763
1764 std::vector<OperandInfo> InsnOperands;
1765
1766 // If the instruction has specified a custom decoding hook, use that instead
1767 // of trying to auto-generate the decoder.
1768 StringRef InstDecoder = Def.getValueAsString("DecoderMethod");
1769 if (InstDecoder != "") {
1770 bool HasCompleteInstDecoder = Def.getValueAsBit("hasCompleteDecoder");
1771 InsnOperands.push_back(OperandInfo(InstDecoder, HasCompleteInstDecoder));
1772 Operands[Opc] = InsnOperands;
1773 return true;
1774 }
1775
1776 // Generate a description of the operand of the instruction that we know
1777 // how to decode automatically.
1778 // FIXME: We'll need to have a way to manually override this as needed.
1779
1780 // Gather the outputs/inputs of the instruction, so we can find their
1781 // positions in the encoding. This assumes for now that they appear in the
1782 // MCInst in the order that they're listed.
1783 std::vector<std::pair<Init*, StringRef>> InOutOperands;
1784 DagInit *Out = Def.getValueAsDag("OutOperandList");
1785 DagInit *In = Def.getValueAsDag("InOperandList");
1786 for (unsigned i = 0; i < Out->getNumArgs(); ++i)
1787 InOutOperands.push_back(std::make_pair(Out->getArg(i),
1788 Out->getArgNameStr(i)));
1789 for (unsigned i = 0; i < In->getNumArgs(); ++i)
1790 InOutOperands.push_back(std::make_pair(In->getArg(i),
1791 In->getArgNameStr(i)));
1792
1793 // Search for tied operands, so that we can correctly instantiate
1794 // operands that are not explicitly represented in the encoding.
1795 std::map<std::string, std::string> TiedNames;
1796 for (unsigned i = 0; i < CGI.Operands.size(); ++i) {
1797 int tiedTo = CGI.Operands[i].getTiedRegister();
1798 if (tiedTo != -1) {
1799 std::pair<unsigned, unsigned> SO =
1800 CGI.Operands.getSubOperandNumber(tiedTo);
1801 TiedNames[InOutOperands[i].second] = InOutOperands[SO.first].second;
1802 TiedNames[InOutOperands[SO.first].second] = InOutOperands[i].second;
1803 }
1804 }
1805
1806 std::map<std::string, std::vector<OperandInfo>> NumberedInsnOperands;
1807 std::set<std::string> NumberedInsnOperandsNoTie;
1808 if (Target.getInstructionSet()->
1809 getValueAsBit("decodePositionallyEncodedOperands")) {
1810 const std::vector<RecordVal> &Vals = Def.getValues();
1811 unsigned NumberedOp = 0;
1812
1813 std::set<unsigned> NamedOpIndices;
1814 if (Target.getInstructionSet()->
1815 getValueAsBit("noNamedPositionallyEncodedOperands"))
1816 // Collect the set of operand indices that might correspond to named
1817 // operand, and skip these when assigning operands based on position.
1818 for (unsigned i = 0, e = Vals.size(); i != e; ++i) {
1819 unsigned OpIdx;
1820 if (!CGI.Operands.hasOperandNamed(Vals[i].getName(), OpIdx))
1821 continue;
1822
1823 NamedOpIndices.insert(OpIdx);
1824 }
1825
1826 for (unsigned i = 0, e = Vals.size(); i != e; ++i) {
1827 // Ignore fixed fields in the record, we're looking for values like:
1828 // bits<5> RST = { ?, ?, ?, ?, ? };
1829 if (Vals[i].getPrefix() || Vals[i].getValue()->isComplete())
1830 continue;
1831
1832 // Determine if Vals[i] actually contributes to the Inst encoding.
1833 unsigned bi = 0;
1834 for (; bi < Bits.getNumBits(); ++bi) {
1835 VarInit *Var = nullptr;
1836 VarBitInit *BI = dyn_cast<VarBitInit>(Bits.getBit(bi));
1837 if (BI)
1838 Var = dyn_cast<VarInit>(BI->getBitVar());
1839 else
1840 Var = dyn_cast<VarInit>(Bits.getBit(bi));
1841
1842 if (Var && Var->getName() == Vals[i].getName())
1843 break;
1844 }
1845
1846 if (bi == Bits.getNumBits())
1847 continue;
1848
1849 // Skip variables that correspond to explicitly-named operands.
1850 unsigned OpIdx;
1851 if (CGI.Operands.hasOperandNamed(Vals[i].getName(), OpIdx))
1852 continue;
1853
1854 // Get the bit range for this operand:
1855 unsigned bitStart = bi++, bitWidth = 1;
1856 for (; bi < Bits.getNumBits(); ++bi) {
1857 VarInit *Var = nullptr;
1858 VarBitInit *BI = dyn_cast<VarBitInit>(Bits.getBit(bi));
1859 if (BI)
1860 Var = dyn_cast<VarInit>(BI->getBitVar());
1861 else
1862 Var = dyn_cast<VarInit>(Bits.getBit(bi));
1863
1864 if (!Var)
1865 break;
1866
1867 if (Var->getName() != Vals[i].getName())
1868 break;
1869
1870 ++bitWidth;
1871 }
1872
1873 unsigned NumberOps = CGI.Operands.size();
1874 while (NumberedOp < NumberOps &&
1875 (CGI.Operands.isFlatOperandNotEmitted(NumberedOp) ||
1876 (!NamedOpIndices.empty() && NamedOpIndices.count(
1877 CGI.Operands.getSubOperandNumber(NumberedOp).first))))
1878 ++NumberedOp;
1879
1880 OpIdx = NumberedOp++;
1881
1882 // OpIdx now holds the ordered operand number of Vals[i].
1883 std::pair<unsigned, unsigned> SO =
1884 CGI.Operands.getSubOperandNumber(OpIdx);
1885 const std::string &Name = CGI.Operands[SO.first].Name;
1886
1887 LLVM_DEBUG(dbgs() << "Numbered operand mapping for " << Def.getName()
1888 << ": " << Name << "(" << SO.first << ", " << SO.second
1889 << ") => " << Vals[i].getName() << "\n");
1890
1891 std::string Decoder;
1892 Record *TypeRecord = CGI.Operands[SO.first].Rec;
1893
1894 RecordVal *DecoderString = TypeRecord->getValue("DecoderMethod");
1895 StringInit *String = DecoderString ?
1896 dyn_cast<StringInit>(DecoderString->getValue()) : nullptr;
1897 if (String && String->getValue() != "")
1898 Decoder = String->getValue();
1899
1900 if (Decoder == "" &&
1901 CGI.Operands[SO.first].MIOperandInfo &&
1902 CGI.Operands[SO.first].MIOperandInfo->getNumArgs()) {
1903 Init *Arg = CGI.Operands[SO.first].MIOperandInfo->
1904 getArg(SO.second);
1905 if (DefInit *DI = cast<DefInit>(Arg))
1906 TypeRecord = DI->getDef();
1907 }
1908
1909 bool isReg = false;
1910 if (TypeRecord->isSubClassOf("RegisterOperand"))
1911 TypeRecord = TypeRecord->getValueAsDef("RegClass");
1912 if (TypeRecord->isSubClassOf("RegisterClass")) {
1913 Decoder = "Decode" + TypeRecord->getName().str() + "RegisterClass";
1914 isReg = true;
1915 } else if (TypeRecord->isSubClassOf("PointerLikeRegClass")) {
1916 Decoder = "DecodePointerLikeRegClass" +
1917 utostr(TypeRecord->getValueAsInt("RegClassKind"));
1918 isReg = true;
1919 }
1920
1921 DecoderString = TypeRecord->getValue("DecoderMethod");
1922 String = DecoderString ?
1923 dyn_cast<StringInit>(DecoderString->getValue()) : nullptr;
1924 if (!isReg && String && String->getValue() != "")
1925 Decoder = String->getValue();
1926
1927 RecordVal *HasCompleteDecoderVal =
1928 TypeRecord->getValue("hasCompleteDecoder");
1929 BitInit *HasCompleteDecoderBit = HasCompleteDecoderVal ?
1930 dyn_cast<BitInit>(HasCompleteDecoderVal->getValue()) : nullptr;
1931 bool HasCompleteDecoder = HasCompleteDecoderBit ?
1932 HasCompleteDecoderBit->getValue() : true;
1933
1934 OperandInfo OpInfo(Decoder, HasCompleteDecoder);
1935 OpInfo.addField(bitStart, bitWidth, 0);
1936
1937 NumberedInsnOperands[Name].push_back(OpInfo);
1938
1939 // FIXME: For complex operands with custom decoders we can't handle tied
1940 // sub-operands automatically. Skip those here and assume that this is
1941 // fixed up elsewhere.
1942 if (CGI.Operands[SO.first].MIOperandInfo &&
1943 CGI.Operands[SO.first].MIOperandInfo->getNumArgs() > 1 &&
1944 String && String->getValue() != "")
1945 NumberedInsnOperandsNoTie.insert(Name);
1946 }
1947 }
1948
1949 // For each operand, see if we can figure out where it is encoded.
1950 for (const auto &Op : InOutOperands) {
1951 if (!NumberedInsnOperands[Op.second].empty()) {
1952 InsnOperands.insert(InsnOperands.end(),
1953 NumberedInsnOperands[Op.second].begin(),
1954 NumberedInsnOperands[Op.second].end());
1955 continue;
1956 }
1957 if (!NumberedInsnOperands[TiedNames[Op.second]].empty()) {
1958 if (!NumberedInsnOperandsNoTie.count(TiedNames[Op.second])) {
1959 // Figure out to which (sub)operand we're tied.
1960 unsigned i = CGI.Operands.getOperandNamed(TiedNames[Op.second]);
1961 int tiedTo = CGI.Operands[i].getTiedRegister();
1962 if (tiedTo == -1) {
1963 i = CGI.Operands.getOperandNamed(Op.second);
1964 tiedTo = CGI.Operands[i].getTiedRegister();
1965 }
1966
1967 if (tiedTo != -1) {
1968 std::pair<unsigned, unsigned> SO =
1969 CGI.Operands.getSubOperandNumber(tiedTo);
1970

1972 [SO.second]);
1973 }
1974 }
1975 continue;
1976 }
1977
1978 TypedInit *TI = cast<TypedInit>(Op.first);
1979
1980 // At this point, we can locate the decoder field, but we need to know how
1981 // to interpret it. As a first step, require the target to provide
1982 // callbacks for decoding register classes.
1983 std::string Decoder = findOperandDecoderMethod(TI);
1984 Record *TypeRecord = cast<DefInit>(TI)->getDef();
1985
1986 RecordVal *HasCompleteDecoderVal =
1987 TypeRecord->getValue("hasCompleteDecoder");
1988 BitInit *HasCompleteDecoderBit = HasCompleteDecoderVal ?
1989 dyn_cast<BitInit>(HasCompleteDecoderVal->getValue()) : nullptr;
1990 bool HasCompleteDecoder = HasCompleteDecoderBit ?
1991 HasCompleteDecoderBit->getValue() : true;
1992
1993 OperandInfo OpInfo(Decoder, HasCompleteDecoder);
1994 unsigned Base = ~0U;
1995 unsigned Width = 0;
1996 unsigned Offset = 0;
1997
1998 for (unsigned bi = 0; bi < Bits.getNumBits(); ++bi) {
1999 VarInit *Var = nullptr;
2000 VarBitInit *BI = dyn_cast<VarBitInit>(Bits.getBit(bi));
2001 if (BI)
2002 Var = dyn_cast<VarInit>(BI->getBitVar());
2003 else
2004 Var = dyn_cast<VarInit>(Bits.getBit(bi));
2005
2006 if (!Var) {
2007 if (Base != ~0U) {
2008 OpInfo.addField(Base, Width, Offset);
2009 Base = ~0U;
2010 Width = 0;
2011 Offset = 0;
2012 }
2013 continue;
2014 }
2015
2016 if (Var->getName() != Op.second &&
2017 Var->getName() != TiedNames[Op.second]) {
2018 if (Base != ~0U) {
2019 OpInfo.addField(Base, Width, Offset);
2020 Base = ~0U;
2021 Width = 0;
2022 Offset = 0;
2023 }
2024 continue;
2025 }
2026
2027 if (Base == ~0U) {
2028 Base = bi;
2029 Width = 1;
2030 Offset = BI ? BI->getBitNum() : 0;
2031 } else if (BI && BI->getBitNum() != Offset + Width) {
2032 OpInfo.addField(Base, Width, Offset);
2033 Base = bi;
2034 Width = 1;
2035 Offset = BI->getBitNum();
2036 } else {
2037 ++Width;
2038 }
2039 }
2040
2041 if (Base != ~0U)
2042 OpInfo.addField(Base, Width, Offset);
2043
2044 if (OpInfo.numFields() > 0)
2045 InsnOperands.push_back(OpInfo);
2046 }
2047
2048 Operands[Opc] = InsnOperands;
2049
2050#if 0
2051 LLVM_DEBUG({
2052 // Dumps the instruction encoding bits.
2053 dumpBits(errs(), Bits);
2054
2055 errs() << '\n';
2056
2057 // Dumps the list of operand info.
2058 for (unsigned i = 0, e = CGI.Operands.size(); i != e; ++i) {
2059 const CGIOperandList::OperandInfo &Info = CGI.Operands[i];
2060 const std::string &OperandName = Info.Name;
2061 const Record &OperandDef = *Info.Rec;
2062
2063 errs() << "\t" << OperandName << " (" << OperandDef.getName() << ")\n";
2064 }
2065 });
2066#endif
2067
2068 return true;
2069}

2320void FixedLenDecoderEmitter::run(raw_ostream &o) {
2321 formatted_raw_ostream OS(o);
2322 OS << "#include \"llvm/MC/MCInst.h\"\n";
2323 OS << "#include \"llvm/Support/Debug.h\"\n";
2324 OS << "#include \"llvm/Support/DataTypes.h\"\n";
2325 OS << "#include \"llvm/Support/LEB128.h\"\n";
2326 OS << "#include \"llvm/Support/raw_ostream.h\"\n";
2327 OS << "#include <assert.h>\n";
2328 OS << '\n';
2329 OS << "namespace llvm {\n\n";
2330
2331 emitFieldFromInstruction(OS);
2332
2333 Target.reverseBitsForLittleEndianEncoding();
2334
2335 // Parameterize the decoders based on namespace and instruction width.
2336 const auto &NumberedInstructions = Target.getInstructionsByEnumValue();
2337 NumberedEncodings.reserve(NumberedInstructions.size());
2338 for (const auto &NumberedInstruction : NumberedInstructions)
2339 NumberedEncodings.emplace_back(NumberedInstruction->TheDef, NumberedInstruction);
2340
2341 std::map<std::pair<std::string, unsigned>,
2342 std::vector<unsigned>> OpcMap;
2343 std::map<unsigned, std::vector<OperandInfo>> Operands;
2344
2345 for (unsigned i = 0; i < NumberedEncodings.size(); ++i) {
2346 const CodeGenInstruction *Inst = NumberedEncodings[i].Inst;
2347 const Record *Def = Inst->TheDef;
2348 unsigned Size = Def->getValueAsInt("Size");
2349 if (Def->getValueAsString("Namespace") == "TargetOpcode" ||
2350 Def->getValueAsBit("isPseudo") ||
2351 Def->getValueAsBit("isAsmParserOnly") ||
2352 Def->getValueAsBit("isCodeGenOnly"))
2353 continue;
2354
2355 StringRef DecoderNamespace = Def->getValueAsString("DecoderNamespace");
2356
2357 if (Size) {
2358 if (populateInstruction(Target, *Inst, i, Operands)) {
2359 OpcMap[std::make_pair(DecoderNamespace, Size)].push_back(i);
2360 }
2361 }
2362 }
2363
2364 DecoderTableInfo TableInfo;
2365 for (const auto &Opc : OpcMap) {
2366 // Emit the decoder for this namespace+width combination.
2367 ArrayRef<EncodingAndInst> NumberedEncodingsRef(NumberedEncodings.data(),
2368 NumberedEncodings.size());
2369 FilterChooser FC(NumberedEncodingsRef, Opc.second, Operands,
2370 8 * Opc.first.second, this);
2371
2372 // The decode table is cleared for each top level decoder function. The
2373 // predicates and decoders themselves, however, are shared across all
2374 // decoders to give more opportunities for uniqueing.
2375 TableInfo.Table.clear();
2376 TableInfo.FixupStack.clear();
2377 TableInfo.Table.reserve(16384);
2378 TableInfo.FixupStack.emplace_back();
2379 FC.emitTableEntries(TableInfo);
2380 // Any NumToSkip fixups in the top level scope can resolve to the
2381 // OPC_Fail at the end of the table.
2382 assert(TableInfo.FixupStack.size() == 1 && "fixup stack phasing error!");
2383 // Resolve any NumToSkip fixups in the current scope.
2384 resolveTableFixups(TableInfo.Table, TableInfo.FixupStack.back(),
2385 TableInfo.Table.size());
2386 TableInfo.FixupStack.clear();
2387
2388 TableInfo.Table.push_back(MCD::OPC_Fail);
2389
2390 // Print the table to the output stream.
2391 emitTable(OS, TableInfo.Table, 0, FC.getBitWidth(), Opc.first.first);
2392 OS.flush();
2393 }
2394
2395 // Emit the predicate function.
2396 emitPredicateFunction(OS, TableInfo.Predicates, 0);
2397
2398 // Emit the decoder function.
2399 emitDecoderFunction(OS, TableInfo.Decoders, 0);
2400
2401 // Emit the main entry point for the decoder, decodeInstruction().
2402 emitDecodeInstruction(OS);
2403
2404 OS << "\n} // End llvm namespace\n";
2405}
