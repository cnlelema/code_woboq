21class TargetLibraryInfoTest : public testing::Test {
22protected:
23 LLVMContext Context;
24 TargetLibraryInfoImpl TLII;
25 TargetLibraryInfo TLI;
26
27 std::unique_ptr<Module> M;
28
29 TargetLibraryInfoTest() : TLI(TLII) {}
30
31 void parseAssembly(const char *Assembly) {
32 SMDiagnostic Error;
33 M = parseAssemblyString(Assembly, Error, Context);
34
35 std::string errMsg;
36 raw_string_ostream os(errMsg);
37 Error.print("", os);
38
39 if (!M)
40 report_fatal_error(os.str());
41 }
42
43 ::testing::AssertionResult isLibFunc(const Function *FDecl,
44 LibFunc ExpectedLF) {
45 StringRef ExpectedLFName = TLI.getName(ExpectedLF);
46
47 if (!FDecl)
48 return ::testing::AssertionFailure() << ExpectedLFName << " not found";
49
50 LibFunc F;
51 if (!TLI.getLibFunc(*FDecl, F))
52 return ::testing::AssertionFailure() << ExpectedLFName << " invalid";
53
54 return ::testing::AssertionSuccess() << ExpectedLFName << " is LibFunc";
55 }
56};
57
58} // end anonymous namespace
59
60// Check that we don't accept egregiously incorrect prototypes.
61TEST_F(TargetLibraryInfoTest, InvalidProto) {
62 parseAssembly("%foo = type { %foo }\n");
63
64 auto *StructTy = M->getTypeByName("foo");
65 auto *InvalidFTy = FunctionType::get(StructTy, /*isVarArg=*/false);
66
67 for (unsigned FI = 0; FI != LibFunc::NumLibFuncs; ++FI) {
68 LibFunc LF = (LibFunc)FI;
69 auto *F = cast<Function>(
70 M->getOrInsertFunction(TLI.getName(LF), InvalidFTy).getCallee());
71 EXPECT_FALSE(isLibFunc(F, LF));
72 }
73}
