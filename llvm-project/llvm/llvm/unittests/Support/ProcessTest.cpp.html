21TEST(ProcessTest, GetRandomNumberTest) {
22 const unsigned r1 = Process::GetRandomNumber();
23 const unsigned r2 = Process::GetRandomNumber();
24 // It should be extremely unlikely that both r1 and r2 are 0.
25 EXPECT_NE((r1 | r2), 0u);
26}

28#ifdef _MSC_VER
29#define setenv(name, var, ignore) _putenv_s(name, var)
30#endif
31
32#if HAVE_SETENV || _MSC_VER
33TEST(ProcessTest, Basic) {
34 setenv("__LLVM_TEST_ENVIRON_VAR__", "abc", true);
35 Optional<std::string> val(Process::GetEnv("__LLVM_TEST_ENVIRON_VAR__"));
36 EXPECT_TRUE(val.hasValue());
37 EXPECT_STREQ("abc", val->c_str());
38}
39
40TEST(ProcessTest, None) {
41 Optional<std::string> val(
42 Process::GetEnv("__LLVM_TEST_ENVIRON_NO_SUCH_VAR__"));
43 EXPECT_FALSE(val.hasValue());
44}
45#endif
46
47#ifdef _WIN32
48
49TEST(ProcessTest, EmptyVal) {
50 SetEnvironmentVariableA("__LLVM_TEST_ENVIRON_VAR__", "");
51 Optional<std::string> val(Process::GetEnv("__LLVM_TEST_ENVIRON_VAR__"));
52 EXPECT_TRUE(val.hasValue());
53 EXPECT_STREQ("", val->c_str());
54}
55
56TEST(ProcessTest, Wchar) {
57 SetEnvironmentVariableW(L"__LLVM_TEST_ENVIRON_VAR__", L"abcdefghijklmnopqrs");
58 Optional<std::string> val(Process::GetEnv("__LLVM_TEST_ENVIRON_VAR__"));
59 EXPECT_TRUE(val.hasValue());
60 EXPECT_STREQ("abcdefghijklmnopqrs", val->c_str());
61}
