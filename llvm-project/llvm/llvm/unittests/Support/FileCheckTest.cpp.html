17TEST_F(FileCheckTest, NumericVariable) {
18 // Undefined variable: getValue and clearValue fails, setValue works.
19 FileCheckNumericVariable FooVar = FileCheckNumericVariable(1, "FOO");
20 EXPECT_EQ("FOO", FooVar.getName());
21 llvm::Optional<uint64_t> Value = FooVar.getValue();
22 EXPECT_FALSE(Value);
23 EXPECT_TRUE(FooVar.clearValue());
24 EXPECT_FALSE(FooVar.setValue(42));
25
26 // Defined variable: getValue returns value set, setValue fails.
27 Value = FooVar.getValue();
28 EXPECT_TRUE(Value);
29 EXPECT_EQ(42U, *Value);
30 EXPECT_TRUE(FooVar.setValue(43));
31 Value = FooVar.getValue();
32 EXPECT_TRUE(Value);
33 EXPECT_EQ(42U, *Value);
34
35 // Clearing variable: getValue fails, clearValue again fails.
36 EXPECT_FALSE(FooVar.clearValue());
37 Value = FooVar.getValue();
38 EXPECT_FALSE(Value);
39 EXPECT_TRUE(FooVar.clearValue());
40}

42uint64_t doAdd(uint64_t OpL, uint64_t OpR) { return OpL + OpR; }

44TEST_F(FileCheckTest, NumExpr) {
45 FileCheckNumericVariable FooVar = FileCheckNumericVariable("FOO", 42);
46 FileCheckNumExpr NumExpr = FileCheckNumExpr(doAdd, &FooVar, 18);
47
48 // Defined variable: eval returns right value, no undefined variable
49 // returned.
50 llvm::Optional<uint64_t> Value = NumExpr.eval();
51 EXPECT_TRUE(Value);
52 EXPECT_EQ(60U, *Value);
53 StringRef UndefVar = NumExpr.getUndefVarName();
54 EXPECT_EQ("", UndefVar);
55
56 // Undefined variable: eval fails, undefined variable returned. We call
57 // getUndefVarName first to check that it can be called without calling
58 // eval() first.
59 FooVar.clearValue();
60 UndefVar = NumExpr.getUndefVarName();
61 EXPECT_EQ("FOO", UndefVar);
62 Value = NumExpr.eval();
63 EXPECT_FALSE(Value);
64}

78TEST_F(FileCheckTest, ParseVar) {
79 StringRef OrigVarName = "GoodVar42";
80 StringRef VarName = OrigVarName;
81 StringRef ParsedName;
82 bool IsPseudo = true;
83 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
84 EXPECT_EQ(ParsedName, OrigVarName);
85 EXPECT_TRUE(VarName.empty());
86 EXPECT_FALSE(IsPseudo);
87
88 VarName = OrigVarName = "$GoodGlobalVar";
89 IsPseudo = true;
90 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
91 EXPECT_EQ(ParsedName, OrigVarName);
92 EXPECT_TRUE(VarName.empty());
93 EXPECT_FALSE(IsPseudo);
94
95 VarName = OrigVarName = "@GoodPseudoVar";
96 IsPseudo = true;
97 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
98 EXPECT_EQ(ParsedName, OrigVarName);
99 EXPECT_TRUE(VarName.empty());
100 EXPECT_TRUE(IsPseudo);
101
102 VarName = "42BadVar";
103 EXPECT_TRUE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
104
105 VarName = "$@";
106 EXPECT_TRUE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
107
108 VarName = OrigVarName = "B@dVar";
109 IsPseudo = true;
110 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
111 EXPECT_EQ(VarName, OrigVarName.substr(1));
112 EXPECT_EQ(ParsedName, "B");
113 EXPECT_FALSE(IsPseudo);
114
115 VarName = OrigVarName = "B$dVar";
116 IsPseudo = true;
117 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
118 EXPECT_EQ(VarName, OrigVarName.substr(1));
119 EXPECT_EQ(ParsedName, "B");
120 EXPECT_FALSE(IsPseudo);
121
122 VarName = "BadVar+";
123 IsPseudo = true;
124 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
125 EXPECT_EQ(VarName, "+");
126 EXPECT_EQ(ParsedName, "BadVar");
127 EXPECT_FALSE(IsPseudo);
128
129 VarName = "BadVar-";
130 IsPseudo = true;
131 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
132 EXPECT_EQ(VarName, "-");
133 EXPECT_EQ(ParsedName, "BadVar");
134 EXPECT_FALSE(IsPseudo);
135
136 VarName = "BadVar:";
137 IsPseudo = true;
138 EXPECT_FALSE(FileCheckPattern::parseVariable(VarName, ParsedName, IsPseudo));
139 EXPECT_EQ(VarName, ":");
140 EXPECT_EQ(ParsedName, "BadVar");
141 EXPECT_FALSE(IsPseudo);
142}

144static StringRef bufferize(SourceMgr &SM, StringRef Str) {
145 std::unique_ptr<MemoryBuffer> Buffer =
146 MemoryBuffer::getMemBufferCopy(Str, "TestBuffer");
147 StringRef StrBufferRef = Buffer->getBuffer();
148 SM.AddNewSourceBuffer(std::move(Buffer), SMLoc());
149 return StrBufferRef;
150}

152class PatternTester {
153private:
154 size_t LineNumber = 1;
155 SourceMgr SM;
156 FileCheckRequest Req;
157 FileCheckPatternContext Context;
158 FileCheckPattern P =
159 FileCheckPattern(Check::CheckPlain, &Context, LineNumber++);
160
161public:
162 PatternTester() {
163 std::vector<std::string> GlobalDefines;
164 GlobalDefines.emplace_back(std::string("#FOO=42"));
165 GlobalDefines.emplace_back(std::string("BAR=BAZ"));
166 Context.defineCmdlineVariables(GlobalDefines, SM);
167 // Call parsePattern to have @LINE defined.
168 P.parsePattern("N/A", "CHECK", SM, Req);
169 // parsePattern does not expect to be called twice for the same line and
170 // will set FixedStr and RegExStr incorrectly if it is. Therefore prepare
171 // a pattern for a different line.
172 initNextPattern();
173 }
174
175 void initNextPattern() {
176 P = FileCheckPattern(Check::CheckPlain, &Context, LineNumber++);
177 }
178
179 bool parseNumVarDefExpect(StringRef Expr) {
180 StringRef ExprBufferRef = bufferize(SM, Expr);
181 StringRef Name;
182 return FileCheckPattern::parseNumericVariableDefinition(ExprBufferRef, Name,
183 &Context, SM);
184 }
185
186 bool parseSubstExpect(StringRef Expr) {
187 StringRef ExprBufferRef = bufferize(SM, Expr);
188 FileCheckNumericVariable *DefinedNumericVariable;
189 return P.parseNumericSubstitutionBlock(
190 ExprBufferRef, DefinedNumericVariable, SM) == nullptr;
191 }
192
193 bool parsePatternExpect(StringRef Pattern) {
194 StringRef PatBufferRef = bufferize(SM, Pattern);
195 return P.parsePattern(PatBufferRef, "CHECK", SM, Req);
196 }
197
198 bool matchExpect(StringRef Buffer) {
199 StringRef BufferRef = bufferize(SM, Buffer);
200 size_t MatchLen;
201 return P.match(BufferRef, MatchLen, SM);
202 }
203};

307TEST_F(FileCheckTest, Match) {
308 PatternTester Tester;
309
310 // Check matching a definition only matches a number.
311 Tester.parsePatternExpect("[[#NUMVAR:]]");
312 EXPECT_TRUE(Tester.matchExpect("FAIL"));
313 EXPECT_FALSE(Tester.matchExpect("18"));
314
315 // Check matching the variable defined matches the correct number only
316 Tester.initNextPattern();
317 Tester.parsePatternExpect("[[#NUMVAR]] [[#NUMVAR+2]]");
318 EXPECT_TRUE(Tester.matchExpect("19 21"));
319 EXPECT_TRUE(Tester.matchExpect("18 21"));
320 EXPECT_FALSE(Tester.matchExpect("18 20"));

323TEST_F(FileCheckTest, Substitution) {
324 SourceMgr SM;
325 FileCheckPatternContext Context;
326 std::vector<std::string> GlobalDefines;
327 GlobalDefines.emplace_back(std::string("FOO=BAR"));
328 Context.defineCmdlineVariables(GlobalDefines, SM);
329
330 // Substitution of an undefined string variable fails.
331 FileCheckStringSubstitution StringSubstitution =
332 FileCheckStringSubstitution(&Context, "VAR404", 42);
333 EXPECT_FALSE(StringSubstitution.getResult());
334
335 // Substitutions of defined pseudo and non-pseudo numeric variables return
336 // the right value.
337 FileCheckNumericVariable LineVar = FileCheckNumericVariable("@LINE", 42);
338 FileCheckNumericVariable NVar = FileCheckNumericVariable("N", 10);
339 FileCheckNumExpr NumExprLine = FileCheckNumExpr(doAdd, &LineVar, 0);
340 FileCheckNumExpr NumExprN = FileCheckNumExpr(doAdd, &NVar, 3);
341 FileCheckNumericSubstitution SubstitutionLine =
342 FileCheckNumericSubstitution(&Context, "@LINE", &NumExprLine, 12);
343 FileCheckNumericSubstitution SubstitutionN =
344 FileCheckNumericSubstitution(&Context, "N", &NumExprN, 30);
345 llvm::Optional<std::string> Value = SubstitutionLine.getResult();
346 EXPECT_TRUE(Value);
347 EXPECT_EQ("42", *Value);
348 Value = SubstitutionN.getResult();
349 EXPECT_TRUE(Value);
350 EXPECT_EQ("13", *Value);
351
352 // Substitution of an undefined numeric variable fails.
353 LineVar.clearValue();
354 EXPECT_FALSE(SubstitutionLine.getResult());
355 NVar.clearValue();
356 EXPECT_FALSE(SubstitutionN.getResult());
357
358 // Substitution of a defined string variable returns the right value.
359 FileCheckPattern P = FileCheckPattern(Check::CheckPlain, &Context, 1);
360 StringSubstitution = FileCheckStringSubstitution(&Context, "FOO", 42);
361 Value = StringSubstitution.getResult();
362 EXPECT_TRUE(Value);
363 EXPECT_EQ("BAR", *Value);
364}

366TEST_F(FileCheckTest, UndefVars) {
367 SourceMgr SM;
368 FileCheckPatternContext Context;
369 std::vector<std::string> GlobalDefines;
370 GlobalDefines.emplace_back(std::string("FOO=BAR"));
371 Context.defineCmdlineVariables(GlobalDefines, SM);
372
373 // getUndefVarName() on a string substitution with an undefined variable
374 // returns that variable.
375 FileCheckStringSubstitution StringSubstitution =
376 FileCheckStringSubstitution(&Context, "VAR404", 42);
377 StringRef UndefVar = StringSubstitution.getUndefVarName();
378 EXPECT_EQ("VAR404", UndefVar);
379
380 // getUndefVarName() on a string substitution with a defined variable returns
381 // an empty string.
382 StringSubstitution = FileCheckStringSubstitution(&Context, "FOO", 42);
383 UndefVar = StringSubstitution.getUndefVarName();
384 EXPECT_EQ("", UndefVar);
385
386 // getUndefVarName() on a numeric substitution with a defined variable
387 // returns an empty string.
388 FileCheckNumericVariable LineVar = FileCheckNumericVariable("@LINE", 42);
389 FileCheckNumExpr NumExpr = FileCheckNumExpr(doAdd, &LineVar, 0);
390 FileCheckNumericSubstitution NumericSubstitution =
391 FileCheckNumericSubstitution(&Context, "@LINE", &NumExpr, 12);
392 UndefVar = NumericSubstitution.getUndefVarName();
393 EXPECT_EQ("", UndefVar);
394
395 // getUndefVarName() on a numeric substitution with an undefined variable
396 // returns that variable.
397 LineVar.clearValue();
398 UndefVar = NumericSubstitution.getUndefVarName();
399 EXPECT_EQ("@LINE", UndefVar);
400}

402TEST_F(FileCheckTest, FileCheckContext) {
403 FileCheckPatternContext Cxt = FileCheckPatternContext();
404 std::vector<std::string> GlobalDefines;
405 SourceMgr SM;
406
407 // Missing equal sign.
408 GlobalDefines.emplace_back(std::string("LocalVar"));
409 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
410 GlobalDefines.clear();
411 GlobalDefines.emplace_back(std::string("#LocalNumVar"));
412 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
413
414 // Empty variable name.
415 GlobalDefines.clear();
416 GlobalDefines.emplace_back(std::string("=18"));
417 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
418 GlobalDefines.clear();
419 GlobalDefines.emplace_back(std::string("#=18"));
420 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
421
422 // Invalid variable name.
423 GlobalDefines.clear();
424 GlobalDefines.emplace_back(std::string("18LocalVar=18"));
425 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
426 GlobalDefines.clear();
427 GlobalDefines.emplace_back(std::string("#18LocalNumVar=18"));
428 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
429
430 // Name conflict between pattern and numeric variable.
431 GlobalDefines.clear();
432 GlobalDefines.emplace_back(std::string("LocalVar=18"));
433 GlobalDefines.emplace_back(std::string("#LocalVar=36"));
434 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
435 Cxt = FileCheckPatternContext();
436 GlobalDefines.clear();
437 GlobalDefines.emplace_back(std::string("#LocalNumVar=18"));
438 GlobalDefines.emplace_back(std::string("LocalNumVar=36"));
439 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
440 Cxt = FileCheckPatternContext();
441
442 // Invalid numeric value for numeric variable.
443 GlobalDefines.clear();
444 GlobalDefines.emplace_back(std::string("#LocalNumVar=x"));
445 EXPECT_TRUE(Cxt.defineCmdlineVariables(GlobalDefines, SM));
446
447 // Define local variables from command-line.
448 GlobalDefines.clear();
449 GlobalDefines.emplace_back(std::string("LocalVar=FOO"));
450 GlobalDefines.emplace_back(std::string("EmptyVar="));
451 GlobalDefines.emplace_back(std::string("#LocalNumVar=18"));
452 bool GotError = Cxt.defineCmdlineVariables(GlobalDefines, SM);
453 EXPECT_FALSE(GotError);
454
455 // Check defined variables are present and undefined is absent.
456 StringRef LocalVarStr = "LocalVar";
457 StringRef LocalNumVarRef = bufferize(SM, "LocalNumVar");
458 StringRef EmptyVarStr = "EmptyVar";
459 StringRef UnknownVarStr = "UnknownVar";
460 llvm::Optional<StringRef> LocalVar = Cxt.getPatternVarValue(LocalVarStr);
461 FileCheckPattern P = FileCheckPattern(Check::CheckPlain, &Cxt, 1);
462 FileCheckNumericVariable *DefinedNumericVariable;
463 FileCheckNumExpr *NumExpr = P.parseNumericSubstitutionBlock(
464 LocalNumVarRef, DefinedNumericVariable, SM);
465 llvm::Optional<StringRef> EmptyVar = Cxt.getPatternVarValue(EmptyVarStr);
466 llvm::Optional<StringRef> UnknownVar = Cxt.getPatternVarValue(UnknownVarStr);
467 EXPECT_TRUE(LocalVar);
468 EXPECT_EQ(*LocalVar, "FOO");
469 EXPECT_TRUE(NumExpr);
470 llvm::Optional<uint64_t> NumExprVal = NumExpr->eval();
471 EXPECT_TRUE(NumExprVal);
472 EXPECT_EQ(*NumExprVal, 18U);
473 EXPECT_TRUE(EmptyVar);
474 EXPECT_EQ(*EmptyVar, "");
475 EXPECT_FALSE(UnknownVar);
476
477 // Clear local variables and check they become absent.
478 Cxt.clearLocalVars();
479 LocalVar = Cxt.getPatternVarValue(LocalVarStr);
480 EXPECT_FALSE(LocalVar);
481 // Check a numeric expression's evaluation fails if called after clearing of
482 // local variables, if it was created before. This is important because local
483 // variable clearing due to --enable-var-scope happens after numeric
484 // expressions are linked to the numeric variables they use.
485 EXPECT_FALSE(NumExpr->eval());
486 P = FileCheckPattern(Check::CheckPlain, &Cxt, 2);
487 NumExpr = P.parseNumericSubstitutionBlock(LocalNumVarRef,
488 DefinedNumericVariable, SM);
489 EXPECT_FALSE(NumExpr);
490 EmptyVar = Cxt.getPatternVarValue(EmptyVarStr);
491 EXPECT_FALSE(EmptyVar);
492
493 // Redefine global variables and check variables are defined again.
494 GlobalDefines.emplace_back(std::string("$GlobalVar=BAR"));
495 GlobalDefines.emplace_back(std::string("#$GlobalNumVar=36"));
496 GotError = Cxt.defineCmdlineVariables(GlobalDefines, SM);
497 EXPECT_FALSE(GotError);
498 StringRef GlobalVarStr = "$GlobalVar";
499 StringRef GlobalNumVarRef = bufferize(SM, "$GlobalNumVar");
500 llvm::Optional<StringRef> GlobalVar = Cxt.getPatternVarValue(GlobalVarStr);
501 EXPECT_TRUE(GlobalVar);
502 EXPECT_EQ(*GlobalVar, "BAR");
503 P = FileCheckPattern(Check::CheckPlain, &Cxt, 3);
504 NumExpr = P.parseNumericSubstitutionBlock(GlobalNumVarRef,
505 DefinedNumericVariable, SM);
506 EXPECT_TRUE(NumExpr);
507 NumExprVal = NumExpr->eval();
508 EXPECT_TRUE(NumExprVal);
509 EXPECT_EQ(*NumExprVal, 36U);
510
511 // Clear local variables and check global variables remain defined.
512 Cxt.clearLocalVars();
513 GlobalVar = Cxt.getPatternVarValue(GlobalVarStr);
514 EXPECT_TRUE(GlobalVar);
515 P = FileCheckPattern(Check::CheckPlain, &Cxt, 4);
516 NumExpr = P.parseNumericSubstitutionBlock(GlobalNumVarRef,
517 DefinedNumericVariable, SM);
518 EXPECT_TRUE(NumExpr);
519 NumExprVal = NumExpr->eval();
520 EXPECT_TRUE(NumExprVal);
521 EXPECT_EQ(*NumExprVal, 36U);
522}
