55/// Create a TargetMachine. As we lack a dedicated always available target for
56/// unittests, we go for "AArch64".
57static std::unique_ptr<LLVMTargetMachine> createTargetMachine() {
58 Triple TargetTriple("aarch64--");
59 std::string Error;
60 const Target *T = TargetRegistry::lookupTarget("", TargetTriple, Error);
61 if (!T)
62 return nullptr;
63
64 TargetOptions Options;
65 return std::unique_ptr<LLVMTargetMachine>(
66 static_cast<LLVMTargetMachine *>(T->createTargetMachine(
67 "AArch64", "", "", Options, None, None, CodeGenOpt::Aggressive)));
68}

70static std::unique_ptr<Module> parseMIR(LLVMContext &Context,
71 std::unique_ptr<MIRParser> &MIR,
72 const TargetMachine &TM,
73 StringRef MIRCode, const char *FuncName,
74 MachineModuleInfo &MMI) {
75 SMDiagnostic Diagnostic;
76 std::unique_ptr<MemoryBuffer> MBuffer = MemoryBuffer::getMemBuffer(MIRCode);
77 MIR = createMIRParser(std::move(MBuffer), Context);
78 if (!MIR)
79 return nullptr;
80
81 std::unique_ptr<Module> M = MIR->parseIRModule();
82 if (!M)
83 return nullptr;
84
85 M->setDataLayout(TM.createDataLayout());
86
87 if (MIR->parseMachineFunctions(*M, MMI))
88 return nullptr;
89
90 return M;
91}

93static std::pair<std::unique_ptr<Module>, std::unique_ptr<MachineModuleInfo>>
94createDummyModule(LLVMContext &Context, const LLVMTargetMachine &TM,
95 StringRef MIRFunc) {
96 SmallString<512> S;
97 StringRef MIRString = (Twine(R"MIR(
98---
99...
100name: func
101registers:
102 - { id: 0, class: _ }
103 - { id: 1, class: _ }
104 - { id: 2, class: _ }
105 - { id: 3, class: _ }
106body: |
107 bb.1:
108 %0(s64) = COPY $x0
109 %1(s64) = COPY $x1
110 %2(s64) = COPY $x2
111)MIR") + Twine(MIRFunc) + Twine("...\n"))
112 .toNullTerminatedStringRef(S);
113 std::unique_ptr<MIRParser> MIR;
114 auto MMI = make_unique<MachineModuleInfo>(&TM);
115 std::unique_ptr<Module> M =
116 parseMIR(Context, MIR, TM, MIRString, "func", *MMI);
117 return make_pair(std::move(M), std::move(MMI));
118}

120static MachineFunction *getMFFromMMI(const Module *M,
121 const MachineModuleInfo *MMI) {
122 Function *F = M->getFunction("func");
123 auto *MF = MMI->getMachineFunction(*F);
124 return MF;
125}

127static void collectCopies(SmallVectorImpl<unsigned> &Copies,
128 MachineFunction *MF) {
129 for (auto &MBB : *MF)
130 for (MachineInstr &MI : MBB) {
131 if (MI.getOpcode() == TargetOpcode::COPY)
132 Copies.push_back(MI.getOperand(0).getReg());
133 }
134}

136class GISelMITest : public ::testing::Test {
137protected:
138 GISelMITest() : ::testing::Test() {
139 TM = createTargetMachine();
140 if (!TM)
141 return;
142 ModuleMMIPair = createDummyModule(Context, *TM, "");
143 MF = getMFFromMMI(ModuleMMIPair.first.get(), ModuleMMIPair.second.get());
144 collectCopies(Copies, MF);
145 EntryMBB = &*MF->begin();
146 B.setMF(*MF);
147 MRI = &MF->getRegInfo();
148 B.setInsertPt(*EntryMBB, EntryMBB->end());
149 }
150 LLVMContext Context;
151 std::unique_ptr<LLVMTargetMachine> TM;
152 MachineFunction *MF;
153 std::pair<std::unique_ptr<Module>, std::unique_ptr<MachineModuleInfo>>
154 ModuleMMIPair;
155 SmallVector<unsigned, 4> Copies;
156 MachineBasicBlock *EntryMBB;
157 MachineIRBuilder B;
158 MachineRegisterInfo *MRI;
159};

181static inline bool CheckMachineFunction(const MachineFunction &MF,
182 StringRef CheckStr) {
183 SmallString<512> Msg;
184 raw_svector_ostream OS(Msg);
185 MF.print(OS);
186 auto OutputBuf = MemoryBuffer::getMemBuffer(Msg, "Output", false);
187 auto CheckBuf = MemoryBuffer::getMemBuffer(CheckStr, "");
188 SmallString<4096> CheckFileBuffer;
189 FileCheckRequest Req;
190 FileCheck FC(Req);
191 StringRef CheckFileText =
192 FC.CanonicalizeFile(*CheckBuf.get(), CheckFileBuffer);
193 SourceMgr SM;
194 SM.AddNewSourceBuffer(MemoryBuffer::getMemBuffer(CheckFileText, "CheckFile"),
195 SMLoc());
196 Regex PrefixRE = FC.buildCheckPrefixRegex();
197 std::vector<FileCheckString> CheckStrings;
198 if (FC.ReadCheckFile(SM, CheckFileText, PrefixRE, CheckStrings))
199 return false;
200
201 auto OutBuffer = OutputBuf->getBuffer();
202 SM.AddNewSourceBuffer(std::move(OutputBuf), SMLoc());
203 return FC.CheckInput(SM, OutBuffer, CheckStrings);
204}
