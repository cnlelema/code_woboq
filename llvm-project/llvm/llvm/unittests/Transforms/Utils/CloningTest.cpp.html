31class CloneInstruction : public ::testing::Test {
32protected:
33 void SetUp() override { V = nullptr; }
34
35 template <typename T>
36 T *clone(T *V1) {
37 Value *V2 = V1->clone();
38 Orig.insert(V1);
39 Clones.insert(V2);
40 return cast<T>(V2);
41 }
42
43 void eraseClones() {
44 for (Value *V : Clones)
45 V->deleteValue();
46 Clones.clear();
47 }
48
49 void TearDown() override {
50 eraseClones();
51 for (Value *V : Orig)
52 V->deleteValue();
53 Orig.clear();
54 if (V)
55 V->deleteValue();
56 }
57
58 SmallPtrSet<Value *, 4> Orig; // Erase on exit
59 SmallPtrSet<Value *, 4> Clones; // Erase in eraseClones
60
61 LLVMContext context;
62 Value *V;
63};

65TEST_F(CloneInstruction, OverflowBits) {
66 V = new Argument(Type::getInt32Ty(context));
67
68 BinaryOperator *Add = BinaryOperator::Create(Instruction::Add, V, V);
69 BinaryOperator *Sub = BinaryOperator::Create(Instruction::Sub, V, V);
70 BinaryOperator *Mul = BinaryOperator::Create(Instruction::Mul, V, V);
71
72 BinaryOperator *AddClone = this->clone(Add);
73 BinaryOperator *SubClone = this->clone(Sub);
74 BinaryOperator *MulClone = this->clone(Mul);
75
76 EXPECT_FALSE(AddClone->hasNoUnsignedWrap());
77 EXPECT_FALSE(AddClone->hasNoSignedWrap());
78 EXPECT_FALSE(SubClone->hasNoUnsignedWrap());
79 EXPECT_FALSE(SubClone->hasNoSignedWrap());
80 EXPECT_FALSE(MulClone->hasNoUnsignedWrap());
81 EXPECT_FALSE(MulClone->hasNoSignedWrap());
82
83 eraseClones();
84
85 Add->setHasNoUnsignedWrap();
86 Sub->setHasNoUnsignedWrap();
87 Mul->setHasNoUnsignedWrap();
88
89 AddClone = this->clone(Add);
90 SubClone = this->clone(Sub);
91 MulClone = this->clone(Mul);
92
93 EXPECT_TRUE(AddClone->hasNoUnsignedWrap());
94 EXPECT_FALSE(AddClone->hasNoSignedWrap());
95 EXPECT_TRUE(SubClone->hasNoUnsignedWrap());
96 EXPECT_FALSE(SubClone->hasNoSignedWrap());
97 EXPECT_TRUE(MulClone->hasNoUnsignedWrap());
98 EXPECT_FALSE(MulClone->hasNoSignedWrap());
99
100 eraseClones();
101
102 Add->setHasNoSignedWrap();
103 Sub->setHasNoSignedWrap();
104 Mul->setHasNoSignedWrap();
105
106 AddClone = this->clone(Add);
107 SubClone = this->clone(Sub);
108 MulClone = this->clone(Mul);
109
110 EXPECT_TRUE(AddClone->hasNoUnsignedWrap());
111 EXPECT_TRUE(AddClone->hasNoSignedWrap());
112 EXPECT_TRUE(SubClone->hasNoUnsignedWrap());
113 EXPECT_TRUE(SubClone->hasNoSignedWrap());
114 EXPECT_TRUE(MulClone->hasNoUnsignedWrap());
115 EXPECT_TRUE(MulClone->hasNoSignedWrap());
116
117 eraseClones();
118
119 Add->setHasNoUnsignedWrap(false);
120 Sub->setHasNoUnsignedWrap(false);
121 Mul->setHasNoUnsignedWrap(false);
122
123 AddClone = this->clone(Add);
124 SubClone = this->clone(Sub);
125 MulClone = this->clone(Mul);
126
127 EXPECT_FALSE(AddClone->hasNoUnsignedWrap());
128 EXPECT_TRUE(AddClone->hasNoSignedWrap());
129 EXPECT_FALSE(SubClone->hasNoUnsignedWrap());
130 EXPECT_TRUE(SubClone->hasNoSignedWrap());
131 EXPECT_FALSE(MulClone->hasNoUnsignedWrap());
132 EXPECT_TRUE(MulClone->hasNoSignedWrap());

207TEST_F(CloneInstruction, DuplicateInstructionsToSplit) {
208 Type *ArgTy1[] = {Type::getInt32PtrTy(context)};
209 FunctionType *FT = FunctionType::get(Type::getVoidTy(context), ArgTy1, false);
210 V = new Argument(Type::getInt32Ty(context));
211
212 Function *F = Function::Create(FT, Function::ExternalLinkage, 0);
213
214 BasicBlock *BB1 = BasicBlock::Create(context, "", F);
215 IRBuilder<> Builder1(BB1);
216
217 BasicBlock *BB2 = BasicBlock::Create(context, "", F);
218 IRBuilder<> Builder2(BB2);
219
220 Builder1.CreateBr(BB2);
221
222 Instruction *AddInst = cast<Instruction>(Builder2.CreateAdd(V, V));
223 Instruction *MulInst = cast<Instruction>(Builder2.CreateMul(AddInst, V));
224 Instruction *SubInst = cast<Instruction>(Builder2.CreateSub(MulInst, V));
225 Builder2.CreateRetVoid();
226
227 // Dummy DTU.
228 ValueToValueMapTy Mapping;
229 DomTreeUpdater DTU(DomTreeUpdater::UpdateStrategy::Lazy);
230 auto Split =
231 DuplicateInstructionsInSplitBetween(BB2, BB1, SubInst, Mapping, DTU);
232
233 EXPECT_TRUE(Split);
234 EXPECT_EQ(Mapping.size(), 2u);
235 EXPECT_TRUE(Mapping.find(AddInst) != Mapping.end());
236 EXPECT_TRUE(Mapping.find(MulInst) != Mapping.end());
237
238 auto AddSplit = dyn_cast<Instruction>(Mapping[AddInst]);
239 EXPECT_TRUE(AddSplit);
240 EXPECT_EQ(AddSplit->getOperand(0), V);
241 EXPECT_EQ(AddSplit->getOperand(1), V);
242 EXPECT_EQ(AddSplit->getParent(), Split);
243
244 auto MulSplit = dyn_cast<Instruction>(Mapping[MulInst]);
245 EXPECT_TRUE(MulSplit);
246 EXPECT_EQ(MulSplit->getOperand(0), AddSplit);
247 EXPECT_EQ(MulSplit->getOperand(1), V);
248 EXPECT_EQ(MulSplit->getParent(), Split);
249
250 EXPECT_EQ(AddSplit->getNextNode(), MulSplit);
251 EXPECT_EQ(MulSplit->getNextNode(), Split->getTerminator());
252
253 delete F;
254}

373 void CreateOldFunc() {
374 FunctionType* FuncType = FunctionType::get(Type::getVoidTy(C), false);
375 OldFunc = Function::Create(FuncType, GlobalValue::PrivateLinkage, "f", M);

379 void CreateOldFunctionBodyAndDI() {
380 DIBuilder DBuilder(*M);
381 IRBuilder<> IBuilder(C);
382
383 // Function DI
384 auto *File = DBuilder.createFile("filename.c", "/file/dir/");
385 DITypeRefArray ParamTypes = DBuilder.getOrCreateTypeArray(None);
386 DISubroutineType *FuncType =
387 DBuilder.createSubroutineType(ParamTypes);
388 auto *CU = DBuilder.createCompileUnit(dwarf::DW_LANG_C99,
389 DBuilder.createFile("filename.c",
390 "/file/dir"),
391 "CloneFunc", false, "", 0);
392
393 auto *Subprogram = DBuilder.createFunction(
394 CU, "f", "f", File, 4, FuncType, 3, DINode::FlagZero,
395 DISubprogram::SPFlagLocalToUnit | DISubprogram::SPFlagDefinition);
396 OldFunc->setSubprogram(Subprogram);
397
398 // Function body
399 BasicBlock* Entry = BasicBlock::Create(C, "", OldFunc);
400 IBuilder.SetInsertPoint(Entry);
401 DebugLoc Loc = DebugLoc::get(3, 2, Subprogram);
402 IBuilder.SetCurrentDebugLocation(Loc);
403 AllocaInst* Alloca = IBuilder.CreateAlloca(IntegerType::getInt32Ty(C));
404 IBuilder.SetCurrentDebugLocation(DebugLoc::get(4, 2, Subprogram));
405 Value* AllocaContent = IBuilder.getInt32(1);
406 Instruction* Store = IBuilder.CreateStore(AllocaContent, Alloca);
407 IBuilder.SetCurrentDebugLocation(DebugLoc::get(5, 2, Subprogram));
408
409 // Create a local variable around the alloca
410 auto *IntType = DBuilder.createBasicType("int", 32, dwarf::DW_ATE_signed);
411 auto *E = DBuilder.createExpression();
412 auto *Variable =
413 DBuilder.createAutoVariable(Subprogram, "x", File, 5, IntType, true);
414 auto *DL = DILocation::get(Subprogram->getContext(), 5, 0, Subprogram);
415 DBuilder.insertDeclare(Alloca, Variable, E, DL, Store);
416 DBuilder.insertDbgValueIntrinsic(AllocaContent, Variable, E, DL, Entry);
417 // Also create an inlined variable.
418 // Create a distinct struct type that we should not duplicate during
419 // cloning).
420 auto *StructType = DICompositeType::getDistinct(
421 C, dwarf::DW_TAG_structure_type, "some_struct", nullptr, 0, nullptr,
422 nullptr, 32, 32, 0, DINode::FlagZero, nullptr, 0, nullptr, nullptr);
423 auto *InlinedSP = DBuilder.createFunction(
424 CU, "inlined", "inlined", File, 8, FuncType, 9, DINode::FlagZero,
425 DISubprogram::SPFlagLocalToUnit | DISubprogram::SPFlagDefinition);
426 auto *InlinedVar =
427 DBuilder.createAutoVariable(InlinedSP, "inlined", File, 5, StructType, true);
428 auto *Scope = DBuilder.createLexicalBlock(
429 DBuilder.createLexicalBlockFile(InlinedSP, File), File, 1, 1);
430 auto InlinedDL =
431 DebugLoc::get(9, 4, Scope, DebugLoc::get(5, 2, Subprogram));
432 IBuilder.SetCurrentDebugLocation(InlinedDL);
433 DBuilder.insertDeclare(Alloca, InlinedVar, E, InlinedDL, Store);
434 IBuilder.CreateStore(IBuilder.getInt32(2), Alloca);
435 // Finalize the debug info.
436 DBuilder.finalize();
437 IBuilder.CreateRetVoid();
438
439 // Create another, empty, compile unit.
440 DIBuilder DBuilder2(*M);
441 DBuilder2.createCompileUnit(dwarf::DW_LANG_C99,
442 DBuilder.createFile("extra.c", "/file/dir"),
443 "CloneFunc", false, "", 0);
444 DBuilder2.finalize();
445 }

477// Test that instructions in the old function still belong to it in the
478// metadata, while instruction in the new function belong to the new one.
479TEST_F(CloneFunc, InstructionOwnership) {
480 EXPECT_FALSE(verifyModule(*M));
481
482 inst_iterator OldIter = inst_begin(OldFunc);
483 inst_iterator OldEnd = inst_end(OldFunc);
484 inst_iterator NewIter = inst_begin(NewFunc);
485 inst_iterator NewEnd = inst_end(NewFunc);
486 while (OldIter != OldEnd && NewIter != NewEnd) {
487 Instruction& OldI = *OldIter;
488 Instruction& NewI = *NewIter;
489 EXPECT_NE(&OldI, &NewI);
490
491 EXPECT_EQ(OldI.hasMetadata(), NewI.hasMetadata());
492 if (OldI.hasMetadata()) {
493 const DebugLoc& OldDL = OldI.getDebugLoc();
494 const DebugLoc& NewDL = NewI.getDebugLoc();
495
496 // Verify that the debug location data is the same
497 EXPECT_EQ(OldDL.getLine(), NewDL.getLine());
498 EXPECT_EQ(OldDL.getCol(), NewDL.getCol());
499
500 // But that they belong to different functions
501 auto *OldSubprogram = cast<DISubprogram>(OldDL.getInlinedAtScope());
502 auto *NewSubprogram = cast<DISubprogram>(NewDL.getInlinedAtScope());
503 EXPECT_EQ(OldFunc->getSubprogram(), OldSubprogram);
504 EXPECT_EQ(NewFunc->getSubprogram(), NewSubprogram);
505 }
506
507 ++OldIter;
508 ++NewIter;
509 }
510 EXPECT_EQ(OldEnd, OldIter);
511 EXPECT_EQ(NewEnd, NewIter);
512}

514// Test that the arguments for debug intrinsics in the new function were
515// properly cloned
516TEST_F(CloneFunc, DebugIntrinsics) {
517 EXPECT_FALSE(verifyModule(*M));
518
519 inst_iterator OldIter = inst_begin(OldFunc);
520 inst_iterator OldEnd = inst_end(OldFunc);
521 inst_iterator NewIter = inst_begin(NewFunc);
522 inst_iterator NewEnd = inst_end(NewFunc);
523 while (OldIter != OldEnd && NewIter != NewEnd) {
524 Instruction& OldI = *OldIter;
525 Instruction& NewI = *NewIter;
526 if (DbgDeclareInst* OldIntrin = dyn_cast<DbgDeclareInst>(&OldI)) {
527 DbgDeclareInst* NewIntrin = dyn_cast<DbgDeclareInst>(&NewI);
528 EXPECT_TRUE(NewIntrin);
529
530 // Old address must belong to the old function
531 EXPECT_EQ(OldFunc, cast<AllocaInst>(OldIntrin->getAddress())->
532 getParent()->getParent());
533 // New address must belong to the new function
534 EXPECT_EQ(NewFunc, cast<AllocaInst>(NewIntrin->getAddress())->
535 getParent()->getParent());
536
537 if (OldIntrin->getDebugLoc()->getInlinedAt()) {
538 // Inlined variable should refer to the same DILocalVariable as in the
539 // Old Function
540 EXPECT_EQ(OldIntrin->getVariable(), NewIntrin->getVariable());
541 } else {
542 // Old variable must belong to the old function.
543 EXPECT_EQ(OldFunc->getSubprogram(),
544 cast<DISubprogram>(OldIntrin->getVariable()->getScope()));
545 // New variable must belong to the new function.
546 EXPECT_EQ(NewFunc->getSubprogram(),
547 cast<DISubprogram>(NewIntrin->getVariable()->getScope()));
548 }
549 } else if (DbgValueInst* OldIntrin = dyn_cast<DbgValueInst>(&OldI)) {
550 DbgValueInst* NewIntrin = dyn_cast<DbgValueInst>(&NewI);
551 EXPECT_TRUE(NewIntrin);
552
553 if (!OldIntrin->getDebugLoc()->getInlinedAt()) {
554 // Old variable must belong to the old function.
555 EXPECT_EQ(OldFunc->getSubprogram(),
556 cast<DISubprogram>(OldIntrin->getVariable()->getScope()));
557 // New variable must belong to the new function.
558 EXPECT_EQ(NewFunc->getSubprogram(),
559 cast<DISubprogram>(NewIntrin->getVariable()->getScope()));
560 }
561 }
562
563 ++OldIter;
564 ++NewIter;
565 }
