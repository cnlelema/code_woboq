24// stand-in for object::ObjectFile
25typedef int MockObjectFile;
26
27// stand-in for llvm::MemoryBuffer set
28typedef int MockMemoryBuffer;
29
30// Mock transform that operates on unique pointers to object files, and
31// allocates new object files rather than mutating the given ones.
32struct AllocatingTransform {
33 std::shared_ptr<MockObjectFile>
34 operator()(std::shared_ptr<MockObjectFile> Obj) const {
35 return std::make_shared<MockObjectFile>(*Obj + 1);
36 }
37};
38
39// Mock base layer for verifying behavior of transform layer.
40// Each method "T foo(args)" is accompanied by two auxiliary methods:
41// - "void expectFoo(args)", to be called before calling foo on the transform
42// layer; saves values of args, which mock layer foo then verifies against.
43// - "void verifyFoo(T)", to be called after foo, which verifies that the
44// transform layer called the base layer and forwarded any return value.
45class MockBaseLayer {
46public:
47 MockBaseLayer() : MockSymbol(nullptr) { resetExpectations(); }
48
49 template <typename ObjPtrT> llvm::Error addObject(VModuleKey K, ObjPtrT Obj) {
50 EXPECT_EQ(MockKey, K) << "Key should pass through";
51 EXPECT_EQ(MockObject + 1, *Obj) << "Transform should be applied";
52 LastCalled = "addObject";
53 return llvm::Error::success();
54 }
55
56 template <typename ObjPtrT> void expectAddObject(VModuleKey K, ObjPtrT Obj) {
57 MockKey = K;
58 MockObject = *Obj;
59 }
60
61 void verifyAddObject() {
62 EXPECT_EQ("addObject", LastCalled);
63 resetExpectations();
64 }
65
66 llvm::Error removeObject(VModuleKey K) {
67 EXPECT_EQ(MockKey, K);
68 LastCalled = "removeObject";
69 return llvm::Error::success();
70 }
71
72 void expectRemoveObject(VModuleKey K) { MockKey = K; }
73 void verifyRemoveObject() {
74 EXPECT_EQ("removeObject", LastCalled);
75 resetExpectations();
76 }
77
78 llvm::JITSymbol findSymbol(const std::string &Name,
79 bool ExportedSymbolsOnly) {
80 EXPECT_EQ(MockName, Name) << "Name should pass through";
81 EXPECT_EQ(MockBool, ExportedSymbolsOnly) << "Flag should pass through";
82 LastCalled = "findSymbol";
83 MockSymbol = llvm::JITSymbol(122, llvm::JITSymbolFlags::None);
84 return llvm::JITSymbol(122, llvm::JITSymbolFlags::None);
85 }
86 void expectFindSymbol(const std::string &Name, bool ExportedSymbolsOnly) {
87 MockName = Name;
88 MockBool = ExportedSymbolsOnly;
89 }
90 void verifyFindSymbol(llvm::JITSymbol Returned) {
91 EXPECT_EQ("findSymbol", LastCalled);
92 EXPECT_EQ(cantFail(MockSymbol.getAddress()),
93 cantFail(Returned.getAddress()))
94 << "Return should pass through";
95 resetExpectations();
96 }
97
98 llvm::JITSymbol findSymbolIn(VModuleKey K, const std::string &Name,
99 bool ExportedSymbolsOnly) {
100 EXPECT_EQ(MockKey, K) << "VModuleKey should pass through";
101 EXPECT_EQ(MockName, Name) << "Name should pass through";
102 EXPECT_EQ(MockBool, ExportedSymbolsOnly) << "Flag should pass through";
103 LastCalled = "findSymbolIn";
104 MockSymbol = llvm::JITSymbol(122, llvm::JITSymbolFlags::None);
105 return llvm::JITSymbol(122, llvm::JITSymbolFlags::None);
106 }
107 void expectFindSymbolIn(VModuleKey K, const std::string &Name,
108 bool ExportedSymbolsOnly) {
109 MockKey = K;
110 MockName = Name;
111 MockBool = ExportedSymbolsOnly;
112 }
113 void verifyFindSymbolIn(llvm::JITSymbol Returned) {
114 EXPECT_EQ("findSymbolIn", LastCalled);
115 EXPECT_EQ(cantFail(MockSymbol.getAddress()),
116 cantFail(Returned.getAddress()))
117 << "Return should pass through";
118 resetExpectations();
119 }
120
121 llvm::Error emitAndFinalize(VModuleKey K) {
122 EXPECT_EQ(MockKey, K) << "VModuleKey should pass through";
123 LastCalled = "emitAndFinalize";
124 return llvm::Error::success();
125 }
126
127 void expectEmitAndFinalize(VModuleKey K) { MockKey = K; }
128
129 void verifyEmitAndFinalize() {
130 EXPECT_EQ("emitAndFinalize", LastCalled);
131 resetExpectations();
132 }
133
134 void mapSectionAddress(VModuleKey K, const void *LocalAddress,
135 llvm::JITTargetAddress TargetAddr) {
136 EXPECT_EQ(MockKey, K);
137 EXPECT_EQ(MockLocalAddress, LocalAddress);
138 EXPECT_EQ(MockTargetAddress, TargetAddr);
139 LastCalled = "mapSectionAddress";
140 }
141 void expectMapSectionAddress(VModuleKey K, const void *LocalAddress,
142 llvm::JITTargetAddress TargetAddr) {
143 MockKey = K;
144 MockLocalAddress = LocalAddress;
145 MockTargetAddress = TargetAddr;
146 }
147 void verifyMapSectionAddress() {
148 EXPECT_EQ("mapSectionAddress", LastCalled);
149 resetExpectations();
150 }
151
152private:
153 // Backing fields for remembering parameter/return values
154 std::string LastCalled;
155 VModuleKey MockKey;
156 MockObjectFile MockObject;
157 std::string MockName;
158 bool MockBool;
159 llvm::JITSymbol MockSymbol;
160 const void *MockLocalAddress;
161 llvm::JITTargetAddress MockTargetAddress;
162 MockMemoryBuffer MockBuffer;
163
164 // Clear remembered parameters between calls
165 void resetExpectations() {
166 LastCalled = "nothing";
167 MockKey = 0;
168 MockObject = 0;
169 MockName = "bogus";
170 MockSymbol = llvm::JITSymbol(nullptr);
171 MockLocalAddress = nullptr;
172 MockTargetAddress = 0;
173 MockBuffer = 0;
174 }
175};

177// Test each operation on LegacyObjectTransformLayer.
178TEST(LegacyObjectTransformLayerTest, Main) {
179 MockBaseLayer M;
180
181 ExecutionSession ES(std::make_shared<SymbolStringPool>());
182
183 // Create one object transform layer using a transform (as a functor)
184 // that allocates new objects, and deals in unique pointers.
185 LegacyObjectTransformLayer<MockBaseLayer, AllocatingTransform> T1(M);
186
187 // Create a second object transform layer using a transform (as a lambda)
188 // that mutates objects in place, and deals in naked pointers
189 LegacyObjectTransformLayer<MockBaseLayer,
190 std::function<std::shared_ptr<MockObjectFile>(
191 std::shared_ptr<MockObjectFile>)>>
192 T2(M, [](std::shared_ptr<MockObjectFile> Obj) {
193 ++(*Obj);
194 return Obj;
195 });
196
197 // Test addObject with T1 (allocating)
198 auto K1 = ES.allocateVModule();
199 auto Obj1 = std::make_shared<MockObjectFile>(211);
200 M.expectAddObject(K1, Obj1);
201 cantFail(T1.addObject(K1, std::move(Obj1)));
202 M.verifyAddObject();
203
204 // Test addObjectSet with T2 (mutating)
205 auto K2 = ES.allocateVModule();
206 auto Obj2 = std::make_shared<MockObjectFile>(222);
207 M.expectAddObject(K2, Obj2);
208 cantFail(T2.addObject(K2, Obj2));
209 M.verifyAddObject();
210 EXPECT_EQ(223, *Obj2) << "Expected mutation";
211
212 // Test removeObjectSet
213 M.expectRemoveObject(K2);
214 cantFail(T1.removeObject(K2));
215 M.verifyRemoveObject();
216
217 // Test findSymbol
218 std::string Name = "foo";
219 bool ExportedOnly = true;
220 M.expectFindSymbol(Name, ExportedOnly);
221 llvm::JITSymbol Sym1 = T2.findSymbol(Name, ExportedOnly);
222 M.verifyFindSymbol(std::move(Sym1));
223
224 // Test findSymbolIn
225 Name = "bar";
226 ExportedOnly = false;
227 M.expectFindSymbolIn(K1, Name, ExportedOnly);
228 llvm::JITSymbol Sym2 = T1.findSymbolIn(K1, Name, ExportedOnly);
229 M.verifyFindSymbolIn(std::move(Sym2));
230
231 // Test emitAndFinalize
232 M.expectEmitAndFinalize(K1);
233 cantFail(T2.emitAndFinalize(K1));
234 M.verifyEmitAndFinalize();
235
236 // Test mapSectionAddress
237 char Buffer[24];
238 llvm::JITTargetAddress MockAddress = 255;
239 M.expectMapSectionAddress(K1, Buffer, MockAddress);
240 T1.mapSectionAddress(K1, Buffer, MockAddress);
241 M.verifyMapSectionAddress();
242
243 // Verify transform getter (non-const)
244 auto Mutatee = std::make_shared<MockObjectFile>(277);
245 auto Out = T2.getTransform()(Mutatee);
246 EXPECT_EQ(*Mutatee, *Out) << "Expected in-place transform";
247 EXPECT_EQ(278, *Mutatee) << "Expected incrementing transform";
248
249 // Verify transform getter (const)
250 auto OwnedObj = std::make_shared<MockObjectFile>(288);
251 const auto &T1C = T1;
252 OwnedObj = T1C.getTransform()(std::move(OwnedObj));
253 EXPECT_EQ(289, *OwnedObj) << "Expected incrementing transform";
254
255 volatile bool RunStaticChecks = false;
256 if (!RunStaticChecks)
257 return;
258
259 // Make sure that LegacyObjectTransformLayer implements the object layer concept
260 // correctly by sandwitching one between an ObjectLinkingLayer and an
261 // LegacyIRCompileLayer, verifying that it compiles if we have a call to the
262 // IRComileLayer's addModule that should call the transform layer's
263 // addObject, and also calling the other public transform layer methods
264 // directly to make sure the methods they intend to forward to exist on
265 // the ObjectLinkingLayer.
266
267 // We'll need a concrete MemoryManager class.
268 class NullManager : public llvm::RuntimeDyld::MemoryManager {
269 public:
270 uint8_t *allocateCodeSection(uintptr_t, unsigned, unsigned,
271 llvm::StringRef) override {
272 return nullptr;
273 }
274 uint8_t *allocateDataSection(uintptr_t, unsigned, unsigned, llvm::StringRef,
275 bool) override {
276 return nullptr;
277 }
278 void registerEHFrames(uint8_t *, uint64_t, size_t) override {}
279 void deregisterEHFrames() override {}
280 bool finalizeMemory(std::string *) override { return false; }
281 };
282
283 // Construct the jit layers.
284 LegacyRTDyldObjectLinkingLayer BaseLayer(ES, [](VModuleKey) {
285 return LegacyRTDyldObjectLinkingLayer::Resources{
286 std::make_shared<llvm::SectionMemoryManager>(),
287 std::make_shared<NullResolver>()};
288 });
289
290 auto IdentityTransform = [](std::unique_ptr<llvm::MemoryBuffer> Obj) {
291 return Obj;
292 };
293 LegacyObjectTransformLayer<decltype(BaseLayer), decltype(IdentityTransform)>
294 TransformLayer(BaseLayer, IdentityTransform);
295 auto NullCompiler = [](llvm::Module &) {
296 return std::unique_ptr<llvm::MemoryBuffer>(nullptr);
297 };
298 LegacyIRCompileLayer<decltype(TransformLayer), decltype(NullCompiler)>
299 CompileLayer(TransformLayer, NullCompiler);
300
301 // Make sure that the calls from LegacyIRCompileLayer to LegacyObjectTransformLayer
302 // compile.
303 cantFail(CompileLayer.addModule(ES.allocateVModule(),
304 std::unique_ptr<llvm::Module>()));
305
306 // Make sure that the calls from LegacyObjectTransformLayer to ObjectLinkingLayer
307 // compile.
308 VModuleKey DummyKey = ES.allocateVModule();
309 cantFail(TransformLayer.emitAndFinalize(DummyKey));
310 TransformLayer.findSymbolIn(DummyKey, Name, false);
311 TransformLayer.findSymbol(Name, true);
312 TransformLayer.mapSectionAddress(DummyKey, nullptr, 0);
313 cantFail(TransformLayer.removeObject(DummyKey));
314}
