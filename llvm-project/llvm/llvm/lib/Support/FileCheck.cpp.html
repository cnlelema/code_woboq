88bool FileCheckPattern::parseVariable(StringRef &Str, StringRef &Name,
89 bool &IsPseudo) {
90 if (Str.empty())
91 return true;
92
93 bool ParsedOneChar = false;
94 unsigned I = 0;
95 IsPseudo = Str[0] == '@';
96
97 // Global vars start with '$'.
98 if (Str[0] == '$' || IsPseudo)
99 ++I;
100
101 for (unsigned E = Str.size(); I != E; ++I) {
102 if (!ParsedOneChar && !isValidVarNameStart(Str[I]))
103 return true;
104
105 // Variable names are composed of alphanumeric characters and underscores.
106 if (Str[I] != '_' && !isalnum(Str[I]))
107 break;
108 ParsedOneChar = true;
109 }
110
111 Name = Str.take_front(I);
112 Str = Str.substr(I);
113 return false;
114}
115
116// StringRef holding all characters considered as horizontal whitespaces by
117// FileCheck input canonicalization.
118StringRef SpaceChars = " \t";
119
120// Parsing helper function that strips the first character in S and returns it.
121static char popFront(StringRef &S) {
122 char C = S.front();
123 S = S.drop_front();
124 return C;
125}

127bool FileCheckPattern::parseNumericVariableDefinition(
128 StringRef &Expr, StringRef &Name, FileCheckPatternContext *Context,
129 const SourceMgr &SM) {
130 bool IsPseudo;
131 if (parseVariable(Expr, Name, IsPseudo)) {
132 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
133 "invalid variable name");
134 return true;
135 }
136
137 if (IsPseudo) {
138 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
139 "definition of pseudo numeric variable unsupported");
140 return true;
141 }
142
143 // Detect collisions between string and numeric variables when the latter
144 // is created later than the former.
145 if (Context->DefinedVariableTable.find(Name) !=
146 Context->DefinedVariableTable.end()) {
147 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
148 "string variable with name '" + Name + "' already exists");
149 return true;
150 }
151
152 return false;
153}

155FileCheckNumericVariable *
156FileCheckPattern::parseNumericVariableUse(StringRef &Expr,
157 const SourceMgr &SM) const {
158 bool IsPseudo;
159 StringRef Name;
160 if (parseVariable(Expr, Name, IsPseudo)) {
161 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
162 "invalid variable name");
163 return nullptr;
164 }
165
166 if (IsPseudo && !Name.equals("@LINE")) {
167 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
168 "invalid pseudo numeric variable '" + Name + "'");
169 return nullptr;
170 }
171
172 // This method is indirectly called from parsePattern for all numeric
173 // variable definitions and uses in the order in which they appear in the
174 // CHECK pattern. For each definition, the pointer to the class instance of
175 // the corresponding numeric variable definition is stored in
176 // GlobalNumericVariableTable. Therefore, the pointer we get below is for the
177 // class instance corresponding to the last definition of this variable use.
178 auto VarTableIter = Context->GlobalNumericVariableTable.find(Name);
179 if (VarTableIter == Context->GlobalNumericVariableTable.end()) {
180 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
181 "using undefined numeric variable '" + Name + "'");
182 return nullptr;
183 }
184
185 FileCheckNumericVariable *NumericVariable = VarTableIter->second;
186 if (!IsPseudo && NumericVariable->getDefLineNumber() == LineNumber) {
187 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
188 "numeric variable '" + Name +
189 "' defined on the same line as used");
190 return nullptr;
191 }
192
193 return NumericVariable;
194}

212FileCheckNumExpr *FileCheckPattern::parseBinop(StringRef &Expr,
213 const SourceMgr &SM) const {
214 FileCheckNumericVariable *LeftOp = parseNumericVariableUse(Expr, SM);
215 if (!LeftOp) {
216 // Error reporting done in parseNumericVariableUse().
217 return nullptr;
218 }
219
220 // Check if this is a supported operation and select a function to perform
221 // it.
222 Expr = Expr.ltrim(SpaceChars);
223 if (Expr.empty())
224 return Context->makeNumExpr(add, LeftOp, 0);
225 SMLoc OpLoc = SMLoc::getFromPointer(Expr.data());
226 char Operator = popFront(Expr);
227 binop_eval_t EvalBinop;
228 switch (Operator) {
229 case '+':
230 EvalBinop = add;
231 break;
232 case '-':
233 EvalBinop = sub;
234 break;
235 case '*':
236 EvalBinop = mul;
237 break;
238 case '/':
239 EvalBinop = div;
240 break;
241 default:
242 SM.PrintMessage(OpLoc, SourceMgr::DK_Error,
243 Twine("unsupported numeric operation '") + Twine(Operator) +
244 "'");
245 return nullptr;
246 }
247
248 // Parse right operand.
249 Expr = Expr.ltrim(SpaceChars);
250 if (Expr.empty()) {
251 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
252 "missing operand in numeric expression");
253 return nullptr;
254 }
255 uint64_t RightOp;
256 if (Expr.consumeInteger(10, RightOp)) {
257 if (Expr.consume_front("#")) {
258 FileCheckNumericVariable *RightOpVar = parseNumericVariableUse(Expr, SM);
259 if (!RightOpVar)
260 // Error reporting done in parseNumericVariableUse().
261 return nullptr;
262 RightOp = *RightOpVar->getValue();
263 } else if (Expr.consume_front("(#") && Expr.consume_back(")")) {
264 FileCheckNumericVariable *NewVar = nullptr;
265 auto SubExpr =
266 parseNumericSubstitutionBlock(Expr, NewVar, SM);
267 if (NewVar) {
268 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
269 "invalid definition in nested expression '" + Expr + "'");
270 return nullptr;
271 }
272 if (!SubExpr)
273 // Error reporting done in parseNumericSubstitutionBlock().
274 return nullptr;
275 RightOp = *SubExpr->eval();
276 } else {
277 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
278 "invalid offset in numeric expression '" + Expr + "'");
279 return nullptr;
280 }
281 } else {
282 Expr = Expr.ltrim(SpaceChars);
283 if (!Expr.empty()) {
284 SM.PrintMessage(SMLoc::getFromPointer(Expr.data()), SourceMgr::DK_Error,
285 "unexpected characters at end of numeric expression '" +
286 Expr + "'");
287 return nullptr;
288 }
289 }
290
291 return Context->makeNumExpr(EvalBinop, LeftOp, RightOp);
292}

294FileCheckNumExpr *FileCheckPattern::parseNumericSubstitutionBlock(
295 StringRef Expr, FileCheckNumericVariable *&DefinedNumericVariable,
296 const SourceMgr &SM) const {
297 // Parse the numeric variable definition.
298 DefinedNumericVariable = nullptr;
299 size_t DefEnd = Expr.find(':');
300 if (DefEnd != StringRef::npos) {
301 StringRef DefExpr = Expr.substr(0, DefEnd);
302 StringRef UseExpr = Expr = Expr.substr(DefEnd + 1);
303
304 DefExpr = DefExpr.ltrim(SpaceChars);
305 StringRef Name;
306 if (parseNumericVariableDefinition(DefExpr, Name, Context, SM)) {
307 // Invalid variable definition. Error reporting done in parsing function.
308 return nullptr;
309 }
310
311 DefinedNumericVariable =
312 Context->makeNumericVariable(this->LineNumber, Name);
313
314 DefExpr = DefExpr.ltrim(SpaceChars);
315 if (!DefExpr.empty()) {
316 SM.PrintMessage(SMLoc::getFromPointer(DefExpr.data()),
317 SourceMgr::DK_Error,
318 "invalid numeric variable definition");
319 return nullptr;
320 }
321 UseExpr = UseExpr.ltrim(SpaceChars);
322 if (!UseExpr.empty()) {
323 SM.PrintMessage(
324 SMLoc::getFromPointer(UseExpr.data()), SourceMgr::DK_Error,
325 "unexpected string after variable definition: '" + UseExpr + "'");
326 return nullptr;
327 }
328 return Context->makeNumExpr(add, nullptr, 0);
329 }
330
331 // Parse the numeric expression itself.
332 Expr = Expr.ltrim(SpaceChars);
333 return parseBinop(Expr, SM);
334}

336bool FileCheckPattern::parsePattern(StringRef PatternStr, StringRef Prefix,
337 SourceMgr &SM,
338 const FileCheckRequest &Req) {
339 bool MatchFullLinesHere = Req.MatchFullLines && CheckTy != Check::CheckNot;
340
341 PatternLoc = SMLoc::getFromPointer(PatternStr.data());
342
343 // Create fake @LINE pseudo variable definition.
344 StringRef LinePseudo = "@LINE";
345 uint64_t LineNumber64 = LineNumber;
346 FileCheckNumericVariable *LinePseudoVar =
347 Context->makeNumericVariable(LinePseudo, LineNumber64);
348 Context->GlobalNumericVariableTable[LinePseudo] = LinePseudoVar;
349
350 if (!(Req.NoCanonicalizeWhiteSpace && Req.MatchFullLines))
351 // Ignore trailing whitespace.
352 while (!PatternStr.empty() &&
353 (PatternStr.back() == ' ' || PatternStr.back() == '\t'))
354 PatternStr = PatternStr.substr(0, PatternStr.size() - 1);
355
356 // Check that there is something on the line.
357 if (PatternStr.empty() && CheckTy != Check::CheckEmpty) {
358 SM.PrintMessage(PatternLoc, SourceMgr::DK_Error,
359 "found empty check string with prefix '" + Prefix + ":'");
360 return true;
361 }
362
363 if (!PatternStr.empty() && CheckTy == Check::CheckEmpty) {
364 SM.PrintMessage(
365 PatternLoc, SourceMgr::DK_Error,
366 "found non-empty check string for empty check with prefix '" + Prefix +
367 ":'");
368 return true;
369 }
370
371 if (CheckTy == Check::CheckEmpty) {
372 RegExStr = "(\n$)";
373 return false;
374 }
375
376 // Check to see if this is a fixed string, or if it has regex pieces.
377 if (!MatchFullLinesHere &&
378 (PatternStr.size() < 2 || (PatternStr.find("{{") == StringRef::npos &&
379 PatternStr.find("[[") == StringRef::npos))) {
380 FixedStr = PatternStr;
381 return false;
382 }
383
384 if (MatchFullLinesHere) {
385 RegExStr += '^';
386 if (!Req.NoCanonicalizeWhiteSpace)
387 RegExStr += " *";
388 }
389
390 // Paren value #0 is for the fully matched string. Any new parenthesized
391 // values add from there.
392 unsigned CurParen = 1;
393
394 // Otherwise, there is at least one regex piece. Build up the regex pattern
395 // by escaping scary characters in fixed strings, building up one big regex.
396 while (!PatternStr.empty()) {
397 // RegEx matches.
398 if (PatternStr.startswith("{{")) {
399 // This is the start of a regex match. Scan for the }}.
400 size_t End = PatternStr.find("}}");
401 if (End == StringRef::npos) {
402 SM.PrintMessage(SMLoc::getFromPointer(PatternStr.data()),
403 SourceMgr::DK_Error,
404 "found start of regex string with no end '}}'");
405 return true;
406 }
407
408 // Enclose {{}} patterns in parens just like [[]] even though we're not
409 // capturing the result for any purpose. This is required in case the
410 // expression contains an alternation like: CHECK: abc{{x|z}}def. We
411 // want this to turn into: "abc(x|z)def" not "abcx|zdef".
412 RegExStr += '(';
413 ++CurParen;
414
415 if (AddRegExToRegEx(PatternStr.substr(2, End - 2), CurParen, SM))
416 return true;
417 RegExStr += ')';
418
419 PatternStr = PatternStr.substr(End + 2);
420 continue;
421 }
422
423 // String and numeric substitution blocks. String substitution blocks come
424 // in two forms: [[foo:.*]] and [[foo]]. The former matches .* (or some
425 // other regex) and assigns it to the string variable 'foo'. The latter
426 // substitutes foo's value. Numeric substitution blocks work the same way
427 // as string ones, but start with a '#' sign after the double brackets.
428 // Both string and numeric variable names must satisfy the regular
429 // expression "[a-zA-Z_][0-9a-zA-Z_]*" to be valid, as this helps catch
430 // some common errors.
431 if (PatternStr.startswith("[[")) {
432 StringRef UnparsedPatternStr = PatternStr.substr(2);
433 // Find the closing bracket pair ending the match. End is going to be an
434 // offset relative to the beginning of the match string.
435 size_t End = FindRegexVarEnd(UnparsedPatternStr, SM);
436 StringRef MatchStr = UnparsedPatternStr.substr(0, End);
437 bool IsNumBlock = MatchStr.consume_front("#");
438
439 if (End == StringRef::npos) {
440 SM.PrintMessage(SMLoc::getFromPointer(PatternStr.data()),
441 SourceMgr::DK_Error,
442 "Invalid substitution block, no ]] found");
443 return true;
444 }
445 // Strip the substitution block we are parsing. End points to the start
446 // of the "]]" closing the expression so account for it in computing the
447 // index of the first unparsed character.
448 PatternStr = UnparsedPatternStr.substr(End + 2);
449
450 bool IsDefinition = false;
451 StringRef DefName;
452 StringRef SubstStr;
453 StringRef MatchRegexp;
454 size_t SubstInsertIdx = RegExStr.size();
455
456 // Parse string variable or legacy numeric expression.
457 if (!IsNumBlock) {
458 size_t VarEndIdx = MatchStr.find(":");
459 size_t SpacePos = MatchStr.substr(0, VarEndIdx).find_first_of(" \t");
460 if (SpacePos != StringRef::npos) {
461 SM.PrintMessage(SMLoc::getFromPointer(MatchStr.data() + SpacePos),
462 SourceMgr::DK_Error, "unexpected whitespace");
463 return true;
464 }
465
466 // Get the name (e.g. "foo") and verify it is well formed.
467 bool IsPseudo;
468 StringRef Name;
469 StringRef OrigMatchStr = MatchStr;
470 if (parseVariable(MatchStr, Name, IsPseudo)) {
471 SM.PrintMessage(SMLoc::getFromPointer(MatchStr.data()),
472 SourceMgr::DK_Error, "invalid variable name");
473 return true;
474 }
475
476 IsDefinition = (VarEndIdx != StringRef::npos);
477 if (IsDefinition) {
478 if ((IsPseudo || !MatchStr.consume_front(":"))) {
479 SM.PrintMessage(SMLoc::getFromPointer(Name.data()),
480 SourceMgr::DK_Error,
481 "invalid name in string variable definition");
482 return true;
483 }
484
485 // Detect collisions between string and numeric variables when the
486 // former is created later than the latter.
487 if (Context->GlobalNumericVariableTable.find(Name) !=
488 Context->GlobalNumericVariableTable.end()) {
489 SM.PrintMessage(
490 SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
491 "numeric variable with name '" + Name + "' already exists");
492 return true;
493 }
494 DefName = Name;
495 MatchRegexp = MatchStr;
496 } else {
497 if (IsPseudo) {
498 MatchStr = OrigMatchStr;
499 IsNumBlock = true;
500 } else
501 SubstStr = Name;
502 }
503 }
504
505 // Parse numeric substitution block.
506 FileCheckNumExpr *NumExpr;
507 FileCheckNumericVariable *DefinedNumericVariable;
508 if (IsNumBlock) {
509 NumExpr =
510 parseNumericSubstitutionBlock(MatchStr, DefinedNumericVariable, SM);
511 if (NumExpr == nullptr)
512 return true;
513 if (DefinedNumericVariable) {
514 IsDefinition = true;
515 DefName = DefinedNumericVariable->getName();
516 MatchRegexp = StringRef("[0-9]+");
517 } else
518 SubstStr = MatchStr;
519 }
520
521 // Handle substitutions: [[foo]] and [[#<foo expr>]].
522 if (!IsDefinition) {
523 // Handle substitution of string variables that were defined earlier on
524 // the same line by emitting a backreference. Numeric expressions do
525 // not support substituting a numeric variable defined on the same
526 // line.
527 if (!IsNumBlock && VariableDefs.find(SubstStr) != VariableDefs.end()) {
528 unsigned CaptureParenGroup = VariableDefs[SubstStr];
529 if (CaptureParenGroup < 1 || CaptureParenGroup > 9) {
530 SM.PrintMessage(SMLoc::getFromPointer(SubstStr.data()),
531 SourceMgr::DK_Error,
532 "Can't back-reference more than 9 variables");
533 return true;
534 }
535 AddBackrefToRegEx(CaptureParenGroup);
536 } else {
537 // Handle substitution of string variables ([[<var>]]) defined in
538 // previous CHECK patterns, and substitution of numeric expressions.
539 FileCheckSubstitution *Substitution =
540 IsNumBlock
541 ? Context->makeNumericSubstitution(SubstStr, NumExpr,
542 SubstInsertIdx)
543 : Context->makeStringSubstitution(SubstStr, SubstInsertIdx);
544 Substitutions.push_back(Substitution);
545 }
546 continue;
547 }
548
549 // Handle variable definitions: [[<def>:(...)]] and
550 // [[#(...)<def>:(...)]].
551 if (IsNumBlock) {
552 FileCheckNumExprMatch NumExprDef = {DefinedNumericVariable, CurParen};
553 NumericVariableDefs[DefName] = NumExprDef;
554 // This store is done here rather than in match() to allow
555 // parseNumericVariableUse() to get the pointer to the class instance
556 // of the right variable definition corresponding to a given numeric
557 // variable use.
558 Context->GlobalNumericVariableTable[DefName] = DefinedNumericVariable;
559 } else {
560 VariableDefs[DefName] = CurParen;
561 // Mark the string variable as defined to detect collisions between
562 // string and numeric variables in parseNumericVariableUse() and
563 // DefineCmdlineVariables() when the latter is created later than the
564 // former. We cannot reuse GlobalVariableTable for this by populating
565 // it with an empty string since we would then lose the ability to
566 // detect the use of an undefined variable in match().
567 Context->DefinedVariableTable[DefName] = true;
568 }
569 RegExStr += '(';
570 ++CurParen;
571
572 if (AddRegExToRegEx(MatchRegexp, CurParen, SM))
573 return true;
574
575 RegExStr += ')';
576 }
577
578 // Handle fixed string matches.
579 // Find the end, which is the start of the next regex.
580 size_t FixedMatchEnd = PatternStr.find("{{");
581 FixedMatchEnd = std::min(FixedMatchEnd, PatternStr.find("[["));
582 RegExStr += Regex::escape(PatternStr.substr(0, FixedMatchEnd));
583 PatternStr = PatternStr.substr(FixedMatchEnd);
584 }
585
586 if (MatchFullLinesHere) {
587 if (!Req.NoCanonicalizeWhiteSpace)
588 RegExStr += " *";
589 RegExStr += '$';
590 }
591
592 return false;
593}

595bool FileCheckPattern::AddRegExToRegEx(StringRef RS, unsigned &CurParen, SourceMgr &SM) {
596 Regex R(RS);
597 std::string Error;
598 if (!R.isValid(Error)) {
599 SM.PrintMessage(SMLoc::getFromPointer(RS.data()), SourceMgr::DK_Error,
600 "invalid regex: " + Error);
601 return true;
602 }
603
604 RegExStr += RS.str();
605 CurParen += R.getNumMatches();
606 return false;
607}

609void FileCheckPattern::AddBackrefToRegEx(unsigned BackrefNum) {
610 assert(BackrefNum >= 1 && BackrefNum <= 9 && "Invalid backref number");
611 std::string Backref = std::string("\\") + std::string(1, '0' + BackrefNum);
612 RegExStr += Backref;
613}

615size_t FileCheckPattern::match(StringRef Buffer, size_t &MatchLen,
616 const SourceMgr &SM) const {
617 // If this is the EOF pattern, match it immediately.
618 if (CheckTy == Check::CheckEOF) {
619 MatchLen = 0;
620 return Buffer.size();
621 }
622
623 // If this is a fixed string pattern, just match it now.
624 if (!FixedStr.empty()) {
625 MatchLen = FixedStr.size();
626 return Buffer.find(FixedStr);
627 }
628
629 // Regex match.
630
631 // If there are substitutions, we need to create a temporary string with the
632 // actual value.
633 StringRef RegExToMatch = RegExStr;
634 std::string TmpStr;
635 if (!Substitutions.empty()) {
636 TmpStr = RegExStr;
637
638 size_t InsertOffset = 0;
639 // Substitute all string variables and numeric expressions whose values are
640 // only now known. Use of string variables defined on the same line are
641 // handled by back-references.
642 for (const auto &Substitution : Substitutions) {
643 // Substitute and check for failure (e.g. use of undefined variable).
644 Optional<std::string> Value = Substitution->getResult();
645 if (!Value)
646 return StringRef::npos;
647
648 // Plop it into the regex at the adjusted offset.
649 TmpStr.insert(TmpStr.begin() + Substitution->getIndex() + InsertOffset,
650 Value->begin(), Value->end());
651 InsertOffset += Value->size();
652 }
653
654 // Match the newly constructed regex.
655 RegExToMatch = TmpStr;
656 }
657
658 SmallVector<StringRef, 4> MatchInfo;
659 if (!Regex(RegExToMatch, Regex::Newline).match(Buffer, &MatchInfo))
660 return StringRef::npos;
661
662 // Successful regex match.
663 assert(!MatchInfo.empty() && "Didn't get any match");
664 StringRef FullMatch = MatchInfo[0];
665
666 // If this defines any string variables, remember their values.
667 for (const auto &VariableDef : VariableDefs) {
668 assert(VariableDef.second < MatchInfo.size() && "Internal paren error");
669 Context->GlobalVariableTable[VariableDef.first] =
670 MatchInfo[VariableDef.second];
671 }
672
673 // If this defines any numeric variables, remember their values.
674 for (const auto &NumericVariableDef : NumericVariableDefs) {
675 const FileCheckNumExprMatch &NumericVariableMatch =
676 NumericVariableDef.getValue();
677 unsigned CaptureParenGroup = NumericVariableMatch.CaptureParenGroup;
678 assert(CaptureParenGroup < MatchInfo.size() && "Internal paren error");
679 FileCheckNumericVariable *DefinedNumericVariable =
680 NumericVariableMatch.DefinedNumericVariable;
681
682 StringRef MatchedValue = MatchInfo[CaptureParenGroup];
683 uint64_t Val;
684 if (MatchedValue.getAsInteger(10, Val)) {
685 SM.PrintMessage(SMLoc::getFromPointer(MatchedValue.data()),
686 SourceMgr::DK_Error, "Unable to represent numeric value");
687 }
688 if (DefinedNumericVariable->setValue(Val))
689 assert(false && "Numeric variable redefined");
690 }
691
692 // Like CHECK-NEXT, CHECK-EMPTY's match range is considered to start after
693 // the required preceding newline, which is consumed by the pattern in the
694 // case of CHECK-EMPTY but not CHECK-NEXT.
695 size_t MatchStartSkip = CheckTy == Check::CheckEmpty;
696 MatchLen = FullMatch.size() - MatchStartSkip;
697 return FullMatch.data() - Buffer.data() + MatchStartSkip;
698}

700unsigned FileCheckPattern::computeMatchDistance(StringRef Buffer) const {
701 // Just compute the number of matching characters. For regular expressions, we
702 // just compare against the regex itself and hope for the best.
703 //
704 // FIXME: One easy improvement here is have the regex lib generate a single
705 // example regular expression which matches, and use that as the example
706 // string.
707 StringRef ExampleString(FixedStr);
708 if (ExampleString.empty())
709 ExampleString = RegExStr;
710
711 // Only compare up to the first line in the buffer, or the string size.
712 StringRef BufferPrefix = Buffer.substr(0, ExampleString.size());
713 BufferPrefix = BufferPrefix.split('\n').first;

769void FileCheckPattern::printFuzzyMatch(
770 const SourceMgr &SM, StringRef Buffer,
771 std::vector<FileCheckDiag> *Diags) const {
772 // Attempt to find the closest/best fuzzy match. Usually an error happens
773 // because some string in the output didn't exactly match. In these cases, we
774 // would like to show the user a best guess at what "should have" matched, to
775 // save them having to actually check the input manually.
776 size_t NumLinesForward = 0;
777 size_t Best = StringRef::npos;
778 double BestQuality = 0;
779
780 // Use an arbitrary 4k limit on how far we will search.
781 for (size_t i = 0, e = std::min(size_t(4096), Buffer.size()); i != e; ++i) {
782 if (Buffer[i] == '\n')
783 ++NumLinesForward;
784
785 // Patterns have leading whitespace stripped, so skip whitespace when
786 // looking for something which looks like a pattern.
787 if (Buffer[i] == ' ' || Buffer[i] == '\t')
788 continue;
789
790 // Compute the "quality" of this match as an arbitrary combination of the
791 // match distance and the number of lines skipped to get to this match.
792 unsigned Distance = computeMatchDistance(Buffer.substr(i));
793 double Quality = Distance + (NumLinesForward / 100.);
794
795 if (Quality < BestQuality || Best == StringRef::npos) {
796 Best = i;
797 BestQuality = Quality;
798 }
799 }
800
801 // Print the "possible intended match here" line if we found something
802 // reasonable and not equal to what we showed in the "scanning from here"
803 // line.
804 if (Best && Best != StringRef::npos && BestQuality < 50) {
805 SMRange MatchRange =
806 ProcessMatchResult(FileCheckDiag::MatchFuzzy, SM, getLoc(),
807 getCheckTy(), Buffer, Best, 0, Diags);
808 SM.PrintMessage(MatchRange.Start, SourceMgr::DK_Note,
809 "possible intended match here");
810
811 // FIXME: If we wanted to be really friendly we would show why the match
812 // failed, as it can be hard to spot simple one character differences.
813 }
814}

857size_t FileCheckPattern::FindRegexVarEnd(StringRef Str, SourceMgr &SM) {
858 // Offset keeps track of the current offset within the input Str
859 size_t Offset = 0;
860 // [...] Nesting depth
861 size_t BracketDepth = 0;
862
863 while (!Str.empty()) {
864 if (Str.startswith("]]") && BracketDepth == 0)
865 return Offset;
866 if (Str[0] == '\\') {
867 // Backslash escapes the next char within regexes, so skip them both.
868 Str = Str.substr(2);
869 Offset += 2;
870 } else {
871 switch (Str[0]) {
872 default:
873 break;
874 case '[':
875 BracketDepth++;
876 break;
877 case ']':
878 if (BracketDepth == 0) {
879 SM.PrintMessage(SMLoc::getFromPointer(Str.data()),
880 SourceMgr::DK_Error,
881 "missing closing \"]\" for regex variable");
882 exit(1);
883 }
884 BracketDepth--;
885 break;
886 }
887 Str = Str.substr(1);
888 Offset++;
889 }
890 }
891
892 return StringRef::npos;
893}

895StringRef FileCheck::CanonicalizeFile(MemoryBuffer &MB,
896 SmallVectorImpl<char> &OutputBuffer) {
897 OutputBuffer.reserve(MB.getBufferSize());
898
899 for (const char *Ptr = MB.getBufferStart(), *End = MB.getBufferEnd();
900 Ptr != End; ++Ptr) {
901 // Eliminate trailing dosish \r.
902 if (Ptr <= End - 2 && Ptr[0] == '\r' && Ptr[1] == '\n') {
903 continue;
904 }
905
906 // If current char is not a horizontal whitespace or if horizontal
907 // whitespace canonicalization is disabled, dump it to output as is.
908 if (Req.NoCanonicalizeWhiteSpace || (*Ptr != ' ' && *Ptr != '\t')) {
909 OutputBuffer.push_back(*Ptr);
910 continue;
911 }
912
913 // Otherwise, add one space and advance over neighboring space.
914 OutputBuffer.push_back(' ');
915 while (Ptr + 1 != End && (Ptr[1] == ' ' || Ptr[1] == '\t'))
916 ++Ptr;
917 }
918
919 // Add a null byte and then return all but that byte.
920 OutputBuffer.push_back('\0');
921 return StringRef(OutputBuffer.data(), OutputBuffer.size() - 1);
922}

982static std::pair<Check::FileCheckType, StringRef>
983FindCheckType(StringRef Buffer, StringRef Prefix) {
984 if (Buffer.size() <= Prefix.size())
985 return {Check::CheckNone, StringRef()};
986
987 char NextChar = Buffer[Prefix.size()];
988
989 StringRef Rest = Buffer.drop_front(Prefix.size() + 1);
990 // Verify that the : is present after the prefix.
991 if (NextChar == ':')
992 return {Check::CheckPlain, Rest};
993
994 if (NextChar != '-')
995 return {Check::CheckNone, StringRef()};
996
997 if (Rest.consume_front("COUNT-")) {
998 int64_t Count;
999 if (Rest.consumeInteger(10, Count))
1000 // Error happened in parsing integer.
1001 return {Check::CheckBadCount, Rest};
1002 if (Count <= 0 || Count > INT32_MAX)
1003 return {Check::CheckBadCount, Rest};
1004 if (!Rest.consume_front(":"))
1005 return {Check::CheckBadCount, Rest};
1006 return {Check::FileCheckType(Check::CheckPlain).setCount(Count), Rest};
1007 }
1008
1009 if (Rest.consume_front("NEXT:"))
1010 return {Check::CheckNext, Rest};
1011
1012 if (Rest.consume_front("SAME:"))
1013 return {Check::CheckSame, Rest};
1014
1015 if (Rest.consume_front("NOT:"))
1016 return {Check::CheckNot, Rest};
1017
1018 if (Rest.consume_front("DAG:"))
1019 return {Check::CheckDAG, Rest};
1020
1021 if (Rest.consume_front("LABEL:"))
1022 return {Check::CheckLabel, Rest};
1023
1024 if (Rest.consume_front("EMPTY:"))
1025 return {Check::CheckEmpty, Rest};
1026
1027 // You can't combine -NOT with another suffix.
1028 if (Rest.startswith("DAG-NOT:") || Rest.startswith("NOT-DAG:") ||
1029 Rest.startswith("NEXT-NOT:") || Rest.startswith("NOT-NEXT:") ||
1030 Rest.startswith("SAME-NOT:") || Rest.startswith("NOT-SAME:") ||
1031 Rest.startswith("EMPTY-NOT:") || Rest.startswith("NOT-EMPTY:"))
1032 return {Check::CheckBadNot, Rest};
1033
1034 return {Check::CheckNone, Rest};
1035}

1037// From the given position, find the next character after the word.
1038static size_t SkipWord(StringRef Str, size_t Loc) {
1039 while (Loc < Str.size() && IsPartOfWord(Str[Loc]))
1040 ++Loc;
1041 return Loc;
1042}
1043
1044/// Searches the buffer for the first prefix in the prefix regular expression.
1045///
1046/// This searches the buffer using the provided regular expression, however it
1047/// enforces constraints beyond that:
1048/// 1) The found prefix must not be a suffix of something that looks like
1049/// a valid prefix.
1050/// 2) The found prefix must be followed by a valid check type suffix using \c
1051/// FindCheckType above.
1052///
1053/// \returns a pair of StringRefs into the Buffer, which combines:
1054/// - the first match of the regular expression to satisfy these two is
1055/// returned,
1056/// otherwise an empty StringRef is returned to indicate failure.
1057/// - buffer rewound to the location right after parsed suffix, for parsing
1058/// to continue from
1059///
1060/// If this routine returns a valid prefix, it will also shrink \p Buffer to
1061/// start at the beginning of the returned prefix, increment \p LineNumber for
1062/// each new line consumed from \p Buffer, and set \p CheckTy to the type of
1063/// check found by examining the suffix.
1064///
1065/// If no valid prefix is found, the state of Buffer, LineNumber, and CheckTy
1066/// is unspecified.
1067static std::pair<StringRef, StringRef>
1068FindFirstMatchingPrefix(Regex &PrefixRE, StringRef &Buffer,
1069 unsigned &LineNumber, Check::FileCheckType &CheckTy) {
1070 SmallVector<StringRef, 2> Matches;
1071
1072 while (!Buffer.empty()) {
1073 // Find the first (longest) match using the RE.
1074 if (!PrefixRE.match(Buffer, &Matches))
1075 // No match at all, bail.
1076 return {StringRef(), StringRef()};
1077
1078 StringRef Prefix = Matches[0];
1079 Matches.clear();
1080
1081 assert(Prefix.data() >= Buffer.data() &&
1082 Prefix.data() < Buffer.data() + Buffer.size() &&
1083 "Prefix doesn't start inside of buffer!");
1084 size_t Loc = Prefix.data() - Buffer.data();
1085 StringRef Skipped = Buffer.substr(0, Loc);
1086 Buffer = Buffer.drop_front(Loc);
1087 LineNumber += Skipped.count('\n');
1088
1089 // Check that the matched prefix isn't a suffix of some other check-like
1090 // word.
1091 // FIXME: This is a very ad-hoc check. it would be better handled in some
1092 // other way. Among other things it seems hard to distinguish between
1093 // intentional and unintentional uses of this feature.
1094 if (Skipped.empty() || !IsPartOfWord(Skipped.back())) {
1095 // Now extract the type.
1096 StringRef AfterSuffix;
1097 std::tie(CheckTy, AfterSuffix) = FindCheckType(Buffer, Prefix);
1098
1099 // If we've found a valid check type for this prefix, we're done.
1100 if (CheckTy != Check::CheckNone)
1101 return {Prefix, AfterSuffix};
1102 }
1103
1104 // If we didn't successfully find a prefix, we need to skip this invalid
1105 // prefix and continue scanning. We directly skip the prefix that was
1106 // matched and any additional parts of that check-like word.
1107 Buffer = Buffer.drop_front(SkipWord(Buffer, Prefix.size()));
1108 }
1109
1110 // We ran out of buffer while skipping partial matches so give up.
1111 return {StringRef(), StringRef()};
1112}

1114bool FileCheck::ReadCheckFile(SourceMgr &SM, StringRef Buffer, Regex &PrefixRE,
1115 std::vector<FileCheckString> &CheckStrings) {
1116 if (PatternContext.defineCmdlineVariables(Req.GlobalDefines, SM))
1117 return true;
1118
1119 std::vector<FileCheckPattern> ImplicitNegativeChecks;
1120 for (const auto &PatternString : Req.ImplicitCheckNot) {
1121 // Create a buffer with fake command line content in order to display the
1122 // command line option responsible for the specific implicit CHECK-NOT.
1123 std::string Prefix = "-implicit-check-not='";
1124 std::string Suffix = "'";
1125 std::unique_ptr<MemoryBuffer> CmdLine = MemoryBuffer::getMemBufferCopy(
1126 Prefix + PatternString + Suffix, "command line");
1127
1128 StringRef PatternInBuffer =
1129 CmdLine->getBuffer().substr(Prefix.size(), PatternString.size());
1130 SM.AddNewSourceBuffer(std::move(CmdLine), SMLoc());
1131
1132 ImplicitNegativeChecks.push_back(
1133 FileCheckPattern(Check::CheckNot, &PatternContext, 0));
1134 ImplicitNegativeChecks.back().parsePattern(PatternInBuffer,
1135 "IMPLICIT-CHECK", SM, Req);
1136 }
1137
1138 std::vector<FileCheckPattern> DagNotMatches = ImplicitNegativeChecks;
1139
1140 // LineNumber keeps track of the line on which CheckPrefix instances are
1141 // found.
1142 unsigned LineNumber = 1;
1143
1144 while (1) {
1145 Check::FileCheckType CheckTy;
1146
1147 // See if a prefix occurs in the memory buffer.
1148 StringRef UsedPrefix;
1149 StringRef AfterSuffix;
1150 std::tie(UsedPrefix, AfterSuffix) =
1151 FindFirstMatchingPrefix(PrefixRE, Buffer, LineNumber, CheckTy);
1152 if (UsedPrefix.empty())
1153 break;
1154 assert(UsedPrefix.data() == Buffer.data() &&
1155 "Failed to move Buffer's start forward, or pointed prefix outside "
1156 "of the buffer!");
1157 assert(AfterSuffix.data() >= Buffer.data() &&
1158 AfterSuffix.data() < Buffer.data() + Buffer.size() &&
1159 "Parsing after suffix doesn't start inside of buffer!");
1160
1161 // Location to use for error messages.
1162 const char *UsedPrefixStart = UsedPrefix.data();
1163
1164 // Skip the buffer to the end of parsed suffix (or just prefix, if no good
1165 // suffix was processed).
1166 Buffer = AfterSuffix.empty() ? Buffer.drop_front(UsedPrefix.size())
1167 : AfterSuffix;
1168
1169 // Complain about useful-looking but unsupported suffixes.
1170 if (CheckTy == Check::CheckBadNot) {
1171 SM.PrintMessage(SMLoc::getFromPointer(Buffer.data()), SourceMgr::DK_Error,
1172 "unsupported -NOT combo on prefix '" + UsedPrefix + "'");
1173 return true;
1174 }
1175
1176 // Complain about invalid count specification.
1177 if (CheckTy == Check::CheckBadCount) {
1178 SM.PrintMessage(SMLoc::getFromPointer(Buffer.data()), SourceMgr::DK_Error,
1179 "invalid count in -COUNT specification on prefix '" +
1180 UsedPrefix + "'");
1181 return true;
1182 }
1183
1184 // Okay, we found the prefix, yay. Remember the rest of the line, but ignore
1185 // leading whitespace.
1186 if (!(Req.NoCanonicalizeWhiteSpace && Req.MatchFullLines))
1187 Buffer = Buffer.substr(Buffer.find_first_not_of(" \t"));
1188
1189 // Scan ahead to the end of line.
1190 size_t EOL = Buffer.find_first_of("\n\r");
1191
1192 // Remember the location of the start of the pattern, for diagnostics.
1193 SMLoc PatternLoc = SMLoc::getFromPointer(Buffer.data());
1194
1195 // Parse the pattern.
1196 FileCheckPattern P(CheckTy, &PatternContext, LineNumber);
1197 if (P.parsePattern(Buffer.substr(0, EOL), UsedPrefix, SM, Req))
1198 return true;
1199
1200 // Verify that CHECK-LABEL lines do not define or use variables
1201 if ((CheckTy == Check::CheckLabel) && P.hasVariable()) {
1202 SM.PrintMessage(
1203 SMLoc::getFromPointer(UsedPrefixStart), SourceMgr::DK_Error,
1204 "found '" + UsedPrefix + "-LABEL:'"
1205 " with variable definition or use");
1206 return true;
1207 }
1208
1209 Buffer = Buffer.substr(EOL);
1210
1211 // Verify that CHECK-NEXT/SAME/EMPTY lines have at least one CHECK line before them.
1212 if ((CheckTy == Check::CheckNext || CheckTy == Check::CheckSame ||
1213 CheckTy == Check::CheckEmpty) &&
1214 CheckStrings.empty()) {
1215 StringRef Type = CheckTy == Check::CheckNext
1216 ? "NEXT"
1217 : CheckTy == Check::CheckEmpty ? "EMPTY" : "SAME";
1218 SM.PrintMessage(SMLoc::getFromPointer(UsedPrefixStart),
1219 SourceMgr::DK_Error,
1220 "found '" + UsedPrefix + "-" + Type +
1221 "' without previous '" + UsedPrefix + ": line");
1222 return true;
1223 }
1224
1225 // Handle CHECK-DAG/-NOT.
1226 if (CheckTy == Check::CheckDAG || CheckTy == Check::CheckNot) {
1227 DagNotMatches.push_back(P);
1228 continue;
1229 }
1230
1231 // Okay, add the string we captured to the output vector and move on.
1232 CheckStrings.emplace_back(P, UsedPrefix, PatternLoc);
1233 std::swap(DagNotMatches, CheckStrings.back().DagNotStrings);
1234 DagNotMatches = ImplicitNegativeChecks;
1235 }
1236
1237 // Add an EOF pattern for any trailing CHECK-DAG/-NOTs, and use the first
1238 // prefix as a filler for the error message.
1239 if (!DagNotMatches.empty()) {
1240 CheckStrings.emplace_back(
1241 FileCheckPattern(Check::CheckEOF, &PatternContext, LineNumber + 1),
1242 *Req.CheckPrefixes.begin(), SMLoc::getFromPointer(Buffer.data()));
1243 std::swap(DagNotMatches, CheckStrings.back().DagNotStrings);
1244 }
1245
1246 if (CheckStrings.empty()) {
1247 errs() << "error: no check strings found with prefix"
1248 << (Req.CheckPrefixes.size() > 1 ? "es " : " ");
1249 auto I = Req.CheckPrefixes.begin();
1250 auto E = Req.CheckPrefixes.end();
1251 if (I != E) {
1252 errs() << "\'" << *I << ":'";
1253 ++I;
1254 }
1255 for (; I != E; ++I)
1256 errs() << ", \'" << *I << ":'";
1257
1258 errs() << '\n';
1259 return true;
1260 }
1261
1262 return false;
1263}

1364/// Counts the number of newlines in the specified range.
1365static unsigned CountNumNewlinesBetween(StringRef Range,
1366 const char *&FirstNewLine) {
1367 unsigned NumNewLines = 0;
1368 while (1) {
1369 // Scan for newline.
1370 Range = Range.substr(Range.find_first_of("\n\r"));
1371 if (Range.empty())
1372 return NumNewLines;
1373
1374 ++NumNewLines;
1375
1376 // Handle \n\r and \r\n as a single newline.
1377 if (Range.size() > 1 && (Range[1] == '\n' || Range[1] == '\r') &&
1378 (Range[0] != Range[1]))
1379 Range = Range.substr(1);
1380 Range = Range.substr(1);
1381
1382 if (NumNewLines == 1)
1383 FirstNewLine = Range.begin();
1384 }

1387size_t FileCheckString::Check(const SourceMgr &SM, StringRef Buffer,
1388 bool IsLabelScanMode, size_t &MatchLen,
1389 FileCheckRequest &Req,
1390 std::vector<FileCheckDiag> *Diags) const {
1391 size_t LastPos = 0;
1392 std::vector<const FileCheckPattern *> NotStrings;
1393
1394 // IsLabelScanMode is true when we are scanning forward to find CHECK-LABEL
1395 // bounds; we have not processed variable definitions within the bounded block
1396 // yet so cannot handle any final CHECK-DAG yet; this is handled when going
1397 // over the block again (including the last CHECK-LABEL) in normal mode.
1398 if (!IsLabelScanMode) {
1399 // Match "dag strings" (with mixed "not strings" if any).
1400 LastPos = CheckDag(SM, Buffer, NotStrings, Req, Diags);
1401 if (LastPos == StringRef::npos)
1402 return StringRef::npos;
1403 }
1404
1405 // Match itself from the last position after matching CHECK-DAG.
1406 size_t LastMatchEnd = LastPos;
1407 size_t FirstMatchPos = 0;
1408 // Go match the pattern Count times. Majority of patterns only match with
1409 // count 1 though.
1410 assert(Pat.getCount() != 0 && "pattern count can not be zero");
1411 for (int i = 1; i <= Pat.getCount(); i++) {
1412 StringRef MatchBuffer = Buffer.substr(LastMatchEnd);
1413 size_t CurrentMatchLen;
1414 // get a match at current start point
1415 size_t MatchPos = Pat.match(MatchBuffer, CurrentMatchLen, SM);
1416 if (i == 1)
1417 FirstMatchPos = LastPos + MatchPos;
1418
1419 // report
1420 if (MatchPos == StringRef::npos) {
1421 PrintNoMatch(true, SM, *this, i, MatchBuffer, Req.VerboseVerbose, Diags);
1422 return StringRef::npos;
1423 }
1424 PrintMatch(true, SM, *this, i, MatchBuffer, MatchPos, CurrentMatchLen, Req,
1425 Diags);
1426
1427 // move start point after the match
1428 LastMatchEnd += MatchPos + CurrentMatchLen;
1429 }
1430 // Full match len counts from first match pos.
1431 MatchLen = LastMatchEnd - FirstMatchPos;
1432
1433 // Similar to the above, in "label-scan mode" we can't yet handle CHECK-NEXT
1434 // or CHECK-NOT
1435 if (!IsLabelScanMode) {
1436 size_t MatchPos = FirstMatchPos - LastPos;
1437 StringRef MatchBuffer = Buffer.substr(LastPos);
1438 StringRef SkippedRegion = Buffer.substr(LastPos, MatchPos);
1439
1440 // If this check is a "CHECK-NEXT", verify that the previous match was on
1441 // the previous line (i.e. that there is one newline between them).
1442 if (CheckNext(SM, SkippedRegion)) {
1443 ProcessMatchResult(FileCheckDiag::MatchFoundButWrongLine, SM, Loc,
1444 Pat.getCheckTy(), MatchBuffer, MatchPos, MatchLen,
1445 Diags, Req.Verbose);
1446 return StringRef::npos;
1447 }
1448
1449 // If this check is a "CHECK-SAME", verify that the previous match was on
1450 // the same line (i.e. that there is no newline between them).
1451 if (CheckSame(SM, SkippedRegion)) {
1452 ProcessMatchResult(FileCheckDiag::MatchFoundButWrongLine, SM, Loc,
1453 Pat.getCheckTy(), MatchBuffer, MatchPos, MatchLen,
1454 Diags, Req.Verbose);
1455 return StringRef::npos;
1456 }
1457
1458 // If this match had "not strings", verify that they don't exist in the
1459 // skipped region.
1460 if (CheckNot(SM, SkippedRegion, NotStrings, Req, Diags))
1461 return StringRef::npos;
1462 }
1463
1464 return FirstMatchPos;
1465}

1467bool FileCheckString::CheckNext(const SourceMgr &SM, StringRef Buffer) const {
1468 if (Pat.getCheckTy() != Check::CheckNext &&
1469 Pat.getCheckTy() != Check::CheckEmpty)
1470 return false;
1471
1472 Twine CheckName =
1473 Prefix +
1474 Twine(Pat.getCheckTy() == Check::CheckEmpty ? "-EMPTY" : "-NEXT");
1475
1476 // Count the number of newlines between the previous match and this one.
1477 const char *FirstNewLine = nullptr;
1478 unsigned NumNewLines = CountNumNewlinesBetween(Buffer, FirstNewLine);
1479
1480 if (NumNewLines == 0) {
1481 SM.PrintMessage(Loc, SourceMgr::DK_Error,
1482 CheckName + ": is on the same line as previous match");
1483 SM.PrintMessage(SMLoc::getFromPointer(Buffer.end()), SourceMgr::DK_Note,
1484 "'next' match was here");
1485 SM.PrintMessage(SMLoc::getFromPointer(Buffer.data()), SourceMgr::DK_Note,
1486 "previous match ended here");
1487 return true;
1488 }
1489
1490 if (NumNewLines != 1) {
1491 SM.PrintMessage(Loc, SourceMgr::DK_Error,
1492 CheckName +
1493 ": is not on the line after the previous match");
1494 SM.PrintMessage(SMLoc::getFromPointer(Buffer.end()), SourceMgr::DK_Note,
1495 "'next' match was here");
1496 SM.PrintMessage(SMLoc::getFromPointer(Buffer.data()), SourceMgr::DK_Note,
1497 "previous match ended here");
1498 SM.PrintMessage(SMLoc::getFromPointer(FirstNewLine), SourceMgr::DK_Note,
1499 "non-matching line after previous match is here");
1500 return true;
1501 }
1502
1503 return false;
1504}

1506bool FileCheckString::CheckSame(const SourceMgr &SM, StringRef Buffer) const {
1507 if (Pat.getCheckTy() != Check::CheckSame)
1508 return false;
1509
1510 // Count the number of newlines between the previous match and this one.
1511 const char *FirstNewLine = nullptr;
1512 unsigned NumNewLines = CountNumNewlinesBetween(Buffer, FirstNewLine);
1513
1514 if (NumNewLines != 0) {
1515 SM.PrintMessage(Loc, SourceMgr::DK_Error,
1516 Prefix +
1517 "-SAME: is not on the same line as the previous match");
1518 SM.PrintMessage(SMLoc::getFromPointer(Buffer.end()), SourceMgr::DK_Note,
1519 "'next' match was here");
1520 SM.PrintMessage(SMLoc::getFromPointer(Buffer.data()), SourceMgr::DK_Note,
1521 "previous match ended here");
1522 return true;
1523 }
1524
1525 return false;

1528bool FileCheckString::CheckNot(
1529 const SourceMgr &SM, StringRef Buffer,
1530 const std::vector<const FileCheckPattern *> &NotStrings,
1531 const FileCheckRequest &Req, std::vector<FileCheckDiag> *Diags) const {
1532 for (const FileCheckPattern *Pat : NotStrings) {
1533 assert((Pat->getCheckTy() == Check::CheckNot) && "Expect CHECK-NOT!");
1534
1535 size_t MatchLen = 0;
1536 size_t Pos = Pat->match(Buffer, MatchLen, SM);
1537
1538 if (Pos == StringRef::npos) {
1539 PrintNoMatch(false, SM, Prefix, Pat->getLoc(), *Pat, 1, Buffer,
1540 Req.VerboseVerbose, Diags);
1541 continue;
1542 }
1543
1544 PrintMatch(false, SM, Prefix, Pat->getLoc(), *Pat, 1, Buffer, Pos, MatchLen,
1545 Req, Diags);
1546
1547 return true;
1548 }
1549
1550 return false;

1553size_t
1554FileCheckString::CheckDag(const SourceMgr &SM, StringRef Buffer,
1555 std::vector<const FileCheckPattern *> &NotStrings,
1556 const FileCheckRequest &Req,
1557 std::vector<FileCheckDiag> *Diags) const {
1558 if (DagNotStrings.empty())
1559 return 0;
1560
1561 // The start of the search range.
1562 size_t StartPos = 0;
1563
1564 struct MatchRange {
1565 size_t Pos;
1566 size_t End;
1567 };
1568 // A sorted list of ranges for non-overlapping CHECK-DAG matches. Match
1569 // ranges are erased from this list once they are no longer in the search
1570 // range.
1571 std::list<MatchRange> MatchRanges;
1572
1573 // We need PatItr and PatEnd later for detecting the end of a CHECK-DAG
1574 // group, so we don't use a range-based for loop here.
1575 for (auto PatItr = DagNotStrings.begin(), PatEnd = DagNotStrings.end();
1576 PatItr != PatEnd; ++PatItr) {
1577 const FileCheckPattern &Pat = *PatItr;
1578 assert((Pat.getCheckTy() == Check::CheckDAG ||
1579 Pat.getCheckTy() == Check::CheckNot) &&
1580 "Invalid CHECK-DAG or CHECK-NOT!");
1581
1582 if (Pat.getCheckTy() == Check::CheckNot) {
1583 NotStrings.push_back(&Pat);
1584 continue;
1585 }
1586
1587 assert((Pat.getCheckTy() == Check::CheckDAG) && "Expect CHECK-DAG!");
1588
1589 // CHECK-DAG always matches from the start.
1590 size_t MatchLen = 0, MatchPos = StartPos;
1591
1592 // Search for a match that doesn't overlap a previous match in this
1593 // CHECK-DAG group.
1594 for (auto MI = MatchRanges.begin(), ME = MatchRanges.end(); true; ++MI) {
1595 StringRef MatchBuffer = Buffer.substr(MatchPos);
1596 size_t MatchPosBuf = Pat.match(MatchBuffer, MatchLen, SM);
1597 // With a group of CHECK-DAGs, a single mismatching means the match on
1598 // that group of CHECK-DAGs fails immediately.
1599 if (MatchPosBuf == StringRef::npos) {
1600 PrintNoMatch(true, SM, Prefix, Pat.getLoc(), Pat, 1, MatchBuffer,
1601 Req.VerboseVerbose, Diags);
1602 return StringRef::npos;
1603 }
1604 // Re-calc it as the offset relative to the start of the original string.
1605 MatchPos += MatchPosBuf;
1606 if (Req.VerboseVerbose)
1607 PrintMatch(true, SM, Prefix, Pat.getLoc(), Pat, 1, Buffer, MatchPos,
1608 MatchLen, Req, Diags);
1609 MatchRange M{MatchPos, MatchPos + MatchLen};
1610 if (Req.AllowDeprecatedDagOverlap) {
1611 // We don't need to track all matches in this mode, so we just maintain
1612 // one match range that encompasses the current CHECK-DAG group's
1613 // matches.
1614 if (MatchRanges.empty())
1615 MatchRanges.insert(MatchRanges.end(), M);
1616 else {
1617 auto Block = MatchRanges.begin();
1618 Block->Pos = std::min(Block->Pos, M.Pos);
1619 Block->End = std::max(Block->End, M.End);
1620 }
1621 break;
1622 }
1623 // Iterate previous matches until overlapping match or insertion point.
1624 bool Overlap = false;
1625 for (; MI != ME; ++MI) {
1626 if (M.Pos < MI->End) {
1627 // !Overlap => New match has no overlap and is before this old match.
1628 // Overlap => New match overlaps this old match.
1629 Overlap = MI->Pos < M.End;
1630 break;
1631 }
1632 }
1633 if (!Overlap) {
1634 // Insert non-overlapping match into list.
1635 MatchRanges.insert(MI, M);
1636 break;
1637 }
1638 if (Req.VerboseVerbose) {
1639 // Due to their verbosity, we don't print verbose diagnostics here if
1640 // we're gathering them for a different rendering, but we always print
1641 // other diagnostics.
1642 if (!Diags) {
1643 SMLoc OldStart = SMLoc::getFromPointer(Buffer.data() + MI->Pos);
1644 SMLoc OldEnd = SMLoc::getFromPointer(Buffer.data() + MI->End);
1645 SMRange OldRange(OldStart, OldEnd);
1646 SM.PrintMessage(OldStart, SourceMgr::DK_Note,
1647 "match discarded, overlaps earlier DAG match here",
1648 {OldRange});
1649 } else
1650 Diags->rbegin()->MatchTy = FileCheckDiag::MatchFoundButDiscarded;
1651 }
1652 MatchPos = MI->End;
1653 }
1654 if (!Req.VerboseVerbose)
1655 PrintMatch(true, SM, Prefix, Pat.getLoc(), Pat, 1, Buffer, MatchPos,
1656 MatchLen, Req, Diags);
1657
1658 // Handle the end of a CHECK-DAG group.
1659 if (std::next(PatItr) == PatEnd ||
1660 std::next(PatItr)->getCheckTy() == Check::CheckNot) {
1661 if (!NotStrings.empty()) {
1662 // If there are CHECK-NOTs between two CHECK-DAGs or from CHECK to
1663 // CHECK-DAG, verify that there are no 'not' strings occurred in that
1664 // region.
1665 StringRef SkippedRegion =
1666 Buffer.slice(StartPos, MatchRanges.begin()->Pos);
1667 if (CheckNot(SM, SkippedRegion, NotStrings, Req, Diags))
1668 return StringRef::npos;
1669 // Clear "not strings".
1670 NotStrings.clear();
1671 }
1672 // All subsequent CHECK-DAGs and CHECK-NOTs should be matched from the
1673 // end of this CHECK-DAG group's match range.
1674 StartPos = MatchRanges.rbegin()->End;
1675 // Don't waste time checking for (impossible) overlaps before that.
1676 MatchRanges.clear();
1677 }
1678 }
1679
1680 return StartPos;
1681}

1726bool FileCheckPatternContext::defineCmdlineVariables(
1727 std::vector<std::string> &CmdlineDefines, SourceMgr &SM) {
1728 assert(GlobalVariableTable.empty() && GlobalNumericVariableTable.empty() &&
1729 "Overriding defined variable with command-line variable definitions");
1730
1731 if (CmdlineDefines.empty())
1732 return false;
1733
1734 // Create a string representing the vector of command-line definitions. Each
1735 // definition is on its own line and prefixed with a definition number to
1736 // clarify which definition a given diagnostic corresponds to.
1737 unsigned I = 0;
1738 bool ErrorFound = false;
1739 std::string CmdlineDefsDiag;
1740 StringRef Prefix1 = "Global define #";
1741 StringRef Prefix2 = ": ";
1742 for (StringRef CmdlineDef : CmdlineDefines)
1743 CmdlineDefsDiag +=
1744 (Prefix1 + Twine(++I) + Prefix2 + CmdlineDef + "\n").str();
1745
1746 // Create a buffer with fake command line content in order to display
1747 // parsing diagnostic with location information and point to the
1748 // global definition with invalid syntax.
1749 std::unique_ptr<MemoryBuffer> CmdLineDefsDiagBuffer =
1750 MemoryBuffer::getMemBufferCopy(CmdlineDefsDiag, "Global defines");
1751 StringRef CmdlineDefsDiagRef = CmdLineDefsDiagBuffer->getBuffer();
1752 SM.AddNewSourceBuffer(std::move(CmdLineDefsDiagBuffer), SMLoc());
1753
1754 SmallVector<StringRef, 4> CmdlineDefsDiagVec;
1755 CmdlineDefsDiagRef.split(CmdlineDefsDiagVec, '\n', -1 /*MaxSplit*/,
1756 false /*KeepEmpty*/);
1757 for (StringRef CmdlineDefDiag : CmdlineDefsDiagVec) {
1758 unsigned DefStart = CmdlineDefDiag.find(Prefix2) + Prefix2.size();
1759 StringRef CmdlineDef = CmdlineDefDiag.substr(DefStart);
1760 size_t EqIdx = CmdlineDef.find('=');
1761 if (EqIdx == StringRef::npos) {
1762 SM.PrintMessage(SMLoc::getFromPointer(CmdlineDef.data()),
1763 SourceMgr::DK_Error,
1764 "Missing equal sign in global definition");
1765 ErrorFound = true;
1766 continue;
1767 }
1768
1769 // Numeric variable definition.
1770 if (CmdlineDef[0] == '#') {
1771 StringRef CmdlineName = CmdlineDef.substr(1, EqIdx - 1);
1772 StringRef VarName;
1773 SMLoc CmdlineNameLoc = SMLoc::getFromPointer(CmdlineName.data());
1774 bool ParseError = FileCheckPattern::parseNumericVariableDefinition(
1775 CmdlineName, VarName, this, SM);
1776 // Check that CmdlineName starts with a valid numeric variable to be
1777 // defined and that it is not followed that anything. That second check
1778 // detects cases like "FOO+2" in a "FOO+2=10" definition.
1779 if (ParseError || !CmdlineName.empty()) {
1780 if (!ParseError)
1781 SM.PrintMessage(CmdlineNameLoc, SourceMgr::DK_Error,
1782 "invalid variable name");
1783 ErrorFound = true;
1784 continue;
1785 }
1786
1787 // Detect collisions between string and numeric variables when the latter
1788 // is created later than the former.
1789 if (DefinedVariableTable.find(VarName) != DefinedVariableTable.end()) {
1790 SM.PrintMessage(
1791 SMLoc::getFromPointer(VarName.data()), SourceMgr::DK_Error,
1792 "string variable with name '" + VarName + "' already exists");
1793 ErrorFound = true;
1794 continue;
1795 }
1796
1797 StringRef CmdlineVal = CmdlineDef.substr(EqIdx + 1);
1798 uint64_t Val;
1799 if (CmdlineVal.getAsInteger(10, Val)) {
1800 SM.PrintMessage(SMLoc::getFromPointer(CmdlineVal.data()),
1801 SourceMgr::DK_Error,
1802 "invalid value in numeric variable definition '" +
1803 CmdlineVal + "'");
1804 ErrorFound = true;
1805 continue;
1806 }
1807 auto DefinedNumericVariable = makeNumericVariable(0, VarName);
1808 DefinedNumericVariable->setValue(Val);
1809
1810 // Record this variable definition.
1811 GlobalNumericVariableTable[DefinedNumericVariable->getName()] =
1812 DefinedNumericVariable;
1813 } else {
1814 // String variable definition.
1815 std::pair<StringRef, StringRef> CmdlineNameVal = CmdlineDef.split('=');
1816 StringRef CmdlineName = CmdlineNameVal.first;
1817 StringRef OrigCmdlineName = CmdlineName;
1818 StringRef Name;
1819 bool IsPseudo;
1820 if (FileCheckPattern::parseVariable(CmdlineName, Name, IsPseudo) ||
1821 IsPseudo || !CmdlineName.empty()) {
1822 SM.PrintMessage(SMLoc::getFromPointer(OrigCmdlineName.data()),
1823 SourceMgr::DK_Error,
1824 "invalid name in string variable definition '" +
1825 OrigCmdlineName + "'");
1826 ErrorFound = true;
1827 continue;
1828 }
1829
1830 // Detect collisions between string and numeric variables when the former
1831 // is created later than the latter.
1832 if (GlobalNumericVariableTable.find(Name) !=
1833 GlobalNumericVariableTable.end()) {
1834 SM.PrintMessage(SMLoc::getFromPointer(Name.data()), SourceMgr::DK_Error,
1835 "numeric variable with name '" + Name +
1836 "' already exists");
1837 ErrorFound = true;
1838 continue;
1839 }
1840 GlobalVariableTable.insert(CmdlineNameVal);
1841 // Mark the string variable as defined to detect collisions between
1842 // string and numeric variables in DefineCmdlineVariables when the latter
1843 // is created later than the former. We cannot reuse GlobalVariableTable
1844 // for this by populating it with an empty string since we would then
1845 // lose the ability to detect the use of an undefined variable in
1846 // match().
1847 DefinedVariableTable[Name] = true;
1848 }
1849 }
1850
1851 return ErrorFound;
1852}

1854void FileCheckPatternContext::clearLocalVars() {
1855 SmallVector<StringRef, 16> LocalPatternVars, LocalNumericVars;
1856 for (const StringMapEntry<StringRef> &Var : GlobalVariableTable)
1857 if (Var.first()[0] != '$')
1858 LocalPatternVars.push_back(Var.first());
1859
1860 // Numeric substitution reads the value of a variable directly, not via
1861 // GlobalNumericVariableTable. Therefore, we clear local variables by
1862 // clearing their value which will lead to a numeric substitution failure. We
1863 // also mark the variable for removal from GlobalNumericVariableTable since
1864 // this is what defineCmdlineVariables checks to decide that no global
1865 // variable has been defined.
1866 for (const auto &Var : GlobalNumericVariableTable)
1867 if (Var.first()[0] != '$') {
1868 Var.getValue()->clearValue();
1869 LocalNumericVars.push_back(Var.first());
1870 }
1871
1872 for (const auto &Var : LocalPatternVars)
1873 GlobalVariableTable.erase(Var);
1874 for (const auto &Var : LocalNumericVars)
1875 GlobalNumericVariableTable.erase(Var);
1876}

1878bool FileCheck::CheckInput(SourceMgr &SM, StringRef Buffer,
1879 ArrayRef<FileCheckString> CheckStrings,
1880 std::vector<FileCheckDiag> *Diags) {
1881 bool ChecksFailed = false;
1882
1883 unsigned i = 0, j = 0, e = CheckStrings.size();
1884 while (true) {
1885 StringRef CheckRegion;
1886 if (j == e) {
1887 CheckRegion = Buffer;
1888 } else {
1889 const FileCheckString &CheckLabelStr = CheckStrings[j];
1890 if (CheckLabelStr.Pat.getCheckTy() != Check::CheckLabel) {
1891 ++j;
1892 continue;
1893 }
1894
1895 // Scan to next CHECK-LABEL match, ignoring CHECK-NOT and CHECK-DAG
1896 size_t MatchLabelLen = 0;
1897 size_t MatchLabelPos =
1898 CheckLabelStr.Check(SM, Buffer, true, MatchLabelLen, Req, Diags);
1899 if (MatchLabelPos == StringRef::npos)
1900 // Immediately bail if CHECK-LABEL fails, nothing else we can do.
1901 return false;
1902
1903 CheckRegion = Buffer.substr(0, MatchLabelPos + MatchLabelLen);
1904 Buffer = Buffer.substr(MatchLabelPos + MatchLabelLen);
1905 ++j;
1906 }
1907
1908 // Do not clear the first region as it's the one before the first
1909 // CHECK-LABEL and it would clear variables defined on the command-line
1910 // before they get used.
1911 if (i != 0 && Req.EnableVarScope)
1912 PatternContext.clearLocalVars();
1913
1914 for (; i != j; ++i) {
1915 const FileCheckString &CheckStr = CheckStrings[i];
1916
1917 // Check each string within the scanned region, including a second check
1918 // of any final CHECK-LABEL (to verify CHECK-NOT and CHECK-DAG)
1919 size_t MatchLen = 0;
1920 size_t MatchPos =
1921 CheckStr.Check(SM, CheckRegion, false, MatchLen, Req, Diags);
1922
1923 if (MatchPos == StringRef::npos) {
1924 ChecksFailed = true;
1925 i = j;
1926 break;
1927 }
1928
1929 CheckRegion = CheckRegion.substr(MatchPos + MatchLen);
1930 }
1931
1932 if (j == e)
1933 break;
1934 }
1935
1936 // Success if no checks failed.
1937 return !ChecksFailed;
1938}
