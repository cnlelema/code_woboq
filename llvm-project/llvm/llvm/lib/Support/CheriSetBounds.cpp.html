23static llvm::cl::opt<llvm::cheri::StatsFormat, true> CollectSetBoundsStats(
24 "collect-csetbounds-stats", llvm::cl::ZeroOrMore, llvm::cl::Hidden,
25 llvm::cl::desc("Collect statistics on CSetBounds uses"),
26 llvm::cl::location(llvm::cheri::ShouldCollectCSetBoundsStats),
27 llvm::cl::values(
28 clEnumValN(llvm::cheri::StatsOff, "off", "Do not collect statistics"),

31 "Print stats in JSON format")));

44void CSetBoundsStatistics::add(unsigned KnownAlignment,
45 Optional<uint64_t> KnownSize, StringRef Pass,
46 SetBoundsPointerSource Kind,
47 const Twine &Details, std::string SourceLoc,
48 Optional<uint64_t> SizeMultipleOf) {
49 Entries.push_back({KnownAlignment, KnownSize, SizeMultipleOf, Kind,
50 std::move(SourceLoc), Pass, Details.str()});

59void CSetBoundsStatistics::print(llvm::raw_ostream &OS, StringRef MainFile,
60 bool PrintHeader) {
61 if (ShouldCollectCSetBoundsStats == StatsJSON) {
62 OS << "{ \"csetbounds_stats\": {";
63 OS << "\n\t\"count\": " << Entries.size() << ',';
64 OS << "\n\t\"details\": [";
65 bool Comma = false;
66 for (const Entry &E : Entries) {
67 if (!Comma) {
68 Comma = true;
69 OS << "\n\t{";
70 } else {
71 OS << ",\n\t{";
72 }
73 OS << "\n\t\t\"alignment\": " << E.KnownAlignment;
74 if (E.RequestedSize)
75 OS << ",\n\t\t\"size\": " << *E.RequestedSize;
76 else
77 OS << ",\n\t\t\"size\": null,";
78 OS << ",\n\t\t\"location\": \"" << llvm::yaml::escape(E.SourceLocation)
79 << '"';
80 if (!E.Pass.empty())
81 OS << ",\n\t\t\"pass\": \"" << llvm::yaml::escape(E.Pass) << '"';
82 if (!E.Details.empty())
83 OS << ",\n\t\t\"details\": \"" << llvm::yaml::escape(E.Details) << '"';
84 OS << "\n\t}";
85 }
86 OS << "\n\t]\n} }\n";
87 } else if (ShouldCollectCSetBoundsStats == StatsCSV) {
88 if (PrintHeader) {
89 OS << "alignment_bits,size,kind,source_loc,compiler_pass,details\n";
90 }
91 for (const Entry &E : Entries) {
92 // The analysis scripts expect alignment as a power of two instead of the
93 // value
94 unsigned AlignmentBits = 0;
95 // Log2_64 will return -1 for 0 but we want 0 in that case
96 if (E.KnownAlignment != 0)
97 AlignmentBits = Log2_64(E.KnownAlignment);
98 OS << AlignmentBits << ',';
99 if (E.RequestedSize) {
100 OS << *E.RequestedSize;
101 } else {
102 // IF the size is unknown
103 if (E.RequestedSizeMultipleOf)
104 OS << "<unknown multiple of " << *E.RequestedSizeMultipleOf << ">";
105 else
106 OS << "<unknown>";
107 }
108
109 if (E.PointerKind == SetBoundsPointerSource::Stack) {
110 OS << ",s";
111 } else if (E.PointerKind == SetBoundsPointerSource::Heap) {
112 OS << ",h";
113 } else if (E.PointerKind == SetBoundsPointerSource::SubObject) {
114 OS << ",o";
115 } else if (E.PointerKind == SetBoundsPointerSource::GlobalVar) {
116 OS << ",g";
117 } else if (E.PointerKind == SetBoundsPointerSource::CodePointer) {
118 OS << ",c";
119 } else {
120 OS << ",?";
121 }
122 OS << ",\"" << llvm::yaml::escape(E.SourceLocation) << '"';
123 OS << ",\"" << llvm::yaml::escape(E.Pass) << '"';
124 OS << ",\"" << llvm::yaml::escape(E.Details) << '"';
125 OS << "\n";
126 }
127 }

138uint64_t StatsOutputFile::size() {
139 sys::fs::file_status s;
140 if (sys::fs::status(FD, s)) {
141 return 0;
142 }
143 return s.getSize();
144}
145
146StatsOutputFile::~StatsOutputFile() {
147 if (flock(FD, LOCK_UN) != 0 && errno != ENOTSUP) {
148 errs() << "WARNING: unlocking statistics FD failed. Should be "
149 "released automatically on exit anyway.\n";
150 }
151}

153std::unique_ptr<StatsOutputFile>
154StatsOutputFile::open(StringRef File, ErrorCallback OnOpenError,
155 ErrorCallback OnLockError) {
156 // raw_fd_ostream has no way of getting the FD and I don't want to change
157 // the interface just for this function to enable file locking since
158 // that would require a recompile of all of LLVM....
159 int StatsFD = STDERR_FILENO;
160 bool CloseOnStreamDelete = true;
161 if (!File.empty()) {
162 std::error_code EC = sys::fs::openFileForWrite(
163 File, StatsFD, sys::fs::CD_CreateAlways, sys::fs::OF_Append);
164 if (EC) {
165 OnOpenError(File, EC);
166 return nullptr;
167 }
168 } else {
169 File = "/dev/stderr";
170 CloseOnStreamDelete = false;
171 }
172 assert(StatsFD != -1);
173 if (flock(StatsFD, LOCK_EX) != 0 && errno != ENOTSUP) {
174 // Lock error is not fatal we just get mixed output in the stats file
175 OnLockError(File, std::error_code(errno, std::generic_category()));
176 }
177 // can't use make_unique here since the ctor is private
178 return std::unique_ptr<StatsOutputFile>(new StatsOutputFile(
179 StatsFD,
180 llvm::make_unique<llvm::raw_fd_ostream>(StatsFD, CloseOnStreamDelete)));
181}
