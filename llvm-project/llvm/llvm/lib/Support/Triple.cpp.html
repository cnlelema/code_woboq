250static Triple::ArchType parseBPFArch(StringRef ArchName) {
251 if (ArchName.equals("bpf")) {
252 if (sys::IsLittleEndianHost)
253 return Triple::bpfel;
254 else
255 return Triple::bpfeb;
256 } else if (ArchName.equals("bpf_be") || ArchName.equals("bpfeb")) {
257 return Triple::bpfeb;
258 } else if (ArchName.equals("bpf_le") || ArchName.equals("bpfel")) {
259 return Triple::bpfel;
260 } else {
261 return Triple::UnknownArch;
262 }
263}

324static Triple::ArchType parseARMArch(StringRef ArchName) {
325 ARM::ISAKind ISA = ARM::parseArchISA(ArchName);
326 ARM::EndianKind ENDIAN = ARM::parseArchEndian(ArchName);
327
328 Triple::ArchType arch = Triple::UnknownArch;
329 switch (ENDIAN) {
330 case ARM::EndianKind::LITTLE: {
331 switch (ISA) {
332 case ARM::ISAKind::ARM:
333 arch = Triple::arm;
334 break;
335 case ARM::ISAKind::THUMB:
336 arch = Triple::thumb;
337 break;
338 case ARM::ISAKind::AARCH64:
339 arch = Triple::aarch64;
340 break;
341 case ARM::ISAKind::INVALID:
342 break;
343 }
344 break;
345 }
346 case ARM::EndianKind::BIG: {
347 switch (ISA) {
348 case ARM::ISAKind::ARM:
349 arch = Triple::armeb;
350 break;
351 case ARM::ISAKind::THUMB:
352 arch = Triple::thumbeb;
353 break;
354 case ARM::ISAKind::AARCH64:
355 arch = Triple::aarch64_be;
356 break;
357 case ARM::ISAKind::INVALID:
358 break;
359 }
360 break;
361 }
362 case ARM::EndianKind::INVALID: {
363 break;
364 }
365 }
366
367 ArchName = ARM::getCanonicalArchName(ArchName);
368 if (ArchName.empty())
369 return Triple::UnknownArch;
370
371 // Thumb only exists in v4+
372 if (ISA == ARM::ISAKind::THUMB &&
373 (ArchName.startswith("v2") || ArchName.startswith("v3")))
374 return Triple::UnknownArch;
375
376 // Thumb only for v6m
377 ARM::ProfileKind Profile = ARM::parseArchProfile(ArchName);
378 unsigned Version = ARM::parseArchVersion(ArchName);
379 if (Profile == ARM::ProfileKind::M && Version == 6) {
380 if (ENDIAN == ARM::EndianKind::BIG)
381 return Triple::thumbeb;
382 else
383 return Triple::thumb;
384 }
385
386 return arch;
387}

389static Triple::ArchType parseArch(StringRef ArchName) {
390 auto AT = StringSwitch<Triple::ArchType>(ArchName)
391 .Cases("i386", "i486", "i586", "i686", Triple::x86)
392 // FIXME: Do we need to support these?
393 .Cases("i786", "i886", "i986", Triple::x86)
394 .Cases("amd64", "x86_64", "x86_64h", Triple::x86_64)
395 .Cases("powerpc", "ppc", "ppc32", Triple::ppc)
396 .Cases("powerpc64", "ppu", "ppc64", Triple::ppc64)
397 .Cases("powerpc64le", "ppc64le", Triple::ppc64le)
398 .Case("xscale", Triple::arm)
399 .Case("xscaleeb", Triple::armeb)
400 .Case("aarch64", Triple::aarch64)
401 .Case("aarch64_be", Triple::aarch64_be)
402 .Case("aarch64_32", Triple::aarch64_32)
403 .Case("arc", Triple::arc)
404 .Case("arm64", Triple::aarch64)
405 .Case("arm64_32", Triple::aarch64_32)
406 .Case("arm", Triple::arm)
407 .Case("armeb", Triple::armeb)
408 .Case("thumb", Triple::thumb)
409 .Case("thumbeb", Triple::thumbeb)
410 .Case("avr", Triple::avr)
411 .Case("msp430", Triple::msp430)
412 .Cases("mips", "mipseb", "mipsallegrex", "mipsisa32r6",
413 "mipsr6", Triple::mips)
414 .Cases("mipsel", "mipsallegrexel", "mipsisa32r6el", "mipsr6el",
415 Triple::mipsel)
416 .Cases("mips64", "mips64eb", "mipsn32", "mipsisa64r6",
417 "mips64r6", "mipsn32r6", Triple::mips64)
418 .StartsWith("mips64c", Triple::mips64) // purecap/hybrid CHERI
419 .Case("cheri", Triple::cheri) // TODO: remove
420 .Cases("mips64el", "mipsn32el", "mipsisa64r6el", "mips64r6el",
421 "mipsn32r6el", Triple::mips64el)
422 .Case("r600", Triple::r600)
423 .Case("amdgcn", Triple::amdgcn)
424 .Case("riscv32", Triple::riscv32)
425 .Case("riscv64", Triple::riscv64)
426 .Case("hexagon", Triple::hexagon)
427 .Cases("s390x", "systemz", Triple::systemz)
428 .Case("sparc", Triple::sparc)
429 .Case("sparcel", Triple::sparcel)
430 .Cases("sparcv9", "sparc64", Triple::sparcv9)
431 .Case("tce", Triple::tce)
432 .Case("tcele", Triple::tcele)
433 .Case("xcore", Triple::xcore)
434 .Case("nvptx", Triple::nvptx)
435 .Case("nvptx64", Triple::nvptx64)
436 .Case("le32", Triple::le32)
437 .Case("le64", Triple::le64)
438 .Case("amdil", Triple::amdil)
439 .Case("amdil64", Triple::amdil64)
440 .Case("hsail", Triple::hsail)
441 .Case("hsail64", Triple::hsail64)
442 .Case("spir", Triple::spir)
443 .Case("spir64", Triple::spir64)
444 .StartsWith("kalimba", Triple::kalimba)
445 .Case("lanai", Triple::lanai)
446 .Case("shave", Triple::shave)
447 .Case("wasm32", Triple::wasm32)
448 .Case("wasm64", Triple::wasm64)
449 .Case("renderscript32", Triple::renderscript32)
450 .Case("renderscript64", Triple::renderscript64)
451 .Default(Triple::UnknownArch);
452
453 // Some architectures require special parsing logic just to compute the
454 // ArchType result.
455 if (AT == Triple::UnknownArch) {
456 if (ArchName.startswith("arm") || ArchName.startswith("thumb") ||
457 ArchName.startswith("aarch64"))
458 return parseARMArch(ArchName);
459 if (ArchName.startswith("bpf"))
460 return parseBPFArch(ArchName);
461 }
462
463 return AT;
464}

567static Triple::SubArchType parseSubArch(StringRef SubArchName) {
568 if (SubArchName.startswith("mips")) {
569 if (SubArchName.endswith("r6el") || SubArchName.endswith("r6")) {
570 return Triple::MipsSubArch_r6;
571 }
572 // Support encoding the CHERI abi and size in the triple name
573 return StringSwitch<Triple::SubArchType>(SubArchName)
574 .EndsWith("c128", Triple::MipsSubArch_cheri128)
575 .EndsWith("c128hybrid", Triple::MipsSubArch_cheri128)
576 .EndsWith("c256", Triple::MipsSubArch_cheri256)
577 .EndsWith("c256hybrid", Triple::MipsSubArch_cheri256)
578 .EndsWith("c64", Triple::MipsSubArch_cheri64)
579 .EndsWith("c64hybrid", Triple::MipsSubArch_cheri64)
580 .Default(Triple::NoSubArch);
581 }
582#ifdef NOTYET
583 if (SubArchName.startswith("cheri")) {
584 // Support encoding the CHERI abi and size in the triple name
585 return StringSwitch<Triple::SubArchType>(SubArchName)
586 .StartsWith("cheri128", Triple::MipsSubArch_cheri128)
587 .StartsWith("cheri256", Triple::MipsSubArch_cheri256)
588 .StartsWith("cheri64", Triple::MipsSubArch_cheri64)
589 .Default(Triple::MipsSubArch_cheri128);
590 }
591#endif
592
593 StringRef ARMSubArch = ARM::getCanonicalArchName(SubArchName);
594
595 // For now, this is the small part. Early return.
596 if (ARMSubArch.empty())
597 return StringSwitch<Triple::SubArchType>(SubArchName)
598 .EndsWith("kalimba3", Triple::KalimbaSubArch_v3)
599 .EndsWith("kalimba4", Triple::KalimbaSubArch_v4)
600 .EndsWith("kalimba5", Triple::KalimbaSubArch_v5)
601 .Default(Triple::NoSubArch);
602
603 // ARM sub arch.
604 switch(ARM::parseArch(ARMSubArch)) {
605 case ARM::ArchKind::ARMV4:
606 return Triple::NoSubArch;
607 case ARM::ArchKind::ARMV4T:
608 return Triple::ARMSubArch_v4t;
609 case ARM::ArchKind::ARMV5T:
610 return Triple::ARMSubArch_v5;
611 case ARM::ArchKind::ARMV5TE:
612 case ARM::ArchKind::IWMMXT:
613 case ARM::ArchKind::IWMMXT2:
614 case ARM::ArchKind::XSCALE:
615 case ARM::ArchKind::ARMV5TEJ:
616 return Triple::ARMSubArch_v5te;
617 case ARM::ArchKind::ARMV6:
618 return Triple::ARMSubArch_v6;
619 case ARM::ArchKind::ARMV6K:
620 case ARM::ArchKind::ARMV6KZ:
621 return Triple::ARMSubArch_v6k;
622 case ARM::ArchKind::ARMV6T2:
623 return Triple::ARMSubArch_v6t2;
624 case ARM::ArchKind::ARMV6M:
625 return Triple::ARMSubArch_v6m;
626 case ARM::ArchKind::ARMV7A:
627 case ARM::ArchKind::ARMV7R:
628 return Triple::ARMSubArch_v7;
629 case ARM::ArchKind::ARMV7VE:
630 return Triple::ARMSubArch_v7ve;
631 case ARM::ArchKind::ARMV7K:

830std::string Triple::normalize(StringRef Str) {
831 bool IsMinGW32 = false;
832 bool IsCygwin = false;
833
834 // Parse into components.
835 SmallVector<StringRef, 4> Components;
836 Str.split(Components, '-');
837
838 // If the first component corresponds to a known architecture, preferentially
839 // use it for the architecture. If the second component corresponds to a
840 // known vendor, preferentially use it for the vendor, etc. This avoids silly
841 // component movement when a component parses as (eg) both a valid arch and a
842 // valid os.
843 ArchType Arch = UnknownArch;
844 if (Components.size() > 0)
845 Arch = parseArch(Components[0]);
846 VendorType Vendor = UnknownVendor;
847 if (Components.size() > 1)
848 Vendor = parseVendor(Components[1]);
849 OSType OS = UnknownOS;
850 if (Components.size() > 2) {
851 OS = parseOS(Components[2]);
852 IsCygwin = Components[2].startswith("cygwin");
853 IsMinGW32 = Components[2].startswith("mingw");
854 }
855 EnvironmentType Environment = UnknownEnvironment;
856 if (Components.size() > 3)
857 Environment = parseEnvironment(Components[3]);
858 ObjectFormatType ObjectFormat = UnknownObjectFormat;
859 if (Components.size() > 4)
860 ObjectFormat = parseFormat(Components[4]);
861
862 // Note which components are already in their final position. These will not
863 // be moved.
864 bool Found[4];
865 Found[0] = Arch != UnknownArch;
866 Found[1] = Vendor != UnknownVendor;
867 Found[2] = OS != UnknownOS;
868 Found[3] = Environment != UnknownEnvironment;
869
870 // If they are not there already, permute the components into their canonical
871 // positions by seeing if they parse as a valid architecture, and if so moving
872 // the component to the architecture position etc.
873 for (unsigned Pos = 0; Pos != array_lengthof(Found); ++Pos) {
874 if (Found[Pos])
875 continue; // Already in the canonical position.
876
877 for (unsigned Idx = 0; Idx != Components.size(); ++Idx) {
878 // Do not reparse any components that already matched.
879 if (Idx < array_lengthof(Found) && Found[Idx])
880 continue;
881
882 // Does this component parse as valid for the target position?
883 bool Valid = false;
884 StringRef Comp = Components[Idx];
885 switch (Pos) {
886 default: llvm_unreachable("unexpected component type!");
887 case 0:
888 Arch = parseArch(Comp);
889 Valid = Arch != UnknownArch;
890 break;
891 case 1:
892 Vendor = parseVendor(Comp);
893 Valid = Vendor != UnknownVendor;
894 break;
895 case 2:
896 OS = parseOS(Comp);
897 IsCygwin = Comp.startswith("cygwin");
898 IsMinGW32 = Comp.startswith("mingw");
899 Valid = OS != UnknownOS || IsCygwin || IsMinGW32;
900 break;
901 case 3:
902 Environment = parseEnvironment(Comp);
903 Valid = Environment != UnknownEnvironment;
904 if (!Valid) {
905 ObjectFormat = parseFormat(Comp);
906 Valid = ObjectFormat != UnknownObjectFormat;
907 }
908 break;
909 }
910 if (!Valid)
911 continue; // Nope, try the next component.
912
913 // Move the component to the target position, pushing any non-fixed
914 // components that are in the way to the right. This tends to give
915 // good results in the common cases of a forgotten vendor component
916 // or a wrongly positioned environment.
917 if (Pos < Idx) {
918 // Insert left, pushing the existing components to the right. For
919 // example, a-b-i386 -> i386-a-b when moving i386 to the front.
920 StringRef CurrentComponent(""); // The empty component.
921 // Replace the component we are moving with an empty component.
922 std::swap(CurrentComponent, Components[Idx]);
923 // Insert the component being moved at Pos, displacing any existing
924 // components to the right.
925 for (unsigned i = Pos; !CurrentComponent.empty(); ++i) {
926 // Skip over any fixed components.
927 while (i < array_lengthof(Found) && Found[i])
928 ++i;
929 // Place the component at the new position, getting the component
930 // that was at this position - it will be moved right.
931 std::swap(CurrentComponent, Components[i]);
932 }
933 } else if (Pos > Idx) {
934 // Push right by inserting empty components until the component at Idx
935 // reaches the target position Pos. For example, pc-a -> -pc-a when
936 // moving pc to the second position.
937 do {
938 // Insert one empty component at Idx.
939 StringRef CurrentComponent(""); // The empty component.
940 for (unsigned i = Idx; i < Components.size();) {
941 // Place the component at the new position, getting the component
942 // that was at this position - it will be moved right.
943 std::swap(CurrentComponent, Components[i]);
944 // If it was placed on top of an empty component then we are done.
945 if (CurrentComponent.empty())
946 break;
947 // Advance to the next component, skipping any fixed components.
948 while (++i < array_lengthof(Found) && Found[i])
949 ;
950 }
951 // The last component was pushed off the end - append it.
952 if (!CurrentComponent.empty())
953 Components.push_back(CurrentComponent);
954
955 // Advance Idx to the component's new position.
956 while (++Idx < array_lengthof(Found) && Found[Idx])
957 ;
958 } while (Idx < Pos); // Add more until the final position is reached.
959 }
960 assert(Pos < Components.size() && Components[Pos] == Comp &&
961 "Component moved wrong!");
962 Found[Pos] = true;
963 break;
964 }
965 }
966
967 // Replace empty components with "unknown" value.
968 for (unsigned i = 0, e = Components.size(); i < e; ++i) {
969 if (Components[i].empty())
970 Components[i] = "unknown";
971 }
972
973 // Special case logic goes here. At this point Arch, Vendor and OS have the
974 // correct values for the computed components.
975 std::string NormalizedEnvironment;
976 if (Environment == Triple::Android && Components[3].startswith("androideabi")) {
977 StringRef AndroidVersion = Components[3].drop_front(strlen("androideabi"));
978 if (AndroidVersion.empty()) {
979 Components[3] = "android";
980 } else {
981 NormalizedEnvironment = Twine("android", AndroidVersion).str();
982 Components[3] = NormalizedEnvironment;
983 }
984 }
985
986 // SUSE uses "gnueabi" to mean "gnueabihf"
987 if (Vendor == Triple::SUSE && Environment == llvm::Triple::GNUEABI)
988 Components[3] = "gnueabihf";
989
990 if (OS == Triple::Win32) {
991 Components.resize(4);
992 Components[2] = "windows";
993 if (Environment == UnknownEnvironment) {
994 if (ObjectFormat == UnknownObjectFormat || ObjectFormat == Triple::COFF)
995 Components[3] = "msvc";
996 else
997 Components[3] = getObjectFormatTypeName(ObjectFormat);
998 }
999 } else if (IsMinGW32) {
1000 Components.resize(4);
1001 Components[2] = "windows";
1002 Components[3] = "gnu";
1003 } else if (IsCygwin) {
1004 Components.resize(4);
1005 Components[2] = "windows";
1006 Components[3] = "cygnus";
1007 }
1008 if (IsMinGW32 || IsCygwin ||
1009 (OS == Triple::Win32 && Environment != UnknownEnvironment)) {
1010 if (ObjectFormat != UnknownObjectFormat && ObjectFormat != Triple::COFF) {
1011 Components.resize(5);
1012 Components[4] = getObjectFormatTypeName(ObjectFormat);
1013 }
1014 }
1015
1016 // Stick the corrected components back together to form the normalized string.
1017 std::string Normalized;
1018 for (unsigned i = 0, e = Components.size(); i != e; ++i) {
1019 if (i) Normalized += '-';
1020 Normalized += Components[i];
1021 }
1022 return Normalized;
1023}

1051static unsigned EatNumber(StringRef &Str) {
1052 assert(!Str.empty() && Str[0] >= '0' && Str[0] <= '9' && "Not a number");
1053 unsigned Result = 0;
1054
1055 do {
1056 // Consume the leading digit.
1057 Result = Result*10 + (Str[0] - '0');
1058
1059 // Eat the digit.
1060 Str = Str.substr(1);
1061 } while (!Str.empty() && Str[0] >= '0' && Str[0] <= '9');
1062
1063 return Result;
1064}

1066static void parseVersionFromName(StringRef Name, unsigned &Major,
1067 unsigned &Minor, unsigned &Micro) {
1068 // Any unset version defaults to 0.
1069 Major = Minor = Micro = 0;
1070
1071 // Parse up to three components.
1072 unsigned *Components[3] = {&Major, &Minor, &Micro};
1073 for (unsigned i = 0; i != 3; ++i) {
1074 if (Name.empty() || Name[0] < '0' || Name[0] > '9')
1075 break;
1076
1077 // Consume the leading number.
1078 *Components[i] = EatNumber(Name);
1079
1080 // Consume the separator, if present.
1081 if (Name.startswith("."))
1082 Name = Name.substr(1);
1083 }
1084}

1109bool Triple::getMacOSXVersion(unsigned &Major, unsigned &Minor,
1110 unsigned &Micro) const {
1111 getOSVersion(Major, Minor, Micro);
1112
1113 switch (getOS()) {
1114 default: llvm_unreachable("unexpected OS for Darwin triple");
1115 case Darwin:
1116 // Default to darwin8, i.e., MacOSX 10.4.
1117 if (Major == 0)
1118 Major = 8;
1119 // Darwin version numbers are skewed from OS X versions.
1120 if (Major < 4)
1121 return false;
1122 Micro = 0;
1123 Minor = Major - 4;
1124 Major = 10;
1125 break;
1126 case MacOSX:
1127 // Default to 10.4.
1128 if (Major == 0) {
1129 Major = 10;
1130 Minor = 4;
1131 }

1133 return false;
1134 break;
1135 case IOS:
1136 case TvOS:
1137 case WatchOS:
1138 // Ignore the version from the triple. This is only handled because the
1139 // the clang driver combines OS X and IOS support into a common Darwin
1140 // toolchain that wants to know the OS X version number even when targeting
1141 // IOS.
1142 Major = 10;
1143 Minor = 4;
1144 Micro = 0;

1602bool Triple::isCompatibleWith(const Triple &Other) const {
1603 // ARM and Thumb triples are compatible, if subarch, vendor and OS match.
1604 if ((getArch() == Triple::thumb && Other.getArch() == Triple::arm) ||
1605 (getArch() == Triple::arm && Other.getArch() == Triple::thumb) ||
1606 (getArch() == Triple::thumbeb && Other.getArch() == Triple::armeb) ||
1607 (getArch() == Triple::armeb && Other.getArch() == Triple::thumbeb)) {
1608 if (getVendor() == Triple::Apple)
1609 return getSubArch() == Other.getSubArch() &&
1610 getVendor() == Other.getVendor() && getOS() == Other.getOS();
1611 else
1612 return getSubArch() == Other.getSubArch() &&
1613 getVendor() == Other.getVendor() && getOS() == Other.getOS() &&
1614 getEnvironment() == Other.getEnvironment() &&
1615 getObjectFormat() == Other.getObjectFormat();
1616 }
1617
1618 // If vendor is apple, ignore the version number.
1619 if (getVendor() == Triple::Apple)
1620 return getArch() == Other.getArch() && getSubArch() == Other.getSubArch() &&
1621 getVendor() == Other.getVendor() && getOS() == Other.getOS();
1622
1623 return *this == Other;
1624}

1635StringRef Triple::getARMCPUForArch(StringRef MArch) const {
1636 if (MArch.empty())
1637 MArch = getArchName();
1638 MArch = ARM::getCanonicalArchName(MArch);
1639
1640 // Some defaults are forced.
1641 switch (getOS()) {
1642 case llvm::Triple::FreeBSD:
1643 case llvm::Triple::NetBSD:
1644 if (!MArch.empty() && MArch == "v6")
1645 return "arm1176jzf-s";
1646 break;
1647 case llvm::Triple::Win32:
1648 // FIXME: this is invalid for WindowsCE
1649 return "cortex-a9";

1651 case llvm::Triple::IOS:
1652 case llvm::Triple::WatchOS:
1653 case llvm::Triple::TvOS:
1654 if (MArch == "v7k")
1655 return "cortex-a7";
1656 break;
1657 default:
1658 break;
1659 }

1661 if (MArch.empty())
1662 return StringRef();

1664 StringRef CPU = ARM::getDefaultCPU(MArch);
1665 if (!CPU.empty() && !CPU.equals("invalid"))
1666 return CPU;

1668 // If no specific architecture version is requested, return the minimum CPU
1669 // required by the OS and environment.
1670 switch (getOS()) {
1671 case llvm::Triple::NetBSD:

1675 case llvm::Triple::EABIHF:

1680 }

1686 case llvm::Triple::EABIHF:
