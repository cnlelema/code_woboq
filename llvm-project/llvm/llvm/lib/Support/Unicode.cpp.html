215/// Gets the number of positions a character is likely to occupy when output
216/// on a terminal ("character width"). This depends on the implementation of the
217/// terminal, and there's no standard definition of character width.
218/// The implementation defines it in a way that is expected to be compatible
219/// with a generic Unicode-capable terminal.
220/// \return Character width:
221/// * ErrorNonPrintableCharacter (-1) for non-printable characters (as
222/// identified by isPrintable);
223/// * 0 for non-spacing and enclosing combining marks;
224/// * 2 for CJK characters excluding halfwidth forms;
225/// * 1 for all remaining characters.
226static inline int charWidth(int UCS)
227{
228 if (!isPrintable(UCS))
229 return ErrorNonPrintableCharacter;
230
231 // Sorted list of non-spacing and enclosing combining mark intervals as
232 // defined in "3.6 Combination" of

342int columnWidthUTF8(StringRef Text) {
343 unsigned ColumnWidth = 0;
344 unsigned Length;
345 for (size_t i = 0, e = Text.size(); i < e; i += Length) {
346 Length = getNumBytesForUTF8(Text[i]);
347 if (Length <= 0 || i + Length > Text.size())
348 return ErrorInvalidUTF8;
349 UTF32 buf[1];
350 const UTF8 *Start = reinterpret_cast<const UTF8 *>(Text.data() + i);
351 UTF32 *Target = &buf[0];
352 if (conversionOK != ConvertUTF8toUTF32(&Start, Start + Length, &Target,
353 Target + 1, strictConversion))
354 return ErrorInvalidUTF8;
355 int Width = charWidth(buf[0]);
356 if (Width < 0)
357 return ErrorNonPrintableCharacter;
358 ColumnWidth += Width;
359 }
360 return ColumnWidth;
361}
