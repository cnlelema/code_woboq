504void ScalarBitSetTraits<ELFYAML::ELF_SHF>::bitset(IO &IO,
505 ELFYAML::ELF_SHF &Value) {
506 const auto *Object = static_cast<ELFYAML::Object *>(IO.getContext());
507#define BCase(X) IO.bitSetCase(Value, #X, ELF::X)
508 BCase(SHF_WRITE);
509 BCase(SHF_ALLOC);
510 BCase(SHF_EXCLUDE);
511 BCase(SHF_EXECINSTR);
512 BCase(SHF_MERGE);
513 BCase(SHF_STRINGS);
514 BCase(SHF_INFO_LINK);
515 BCase(SHF_LINK_ORDER);
516 BCase(SHF_OS_NONCONFORMING);
517 BCase(SHF_GROUP);
518 BCase(SHF_TLS);
519 BCase(SHF_COMPRESSED);
520 switch (Object->Header.Machine) {
521 case ELF::EM_ARM:

994void MappingTraits<std::unique_ptr<ELFYAML::Section>>::mapping(
995 IO &IO, std::unique_ptr<ELFYAML::Section> &Section) {
996 ELFYAML::ELF_SHT sectionType;
997 if (IO.outputting())
998 sectionType = Section->Type;
999 else
1000 IO.mapRequired("Type", sectionType);
1001
1002 switch (sectionType) {
1003 case ELF::SHT_DYNAMIC:
1004 if (!IO.outputting())
1005 Section.reset(new ELFYAML::DynamicSection());
1006 sectionMapping(IO, *cast<ELFYAML::DynamicSection>(Section.get()));
1007 break;
1008 case ELF::SHT_REL:
1009 case ELF::SHT_RELA:
1010 if (!IO.outputting())
1011 Section.reset(new ELFYAML::RelocationSection());
1012 sectionMapping(IO, *cast<ELFYAML::RelocationSection>(Section.get()));
1013 break;
1014 case ELF::SHT_GROUP:
1015 if (!IO.outputting())
1016 Section.reset(new ELFYAML::Group());
1017 groupSectionMapping(IO, *cast<ELFYAML::Group>(Section.get()));
1018 break;
1019 case ELF::SHT_NOBITS:
1020 if (!IO.outputting())
1021 Section.reset(new ELFYAML::NoBitsSection());
1022 sectionMapping(IO, *cast<ELFYAML::NoBitsSection>(Section.get()));
1023 break;
1024 case ELF::SHT_MIPS_ABIFLAGS:
1025 if (!IO.outputting())
1026 Section.reset(new ELFYAML::MipsABIFlags());
1027 sectionMapping(IO, *cast<ELFYAML::MipsABIFlags>(Section.get()));
1028 break;
1029 case ELF::SHT_GNU_verdef:
1030 if (!IO.outputting())
1031 Section.reset(new ELFYAML::VerdefSection());
1032 sectionMapping(IO, *cast<ELFYAML::VerdefSection>(Section.get()));
1033 break;
1034 case ELF::SHT_GNU_versym:
1035 if (!IO.outputting())
1036 Section.reset(new ELFYAML::SymverSection());
1037 sectionMapping(IO, *cast<ELFYAML::SymverSection>(Section.get()));
1038 break;
1039 case ELF::SHT_GNU_verneed:
1040 if (!IO.outputting())
1041 Section.reset(new ELFYAML::VerneedSection());
1042 sectionMapping(IO, *cast<ELFYAML::VerneedSection>(Section.get()));
1043 break;
1044 default:
1045 if (!IO.outputting())
1046 Section.reset(new ELFYAML::RawContentSection());
1047 sectionMapping(IO, *cast<ELFYAML::RawContentSection>(Section.get()));
1048 }
1049}

1051StringRef MappingTraits<std::unique_ptr<ELFYAML::Section>>::validate(
1052 IO &io, std::unique_ptr<ELFYAML::Section> &Section) {
1053 const auto *RawSection = dyn_cast<ELFYAML::RawContentSection>(Section.get());
1054 if (!RawSection)
1055 return {};
1056 if (RawSection->Size && RawSection->Content &&
1057 (uint64_t)(*RawSection->Size) < RawSection->Content->binary_size())
1058 return "Section size must be greater than or equal to the content size";
1059 return {};
1060}
