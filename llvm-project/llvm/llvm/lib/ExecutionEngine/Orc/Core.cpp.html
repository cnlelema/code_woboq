59bool flagsMatchCLOpts(const JITSymbolFlags &Flags) {
60#ifndef NDEBUG
61 // Bail out early if this is a hidden symbol and we're not printing hiddens.
62 if (!PrintHidden && !Flags.isExported())
63 return false;
64
65 // Return true if this is callable and we're printing callables.
66 if (PrintCallable && Flags.isCallable())
67 return true;
68
69 // Return true if this is data and we're printing data.
70 if (PrintData && !Flags.isCallable())
71 return true;
72
73 // otherwise return false.
74 return false;
75#else
76 return false;
77#endif // NDEBUG
78}

80// Prints a set of items, filtered by an user-supplied predicate.
81template <typename Set, typename Pred = PrintAll<typename Set::value_type>>
82class SetPrinter {
83public:
84 SetPrinter(const Set &S, Pred ShouldPrint = Pred())
85 : S(S), ShouldPrint(std::move(ShouldPrint)) {}
86
87 void printTo(llvm::raw_ostream &OS) const {
88 bool PrintComma = false;
89 OS << "{";
90 for (auto &E : S) {
91 if (ShouldPrint(E)) {
92 if (PrintComma)
93 OS << ',';
94 OS << ' ' << E;
95 PrintComma = true;
96 }
97 }
98 OS << " }";
99 }
100
101private:
102 const Set &S;
103 mutable Pred ShouldPrint;
104};

106template <typename Set, typename Pred>
107SetPrinter<Set, Pred> printSet(const Set &S, Pred P = Pred()) {
108 return SetPrinter<Set, Pred>(S, std::move(P));
109}

300void AsynchronousSymbolQuery::notifySymbolMetRequiredState(
301 const SymbolStringPtr &Name, JITEvaluatedSymbol Sym) {
302 auto I = ResolvedSymbols.find(Name);
303 assert(I != ResolvedSymbols.end() &&
304 "Resolving symbol outside the requested set");
305 assert(I->second.getAddress() == 0 && "Redundantly resolving symbol Name");
306 I->second = std::move(Sym);
307 --OutstandingSymbolsCount;
308}

310void AsynchronousSymbolQuery::handleComplete() {
311 assert(OutstandingSymbolsCount == 0 &&
312 "Symbols remain, handleComplete called prematurely");
313
314 auto TmpNotifyComplete = std::move(NotifyComplete);
315 NotifyComplete = SymbolsResolvedCallback();
316 TmpNotifyComplete(std::move(ResolvedSymbols));
317}

326 NotifyComplete = SymbolsResolvedCallback();

370void MaterializationResponsibility::resolve(const SymbolMap &Symbols) {
371 LLVM_DEBUG(
372 { dbgs() << "In " << JD.getName() << " resolving " << Symbols << "\n"; });
373#ifndef NDEBUG
374 for (auto &KV : Symbols) {
375 auto I = SymbolFlags.find(KV.first);
376 assert(I != SymbolFlags.end() &&
377 "Resolving symbol outside this responsibility set");
378 if (I->second.isWeak())
379 assert(I->second == (KV.second.getFlags() | JITSymbolFlags::Weak) &&
380 "Resolving symbol with incorrect flags");
381 else
382 assert(I->second == KV.second.getFlags() &&
383 "Resolving symbol with incorrect flags");
384 }
385#endif
386
387 JD.resolve(Symbols);
388}

516void ReExportsMaterializationUnit::materialize(
517 MaterializationResponsibility R) {
518
519 auto &ES = R.getTargetJITDylib().getExecutionSession();
520 JITDylib &TgtJD = R.getTargetJITDylib();
521 JITDylib &SrcJD = SourceJD ? *SourceJD : TgtJD;
522
523 // Find the set of requested aliases and aliasees. Return any unrequested
524 // aliases back to the JITDylib so as to not prematurely materialize any
525 // aliasees.
526 auto RequestedSymbols = R.getRequestedSymbols();
527 SymbolAliasMap RequestedAliases;
528
529 for (auto &Name : RequestedSymbols) {
530 auto I = Aliases.find(Name);
531 assert(I != Aliases.end() && "Symbol not found in aliases map?");
532 RequestedAliases[Name] = std::move(I->second);
533 Aliases.erase(I);
534 }
535
536 LLVM_DEBUG({
537 ES.runSessionLocked([&]() {
538 dbgs() << "materializing reexports: target = " << TgtJD.getName()
539 << ", source = " << SrcJD.getName() << " " << RequestedAliases
540 << "\n";
541 });
542 });
543
544 if (!Aliases.empty()) {
545 if (SourceJD)
546 R.replace(reexports(*SourceJD, std::move(Aliases), MatchNonExported));
547 else
548 R.replace(symbolAliases(std::move(Aliases)));
549 }
550
551 // The OnResolveInfo struct will hold the aliases and responsibilty for each
552 // query in the list.
553 struct OnResolveInfo {
554 OnResolveInfo(MaterializationResponsibility R, SymbolAliasMap Aliases)
555 : R(std::move(R)), Aliases(std::move(Aliases)) {}
556
557 MaterializationResponsibility R;
558 SymbolAliasMap Aliases;
559 };
560
561 // Build a list of queries to issue. In each round we build the largest set of
562 // aliases that we can resolve without encountering a chain definition of the
563 // form Foo -> Bar, Bar -> Baz. Such a form would deadlock as the query would
564 // be waitin on a symbol that it itself had to resolve. Usually this will just
565 // involve one round and a single query.
566
567 std::vector<std::pair<SymbolNameSet, std::shared_ptr<OnResolveInfo>>>
568 QueryInfos;
569 while (!RequestedAliases.empty()) {
570 SymbolNameSet ResponsibilitySymbols;
571 SymbolNameSet QuerySymbols;
572 SymbolAliasMap QueryAliases;
573
574 // Collect as many aliases as we can without including a chain.
575 for (auto &KV : RequestedAliases) {
576 // Chain detected. Skip this symbol for this round.
577 if (&SrcJD == &TgtJD && (QueryAliases.count(KV.second.Aliasee) ||
578 RequestedAliases.count(KV.second.Aliasee)))
579 continue;
580
581 ResponsibilitySymbols.insert(KV.first);
582 QuerySymbols.insert(KV.second.Aliasee);
583 QueryAliases[KV.first] = std::move(KV.second);
584 }
585
586 // Remove the aliases collected this round from the RequestedAliases map.
587 for (auto &KV : QueryAliases)
588 RequestedAliases.erase(KV.first);
589
590 assert(!QuerySymbols.empty() && "Alias cycle detected!");
591
592 auto QueryInfo = std::make_shared<OnResolveInfo>(
593 R.delegate(ResponsibilitySymbols), std::move(QueryAliases));
594 QueryInfos.push_back(
595 make_pair(std::move(QuerySymbols), std::move(QueryInfo)));
596 }
597
598 // Issue the queries.
599 while (!QueryInfos.empty()) {
600 auto QuerySymbols = std::move(QueryInfos.back().first);
601 auto QueryInfo = std::move(QueryInfos.back().second);
602
603 QueryInfos.pop_back();
604
605 auto RegisterDependencies = [QueryInfo,
606 &SrcJD](const SymbolDependenceMap &Deps) {
607 // If there were no materializing symbols, just bail out.
608 if (Deps.empty())
609 return;
610
611 // Otherwise the only deps should be on SrcJD.
612 assert(Deps.size() == 1 && Deps.count(&SrcJD) &&
613 "Unexpected dependencies for reexports");
614
615 auto &SrcJDDeps = Deps.find(&SrcJD)->second;
616 SymbolDependenceMap PerAliasDepsMap;
617 auto &PerAliasDeps = PerAliasDepsMap[&SrcJD];
618
619 for (auto &KV : QueryInfo->Aliases)
620 if (SrcJDDeps.count(KV.second.Aliasee)) {
621 PerAliasDeps = {KV.second.Aliasee};
622 QueryInfo->R.addDependencies(KV.first, PerAliasDepsMap);
623 }
624 };
625
626 auto OnComplete = [QueryInfo](Expected<SymbolMap> Result) {
627 if (Result) {
628 SymbolMap ResolutionMap;
629 for (auto &KV : QueryInfo->Aliases) {
630 assert(Result->count(KV.second.Aliasee) &&
631 "Result map missing entry?");
632 ResolutionMap[KV.first] = JITEvaluatedSymbol(
633 (*Result)[KV.second.Aliasee].getAddress(), KV.second.AliasFlags);
634 }
635 QueryInfo->R.resolve(ResolutionMap);
636 QueryInfo->R.emit();
637 } else {
638 auto &ES = QueryInfo->R.getTargetJITDylib().getExecutionSession();
639 ES.reportError(Result.takeError());
640 QueryInfo->R.failMaterialization();
641 }
642 };
643
644 ES.lookup(JITDylibSearchList({{&SrcJD, MatchNonExported}}), QuerySymbols,
645 SymbolState::Resolved, std::move(OnComplete),
646 std::move(RegisterDependencies));
647 }
648}

666Expected<SymbolAliasMap>
667buildSimpleReexportsAliasMap(JITDylib &SourceJD, const SymbolNameSet &Symbols) {
668 auto Flags = SourceJD.lookupFlags(Symbols);
669
670 if (!Flags)
671 return Flags.takeError();
672
673 if (Flags->size() != Symbols.size()) {
674 SymbolNameSet Unresolved = Symbols;
675 for (auto &KV : *Flags)
676 Unresolved.erase(KV.first);
677 return make_error<SymbolsNotFound>(std::move(Unresolved));
678 }
679
680 SymbolAliasMap Result;
681 for (auto &Name : Symbols) {
682 assert(Flags->count(Name) && "Missing entry in flags map");
683 Result[Name] = SymbolAliasMapEntry(Name, (*Flags)[Name]);
684 }
685
686 return Result;
687}

689ReexportsGenerator::ReexportsGenerator(JITDylib &SourceJD,
690 bool MatchNonExported,
691 SymbolPredicate Allow)
692 : SourceJD(SourceJD), MatchNonExported(MatchNonExported),
693 Allow(std::move(Allow)) {}
694
695Expected<SymbolNameSet>
696ReexportsGenerator::operator()(JITDylib &JD, const SymbolNameSet &Names) {
697 orc::SymbolNameSet Added;
698 orc::SymbolAliasMap AliasMap;
699
700 auto Flags = SourceJD.lookupFlags(Names);
701
702 if (!Flags)
703 return Flags.takeError();
704
705 for (auto &KV : *Flags) {
706 if (Allow && !Allow(KV.first))
707 continue;
708 AliasMap[KV.first] = SymbolAliasMapEntry(KV.first, KV.second);
709 Added.insert(KV.first);
710 }
711
712 if (!Added.empty())
713 cantFail(JD.define(reexports(SourceJD, AliasMap, MatchNonExported)));
714
715 return Added;
716}

718Error JITDylib::defineMaterializing(const SymbolFlagsMap &SymbolFlags) {
719 return ES.runSessionLocked([&]() -> Error {
720 std::vector<SymbolTable::iterator> AddedSyms;
721
722 for (auto &KV : SymbolFlags) {
723 SymbolTable::iterator EntryItr;
724 bool Added;
725
726 std::tie(EntryItr, Added) =
727 Symbols.insert(std::make_pair(KV.first, SymbolTableEntry(KV.second)));
728
729 if (Added) {
730 AddedSyms.push_back(EntryItr);
731 EntryItr->second.setState(SymbolState::Materializing);
732 } else {
733 // Remove any symbols already added.
734 for (auto &SI : AddedSyms)
735 Symbols.erase(SI);
736
737 // FIXME: Return all duplicates.
738 return make_error<DuplicateDefinition>(*KV.first);
739 }
740 }
741
742 return Error::success();
743 });
744}

746void JITDylib::replace(std::unique_ptr<MaterializationUnit> MU) {
747 assert(MU != nullptr && "Can not replace with a null MaterializationUnit");
748
749 auto MustRunMU =
750 ES.runSessionLocked([&, this]() -> std::unique_ptr<MaterializationUnit> {
751
752#ifndef NDEBUG
753 for (auto &KV : MU->getSymbols()) {
754 auto SymI = Symbols.find(KV.first);
755 assert(SymI != Symbols.end() && "Replacing unknown symbol");
756 assert(SymI->second.isInMaterializationPhase() &&
757 "Can not call replace on a symbol that is not materializing");
758 assert(!SymI->second.hasMaterializerAttached() &&
759 "Symbol should not have materializer attached already");
760 assert(UnmaterializedInfos.count(KV.first) == 0 &&
761 "Symbol being replaced should have no UnmaterializedInfo");
762 }
763#endif // NDEBUG
764
765 // If any symbol has pending queries against it then we need to
766 // materialize MU immediately.
767 for (auto &KV : MU->getSymbols()) {
768 auto MII = MaterializingInfos.find(KV.first);
769 if (MII != MaterializingInfos.end()) {
770 if (MII->second.hasQueriesPending())
771 return std::move(MU);
772 }
773 }
774
775 // Otherwise, make MU responsible for all the symbols.
776 auto UMI = std::make_shared<UnmaterializedInfo>(std::move(MU));
777 for (auto &KV : UMI->MU->getSymbols()) {
778 auto SymI = Symbols.find(KV.first);
779 assert(SymI->second.getState() == SymbolState::Materializing &&
780 "Can not replace a symbol that is not materializing");
781 assert(!SymI->second.hasMaterializerAttached() &&
782 "Can not replace a symbol that has a materializer attached");
783 assert(UnmaterializedInfos.count(KV.first) == 0 &&
784 "Unexpected materializer entry in map");
785 SymI->second.setAddress(SymI->second.getAddress());
786 SymI->second.setMaterializerAttached(true);
787 UnmaterializedInfos[KV.first] = UMI;
788 }
789
790 return nullptr;
791 });
792
793 if (MustRunMU)
794 ES.dispatchMaterialization(*this, std::move(MustRunMU));
795}

819void JITDylib::addDependencies(const SymbolStringPtr &Name,
820 const SymbolDependenceMap &Dependencies) {
821 assert(Symbols.count(Name) && "Name not in symbol table");
822 assert(Symbols[Name].isInMaterializationPhase() &&
823 "Can not add dependencies for a symbol that is not materializing");
824

826 assert(!MI.IsEmitted && "Can not add dependencies to an emitted symbol");
827
828 for (auto &KV : Dependencies) {
829 assert(KV.first && "Null JITDylib in dependency?");
830 auto &OtherJITDylib = *KV.first;

832
833 for (auto &OtherSymbol : KV.second) {
834#ifndef NDEBUG
835 // Assert that this symbol exists and has not been emitted already.
836 auto SymI = OtherJITDylib.Symbols.find(OtherSymbol);
837 assert(SymI != OtherJITDylib.Symbols.end() &&

843

845 transferEmittedNodeDependencies(MI, Name, OtherMI);

848 DepsOnOtherJITDylib.insert(OtherSymbol);
849 }
850 }
851
852 if (DepsOnOtherJITDylib.empty())
853 MI.UnemittedDependencies.erase(&OtherJITDylib);

857void JITDylib::resolve(const SymbolMap &Resolved) {
858 auto CompletedQueries = ES.runSessionLocked([&, this]() {
859 AsynchronousSymbolQuerySet CompletedQueries;
860 for (const auto &KV : Resolved) {
861 auto &Name = KV.first;
862 auto Sym = KV.second;
863
864 auto I = Symbols.find(Name);
865
866 assert(I != Symbols.end() && "Symbol not found");
867 assert(!I->second.hasMaterializerAttached() &&
868 "Resolving symbol with materializer attached?");
869 assert(I->second.getState() == SymbolState::Materializing &&
870 "Symbol should be materializing");
871 assert(I->second.getAddress() == 0 && "Symbol has already been resolved");
872
873 assert((Sym.getFlags() & ~JITSymbolFlags::Weak) ==
874 (I->second.getFlags() & ~JITSymbolFlags::Weak) &&
875 "Resolved flags should match the declared flags");
876
877 // Once resolved, symbols can never be weak.
878 JITSymbolFlags ResolvedFlags = Sym.getFlags();
879 ResolvedFlags &= ~JITSymbolFlags::Weak;
880 I->second.setAddress(Sym.getAddress());
881 I->second.setFlags(ResolvedFlags);
882 I->second.setState(SymbolState::Resolved);
883
884 auto &MI = MaterializingInfos[Name];
885 for (auto &Q : MI.takeQueriesMeeting(SymbolState::Resolved)) {
886 Q->notifySymbolMetRequiredState(Name, Sym);
887 if (Q->isComplete())
888 CompletedQueries.insert(std::move(Q));
889 }
890 }
891
892 return CompletedQueries;
893 });
894
895 for (auto &Q : CompletedQueries) {
896 assert(Q->isComplete() && "Q not completed");
897 Q->handleComplete();
898 }

901void JITDylib::emit(const SymbolFlagsMap &Emitted) {
902 auto CompletedQueries = ES.runSessionLocked([&, this]() {
903 AsynchronousSymbolQuerySet CompletedQueries;
904
905 for (const auto &KV : Emitted) {
906 const auto &Name = KV.first;
907
908 auto MII = MaterializingInfos.find(Name);
909 assert(MII != MaterializingInfos.end() &&
910 "Missing MaterializingInfo entry");
911
912 auto &MI = MII->second;
913
914 // For each dependant, transfer this node's emitted dependencies to
915 // it. If the dependant node is ready (i.e. has no unemitted
916 // dependencies) then notify any pending queries.
917 for (auto &KV : MI.Dependants) {
918 auto &DependantJD = *KV.first;
919 for (auto &DependantName : KV.second) {
920 auto DependantMII =
921 DependantJD.MaterializingInfos.find(DependantName);
922 assert(DependantMII != DependantJD.MaterializingInfos.end() &&
923 "Dependant should have MaterializingInfo");
924
925 auto &DependantMI = DependantMII->second;
926
927 // Remove the dependant's dependency on this node.
928 assert(DependantMI.UnemittedDependencies[this].count(Name) &&
929 "Dependant does not count this symbol as a dependency?");
930 DependantMI.UnemittedDependencies[this].erase(Name);
931 if (DependantMI.UnemittedDependencies[this].empty())
932 DependantMI.UnemittedDependencies.erase(this);
933
934 // Transfer unemitted dependencies from this node to the dependant.
935 DependantJD.transferEmittedNodeDependencies(DependantMI,
936 DependantName, MI);
937
938 // If the dependant is emitted and this node was the last of its
939 // unemitted dependencies then the dependant node is now ready, so
940 // notify any pending queries on the dependant node.
941 if (DependantMI.IsEmitted &&
942 DependantMI.UnemittedDependencies.empty()) {
943 assert(DependantMI.Dependants.empty() &&
944 "Dependants should be empty by now");
945
946 // Since this dependant is now ready, we erase its MaterializingInfo
947 // and update its materializing state.
948 auto DependantSymI = DependantJD.Symbols.find(DependantName);
949 assert(DependantSymI != DependantJD.Symbols.end() &&
950 "Dependant has no entry in the Symbols table");
951 DependantSymI->second.setState(SymbolState::Ready);
952
953 for (auto &Q : DependantMI.takeQueriesMeeting(SymbolState::Ready)) {
954 Q->notifySymbolMetRequiredState(
955 DependantName, DependantSymI->second.getSymbol());
956 if (Q->isComplete())
957 CompletedQueries.insert(Q);
958 Q->removeQueryDependence(DependantJD, DependantName);
959 }
960
961 DependantJD.MaterializingInfos.erase(DependantMII);
962 }
963 }
964 }
965 MI.Dependants.clear();
966 MI.IsEmitted = true;
967
968 if (MI.UnemittedDependencies.empty()) {
969 auto SymI = Symbols.find(Name);
970 assert(SymI != Symbols.end() && "Symbol has no entry in Symbols table");
971 SymI->second.setState(SymbolState::Ready);
972 for (auto &Q : MI.takeQueriesMeeting(SymbolState::Ready)) {
973 Q->notifySymbolMetRequiredState(Name, SymI->second.getSymbol());
974 if (Q->isComplete())
975 CompletedQueries.insert(Q);
976 Q->removeQueryDependence(*this, Name);
977 }
978 MaterializingInfos.erase(MII);
979 }
980 }
981
982 return CompletedQueries;
983 });
984
985 for (auto &Q : CompletedQueries) {
986 assert(Q->isComplete() && "Q is not complete");
987 Q->handleComplete();
988 }
989}

991void JITDylib::notifyFailed(const SymbolNameSet &FailedSymbols) {
992
993 // FIXME: This should fail any transitively dependant symbols too.
994
995 auto FailedQueriesToNotify = ES.runSessionLocked([&, this]() {
996 AsynchronousSymbolQuerySet FailedQueries;
997 std::vector<MaterializingInfosMap::iterator> MIIsToRemove;
998
999 for (auto &Name : FailedSymbols) {
1000 auto I = Symbols.find(Name);
1001 assert(I != Symbols.end() && "Symbol not present in this JITDylib");
1002 Symbols.erase(I);
1003
1004 auto MII = MaterializingInfos.find(Name);
1005
1006 // If we have not created a MaterializingInfo for this symbol yet then
1007 // there is nobody to notify.
1008 if (MII == MaterializingInfos.end())
1009 continue;
1010
1011 // Remove this symbol from the dependants list of any dependencies.
1012 for (auto &KV : MII->second.UnemittedDependencies) {
1013 auto *DependencyJD = KV.first;
1014 auto &Dependencies = KV.second;
1015 for (auto &DependencyName : Dependencies) {
1016 auto DependencyMII =
1017 DependencyJD->MaterializingInfos.find(DependencyName);
1018 assert(DependencyMII != DependencyJD->MaterializingInfos.end() &&
1019 "Unemitted dependency must have a MaterializingInfo entry");
1020 assert(DependencyMII->second.Dependants.count(this) &&
1021 "Dependency's dependants list does not contain this JITDylib");
1022 assert(DependencyMII->second.Dependants[this].count(Name) &&
1023 "Dependency's dependants list does not contain dependant");
1024 DependencyMII->second.Dependants[this].erase(Name);
1025 }
1026 }
1027
1028 // Copy all the queries to the FailedQueries list, then abandon them.
1029 // This has to be a copy, and the copy has to come before the abandon
1030 // operation: Each Q.detach() call will reach back into this
1031 // PendingQueries list to remove Q.
1032 for (auto &Q : MII->second.pendingQueries())
1033 FailedQueries.insert(Q);
1034
1035 MIIsToRemove.push_back(std::move(MII));
1036 }
1037
1038 // Detach failed queries.
1039 for (auto &Q : FailedQueries)
1040 Q->detach();
1041
1042 // Remove the MaterializingInfos.
1043 for (auto &MII : MIIsToRemove) {
1044 assert(!MII->second.hasQueriesPending() &&
1045 "Queries remain after symbol was failed");
1046
1047 MaterializingInfos.erase(MII);
1048 }
1049
1050 return FailedQueries;
1051 });
1052
1053 for (auto &Q : FailedQueriesToNotify)
1054 Q->handleFailed(make_error<FailedToMaterialize>(FailedSymbols));
1055}

1099Error JITDylib::remove(const SymbolNameSet &Names) {
1100 return ES.runSessionLocked([&]() -> Error {
1101 using SymbolMaterializerItrPair =
1102 std::pair<SymbolTable::iterator, UnmaterializedInfosMap::iterator>;
1103 std::vector<SymbolMaterializerItrPair> SymbolsToRemove;
1104 SymbolNameSet Missing;
1105 SymbolNameSet Materializing;
1106
1107 for (auto &Name : Names) {
1108 auto I = Symbols.find(Name);
1109
1110 // Note symbol missing.
1111 if (I == Symbols.end()) {
1112 Missing.insert(Name);
1113 continue;
1114 }
1115
1116 // Note symbol materializing.
1117 if (I->second.isInMaterializationPhase()) {
1118 Materializing.insert(Name);
1119 continue;
1120 }
1121
1122 auto UMII = I->second.hasMaterializerAttached()
1123 ? UnmaterializedInfos.find(Name)
1124 : UnmaterializedInfos.end();
1125 SymbolsToRemove.push_back(std::make_pair(I, UMII));
1126 }
1127
1128 // If any of the symbols are not defined, return an error.
1129 if (!Missing.empty())
1130 return make_error<SymbolsNotFound>(std::move(Missing));
1131
1132 // If any of the symbols are currently materializing, return an error.
1133 if (!Materializing.empty())
1134 return make_error<SymbolsCouldNotBeRemoved>(std::move(Materializing));
1135
1136 // Remove the symbols.
1137 for (auto &SymbolMaterializerItrPair : SymbolsToRemove) {
1138 auto UMII = SymbolMaterializerItrPair.second;
1139
1140 // If there is a materializer attached, call discard.
1141 if (UMII != UnmaterializedInfos.end()) {
1142 UMII->second->MU->doDiscard(*this, UMII->first);
1143 UnmaterializedInfos.erase(UMII);
1144 }
1145
1146 auto SymI = SymbolMaterializerItrPair.first;
1147 Symbols.erase(SymI);
1148 }
1149
1150 return Error::success();
1151 });
1152}

1216void JITDylib::lodgeQueryImpl(
1217 std::shared_ptr<AsynchronousSymbolQuery> &Q, SymbolNameSet &Unresolved,
1218 bool MatchNonExported,
1219 std::vector<std::unique_ptr<MaterializationUnit>> &MUs) {
1220
1221 std::vector<SymbolStringPtr> ToRemove;
1222 for (auto Name : Unresolved) {
1223
1224 // Search for the name in Symbols. Skip it if not found.
1225 auto SymI = Symbols.find(Name);
1226 if (SymI == Symbols.end())
1227 continue;
1228
1229 // If this is a non exported symbol and we're skipping those then skip it.
1230 if (!SymI->second.getFlags().isExported() && !MatchNonExported)
1231 continue;
1232
1233 // If we matched against Name in JD, mark it to be removed from the
1234 // Unresolved set.
1235 ToRemove.push_back(Name);
1236
1237 // If this symbol already meets the required state for then notify the
1238 // query and continue.
1239 if (SymI->second.getState() >= Q->getRequiredState()) {
1240 Q->notifySymbolMetRequiredState(Name, SymI->second.getSymbol());
1241 continue;
1242 }
1243
1244 // Otherwise this symbol does not yet meet the required state. Check whether
1245 // it has a materializer attached, and if so prepare to run it.
1246 if (SymI->second.hasMaterializerAttached()) {
1247 assert(SymI->second.getAddress() == 0 &&
1248 "Symbol not resolved but already has address?");
1249 auto UMII = UnmaterializedInfos.find(Name);
1250 assert(UMII != UnmaterializedInfos.end() &&
1251 "Lazy symbol should have UnmaterializedInfo");
1252 auto MU = std::move(UMII->second->MU);
1253 assert(MU != nullptr && "Materializer should not be null");
1254
1255 // Move all symbols associated with this MaterializationUnit into
1256 // materializing state.
1257 for (auto &KV : MU->getSymbols()) {
1258 auto SymK = Symbols.find(KV.first);
1259 SymK->second.setMaterializerAttached(false);
1260 SymK->second.setState(SymbolState::Materializing);
1261 UnmaterializedInfos.erase(KV.first);
1262 }
1263
1264 // Add MU to the list of MaterializationUnits to be materialized.
1265 MUs.push_back(std::move(MU));
1266 }
1267
1268 // Add the query to the PendingQueries list.
1269 assert(SymI->second.isInMaterializationPhase() &&
1270 "By this line the symbol should be materializing");
1271 auto &MI = MaterializingInfos[Name];
1272 MI.addQuery(Q);
1273 Q->addQueryDependence(*this, Name);
1274 }
1275
1276 // Remove any symbols that we found.
1277 for (auto &Name : ToRemove)
1278 Unresolved.erase(Name);
1279}

1336bool JITDylib::lookupImpl(
1337 std::shared_ptr<AsynchronousSymbolQuery> &Q,
1338 std::vector<std::unique_ptr<MaterializationUnit>> &MUs,
1339 SymbolNameSet &Unresolved) {
1340 bool QueryComplete = false;
1341
1342 std::vector<SymbolStringPtr> ToRemove;
1343 for (auto Name : Unresolved) {
1344
1345 // Search for the name in Symbols. Skip it if not found.
1346 auto SymI = Symbols.find(Name);
1347 if (SymI == Symbols.end())
1348 continue;
1349
1350 // If we found Name, mark it to be removed from the Unresolved set.
1351 ToRemove.push_back(Name);
1352
1353 if (SymI->second.getState() >= Q->getRequiredState()) {
1354 Q->notifySymbolMetRequiredState(Name, SymI->second.getSymbol());
1355 if (Q->isComplete())
1356 QueryComplete = true;
1357 continue;
1358 }
1359
1360 // If the symbol is lazy, get the MaterialiaztionUnit for it.
1361 if (SymI->second.hasMaterializerAttached()) {
1362 assert(SymI->second.getAddress() == 0 &&
1363 "Lazy symbol should not have a resolved address");
1364 auto UMII = UnmaterializedInfos.find(Name);
1365 assert(UMII != UnmaterializedInfos.end() &&
1366 "Lazy symbol should have UnmaterializedInfo");
1367 auto MU = std::move(UMII->second->MU);
1368 assert(MU != nullptr && "Materializer should not be null");
1369
1370 // Kick all symbols associated with this MaterializationUnit into
1371 // materializing state.
1372 for (auto &KV : MU->getSymbols()) {
1373 auto SymK = Symbols.find(KV.first);
1374 assert(SymK != Symbols.end() && "Missing symbol table entry");
1375 SymK->second.setState(SymbolState::Materializing);
1376 SymK->second.setMaterializerAttached(false);
1377 UnmaterializedInfos.erase(KV.first);
1378 }
1379
1380 // Add MU to the list of MaterializationUnits to be materialized.
1381 MUs.push_back(std::move(MU));
1382 }
1383
1384 // Add the query to the PendingQueries list.
1385 assert(SymI->second.isInMaterializationPhase() &&
1386 "By this line the symbol should be materializing");
1387 auto &MI = MaterializingInfos[Name];
1388 MI.addQuery(Q);
1389 Q->addQueryDependence(*this, Name);
1390 }
1391
1392 // Remove any marked symbols from the Unresolved set.
1393 for (auto &Name : ToRemove)
1394 Unresolved.erase(Name);
1395
1396 return QueryComplete;
1397}

1500Error JITDylib::defineImpl(MaterializationUnit &MU) {
1501 SymbolNameSet Duplicates;
1502 std::vector<SymbolStringPtr> ExistingDefsOverridden;
1503 std::vector<SymbolStringPtr> MUDefsOverridden;
1504
1505 for (const auto &KV : MU.getSymbols()) {
1506 auto I = Symbols.find(KV.first);
1507
1508 if (I != Symbols.end()) {
1509 if (KV.second.isStrong()) {
1510 if (I->second.getFlags().isStrong() ||
1511 I->second.getState() > SymbolState::NeverSearched)
1512 Duplicates.insert(KV.first);
1513 else {
1514 assert(I->second.getState() == SymbolState::NeverSearched &&
1515 "Overridden existing def should be in the never-searched "
1516 "state");
1517 ExistingDefsOverridden.push_back(KV.first);
1518 }
1519 } else
1520 MUDefsOverridden.push_back(KV.first);
1521 }
1522 }
1523
1524 // If there were any duplicate definitions then bail out.
1525 if (!Duplicates.empty())
1526 return make_error<DuplicateDefinition>(**Duplicates.begin());
1527
1528 // Discard any overridden defs in this MU.
1529 for (auto &S : MUDefsOverridden)
1530 MU.doDiscard(*this, S);
1531
1532 // Discard existing overridden defs.
1533 for (auto &S : ExistingDefsOverridden) {
1534
1535 auto UMII = UnmaterializedInfos.find(S);
1536 assert(UMII != UnmaterializedInfos.end() &&
1537 "Overridden existing def should have an UnmaterializedInfo");
1538 UMII->second->MU->doDiscard(*this, S);
1539 }
1540
1541 // Finally, add the defs from this MU.
1542 for (auto &KV : MU.getSymbols()) {
1543 auto &SymEntry = Symbols[KV.first];
1544 SymEntry.setFlags(KV.second);
1545 SymEntry.setState(SymbolState::NeverSearched);
1546 SymEntry.setMaterializerAttached(true);
1547 }
1548
1549 return Error::success();
1550}

1583 UnemittedDependenciesOnDependencyJD->insert(DependencyName);

1701void ExecutionSession::lookup(
1702 const JITDylibSearchList &SearchOrder, SymbolNameSet Symbols,
1703 SymbolState RequiredState, SymbolsResolvedCallback NotifyComplete,
1704 RegisterDependenciesFunction RegisterDependencies) {
1705
1706 LLVM_DEBUG({
1707 runSessionLocked([&]() {
1708 dbgs() << "Looking up " << Symbols << " in " << SearchOrder
1709 << " (required state: " << RequiredState << ")\n";
1710 });
1711 });
1712
1713 // lookup can be re-entered recursively if running on a single thread. Run any
1714 // outstanding MUs in case this query depends on them, otherwise this lookup
1715 // will starve waiting for a result from an MU that is stuck in the queue.
1716 runOutstandingMUs();
1717
1718 auto Unresolved = std::move(Symbols);
1719 std::map<JITDylib *, MaterializationUnitList> CollectedMUsMap;
1720 auto Q = std::make_shared<AsynchronousSymbolQuery>(Unresolved, RequiredState,
1721 std::move(NotifyComplete));
1722 bool QueryComplete = false;
1723
1724 auto LodgingErr = runSessionLocked([&]() -> Error {
1725 auto LodgeQuery = [&]() -> Error {
1726 for (auto &KV : SearchOrder) {
1727 assert(KV.first && "JITDylibList entries must not be null");
1728 assert(!CollectedMUsMap.count(KV.first) &&
1729 "JITDylibList should not contain duplicate entries");
1730
1731 auto &JD = *KV.first;
1732 auto MatchNonExported = KV.second;
1733 if (auto Err = JD.lodgeQuery(Q, Unresolved, MatchNonExported,
1734 CollectedMUsMap[&JD]))
1735 return Err;
1736 }
1737
1738 if (!Unresolved.empty())
1739 return make_error<SymbolsNotFound>(std::move(Unresolved));
1740
1741 return Error::success();
1742 };
1743
1744 if (auto Err = LodgeQuery()) {
1745 // Query failed.
1746
1747 // Disconnect the query from its dependencies.
1748 Q->detach();
1749
1750 // Replace the MUs.
1751 for (auto &KV : CollectedMUsMap)
1752 for (auto &MU : KV.second)
1753 KV.first->replace(std::move(MU));
1754
1755 return Err;
1756 }
1757
1758 // Query lodged successfully.
1759
1760 // Record whether this query is fully ready / resolved. We will use
1761 // this to call handleFullyResolved/handleFullyReady outside the session
1762 // lock.
1763 QueryComplete = Q->isComplete();
1764
1765 // Call the register dependencies function.
1766 if (RegisterDependencies && !Q->QueryRegistrations.empty())
1767 RegisterDependencies(Q->QueryRegistrations);
1768
1769 return Error::success();
1770 });
1771
1772 if (LodgingErr) {
1773 Q->handleFailed(std::move(LodgingErr));
1774 return;
1775 }
1776
1777 if (QueryComplete)
1778 Q->handleComplete();
1779
1780 // Move the MUs to the OutstandingMUs list, then materialize.
1781 {
1782 std::lock_guard<std::recursive_mutex> Lock(OutstandingMUsMutex);
1783
1784 for (auto &KV : CollectedMUsMap)
1785 for (auto &MU : KV.second)
1786 OutstandingMUs.push_back(std::make_pair(KV.first, std::move(MU)));
1787 }
1788
1789 runOutstandingMUs();
1790}
