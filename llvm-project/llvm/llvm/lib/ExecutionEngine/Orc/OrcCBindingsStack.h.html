55
56 virtual JITSymbol findSymbolIn(orc::VModuleKey K, const std::string &Name,

61 template <typename LayerT> class GenericLayerImpl : public GenericLayer {
62 public:
63 GenericLayerImpl(LayerT &Layer) : Layer(Layer) {}
64
65 JITSymbol findSymbolIn(orc::VModuleKey K, const std::string &Name,
66 bool ExportedSymbolsOnly) override {
67 return Layer.findSymbolIn(K, Name, ExportedSymbolsOnly);
68 }
69
70 Error removeModule(orc::VModuleKey K) override {
71 return Layer.removeModule(K);
72 }
73
74 private:
75 LayerT &Layer;
76 };

123 class CBindingsResolver : public orc::SymbolResolver {
124 public:
125 CBindingsResolver(OrcCBindingsStack &Stack,
126 LLVMOrcSymbolResolverFn ExternalResolver,
127 void *ExternalResolverCtx)
128 : Stack(Stack), ExternalResolver(std::move(ExternalResolver)),
129 ExternalResolverCtx(std::move(ExternalResolverCtx)) {}
130
131 orc::SymbolNameSet
132 getResponsibilitySet(const orc::SymbolNameSet &Symbols) override {
133 orc::SymbolNameSet Result;
134
135 for (auto &S : Symbols) {
136 if (auto Sym = findSymbol(*S)) {
137 if (!Sym.getFlags().isStrong())
138 Result.insert(S);
139 } else if (auto Err = Sym.takeError()) {
140 Stack.reportError(std::move(Err));
141 return orc::SymbolNameSet();
142 }
143 }
144
145 return Result;
146 }
147
148 orc::SymbolNameSet
149 lookup(std::shared_ptr<orc::AsynchronousSymbolQuery> Query,
150 orc::SymbolNameSet Symbols) override {
151 orc::SymbolNameSet UnresolvedSymbols;
152
153 for (auto &S : Symbols) {
154 if (auto Sym = findSymbol(*S)) {
155 if (auto Addr = Sym.getAddress()) {
156 Query->notifySymbolMetRequiredState(
157 S, JITEvaluatedSymbol(*Addr, Sym.getFlags()));
158 } else {
159 Stack.ES.legacyFailQuery(*Query, Addr.takeError());
160 return orc::SymbolNameSet();
161 }
162 } else if (auto Err = Sym.takeError()) {
163 Stack.ES.legacyFailQuery(*Query, std::move(Err));
164 return orc::SymbolNameSet();
165 } else
166 UnresolvedSymbols.insert(S);
167 }
168
169 if (Query->isComplete())
170 Query->handleComplete();
171
172 return UnresolvedSymbols;
173 }
174
175 private:
176 JITSymbol findSymbol(const std::string &Name) {
177 // Search order:
178 // 1. JIT'd symbols.
179 // 2. Runtime overrides.
180 // 3. External resolver (if present).
181
182 if (Stack.CODLayer) {
183 if (auto Sym = Stack.CODLayer->findSymbol(Name, true))
184 return Sym;
185 else if (auto Err = Sym.takeError())
186 return Sym.takeError();
187 } else {
188 if (auto Sym = Stack.CompileLayer.findSymbol(Name, true))
189 return Sym;
190 else if (auto Err = Sym.takeError())
191 return Sym.takeError();
192 }
193
194 if (auto Sym = Stack.CXXRuntimeOverrides.searchOverrides(Name))
195 return Sym;
196
197 if (ExternalResolver)
198 return JITSymbol(ExternalResolver(Name.c_str(), ExternalResolverCtx),
199 JITSymbolFlags::Exported);
200
201 return JITSymbol(nullptr);
202 }
203
204 OrcCBindingsStack &Stack;
205 LLVMOrcSymbolResolverFn ExternalResolver;
206 void *ExternalResolverCtx = nullptr;
207 };
208
209public:
210 OrcCBindingsStack(TargetMachine &TM,
211 IndirectStubsManagerBuilder IndirectStubsMgrBuilder)
212 : CCMgr(createCompileCallbackManager(TM, ES)), DL(TM.createDataLayout()),
213 IndirectStubsMgr(IndirectStubsMgrBuilder()),
214 ObjectLayer(ES,
215 [this](orc::VModuleKey K) {
216 auto ResolverI = Resolvers.find(K);
217 assert(ResolverI != Resolvers.end() &&
218 "No resolver for module K");
219 auto Resolver = std::move(ResolverI->second);
220 Resolvers.erase(ResolverI);
221 return ObjLayerT::Resources{
222 std::make_shared<SectionMemoryManager>(), Resolver};
223 },
224 nullptr,
225 [this](orc::VModuleKey K, const object::ObjectFile &Obj,
226 const RuntimeDyld::LoadedObjectInfo &LoadedObjInfo) {
227 this->notifyFinalized(K, Obj, LoadedObjInfo);
228 },
229 [this](orc::VModuleKey K, const object::ObjectFile &Obj) {
230 this->notifyFreed(K, Obj);
231 }),
232 CompileLayer(ObjectLayer, orc::SimpleCompiler(TM)),
233 CODLayer(createCODLayer(ES, CompileLayer, CCMgr.get(),
234 std::move(IndirectStubsMgrBuilder), Resolvers)),
235 CXXRuntimeOverrides(
236 [this](const std::string &S) { return mangle(S); }) {}
237
238 Error shutdown() {
239 // Run any destructors registered with __cxa_atexit.
240 CXXRuntimeOverrides.runDestructors();
241 // Run any IR destructors.
242 for (auto &DtorRunner : IRStaticDestructorRunners)
243 if (auto Err = DtorRunner.runViaLayer(*this))
244 return Err;
245 return Error::success();
246 }
247
248 std::string mangle(StringRef Name) {
249 std::string MangledName;
250 {
251 raw_string_ostream MangledNameStream(MangledName);
252 Mangler::getNameWithPrefix(MangledNameStream, Name, DL);
253 }
254 return MangledName;
255 }

281 template <typename LayerT>
282 Expected<orc::VModuleKey>
283 addIRModule(LayerT &Layer, std::unique_ptr<Module> M,
284 std::unique_ptr<RuntimeDyld::MemoryManager> MemMgr,
285 LLVMOrcSymbolResolverFn ExternalResolver,
286 void *ExternalResolverCtx) {
287
288 // Attach a data-layout if one isn't already present.
289 if (M->getDataLayout().isDefault())
290 M->setDataLayout(DL);
291
292 // Record the static constructors and destructors. We have to do this before
293 // we hand over ownership of the module to the JIT.
294 std::vector<std::string> CtorNames, DtorNames;
295 for (auto Ctor : orc::getConstructors(*M))
296 CtorNames.push_back(mangle(Ctor.Func->getName()));
297 for (auto Dtor : orc::getDestructors(*M))
298 DtorNames.push_back(mangle(Dtor.Func->getName()));
299
300 // Add the module to the JIT.
301 auto K = ES.allocateVModule();
302 Resolvers[K] = std::make_shared<CBindingsResolver>(*this, ExternalResolver,
303 ExternalResolverCtx);
304 if (auto Err = Layer.addModule(K, std::move(M)))
305 return std::move(Err);
306
307 KeyLayers[K] = detail::createGenericLayer(Layer);
308
309 // Run the static constructors, and save the static destructor runner for
310 // execution when the JIT is torn down.
311 orc::LegacyCtorDtorRunner<OrcCBindingsStack> CtorRunner(std::move(CtorNames), K);
312 if (auto Err = CtorRunner.runViaLayer(*this))
313 return std::move(Err);
314
315 IRStaticDestructorRunners.emplace_back(std::move(DtorNames), K);
316
317 return K;
318 }

343 Error removeModule(orc::VModuleKey K) {
344 // FIXME: Should error release the module key?
345 if (auto Err = KeyLayers[K]->removeModule(K))
346 return Err;
347 ES.releaseVModule(K);
348 KeyLayers.erase(K);
349 return Error::success();
350 }
351
352 Expected<orc::VModuleKey> addObject(std::unique_ptr<MemoryBuffer> ObjBuffer,
353 LLVMOrcSymbolResolverFn ExternalResolver,
354 void *ExternalResolverCtx) {
355 if (auto Obj = object::ObjectFile::createObjectFile(
356 ObjBuffer->getMemBufferRef())) {
357
358 auto K = ES.allocateVModule();
359 Resolvers[K] = std::make_shared<CBindingsResolver>(
360 *this, ExternalResolver, ExternalResolverCtx);
361
362 if (auto Err = ObjectLayer.addObject(K, std::move(ObjBuffer)))
363 return std::move(Err);
364
365 KeyLayers[K] = detail::createGenericLayer(ObjectLayer);
366
367 return K;
368 } else
369 return Obj.takeError();
370 }

372 JITSymbol findSymbol(const std::string &Name,
373 bool ExportedSymbolsOnly) {
374 if (auto Sym = IndirectStubsMgr->findStub(Name, ExportedSymbolsOnly))
375 return Sym;
376 if (CODLayer)
377 return CODLayer->findSymbol(mangle(Name), ExportedSymbolsOnly);
378 return CompileLayer.findSymbol(mangle(Name), ExportedSymbolsOnly);
379 }

381 JITSymbol findSymbolIn(orc::VModuleKey K, const std::string &Name,
382 bool ExportedSymbolsOnly) {
383 assert(KeyLayers.count(K) && "looking up symbol in unknown module");
384 return KeyLayers[K]->findSymbolIn(K, mangle(Name), ExportedSymbolsOnly);
385 }
386
387 Expected<JITTargetAddress> findSymbolAddress(const std::string &Name,
388 bool ExportedSymbolsOnly) {
389 if (auto Sym = findSymbol(Name, ExportedSymbolsOnly)) {
390 // Successful lookup, non-null symbol:
391 if (auto AddrOrErr = Sym.getAddress())
392 return *AddrOrErr;
393 else
394 return AddrOrErr.takeError();
395 } else if (auto Err = Sym.takeError()) {
396 // Lookup failure - report error.
397 return std::move(Err);
398 }
399
400 // No symbol not found. Return 0.
401 return 0;
402 }
403
404 Expected<JITTargetAddress> findSymbolAddressIn(orc::VModuleKey K,
405 const std::string &Name,
406 bool ExportedSymbolsOnly) {
407 if (auto Sym = findSymbolIn(K, Name, ExportedSymbolsOnly)) {
408 // Successful lookup, non-null symbol:
409 if (auto AddrOrErr = Sym.getAddress())
410 return *AddrOrErr;
411 else
412 return AddrOrErr.takeError();
413 } else if (auto Err = Sym.takeError()) {
414 // Lookup failure - report error.
415 return std::move(Err);
416 }
417
418 // Symbol not found. Return 0.
419 return 0;
420 }

424 void RegisterJITEventListener(JITEventListener *L) {
425 if (!L)
426 return;
427 EventListeners.push_back(L);
428 }
429
430 void UnregisterJITEventListener(JITEventListener *L) {
431 if (!L)
432 return;
433
434 auto I = find(reverse(EventListeners), L);
435 if (I != EventListeners.rend()) {
436 std::swap(*I, EventListeners.back());
437 EventListeners.pop_back();
438 }
439 }

445 static std::unique_ptr<CompileCallbackMgr>
446 createCompileCallbackManager(TargetMachine &TM, orc::ExecutionSession &ES) {
447 auto CCMgr = createLocalCompileCallbackManager(TM.getTargetTriple(), ES, 0);
448 if (!CCMgr) {
449 // FIXME: It would be good if we could report this somewhere, but we do
450 // have an instance yet.
451 logAllUnhandledErrors(CCMgr.takeError(), errs(), "ORC error: ");
452 return nullptr;
453 }
454 return std::move(*CCMgr);
455 }
456
457 static std::unique_ptr<CODLayerT>
458 createCODLayer(orc::ExecutionSession &ES, CompileLayerT &CompileLayer,
459 CompileCallbackMgr *CCMgr,
460 IndirectStubsManagerBuilder IndirectStubsMgrBuilder,
461 ResolverMap &Resolvers) {
462 // If there is no compile callback manager available we can not create a
463 // compile on demand layer.
464 if (!CCMgr)
465 return nullptr;
466
467 return llvm::make_unique<CODLayerT>(
468 ES, CompileLayer,
469 [&Resolvers](orc::VModuleKey K) {
470 auto ResolverI = Resolvers.find(K);
471 assert(ResolverI != Resolvers.end() && "No resolver for module K");
472 return ResolverI->second;
473 },
474 [&Resolvers](orc::VModuleKey K,
475 std::shared_ptr<orc::SymbolResolver> Resolver) {
476 assert(!Resolvers.count(K) && "Resolver already present");
477 Resolvers[K] = std::move(Resolver);
478 },
479 [](Function &F) { return std::set<Function *>({&F}); }, *CCMgr,
480 std::move(IndirectStubsMgrBuilder), false);
481 }

483 void reportError(Error Err) {
484 // FIXME: Report errors on the execution session.
485 logAllUnhandledErrors(std::move(Err), errs(), "ORC error: ");
486 };
487
488 void notifyFinalized(orc::VModuleKey K,
489 const object::ObjectFile &Obj,
490 const RuntimeDyld::LoadedObjectInfo &LoadedObjInfo) {
491 uint64_t Key = static_cast<uint64_t>(
492 reinterpret_cast<uintptr_t>(Obj.getData().data()));
493 for (auto &Listener : EventListeners)
494 Listener->notifyObjectLoaded(Key, Obj, LoadedObjInfo);
495 }
496
497 void notifyFreed(orc::VModuleKey K, const object::ObjectFile &Obj) {
498 uint64_t Key = static_cast<uint64_t>(
499 reinterpret_cast<uintptr_t>(Obj.getData().data()));
500 for (auto &Listener : EventListeners)
501 Listener->notifyFreeingObject(Key);
502 }
