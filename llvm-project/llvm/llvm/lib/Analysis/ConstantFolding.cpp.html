61static Constant *foldConstVectorToAPInt(APInt &Result, Type *DestTy,
62 Constant *C, Type *SrcEltTy,
63 unsigned NumSrcElts,
64 const DataLayout &DL) {
65 // Now that we know that the input value is a vector of integers, just shift
66 // and insert them into our result.
67 unsigned BitShift = DL.getTypeSizeInBits(SrcEltTy);
68 for (unsigned i = 0; i != NumSrcElts; ++i) {
69 Constant *Element;
70 if (DL.isLittleEndian())
71 Element = C->getAggregateElement(NumSrcElts - i - 1);
72 else
73 Element = C->getAggregateElement(i);
74
75 if (Element && isa<UndefValue>(Element)) {
76 Result <<= BitShift;
77 continue;
78 }
79
80 auto *ElementCI = dyn_cast_or_null<ConstantInt>(Element);
81 if (!ElementCI)
82 return ConstantExpr::getBitCast(C, DestTy);
83
84 Result <<= BitShift;
85 Result |= ElementCI->getValue().zextOrSelf(Result.getBitWidth());
86 }
87
88 return nullptr;
89}

91/// Constant fold bitcast, symbolically evaluating it with DataLayout.
92/// This always returns a non-null constant, but it may be a
93/// ConstantExpr if unfoldable.
94Constant *FoldBitCast(Constant *C, Type *DestTy, const DataLayout &DL) {
95 // Catch the obvious splat cases.
96 if (C->isNullValue() && !DestTy->isX86_MMXTy())
97 return Constant::getNullValue(DestTy);
98 if (C->isAllOnesValue() && !DestTy->isX86_MMXTy() &&
99 !DestTy->isPtrOrPtrVectorTy()) // Don't get ones for ptr types!
100 return Constant::getAllOnesValue(DestTy);
101
102 if (auto *VTy = dyn_cast<VectorType>(C->getType())) {
103 // Handle a vector->scalar integer/fp cast.
104 if (isa<IntegerType>(DestTy) || DestTy->isFloatingPointTy()) {
105 unsigned NumSrcElts = VTy->getNumElements();
106 Type *SrcEltTy = VTy->getElementType();
107
108 // If the vector is a vector of floating point, convert it to vector of int
109 // to simplify things.
110 if (SrcEltTy->isFloatingPointTy()) {
111 unsigned FPWidth = SrcEltTy->getPrimitiveSizeInBits();
112 Type *SrcIVTy =
113 VectorType::get(IntegerType::get(C->getContext(), FPWidth), NumSrcElts);
114 // Ask IR to do the conversion now that #elts line up.
115 C = ConstantExpr::getBitCast(C, SrcIVTy);
116 }
117
118 APInt Result(DL.getTypeSizeInBits(DestTy), 0);
119 if (Constant *CE = foldConstVectorToAPInt(Result, DestTy, C,
120 SrcEltTy, NumSrcElts, DL))
121 return CE;
122
123 if (isa<IntegerType>(DestTy))
124 return ConstantInt::get(DestTy, Result);
125
126 APFloat FP(DestTy->getFltSemantics(), Result);
127 return ConstantFP::get(DestTy->getContext(), FP);
128 }
129 }
130
131 // The code below only handles casts to vectors currently.
132 auto *DestVTy = dyn_cast<VectorType>(DestTy);
133 if (!DestVTy)
134 return ConstantExpr::getBitCast(C, DestTy);
135
136 // If this is a scalar -> vector cast, convert the input into a <1 x scalar>
137 // vector so the code below can handle it uniformly.
138 if (isa<ConstantFP>(C) || isa<ConstantInt>(C)) {
139 Constant *Ops = C; // don't take the address of C!
140 return FoldBitCast(ConstantVector::get(Ops), DestTy, DL);
141 }
142
143 // If this is a bitcast from constant vector -> vector, fold it.
144 if (!isa<ConstantDataVector>(C) && !isa<ConstantVector>(C))
145 return ConstantExpr::getBitCast(C, DestTy);
146
147 // If the element types match, IR can fold it.
148 unsigned NumDstElt = DestVTy->getNumElements();
149 unsigned NumSrcElt = C->getType()->getVectorNumElements();
150 if (NumDstElt == NumSrcElt)
151 return ConstantExpr::getBitCast(C, DestTy);
152
153 Type *SrcEltTy = C->getType()->getVectorElementType();
154 Type *DstEltTy = DestVTy->getElementType();
155
156 // Otherwise, we're changing the number of elements in a vector, which
157 // requires endianness information to do the right thing. For example,
158 // bitcast (<2 x i64> <i64 0, i64 1> to <4 x i32>)
159 // folds to (little endian):
160 // <4 x i32> <i32 0, i32 0, i32 1, i32 0>
161 // and to (big endian):
162 // <4 x i32> <i32 0, i32 0, i32 0, i32 1>
163
164 // First thing is first. We only want to think about integer here, so if
165 // we have something in FP form, recast it as integer.
166 if (DstEltTy->isFloatingPointTy()) {
167 // Fold to an vector of integers with same size as our FP type.
168 unsigned FPWidth = DstEltTy->getPrimitiveSizeInBits();
169 Type *DestIVTy =
170 VectorType::get(IntegerType::get(C->getContext(), FPWidth), NumDstElt);
171 // Recursively handle this integer conversion, if possible.
172 C = FoldBitCast(C, DestIVTy, DL);
173
174 // Finally, IR can handle this now that #elts line up.
175 return ConstantExpr::getBitCast(C, DestTy);
176 }
177
178 // Okay, we know the destination is integer, if the input is FP, convert
179 // it to integer first.
180 if (SrcEltTy->isFloatingPointTy()) {
181 unsigned FPWidth = SrcEltTy->getPrimitiveSizeInBits();
182 Type *SrcIVTy =
183 VectorType::get(IntegerType::get(C->getContext(), FPWidth), NumSrcElt);
184 // Ask IR to do the conversion now that #elts line up.
185 C = ConstantExpr::getBitCast(C, SrcIVTy);
186 // If IR wasn't able to fold it, bail out.
187 if (!isa<ConstantVector>(C) && // FIXME: Remove ConstantVector.
188 !isa<ConstantDataVector>(C))
189 return C;
190 }
191
192 // Now we know that the input and output vectors are both integer vectors
193 // of the same size, and that their #elements is not the same. Do the
194 // conversion here, which depends on whether the input or output has
195 // more elements.
196 bool isLittleEndian = DL.isLittleEndian();
197
198 SmallVector<Constant*, 32> Result;
199 if (NumDstElt < NumSrcElt) {
200 // Handle: bitcast (<4 x i32> <i32 0, i32 1, i32 2, i32 3> to <2 x i64>)
201 Constant *Zero = Constant::getNullValue(DstEltTy);
202 unsigned Ratio = NumSrcElt/NumDstElt;
203 unsigned SrcBitSize = SrcEltTy->getPrimitiveSizeInBits();
204 unsigned SrcElt = 0;
205 for (unsigned i = 0; i != NumDstElt; ++i) {
206 // Build each element of the result.
207 Constant *Elt = Zero;
208 unsigned ShiftAmt = isLittleEndian ? 0 : SrcBitSize*(Ratio-1);
209 for (unsigned j = 0; j != Ratio; ++j) {
210 Constant *Src = C->getAggregateElement(SrcElt++);
211 if (Src && isa<UndefValue>(Src))
212 Src = Constant::getNullValue(C->getType()->getVectorElementType());
213 else
214 Src = dyn_cast_or_null<ConstantInt>(Src);
215 if (!Src) // Reject constantexpr elements.
216 return ConstantExpr::getBitCast(C, DestTy);
217
218 // Zero extend the element to the right size.
219 Src = ConstantExpr::getZExt(Src, Elt->getType());
220
221 // Shift it to the right place, depending on endianness.
222 Src = ConstantExpr::getShl(Src,
223 ConstantInt::get(Src->getType(), ShiftAmt));
224 ShiftAmt += isLittleEndian ? SrcBitSize : -SrcBitSize;
225
226 // Mix it in.
227 Elt = ConstantExpr::getOr(Elt, Src);
228 }
229 Result.push_back(Elt);
230 }
231 return ConstantVector::get(Result);
232 }
233
234 // Handle: bitcast (<2 x i64> <i64 0, i64 1> to <4 x i32>)
235 unsigned Ratio = NumDstElt/NumSrcElt;
236 unsigned DstBitSize = DL.getTypeSizeInBits(DstEltTy);
237
238 // Loop over each source value, expanding into multiple results.
239 for (unsigned i = 0; i != NumSrcElt; ++i) {
240 auto *Element = C->getAggregateElement(i);
241
242 if (!Element) // Reject constantexpr elements.
243 return ConstantExpr::getBitCast(C, DestTy);
244
245 if (isa<UndefValue>(Element)) {
246 // Correctly Propagate undef values.
247 Result.append(Ratio, UndefValue::get(DstEltTy));
248 continue;
249 }
250
251 auto *Src = dyn_cast<ConstantInt>(Element);
252 if (!Src)
253 return ConstantExpr::getBitCast(C, DestTy);
254
255 unsigned ShiftAmt = isLittleEndian ? 0 : DstBitSize*(Ratio-1);
256 for (unsigned j = 0; j != Ratio; ++j) {
257 // Shift the piece of the value into the right place, depending on
258 // endianness.
259 Constant *Elt = ConstantExpr::getLShr(Src,
260 ConstantInt::get(Src->getType(), ShiftAmt));
261 ShiftAmt += isLittleEndian ? DstBitSize : -DstBitSize;
262
263 // Truncate the element to an integer with the same pointer size and
264 // convert the element back to a pointer using a inttoptr.
265 if (DstEltTy->isPointerTy()) {
266 IntegerType *DstIntTy = Type::getIntNTy(C->getContext(), DstBitSize);
267 Constant *CE = ConstantExpr::getTrunc(Elt, DstIntTy);
268 Result.push_back(ConstantExpr::getIntToPtr(CE, DstEltTy));
269 continue;
270 }
271
272 // Truncate and remember this piece.
273 Result.push_back(ConstantExpr::getTrunc(Elt, DstEltTy));
274 }
275 }
276
277 return ConstantVector::get(Result);
278}

282/// If this constant is a constant offset from a global, return the global and
283/// the constant. Because of constantexprs, this function is recursive.
284bool llvm::IsConstantOffsetFromGlobal(Constant *C, GlobalValue *&GV,
285 APInt &Offset, const DataLayout &DL,
286 bool LookThroughAddrSpaces) {
287 // Trivial case, constant is the global.
288 if ((GV = dyn_cast<GlobalValue>(C))) {
289 unsigned BitWidth = DL.getIndexTypeSizeInBits(GV->getType());
290 Offset = APInt(BitWidth, 0);
291 return true;
292 }
293
294 // Otherwise, if this isn't a constant expr, bail out.
295 auto *CE = dyn_cast<ConstantExpr>(C);
296 if (!CE) return false;
297
298 // For CHERI we also need to look through AddrSpaceCasts
299 // Look through ptr->int and ptr->ptr casts.
300 if (CE->getOpcode() == Instruction::PtrToInt ||
301 CE->getOpcode() == Instruction::BitCast ||
302 (LookThroughAddrSpaces && CE->getOpcode() == Instruction::AddrSpaceCast))
303 return IsConstantOffsetFromGlobal(CE->getOperand(0), GV, Offset, DL, LookThroughAddrSpaces);
304
305 // i32* getelementptr ([5 x i32]* @a, i32 0, i32 5)
306 auto *GEP = dyn_cast<GEPOperator>(CE);
307 if (!GEP)
308 return false;
309
310 unsigned BitWidth = DL.getIndexTypeSizeInBits(GEP->getType());
311 APInt TmpOffset(BitWidth, 0);
312
313 // If the base isn't a global+constant, we aren't either.
314 if (!IsConstantOffsetFromGlobal(CE->getOperand(0), GV, TmpOffset, DL, LookThroughAddrSpaces))
315 return false;
316
317 // Otherwise, add any offset that our operands provide.
318 if (!GEP->accumulateConstantOffset(DL, TmpOffset))
319 return false;
320
321 Offset = TmpOffset;
322 return true;
323}

325Constant *llvm::ConstantFoldLoadThroughBitcast(Constant *C, Type *DestTy,
326 const DataLayout &DL) {
327 do {
328 Type *SrcTy = C->getType();
329
330 // If the type sizes are the same and a cast is legal, just directly
331 // cast the constant.
332 if (DL.getTypeSizeInBits(DestTy) == DL.getTypeSizeInBits(SrcTy)) {
333 Instruction::CastOps Cast = Instruction::BitCast;
334 // If we are going from a pointer to int or vice versa, we spell the cast
335 // differently.
336 if (SrcTy->isIntegerTy() && DestTy->isPointerTy())
337 Cast = Instruction::IntToPtr;
338 else if (SrcTy->isPointerTy() && DestTy->isIntegerTy())
339 Cast = Instruction::PtrToInt;
340
341 if (CastInst::castIsValid(Cast, C, DestTy))
342 return ConstantExpr::getCast(Cast, C, DestTy);
343 }
344
345 // If this isn't an aggregate type, there is nothing we can do to drill down
346 // and find a bitcastable constant.
347 if (!SrcTy->isAggregateType())
348 return nullptr;
349
350 // We're simulating a load through a pointer that was bitcast to point to
351 // a different type, so we can try to walk down through the initial
352 // elements of an aggregate to see if some part of the aggregate is
353 // castable to implement the "load" semantic model.
354 if (SrcTy->isStructTy()) {
355 // Struct types might have leading zero-length elements like [0 x i32],
356 // which are certainly not what we are looking for, so skip them.
357 unsigned Elem = 0;
358 Constant *ElemC;
359 do {
360 ElemC = C->getAggregateElement(Elem++);
361 } while (ElemC && DL.getTypeSizeInBits(ElemC->getType()) == 0);
362 C = ElemC;
363 } else {
364 C = C->getAggregateElement(0u);
365 }
366 } while (C);
367
368 return nullptr;
369}

373/// Recursive helper to read bits out of global. C is the constant being copied
374/// out of. ByteOffset is an offset into C. CurPtr is the pointer to copy
375/// results into and BytesLeft is the number of bytes left in
376/// the CurPtr buffer. DL is the DataLayout.
377bool ReadDataFromGlobal(Constant *C, uint64_t ByteOffset, unsigned char *CurPtr,
378 unsigned BytesLeft, const DataLayout &DL) {
379 assert(ByteOffset <= DL.getTypeAllocSize(C->getType()) &&
380 "Out of range access");
381
382 // If this element is zero or undefined, we can just return since *CurPtr is
383 // zero initialized.
384 if (isa<ConstantAggregateZero>(C) || isa<UndefValue>(C))
385 return true;
386
387 if (auto *CI = dyn_cast<ConstantInt>(C)) {
388 if (CI->getBitWidth() > 64 ||
389 (CI->getBitWidth() & 7) != 0)
390 return false;
391
392 uint64_t Val = CI->getZExtValue();
393 unsigned IntBytes = unsigned(CI->getBitWidth()/8);
394
395 for (unsigned i = 0; i != BytesLeft && ByteOffset != IntBytes; ++i) {
396 int n = ByteOffset;
397 if (!DL.isLittleEndian())
398 n = IntBytes - n - 1;
399 CurPtr[i] = (unsigned char)(Val >> (n * 8));
400 ++ByteOffset;
401 }
402 return true;
403 }
404
405 if (auto *CFP = dyn_cast<ConstantFP>(C)) {
406 if (CFP->getType()->isDoubleTy()) {
407 C = FoldBitCast(C, Type::getInt64Ty(C->getContext()), DL);
408 return ReadDataFromGlobal(C, ByteOffset, CurPtr, BytesLeft, DL);
409 }
410 if (CFP->getType()->isFloatTy()){
411 C = FoldBitCast(C, Type::getInt32Ty(C->getContext()), DL);
412 return ReadDataFromGlobal(C, ByteOffset, CurPtr, BytesLeft, DL);
413 }
414 if (CFP->getType()->isHalfTy()){
415 C = FoldBitCast(C, Type::getInt16Ty(C->getContext()), DL);
416 return ReadDataFromGlobal(C, ByteOffset, CurPtr, BytesLeft, DL);
417 }
418 return false;
419 }
420
421 if (auto *CS = dyn_cast<ConstantStruct>(C)) {
422 const StructLayout *SL = DL.getStructLayout(CS->getType());
423 unsigned Index = SL->getElementContainingOffset(ByteOffset);
424 uint64_t CurEltOffset = SL->getElementOffset(Index);
425 ByteOffset -= CurEltOffset;
426
427 while (true) {
428 // If the element access is to the element itself and not to tail padding,
429 // read the bytes from the element.
430 uint64_t EltSize = DL.getTypeAllocSize(CS->getOperand(Index)->getType());
431
432 if (ByteOffset < EltSize &&
433 !ReadDataFromGlobal(CS->getOperand(Index), ByteOffset, CurPtr,
434 BytesLeft, DL))
435 return false;
436
437 ++Index;
438
439 // Check to see if we read from the last struct element, if so we're done.
440 if (Index == CS->getType()->getNumElements())
441 return true;
442
443 // If we read all of the bytes we needed from this element we're done.
444 uint64_t NextEltOffset = SL->getElementOffset(Index);
445
446 if (BytesLeft <= NextEltOffset - CurEltOffset - ByteOffset)
447 return true;
448
449 // Move to the next element of the struct.
450 CurPtr += NextEltOffset - CurEltOffset - ByteOffset;
451 BytesLeft -= NextEltOffset - CurEltOffset - ByteOffset;
452 ByteOffset = 0;
453 CurEltOffset = NextEltOffset;
454 }
455 // not reached.
456 }
457
458 if (isa<ConstantArray>(C) || isa<ConstantVector>(C) ||
459 isa<ConstantDataSequential>(C)) {
460 Type *EltTy = C->getType()->getSequentialElementType();
461 uint64_t EltSize = DL.getTypeAllocSize(EltTy);
462 uint64_t Index = ByteOffset / EltSize;
463 uint64_t Offset = ByteOffset - Index * EltSize;
464 uint64_t NumElts;
465 if (auto *AT = dyn_cast<ArrayType>(C->getType()))
466 NumElts = AT->getNumElements();
467 else
468 NumElts = C->getType()->getVectorNumElements();
469
470 for (; Index != NumElts; ++Index) {
471 if (!ReadDataFromGlobal(C->getAggregateElement(Index), Offset, CurPtr,
472 BytesLeft, DL))
473 return false;
474
475 uint64_t BytesWritten = EltSize - Offset;
476 assert(BytesWritten <= EltSize && "Not indexing into this element?");
477 if (BytesWritten >= BytesLeft)
478 return true;
479
480 Offset = 0;
481 BytesLeft -= BytesWritten;
482 CurPtr += BytesWritten;
483 }
484 return true;
485 }
486
487 if (auto *CE = dyn_cast<ConstantExpr>(C)) {
488 if (CE->getOpcode() == Instruction::IntToPtr &&
489 CE->getOperand(0)->getType() == DL.getIntPtrType(CE->getType())) {
490 return ReadDataFromGlobal(CE->getOperand(0), ByteOffset, CurPtr,
491 BytesLeft, DL);
492 }
493 }
494
495 // Otherwise, unknown initializer type.
496 return false;
497}

499Constant *FoldReinterpretLoadFromConstPtr(Constant *C, Type *LoadTy,
500 const DataLayout &DL) {
501 auto *PTy = cast<PointerType>(C->getType());
502 auto *IntType = dyn_cast<IntegerType>(LoadTy);
503
504 // If this isn't an integer load we can't fold it directly.
505 if (!IntType) {
506 unsigned AS = PTy->getAddressSpace();
507
508 // If this is a float/double load, we can try folding it as an int32/64 load
509 // and then bitcast the result. This can be useful for union cases. Note
510 // that address spaces don't matter here since we're not going to result in
511 // an actual new load.
512 Type *MapTy;
513 if (LoadTy->isHalfTy())
514 MapTy = Type::getInt16Ty(C->getContext());
515 else if (LoadTy->isFloatTy())
516 MapTy = Type::getInt32Ty(C->getContext());
517 else if (LoadTy->isDoubleTy())
518 MapTy = Type::getInt64Ty(C->getContext());
519 else if (LoadTy->isVectorTy()) {
520 MapTy = PointerType::getIntNTy(C->getContext(),
521 DL.getTypeAllocSizeInBits(LoadTy));
522 } else
523 return nullptr;
524
525 C = FoldBitCast(C, MapTy->getPointerTo(AS), DL);
526 if (Constant *Res = FoldReinterpretLoadFromConstPtr(C, MapTy, DL))
527 return FoldBitCast(Res, LoadTy, DL);
528 return nullptr;
529 }
530
531 unsigned BytesLoaded = (IntType->getBitWidth() + 7) / 8;
532 if (BytesLoaded > 32 || BytesLoaded == 0)
533 return nullptr;
534
535 GlobalValue *GVal;
536 APInt OffsetAI;
537 if (!IsConstantOffsetFromGlobal(C, GVal, OffsetAI, DL, false))
538 return nullptr;
539
540 auto *GV = dyn_cast<GlobalVariable>(GVal);
541 if (!GV || !GV->isConstant() || !GV->hasDefinitiveInitializer() ||
542 !GV->getInitializer()->getType()->isSized())
543 return nullptr;
544
545 int64_t Offset = OffsetAI.getSExtValue();
546 int64_t InitializerSize = DL.getTypeAllocSize(GV->getInitializer()->getType());
547
548 // If we're not accessing anything in this constant, the result is undefined.
549 if (Offset + BytesLoaded <= 0)
550 return UndefValue::get(IntType);
551
552 // If we're not accessing anything in this constant, the result is undefined.
553 if (Offset >= InitializerSize)
554 return UndefValue::get(IntType);
555
556 unsigned char RawBytes[32] = {0};
557 unsigned char *CurPtr = RawBytes;
558 unsigned BytesLeft = BytesLoaded;
559
560 // If we're loading off the beginning of the global, some bytes may be valid.
561 if (Offset < 0) {
562 CurPtr += -Offset;
563 BytesLeft += Offset;
564 Offset = 0;
565 }
566
567 if (!ReadDataFromGlobal(GV->getInitializer(), Offset, CurPtr, BytesLeft, DL))
568 return nullptr;
569
570 APInt ResultVal = APInt(IntType->getBitWidth(), 0);
571 if (DL.isLittleEndian()) {
572 ResultVal = RawBytes[BytesLoaded - 1];
573 for (unsigned i = 1; i != BytesLoaded; ++i) {
574 ResultVal <<= 8;
575 ResultVal |= RawBytes[BytesLoaded - 1 - i];
576 }
577 } else {
578 ResultVal = RawBytes[0];
579 for (unsigned i = 1; i != BytesLoaded; ++i) {
580 ResultVal <<= 8;
581 ResultVal |= RawBytes[i];
582 }
583 }
584
585 return ConstantInt::get(IntType->getContext(), ResultVal);
586}

588Constant *ConstantFoldLoadThroughBitcastExpr(ConstantExpr *CE, Type *DestTy,
589 const DataLayout &DL) {
590 auto *SrcPtr = CE->getOperand(0);

592 if (!SrcPtrTy)
593 return nullptr;
594 Type *SrcTy = SrcPtrTy->getPointerElementType();

596 Constant *C = ConstantFoldLoadFromConstPtr(SrcPtr, SrcTy, DL);
597 if (!C)
598 return nullptr;
599
600 return llvm::ConstantFoldLoadThroughBitcast(C, DestTy, DL);
601}

605Constant *llvm::ConstantFoldLoadFromConstPtr(Constant *C, Type *Ty,
606 const DataLayout &DL) {
607 // First, try the easy cases:
608 if (auto *GV = dyn_cast<GlobalVariable>(C))
609 if (GV->isConstant() && GV->hasDefinitiveInitializer())
610 return GV->getInitializer();
611
612 if (auto *GA = dyn_cast<GlobalAlias>(C))
613 if (GA->getAliasee() && !GA->isInterposable())
614 return ConstantFoldLoadFromConstPtr(GA->getAliasee(), Ty, DL);
615
616 // If the loaded value isn't a constant expr, we can't handle it.
617 auto *CE = dyn_cast<ConstantExpr>(C);
618 if (!CE)
619 return nullptr;
620
621 if (CE->getOpcode() == Instruction::GetElementPtr) {
622 if (auto *GV = dyn_cast<GlobalVariable>(CE->getOperand(0))) {
623 if (GV->isConstant() && GV->hasDefinitiveInitializer()) {
624 if (Constant *V =
625 ConstantFoldLoadThroughGEPConstantExpr(GV->getInitializer(), CE))
626 return V;
627 }
628 }
629 }
630
631 if (CE->getOpcode() == Instruction::BitCast)
632 if (Constant *LoadedC = ConstantFoldLoadThroughBitcastExpr(CE, Ty, DL))
633 return LoadedC;
634
635 // Instead of loading constant c string, use corresponding integer value
636 // directly if string length is small enough.
637 StringRef Str;
638 if (getConstantStringInfo(CE, Str) && !Str.empty()) {
639 size_t StrLen = Str.size();
640 unsigned NumBits = Ty->getPrimitiveSizeInBits();
641 // Replace load with immediate integer if the result is an integer or fp
642 // value.
643 if ((NumBits >> 3) == StrLen + 1 && (NumBits & 7) == 0 &&
644 (isa<IntegerType>(Ty) || Ty->isFloatingPointTy())) {
645 APInt StrVal(NumBits, 0);
646 APInt SingleChar(NumBits, 0);
647 if (DL.isLittleEndian()) {
648 for (unsigned char C : reverse(Str.bytes())) {
649 SingleChar = static_cast<uint64_t>(C);
650 StrVal = (StrVal << 8) | SingleChar;
651 }
652 } else {
653 for (unsigned char C : Str.bytes()) {
654 SingleChar = static_cast<uint64_t>(C);
655 StrVal = (StrVal << 8) | SingleChar;
656 }
657 // Append NULL at the end.
658 SingleChar = 0;
659 StrVal = (StrVal << 8) | SingleChar;
660 }
661
662 Constant *Res = ConstantInt::get(CE->getContext(), StrVal);
663 if (Ty->isFloatingPointTy())
664 Res = ConstantExpr::getBitCast(Res, Ty);
665 return Res;
666 }
667 }
668
669 // If this load comes from anywhere in a constant global, and if the global
670 // is all undef or zero, we know what it loads.
671 if (auto *GV = dyn_cast<GlobalVariable>(GetUnderlyingObject(CE, DL))) {
672 if (GV->isConstant() && GV->hasDefinitiveInitializer()) {
673 if (GV->getInitializer()->isNullValue())
674 return Constant::getNullValue(Ty);
675 if (isa<UndefValue>(GV->getInitializer()))
676 return UndefValue::get(Ty);
677 }
678 }
679
680 // Try hard to fold loads from bitcasted strange and non-type-safe things.
681 return FoldReinterpretLoadFromConstPtr(CE, Ty, DL);
682}

695/// One of Op0/Op1 is a constant expression.
696/// Attempt to symbolically evaluate the result of a binary operator merging
697/// these together. If target data info is available, it is provided as DL,
698/// otherwise DL is null.
699Constant *SymbolicallyEvaluateBinop(unsigned Opc, Constant *Op0, Constant *Op1,
700 const DataLayout &DL) {
701 // SROA
702
703 // Fold (and 0xffffffff00000000, (shl x, 32)) -> shl.
704 // Fold (lshr (or X, Y), 32) -> (lshr [X/Y], 32) if one doesn't contribute
705 // bits.
706
707 if (Opc == Instruction::And) {
708 KnownBits Known0 = computeKnownBits(Op0, DL);
709 KnownBits Known1 = computeKnownBits(Op1, DL);
710 if ((Known1.One | Known0.Zero).isAllOnesValue()) {
711 // All the bits of Op0 that the 'and' could be masking are already zero.
712 return Op0;
713 }
714 if ((Known0.One | Known1.Zero).isAllOnesValue()) {
715 // All the bits of Op1 that the 'and' could be masking are already zero.
716 return Op1;
717 }
718
719 Known0.Zero |= Known1.Zero;
720 Known0.One &= Known1.One;
721 if (Known0.isConstant())
722 return ConstantInt::get(Op0->getType(), Known0.getConstant());
723 }
724
725 // If the constant expr is something like &A[123] - &A[4].f, fold this into a
726 // constant. This happens frequently when iterating over a global array.
727 if (Opc == Instruction::Sub) {
728 GlobalValue *GV1, *GV2;
729 APInt Offs1, Offs2;
730
731 if (IsConstantOffsetFromGlobal(Op0, GV1, Offs1, DL, false))
732 if (IsConstantOffsetFromGlobal(Op1, GV2, Offs2, DL, false) && GV1 == GV2) {
733 unsigned OpSize = DL.getTypeSizeInBits(Op0->getType());
734
735 // (&GV+C1) - (&GV+C2) -> C1-C2, pointer arithmetic cannot overflow.
736 // PtrToInt may change the bitwidth so we have convert to the right size
737 // first.
738 return ConstantInt::get(Op0->getType(), Offs1.zextOrTrunc(OpSize) -
739 Offs2.zextOrTrunc(OpSize));
740 }
741 }
742
743 return nullptr;
744}

746/// If array indices are not pointer-sized integers, explicitly cast them so
747/// that they aren't implicitly casted by the getelementptr.
748Constant *CastGEPIndices(Type *SrcElemTy, ArrayRef<Constant *> Ops,
749 Type *ResultTy, Optional<unsigned> InRangeIndex,
750 const DataLayout &DL, const TargetLibraryInfo *TLI) {
751 Type *IntPtrTy = DL.getIntPtrType(ResultTy);
752 Type *IntPtrScalarTy = IntPtrTy->getScalarType();
753
754 bool Any = false;
755 SmallVector<Constant*, 32> NewIdxs;
756 for (unsigned i = 1, e = Ops.size(); i != e; ++i) {
757 if ((i == 1 ||
758 !isa<StructType>(GetElementPtrInst::getIndexedType(
759 SrcElemTy, Ops.slice(1, i - 1)))) &&
760 Ops[i]->getType()->getScalarType() != IntPtrScalarTy) {
761 Any = true;
762 Type *NewType = Ops[i]->getType()->isVectorTy()
763 ? IntPtrTy
764 : IntPtrTy->getScalarType();
765 NewIdxs.push_back(ConstantExpr::getCast(CastInst::getCastOpcode(Ops[i],
766 true,
767 NewType,
768 true),
769 Ops[i], NewType));
770 } else
771 NewIdxs.push_back(Ops[i]);
772 }
773
774 if (!Any)
775 return nullptr;
776
777 Constant *C = ConstantExpr::getGetElementPtr(
778 SrcElemTy, Ops[0], NewIdxs, /*InBounds=*/false, InRangeIndex);
779 if (Constant *Folded = ConstantFoldConstant(C, DL, TLI))
780 C = Folded;
781
782 return C;
783}

802/// If we can symbolically evaluate the GEP constant expression, do so.
803Constant *SymbolicallyEvaluateGEP(const GEPOperator *GEP,
804 ArrayRef<Constant *> Ops,
805 const DataLayout &DL,
806 const TargetLibraryInfo *TLI) {
807 const GEPOperator *InnermostGEP = GEP;
808 bool InBounds = GEP->isInBounds();
809
810 Type *SrcElemTy = GEP->getSourceElementType();
811 Type *ResElemTy = GEP->getResultElementType();
812 Type *ResTy = GEP->getType();
813 if (!SrcElemTy->isSized())
814 return nullptr;
815
816 if (Constant *C = CastGEPIndices(SrcElemTy, Ops, ResTy,
817 GEP->getInRangeIndex(), DL, TLI))
818 return C;
819
820 Constant *Ptr = Ops[0];
821 if (!Ptr->getType()->isPointerTy())
822 return nullptr;
823
824 Type *IntPtrTy = DL.getIntPtrType(Ptr->getType());
825
826 // If this is a constant expr gep that is effectively computing an
827 // "offsetof", fold it into 'cast int Size to T*' instead of 'gep 0, 0, 12'
828 for (unsigned i = 1, e = Ops.size(); i != e; ++i)
829 if (!isa<ConstantInt>(Ops[i])) {
830
831 // If this is "gep i8* Ptr, (sub 0, V)", fold this as:
832 // "inttoptr (sub (ptrtoint Ptr), V)"
833 if (Ops.size() == 2 && ResElemTy->isIntegerTy(8)) {
834 auto *CE = dyn_cast<ConstantExpr>(Ops[1]);
835 assert((!CE || CE->getType() == IntPtrTy) &&
836 "CastGEPIndices didn't canonicalize index types!");
837 if (CE && CE->getOpcode() == Instruction::Sub &&
838 CE->getOperand(0)->isNullValue()) {
839 Constant *Res = ConstantExpr::getPtrToInt(Ptr, CE->getType());
840 Res = ConstantExpr::getSub(Res, CE->getOperand(1));
841 Res = ConstantExpr::getIntToPtr(Res, ResTy);
842 if (auto *FoldedRes = ConstantFoldConstant(Res, DL, TLI))
843 Res = FoldedRes;
844 return Res;
845 }
846 }
847 return nullptr;
848 }
849
850 unsigned BitWidth = DL.getTypeSizeInBits(IntPtrTy);
851 APInt Offset =
852 APInt(BitWidth,
853 DL.getIndexedOffsetInType(
854 SrcElemTy,
855 makeArrayRef((Value * const *)Ops.data() + 1, Ops.size() - 1)));
856 Ptr = StripPtrCastKeepAS(Ptr, SrcElemTy);
857
858 // If this is a GEP of a GEP, fold it all into a single GEP.
859 while (auto *GEP = dyn_cast<GEPOperator>(Ptr)) {
860 InnermostGEP = GEP;
861 InBounds &= GEP->isInBounds();
862
863 SmallVector<Value *, 4> NestedOps(GEP->op_begin() + 1, GEP->op_end());
864
865 // Do not try the incorporate the sub-GEP if some index is not a number.
866 bool AllConstantInt = true;
867 for (Value *NestedOp : NestedOps)
868 if (!isa<ConstantInt>(NestedOp)) {
869 AllConstantInt = false;
870 break;
871 }
872 if (!AllConstantInt)
873 break;
874
875 Ptr = cast<Constant>(GEP->getOperand(0));
876 SrcElemTy = GEP->getSourceElementType();
877 Offset += APInt(BitWidth, DL.getIndexedOffsetInType(SrcElemTy, NestedOps));
878 Ptr = StripPtrCastKeepAS(Ptr, SrcElemTy);
879 }
880
881 // If the base value for this address is a literal integer value, fold the
882 // getelementptr to the resulting integer value casted to the pointer type.
883 APInt BasePtr(BitWidth, 0);
884 if (auto *CE = dyn_cast<ConstantExpr>(Ptr)) {
885 if (CE->getOpcode() == Instruction::IntToPtr) {
886 if (auto *Base = dyn_cast<ConstantInt>(CE->getOperand(0)))
887 BasePtr = Base->getValue().zextOrTrunc(BitWidth);
888 }
889 }
890
891 auto *PTy = cast<PointerType>(Ptr->getType());
892 if ((Ptr->isNullValue() || BasePtr != 0) &&
893 !DL.isNonIntegralPointerType(PTy)) {
894 Constant *C = ConstantInt::get(Ptr->getContext(), Offset + BasePtr);
895 return ConstantExpr::getIntToPtr(C, ResTy);
896 }
897
898 // Otherwise form a regular getelementptr. Recompute the indices so that
899 // we eliminate over-indexing of the notional static type array bounds.
900 // This makes it easy to determine if the getelementptr is "inbounds".
901 // Also, this helps GlobalOpt do SROA on GlobalVariables.
902 Type *Ty = PTy;
903 SmallVector<Constant *, 32> NewIdxs;
904
905 do {
906 if (!Ty->isStructTy()) {
907 if (Ty->isPointerTy()) {
908 // The only pointer indexing we'll do is on the first index of the GEP.
909 if (!NewIdxs.empty())
910 break;
911
912 Ty = SrcElemTy;
913
914 // Only handle pointers to sized types, not pointers to functions.
915 if (!Ty->isSized())
916 return nullptr;
917 } else if (auto *ATy = dyn_cast<SequentialType>(Ty)) {
918 Ty = ATy->getElementType();
919 } else {
920 // We've reached some non-indexable type.
921 break;
922 }
923
924 // Determine which element of the array the offset points into.
925 APInt ElemSize(BitWidth, DL.getTypeAllocSize(Ty));
926 if (ElemSize == 0) {
927 // The element size is 0. This may be [0 x Ty]*, so just use a zero
928 // index for this level and proceed to the next level to see if it can
929 // accommodate the offset.
930 NewIdxs.push_back(ConstantInt::get(IntPtrTy, 0));
931 } else {
932 // The element size is non-zero divide the offset by the element
933 // size (rounding down), to compute the index at this level.
934 bool Overflow;
935 APInt NewIdx = Offset.sdiv_ov(ElemSize, Overflow);
936 if (Overflow)
937 break;
938 Offset -= NewIdx * ElemSize;
939 NewIdxs.push_back(ConstantInt::get(IntPtrTy, NewIdx));
940 }
941 } else {
942 auto *STy = cast<StructType>(Ty);
943 // If we end up with an offset that isn't valid for this struct type, we
944 // can't re-form this GEP in a regular form, so bail out. The pointer
945 // operand likely went through casts that are necessary to make the GEP
946 // sensible.
947 const StructLayout &SL = *DL.getStructLayout(STy);
948 if (Offset.isNegative() || Offset.uge(SL.getSizeInBytes()))
949 break;
950
951 // Determine which field of the struct the offset points into. The
952 // getZExtValue is fine as we've already ensured that the offset is
953 // within the range representable by the StructLayout API.
954 unsigned ElIdx = SL.getElementContainingOffset(Offset.getZExtValue());
955 NewIdxs.push_back(ConstantInt::get(Type::getInt32Ty(Ty->getContext()),
956 ElIdx));
957 Offset -= APInt(BitWidth, SL.getElementOffset(ElIdx));
958 Ty = STy->getTypeAtIndex(ElIdx);
959 }
960 } while (Ty != ResElemTy);
961
962 // If we haven't used up the entire offset by descending the static
963 // type, then the offset is pointing into the middle of an indivisible
964 // member, so we can't simplify it.
965 if (Offset != 0)
966 return nullptr;
967
968 // Preserve the inrange index from the innermost GEP if possible. We must
969 // have calculated the same indices up to and including the inrange index.
970 Optional<unsigned> InRangeIndex;
971 if (Optional<unsigned> LastIRIndex = InnermostGEP->getInRangeIndex())
972 if (SrcElemTy == InnermostGEP->getSourceElementType() &&
973 NewIdxs.size() > *LastIRIndex) {
974 InRangeIndex = LastIRIndex;
975 for (unsigned I = 0; I <= *LastIRIndex; ++I)
976 if (NewIdxs[I] != InnermostGEP->getOperand(I + 1))
977 return nullptr;
978 }
979
980 // Create a GEP.
981 Constant *C = ConstantExpr::getGetElementPtr(SrcElemTy, Ptr, NewIdxs,
982 InBounds, InRangeIndex);
983 assert(C->getType()->getPointerElementType() == Ty &&
984 "Computed GetElementPtr has unexpected type!");
985
986 // If we ended up indexing a member with a type that doesn't match
987 // the type of what the original indices indexed, add a cast.
988 if (Ty != ResElemTy)
989 C = FoldBitCast(C, ResTy, DL);
990
991 return C;
992}

994/// Attempt to constant fold an instruction with the
995/// specified opcode and operands. If successful, the constant result is
996/// returned, if not, null is returned. Note that this function can fail when
997/// attempting to fold instructions like loads and stores, which have no
998/// constant expression form.
999Constant *ConstantFoldInstOperandsImpl(const Value *InstOrCE, unsigned Opcode,
1000 ArrayRef<Constant *> Ops,
1001 const DataLayout &DL,
1002 const TargetLibraryInfo *TLI) {
1003 Type *DestTy = InstOrCE->getType();
1004
1005 if (Instruction::isUnaryOp(Opcode))
1006 return ConstantFoldUnaryOpOperand(Opcode, Ops[0], DL);
1007
1008 if (Instruction::isBinaryOp(Opcode))
1009 return ConstantFoldBinaryOpOperands(Opcode, Ops[0], Ops[1], DL);
1010
1011 if (Instruction::isCast(Opcode))
1012 return ConstantFoldCastOperand(Opcode, Ops[0], DestTy, DL);
1013
1014 if (auto *GEP = dyn_cast<GEPOperator>(InstOrCE)) {
1015 if (Constant *C = SymbolicallyEvaluateGEP(GEP, Ops, DL, TLI))
1016 return C;
1017
1018 return ConstantExpr::getGetElementPtr(GEP->getSourceElementType(), Ops[0],
1019 Ops.slice(1), GEP->isInBounds(),
1020 GEP->getInRangeIndex());
1021 }
1022
1023 if (auto *CE = dyn_cast<ConstantExpr>(InstOrCE))
1024 return CE->getWithOperands(Ops);
1025
1026 switch (Opcode) {
1027 default: return nullptr;
1028 case Instruction::ICmp:
1029 case Instruction::FCmp: llvm_unreachable("Invalid for compares");
1030 case Instruction::Call:
1031 if (auto *F = dyn_cast<Function>(Ops.back())) {
1032 const auto *Call = cast<CallBase>(InstOrCE);
1033 if (canConstantFoldCallTo(Call, F))
1034 return ConstantFoldCall(Call, F, Ops.slice(0, Ops.size() - 1), TLI);
1035 }
1036 return nullptr;
1037 case Instruction::Select:
1038 return ConstantExpr::getSelect(Ops[0], Ops[1], Ops[2]);
1039 case Instruction::ExtractElement:
1040 return ConstantExpr::getExtractElement(Ops[0], Ops[1]);
1041 case Instruction::InsertElement:
1042 return ConstantExpr::getInsertElement(Ops[0], Ops[1], Ops[2]);
1043 case Instruction::ShuffleVector:
1044 return ConstantExpr::getShuffleVector(Ops[0], Ops[1], Ops[2]);
1045 }
1046}

1056Constant *
1057ConstantFoldConstantImpl(const Constant *C, const DataLayout &DL,
1058 const TargetLibraryInfo *TLI,
1059 SmallDenseMap<Constant *, Constant *> &FoldedOps) {
1060 if (!isa<ConstantVector>(C) && !isa<ConstantExpr>(C))
1061 return nullptr;
1062
1063 SmallVector<Constant *, 8> Ops;
1064 for (const Use &NewU : C->operands()) {
1065 auto *NewC = cast<Constant>(&NewU);
1066 // Recursively fold the ConstantExpr's operands. If we have already folded
1067 // a ConstantExpr, we don't have to process it again.
1068 if (isa<ConstantVector>(NewC) || isa<ConstantExpr>(NewC)) {
1069 auto It = FoldedOps.find(NewC);
1070 if (It == FoldedOps.end()) {
1071 if (auto *FoldedC =
1072 ConstantFoldConstantImpl(NewC, DL, TLI, FoldedOps)) {
1073 FoldedOps.insert({NewC, FoldedC});
1074 NewC = FoldedC;
1075 } else {
1076 FoldedOps.insert({NewC, NewC});
1077 }
1078 } else {
1079 NewC = It->second;
1080 }
1081 }
1082 Ops.push_back(NewC);
1083 }
1084
1085 if (auto *CE = dyn_cast<ConstantExpr>(C)) {
1086 if (CE->isCompare())
1087 return ConstantFoldCompareInstOperands(CE->getPredicate(), Ops[0], Ops[1],
1088 DL, TLI);
1089
1090 return ConstantFoldInstOperandsImpl(CE, CE->getOpcode(), Ops, DL, TLI);
1091 }
1092
1093 assert(isa<ConstantVector>(C));
1094 return ConstantVector::get(Ops);
1095}

1099Constant *llvm::ConstantFoldInstruction(Instruction *I, const DataLayout &DL,
1100 const TargetLibraryInfo *TLI) {
1101 // Handle PHI nodes quickly here...
1102 if (auto *PN = dyn_cast<PHINode>(I)) {
1103 Constant *CommonValue = nullptr;
1104
1105 SmallDenseMap<Constant *, Constant *> FoldedOps;
1106 for (Value *Incoming : PN->incoming_values()) {
1107 // If the incoming value is undef then skip it. Note that while we could
1108 // skip the value if it is equal to the phi node itself we choose not to
1109 // because that would break the rule that constant folding only applies if
1110 // all operands are constants.
1111 if (isa<UndefValue>(Incoming))
1112 continue;
1113 // If the incoming value is not a constant, then give up.
1114 auto *C = dyn_cast<Constant>(Incoming);
1115 if (!C)
1116 return nullptr;
1117 // Fold the PHI's operands.
1118 if (auto *FoldedC = ConstantFoldConstantImpl(C, DL, TLI, FoldedOps))
1119 C = FoldedC;
1120 // If the incoming value is a different constant to
1121 // the one we saw previously, then give up.
1122 if (CommonValue && C != CommonValue)
1123 return nullptr;
1124 CommonValue = C;
1125 }
1126
1127 // If we reach here, all incoming values are the same constant or undef.
1128 return CommonValue ? CommonValue : UndefValue::get(PN->getType());
1129 }
1130
1131 // Scan the operand list, checking to see if they are all constants, if so,
1132 // hand off to ConstantFoldInstOperandsImpl.
1133 if (!all_of(I->operands(), [](Use &U) { return isa<Constant>(U); }))
1134 return nullptr;
1135
1136 SmallDenseMap<Constant *, Constant *> FoldedOps;
1137 SmallVector<Constant *, 8> Ops;
1138 for (const Use &OpU : I->operands()) {
1139 auto *Op = cast<Constant>(&OpU);
1140 // Fold the Instruction's operands.
1141 if (auto *FoldedOp = ConstantFoldConstantImpl(Op, DL, TLI, FoldedOps))
1142 Op = FoldedOp;
1143
1144 Ops.push_back(Op);
1145 }
1146
1147 if (const auto *CI = dyn_cast<CmpInst>(I))
1148 return ConstantFoldCompareInstOperands(CI->getPredicate(), Ops[0], Ops[1],
1149 DL, TLI);
1150
1151 if (const auto *LI = dyn_cast<LoadInst>(I))
1152 return ConstantFoldLoadInst(LI, DL);
1153
1154 if (auto *IVI = dyn_cast<InsertValueInst>(I)) {
1155 return ConstantExpr::getInsertValue(
1156 cast<Constant>(IVI->getAggregateOperand()),
1157 cast<Constant>(IVI->getInsertedValueOperand()),
1158 IVI->getIndices());
1159 }
1160
1161 if (auto *EVI = dyn_cast<ExtractValueInst>(I)) {
1162 return ConstantExpr::getExtractValue(
1163 cast<Constant>(EVI->getAggregateOperand()),
1164 EVI->getIndices());
1165 }
1166
1167 return ConstantFoldInstOperands(I, Ops, DL, TLI);
1168}

1183Constant *llvm::ConstantFoldCompareInstOperands(unsigned Predicate,
1184 Constant *Ops0, Constant *Ops1,
1185 const DataLayout &DL,
1186 const TargetLibraryInfo *TLI) {
1187 // fold: icmp (inttoptr x), null -> icmp x, 0
1188 // fold: icmp null, (inttoptr x) -> icmp 0, x
1189 // fold: icmp (ptrtoint x), 0 -> icmp x, null
1190 // fold: icmp 0, (ptrtoint x) -> icmp null, x
1191 // fold: icmp (inttoptr x), (inttoptr y) -> icmp trunc/zext x, trunc/zext y
1192 // fold: icmp (ptrtoint x), (ptrtoint y) -> icmp x, y
1193 //
1194 // FIXME: The following comment is out of data and the DataLayout is here now.
1195 // ConstantExpr::getCompare cannot do this, because it doesn't have DL
1196 // around to know if bit truncation is happening.
1197 if (auto *CE0 = dyn_cast<ConstantExpr>(Ops0)) {
1198 if (Ops1->isNullValue()) {
1199 if (CE0->getOpcode() == Instruction::IntToPtr) {
1200 Type *IntPtrTy = DL.getIntPtrType(CE0->getType());
1201 // Convert the integer value to the right size to ensure we get the
1202 // proper extension or truncation.
1203 Constant *C = ConstantExpr::getIntegerCast(CE0->getOperand(0),
1204 IntPtrTy, false);
1205 Constant *Null = Constant::getNullValue(C->getType());
1206 return ConstantFoldCompareInstOperands(Predicate, C, Null, DL, TLI);
1207 }
1208
1209 // Only do this transformation if the int is intptrty in size, otherwise
1210 // there is a truncation or extension that we aren't modeling.
1211 if (CE0->getOpcode() == Instruction::PtrToInt) {
1212 Type *IntPtrTy = DL.getIntPtrType(CE0->getOperand(0)->getType());
1213 if (CE0->getType() == IntPtrTy) {
1214 Constant *C = CE0->getOperand(0);
1215 Constant *Null = Constant::getNullValue(C->getType());
1216 return ConstantFoldCompareInstOperands(Predicate, C, Null, DL, TLI);
1217 }
1218 }
1219 }
1220
1221 if (auto *CE1 = dyn_cast<ConstantExpr>(Ops1)) {
1222 if (CE0->getOpcode() == CE1->getOpcode()) {
1223 if (CE0->getOpcode() == Instruction::IntToPtr) {
1224 Type *IntPtrTy = DL.getIntPtrType(CE0->getType());
1225
1226 // Convert the integer value to the right size to ensure we get the
1227 // proper extension or truncation.
1228 Constant *C0 = ConstantExpr::getIntegerCast(CE0->getOperand(0),
1229 IntPtrTy, false);
1230 Constant *C1 = ConstantExpr::getIntegerCast(CE1->getOperand(0),
1231 IntPtrTy, false);
1232 return ConstantFoldCompareInstOperands(Predicate, C0, C1, DL, TLI);
1233 }
1234
1235 // Only do this transformation if the int is intptrty in size, otherwise
1236 // there is a truncation or extension that we aren't modeling.
1237 if (CE0->getOpcode() == Instruction::PtrToInt) {
1238 Type *IntPtrTy = DL.getIntPtrType(CE0->getOperand(0)->getType());
1239 if (CE0->getType() == IntPtrTy &&
1240 CE0->getOperand(0)->getType() == CE1->getOperand(0)->getType()) {
1241 return ConstantFoldCompareInstOperands(
1242 Predicate, CE0->getOperand(0), CE1->getOperand(0), DL, TLI);
1243 }
1244 }
1245 }
1246 }
1247
1248 // icmp eq (or x, y), 0 -> (icmp eq x, 0) & (icmp eq y, 0)
1249 // icmp ne (or x, y), 0 -> (icmp ne x, 0) | (icmp ne y, 0)
1250 if ((Predicate == ICmpInst::ICMP_EQ || Predicate == ICmpInst::ICMP_NE) &&
1251 CE0->getOpcode() == Instruction::Or && Ops1->isNullValue()) {
1252 Constant *LHS = ConstantFoldCompareInstOperands(
1253 Predicate, CE0->getOperand(0), Ops1, DL, TLI);
1254 Constant *RHS = ConstantFoldCompareInstOperands(
1255 Predicate, CE0->getOperand(1), Ops1, DL, TLI);
1256 unsigned OpC =
1257 Predicate == ICmpInst::ICMP_EQ ? Instruction::And : Instruction::Or;
1258 return ConstantFoldBinaryOpOperands(OpC, LHS, RHS, DL);
1259 }
1260 } else if (isa<ConstantExpr>(Ops1)) {
1261 // If RHS is a constant expression, but the left side isn't, swap the
1262 // operands and try again.
1263 Predicate = ICmpInst::getSwappedPredicate((ICmpInst::Predicate)Predicate);
1264 return ConstantFoldCompareInstOperands(Predicate, Ops1, Ops0, DL, TLI);
1265 }
1266
1267 return ConstantExpr::getCompare(Predicate, Ops0, Ops1);
1268}

1288Constant *llvm::ConstantFoldCastOperand(unsigned Opcode, Constant *C,
1289 Type *DestTy, const DataLayout &DL) {
1290 assert(Instruction::isCast(Opcode));
1291 switch (Opcode) {
1292 default:
1293 llvm_unreachable("Missing case");
1294 case Instruction::PtrToInt:
1295 // If the input is a inttoptr, eliminate the pair. This requires knowing
1296 // the width of a pointer, so it can't be done in ConstantExpr::getCast.
1297 if (auto *CE = dyn_cast<ConstantExpr>(C)) {
1298 if (CE->getOpcode() == Instruction::IntToPtr) {
1299 Constant *Input = CE->getOperand(0);
1300 unsigned InWidth = Input->getType()->getScalarSizeInBits();
1301 unsigned PtrWidth = DL.getPointerTypeSizeInBits(CE->getType());
1302 if (PtrWidth < InWidth) {
1303 Constant *Mask =
1304 ConstantInt::get(CE->getContext(),
1305 APInt::getLowBitsSet(InWidth, PtrWidth));
1306 Input = ConstantExpr::getAnd(Input, Mask);
1307 }
1308 // Do a zext or trunc to get to the dest size.
1309 return ConstantExpr::getIntegerCast(Input, DestTy, false);
1310 }
1311 }
1312 return ConstantExpr::getCast(Opcode, C, DestTy);
1313 case Instruction::IntToPtr:
1314 // If the input is a ptrtoint, turn the pair into a ptr to ptr bitcast if
1315 // the int size is >= the ptr size and the address spaces are the same.
1316 // This requires knowing the width of a pointer, so it can't be done in
1317 // ConstantExpr::getCast.
1318 if (auto *CE = dyn_cast<ConstantExpr>(C)) {
1319 if (CE->getOpcode() == Instruction::PtrToInt) {
1320 Constant *SrcPtr = CE->getOperand(0);
1321 unsigned SrcPtrSize = DL.getPointerTypeSizeInBits(SrcPtr->getType());
1322 unsigned MidIntSize = CE->getType()->getScalarSizeInBits();
1323
1324 if (MidIntSize >= SrcPtrSize) {
1325 unsigned SrcAS = SrcPtr->getType()->getPointerAddressSpace();
1326 if (SrcAS == DestTy->getPointerAddressSpace())
1327 return FoldBitCast(CE->getOperand(0), DestTy, DL);
1328 }
1329 }
1330 }
1331
1332 return ConstantExpr::getCast(Opcode, C, DestTy);
1333 case Instruction::Trunc:
1334 case Instruction::ZExt:
1335 case Instruction::SExt:
1336 case Instruction::FPTrunc:
1337 case Instruction::FPExt:
1338 case Instruction::UIToFP:
1339 case Instruction::SIToFP:
1340 case Instruction::FPToUI:
1341 case Instruction::FPToSI:
1342 case Instruction::AddrSpaceCast:
1343 return ConstantExpr::getCast(Opcode, C, DestTy);
1344 case Instruction::BitCast:
1345 return FoldBitCast(C, DestTy, DL);
1346 }
1347}

1653static Constant *ConstantFoldScalarCall(StringRef Name, unsigned IntrinsicID,
1654 Type *Ty,
1655 ArrayRef<Constant *> Operands,
1656 const TargetLibraryInfo *TLI,
1657 const CallBase *Call) {
1658 if (Operands.size() == 1) {
1659 if (IntrinsicID == Intrinsic::is_constant) {
1660 // We know we have a "Constant" argument. But we want to only
1661 // return true for manifest constants, not those that depend on
1662 // constants with unknowable values, e.g. GlobalValue or BlockAddress.
1663 if (isManifestConstant(Operands[0]))
1664 return ConstantInt::getTrue(Ty->getContext());
1665 return nullptr;
1666 }
1667 if (isa<UndefValue>(Operands[0])) {
1668 // cosine(arg) is between -1 and 1. cosine(invalid arg) is NaN.
1669 // ctpop() is between 0 and bitwidth, pick 0 for undef.
1670 if (IntrinsicID == Intrinsic::cos ||
1671 IntrinsicID == Intrinsic::ctpop)
1672 return Constant::getNullValue(Ty);
1673 if (IntrinsicID == Intrinsic::bswap ||
1674 IntrinsicID == Intrinsic::bitreverse ||
1675 IntrinsicID == Intrinsic::launder_invariant_group ||
1676 IntrinsicID == Intrinsic::strip_invariant_group)
1677 return Operands[0];
1678 }
1679
1680 if (isa<ConstantPointerNull>(Operands[0])) {
1681 // launder(null) == null == strip(null) iff in addrspace 0
1682 if (IntrinsicID == Intrinsic::launder_invariant_group ||
1683 IntrinsicID == Intrinsic::strip_invariant_group) {
1684 // If instruction is not yet put in a basic block (e.g. when cloning
1685 // a function during inlining), Call's caller may not be available.
1686 // So check Call's BB first before querying Call->getCaller.
1687 const Function *Caller =
1688 Call->getParent() ? Call->getCaller() : nullptr;
1689 if (Caller &&
1690 !NullPointerIsDefined(
1691 Caller, Operands[0]->getType()->getPointerAddressSpace())) {
1692 return Operands[0];
1693 }
1694 return nullptr;
1695 }
1696 }
1697
1698 if (auto *Op = dyn_cast<ConstantFP>(Operands[0])) {
1699 if (IntrinsicID == Intrinsic::convert_to_fp16) {
1700 APFloat Val(Op->getValueAPF());
1701
1702 bool lost = false;
1703 Val.convert(APFloat::IEEEhalf(), APFloat::rmNearestTiesToEven, &lost);
1704
1705 return ConstantInt::get(Ty->getContext(), Val.bitcastToAPInt());
1706 }
1707
1708 if (!Ty->isHalfTy() && !Ty->isFloatTy() && !Ty->isDoubleTy())
1709 return nullptr;
1710
1711 if (IntrinsicID == Intrinsic::round) {
1712 APFloat V = Op->getValueAPF();
1713 V.roundToIntegral(APFloat::rmNearestTiesToAway);
1714 return ConstantFP::get(Ty->getContext(), V);
1715 }
1716
1717 if (IntrinsicID == Intrinsic::floor) {
1718 APFloat V = Op->getValueAPF();
1719 V.roundToIntegral(APFloat::rmTowardNegative);
1720 return ConstantFP::get(Ty->getContext(), V);
1721 }
1722
1723 if (IntrinsicID == Intrinsic::ceil) {
1724 APFloat V = Op->getValueAPF();
1725 V.roundToIntegral(APFloat::rmTowardPositive);
1726 return ConstantFP::get(Ty->getContext(), V);
1727 }
1728
1729 if (IntrinsicID == Intrinsic::trunc) {
1730 APFloat V = Op->getValueAPF();
1731 V.roundToIntegral(APFloat::rmTowardZero);
1732 return ConstantFP::get(Ty->getContext(), V);
1733 }
1734
1735 if (IntrinsicID == Intrinsic::rint) {
1736 APFloat V = Op->getValueAPF();
1737 V.roundToIntegral(APFloat::rmNearestTiesToEven);
1738 return ConstantFP::get(Ty->getContext(), V);
1739 }
1740
1741 if (IntrinsicID == Intrinsic::nearbyint) {
1742 APFloat V = Op->getValueAPF();
1743 V.roundToIntegral(APFloat::rmNearestTiesToEven);
1744 return ConstantFP::get(Ty->getContext(), V);
1745 }
1746
1747 /// We only fold functions with finite arguments. Folding NaN and inf is
1748 /// likely to be aborted with an exception anyway, and some host libms
1749 /// have known errors raising exceptions.
1750 if (Op->getValueAPF().isNaN() || Op->getValueAPF().isInfinity())
1751 return nullptr;
1752
1753 /// Currently APFloat versions of these functions do not exist, so we use
1754 /// the host native double versions. Float versions are not called
1755 /// directly but for all these it is true (float)(f((double)arg)) ==
1756 /// f(arg). Long double not supported yet.
1757 double V = getValueAsDouble(Op);
1758
1759 switch (IntrinsicID) {
1760 default: break;
1761 case Intrinsic::fabs:
1762 return ConstantFoldFP(fabs, V, Ty);
1763 case Intrinsic::log2:
1764 return ConstantFoldFP(Log2, V, Ty);
1765 case Intrinsic::log:
1766 return ConstantFoldFP(log, V, Ty);
1767 case Intrinsic::log10:
1768 return ConstantFoldFP(log10, V, Ty);
1769 case Intrinsic::exp:
1770 return ConstantFoldFP(exp, V, Ty);
1771 case Intrinsic::exp2:
1772 return ConstantFoldFP(exp2, V, Ty);
1773 case Intrinsic::sin:
1774 return ConstantFoldFP(sin, V, Ty);
1775 case Intrinsic::cos:
1776 return ConstantFoldFP(cos, V, Ty);
1777 case Intrinsic::sqrt:
1778 return ConstantFoldFP(sqrt, V, Ty);
1779 }
1780
1781 if (!TLI)
1782 return nullptr;
1783
1784 char NameKeyChar = Name[0];
1785 if (Name[0] == '_' && Name.size() > 2 && Name[1] == '_')
1786 NameKeyChar = Name[2];
1787
1788 switch (NameKeyChar) {
1789 case 'a':
1790 if ((Name == "acos" && TLI->has(LibFunc_acos)) ||
1791 (Name == "acosf" && TLI->has(LibFunc_acosf)) ||
1792 (Name == "__acos_finite" && TLI->has(LibFunc_acos_finite)) ||
1793 (Name == "__acosf_finite" && TLI->has(LibFunc_acosf_finite)))
1794 return ConstantFoldFP(acos, V, Ty);
1795 else if ((Name == "asin" && TLI->has(LibFunc_asin)) ||
1796 (Name == "asinf" && TLI->has(LibFunc_asinf)) ||
1797 (Name == "__asin_finite" && TLI->has(LibFunc_asin_finite)) ||
1798 (Name == "__asinf_finite" && TLI->has(LibFunc_asinf_finite)))
1799 return ConstantFoldFP(asin, V, Ty);
1800 else if ((Name == "atan" && TLI->has(LibFunc_atan)) ||
1801 (Name == "atanf" && TLI->has(LibFunc_atanf)))
1802 return ConstantFoldFP(atan, V, Ty);
1803 break;
1804 case 'c':
1805 if ((Name == "ceil" && TLI->has(LibFunc_ceil)) ||
1806 (Name == "ceilf" && TLI->has(LibFunc_ceilf)))
1807 return ConstantFoldFP(ceil, V, Ty);
1808 else if ((Name == "cos" && TLI->has(LibFunc_cos)) ||
1809 (Name == "cosf" && TLI->has(LibFunc_cosf)))
1810 return ConstantFoldFP(cos, V, Ty);
1811 else if ((Name == "cosh" && TLI->has(LibFunc_cosh)) ||
1812 (Name == "coshf" && TLI->has(LibFunc_coshf)) ||
1813 (Name == "__cosh_finite" && TLI->has(LibFunc_cosh_finite)) ||
1814 (Name == "__coshf_finite" && TLI->has(LibFunc_coshf_finite)))
1815 return ConstantFoldFP(cosh, V, Ty);
1816 break;
1817 case 'e':
1818 if ((Name == "exp" && TLI->has(LibFunc_exp)) ||
1819 (Name == "expf" && TLI->has(LibFunc_expf)) ||
1820 (Name == "__exp_finite" && TLI->has(LibFunc_exp_finite)) ||
1821 (Name == "__expf_finite" && TLI->has(LibFunc_expf_finite)))

2392 if (Ty->isDoubleTy() || Ty->isFloatTy() || Ty->isHalfTy()) {
2393 double OpV = getValueAsDouble(OpC);
2394 return ConstantFoldFP(tan, OpV, Ty) != nullptr;
2395 }
2396 break;
2397 }
2398
2399 case LibFunc_asinl:
2400 case LibFunc_asin:
2401 case LibFunc_asinf:
2402 case LibFunc_acosl:
2403 case LibFunc_acos:
2404 case LibFunc_acosf:
2405 return Op.compare(APFloat(Op.getSemantics(), "-1")) !=
2406 APFloat::cmpLessThan &&
2407 Op.compare(APFloat(Op.getSemantics(), "1")) !=
2408 APFloat::cmpGreaterThan;
2409
2410 case LibFunc_sinh:
2411 case LibFunc_cosh:
2412 case LibFunc_sinhf:
2413 case LibFunc_coshf:
2414 case LibFunc_sinhl:
2415 case LibFunc_coshl:
2416 // FIXME: These boundaries are slightly conservative.
2417 if (OpC->getType()->isDoubleTy())
2418 return Op.compare(APFloat(-710.0)) != APFloat::cmpLessThan &&
2419 Op.compare(APFloat(710.0)) != APFloat::cmpGreaterThan;
2420 if (OpC->getType()->isFloatTy())
2421 return Op.compare(APFloat(-89.0f)) != APFloat::cmpLessThan &&
2422 Op.compare(APFloat(89.0f)) != APFloat::cmpGreaterThan;
2423 break;
2424
2425 case LibFunc_sqrtl:
2426 case LibFunc_sqrt:
2427 case LibFunc_sqrtf:
2428 return Op.isNaN() || Op.isZero() || !Op.isNegative();
2429
2430 // FIXME: Add more functions: sqrt_finite, atanh, expm1, log1p,
2431 // maybe others?
2432 default:
2433 break;
2434 }
2435 }
2436 }
2437
2438 if (Call->getNumArgOperands() == 2) {
2439 ConstantFP *Op0C = dyn_cast<ConstantFP>(Call->getArgOperand(0));
2440 ConstantFP *Op1C = dyn_cast<ConstantFP>(Call->getArgOperand(1));
2441 if (Op0C && Op1C) {
2442 const APFloat &Op0 = Op0C->getValueAPF();
2443 const APFloat &Op1 = Op1C->getValueAPF();
2444
2445 switch (Func) {
2446 case LibFunc_powl:
2447 case LibFunc_pow:
2448 case LibFunc_powf: {
2449 // FIXME: Stop using the host math library.
2450 // FIXME: The computation isn't done in the right precision.
2451 Type *Ty = Op0C->getType();
2452 if (Ty->isDoubleTy() || Ty->isFloatTy() || Ty->isHalfTy()) {
2453 if (Ty == Op1C->getType()) {
2454 double Op0V = getValueAsDouble(Op0C);
2455 double Op1V = getValueAsDouble(Op1C);
2456 return ConstantFoldBinaryFP(pow, Op0V, Op1V, Ty) != nullptr;
2457 }
2458 }
2459 break;
2460 }
2461
2462 case LibFunc_fmodl:
2463 case LibFunc_fmod:
2464 case LibFunc_fmodf:
2465 return Op0.isNaN() || Op1.isNaN() ||
2466 (!Op0.isInfinity() && !Op1.isZero());
2467
2468 default:
2469 break;
2470 }
2471 }
2472 }

2474 return false;
