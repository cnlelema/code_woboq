24static bool useCompactUnwind(const Triple &T) {
25 // Only on darwin.
26 if (!T.isOSDarwin())
27 return false;
28
29 // aarch64 always has it.
30 if (T.getArch() == Triple::aarch64)
31 return true;
32
33 // armv7k always has it.
34 if (T.isWatchABI())
35 return true;
36
37 // Use it on newer version of OS X.
38 if (T.isMacOSX() && !T.isMacOSXVersionLT(10, 6))
39 return true;
40
41 // And the iOS simulator.
42 if (T.isiOS() &&
43 (T.getArch() == Triple::x86_64 || T.getArch() == Triple::x86))
44 return true;
45
46 return false;
47}

299void MCObjectFileInfo::initELFMCObjectFileInfo(const Triple &T, bool Large) {
300 switch (T.getArch()) {
301 case Triple::mips:
302 case Triple::mipsel:
303 case Triple::mips64:
304 case Triple::mips64el:
305 case Triple::cheri:
306 FDECFIEncoding = Ctx->getAsmInfo()->getCodePointerSize() == 4
307 ? dwarf::DW_EH_PE_sdata4
308 : dwarf::DW_EH_PE_sdata8;
309 break;
310 case Triple::ppc64:
311 case Triple::ppc64le:
312 case Triple::x86_64:
313 FDECFIEncoding = dwarf::DW_EH_PE_pcrel |
314 (Large ? dwarf::DW_EH_PE_sdata8 : dwarf::DW_EH_PE_sdata4);
315 break;
316 case Triple::bpfel:
317 case Triple::bpfeb:
318 FDECFIEncoding = dwarf::DW_EH_PE_sdata8;
319 break;
320 case Triple::hexagon:
321 FDECFIEncoding =
322 PositionIndependent ? dwarf::DW_EH_PE_pcrel : dwarf::DW_EH_PE_absptr;
323 break;
324 default:
325 FDECFIEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
326 break;
327 }
328
329 unsigned EHSectionType = T.getArch() == Triple::x86_64
330 ? ELF::SHT_X86_64_UNWIND
331 : ELF::SHT_PROGBITS;
332
333 // Solaris requires different flags for .eh_frame to seemingly every other
334 // platform.
335 unsigned EHSectionFlags = ELF::SHF_ALLOC;
336 if (T.isOSSolaris() && T.getArch() != Triple::x86_64)
337 EHSectionFlags |= ELF::SHF_WRITE;
338
339 // This is (currently) also true for MIPS
340 // TODO: add a T.isMIPS()?
341 switch (T.getArch()) {
342 case Triple::mips:
343 case Triple::mipsel:
344 case Triple::mips64:
345 case Triple::mips64el:
346 case Triple::cheri:
347 if (PositionIndependent) {
348 EHSectionFlags |= ELF::SHF_WRITE;
349 }
350 break;
351 default:
352 break;
353 }
354
355 // ELF
356 BSSSection = Ctx->getELFSection(".bss", ELF::SHT_NOBITS,
357 ELF::SHF_WRITE | ELF::SHF_ALLOC);
358
359 TextSection = Ctx->getELFSection(".text", ELF::SHT_PROGBITS,
360 ELF::SHF_EXECINSTR | ELF::SHF_ALLOC);
361
362 DataSection = Ctx->getELFSection(".data", ELF::SHT_PROGBITS,
363 ELF::SHF_WRITE | ELF::SHF_ALLOC);
364
365 ReadOnlySection =
366 Ctx->getELFSection(".rodata", ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
367
368 TLSDataSection =
369 Ctx->getELFSection(".tdata", ELF::SHT_PROGBITS,
370 ELF::SHF_ALLOC | ELF::SHF_TLS | ELF::SHF_WRITE);
371
372 TLSBSSSection = Ctx->getELFSection(
373 ".tbss", ELF::SHT_NOBITS, ELF::SHF_ALLOC | ELF::SHF_TLS | ELF::SHF_WRITE);
374
375 DataRelROSection = Ctx->getELFSection(".data.rel.ro", ELF::SHT_PROGBITS,
376 ELF::SHF_ALLOC | ELF::SHF_WRITE);
377
378 MergeableConst4Section =
379 Ctx->getELFSection(".rodata.cst4", ELF::SHT_PROGBITS,
380 ELF::SHF_ALLOC | ELF::SHF_MERGE, 4, "");
381
382 MergeableConst8Section =
383 Ctx->getELFSection(".rodata.cst8", ELF::SHT_PROGBITS,
384 ELF::SHF_ALLOC | ELF::SHF_MERGE, 8, "");
385
386 MergeableConst16Section =
387 Ctx->getELFSection(".rodata.cst16", ELF::SHT_PROGBITS,
388 ELF::SHF_ALLOC | ELF::SHF_MERGE, 16, "");
389
390 MergeableConst32Section =
391 Ctx->getELFSection(".rodata.cst32", ELF::SHT_PROGBITS,
392 ELF::SHF_ALLOC | ELF::SHF_MERGE, 32, "");
393
394 // Exception Handling Sections.
395
396 // FIXME: We're emitting LSDA info into a readonly section on ELF, even though
397 // it contains relocatable pointers. In PIC mode, this is probably a big
398 // runtime hit for C++ apps. Either the contents of the LSDA need to be
399 // adjusted or this should be a data section.
400 LSDASection = Ctx->getELFSection(".gcc_except_table", ELF::SHT_PROGBITS,
401 ELF::SHF_ALLOC);
402
403 COFFDebugSymbolsSection = nullptr;
404 COFFDebugTypesSection = nullptr;
405
406 unsigned DebugSecType = ELF::SHT_PROGBITS;
407
408 // MIPS .debug_* sections should have SHT_MIPS_DWARF section type
409 // to distinguish among sections contain DWARF and ECOFF debug formats.
410 // Sections with ECOFF debug format are obsoleted and marked by SHT_PROGBITS.
411 if (T.isMIPS())
412 DebugSecType = ELF::SHT_MIPS_DWARF;
413
414 // Debug Info Sections.
415 DwarfAbbrevSection =
416 Ctx->getELFSection(".debug_abbrev", DebugSecType, 0);
417 DwarfInfoSection = Ctx->getELFSection(".debug_info", DebugSecType, 0);
418 DwarfLineSection = Ctx->getELFSection(".debug_line", DebugSecType, 0);
419 DwarfLineStrSection =
420 Ctx->getELFSection(".debug_line_str", DebugSecType,
421 ELF::SHF_MERGE | ELF::SHF_STRINGS, 1, "");
422 DwarfFrameSection = Ctx->getELFSection(".debug_frame", DebugSecType, 0);
423 DwarfPubNamesSection =
424 Ctx->getELFSection(".debug_pubnames", DebugSecType, 0);
425 DwarfPubTypesSection =
426 Ctx->getELFSection(".debug_pubtypes", DebugSecType, 0);
427 DwarfGnuPubNamesSection =
428 Ctx->getELFSection(".debug_gnu_pubnames", DebugSecType, 0);
429 DwarfGnuPubTypesSection =
430 Ctx->getELFSection(".debug_gnu_pubtypes", DebugSecType, 0);
431 DwarfStrSection =
432 Ctx->getELFSection(".debug_str", DebugSecType,
433 ELF::SHF_MERGE | ELF::SHF_STRINGS, 1, "");
434 DwarfLocSection = Ctx->getELFSection(".debug_loc", DebugSecType, 0);
435 DwarfARangesSection =
436 Ctx->getELFSection(".debug_aranges", DebugSecType, 0);
437 DwarfRangesSection =
438 Ctx->getELFSection(".debug_ranges", DebugSecType, 0);
439 DwarfMacinfoSection =
440 Ctx->getELFSection(".debug_macinfo", DebugSecType, 0);
441
442 // DWARF5 Experimental Debug Info
443
444 // Accelerator Tables
445 DwarfDebugNamesSection =
446 Ctx->getELFSection(".debug_names", ELF::SHT_PROGBITS, 0);
447 DwarfAccelNamesSection =
448 Ctx->getELFSection(".apple_names", ELF::SHT_PROGBITS, 0);
449 DwarfAccelObjCSection =
450 Ctx->getELFSection(".apple_objc", ELF::SHT_PROGBITS, 0);
451 DwarfAccelNamespaceSection =
452 Ctx->getELFSection(".apple_namespaces", ELF::SHT_PROGBITS, 0);
453 DwarfAccelTypesSection =
454 Ctx->getELFSection(".apple_types", ELF::SHT_PROGBITS, 0);
455
456 // String Offset and Address Sections
457 DwarfStrOffSection =
458 Ctx->getELFSection(".debug_str_offsets", DebugSecType, 0);
459 DwarfAddrSection = Ctx->getELFSection(".debug_addr", DebugSecType, 0);
460 DwarfRnglistsSection = Ctx->getELFSection(".debug_rnglists", DebugSecType, 0);
461 DwarfLoclistsSection = Ctx->getELFSection(".debug_loclists", DebugSecType, 0);
462
463 // Fission Sections
464 DwarfInfoDWOSection =
465 Ctx->getELFSection(".debug_info.dwo", DebugSecType, ELF::SHF_EXCLUDE);
466 DwarfTypesDWOSection =
467 Ctx->getELFSection(".debug_types.dwo", DebugSecType, ELF::SHF_EXCLUDE);
468 DwarfAbbrevDWOSection =
469 Ctx->getELFSection(".debug_abbrev.dwo", DebugSecType, ELF::SHF_EXCLUDE);
470 DwarfStrDWOSection = Ctx->getELFSection(
471 ".debug_str.dwo", DebugSecType,
472 ELF::SHF_MERGE | ELF::SHF_STRINGS | ELF::SHF_EXCLUDE, 1, "");
473 DwarfLineDWOSection =
474 Ctx->getELFSection(".debug_line.dwo", DebugSecType, ELF::SHF_EXCLUDE);
475 DwarfLocDWOSection =
476 Ctx->getELFSection(".debug_loc.dwo", DebugSecType, ELF::SHF_EXCLUDE);
477 DwarfStrOffDWOSection = Ctx->getELFSection(".debug_str_offsets.dwo",
478 DebugSecType, ELF::SHF_EXCLUDE);
479 DwarfRnglistsDWOSection =
480 Ctx->getELFSection(".debug_rnglists.dwo", DebugSecType, ELF::SHF_EXCLUDE);
481
482 // DWP Sections
483 DwarfCUIndexSection =
484 Ctx->getELFSection(".debug_cu_index", DebugSecType, 0);
485 DwarfTUIndexSection =
486 Ctx->getELFSection(".debug_tu_index", DebugSecType, 0);
487
488 StackMapSection =
489 Ctx->getELFSection(".llvm_stackmaps", ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
490
491 FaultMapSection =
492 Ctx->getELFSection(".llvm_faultmaps", ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
493
494 EHFrameSection =
495 Ctx->getELFSection(".eh_frame", EHSectionType, EHSectionFlags);
496
497 StackSizesSection = Ctx->getELFSection(".stack_sizes", ELF::SHT_PROGBITS, 0);
498
499 RemarksSection =
500 Ctx->getELFSection(".remarks", ELF::SHT_PROGBITS, ELF::SHF_EXCLUDE);
501}

503void MCObjectFileInfo::initCOFFMCObjectFileInfo(const Triple &T) {
504 EHFrameSection =
505 Ctx->getCOFFSection(".eh_frame", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
506 COFF::IMAGE_SCN_MEM_READ,
507 SectionKind::getData());
508
509 // Set the `IMAGE_SCN_MEM_16BIT` flag when compiling for thumb mode. This is
510 // used to indicate to the linker that the text segment contains thumb instructions
511 // and to set the ISA selection bit for calls accordingly.
512 const bool IsThumb = T.getArch() == Triple::thumb;
513
514 CommDirectiveSupportsAlignment = true;
515
516 // COFF
517 BSSSection = Ctx->getCOFFSection(
518 ".bss", COFF::IMAGE_SCN_CNT_UNINITIALIZED_DATA |
519 COFF::IMAGE_SCN_MEM_READ | COFF::IMAGE_SCN_MEM_WRITE,
520 SectionKind::getBSS());
521 TextSection = Ctx->getCOFFSection(
522 ".text",
523 (IsThumb ? COFF::IMAGE_SCN_MEM_16BIT : (COFF::SectionCharacteristics)0) |
524 COFF::IMAGE_SCN_CNT_CODE | COFF::IMAGE_SCN_MEM_EXECUTE |
525 COFF::IMAGE_SCN_MEM_READ,
526 SectionKind::getText());
527 DataSection = Ctx->getCOFFSection(
528 ".data", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ |
529 COFF::IMAGE_SCN_MEM_WRITE,
530 SectionKind::getData());
531 ReadOnlySection = Ctx->getCOFFSection(
532 ".rdata", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ,
533 SectionKind::getReadOnly());
534
535 if (T.getArch() == Triple::x86_64 || T.getArch() == Triple::aarch64) {
536 // On Windows 64 with SEH, the LSDA is emitted into the .xdata section
537 LSDASection = nullptr;
538 } else {
539 LSDASection = Ctx->getCOFFSection(".gcc_except_table",
540 COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
541 COFF::IMAGE_SCN_MEM_READ,
542 SectionKind::getReadOnly());
543 }
544
545 // Debug info.
546 COFFDebugSymbolsSection =
547 Ctx->getCOFFSection(".debug$S", (COFF::IMAGE_SCN_MEM_DISCARDABLE |
548 COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
549 COFF::IMAGE_SCN_MEM_READ),
550 SectionKind::getMetadata());
551 COFFDebugTypesSection =
552 Ctx->getCOFFSection(".debug$T", (COFF::IMAGE_SCN_MEM_DISCARDABLE |
553 COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |

781void MCObjectFileInfo::InitMCObjectFileInfo(const Triple &TheTriple, bool PIC,
782 MCContext &ctx,
783 bool LargeCodeModel) {
784 PositionIndependent = PIC;
785 Ctx = &ctx;
786
787 // Common.
788 CommDirectiveSupportsAlignment = true;
789 SupportsWeakOmittedEHFrame = true;
790 SupportsCompactUnwindWithoutEHFrame = false;
791 OmitDwarfIfHaveCompactUnwind = false;
792
793 FDECFIEncoding = dwarf::DW_EH_PE_absptr;
794
795 CompactUnwindDwarfEHFrameOnly = 0;
796
797 EHFrameSection = nullptr; // Created on demand.
798 CompactUnwindSection = nullptr; // Used only by selected targets.
799 DwarfAccelNamesSection = nullptr; // Used only by selected targets.
800 DwarfAccelObjCSection = nullptr; // Used only by selected targets.
801 DwarfAccelNamespaceSection = nullptr; // Used only by selected targets.
802 DwarfAccelTypesSection = nullptr; // Used only by selected targets.
803
804 TT = TheTriple;
805
806 switch (TT.getObjectFormat()) {
807 case Triple::MachO:
808 Env = IsMachO;
809 initMachOMCObjectFileInfo(TT);
810 break;
811 case Triple::COFF:
812 if (!TT.isOSWindows())
813 report_fatal_error(
814 "Cannot initialize MC for non-Windows COFF object files.");
815
816 Env = IsCOFF;
817 initCOFFMCObjectFileInfo(TT);
818 break;
819 case Triple::ELF:
820 Env = IsELF;
821 initELFMCObjectFileInfo(TT, LargeCodeModel);
822 break;
823 case Triple::Wasm:
824 Env = IsWasm;
825 initWasmMCObjectFileInfo(TT);
826 break;
827 case Triple::XCOFF:
828 Env = IsXCOFF;
829 // TODO: Initialize MCObjectFileInfo for XCOFF format when
830 // MCSectionXCOFF is ready.
831 break;
832 case Triple::UnknownObjectFormat:
833 report_fatal_error("Cannot initialize MC for unknown object file format.");
834 break;
835 }
836}

838MCSection *MCObjectFileInfo::getDwarfComdatSection(const char *Name,
839 uint64_t Hash) const {
840 switch (TT.getObjectFormat()) {
841 case Triple::ELF:
842 return Ctx->getELFSection(Name, ELF::SHT_PROGBITS, ELF::SHF_GROUP, 0,
843 utostr(Hash));
844 case Triple::MachO:
845 case Triple::COFF:
846 case Triple::Wasm:
847 case Triple::XCOFF:
848 case Triple::UnknownObjectFormat:
849 report_fatal_error("Cannot get DWARF comdat section for this object file "
850 "format: not implemented.");
851 break;
852 }
853 llvm_unreachable("Unknown ObjectFormatType");
854}

856MCSection *
857MCObjectFileInfo::getStackSizesSection(const MCSection &TextSec) const {
858 if (Env != IsELF)
859 return StackSizesSection;
860
861 const MCSectionELF &ElfSec = static_cast<const MCSectionELF &>(TextSec);
862 unsigned Flags = ELF::SHF_LINK_ORDER;
863 StringRef GroupName;
864 if (const MCSymbol *Group = ElfSec.getGroup()) {
865 GroupName = Group->getName();
866 Flags |= ELF::SHF_GROUP;
867 }
868
869 const MCSymbol *Link = TextSec.getBeginSymbol();
870 auto It = StackSizesUniquing.insert({Link, StackSizesUniquing.size()});
871 unsigned UniqueID = It.first->second;
872
873 return Ctx->getELFSection(".stack_sizes", ELF::SHT_PROGBITS, Flags, 0,
874 GroupName, UniqueID, cast<MCSymbolELF>(Link));
875}
