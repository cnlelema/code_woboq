13using namespace llvm;
14
15static cl::opt<CheriCapabilityTableABI> CapTableABI(
16 "cheri-cap-table-abi", cl::desc("ABI to use for :"),
17 cl::init(CheriCapabilityTableABI::Pcrel),
18 cl::values(clEnumValN(CheriCapabilityTableABI::Legacy, "legacy",
19 "Disable capability table and use the legacy ABI"),
20 clEnumValN(CheriCapabilityTableABI::PLT, "plt",
21 "Use PLT stubs to setup $cgp correctly"),
22 clEnumValN(CheriCapabilityTableABI::Pcrel, "pcrel",
23 "Derive $cgp from $pcc in every function"),
24 clEnumValN(CheriCapabilityTableABI::FunctionDescriptor,
25 "fn-desc",
26 "Use function descriptors to setup $cgp correctly")));
27
28bool MCTargetOptions::cheriUsesCapabilityTable() {
29 return CapTableABI != CheriCapabilityTableABI::Legacy;
30}
31
32CheriCapabilityTableABI MCTargetOptions::cheriCapabilityTableABI() {
33 return CapTableABI;
34}
35
36static cl::opt<CheriCapabilityTlsABI> CapTlsABI(
37 "cheri-cap-tls-abi", cl::desc("ABI to use for :"),
38 cl::values(clEnumValN(CheriCapabilityTlsABI::Legacy, "legacy",
39 "Disable capability TLS and use the legacy ABI"),
40 clEnumValN(CheriCapabilityTlsABI::CapEquiv, "cap-equiv",
41 "Use a capability equivalent of the normal ABI")));
42
43bool MCTargetOptions::cheriUsesCapabilityTls() {
44 return cheriCapabilityTlsABI() != CheriCapabilityTlsABI::Legacy &&
45 cheriUsesCapabilityTable();
46}
47
48CheriCapabilityTlsABI MCTargetOptions::cheriCapabilityTlsABI() {
49 if (!cheriUsesCapabilityTable())
50 return CheriCapabilityTlsABI::Legacy;
51 // For cap-table default to CapEquiv unless there is an explicit =legacy set.
52 if (CapTlsABI.getNumOccurrences() > 0)
53 return CapTlsABI;
54 return CheriCapabilityTlsABI::CapEquiv;
55}

57MCTargetOptions::MCTargetOptions()
58 : MCRelaxAll(false), MCNoExecStack(false), MCFatalWarnings(false),
59 MCNoWarn(false), MCNoDeprecatedWarn(false), MCSaveTempLabels(false),
60 MCUseDwarfDirectory(false), MCIncrementalLinkerCompatible(false),
61 MCPIECopyRelocations(false), ShowMCEncoding(false), ShowMCInst(false),
62 AsmVerbose(false), PreserveAsmComments(true) {}
