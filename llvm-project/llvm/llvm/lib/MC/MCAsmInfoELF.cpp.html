23MCSection *MCAsmInfoELF::getNonexecutableStackSection(MCContext &Ctx) const {
24 if (!UsesNonexecutableStackSection)
25 return nullptr;
26 return Ctx.getELFSection(".note.GNU-stack", ELF::SHT_PROGBITS, 0);
27}
