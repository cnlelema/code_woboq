22void llvm::dumpBytes(ArrayRef<uint8_t> bytes, raw_ostream &OS) {
23 static const char hex_rep[] = "0123456789abcdef";
24 bool First = true;
25 for (char i: bytes) {
26 if (First)
27 First = false;
28 else
29 OS << ' ';
30 OS << hex_rep[(i & 0xF0) >> 4];
31 OS << hex_rep[i & 0xF];
32 }
33}

47void MCInstPrinter::printAnnotation(raw_ostream &OS, StringRef Annot) {
48 if (!Annot.empty()) {
49 if (CommentStream) {
50 (*CommentStream) << Annot;
51 // By definition (see MCInstPrinter.h), CommentStream must end with
52 // a newline after each comment.
53 if (Annot.back() != '\n')
54 (*CommentStream) << '\n';
55 } else
56 OS << " " << MAI.getCommentString() << " " << Annot;
57 }
58}

74// For asm-style hex (e.g. 0ffh) the first digit always has to be a number.
75static bool needsLeadingZero(uint64_t Value)
76{
77 while (Value)
78 {
79 uint64_t digit = (Value >> 60) & 0xf;
80 if (digit != 0)
81 return (digit >= 0xa);
82 Value <<= 4;
83 }
84 return false;
85}

91format_object<int64_t> MCInstPrinter::formatHex(int64_t Value) const {
92 switch(PrintHexStyle) {
93 case HexStyle::C:
94 if (Value < 0)
95 return format("-0x%" PRIx64, -Value);
96 else
97 return format("0x%" PRIx64, Value);
98 case HexStyle::Asm:
99 if (Value < 0) {
100 if (needsLeadingZero((uint64_t)(-Value)))
101 return format("-0%" PRIx64 "h", -Value);
102 else
103 return format("-%" PRIx64 "h", -Value);
104 } else {
105 if (needsLeadingZero((uint64_t)(Value)))
106 return format("0%" PRIx64 "h", Value);
107 else
108 return format("%" PRIx64 "h", Value);
109 }
110 }
111 llvm_unreachable("unsupported print style");
112}

114format_object<uint64_t> MCInstPrinter::formatHex(uint64_t Value) const {
115 switch(PrintHexStyle) {
116 case HexStyle::C:
117 return format("0x%" PRIx64, Value);
118 case HexStyle::Asm:
119 if (needsLeadingZero(Value))
120 return format("0%" PRIx64 "h", Value);
121 else
122 return format("%" PRIx64 "h", Value);
123 }
124 llvm_unreachable("unsupported print style");
125}
