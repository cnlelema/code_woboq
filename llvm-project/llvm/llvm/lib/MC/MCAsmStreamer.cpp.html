339void MCAsmStreamer::AddComment(const Twine &T, bool EOL) {
340 if (!IsVerboseAsm) return;
341
342 T.toVector(CommentToEmit);
343
344 if (EOL)
345 CommentToEmit.push_back('\n'); // Place comment in a new line.
346}
347
348void MCAsmStreamer::EmitCommentsAndEOL() {
349 if (CommentToEmit.empty() && CommentStream.GetNumBytesInBuffer() == 0) {
350 OS << '\n';
351 return;
352 }
353
354 StringRef Comments = CommentToEmit;
355
356 assert(Comments.back() == '\n' &&
357 "Comment array not newline terminated");
358 do {
359 // Emit a line of comments.
360 OS.PadToColumn(MAI->getCommentColumn());
361 size_t Position = Comments.find('\n');
362 OS << MAI->getCommentString() << ' ' << Comments.substr(0, Position) <<'\n';
363
364 Comments = Comments.substr(Position+1);
365 } while (!Comments.empty());
366
367 CommentToEmit.clear();
368}

382void MCAsmStreamer::addExplicitComment(const Twine &T) {
383 StringRef c = T.getSingleStringRef();
384 if (c.equals(StringRef(MAI->getSeparatorString())))
385 return;
386 if (c.startswith(StringRef("//"))) {
387 ExplicitCommentToEmit.append("\t");
388 ExplicitCommentToEmit.append(MAI->getCommentString());
389 // drop //
390 ExplicitCommentToEmit.append(c.slice(2, c.size()).str());
391 } else if (c.startswith(StringRef("/*"))) {
392 size_t p = 2, len = c.size() - 2;
393 // emit each line in comment as separate newline.
394 do {
395 size_t newp = std::min(len, c.find_first_of("\r\n", p));
396 ExplicitCommentToEmit.append("\t");
397 ExplicitCommentToEmit.append(MAI->getCommentString());
398 ExplicitCommentToEmit.append(c.slice(p, newp).str());
399 // If we have another line in this comment add line
400 if (newp < len)
401 ExplicitCommentToEmit.append("\n");
402 p = newp + 1;
403 } while (p < len);
404 } else if (c.startswith(StringRef(MAI->getCommentString()))) {
405 ExplicitCommentToEmit.append("\t");
406 ExplicitCommentToEmit.append(c.str());
407 } else if (c.front() == '#') {
408
409 ExplicitCommentToEmit.append("\t");
410 ExplicitCommentToEmit.append(MAI->getCommentString());
411 ExplicitCommentToEmit.append(c.slice(1, c.size()).str());
412 } else
413 assert(false && "Unexpected Assembly Comment");
414 // full line comments immediately output
415 if (c.back() == '\n')
416 emitExplicitComments();
417}

468 OS << ", ";

471 }
472 EmitEOL();

843static inline char toOctal(int X) { return (X&7)+'0'; }
844
845static void PrintQuotedString(StringRef Data, raw_ostream &OS) {
846 OS << '"';
847
848 for (unsigned i = 0, e = Data.size(); i != e; ++i) {
849 unsigned char C = Data[i];
850 if (C == '"' || C == '\\') {
851 OS << '\\' << (char)C;
852 continue;
853 }
854
855 if (isPrint((unsigned char)C)) {
856 OS << (char)C;
857 continue;
858 }
859
860 switch (C) {
861 case '\b': OS << "\\b"; break;
862 case '\f': OS << "\\f"; break;
863 case '\n': OS << "\\n"; break;
864 case '\r': OS << "\\r"; break;
865 case '\t': OS << "\\t"; break;
866 default:
867 OS << '\\';
868 OS << toOctal(C >> 6);
869 OS << toOctal(C >> 3);
870 OS << toOctal(C >> 0);
871 break;
872 }
873 }
874
875 OS << '"';
876}

878void MCAsmStreamer::EmitBytes(StringRef Data) {
879 assert(getCurrentSectionOnly() &&
880 "Cannot emit contents before setting section!");
881 if (Data.empty()) return;
882
883 // If only single byte is provided or no ascii or asciz directives is
884 // supported, emit as vector of 8bits data.
885 if (Data.size() == 1 ||
886 !(MAI->getAscizDirective() || MAI->getAsciiDirective())) {
887 if (MCTargetStreamer *TS = getTargetStreamer()) {
888 TS->emitRawBytes(Data);
889 } else {
890 const char *Directive = MAI->getData8bitsDirective();
891 for (const unsigned char C : Data.bytes()) {
892 OS << Directive << (unsigned)C;
893 EmitEOL();
894 }
895 }
896 return;
897 }
898
899 // If the data ends with 0 and the target supports .asciz, use it, otherwise
900 // use .ascii
901 if (MAI->getAscizDirective() && Data.back() == 0) {
902 OS << MAI->getAscizDirective();
903 Data = Data.substr(0, Data.size()-1);
904 } else {
905 OS << MAI->getAsciiDirective();
906 }
907
908 PrintQuotedString(Data, OS);
909 EmitEOL();
910}

912void MCAsmStreamer::EmitBinaryData(StringRef Data) {
913 // This is binary data. Print it in a grid of hex bytes for readability.
914 const size_t Cols = 4;
915 for (size_t I = 0, EI = alignTo(Data.size(), Cols); I < EI; I += Cols) {
916 size_t J = I, EJ = std::min(I + Cols, Data.size());

918 OS << MAI->getData8bitsDirective();

921 OS << format("0x%02x", uint8_t(Data[J]));
922 EmitEOL();

930void MCAsmStreamer::EmitValueImpl(const MCExpr *Value, unsigned Size,
931 SMLoc Loc) {
932 assert(Size <= 8 && "Invalid size");
933 assert(getCurrentSectionOnly() &&
934 "Cannot emit contents before setting section!");
935 const char *Directive = nullptr;
936 switch (Size) {
937 default: break;
938 case 1: Directive = MAI->getData8bitsDirective(); break;
939 case 2: Directive = MAI->getData16bitsDirective(); break;
940 case 4: Directive = MAI->getData32bitsDirective(); break;
941 case 8: Directive = MAI->getData64bitsDirective(); break;
942 }
943
944 if (!Directive) {
945 int64_t IntValue;
946 if (!Value->evaluateAsAbsolute(IntValue))
947 report_fatal_error("Don't know how to emit this value.");
948
949 // We couldn't handle the requested integer size so we fallback by breaking
950 // the request down into several, smaller, integers.
951 // Since sizes greater or equal to "Size" are invalid, we use the greatest
952 // power of 2 that is less than "Size" as our largest piece of granularity.
953 bool IsLittleEndian = MAI->isLittleEndian();
954 for (unsigned Emitted = 0; Emitted != Size;) {
955 unsigned Remaining = Size - Emitted;
956 // The size of our partial emission must be a power of two less than
957 // Size.
958 unsigned EmissionSize = PowerOf2Floor(std::min(Remaining, Size - 1));
959 // Calculate the byte offset of our partial emission taking into account
960 // the endianness of the target.
961 unsigned ByteOffset =
962 IsLittleEndian ? Emitted : (Remaining - EmissionSize);
963 uint64_t ValueToEmit = IntValue >> (ByteOffset * 8);
964 // We truncate our partial emission to fit within the bounds of the
965 // emission domain. This produces nicer output and silences potential
966 // truncation warnings when round tripping through another assembler.
967 uint64_t Shift = 64 - EmissionSize * 8;
968 assert(Shift < static_cast<uint64_t>(
969 std::numeric_limits<unsigned long long>::digits) &&
970 "undefined behavior");
971 ValueToEmit &= ~0ULL >> Shift;
972 EmitIntValue(ValueToEmit, EmissionSize);
973 Emitted += EmissionSize;
974 }
975 return;
976 }
977
978 assert(Directive && "Invalid size for machine code value!");
979 OS << Directive;
980 if (MCTargetStreamer *TS = getTargetStreamer()) {
981 TS->emitValue(Value);
982 } else {
983 Value->print(OS, MAI);
984 EmitEOL();
985 }
986}

1191 unsigned FileNo, StringRef Directory, StringRef Filename,

1199 getContext().getDwarfVersion(), FileNo);

1212 TS->emitDwarfFileDirective(OS1.str());
1213 else
1214 EmitRawText(OS1.str());
1215
1216 return FileNo;

1243void MCAsmStreamer::EmitDwarfLocDirective(unsigned FileNo, unsigned Line,
1244 unsigned Column, unsigned Flags,
1245 unsigned Isa,
1246 unsigned Discriminator,
1247 StringRef FileName) {
1248 OS << "\t.loc\t" << FileNo << " " << Line << " " << Column;
1249 if (MAI->supportsExtendedDwarfLocDirective()) {
1250 if (Flags & DWARF2_FLAG_BASIC_BLOCK)
1251 OS << " basic_block";
1252 if (Flags & DWARF2_FLAG_PROLOGUE_END)
1253 OS << " prologue_end";
1254 if (Flags & DWARF2_FLAG_EPILOGUE_BEGIN)
1255 OS << " epilogue_begin";
1256
1257 unsigned OldFlags = getContext().getCurrentDwarfLoc().getFlags();
1258 if ((Flags & DWARF2_FLAG_IS_STMT) != (OldFlags & DWARF2_FLAG_IS_STMT)) {
1259 OS << " is_stmt ";
1260
1261 if (Flags & DWARF2_FLAG_IS_STMT)
1262 OS << "1";
1263 else
1264 OS << "0";
1265 }
1266
1267 if (Isa)
1268 OS << " isa " << Isa;
1269 if (Discriminator)
1270 OS << " discriminator " << Discriminator;
1271 }

1743void MCAsmStreamer::AddEncodingComment(const MCInst &Inst,
1744 const MCSubtargetInfo &STI) {
1745 raw_ostream &OS = GetCommentOS();
1746 SmallString<256> Code;
1747 SmallVector<MCFixup, 4> Fixups;
1748 raw_svector_ostream VecOS(Code);
1749
1750 // If we have no code emitter, don't emit code.
1751 if (!getAssembler().getEmitterPtr())
1752 return;
1753
1754 getAssembler().getEmitter().encodeInstruction(Inst, VecOS, Fixups, STI);
1755
1756 // If we are showing fixups, create symbolic markers in the encoded
1757 // representation. We do this by making a per-bit map to the fixup item index,
1758 // then trying to display it as nicely as possible.
1759 SmallVector<uint8_t, 64> FixupMap;
1760 FixupMap.resize(Code.size() * 8);
1761 for (unsigned i = 0, e = Code.size() * 8; i != e; ++i)
1762 FixupMap[i] = 0;
1763
1764 for (unsigned i = 0, e = Fixups.size(); i != e; ++i) {
1765 MCFixup &F = Fixups[i];
1766 const MCFixupKindInfo &Info =
1767 getAssembler().getBackend().getFixupKindInfo(F.getKind());
1768 for (unsigned j = 0; j != Info.TargetSize; ++j) {
1769 unsigned Index = F.getOffset() * 8 + Info.TargetOffset + j;
1770 assert(Index < Code.size() * 8 && "Invalid offset in fixup!");
1771 FixupMap[Index] = 1 + i;
1772 }
1773 }
1774
1775 // FIXME: Note the fixup comments for Thumb2 are completely bogus since the
1776 // high order halfword of a 32-bit Thumb2 instruction is emitted first.
1777 OS << "encoding: [";
1778 for (unsigned i = 0, e = Code.size(); i != e; ++i) {
1779 if (i)
1780 OS << ',';
1781
1782 // See if all bits are the same map entry.
1783 uint8_t MapEntry = FixupMap[i * 8 + 0];
1784 for (unsigned j = 1; j != 8; ++j) {
1785 if (FixupMap[i * 8 + j] == MapEntry)
1786 continue;
1787
1788 MapEntry = uint8_t(~0U);
1789 break;
1790 }
1791
1792 if (MapEntry != uint8_t(~0U)) {
1793 if (MapEntry == 0) {
1794 OS << format("0x%02x", uint8_t(Code[i]));
1795 } else {
1796 if (Code[i]) {
1797 // FIXME: Some of the 8 bits require fix up.
1798 OS << format("0x%02x", uint8_t(Code[i])) << '\''
1799 << char('A' + MapEntry - 1) << '\'';
1800 } else
1801 OS << char('A' + MapEntry - 1);
1802 }
1803 } else {
1804 // Otherwise, write out in binary.
1805 OS << "0b";
1806 for (unsigned j = 8; j--;) {
1807 unsigned Bit = (Code[i] >> j) & 1;
1808
1809 unsigned FixupBit;
1810 if (MAI->isLittleEndian())
1811 FixupBit = i * 8 + j;
1812 else
1813 FixupBit = i * 8 + (7-j);
1814
1815 if (uint8_t MapEntry = FixupMap[FixupBit]) {
1816 assert(Bit == 0 && "Encoder wrote into fixed up bit!");
1817 OS << char('A' + MapEntry - 1);
1818 } else
1819 OS << Bit;
1820 }
1821 }
1822 }
1823 OS << "]\n";
1824
1825 for (unsigned i = 0, e = Fixups.size(); i != e; ++i) {
1826 MCFixup &F = Fixups[i];
1827 const MCFixupKindInfo &Info =
1828 getAssembler().getBackend().getFixupKindInfo(F.getKind());
1829 OS << " fixup " << char('A' + i) << " - " << "offset: " << F.getOffset()
1830 << ", value: " << *F.getValue() << ", kind: " << Info.Name << "\n";
1831 }
1832}

1837 "Cannot emit contents before setting section!");

1931void MCAsmStreamer::FinishImpl() {
1932 // If we are generating dwarf for assembly source files dump out the sections.
1933 if (getContext().getGenDwarfForAssembly())
1934 MCGenDwarfInfo::Emit(this);
1935
1936 // Emit the label for the line table, if requested - since the rest of the
1937 // line table will be defined by .loc/.file directives, and not emitted
1938 // directly, the label is the only work required here.
1939 const auto &Tables = getContext().getMCDwarfLineTables();
1940 if (!Tables.empty()) {
1941 assert(Tables.size() == 1 && "asm output only supports one line table");
1942 if (auto *Label = Tables.begin()->second.getLabel()) {
1943 SwitchSection(getContext().getObjectFileInfo()->getDwarfLineSection());
1944 EmitLabel(Label);
1945 }
1946 }
