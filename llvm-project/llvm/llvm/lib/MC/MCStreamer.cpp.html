77void MCTargetStreamer::emitRawBytes(StringRef Data) {
78 const MCAsmInfo *MAI = Streamer.getContext().getAsmInfo();
79 const char *Directive = MAI->getData8bitsDirective();
80 for (const unsigned char C : Data.bytes()) {
81 SmallString<128> Str;
82 raw_svector_ostream OS(Str);
83
84 OS << Directive << (unsigned)C;
85 Streamer.EmitRawText(OS.str());
86 }
87}

137/// EmitIntValue - Special case of EmitValue that avoids the client having to
138/// pass in a MCExpr for constant integers.
139void MCStreamer::EmitIntValue(uint64_t Value, unsigned Size) {
140 assert(1 <= Size && Size <= 8 && "Invalid size");
141 assert((isUIntN(8 * Size, Value) || isIntN(8 * Size, Value)) &&
142 "Invalid size");
143 char buf[8];
144 const bool isLittleEndian = Context.getAsmInfo()->isLittleEndian();
145 for (unsigned i = 0; i != Size; ++i) {
146 unsigned index = isLittleEndian ? i : (Size - i - 1);
147 buf[i] = uint8_t(Value >> (index * 8));
148 }
149 EmitBytes(StringRef(buf, Size));
150}

215void MCStreamer::EmitCheriCapability(const MCSymbol *Value, int64_t Addend,
216 unsigned CapSize, SMLoc Loc) {
217 if (LLVM_UNLIKELY(TargetStreamer->useLegacyCapRelocs())) {
218 // XXXAR: The legacy path still exists to allow comparing bounds quality vs
219 // the R_CHERI_CAPABILITY approach.
220 // TODO: remove this at some point in the future
221 assert(Value);
222 auto *E = MCBinaryExpr::createAdd(MCSymbolRefExpr::create(Value, Context),
223 MCConstantExpr::create(Addend, Context),
224 Context);
225 EmitLegacyCHERICapability(E, CapSize, Loc);
226 } else {
227 EmitCheriCapabilityImpl(Value, Addend, CapSize, Loc);
228 }
229}
230
231void MCStreamer::EmitLegacyCHERICapability(const MCExpr *Value,
232 unsigned CapSize, SMLoc Loc) {
233 assert(TargetStreamer->useLegacyCapRelocs());
234 if (const MCConstantExpr *CE = dyn_cast<MCConstantExpr>(Value)) {
235 EmitCheriIntcap(CE->getValue(), CapSize);
236 return;
237 }
238 assert(CapSize == 32 || CapSize == 16);
239 // MCStreamers with proper capability support will emit real relocations
240 // instead of using the __cap_relocs hack
241 MCSymbol *Here = Context.createTempSymbol();
242 EmitLabel(Here);
243 const MCSectionELF *Section =
244 dyn_cast<const MCSectionELF>(SectionStack.back().first.first);
245 const MCSymbolELF *Group = nullptr;
246 if (Section && Section->getGroup())
247 Group = Section->getGroup();
248
249 if (Group) {
250 Context.getELFSection("__cap_relocs", ELF::SHT_PROGBITS,
251 ELF::SHF_ALLOC | ELF::SHF_GROUP, 0, Group->getName());
252 FatRelocs.push_back(std::make_tuple(Here, Value, Group->getName()));
253 } else {
254 Context.getELFSection("__cap_relocs", ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
255 FatRelocs.push_back(std::make_tuple(Here, Value, StringRef()));
256 }
257 // We do this here to ensure that the section exists.
258 EmitZeros(CapSize);
259}

370bool MCStreamer::checkCVLocSection(unsigned FuncId, unsigned FileNo,
371 SMLoc Loc) {
372 CodeViewContext &CVC = getContext().getCVContext();
373 MCCVFunctionInfo *FI = CVC.getCVFunctionInfo(FuncId);
374 if (!FI) {
375 getContext().reportError(
376 Loc, "function id not introduced by .cv_func_id or .cv_inline_site_id");
377 return false;
378 }
379
380 // Track the section
381 if (FI->Section == nullptr)
382 FI->Section = getCurrentSectionOnly();
383 else if (FI->Section != getCurrentSectionOnly()) {
384 getContext().reportError(
385 Loc,
386 "all .cv_loc directives for a function must be in the same section");
387 return false;
388 }
389 return true;
390}

416 Symbol->setFragment(Fragment);

423void MCStreamer::EmitLabel(MCSymbol *Symbol, SMLoc Loc) {
424 Symbol->redefineIfPossible();
425
426 if (!Symbol->isUndefined() || Symbol->isVariable())
427 return getContext().reportError(Loc, "invalid symbol redefinition of " + Symbol->getName());
428
429 assert(!Symbol->isVariable() && "Cannot emit a variable symbol!");
430 assert(getCurrentSectionOnly() && "Cannot emit before setting section!");
431 assert(!Symbol->getFragment() && "Unexpected fragment on symbol data!");
432 assert(Symbol->isUndefined() && "Cannot define a symbol twice!");
433
434 Symbol->setFragment(&getCurrentSectionOnly()->getDummyFragment());
435
436 MCTargetStreamer *TS = getTargetStreamer();
437 if (TS)
438 TS->emitLabel(Symbol);
439}

445void MCStreamer::EmitCFIStartProc(bool IsSimple, SMLoc Loc) {
446 if (hasUnfinishedDwarfFrameInfo())
447 return getContext().reportError(
448 Loc, "starting new .cfi frame before finishing the previous one");
449
450 MCDwarfFrameInfo Frame;
451 Frame.IsSimple = IsSimple;
452 EmitCFIStartProcImpl(Frame);
453
454 const MCAsmInfo* MAI = Context.getAsmInfo();
455 if (MAI) {
456 for (const MCCFIInstruction& Inst : MAI->getInitialFrameState()) {
457 if (Inst.getOperation() == MCCFIInstruction::OpDefCfa ||
458 Inst.getOperation() == MCCFIInstruction::OpDefCfaRegister) {
459 Frame.CurrentCfaRegister = Inst.getRegister();
460 }
461 }
462 }
463
464 DwarfFrameInfos.push_back(Frame);
465}

789static MCSection *getWinCFISection(MCContext &Context, unsigned *NextWinCFIID,
790 MCSection *MainCFISec,
791 const MCSection *TextSec) {
792 // If this is the main .text section, use the main unwind info section.
793 if (TextSec == Context.getObjectFileInfo()->getTextSection())
794 return MainCFISec;
795
796 const auto *TextSecCOFF = cast<MCSectionCOFF>(TextSec);
797 auto *MainCFISecCOFF = cast<MCSectionCOFF>(MainCFISec);
798 unsigned UniqueID = TextSecCOFF->getOrAssignWinCFISectionID(NextWinCFIID);
799
800 // If this section is COMDAT, this unwind section should be COMDAT associative
801 // with its group.
802 const MCSymbol *KeySym = nullptr;
803 if (TextSecCOFF->getCharacteristics() & COFF::IMAGE_SCN_LNK_COMDAT) {
804 KeySym = TextSecCOFF->getCOMDATSymbol();
805
806 // In a GNU environment, we can't use associative comdats. Instead, do what
807 // GCC does, which is to make plain comdat selectany section named like
808 // ".[px]data$_Z3foov".
809 if (!Context.getAsmInfo()->hasCOFFAssociativeComdats()) {
810 std::string SectionName =
811 (MainCFISecCOFF->getSectionName() + "$" +
812 TextSecCOFF->getSectionName().split('$').second)
813 .str();
814 return Context.getCOFFSection(
815 SectionName,
816 MainCFISecCOFF->getCharacteristics() | COFF::IMAGE_SCN_LNK_COMDAT,
817 MainCFISecCOFF->getKind(), "", COFF::IMAGE_COMDAT_SELECT_ANY);
818 }
819 }
820
821 return Context.getAssociativeCOFFSection(MainCFISecCOFF, KeySym, UniqueID);
822}

974void MCStreamer::Finish() {
975 if ((!DwarfFrameInfos.empty() && !DwarfFrameInfos.back().End) ||
976 (!WinFrameInfos.empty() && !WinFrameInfos.back()->End)) {
977 getContext().reportError(SMLoc(), "Unfinished frame!");
978 return;
979 }
980
981 if (!FatRelocs.empty()) {
982 MCSection *DefaultRelocSection = Context.getELFSection("__cap_relocs",
983 ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
984 DefaultRelocSection->setAlignment(8);
985 for (auto &R : FatRelocs) {
986 MCSymbol *Sym;
987 const MCExpr *Value;
988 MCSection *RelocSection;
989 StringRef GroupName;
990
991 std::tie(Sym, Value, GroupName) = R;
992 if (GroupName != StringRef()) {
993 RelocSection =
994 Context.getELFSection("__cap_relocs", ELF::SHT_PROGBITS,
995 ELF::SHF_ALLOC | ELF::SHF_GROUP, 0, GroupName);
996 RelocSection->setAlignment(8);
997 } else {
998 RelocSection = DefaultRelocSection;
999 }
1000
1001 SwitchSection(RelocSection);
1002
1003 EmitValue(MCSymbolRefExpr::create(Sym, Context), 8);
1004 if (const MCSymbolRefExpr *Sym = dyn_cast<MCSymbolRefExpr>(Value)) {
1005 EmitValue(Sym, 8);
1006 EmitZeros(8);
1007 } else {
1008 const MCBinaryExpr *Bin = cast<MCBinaryExpr>(Value);
1009 EmitValue(cast<MCSymbolRefExpr>(Bin->getLHS()), 8);
1010 EmitValue(Bin->getRHS(), 8);
1011 }
1012 // TODO: Emit size / perms here.
1013 EmitZeros(16);
1014 }
1015 }
1016
1017 MCTargetStreamer *TS = getTargetStreamer();
1018 if (TS)
1019 TS->finish();
1020
1021 FinishImpl();
1022}

1042void MCStreamer::visitUsedExpr(const MCExpr &Expr) {
1043 switch (Expr.getKind()) {
1044 case MCExpr::Target:
1045 cast<MCTargetExpr>(Expr).visitUsedExpr(*this);
1046 break;
1047
1048 case MCExpr::Constant:
1049 break;
1050
1051 case MCExpr::Binary: {
1052 const MCBinaryExpr &BE = cast<MCBinaryExpr>(Expr);
1053 visitUsedExpr(*BE.getLHS());
1054 visitUsedExpr(*BE.getRHS());
1055 break;
1056 }
1057
1058 case MCExpr::SymbolRef:
1059 visitUsedSymbol(cast<MCSymbolRefExpr>(Expr).getSymbol());
1060 break;
1061
1062 case MCExpr::Unary:
1063 visitUsedExpr(*cast<MCUnaryExpr>(Expr).getSubExpr());
1064 break;
1065 }
1066}

1151void MCStreamer::SwitchSection(MCSection *Section, const MCExpr *Subsection) {
1152 assert(Section && "Cannot switch to a null section!");
1153 MCSectionSubPair curSection = SectionStack.back().first;
1154 SectionStack.back().second = curSection;
1155 if (MCSectionSubPair(Section, Subsection) != curSection) {
1156 ChangeSection(Section, Subsection);
1157 SectionStack.back().first = MCSectionSubPair(Section, Subsection);
1158 assert(!Section->hasEnded() && "Section already ended");
1159 MCSymbol *Sym = Section->getBeginSymbol();
1160 if (Sym && !Sym->isInSection())
1161 EmitLabel(Sym);
1162 }
1163}

1177void MCStreamer::EmitVersionForTarget(const Triple &Target,
1178 const VersionTuple &SDKVersion) {
1179 if (!Target.isOSBinFormatMachO() || !Target.isOSDarwin())
1180 return;
1181 // Do we even know the version?
1182 if (Target.getOSMajorVersion() == 0)
1183 return;
1184
1185 unsigned Major;
1186 unsigned Minor;
1187 unsigned Update;
1188 MCVersionMinType VersionType;
1189 if (Target.isWatchOS()) {
1190 VersionType = MCVM_WatchOSVersionMin;
1191 Target.getWatchOSVersion(Major, Minor, Update);
1192 } else if (Target.isTvOS()) {
1193 VersionType = MCVM_TvOSVersionMin;
1194 Target.getiOSVersion(Major, Minor, Update);
1195 } else if (Target.isMacOSX()) {
1196 VersionType = MCVM_OSXVersionMin;
1197 if (!Target.getMacOSXVersion(Major, Minor, Update))
1198 Major = 0;
1199 } else {
1200 VersionType = MCVM_IOSVersionMin;
1201 Target.getiOSVersion(Major, Minor, Update);
1202 }
1203 if (Major != 0)
1204 EmitVersionMin(VersionType, Major, Minor, Update, SDKVersion);
1205}
