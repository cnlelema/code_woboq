143unsigned PMDataManager::initSizeRemarkInfo(
144 Module &M, StringMap<std::pair<unsigned, unsigned>> &FunctionToInstrCount) {
145 // Only calculate getInstructionCount if the size-info remark is requested.
146 unsigned InstrCount = 0;
147
148 // Collect instruction counts for every function. We'll use this to emit
149 // per-function size remarks later.
150 for (Function &F : M) {
151 unsigned FCount = F.getInstructionCount();
152
153 // Insert a record into FunctionToInstrCount keeping track of the current
154 // size of the function as the first member of a pair. Set the second
155 // member to 0; if the function is deleted by the pass, then when we get
156 // here, we'll be able to let the user know that F no longer contributes to
157 // the module.
158 FunctionToInstrCount[F.getName().str()] =
159 std::pair<unsigned, unsigned>(FCount, 0);
160 InstrCount += FCount;
161 }
162 return InstrCount;
163}

165void PMDataManager::emitInstrCountChangedRemark(
166 Pass *P, Module &M, int64_t Delta, unsigned CountBefore,
167 StringMap<std::pair<unsigned, unsigned>> &FunctionToInstrCount,
168 Function *F) {
169 // If it's a pass manager, don't emit a remark. (This hinges on the assumption
170 // that the only passes that return non-null with getAsPMDataManager are pass
171 // managers.) The reason we have to do this is to avoid emitting remarks for
172 // CGSCC passes.
173 if (P->getAsPMDataManager())
174 return;
175
176 // Set to true if this isn't a module pass or CGSCC pass.
177 bool CouldOnlyImpactOneFunction = (F != nullptr);
178
179 // Helper lambda that updates the changes to the size of some function.
180 auto UpdateFunctionChanges =
181 [&FunctionToInstrCount](Function &MaybeChangedFn) {
182 // Update the total module count.
183 unsigned FnSize = MaybeChangedFn.getInstructionCount();
184 auto It = FunctionToInstrCount.find(MaybeChangedFn.getName());
185
186 // If we created a new function, then we need to add it to the map and
187 // say that it changed from 0 instructions to FnSize.
188 if (It == FunctionToInstrCount.end()) {
189 FunctionToInstrCount[MaybeChangedFn.getName()] =
190 std::pair<unsigned, unsigned>(0, FnSize);
191 return;
192 }
193 // Insert the new function size into the second member of the pair. This
194 // tells us whether or not this function changed in size.
195 It->second.second = FnSize;
196 };
197
198 // We need to initially update all of the function sizes.
199 // If no function was passed in, then we're either a module pass or an
200 // CGSCC pass.
201 if (!CouldOnlyImpactOneFunction)
202 std::for_each(M.begin(), M.end(), UpdateFunctionChanges);
203 else
204 UpdateFunctionChanges(*F);
205
206 // Do we have a function we can use to emit a remark?
207 if (!CouldOnlyImpactOneFunction) {
208 // We need a function containing at least one basic block in order to output
209 // remarks. Since it's possible that the first function in the module
210 // doesn't actually contain a basic block, we have to go and find one that's
211 // suitable for emitting remarks.
212 auto It = std::find_if(M.begin(), M.end(),
213 [](const Function &Fn) { return !Fn.empty(); });
214
215 // Didn't find a function. Quit.
216 if (It == M.end())
217 return;
218
219 // We found a function containing at least one basic block.
220 F = &*It;
221 }
222 int64_t CountAfter = static_cast<int64_t>(CountBefore) + Delta;
223 BasicBlock &BB = *F->begin();
224 OptimizationRemarkAnalysis R("size-info", "IRSizeChange",
225 DiagnosticLocation(), &BB);
226 // FIXME: Move ore namespace to DiagnosticInfo so that we can use it. This
227 // would let us use NV instead of DiagnosticInfoOptimizationBase::Argument.
228 R << DiagnosticInfoOptimizationBase::Argument("Pass", P->getPassName())
229 << ": IR instruction count changed from "
230 << DiagnosticInfoOptimizationBase::Argument("IRInstrsBefore", CountBefore)
231 << " to "
232 << DiagnosticInfoOptimizationBase::Argument("IRInstrsAfter", CountAfter)
233 << "; Delta: "
234 << DiagnosticInfoOptimizationBase::Argument("DeltaInstrCount", Delta);
235 F->getContext().diagnose(R); // Not using ORE for layering reasons.
236
237 // Emit per-function size change remarks separately.
238 std::string PassName = P->getPassName().str();
239
240 // Helper lambda that emits a remark when the size of a function has changed.
241 auto EmitFunctionSizeChangedRemark = [&FunctionToInstrCount, &F, &BB,
242 &PassName](const std::string &Fname) {
243 unsigned FnCountBefore, FnCountAfter;
244 std::pair<unsigned, unsigned> &Change = FunctionToInstrCount[Fname];
245 std::tie(FnCountBefore, FnCountAfter) = Change;
246 int64_t FnDelta = static_cast<int64_t>(FnCountAfter) -
247 static_cast<int64_t>(FnCountBefore);
248
249 if (FnDelta == 0)
250 return;
251
252 // FIXME: We shouldn't use BB for the location here. Unfortunately, because
253 // the function that we're looking at could have been deleted, we can't use
254 // it for the source location. We *want* remarks when a function is deleted
255 // though, so we're kind of stuck here as is. (This remark, along with the
256 // whole-module size change remarks really ought not to have source
257 // locations at all.)
258 OptimizationRemarkAnalysis FR("size-info", "FunctionIRSizeChange",
259 DiagnosticLocation(), &BB);
260 FR << DiagnosticInfoOptimizationBase::Argument("Pass", PassName)
261 << ": Function: "
262 << DiagnosticInfoOptimizationBase::Argument("Function", Fname)
263 << ": IR instruction count changed from "
264 << DiagnosticInfoOptimizationBase::Argument("IRInstrsBefore",
265 FnCountBefore)
266 << " to "
267 << DiagnosticInfoOptimizationBase::Argument("IRInstrsAfter",
268 FnCountAfter)
269 << "; Delta: "
270 << DiagnosticInfoOptimizationBase::Argument("DeltaInstrCount", FnDelta);
271 F->getContext().diagnose(FR);
272
273 // Update the function size.
274 Change.first = FnCountAfter;
275 };
276
277 // Are we looking at more than one function? If so, emit remarks for all of
278 // the functions in the module. Otherwise, only emit one remark.
279 if (!CouldOnlyImpactOneFunction)
280 std::for_each(FunctionToInstrCount.keys().begin(),
281 FunctionToInstrCount.keys().end(),
282 EmitFunctionSizeChangedRemark);
283 else
284 EmitFunctionSizeChangedRemark(F->getName().str());
285}

287void PassManagerPrettyStackEntry::print(raw_ostream &OS) const {
288 if (!V && !M)
289 OS << "Releasing pass '";
290 else
291 OS << "Running pass '";
292
293 OS << P->getPassName() << "'";
294
295 if (M) {
296 OS << " on module '" << M->getModuleIdentifier() << "'.\n";
297 return;
298 }
299 if (!V) {
300 OS << '\n';
301 return;
302 }
303
304 OS << " on ";
305 if (isa<Function>(V))
306 OS << "function";
307 else if (isa<BasicBlock>(V))
308 OS << "basic block";
309 else
310 OS << "value";
311
312 OS << " '";
313 V->printAsOperand(OS, /*PrintTy=*/false, M);
314 OS << "'\n";
315}

601/// Set pass P as the last user of the given analysis passes.
602void
603PMTopLevelManager::setLastUser(ArrayRef<Pass*> AnalysisPasses, Pass *P) {
604 unsigned PDepth = 0;
605 if (P->getResolver())
606 PDepth = P->getResolver()->getPMDataManager().getDepth();
607
608 for (Pass *AP : AnalysisPasses) {
609 LastUser[AP] = P;
610
611 if (P == AP)
612 continue;
613
614 // Update the last users of passes that are required transitive by AP.
615 AnalysisUsage *AnUsage = findAnalysisUsage(AP);
616 const AnalysisUsage::VectorType &IDs = AnUsage->getRequiredTransitiveSet();
617 SmallVector<Pass *, 12> LastUses;
618 SmallVector<Pass *, 12> LastPMUses;
619 for (AnalysisID ID : IDs) {
620 Pass *AnalysisPass = findAnalysisPass(ID);
621 assert(AnalysisPass && "Expected analysis pass to exist.");
622 AnalysisResolver *AR = AnalysisPass->getResolver();
623 assert(AR && "Expected analysis resolver to exist.");
624 unsigned APDepth = AR->getPMDataManager().getDepth();
625
626 if (PDepth == APDepth)
627 LastUses.push_back(AnalysisPass);
628 else if (PDepth > APDepth)
629 LastPMUses.push_back(AnalysisPass);
630 }
631
632 setLastUser(LastUses, P);
633
634 // If this pass has a corresponding pass manager, push higher level
635 // analysis to this pass manager.
636 if (P->getResolver())
637 setLastUser(LastPMUses, P->getResolver()->getPMDataManager().getAsPass());
638
639
640 // If AP is the last user of other passes then make P last user of
641 // such passes.
642 for (auto LU : LastUser) {
643 if (LU.second == AP)
644 // DenseMap iterator is not invalidated here because
645 // this is just updating existing entries.
646 LastUser[LU.first] = P;
647 }
648 }
649}

651/// Collect passes whose last user is P
652void PMTopLevelManager::collectLastUses(SmallVectorImpl<Pass *> &LastUses,
653 Pass *P) {
654 DenseMap<Pass *, SmallPtrSet<Pass *, 8> >::iterator DMI =
655 InversedLastUser.find(P);
656 if (DMI == InversedLastUser.end())
657 return;
658
659 SmallPtrSet<Pass *, 8> &LU = DMI->second;
660 for (Pass *LUP : LU) {
661 LastUses.push_back(LUP);
662 }

666AnalysisUsage *PMTopLevelManager::findAnalysisUsage(Pass *P) {
667 AnalysisUsage *AnUsage = nullptr;
668 auto DMI = AnUsageMap.find(P);
669 if (DMI != AnUsageMap.end())
670 AnUsage = DMI->second;
671 else {
672 // Look up the analysis usage from the pass instance (different instances
673 // of the same pass can produce different results), but unique the
674 // resulting object to reduce memory usage. This helps to greatly reduce
675 // memory usage when we have many instances of only a few pass types
676 // (e.g. instcombine, simplifycfg, etc...) which tend to share a fixed set
677 // of dependencies.
678 AnalysisUsage AU;
679 P->getAnalysisUsage(AU);
680
681 AUFoldingSetNode* Node = nullptr;
682 FoldingSetNodeID ID;
683 AUFoldingSetNode::Profile(ID, AU);
684 void *IP = nullptr;
685 if (auto *N = UniqueAnalysisUsages.FindNodeOrInsertPos(ID, IP))
686 Node = N;
687 else {
688 Node = new (AUFoldingSetNodeAllocator.Allocate()) AUFoldingSetNode(AU);
689 UniqueAnalysisUsages.InsertNode(Node, IP);
690 }
691 assert(Node && "cached analysis usage must be non null");
692
693 AnUsageMap[P] = &Node->AU;
694 AnUsage = &Node->AU;
695 }
696 return AnUsage;
697}

699/// Schedule pass P for execution. Make sure that passes required by
700/// P are run before P is run. Update analysis info maintained by
701/// the manager. Remove dead passes. This is a recursive function.
702void PMTopLevelManager::schedulePass(Pass *P) {
703
704 // TODO : Allocate function manager for this pass, other wise required set
705 // may be inserted into previous function manager
706
707 // Give pass a chance to prepare the stage.
708 P->preparePassManager(activeStack);
709
710 // If P is an analysis pass and it is available then do not
711 // generate the analysis again. Stale analysis info should not be
712 // available at this point.
713 const PassInfo *PI = findAnalysisPassInfo(P->getPassID());
714 if (PI && PI->isAnalysis() && findAnalysisPass(P->getPassID())) {
715 // Remove any cached AnalysisUsage information.
716 AnUsageMap.erase(P);
717 delete P;
718 return;
719 }
720
721 AnalysisUsage *AnUsage = findAnalysisUsage(P);
722
723 bool checkAnalysis = true;
724 while (checkAnalysis) {
725 checkAnalysis = false;
726
727 const AnalysisUsage::VectorType &RequiredSet = AnUsage->getRequiredSet();
728 for (const AnalysisID ID : RequiredSet) {
729
730 Pass *AnalysisPass = findAnalysisPass(ID);
731 if (!AnalysisPass) {
732 const PassInfo *PI = findAnalysisPassInfo(ID);
733
734 if (!PI) {
735 // Pass P is not in the global PassRegistry
736 dbgs() << "Pass '" << P->getPassName() << "' is not initialized." << "\n";
737 dbgs() << "Verify if there is a pass dependency cycle." << "\n";
738 dbgs() << "Required Passes:" << "\n";
739 for (const AnalysisID ID2 : RequiredSet) {
740 if (ID == ID2)
741 break;
742 Pass *AnalysisPass2 = findAnalysisPass(ID2);
743 if (AnalysisPass2) {
744 dbgs() << "\t" << AnalysisPass2->getPassName() << "\n";
745 } else {
746 dbgs() << "\t" << "Error: Required pass not found! Possible causes:" << "\n";
747 dbgs() << "\t\t" << "- Pass misconfiguration (e.g.: missing macros)" << "\n";
748 dbgs() << "\t\t" << "- Corruption of the global PassRegistry" << "\n";
749 }
750 }
751 }
752
753 assert(PI && "Expected required passes to be initialized");
754 AnalysisPass = PI->createPass();
755 if (P->getPotentialPassManagerType () ==
756 AnalysisPass->getPotentialPassManagerType())
757 // Schedule analysis pass that is managed by the same pass manager.
758 schedulePass(AnalysisPass);
759 else if (P->getPotentialPassManagerType () >
760 AnalysisPass->getPotentialPassManagerType()) {
761 // Schedule analysis pass that is managed by a new manager.
762 schedulePass(AnalysisPass);
763 // Recheck analysis passes to ensure that required analyses that
764 // are already checked are still available.
765 checkAnalysis = true;
766 } else
767 // Do not schedule this analysis. Lower level analysis
768 // passes are run on the fly.
769 delete AnalysisPass;
770 }
771 }
772 }
773
774 // Now all required passes are available.
775 if (ImmutablePass *IP = P->getAsImmutablePass()) {
776 // P is a immutable pass and it will be managed by this
777 // top level manager. Set up analysis resolver to connect them.
778 PMDataManager *DM = getAsPMDataManager();
779 AnalysisResolver *AR = new AnalysisResolver(*DM);
780 P->setResolver(AR);
781 DM->initializeAnalysisImpl(P);
782 addImmutablePass(IP);
783 DM->recordAvailableAnalysis(IP);
784 return;
785 }
786
787 if ((PI && !PI->isAnalysis() && shouldPrintBeforePass(PI->getPassArgument())) ||
788 (!PI && PrintBeforeAll)) {
789 Pass *PP = P->createPrinterPass(
790 dbgs(), ("*** IR Dump Before " + P->getPassName() + " ***").str());
791 PP->assignPassManager(activeStack, getTopLevelPassManagerType());
792 }
793
794 // Add the requested pass to the best available pass manager.
795 P->assignPassManager(activeStack, getTopLevelPassManagerType());
796
797 if ((PI && !PI->isAnalysis() && shouldPrintAfterPass(PI->getPassArgument())) ||
798 (!PI && PrintAfterAll)) {
799 Pass *PP = P->createPrinterPass(
800 dbgs(), ("*** IR Dump After " + P->getPassName() + " ***").str());
801 PP->assignPassManager(activeStack, getTopLevelPassManagerType());
802 }
803}

1068/// Add pass P into the PassVector. Update
1069/// AvailableAnalysis appropriately if ProcessAnalysis is true.
1070void PMDataManager::add(Pass *P, bool ProcessAnalysis) {
1071 // This manager is going to manage pass P. Set up analysis resolver
1072 // to connect them.
1073 AnalysisResolver *AR = new AnalysisResolver(*this);
1074 P->setResolver(AR);
1075
1076 // If a FunctionPass F is the last user of ModulePass info M
1077 // then the F's manager, not F, records itself as a last user of M.
1078 SmallVector<Pass *, 12> TransferLastUses;
1079
1080 if (!ProcessAnalysis) {
1081 // Add pass
1082 PassVector.push_back(P);
1083 return;
1084 }
1085
1086 // At the moment, this pass is the last user of all required passes.
1087 SmallVector<Pass *, 12> LastUses;
1088 SmallVector<Pass *, 8> UsedPasses;
1089 SmallVector<AnalysisID, 8> ReqAnalysisNotAvailable;
1090
1091 unsigned PDepth = this->getDepth();
1092
1093 collectRequiredAndUsedAnalyses(UsedPasses, ReqAnalysisNotAvailable, P);
1094 for (Pass *PUsed : UsedPasses) {
1095 unsigned RDepth = 0;
1096
1097 assert(PUsed->getResolver() && "Analysis Resolver is not set");
1098 PMDataManager &DM = PUsed->getResolver()->getPMDataManager();
1099 RDepth = DM.getDepth();
1100
1101 if (PDepth == RDepth)
1102 LastUses.push_back(PUsed);
1103 else if (PDepth > RDepth) {
1104 // Let the parent claim responsibility of last use
1105 TransferLastUses.push_back(PUsed);
1106 // Keep track of higher level analysis used by this manager.
1107 HigherLevelAnalysis.push_back(PUsed);
1108 } else
1109 llvm_unreachable("Unable to accommodate Used Pass");
1110 }
1111
1112 // Set P as P's last user until someone starts using P.
1113 // However, if P is a Pass Manager then it does not need
1114 // to record its last user.
1115 if (!P->getAsPMDataManager())
1116 LastUses.push_back(P);
1117 TPM->setLastUser(LastUses, P);
1118
1119 if (!TransferLastUses.empty()) {
1120 Pass *My_PM = getAsPass();
1121 TPM->setLastUser(TransferLastUses, My_PM);
1122 TransferLastUses.clear();
1123 }
1124
1125 // Now, take care of required analyses that are not available.
1126 for (AnalysisID ID : ReqAnalysisNotAvailable) {
1127 const PassInfo *PI = TPM->findAnalysisPassInfo(ID);
1128 Pass *AnalysisPass = PI->createPass();
1129 this->addLowerLevelRequiredPass(P, AnalysisPass);
1130 }
1131
1132 // Take a note of analysis required and made available by this pass.
1133 // Remove the analysis not preserved by this pass
1134 removeNotPreservedAnalysis(P);
1135 recordAvailableAnalysis(P);
1136
1137 // Add pass
1138 PassVector.push_back(P);
1139}

1375/// Execute all of the passes scheduled for execution by invoking
1376/// runOnBasicBlock method. Keep track of whether any of the passes modifies
1377/// the function, and if so, return true.
1378bool BBPassManager::runOnFunction(Function &F) {
1379 if (F.isDeclaration())
1380 return false;
1381
1382 bool Changed = doInitialization(F);
1383 Module &M = *F.getParent();
1384
1385 unsigned InstrCount, BBSize = 0;
1386 StringMap<std::pair<unsigned, unsigned>> FunctionToInstrCount;
1387 bool EmitICRemark = M.shouldEmitInstrCountChangedRemark();
1388 if (EmitICRemark)
1389 InstrCount = initSizeRemarkInfo(M, FunctionToInstrCount);
1390
1391 for (BasicBlock &BB : F) {
1392 // Collect the initial size of the basic block.
1393 if (EmitICRemark)
1394 BBSize = BB.size();
1395 for (unsigned Index = 0; Index < getNumContainedPasses(); ++Index) {
1396 BasicBlockPass *BP = getContainedPass(Index);
1397 bool LocalChanged = false;
1398
1399 dumpPassInfo(BP, EXECUTION_MSG, ON_BASICBLOCK_MSG, BB.getName());
1400 dumpRequiredSet(BP);
1401
1402 initializeAnalysisImpl(BP);
1403
1404 {
1405 // If the pass crashes, remember this.
1406 PassManagerPrettyStackEntry X(BP, BB);
1407 TimeRegion PassTimer(getPassTimer(BP));
1408 LocalChanged |= BP->runOnBasicBlock(BB);

1416 emitInstrCountChangedRemark(BP, M, Delta, InstrCount,
1417 FunctionToInstrCount, &F);
1418 InstrCount = static_cast<int64_t>(InstrCount) + Delta;
1419 BBSize = NewSize;
1420 }
1421 }
1422 }
1423
1424 Changed |= LocalChanged;
1425 if (LocalChanged)
1426 dumpPassInfo(BP, MODIFICATION_MSG, ON_BASICBLOCK_MSG,
1427 BB.getName());
1428 dumpPreservedSet(BP);
1429 dumpUsedSet(BP);
1430
1431 verifyPreservedAnalysis(BP);
1432 removeNotPreservedAnalysis(BP);
1433 recordAvailableAnalysis(BP);
1434 removeDeadPasses(BP, BB.getName(), ON_BASICBLOCK_MSG);
1435 }
1436 }

1438 return doFinalization(F) || Changed;
1439}

1613/// Execute all of the passes scheduled for execution by invoking
1614/// runOnFunction method. Keep track of whether any of the passes modifies
1615/// the function, and if so, return true.
1616bool FPPassManager::runOnFunction(Function &F) {
1617 if (F.isDeclaration())
1618 return false;
1619
1620 bool Changed = false;
1621 Module &M = *F.getParent();
1622 // Collect inherited analysis from Module level pass manager.
1623 populateInheritedAnalysis(TPM->activeStack);
1624
1625 unsigned InstrCount, FunctionSize = 0;
1626 StringMap<std::pair<unsigned, unsigned>> FunctionToInstrCount;
1627 bool EmitICRemark = M.shouldEmitInstrCountChangedRemark();
1628 // Collect the initial size of the module.
1629 if (EmitICRemark) {
1630 InstrCount = initSizeRemarkInfo(M, FunctionToInstrCount);
1631 FunctionSize = F.getInstructionCount();
1632 }
1633
1634 llvm::TimeTraceScope FunctionScope("OptFunction", F.getName());
1635
1636 for (unsigned Index = 0; Index < getNumContainedPasses(); ++Index) {
1637 FunctionPass *FP = getContainedPass(Index);
1638 bool LocalChanged = false;
1639
1640 llvm::TimeTraceScope PassScope("RunPass", FP->getPassName());
1641
1642 dumpPassInfo(FP, EXECUTION_MSG, ON_FUNCTION_MSG, F.getName());
1643 dumpRequiredSet(FP);
1644
1645 initializeAnalysisImpl(FP);
1646
1647 {
1648 PassManagerPrettyStackEntry X(FP, F);
1649 TimeRegion PassTimer(getPassTimer(FP));
1650 LocalChanged |= FP->runOnFunction(F);
1651 if (EmitICRemark) {
1652 unsigned NewSize = F.getInstructionCount();
1653
1654 // Update the size of the function, emit a remark, and update the size
1655 // of the module.
1656 if (NewSize != FunctionSize) {
1657 int64_t Delta = static_cast<int64_t>(NewSize) -
1658 static_cast<int64_t>(FunctionSize);
1659 emitInstrCountChangedRemark(FP, M, Delta, InstrCount,
1660 FunctionToInstrCount, &F);
1661 InstrCount = static_cast<int64_t>(InstrCount) + Delta;
1662 FunctionSize = NewSize;
1663 }
1664 }
1665 }
1666
1667 Changed |= LocalChanged;
1668 if (LocalChanged)
1669 dumpPassInfo(FP, MODIFICATION_MSG, ON_FUNCTION_MSG, F.getName());
1670 dumpPreservedSet(FP);
1671 dumpUsedSet(FP);
1672
1673 verifyPreservedAnalysis(FP);
1674 removeNotPreservedAnalysis(FP);
1675 recordAvailableAnalysis(FP);
1676 removeDeadPasses(FP, F.getName(), ON_FUNCTION_MSG);
1677 }
1678
1679 return Changed;

1713/// Execute all of the passes scheduled for execution by invoking
1714/// runOnModule method. Keep track of whether any of the passes modifies
1715/// the module, and if so, return true.
1716bool
1717MPPassManager::runOnModule(Module &M) {
1718 llvm::TimeTraceScope TimeScope("OptModule", M.getName());
1719
1720 bool Changed = false;
1721
1722 // Initialize on-the-fly passes
1723 for (auto &OnTheFlyManager : OnTheFlyManagers) {
1724 FunctionPassManagerImpl *FPP = OnTheFlyManager.second;
1725 Changed |= FPP->doInitialization(M);
1726 }
1727
1728 // Initialize module passes
1729 for (unsigned Index = 0; Index < getNumContainedPasses(); ++Index)
1730 Changed |= getContainedPass(Index)->doInitialization(M);
1731
1732 unsigned InstrCount, ModuleCount = 0;
1733 StringMap<std::pair<unsigned, unsigned>> FunctionToInstrCount;
1734 bool EmitICRemark = M.shouldEmitInstrCountChangedRemark();
1735 // Collect the initial size of the module.
1736 if (EmitICRemark) {
1737 InstrCount = initSizeRemarkInfo(M, FunctionToInstrCount);
1738 ModuleCount = InstrCount;
1739 }
1740
1741 for (unsigned Index = 0; Index < getNumContainedPasses(); ++Index) {
1742 ModulePass *MP = getContainedPass(Index);
1743 bool LocalChanged = false;
1744
1745 dumpPassInfo(MP, EXECUTION_MSG, ON_MODULE_MSG, M.getModuleIdentifier());
1746 dumpRequiredSet(MP);
1747
1748 initializeAnalysisImpl(MP);
1749
1750 {
1751 PassManagerPrettyStackEntry X(MP, M);
1752 TimeRegion PassTimer(getPassTimer(MP));
1753
1754 LocalChanged |= MP->runOnModule(M);
1755 if (EmitICRemark) {
1756 // Update the size of the module.
1757 ModuleCount = M.getInstructionCount();
1758 if (ModuleCount != InstrCount) {
1759 int64_t Delta = static_cast<int64_t>(ModuleCount) -
1760 static_cast<int64_t>(InstrCount);
1761 emitInstrCountChangedRemark(MP, M, Delta, InstrCount,
1762 FunctionToInstrCount);
1763 InstrCount = ModuleCount;
1764 }
1765 }
1766 }
1767
1768 Changed |= LocalChanged;
1769 if (LocalChanged)
1770 dumpPassInfo(MP, MODIFICATION_MSG, ON_MODULE_MSG,
1771 M.getModuleIdentifier());
1772 dumpPreservedSet(MP);
1773 dumpUsedSet(MP);
1774
1775 verifyPreservedAnalysis(MP);
1776 removeNotPreservedAnalysis(MP);
1777 recordAvailableAnalysis(MP);
1778 removeDeadPasses(MP, M.getModuleIdentifier(), ON_MODULE_MSG);
1779 }
1780
1781 // Finalize module passes
1782 for (int Index = getNumContainedPasses() - 1; Index >= 0; --Index)
1783 Changed |= getContainedPass(Index)->doFinalization(M);
1784
1785 // Finalize on-the-fly passes
1786 for (auto &OnTheFlyManager : OnTheFlyManagers) {
1787 FunctionPassManagerImpl *FPP = OnTheFlyManager.second;
1788 // We don't know when is the last time an on-the-fly pass is run,
1789 // so we need to releaseMemory / finalize here
1790 FPP->releaseMemoryOnTheFly();
1791 Changed |= FPP->doFinalization(M);
1792 }
1793
1794 return Changed;
1795}

1797/// Add RequiredPass into list of lower level passes required by pass P.
1798/// RequiredPass is run on the fly by Pass Manager when P requests it
1799/// through getAnalysis interface.
1800void MPPassManager::addLowerLevelRequiredPass(Pass *P, Pass *RequiredPass) {
1801 assert(P->getPotentialPassManagerType() == PMT_ModulePassManager &&
1802 "Unable to handle Pass that requires lower level Analysis pass");
1803 assert((P->getPotentialPassManagerType() <
1804 RequiredPass->getPotentialPassManagerType()) &&
1805 "Unable to handle Pass that requires lower level Analysis pass");
1806 if (!RequiredPass)
1807 return;
1808
1809 FunctionPassManagerImpl *FPP = OnTheFlyManagers[P];
1810 if (!FPP) {
1811 FPP = new FunctionPassManagerImpl();
1812 // FPP is the top level manager.
1813 FPP->setTopLevelManager(FPP);
1814
1815 OnTheFlyManagers[P] = FPP;
1816 }
1817 const PassInfo *RequiredPassPI =
1818 TPM->findAnalysisPassInfo(RequiredPass->getPassID());
1819
1820 Pass *FoundPass = nullptr;
1821 if (RequiredPassPI && RequiredPassPI->isAnalysis()) {
1822 FoundPass =
1823 ((PMTopLevelManager*)FPP)->findAnalysisPass(RequiredPass->getPassID());
1824 }
1825 if (!FoundPass) {
1826 FoundPass = RequiredPass;
1827 // This should be guaranteed to add RequiredPass to the passmanager given
1828 // that we checked for an available analysis above.
1829 FPP->add(RequiredPass);
1830 }
1831 // Register P as the last user of FoundPass or RequiredPass.
1832 SmallVector<Pass *, 1> LU;
1833 LU.push_back(FoundPass);
1834 FPP->setLastUser(LU, P);
1835}

1947/// Find appropriate Module Pass Manager in the PM Stack and
1948/// add self into that manager.
1949void ModulePass::assignPassManager(PMStack &PMS,
1950 PassManagerType PreferredType) {
1951 // Find Module Pass Manager
1952 while (!PMS.empty()) {
1953 PassManagerType TopPMType = PMS.top()->getPassManagerType();
1954 if (TopPMType == PreferredType)
1955 break; // We found desired pass manager
1956 else if (TopPMType > PMT_ModulePassManager)
1957 PMS.pop(); // Pop children pass managers
1958 else
1959 break;
1960 }
1961 assert(!PMS.empty() && "Unable to find appropriate Pass Manager");
1962 PMS.top()->add(this);

1965/// Find appropriate Function Pass Manager or Call Graph Pass Manager
1966/// in the PM Stack and add self into that manager.
1967void FunctionPass::assignPassManager(PMStack &PMS,
1968 PassManagerType PreferredType) {
1969
1970 // Find Function Pass Manager
1971 while (!PMS.empty()) {
1972 if (PMS.top()->getPassManagerType() > PMT_FunctionPassManager)
1973 PMS.pop();
1974 else
1975 break;
1976 }
1977
1978 // Create new Function Pass Manager if needed.
1979 FPPassManager *FPP;
1980 if (PMS.top()->getPassManagerType() == PMT_FunctionPassManager) {
1981 FPP = (FPPassManager *)PMS.top();
1982 } else {
1983 assert(!PMS.empty() && "Unable to create Function Pass Manager");
1984 PMDataManager *PMD = PMS.top();
1985
1986 // [1] Create new Function Pass Manager
1987 FPP = new FPPassManager();
1988 FPP->populateInheritedAnalysis(PMS);
1989
1990 // [2] Set up new manager's top level manager
1991 PMTopLevelManager *TPM = PMD->getTopLevelManager();
1992 TPM->addIndirectPassManager(FPP);
1993
1994 // [3] Assign manager to manage this new manager. This may create
1995 // and push new managers into PMS
1996 FPP->assignPassManager(PMS, PMD->getPassManagerType());
1997
1998 // [4] Push new manager into PMS
1999 PMS.push(FPP);
2000 }
2001
2002 // Assign FPP as the manager of this pass.
2003 FPP->add(this);
2004}

2006/// Find appropriate Basic Pass Manager or Call Graph Pass Manager
2007/// in the PM Stack and add self into that manager.
2008void BasicBlockPass::assignPassManager(PMStack &PMS,
2009 PassManagerType PreferredType) {

2027 // Basic Block Pass Manager does not live by itself

2032 // and push new managers into PMS

2036 PMS.push(BBP);
2037 }
