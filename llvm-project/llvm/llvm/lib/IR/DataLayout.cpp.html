46StructLayout::StructLayout(StructType *ST, const DataLayout &DL) {
47 assert(!ST->isOpaque() && "Cannot get layout of opaque structs");
48 StructAlignment = 0;
49 StructSize = 0;
50 IsPadded = false;
51 NumElements = ST->getNumElements();
52
53 // Loop over each of the elements, placing them in memory.
54 for (unsigned i = 0, e = NumElements; i != e; ++i) {
55 Type *Ty = ST->getElementType(i);
56 unsigned TyAlign = ST->isPacked() ? 1 : DL.getABITypeAlignment(Ty);
57
58 // Add padding if necessary to align the data element properly.
59 if ((StructSize & (TyAlign-1)) != 0) {
60 IsPadded = true;
61 StructSize = alignTo(StructSize, TyAlign);
62 }
63
64 // Keep track of maximum alignment constraint.
65 StructAlignment = std::max(TyAlign, StructAlignment);
66
67 MemberOffsets[i] = StructSize;
68 StructSize += DL.getTypeAllocSize(Ty); // Consume space for this data item
69 }
70
71 // Empty structures have alignment of 1 byte.
72 if (StructAlignment == 0) StructAlignment = 1;
73
74 // Add padding to the end of the struct so that it could be put in an array
75 // and all array elements would be aligned correctly.
76 if ((StructSize & (StructAlignment-1)) != 0) {
77 IsPadded = true;
78 StructSize = alignTo(StructSize, StructAlignment);
79 }
80}

82/// getElementContainingOffset - Given a valid offset into the structure,
83/// return the structure index that contains it.
84unsigned StructLayout::getElementContainingOffset(uint64_t Offset) const {
85 const uint64_t *SI =
86 std::upper_bound(&MemberOffsets[0], &MemberOffsets[NumElements], Offset);
87 assert(SI != &MemberOffsets[0] && "Offset not in structure type!");
88 --SI;
89 assert(*SI <= Offset && "upper_bound didn't work");
90 assert((SI == &MemberOffsets[0] || *(SI-1) <= Offset) &&
91 (SI+1 == &MemberOffsets[NumElements] || *(SI+1) > Offset) &&
92 "Upper bound didn't work!");
93
94 // Multiple fields can have the same offset if any of them are zero sized.
95 // For example, in { i32, [0 x i32], i32 }, searching for offset 4 will stop
96 // at the i32 element, because it is the last element at that offset. This is
97 // the right one to return, because anything after it will have a higher
98 // offset, implying that this element is non-empty.
99 return SI-&MemberOffsets[0];
100}

106LayoutAlignElem
107LayoutAlignElem::get(AlignTypeEnum align_type, unsigned abi_align,
108 unsigned pref_align, uint32_t bit_width) {
109 assert(abi_align <= pref_align && "Preferred alignment worse than ABI!");
110 LayoutAlignElem retval;
111 retval.AlignType = align_type;
112 retval.ABIAlign = abi_align;
113 retval.PrefAlign = pref_align;
114 retval.TypeBitWidth = bit_width;
115 return retval;
116}

118bool
119LayoutAlignElem::operator==(const LayoutAlignElem &rhs) const {
120 return (AlignType == rhs.AlignType
121 && ABIAlign == rhs.ABIAlign
122 && PrefAlign == rhs.PrefAlign
123 && TypeBitWidth == rhs.TypeBitWidth);
124}

179void DataLayout::reset(StringRef Desc) {
180 clear();
181
182 LayoutMap = nullptr;
183 BigEndian = false;
184 AllocaAddrSpace = 0;
185 StackNaturalAlign = 0;
186 ProgramAddrSpace = 0;
187 GlobalsAddrSpace = 0;
188 FunctionPtrAlign = 0;
189 TheFunctionPtrAlignType = FunctionPtrAlignType::Independent;
190 ManglingMode = MM_None;
191 NonIntegralAddressSpaces.clear();
192
193 // Default alignments
194 for (const LayoutAlignElem &E : DefaultAlignments) {
195 setAlignment((AlignTypeEnum)E.AlignType, E.ABIAlign, E.PrefAlign,
196 E.TypeBitWidth);
197 }
198 setPointerAlignment(0, 8, 8, 8, 8, false);
199 parseSpecifier(Desc);
200}

202/// Checked version of split, to ensure mandatory subparts.
203static std::pair<StringRef, StringRef> split(StringRef Str, char Separator) {
204 assert(!Str.empty() && "parse error, string can't be empty here");
205 std::pair<StringRef, StringRef> Split = Str.split(Separator);
206 if (Split.second.empty() && Split.first != Str)
207 report_fatal_error("Trailing separator in datalayout string");
208 if (!Split.second.empty() && Split.first.empty())
209 report_fatal_error("Expected token before separator in datalayout string");
210 return Split;
211}
212
213/// Get an unsigned integer, including error checks.
214static unsigned getInt(StringRef R) {
215 unsigned Result;
216 bool error = R.getAsInteger(10, Result); (void)error;
217 if (error)
218 report_fatal_error("not a number, or does not fit in an unsigned int");
219 return Result;
220}
221
222/// Convert bits into bytes. Assert if not a byte width multiple.
223static unsigned inBytes(unsigned Bits) {
224 if (Bits % 8)
225 report_fatal_error("number of bits must be a byte width multiple");
226 return Bits / 8;
227}
228
229static unsigned getAddrSpace(StringRef R) {
230 unsigned AddrSpace = getInt(R);
231 if (!isUInt<24>(AddrSpace))
232 report_fatal_error("Invalid address space, must be a 24-bit integer");
233 return AddrSpace;
234}

236void DataLayout::parseSpecifier(StringRef Desc) {
237 StringRepresentation = Desc;
238 while (!Desc.empty()) {
239 // Split at '-'.
240 std::pair<StringRef, StringRef> Split = split(Desc, '-');
241 Desc = Split.second;
242
243 // Split at ':'.
244 Split = split(Split.first, ':');
245
246 // Aliases used below.
247 StringRef &Tok = Split.first; // Current token.
248 StringRef &Rest = Split.second; // The rest of the string.
249
250 if (Tok == "ni") {
251 do {
252 Split = split(Rest, ':');
253 Rest = Split.second;
254 unsigned AS = getInt(Split.first);
255 if (AS == 0)
256 report_fatal_error("Address space 0 can never be non-integral");
257 NonIntegralAddressSpaces.push_back(AS);
258 } while (!Rest.empty());
259
260 continue;
261 }
262
263 char Specifier = Tok.front();
264 Tok = Tok.substr(1);
265
266 switch (Specifier) {
267 case 's':
268 // Ignored for backward compatibility.
269 // FIXME: remove this on LLVM 4.0.
270 break;
271 case 'E':
272 BigEndian = true;
273 break;
274 case 'e':
275 BigEndian = false;
276 break;
277 case 'p': {
278 bool isFat = false;
279 if (!Tok.empty() && (Tok[0] == 'f')) {
280 isFat = true;
281 Tok = Tok.drop_front(1);
282 }
283 // Address space.
284 unsigned AddrSpace = Tok.empty() ? 0 : getInt(Tok);
285 if (!isUInt<24>(AddrSpace))
286 report_fatal_error("Invalid address space, must be a 24bit integer");
287 assert(isFat == isCheriPointer(AddrSpace, nullptr));
288
289 // Size.
290 if (Rest.empty())
291 report_fatal_error(
292 "Missing size specification for pointer in datalayout string");
293 Split = split(Rest, ':');
294 unsigned PointerMemSize = inBytes(getInt(Tok));
295 if (!PointerMemSize)
296 report_fatal_error("Invalid pointer size of 0 bytes");
297
298 // ABI alignment.
299 if (Rest.empty())
300 report_fatal_error(
301 "Missing alignment specification for pointer in datalayout string");
302 Split = split(Rest, ':');
303 unsigned PointerABIAlign = inBytes(getInt(Tok));
304 if (!isPowerOf2_64(PointerABIAlign))
305 report_fatal_error(
306 "Pointer ABI alignment must be a power of 2");
307
308 // Size of index used in GEP for address calculation.
309 // The parameter is optional. By default it is equal to size of pointer.
310 // XXXAR: For compatibility make isFat default to index width = 64 bits so
311 // we don't have to add the index width to the datalayout immediately
312 unsigned IndexSize = isFat ? inBytes(64) : PointerMemSize;
313
314 // Preferred alignment.
315 unsigned PointerPrefAlign = PointerABIAlign;
316 if (!Rest.empty()) {
317 Split = split(Rest, ':');
318 PointerPrefAlign = inBytes(getInt(Tok));
319 if (!isPowerOf2_64(PointerPrefAlign))
320 report_fatal_error(
321 "Pointer preferred alignment must be a power of 2");
322
323 // Now read the index. It is the second optional parameter here.
324 if (!Rest.empty()) {
325 Split = split(Rest, ':');
326 IndexSize = inBytes(getInt(Tok));
327 if (!IndexSize)
328 report_fatal_error("Invalid index size of 0 bytes");
329 }
330 }
331 setPointerAlignment(AddrSpace, PointerABIAlign, PointerPrefAlign,
332 PointerMemSize, IndexSize, isFat);
333 break;
334 }
335 case 'i':
336 case 'v':
337 case 'f':
338 case 'a': {
339 AlignTypeEnum AlignType;
340 switch (Specifier) {
341 default: llvm_unreachable("Unexpected specifier!");
342 case 'i': AlignType = INTEGER_ALIGN; break;
343 case 'v': AlignType = VECTOR_ALIGN; break;
344 case 'f': AlignType = FLOAT_ALIGN; break;
345 case 'a': AlignType = AGGREGATE_ALIGN; break;
346 }
347
348 // Bit size.
349 unsigned Size = Tok.empty() ? 0 : getInt(Tok);
350
351 if (AlignType == AGGREGATE_ALIGN && Size != 0)
352 report_fatal_error(
353 "Sized aggregate specification in datalayout string");
354
355 // ABI alignment.
356 if (Rest.empty())
357 report_fatal_error(
358 "Missing alignment specification in datalayout string");
359 Split = split(Rest, ':');
360 unsigned ABIAlign = inBytes(getInt(Tok));
361 if (AlignType != AGGREGATE_ALIGN && !ABIAlign)
362 report_fatal_error(
363 "ABI alignment specification must be >0 for non-aggregate types");
364
365 // Preferred alignment.
366 unsigned PrefAlign = ABIAlign;
367 if (!Rest.empty()) {
368 Split = split(Rest, ':');
369 PrefAlign = inBytes(getInt(Tok));
370 }
371
372 setAlignment(AlignType, ABIAlign, PrefAlign, Size);
373
374 break;
375 }
376 case 'n': // Native integer types.
377 while (true) {
378 unsigned Width = getInt(Tok);
379 if (Width == 0)
380 report_fatal_error(
381 "Zero width native integer type in datalayout string");
382 LegalIntWidths.push_back(Width);
383 if (Rest.empty())
384 break;
385 Split = split(Rest, ':');
386 }
387 break;
388 case 'S': { // Stack natural alignment.
389 StackNaturalAlign = inBytes(getInt(Tok));
390 break;
391 }
392 case 'F': {
393 switch (Tok.front()) {
394 case 'i':
395 TheFunctionPtrAlignType = FunctionPtrAlignType::Independent;
396 break;
397 case 'n':
398 TheFunctionPtrAlignType = FunctionPtrAlignType::MultipleOfFunctionAlign;
399 break;
400 default:
401 report_fatal_error("Unknown function pointer alignment type in "
402 "datalayout string");
403 }
404 Tok = Tok.substr(1);
405 FunctionPtrAlign = inBytes(getInt(Tok));
406 break;
407 }
408 case 'P': { // Function address space.
409 ProgramAddrSpace = getAddrSpace(Tok);
410 break;
411 }
412 case 'A': { // Default stack/alloca address space.
413 AllocaAddrSpace = getAddrSpace(Tok);
414 break;
415 }
416 case 'G': { // Default address space for global variables.
417 GlobalsAddrSpace = getAddrSpace(Tok);
418 break;
419 }
420 case 'm':
421 if (!Tok.empty())
422 report_fatal_error("Unexpected trailing characters after mangling specifier in datalayout string");
423 if (Rest.empty())
424 report_fatal_error("Expected mangling specifier in datalayout string");
425 if (Rest.size() > 1)
426 report_fatal_error("Unknown mangling specifier in datalayout string");
427 switch(Rest[0]) {
428 default:
429 report_fatal_error("Unknown mangling in datalayout string");
430 case 'e':
431 ManglingMode = MM_ELF;
432 break;
433 case 'o':
434 ManglingMode = MM_MachO;
435 break;
436 case 'm':
437 ManglingMode = MM_Mips;
438 break;
439 case 'w':
440 ManglingMode = MM_WinCOFF;
441 break;
442 case 'x':
443 ManglingMode = MM_WinCOFFX86;
444 break;
445 }
446 break;
447 default:
448 report_fatal_error("Unknown specifier in datalayout string");
449 break;
450 }
451 }
452}

460bool DataLayout::operator==(const DataLayout &Other) const {
461 bool Ret = BigEndian == Other.BigEndian &&
462 AllocaAddrSpace == Other.AllocaAddrSpace &&
463 StackNaturalAlign == Other.StackNaturalAlign &&
464 ProgramAddrSpace == Other.ProgramAddrSpace &&
465 FunctionPtrAlign == Other.FunctionPtrAlign &&
466 TheFunctionPtrAlignType == Other.TheFunctionPtrAlignType &&
467 ManglingMode == Other.ManglingMode &&
468 LegalIntWidths == Other.LegalIntWidths &&
469 Alignments == Other.Alignments && Pointers == Other.Pointers;
470 // Note: getStringRepresentation() might differs, it is not canonicalized
471 return Ret;
472}

474DataLayout::AlignmentsTy::iterator
475DataLayout::findAlignmentLowerBound(AlignTypeEnum AlignType,
476 uint32_t BitWidth) {
477 auto Pair = std::make_pair((unsigned)AlignType, BitWidth);
478 return std::lower_bound(Alignments.begin(), Alignments.end(), Pair,
479 [](const LayoutAlignElem &LHS,
480 const std::pair<unsigned, uint32_t> &RHS) {
481 return std::tie(LHS.AlignType, LHS.TypeBitWidth) <
482 std::tie(RHS.first, RHS.second);
483 });
484}

486void
487DataLayout::setAlignment(AlignTypeEnum align_type, unsigned abi_align,
488 unsigned pref_align, uint32_t bit_width) {
489 if (!isUInt<24>(bit_width))
490 report_fatal_error("Invalid bit width, must be a 24bit integer");
491 if (!isUInt<16>(abi_align))
492 report_fatal_error("Invalid ABI alignment, must be a 16bit integer");
493 if (!isUInt<16>(pref_align))
494 report_fatal_error("Invalid preferred alignment, must be a 16bit integer");
495 if (abi_align != 0 && !isPowerOf2_64(abi_align))
496 report_fatal_error("Invalid ABI alignment, must be a power of 2");
497 if (pref_align != 0 && !isPowerOf2_64(pref_align))
498 report_fatal_error("Invalid preferred alignment, must be a power of 2");
499
500 if (pref_align < abi_align)
501 report_fatal_error(
502 "Preferred alignment cannot be less than the ABI alignment");
503
504 AlignmentsTy::iterator I = findAlignmentLowerBound(align_type, bit_width);
505 if (I != Alignments.end() &&
506 I->AlignType == (unsigned)align_type && I->TypeBitWidth == bit_width) {
507 // Update the abi, preferred alignments.
508 I->ABIAlign = abi_align;
509 I->PrefAlign = pref_align;
510 } else {
511 // Insert before I to keep the vector sorted.
512 Alignments.insert(I, LayoutAlignElem::get(align_type, abi_align,
513 pref_align, bit_width));
514 }
515}

517DataLayout::PointersTy::iterator
518DataLayout::findPointerLowerBound(uint32_t AddressSpace) {
519 return std::lower_bound(Pointers.begin(), Pointers.end(), AddressSpace,
520 [](const PointerAlignElem &A, uint32_t AddressSpace) {
521 return A.AddressSpace < AddressSpace;
522 });
523}

525void DataLayout::setPointerAlignment(uint32_t AddrSpace, unsigned ABIAlign,
526 unsigned PrefAlign, uint32_t TypeByteWidth,
527 uint32_t IndexWidth, bool IsFatPointer) {
528 if (PrefAlign < ABIAlign)
529 report_fatal_error(
530 "Preferred alignment cannot be less than the ABI alignment");
531
532 PointersTy::iterator I = findPointerLowerBound(AddrSpace);
533 if (I == Pointers.end() || I->AddressSpace != AddrSpace) {
534 Pointers.insert(I, PointerAlignElem::get(AddrSpace, ABIAlign, PrefAlign,
535 TypeByteWidth, IndexWidth,
536 IsFatPointer));
537 } else {
538 I->ABIAlign = ABIAlign;
539 I->PrefAlign = PrefAlign;
540 I->TypeByteWidth = TypeByteWidth;
541 I->IndexWidth = IndexWidth;
542 I->IsFatPointer = IsFatPointer;
543 }
544}

546/// getAlignmentInfo - Return the alignment (either ABI if ABIInfo = true or
547/// preferred if ABIInfo = false) the layout wants for the specified datatype.
548unsigned DataLayout::getAlignmentInfo(AlignTypeEnum AlignType,
549 uint32_t BitWidth, bool ABIInfo,
550 Type *Ty) const {
551 AlignmentsTy::const_iterator I = findAlignmentLowerBound(AlignType, BitWidth);
552 // See if we found an exact match. Of if we are looking for an integer type,
553 // but don't have an exact match take the next largest integer. This is where
554 // the lower_bound will point to when it fails an exact match.
555 if (I != Alignments.end() && I->AlignType == (unsigned)AlignType &&
556 (I->TypeBitWidth == BitWidth || AlignType == INTEGER_ALIGN))
557 return ABIInfo ? I->ABIAlign : I->PrefAlign;
558
559 if (AlignType == INTEGER_ALIGN) {
560 // If we didn't have a larger value try the largest value we have.
561 if (I != Alignments.begin()) {
562 --I; // Go to the previous entry and see if its an integer.
563 if (I->AlignType == INTEGER_ALIGN)
564 return ABIInfo ? I->ABIAlign : I->PrefAlign;
565 }
566 } else if (AlignType == VECTOR_ALIGN) {
567 // By default, use natural alignment for vector types. This is consistent
568 // with what clang and llvm-gcc do.
569 unsigned Align = getTypeAllocSize(cast<VectorType>(Ty)->getElementType());
570 Align *= cast<VectorType>(Ty)->getNumElements();
571 Align = PowerOf2Ceil(Align);
572 return Align;
573 }
574
575 // If we still couldn't find a reasonable default alignment, fall back
576 // to a simple heuristic that the alignment is the first power of two
577 // greater-or-equal to the store size of the type. This is a reasonable
578 // approximation of reality, and if the user wanted something less
579 // less conservative, they should have specified it explicitly in the data
580 // layout.
581 unsigned Align = getTypeStoreSize(Ty);
582 Align = PowerOf2Ceil(Align);
583 return Align;
584}

621const StructLayout *DataLayout::getStructLayout(StructType *Ty) const {
622 if (!LayoutMap)
623 LayoutMap = new StructLayoutMap();
624
625 StructLayoutMap *STM = static_cast<StructLayoutMap*>(LayoutMap);
626 StructLayout *&SL = (*STM)[Ty];
627 if (SL) return SL;
628
629 // Otherwise, create the struct layout. Because it is variable length, we
630 // malloc it, then use placement new.
631 int NumElts = Ty->getNumElements();
632 StructLayout *L = (StructLayout *)
633 safe_malloc(sizeof(StructLayout)+(NumElts-1) * sizeof(uint64_t));
634
635 // Set SL before calling StructLayout's ctor. The ctor could cause other
636 // entries to be added to TheMap, invalidating our reference.
637 SL = L;
638
639 new (L) StructLayout(Ty, *this);
640
641 return L;
642}

711/*!
712 \param abi_or_pref Flag that determines which alignment is returned. true
713 returns the ABI alignment, false returns the preferred alignment.
714 \param Ty The underlying type for which alignment is determined.
715
716 Get the ABI (\a abi_or_pref == true) or preferred alignment (\a abi_or_pref
717 == false) for the requested type \a Ty.
718 */
719unsigned DataLayout::getAlignment(Type *Ty, bool abi_or_pref) const {
720 AlignTypeEnum AlignType;
721
722 assert(Ty->isSized() && "Cannot getTypeInfo() on a type that is unsized!");
723 switch (Ty->getTypeID()) {
724 // Early escape for the non-numeric types.
725 case Type::LabelTyID:
726 return (abi_or_pref
727 ? getPointerABIAlignment(0)
728 : getPointerPrefAlignment(0));
729 case Type::PointerTyID: {
730 unsigned AS = cast<PointerType>(Ty)->getAddressSpace();
731 return (abi_or_pref
732 ? getPointerABIAlignment(AS)
733 : getPointerPrefAlignment(AS));
734 }
735 case Type::ArrayTyID:
736 return getAlignment(cast<ArrayType>(Ty)->getElementType(), abi_or_pref);
737
738 case Type::StructTyID: {
739 // Packed structure types always have an ABI alignment of one.
740 if (cast<StructType>(Ty)->isPacked() && abi_or_pref)
741 return 1;
742
743 // Get the layout annotation... which is lazily created on demand.
744 const StructLayout *Layout = getStructLayout(cast<StructType>(Ty));
745 unsigned Align = getAlignmentInfo(AGGREGATE_ALIGN, 0, abi_or_pref, Ty);
746 return std::max(Align, Layout->getAlignment());
747 }
748 case Type::IntegerTyID:
749 AlignType = INTEGER_ALIGN;
750 break;
751 case Type::HalfTyID:
752 case Type::FloatTyID:
753 case Type::DoubleTyID:
754 // PPC_FP128TyID and FP128TyID have different data contents, but the
755 // same size and alignment, so they look the same here.
756 case Type::PPC_FP128TyID:
757 case Type::FP128TyID:
758 case Type::X86_FP80TyID:
759 AlignType = FLOAT_ALIGN;
760 break;
761 case Type::X86_MMXTyID:
762 case Type::VectorTyID:
763 AlignType = VECTOR_ALIGN;
764 break;
765 default:
766 llvm_unreachable("Bad type for getAlignment!!!");
767 }
768
769 return getAlignmentInfo(AlignType, getTypeSizeInBits(Ty), abi_or_pref, Ty);
770}

772unsigned DataLayout::getABITypeAlignment(Type *Ty) const {
773 return getAlignment(Ty, true);
774}

829int64_t DataLayout::getIndexedOffsetInType(Type *ElemTy,
830 ArrayRef<Value *> Indices) const {
831 int64_t Result = 0;
832
833 generic_gep_type_iterator<Value* const*>
834 GTI = gep_type_begin(ElemTy, Indices),
835 GTE = gep_type_end(ElemTy, Indices);
836 for (; GTI != GTE; ++GTI) {
837 Value *Idx = GTI.getOperand();
838 if (StructType *STy = GTI.getStructTypeOrNull()) {
839 assert(Idx->getType()->isIntegerTy(32) && "Illegal struct idx");
840 unsigned FieldNo = cast<ConstantInt>(Idx)->getZExtValue();
841
842 // Get structure layout information...
843 const StructLayout *Layout = getStructLayout(STy);
844
845 // Add in the offset, as calculated by the structure layout info...
846 Result += Layout->getElementOffset(FieldNo);
847 } else {
848 // Get the array index and the size of each array element.
849 if (int64_t arrayIdx = cast<ConstantInt>(Idx)->getSExtValue())
850 Result += arrayIdx * getTypeAllocSize(GTI.getIndexedType());
851 }
852 }
853
854 return Result;
855}

857/// getPreferredAlignment - Return the preferred alignment of the specified
858/// global. This includes an explicitly requested alignment (if the global
859/// has one).
860unsigned DataLayout::getPreferredAlignment(const GlobalVariable *GV) const {
861 unsigned GVAlignment = GV->getAlignment();
862 // If a section is specified, always precisely honor explicit alignment,
863 // so we don't insert padding into a section we don't control.
864 if (GVAlignment && GV->hasSection())
865 return GVAlignment;
866
867 // If no explicit alignment is specified, compute the alignment based on
868 // the IR type. If an alignment is specified, increase it to match the ABI
869 // alignment of the IR type.
870 //
871 // FIXME: Not sure it makes sense to use the alignment of the type if
872 // there's already an explicit alignment specification.
873 Type *ElemType = GV->getValueType();
874 unsigned Alignment = getPrefTypeAlignment(ElemType);
875 if (GVAlignment >= Alignment) {
876 Alignment = GVAlignment;
877 } else if (GVAlignment != 0) {
878 Alignment = std::max(GVAlignment, getABITypeAlignment(ElemType));
879 }
880
881 // If no explicit alignment is specified, and the global is large, increase
882 // the alignment to 16.
883 // FIXME: Why 16, specifically?
884 if (GV->hasInitializer() && GVAlignment == 0) {
885 if (Alignment < 16) {
886 // If the global is not external, see if it is large. If so, give it a
887 // larger alignment.
888 if (getTypeSizeInBits(ElemType) > 128)
889 Alignment = 16; // 16-byte alignment.
890 }
891 }
892 return Alignment;
893}

895/// getPreferredAlignmentLog - Return the preferred alignment of the
896/// specified global, returned in log form. This includes an explicitly
897/// requested alignment (if the global has one).
898unsigned DataLayout::getPreferredAlignmentLog(const GlobalVariable *GV) const {
899 return Log2_32(getPreferredAlignment(GV));
