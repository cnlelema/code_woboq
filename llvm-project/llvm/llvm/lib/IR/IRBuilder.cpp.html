38/// CreateGlobalString - Make a new global variable with an initializer that
39/// has array of i8 type filled in with the nul terminated string value
40/// specified. If Name is specified, it is the name of the global variable
41/// created.
42GlobalVariable *IRBuilderBase::CreateGlobalString(StringRef Str,
43 const Twine &Name,
44 unsigned AddressSpace) {
45 Constant *StrConstant = ConstantDataArray::getString(Context, Str);
46 Module &M = *BB->getParent()->getParent();
47 auto *GV = new GlobalVariable(M, StrConstant->getType(), true,
48 GlobalValue::PrivateLinkage, StrConstant, Name,
49 nullptr, GlobalVariable::NotThreadLocal,
50 AddressSpace);
51 GV->setUnnamedAddr(GlobalValue::UnnamedAddr::Global);
52 GV->setAlignment(1);
53 return GV;
54}

61Value *IRBuilderBase::getCastedInt8PtrValue(Value *Ptr, unsigned TargetAS) {
62 PointerType *PT = cast<PointerType>(Ptr->getType());
63 unsigned AS = PT->getAddressSpace();
64 if (PT->getElementType()->isIntegerTy(8) && AS == TargetAS)
65 return Ptr;
66
67 // Otherwise, we need to insert a bitcast.
68 PT = getInt8PtrTy(TargetAS);
69 Instruction *I;
70 if (TargetAS == AS)
71 I = new BitCastInst(Ptr, PT, "");
72 else
73 I = new AddrSpaceCastInst(Ptr, PT, "");
74 BB->getInstList().insert(InsertPt, I);
75 SetInstDebugLocation(I);
76 return I;
77}
78
79Value *IRBuilderBase::getCastedInt8PtrValue(Value *Ptr) {
80 auto *PT = cast<PointerType>(Ptr->getType());
81 if (PT->getElementType()->isIntegerTy(8))
82 return Ptr;
83
84 // Otherwise, we need to insert a bitcast.
85 PT = getInt8PtrTy(PT->getAddressSpace());
86 BitCastInst *BCI = new BitCastInst(Ptr, PT, "");
87 BB->getInstList().insert(InsertPt, BCI);
88 SetInstDebugLocation(BCI);
89 return BCI;
90}

92static CallInst *createCallHelper(Function *Callee, ArrayRef<Value *> Ops,
93 IRBuilderBase *Builder,
94 const Twine &Name = "",
95 Instruction *FMFSource = nullptr) {
96 CallInst *CI = CallInst::Create(Callee, Ops, Name);
97 if (FMFSource)
98 CI->copyFastMathFlags(FMFSource);
99 Builder->GetInsertBlock()->getInstList().insert(Builder->GetInsertPoint(),CI);
100 Builder->SetInstDebugLocation(CI);
101 return CI;
102}

117CallInst *IRBuilderBase::
118CreateMemSet(Value *Ptr, Value *Val, Value *Size, unsigned Align,
119 bool isVolatile, MDNode *TBAATag, MDNode *ScopeTag,
120 MDNode *NoAliasTag) {
121 Ptr = getCastedInt8PtrValue(Ptr);
122 Value *Ops[] = {Ptr, Val, Size, getInt1(isVolatile)};
123 Type *Tys[] = { Ptr->getType(), Size->getType() };
124 Module *M = BB->getParent()->getParent();
125 Function *TheFn = Intrinsic::getDeclaration(M, Intrinsic::memset, Tys);
126
127 CallInst *CI = createCallHelper(TheFn, Ops, this);
128
129 if (Align > 0)
130 cast<MemSetInst>(CI)->setDestAlignment(Align);
131
132 // Set the TBAA info if present.
133 if (TBAATag)
134 CI->setMetadata(LLVMContext::MD_tbaa, TBAATag);
135
136 if (ScopeTag)
137 CI->setMetadata(LLVMContext::MD_alias_scope, ScopeTag);
138
139 if (NoAliasTag)
140 CI->setMetadata(LLVMContext::MD_noalias, NoAliasTag);
141
142 return CI;
143}

196

478/// Create a call to a Masked Load intrinsic.
479/// \p Ptr - base pointer for the load
480/// \p Align - alignment of the source location
481/// \p Mask - vector of booleans which indicates what vector lanes should
482/// be accessed in memory
483/// \p PassThru - pass-through value that is used to fill the masked-off lanes
484/// of the result
485/// \p Name - name of the result variable
486CallInst *IRBuilderBase::CreateMaskedLoad(Value *Ptr, unsigned Align,
487 Value *Mask, Value *PassThru,
488 const Twine &Name) {
489 auto *PtrTy = cast<PointerType>(Ptr->getType());
490 Type *DataTy = PtrTy->getElementType();
491 assert(DataTy->isVectorTy() && "Ptr should point to a vector");
492 assert(Mask && "Mask should not be all-ones (null)");
493 if (!PassThru)
494 PassThru = UndefValue::get(DataTy);
495 Type *OverloadedTypes[] = { DataTy, PtrTy };
496 Value *Ops[] = { Ptr, getInt32(Align), Mask, PassThru};
497 return CreateMaskedIntrinsic(Intrinsic::masked_load, Ops,
498 OverloadedTypes, Name);
499}

530/// Create a call to a Masked Gather intrinsic.
531/// \p Ptrs - vector of pointers for loading
532/// \p Align - alignment for one element
533/// \p Mask - vector of booleans which indicates what vector lanes should
534/// be accessed in memory
535/// \p PassThru - pass-through value that is used to fill the masked-off lanes
536/// of the result
537/// \p Name - name of the result variable
538CallInst *IRBuilderBase::CreateMaskedGather(Value *Ptrs, unsigned Align,
539 Value *Mask, Value *PassThru,
540 const Twine& Name) {
541 auto PtrsTy = cast<VectorType>(Ptrs->getType());
542 auto PtrTy = cast<PointerType>(PtrsTy->getElementType());
543 unsigned NumElts = PtrsTy->getVectorNumElements();
544 Type *DataTy = VectorType::get(PtrTy->getElementType(), NumElts);
545
546 if (!Mask)
547 Mask = Constant::getAllOnesValue(VectorType::get(Type::getInt1Ty(Context),
548 NumElts));
549
550 if (!PassThru)
551 PassThru = UndefValue::get(DataTy);
552
553 Type *OverloadedTypes[] = {DataTy, PtrsTy};
554 Value * Ops[] = {Ptrs, getInt32(Align), Mask, PassThru};
555
556 // We specify only one type when we create this intrinsic. Types of other
557 // arguments are derived from this type.
558 return CreateMaskedIntrinsic(Intrinsic::masked_gather, Ops, OverloadedTypes,
559 Name);
560}

562/// Create a call to a Masked Scatter intrinsic.
563/// \p Data - data to be stored,
564/// \p Ptrs - the vector of pointers, where the \p Data elements should be
565/// stored
566/// \p Align - alignment for one element
567/// \p Mask - vector of booleans which indicates what vector lanes should
568/// be accessed in memory
569CallInst *IRBuilderBase::CreateMaskedScatter(Value *Data, Value *Ptrs,
570 unsigned Align, Value *Mask) {
571 auto PtrsTy = cast<VectorType>(Ptrs->getType());
572 auto DataTy = cast<VectorType>(Data->getType());
573 unsigned NumElts = PtrsTy->getVectorNumElements();
574
575#ifndef NDEBUG
576 auto PtrTy = cast<PointerType>(PtrsTy->getElementType());
577 assert(NumElts == DataTy->getVectorNumElements() &&
578 PtrTy->getElementType() == DataTy->getElementType() &&
579 "Incompatible pointer and data types");
580#endif
581
582 if (!Mask)
583 Mask = Constant::getAllOnesValue(VectorType::get(Type::getInt1Ty(Context),
584 NumElts));
585
586 Type *OverloadedTypes[] = {DataTy, PtrsTy};
587 Value * Ops[] = {Data, Ptrs, getInt32(Align), Mask};
588
589 // We specify only one type when we create this intrinsic. Types of other
590 // arguments are derived from this type.
591 return CreateMaskedIntrinsic(Intrinsic::masked_scatter, Ops, OverloadedTypes);
592}

594template <typename T0, typename T1, typename T2, typename T3>
595static std::vector<Value *>
596getStatepointArgs(IRBuilderBase &B, uint64_t ID, uint32_t NumPatchBytes,
597 Value *ActualCallee, uint32_t Flags, ArrayRef<T0> CallArgs,
598 ArrayRef<T1> TransitionArgs, ArrayRef<T2> DeoptArgs,
599 ArrayRef<T3> GCArgs) {
600 std::vector<Value *> Args;
601 Args.push_back(B.getInt64(ID));
602 Args.push_back(B.getInt32(NumPatchBytes));
603 Args.push_back(ActualCallee);
604 Args.push_back(B.getInt32(CallArgs.size()));
605 Args.push_back(B.getInt32(Flags));
606 Args.insert(Args.end(), CallArgs.begin(), CallArgs.end());
607 Args.push_back(B.getInt32(TransitionArgs.size()));
608 Args.insert(Args.end(), TransitionArgs.begin(), TransitionArgs.end());
609 Args.push_back(B.getInt32(DeoptArgs.size()));
610 Args.insert(Args.end(), DeoptArgs.begin(), DeoptArgs.end());
611 Args.insert(Args.end(), GCArgs.begin(), GCArgs.end());
612
613 return Args;
614}
