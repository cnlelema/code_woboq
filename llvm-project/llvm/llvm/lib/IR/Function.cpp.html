254static unsigned computeAddrSpace(unsigned AddrSpace, Module *M) {
255 // If AS == -1 and we are passed a valid module pointer we place the function
256 // in the program address space and otherwise we default to AS0
257 // TODO: assert(M || AddrSpace != static_cast<unsigned>(-1));
258 if (AddrSpace == static_cast<unsigned>(-1)) {
259 assert(M && "Need either Module* or an explicit address space");
260 return M ? DefaultAddrSpaceForFunctions(*M) : 0;
261 }
262 if (IgnoreProgramASForFunctions)
263 return 0;
264 return AddrSpace;
265}

267Function::Function(FunctionType *Ty, LinkageTypes Linkage, unsigned AddrSpace,
268 const Twine &name, Module *ParentModule)
269 : GlobalObject(Ty, Value::FunctionVal,
270 OperandTraits<Function>::op_begin(this), 0, Linkage, name,
271 computeAddrSpace(AddrSpace, ParentModule)),
272 NumArgs(Ty->getNumParams()) {
273 assert(FunctionType::isValidReturnType(getReturnType()) &&
274 "invalid return type");
275 setGlobalObjectSubClassData(0);
276
277 // We only need a symbol table for a function if the context keeps value names
278 if (!getContext().shouldDiscardValueNames())
279 SymTab = make_unique<ValueSymbolTable>();
280
281 // If the function has arguments, mark them as lazily built.
282 if (Ty->getNumParams())
283 setValueSubclassData(1); // Set the "has lazy arguments" bit.
284
285 if (ParentModule)
286 ParentModule->getFunctionList().push_back(this);
287
288 HasLLVMReservedName = getName().startswith("llvm.");
289 // Ensure intrinsics have the right parameter attributes.
290 // Note, the IntID field will have been set in Value::setName if this function
291 // name is a valid intrinsic ID.
292 if (IntID)
293 setAttributes(Intrinsic::getAttributes(getContext(), IntID));
294}

296Function::~Function() {
297 dropAllReferences(); // After this it is safe to delete instructions.
298
299 // Delete all of the method arguments and unlink from symbol table...
300 if (Arguments)
301 clearArguments();
302
303 // Remove the function from the on-the-side GC table.
304 clearGC();
305}

307void Function::BuildLazyArguments() const {
308 // Create the arguments vector, all arguments start out unnamed.
309 auto *FT = getFunctionType();
310 if (NumArgs > 0) {
311 Arguments = std::allocator<Argument>().allocate(NumArgs);
312 for (unsigned i = 0, e = NumArgs; i != e; ++i) {
313 Type *ArgTy = FT->getParamType(i);
314 assert(!ArgTy->isVoidTy() && "Cannot have void typed arguments!");
315 new (Arguments + i) Argument(ArgTy, "", const_cast<Function *>(this), i);
316 }
317 }
318
319 // Clear the lazy arguments bit.
320 unsigned SDC = getSubclassDataFromValue();
321 const_cast<Function*>(this)->setValueSubclassData(SDC &= ~(1<<0));
322 assert(!hasLazyArguments());
323}

338void Function::stealArgumentListFrom(Function &Src) {
339 assert(isDeclaration() && "Expected no references to current arguments");
340
341 // Drop the current arguments, if any, and set the lazy argument bit.
342 if (!hasLazyArguments()) {
343 assert(llvm::all_of(makeArgArray(Arguments, NumArgs),
344 [](const Argument &A) { return A.use_empty(); }) &&
345 "Expected arguments to be unused in declaration");
346 clearArguments();
347 setValueSubclassData(getSubclassDataFromValue() | (1 << 0));
348 }
349
350 // Nothing to steal if Src has lazy arguments.
351 if (Src.hasLazyArguments())
352 return;
353
354 // Steal arguments from Src, and fix the lazy argument bits.
355 assert(arg_size() == Src.arg_size());
356 Arguments = Src.Arguments;
357 Src.Arguments = nullptr;
358 for (Argument &A : makeArgArray(Arguments, NumArgs)) {
359 // FIXME: This does the work of transferNodesFromList inefficiently.
360 SmallString<128> Name;
361 if (A.hasName())
362 Name = A.getName();
363 if (!Name.empty())
364 A.setName("");
365 A.setParent(this);
366 if (!Name.empty())
367 A.setName(Name);
368 }
369
370 setValueSubclassData(getSubclassDataFromValue() & ~(1 << 0));
371 assert(!hasLazyArguments());
372 Src.setValueSubclassData(Src.getSubclassDataFromValue() | (1 << 0));
373}

375// dropAllReferences() - This function causes all the subinstructions to "let
376// go" of all references that they are maintaining. This allows one to
377// 'delete' a whole class at a time, even though there may be circular
378// references... first all references are dropped, and all use counts go to
379// zero. Then everything is deleted for real. Note that no operations are
380// valid on an object that has "dropped all references", except operator
381// delete.
382//
383void Function::dropAllReferences() {
384 setIsMaterializable(false);
385
386 for (BasicBlock &BB : *this)
387 BB.dropAllReferences();
388
389 // Delete all basic blocks. They are now unused, except possibly by
390 // blockaddresses, but BasicBlock's destructor takes care of those.
391 while (!BasicBlocks.empty())
392 BasicBlocks.begin()->eraseFromParent();
393
394 // Drop uses of any optional data (real or placeholder).
395 if (getNumOperands()) {
396 User::dropAllReferences();
397 setNumHungOffUseOperands(0);
398 setValueSubclassData(getSubclassDataFromValue() & ~0xe);
399 }
400
401 // Metadata is stored in a side-table.
402 clearMetadata();
403}

550/// Find the segment of \c IntrinsicNameTable for intrinsics with the same
551/// target as \c Name, or the generic table if \c Name is not target specific.
552///
553/// Returns the relevant slice of \c IntrinsicNameTable
554static ArrayRef<const char *> findTargetSubtable(StringRef Name) {
555 assert(Name.startswith("llvm."));
556
557 ArrayRef<IntrinsicTargetInfo> Targets(TargetInfos);
558 // Drop "llvm." and take the first dotted component. That will be the target
559 // if this is target specific.
560 StringRef Target = Name.drop_front(5).split('.').first;
561 auto It = std::lower_bound(Targets.begin(), Targets.end(), Target,
562 [](const IntrinsicTargetInfo &TI,
563 StringRef Target) { return TI.Name < Target; });
564 // We've either found the target or just fall back to the generic set, which
565 // is always first.
566 const auto &TI = It != Targets.end() && It->Name == Target ? *It : Targets[0];
567 return makeArrayRef(&IntrinsicNameTable[1] + TI.Offset, TI.Count);
568}

570/// This does the actual lookup of an intrinsic ID which
571/// matches the given function name.
572Intrinsic::ID Function::lookupIntrinsicID(StringRef Name) {
573 ArrayRef<const char *> NameTable = findTargetSubtable(Name);
574 int Idx = Intrinsic::lookupLLVMIntrinsicByName(NameTable, Name);
575 if (Idx == -1)
576 return Intrinsic::not_intrinsic;
577
578 // Intrinsic IDs correspond to the location in IntrinsicNameTable, but we have
579 // an index into a sub-table.
580 int Adjust = NameTable.data() - IntrinsicNameTable;
581 Intrinsic::ID ID = static_cast<Intrinsic::ID>(Idx + Adjust);
582
583 // If the intrinsic is not overloaded, require an exact match. If it is
584 // overloaded, require either exact or prefix match.
585 const auto MatchSize = strlen(NameTable[Idx]);
586 assert(Name.size() >= MatchSize && "Expected either exact or prefix match");
587 bool IsExactMatch = Name.size() == MatchSize;
588 return IsExactMatch || isOverloaded(ID) ? ID : Intrinsic::not_intrinsic;
589}

602/// Returns a stable mangling for the type specified for use in the name
603/// mangling scheme used by 'any' types in intrinsic signatures. The mangling
604/// of named types is simply their name. Manglings for unnamed types consist
605/// of a prefix ('p' for pointers, 'a' for arrays, 'f_' for functions)
606/// combined with the mangling of their component types. A vararg function
607/// type will have a suffix of 'vararg'. Since function types can contain
608/// other function types, we close a function type mangling with suffix 'f'
609/// which can't be confused with it's prefix. This ensures we don't have
610/// collisions between two unrelated function types. Otherwise, you might
611/// parse ffXX as f(fXX) or f(fX)X. (X is a placeholder for any other type.)
612///
613static std::string getMangledTypeStr(Type* Ty) {
614 std::string Result;
615 if (PointerType* PTyp = dyn_cast<PointerType>(Ty)) {
616 Result += "p" + utostr(PTyp->getAddressSpace()) +
617 getMangledTypeStr(PTyp->getElementType());
618 } else if (ArrayType* ATyp = dyn_cast<ArrayType>(Ty)) {
619 Result += "a" + utostr(ATyp->getNumElements()) +
620 getMangledTypeStr(ATyp->getElementType());
621 } else if (StructType *STyp = dyn_cast<StructType>(Ty)) {
622 if (!STyp->isLiteral()) {
623 Result += "s_";
624 Result += STyp->getName();
625 } else {
626 Result += "sl_";
627 for (auto Elem : STyp->elements())
628 Result += getMangledTypeStr(Elem);
629 }
630 // Ensure nested structs are distinguishable.
631 Result += "s";
632 } else if (FunctionType *FT = dyn_cast<FunctionType>(Ty)) {
633 Result += "f_" + getMangledTypeStr(FT->getReturnType());
634 for (size_t i = 0; i < FT->getNumParams(); i++)
635 Result += getMangledTypeStr(FT->getParamType(i));
636 if (FT->isVarArg())
637 Result += "vararg";
638 // Ensure nested function types are distinguishable.
639 Result += "f";
640 } else if (isa<VectorType>(Ty)) {
641 Result += "v" + utostr(Ty->getVectorNumElements()) +
642 getMangledTypeStr(Ty->getVectorElementType());
643 } else if (Ty) {
644 switch (Ty->getTypeID()) {
645 default: llvm_unreachable("Unhandled type");
646 case Type::VoidTyID: Result += "isVoid"; break;
647 case Type::MetadataTyID: Result += "Metadata"; break;
648 case Type::HalfTyID: Result += "f16"; break;
649 case Type::FloatTyID: Result += "f32"; break;
650 case Type::DoubleTyID: Result += "f64"; break;
651 case Type::X86_FP80TyID: Result += "f80"; break;
652 case Type::FP128TyID: Result += "f128"; break;
653 case Type::PPC_FP128TyID: Result += "ppcf128"; break;
654 case Type::X86_MMXTyID: Result += "x86mmx"; break;
655 case Type::IntegerTyID:
656 Result += "i" + utostr(cast<IntegerType>(Ty)->getBitWidth());
657 break;
658 }
659 }
660 return Result;
661}

737static void DecodeIITType(unsigned &NextElt, ArrayRef<unsigned char> Infos,
738 SmallVectorImpl<Intrinsic::IITDescriptor> &OutputTable) {
739 using namespace Intrinsic;
740
741 IIT_Info Info = IIT_Info(Infos[NextElt++]);
742 unsigned StructElts = 2;
743
744 switch (Info) {
745 case IIT_Done:
746 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Void, 0));
747 return;
748 case IIT_VARARG:
749 OutputTable.push_back(IITDescriptor::get(IITDescriptor::VarArg, 0));
750 return;
751 case IIT_MMX:
752 OutputTable.push_back(IITDescriptor::get(IITDescriptor::MMX, 0));
753 return;
754 case IIT_TOKEN:
755 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Token, 0));
756 return;
757 case IIT_METADATA:
758 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Metadata, 0));
759 return;
760 case IIT_F16:
761 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Half, 0));
762 return;
763 case IIT_F32:
764 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Float, 0));
765 return;
766 case IIT_F64:
767 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Double, 0));
768 return;
769 case IIT_F128:
770 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Quad, 0));
771 return;
772 case IIT_I1:
773 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer, 1));
774 return;
775 case IIT_I8:
776 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer, 8));
777 return;
778 case IIT_I16:
779 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer,16));
780 return;
781 case IIT_I32:
782 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer, 32));
783 return;
784 case IIT_I64:
785 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer, 64));
786 return;
787 case IIT_I128:
788 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Integer, 128));
789 return;
790 case IIT_V1:
791 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 1));
792 DecodeIITType(NextElt, Infos, OutputTable);
793 return;
794 case IIT_V2:
795 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 2));
796 DecodeIITType(NextElt, Infos, OutputTable);
797 return;
798 case IIT_V4:
799 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 4));
800 DecodeIITType(NextElt, Infos, OutputTable);
801 return;
802 case IIT_V8:
803 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 8));
804 DecodeIITType(NextElt, Infos, OutputTable);
805 return;
806 case IIT_V16:
807 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 16));
808 DecodeIITType(NextElt, Infos, OutputTable);
809 return;
810 case IIT_V32:
811 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 32));
812 DecodeIITType(NextElt, Infos, OutputTable);
813 return;
814 case IIT_V64:
815 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 64));
816 DecodeIITType(NextElt, Infos, OutputTable);
817 return;
818 case IIT_V512:
819 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 512));
820 DecodeIITType(NextElt, Infos, OutputTable);
821 return;
822 case IIT_V1024:
823 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Vector, 1024));
824 DecodeIITType(NextElt, Infos, OutputTable);
825 return;
826 case IIT_PTR:
827 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Pointer, 0));
828 DecodeIITType(NextElt, Infos, OutputTable);
829 return;
830 case IIT_ANYPTR: { // [ANYPTR addrspace, subtype]
831 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Pointer,
832 Infos[NextElt++]));
833 DecodeIITType(NextElt, Infos, OutputTable);
834 return;
835 }
836 case IIT_IFATPTR64:
837 case IIT_IFATPTR128:
838 case IIT_IFATPTR256:
839 case IIT_IFATPTR512:
840 case IIT_IFATPTRAny: {
841 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Pointer,
842 200));
843 DecodeIITType(++NextElt, Infos, OutputTable);
844 return;
845 }
846 case IIT_ARG: {
847 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
848 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Argument, ArgInfo));
849 return;
850 }
851 case IIT_EXTEND_ARG: {
852 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
853 OutputTable.push_back(IITDescriptor::get(IITDescriptor::ExtendArgument,
854 ArgInfo));
855 return;
856 }
857 case IIT_TRUNC_ARG: {
858 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
859 OutputTable.push_back(IITDescriptor::get(IITDescriptor::TruncArgument,
860 ArgInfo));
861 return;
862 }
863 case IIT_HALF_VEC_ARG: {
864 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
865 OutputTable.push_back(IITDescriptor::get(IITDescriptor::HalfVecArgument,
866 ArgInfo));
867 return;
868 }
869 case IIT_SAME_VEC_WIDTH_ARG: {
870 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
871 OutputTable.push_back(IITDescriptor::get(IITDescriptor::SameVecWidthArgument,
872 ArgInfo));
873 return;
874 }
875 case IIT_PTR_TO_ARG: {
876 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
877 OutputTable.push_back(IITDescriptor::get(IITDescriptor::PtrToArgument,
878 ArgInfo));
879 return;
880 }
881 case IIT_PTR_TO_ELT: {
882 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
883 OutputTable.push_back(IITDescriptor::get(IITDescriptor::PtrToElt, ArgInfo));
884 return;
885 }
886 case IIT_VEC_OF_ANYPTRS_TO_ELT: {
887 unsigned short ArgNo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
888 unsigned short RefNo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
889 OutputTable.push_back(
890 IITDescriptor::get(IITDescriptor::VecOfAnyPtrsToElt, ArgNo, RefNo));
891 return;
892 }
893 case IIT_EMPTYSTRUCT:
894 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Struct, 0));
895 return;
896 case IIT_STRUCT8: ++StructElts; LLVM_FALLTHROUGH;
897 case IIT_STRUCT7: ++StructElts; LLVM_FALLTHROUGH;
898 case IIT_STRUCT6: ++StructElts; LLVM_FALLTHROUGH;
899 case IIT_STRUCT5: ++StructElts; LLVM_FALLTHROUGH;
900 case IIT_STRUCT4: ++StructElts; LLVM_FALLTHROUGH;
901 case IIT_STRUCT3: ++StructElts; LLVM_FALLTHROUGH;
902 case IIT_STRUCT2: {
903 OutputTable.push_back(IITDescriptor::get(IITDescriptor::Struct,StructElts));
904
905 for (unsigned i = 0; i != StructElts; ++i)
906 DecodeIITType(NextElt, Infos, OutputTable);
907 return;
908 }
909 case IIT_VEC_ELEMENT: {
910 unsigned ArgInfo = (NextElt == Infos.size() ? 0 : Infos[NextElt++]);
911 OutputTable.push_back(IITDescriptor::get(IITDescriptor::VecElementArgument,
912 ArgInfo));
913 return;
914 }
915 }
916 llvm_unreachable("unhandled");
917}

923void Intrinsic::getIntrinsicInfoTableEntries(ID id,
924 SmallVectorImpl<IITDescriptor> &T){
925 // Check to see if the intrinsic's type was expressible by the table.
926 unsigned TableVal = IIT_Table[id-1];
927
928 // Decode the TableVal into an array of IITValues.
929 SmallVector<unsigned char, 8> IITValues;
930 ArrayRef<unsigned char> IITEntries;
931 unsigned NextElt = 0;
932 if ((TableVal >> 31) != 0) {
933 // This is an offset into the IIT_LongEncodingTable.
934 IITEntries = IIT_LongEncodingTable;
935
936 // Strip sentinel bit.
937 NextElt = (TableVal << 1) >> 1;
938 } else {
939 // Decode the TableVal into an array of IITValues. If the entry was encoded
940 // into a single word in the table itself, decode it now.
941 do {
942 IITValues.push_back(TableVal & 0xF);
943 TableVal >>= 4;
944 } while (TableVal);
945
946 IITEntries = IITValues;
947 NextElt = 0;
948 }
949
950 // Okay, decode the table into the output vector of IITDescriptors.
951 DecodeIITType(NextElt, IITEntries, T);
952 while (NextElt != IITEntries.size() && IITEntries[NextElt] != 0)
953 DecodeIITType(NextElt, IITEntries, T);
954}

956static Type *DecodeFixedType(ArrayRef<Intrinsic::IITDescriptor> &Infos,
957 ArrayRef<Type*> Tys, LLVMContext &Context) {
958 using namespace Intrinsic;
959
960 IITDescriptor D = Infos.front();
961 Infos = Infos.slice(1);
962
963 switch (D.Kind) {
964 case IITDescriptor::Void: return Type::getVoidTy(Context);
965 case IITDescriptor::VarArg: return Type::getVoidTy(Context);
966 case IITDescriptor::MMX: return Type::getX86_MMXTy(Context);
967 case IITDescriptor::Token: return Type::getTokenTy(Context);
968 case IITDescriptor::Metadata: return Type::getMetadataTy(Context);
969 case IITDescriptor::Half: return Type::getHalfTy(Context);
970 case IITDescriptor::Float: return Type::getFloatTy(Context);
971 case IITDescriptor::Double: return Type::getDoubleTy(Context);
972 case IITDescriptor::Quad: return Type::getFP128Ty(Context);
973
974 case IITDescriptor::Integer:
975 return IntegerType::get(Context, D.Integer_Width);
976 case IITDescriptor::Vector:
977 return VectorType::get(DecodeFixedType(Infos, Tys, Context),D.Vector_Width);
978 case IITDescriptor::Pointer:
979 return PointerType::get(DecodeFixedType(Infos, Tys, Context),
980 D.Pointer_AddressSpace);
981 case IITDescriptor::Struct: {
982 SmallVector<Type *, 8> Elts;
983 for (unsigned i = 0, e = D.Struct_NumElements; i != e; ++i)
984 Elts.push_back(DecodeFixedType(Infos, Tys, Context));
985 return StructType::get(Context, Elts);
986 }
987 case IITDescriptor::Argument:
988 return Tys[D.getArgumentNumber()];
989 case IITDescriptor::ExtendArgument: {
990 Type *Ty = Tys[D.getArgumentNumber()];
991 if (VectorType *VTy = dyn_cast<VectorType>(Ty))
992 return VectorType::getExtendedElementVectorType(VTy);
993
994 return IntegerType::get(Context, 2 * cast<IntegerType>(Ty)->getBitWidth());
995 }
996 case IITDescriptor::TruncArgument: {
997 Type *Ty = Tys[D.getArgumentNumber()];
998 if (VectorType *VTy = dyn_cast<VectorType>(Ty))
999 return VectorType::getTruncatedElementVectorType(VTy);
1000
1001 IntegerType *ITy = cast<IntegerType>(Ty);
1002 assert(ITy->getBitWidth() % 2 == 0);
1003 return IntegerType::get(Context, ITy->getBitWidth() / 2);
1004 }
1005 case IITDescriptor::HalfVecArgument:
1006 return VectorType::getHalfElementsVectorType(cast<VectorType>(
1007 Tys[D.getArgumentNumber()]));
1008 case IITDescriptor::SameVecWidthArgument: {
1009 Type *EltTy = DecodeFixedType(Infos, Tys, Context);
1010 Type *Ty = Tys[D.getArgumentNumber()];
1011 if (auto *VTy = dyn_cast<VectorType>(Ty))
1012 return VectorType::get(EltTy, VTy->getNumElements());
1013 return EltTy;
1014 }
1015 case IITDescriptor::PtrToArgument: {
1016 Type *Ty = Tys[D.getArgumentNumber()];
1017 return PointerType::getUnqual(Ty);
1018 }
1019 case IITDescriptor::PtrToElt: {
1020 Type *Ty = Tys[D.getArgumentNumber()];
1021 VectorType *VTy = dyn_cast<VectorType>(Ty);
1022 if (!VTy)
1023 llvm_unreachable("Expected an argument of Vector Type");
1024 Type *EltTy = VTy->getVectorElementType();
1025 return PointerType::getUnqual(EltTy);
1026 }
1027 case IITDescriptor::VecElementArgument: {
1028 Type *Ty = Tys[D.getArgumentNumber()];
1029 if (VectorType *VTy = dyn_cast<VectorType>(Ty))
1030 return VTy->getElementType();
1031 llvm_unreachable("Expected an argument of Vector Type");
1032 }
1033 case IITDescriptor::VecOfAnyPtrsToElt:
1034 // Return the overloaded type (which determines the pointers address space)
1035 return Tys[D.getOverloadArgNumber()];
1036 }
1037 llvm_unreachable("unhandled");
1038}

1040FunctionType *Intrinsic::getType(LLVMContext &Context,
1041 ID id, ArrayRef<Type*> Tys) {
1042 SmallVector<IITDescriptor, 8> Table;
1043 getIntrinsicInfoTableEntries(id, Table);
1044
1045 ArrayRef<IITDescriptor> TableRef = Table;
1046 Type *ResultTy = DecodeFixedType(TableRef, Tys, Context);
1047
1048 SmallVector<Type*, 8> ArgTys;
1049 while (!TableRef.empty())
1050 ArgTys.push_back(DecodeFixedType(TableRef, Tys, Context));
1051
1052 // DecodeFixedType returns Void for IITDescriptor::Void and IITDescriptor::VarArg
1053 // If we see void type as the type of the last argument, it is vararg intrinsic
1054 if (!ArgTys.empty() && ArgTys.back()->isVoidTy()) {
1055 ArgTys.pop_back();
1056 return FunctionType::get(ResultTy, ArgTys, true);
1057 }
1058 return FunctionType::get(ResultTy, ArgTys, false);
1059}

1103using DeferredIntrinsicMatchPair =
1104 std::pair<Type *, ArrayRef<Intrinsic::IITDescriptor>>;
1105
1106static bool matchIntrinsicType(
1107 Type *Ty, ArrayRef<Intrinsic::IITDescriptor> &Infos,
1108 SmallVectorImpl<Type *> &ArgTys,
1109 SmallVectorImpl<DeferredIntrinsicMatchPair> &DeferredChecks,
1110 bool IsDeferredCheck) {
1111 using namespace Intrinsic;
1112
1113 // If we ran out of descriptors, there are too many arguments.
1114 if (Infos.empty()) return true;
1115
1116 // Do this before slicing off the 'front' part
1117 auto InfosRef = Infos;
1118 auto DeferCheck = [&DeferredChecks, &InfosRef](Type *T) {
1119 DeferredChecks.emplace_back(T, InfosRef);
1120 return false;
1121 };
1122
1123 IITDescriptor D = Infos.front();
1124 Infos = Infos.slice(1);
1125
1126 switch (D.Kind) {
1127 case IITDescriptor::Void: return !Ty->isVoidTy();
1128 case IITDescriptor::VarArg: return true;
1129 case IITDescriptor::MMX: return !Ty->isX86_MMXTy();
1130 case IITDescriptor::Token: return !Ty->isTokenTy();
1131 case IITDescriptor::Metadata: return !Ty->isMetadataTy();
1132 case IITDescriptor::Half: return !Ty->isHalfTy();
1133 case IITDescriptor::Float: return !Ty->isFloatTy();
1134 case IITDescriptor::Double: return !Ty->isDoubleTy();
1135 case IITDescriptor::Quad: return !Ty->isFP128Ty();
1136 case IITDescriptor::Integer: return !Ty->isIntegerTy(D.Integer_Width);
1137 case IITDescriptor::Vector: {
1138 VectorType *VT = dyn_cast<VectorType>(Ty);
1139 return !VT || VT->getNumElements() != D.Vector_Width ||
1140 matchIntrinsicType(VT->getElementType(), Infos, ArgTys,
1141 DeferredChecks, IsDeferredCheck);
1142 }
1143 case IITDescriptor::Pointer: {
1144 PointerType *PT = dyn_cast<PointerType>(Ty);
1145 return !PT || PT->getAddressSpace() != D.Pointer_AddressSpace ||
1146 matchIntrinsicType(PT->getElementType(), Infos, ArgTys,
1147 DeferredChecks, IsDeferredCheck);
1148 }
1149
1150 case IITDescriptor::Struct: {
1151 StructType *ST = dyn_cast<StructType>(Ty);
1152 if (!ST || ST->getNumElements() != D.Struct_NumElements)
1153 return true;
1154
1155 for (unsigned i = 0, e = D.Struct_NumElements; i != e; ++i)
1156 if (matchIntrinsicType(ST->getElementType(i), Infos, ArgTys,
1157 DeferredChecks, IsDeferredCheck))
1158 return true;
1159 return false;
1160 }
1161
1162 case IITDescriptor::Argument:
1163 // If this is the second occurrence of an argument,
1164 // verify that the later instance matches the previous instance.
1165 if (D.getArgumentNumber() < ArgTys.size())
1166 return Ty != ArgTys[D.getArgumentNumber()];
1167
1168 if (D.getArgumentNumber() > ArgTys.size() ||
1169 D.getArgumentKind() == IITDescriptor::AK_MatchType)
1170 return IsDeferredCheck || DeferCheck(Ty);
1171
1172 assert(D.getArgumentNumber() == ArgTys.size() && !IsDeferredCheck &&
1173 "Table consistency error");
1174 ArgTys.push_back(Ty);
1175
1176 switch (D.getArgumentKind()) {
1177 case IITDescriptor::AK_Any: return false; // Success
1178 case IITDescriptor::AK_AnyInteger: return !Ty->isIntOrIntVectorTy();
1179 case IITDescriptor::AK_AnyFloat: return !Ty->isFPOrFPVectorTy();
1180 case IITDescriptor::AK_AnyVector: return !isa<VectorType>(Ty);
1181 case IITDescriptor::AK_AnyPointer: return !isa<PointerType>(Ty);
1182 default: break;
1183 }
1184 llvm_unreachable("all argument kinds not covered");
1185
1186 case IITDescriptor::ExtendArgument: {
1187 // If this is a forward reference, defer the check for later.
1188 if (D.getArgumentNumber() >= ArgTys.size())
1189 return IsDeferredCheck || DeferCheck(Ty);
1190
1191 Type *NewTy = ArgTys[D.getArgumentNumber()];
1192 if (VectorType *VTy = dyn_cast<VectorType>(NewTy))
1193 NewTy = VectorType::getExtendedElementVectorType(VTy);
1194 else if (IntegerType *ITy = dyn_cast<IntegerType>(NewTy))
1195 NewTy = IntegerType::get(ITy->getContext(), 2 * ITy->getBitWidth());
1196 else
1197 return true;
1198
1199 return Ty != NewTy;
1200 }
1201 case IITDescriptor::TruncArgument: {
1202 // If this is a forward reference, defer the check for later.
1203 if (D.getArgumentNumber() >= ArgTys.size())
1204 return IsDeferredCheck || DeferCheck(Ty);
1205
1206 Type *NewTy = ArgTys[D.getArgumentNumber()];
1207 if (VectorType *VTy = dyn_cast<VectorType>(NewTy))
1208 NewTy = VectorType::getTruncatedElementVectorType(VTy);
1209 else if (IntegerType *ITy = dyn_cast<IntegerType>(NewTy))
1210 NewTy = IntegerType::get(ITy->getContext(), ITy->getBitWidth() / 2);
1211 else
1212 return true;
1213
1214 return Ty != NewTy;
1215 }
1216 case IITDescriptor::HalfVecArgument:
1217 // If this is a forward reference, defer the check for later.
1218 return D.getArgumentNumber() >= ArgTys.size() ||
1219 !isa<VectorType>(ArgTys[D.getArgumentNumber()]) ||
1220 VectorType::getHalfElementsVectorType(
1221 cast<VectorType>(ArgTys[D.getArgumentNumber()])) != Ty;
1222 case IITDescriptor::SameVecWidthArgument: {
1223 if (D.getArgumentNumber() >= ArgTys.size()) {
1224 // Defer check and subsequent check for the vector element type.
1225 Infos = Infos.slice(1);
1226 return IsDeferredCheck || DeferCheck(Ty);
1227 }
1228 auto *ReferenceType = dyn_cast<VectorType>(ArgTys[D.getArgumentNumber()]);
1229 auto *ThisArgType = dyn_cast<VectorType>(Ty);
1230 // Both must be vectors of the same number of elements or neither.
1231 if ((ReferenceType != nullptr) != (ThisArgType != nullptr))
1232 return true;
1233 Type *EltTy = Ty;
1234 if (ThisArgType) {
1235 if (ReferenceType->getVectorNumElements() !=
1236 ThisArgType->getVectorNumElements())
1237 return true;
1238 EltTy = ThisArgType->getVectorElementType();
1239 }
1240 return matchIntrinsicType(EltTy, Infos, ArgTys, DeferredChecks,
1241 IsDeferredCheck);
1242 }
1243 case IITDescriptor::PtrToArgument: {
1244 if (D.getArgumentNumber() >= ArgTys.size())
1245 return IsDeferredCheck || DeferCheck(Ty);
1246 Type * ReferenceType = ArgTys[D.getArgumentNumber()];
1247 PointerType *ThisArgType = dyn_cast<PointerType>(Ty);
1248 return (!ThisArgType || ThisArgType->getElementType() != ReferenceType);
1249 }
1250 case IITDescriptor::PtrToElt: {
1251 if (D.getArgumentNumber() >= ArgTys.size())
1252 return IsDeferredCheck || DeferCheck(Ty);
1253 VectorType * ReferenceType =
1254 dyn_cast<VectorType> (ArgTys[D.getArgumentNumber()]);
1255 PointerType *ThisArgType = dyn_cast<PointerType>(Ty);
1256
1257 return (!ThisArgType || !ReferenceType ||
1258 ThisArgType->getElementType() != ReferenceType->getElementType());
1259 }
1260 case IITDescriptor::VecOfAnyPtrsToElt: {
1261 unsigned RefArgNumber = D.getRefArgNumber();
1262 if (RefArgNumber >= ArgTys.size()) {
1263 if (IsDeferredCheck)
1264 return true;
1265 // If forward referencing, already add the pointer-vector type and
1266 // defer the checks for later.
1267 ArgTys.push_back(Ty);
1268 return DeferCheck(Ty);
1269 }
1270
1271 if (!IsDeferredCheck){
1272 assert(D.getOverloadArgNumber() == ArgTys.size() &&
1273 "Table consistency error");
1274 ArgTys.push_back(Ty);
1275 }
1276
1277 // Verify the overloaded type "matches" the Ref type.
1278 // i.e. Ty is a vector with the same width as Ref.
1279 // Composed of pointers to the same element type as Ref.
1280 VectorType *ReferenceType = dyn_cast<VectorType>(ArgTys[RefArgNumber]);
1281 VectorType *ThisArgVecTy = dyn_cast<VectorType>(Ty);
1282 if (!ThisArgVecTy || !ReferenceType ||
1283 (ReferenceType->getVectorNumElements() !=
1284 ThisArgVecTy->getVectorNumElements()))
1285 return true;
1286 PointerType *ThisArgEltTy =
1287 dyn_cast<PointerType>(ThisArgVecTy->getVectorElementType());
1288 if (!ThisArgEltTy)
1289 return true;
1290 return ThisArgEltTy->getElementType() !=
1291 ReferenceType->getVectorElementType();
1292 }
1293 case IITDescriptor::VecElementArgument: {
1294 if (D.getArgumentNumber() >= ArgTys.size())
1295 return IsDeferredCheck ? true : DeferCheck(Ty);
1296 auto *ReferenceType = dyn_cast<VectorType>(ArgTys[D.getArgumentNumber()]);
1297 return !ReferenceType || Ty != ReferenceType->getElementType();
1298 }
1299 }
1300 llvm_unreachable("unhandled");
1301}

1303Intrinsic::MatchIntrinsicTypesResult
1304Intrinsic::matchIntrinsicSignature(FunctionType *FTy,
1305 ArrayRef<Intrinsic::IITDescriptor> &Infos,
1306 SmallVectorImpl<Type *> &ArgTys) {
1307 SmallVector<DeferredIntrinsicMatchPair, 2> DeferredChecks;
1308 if (matchIntrinsicType(FTy->getReturnType(), Infos, ArgTys, DeferredChecks,
1309 false))
1310 return MatchIntrinsicTypes_NoMatchRet;
1311
1312 unsigned NumDeferredReturnChecks = DeferredChecks.size();
1313
1314 for (auto Ty : FTy->params())
1315 if (matchIntrinsicType(Ty, Infos, ArgTys, DeferredChecks, false))
1316 return MatchIntrinsicTypes_NoMatchArg;
1317
1318 for (unsigned I = 0, E = DeferredChecks.size(); I != E; ++I) {
1319 DeferredIntrinsicMatchPair &Check = DeferredChecks[I];
1320 if (matchIntrinsicType(Check.first, Check.second, ArgTys, DeferredChecks,
1321 true))
1322 return I < NumDeferredReturnChecks ? MatchIntrinsicTypes_NoMatchRet
1323 : MatchIntrinsicTypes_NoMatchArg;
1324 }
1325
1326 return MatchIntrinsicTypes_Match;
1327}

1329bool
1330Intrinsic::matchIntrinsicVarArg(bool isVarArg,
1331 ArrayRef<Intrinsic::IITDescriptor> &Infos) {
1332 // If there are no descriptors left, then it can't be a vararg.
1333 if (Infos.empty())
1334 return isVarArg;
1335
1336 // There should be only one descriptor remaining at this point.
1337 if (Infos.size() != 1)
1338 return true;
1339
1340 // Check and verify the descriptor.
1341 IITDescriptor D = Infos.front();
1342 Infos = Infos.slice(1);
1343 if (D.Kind == IITDescriptor::VarArg)
1344 return !isVarArg;
1345
1346 return true;

1349Optional<Function*> Intrinsic::remangleIntrinsicFunction(Function *F) {
1350 Intrinsic::ID ID = F->getIntrinsicID();
1351 if (!ID)
1352 return None;
1353
1354 FunctionType *FTy = F->getFunctionType();
1355 // Accumulate an array of overloaded types for the given intrinsic
1356 SmallVector<Type *, 4> ArgTys;
1357 {
1358 SmallVector<Intrinsic::IITDescriptor, 8> Table;
1359 getIntrinsicInfoTableEntries(ID, Table);
1360 ArrayRef<Intrinsic::IITDescriptor> TableRef = Table;
1361
1362 if (Intrinsic::matchIntrinsicSignature(FTy, TableRef, ArgTys))
1363 return None;
1364 if (Intrinsic::matchIntrinsicVarArg(FTy->isVarArg(), TableRef))
1365 return None;
1366 }
1367
1368 StringRef Name = F->getName();
1369 if (Name == Intrinsic::getName(ID, ArgTys))
1370 return None;
1371
1372 // Don't remangle va_start, etc if it already had an address space qualifier
1373 if (ID == Intrinsic::vastart || ID == Intrinsic::vaend ||
1374 ID == Intrinsic::vacopy) {
1375 if (Name.contains('.'))
1376 return None;
1377 }
1378
1379 auto NewDecl = Intrinsic::getDeclaration(F->getParent(), ID, ArgTys);
1380 NewDecl->setCallingConv(F->getCallingConv());
1381 assert(NewDecl->getFunctionType() == FTy && "Shouldn't change the signature");
1382 return NewDecl;
1383}

1385/// hasAddressTaken - returns true if there are any uses of this function
1386/// other than direct calls or invokes to it.
1387bool Function::hasAddressTaken(const User* *PutOffender) const {
1388 for (const Use &U : uses()) {
1389 const User *FU = U.getUser();
1390 if (isa<BlockAddress>(FU))
1391 continue;
1392 const auto *Call = dyn_cast<CallBase>(FU);
1393 if (!Call) {
1394 if (PutOffender)
1395 *PutOffender = FU;
1396 return true;
1397 }
1398 if (!Call->isCallee(&U)) {
1399 if (PutOffender)
1400 *PutOffender = FU;
1401 return true;
1402 }
1403 }
1404 return false;
1405}
