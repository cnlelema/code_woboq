15#define PDB_ENUM_CLASS_ENT(enum_class, enum) \
16 { #enum, std::underlying_type < enum_class > ::type(enum_class::enum) }
17
18#define PDB_ENUM_ENT(ns, enum) \
19 { #enum, ns::enum }
