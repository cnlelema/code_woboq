22SymbolStream::SymbolStream(std::unique_ptr<MappedBlockStream> Stream)
23 : Stream(std::move(Stream)) {}
24
25SymbolStream::~SymbolStream() {}
26
27Error SymbolStream::reload() {
28 BinaryStreamReader Reader(*Stream);
29
30 if (auto EC = Reader.readArray(SymbolRecords, Stream->getLength()))
31 return EC;
32
33 return Error::success();
34}
35
36iterator_range<codeview::CVSymbolArray::Iterator>
37SymbolStream::getSymbols(bool *HadError) const {
38 return llvm::make_range(SymbolRecords.begin(HadError), SymbolRecords.end());
39}
40
41Error SymbolStream::commit() { return Error::success(); }
42
43codeview::CVSymbol SymbolStream::readRecord(uint32_t Offset) const {
44 return *SymbolRecords.at(Offset);
45}
