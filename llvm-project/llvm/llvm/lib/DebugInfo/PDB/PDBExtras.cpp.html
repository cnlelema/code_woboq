119raw_ostream &llvm::pdb::operator<<(raw_ostream &OS,
120 const llvm::codeview::CPURegister &CpuReg) {
121 if (CpuReg.Cpu == llvm::codeview::CPUType::ARM64) {
122 switch (CpuReg.Reg) {
123#define CV_REGISTERS_ARM64
124#define CV_REGISTER(name, val) \
125 case codeview::RegisterId::name: \
126 OS << #name; \
127 return OS;
128#include "llvm/DebugInfo/CodeView/CodeViewRegisters.def"
129#undef CV_REGISTER
130#undef CV_REGISTERS_ARM64
131
132 default:
133 break;
134 }
135 } else {
136 switch (CpuReg.Reg) {
137#define CV_REGISTERS_X86
138#define CV_REGISTER(name, val) \
139 case codeview::RegisterId::name: \
140 OS << #name; \
141 return OS;
142#include "llvm/DebugInfo/CodeView/CodeViewRegisters.def"
143#undef CV_REGISTER
144#undef CV_REGISTERS_X86
145
146 default:
147 break;
148 }
149 }
150 OS << static_cast<int>(CpuReg.Reg);
151 return OS;
152}

335raw_ostream &llvm::pdb::operator<<(raw_ostream &OS, const Variant &Value) {
336 switch (Value.Type) {
337 case PDB_VariantType::Bool:
338 OS << (Value.Value.Bool ? "true" : "false");
339 break;
340 case PDB_VariantType::Double:
341 OS << Value.Value.Double;
342 break;
343 case PDB_VariantType::Int16:
344 OS << Value.Value.Int16;
345 break;
346 case PDB_VariantType::Int32:
347 OS << Value.Value.Int32;
348 break;
349 case PDB_VariantType::Int64:
350 OS << Value.Value.Int64;
351 break;
352 case PDB_VariantType::Int8:
353 OS << static_cast<int>(Value.Value.Int8);
354 break;
355 case PDB_VariantType::Single:
356 OS << Value.Value.Single;
357 break;
358 case PDB_VariantType::UInt16:
359 OS << Value.Value.UInt16;
360 break;
361 case PDB_VariantType::UInt32:
362 OS << Value.Value.UInt32;
363 break;
364 case PDB_VariantType::UInt64:
365 OS << Value.Value.UInt64;
366 break;
367 case PDB_VariantType::UInt8:
368 OS << static_cast<unsigned>(Value.Value.UInt8);
369 break;
370 case PDB_VariantType::String:
371 OS << Value.Value.String;
372 break;
373 default:
374 OS << Value.Type;
375 }
376 return OS;
377}
