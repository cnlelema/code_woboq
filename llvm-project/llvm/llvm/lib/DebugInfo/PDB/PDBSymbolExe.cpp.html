21uint32_t PDBSymbolExe::getPointerByteSize() const {
22 auto Pointer = findOneChild<PDBSymbolTypePointer>();
23 if (Pointer)
24 return Pointer->getLength();
25
26 if (getMachineType() == PDB_Machine::x86)
27 return 4;
28 return 8;
29}
