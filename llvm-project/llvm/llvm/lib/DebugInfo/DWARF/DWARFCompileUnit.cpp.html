15using namespace llvm;
16
17void DWARFCompileUnit::dump(raw_ostream &OS, DIDumpOptions DumpOpts) {
18 OS << format("0x%08x", getOffset()) << ": Compile Unit:"
19 << " length = " << format("0x%08x", getLength())
20 << " version = " << format("0x%04x", getVersion());
21 if (getVersion() >= 5)
22 OS << " unit_type = " << dwarf::UnitTypeString(getUnitType());
23 OS << " abbr_offset = " << format("0x%04x", getAbbreviations()->getOffset())
24 << " addr_size = " << format("0x%02x", getAddressByteSize());
25 if (getVersion() >= 5 && getUnitType() != dwarf::DW_UT_compile)
26 OS << " DWO_id = " << format("0x%016" PRIx64, *getDWOId());
27 OS << " (next unit at " << format("0x%08x", getNextUnitOffset()) << ")\n";
28
29 if (DWARFDie CUDie = getUnitDIE(false))
30 CUDie.dump(OS, 0, DumpOpts);
31 else
32 OS << "<compile unit can't be parsed!>\n\n";
33}
