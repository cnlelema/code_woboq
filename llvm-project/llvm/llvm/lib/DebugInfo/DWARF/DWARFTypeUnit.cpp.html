20void DWARFTypeUnit::dump(raw_ostream &OS, DIDumpOptions DumpOpts) {
21 DWARFDie TD = getDIEForOffset(getTypeOffset() + getOffset());
22 const char *Name = TD.getName(DINameKind::ShortName);
23
24 if (DumpOpts.SummarizeTypes) {
25 OS << "name = '" << Name << "'"
26 << " type_signature = " << format("0x%016" PRIx64, getTypeHash())
27 << " length = " << format("0x%08x", getLength()) << '\n';
28 return;
29 }
30
31 OS << format("0x%08x", getOffset()) << ": Type Unit:"
32 << " length = " << format("0x%08x", getLength())
33 << " version = " << format("0x%04x", getVersion());
34 if (getVersion() >= 5)
35 OS << " unit_type = " << dwarf::UnitTypeString(getUnitType());
36 OS << " abbr_offset = " << format("0x%04x", getAbbreviations()->getOffset())
37 << " addr_size = " << format("0x%02x", getAddressByteSize())
38 << " name = '" << Name << "'"
39 << " type_signature = " << format("0x%016" PRIx64, getTypeHash())
40 << " type_offset = " << format("0x%04x", getTypeOffset())
41 << " (next unit at " << format("0x%08x", getNextUnitOffset()) << ")\n";
42
43 if (DWARFDie TU = getUnitDIE(false))
44 TU.dump(OS, 0, DumpOpts);
45 else
46 OS << "<type unit can't be parsed!>\n\n";
47}
