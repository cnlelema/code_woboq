106static DWARFExpression::Operation::Description getOpDesc(unsigned OpCode) {
107 // FIXME: Make this constexpr once all compilers are smart enough to do it.
108 static DescVector Descriptions = getDescriptions();
109 // Handle possible corrupted or unsupported operation.
110 if (OpCode >= Descriptions.size())
111 return {};
112 return Descriptions[OpCode];
113}

115static uint8_t getRefAddrSize(uint8_t AddrSize, uint16_t Version) {
116 return (Version == 2) ? AddrSize : 4;
117}

119bool DWARFExpression::Operation::extract(DataExtractor Data, uint16_t Version,
120 uint8_t AddressSize, uint32_t Offset) {
121 Opcode = Data.getU8(&Offset);
122
123 Desc = getOpDesc(Opcode);
124 if (Desc.Version == Operation::DwarfNA) {
125 EndOffset = Offset;
126 return false;
127 }
128
129 for (unsigned Operand = 0; Operand < 2; ++Operand) {
130 unsigned Size = Desc.Op[Operand];
131 unsigned Signed = Size & Operation::SignBit;
132
133 if (Size == Operation::SizeNA)
134 break;
135
136 switch (Size & ~Operation::SignBit) {
137 case Operation::Size1:
138 Operands[Operand] = Data.getU8(&Offset);
139 if (Signed)
140 Operands[Operand] = (int8_t)Operands[Operand];
141 break;
142 case Operation::Size2:
143 Operands[Operand] = Data.getU16(&Offset);
144 if (Signed)
145 Operands[Operand] = (int16_t)Operands[Operand];
146 break;
147 case Operation::Size4:
148 Operands[Operand] = Data.getU32(&Offset);
149 if (Signed)
150 Operands[Operand] = (int32_t)Operands[Operand];
151 break;
152 case Operation::Size8:
153 Operands[Operand] = Data.getU64(&Offset);
154 break;
155 case Operation::SizeAddr:
156 if (AddressSize == 8) {
157 Operands[Operand] = Data.getU64(&Offset);
158 } else if (AddressSize == 4) {
159 Operands[Operand] = Data.getU32(&Offset);
160 } else {
161 assert(AddressSize == 2);
162 Operands[Operand] = Data.getU16(&Offset);
163 }
164 break;
165 case Operation::SizeRefAddr:
166 if (getRefAddrSize(AddressSize, Version) == 8) {
167 Operands[Operand] = Data.getU64(&Offset);
168 } else if (getRefAddrSize(AddressSize, Version) == 4) {
169 Operands[Operand] = Data.getU32(&Offset);
170 } else {
171 assert(getRefAddrSize(AddressSize, Version) == 2);
172 Operands[Operand] = Data.getU16(&Offset);
173 }
174 break;
175 case Operation::SizeLEB:
176 if (Signed)
177 Operands[Operand] = Data.getSLEB128(&Offset);
178 else
179 Operands[Operand] = Data.getULEB128(&Offset);
180 break;
181 case Operation::BaseTypeRef:
182 Operands[Operand] = Data.getULEB128(&Offset);
183 break;
184 case Operation::SizeBlock:
185 // We need a size, so this cannot be the first operand
186 if (Operand == 0)
187 return false;
188 // Store the offset of the block as the value.
189 Operands[Operand] = Offset;
190 Offset += Operands[Operand - 1];
191 break;
192 default:
193 llvm_unreachable("Unknown DWARFExpression Op size");
194 }
195
196 OperandEndOffsets[Operand] = Offset;
197 }
198
199 EndOffset = Offset;
200 return true;
201}

203static bool prettyPrintRegisterOp(raw_ostream &OS, uint8_t Opcode,
204 uint64_t Operands[2],
205 const MCRegisterInfo *MRI, bool isEH) {
206 if (!MRI)
207 return false;
208
209 uint64_t DwarfRegNum;
210 unsigned OpNum = 0;
211
212 if (Opcode == DW_OP_bregx || Opcode == DW_OP_regx)
213 DwarfRegNum = Operands[OpNum++];
214 else if (Opcode >= DW_OP_breg0 && Opcode < DW_OP_bregx)
215 DwarfRegNum = Opcode - DW_OP_breg0;
216 else
217 DwarfRegNum = Opcode - DW_OP_reg0;
218
219 int LLVMRegNum = MRI->getLLVMRegNum(DwarfRegNum, isEH);
220 if (LLVMRegNum >= 0) {
221 if (const char *RegName = MRI->getName(LLVMRegNum)) {
222 if ((Opcode >= DW_OP_breg0 && Opcode <= DW_OP_breg31) ||
223 Opcode == DW_OP_bregx)
224 OS << format(" %s%+" PRId64, RegName, Operands[OpNum]);
225 else
226 OS << ' ' << RegName;
227 return true;
228 }
229 }
230
231 return false;
232}

234bool DWARFExpression::Operation::print(raw_ostream &OS,
235 const DWARFExpression *Expr,
236 const MCRegisterInfo *RegInfo,
237 DWARFUnit *U,
238 bool isEH) {
239 if (Error) {
240 OS << "<decoding error>";
241 return false;
242 }
243
244 StringRef Name = OperationEncodingString(Opcode);
245 assert(!Name.empty() && "DW_OP has no name!");
246 OS << Name;
247
248 if ((Opcode >= DW_OP_breg0 && Opcode <= DW_OP_breg31) ||
249 (Opcode >= DW_OP_reg0 && Opcode <= DW_OP_reg31) ||
250 Opcode == DW_OP_bregx || Opcode == DW_OP_regx)
251 if (prettyPrintRegisterOp(OS, Opcode, Operands, RegInfo, isEH))
252 return true;
253
254 for (unsigned Operand = 0; Operand < 2; ++Operand) {
255 unsigned Size = Desc.Op[Operand];
256 unsigned Signed = Size & Operation::SignBit;
257
258 if (Size == Operation::SizeNA)
259 break;
260
261 if (Size == Operation::BaseTypeRef && U) {
262 auto Die = U->getDIEForOffset(U->getOffset() + Operands[Operand]);
263 if (Die && Die.getTag() == dwarf::DW_TAG_base_type) {
264 OS << format(" (0x%08x)", U->getOffset() + Operands[Operand]);
265 if (auto Name = Die.find(dwarf::DW_AT_name))
266 OS << " \"" << Name->getAsCString() << "\"";
267 } else {
268 OS << format(" <invalid base_type ref: 0x%" PRIx64 ">",
269 Operands[Operand]);
270 }
271 } else if (Size == Operation::SizeBlock) {
272 uint32_t Offset = Operands[Operand];
273 for (unsigned i = 0; i < Operands[Operand - 1]; ++i)
274 OS << format(" 0x%02x", Expr->Data.getU8(&Offset));
275 } else {
276 if (Signed)
277 OS << format(" %+" PRId64, (int64_t)Operands[Operand]);
278 else
279 OS << format(" 0x%" PRIx64, Operands[Operand]);
280 }
281 }
282 return true;
283}

285void DWARFExpression::print(raw_ostream &OS, const MCRegisterInfo *RegInfo,
286 DWARFUnit *U, bool IsEH) const {
287 for (auto &Op : *this) {
288 if (!Op.print(OS, this, RegInfo, U, IsEH)) {
289 uint32_t FailOffset = Op.getEndOffset();
290 while (FailOffset < Data.getData().size())
291 OS << format(" %02x", Data.getU8(&FailOffset));
292 return;
293 }
294 if (Op.getEndOffset() < Data.getData().size())
295 OS << ", ";
296 }
297}

299bool DWARFExpression::Operation::verify(DWARFUnit *U) {
300
301 for (unsigned Operand = 0; Operand < 2; ++Operand) {
302 unsigned Size = Desc.Op[Operand];
303
304 if (Size == Operation::SizeNA)
305 break;
306
307 if (Size == Operation::BaseTypeRef) {
308 auto Die = U->getDIEForOffset(U->getOffset() + Operands[Operand]);
309 if (!Die || Die.getTag() != dwarf::DW_TAG_base_type) {
310 Error = true;
311 return false;
312 }
313 }
314 }
315
316 return true;
317}
318
319bool DWARFExpression::verify(DWARFUnit *U) {
320 for (auto &Op : *this)
321 if (!Op.verify(U))
322 return false;
323
324 return true;
325}
