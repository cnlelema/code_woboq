45Expected<std::unique_ptr<Binary>> object::createBinary(MemoryBufferRef Buffer,
46 LLVMContext *Context) {
47 file_magic Type = identify_magic(Buffer.getBuffer());
48
49 switch (Type) {
50 case file_magic::archive:
51 return Archive::create(Buffer);
52 case file_magic::elf:
53 case file_magic::elf_relocatable:
54 case file_magic::elf_executable:
55 case file_magic::elf_shared_object:
56 case file_magic::elf_core:
57 case file_magic::macho_object:
58 case file_magic::macho_executable:
59 case file_magic::macho_fixed_virtual_memory_shared_lib:
60 case file_magic::macho_core:
61 case file_magic::macho_preload_executable:
62 case file_magic::macho_dynamically_linked_shared_lib:
63 case file_magic::macho_dynamic_linker:
64 case file_magic::macho_bundle:
65 case file_magic::macho_dynamically_linked_shared_lib_stub:
66 case file_magic::macho_dsym_companion:
67 case file_magic::macho_kext_bundle:
68 case file_magic::coff_object:
69 case file_magic::coff_import_library:
70 case file_magic::pecoff_executable:
71 case file_magic::bitcode:
72 case file_magic::xcoff_object_32:
73 case file_magic::wasm_object:
74 return ObjectFile::createSymbolicFile(Buffer, Type, Context);
75 case file_magic::macho_universal_binary:
76 return MachOUniversalBinary::create(Buffer);
77 case file_magic::windows_resource:
78 return WindowsResource::createWindowsResource(Buffer);
79 case file_magic::pdb:
80 // PDB does not support the Binary interface.
81 return errorCodeToError(object_error::invalid_file_type);
82 case file_magic::unknown:
83 case file_magic::coff_cl_gl_object:
84 // Unrecognized object file format.
85 return errorCodeToError(object_error::invalid_file_type);
86 case file_magic::minidump:
87 return MinidumpFile::create(Buffer);
88 }
89 llvm_unreachable("Unexpected Binary File Type");
90}

92Expected<OwningBinary<Binary>> object::createBinary(StringRef Path) {
93 ErrorOr<std::unique_ptr<MemoryBuffer>> FileOrErr =
94 MemoryBuffer::getFileOrSTDIN(Path, /*FileSize=*/-1,
95 /*RequiresNullTerminator=*/false);
96 if (std::error_code EC = FileOrErr.getError())
97 return errorCodeToError(EC);
98 std::unique_ptr<MemoryBuffer> &Buffer = FileOrErr.get();
99
100 Expected<std::unique_ptr<Binary>> BinOrErr =
101 createBinary(Buffer->getMemBufferRef());
102 if (!BinOrErr)
103 return BinOrErr.takeError();
104 std::unique_ptr<Binary> &Bin = BinOrErr.get();
105
106 return OwningBinary<Binary>(std::move(Bin), std::move(Buffer));
107}
