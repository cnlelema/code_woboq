31/// The base class for BTF type generation.
32class BTFTypeBase {
33protected:
34 uint8_t Kind;
35 uint32_t Id;
36 struct BTF::CommonType BTFType;
37
38public:
39 virtual ~BTFTypeBase() = default;
40 void setId(uint32_t Id) { this->Id = Id; }
41 uint32_t getId() { return Id; }
42 uint32_t roundupToBytes(uint32_t NumBits) { return (NumBits + 7) >> 3; }
43 /// Get the size of this BTF type entry.
44 virtual uint32_t getSize() { return BTF::CommonTypeSize; }
45 /// Complete BTF type generation after all related DebugInfo types
46 /// have been visited so their BTF type id's are available
47 /// for cross referece.
48 virtual void completeType(BTFDebug &BDebug) {}
49 /// Emit types for this BTF type entry.
50 virtual void emitType(MCStreamer &OS);
51};
52

74/// Handle int type.
75class BTFTypeInt : public BTFTypeBase {
76 StringRef Name;
77 uint32_t IntVal; ///< Encoding, offset, bits
78
79public:
80 BTFTypeInt(uint32_t Encoding, uint32_t SizeInBits, uint32_t OffsetInBits,
81 StringRef TypeName);
82 uint32_t getSize() { return BTFTypeBase::getSize() + sizeof(uint32_t); }
83 void completeType(BTFDebug &BDebug);
84 void emitType(MCStreamer &OS);
85};

87/// Handle enumerate type.
88class BTFTypeEnum : public BTFTypeBase {
89 const DICompositeType *ETy;
90 std::vector<struct BTF::BTFEnum> EnumValues;
91
92public:
93 BTFTypeEnum(const DICompositeType *ETy, uint32_t NumValues);
94 uint32_t getSize() {
95 return BTFTypeBase::getSize() + EnumValues.size() * BTF::BTFEnumSize;
96 }
97 void completeType(BTFDebug &BDebug);
98 void emitType(MCStreamer &OS);
99};

101/// Handle array type.
102class BTFTypeArray : public BTFTypeBase {

108 void completeType(BTFDebug &BDebug);

112/// Handle struct/union type.

128/// Handle function pointer.

134public:

155/// Handle variable instances

167/// Handle data sections
168class BTFKindDataSec : public BTFTypeBase {
169 AsmPrinter *Asm;
170 std::string Name;
171 std::vector<std::tuple<uint32_t, const MCSymbol *, uint32_t>> Vars;
172
173public:
174 BTFKindDataSec(AsmPrinter *AsmPrt, std::string SecName);
175 uint32_t getSize() {
176 return BTFTypeBase::getSize() + BTF::BTFDataSecVarSize * Vars.size();
177 }
178 void addVar(uint32_t Id, const MCSymbol *Sym, uint32_t Size) {
179 Vars.push_back(std::make_tuple(Id, Sym, Size));
180 }
181 std::string getName() { return Name; }
182 void completeType(BTFDebug &BDebug);
183 void emitType(MCStreamer &OS);
184};

186/// String table.
187class BTFStringTable {
188 /// String table size in bytes.
189 uint32_t Size;
190 /// A mapping from string table offset to the index
191 /// of the Table. It is used to avoid putting
192 /// duplicated strings in the table.
193 std::unordered_map<uint32_t, uint32_t> OffsetToIdMap;
194 /// A vector of strings to represent the string table.
195 std::vector<std::string> Table;
196
197public:
198 BTFStringTable() : Size(0) {}
199 uint32_t getSize() { return Size; }
200 std::vector<std::string> &getTable() { return Table; }
201 /// Add a string to the string table and returns its offset
202 /// in the table.
203 uint32_t addString(StringRef S);
204};

206/// Represent one func and its type id.
207struct BTFFuncInfo {
208 const MCSymbol *Label; ///< Func MCSymbol
209 uint32_t TypeId; ///< Type id referring to .BTF type section
210};
211
212/// Represent one line info.
213struct BTFLineInfo {
214 MCSymbol *Label; ///< MCSymbol identifying insn for the lineinfo
215 uint32_t FileNameOff; ///< file name offset in the .BTF string table
216 uint32_t LineOff; ///< line offset in the .BTF string table
217 uint32_t LineNum; ///< the line number
218 uint32_t ColumnNum; ///< the column number
219};

221/// Collect and emit BTF information.
222class BTFDebug : public DebugHandlerBase {
223 MCStreamer &OS;
224 bool SkipInstruction;
225 bool LineInfoGenerated;
226 uint32_t SecNameOff;
227 uint32_t ArrayIndexTypeId;
228 BTFStringTable StringTable;
229 std::vector<std::unique_ptr<BTFTypeBase>> TypeEntries;
230 std::unordered_map<const DIType *, uint32_t> DIToIdMap;
231 std::map<uint32_t, std::vector<BTFFuncInfo>> FuncInfoTable;
232 std::map<uint32_t, std::vector<BTFLineInfo>> LineInfoTable;
233 StringMap<std::vector<std::string>> FileContent;
234 std::map<std::string, std::unique_ptr<BTFKindDataSec>>
235 DataSecEntries;

237 /// Add types to TypeEntries.
238 /// @{
239 /// Add types to TypeEntries and DIToIdMap.
240 uint32_t addType(std::unique_ptr<BTFTypeBase> TypeEntry, const DIType *Ty);
241 /// Add types to TypeEntries only and return type id.
242 uint32_t addType(std::unique_ptr<BTFTypeBase> TypeEntry);
243 /// @}
244
245 /// IR type visiting functions.
246 /// @{
247 void visitTypeEntry(const DIType *Ty);
248 void visitTypeEntry(const DIType *Ty, uint32_t &TypeId);
249 void visitBasicType(const DIBasicType *BTy, uint32_t &TypeId);
250 void visitSubroutineType(
251 const DISubroutineType *STy, bool ForSubprog,
252 const std::unordered_map<uint32_t, StringRef> &FuncArgNames,
253 uint32_t &TypeId);
254 void visitFwdDeclType(const DICompositeType *CTy, bool IsUnion,
255 uint32_t &TypeId);
256 void visitCompositeType(const DICompositeType *CTy, uint32_t &TypeId);
257 void visitStructType(const DICompositeType *STy, bool IsStruct,
258 uint32_t &TypeId);
259 void visitArrayType(const DICompositeType *ATy, uint32_t &TypeId);
260 void visitEnumType(const DICompositeType *ETy, uint32_t &TypeId);
261 void visitDerivedType(const DIDerivedType *DTy, uint32_t &TypeId);
262 /// @}
263
264 /// Get the file content for the subprogram. Certain lines of the file
265 /// later may be put into string table and referenced by line info.
266 std::string populateFileContent(const DISubprogram *SP);

268 /// Construct a line info.
269 void constructLineInfo(const DISubprogram *SP, MCSymbol *Label, uint32_t Line,

272 /// Generate types and variables for globals.
273 void processGlobals(void);

275 /// Emit common header of .BTF and .BTF.ext sections.
276 void emitCommonHeader();
277
278 /// Emit the .BTF section.
279 void emitBTFSection();

281 /// Emit the .BTF.ext section.
282 void emitBTFExtSection();
283
284protected:
285 /// Gather pre-function debug information.
286 void beginFunctionImpl(const MachineFunction *MF) override;
287
288 /// Post process after all instructions in this function are processed.
289 void endFunctionImpl(const MachineFunction *MF) override;
290
291public:
292 BTFDebug(AsmPrinter *AP);
293
294 /// Get the special array index type id.
295 uint32_t getArrayIndexTypeId() {
296 assert(ArrayIndexTypeId);
297 return ArrayIndexTypeId;
298 }
299
300 /// Add string to the string table.
301 size_t addString(StringRef S) { return StringTable.addString(S); }
302
303 /// Get the type id for a particular DIType.
304 uint32_t getTypeId(const DIType *Ty) {
305 assert(Ty && "Invalid null Type");
306 assert(DIToIdMap.find(Ty) != DIToIdMap.end() &&
307 "DIType not added in the BDIToIdMap");
308 return DIToIdMap[Ty];
309 }

311 void setSymbolSize(const MCSymbol *Symbol, uint64_t Size) override {}
312
313 /// Process beginning of an instruction.
314 void beginInstruction(const MachineInstr *MI) override;
315
316 /// Complete all the types and emit the BTF sections.
317 void endModule() override;
