24unsigned TargetIntrinsicInfo::getIntrinsicID(const Function *F) const {
25 const ValueName *ValName = F->getValueName();
26 if (!ValName)
27 return 0;
28 return lookupName(ValName->getKeyData(), ValName->getKeyLength());
29}
