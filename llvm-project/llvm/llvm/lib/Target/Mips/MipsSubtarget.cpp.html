76MipsSubtarget::MipsSubtarget(const Triple &TT, StringRef CPU, StringRef FS,
77 bool little, const MipsTargetMachine &TM,
78 unsigned StackAlignOverride)
79 : MipsGenSubtargetInfo(TT, CPU, FS), MipsArchVersion(MipsDefault),
80 IsLittle(little), IsSoftFloat(false), IsSingleFloat(false), IsFPXX(false),
81 NoABICalls(false), Abs2008(false), IsFP64bit(false), UseOddSPReg(true),
82 IsNaN2008bit(false), IsGP64bit(false), HasVFPU(false), HasCnMips(false),
83 HasMips3_32(false), HasMips3_32r2(false), HasMips4_32(false),
84 HasMips4_32r2(false), HasMips5_32r2(false), IsCheri64(false),
85 IsCheri128(false), IsCheri256(false), IsCheri(false),
86 UseCheriExactEquals(CheriExactEqualsOpt), InMips16Mode(false),
87 InMips16HardFloat(Mips16HardFloat), InMicroMipsMode(false), HasDSP(false),
88 HasDSPR2(false), HasDSPR3(false), AllowMixed16_32(Mixed16_32 | Mips_Os16),
89 Os16(Mips_Os16), HasMSA(false), UseTCCInDIV(false), HasSym32(false),
90 HasEVA(false), DisableMadd4(false), HasMT(false), HasCRC(false),
91 HasVirt(false), HasGINV(false), UseIndirectJumpsHazard(false),
92 StackAlignOverride(StackAlignOverride), TM(TM), TargetTriple(TT),
93 TSInfo(), InstrInfo(MipsInstrInfo::create(
94 initializeSubtargetDependencies(CPU, FS, TM))),
95 FrameLowering(MipsFrameLowering::create(*this)),
96 TLInfo(MipsTargetLowering::create(TM, *this)) {
97
98 if (MipsArchVersion == MipsDefault)
99 MipsArchVersion = Mips32;
100
101 // Don't even attempt to generate code for MIPS-I and MIPS-V. They have not
102 // been tested and currently exist for the integrated assembler only.
103 if (MipsArchVersion == Mips1)
104 report_fatal_error("Code generation for MIPS-I is not implemented", false);
105 if (MipsArchVersion == Mips5)
106 report_fatal_error("Code generation for MIPS-V is not implemented", false);
107
108 // Check if Architecture and ABI are compatible.
109 assert(((!isGP64bit() && isABI_O32()) ||
110 (isGP64bit() && (isABI_N32() || isABI_N64()))) &&
111 "Invalid Arch & ABI pair.");
112
113 if (hasMSA() && !isFP64bit())
114 report_fatal_error("MSA requires a 64-bit FPU register file (FR=1 mode). "
115 "See -mattr=+fp64.",
116 false);
117
118 if (!isABI_O32() && !useOddSPReg())
119 report_fatal_error("-mattr=+nooddspreg requires the O32 ABI.", false);
120
121 if (IsFPXX && (isABI_N32() || isABI_N64()))
122 report_fatal_error("FPXX is not permitted for the N32/N64 ABI's.", false);
123
124 if (hasMips64r6() && InMicroMipsMode)
125 report_fatal_error("microMIPS64R6 is not supported", false);
126
127 if (!isABI_O32() && InMicroMipsMode)
128 report_fatal_error("microMIPS64 is not supported.", false);
129
130 if (UseIndirectJumpsHazard) {
131 if (InMicroMipsMode)
132 report_fatal_error(
133 "cannot combine indirect jumps with hazard barriers and microMIPS");
134 if (!hasMips32r2())
135 report_fatal_error(
136 "indirect jumps with hazard barriers requires MIPS32R2 or later");
137 }
138 if (inAbs2008Mode() && hasMips32() && !hasMips32r2()) {
139 report_fatal_error("IEEE 754-2008 abs.fmt is not supported for the given "
140 "architecture.",
141 false);
142 }
143
144 if (hasMips32r6()) {
145 StringRef ISA = hasMips64r6() ? "MIPS64r6" : "MIPS32r6";
146
147 assert(isFP64bit());
148 assert(isNaN2008());
149 assert(inAbs2008Mode());
150 if (hasDSP())
151 report_fatal_error(ISA + " is not compatible with the DSP ASE", false);
152 }
153
154 if (NoABICalls && TM.isPositionIndependent())
155 report_fatal_error("position-independent code requires '-mabicalls'");
156
157 if (isABI_N64() && !TM.isPositionIndependent() && !hasSym32())
158 NoABICalls = true;
159
160 // Set UseSmallSection.
161 UseSmallSection = GPOpt;
162 if (!NoABICalls && GPOpt) {
163 errs() << "warning: cannot use small-data accesses for '-mabicalls'"
164 << "\n";
165 UseSmallSection = false;
166 }
167
168 if (hasDSPR2() && !DspWarningPrinted) {
169 if (hasMips64() && !hasMips64r2()) {
170 errs() << "warning: the 'dspr2' ASE requires MIPS64 revision 2 or "
171 << "greater\n";
172 DspWarningPrinted = true;
173 } else if (hasMips32() && !hasMips32r2()) {
174 errs() << "warning: the 'dspr2' ASE requires MIPS32 revision 2 or "
175 << "greater\n";
176 DspWarningPrinted = true;
177 }
178 } else if (hasDSP() && !DspWarningPrinted) {
179 if (hasMips64() && !hasMips64r2()) {
180 errs() << "warning: the 'dsp' ASE requires MIPS64 revision 2 or "
181 << "greater\n";
182 DspWarningPrinted = true;
183 } else if (hasMips32() && !hasMips32r2()) {
184 errs() << "warning: the 'dsp' ASE requires MIPS32 revision 2 or "
185 << "greater\n";
186 DspWarningPrinted = true;
187 }
188 }
189
190 StringRef ArchName = hasMips64() ? "MIPS64" : "MIPS32";
191
192 if (!hasMips32r5() && hasMSA() && !MSAWarningPrinted) {
193 errs() << "warning: the 'msa' ASE requires " << ArchName
194 << " revision 5 or greater\n";
195 MSAWarningPrinted = true;
196 }
197 if (!hasMips32r5() && hasVirt() && !VirtWarningPrinted) {
198 errs() << "warning: the 'virt' ASE requires " << ArchName
199 << " revision 5 or greater\n";
200 VirtWarningPrinted = true;
201 }
202 if (!hasMips32r6() && hasCRC() && !CRCWarningPrinted) {
203 errs() << "warning: the 'crc' ASE requires " << ArchName
204 << " revision 6 or greater\n";
205 CRCWarningPrinted = true;
206 }
207 if (!hasMips32r6() && hasGINV() && !GINVWarningPrinted) {
208 errs() << "warning: the 'ginv' ASE requires " << ArchName
209 << " revision 6 or greater\n";
210 GINVWarningPrinted = true;
211 }
212
213 CallLoweringInfo.reset(new MipsCallLowering(*getTargetLowering()));
214 Legalizer.reset(new MipsLegalizerInfo(*this));
215
216 auto *RBI = new MipsRegisterBankInfo(*getRegisterInfo());
217 RegBankInfo.reset(RBI);
218 InstSelector.reset(createMipsInstructionSelector(
219 *static_cast<const MipsTargetMachine *>(&TM), *this, *RBI));
220}

241MipsSubtarget &
242MipsSubtarget::initializeSubtargetDependencies(StringRef CPU, StringRef FS,
243 const TargetMachine &TM) {
244 std::string CPUName = MIPS_MC::selectMipsCPU(TM.getTargetTriple(), CPU);
245 std::string CheriFeatures;
246 // enable capabilties for all cheri-*-* triples even if CPUName != cheri
247 if (TM.getTargetTriple().getArch() == llvm::Triple::cheri) {
248 if (FS.empty())
249 FS = "+chericap,+cheri128";
250 else {
251 CheriFeatures = FS;
252 CheriFeatures += ",+chericap";
253 if (!FS.contains("+cheri128") && !FS.contains("+cheri64"))
254 CheriFeatures += ",+cheri256";
255 FS = CheriFeatures;
256 }
257 }
258
259 // Parse features string.
260 ParseSubtargetFeatures(CPUName, FS);
261 // Initialize scheduling itinerary for the specified CPU.
262 InstrItins = getInstrItineraryForCPU(CPUName);
263
264 if (InMips16Mode && !IsSoftFloat)
265 InMips16HardFloat = true;
266
267 if (StackAlignOverride)
268 stackAlignment = StackAlignOverride;
269 else if (isCheri()) {
270 if (isCheri128())
271 stackAlignment = 16;
272 else
273 stackAlignment = 32;
274 } else if (isABI_N32() || isABI_N64())
275 stackAlignment = 16;
276 else {
277 assert(isABI_O32() && "Unknown ABI for stack alignment!");
278 stackAlignment = 8;
279 }
280
281 return *this;
282}

314void MipsSubtarget::overrideSchedPolicy(MachineSchedPolicy &Policy,
315 unsigned NumRegionInstrs) const {
316 // copied from AMDGPU:
317#if 0
318
319 // Track register pressure so the scheduler can try to decrease
320 // pressure once register usage is above the threshold defined by
321 // SIRegisterInfo::getRegPressureSetLimit()
322 Policy.ShouldTrackPressure = true;
323 // Enabling both top down and bottom up scheduling seems to give us less
324 // register spills than just using one of these approaches on its own.
325 Policy.OnlyTopDown = false;
326 Policy.OnlyBottomUp = true;
327
328 Policy.ShouldTrackLaneMasks = false;
329 Policy.DisableLatencyHeuristic = true;
