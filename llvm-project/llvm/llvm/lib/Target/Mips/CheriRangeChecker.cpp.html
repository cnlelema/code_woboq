28class CheriRangeChecker : public FunctionPass,
29 public InstVisitor<CheriRangeChecker> {
30 // Operands for an allocation. Either one or two integers (constant or
31 // variable). If there are two, then they must be multiplied together.
32 typedef std::tuple<Value *, Value *, cheri::SetBoundsPointerSource>
33 AllocOperands;
34 struct ConstantCast {
35 Instruction *Instr;
36 unsigned OpNo;
37 User *Origin;
38 };
39 std::unique_ptr<DataLayout> TD;
40 Module *M;
41 IntegerType *SizeTy;
42 PointerType *CapPtrTy;
43 SmallVector<pair<AllocOperands, Instruction *>, 32> Casts;
44 SmallVector<pair<AllocOperands, ConstantCast>, 32> ConstantCasts;
45 Function *SetLengthFn;
46
47 AllocOperands getRangeForAllocation(Value *Src) {
48 CallSite Malloc = CallSite(Src);
49 if (Malloc != CallSite()) {
50 Function *Fn = Malloc.getCalledFunction();
51 if (!Fn)
52 return AllocOperands();
53 switch (StringSwitch<int>(Fn->getName())
54 .Case("malloc", 1)
55 .Case("valloc", 1)
56 .Case("realloc", 2)
57 .Case("aligned_alloc", 2)
58 .Case("reallocf", 2)
59 .Case("calloc", 3)
60 .Default(-1)) {
61 default:
62 return AllocOperands();
63 case 1:
64 return AllocOperands(Malloc.getArgument(0), nullptr,
65 cheri::SetBoundsPointerSource::Heap);
66 case 2:
67 return AllocOperands(Malloc.getArgument(1), nullptr,
68 cheri::SetBoundsPointerSource::Heap);
69 case 3:
70 return AllocOperands(Malloc.getArgument(0), Malloc.getArgument(1),
71 cheri::SetBoundsPointerSource::Heap);
72 }
73 } else if (AllocaInst *AI = dyn_cast<AllocaInst>(Src)) {
74 PointerType *AllocaTy = AI->getType();
75 Value *ArraySize = AI->getArraySize();
76 Type *AllocationTy = AllocaTy->getElementType();
77 unsigned ElementSize = TD->getTypeAllocSize(AllocationTy);
78 if (ElementSize == 1)
79 return AllocOperands(ArraySize, nullptr,
80 cheri::SetBoundsPointerSource::Stack);
81 Value *Size = ConstantInt::get(ArraySize->getType(), ElementSize);
82 return AllocOperands(Size, ArraySize,
83 cheri::SetBoundsPointerSource::Stack);
84 }
85 return AllocOperands();
86 }
87 Value *RangeCheckedValue(Instruction *InsertPt, AllocOperands AO, Value *I2P,
88 Value *&BitCast) {
89 IRBuilder<> B(InsertPt);
90 Value *Size = (std::get<1>(AO))
91 ? B.CreateMul(std::get<0>(AO), std::get<1>(AO))
92 : std::get<0>(AO);
93 BitCast = B.CreateBitCast(I2P, CapPtrTy);
94 if (Size->getType() != SizeTy)
95 Size = B.CreateZExt(Size, SizeTy);
96 CallInst *SetLength = B.CreateCall(SetLengthFn, {BitCast, Size});
97 if (cheri::ShouldCollectCSetBoundsStats) {
98 // FIXME: can't be in add due to layering issues
99 Value *AlignmentSource = BitCast;
100 Instruction *DebugInst = dyn_cast<Instruction>(AlignmentSource);
101 if (!DebugInst)
102 DebugInst = InsertPt;
103 cheri::addSetBoundsStats(getKnownAlignment(AlignmentSource, *TD), Size,
104 getPassName(), std::get<2>(AO), "",
105 cheri::inferSourceLocation(DebugInst));
106 }
107 if (BitCast == I2P)
108 BitCast = SetLength;
109 return B.CreateBitCast(SetLength, I2P->getType());
110 }

112public:
113 static char ID;
114 CheriRangeChecker() : FunctionPass(ID) {}
115 StringRef getPassName() const override { return "CHERI range checker"; }
116 bool doInitialization(Module &Mod) override {
117 M = &Mod;
118 TD = llvm::make_unique<DataLayout>(M);
119 SizeTy = IntegerType::get(M->getContext(), TD->getIndexSizeInBits(200));
120 CapPtrTy = PointerType::get(IntegerType::get(M->getContext(), 8), 200);
121 return true;
122 }
123 virtual ~CheriRangeChecker() {}
124 bool checkOpcode(Value *V, unsigned Opcode) {
125 if (Instruction *I = dyn_cast<Instruction>(V))
126 return I->getOpcode() == Opcode;
127 if (ConstantExpr *E = dyn_cast<ConstantExpr>(V))
128 return E->getOpcode() == Opcode;
129 return false;
130 }
131 User *testI2P(User &I2P) {
132 PointerType *DestTy = dyn_cast<PointerType>(I2P.getType());
133 if (DestTy && isCheriPointer(DestTy, TD.get())) {
134 if (checkOpcode(I2P.getOperand(0), Instruction::PtrToInt)) {
135 User *P2I = cast<User>(I2P.getOperand(0));
136 PointerType *SrcTy =
137 dyn_cast<PointerType>(P2I->getOperand(0)->getType());
138 if (SrcTy && SrcTy->getAddressSpace() == 0) {
139 Value *Src = P2I->getOperand(0)->stripPointerCasts();
140 if (GlobalVariable *GV = dyn_cast<GlobalVariable>(Src))
141 return GV->hasExternalLinkage() ? 0 : P2I;
142 if (isa<AllocaInst>(Src) || CallSite(Src) != CallSite()) {
143 return P2I;
144 }
145 }
146 }
147 }
148 return 0;
149 }
150 void visitAddrSpaceCast(AddrSpaceCastInst &ASC) {
151 PointerType *DestTy = dyn_cast<PointerType>(ASC.getType());
152 Value *Src = ASC.getOperand(0);
153 PointerType *SrcTy = dyn_cast<PointerType>(Src->getType());
154 if ((DestTy && isCheriPointer(DestTy, TD.get())) &&
155 (SrcTy && SrcTy->getAddressSpace() == 0)) {
156 Src = Src->stripPointerCasts();
157 if (GlobalVariable *GV = dyn_cast<GlobalVariable>(Src)) {
158 if (GV->hasExternalLinkage())
159 return;
160 } else if (!(isa<AllocaInst>(Src) || CallSite(Src) != CallSite()))
161 return;
162 AllocOperands AO = getRangeForAllocation(Src);
163 if (AO != AllocOperands())
164 Casts.push_back(pair<AllocOperands, Instruction *>(AO, &ASC));
165 }
166 }
167 void visitIntToPtrInst(IntToPtrInst &I2P) {
168 if (User *P2I = testI2P(I2P)) {
169 Value *Src = P2I->getOperand(0)->stripPointerCasts();
170 AllocOperands AO = getRangeForAllocation(Src);
171 if (AO != AllocOperands())
172 Casts.push_back(pair<AllocOperands, Instruction *>(AO, &I2P));
173 }
174 }
175 void visitRet(ReturnInst &RI) {
176 Value *RV = RI.getReturnValue();
177 if (RV && isa<ConstantExpr>(RV)) {
178 ConstantCast C = {&RI, 0, testI2P(*cast<User>(RV))};
179 if (C.Origin) {
180 Value *Src = C.Origin->getOperand(0)->stripPointerCasts();
181 AllocOperands AO = getRangeForAllocation(Src);
182 if (AO != AllocOperands())
183 ConstantCasts.push_back(pair<AllocOperands, ConstantCast>(AO, C));
184 }
185 }
186 }
187 void visitCall(CallInst &CI) {
188 for (unsigned i = 0; i < CI.getNumOperands(); i++) {
189 Value *AV = CI.getOperand(i);
190 if (AV && isa<ConstantExpr>(AV)) {
191 ConstantCast C = {&CI, i, testI2P(*cast<User>(AV))};
192 if (C.Origin) {
193 Value *Src = C.Origin->getOperand(0)->stripPointerCasts();
194 AllocOperands AO = getRangeForAllocation(Src);
195 if (AO != AllocOperands())
196 ConstantCasts.push_back(pair<AllocOperands, ConstantCast>(AO, C));
197 }
198 }
199 }
200 }
201 bool runOnFunction(Function &F) override{
202 Casts.clear();
203 ConstantCasts.clear();
204
205 visit(F);
206
207 if (!(Casts.empty() && ConstantCasts.empty())) {
208 Intrinsic::ID SetLength = Intrinsic::cheri_cap_bounds_set;
209 SetLengthFn = Intrinsic::getDeclaration(M, SetLength, SizeTy);
210 Value *BitCast = 0;
211
212 for (auto *i = Casts.begin(), *e = Casts.end(); i != e; ++i) {
213 Instruction *I2P = i->second;
214 auto InsertPt = I2P->getParent()->begin();
215 while (&(*InsertPt) != I2P) {
216 ++InsertPt;
217 }
218 ++InsertPt;
219 Value *New = RangeCheckedValue(&*InsertPt, i->first,
220 I2P, BitCast);
221 I2P->replaceAllUsesWith(New);
222 cast<Instruction>(BitCast)->setOperand(0, I2P);
223 }
224 for (pair<AllocOperands, ConstantCast> *i = ConstantCasts.begin(),
225 *e = ConstantCasts.end();
226 i != e; ++i) {
227 Value *I2P = i->second.Instr->getOperand(i->second.OpNo);
228 Value *New = RangeCheckedValue(i->second.Instr, i->first, I2P, BitCast);
229 i->second.Instr->setOperand(i->second.OpNo, New);
230 }
231 return true;
232 }
233 return false;
234 }
235};
