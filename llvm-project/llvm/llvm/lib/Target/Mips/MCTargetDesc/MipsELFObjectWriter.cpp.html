71/// The possible results of the Predicate function used by find_best.
72enum FindBestPredicateResult {
73 FindBest_NoMatch = 0, ///< The current element is not a match.
74 FindBest_Match, ///< The current element is a match but better ones are
75 /// possible.
76 FindBest_PerfectMatch, ///< The current element is an unbeatable match.
77};

81/// Copy elements in the range [First, Last) to d1 when the predicate is true or
82/// d2 when the predicate is false. This is essentially both std::copy_if and
83/// std::remove_copy_if combined into a single pass.
84template <class InputIt, class OutputIt1, class OutputIt2, class UnaryPredicate>
85static std::pair<OutputIt1, OutputIt2> copy_if_else(InputIt First, InputIt Last,
86 OutputIt1 d1, OutputIt2 d2,
87 UnaryPredicate Predicate) {
88 for (InputIt I = First; I != Last; ++I) {
89 if (Predicate(*I)) {
90 *d1 = *I;
91 d1++;
92 } else {
93 *d2 = *I;
94 d2++;
95 }
96 }
97
98 return std::make_pair(d1, d2);
99}

101/// Find the best match in the range [First, Last).
102///
103/// An element matches when Predicate(X) returns FindBest_Match or
104/// FindBest_PerfectMatch. A value of FindBest_PerfectMatch also terminates
105/// the search. BetterThan(A, B) is a comparator that returns true when A is a
106/// better match than B. The return value is the position of the best match.
107///
108/// This is similar to std::find_if but finds the best of multiple possible
109/// matches.
110template <class InputIt, class UnaryPredicate, class Comparator>
111static InputIt find_best(InputIt First, InputIt Last, UnaryPredicate Predicate,
112 Comparator BetterThan) {
113 InputIt Best = Last;
114
115 for (InputIt I = First; I != Last; ++I) {
116 unsigned Matched = Predicate(*I);
117 if (Matched != FindBest_NoMatch) {
118 LLVM_DEBUG(dbgs() << std::distance(First, I) << " is a match (";
119 I->print(dbgs()); dbgs() << ")\n");
120 if (Best == Last || BetterThan(*I, *Best)) {
121 LLVM_DEBUG(dbgs() << ".. and it beats the last one\n");
122 Best = I;
123 }
124 }
125 if (Matched == FindBest_PerfectMatch) {
126 LLVM_DEBUG(dbgs() << ".. and it is unbeatable\n");
127 break;
128 }
129 }
130
131 return Best;
132}

134/// Determine the low relocation that matches the given relocation.
135/// If the relocation does not need a low relocation then the return value
136/// is ELF::R_MIPS_NONE.
137///
138/// The relocations that need a matching low part are
139/// R_(MIPS|MICROMIPS|MIPS16)_HI16 for all symbols and
140/// R_(MIPS|MICROMIPS|MIPS16)_GOT16 for local symbols only.
141static unsigned getMatchingLoType(const ELFRelocationEntry &Reloc) {
142 unsigned Type = Reloc.Type;
143 if (Type == ELF::R_MIPS_HI16)
144 return ELF::R_MIPS_LO16;
145 if (Type == ELF::R_MICROMIPS_HI16)
146 return ELF::R_MICROMIPS_LO16;
147 if (Type == ELF::R_MIPS16_HI16)
148 return ELF::R_MIPS16_LO16;
149
150 if (Reloc.OriginalSymbol &&
151 Reloc.OriginalSymbol->getBinding() != ELF::STB_LOCAL)
152 return ELF::R_MIPS_NONE;
153
154 if (Type == ELF::R_MIPS_GOT16)
155 return ELF::R_MIPS_LO16;
156 if (Type == ELF::R_MICROMIPS_GOT16)
157 return ELF::R_MICROMIPS_LO16;
158 if (Type == ELF::R_MIPS16_GOT16)
159 return ELF::R_MIPS16_LO16;
160
161 return ELF::R_MIPS_NONE;
162}

164/// Determine whether a relocation (X) matches the one given in R.
165///
166/// A relocation matches if:
167/// - It's type matches that of a corresponding low part. This is provided in
168/// MatchingType for efficiency.
169/// - It's based on the same symbol.
170/// - It's offset of greater or equal to that of the one given in R.
171/// It should be noted that this rule assumes the programmer does not use
172/// offsets that exceed the alignment of the symbol. The carry-bit will be
173/// incorrect if this is not true.
174///
175/// A matching relocation is unbeatable if:
176/// - It is not already involved in a match.
177/// - It's offset is exactly that of the one given in R.
178static FindBestPredicateResult isMatchingReloc(const MipsRelocationEntry &X,
179 const ELFRelocationEntry &R,
180 unsigned MatchingType) {
181 if (X.R.Type == MatchingType && X.R.OriginalSymbol == R.OriginalSymbol) {
182 if (!X.Matched &&
183 X.R.OriginalAddend == R.OriginalAddend)
184 return FindBest_PerfectMatch;
185 else if (X.R.OriginalAddend >= R.OriginalAddend)
186 return FindBest_Match;
187 }
188 return FindBest_NoMatch;
189}
190
191/// Determine whether Candidate or PreviousBest is the better match.
192/// The return value is true if Candidate is the better match.
193///
194/// A matching relocation is a better match if:
195/// - It has a smaller addend.
196/// - It is not already involved in a match.
197static bool compareMatchingRelocs(const MipsRelocationEntry &Candidate,
198 const MipsRelocationEntry &PreviousBest) {
199 if (Candidate.R.OriginalAddend != PreviousBest.R.OriginalAddend)
200 return Candidate.R.OriginalAddend < PreviousBest.R.OriginalAddend;
201 return PreviousBest.Matched && !Candidate.Matched;
202}

470/// Sort relocation table entries by offset except where another order is
471/// required by the MIPS ABI.
472///
473/// MIPS has a few relocations that have an AHL component in the expression used
474/// to evaluate them. This AHL component is an addend with the same number of
475/// bits as a symbol value but not all of our ABI's are able to supply a
476/// sufficiently sized addend in a single relocation.
477///
478/// The O32 ABI for example, uses REL relocations which store the addend in the
479/// section data. All the relocations with AHL components affect 16-bit fields
480/// so the addend for a single relocation is limited to 16-bit. This ABI
481/// resolves the limitation by linking relocations (e.g. R_MIPS_HI16 and
482/// R_MIPS_LO16) and distributing the addend between the linked relocations. The
483/// ABI mandates that such relocations must be next to each other in a
484/// particular order (e.g. R_MIPS_HI16 must be immediately followed by a
485/// matching R_MIPS_LO16) but the rule is less strict in practice.
486///
487/// The de facto standard is lenient in the following ways:
488/// - 'Immediately following' does not refer to the next relocation entry but
489/// the next matching relocation.
490/// - There may be multiple high parts relocations for one low part relocation.
491/// - There may be multiple low part relocations for one high part relocation.
492/// - The AHL addend in each part does not have to be exactly equal as long as
493/// the difference does not affect the carry bit from bit 15 into 16. This is
494/// to allow, for example, the use of %lo(foo) and %lo(foo+4) when loading
495/// both halves of a long long.
496///
497/// See getMatchingLoType() for a description of which high part relocations
498/// match which low part relocations. One particular thing to note is that
499/// R_MIPS_GOT16 and similar only have AHL addends if they refer to local
500/// symbols.
501///
502/// It should also be noted that this function is not affected by whether
503/// the symbol was kept or rewritten into a section-relative equivalent. We
504/// always match using the expressions from the source.
505void MipsELFObjectWriter::sortRelocs(const MCAssembler &Asm,
506 std::vector<ELFRelocationEntry> &Relocs) {
507 // We do not need to sort the relocation table for RELA relocations which
508 // N32/N64 uses as the relocation addend contains the value we require,
509 // rather than it being split across a pair of relocations.
510 if (hasRelocationAddend())
511 return;
512
513 if (Relocs.size() < 2)
514 return;
515
516 // Sort relocations by the address they are applied to.
517 llvm::sort(Relocs,
518 [](const ELFRelocationEntry &A, const ELFRelocationEntry &B) {
519 return A.Offset < B.Offset;
520 });
521
522 std::list<MipsRelocationEntry> Sorted;
523 std::list<ELFRelocationEntry> Remainder;
524
525 LLVM_DEBUG(dumpRelocs("R: ", Relocs));
526
527 // Separate the movable relocations (AHL relocations using the high bits) from
528 // the immobile relocations (everything else). This does not preserve high/low
529 // matches that already existed in the input.
530 copy_if_else(Relocs.begin(), Relocs.end(), std::back_inserter(Remainder),
531 std::back_inserter(Sorted), [](const ELFRelocationEntry &Reloc) {
532 return getMatchingLoType(Reloc) != ELF::R_MIPS_NONE;
533 });
534
535 for (auto &R : Remainder) {
536 LLVM_DEBUG(dbgs() << "Matching: " << R << "\n");
537
538 unsigned MatchingType = getMatchingLoType(R);
539 assert(MatchingType != ELF::R_MIPS_NONE &&
540 "Wrong list for reloc that doesn't need a match");
541
542 // Find the best matching relocation for the current high part.
543 // See isMatchingReloc for a description of a matching relocation and
544 // compareMatchingRelocs for a description of what 'best' means.
545 auto InsertionPoint =
546 find_best(Sorted.begin(), Sorted.end(),
547 [&R, &MatchingType](const MipsRelocationEntry &X) {
548 return isMatchingReloc(X, R, MatchingType);
549 },
550 compareMatchingRelocs);
551
552 // If we matched then insert the high part in front of the match and mark
553 // both relocations as being involved in a match. We only mark the high
554 // part for cosmetic reasons in the debug output.
555 //
556 // If we failed to find a match then the high part is orphaned. This is not
557 // permitted since the relocation cannot be evaluated without knowing the
558 // carry-in. We can sometimes handle this using a matching low part that is
559 // already used in a match but we already cover that case in
560 // isMatchingReloc and compareMatchingRelocs. For the remaining cases we
561 // should insert the high part at the end of the list. This will cause the
562 // linker to fail but the alternative is to cause the linker to bind the
563 // high part to a semi-matching low part and silently calculate the wrong
564 // value. Unfortunately we have no means to warn the user that we did this
565 // so leave it up to the linker to complain about it.
566 if (InsertionPoint != Sorted.end())
567 InsertionPoint->Matched = true;
568 Sorted.insert(InsertionPoint, R)->Matched = true;
569 }
570
571 LLVM_DEBUG(dumpRelocs("S: ", Sorted));
572
573 assert(Relocs.size() == Sorted.size() && "Some relocs were not consumed");
574
575 // Overwrite the original vector with the sorted elements. The caller expects
576 // them in reverse order.
577 unsigned CopyTo = 0;
578 for (const auto &R : reverse(Sorted))
579 Relocs[CopyTo++] = R.R;
580}

582bool MipsELFObjectWriter::needsRelocateWithSymbol(const MCSymbol &Sym,
583 unsigned Type) const {
584 // If it's a compound relocation for N64 then we need the relocation if any
585 // sub-relocation needs it.
586 if (!isUInt<8>(Type))
587 return needsRelocateWithSymbol(Sym, Type & 0xff) ||
588 needsRelocateWithSymbol(Sym, (Type >> 8) & 0xff) ||
589 needsRelocateWithSymbol(Sym, (Type >> 16) & 0xff);
590
591 switch (Type) {
592 default:
593 errs() << Type << "\n";
594 llvm_unreachable("Unexpected relocation");
595 return true;
596
597 // This relocation doesn't affect the section data.
598 case ELF::R_MIPS_NONE:
599 return false;
600
601 // On REL ABI's (e.g. O32), these relocations form pairs. The pairing is done
602 // by the static linker by matching the symbol and offset.
603 // We only see one relocation at a time but it's still safe to relocate with
604 // the section so long as both relocations make the same decision.
605 //
606 // Some older linkers may require the symbol for particular cases. Such cases
607 // are not supported yet but can be added as required.
608 case ELF::R_MIPS_GOT16:
609 case ELF::R_MIPS16_GOT16:
610 case ELF::R_MICROMIPS_GOT16:
611 case ELF::R_MIPS_HIGHER:
612 case ELF::R_MIPS_HIGHEST:
613 case ELF::R_MIPS_HI16:
614 case ELF::R_MIPS16_HI16:
615 case ELF::R_MICROMIPS_HI16:
616 case ELF::R_MIPS_LO16:
617 case ELF::R_MIPS16_LO16:
618 case ELF::R_MICROMIPS_LO16:
619 // FIXME: It should be safe to return false for the STO_MIPS_MICROMIPS but
620 // we neglect to handle the adjustment to the LSB of the addend that
621 // it causes in applyFixup() and similar.
622 if (cast<MCSymbolELF>(Sym).getOther() & ELF::STO_MIPS_MICROMIPS)
623 return true;
624 return false;
625
626 case ELF::R_MIPS_GOT_PAGE:
627 case ELF::R_MICROMIPS_GOT_PAGE:
628 case ELF::R_MIPS_GOT_OFST:
629 case ELF::R_MICROMIPS_GOT_OFST:
630 case ELF::R_MIPS_16:
631 case ELF::R_MIPS_32:
632 case ELF::R_MIPS_GPREL32:
633 if (cast<MCSymbolELF>(Sym).getOther() & ELF::STO_MIPS_MICROMIPS)
634 return true;
635 LLVM_FALLTHROUGH;
636 case ELF::R_MIPS_26:
637 case ELF::R_MIPS_64:
638 case ELF::R_MIPS_GPREL16:
639 case ELF::R_MIPS_PC16:
640 case ELF::R_MIPS_SUB:
641 return false;
642
643 // CHERI Capability relocations need to preserve the symbol in order to be
644 // able to get tight bounds on the resulting capability.
645 case ELF::R_MIPS_CHERI_CAPTAB_LO16:
646 case ELF::R_MIPS_CHERI_CAPTAB_HI16:
647 case ELF::R_MIPS_CHERI_CAPTAB_CLC11:
648 case ELF::R_MIPS_CHERI_CAPTAB20:
649 case ELF::R_MIPS_CHERI_CAPCALL_LO16:
650 case ELF::R_MIPS_CHERI_CAPCALL_HI16:
651 case ELF::R_MIPS_CHERI_CAPCALL_CLC11:
652 case ELF::R_MIPS_CHERI_CAPCALL20:
653 case ELF::R_MIPS_CHERI_CAPABILITY:
654 return true;
655 case ELF::R_MIPS_CHERI_SETTAG:
656 case ELF::R_MIPS_CHERI_SETTAG_LOADADDR:
657 case ELF::R_MIPS_CHERI_LOCAL_CAPABILITY:
658 case ELF::R_MIPS_CHERI_GLOBALS_TABLE:
659 llvm_unreachable("This relocation should not be genreated yet!");
660 return true;
661 // We want to keep the symbol for CAPTABLEREL since otherwise the offset to
662 // $pcc could be wrong
663 case ELF::R_MIPS_CHERI_CAPTABLEREL16:
664 return true;
665
666 // FIXME: Many of these relocations should probably return false but this
667 // hasn't been confirmed to be safe yet.
668 case ELF::R_MIPS_REL32:
669 case ELF::R_MIPS_LITERAL:
670 case ELF::R_MIPS_CALL16:
671 case ELF::R_MIPS_SHIFT5:
672 case ELF::R_MIPS_SHIFT6:
673 case ELF::R_MIPS_GOT_DISP:
674 case ELF::R_MIPS_GOT_HI16:
675 case ELF::R_MIPS_GOT_LO16:
676 case ELF::R_MIPS_INSERT_A:
677 case ELF::R_MIPS_INSERT_B:
678 case ELF::R_MIPS_DELETE:
679 case ELF::R_MIPS_CALL_HI16:
680 case ELF::R_MIPS_CALL_LO16:
681 case ELF::R_MIPS_SCN_DISP:
682 case ELF::R_MIPS_REL16:
683 case ELF::R_MIPS_ADD_IMMEDIATE:
684 case ELF::R_MIPS_PJUMP:
685 case ELF::R_MIPS_RELGOT:
686 case ELF::R_MIPS_JALR:
687 case ELF::R_MIPS_TLS_DTPMOD32:
688 case ELF::R_MIPS_TLS_DTPREL32:
689 case ELF::R_MIPS_TLS_DTPMOD64:
690 case ELF::R_MIPS_TLS_DTPREL64:
691 case ELF::R_MIPS_TLS_GD:
692 case ELF::R_MIPS_TLS_LDM:
693 case ELF::R_MIPS_TLS_DTPREL_HI16:
694 case ELF::R_MIPS_TLS_DTPREL_LO16:
695 case ELF::R_MIPS_TLS_GOTTPREL:
696 case ELF::R_MIPS_TLS_TPREL32:
697 case ELF::R_MIPS_TLS_TPREL64:
698 case ELF::R_MIPS_TLS_TPREL_HI16:
699 case ELF::R_MIPS_TLS_TPREL_LO16:
700 case ELF::R_MIPS_GLOB_DAT:
701 case ELF::R_MIPS_PC21_S2:
702 case ELF::R_MIPS_PC26_S2:
703 case ELF::R_MIPS_PC18_S3:
704 case ELF::R_MIPS_PC19_S2:
705 case ELF::R_MIPS_PCHI16:
706 case ELF::R_MIPS_PCLO16:
707 case ELF::R_MIPS_COPY:
708 case ELF::R_MIPS_JUMP_SLOT:
709 case ELF::R_MIPS_NUM:
710 case ELF::R_MIPS_PC32:
711 case ELF::R_MIPS_EH:
712 case ELF::R_MICROMIPS_26_S1:
713 case ELF::R_MICROMIPS_GPREL16:
714 case ELF::R_MICROMIPS_LITERAL:
715 case ELF::R_MICROMIPS_PC7_S1:
716 case ELF::R_MICROMIPS_PC10_S1:
717 case ELF::R_MICROMIPS_PC16_S1:
718 case ELF::R_MICROMIPS_CALL16:
719 case ELF::R_MICROMIPS_GOT_DISP:
720 case ELF::R_MICROMIPS_GOT_HI16:
721 case ELF::R_MICROMIPS_GOT_LO16:
722 case ELF::R_MICROMIPS_SUB:
723 case ELF::R_MICROMIPS_HIGHER:
724 case ELF::R_MICROMIPS_HIGHEST:
725 case ELF::R_MICROMIPS_CALL_HI16:
726 case ELF::R_MICROMIPS_CALL_LO16:
727 case ELF::R_MICROMIPS_SCN_DISP:
728 case ELF::R_MICROMIPS_JALR:
729 case ELF::R_MICROMIPS_HI0_LO16:
730 case ELF::R_MICROMIPS_TLS_GD:
731 case ELF::R_MICROMIPS_TLS_LDM:
732 case ELF::R_MICROMIPS_TLS_DTPREL_HI16:
733 case ELF::R_MICROMIPS_TLS_DTPREL_LO16:
734 case ELF::R_MICROMIPS_TLS_GOTTPREL:
735 case ELF::R_MICROMIPS_TLS_TPREL_HI16:
736 case ELF::R_MICROMIPS_TLS_TPREL_LO16:
737 case ELF::R_MICROMIPS_GPREL7_S2:
738 case ELF::R_MICROMIPS_PC23_S2:
739 case ELF::R_MICROMIPS_PC21_S1:
740 case ELF::R_MICROMIPS_PC26_S1:
741 case ELF::R_MICROMIPS_PC18_S3:
742 case ELF::R_MICROMIPS_PC19_S2:
743 return true;
744
745 // FIXME: Many of these should probably return false but MIPS16 isn't
746 // supported by the integrated assembler.
747 case ELF::R_MIPS16_26:
748 case ELF::R_MIPS16_GPREL:
749 case ELF::R_MIPS16_CALL16:
750 case ELF::R_MIPS16_TLS_GD:
751 case ELF::R_MIPS16_TLS_LDM:
752 case ELF::R_MIPS16_TLS_DTPREL_HI16:
753 case ELF::R_MIPS16_TLS_DTPREL_LO16:
754 case ELF::R_MIPS16_TLS_GOTTPREL:
755 case ELF::R_MIPS16_TLS_TPREL_HI16:
756 case ELF::R_MIPS16_TLS_TPREL_LO16:
757 llvm_unreachable("Unsupported MIPS16 relocation");
758 return true;
759 }

762std::unique_ptr<MCObjectTargetWriter>
763llvm::createMipsELFObjectWriter(const Triple &TT, bool IsN32) {
764 uint8_t OSABI = MCELFObjectTargetWriter::getOSABI(TT.getOS());
765 bool IsN64 = TT.isArch64Bit() && !IsN32;
766 bool HasRelocationAddend = TT.isArch64Bit();
767 return llvm::make_unique<MipsELFObjectWriter>(OSABI, HasRelocationAddend,
768 IsN64);
769}
