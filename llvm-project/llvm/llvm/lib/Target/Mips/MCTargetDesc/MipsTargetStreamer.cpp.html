39llvm::Optional<unsigned> llvm::getCheriCapabilitySize(FeatureBitset Features) {
40 if (Features[Mips::FeatureMipsCheri256]) {
41 assert(Features[Mips::FeatureMipsCheri]);
42 return 32;
43 }
44 if (Features[Mips::FeatureMipsCheri128]) {
45 assert(Features[Mips::FeatureMipsCheri]);
46 return 16;
47 }
48 if (Features[Mips::FeatureMipsCheri64]) {
49 assert(Features[Mips::FeatureMipsCheri]);
50 return 8;
51 }
52 assert(!Features[Mips::FeatureMipsCheri]);
53 return None;
54}

285/// Emit a store instruction with an immediate offset.
286void MipsTargetStreamer::emitStoreWithImmOffset(
287 unsigned Opcode, unsigned SrcReg, unsigned BaseReg, int64_t Offset,
288 function_ref<unsigned()> GetATReg, SMLoc IDLoc,
289 const MCSubtargetInfo *STI) {
290 if (isInt<16>(Offset)) {
291 emitRRI(Opcode, SrcReg, BaseReg, Offset, IDLoc, STI);
292 return;
293 }
294
295 // sw $8, offset($8) => lui $at, %hi(offset)
296 // add $at, $at, $8
297 // sw $8, %lo(offset)($at)
298
299 unsigned ATReg = GetATReg();
300 if (!ATReg)
301 return;
302
303 unsigned LoOffset = Offset & 0x0000ffff;
304 unsigned HiOffset = (Offset & 0xffff0000) >> 16;
305
306 // If msb of LoOffset is 1(negative number) we must increment HiOffset
307 // to account for the sign-extension of the low part.
308 if (LoOffset & 0x8000)
309 HiOffset++;
310
311 // Generate the base address in ATReg.
312 emitRI(Mips::LUi, ATReg, HiOffset, IDLoc, STI);
313 if (BaseReg != Mips::ZERO)
314 emitRRR(Mips::ADDu, ATReg, ATReg, BaseReg, IDLoc, STI);
315 // Emit the store with the adjusted base and offset.
316 emitRRI(Opcode, SrcReg, ATReg, LoOffset, IDLoc, STI);
317}

319/// Emit a store instruction with an symbol offset. Symbols are assumed to be
320/// out of range for a simm16 will be expanded to appropriate instructions.
321void MipsTargetStreamer::emitStoreWithSymOffset(
322 unsigned Opcode, unsigned SrcReg, unsigned BaseReg, MCOperand &HiOperand,
323 MCOperand &LoOperand, unsigned ATReg, SMLoc IDLoc,
324 const MCSubtargetInfo *STI) {
325 // sw $8, sym => lui $at, %hi(sym)
326 // sw $8, %lo(sym)($at)
327
328 // Generate the base address in ATReg.
329 emitRX(Mips::LUi, ATReg, HiOperand, IDLoc, STI);
330 if (BaseReg != Mips::ZERO)
331 emitRRR(Mips::ADDu, ATReg, ATReg, BaseReg, IDLoc, STI);
332 // Emit the store with the adjusted base and offset.
333 emitRRX(Opcode, SrcReg, ATReg, LoOperand, IDLoc, STI);
334}

336/// Emit a load instruction with an immediate offset. DstReg and TmpReg are
337/// permitted to be the same register iff DstReg is distinct from BaseReg and
338/// DstReg is a GPR. It is the callers responsibility to identify such cases
339/// and pass the appropriate register in TmpReg.
340void MipsTargetStreamer::emitLoadWithImmOffset(unsigned Opcode, unsigned DstReg,
341 unsigned BaseReg, int64_t Offset,
342 unsigned TmpReg, SMLoc IDLoc,
343 const MCSubtargetInfo *STI) {
344 if (isInt<16>(Offset)) {
345 emitRRI(Opcode, DstReg, BaseReg, Offset, IDLoc, STI);
346 return;
347 }
348
349 // 1) lw $8, offset($9) => lui $8, %hi(offset)
350 // add $8, $8, $9
351 // lw $8, %lo(offset)($9)
352 // 2) lw $8, offset($8) => lui $at, %hi(offset)
353 // add $at, $at, $8
354 // lw $8, %lo(offset)($at)
355
356 unsigned LoOffset = Offset & 0x0000ffff;
357 unsigned HiOffset = (Offset & 0xffff0000) >> 16;
358
359 // If msb of LoOffset is 1(negative number) we must increment HiOffset
360 // to account for the sign-extension of the low part.
361 if (LoOffset & 0x8000)
362 HiOffset++;
363
364 // Generate the base address in TmpReg.
365 emitRI(Mips::LUi, TmpReg, HiOffset, IDLoc, STI);
366 if (BaseReg != Mips::ZERO)
367 emitRRR(Mips::ADDu, TmpReg, TmpReg, BaseReg, IDLoc, STI);
368 // Emit the load with the adjusted base and offset.
369 emitRRI(Opcode, DstReg, TmpReg, LoOffset, IDLoc, STI);
370}

372/// Emit a load instruction with an symbol offset. Symbols are assumed to be
373/// out of range for a simm16 will be expanded to appropriate instructions.
374/// DstReg and TmpReg are permitted to be the same register iff DstReg is a
375/// GPR. It is the callers responsibility to identify such cases and pass the
376/// appropriate register in TmpReg.
377void MipsTargetStreamer::emitLoadWithSymOffset(unsigned Opcode, unsigned DstReg,
378 unsigned BaseReg,
379 MCOperand &HiOperand,
380 MCOperand &LoOperand,
381 unsigned TmpReg, SMLoc IDLoc,
382 const MCSubtargetInfo *STI) {
383 // 1) lw $8, sym => lui $8, %hi(sym)
384 // lw $8, %lo(sym)($8)
385 // 2) ldc1 $f0, sym => lui $at, %hi(sym)
386 // ldc1 $f0, %lo(sym)($at)
387
388 // Generate the base address in TmpReg.
389 emitRX(Mips::LUi, TmpReg, HiOperand, IDLoc, STI);
390 if (BaseReg != Mips::ZERO)
391 emitRRR(Mips::ADDu, TmpReg, TmpReg, BaseReg, IDLoc, STI);
392 // Emit the load with the adjusted base and offset.
393 emitRRX(Opcode, DstReg, TmpReg, LoOperand, IDLoc, STI);
394}

817// This part is for ELF object output.
818MipsTargetELFStreamer::MipsTargetELFStreamer(MCStreamer &S,
819 const MCSubtargetInfo &STI)
820 : MipsTargetStreamer(S, getCheriCapabilitySize(STI.getFeatureBits())),
821 MicroMipsEnabled(false), STI(STI) {
822 MCAssembler &MCA = getStreamer().getAssembler();
823
824 // It's possible that MCObjectFileInfo isn't fully initialized at this point
825 // due to an initialization order problem where LLVMTargetMachine creates the
826 // target streamer before TargetLoweringObjectFile calls
827 // InitializeMCObjectFileInfo. There doesn't seem to be a single place that
828 // covers all cases so this statement covers most cases and direct object
829 // emission must call setPic() once MCObjectFileInfo has been initialized. The
830 // cases we don't handle here are covered by MipsAsmPrinter.
831 Pic = MCA.getContext().getObjectFileInfo()->isPositionIndependent();
832
833 const FeatureBitset &Features = STI.getFeatureBits();
834
835 // Set the header flags that we can in the constructor.
836 // FIXME: This is a fairly terrible hack. We set the rest
837 // of these in the destructor. The problem here is two-fold:
838 //
839 // a: Some of the eflags can be set/reset by directives.
840 // b: There aren't any usage paths that initialize the ABI
841 // pointer until after we initialize either an assembler
842 // or the target machine.
843 // We can fix this by making the target streamer construct
844 // the ABI, but this is fraught with wide ranging dependency
845 // issues as well.
846 unsigned EFlags = MCA.getELFHeaderEFlags();
847
848 // FIXME: Fix a dependency issue by instantiating the ABI object to some
849 // default based off the triple. The triple doesn't describe the target
850 // fully, but any external user of the API that uses the MCTargetStreamer
851 // would otherwise crash on assertion failure.
852
853 ABI = MipsABIInfo(
854 STI.getTargetTriple().getArch() == Triple::ArchType::mipsel ||
855 STI.getTargetTriple().getArch() == Triple::ArchType::mips
856 ? MipsABIInfo::O32()
857 : MipsABIInfo::N64());
858
859 // Architecture
860 if (Features[Mips::FeatureMips64r6])
861 EFlags |= ELF::EF_MIPS_ARCH_64R6;
862 else if (Features[Mips::FeatureMips64r2] ||
863 Features[Mips::FeatureMips64r3] ||
864 Features[Mips::FeatureMips64r5])
865 EFlags |= ELF::EF_MIPS_ARCH_64R2;
866 else if (Features[Mips::FeatureMips64])
867 EFlags |= ELF::EF_MIPS_ARCH_64;
868 else if (Features[Mips::FeatureMips5])
869 EFlags |= ELF::EF_MIPS_ARCH_5;
870 else if (Features[Mips::FeatureMips4])
871 EFlags |= ELF::EF_MIPS_ARCH_4;
872 else if (Features[Mips::FeatureMips3])
873 EFlags |= ELF::EF_MIPS_ARCH_3;
874 else if (Features[Mips::FeatureMips32r6])
875 EFlags |= ELF::EF_MIPS_ARCH_32R6;
876 else if (Features[Mips::FeatureMips32r2] ||
877 Features[Mips::FeatureMips32r3] ||
878 Features[Mips::FeatureMips32r5])
879 EFlags |= ELF::EF_MIPS_ARCH_32R2;
880 else if (Features[Mips::FeatureMips32])
881 EFlags |= ELF::EF_MIPS_ARCH_32;
882 else if (Features[Mips::FeatureMips2])
883 EFlags |= ELF::EF_MIPS_ARCH_2;
884 else
885 EFlags |= ELF::EF_MIPS_ARCH_1;
886
887 // Machine
888 if (Features[Mips::FeatureCnMips])
889 EFlags |= ELF::EF_MIPS_MACH_OCTEON;
890 else if (Features[Mips::FeatureMipsCheri128])
891 EFlags |= ELF::EF_MIPS_MACH_CHERI128;
892 else if (Features[Mips::FeatureMipsCheri])
893 EFlags |= ELF::EF_MIPS_MACH_CHERI256;
894 else if (Features[Mips::FeatureMipsBeri])
895 EFlags |= ELF::EF_MIPS_MACH_BERI;
896
897 // Other options.
898 if (Features[Mips::FeatureNaN2008])
899 EFlags |= ELF::EF_MIPS_NAN2008;
900
901 MCA.setELFHeaderEFlags(EFlags);
902}

915void MipsTargetELFStreamer::finish() {
916 MCAssembler &MCA = getStreamer().getAssembler();
917 const MCObjectFileInfo &OFI = *MCA.getContext().getObjectFileInfo();
918
919 // .bss, .text and .data are always at least 16-byte aligned.
920 MCSection &TextSection = *OFI.getTextSection();
921 MCA.registerSection(TextSection);
922 MCSection &DataSection = *OFI.getDataSection();
923 MCA.registerSection(DataSection);
924 MCSection &BSSSection = *OFI.getBSSSection();
925 MCA.registerSection(BSSSection);
926
927 TextSection.setAlignment(std::max(16u, TextSection.getAlignment()));
928 DataSection.setAlignment(std::max(16u, DataSection.getAlignment()));
929 BSSSection.setAlignment(std::max(16u, BSSSection.getAlignment()));
930
931 if (RoundSectionSizes) {
932 // Make sections sizes a multiple of the alignment. This is useful for
933 // verifying the output of IAS against the output of other assemblers but
934 // it's not necessary to produce a correct object and increases section
935 // size.
936 MCStreamer &OS = getStreamer();
937 for (MCSection &S : MCA) {
938 MCSectionELF &Section = static_cast<MCSectionELF &>(S);
939
940 unsigned Alignment = Section.getAlignment();
941 if (Alignment) {
942 OS.SwitchSection(&Section);
943 if (Section.UseCodeAlign())
944 OS.EmitCodeAlignment(Alignment, Alignment);
945 else
946 OS.EmitValueToAlignment(Alignment, 0, 1, Alignment);
947 }
948 }
949 }
950
951 const FeatureBitset &Features = STI.getFeatureBits();
952
953 // Update e_header flags. See the FIXME and comment above in
954 // the constructor for a full rundown on this.
955 unsigned EFlags = MCA.getELFHeaderEFlags();
956
957 // ABI
958 // N64 does not require any ABI bits.
959 if (getABI().IsO32())
960 EFlags |= ELF::EF_MIPS_ABI_O32;
961 else if (getABI().IsN32())
962 EFlags |= ELF::EF_MIPS_ABI2;
963
964 if (Features[Mips::FeatureGP64Bit]) {
965 if (getABI().IsO32())
966 EFlags |= ELF::EF_MIPS_32BITMODE; /* Compatibility Mode */
967 } else if (Features[Mips::FeatureMips64r2] || Features[Mips::FeatureMips64])
968 EFlags |= ELF::EF_MIPS_32BITMODE;
969
970 // -mplt is not implemented but we should act as if it was
971 // given.
972 if (!Features[Mips::FeatureNoABICalls])
973 EFlags |= ELF::EF_MIPS_CPIC;
974
975 if (Pic)
976 EFlags |= ELF::EF_MIPS_PIC | ELF::EF_MIPS_CPIC;
977
978 if (getABI().IsCheriPureCap())
979 EFlags |= ELF::EF_MIPS_ABI_CHERIABI;
980
981 MCA.setELFHeaderEFlags(EFlags);
982
983 // Emit all the option records.
984 // At the moment we are only emitting .Mips.options (ODK_REGINFO) and
985 // .reginfo.
986 MipsELFStreamer &MEF = static_cast<MipsELFStreamer &>(Streamer);
987 MEF.EmitMipsOptionRecords();
988
989 emitMipsAbiFlags();
990}

1043void MipsTargetELFStreamer::emitDirectiveEnd(StringRef Name) {
1044 MCAssembler &MCA = getStreamer().getAssembler();
1045 MCContext &Context = MCA.getContext();
1046 MCStreamer &OS = getStreamer();
1047
1048 MCSectionELF *Sec = Context.getELFSection(".pdr", ELF::SHT_PROGBITS, 0);
1049
1050 MCSymbol *Sym = Context.getOrCreateSymbol(Name);
1051 const MCSymbolRefExpr *ExprRef =
1052 MCSymbolRefExpr::create(Sym, MCSymbolRefExpr::VK_None, Context);
1053
1054 MCA.registerSection(*Sec);
1055 Sec->setAlignment(4);
1056
1057 OS.PushSection();
1058
1059 OS.SwitchSection(Sec);
1060
1061 OS.EmitValueImpl(ExprRef, 4);
1062
1063 OS.EmitIntValue(GPRInfoSet ? GPRBitMask : 0, 4); // reg_mask
1064 OS.EmitIntValue(GPRInfoSet ? GPROffset : 0, 4); // reg_offset
1065
1066 OS.EmitIntValue(FPRInfoSet ? FPRBitMask : 0, 4); // fpreg_mask
1067 OS.EmitIntValue(FPRInfoSet ? FPROffset : 0, 4); // fpreg_offset
1068
1069 OS.EmitIntValue(FrameInfoSet ? FrameOffset : 0, 4); // frame_offset
1070 OS.EmitIntValue(FrameInfoSet ? FrameReg : 0, 4); // frame_reg
1071 OS.EmitIntValue(FrameInfoSet ? ReturnReg : 0, 4); // return_reg
1072
1073 // The .end directive marks the end of a procedure. Invalidate
1074 // the information gathered up until this point.
1075 GPRInfoSet = FPRInfoSet = FrameInfoSet = false;
1076
1077 OS.PopSection();
1078
1079 // .end also implicitly sets the size.
1080 MCSymbol *CurPCSym = Context.createTempSymbol();
1081 OS.EmitLabel(CurPCSym);
1082 const MCExpr *Size = MCBinaryExpr::createSub(
1083 MCSymbolRefExpr::create(CurPCSym, MCSymbolRefExpr::VK_None, Context),
1084 ExprRef, Context);
1085
1086 // The ELFObjectWriter can determine the absolute size as it has access to
1087 // the layout information of the assembly file, so a size expression rather
1088 // than an absolute value is ok here.
1089 static_cast<MCSymbolELF *>(Sym)->setSize(Size);
1090}

1172void MipsTargetELFStreamer::emitDirectiveCpLoad(unsigned RegNo) {
1173 // .cpload $reg
1174 // This directive expands to:
1175 // lui $gp, %hi(_gp_disp)
1176 // addui $gp, $gp, %lo(_gp_disp)
1177 // addu $gp, $gp, $reg
1178 // when support for position independent code is enabled.
1179 if (!Pic || (getABI().IsN32() || getABI().IsN64()))
1180 return;
1181
1182 // There's a GNU extension controlled by -mno-shared that allows
1183 // locally-binding symbols to be accessed using absolute addresses.
1184 // This is currently not supported. When supported -mno-shared makes
1185 // .cpload expand to:
1186 // lui $gp, %hi(__gnu_local_gp)
1187 // addiu $gp, $gp, %lo(__gnu_local_gp)
1188
1189 StringRef SymName("_gp_disp");
1190 MCAssembler &MCA = getStreamer().getAssembler();
1191 MCSymbol *GP_Disp = MCA.getContext().getOrCreateSymbol(SymName);
1192 MCA.registerSymbol(*GP_Disp);
1193
1194 MCInst TmpInst;
1195 TmpInst.setOpcode(Mips::LUi);
1196 TmpInst.addOperand(MCOperand::createReg(Mips::GP));
1197 const MCExpr *HiSym = MipsMCExpr::create(
1198 MipsMCExpr::MEK_HI,
1199 MCSymbolRefExpr::create("_gp_disp", MCSymbolRefExpr::VK_None,
1200 MCA.getContext()),
1201 MCA.getContext());
1202 TmpInst.addOperand(MCOperand::createExpr(HiSym));
1203 getStreamer().EmitInstruction(TmpInst, STI);
1204
1205 TmpInst.clear();
1206
1207 TmpInst.setOpcode(Mips::ADDiu);
1208 TmpInst.addOperand(MCOperand::createReg(Mips::GP));
1209 TmpInst.addOperand(MCOperand::createReg(Mips::GP));
1210 const MCExpr *LoSym = MipsMCExpr::create(
1211 MipsMCExpr::MEK_LO,
1212 MCSymbolRefExpr::create("_gp_disp", MCSymbolRefExpr::VK_None,
1213 MCA.getContext()),
1214 MCA.getContext());
1215 TmpInst.addOperand(MCOperand::createExpr(LoSym));
1216 getStreamer().EmitInstruction(TmpInst, STI);
1217
1218 TmpInst.clear();
1219
1220 TmpInst.setOpcode(Mips::ADDu);
1221 TmpInst.addOperand(MCOperand::createReg(Mips::GP));
1222 TmpInst.addOperand(MCOperand::createReg(Mips::GP));
1223 TmpInst.addOperand(MCOperand::createReg(RegNo));
1224 getStreamer().EmitInstruction(TmpInst, STI);
1225
1226 forbidModuleDirective();
1227}

1230 int Offset, function_ref<unsigned()> GetATReg, SMLoc IDLoc,
1231 const MCSubtargetInfo *STI) {

1233 // .cprestore offset
1234 // When PIC mode is enabled and the O32 ABI is used, this directive expands
1235 // to:
1236 // sw $gp, offset($sp)
1237 // and adds a corresponding LW after every JAL.

1240 // is used in non-PIC mode.

1242 return true;

1250void MipsTargetELFStreamer::emitDirectiveCpsetup(unsigned RegNo,
1251 int RegOrOffset,
1252 const MCSymbol &Sym,
1253 bool IsReg) {
1254 // Only N32 and N64 emit anything for .cpsetup iff PIC is set.
1255 if (!Pic || !(getABI().IsN32() || getABI().IsN64()))
1256 return;
1257
1258 forbidModuleDirective();
1259
1260 MCAssembler &MCA = getStreamer().getAssembler();
1261 MCInst Inst;
1262
1263 // Either store the old $gp in a register or on the stack
1264 if (IsReg) {
1265 // move $save, $gpreg
1266 emitRRR(Mips::OR64, RegOrOffset, Mips::GP, Mips::ZERO, SMLoc(), &STI);
1267 } else {
1268 // sd $gpreg, offset($sp)
1269 emitRRI(Mips::SD, Mips::GP, Mips::SP, RegOrOffset, SMLoc(), &STI);
1270 }
1271
1272 if (getABI().IsN32()) {
1273 MCSymbol *GPSym = MCA.getContext().getOrCreateSymbol("__gnu_local_gp");
1274 const MipsMCExpr *HiExpr = MipsMCExpr::create(
1275 MipsMCExpr::MEK_HI, MCSymbolRefExpr::create(GPSym, MCA.getContext()),
1276 MCA.getContext());
1277 const MipsMCExpr *LoExpr = MipsMCExpr::create(
1278 MipsMCExpr::MEK_LO, MCSymbolRefExpr::create(GPSym, MCA.getContext()),
1279 MCA.getContext());
1280
1281 // lui $gp, %hi(__gnu_local_gp)
1282 emitRX(Mips::LUi, Mips::GP, MCOperand::createExpr(HiExpr), SMLoc(), &STI);
1283
1284 // addiu $gp, $gp, %lo(__gnu_local_gp)
1285 emitRRX(Mips::ADDiu, Mips::GP, Mips::GP, MCOperand::createExpr(LoExpr),
1286 SMLoc(), &STI);
1287
1288 return;
1289 }
1290
1291 const MipsMCExpr *HiExpr = MipsMCExpr::createGpOff(
1292 MipsMCExpr::MEK_HI, MCSymbolRefExpr::create(&Sym, MCA.getContext()),
1293 MCA.getContext());
1294 const MipsMCExpr *LoExpr = MipsMCExpr::createGpOff(
1295 MipsMCExpr::MEK_LO, MCSymbolRefExpr::create(&Sym, MCA.getContext()),
1296 MCA.getContext());
1297
1298 // lui $gp, %hi(%neg(%gp_rel(funcSym)))
1299 emitRX(Mips::LUi, Mips::GP, MCOperand::createExpr(HiExpr), SMLoc(), &STI);
1300
1301 // addiu $gp, $gp, %lo(%neg(%gp_rel(funcSym)))
1302 emitRRX(Mips::ADDiu, Mips::GP, Mips::GP, MCOperand::createExpr(LoExpr),
1303 SMLoc(), &STI);
1304
1305 // daddu $gp, $gp, $funcreg
1306 emitRRR(Mips::DADDu, Mips::GP, Mips::GP, RegNo, SMLoc(), &STI);
1307}

1311 // Only N32 and N64 emit anything for .cpreturn iff PIC is set.
1312 if (!Pic || !(getABI().IsN32() || getABI().IsN64()))
1313 return;
