27MipsELFStreamer::MipsELFStreamer(MCContext &Context,
28 std::unique_ptr<MCAsmBackend> MAB,
29 std::unique_ptr<MCObjectWriter> OW,
30 std::unique_ptr<MCCodeEmitter> Emitter)
31 : MCELFStreamer(Context, std::move(MAB), std::move(OW),
32 std::move(Emitter)) {
33 RegInfoRecord = new MipsRegInfoRecord(this, Context);
34 MipsOptionRecords.push_back(
35 std::unique_ptr<MipsRegInfoRecord>(RegInfoRecord));
36}

38void MipsELFStreamer::EmitInstruction(const MCInst &Inst,
39 const MCSubtargetInfo &STI) {
40 MCELFStreamer::EmitInstruction(Inst, STI);
41
42 MCContext &Context = getContext();
43 const MCRegisterInfo *MCRegInfo = Context.getRegisterInfo();
44
45 for (unsigned OpIndex = 0; OpIndex < Inst.getNumOperands(); ++OpIndex) {
46 const MCOperand &Op = Inst.getOperand(OpIndex);
47
48 if (!Op.isReg())
49 continue;
50
51 unsigned Reg = Op.getReg();
52 RegInfoRecord->SetPhysRegUsed(Reg, MCRegInfo);
53 }
54
55 createPendingLabelRelocs();
56}

74void MipsELFStreamer::createPendingLabelRelocs() {
75 MipsTargetELFStreamer *ELFTargetStreamer =
76 static_cast<MipsTargetELFStreamer *>(getTargetStreamer());
77
78 // FIXME: Also mark labels when in MIPS16 mode.
79 if (ELFTargetStreamer->isMicroMipsEnabled()) {
80 for (auto *L : Labels) {
81 auto *Label = cast<MCSymbolELF>(L);
82 getAssembler().registerSymbol(*Label);
83 Label->setOther(ELF::STO_MIPS_MICROMIPS);
84 }
85 }
86
87 Labels.clear();
88}

107void MipsELFStreamer::EmitCheriCapabilityImpl(const MCSymbol *Symbol,
108 int64_t Offset, unsigned CapSize,
109 SMLoc Loc) {
110 visitUsedSymbol(*Symbol);
111 MCContext &Context = getContext();
112
113 const MCSymbolRefExpr *SRE =
114 MCSymbolRefExpr::create(Symbol, MCSymbolRefExpr::VK_None, Context);
115 const MCBinaryExpr *CapExpr = MCBinaryExpr::createAdd(
116 MipsMCExpr::create(MipsMCExpr::MEK_CHERI_CAP, SRE, Context),
117 MCConstantExpr::create(Offset, Context), Context);
118
119 const unsigned ByteAlignment = CapSize;
120 insert(new MCAlignFragment(ByteAlignment, 0, 1, ByteAlignment));
121 // Update the maximum alignment on the current section if necessary.
122 MCSection *CurSec = getCurrentSectionOnly();
123 if (ByteAlignment > CurSec->getAlignment())
124 CurSec->setAlignment(ByteAlignment);
125
126 MCDataFragment *DF = new MCDataFragment();
127 MCFixup cheriFixup =
128 MCFixup::create(0, CapExpr, MCFixupKind(Mips::fixup_CHERI_CAPABILITY));
129 DF->getFixups().push_back(cheriFixup);
130 DF->getContents().resize(DF->getContents().size() + CapSize, '\xca');
131 insert(DF);
132}

134void MipsELFStreamer::EmitCheriIntcap(int64_t Value, unsigned CapSize, SMLoc) {
135 assert(CapSize == 32 || CapSize == 16);
136 if (Value == 0) {
137 EmitZeros(CapSize);
138 } else {
139 // TODO: we should probably move the CHERI capability encoding somewhere else.
140 // Maybe to BinaryFormat or Object?
141 EmitIntValue(0, 8);
142 EmitIntValue(Value, 8);
143 if (CapSize == 32) {
144 EmitIntValue(0, 8);
145 EmitIntValue(0, 8);
146 }
147 }
148}
