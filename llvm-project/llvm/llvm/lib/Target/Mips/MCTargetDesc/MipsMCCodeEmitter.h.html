90 template<int Width, int Shift>
91 unsigned getShiftedImmediate(const MCInst &MI, unsigned OpNo,
92 SmallVectorImpl<MCFixup> &Fixups,
93 const MCSubtargetInfo &STI) const {
94 int Value = getMachineOpValue(MI, MI.getOperand(OpNo), Fixups, STI);
95 assert((isShiftedInt<Width, Shift>(Value)));
96 return ((unsigned)Value) >> Shift;
97 }
98
99 template<int Width, int Shift>
100 unsigned getShiftedUnsignedImmediate(const MCInst &MI, unsigned OpNo,
101 SmallVectorImpl<MCFixup> &Fixups,
102 const MCSubtargetInfo &STI) const {
103 unsigned Value = getMachineOpValue(MI, MI.getOperand(OpNo), Fixups, STI);
104 assert((isShiftedUInt<Width, Shift>(Value)));
105 return Value >> Shift;
106 }
