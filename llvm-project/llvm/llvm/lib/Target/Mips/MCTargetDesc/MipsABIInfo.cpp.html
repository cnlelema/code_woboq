39ArrayRef<MCPhysReg> MipsABIInfo::GetByValArgRegs() const {
40 if (IsCheriPureCap())
41 return makeArrayRef(CheriCapArgRegs);
42 if (IsO32())
43 return makeArrayRef(O32IntRegs);
44 if (IsN32() || IsN64())
45 return makeArrayRef(Mips64IntRegs);
46 llvm_unreachable("Unhandled ABI");
47}

54 if (IsN32() || IsN64())

59unsigned MipsABIInfo::GetCalleeAllocdArgSizeInBytes(CallingConv::ID CC) const {

67MipsABIInfo MipsABIInfo::computeTargetABI(const Triple &TT, StringRef CPU,
68 const MCTargetOptions &Options) {
69 StringRef ABIName = Options.getABIName();
70 if (ABIName.startswith("o32"))
71 return MipsABIInfo::O32();
72 if (ABIName.startswith("n32"))
73 return MipsABIInfo::N32();
74 if (ABIName.startswith("n64"))
75 return MipsABIInfo::N64();
76 if (ABIName.startswith("sandbox") || ABIName.startswith("purecap"))
77 return MipsABIInfo::CheriPureCap();
78 if (TT.getEnvironment() == llvm::Triple::GNUABIN32)
79 return MipsABIInfo::N32();
80 assert(Options.getABIName().empty() && "Unknown ABI option for MIPS");
81
82 if (TT.isMIPS64())
83 return MipsABIInfo::N64();
84 return MipsABIInfo::O32();
85}

112// HACK: Update the default CFA register for CHERI purecap
113void MipsABIInfo::updateCheriInitialFrameStateHack(const MCAsmInfo &MAI,
114 const MCRegisterInfo &MRI) {
115 if (!IsCheriPureCap()) {
116 assert((MRI.getRARegister() == Mips::RA ||
117 MRI.getRARegister() == Mips::RA_64) &&
118 "Wrong RA register");
119 return;
120 }
121
122 // Some general sanity checks
123 auto ReturnReg = GetReturnAddress();
124 assert(ReturnReg == Mips::C17 && "Wrong RA register");
125 auto StackReg = GetStackPtr();
126 assert(StackReg == Mips::C11 && "Wrong SP register");
127 assert(GetFramePtr() == Mips::C24 && "Wrong FP register");
128
129 // Update the value of the return register
130 if (MRI.getRARegister() != ReturnReg) {
131 const_cast<MCRegisterInfo&>(MRI).RAReg = ReturnReg;
132 }
133 assert(MRI.getRARegister() == ReturnReg && "Wrong RA register");
134 // Check that we didn't set the PC register:
135 assert(MRI.getProgramCounter() == 0 && "Wrong PC register");
136
137 // Update the value of the initial frame state (since C11 is set too late)
138 auto &InitialState = MAI.getInitialFrameState();
139 unsigned C11Dwarf = MRI.getDwarfRegNum(StackReg, true);
140 for (const MCCFIInstruction &Inst : InitialState) {
141 if (Inst.getOperation() == MCCFIInstruction::OpDefCfaRegister) {
142 if (Inst.getRegister() != C11Dwarf) {
143 // errs() << __func__ << ": Updating default OpDefCfaRegister from "
144 // << Inst.getRegister() << " to " << C11Dwarf << "\n";
145 const_cast<MCCFIInstruction &>(Inst).setRegister(C11Dwarf);
146 }
147 } else if (Inst.getOperation() == MCCFIInstruction::OpDefCfa) {
148 if (Inst.getRegister() != C11Dwarf) {
149 // errs() << __func__ << ": Updating default OpDefCfa from "
150 // << Inst.getRegister() << "to" << C11Dwarf << "\n";
151 const_cast<MCCFIInstruction &>(Inst).setRegister(C11Dwarf);
152 }
153 }
154 }

215 Mips::A0_64, Mips::A1_64, Mips::A2_64, Mips::A3_64
