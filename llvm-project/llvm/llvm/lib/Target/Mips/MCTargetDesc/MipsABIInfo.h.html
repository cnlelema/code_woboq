9#ifndef LLVM_LIB_TARGET_MIPS_MCTARGETDESC_MIPSABIINFO_H
10#define LLVM_LIB_TARGET_MIPS_MCTARGETDESC_MIPSABIINFO_H
11
12#include "llvm/ADT/Triple.h"
13#include "llvm/IR/CallingConv.h"
14#include "llvm/MC/MCRegisterInfo.h"
15#include "llvm/MC/MCTargetOptions.h"
16
17namespace llvm {
18
19template <typename T> class ArrayRef;
20class MCTargetOptions;
21class MCAsmInfo;
22class StringRef;
23class TargetRegisterClass;
24
25class MipsABIInfo {
26public:
27 enum class ABI { Unknown, O32, N32, N64 };
28
29protected:
30 ABI ThisABI;
31 bool isCheriPureCap = false;
32
33public:
34 MipsABIInfo(ABI ThisABI, bool isAllCap=false) : ThisABI(ThisABI),
35 isCheriPureCap(isAllCap) {}
36
37 static MipsABIInfo Unknown() { return MipsABIInfo(ABI::Unknown); }
38 static MipsABIInfo O32() { return MipsABIInfo(ABI::O32); }
39 static MipsABIInfo N32() { return MipsABIInfo(ABI::N32); }
40 static MipsABIInfo N64() { return MipsABIInfo(ABI::N64); }
41 static MipsABIInfo CheriPureCap() { return MipsABIInfo(ABI::N64, true); }
42 static MipsABIInfo computeTargetABI(const Triple &TT, StringRef CPU,
43 const MCTargetOptions &Options);
44
45 bool IsKnown() const { return ThisABI != ABI::Unknown; }
46 bool IsO32() const { return ThisABI == ABI::O32; }
47 bool IsN32() const { return ThisABI == ABI::N32; }
48 bool IsN64() const { return ThisABI == ABI::N64; }
49 bool IsCheriPureCap() const { return isCheriPureCap; }
50 bool UsesCapabilityTable() const;
51 CheriCapabilityTableABI CapabilityTableABI() const {
52 assert(IsCheriPureCap());
53 return MCTargetOptions::cheriCapabilityTableABI();
54 }
55 bool UsesCapabilityTls() const;
56 CheriCapabilityTlsABI CapabilityTlsABI() const {
57 assert(IsCheriPureCap());
58 return MCTargetOptions::cheriCapabilityTlsABI();
59 }
60 unsigned StackAddrSpace() const { return isCheriPureCap ? 200 : 0; }
61 ABI GetEnumValue() const { return ThisABI; }
62
63 /// The registers to use for byval arguments.
64 ArrayRef<MCPhysReg> GetByValArgRegs() const;
65
66 /// The registers to use for the variable argument list.
67 ArrayRef<MCPhysReg> GetVarArgRegs() const;
68
69 /// Obtain the size of the area allocated by the callee for arguments.
70 /// CallingConv::FastCall affects the value for O32.
71 unsigned GetCalleeAllocdArgSizeInBytes(CallingConv::ID CC) const;
72
73 /// Ordering of ABI's
74 /// MipsGenSubtargetInfo.inc will use this to resolve conflicts when given
75 /// multiple ABI options.
76 bool operator<(const MipsABIInfo Other) const {
77 return ThisABI < Other.GetEnumValue();
78 }
79
80 unsigned GetDefaultDataCapability() const;
81 unsigned GetReturnAddress() const;
82 unsigned GetStackPtr() const;
83 unsigned GetFramePtr() const;
84 unsigned GetBasePtr() const;
85 unsigned GetGlobalPtr() const;
86 /// This method will eventually be replaced by GetGlobalPtr in
87 /// pure-capability mode, but until all of the new linker work is done we
88 /// need a separate $gp and $cgp as a transition step.
89 unsigned GetGlobalCapability() const;
90 unsigned GetNullPtr() const;
91 unsigned GetZeroReg() const;
92 unsigned GetPtrAdduOp() const;
93 unsigned GetPtrAddiuOp() const;
94 unsigned GetPtrSubuOp() const;
95 unsigned GetPtrAndOp() const;
96 unsigned GetGPRMoveOp() const;
97 unsigned GetSPMoveOp() const;
98 inline bool ArePtrs64bit() const { return IsN64(); }
99 inline bool AreGprs64bit() const { return IsN32() || IsN64(); }
100
101 unsigned GetEhDataReg(unsigned I) const;
102
103 // Update the initial CFA register from SP to C11 if needed
104 void updateCheriInitialFrameStateHack(const MCAsmInfo &MAI,
105 const MCRegisterInfo &MRI);
106};
107}
108
109#endif
