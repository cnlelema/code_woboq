42void MipsMCExpr::printImpl(raw_ostream &OS, const MCAsmInfo *MAI) const {
43 int64_t AbsVal;
44
45 switch (Kind) {
46 case MEK_None:
47 case MEK_Special:
48 llvm_unreachable("MEK_None and MEK_Special are invalid");
49 break;
50 case MEK_DTPREL:
51 // MEK_DTPREL is used for marking TLS DIEExpr only
52 // and contains a regular sub-expression.
53 getSubExpr()->print(OS, MAI, true);
54 return;
55 case MEK_CALL_HI16:
56 OS << "%call_hi";
57 break;
58 case MEK_CALL_LO16:
59 OS << "%call_lo";
60 break;
61 case MEK_DTPREL_HI:
62 OS << "%dtprel_hi";
63 break;
64 case MEK_DTPREL_LO:
65 OS << "%dtprel_lo";
66 break;
67 case MEK_GOT:
68 OS << "%got";
69 break;
70 case MEK_GOTTPREL:
71 OS << "%gottprel";
72 break;
73 case MEK_GOT_CALL:
74 OS << "%call16";
75 break;
76 case MEK_GOT_DISP:
77 OS << "%got_disp";
78 break;
79 case MEK_GOT_HI16:
80 OS << "%got_hi";
81 break;
82 case MEK_GOT_LO16:
83 OS << "%got_lo";
84 break;
85 case MEK_GOT_PAGE:
86 OS << "%got_page";
87 break;
88 case MEK_GOT_OFST:
89 OS << "%got_ofst";
90 break;
91 case MEK_GPREL:
92 OS << "%gp_rel";
93 break;
94 case MEK_CAPTABLEREL:
95 OS << "%captab_rel";
96 break;
97 case MEK_HI:
98 OS << "%hi";
99 break;
100 case MEK_HIGHER:
101 OS << "%higher";
102 break;
103 case MEK_HIGHEST:
104 OS << "%highest";
105 break;
106 case MEK_LO:
107 OS << "%lo";
108 break;
109 case MEK_NEG:
110 OS << "%neg";
111 break;
112 case MEK_PCREL_HI16:
113 OS << "%pcrel_hi";
114 break;
115 case MEK_PCREL_LO16:
116 OS << "%pcrel_lo";
117 break;
118 case MEK_TLSGD:
119 OS << "%tlsgd";
120 break;
121 case MEK_TLSLDM:
122 OS << "%tlsldm";
123 break;
124 case MEK_TPREL_HI:
125 OS << "%tprel_hi";
126 break;
127 case MEK_TPREL_LO:
128 OS << "%tprel_lo";
129 break;
130 case MEK_CAPCALL11:
131 OS << "%capcall";
132 break;
133 case MEK_CAPCALL20:
134 OS << "%capcall20";
135 break;
136 case MEK_CAPCALL_HI16:
137 OS << "%capcall_hi";
138 break;
139 case MEK_CAPCALL_LO16:
140 OS << "%capcall_lo";
141 break;
142 case MEK_CAPTABLE11:
143 OS << "%captab";
144 break;
145 case MEK_CAPTABLE20:
146 OS << "%captab20";
147 break;
148 case MEK_CAPTABLE_HI16:
149 OS << "%captab_hi";
150 break;
151 case MEK_CAPTABLE_LO16:
152 OS << "%captab_lo";
153 break;
154 case MEK_CHERI_CAP:
155 // FIXME: should we really end up here?
156 OS << "%chericap";
157 break;
158 case MEK_CAPTAB_TLSGD_HI16:
159 OS << "%captab_tlsgd_hi";
160 break;
161 case MEK_CAPTAB_TLSGD_LO16:
162 OS << "%captab_tlsgd_lo";
163 break;
164 case MEK_CAPTAB_TLSLDM_HI16:
165 OS << "%captab_tlsldm_hi";
166 break;
167 case MEK_CAPTAB_TLSLDM_LO16:
168 OS << "%captab_tlsldm_lo";
169 break;
170 case MEK_CAPTAB_TPREL_HI16:
171 OS << "%captab_tprel_hi";
172 break;
173 case MEK_CAPTAB_TPREL_LO16:
174 OS << "%captab_tprel_lo";
175 break;
176 }
177
178 OS << '(';
179 if (Expr->evaluateAsAbsolute(AbsVal))
180 OS << AbsVal;
181 else
182 Expr->print(OS, MAI, true);
183 OS << ')';
184}

186bool
187MipsMCExpr::evaluateAsRelocatableImpl(MCValue &Res,
188 const MCAsmLayout *Layout,
189 const MCFixup *Fixup) const {
190 // Look for the %hi(%neg(%gp_rel(X))) and %lo(%neg(%gp_rel(X))) special cases.
191 if (isGpOff() || isCaptableOff()) {
192 const MCExpr *SubExpr =
193 cast<MipsMCExpr>(cast<MipsMCExpr>(getSubExpr())->getSubExpr())
194 ->getSubExpr();
195 if (!SubExpr->evaluateAsRelocatable(Res, Layout, Fixup))
196 return false;
197
198 Res = MCValue::get(Res.getSymA(), Res.getSymB(), Res.getConstant(),
199 MEK_Special);
200 return true;
201 }
202
203 if (!getSubExpr()->evaluateAsRelocatable(Res, Layout, Fixup))
204 return false;
205
206 if (Res.getRefKind() != MCSymbolRefExpr::VK_None)
207 return false;
208
209 // evaluateAsAbsolute() and evaluateAsValue() require that we evaluate the
210 // %hi/%lo/etc. here. Fixup is a null pointer when either of these is the
211 // caller.
212 if (Res.isAbsolute() && Fixup == nullptr) {
213 int64_t AbsVal = Res.getConstant();
214 switch (Kind) {
215 case MEK_None:
216 case MEK_Special:
217 llvm_unreachable("MEK_None and MEK_Special are invalid");
218 case MEK_CHERI_CAP:
219 llvm_unreachable("MEK_CHERI_CAP is invalid");
220 case MEK_DTPREL:
221 // MEK_DTPREL is used for marking TLS DIEExpr only
222 // and contains a regular sub-expression.
223 return getSubExpr()->evaluateAsRelocatable(Res, Layout, Fixup);
224 case MEK_DTPREL_HI:
225 case MEK_DTPREL_LO:
226 case MEK_GOT:
227 case MEK_GOTTPREL:
228 case MEK_GOT_CALL:
229 case MEK_GOT_DISP:
230 case MEK_GOT_HI16:
231 case MEK_GOT_LO16:
232 case MEK_GOT_OFST:
233 case MEK_GOT_PAGE:
234 case MEK_GPREL:
235 case MEK_CAPTABLEREL:
236 case MEK_PCREL_HI16:
237 case MEK_PCREL_LO16:
238 case MEK_TLSGD:
239 case MEK_TLSLDM:
240 case MEK_TPREL_HI:
241 case MEK_TPREL_LO:
242 case MEK_CAPCALL11:
243 case MEK_CAPCALL20:
244 case MEK_CAPCALL_LO16:
245 case MEK_CAPCALL_HI16:
246 case MEK_CAPTABLE11:
247 case MEK_CAPTABLE20:
248 case MEK_CAPTABLE_HI16:
249 case MEK_CAPTABLE_LO16:
250 case MEK_CAPTAB_TLSGD_HI16:
251 case MEK_CAPTAB_TLSGD_LO16:
252 case MEK_CAPTAB_TLSLDM_HI16:
253 case MEK_CAPTAB_TLSLDM_LO16:
254 case MEK_CAPTAB_TPREL_HI16:
255 case MEK_CAPTAB_TPREL_LO16:
256 return false;
257 case MEK_LO:
258 case MEK_CALL_LO16:
259 AbsVal = SignExtend64<16>(AbsVal);
260 break;
261 case MEK_CALL_HI16:
262 case MEK_HI:
263 AbsVal = SignExtend64<16>((AbsVal + 0x8000) >> 16);
264 break;
265 case MEK_HIGHER:
266 AbsVal = SignExtend64<16>((AbsVal + 0x80008000LL) >> 32);
267 break;
268 case MEK_HIGHEST:
269 AbsVal = SignExtend64<16>((AbsVal + 0x800080008000LL) >> 48);
270 break;
271 case MEK_NEG:
272 AbsVal = -AbsVal;
273 break;
274 }
275 Res = MCValue::get(AbsVal);
276 return true;
277 }
278
279 // We want to defer it for relocatable expressions since the constant is
280 // applied to the whole symbol value.
281 //
282 // The value of getKind() that is given to MCValue is only intended to aid
283 // debugging when inspecting MCValue objects. It shouldn't be relied upon
284 // for decision making.
285 Res = MCValue::get(Res.getSymA(), Res.getSymB(), Res.getConstant(), getKind());
286
287 return true;
288}

294static void fixELFSymbolsInTLSFixupsImpl(const MCExpr *Expr, MCAssembler &Asm) {
295 switch (Expr->getKind()) {
296 case MCExpr::Target:
297 fixELFSymbolsInTLSFixupsImpl(cast<MipsMCExpr>(Expr)->getSubExpr(), Asm);
298 break;
299 case MCExpr::Constant:
300 break;
301 case MCExpr::Binary: {
302 const MCBinaryExpr *BE = cast<MCBinaryExpr>(Expr);
303 fixELFSymbolsInTLSFixupsImpl(BE->getLHS(), Asm);
304 fixELFSymbolsInTLSFixupsImpl(BE->getRHS(), Asm);
305 break;
306 }
307 case MCExpr::SymbolRef: {
308 // We're known to be under a TLS fixup, so any symbol should be
309 // modified. There should be only one.
310 const MCSymbolRefExpr &SymRef = *cast<MCSymbolRefExpr>(Expr);
311 cast<MCSymbolELF>(SymRef.getSymbol()).setType(ELF::STT_TLS);
312 break;
313 }
314 case MCExpr::Unary:
315 fixELFSymbolsInTLSFixupsImpl(cast<MCUnaryExpr>(Expr)->getSubExpr(), Asm);
316 break;
317 }
318}

320void MipsMCExpr::fixELFSymbolsInTLSFixups(MCAssembler &Asm) const {
321 switch (getKind()) {
322 case MEK_None:
323 case MEK_Special:
324 llvm_unreachable("MEK_None and MEK_Special are invalid");
325 break;
326 case MEK_CHERI_CAP:
327 llvm_unreachable("MEK_CHERI_CAP is not handled here");
328 break;
329 case MEK_CALL_HI16:
330 case MEK_CALL_LO16:
331 case MEK_GOT:
332 case MEK_GOT_CALL:
333 case MEK_GOT_DISP:
334 case MEK_GOT_HI16:
335 case MEK_GOT_LO16:
336 case MEK_GOT_OFST:
337 case MEK_GOT_PAGE:
338 case MEK_GPREL:
339 case MEK_CAPTABLEREL:
340 case MEK_HI:
341 case MEK_HIGHER:
342 case MEK_HIGHEST:
343 case MEK_LO:
344 case MEK_NEG:
345 case MEK_PCREL_HI16:
346 case MEK_PCREL_LO16:
347 case MEK_CAPCALL11:
348 case MEK_CAPCALL20:
349 case MEK_CAPCALL_LO16:
350 case MEK_CAPCALL_HI16:
351 case MEK_CAPTABLE11:
352 case MEK_CAPTABLE20:
353 case MEK_CAPTABLE_HI16:
354 case MEK_CAPTABLE_LO16:
355 // If we do have nested target-specific expressions, they will be in
356 // a consecutive chain.
357 if (const MipsMCExpr *E = dyn_cast<const MipsMCExpr>(getSubExpr()))
358 E->fixELFSymbolsInTLSFixups(Asm);
359 break;
360 case MEK_DTPREL:
361 case MEK_DTPREL_HI:
362 case MEK_DTPREL_LO:
363 case MEK_TLSLDM:
364 case MEK_TLSGD:
365 case MEK_GOTTPREL:
366 case MEK_TPREL_HI:
367 case MEK_TPREL_LO:
368 case MEK_CAPTAB_TLSGD_HI16:
369 case MEK_CAPTAB_TLSGD_LO16:
370 case MEK_CAPTAB_TLSLDM_HI16:
371 case MEK_CAPTAB_TLSLDM_LO16:
372 case MEK_CAPTAB_TPREL_HI16:
373 case MEK_CAPTAB_TPREL_LO16:
374 fixELFSymbolsInTLSFixupsImpl(getSubExpr(), Asm);
375 break;
376 }
377}

379bool MipsMCExpr::isOffImpl(MipsExprKind &Kind, MipsExprKind Expected) const {
380 if (getKind() == MEK_HI || getKind() == MEK_LO) {
381 if (const MipsMCExpr *S1 = dyn_cast<const MipsMCExpr>(getSubExpr())) {
382 if (const MipsMCExpr *S2 = dyn_cast<const MipsMCExpr>(S1->getSubExpr())) {
383 if (S1->getKind() == MEK_NEG && S2->getKind() == Expected) {
384 Kind = getKind();
385 return true;
386 }
387 }
388 }
389 }
390 return false;
391}
