47/// Select the Mips CPU for the given triple and cpu name.
48/// FIXME: Merge with the copy in MipsSubtarget.cpp
49StringRef MIPS_MC::selectMipsCPU(const Triple &TT, StringRef CPU) {
50 if (CPU.empty() || CPU == "generic") {
51 if (TT.getSubArch() == llvm::Triple::MipsSubArch_r6) {
52 if (TT.isMIPS32())
53 CPU = "mips32r6";
54 else
55 CPU = "mips64r6";
56 } else if (TT.getArch() == Triple::cheri) {
57 // FIXME: do we really want this?
58 CPU = "cheri128";
59 } else {
60 if (TT.isMIPS32())
61 CPU = "mips32";
62 else
63 CPU = "mips64";
64 }
65 }
66 return CPU;
67}

75static MCRegisterInfo *createMipsMCRegisterInfo(const Triple &TT) {
76 MCRegisterInfo *X = new MCRegisterInfo();

90static MCAsmInfo *createMipsMCAsmInfo(const MCRegisterInfo &MRI,
91 const Triple &TT) {
92 MCAsmInfo *MAI = new MipsMCAsmInfo(TT);
93 // FIXME: would be nice if we had a MCTargetOptions instance here instead of
94 // just the triple.
95 unsigned SP = MRI.getDwarfRegNum(
96 TT.getEnvironment() == Triple::CheriPurecap ? Mips::C11 : Mips::SP, true);
97 MCCFIInstruction Inst = MCCFIInstruction::createDefCfaRegister(nullptr, SP);
98 MAI->addInitialFrameState(Inst);
99
100 return MAI;
101}

111static MCStreamer *createMCStreamer(const Triple &T, MCContext &Context,
112 std::unique_ptr<MCAsmBackend> &&MAB,
113 std::unique_ptr<MCObjectWriter> &&OW,
114 std::unique_ptr<MCCodeEmitter> &&Emitter,
115 bool RelaxAll) {
116 MCStreamer *S;
117 if (!T.isOSNaCl())
118 S = createMipsELFStreamer(Context, std::move(MAB), std::move(OW),
119 std::move(Emitter), RelaxAll);
120 else
121 S = createMipsNaClELFStreamer(Context, std::move(MAB), std::move(OW),
122 std::move(Emitter), RelaxAll);
123 return S;
124}

145class MipsMCInstrAnalysis : public MCInstrAnalysis {
146public:
147 MipsMCInstrAnalysis(const MCInstrInfo *Info) : MCInstrAnalysis(Info) {}
148
149 bool evaluateBranch(const MCInst &Inst, uint64_t Addr, uint64_t Size,
150 uint64_t &Target) const override {
151 unsigned NumOps = Inst.getNumOperands();
152 if (NumOps == 0)
153 return false;
154 switch (Info->get(Inst.getOpcode()).OpInfo[NumOps - 1].OperandType) {
155 case MCOI::OPERAND_UNKNOWN:
156 case MCOI::OPERAND_IMMEDIATE:
157 // jal, bal ...
158 Target = Inst.getOperand(NumOps - 1).getImm();
159 return true;
160 case MCOI::OPERAND_PCREL:
161 // b, j, beq ...
162 Target = Addr + Inst.getOperand(NumOps - 1).getImm();
163 return true;
164 default:
165 return false;
166 }
167 }
168
169 virtual bool isCapTableLoad(const MCInst &Inst,
170 int64_t &Offset) const override {
171 // cap table loads have a zero register offset and are relative to $c26
172 if (Inst.getOpcode() == Mips::LOADCAP &&
173 (Inst.getOperand(1).getReg() == Mips::ZERO_64 ||
174 Inst.getOperand(1).getReg() == Mips::ZERO) &&
175 Inst.getOperand(3).getReg() == Mips::C26) {
176 // For classic clc it register has to be $zero (mxcaptable not handled yet)
177 // TODO: can I somehow handle the mxcaptable case by inferring the value of $at?
178 Offset = Inst.getOperand(2).getImm();
179 return true;
180 } else if (Inst.getOpcode() == Mips::LOADCAP_BigImm &&
181 Inst.getOperand(2).getReg() == Mips::C26) {
182 Offset = Inst.getOperand(1).getImm();
183 return true;
184 }
185 return false;
186 }
187};
188}

194extern "C" void LLVMInitializeMipsTargetMC() {
195 for (Target *T : {&getTheMipsTarget(), &getTheMipselTarget(),
196 &getTheMips64Target(), &getTheMips64elTarget(),
197 &getTheMipsCheriTarget()}) {
198 // Register the MC asm info.
199 RegisterMCAsmInfoFn X(*T, createMipsMCAsmInfo);
200
201 // Register the MC instruction info.
202 TargetRegistry::RegisterMCInstrInfo(*T, createMipsMCInstrInfo);
203
204 // Register the MC register info.
205 TargetRegistry::RegisterMCRegInfo(*T, createMipsMCRegisterInfo);
206
207 // Register the elf streamer.
208 TargetRegistry::RegisterELFStreamer(*T, createMCStreamer);
209
210 // Register the asm target streamer.
211 TargetRegistry::RegisterAsmTargetStreamer(*T, createMipsAsmTargetStreamer);
212
213 TargetRegistry::RegisterNullTargetStreamer(*T,
214 createMipsNullTargetStreamer);
215
216 // Register the MC subtarget info.
217 TargetRegistry::RegisterMCSubtargetInfo(*T, createMipsMCSubtargetInfo);
218
219 // Register the MC instruction analyzer.
220 TargetRegistry::RegisterMCInstrAnalysis(*T, createMipsMCInstrAnalysis);
221
222 // Register the MCInstPrinter.
223 TargetRegistry::RegisterMCInstPrinter(*T, createMipsMCInstPrinter);
224
225 TargetRegistry::RegisterObjectTargetStreamer(
226 *T, createMipsObjectTargetStreamer);
227
228 // Register the asm backend.
229 TargetRegistry::RegisterMCAsmBackend(*T, createMipsAsmBackend);
230 }
231
232 // Register the MC Code Emitter
233 for (Target *T : {&getTheMipsTarget(), &getTheMips64Target(), &getTheMipsCheriTarget()})
234 TargetRegistry::RegisterMCCodeEmitter(*T, createMipsMCCodeEmitterEB);
235
236 for (Target *T : {&getTheMipselTarget(), &getTheMips64elTarget()})
237 TargetRegistry::RegisterMCCodeEmitter(*T, createMipsMCCodeEmitterEL);
238}
