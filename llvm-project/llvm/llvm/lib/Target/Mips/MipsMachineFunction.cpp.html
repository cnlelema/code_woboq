31static const TargetRegisterClass &getGlobalBaseRegClass(MachineFunction &MF) {
32 auto &STI = static_cast<const MipsSubtarget &>(MF.getSubtarget());
33 auto &TM = static_cast<const MipsTargetMachine &>(MF.getTarget());
34
35 if (STI.inMips16Mode())
36 return Mips::CPU16RegsRegClass;
37
38 if (STI.inMicroMipsMode())
39 return Mips::GPRMM16RegClass;
40
41 if (TM.getABI().IsN64())
42 return Mips::GPR64RegClass;
43
44 return Mips::GPR32RegClass;
45}

52unsigned MipsFunctionInfo::getGlobalBaseReg(bool IsForTls) {
53 if (IsForTls)
54 UsesTlsViaGlobalReg = true;
55 else
56 assert(!static_cast<const MipsSubtarget &>(MF.getSubtarget()).useCheriCapTable() &&
57 "$gp should only be used for TLS in cap-table mode");
58
59 // Return if it has already been initialized.
60 if (!GlobalBaseReg)
61 GlobalBaseReg =
62 MF.getRegInfo().createVirtualRegister(&getGlobalBaseRegClass(MF));
63 return GlobalBaseReg;
64}

82 getCapGlobalBaseReg();

88unsigned MipsFunctionInfo::getGlobalBaseRegForGlobalISel() {
89 if (static_cast<const MipsSubtarget&>(MF.getSubtarget()).useCheriCapTable()) {
90 if (!CapGlobalBaseReg) {
91 getCapGlobalBaseReg();
92 initCapGlobalBaseReg();
93 }
94 return CapGlobalBaseReg;
95 }
96 if (!GlobalBaseReg) {
97 getGlobalBaseReg(false);
98 initGlobalBaseReg();
99 }
100 return GlobalBaseReg;
101}

103void MipsFunctionInfo::initGlobalBaseReg() {
104 if (!GlobalBaseReg)
105 return;
106
107 MachineBasicBlock &MBB = MF.front();
108 MachineBasicBlock::iterator I = MBB.begin();
109 MachineRegisterInfo &RegInfo = MF.getRegInfo();
110 const TargetInstrInfo &TII = *MF.getSubtarget().getInstrInfo();
111 DebugLoc DL;
112 unsigned V0, V1;
113 const TargetRegisterClass *RC;
114 const MipsABIInfo &ABI =
115 static_cast<const MipsTargetMachine &>(MF.getTarget()).getABI();
116 RC = (ABI.IsN64()) ? &Mips::GPR64RegClass : &Mips::GPR32RegClass;
117
118 V0 = RegInfo.createVirtualRegister(RC);
119 V1 = RegInfo.createVirtualRegister(RC);
120
121 if (ABI.IsN64()) {
122 if (ABI.IsCheriPureCap()) {
123 if (ABI.UsesCapabilityTable()) {
124 assert(usesTlsViaGlobalReg() && "$gp should only be used for TLS");
125 assert((CapGlobalBaseReg || !MF.getRegInfo().isLiveIn(Mips::C12)) &&
126 "C12 should not be used yet");
127 }
128 if (!MF.getRegInfo().isLiveIn(Mips::C12))
129 MF.getRegInfo().addLiveIn(Mips::C12);
130 MBB.addLiveIn(Mips::C12);
131 BuildMI(MBB, I, DL, TII.get(Mips::CGetOffset))
132 .addReg(Mips::T9_64, RegState::Define)
133 .addReg(Mips::C12);
134 } else {
135 MF.getRegInfo().addLiveIn(Mips::T9_64);
136 MBB.addLiveIn(Mips::T9_64);
137 }
138
139 // lui $v0, %hi(%neg(%gp_rel(fname)))
140 // daddu $v1, $v0, $t9
141 // daddiu $globalbasereg, $v1, %lo(%neg(%gp_rel(fname)))
142 const GlobalValue *FName = &MF.getFunction();
143 BuildMI(MBB, I, DL, TII.get(Mips::LUi64), V0)
144 .addGlobalAddress(FName, 0, MipsII::MO_GPOFF_HI);
145 BuildMI(MBB, I, DL, TII.get(Mips::DADDu), V1).addReg(V0)
146 .addReg(Mips::T9_64);
147 BuildMI(MBB, I, DL, TII.get(Mips::DADDiu), GlobalBaseReg).addReg(V1)
148 .addGlobalAddress(FName, 0, MipsII::MO_GPOFF_LO);
149 return;
150 }
151
152 if (!MF.getTarget().isPositionIndependent()) {
153 // Set global register to __gnu_local_gp.
154 //
155 // lui $v0, %hi(__gnu_local_gp)
156 // addiu $globalbasereg, $v0, %lo(__gnu_local_gp)
157 BuildMI(MBB, I, DL, TII.get(Mips::LUi), V0)
158 .addExternalSymbol("__gnu_local_gp", MipsII::MO_ABS_HI);
159 BuildMI(MBB, I, DL, TII.get(Mips::ADDiu), GlobalBaseReg).addReg(V0)
160 .addExternalSymbol("__gnu_local_gp", MipsII::MO_ABS_LO);
161 return;
162 }
163
164 MF.getRegInfo().addLiveIn(Mips::T9);
165 MBB.addLiveIn(Mips::T9);
166
167 if (ABI.IsN32()) {
168 // lui $v0, %hi(%neg(%gp_rel(fname)))
169 // addu $v1, $v0, $t9
170 // addiu $globalbasereg, $v1, %lo(%neg(%gp_rel(fname)))
171 const GlobalValue *FName = &MF.getFunction();
172 BuildMI(MBB, I, DL, TII.get(Mips::LUi), V0)
173 .addGlobalAddress(FName, 0, MipsII::MO_GPOFF_HI);
174 BuildMI(MBB, I, DL, TII.get(Mips::ADDu), V1).addReg(V0).addReg(Mips::T9);
175 BuildMI(MBB, I, DL, TII.get(Mips::ADDiu), GlobalBaseReg).addReg(V1)
176 .addGlobalAddress(FName, 0, MipsII::MO_GPOFF_LO);
177 return;
178 }
179
180 assert(ABI.IsO32());
181
182 // For O32 ABI, the following instruction sequence is emitted to initialize
183 // the global base register:
184 //
185 // 0. lui $2, %hi(_gp_disp)
186 // 1. addiu $2, $2, %lo(_gp_disp)
187 // 2. addu $globalbasereg, $2, $t9
188 //
189 // We emit only the last instruction here.
190 //
191 // GNU linker requires that the first two instructions appear at the beginning
192 // of a function and no instructions be inserted before or between them.
193 // The two instructions are emitted during lowering to MC layer in order to
194 // avoid any reordering.
195 //
196 // Register $2 (Mips::V0) is added to the list of live-in registers to ensure
197 // the value instruction 1 (addiu) defines is valid when instruction 2 (addu)
198 // reads it.
199 MF.getRegInfo().addLiveIn(Mips::V0);
200 MBB.addLiveIn(Mips::V0);
201 BuildMI(MBB, I, DL, TII.get(Mips::ADDu), GlobalBaseReg)
202 .addReg(Mips::V0).addReg(Mips::T9);
203}

205void MipsFunctionInfo::initCapGlobalBaseReg() {
206 if (!CapGlobalBaseReg)
207 return;
208
209 auto &TM = static_cast<const MipsTargetMachine &>(MF.getTarget());
210 assert(TM.getABI().UsesCapabilityTable());
211 // assert(MF.getTarget().isPositionIndependent() && "CHERI CODEGEN REQUIRES -fPIC");
212
213 MachineBasicBlock &MBB = MF.front();
214 MachineBasicBlock::iterator I = MBB.begin();
215 const TargetInstrInfo &TII = *MF.getSubtarget().getInstrInfo();
216 DebugLoc DL;
217 const MipsABIInfo &ABI = TM.getABI();
218
219 assert(ABI.IsCheriPureCap());
220 const unsigned ABIGlobalCapReg = ABI.GetGlobalCapability();
221
222 // For the purecap ABI, $cgp is required to point to the function's/DSOs
223 // capability table on function entry, so emit a single COPY
224 // (which may be optimised away):
225 // COPY $capglobalbasereg, $c26
226 if (MCTargetOptions::cheriCapabilityTableABI() ==
227 CheriCapabilityTableABI::Pcrel) {
228 // However, we also support an experimental mode where we derive the $cgp
229 // register from $pcc (this is closer to what MIPS does and has the
230 // advantage that we don't need to reserve $cgp since we can always derive
231 // it from $pcc. However, in this mode we can't restrict the bounds or
232 // permissions on $pcc very much since we need it to always contain the
233 // whole cap table as well as Load_Capability permission. If we decide to
234 // also inline global statics into the cap-table it also needs to contain
235 // write permissions which means we're back to relying only on the MMU to
236 // protect the .text segment.
237
238 // XXXAR: Mips::C12 may already be marked as live-in if we use TLS
239 if (MF.getRegInfo().isLiveIn(Mips::C12)) {
240 assert(usesTlsViaGlobalReg() && "$gp should only be used for TLS");
241 assert(MBB.isLiveIn(Mips::C12)); // should have been added by the TLS hack
242 } else {
243 MF.getRegInfo().addLiveIn(Mips::C12);
244 MBB.addLiveIn(Mips::C12);
245 }
246
247 // FIXME: there doesn't seem to be an easy way to get a label difference
248 // For now I'll just add a new relocation or see if I can convert itm
249 MachineRegisterInfo &RegInfo = MF.getRegInfo();
250 const GlobalValue *FName = &MF.getFunction();
251 unsigned Tmp1 = RegInfo.createVirtualRegister(&Mips::GPR64RegClass);
252 unsigned Tmp2 = RegInfo.createVirtualRegister(&Mips::GPR64RegClass);
253 BuildMI(MBB, I, DL, TII.get(Mips::LUi64), Tmp1)
254 .addGlobalAddress(FName, 0, MipsII::MO_CAPTABLE_OFF_HI);
255 BuildMI(MBB, I, DL, TII.get(Mips::DADDiu), Tmp2)
256 .addReg(Tmp1)
257 .addGlobalAddress(FName, 0, MipsII::MO_CAPTABLE_OFF_LO);
258#if 0
259 BuildMI(MBB, I, DL, TII.get(Mips::CIncOffset), CapGlobalBaseReg)
260 .addReg(Mips::C12)
261 .addReg(Tmp2);
262 // XXXAR: the hints are being ignored (probably since $cgp is reserved?)
263 RegInfo.addRegAllocationHint(CapGlobalBaseReg, ABIGlobalCapReg);
264 RegInfo.setSimpleHint(CapGlobalBaseReg, ABIGlobalCapReg);
265#else
266 // Generate the globals in $cgp to make the llvm-objdump output more useful
267 // since it will guess which global is being loaded. Also I think that if
268 // we explicitly use $cgp here the register allocator will be able to use
269 // one additional register (since it can optimize away the copy node and
270 // just use $cgp directly for loads)
271 if (MF.getTarget().getOptLevel() == CodeGenOpt::None) {
272 // At -O0 ensure that the global register ends up in $c26
273 BuildMI(MBB, I, DL, TII.get(Mips::CIncOffset), ABIGlobalCapReg)
274 .addReg(Mips::C12)
275 .addReg(Tmp2);
276 BuildMI(MBB, I, DL, TII.get(TargetOpcode::COPY), CapGlobalBaseReg)
277 .addReg(ABIGlobalCapReg);
278 } else {
279 // Otherwise we will use the first available callee-save register for $cgp
280 BuildMI(MBB, I, DL, TII.get(Mips::CIncOffset), CapGlobalBaseReg)
281 .addReg(Mips::C12)
282 .addReg(Tmp2);
283 }
284#endif
285 } else {
286 MF.getRegInfo().addLiveIn(ABIGlobalCapReg);
287 MBB.addLiveIn(ABIGlobalCapReg);
288 BuildMI(MBB, I, DL, TII.get(TargetOpcode::COPY), CapGlobalBaseReg)
289 .addReg(ABIGlobalCapReg);
290 }
291}

294void MipsFunctionInfo::createEhDataRegsFI() {
295 const TargetRegisterInfo &TRI = *MF.getSubtarget().getRegisterInfo();
296 for (int I = 0; I < 4; ++I) {
297 const TargetRegisterClass &RC =
298 static_cast<const MipsTargetMachine &>(MF.getTarget()).getABI().IsN64()
299 ? Mips::GPR64RegClass
300 : Mips::GPR32RegClass;
301
302 EhDataRegFI[I] = MF.getFrameInfo().CreateStackObject(TRI.getSpillSize(RC),
303 TRI.getSpillAlignment(RC), false);
304 }
305}

307void MipsFunctionInfo::createISRRegFI() {
308 // ISRs require spill slots for Status & ErrorPC Coprocessor 0 registers.
309 // The current implementation only supports Mips32r2+ not Mips64rX. Status
310 // is always 32 bits, ErrorPC is 32 or 64 bits dependent on architecture,
311 // however Mips32r2+ is the supported architecture.
312 const TargetRegisterClass &RC = Mips::GPR32RegClass;
313 const TargetRegisterInfo &TRI = *MF.getSubtarget().getRegisterInfo();
314
315 for (int I = 0; I < 2; ++I)
316 ISRDataRegFI[I] = MF.getFrameInfo().CreateStackObject(
317 TRI.getSpillSize(RC), TRI.getSpillAlignment(RC), false);
318}

320bool MipsFunctionInfo::isEhDataRegFI(int FI) const {
321 return CallsEhReturn && (FI == EhDataRegFI[0] || FI == EhDataRegFI[1]
322 || FI == EhDataRegFI[2] || FI == EhDataRegFI[3]);
323}

325bool MipsFunctionInfo::isISRRegFI(int FI) const {
326 return IsISR && (FI == ISRDataRegFI[0] || FI == ISRDataRegFI[1]);
327}
