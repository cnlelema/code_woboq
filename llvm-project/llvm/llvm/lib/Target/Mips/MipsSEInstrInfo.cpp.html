28static unsigned getUnconditionalBranch(const MipsSubtarget &STI) {
29 if (STI.inMicroMipsMode())
30 return STI.isPositionIndependent() ? Mips::B_MM : Mips::J_MM;
31 return STI.isPositionIndependent() ? Mips::B : Mips::J;

41/// isLoadFromStackSlot - If the specified machine instruction is a direct
42/// load from a stack slot, return the virtual or physical register number of
43/// the destination along with the FrameIndex of the loaded stack slot. If
44/// not, return 0. This predicate must return 0 if the instruction has
45/// any side effects other than loading from the stack slot.
46unsigned MipsSEInstrInfo::isLoadFromStackSlot(const MachineInstr &MI,
47 int &FrameIndex) const {
48 if (MI.mayLoad()) { // Is a load...
49 if ((MI.getOperand(1).isFI()) && // is a stack slot
50 (MI.getOperand(2).isImm()) && // the imm is zero
51 (isZeroImm(MI.getOperand(2)))) {
52 FrameIndex = MI.getOperand(1).getIndex();
53 return MI.getOperand(0).getReg();
54 }
55 }
56
57 return 0;
58}
59
60/// isStoreToStackSlot - If the specified machine instruction is a direct
61/// store to a stack slot, return the virtual or physical register number of
62/// the source reg along with the FrameIndex of the loaded stack slot. If
63/// not, return 0. This predicate must return 0 if the instruction has
64/// any side effects other than storing to the stack slot.
65unsigned MipsSEInstrInfo::isStoreToStackSlot(const MachineInstr &MI,
66 int &FrameIndex) const {
67 if (MI.mayStore()) { // is a store...
68 if ((MI.getOperand(1).isFI()) && // is a stack slot
69 (MI.getOperand(2).isImm()) && // the imm is zero
70 (isZeroImm(MI.getOperand(2)))) {
71 FrameIndex = MI.getOperand(1).getIndex();
72 return MI.getOperand(0).getReg();
73 }
74 }
75
76 return 0;
77}

79void MipsSEInstrInfo::copyPhysReg(MachineBasicBlock &MBB,
80 MachineBasicBlock::iterator I,
81 const DebugLoc &DL, unsigned DestReg,
82 unsigned SrcReg, bool KillSrc) const {
83 unsigned Opc = 0, ZeroReg = 0;
84 bool isMicroMips = Subtarget.inMicroMipsMode();
85
86 if (Mips::GPR32RegClass.contains(DestReg)) { // Copy to CPU Reg.
87 if (Mips::GPR32RegClass.contains(SrcReg)) {
88 if (isMicroMips)
89 Opc = Mips::MOVE16_MM;
90 else
91 Opc = Mips::OR, ZeroReg = Mips::ZERO;
92 } else if (Mips::CCRRegClass.contains(SrcReg))
93 Opc = Mips::CFC1;
94 else if (Mips::FGR32RegClass.contains(SrcReg))
95 Opc = Mips::MFC1;
96 else if (Mips::HI32RegClass.contains(SrcReg)) {
97 Opc = isMicroMips ? Mips::MFHI16_MM : Mips::MFHI;
98 SrcReg = 0;
99 } else if (Mips::LO32RegClass.contains(SrcReg)) {
100 Opc = isMicroMips ? Mips::MFLO16_MM : Mips::MFLO;
101 SrcReg = 0;
102 } else if (Mips::HI32DSPRegClass.contains(SrcReg))
103 Opc = Mips::MFHI_DSP;
104 else if (Mips::LO32DSPRegClass.contains(SrcReg))
105 Opc = Mips::MFLO_DSP;
106 else if (Mips::DSPCCRegClass.contains(SrcReg)) {
107 BuildMI(MBB, I, DL, get(Mips::RDDSP), DestReg).addImm(1 << 4)
108 .addReg(SrcReg, RegState::Implicit | getKillRegState(KillSrc));
109 return;
110 }
111 else if (Mips::MSACtrlRegClass.contains(SrcReg))
112 Opc = Mips::CFCMSA;
113 }
114 else if (Mips::GPR32RegClass.contains(SrcReg)) { // Copy from CPU Reg.
115 if (Mips::CCRRegClass.contains(DestReg))
116 Opc = Mips::CTC1;
117 else if (Mips::FGR32RegClass.contains(DestReg))
118 Opc = Mips::MTC1;
119 else if (Mips::HI32RegClass.contains(DestReg))
120 Opc = Mips::MTHI, DestReg = 0;
121 else if (Mips::LO32RegClass.contains(DestReg))
122 Opc = Mips::MTLO, DestReg = 0;
123 else if (Mips::HI32DSPRegClass.contains(DestReg))
124 Opc = Mips::MTHI_DSP;
125 else if (Mips::LO32DSPRegClass.contains(DestReg))
126 Opc = Mips::MTLO_DSP;
127 else if (Mips::DSPCCRegClass.contains(DestReg)) {
128 BuildMI(MBB, I, DL, get(Mips::WRDSP))
129 .addReg(SrcReg, getKillRegState(KillSrc)).addImm(1 << 4)
130 .addReg(DestReg, RegState::ImplicitDefine);
131 return;
132 } else if (Mips::MSACtrlRegClass.contains(DestReg)) {
133 BuildMI(MBB, I, DL, get(Mips::CTCMSA))
134 .addReg(DestReg)
135 .addReg(SrcReg, getKillRegState(KillSrc));
136 return;
137 }
138 }
139 else if (Mips::FGR32RegClass.contains(DestReg, SrcReg))
140 Opc = Mips::FMOV_S;
141 else if (Mips::AFGR64RegClass.contains(DestReg, SrcReg))
142 Opc = Mips::FMOV_D32;
143 else if (Mips::FGR64RegClass.contains(DestReg, SrcReg))
144 Opc = Mips::FMOV_D64;
145 else if (Mips::GPR64RegClass.contains(DestReg)) { // Copy to CPU64 Reg.
146 if (Mips::GPR64RegClass.contains(SrcReg))
147 Opc = Mips::OR64, ZeroReg = Mips::ZERO_64;
148 else if (Mips::HI64RegClass.contains(SrcReg))
149 Opc = Mips::MFHI64, SrcReg = 0;
150 else if (Mips::LO64RegClass.contains(SrcReg))
151 Opc = Mips::MFLO64, SrcReg = 0;
152 else if (Mips::FGR64RegClass.contains(SrcReg))
153 Opc = Mips::DMFC1;
154 }
155 else if (Mips::GPR64RegClass.contains(SrcReg)) { // Copy from CPU64 Reg.
156 if (Mips::HI64RegClass.contains(DestReg))
157 Opc = Mips::MTHI64, DestReg = 0;
158 else if (Mips::LO64RegClass.contains(DestReg))
159 Opc = Mips::MTLO64, DestReg = 0;
160 else if (Mips::FGR64RegClass.contains(DestReg))
161 Opc = Mips::DMTC1;
162 } else if (Mips::CheriGPROrCNullRegClass.contains(SrcReg)) {
163 BuildMI(MBB, I, DL, get(Mips::CMove))
164 .addReg(DestReg, RegState::Define)
165 .addReg(SrcReg, getKillRegState(KillSrc));
166 return;
167 } else if (Mips::CheriHWRegsRegClass.contains(SrcReg)) {
168 BuildMI(MBB, I, DL, get(Mips::CReadHwr))
169 .addReg(DestReg, RegState::Define)
170 .addReg(SrcReg, getKillRegState(KillSrc));
171 return;
172 } else if (Mips::CheriHWRegsRegClass.contains(DestReg)) {
173 BuildMI(MBB, I, DL, get(Mips::CWriteHwr))
174 .addReg(SrcReg, RegState::Define)
175 .addReg(DestReg, getKillRegState(KillSrc));
176 return;
177 } else if (Mips::MSA128BRegClass.contains(DestReg)) { // Copy to MSA reg
178 if (Mips::MSA128BRegClass.contains(SrcReg))
179 Opc = Mips::MOVE_V;
180 }
181
182 assert(Opc && "Cannot copy registers");
183
184 MachineInstrBuilder MIB = BuildMI(MBB, I, DL, get(Opc));
185
186 if (DestReg)
187 MIB.addReg(DestReg, RegState::Define);
188
189 if (SrcReg)
190 MIB.addReg(SrcReg, getKillRegState(KillSrc));
191
192 if (ZeroReg)
193 MIB.addReg(ZeroReg);
194}

196static bool isORCopyInst(const MachineInstr &MI) {
197 switch (MI.getOpcode()) {
198 default:
199 break;
200 case Mips::OR_MM:
201 case Mips::OR:
202 if (MI.getOperand(2).getReg() == Mips::ZERO)
203 return true;
204 break;
205 case Mips::OR64:
206 if (MI.getOperand(2).getReg() == Mips::ZERO_64)
207 return true;
208 break;
209 }
210 return false;
211}

213static bool isCIncOffsetCopyInst(const MachineInstr &MI) {
214 if (MI.getOpcode() == Mips::CIncOffset &&
215 MI.getOperand(2).getReg() == Mips::ZERO_64)
216 return true;
217 if (MI.getOpcode() == Mips::CIncOffsetImm && MI.getOperand(2).getImm() == 0)
218 return true;
219 return false;
220}

222/// If @MI is WRDSP/RRDSP instruction return true with @isWrite set to true
223/// if it is WRDSP instruction.
224static bool isReadOrWriteToDSPReg(const MachineInstr &MI, bool &isWrite) {
225 switch (MI.getOpcode()) {
226 default:
227 return false;
228 case Mips::WRDSP:
229 case Mips::WRDSP_MM:
230 isWrite = true;
231 break;
232 case Mips::RDDSP:
233 case Mips::RDDSP_MM:
234 isWrite = false;
235 break;
236 }
237 return true;
238}

240/// We check for the common case of 'or', as it's MIPS' preferred instruction
241/// for GPRs but we have to check the operands to ensure that is the case.
242/// Other move instructions for MIPS are directly identifiable.
243bool MipsSEInstrInfo::isCopyInstrImpl(const MachineInstr &MI,
244 const MachineOperand *&Src,
245 const MachineOperand *&Dest) const {
246 bool isDSPControlWrite = false;
247 // Condition is made to match the creation of WRDSP/RDDSP copy instruction
248 // from copyPhysReg function.
249
250 // FIXME: Handle CRead/WriteHWR here as well?
251 if (isReadOrWriteToDSPReg(MI, isDSPControlWrite)) {
252 if (!MI.getOperand(1).isImm() || MI.getOperand(1).getImm() != (1<<4))
253 return false;
254 else if (isDSPControlWrite) {
255 Src = &MI.getOperand(0);
256 Dest = &MI.getOperand(2);
257 } else {
258 Dest = &MI.getOperand(0);
259 Src = &MI.getOperand(2);
260 }
261 return true;
262 } else if (MI.isMoveReg() || isORCopyInst(MI) || isCIncOffsetCopyInst(MI)) {
263 Dest = &MI.getOperand(0);
264 Src = &MI.getOperand(1);
265 return true;
266 }
267 return false;
268}

270void MipsSEInstrInfo::
271storeRegToStack(MachineBasicBlock &MBB, MachineBasicBlock::iterator I,
272 unsigned SrcReg, bool isKill, int FI,
273 const TargetRegisterClass *RC, const TargetRegisterInfo *TRI,
274 int64_t Offset) const {
275 DebugLoc DL;
276 MachineMemOperand *MMO = GetMemOperand(MBB, FI, MachineMemOperand::MOStore);
277
278 unsigned Opc = 0;
279
280 // The ACC64/128 registers are handled by STORE_ACC64/128 pseudos, which call this function again with more ordinary
281 // registers when they are lowered: so no special treatment for CHERI is required.
282 if (Subtarget.isABI_CheriPureCap() &&
283 !Mips::ACC64RegClass.hasSubClassEq(RC) &&
284 !Mips::ACC128RegClass.hasSubClassEq(RC)) {
285 if (Mips::GPR32RegClass.hasSubClassEq(RC))
286 Opc = Mips::CAPSTORE32;
287 else if (Mips::GPR64RegClass.hasSubClassEq(RC))
288 Opc = Mips::CAPSTORE64;
289 else if (Mips::FGR64RegClass.hasSubClassEq(RC)) {
290 DebugLoc DL = I->getDebugLoc();
291 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
292 unsigned IntReg = RegInfo.createVirtualRegister(&Mips::GPR64RegClass);
293 BuildMI(MBB, I, DL, get(Mips::DMFC1), IntReg)
294 .addReg(SrcReg);
295 BuildMI(MBB, I, DL, get(Mips::CAPSTORE64)).addReg(IntReg, getKillRegState(true))
296 .addReg(Mips::ZERO_64).addFrameIndex(FI).addImm(Offset).addMemOperand(MMO);
297 return;
298 } else if (Mips::CheriGPROrCNullRegClass.hasSubClassEq(RC)) {
299 Opc = Mips::STORECAP;
300 // Ensure that capabilities have a 32-byte alignment
301 // FIXME: This shouldn't be needed. Whatever is allocating the frame index
302 // ought to set it.
303 MachineFrameInfo &MFI = MBB.getParent()->getFrameInfo();
304 MFI.setObjectAlignment(FI, Subtarget.isCheri128() ? 16 : 32);
305 } else {
306 llvm_unreachable("Unexpected register type for CHERI!");
307 }
308 BuildMI(MBB, I, DL, get(Opc)).addReg(SrcReg, getKillRegState(isKill))
309 .addReg(Mips::ZERO_64).addFrameIndex(FI).addImm(Offset).addMemOperand(MMO);
310 return;
311 }
312 if (Mips::GPR32RegClass.hasSubClassEq(RC))
313 Opc = Mips::SW;
314 else if (Mips::GPR64RegClass.hasSubClassEq(RC))
315 Opc = Mips::SD;
316 else if (Mips::ACC64RegClass.hasSubClassEq(RC))
317 Opc = Mips::STORE_ACC64;
318 else if (Mips::ACC64DSPRegClass.hasSubClassEq(RC))
319 Opc = Mips::STORE_ACC64DSP;
320 else if (Mips::ACC128RegClass.hasSubClassEq(RC))
321 Opc = Mips::STORE_ACC128;
322 else if (Mips::DSPCCRegClass.hasSubClassEq(RC))
323 Opc = Mips::STORE_CCOND_DSP;
324 else if (Mips::FGR32RegClass.hasSubClassEq(RC))
325 Opc = Mips::SWC1;
326 else if (Mips::AFGR64RegClass.hasSubClassEq(RC))
327 Opc = Mips::SDC1;
328 else if (Mips::FGR64RegClass.hasSubClassEq(RC))
329 Opc = Mips::SDC164;
330 else if (TRI->isTypeLegalForClass(*RC, MVT::v16i8))
331 Opc = Mips::ST_B;
332 else if (TRI->isTypeLegalForClass(*RC, MVT::v8i16) ||
333 TRI->isTypeLegalForClass(*RC, MVT::v8f16))
334 Opc = Mips::ST_H;
335 else if (TRI->isTypeLegalForClass(*RC, MVT::v4i32) ||
336 TRI->isTypeLegalForClass(*RC, MVT::v4f32))
337 Opc = Mips::ST_W;
338 else if (TRI->isTypeLegalForClass(*RC, MVT::v2i64) ||
339 TRI->isTypeLegalForClass(*RC, MVT::v2f64))
340 Opc = Mips::ST_D;
341 else if (Mips::CheriGPROrCNullRegClass.hasSubClassEq(RC)) {
342 Opc = Mips::STORECAP;
343 // Ensure that capabilities have a 32-byte alignment
344 // FIXME: This shouldn't be needed. Whatever is allocating the frame index
345 // ought to set it.
346 MachineFrameInfo &MFI = MBB.getParent()->getFrameInfo();
347 MFI.setObjectAlignment(FI, Subtarget.isCheri128() ? 16 : 32);
348 BuildMI(MBB, I, DL, get(Opc))
349 .addReg(SrcReg, getKillRegState(isKill))
350 .addFrameIndex(FI)
351 .addImm(Offset)
352 .addMemOperand(MMO)
353 .addReg(Mips::DDC);
354 return;
355 } else if (Mips::LO32RegClass.hasSubClassEq(RC))
356 Opc = Mips::SW;
357 else if (Mips::LO64RegClass.hasSubClassEq(RC))
358 Opc = Mips::SD;
359 else if (Mips::HI32RegClass.hasSubClassEq(RC))
360 Opc = Mips::SW;
361 else if (Mips::HI64RegClass.hasSubClassEq(RC))
362 Opc = Mips::SD;
363 else if (Mips::DSPRRegClass.hasSubClassEq(RC))
364 Opc = Mips::SWDSP;
365
366 // Hi, Lo are normally caller save but they are callee save
367 // for interrupt handling.
368 const Function &Func = MBB.getParent()->getFunction();
369 if (Func.hasFnAttribute("interrupt")) {
370 if (Mips::HI32RegClass.hasSubClassEq(RC)) {
371 BuildMI(MBB, I, DL, get(Mips::MFHI), Mips::K0);
372 SrcReg = Mips::K0;
373 } else if (Mips::HI64RegClass.hasSubClassEq(RC)) {
374 BuildMI(MBB, I, DL, get(Mips::MFHI64), Mips::K0_64);
375 SrcReg = Mips::K0_64;
376 } else if (Mips::LO32RegClass.hasSubClassEq(RC)) {
377 BuildMI(MBB, I, DL, get(Mips::MFLO), Mips::K0);
378 SrcReg = Mips::K0;
379 } else if (Mips::LO64RegClass.hasSubClassEq(RC)) {
380 BuildMI(MBB, I, DL, get(Mips::MFLO64), Mips::K0_64);
381 SrcReg = Mips::K0_64;
382 }
383 }
384
385 assert(Opc && "Register class not handled!");
386 BuildMI(MBB, I, DL, get(Opc)).addReg(SrcReg, getKillRegState(isKill))
387 .addFrameIndex(FI).addImm(Offset).addMemOperand(MMO);
388}

390void MipsSEInstrInfo::
391loadRegFromStack(MachineBasicBlock &MBB, MachineBasicBlock::iterator I,
392 unsigned DestReg, int FI, const TargetRegisterClass *RC,
393 const TargetRegisterInfo *TRI, int64_t Offset) const {
394 DebugLoc DL;
395 if (I != MBB.end()) DL = I->getDebugLoc();
396 MachineMemOperand *MMO = GetMemOperand(MBB, FI, MachineMemOperand::MOLoad);
397 unsigned Opc = 0;
398
399 // The ACC64/128 registers are handled by LOAD_ACC64/128 pseudos, which call this function again with more ordinary
400 // registers when they are lowered: so no special treatment for CHERI is required.
401 if (Subtarget.isABI_CheriPureCap() &&
402 !Mips::ACC64RegClass.hasSubClassEq(RC) &&
403 !Mips::ACC128RegClass.hasSubClassEq(RC)) {
404 if (Mips::GPR32RegClass.hasSubClassEq(RC))
405 Opc = Mips::CAPLOAD32;
406 else if (Mips::GPR64RegClass.hasSubClassEq(RC))
407 Opc = Mips::CAPLOAD64;
408 else if (Mips::FGR64RegClass.hasSubClassEq(RC)) {
409 DebugLoc DL = I->getDebugLoc();
410 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
411 unsigned IntReg = RegInfo.createVirtualRegister(&Mips::GPR64RegClass);
412 BuildMI(MBB, I, DL, get(Mips::CAPLOAD64), IntReg)
413 .addReg(Mips::ZERO_64).addFrameIndex(FI).addImm(Offset)
414 .addMemOperand(MMO);
415 BuildMI(MBB, I, DL, get(Mips::DMTC1), DestReg)
416 .addReg(IntReg, getKillRegState(true));
417 return;
418 } else if (Mips::CheriGPROrCNullRegClass.hasSubClassEq(RC)) {
419 Opc = Mips::LOADCAP;
420 } else {
421 llvm_unreachable("Unexpected register type for CHERI!");
422 }
423 BuildMI(MBB, I, DL, get(Opc), DestReg)
424 .addReg(Mips::ZERO_64).addFrameIndex(FI).addImm(Offset)
425 .addMemOperand(MMO);
426 return;
427 }
428 const Function &Func = MBB.getParent()->getFunction();
429 bool ReqIndirectLoad = Func.hasFnAttribute("interrupt") &&
430 (DestReg == Mips::LO0 || DestReg == Mips::LO0_64 ||
431 DestReg == Mips::HI0 || DestReg == Mips::HI0_64);
432
433 if (Mips::GPR32RegClass.hasSubClassEq(RC))
434 Opc = Mips::LW;
435 else if (Mips::GPR64RegClass.hasSubClassEq(RC))
436 Opc = Mips::LD;
437 else if (Mips::ACC64RegClass.hasSubClassEq(RC))
438 Opc = Mips::LOAD_ACC64;
439 else if (Mips::ACC64DSPRegClass.hasSubClassEq(RC))
440 Opc = Mips::LOAD_ACC64DSP;
441 else if (Mips::ACC128RegClass.hasSubClassEq(RC))
442 Opc = Mips::LOAD_ACC128;
443 else if (Mips::DSPCCRegClass.hasSubClassEq(RC))
444 Opc = Mips::LOAD_CCOND_DSP;
445 else if (Mips::FGR32RegClass.hasSubClassEq(RC))
446 Opc = Mips::LWC1;
447 else if (Mips::AFGR64RegClass.hasSubClassEq(RC))
448 Opc = Mips::LDC1;
449 else if (Mips::FGR64RegClass.hasSubClassEq(RC))
450 Opc = Mips::LDC164;
451 else if (TRI->isTypeLegalForClass(*RC, MVT::v16i8))
452 Opc = Mips::LD_B;
453 else if (TRI->isTypeLegalForClass(*RC, MVT::v8i16) ||
454 TRI->isTypeLegalForClass(*RC, MVT::v8f16))
455 Opc = Mips::LD_H;
456 else if (TRI->isTypeLegalForClass(*RC, MVT::v4i32) ||
457 TRI->isTypeLegalForClass(*RC, MVT::v4f32))
458 Opc = Mips::LD_W;
459 else if (TRI->isTypeLegalForClass(*RC, MVT::v2i64) ||
460 TRI->isTypeLegalForClass(*RC, MVT::v2f64))
461 Opc = Mips::LD_D;
462 else if (Mips::CheriGPROrCNullRegClass.hasSubClassEq(RC)) {
463 Opc = Mips::LOADCAP;
464 BuildMI(MBB, I, DL, get(Opc), DestReg)
465 .addFrameIndex(FI)
466 .addImm(Offset)
467 .addMemOperand(MMO)
468 .addReg(Mips::DDC);
469 return;
470 } else if (Mips::HI32RegClass.hasSubClassEq(RC))
471 Opc = Mips::LW;
472 else if (Mips::HI64RegClass.hasSubClassEq(RC))
473 Opc = Mips::LD;
474 else if (Mips::LO32RegClass.hasSubClassEq(RC))
475 Opc = Mips::LW;
476 else if (Mips::LO64RegClass.hasSubClassEq(RC))
477 Opc = Mips::LD;
478 else if (Mips::DSPRRegClass.hasSubClassEq(RC))
479 Opc = Mips::LWDSP;
480
481 assert(Opc && "Register class not handled!");
482
483 if (!ReqIndirectLoad)
484 BuildMI(MBB, I, DL, get(Opc), DestReg)
485 .addFrameIndex(FI)
486 .addImm(Offset)
487 .addMemOperand(MMO);
488 else {
489 // Load HI/LO through K0. Notably the DestReg is encoded into the
490 // instruction itself.
491 unsigned Reg = Mips::K0;
492 unsigned LdOp = Mips::MTLO;
493 if (DestReg == Mips::HI0)
494 LdOp = Mips::MTHI;
495
496 if (Subtarget.getABI().ArePtrs64bit()) {
497 Reg = Mips::K0_64;
498 if (DestReg == Mips::HI0_64)
499 LdOp = Mips::MTHI64;
500 else
501 LdOp = Mips::MTLO64;
502 }
503
504 BuildMI(MBB, I, DL, get(Opc), Reg)
505 .addFrameIndex(FI)
506 .addImm(Offset)
507 .addMemOperand(MMO);
508 BuildMI(MBB, I, DL, get(LdOp)).addReg(Reg);
509 }

512bool MipsSEInstrInfo::expandPostRAPseudo(MachineInstr &MI) const {
513 MachineBasicBlock &MBB = *MI.getParent();
514 bool isMicroMips = Subtarget.inMicroMipsMode();
515 unsigned Opc;
516
517 switch (MI.getDesc().getOpcode()) {
518 default:
519 return false;
520 case Mips::RetRA:
521 expandRetRA(MBB, MI);
522 break;
523 case Mips::ERet:
524 expandERet(MBB, MI);
525 break;
526 case Mips::PseudoMFHI:
527 expandPseudoMFHiLo(MBB, MI, Mips::MFHI);
528 break;
529 case Mips::PseudoMFHI_MM:
530 expandPseudoMFHiLo(MBB, MI, Mips::MFHI16_MM);
531 break;
532 case Mips::PseudoMFLO:
533 expandPseudoMFHiLo(MBB, MI, Mips::MFLO);
534 break;
535 case Mips::PseudoMFLO_MM:
536 expandPseudoMFHiLo(MBB, MI, Mips::MFLO16_MM);
537 break;
538 case Mips::PseudoMFHI64:
539 expandPseudoMFHiLo(MBB, MI, Mips::MFHI64);
540 break;
541 case Mips::PseudoMFLO64:
542 expandPseudoMFHiLo(MBB, MI, Mips::MFLO64);
543 break;
544 case Mips::PseudoMTLOHI:
545 expandPseudoMTLoHi(MBB, MI, Mips::MTLO, Mips::MTHI, false);
546 break;
547 case Mips::PseudoMTLOHI64:
548 expandPseudoMTLoHi(MBB, MI, Mips::MTLO64, Mips::MTHI64, false);
549 break;
550 case Mips::PseudoMTLOHI_DSP:
551 expandPseudoMTLoHi(MBB, MI, Mips::MTLO_DSP, Mips::MTHI_DSP, true);
552 break;
553 case Mips::PseudoMTLOHI_MM:
554 expandPseudoMTLoHi(MBB, MI, Mips::MTLO_MM, Mips::MTHI_MM, false);
555 break;
556 case Mips::PseudoCVT_S_W:
557 expandCvtFPInt(MBB, MI, Mips::CVT_S_W, Mips::MTC1, false);
558 break;
559 case Mips::PseudoCVT_D32_W:
560 Opc = isMicroMips ? Mips::CVT_D32_W_MM : Mips::CVT_D32_W;
561 expandCvtFPInt(MBB, MI, Opc, Mips::MTC1, false);
562 break;
563 case Mips::PseudoCVT_S_L:
564 expandCvtFPInt(MBB, MI, Mips::CVT_S_L, Mips::DMTC1, true);
565 break;
566 case Mips::PseudoCVT_D64_W:
567 Opc = isMicroMips ? Mips::CVT_D64_W_MM : Mips::CVT_D64_W;
568 expandCvtFPInt(MBB, MI, Opc, Mips::MTC1, true);
569 break;
570 case Mips::PseudoCVT_D64_L:
571 expandCvtFPInt(MBB, MI, Mips::CVT_D64_L, Mips::DMTC1, true);
572 break;
573 case Mips::BuildPairF64:
574 expandBuildPairF64(MBB, MI, isMicroMips, false);
575 break;
576 case Mips::BuildPairF64_64:
577 expandBuildPairF64(MBB, MI, isMicroMips, true);
578 break;
579 case Mips::ExtractElementF64:
580 expandExtractElementF64(MBB, MI, isMicroMips, false);
581 break;
582 case Mips::ExtractElementF64_64:
583 expandExtractElementF64(MBB, MI, isMicroMips, true);
584 break;
585 case Mips::MIPSeh_return32:
586 case Mips::MIPSeh_return64:
587 expandEhReturn(MBB, MI);
588 break;
589 case Mips::CapRetPseudo:
590 BuildMI(MBB, MI, MI.getDebugLoc(), get(Mips::PseudoReturnCap))
591 .addReg(Mips::C17);
592 break;
593 case Mips::CheriBoundedStackPseudo: {
594 auto Op = MI.getOperand(3).isImm() ? Mips::CSetBoundsImm : Mips::CSetBounds;
595 if (Op == Mips::CSetBoundsImm)
596 assert(isInt<11>(MI.getOperand(3).getImm()));
597 else
598 assert(MI.getOperand(3).isReg());
599 assert(MI.getOperand(2).getImm() == 0 && "This operand is a dummy and must be zero!");
600 BuildMI(MBB, MI, MI.getDebugLoc(), get(Op), MI.getOperand(0).getReg())
601 .addReg(MI.getOperand(1).getReg()).add(MI.getOperand(3));
602 break;
603 }
604 case Mips::CPSETUP:
605 expandCPSETUP(MBB, MI);
606 break;
607 case Mips::CCallPseudo:
608 expandCCallPseudo(MBB, MI);
609 break;
610 }
611
612 MBB.erase(MI);
613 return true;
614}

708/// Adjust SP by Amount bytes.
709void MipsSEInstrInfo::adjustStackPtr(unsigned SP, int64_t Amount,
710 MachineBasicBlock &MBB,
711 MachineBasicBlock::iterator I) const {
712 MipsABIInfo ABI = Subtarget.getABI();
713 DebugLoc DL;
714 unsigned ADDiu = ABI.GetPtrAddiuOp();
715
716 if (Amount == 0)
717 return;
718
719 if (ABI.IsCheriPureCap()) {
720 if (isInt<11>(Amount)) {
721 BuildMI(MBB, I, DL, get(Mips::CIncOffsetImm), SP)
722 .addReg(SP).addImm(Amount);
723 } else {
724 unsigned Reg = loadImmediate(Amount, MBB, I, DL, nullptr);
725 BuildMI(MBB, I, DL, get(Mips::CIncOffset), SP)
726 .addReg(SP).addReg(Reg, RegState::Kill);
727 }
728 } else if (isInt<16>(Amount)) {
729 // addi sp, sp, amount
730 BuildMI(MBB, I, DL, get(ADDiu), SP).addReg(SP).addImm(Amount);
731 } else {
732 // For numbers which are not 16bit integers we synthesize Amount inline
733 // then add or subtract it from sp.
734 unsigned Opc = ABI.GetPtrAdduOp();
735 if (Amount < 0) {
736 Opc = ABI.GetPtrSubuOp();
737 Amount = -Amount;
738 }
739 unsigned Reg = loadImmediate(Amount, MBB, I, DL, nullptr);
740 BuildMI(MBB, I, DL, get(Opc), SP).addReg(SP).addReg(Reg, RegState::Kill);
741 }
742}

744/// This function generates the sequence of instructions needed to get the
745/// result of adding register REG and immediate IMM.
746unsigned MipsSEInstrInfo::loadImmediate(int64_t Imm, MachineBasicBlock &MBB,
747 MachineBasicBlock::iterator II,
748 const DebugLoc &DL,
749 unsigned *NewImm) const {
750 MipsAnalyzeImmediate AnalyzeImm;
751 const MipsSubtarget &STI = Subtarget;
752 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
753 unsigned Size = STI.isABI_N64() ? 64 : 32;
754 unsigned LUi = STI.isABI_N64() ? Mips::LUi64 : Mips::LUi;
755 unsigned ZEROReg = STI.isABI_N64() ? Mips::ZERO_64 : Mips::ZERO;
756 const TargetRegisterClass *RC = STI.isABI_N64() ?
757 &Mips::GPR64RegClass : &Mips::GPR32RegClass;
758 bool LastInstrIsADDiu = NewImm;
759
760 const MipsAnalyzeImmediate::InstSeq &Seq =
761 AnalyzeImm.Analyze(Imm, Size, LastInstrIsADDiu);
762 MipsAnalyzeImmediate::InstSeq::const_iterator Inst = Seq.begin();
763
764 assert(Seq.size() && (!LastInstrIsADDiu || (Seq.size() > 1)));
765
766 // The first instruction can be a LUi, which is different from other
767 // instructions (ADDiu, ORI and SLL) in that it does not have a register
768 // operand.
769 unsigned Reg = RegInfo.createVirtualRegister(RC);
770
771 if (Inst->Opc == LUi)
772 BuildMI(MBB, II, DL, get(LUi), Reg).addImm(SignExtend64<16>(Inst->ImmOpnd));
773 else
774 BuildMI(MBB, II, DL, get(Inst->Opc), Reg).addReg(ZEROReg)
775 .addImm(SignExtend64<16>(Inst->ImmOpnd));
776
777 // Build the remaining instructions in Seq.
778 for (++Inst; Inst != Seq.end() - LastInstrIsADDiu; ++Inst)
779 BuildMI(MBB, II, DL, get(Inst->Opc), Reg).addReg(Reg, RegState::Kill)
780 .addImm(SignExtend64<16>(Inst->ImmOpnd));
781
782 if (LastInstrIsADDiu)
783 *NewImm = Inst->ImmOpnd;
784
785 return Reg;
786}

858void MipsSEInstrInfo::expandPseudoMTLoHi(MachineBasicBlock &MBB,
859 MachineBasicBlock::iterator I,
860 unsigned LoOpc,
861 unsigned HiOpc,
862 bool HasExplicitDef) const {
863 // Expand
864 // lo_hi pseudomtlohi $gpr0, $gpr1
865 // to these two instructions:
866 // mtlo $gpr0
867 // mthi $gpr1
868
869 DebugLoc DL = I->getDebugLoc();
870 const MachineOperand &SrcLo = I->getOperand(1), &SrcHi = I->getOperand(2);
871 MachineInstrBuilder LoInst = BuildMI(MBB, I, DL, get(LoOpc));
872 MachineInstrBuilder HiInst = BuildMI(MBB, I, DL, get(HiOpc));
873
874 // Add lo/hi registers if the mtlo/hi instructions created have explicit
875 // def registers.
876 if (HasExplicitDef) {
877 unsigned DstReg = I->getOperand(0).getReg();
878 unsigned DstLo = getRegisterInfo().getSubReg(DstReg, Mips::sub_lo);
879 unsigned DstHi = getRegisterInfo().getSubReg(DstReg, Mips::sub_hi);
880 LoInst.addReg(DstLo, RegState::Define);
881 HiInst.addReg(DstHi, RegState::Define);
882 }
883
884 LoInst.addReg(SrcLo.getReg(), getKillRegState(SrcLo.isKill()));
885 HiInst.addReg(SrcHi.getReg(), getKillRegState(SrcHi.isKill()));
886}

888void MipsSEInstrInfo::expandCvtFPInt(MachineBasicBlock &MBB,
889 MachineBasicBlock::iterator I,
890 unsigned CvtOpc, unsigned MovOpc,
891 bool IsI64) const {
892 const MCInstrDesc &CvtDesc = get(CvtOpc), &MovDesc = get(MovOpc);
893 const MachineOperand &Dst = I->getOperand(0), &Src = I->getOperand(1);
894 unsigned DstReg = Dst.getReg(), SrcReg = Src.getReg(), TmpReg = DstReg;
895 unsigned KillSrc = getKillRegState(Src.isKill());
896 DebugLoc DL = I->getDebugLoc();
897 bool DstIsLarger, SrcIsLarger;
898
899 std::tie(DstIsLarger, SrcIsLarger) =
900 compareOpndSize(CvtOpc, *MBB.getParent());
901
902 if (DstIsLarger)
903 TmpReg = getRegisterInfo().getSubReg(DstReg, Mips::sub_lo);
904
905 if (SrcIsLarger)
906 DstReg = getRegisterInfo().getSubReg(DstReg, Mips::sub_lo);
907
908 BuildMI(MBB, I, DL, MovDesc, TmpReg).addReg(SrcReg, KillSrc);
909 BuildMI(MBB, I, DL, CvtDesc, DstReg).addReg(TmpReg, RegState::Kill);
910}

912void MipsSEInstrInfo::expandExtractElementF64(MachineBasicBlock &MBB,
913 MachineBasicBlock::iterator I,
914 bool isMicroMips,
915 bool FP64) const {
916 unsigned DstReg = I->getOperand(0).getReg();
917 unsigned SrcReg = I->getOperand(1).getReg();
918 unsigned N = I->getOperand(2).getImm();
919 DebugLoc dl = I->getDebugLoc();
920
921 assert(N < 2 && "Invalid immediate");
922 unsigned SubIdx = N ? Mips::sub_hi : Mips::sub_lo;
923 unsigned SubReg = getRegisterInfo().getSubReg(SrcReg, SubIdx);
924
925 // FPXX on MIPS-II or MIPS32r1 should have been handled with a spill/reload
926 // in MipsSEFrameLowering.cpp.
927 assert(!(Subtarget.isABI_FPXX() && !Subtarget.hasMips32r2()));
928
929 // FP64A (FP64 with nooddspreg) should have been handled with a spill/reload
930 // in MipsSEFrameLowering.cpp.
931 assert(!(Subtarget.isFP64bit() && !Subtarget.useOddSPReg()));
932
933 if (SubIdx == Mips::sub_hi && Subtarget.hasMTHC1()) {
934 // FIXME: Strictly speaking MFHC1 only reads the top 32-bits however, we
935 // claim to read the whole 64-bits as part of a white lie used to
936 // temporarily work around a widespread bug in the -mfp64 support.
937 // The problem is that none of the 32-bit fpu ops mention the fact
938 // that they clobber the upper 32-bits of the 64-bit FPR. Fixing that
939 // requires a major overhaul of the FPU implementation which can't
940 // be done right now due to time constraints.
941 // MFHC1 is one of two instructions that are affected since they are
942 // the only instructions that don't read the lower 32-bits.
943 // We therefore pretend that it reads the bottom 32-bits to
944 // artificially create a dependency and prevent the scheduler
945 // changing the behaviour of the code.
946 BuildMI(MBB, I, dl,
947 get(isMicroMips ? (FP64 ? Mips::MFHC1_D64_MM : Mips::MFHC1_D32_MM)
948 : (FP64 ? Mips::MFHC1_D64 : Mips::MFHC1_D32)),
949 DstReg)
950 .addReg(SrcReg);
951 } else
952 BuildMI(MBB, I, dl, get(Mips::MFC1), DstReg).addReg(SubReg);
953}

955void MipsSEInstrInfo::expandBuildPairF64(MachineBasicBlock &MBB,
956 MachineBasicBlock::iterator I,
957 bool isMicroMips, bool FP64) const {
958 unsigned DstReg = I->getOperand(0).getReg();
959 unsigned LoReg = I->getOperand(1).getReg(), HiReg = I->getOperand(2).getReg();
960 const MCInstrDesc& Mtc1Tdd = get(Mips::MTC1);
961 DebugLoc dl = I->getDebugLoc();
962 const TargetRegisterInfo &TRI = getRegisterInfo();
963
964 // When mthc1 is available, use:
965 // mtc1 Lo, $fp
966 // mthc1 Hi, $fp
967 //
968 // Otherwise, for O32 FPXX ABI:
969 // spill + reload via ldc1
970 // This case is handled by the frame lowering code.
971 //
972 // Otherwise, for FP32:
973 // mtc1 Lo, $fp
974 // mtc1 Hi, $fp + 1
975 //
976 // The case where dmtc1 is available doesn't need to be handled here
977 // because it never creates a BuildPairF64 node.
978
979 // FPXX on MIPS-II or MIPS32r1 should have been handled with a spill/reload
980 // in MipsSEFrameLowering.cpp.
981 assert(!(Subtarget.isABI_FPXX() && !Subtarget.hasMips32r2()));
982
983 // FP64A (FP64 with nooddspreg) should have been handled with a spill/reload
984 // in MipsSEFrameLowering.cpp.
985 assert(!(Subtarget.isFP64bit() && !Subtarget.useOddSPReg()));
986
987 BuildMI(MBB, I, dl, Mtc1Tdd, TRI.getSubReg(DstReg, Mips::sub_lo))
988 .addReg(LoReg);
989
990 if (Subtarget.hasMTHC1()) {
991 // FIXME: The .addReg(DstReg) is a white lie used to temporarily work
992 // around a widespread bug in the -mfp64 support.
993 // The problem is that none of the 32-bit fpu ops mention the fact
994 // that they clobber the upper 32-bits of the 64-bit FPR. Fixing that
995 // requires a major overhaul of the FPU implementation which can't
996 // be done right now due to time constraints.
997 // MTHC1 is one of two instructions that are affected since they are
998 // the only instructions that don't read the lower 32-bits.
999 // We therefore pretend that it reads the bottom 32-bits to
1000 // artificially create a dependency and prevent the scheduler
1001 // changing the behaviour of the code.
1002 BuildMI(MBB, I, dl,
1003 get(isMicroMips ? (FP64 ? Mips::MTHC1_D64_MM : Mips::MTHC1_D32_MM)
1004 : (FP64 ? Mips::MTHC1_D64 : Mips::MTHC1_D32)),
1005 DstReg)
1006 .addReg(DstReg)
1007 .addReg(HiReg);
1008 } else if (Subtarget.isABI_FPXX())
1009 llvm_unreachable("BuildPairF64 not expanded in frame lowering code!");
1010 else
1011 BuildMI(MBB, I, dl, Mtc1Tdd, TRI.getSubReg(DstReg, Mips::sub_hi))
1012 .addReg(HiReg);
1013}

1015void MipsSEInstrInfo::expandEhReturn(MachineBasicBlock &MBB,
1016 MachineBasicBlock::iterator I) const {
1017 // This pseudo instruction is generated as part of the lowering of
1018 // ISD::EH_RETURN. We convert it to a stack increment by OffsetReg, and
1019 // indirect jump to TargetReg
1020 MipsABIInfo ABI = Subtarget.getABI();
1021 unsigned ADDU = ABI.GetPtrAdduOp();
1022 unsigned SP = Subtarget.isGP64bit() ? Mips::SP_64 : Mips::SP;
1023 unsigned RA = Subtarget.isGP64bit() ? Mips::RA_64 : Mips::RA;
1024 unsigned T9 = Subtarget.isGP64bit() ? Mips::T9_64 : Mips::T9;
1025 unsigned ZERO = Subtarget.isGP64bit() ? Mips::ZERO_64 : Mips::ZERO;
1026 unsigned OffsetReg = I->getOperand(0).getReg();
1027 unsigned TargetReg = I->getOperand(1).getReg();
1028
1029 // addu $ra, $v0, $zero
1030 // addu $sp, $sp, $v1
1031 // jr $ra (via RetRA)
1032 const TargetMachine &TM = MBB.getParent()->getTarget();
1033 if (TM.isPositionIndependent())
1034 BuildMI(MBB, I, I->getDebugLoc(), get(ADDU), T9)
1035 .addReg(TargetReg)
1036 .addReg(ZERO);
1037 BuildMI(MBB, I, I->getDebugLoc(), get(ADDU), RA)
1038 .addReg(TargetReg)
1039 .addReg(ZERO);
1040 BuildMI(MBB, I, I->getDebugLoc(), get(ADDU), SP).addReg(SP).addReg(OffsetReg);
1041 expandRetRA(MBB, I);
1042}

1044void MipsSEInstrInfo::expandCCallPseudo(MachineBasicBlock &MBB,
1045 MachineBasicBlock::iterator I) const {
1046 const TargetInstrInfo *TII = Subtarget.getInstrInfo();
1047 DebugLoc DL = I->getDebugLoc();
1048 // Operand 1 contains the mask of registers to clear, with the top 16 bits
1049 // containing the capability registers and the bottom 16 bits containing the
1050 // integer registers (floating point are currently ignored).
1051 uint32_t ClearMask = I->getOperand(1).getImm();
1052 // Clear the capability registers
1053 MachineBasicBlock::iterator BundleStart = BuildMI(MBB, I, DL,
1054 TII->get(Mips::CClearLo)).addImm(ClearMask >> 16);
1055 // Emit the jump
1056 BuildMI(MBB, I, DL, TII->get(Mips::JALR64),
1057 Mips::RA_64).addReg(I->getOperand(0).getReg());
1058 // Clear the integer registers (in the delay slot)
1059 BuildMI(MBB, I, DL, TII->get(Mips::ClearLo)).addImm(ClearMask & 0xffff);
1060 // Ensure that the jump and the delay slot are not split
1061 MIBundleBuilder(MBB, BundleStart, std::next(BundleStart, 3));
1062}

1064// For opcodes with the ReMaterializable flag set, this function is called to
1065// verify the instruction is really rematable.
1066bool MipsSEInstrInfo::isReallyTriviallyReMaterializable(const MachineInstr &MI,
1067 AliasAnalysis *AA) const {
1068 switch(MI.getOpcode()) {
1069 // To allow moving CSetBounds on the stack as late as possible.
1070 case Mips::CheriBoundedStackPseudo:
1071 // LLVM_DEBUG(dbgs() << "isReallyTriviallyReMaterializable: CHECKING "; MI.dump();)
1072 // We cannot trivially rematerialize if the size operand is a GPR since
1073 // That might be dead by the time we use it. Only remat
1074 return MI.getOperand(3).isImm();
1075 case Mips::CIncOffsetImm:
1076 case Mips::CMove:
1077 return MI.getOperand(1).isReg() && MI.getOperand(1).getReg() == Mips::CNULL;
1078 case Mips::LUi64: {
1079 auto Flags = MI.getOperand(1).getTargetFlags();
1080 if (Flags == MipsII::MO_CAPTABLE_OFF_HI) {
1081 return true;
1082 }
1083 return false;
1084 }
1085 default:
1086 return false;
1087 }

1090void MipsSEInstrInfo::expandCPSETUP(MachineBasicBlock &MBB,
1091 MachineBasicBlock::iterator I) const {
1092 unsigned GP = I->getOperand(0).getReg();
1093 assert(GP != Mips::T9_64);
1094 DebugLoc DL = I->getDebugLoc();
1095 const TargetInstrInfo *TII = Subtarget.getInstrInfo();
1096 const GlobalValue& FName = MBB.getParent()->getFunction();
1097 BuildMI(MBB, I, DL, TII->get(Mips::LUi64), GP)
1098 .addGlobalAddress(&FName, 0, MipsII::MO_GPOFF_HI);
1099 BuildMI(MBB, I, DL, TII->get(Mips::DADDu), GP).addReg(GP)
1100 .addReg(Mips::T9_64);
1101 BuildMI(MBB, I, DL, TII->get(Mips::DADDiu), GP).addReg(GP)
1102 .addGlobalAddress(&FName, 0, MipsII::MO_GPOFF_LO);
1103}
