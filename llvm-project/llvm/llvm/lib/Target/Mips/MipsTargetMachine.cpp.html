75 initializeCheriPureCapABIPass(*PR);
76}

78static std::string computeDataLayout(const Triple &TT, StringRef CPU,
79 const TargetOptions &Options,
80 StringRef FS,
81 bool isLittle) {
82 std::string Ret;
83 MipsABIInfo ABI = MipsABIInfo::computeTargetABI(TT, CPU, Options.MCOptions);
84
85 // There are both little and big endian mips.
86 if (isLittle)
87 Ret += "e";
88 else
89 Ret += "E";
90
91 if (ABI.IsO32())
92 Ret += "-m:m";
93 else
94 Ret += "-m:e";
95
96 // For CHERI256 we need to ensure at least capability-aligned stacks
97 unsigned MinStackAlignBits = 1;
98 if (FS.find("+cheri128") != StringRef::npos) {
99 Ret += "-pf200:128:128:128:64";
100 MinStackAlignBits = 128;
101 } else if (FS.find("+cheri64") != StringRef::npos) {
102 Ret += "-pf200:64:64:64:32";
103 MinStackAlignBits = 64;
104 } else if (FS.find("+cheri256") != StringRef::npos ||
105 Triple(TT).getArch() == Triple::cheri) {
106 Ret += "-pf200:256:256:256:64";
107 MinStackAlignBits = 256;
108 }
109
110 // Pointers are 32 bit on some ABIs.
111 if (!ABI.IsN64())
112 Ret += "-p:32:32";
113
114 // 8 and 16 bit integers only need to have natural alignment, but try to
115 // align them to 32 bits. 64 bit integers have natural alignment.
116 Ret += "-i8:8:32-i16:16:32-i64:64";
117
118 // 32 bit registers are always available and the stack is at least 64 bit
119 // aligned. On N64 64 bit registers are also available and the stack is
120 // 128 bit aligned.
121 if (ABI.IsN64() || ABI.IsN32())
122 Ret += "-n32:64-S" + llvm::utostr(std::max(128u, MinStackAlignBits));
123 else
124 Ret += "-n32-S64";
125
126 // TODO: we may want to put functions in AS201 at some point
127 if (ABI.IsCheriPureCap())
128 Ret += "-A200-P200";
129 // Capability wan't all globals in AS200 (except for TLS variables)
130 if (ABI.UsesCapabilityTable())
131 Ret += "-G200";
132
133 return Ret;
134}

136static Reloc::Model getEffectiveRelocModel(bool JIT,
137 Optional<Reloc::Model> RM) {
138 if (!RM.hasValue() || JIT)
139 return Reloc::Static;
140 return *RM;
141}

143// On function prologue, the stack is created by decrementing
144// its pointer. Once decremented, all references are done with positive
145// offset from the stack/frame pointer, using StackGrowsUp enables
146// an easier handling.
147// Using CodeModel::Large enables different CALL behavior.
148MipsTargetMachine::MipsTargetMachine(const Target &T, const Triple &TT,
149 StringRef CPU, StringRef FS,
150 const TargetOptions &Options,
151 Optional<Reloc::Model> RM,
152 Optional<CodeModel::Model> CM,
153 CodeGenOpt::Level OL, bool JIT,
154 bool isLittle)
155 : LLVMTargetMachine(T, computeDataLayout(TT, CPU, Options, FS, isLittle), TT,
156 CPU, FS, Options, getEffectiveRelocModel(JIT, RM),
157 getEffectiveCodeModel(CM, CodeModel::Small), OL),
158 isLittle(isLittle), TLOF(llvm::make_unique<MipsTargetObjectFile>()),
159 ABI(MipsABIInfo::computeTargetABI(TT, CPU, Options.MCOptions)),
160 Subtarget(nullptr), DefaultSubtarget(TT, CPU, FS, isLittle, *this,
161 Options.StackAlignmentOverride),
162 NoMips16Subtarget(TT, CPU, FS.empty() ? "-mips16" : FS.str() + ",-mips16",
163 isLittle, *this, Options.StackAlignmentOverride),
164 Mips16Subtarget(TT, CPU, FS.empty() ? "+mips16" : FS.str() + ",+mips16",
165 isLittle, *this, Options.StackAlignmentOverride) {
166 Subtarget = &DefaultSubtarget;
167 initAsmInfo();
168
169 // HACK: Update the default CFA register for CHERI purecap
170 ABI.updateCheriInitialFrameStateHack(*AsmInfo, *MRI);
171 if (Subtarget->isCheri()) {
172 assert(DL.getStackAlignment() >= Subtarget->getCapSizeInBytes());
173 }
174}

207const MipsSubtarget *
208MipsTargetMachine::getSubtargetImpl(const Function &F) const {
209 Attribute CPUAttr = F.getFnAttribute("target-cpu");
210 Attribute FSAttr = F.getFnAttribute("target-features");
211
212 std::string CPU = !CPUAttr.hasAttribute(Attribute::None)
213 ? CPUAttr.getValueAsString().str()
214 : TargetCPU;
215 std::string FS = !FSAttr.hasAttribute(Attribute::None)
216 ? FSAttr.getValueAsString().str()
217 : TargetFS;
218 bool hasMips16Attr =
219 !F.getFnAttribute("mips16").hasAttribute(Attribute::None);
220 bool hasNoMips16Attr =
221 !F.getFnAttribute("nomips16").hasAttribute(Attribute::None);
222
223 bool HasMicroMipsAttr =
224 !F.getFnAttribute("micromips").hasAttribute(Attribute::None);
225 bool HasNoMicroMipsAttr =
226 !F.getFnAttribute("nomicromips").hasAttribute(Attribute::None);
227
228 // FIXME: This is related to the code below to reset the target options,
229 // we need to know whether or not the soft float flag is set on the
230 // function, so we can enable it as a subtarget feature.
231 bool softFloat =
232 F.hasFnAttribute("use-soft-float") &&
233 F.getFnAttribute("use-soft-float").getValueAsString() == "true";
234
235 if (hasMips16Attr)
236 FS += FS.empty() ? "+mips16" : ",+mips16";
237 else if (hasNoMips16Attr)
238 FS += FS.empty() ? "-mips16" : ",-mips16";
239 if (HasMicroMipsAttr)
240 FS += FS.empty() ? "+micromips" : ",+micromips";
241 else if (HasNoMicroMipsAttr)
242 FS += FS.empty() ? "-micromips" : ",-micromips";
243 if (softFloat)
244 FS += FS.empty() ? "+soft-float" : ",+soft-float";
245
246 auto &I = SubtargetMap[CPU + FS];
247 if (!I) {
248 // This needs to be done before we create a new subtarget since any
249 // creation will depend on the TM and the code generation flags on the
250 // function that reside in TargetOptions.
251 resetTargetOptions(F);
252 I = llvm::make_unique<MipsSubtarget>(TargetTriple, CPU, FS, isLittle, *this,
253 Options.StackAlignmentOverride);
254 }
255 return I.get();
256}

314void MipsPassConfig::addIRPasses() {
315 TargetPassConfig::addIRPasses();
316 addPass(createAtomicExpandPass());
317 if (getMipsSubtarget().os16())
318 addPass(createMipsOs16Pass());
319 if (getMipsSubtarget().inMips16HardFloat())
320 addPass(createMips16HardFloatPass());
321 if (getMipsSubtarget().isCheri()) {
322 if (getOptLevel() != CodeGenOpt::Level::None) {
323 addPass(createCHERICapFoldIntrinsicsPass());
324 }
325 addPass(createCheriLoopPointerDecanonicalize());
326 addPass(createAggressiveDCEPass());
327 addPass(createCheriRangeChecker());
328 addPass(createCheriPureCapABI());
329 }
330}
331// Install an instruction selector pass using
332// the ISelDag to gen Mips code.
333bool MipsPassConfig::addInstSelector() {
334 addPass(createMipsModuleISelDagPass());
335 addPass(createMips16ISelDag(getMipsTargetMachine(), getOptLevel()));
336 addPass(createMipsSEISelDag(getMipsTargetMachine(), getOptLevel()));
337 return false;
338}

340void MipsPassConfig::addPreRegAlloc() {
341 addPass(createMipsOptimizePICCallPass());
342 if (getMipsSubtarget().isCheri()) {
343 addPass(createCheriAddressingModeFolder());
344 // The CheriAddressingModeFolder can sometimes produce new dead instructions
345 // be sure to clean them up:
346 if (getOptLevel() != CodeGenOpt::Level::None)
347 addPass(&DeadMachineInstructionElimID);
348 addPass(createCheri128FailHardPass());
349 }

364// Implemented by targets that want to run passes immediately before
365// machine code is emitted. return true if -print-machineinstrs should
366// print out the code after the passes.
367void MipsPassConfig::addPreEmitPass() {
368 // Expand pseudo instructions that are sensitive to register allocation.
369 addPass(createMipsExpandPseudoPass());
370
371 // The microMIPS size reduction pass performs instruction reselection for
372 // instructions which can be remapped to a 16 bit instruction.
373 addPass(createMicroMipsSizeReducePass());
374
375 // The delay slot filler pass can potientially create forbidden slot hazards
376 // for MIPSR6 and therefore it should go before MipsBranchExpansion pass.
377 addPass(createMipsDelaySlotFillerPass());
378
379 // This pass expands branches and takes care about the forbidden slot hazards.
380 // Expanding branches may potentially create forbidden slot hazards for
381 // MIPSR6, and fixing such hazard may potentially break a branch by extending
382 // its offset out of range. That's why this pass combine these two tasks, and
383 // runs them alternately until one of them finishes without any changes. Only
384 // then we can be sure that all branches are expanded properly and no hazards
385 // exists.
386 // Any new pass should go before this pass.
387 addPass(createMipsBranchExpansion());
388
389 addPass(createMipsConstantIslandPass());
390}
