123/// Return the first MachineOperand of MI if it is a used virtual register.
124static MachineOperand *getCallTargetRegOpnd(MachineInstr &MI) {
125 if (MI.getNumOperands() == 0)
126 return nullptr;
127
128 MachineOperand &MO = MI.getOperand(0);
129
130 if (!MO.isReg() || !MO.isUse() ||
131 !TargetRegisterInfo::isVirtualRegister(MO.getReg()))
132 return nullptr;
133
134 return &MO;
135}
136
137/// Return type of register Reg.
138static MVT::SimpleValueType getRegTy(unsigned Reg, MachineFunction &MF) {
139 const TargetRegisterInfo &TRI = *MF.getSubtarget().getRegisterInfo();
140 const TargetRegisterClass *RC = MF.getRegInfo().getRegClass(Reg);
141 assert(TRI.legalclasstypes_end(*RC) - TRI.legalclasstypes_begin(*RC) == 1);
142 return *TRI.legalclasstypes_begin(*RC);
143}
144
145/// Do the following transformation:
146///
147/// jalr $vreg
148/// =>
149/// copy $t9, $vreg
150/// jalr $t9
151static void setCallTargetReg(MachineBasicBlock *MBB,
152 MachineBasicBlock::iterator I) {
153 MachineFunction &MF = *MBB->getParent();
154 const TargetInstrInfo &TII = *MF.getSubtarget().getInstrInfo();
155 unsigned SrcReg = I->getOperand(0).getReg();
156 unsigned DstReg = getRegTy(SrcReg, MF) == MVT::i32 ? Mips::T9 : Mips::T9_64;
157 if (MVT(getRegTy(SrcReg, MF)).isFatPointer())
158 DstReg = Mips::C12;
159 BuildMI(*MBB, I, I->getDebugLoc(), TII.get(TargetOpcode::COPY), DstReg)
160 .addReg(SrcReg);
161 I->getOperand(0).setReg(DstReg);
162}
163
164/// Search MI's operands for register GP and erase it.
165static void eraseGPOpnd(MachineInstr &MI) {
166 if (!EraseGPOpnd)
167 return;
168
169 MachineFunction &MF = *MI.getParent()->getParent();
170 MVT::SimpleValueType Ty = getRegTy(MI.getOperand(0).getReg(), MF);
171 unsigned Reg = Ty == MVT::i32 ? Mips::GP : Mips::GP_64;
172
173 for (unsigned I = 0; I < MI.getNumOperands(); ++I) {
174 MachineOperand &MO = MI.getOperand(I);
175 if (MO.isReg() && MO.getReg() == Reg) {
176 MI.RemoveOperand(I);
177 return;
178 }
179 }
180
181 // If compiling with -mx-got, we can fail to remove the GP register.
182}

198// MipsOptimizePICCall methods.
199bool MipsOptimizePICCall::runOnMachineFunction(MachineFunction &F) {
200 if (static_cast<const MipsSubtarget &>(F.getSubtarget()).inMips16Mode())
201 return false;
202
203 // Do a pre-order traversal of the dominator tree.
204 MachineDominatorTree *MDT = &getAnalysis<MachineDominatorTree>();
205 bool Changed = false;
206
207 SmallVector<MBBInfo, 8> WorkList(1, MBBInfo(MDT->getRootNode()));
208
209 while (!WorkList.empty()) {
210 MBBInfo &MBBI = WorkList.back();
211
212 // If this MBB has already been visited, destroy the scope for the MBB and
213 // pop it from the work list.
214 if (MBBI.isVisited()) {
215 MBBI.postVisit();
216 WorkList.pop_back();
217 continue;
218 }
219
220 // Visit the MBB and add its children to the work list.
221 MBBI.preVisit(ScopedHT);
222 Changed |= visitNode(MBBI);
223 const MachineDomTreeNode *Node = MBBI.getNode();
224 const std::vector<MachineDomTreeNode *> &Children = Node->getChildren();
225 WorkList.append(Children.begin(), Children.end());
226 }
227
228 return Changed;
229}
230
231bool MipsOptimizePICCall::visitNode(MBBInfo &MBBI) {
232 bool Changed = false;
233 MachineBasicBlock *MBB = MBBI.getNode()->getBlock();
234
235 for (MachineBasicBlock::iterator I = MBB->begin(), E = MBB->end(); I != E;
236 ++I) {
237 unsigned Reg;
238 ValueType Entry;
239
240 // Skip instructions that are not call instructions via registers.
241 if (!isCallViaRegister(*I, Reg, Entry))
242 continue;
243
244 Changed = true;
245 unsigned N = getCount(Entry);
246
247 if (N != 0) {
248 // If a function has been called more than twice, we do not have to emit a
249 // load instruction to get the function address from the GOT, but can
250 // instead reuse the address that has been loaded before.
251 if (N >= 2 && !LoadTargetFromGOT)
252 getCallTargetRegOpnd(*I)->setReg(getReg(Entry));
253
254 // Erase the $gp operand if this isn't the first time a function has
255 // been called. $gp needs to be set up only if the function call can go
256 // through a lazy binding stub.
257 eraseGPOpnd(*I);
258 }
259
260 if (Entry)
261 incCntAndSetReg(Entry, Reg);
262
263 setCallTargetReg(MBB, I);
264 }
265
266 return Changed;
267}

269bool MipsOptimizePICCall::isCallViaRegister(MachineInstr &MI, unsigned &Reg,
270 ValueType &Val) const {
271 if (!MI.isCall())
272 return false;
273
274 MachineOperand *MO = getCallTargetRegOpnd(MI);
275
276 // Return if MI is not a function call via a register.
277 if (!MO)
278 return false;
279
280 // Get the instruction that loads the function address from the GOT.
281 Reg = MO->getReg();
282 Val = nullptr;
283 MachineRegisterInfo &MRI = MI.getParent()->getParent()->getRegInfo();
284 MachineInstr *DefMI = MRI.getVRegDef(Reg);
285
286 assert(DefMI);
287
288 // See if DefMI is an instruction that loads from a GOT entry that holds the
289 // address of a lazy binding stub.
290 if (!DefMI->mayLoad() || DefMI->getNumOperands() < 3)
291 return true;
292
293 unsigned Flags = DefMI->getOperand(2).getTargetFlags();
294
295 if (Flags != MipsII::MO_GOT_CALL && Flags != MipsII::MO_CALL_LO16)
296 return true;
297
298 // Return the underlying object for the GOT entry in Val.
299 assert(DefMI->hasOneMemOperand());
300 Val = (*DefMI->memoperands_begin())->getValue();
301 if (!Val)
302 Val = (*DefMI->memoperands_begin())->getPseudoValue();
303 return true;
304}

306unsigned MipsOptimizePICCall::getCount(ValueType Entry) {
307 return ScopedHT.lookup(Entry).first;
308}
309
310unsigned MipsOptimizePICCall::getReg(ValueType Entry) {
311 unsigned Reg = ScopedHT.lookup(Entry).second;
312 assert(Reg);
313 return Reg;
314}

316void MipsOptimizePICCall::incCntAndSetReg(ValueType Entry, unsigned Reg) {
317 CntRegP P = ScopedHT.lookup(Entry);
318 ScopedHT.insert(Entry, std::make_pair(P.first + 1, Reg));
319}
