22/// Helper function to get the MIPS subclass of the subtarget
23const MipsSubtarget &getMipsSubtarget(SelectionDAG &DAG) {
24 return reinterpret_cast<const MipsSubtarget&>(DAG.getSubtarget());
25}

27/// Helper function that generates a DAG node for calling a memory function.
28SDValue callFunction(SelectionDAG &DAG, SDLoc dl, SDValue Chain, const char
29 *fnName, SDValue Dst, SDValue Src, SDValue Size) {
30 auto &Ctx = *DAG.getContext();
31 auto &STI = getMipsSubtarget(DAG);
32 TargetLowering::ArgListTy Args;
33 auto pushArg = [&](SDValue &Op) {
34 TargetLowering::ArgListEntry Entry;
35 Entry.Node = Op;
36 Entry.Ty = Op.getValueType().getTypeForEVT(Ctx);
37 Args.push_back(Entry);
38 };
39 pushArg(Dst);
40 pushArg(Src);
41 pushArg(Size);
42
43 SDValue memOpFn = DAG.getExternalFunctionSymbol(fnName);
44
45 TargetLowering::CallLoweringInfo CLI(DAG);
46 CLI.setDebugLoc(dl)
47 .setChain(Chain)
48 .setLibCallee(CallingConv::C,
49 Dst.getValueType().getTypeForEVT(Ctx),
50 memOpFn,
51 std::move(Args))
52 .setDiscardResult();
53
54 const MipsTargetLowering *TLI = STI.getTargetLowering();
55 std::pair<SDValue,SDValue> CallResult = TLI->LowerCallTo(CLI);
56 return CallResult.second;
57}

59/// Helper that emits the memcpy / memmove call, as required.
60SDValue EmitTargetCodeForMemOp(SelectionDAG &DAG, const SDLoc &dl,
61 SDValue Chain, SDValue Dst, SDValue Src,
62 SDValue Size, unsigned Align, bool isVolatile,
63 bool AlwaysInline,
64 bool MustPreserveCheriCapabilities,
65 MachinePointerInfo DstPtrInfo,
66 MachinePointerInfo SrcPtrInfo, bool isMemCpy) {
67 // If AlwaysInline is set, let SelectionDAG expand this.
68 if (AlwaysInline) {
69 return SDValue();
70 }
71 // If we're copying AS0 to AS0, do the normal thing.
72 unsigned DstAS = DstPtrInfo.getAddrSpace();
73 unsigned SrcAS = SrcPtrInfo.getAddrSpace();
74 if ((DstAS == 0) && (SrcAS == 0))
75 return SDValue();
76 auto &STI = getMipsSubtarget(DAG);
77 // If either argument is AS0, make it a capability.
78 MVT CapType = STI.typeForCapabilities();
79 if (DstAS == 0)
80 Dst = DAG.getAddrSpaceCast(dl, CapType, Dst, 0, 200);
81 if (SrcAS == 0)
82 Src = DAG.getAddrSpaceCast(dl, CapType, Src, 0, 200);
83
84
85 const char *memFnName = isMemCpy ?
86 (STI.isABI_CheriPureCap() ? "memcpy" : "memcpy_c") :
87 (STI.isABI_CheriPureCap() ? "memmove" : "memmove_c");
88 return callFunction(DAG, dl, Chain, memFnName, Dst, Src, Size);
89}

115SDValue
116MipsSelectionDAGInfo::EmitTargetCodeForMemset(SelectionDAG &DAG, const SDLoc &dl,
117 SDValue Chain, SDValue Dst,
118 SDValue Src, SDValue Size,
119 unsigned Align, bool isVolatile,
120 MachinePointerInfo DstPtrInfo) const {
121 // If we're setting via an AS0 pointer, do the normal thing.
122 unsigned DstAS = DstPtrInfo.getAddrSpace();
123 if (DstAS == 0)
124 return SDValue();
125
126 auto &STI = getMipsSubtarget(DAG);
127 unsigned CapSize = STI.getCapSizeInBytes();
128 // If this is capability aligned, but not a multiple of capability size, we
129 // might have given up too early trying to emit capability instructions.
130 if (Align >= CapSize) {
131 if (auto ConstantSize = dyn_cast<ConstantSDNode>(Size)) {
132 uint64_t SizeVal = ConstantSize->getZExtValue();
133 // If this size is a small constant, and the value we're writing is zero,
134 // then let's emit some stores instead.
135 if ((SizeVal < (CapSize * 8)))
136 if (isa<ConstantSDNode>(Src) && cast<ConstantSDNode>(Src)->isNullValue()) {
137 MVT CapType = STI.typeForCapabilities();
138 SmallVector<SDValue, 8> OutChains;
139 SDValue ZeroCap = DAG.getConstant(0, dl, CapType);
140 for (uint64_t i=0 ; i<(SizeVal / CapSize) ; i++) {
141 uint64_t DstOff = i*CapSize;
142 SDValue Store = DAG.getStore(Chain, dl, ZeroCap,
143 DAG.getMemBasePlusOffset(Dst, DstOff, dl),
144 DstPtrInfo.getWithOffset(DstOff), Align, isVolatile ?
145 MachineMemOperand::MOVolatile : MachineMemOperand::MONone);
146 OutChains.push_back(Store);
147 }
148 uint64_t Remainder = SizeVal % CapSize;
149 SDValue Zero64 = DAG.getConstant(0, dl, MVT::i64);
150 uint64_t Done = (SizeVal / CapSize) * CapSize;
151 // Write zero, one or two doubles.
152 while (Remainder >= 8) {
153 SDValue Store = DAG.getStore(Chain, dl, Zero64,
154 DAG.getMemBasePlusOffset(Dst, Done, dl),
155 DstPtrInfo.getWithOffset(Done), Align, isVolatile ?
156 MachineMemOperand::MOVolatile : MachineMemOperand::MONone);
157 OutChains.push_back(Store);
158 Done += 8;
159 Remainder -= 8;
160 }
161 // We can always do the remaining 1-7 bytes in at most two
162 // instructions, either two adjacent stores or an unaligned
163 // overlapping store (which will need a separate CIncOffset to
164 // calculate the address, because CHERI offsets are scaled).
165 // We prefer the two-store version, because it reduces dependencies
166 // between instructions.
167 while (Remainder > 0) {
168 SDValue Zero = Zero64;
169 uint64_t DstOff;
170 switch (Remainder) {
171 default: llvm_unreachable("Remainder must be < 8");
172 case 7:
173 Zero = Zero64;
174 DstOff = SizeVal - 8;
175 Remainder -= 7;
176 break;
177 case 6: case 5: case 4:
178 Zero = DAG.getConstant(0, dl, MVT::i32);;
179 DstOff = SizeVal - Remainder;
180 Remainder -= 4;
181 break;
182 case 3: case 2:
183 Zero = DAG.getConstant(0, dl, MVT::i16);;
184 DstOff = SizeVal - Remainder;
185 Remainder -= 2;
186 break;
187 case 1:
188 Zero = DAG.getConstant(0, dl, MVT::i8);;
189 DstOff = SizeVal - Remainder;
190 Remainder -= 1;
191 break;
192 }
193 SDValue Store = DAG.getStore(Chain, dl, Zero,
194 DAG.getMemBasePlusOffset(Dst, DstOff, dl),
195 DstPtrInfo.getWithOffset(DstOff), Align, isVolatile ?
196 MachineMemOperand::MOVolatile : MachineMemOperand::MONone);
197 OutChains.push_back(Store);
198 }
199 return DAG.getNode(ISD::TokenFactor, dl, MVT::Other, OutChains);
200 }
201 }
202 }

205 const char *memFnName =
206 STI.isABI_CheriPureCap() ? "memset" : "memset_c";
207 return callFunction(DAG, dl, Chain, memFnName, Dst, Src, Size);
