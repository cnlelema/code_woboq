45 bool tryToFoldAdd(unsigned Op, unsigned vreg, MachineRegisterInfo &RI,
46 MachineInstr *&AddInst, int64_t &offset) {
47 AddInst = RI.getUniqueVRegDef(vreg);
48 // If we can't uniquely identify the definition, give up.
49 if (AddInst == 0)
50 return false;
51 unsigned reg;
52 int64_t off;
53 // If this is an add-immediate then we can possibly fold it
54 switch (AddInst->getOpcode()) {
55 case Mips::DADDi:
56 case Mips::DADDiu:
57 // Don't try to fold in things that have relocations yet
58 if (!AddInst->getOperand(2).isImm())
59 return false;
60 reg = AddInst->getOperand(1).getReg();
61 off = AddInst->getOperand(2).getImm();
62 break;
63 default:
64 return false;
65 }
66 // We potentially could fold non-zero adds, but we'll leave them for now.
67 if (reg != Mips::ZERO_64)
68 return false;
69 off += offset;
70 // If the result is too big to fit in the offset field, give up
71 if (!IsValidOffset(Op, off))
72 return false;
73 offset = off;
74 return true;
75 }
76 bool IsValidOffset(unsigned Op, int immediate) {
77 switch (Op) {
78 default:
79 llvm_unreachable("Unsupported load operation");
80 case Mips::CAPLOAD8:
81 case Mips::CAPSTORE8:
82 case Mips::CAPSTORE832:
83 case Mips::CAPLOADU8:
84 case Mips::CAPLOADU832:
85 case Mips::CAPLOAD832:
86 return isShiftedInt<8,0>(immediate);
87 case Mips::CAPLOAD16:
88 case Mips::CAPLOADU16:
89 case Mips::CAPSTORE16:
90 case Mips::CAPSTORE1632:
91 case Mips::CAPLOAD1632:
92 case Mips::CAPLOADU1632:
93 return isShiftedInt<8,1>(immediate);
94 case Mips::CAPLOAD32:
95 case Mips::CAPLOADU32:
96 case Mips::CAPSTORE32:
97 case Mips::CAPSTORE3264:
98 case Mips::CAPLOAD3264:
99 return isShiftedInt<8,2>(immediate);
100 case Mips::CAPLOAD64:
101 case Mips::CAPSTORE64:
102 return isShiftedInt<8,3>(immediate);
103 case Mips::LOADCAP:
104 case Mips::STORECAP:
105 return isShiftedInt<11,4>(immediate);
106 case Mips::LOADCAP_BigImm:
107 case Mips::STORECAP_BigImm:
108 return isShiftedInt<16,4>(immediate);
109 }
110 }
111 unsigned MipsOpForCHERIOp(unsigned Op) {
112 switch (Op) {
113 default:
114 llvm_unreachable("No MIPS equivalent");
115 case Mips::CAPLOAD8: return Mips::LB64;
116 case Mips::CAPLOADU8: return Mips::LBu64;
117 case Mips::CAPLOAD16: return Mips::LH64;
118 case Mips::CAPLOADU16: return Mips::LHu64;
119 case Mips::CAPLOAD32: return Mips::LW;
120 case Mips::CAPLOADU32: return Mips::LWu;
121 case Mips::CAPLOAD832: return Mips::LB;
122 case Mips::CAPLOADU832: return Mips::LBu;
123 case Mips::CAPLOAD1632: return Mips::LH;
124 case Mips::CAPLOADU1632: return Mips::LHu;
125 case Mips::CAPLOAD3264: return Mips::LW64;
126 case Mips::CAPLOAD64: return Mips::LD;
127 case Mips::CAPSTORE832: return Mips::SB;
128 case Mips::CAPSTORE8: return Mips::SB64;
129 case Mips::CAPSTORE16: return Mips::SH64;
130 case Mips::CAPSTORE1632: return Mips::SH;
131 case Mips::CAPSTORE32: return Mips::SW;
132 case Mips::CAPSTORE3264: return Mips::SW64;
133 case Mips::CAPSTORE64: return Mips::SD;
134 }
135 }

137 template <typename T>
138 void Remove(T &Instrs, MachineRegisterInfo &RI) {
139 for (MachineInstr *I : Instrs) {
140 // If all uses of operand were fully folded we can delete the CIncOffset,
141 // etc. We can ignore DBG_VALUE calls and should then also
142 if (RI.use_nodbg_empty(I->getOperand(0).getReg())) {
143 // If all uses of this variable have folded into immediate args, then
144 // the dbg values are not particularly useful and can be deleted since
145 // there is no register holding the target address.
146 I->eraseFromParentAndMarkDBGValuesForRemoval();
147 }
148 }

151 bool foldMachineFunction(MachineFunction &MF, MachineLoopInfo &MLI,
152 MachineDominatorTree &MDT) {
153
154 MachineRegisterInfo &RI = MF.getRegInfo();
155 std::set<MachineInstr *> IncOffsets;
156 std::set<MachineInstr *> Adds;
157 llvm::SmallVector<std::pair<MachineInstr *, MachineInstr *>, 8> DDCOps;
158 std::set<MachineInstr *> GetPCCs;
159 bool modified = false;
160 for (auto &MBB : MF)
161 // Iterate backwards to update the last use of the CIncOffsets first
162 // This prevents various machine verifier issues
163 for (auto I = MBB.rbegin(), IE = MBB.rend(); I != IE; ++I) {
164 MachineInstr &MI = *I;
165 int Op = MI.getOpcode();
166 if (Op == Mips::CGetPCC) {
167 GetPCCs.insert(&MI);
168 continue;
169 }
170 // Only look at cap-relative loads and stores
171 if (!(Op == Mips::CAPLOAD8 || Op == Mips::CAPLOADU8 ||
172 Op == Mips::CAPLOAD16 || Op == Mips::CAPLOADU16 ||
173 Op == Mips::CAPLOAD32 || Op == Mips::CAPLOADU32 ||
174 Op == Mips::CAPLOAD832 || Op == Mips::CAPLOADU832 ||
175 Op == Mips::CAPLOAD1632 || Op == Mips::CAPLOADU1632 ||
176 Op == Mips::CAPLOAD3264 || Op == Mips::CAPLOAD64 ||
177 Op == Mips::CAPSTORE8 || Op == Mips::CAPSTORE16 ||
178 Op == Mips::CAPSTORE3264 || Op == Mips::CAPSTORE64 ||
179 Op == Mips::LOADCAP || Op == Mips::STORECAP ||
180 Op == Mips::LOADCAP_BigImm || Op == Mips::STORECAP_BigImm ||
181 Op == Mips::CAPSTORE832 || Op == Mips::CAPSTORE1632 ||
182 Op == Mips::CAPSTORE32))
183 continue;
184
185 // If the operand is a relocation expression then skip this pass:
186 // Example: loading a global: clc $c1, $zero, %captab(global)($c26)
187 if (!MI.getOperand(2).isImm())
188 continue;
189
190 LLVM_DEBUG(dbgs() << "Attempting to fold: "; MI.dump(););
191
192 // If the load is not currently at register-zero offset, we can't fix
193 // it up to use relative addressing, but we may be able to modify it so
194 // that it is...
195 int64_t offset = MI.getOperand(2).getImm();
196 if (MI.getOperand(1).getReg() != Mips::ZERO_64) {
197 // Don't try to fold in things that have relocations yet
198 if (!MI.getOperand(2).isImm())
199 continue;
200 MachineInstr *AddInst;
201 // If the register offset is a simple constant, then try to move it
202 // into the memory operation
203 if (tryToFoldAdd(Op, MI.getOperand(1).getReg(), RI, AddInst, offset)) {
204 Adds.insert(AddInst);
205 MI.getOperand(2).setImm(offset);
206 MI.getOperand(1).setReg(Mips::ZERO_64);
207 } else
208 continue;
209 }
210 // If we get to here, then the memory operation has a capability and
211 // (possibly) an immediate offset, but the register offset is $zero.
212 // If the capability is formed by incrementing an offset, then try to
213 // pull that calculation into the memory operation.
214
215 // If this is a frame index or other symbol, skip it.
216 if (!MI.getOperand(3).isReg())
217 continue;
218 MachineInstr *IncOffset = RI.getUniqueVRegDef(MI.getOperand(3).getReg());
219 if (!IncOffset)
220 continue;
221 // If this is CFromPtr-relative load or store, then we may be able to
222 // fold it into a MIPS load.
223 // XXXAR: this optimization is needed for the legacy ABI but is slightly
224 // unsounds (load of null might not longer always trap) so we disable it
225 // when using the cap table.
226 if (!UseCapTable && IncOffset->getOpcode() == Mips::CFromPtr) {
227 if ((Op == Mips::LOADCAP) || (Op == Mips::STORECAP))
228 continue;
229 // FIXME: this does the wrong thing if the MIPS register is zero!
230 // For now just assume it's fine if it's not the zero register
231 if (IncOffset->getOperand(1).getReg() == Mips::DDC &&
232 IncOffset->getOperand(2).getReg() != Mips::ZERO_64)
233 DDCOps.emplace_back(&MI, IncOffset);
234 continue;
235 }
236
237 // We can't fold $cnull since register 0 refers to DDC in loads and
238 // stores.
239 if ((IncOffset->getOpcode() == Mips::CIncOffset ||
240 IncOffset->getOpcode() == Mips::CIncOffsetImm) &&
241 IncOffset->getOperand(1).isReg() &&
242 IncOffset->getOperand(1).getReg() == Mips::CNULL)
243 continue;
244
245 // If this is a CIncOffset with an immediate then try to fold it into
246 // the operation.
247 if (IncOffset->getOpcode() == Mips::CIncOffsetImm) {
248 uint64_t offsetImm = IncOffset->getOperand(2).getImm();
249 bool ImmediateWasFolded = false;
250 LLVM_DEBUG(dbgs() << "Trying to fold input CIncOffsetImm: "; IncOffset->dump(););
251 if (IsValidOffset(Op, offset + offsetImm)) {
252 if (IncOffset->getOperand(1).isReg()) {
253 IncOffset->getOperand(1).setIsKill(false);
254 auto IncOffsetArg = IncOffset->getOperand(1).getReg();
255 MI.getOperand(3).setReg(IncOffsetArg);
256 if (std::distance(RI.use_begin(IncOffsetArg), RI.use_end()) > 2) {
257 // Don't kill the register if there are other uses
258 MI.getOperand(3).setIsKill(false);
259 }
260 } else {
261 // If it is a frame index we can just replace the register operand with a frame index operand
262 assert(IncOffset->getOperand(1).isFI());
263 MI.getOperand(3).ChangeToFrameIndex(IncOffset->getOperand(1).getIndex());
264 }
265 MI.getOperand(2).setImm(offset + offsetImm);
266 IncOffsets.insert(IncOffset);
267 LLVM_DEBUG(dbgs() << "Was able to fold CIncOffsetImm. New Instr: "; MI.dump(););
268 ImmediateWasFolded = true;
269 modified = true;
270 } else if (IncOffset->getOperand(1).isReg()) {
271 LLVM_DEBUG(dbgs() << "Could not fold input CIncOffsetImm due to immediate range: " << Twine(offset + offsetImm) << "\n";);
272 // Check if there the input to the CIncOffsetImm was a non-constant
273 // inc-offset and attempt to fold that into the register operand
274 MachineInstr *ImmInput =
275 RI.getUniqueVRegDef(IncOffset->getOperand(1).getReg());
276 if (ImmInput && ImmInput->getOpcode() == Mips::CIncOffset) {
277 LLVM_DEBUG(dbgs()<< "However, the input Reg for CIncOffsetImm is "
278 "CIncOffset. Trying to fold that: "; ImmInput->dump(););
279 auto MemopRegOff = MI.getOperand(1).getReg();
280 // Note: The first operand to CIncOffset could also be a frame Index in which case we
281 // cannot fold it.
282 assert(ImmInput->getOperand(1).isReg() || ImmInput->getOperand(1).isFI());
283 if (ImmInput->getOperand(1).isReg() && (MemopRegOff == Mips::ZERO_64 || MemopRegOff == Mips::ZERO)) {
284 auto IncRegOff = ImmInput->getOperand(2).getReg();
285 MI.getOperand(1).setReg(IncRegOff);
286 if (std::distance(RI.use_begin(IncRegOff), RI.use_end()) > 2) {
287 // Don't kill the register if there are other uses
288 MI.getOperand(1).setIsKill(false);
289 } else {
290 MI.getOperand(1).setIsKill(true);
291 }
292 assert(IncOffset->getOperand(1).getReg() == ImmInput->getOperand(0).getReg());
293 IncOffset->getOperand(1).setReg(ImmInput->getOperand(1).getReg());
294 if (std::distance(RI.use_begin(IncOffset->getOperand(1).getReg()), RI.use_end()) > 2) {
295 // Don't kill the IncOffsetImm input register if there are other uses
296 IncOffset->getOperand(1).setIsKill(false);
297 } else {
298 IncOffset->getOperand(1).setIsKill(true);
299 }
300 IncOffsets.insert(ImmInput);
301 modified = true;
302 LLVM_DEBUG(dbgs() << "Folded register offset from CIncOffsetImm"
303 " input into the memop. New CIncOffsetImm: ";
304 IncOffset->dump(); dbgs() << "New memop: "; MI.dump(););
305 }
306 }
307 }
308
309 continue;
310 }
311
312 // Ignore ones that are not based on a CIncOffset op
313 if (IncOffset->getOpcode() != Mips::CIncOffset)
314 continue;
315 assert(IncOffset);
316
317 MachineOperand& Cap = IncOffset->getOperand(1);
318 // Cannot handle frame indexes (yet?):
319 if (Cap.isFI())
320 continue;
321 MachineOperand& Offset = IncOffset->getOperand(2);
322 assert(Cap.isReg());
323 // If the IncOffset is in a different basic block we need to be more
324 // careful: The machine verifier currently complains because we
325 // kill vregs here that we use later
326 if (&MBB != IncOffset->getParent())
327 continue;
328
329 // We are going to use the CIncOffset's source capability at the
330 // load/store instruction, so first we need to check it has not been
331 // killed before the use
332 // We also need to do the same for the offset value (but we can ignore
333 // it if the only kill is the CIncOffset instruction)
334 auto* TRI = RI.getTargetRegisterInfo();
335 bool CapKilled = false;
336 bool OffsetKilled = false;
337 // Keep looking backwards towards the incoffset instruction to see if
338 // there is a hazard
339 for (auto J = std::next(I),
340 JE = MachineBasicBlock::reverse_iterator(IncOffset);
341 J != JE; ++J) {
342 if (J->modifiesRegister(Cap.getReg(), TRI) ||
343 J->killsRegister(Cap.getReg(), TRI)) {
344 CapKilled = true;
345 LLVM_DEBUG(dbgs() << "Cannot fold: CAP IS KILLED BY "; J->dump(););
346 break;
347 }
348 if (J->modifiesRegister(Offset.getReg(), TRI) ||
349 J->killsRegister(Offset.getReg(), TRI)) {
350 OffsetKilled = true;
351 LLVM_DEBUG(dbgs() << "Cannot fold: OFFSET IS KILLED BY "; J->dump(););
352 break;
353 }
354 }
355 if (CapKilled || OffsetKilled)
356 continue;
357
358 MachineInstr *AddInst;
359 // If the CIncOffset is of a daddi[u] then we can potentially replace
360 // both by just folding the register and immediate offsets into the
361 // load / store. Don't try to fold in things that have relocations yet.
362 if (MI.getOperand(2).isImm()) {
363 int64_t offset = MI.getOperand(2).getImm();
364 if (tryToFoldAdd(Op, Offset.getReg(), RI, AddInst, offset)) {
365 // If we managed to pull the offset calculation entirely away, then
366 // just use the computed immediate
367 Adds.insert(AddInst);
368 MI.getOperand(1).setReg(Mips::ZERO_64);
369 MI.getOperand(2).setImm(offset);
370 } else {
371 // If we didn't, then use the CIncOffset's register value as our
372 // offset
373 MI.getOperand(1).setReg(Offset.getReg());
374 // transfer the kill state to the capability load instruction
375 MI.getOperand(1).setIsKill(Offset.isKill());
376 Offset.setIsKill(false);
377 }
378 } else {
379 // If it has a relocation, then use the CIncOffset's register value
380 // as our offset
381 MI.getOperand(1).setReg(Offset.getReg());
382 // transfer the kill state to the capability load instruction
383 MI.getOperand(1).setIsKill(Offset.isKill());
384 Offset.setIsKill(false);
385 }
386 // If we were using a unique vreg here it will probably have kill set to
387 // true, so if we retain this kill state the machine instr verifier will
388 // complain if there is any other instruction later on that uses the
389 // cap reg.
390 // We can kill the capreg if the only uses are the CIncOffset
391 bool CanKillCap = RI.hasOneUse(Cap.getReg());
392 if (auto *CapRegDef = RI.getUniqueVRegDef(Cap.getReg())) {
393 // But we can't kill it if it is defined in a different BB since we
394 // might jump somewhere else that requires that vreg
395 if (CapRegDef->getParent() != MI.getParent()) {
396 LLVM_DEBUG(dbgs() << "capreg defined in different BB -> not "
397 "killing operand 3: ";
398 MI.dump(););
399 CanKillCap = false;
400 }
401 }
402 MI.getOperand(3).setReg(Cap.getReg());
403 MI.getOperand(3).setIsKill(CanKillCap);
404 Cap.setIsKill(false);
405
406 IncOffsets.insert(IncOffset);
407 modified = true;
408 }
409
410 assert((!UseCapTable || DDCOps.empty()) &&
411 "This optimization is sometimes wrong -> should "
412 "skip (at least for captable)!");
413 for (auto &I : DDCOps) {
414 IncOffsets.insert(I.second);
415 MachineOperand *BaseOperand = &I.second->getOperand(2);
416 // This can't be a symbolic address (yet) because we don't have
417 // relocations that fit.
418 MachineOperand &Offset = I.first->getOperand(2);
419 // If this was the result of a daddiu, fold the immediate into the result
420 // as well.
421 MachineInstr *AddInst = RI.getUniqueVRegDef(BaseOperand->getReg());
422 if (AddInst && (AddInst->getOpcode() == Mips::DADDiu)) {
423 MachineOperand &MO = AddInst->getOperand(2);
424 // FIXME: We could probably fold the add into the load in some cases
425 // even when it's not initially zero, as our immediate field has just
426 // grown, but that's probably not the common case.
427 if (Offset.isImm() && (Offset.getImm() == 0)) {
428 Offset = MO;
429 BaseOperand = &AddInst->getOperand(1);
430 }
431 } else
432 AddInst = nullptr;
433 MachineBasicBlock *InsertBlock = I.first->getParent();
434 auto FirstOperand = I.first->getOperand(0);
435 unsigned FirstReg = FirstOperand.getReg();
436 BuildMI(*InsertBlock, I.first, I.first->getDebugLoc(),
437 InstrInfo->get(MipsOpForCHERIOp(I.first->getOpcode())))
438 .addReg(FirstReg, getDefRegState(FirstOperand.isDef()))
439 .addReg(BaseOperand->getReg()).add(Offset);
440 BaseOperand->setIsKill(false); // we used this register -> can't kill it
441 I.first->eraseFromBundle();
442 if (AddInst) {
443 // If we've folded the base of the add into the load's immediate, then
444 // we the add is no longer able to kill the register.
445 AddInst->getOperand(1).setIsKill(false);
446 Adds.insert(AddInst);
447 }
448 modified = true;
449 }
450 Remove(GetPCCs, RI);
451 Remove(IncOffsets, RI);
452 Remove(Adds, RI);
453 return modified;
454 }
455 bool runOnMachineFunction(MachineFunction &MF) override {
456 if (DisableAddressingModeFolder)
457 return false;
458 InstrInfo = MF.getSubtarget<MipsSubtarget>().getInstrInfo();
459 bool modified = false;
460 MachineLoopInfo &MLI = getAnalysis<MachineLoopInfo>();
461 MachineDominatorTree &MDT = getAnalysis<MachineDominatorTree>();
462 // Try iterating as long as we are making changs but add a cut-off at 1000
463 // to find inifite loops
464 unsigned NumIterations = 0;
465 while (foldMachineFunction(MF, MLI, MDT)) {
466 modified = true;
467 NumIterations++;
468 assert(NumIterations < 1000 && "Infinite loop in CheriAddrModeFolder?");
469 }
470 return modified;
471 }
