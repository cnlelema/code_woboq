396 /// Returns true if a cast between SrcAS and DestAS is a noop.
397 bool isNoopAddrSpaceCast(unsigned SrcAS, unsigned DestAS) const override {
398 // Mips doesn't have any special address spaces so we just reserve
399 // the first 256 for software use (e.g. OpenCL) and treat casts
400 // between them as noops.
401 // TODO: can we get the datalayout somehow?
402 const bool SrcIsCheri = isCheriPointer(SrcAS, nullptr);
403 const bool DestIsCheri = isCheriPointer(DestAS, nullptr);
404 if ((SrcIsCheri || DestIsCheri) && (SrcIsCheri != DestIsCheri))
405 return false;
406 return SrcAS < 256 && DestAS < 256;
407 }

426 // This method creates the following nodes, which are necessary for
427 // computing a local symbol's address:
428 //
429 // (add (load (wrapper $gp, %got(sym)), %lo(sym))
430 template <class NodeTy>
431 SDValue getAddrLocal(NodeTy *N, const SDLoc &DL, EVT Ty, SelectionDAG &DAG,
432 bool IsN32OrN64, bool IsForTls) const {
433 assert(!ABI.UsesCapabilityTable());
434 assert(!Ty.isFatPointer());
435 unsigned GOTFlag = IsN32OrN64 ? MipsII::MO_GOT_PAGE : MipsII::MO_GOT;
436 SDValue GOT = DAG.getNode(MipsISD::Wrapper, DL, Ty, getGlobalReg(DAG, Ty, IsForTls),
437 getTargetNode(N, Ty, DAG, GOTFlag));
438 SDValue Load =
439 DAG.getLoad(Ty, DL, DAG.getEntryNode(), GOT,
440 MachinePointerInfo::getGOT(DAG.getMachineFunction()));
441 unsigned LoFlag = IsN32OrN64 ? MipsII::MO_GOT_OFST : MipsII::MO_ABS_LO;
442 SDValue Lo = DAG.getNode(MipsISD::Lo, DL, Ty,
443 getTargetNode(N, Ty, DAG, LoFlag));
444 return DAG.getNode(ISD::ADD, DL, Ty, Load, Lo);
445 }

447 // This method creates the following nodes, which are necessary for
448 // computing a global symbol's address:
449 //
450 // (load (wrapper $gp, %got(sym)))
451 template <class NodeTy>
452 SDValue getAddrGlobal(NodeTy *N, const SDLoc &DL, EVT Ty, SelectionDAG &DAG,
453 unsigned Flag, SDValue Chain,
454 const MachinePointerInfo &PtrInfo,
455 bool IsForTls) const {
456 assert(!ABI.UsesCapabilityTable());
457 assert(!Ty.isFatPointer());
458 SDValue Tgt = DAG.getNode(MipsISD::Wrapper, DL, Ty, getGlobalReg(DAG, Ty, IsForTls),
459 getTargetNode(N, Ty, DAG, Flag));
460 if (ABI.IsCheriPureCap())
461 Tgt = cFromDDC(DAG, DL, Tgt);
462 return DAG.getLoad(Ty, DL, Chain, Tgt, PtrInfo);
463 }

465 // This method creates the following nodes, which are necessary for
466 // computing a global symbol's address in large-GOT mode:
467 //
468 // (load (wrapper (add %hi(sym), $gp), %lo(sym)))
469 template <class NodeTy>
470 SDValue getAddrGlobalLargeGOT(NodeTy *N, const SDLoc &DL, EVT Ty,
471 SelectionDAG &DAG, unsigned HiFlag,
472 unsigned LoFlag, SDValue Chain,
473 const MachinePointerInfo &PtrInfo,
474 bool IsForTls) const {
475 assert(!ABI.UsesCapabilityTable());
476 assert(!Ty.isFatPointer());
477 SDValue Hi = DAG.getNode(MipsISD::GotHi, DL, Ty,
478 getTargetNode(N, Ty, DAG, HiFlag));
479 Hi = DAG.getNode(ISD::ADD, DL, Ty, Hi, getGlobalReg(DAG, Ty, IsForTls));
480 SDValue Wrapper = DAG.getNode(MipsISD::Wrapper, DL, Ty, Hi,
481 getTargetNode(N, Ty, DAG, LoFlag));
482 return DAG.getLoad(Ty, DL, Chain, Wrapper, PtrInfo);
483 }

485 // This method creates the following nodes, which are necessary for
486 // computing a symbol's address in non-PIC mode:
487 //
488 // (add %hi(sym), %lo(sym))
489 //
490 // This method covers O32, N32 and N64 in sym32 mode.
491 template <class NodeTy>
492 SDValue getAddrNonPIC(NodeTy *N, const SDLoc &DL, EVT Ty,
493 SelectionDAG &DAG) const {
494 assert(!ABI.UsesCapabilityTable());
495 assert(!Ty.isFatPointer());
496 SDValue Hi = getTargetNode(N, Ty, DAG, MipsII::MO_ABS_HI);
497 SDValue Lo = getTargetNode(N, Ty, DAG, MipsII::MO_ABS_LO);
498 return DAG.getNode(ISD::ADD, DL, Ty,
499 DAG.getNode(MipsISD::Hi, DL, Ty, Hi),
500 DAG.getNode(MipsISD::Lo, DL, Ty, Lo));
501 }
502
503 // This method creates the following nodes, which are necessary for
504 // computing a symbol's address in non-PIC mode for N64.
505 //
506 // (add (shl (add (shl (add %highest(sym), %higher(sim)), 16), %high(sym)),
507 // 16), %lo(%sym))
508 //
509 // FIXME: This method is not efficent for (micro)MIPS64R6.
510 template <class NodeTy>
511 SDValue getAddrNonPICSym64(NodeTy *N, const SDLoc &DL, EVT Ty,
512 SelectionDAG &DAG) const {
513 assert(!ABI.UsesCapabilityTable());
514 assert(!Ty.isFatPointer());
515 SDValue Hi = getTargetNode(N, Ty, DAG, MipsII::MO_ABS_HI);
516 SDValue Lo = getTargetNode(N, Ty, DAG, MipsII::MO_ABS_LO);
517
518 SDValue Highest =
519 DAG.getNode(MipsISD::Highest, DL, Ty,
520 getTargetNode(N, Ty, DAG, MipsII::MO_HIGHEST));
521 SDValue Higher = getTargetNode(N, Ty, DAG, MipsII::MO_HIGHER);
522 SDValue HigherPart =
523 DAG.getNode(ISD::ADD, DL, Ty, Highest,
524 DAG.getNode(MipsISD::Higher, DL, Ty, Higher));
525 SDValue Cst = DAG.getConstant(16, DL, MVT::i32);
526 SDValue Shift = DAG.getNode(ISD::SHL, DL, Ty, HigherPart, Cst);
527 SDValue Add = DAG.getNode(ISD::ADD, DL, Ty, Shift,
528 DAG.getNode(MipsISD::Hi, DL, Ty, Hi));
529 SDValue Shift2 = DAG.getNode(ISD::SHL, DL, Ty, Add, Cst);
530
531 return DAG.getNode(ISD::ADD, DL, Ty, Shift2,
532 DAG.getNode(MipsISD::Lo, DL, Ty, Lo));
533 }

570 template <class NodeTy>

572 SelectionDAG &DAG, SDValue Chain,

598 // This method creates the following nodes, which are necessary for
599 // computing a symbol's capability:
600 //
601 // (load (wrappercapop $cgp, %captab20(sym)))
602 template <class NodeTy>
603 SDValue
604 _getGlobalCapSmallImmediate(NodeTy *N, const SDLoc &DL, EVT Ty,
605 SelectionDAG &DAG, unsigned Flag, SDValue Chain,
606 const MachinePointerInfo *PtrInfo,
607 bool DoLoad) const {
608 assert(Ty.isFatPointer());
609 SDValue Off = getTargetNode(N, MVT::i64, DAG, Flag);
610 SDValue Tgt = DAG.getNode(MipsISD::WrapperCapOp, DL, CapType,
611 getCapGlobalReg(DAG, CapType), Off);
612 // Why can't I use the target node here directly?
613 // SDNode *Addr = DAG.getMachineNode(Mips::CapGlobalAddrPseudo, DL, Ty, getCapGlobalReg(DAG, CapType), Off);
614 if (DoLoad)
615 return DAG.getLoad(Ty, DL, Chain, Tgt, *PtrInfo);
616 else
617 return Tgt;
618 }

620 // This method creates the following nodes, which are necessary for
621 // computing a global symbol's address in large-GOT mode:
622 //
623 // (load (ptradd $cgp, (wrapper %captab_hi(sym), %captab_lo(sym))))
624 template <class NodeTy>
625 SDValue _getGlobalCapBigImmediate(NodeTy *N, const SDLoc &DL, EVT Ty,
626 SelectionDAG &DAG, unsigned HiFlag,
627 unsigned LoFlag, SDValue Chain,
628 const MachinePointerInfo *PtrInfo,
629 bool DoLoad, unsigned HiOpc) const {
630 SDValue Off = DAG.getNode(HiOpc, DL, MVT::i64,
631 getTargetNode(N, MVT::i64, DAG, HiFlag));
632 Off = DAG.getNode(MipsISD::Wrapper, DL, MVT::i64, Off,
633 getTargetNode(N, MVT::i64, DAG, LoFlag));
634 SDValue Tgt = DAG.getNode(ISD::PTRADD, DL, CapType,
635 getCapGlobalReg(DAG, CapType), Off);
636 if (DoLoad)
637 return DAG.getLoad(Ty, DL, Chain, Tgt, *PtrInfo);
638 else
639 return Tgt;
640 }
