58const TargetRegisterClass *
59MipsRegisterInfo::getPointerRegClass(const MachineFunction &MF,
60 unsigned Kind) const {
61 MipsABIInfo ABI = MF.getSubtarget<MipsSubtarget>().getABI();
62 MipsPtrClass PtrClassKind = static_cast<MipsPtrClass>(Kind);
63
64 switch (PtrClassKind) {
65 case MipsPtrClass::Default:
66 return ABI.ArePtrs64bit() ? &Mips::GPR64RegClass : &Mips::GPR32RegClass;
67 case MipsPtrClass::GPR16MM:
68 return &Mips::GPRMM16RegClass;
69 case MipsPtrClass::StackPointer:
70 return ABI.ArePtrs64bit() ? &Mips::SP64RegClass : &Mips::SP32RegClass;
71 case MipsPtrClass::GlobalPointer:
72 return ABI.ArePtrs64bit() ? &Mips::GP64RegClass : &Mips::GP32RegClass;
73 }
74
75 llvm_unreachable("Unknown pointer kind");
76}

78unsigned
79MipsRegisterInfo::getRegPressureLimit(const TargetRegisterClass *RC,
80 MachineFunction &MF) const {
81 switch (RC->getID()) {
82 default:
83 return 0;
84 case Mips::GPR32RegClassID:
85 case Mips::GPR64RegClassID:
86 case Mips::DSPRRegClassID: {
87 const TargetFrameLowering *TFI = MF.getSubtarget().getFrameLowering();
88 return 28 - TFI->hasFP(MF);
89 }
90 case Mips::FGR32RegClassID:
91 return 32;
92 case Mips::AFGR64RegClassID:
93 return 16;
94 case Mips::FGR64RegClassID:
95 return 32;
96 }
97}

103/// Mips Callee Saved Registers
104const MCPhysReg *
105MipsRegisterInfo::getCalleeSavedRegs(const MachineFunction *MF) const {
106 const MipsSubtarget &Subtarget = MF->getSubtarget<MipsSubtarget>();
107 const Function &F = MF->getFunction();
108 if (F.hasFnAttribute("interrupt")) {
109 if (Subtarget.hasMips64())
110 return Subtarget.hasMips64r6() ? CSR_Interrupt_64R6_SaveList
111 : CSR_Interrupt_64_SaveList;
112 else
113 return Subtarget.hasMips32r6() ? CSR_Interrupt_32R6_SaveList
114 : CSR_Interrupt_32_SaveList;
115 }
116
117 if (Subtarget.isSingleFloat())
118 return CSR_SingleFloatOnly_SaveList;
119
120 if (Subtarget.isABI_CheriPureCap())
121 return CSR_Cheri_Purecap_SaveList;
122
123 if (Subtarget.isCheri())
124 return CSR_N64_Cheri_SaveList;
125
126 if (Subtarget.isABI_N64())
127 return CSR_N64_SaveList;
128
129 if (Subtarget.isABI_N32())
130 return CSR_N32_SaveList;
131
132 if (Subtarget.isFP64bit())
133 return CSR_O32_FP64_SaveList;
134
135 if (Subtarget.isFPXX())
136 return CSR_O32_FPXX_SaveList;
137
138 return CSR_O32_SaveList;
139}

173BitVector MipsRegisterInfo::
174getReservedRegs(const MachineFunction &MF) const {
175 static const MCPhysReg ReservedGPR32[] = {
176 Mips::ZERO, Mips::K0, Mips::K1
177 };
178
179 static const MCPhysReg ReservedGPR64[] = {
180 Mips::ZERO_64, Mips::K0_64, Mips::K1_64
181 };
182
183 static const uint16_t ReservedCheriRegs[] = {
184 Mips::CNULL,
185 Mips::DDC,
186 Mips::C25,
187 Mips::C26,
188 Mips::C27,
189 Mips::C28,
190 Mips::C29,
191 Mips::C30,
192 Mips::C31,
193 // Mark the capability HWRegs as reserved to avoid machine verifier errors
194 Mips::CAPHWR0,
195 Mips::CAPHWR1,
196 Mips::CAPHWR8,
197 Mips::CAPHWR22,
198 Mips::CAPHWR23,
199 Mips::CAPHWR29,
200 Mips::CAPHWR30,
201 Mips::CAPHWR31,
202 };
203
204 // C1-C7, C11, C12, C17-C23 allowed
205 static const uint16_t ReservedCheri16Regs[] = {
206 Mips::C8, Mips::C9, Mips::C10, Mips::C13, Mips::C14, Mips::C15, Mips::C16,
207 Mips::C24
208 };
209
210 // C1-C3, C11, C12, C17-C19 allowed
211 static const uint16_t ReservedCheri8Regs[] = {
212 Mips::C4, Mips::C5, Mips::C6, Mips::C7, Mips::C8, Mips::C9, Mips::C10,
213 Mips::C13, Mips::C14, Mips::C15, Mips::C16, Mips::C20, Mips::C21,
214 Mips::C22, Mips::C23, Mips::C24
215 };
216
217 BitVector Reserved(getNumRegs());
218 const MipsSubtarget &Subtarget = MF.getSubtarget<MipsSubtarget>();
219
220 for (unsigned I = 0; I < array_lengthof(ReservedGPR32); ++I)
221 Reserved.set(ReservedGPR32[I]);
222
223 // Reserve registers for the NaCl sandbox.
224 if (Subtarget.isTargetNaCl()) {
225 Reserved.set(Mips::T6); // Reserved for control flow mask.
226 Reserved.set(Mips::T7); // Reserved for memory access mask.
227 Reserved.set(Mips::T8); // Reserved for thread pointer.
228 }
229
230 for (unsigned I = 0; I < array_lengthof(ReservedGPR64); ++I)
231 Reserved.set(ReservedGPR64[I]);
232
233 // In the CHERI pure-capability ABI, $sp is just another temporary register.
234 if (!Subtarget.isABI_CheriPureCap()) {
235 Reserved.set(Mips::SP);
236 Reserved.set(Mips::SP_64);
237 }
238
239 // For mno-abicalls, GP is a program invariant!
240 if (!Subtarget.isABICalls()) {
241 Reserved.set(Mips::GP);
242 Reserved.set(Mips::GP_64);
243 }
244
245 if (Subtarget.isFP64bit()) {
246 // Reserve all registers in AFGR64.
247 for (MCPhysReg Reg : Mips::AFGR64RegClass)
248 Reserved.set(Reg);
249 } else {
250 // Reserve all registers in FGR64.
251 for (MCPhysReg Reg : Mips::FGR64RegClass)
252 Reserved.set(Reg);
253 }
254
255 if (Subtarget.isCheri()) {
256 for (unsigned I = 0; I < array_lengthof(ReservedCheriRegs); ++I)
257 Reserved.set(ReservedCheriRegs[I]);
258 auto &ABI = Subtarget.getABI();
259 auto *FL =
260 static_cast<const MipsFrameLowering*>(Subtarget.getFrameLowering());
261 if (Subtarget.isABI_CheriPureCap()) {
262 Reserved.set(ABI.GetStackPtr());
263 if (FL->hasFP(MF))
264 Reserved.set(ABI.GetFramePtr());
265 if (FL->hasBP(MF))
266 Reserved.set(ABI.GetBasePtr());
267 if (Subtarget.useCheriCapTable())
268 Reserved.set(ABI.GetGlobalCapability());
269
270 }
271 if (Cheri8)
272 for (unsigned I = 0; I < array_lengthof(ReservedCheri8Regs); ++I)
273 Reserved.set(ReservedCheri8Regs[I]);
274 if (Cheri16)
275 for (unsigned I = 0; I < array_lengthof(ReservedCheri16Regs); ++I)
276 Reserved.set(ReservedCheri16Regs[I]);
277 } else
278 Reserved.set(Mips::DDC);
279
280 // Reserve FP if this function should have a dedicated frame pointer register.
281 if (Subtarget.getFrameLowering()->hasFP(MF)) {
282 if (Subtarget.inMips16Mode())
283 Reserved.set(Mips::S0);
284 else {
285 Reserved.set(Mips::FP);
286 Reserved.set(Mips::FP_64);
287
288 // Reserve the base register if we need to both realign the stack and
289 // allocate variable-sized objects at runtime. This should test the
290 // same conditions as MipsFrameLowering::hasBP().
291 if (needsStackRealignment(MF) &&
292 MF.getFrameInfo().hasVarSizedObjects()) {
293 Reserved.set(Mips::S7);
294 Reserved.set(Mips::S7_64);
295 }
296 }
297 }
298
299 // Reserve hardware registers.
300 Reserved.set(Mips::HWR29);
301
302 // Reserve DSP control register.
303 Reserved.set(Mips::DSPPos);
304 Reserved.set(Mips::DSPSCount);
305 Reserved.set(Mips::DSPCarry);
306 Reserved.set(Mips::DSPEFI);
307 Reserved.set(Mips::DSPOutFlag);
308
309 // Reserve MSA control registers.
310 for (MCPhysReg Reg : Mips::MSACtrlRegClass)
311 Reserved.set(Reg);
312
313 // Reserve RA if in mips16 mode.
314 if (Subtarget.inMips16Mode()) {
315 const MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
316 Reserved.set(Mips::RA);
317 Reserved.set(Mips::RA_64);
318 Reserved.set(Mips::T0);
319 Reserved.set(Mips::T1);
320 if (MF.getFunction().hasFnAttribute("saveS2") || MipsFI->hasSaveS2())
321 Reserved.set(Mips::S2);
322 }
323
324 // Reserve GP if small section is used.
325 if (Subtarget.useSmallSection()) {
326 Reserved.set(Mips::GP);
327 Reserved.set(Mips::GP_64);
328 }
329
330 return Reserved;
331}

343// FrameIndex represent objects inside a abstract stack.
344// We must replace FrameIndex with an stack/frame pointer
345// direct reference.
346void MipsRegisterInfo::
347eliminateFrameIndex(MachineBasicBlock::iterator II, int SPAdj,
348 unsigned FIOperandNum, RegScavenger *RS) const {
349 MachineInstr &MI = *II;
350 MachineFunction &MF = *MI.getParent()->getParent();
351
352 LLVM_DEBUG(errs() << "\nFunction : " << MF.getName() << "\n";
353 errs() << "<--------->\n"
354 << MI);
355
356 int FrameIndex = MI.getOperand(FIOperandNum).getIndex();
357 uint64_t stackSize = MF.getFrameInfo().getStackSize();
358 int64_t spOffset = MF.getFrameInfo().getObjectOffset(FrameIndex);
359
360 LLVM_DEBUG(errs() << "FrameIndex : " << FrameIndex << "\n"
361 << "spOffset : " << spOffset << "\n"
362 << "stackSize : " << stackSize << "\n"
363 << "alignment : "
364 << MF.getFrameInfo().getObjectAlignment(FrameIndex)
365 << "\n");
366
367 eliminateFI(MI, FIOperandNum, FrameIndex, stackSize, spOffset, RS);
368}

370unsigned MipsRegisterInfo::
371getFrameRegister(const MachineFunction &MF) const {
372 const MipsSubtarget &Subtarget = MF.getSubtarget<MipsSubtarget>();
373 const TargetFrameLowering *TFI = Subtarget.getFrameLowering();
374 auto &ABI = static_cast<const MipsTargetMachine &>(MF.getTarget()).getABI();
375
376 if (Subtarget.inMips16Mode())
377 return TFI->hasFP(MF) ? Mips::S0 : Mips::SP;
378 else
379 return TFI->hasFP(MF) ? ABI.GetFramePtr() : ABI.GetStackPtr();
380}

386bool MipsRegisterInfo::canRealignStack(const MachineFunction &MF) const {
387 // Avoid realigning functions that explicitly do not want to be realigned.
388 // Normally, we should report an error when a function should be dynamically
389 // realigned but also has the attribute no-realign-stack. Unfortunately,
390 // with this attribute, MachineFrameInfo clamps each new object's alignment
391 // to that of the stack's alignment as specified by the ABI. As a result,
392 // the information of whether we have objects with larger alignment
393 // requirement than the stack's alignment is already lost at this point.
394 if (!TargetRegisterInfo::canRealignStack(MF))
395 return false;
396
397 const MipsSubtarget &Subtarget = MF.getSubtarget<MipsSubtarget>();
398 unsigned FP = Subtarget.isGP32bit() ? Mips::FP : Mips::FP_64;
399 unsigned BP = Subtarget.isGP32bit() ? Mips::S7 : Mips::S7_64;
400
401 // Support dynamic stack realignment for all targets except Mips16.
402 if (Subtarget.inMips16Mode())
403 return false;
404
405 // We can't perform dynamic stack realignment if we can't reserve the
406 // frame pointer register.
407 if (!MF.getRegInfo().canReserveReg(FP))
408 return false;
409
410 // We can realign the stack if we know the maximum call frame size and we
411 // don't have variable sized objects.
412 if (Subtarget.getFrameLowering()->hasReservedCallFrame(MF))
413 return true;
414
415 // We have to reserve the base pointer register in the presence of variable
416 // sized objects.
417 return MF.getRegInfo().canReserveReg(BP);
418}
