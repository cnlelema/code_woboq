55const TargetRegisterClass *
56MipsSERegisterInfo::intRegClass(unsigned Size) const {
57 if (Size == 4)
58 return &Mips::GPR32RegClass;
59
60 assert(Size == 8);
61 return &Mips::GPR64RegClass;
62}

64/// Get the size of the offset supported by the given load/store/inline asm.
65/// The result includes the effects of any scale factors applied to the
66/// instruction immediate.
67static inline unsigned getLoadStoreOffsetSizeInBits(const unsigned Opcode,
68 MachineOperand MO) {
69 switch (Opcode) {
70 case Mips::CAPSTORE8:
71 case Mips::CAPSTORE832:
72 case Mips::CAPLOAD8:
73 case Mips::CAPLOAD832:
74 case Mips::CAPLOADU8:
75 case Mips::CAPLOADU832:
76 return 8;
77 case Mips::CAPSTORE16:
78 case Mips::CAPSTORE1632:
79 case Mips::CAPLOAD16:
80 case Mips::CAPLOAD1632:
81 case Mips::CAPLOADU16:
82 case Mips::CAPLOADU1632:
83 return 8 + 1 /* scale factor */;
84 case Mips::CAPSTORE32:
85 case Mips::CAPSTORE3264:
86 case Mips::CAPLOAD32:
87 case Mips::CAPLOAD3264:
88 case Mips::CAPLOADU32:
89 return 8 + 2 /* scale factor */;
90 case Mips::CAPSTORE64:
91 case Mips::CAPLOAD64:
92 return 8 + 3 /* scale factor */;
93 case Mips::LOADCAP:
94 case Mips::STORECAP:
95 return 11 + 4 /* scale factor */;
96 case Mips::LOADCAP_BigImm:
97 case Mips::STORECAP_BigImm:
98 return 16 + 4 /* scale factor */;
99 case Mips::CIncOffsetImm:
100 return 11;
101 case Mips::CheriBoundedStackPseudo:
102 return 1; // only 0 is valid, otherwise we need a CIncOffset
103 case Mips::LD_B:
104 case Mips::ST_B:
105 return 10;
106 case Mips::LD_H:
107 case Mips::ST_H:
108 return 10 + 1 /* scale factor */;
109 case Mips::LD_W:
110 case Mips::ST_W:
111 return 10 + 2 /* scale factor */;
112 case Mips::LD_D:
113 case Mips::ST_D:
114 return 10 + 3 /* scale factor */;
115 case Mips::LL:
116 case Mips::LL64:
117 case Mips::LLD:
118 case Mips::LLE:
119 case Mips::SC:
120 case Mips::SC64:
121 case Mips::SCD:
122 case Mips::SCE:
123 return 16;
124 case Mips::LLE_MM:
125 case Mips::LL_MM:
126 case Mips::SCE_MM:
127 case Mips::SC_MM:
128 return 12;
129 case Mips::LL64_R6:
130 case Mips::LL_R6:
131 case Mips::LLD_R6:
132 case Mips::SC64_R6:
133 case Mips::SCD_R6:
134 case Mips::SC_R6:
135 case Mips::LL_MMR6:
136 case Mips::SC_MMR6:
137 return 9;
138 case Mips::INLINEASM: {
139 unsigned ConstraintID = InlineAsm::getMemoryConstraintID(MO.getImm());
140 switch (ConstraintID) {
141 case InlineAsm::Constraint_ZC: {
142 const MipsSubtarget &Subtarget = MO.getParent()
143 ->getParent()
144 ->getParent()
145 ->getSubtarget<MipsSubtarget>();
146 if (Subtarget.inMicroMipsMode())
147 return 12;
148
149 if (Subtarget.hasMips32r6())
150 return 9;
151
152 return 16;
153 }
154 default:
155 return 16;
156 }
157 }
158 default:
159 return 16;
160 }
161}

163/// Get the scale factor applied to the immediate in the given load/store.
164static inline unsigned getLoadStoreOffsetAlign(const unsigned Opcode) {
165 switch (Opcode) {
166 case Mips::CheriBoundedStackPseudo:
167 return 64; // Ensure that only 0 is valid (since max witdth is 1)
168 case Mips::CAPSTORE16:
169 case Mips::CAPLOAD16:
170 case Mips::CAPLOAD1632:
171 case Mips::CAPLOADU1632:
172 case Mips::CAPLOADU16:
173 return 2;
174 case Mips::CAPSTORE32:
175 case Mips::CAPLOAD32:
176 case Mips::CAPLOADU32:
177 return 4;
178 case Mips::CAPSTORE64:
179 case Mips::CAPLOAD64:
180 return 8;
181 case Mips::LOADCAP:
182 case Mips::LOADCAP_BigImm:
183 case Mips::STORECAP:
184 case Mips::STORECAP_BigImm:
185 return 16;
186 case Mips::LD_H:
187 case Mips::ST_H:
188 return 2;
189 case Mips::LD_W:
190 case Mips::ST_W:
191 return 4;
192 case Mips::LD_D:
193 case Mips::ST_D:
194 return 8;
195 default:
196 return 1;
197 }
198}

200void MipsSERegisterInfo::eliminateFI(MachineBasicBlock::iterator II,
201 unsigned OpNo, int FrameIndex,
202 uint64_t StackSize,
203 int64_t SPOffset,
204 RegScavenger *RS) const {
205 MachineInstr &MI = *II;
206 MachineFunction &MF = *MI.getParent()->getParent();
207 MachineFrameInfo &MFI = MF.getFrameInfo();
208 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
209
210 auto &TM = static_cast<const MipsTargetMachine &>(MF.getTarget());
211 MipsABIInfo ABI = TM.getABI();
212 const MipsRegisterInfo *RegInfo =
213 static_cast<const MipsRegisterInfo *>(MF.getSubtarget().getRegisterInfo());
214
215 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
216 int MinCSFI = 0;
217 int MaxCSFI = -1;
218
219 if (CSI.size()) {
220 MinCSFI = CSI[0].getFrameIdx();
221 MaxCSFI = CSI[CSI.size() - 1].getFrameIdx();
222 }
223
224 bool EhDataRegFI = MipsFI->isEhDataRegFI(FrameIndex);
225 bool IsISRRegFI = MipsFI->isISRRegFI(FrameIndex);
226 // The following stack frame objects are always referenced relative to $sp:
227 // 1. Outgoing arguments.
228 // 2. Pointer to dynamically allocated stack space.
229 // 3. Locations for callee-saved registers.
230 // 4. Locations for eh data registers.
231 // 5. Locations for ISR saved Coprocessor 0 registers 12 & 14.
232 // Everything else is referenced relative to whatever register
233 // getFrameRegister() returns.
234 unsigned FrameReg;
235
236 if ((FrameIndex >= MinCSFI && FrameIndex <= MaxCSFI) || EhDataRegFI ||
237 IsISRRegFI)
238 FrameReg = ABI.GetStackPtr();
239 else if (RegInfo->needsStackRealignment(MF)) {
240 if (MFI.hasVarSizedObjects() && !MFI.isFixedObjectIndex(FrameIndex))
241 FrameReg = ABI.GetBasePtr();
242 else if (MFI.isFixedObjectIndex(FrameIndex))
243 FrameReg = getFrameRegister(MF);
244 else
245 FrameReg = ABI.GetStackPtr();
246 } else
247 FrameReg = getFrameRegister(MF);
248
249 // Calculate final offset.
250 // - There is no need to change the offset if the frame object is one of the
251 // following: an outgoing argument, pointer to a dynamically allocated
252 // stack space or a $gp restore location,
253 // - If the frame object is any of the following, its offset must be adjusted
254 // by adding the size of the stack:
255 // incoming argument, callee-saved register location or local variable.
256 bool IsKill = false;
257 int64_t Offset;
258 int RegOpNo = OpNo;
259 int ImmOpNo = OpNo + 1;
260 if (ABI.IsCheriPureCap()) {
261 RegOpNo = OpNo;
262 switch (MI.getOpcode()) {
263 case Mips::CAPSTORE64:
264 case Mips::CAPSTORE32:
265 case Mips::CAPSTORE3264:
266 case Mips::CAPSTORE16:
267 case Mips::CAPSTORE1632:
268 case Mips::CAPSTORE8:
269 case Mips::CAPSTORE832:
270 case Mips::STORECAP:
271 case Mips::LOADCAP:
272 case Mips::CAPLOAD64:
273 case Mips::CAPLOAD32:
274 case Mips::CAPLOAD3264:
275 case Mips::CAPLOADU32:
276 case Mips::CAPLOAD16:
277 case Mips::CAPLOAD1632:
278 case Mips::CAPLOADU16:
279 case Mips::CAPLOADU1632:
280 case Mips::CAPLOAD8:
281 case Mips::CAPLOAD832:
282 case Mips::CAPLOADU8:
283 case Mips::CAPLOADU832:
284 ImmOpNo = 2;
285 RegOpNo = 3;
286 break;
287 case Mips::CIncOffsetImm:
288 ImmOpNo = 2;
289 RegOpNo = 1;
290 break;
291 case Mips::CheriBoundedStackPseudo:
292 ImmOpNo = 2;
293 RegOpNo = 1;
294 break;
295 case Mips::CIncOffset:
296 break;
297 case Mips::INLINEASM:
298 break;
299 default:
300 llvm_unreachable("Unsupported instruction in eliminateFI!");
301 }
302 }
303
304
305 Offset = SPOffset + (int64_t)StackSize;
306 if (MI.getOperand(ImmOpNo).isImm())
307 Offset += MI.getOperand(ImmOpNo).getImm();
308
309 // For CHERI spills, we store the offset as an immediate operand from the
310 // FrameIndex where the register operand should go and then replace it with
311 // the real register here.
312 if (MI.getOperand(RegOpNo).isImm())
313 Offset += MI.getOperand(RegOpNo).getImm();
314
315 LLVM_DEBUG(errs() << "Offset : " << Offset << "\n"
316 << "<--------->\n");
317
318 if (!MI.isDebugValue()) {
319 // Make sure Offset fits within the field available.
320 // For MSA instructions, this is a 10-bit signed immediate (scaled by
321 // element size), otherwise it is a 16-bit signed immediate.
322 unsigned OffsetBitSize =
323 getLoadStoreOffsetSizeInBits(MI.getOpcode(), MI.getOperand(OpNo - 1));
324 unsigned OffsetAlign = getLoadStoreOffsetAlign(MI.getOpcode());
325
326 auto *STI = TM.getSubtargetImpl(MF.getFunction());
327 const MipsSEInstrInfo &TII = *static_cast<const MipsSEInstrInfo *>(
328 STI->getInstrInfo());
329 DebugLoc DL = II->getDebugLoc();
330
331 // Convert CIncOffset <-> CIncOffsetImm depending on offset value
332 if (MI.getOpcode() == Mips::CIncOffset || MI.getOpcode() == Mips::CIncOffsetImm) {
333 MI.getOperand(1).ChangeToRegister(FrameReg, false);
334 // If this is an 11-bit offset, then replace the CIncOffset that takes a
335 // register with one that takes an immediate.
336 if (isInt<11>(Offset)) {
337 if (MI.getOpcode() == Mips::CIncOffset)
338 MI.setDesc(TII.get(Mips::CIncOffsetImm));
339 MI.getOperand(2).ChangeToImmediate(Offset);
340 } else {
341 if (MI.getOpcode() == Mips::CIncOffsetImm)
342 MI.setDesc(TII.get(Mips::CIncOffset));
343 MachineBasicBlock &MBB = *MI.getParent();
344 unsigned Reg = TII.loadImmediate(Offset, MBB, II, DL, nullptr);
345 MI.getOperand(2).ChangeToRegister(Reg, false, false, true);
346 }
347 return;
348 }
349
350 if (ABI.IsCheriPureCap()) {
351 if (!isIntN(OffsetBitSize, Offset) || OffsetToAlignment(Offset, OffsetAlign) != 0) {
352 MachineBasicBlock &MBB = *MI.getParent();
353 // If we have an offset that needs to fit into a signed n-bit immediate
354 // (where n < 16) and doesn't, but does fit into 16-bits then use an ADDiu
355 bool isFrameRegLoad = MI.getOperand(0).getReg() == FrameReg && MI.getOpcode() != Mips::STORECAP;
356 // bounded stack caps always need inc-offset (except for offset 0)
357 bool needsIncOffset = MI.isPseudo() || MI.getOperand(1).getReg() != Mips::ZERO_64;
358 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
359#if 0
360 // This was the original code which avoids allocating a new capreg.
361 const TargetRegisterClass *PtrRC =
362 ABI.ArePtrs64bit() ? &Mips::GPR64RegClass : &Mips::GPR32RegClass;
363 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
364 unsigned Reg = TII.loadImmediate(Offset, MBB, II, DL, nullptr);
365 // TODO: use CIncOffsetImmediate here!
366 if (needsIncOffset) {
367 BuildMI(MBB, II, DL, TII.get(Mips::CIncOffset), FrameReg)
368 .addReg(FrameReg)
369 .addReg(Reg, isFrameReg ? RegState::Kill : 0);
370 if (!isFrameReg) {
371 unsigned NegReg = RegInfo.createVirtualRegister(PtrRC);
372 BuildMI(MBB, (++II), DL, TII.get(Mips::DSUBu), NegReg)
373 .addReg(Mips::ZERO_64)
374 .addReg(Reg, RegState::Kill);
375 BuildMI(MBB, II, DL, TII.get(Mips::CIncOffset), FrameReg)
376 .addReg(FrameReg)
377 .addReg(NegReg, RegState::Kill);
378 }
379#endif
380 unsigned TmpCap = -1;
381 if (isFrameRegLoad || MI.getOpcode() == Mips::CheriBoundedStackPseudo) {
382 assert(MI.getOpcode() == Mips::LOADCAP || MI.isPseudo());
383 TmpCap = MI.getOperand(0).getReg();
384 } else {
385 TmpCap = RegInfo.createVirtualRegister(&Mips::CheriGPRRegClass);
386 }
387 // TODO: use CIncOffsetImmediate here!
388 if (needsIncOffset) {
389 if (isInt<11>(Offset)) {
390 BuildMI(MBB, II, DL, TII.get(Mips::CIncOffsetImm), TmpCap)
391 .addReg(FrameReg, isFrameRegLoad ? RegState::Kill : 0)
392 .addImm(Offset);
393 } else {
394 unsigned Reg = TII.loadImmediate(Offset, MBB, II, DL, nullptr);
395 BuildMI(MBB, II, DL, TII.get(Mips::CIncOffset), TmpCap)
396 .addReg(FrameReg, isFrameRegLoad ? RegState::Kill : 0)
397 .addReg(Reg, RegState::Kill);
398 }
399 FrameReg = TmpCap;
400 IsKill = true;
401 } else {
402 unsigned Reg = TII.loadImmediate(Offset, MBB, II, DL, nullptr);
403 MI.getOperand(1).ChangeToRegister(Reg, false, false, true);
404 }
405 Offset = 0;
406 }
407 } else if (OffsetBitSize < 16 && isInt<16>(Offset) &&
408 (!isIntN(OffsetBitSize, Offset) ||
409 OffsetToAlignment(Offset, OffsetAlign) != 0)) {
410 // If we have an offset that needs to fit into a signed n-bit immediate
411 // (where n < 16) and doesn't, but does fit into 16-bits then use an ADDiu
412 MachineBasicBlock &MBB = *MI.getParent();
413 const TargetRegisterClass *PtrRC =
414 ABI.ArePtrs64bit() ? &Mips::GPR64RegClass : &Mips::GPR32RegClass;
415 MachineRegisterInfo &RegInfo = MBB.getParent()->getRegInfo();
416 unsigned Reg = RegInfo.createVirtualRegister(PtrRC);
417 BuildMI(MBB, II, DL, TII.get(ABI.GetPtrAddiuOp()), Reg)
418 .addReg(FrameReg)
419 .addImm(Offset);
420
421 FrameReg = Reg;
422 Offset = 0;
423 IsKill = true;
424 } else if (!isInt<16>(Offset)) {
425 // Otherwise split the offset into 16-bit pieces and add it in multiple
426 // instructions.
427 MachineBasicBlock &MBB = *MI.getParent();
428 DebugLoc DL = II->getDebugLoc();
429 unsigned NewImm = 0;
430 unsigned Reg = TII.loadImmediate(Offset, MBB, II, DL,
431 OffsetBitSize == 16 ? &NewImm : nullptr);
432 BuildMI(MBB, II, DL, TII.get(ABI.GetPtrAdduOp()), Reg).addReg(FrameReg)
433 .addReg(Reg, RegState::Kill);
434
435 FrameReg = Reg;
436 Offset = SignExtend64<16>(NewImm);
437 IsKill = true;
438 }
439 }
440
441 MI.getOperand(RegOpNo).ChangeToRegister(FrameReg, false, false, IsKill);
442 MI.getOperand(ImmOpNo).ChangeToImmediate(Offset);
443}
