21namespace {
22class CheriInvalidatePass : public MachineFunctionPass {
23 const MipsInstrInfo *InstrInfo = nullptr;
24 SmallVector<MachineInstr *, 16> StackStores;
25 SmallVector<MachineInstr *, 4> Returns;
26
27public:
28 static char ID;
29 CheriInvalidatePass() : MachineFunctionPass(ID) {}
30 StringRef getPassName() const override { return "CHERI invalidate pass"; }
31
32 void runOnMachineBasicBlock(MachineBasicBlock &MBB) {
33 if (!InstrInfo)
34 InstrInfo = MBB.getParent()->getSubtarget<MipsSubtarget>().getInstrInfo();
35 for (MachineBasicBlock::iterator I = MBB.instr_begin();
36 I != MBB.instr_end(); ++I) {
37 int FI;
38 MachineInstr &Inst = *I;
39 if (InstrInfo->isStoreToStackSlot(Inst, FI)) {
40 StackStores.push_back(&Inst);
41 } else if (I->isReturn()) {
42 Returns.push_back(&Inst);
43 }
44 }
45 }
46
47 bool runOnMachineFunction(MachineFunction &F) override{
48 if (!InstrInfo)
49 InstrInfo = F.getSubtarget<MipsSubtarget>().getInstrInfo();
50
51// Metadata nodes are no longer allowed to refer to functions, so we need
52// another mechanism for identifying them. We should do it properly by adding
53// a function attribute.
54#if 0
55 const Function *IRFunction = F.getFunction();
56 const Module *Mod = IRFunction->getParent();
57 NamedMDNode *SensitiveFunctions =
58 Mod->getNamedMetadata("cheri.sensitive.functions");
59 if (!SensitiveFunctions) return false;
60 bool foundFunction = false;
61 for (unsigned i=0 ; i<SensitiveFunctions->getNumOperands() ; i++) {
62 Value *SensitiveFunction =
63 cast<MDNode>(SensitiveFunctions->getOperand(i))->getOperand(0);
64 if (SensitiveFunction == IRFunction) {
65 foundFunction = true;
66 break;
67 }
68 }
69 if (!foundFunction) return false;
70
71
72 LLVM_DEBUG(dbgs() << "Zeroing stack spills\n");
73 StackStores.clear();
74 Returns.clear();
75 SmallSet<int, 16> ZeroedLocations;
76 for (MachineFunction::iterator FI = F.begin(), FE = F.end();
77 FI != FE; ++FI)
78 runOnMachineBasicBlock(*FI);
79
80 if (StackStores.size() == 0) return false;
81
82 for (SmallVector<MachineInstr*, 4>::iterator i=Returns.begin(),
83 e=Returns.end() ; i!=e ; ++i) {
84 MachineInstr *Ret = *i;
85 ZeroedLocations.clear();
86 for (SmallVector<MachineInstr*, 16>::iterator si=StackStores.begin(),
87 se=StackStores.end() ; si!=se ; ++si) {
88 MachineInstr *Store = *si;
89 unsigned Opc = Store->getOpcode();
90 MachineBasicBlock &MBB = *Ret->getParent();
91 int FI = Store->getOperand(1).getIndex();
92 // If we've already zeroed this location, skip it.
93 if (ZeroedLocations.count(FI)) continue;
94 ZeroedLocations.insert(FI);
95 // If this is a capability store, then we just do a 64-bit integer
96 // write. This leaks information, but invalidates the capability.
97 if (Opc == Mips::STORECAP) {
98 MachineInstrBuilder MIB(F, Ret);
99 BuildMI(MBB, Ret, Ret->getDebugLoc(), InstrInfo->get(Mips::SD))
100 .addReg(Mips::ZERO)
101 .addFrameIndex(FI).addImm(Store->getOperand(2).getImm())
102 .addMemOperand(InstrInfo->GetMemOperand(MBB, FI, MachineMemOperand::MOStore));
103 LLVM_DEBUG(dbgs() << "Zeroing capability spill\n");
104 } else {
105 // For other stores, we do the same type of store as was used for the spill, now with zeros.
106 BuildMI(MBB, Ret, Ret->getDebugLoc(), InstrInfo->get(Opc))
107 .addReg(Mips::ZERO)
108 .addFrameIndex(FI).addImm(Store->getOperand(2).getImm())
109 .addMemOperand(InstrInfo->GetMemOperand(MBB, FI, MachineMemOperand::MOStore));
110 LLVM_DEBUG(dbgs() << "Zeroing non-capability spill\n");
111 }
112 }
113 }
114
115 return true;
116#else
117 return false;
118#endif
119 }
120};
121}
