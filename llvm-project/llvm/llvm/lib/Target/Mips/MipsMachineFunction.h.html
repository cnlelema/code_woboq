13#ifndef LLVM_LIB_TARGET_MIPS_MIPSMACHINEFUNCTION_H
14#define LLVM_LIB_TARGET_MIPS_MIPSMACHINEFUNCTION_H
15
16#include "Mips16HardFloatInfo.h"
17#include "llvm/CodeGen/MachineFunction.h"
18#include "llvm/CodeGen/MachineMemOperand.h"
19#include <map>
20
21namespace llvm {
22
23/// MipsFunctionInfo - This class is derived from MachineFunction private
24/// Mips target-specific information for each MachineFunction.
25class MipsFunctionInfo : public MachineFunctionInfo {
26public:
27 MipsFunctionInfo(MachineFunction &MF) : MF(MF) {}
28
29 ~MipsFunctionInfo() override;
30
31 unsigned getSRetReturnReg() const { return SRetReturnReg; }
32 void setSRetReturnReg(unsigned Reg) { SRetReturnReg = Reg; }
33
34 bool globalBaseRegSet() const;
35 unsigned getGlobalBaseReg(bool IsForTls);
36 unsigned getGlobalBaseRegUnchecked() const;
37
38 bool capGlobalBaseRegSet() const;
39 unsigned getCapGlobalBaseReg();
40 unsigned getCapGlobalBaseRegForGlobalISel();
41 unsigned getGlobalBaseRegForGlobalISel();
42
43 // Insert instructions to initialize the global base register in the
44 // first MBB of the function.
45 void initGlobalBaseReg();
46 // Insert instructions to initialize the capability global base register in
47 // the first MBB of the function.
48 void initCapGlobalBaseReg();
49
50 int getVarArgsFrameIndex() const { return VarArgsFrameIndex; }
51 void setVarArgsFrameIndex(int Index) { VarArgsFrameIndex = Index; }
52
53 bool hasByvalArg() const { return HasByvalArg; }
54 void setFormalArgInfo(unsigned Size, bool HasByval) {
55 IncomingArgSize = Size;
56 HasByvalArg = HasByval;
57 }
58
59 unsigned getIncomingArgSize() const { return IncomingArgSize; }
60
61 bool callsEhReturn() const { return CallsEhReturn; }
62 void setCallsEhReturn() { CallsEhReturn = true; }
63
64 void createEhDataRegsFI();
65 int getEhDataRegFI(unsigned Reg) const { return EhDataRegFI[Reg]; }
66 bool isEhDataRegFI(int FI) const;
67
68 /// Create a MachinePointerInfo that has an ExternalSymbolPseudoSourceValue
69 /// object representing a GOT entry for an external function.
70 MachinePointerInfo callPtrInfo(const char *ES);
71
72 // Functions with the "interrupt" attribute require special prologues,
73 // epilogues and additional spill slots.
74 bool isISR() const { return IsISR; }
75 void setISR() { IsISR = true; }
76 void createISRRegFI();
77 int getISRRegFI(unsigned Reg) const { return ISRDataRegFI[Reg]; }
78 bool isISRRegFI(int FI) const;
79
80 /// Create a MachinePointerInfo that has a GlobalValuePseudoSourceValue object
81 /// representing a GOT entry for a global function.
82 MachinePointerInfo callPtrInfo(const GlobalValue *GV);
83
84 void setSaveS2() { SaveS2 = true; }
85 bool hasSaveS2() const { return SaveS2; }
86
87 bool usesTlsViaGlobalReg() const { return UsesTlsViaGlobalReg; }
88
89 int getMoveF64ViaSpillFI(const TargetRegisterClass *RC);
90
91 std::map<const char *, const Mips16HardFloatInfo::FuncSignature *>
92 StubsNeeded;
93
94private:
95 virtual void anchor();
96
97 MachineFunction& MF;
98
99 /// SRetReturnReg - Some subtargets require that sret lowering includes
100 /// returning the value of the returned struct in a register. This field
101 /// holds the virtual register into which the sret argument is passed.
102 unsigned SRetReturnReg = 0;
103
104 /// GlobalBaseReg - keeps track of the virtual register initialized for
105 /// use as the global base register. This is used for PIC in some PIC
106 /// relocation models.
107 unsigned GlobalBaseReg = 0;
108
109 /// CapGlobalBaseReg - keeps track of the virtual register initialized for
110 /// use as the capability global base register. This is used for all global
111 /// accesses in the purecap ABI.
112 unsigned CapGlobalBaseReg = 0;
113
114 /// VarArgsFrameIndex - FrameIndex for start of varargs area.
115 int VarArgsFrameIndex = 0;
116
117 /// True if function has a byval argument.
118 bool HasByvalArg;
119
120 /// Size of incoming argument area.
121 unsigned IncomingArgSize;
122
123 /// CallsEhReturn - Whether the function calls llvm.eh.return.
124 bool CallsEhReturn = false;
125
126 /// Frame objects for spilling eh data registers.
127 int EhDataRegFI[4];
128
129 /// ISR - Whether the function is an Interrupt Service Routine.
130 bool IsISR = false;
131
132 /// Frame objects for spilling C0_STATUS, C0_EPC
133 int ISRDataRegFI[2];
134
135 // saveS2
136 bool SaveS2 = false;
137
138 /// FrameIndex for expanding BuildPairF64 nodes to spill and reload when the
139 /// O32 FPXX ABI is enabled. -1 is used to denote invalid index.
140 int MoveF64ViaSpillFI = -1;
141
142 bool UsesTlsViaGlobalReg = false;
143};
144
145} // end namespace llvm
146
147#endif // LLVM_LIB_TARGET_MIPS_MIPSMACHINEFUNCTION_H
