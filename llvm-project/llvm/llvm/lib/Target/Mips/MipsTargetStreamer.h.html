149 /// Emit a store instruction with an offset. If the offset is out of range
150 /// then it will be synthesized using the assembler temporary.
151 ///
152 /// GetATReg() is a callback that can be used to obtain the current assembler
153 /// temporary and is only called when the assembler temporary is required. It
154 /// must handle the case where no assembler temporary is available (typically
155 /// by reporting an error).
156 void emitStoreWithImmOffset(unsigned Opcode, unsigned SrcReg,
157 unsigned BaseReg, int64_t Offset,
158 function_ref<unsigned()> GetATReg, SMLoc IDLoc,
159 const MCSubtargetInfo *STI);

161 unsigned BaseReg, MCOperand &HiOperand,
162 MCOperand &LoOperand, unsigned ATReg, SMLoc IDLoc,
163 const MCSubtargetInfo *STI);

165 int64_t Offset, unsigned TmpReg, SMLoc IDLoc,

173 void forbidModuleDirective() { ModuleDirectiveAllowed = false; }

177 // This method enables template classes to set internal abi flags
178 // structure values.
179 template <class PredicateLibrary>
180 void updateABIInfo(const PredicateLibrary &P) {
181 ABI = P.getABI();
182 ABIFlagsSection.setAllFromPredicates(P, getABI());
183 }
184
185 MipsABIFlagsSection &getABIFlagsSection() { return ABIFlagsSection; }
186 const MipsABIInfo &getABI() const {
187 assert(ABI.hasValue() && "ABI hasn't been set!");
188 return *ABI;
189 }

191 /// CHERI128 uses compressed capabilities. If we would like to guarantee
192 /// non-overlapping bounds for all global symbols we must over-align the
193 /// symbol if the size is no precisely representable. We also add padding at
194 /// the end to ensure that we cannot access another variable that happens to
195 /// be located in the bytes that are accessible after the end of the object
196 /// due to the bounds having been rounded up.
197 TailPaddingAmount getTailPaddingForPreciseBounds(unsigned Size) override;
198 unsigned getAlignmentForPreciseBounds(unsigned Size) override;
