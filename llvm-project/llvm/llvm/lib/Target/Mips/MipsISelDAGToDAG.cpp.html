50 // There are multiple MipsDAGToDAGISel instances added to the pass pipeline.
51 // We need to preserve StackProtector for the next one.
52 AU.addPreserved<StackProtector>();
53 SelectionDAGISel::getAnalysisUsage(AU);
54}

56bool MipsDAGToDAGISel::runOnMachineFunction(MachineFunction &MF) {
57 Subtarget = &static_cast<const MipsSubtarget &>(MF.getSubtarget());
58 bool Ret = SelectionDAGISel::runOnMachineFunction(MF);
59
60 processFunctionAfterISel(MF);
61
62 return Ret;
63}

65/// getGlobalBaseReg - Output the instructions required to put the
66/// GOT address into a register.
67SDNode *MipsDAGToDAGISel::getGlobalBaseReg(bool IsForTls) {
68 unsigned GlobalBaseReg = MF->getInfo<MipsFunctionInfo>()->getGlobalBaseReg(IsForTls);
69 // XXXAR: address space 0 is correct here since it will never be used
70 // for capabilities
71 return CurDAG
72 ->getRegister(GlobalBaseReg, getTargetLowering()->getPointerTy(
73 CurDAG->getDataLayout(), 0))
74 .getNode();
75}

228/// Select instructions not customized! Used for
229/// expanded, promoted and normal instructions
230void MipsDAGToDAGISel::Select(SDNode *Node) {
231 unsigned Opcode = Node->getOpcode();
232
233 // If we have a custom node, we already have selected!
234 if (Node->isMachineOpcode()) {
235 LLVM_DEBUG(errs() << "== "; Node->dump(CurDAG); errs() << "\n");
236 Node->setNodeId(-1);
237 return;
238 }
239
240 // See if subclasses can handle this node.
241 if (trySelect(Node))
242 return;
243
244 switch(Opcode) {
245 default: break;
246
247 // Get target GOT address.
248 case ISD::GLOBAL_OFFSET_TABLE:
249 ReplaceNode(Node, getGlobalBaseReg(/*IsForTls=*/false));
250 return;
251
252#ifndef NDEBUG
253 case ISD::LOAD:
254 case ISD::STORE:
255 assert((Subtarget->systemSupportsUnalignedAccess(cast<MemSDNode>(Node)->getAddressSpace()) ||
256 (cast<MemSDNode>(Node)->getMemoryVT().getSizeInBits() / 8) <=
257 cast<MemSDNode>(Node)->getAlignment()) &&
258 "Unexpected unaligned loads/stores.");
259 break;
260#endif
261 }
262
263 // Select the default instruction
264 SelectCode(Node);
265}

267bool MipsDAGToDAGISel::
268SelectInlineAsmMemoryOperand(const SDValue &Op, unsigned ConstraintID,
269 std::vector<SDValue> &OutOps) {
270 // All memory constraints can at least accept raw pointers.
271 switch(ConstraintID) {
272 default:
273 llvm_unreachable("Unexpected asm memory constraint");
274 case InlineAsm::Constraint_i:
275 case InlineAsm::Constraint_m:
276 case InlineAsm::Constraint_R:
277 case InlineAsm::Constraint_ZC:
278 OutOps.push_back(Op);
279 return false;
280 }
281 return true;
282}
