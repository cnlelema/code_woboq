34extern "C" void LLVMInitializeMipsTargetInfo() {
35 RegisterTarget<Triple::mips,
36 /*HasJIT=*/true>
37 X(getTheMipsTarget(), "mips", "MIPS (32-bit big endian)", "Mips");
38
39 RegisterTarget<Triple::mipsel,
40 /*HasJIT=*/true>
41 Y(getTheMipselTarget(), "mipsel", "MIPS (32-bit little endian)", "Mips");
42
43 RegisterTarget<Triple::mips64,
44 /*HasJIT=*/true>
45 A(getTheMips64Target(), "mips64", "MIPS (64-bit big endian)", "Mips");
46
47 RegisterTarget<Triple::mips64el,
48 /*HasJIT=*/true>
49 B(getTheMips64elTarget(), "mips64el", "MIPS (64-bit little endian)",
50 "Mips");
51
52 RegisterTarget<Triple::cheri,
53 /*HasJIT=*/true>
54 C(getTheMipsCheriTarget(), "cheri", "CHERI", "Mips");
55}
