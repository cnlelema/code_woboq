12static cl::opt<bool> DisableLoopDecanonicalize(
13 "disable-cheri-loop-decanonicalization", cl::init(false),
14 cl::desc("Don't attempt to optimize pointer operations in loops"), cl::Hidden);
15
16
17namespace {
18class CheriLoopPointerDecanonicalize : public FunctionPass {
19 void getAnalysisUsage(AnalysisUsage &AU) const override {
20 AU.setPreservesCFG();
21 AU.addRequired<ScalarEvolutionWrapperPass>();
22 AU.addRequired<LoopInfoWrapperPass>();
23 AU.addRequired<DominatorTreeWrapperPass>();
24 AU.addPreserved<LoopInfoWrapperPass>();
25 FunctionPass::getAnalysisUsage(AU);
26 }
27
28 bool runOnFunction(Function &F) override {
29 if (DisableLoopDecanonicalize)
30 return false;
31 bool Modified = false;
32 LoopInfo &LI = getAnalysis<LoopInfoWrapperPass>().getLoopInfo();
33 auto &DT = getAnalysis<DominatorTreeWrapperPass>().getDomTree();
34 ScalarEvolution &SE = getAnalysis<ScalarEvolutionWrapperPass>().getSE();
35 DenseMap<const SCEV*,Value*> SCEVs;
36 SmallVector<std::tuple<GetElementPtrInst&, BasicBlock*, BasicBlock*,
37 BasicBlock*, Value*, Value*>, 8> Replacements;
38 for (auto &BB : F)
39 for (auto &I : BB) {
40 if (SE.isSCEVable(I.getType()))
41 SCEVs[SE.getSCEV(&I)] = &I;
42 }
43 for (auto &BB : F) {
44 Loop *L = LI.getLoopFor(&BB);
45 // Skip basic blocks that are not inside a loop
46 if (!L)
47 continue;
48 // If the loop doesn't have a preheader, give up because we'll want to
49 // insert there.
50 auto *Preheader = L->getLoopPreheader();
51 if (!Preheader)
52 continue;
53 // Skip loops that have more than a single back edge for now.
54 if (L->getNumBackEdges() != 1)
55 continue;
56 auto *Header = L->getHeader();
57 auto PI = pred_begin(Header), PE = pred_end(Header);
58 while ((PI != PE) && (*PI == Preheader))
59 ++PI;
60 assert(PI != PE);
61 auto BackBB = *PI;
62
63
64 for (auto &I : BB) {
65 // Skip instructions that are not GEPs
66 if (!isa<GetElementPtrInst>(I))
67 continue;
68 auto &GEP = cast<GetElementPtrInst>(I);
69 // Skip GEPs where we can't trivially replace the induction variable
70 // with the original.
71 if (GEP.getType() != GEP.getOperand(0)->getType())
72 continue;
73 // If the GEP base does not dominate the end of the preheader then we
74 // can't hoist the start out.
75 if (auto *I = dyn_cast<Instruction>(GEP.getOperand(0)))
76 if (!DT.dominates(I, Preheader->getTerminator()))
77 continue;
78 // Skip more complex GEPs for now - we mostly care about simple array
79 // iteration anyway.
80 if (GEP.getNumOperands() > 2)
81 continue;
82 // Skip non-capability pointers
83 if (GEP.getPointerAddressSpace() != 200)
84 continue;
85 if (!DT.dominates(&GEP, BackBB))
86 continue;
87 auto *Idx = GEP.getOperand(1);
88 auto *S = SE.getSCEV(Idx);
89 if (isa<SCEVCastExpr>(S))
90 S = cast<SCEVCastExpr>(S)->getOperand();
91 if (auto *SA = dyn_cast<SCEVAddRecExpr>(S)) {
92 auto *Start = SA->getStart();
93 if (!Start)
94 continue;
95 auto *StartValue = SCEVs[Start];
96 if (!StartValue)
97 continue;
98 if (auto *I = dyn_cast<Instruction>(StartValue))
99 if (!DT.dominates(I, Preheader->getTerminator()))
100 continue;
101 auto Rec = SA->getStepRecurrence(SE);
102 if (auto *Step = dyn_cast<SCEVUnknown>(Rec))
103 Replacements.emplace_back(GEP, Header, Preheader, BackBB, Step->getValue(), StartValue);
104 else if (auto Step = dyn_cast<SCEVConstant>(Rec))
105 Replacements.emplace_back(GEP, Header, Preheader, BackBB, Step->getValue(), StartValue);
106 }
107 }
108 }
109 for (auto R : Replacements) {
110 auto &GEP = std::get<0>(R);
111 auto *Header = std::get<1>(R);
112 auto *Preheader = std::get<2>(R);
113 auto *BackBB = std::get<3>(R);
114 auto *StepValue = std::get<4>(R);
115 auto *StartValue = std::get<5>(R);
116 auto *PHI = PHINode::Create(GEP.getType(), 2,
117 "decanonicalized_ptr", &*Header->begin());
118 auto *NewGEP =
119 GetElementPtrInst::Create(GEP.getResultElementType(), PHI, {
120 StepValue }, "decanonicalized", &GEP);
121 PHI->addIncoming(NewGEP, BackBB);
122 auto *StartGEP =
123 GetElementPtrInst::Create(GEP.getResultElementType(), GEP.getOperand(0), {
124 StartValue }, "decanonicalized_start", &*(--(Preheader->end())));
125 PHI->addIncoming(StartGEP, Preheader);
126 GEP.replaceAllUsesWith(PHI);
127 Modified = true;
128 }
129 return Modified;
130 }
131 public:
132 static char ID;
133 CheriLoopPointerDecanonicalize() : FunctionPass(ID) {}
134 StringRef getPassName() const override {
135 return "CheriLoopPointerDecanonicalize";
136 }
137};
