45 MipsDisassembler(const MCSubtargetInfo &STI, MCContext &Ctx, bool IsBigEndian)

47 IsMicroMips(STI.getFeatureBits()[Mips::FeatureMicroMips]),
48 IsBigEndian(IsBigEndian) {}

51 bool hasMips3() const { return STI.getFeatureBits()[Mips::FeatureMips3]; }
52 bool hasMips32() const { return STI.getFeatureBits()[Mips::FeatureMips32]; }
53
54 bool hasMips32r6() const {
55 return STI.getFeatureBits()[Mips::FeatureMips32r6];
56 }
57
58 bool isFP64() const { return STI.getFeatureBits()[Mips::FeatureFP64Bit]; }
59
60 bool isGP64() const { return STI.getFeatureBits()[Mips::FeatureGP64Bit]; }
61
62 bool isPTR64() const { return STI.getFeatureBits()[Mips::FeaturePTR64Bit]; }
63
64 bool hasCnMips() const { return STI.getFeatureBits()[Mips::FeatureCnMips]; }
65
66 bool hasCOP3() const {
67 // Only present in MIPS-I and MIPS-II
68 return !hasMips32() && !hasMips3();

72 ArrayRef<uint8_t> Bytes, uint64_t Address,
73 raw_ostream &VStream,
74 raw_ostream &CStream) const override;
75};

610template <typename InsnType>
611static DecodeStatus DecodeINSVE_DF(MCInst &MI, InsnType insn, uint64_t Address,
612 const void *Decoder) {
613 using DecodeFN = DecodeStatus (*)(MCInst &, unsigned, uint64_t, const void *);
614
615 // The size of the n field depends on the element size
616 // The register class also depends on this.
617 InsnType tmp = fieldFromInstruction(insn, 17, 5);
618 unsigned NSize = 0;
619 DecodeFN RegDecoder = nullptr;
620 if ((tmp & 0x18) == 0x00) { // INSVE_B
621 NSize = 4;
622 RegDecoder = DecodeMSA128BRegisterClass;
623 } else if ((tmp & 0x1c) == 0x10) { // INSVE_H
624 NSize = 3;
625 RegDecoder = DecodeMSA128HRegisterClass;
626 } else if ((tmp & 0x1e) == 0x18) { // INSVE_W
627 NSize = 2;
628 RegDecoder = DecodeMSA128WRegisterClass;
629 } else if ((tmp & 0x1f) == 0x1c) { // INSVE_D
630 NSize = 1;
631 RegDecoder = DecodeMSA128DRegisterClass;
632 } else
633 llvm_unreachable("Invalid encoding");
634
635 assert(NSize != 0 && RegDecoder != nullptr);
636
637 // $wd
638 tmp = fieldFromInstruction(insn, 6, 5);
639 if (RegDecoder(MI, tmp, Address, Decoder) == MCDisassembler::Fail)
640 return MCDisassembler::Fail;
641 // $wd_in
642 if (RegDecoder(MI, tmp, Address, Decoder) == MCDisassembler::Fail)
643 return MCDisassembler::Fail;
644 // $n
645 tmp = fieldFromInstruction(insn, 16, NSize);
646 MI.addOperand(MCOperand::createImm(tmp));
647 // $ws
648 tmp = fieldFromInstruction(insn, 11, 5);
649 if (RegDecoder(MI, tmp, Address, Decoder) == MCDisassembler::Fail)
650 return MCDisassembler::Fail;
651 // $n2
652 MI.addOperand(MCOperand::createImm(0));
653
654 return MCDisassembler::Success;
655}

685template <typename InsnType>
686static DecodeStatus DecodeAddiGroupBranch(MCInst &MI, InsnType insn,
687 uint64_t Address,
688 const void *Decoder) {
689 // If we are called then we can assume that MIPS32r6/MIPS64r6 is enabled
690 // (otherwise we would have matched the ADDI instruction from the earlier
691 // ISA's instead).
692 //
693 // We have:
694 // 0b001000 sssss ttttt iiiiiiiiiiiiiiii
695 // BOVC if rs >= rt
696 // BEQZALC if rs == 0 && rt != 0
697 // BEQC if rs < rt && rs != 0
698
699 InsnType Rs = fieldFromInstruction(insn, 21, 5);
700 InsnType Rt = fieldFromInstruction(insn, 16, 5);
701 int64_t Imm = SignExtend64(fieldFromInstruction(insn, 0, 16), 16) * 4 + 4;
702 bool HasRs = false;
703
704 if (Rs >= Rt) {
705 MI.setOpcode(Mips::BOVC);
706 HasRs = true;
707 } else if (Rs != 0 && Rs < Rt) {

1183/// Read two bytes from the ArrayRef and return 16 bit halfword sorted
1184/// according to the given endianness.
1185static DecodeStatus readInstruction16(ArrayRef<uint8_t> Bytes, uint64_t Address,
1186 uint64_t &Size, uint32_t &Insn,
1187 bool IsBigEndian) {
1188 // We want to read exactly 2 Bytes of data.
1189 if (Bytes.size() < 2) {
1190 Size = 0;
1191 return MCDisassembler::Fail;
1192 }
1193
1194 if (IsBigEndian) {
1195 Insn = (Bytes[0] << 8) | Bytes[1];
1196 } else {
1197 Insn = (Bytes[1] << 8) | Bytes[0];
1198 }
1199
1200 return MCDisassembler::Success;
1201}

1203/// Read four bytes from the ArrayRef and return 32 bit word sorted
1204/// according to the given endianness.
1205static DecodeStatus readInstruction32(ArrayRef<uint8_t> Bytes, uint64_t Address,
1206 uint64_t &Size, uint32_t &Insn,
1207 bool IsBigEndian, bool IsMicroMips) {
1208 // We want to read exactly 4 Bytes of data.
1209 if (Bytes.size() < 4) {
1210 Size = 0;
1211 return MCDisassembler::Fail;
1212 }
1213
1214 // High 16 bits of a 32-bit microMIPS instruction (where the opcode is)
1215 // always precede the low 16 bits in the instruction stream (that is, they
1216 // are placed at lower addresses in the instruction stream).
1217 //
1218 // microMIPS byte ordering:
1219 // Big-endian: 0 | 1 | 2 | 3
1220 // Little-endian: 1 | 0 | 3 | 2
1221
1222 if (IsBigEndian) {
1223 // Encoded as a big-endian 32-bit word in the stream.
1224 Insn =
1225 (Bytes[3] << 0) | (Bytes[2] << 8) | (Bytes[1] << 16) | (Bytes[0] << 24);
1226 } else {
1227 if (IsMicroMips) {
1228 Insn = (Bytes[2] << 0) | (Bytes[3] << 8) | (Bytes[0] << 16) |
1229 (Bytes[1] << 24);
1230 } else {
1231 Insn = (Bytes[0] << 0) | (Bytes[1] << 8) | (Bytes[2] << 16) |
1232 (Bytes[3] << 24);
1233 }
1234 }
1235
1236 return MCDisassembler::Success;
1237}

1239DecodeStatus MipsDisassembler::getInstruction(MCInst &Instr, uint64_t &Size,
1240 ArrayRef<uint8_t> Bytes,
1241 uint64_t Address,
1242 raw_ostream &VStream,
1243 raw_ostream &CStream) const {
1244 uint32_t Insn;
1245 DecodeStatus Result;
1246 Size = 0;
1247
1248 if (IsMicroMips) {
1249 Result = readInstruction16(Bytes, Address, Size, Insn, IsBigEndian);
1250 if (Result == MCDisassembler::Fail)
1251 return MCDisassembler::Fail;
1252
1253 if (hasMips32r6()) {
1254 LLVM_DEBUG(
1255 dbgs() << "Trying MicroMipsR616 table (16-bit instructions):\n");
1256 // Calling the auto-generated decoder function for microMIPS32R6
1257 // 16-bit instructions.
1258 Result = decodeInstruction(DecoderTableMicroMipsR616, Instr, Insn,
1259 Address, this, STI);
1260 if (Result != MCDisassembler::Fail) {
1261 Size = 2;
1262 return Result;
1263 }
1264 }
1265
1266 LLVM_DEBUG(dbgs() << "Trying MicroMips16 table (16-bit instructions):\n");
1267 // Calling the auto-generated decoder function for microMIPS 16-bit
1268 // instructions.
1269 Result = decodeInstruction(DecoderTableMicroMips16, Instr, Insn, Address,
1270 this, STI);
1271 if (Result != MCDisassembler::Fail) {
1272 Size = 2;
1273 return Result;
1274 }
1275
1276 Result = readInstruction32(Bytes, Address, Size, Insn, IsBigEndian, true);
1277 if (Result == MCDisassembler::Fail)
1278 return MCDisassembler::Fail;
1279
1280 if (hasMips32r6()) {
1281 LLVM_DEBUG(
1282 dbgs() << "Trying MicroMips32r632 table (32-bit instructions):\n");
1283 // Calling the auto-generated decoder function.
1284 Result = decodeInstruction(DecoderTableMicroMipsR632, Instr, Insn, Address,
1285 this, STI);
1286 if (Result != MCDisassembler::Fail) {
1287 Size = 4;
1288 return Result;
1289 }
1290 }
1291
1292 LLVM_DEBUG(dbgs() << "Trying MicroMips32 table (32-bit instructions):\n");
1293 // Calling the auto-generated decoder function.
1294 Result = decodeInstruction(DecoderTableMicroMips32, Instr, Insn, Address,
1295 this, STI);
1296 if (Result != MCDisassembler::Fail) {
1297 Size = 4;
1298 return Result;
1299 }
1300
1301 if (isFP64()) {
1302 LLVM_DEBUG(dbgs() << "Trying MicroMipsFP64 table (32-bit opcodes):\n");
1303 Result = decodeInstruction(DecoderTableMicroMipsFP6432, Instr, Insn,
1304 Address, this, STI);
1305 if (Result != MCDisassembler::Fail) {
1306 Size = 4;
1307 return Result;
1308 }
1309 }
1310
1311 // This is an invalid instruction. Claim that the Size is 2 bytes. Since
1312 // microMIPS instructions have a minimum alignment of 2, the next 2 bytes
1313 // could form a valid instruction. The two bytes we rejected as an
1314 // instruction could have actually beeen an inline constant pool that is
1315 // unconditionally branched over.
1316 Size = 2;
1317 return MCDisassembler::Fail;
1318 }
1319
1320 // Attempt to read the instruction so that we can attempt to decode it. If
1321 // the buffer is not 4 bytes long, let the higher level logic figure out
1322 // what to do with a size of zero and MCDisassembler::Fail.
1323 Result = readInstruction32(Bytes, Address, Size, Insn, IsBigEndian, false);
1324 if (Result == MCDisassembler::Fail)
1325 return MCDisassembler::Fail;
1326
1327 // The only instruction size for standard encoded MIPS.
1328 Size = 4;
1329
1330 if (hasCOP3()) {
1331 LLVM_DEBUG(dbgs() << "Trying COP3_ table (32-bit opcodes):\n");
1332 Result =
1333 decodeInstruction(DecoderTableCOP3_32, Instr, Insn, Address, this, STI);
1334 if (Result != MCDisassembler::Fail)
1335 return Result;
1336 }
1337
1338 if (hasMips32r6() && isGP64()) {
1339 LLVM_DEBUG(
1340 dbgs() << "Trying Mips32r6_64r6 (GPR64) table (32-bit opcodes):\n");
1341 Result = decodeInstruction(DecoderTableMips32r6_64r6_GP6432, Instr, Insn,
1342 Address, this, STI);
1343 if (Result != MCDisassembler::Fail)
1344 return Result;
1345 }
1346
1347 if (hasMips32r6() && isPTR64()) {
1348 LLVM_DEBUG(
1349 dbgs() << "Trying Mips32r6_64r6 (PTR64) table (32-bit opcodes):\n");
1350 Result = decodeInstruction(DecoderTableMips32r6_64r6_PTR6432, Instr, Insn,
1351 Address, this, STI);
1352 if (Result != MCDisassembler::Fail)
1353 return Result;
1354 }
1355
1356 if (hasMips32r6()) {
1357 LLVM_DEBUG(dbgs() << "Trying Mips32r6_64r6 table (32-bit opcodes):\n");
1358 Result = decodeInstruction(DecoderTableMips32r6_64r632, Instr, Insn,
1359 Address, this, STI);
1360 if (Result != MCDisassembler::Fail)
1361 return Result;
1362 }
1363
1364 if (hasMips2() && isPTR64()) {
1365 LLVM_DEBUG(
1366 dbgs() << "Trying Mips32r6_64r6 (PTR64) table (32-bit opcodes):\n");
1367 Result = decodeInstruction(DecoderTableMips32_64_PTR6432, Instr, Insn,
1368 Address, this, STI);
1369 if (Result != MCDisassembler::Fail)
1370 return Result;
1371 }
1372
1373 if (hasCnMips()) {
1374 LLVM_DEBUG(dbgs() << "Trying CnMips table (32-bit opcodes):\n");
1375 Result = decodeInstruction(DecoderTableCnMips32, Instr, Insn,
1376 Address, this, STI);
1377 if (Result != MCDisassembler::Fail)
1378 return Result;
1379 }
1380
1381 if (isGP64()) {
1382 LLVM_DEBUG(dbgs() << "Trying Mips64 (GPR64) table (32-bit opcodes):\n");
1383 Result = decodeInstruction(DecoderTableMips6432, Instr, Insn,
1384 Address, this, STI);
1385 if (Result != MCDisassembler::Fail)
1386 return Result;
1387 }
1388
1389 if (isFP64()) {
1390 LLVM_DEBUG(
1391 dbgs() << "Trying MipsFP64 (64 bit FPU) table (32-bit opcodes):\n");
1392 Result = decodeInstruction(DecoderTableMipsFP6432, Instr, Insn,
1393 Address, this, STI);
1394 if (Result != MCDisassembler::Fail)
1395 return Result;
1396 }
1397
1398 LLVM_DEBUG(dbgs() << "Trying Mips table (32-bit opcodes):\n");
1399 // Calling the auto-generated decoder function.
1400 Result =
1401 decodeInstruction(DecoderTableMips32, Instr, Insn, Address, this, STI);
1402 if (Result != MCDisassembler::Fail)
1403 return Result;
1404
1405 return MCDisassembler::Fail;
1406}

1408DecodeStatus CheriDisassembler::getInstruction(MCInst &Instr, uint64_t &Size,
1409 ArrayRef<uint8_t> Bytes,
1410 uint64_t Address,
1411 raw_ostream &VStream,
1412 raw_ostream &CStream) const {
1413 uint32_t Insn;
1414 // If we try deconding an unknown isntruction advance by 4 bytes
1415 // Otherwise llvm-objdump will only advance by one byte and wrongly disassemble
1416 // the rest of the function after it encounters an unknown instruction
1417 Size = 4;
1418 DecodeStatus Result = readInstruction32(Bytes, Address, Size, Insn,
1419 /*IsBigEndian*/ true, /*IsMicroMips*/false);
1420 if (Result == MCDisassembler::Fail)
1421 return MCDisassembler::Fail;
1422
1423 // Try this as a CHERI instruction
1424 Result = decodeInstruction(DecoderTableCHERI32, Instr, Insn, Address,
1425 this, STI);
1426 if (Result != MCDisassembler::Fail) {
1427 Size = 4;
1428 return Result;
1429 }
1430 // If that fails, try it as a generic MIPS64 instruction
1431 Result = decodeInstruction(DecoderTableMips6432, Instr, Insn, Address,
1432 this, STI);
1433 if (Result != MCDisassembler::Fail) {
1434 Size = 4;
1435 return Result;
1436 }
1437 // If we fail to decode in Mips64 decoder space we can try in Mips32
1438 Result = decodeInstruction(DecoderTableMips32, Instr, Insn, Address,
1439 this, STI);
1440 if (Result != MCDisassembler::Fail) {
1441 Size = 4;
1442 return Result;
1443 }
1444 return MCDisassembler::Fail;
1445}

1791static DecodeStatus DecodeMSA128Mem(MCInst &Inst, unsigned Insn,
1792 uint64_t Address, const void *Decoder) {
1793 int Offset = SignExtend32<10>(fieldFromInstruction(Insn, 16, 10));
1794 unsigned Reg = fieldFromInstruction(Insn, 6, 5);
1795 unsigned Base = fieldFromInstruction(Insn, 11, 5);

1801 Inst.addOperand(MCOperand::createReg(Base));

1803 // The immediate field of an LD/ST instruction is scaled which means it must
1804 // be multiplied (when decoding) by the size (in bytes) of the instructions'
1805 // data format.
1806 // .b - 1 byte
1807 // .h - 2 bytes
1808 // .w - 4 bytes
1809 // .d - 8 bytes
1810 switch(Inst.getOpcode())
1811 {
1812 default:
1813 assert(false && "Unexpected instruction");
1814 return MCDisassembler::Fail;
1815 break;
1816 case Mips::LD_B:
1817 case Mips::ST_B:
1818 Inst.addOperand(MCOperand::createImm(Offset));
1819 break;
1820 case Mips::LD_H:
1821 case Mips::ST_H:
1822 Inst.addOperand(MCOperand::createImm(Offset * 2));
1823 break;
1824 case Mips::LD_W:
1825 case Mips::ST_W:
1826 Inst.addOperand(MCOperand::createImm(Offset * 4));
1827 break;
1828 case Mips::LD_D:
1829 case Mips::ST_D:
1830 Inst.addOperand(MCOperand::createImm(Offset * 8));
1831 break;
1832 }
1833
1834 return MCDisassembler::Success;
1835}

1839 uint64_t Address,
1840 const void *Decoder) {
1841 unsigned Offset = Insn & 0xf;
1842 unsigned Reg = fieldFromInstruction(Insn, 7, 3);
1843 unsigned Base = fieldFromInstruction(Insn, 4, 3);
1844
1845 switch (Inst.getOpcode()) {
1846 case Mips::LBU16_MM:
1847 case Mips::LHU16_MM:
1848 case Mips::LW16_MM:
1849 if (DecodeGPRMM16RegisterClass(Inst, Reg, Address, Decoder)
1850 == MCDisassembler::Fail)
1851 return MCDisassembler::Fail;
1852 break;
1853 case Mips::SB16_MM:
1854 case Mips::SB16_MMR6:
1855 case Mips::SH16_MM:

1860 == MCDisassembler::Fail)
1861 return MCDisassembler::Fail;

1864
1865 if (DecodeGPRMM16RegisterClass(Inst, Base, Address, Decoder)

1870 case Mips::LBU16_MM:
1871 if (Offset == 0xf)
1872 Inst.addOperand(MCOperand::createImm(-1));
1873 else
1874 Inst.addOperand(MCOperand::createImm(Offset));

1876 case Mips::SB16_MM:
1877 case Mips::SB16_MMR6:

1880 case Mips::LHU16_MM:
1881 case Mips::SH16_MM:
1882 case Mips::SH16_MMR6:
1883 Inst.addOperand(MCOperand::createImm(Offset << 1));
1884 break;
1885 case Mips::LW16_MM:

1887 case Mips::SW16_MMR6:

1890 }

1893}
