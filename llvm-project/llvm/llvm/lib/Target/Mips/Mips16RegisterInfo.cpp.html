57bool Mips16RegisterInfo::saveScavengerRegister
58 (MachineBasicBlock &MBB,
59 MachineBasicBlock::iterator I,
60 MachineBasicBlock::iterator &UseMI,
61 const TargetRegisterClass *RC,
62 unsigned Reg) const {
63 DebugLoc DL;
64 const TargetInstrInfo &TII = *MBB.getParent()->getSubtarget().getInstrInfo();
65 TII.copyPhysReg(MBB, I, DL, Mips::T0, Reg, true);
66 TII.copyPhysReg(MBB, UseMI, DL, Reg, Mips::T0, true);
67 return true;
68}

76void Mips16RegisterInfo::eliminateFI(MachineBasicBlock::iterator II,
77 unsigned OpNo, int FrameIndex,
78 uint64_t StackSize,
79 int64_t SPOffset,
80 RegScavenger *RS) const {
81 MachineInstr &MI = *II;
82 MachineFunction &MF = *MI.getParent()->getParent();
83 MachineFrameInfo &MFI = MF.getFrameInfo();
84
85 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
86 int MinCSFI = 0;
87 int MaxCSFI = -1;
88
89 if (CSI.size()) {
90 MinCSFI = CSI[0].getFrameIdx();
91 MaxCSFI = CSI[CSI.size() - 1].getFrameIdx();
92 }
93
94 // The following stack frame objects are always
95 // referenced relative to $sp:
96 // 1. Outgoing arguments.
97 // 2. Pointer to dynamically allocated stack space.
98 // 3. Locations for callee-saved registers.
99 // Everything else is referenced relative to whatever register
100 // getFrameRegister() returns.
101 unsigned FrameReg;
102
103 if (FrameIndex >= MinCSFI && FrameIndex <= MaxCSFI)
104 FrameReg = Mips::SP;
105 else {
106 const TargetFrameLowering *TFI = MF.getSubtarget().getFrameLowering();
107 if (TFI->hasFP(MF)) {
108 FrameReg = Mips::S0;
109 }
110 else {
111 if ((MI.getNumOperands()> OpNo+2) && MI.getOperand(OpNo+2).isReg())
112 FrameReg = MI.getOperand(OpNo+2).getReg();
113 else
114 FrameReg = Mips::SP;
115 }
116 }
117 // Calculate final offset.
118 // - There is no need to change the offset if the frame object
119 // is one of the
120 // following: an outgoing argument, pointer to a dynamically allocated
121 // stack space or a $gp restore location,
122 // - If the frame object is any of the following,
123 // its offset must be adjusted
124 // by adding the size of the stack:
125 // incoming argument, callee-saved register location or local variable.
126 int64_t Offset;
127 bool IsKill = false;
128 Offset = SPOffset + (int64_t)StackSize;
129 Offset += MI.getOperand(OpNo + 1).getImm();
130
131 LLVM_DEBUG(errs() << "Offset : " << Offset << "\n"
132 << "<--------->\n");
133
134 if (!MI.isDebugValue() &&
135 !Mips16InstrInfo::validImmediate(MI.getOpcode(), FrameReg, Offset)) {
136 MachineBasicBlock &MBB = *MI.getParent();
137 DebugLoc DL = II->getDebugLoc();
138 unsigned NewImm;
139 const Mips16InstrInfo &TII =
140 *static_cast<const Mips16InstrInfo *>(MF.getSubtarget().getInstrInfo());
141 FrameReg = TII.loadImmediate(FrameReg, Offset, MBB, II, DL, NewImm);
142 Offset = SignExtend64<16>(NewImm);
143 IsKill = true;
144 }
145 MI.getOperand(OpNo).ChangeToRegister(FrameReg, false, false, IsKill);
146 MI.getOperand(OpNo + 1).ChangeToImmediate(Offset);
147
148
149}
