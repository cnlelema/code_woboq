15static cl::opt<bool> EnableCheri128FailHard(
16 "enable-cheri128-fail-hard", cl::init(false),
17 cl::desc("Trap if CHERI128 operations lose their tags"), cl::Hidden);
18
19namespace {
20struct Cher128FailHardPass : public MachineFunctionPass {
21 static char ID;
22 const MipsInstrInfo *InstrInfo = nullptr;
23
24 Cher128FailHardPass() : MachineFunctionPass(ID) {
25 }
26 Cher128FailHardPass(MipsTargetMachine &TM) : MachineFunctionPass(ID) {
27 InstrInfo = TM.getSubtargetImpl()->getInstrInfo();
28 }
29
30 StringRef getPassName() const override { return "Cheri128 fail hard"; }
31
32 bool runOnMachineFunction(MachineFunction &MF) override {
33 if (!EnableCheri128FailHard)
34 return false;
35 InstrInfo = MF.getSubtarget<MipsSubtarget>().getInstrInfo();
36 SmallVector<MachineInstr*, 16> instrs;
37 std::set<MachineBasicBlock*> bbs;
38 for (auto &MBB : MF)
39 for (auto &MI : MBB)
40 if ((MI.getOpcode() == Mips::CIncOffset) ||
41 (MI.getOpcode() == Mips::CSetOffset)) {
42 instrs.push_back(&MI);
43 bbs.insert(&MBB);
44 }
45 if (instrs.empty())
46 return false;
47 MachineRegisterInfo &RI = MF.getRegInfo();
48 for (auto *MI : instrs) {
49 unsigned tag_before = RI.createVirtualRegister(&Mips::GPR64RegClass);
50 unsigned tag_after = RI.createVirtualRegister(&Mips::GPR64RegClass);
51 auto &MBB = *MI->getParent();
52 BuildMI(MBB, MI,
53 MI->getDebugLoc(), InstrInfo->get(Mips::CGetTag))
54 .addDef(tag_before).addUse(MI->getOperand(1).getReg());
55 MachineBasicBlock::iterator after =
56 BuildMI(*MI->getParent(), ++MachineBasicBlock::iterator(MI),
57 MI->getDebugLoc(), InstrInfo->get(Mips::CGetTag))
58 .addDef(tag_after).addUse(MI->getOperand(0).getReg()).getInstr();
59 auto Sink = MF.CreateMachineBasicBlock(MBB.getBasicBlock());
60 MF.insert(++MachineFunction::iterator(MBB), Sink);
61 // Transfer the remainder of BB and its successor edges to Sink.
62 Sink->splice(Sink->begin(), &MBB, std::next(after),
63 MBB.end());
64 Sink->transferSuccessorsAndUpdatePHIs(&MBB);
65 MBB.addSuccessor(Sink);
66 assert(Sink->getParent() == &MF);
67 after = BuildMI(MBB, ++after,
68 MI->getDebugLoc(), InstrInfo->get(Mips::BEQ64))
69 .addUse(tag_after, RegState::Kill)
70 .addUse(tag_before, RegState::Kill)
71 .addMBB(Sink);
72 BuildMI(MBB, ++after, MI->getDebugLoc(), InstrInfo->get(Mips::TRAP));
73 }
74 return true;
75 }
76};
