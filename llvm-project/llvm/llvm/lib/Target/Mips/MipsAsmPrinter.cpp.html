77bool MipsAsmPrinter::runOnMachineFunction(MachineFunction &MF) {
78 Subtarget = &MF.getSubtarget<MipsSubtarget>();
79
80 MipsFI = MF.getInfo<MipsFunctionInfo>();
81 if (Subtarget->inMips16Mode())
82 for (std::map<
83 const char *,
84 const Mips16HardFloatInfo::FuncSignature *>::const_iterator
85 it = MipsFI->StubsNeeded.begin();
86 it != MipsFI->StubsNeeded.end(); ++it) {
87 const char *Symbol = it->first;
88 const Mips16HardFloatInfo::FuncSignature *Signature = it->second;
89 if (StubsNeeded.find(Symbol) == StubsNeeded.end())
90 StubsNeeded[Symbol] = Signature;
91 }
92 MCP = MF.getConstantPool();
93
94 // In NaCl, all indirect jump targets must be aligned to bundle size.
95 if (Subtarget->isTargetNaCl())
96 NaClAlignIndirectJumpTargets(MF);
97
98 AsmPrinter::runOnMachineFunction(MF);
99
100 emitXRayTable();
101
102 return true;
103}

112// Lower PseudoReturn/PseudoIndirectBranch/PseudoIndirectBranch64 to JR, JR_MM,
113// JALR, or JALR64 as appropriate for the target.
114void MipsAsmPrinter::emitPseudoIndirectBranch(MCStreamer &OutStreamer,
115 const MachineInstr *MI) {
116 bool HasLinkReg = false;
117 bool InMicroMipsMode = Subtarget->inMicroMipsMode();
118 MCInst TmpInst0;
119
120 if (Subtarget->hasMips64r6()) {
121 // MIPS64r6 should use (JALR64 ZERO_64, $rs)
122 TmpInst0.setOpcode(Mips::JALR64);
123 HasLinkReg = true;
124 } else if (Subtarget->hasMips32r6()) {
125 // MIPS32r6 should use (JALR ZERO, $rs)
126 if (InMicroMipsMode)
127 TmpInst0.setOpcode(Mips::JRC16_MMR6);
128 else {
129 TmpInst0.setOpcode(Mips::JALR);
130 HasLinkReg = true;
131 }
132 } else if (Subtarget->inMicroMipsMode())
133 // microMIPS should use (JR_MM $rs)
134 TmpInst0.setOpcode(Mips::JR_MM);
135 else if (static_cast<MipsTargetMachine &>(TM).getABI().IsCheriPureCap())
136 // Everything else should use (JR $rs) or (CJR $rs), depending on the register.
137 TmpInst0.setOpcode(
138 Mips::CheriGPROrCNullRegClass.contains(MI->getOperand(0).getReg())
139 ? Mips::CJR
140 : Mips::JR);
141 else {
142 // Everything else should use (JR $rs)
143 TmpInst0.setOpcode(Mips::JR);
144 }
145
146 MCOperand MCOp;
147
148 if (HasLinkReg) {
149 unsigned ZeroReg = Subtarget->isGP64bit() ? Mips::ZERO_64 : Mips::ZERO;
150 TmpInst0.addOperand(MCOperand::createReg(ZeroReg));
151 }
152
153 lowerOperand(MI->getOperand(0), MCOp);
154 TmpInst0.addOperand(MCOp);
155
156 EmitToStreamer(OutStreamer, TmpInst0);
157}
158
159// If there is an MO_JALR operand, insert:
160//
161// .reloc tmplabel, R_{MICRO}MIPS_JALR, symbol
162// tmplabel:
163//
164// This is an optimization hint for the linker which may then replace
165// an indirect call with a direct branch.
166static void emitDirectiveRelocJalr(const MachineInstr &MI,
167 MCContext &OutContext,
168 TargetMachine &TM,
169 MCStreamer &OutStreamer,
170 const MipsSubtarget &Subtarget) {
171 for (unsigned int I = MI.getDesc().getNumOperands(), E = MI.getNumOperands();
172 I < E; ++I) {
173 MachineOperand MO = MI.getOperand(I);
174 if (MO.isMCSymbol() && (MO.getTargetFlags() & MipsII::MO_JALR)) {
175 MCSymbol *Callee = MO.getMCSymbol();
176 if (Callee && !Callee->getName().empty()) {
177 MCSymbol *OffsetLabel = OutContext.createTempSymbol();
178 const MCExpr *OffsetExpr =
179 MCSymbolRefExpr::create(OffsetLabel, OutContext);
180 const MCExpr *CaleeExpr =
181 MCSymbolRefExpr::create(Callee, OutContext);
182 OutStreamer.EmitRelocDirective
183 (*OffsetExpr,
184 Subtarget.inMicroMipsMode() ? "R_MICROMIPS_JALR" : "R_MIPS_JALR",
185 CaleeExpr, SMLoc(), *TM.getMCSubtargetInfo());
186 OutStreamer.EmitLabel(OffsetLabel);
187 return;
188 }
189 }
190 }

193void MipsAsmPrinter::EmitInstruction(const MachineInstr *MI) {
194 MipsTargetStreamer &TS = getTargetStreamer();
195 unsigned Opc = MI->getOpcode();
196 TS.forbidModuleDirective();
197
198 if (MI->isDebugValue()) {
199 SmallString<128> Str;
200 raw_svector_ostream OS(Str);
201
202 PrintDebugValueComment(MI, OS);
203 return;
204 }
205 if (MI->isDebugLabel())
206 return;
207
208 // If we just ended a constant pool, mark it as such.
209 if (InConstantPool && Opc != Mips::CONSTPOOL_ENTRY) {
210 OutStreamer->EmitDataRegion(MCDR_DataRegionEnd);
211 InConstantPool = false;
212 }
213 if (Opc == Mips::CONSTPOOL_ENTRY) {
214 // CONSTPOOL_ENTRY - This instruction represents a floating
215 // constant pool in the function. The first operand is the ID#
216 // for this instruction, the second is the index into the
217 // MachineConstantPool that this is, the third is the size in
218 // bytes of this constant pool entry.
219 // The required alignment is specified on the basic block holding this MI.
220 //
221 unsigned LabelId = (unsigned)MI->getOperand(0).getImm();
222 unsigned CPIdx = (unsigned)MI->getOperand(1).getIndex();
223
224 // If this is the first entry of the pool, mark it.
225 if (!InConstantPool) {
226 OutStreamer->EmitDataRegion(MCDR_DataRegion);
227 InConstantPool = true;
228 }
229
230 OutStreamer->EmitLabel(GetCPISymbol(LabelId));
231
232 const MachineConstantPoolEntry &MCPE = MCP->getConstants()[CPIdx];
233 if (MCPE.isMachineConstantPoolEntry())
234 EmitMachineConstantPoolValue(MCPE.Val.MachineCPVal);
235 else {
236 // XXXAR: should not need tail padding here since all entries are small
237 EmitGlobalConstant(MF->getDataLayout(), MCPE.Val.ConstVal, 0);
238 }
239 return;
240 }
241
242 switch (Opc) {
243 case Mips::PATCHABLE_FUNCTION_ENTER:
244 LowerPATCHABLE_FUNCTION_ENTER(*MI);
245 return;
246 case Mips::PATCHABLE_FUNCTION_EXIT:
247 LowerPATCHABLE_FUNCTION_EXIT(*MI);
248 return;
249 case Mips::PATCHABLE_TAIL_CALL:
250 LowerPATCHABLE_TAIL_CALL(*MI);
251 return;
252 }
253
254 if (EmitJalrReloc &&
255 (MI->isReturn() || MI->isCall() || MI->isIndirectBranch())) {
256 emitDirectiveRelocJalr(*MI, OutContext, TM, *OutStreamer, *Subtarget);
257 }
258
259 MachineBasicBlock::const_instr_iterator I = MI->getIterator();
260 MachineBasicBlock::const_instr_iterator E = MI->getParent()->instr_end();
261
262 do {
263 // Do any auto-generated pseudo lowerings.
264 if (emitPseudoExpansionLowering(*OutStreamer, &*I))
265 continue;
266
267 // FIXME: not sure this is correct
268 if (I->getOpcode() == Mips::BUNDLE)
269 continue;
270
271 if (I->getOpcode() == Mips::PseudoReturn ||
272 I->getOpcode() == Mips::PseudoReturn64 ||
273 I->getOpcode() == Mips::PseudoReturnCap ||
274 I->getOpcode() == Mips::PseudoIndirectBranch ||
275 I->getOpcode() == Mips::PseudoIndirectBranch64 ||
276 I->getOpcode() == Mips::PseudoIndirectBranchCap ||
277 I->getOpcode() == Mips::TAILCALLREG ||
278 I->getOpcode() == Mips::TAILCALLREG64) {
279 emitPseudoIndirectBranch(*OutStreamer, &*I);
280 continue;
281 }
282
283 // The inMips16Mode() test is not permanent.
284 // Some instructions are marked as pseudo right now which
285 // would make the test fail for the wrong reason but
286 // that will be fixed soon. We need this here because we are
287 // removing another test for this situation downstream in the
288 // callchain.
289 //
290 if (I->isPseudo() && !Subtarget->inMips16Mode()
291 && !isLongBranchPseudo(I->getOpcode()))
292 llvm_unreachable("Pseudo opcode found in EmitInstruction()");
293
294 MCInst TmpInst0;
295 MCInstLowering.Lower(&*I, TmpInst0);
296 EmitToStreamer(*OutStreamer, TmpInst0);
297 } while ((++I != E) && I->isInsideBundle()); // Delay slot check
298}

300//===----------------------------------------------------------------------===//
301//
302// Mips Asm Directives
303//
304// -- Frame directive "frame Stackpointer, Stacksize, RARegister"
305// Describe the stack frame.
306//
307// -- Mask directives "(f)mask bitmask, offset"
308// Tells the assembler which registers are saved and where.
309// bitmask - contain a little endian bitset indicating which registers are
310// saved on function prologue (e.g. with a 0x80000000 mask, the
311// assembler knows the register 31 (RA) is saved at prologue.
312// offset - the position before stack pointer subtraction indicating where
313// the first saved register on prologue is located. (e.g. with a
314//
315// Consider the following function prologue:
316//
317// .frame $fp,48,$ra
318// .mask 0xc0000000,-8
319// addiu $sp, $sp, -48
320// sw $ra, 40($sp)
321// sw $fp, 36($sp)
322//
323// With a 0xc0000000 mask, the assembler knows the register 31 (RA) and
324// 30 (FP) are saved at prologue. As the save order on prologue is from
325// left to right, RA is saved first. A -8 offset means that after the
326// stack pointer subtration, the first register in the mask (RA) will be
327// saved at address 48-8=40.
328//
329//===----------------------------------------------------------------------===//
330
331//===----------------------------------------------------------------------===//
332// Mask directives
333//===----------------------------------------------------------------------===//
334
335// Create a bitmask with all callee saved registers for CPU or Floating Point
336// registers. For CPU registers consider RA, GP and FP for saving if necessary.
337void MipsAsmPrinter::printSavedRegsBitmask() {
338 // CPU and FPU Saved Registers Bitmasks
339 unsigned CPUBitmask = 0, FPUBitmask = 0;
340 int CPUTopSavedRegOff, FPUTopSavedRegOff;
341
342 // Set the CPU and FPU Bitmasks
343 const MachineFrameInfo &MFI = MF->getFrameInfo();
344 const TargetRegisterInfo *TRI = MF->getSubtarget().getRegisterInfo();
345 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
346 // size of stack area to which FP callee-saved regs are saved.
347 unsigned CPURegSize = TRI->getRegSizeInBits(Mips::GPR32RegClass) / 8;
348 unsigned FGR32RegSize = TRI->getRegSizeInBits(Mips::FGR32RegClass) / 8;
349 unsigned AFGR64RegSize = TRI->getRegSizeInBits(Mips::AFGR64RegClass) / 8;
350 bool HasAFGR64Reg = false;
351 unsigned CSFPRegsSize = 0;
352
353 for (const auto &I : CSI) {
354 unsigned Reg = I.getReg();
355 unsigned RegNum = TRI->getEncodingValue(Reg);
356
357 // If it's a floating point register, set the FPU Bitmask.
358 // If it's a general purpose register, set the CPU Bitmask.
359 if (Mips::FGR32RegClass.contains(Reg)) {
360 FPUBitmask |= (1 << RegNum);
361 CSFPRegsSize += FGR32RegSize;
362 } else if (Mips::AFGR64RegClass.contains(Reg)) {
363 FPUBitmask |= (3 << RegNum);
364 CSFPRegsSize += AFGR64RegSize;
365 HasAFGR64Reg = true;
366 } else if (Mips::GPR32RegClass.contains(Reg))
367 CPUBitmask |= (1 << RegNum);
368 }
369
370 // FP Regs are saved right below where the virtual frame pointer points to.
371 FPUTopSavedRegOff = FPUBitmask ?
372 (HasAFGR64Reg ? -AFGR64RegSize : -FGR32RegSize) : 0;
373
374 // CPU Regs are saved below FP Regs.
375 CPUTopSavedRegOff = CPUBitmask ? -CSFPRegsSize - CPURegSize : 0;
376
377 MipsTargetStreamer &TS = getTargetStreamer();
378 // Print CPUBitmask
379 TS.emitMask(CPUBitmask, CPUTopSavedRegOff);
380
381 // Print FPUBitmask
382 TS.emitFMask(FPUBitmask, FPUTopSavedRegOff);
383}

410void MipsAsmPrinter::EmitFunctionEntryLabel() {
411 MipsTargetStreamer &TS = getTargetStreamer();
412
413 // NaCl sandboxing requires that indirect call instructions are masked.
414 // This means that function entry points should be bundle-aligned.
415 if (Subtarget->isTargetNaCl())
416 EmitAlignment(std::max(MF->getAlignment(), MIPS_NACL_BUNDLE_ALIGN));
417
418 if (Subtarget->inMicroMipsMode()) {
419 TS.emitDirectiveSetMicroMips();
420 TS.setUsesMicroMips();
421 TS.updateABIInfo(*Subtarget);
422 } else
423 TS.emitDirectiveSetNoMicroMips();
424
425 if (Subtarget->inMips16Mode())
426 TS.emitDirectiveSetMips16();
427 else
428 TS.emitDirectiveSetNoMips16();
429
430 TS.emitDirectiveEnt(*CurrentFnSym);
431 OutStreamer->EmitLabel(CurrentFnSym);
432}

434/// EmitFunctionBodyStart - Targets can override this to emit stuff before
435/// the first basic block in the function.
436void MipsAsmPrinter::EmitFunctionBodyStart() {
437 MipsTargetStreamer &TS = getTargetStreamer();
438
439 MCInstLowering.Initialize(&MF->getContext());
440
441 bool IsNakedFunction = MF->getFunction().hasFnAttribute(Attribute::Naked);
442 if (!IsNakedFunction)
443 emitFrameDirective();
444
445 if (!IsNakedFunction)
446 printSavedRegsBitmask();
447
448 if (!Subtarget->inMips16Mode()) {
449 TS.emitDirectiveSetNoReorder();
450 TS.emitDirectiveSetNoMacro();
451 TS.emitDirectiveSetNoAt();
452 }
453}
454
455/// EmitFunctionBodyEnd - Targets can override this to emit stuff after
456/// the last basic block in the function.
457void MipsAsmPrinter::EmitFunctionBodyEnd() {
458 MipsTargetStreamer &TS = getTargetStreamer();
459
460 // There are instruction for this macros, but they must
461 // always be at the function end, and we can't emit and
462 // break with BB logic.
463 if (!Subtarget->inMips16Mode()) {
464 TS.emitDirectiveSetAt();
465 TS.emitDirectiveSetMacro();
466 TS.emitDirectiveSetReorder();
467 }
468 TS.emitDirectiveEnd(CurrentFnSym->getName());
469 // Make sure to terminate any constant pools that were at the end
470 // of the function.
471 if (!InConstantPool)
472 return;
473 InConstantPool = false;
474 OutStreamer->EmitDataRegion(MCDR_DataRegionEnd);
475}

484/// isBlockOnlyReachableByFallthough - Return true if the basic block has
485/// exactly one predecessor and the control transfer mechanism between
486/// the predecessor and this block is a fall-through.
487bool MipsAsmPrinter::isBlockOnlyReachableByFallthrough(const MachineBasicBlock*
488 MBB) const {
489 // The predecessor has to be immediately before this block.
490 const MachineBasicBlock *Pred = *MBB->pred_begin();
491
492 // If the predecessor is a switch statement, assume a jump table
493 // implementation, so it is not a fall through.
494 if (const BasicBlock *bb = Pred->getBasicBlock())
495 if (isa<SwitchInst>(bb->getTerminator()))
496 return false;
497
498 // If this is a landing pad, it isn't a fall through. If it has no preds,
499 // then nothing falls through to it.
500 if (MBB->isEHPad() || MBB->pred_empty())
501 return false;
502
503 // If there isn't exactly one predecessor, it can't be a fall through.
504 MachineBasicBlock::const_pred_iterator PI = MBB->pred_begin(), PI2 = PI;
505 ++PI2;
506
507 if (PI2 != MBB->pred_end())
508 return false;
509
510 // The predecessor has to be immediately before this block.
511 if (!Pred->isLayoutSuccessor(MBB))
512 return false;
513
514 // If the block is completely empty, then it definitely does fall through.
515 if (Pred->empty())
516 return true;
517
518 // Otherwise, check the last instruction.
519 // Check if the last terminator is an unconditional branch.
520 MachineBasicBlock::const_iterator I = Pred->end();
521 while (I != Pred->begin() && !(--I)->isTerminator()) ;
522
523 return !I->isBarrier();
524}

526// Print out an operand for an inline asm expression.
527bool MipsAsmPrinter::PrintAsmOperand(const MachineInstr *MI, unsigned OpNum,
528 const char *ExtraCode, raw_ostream &O) {
529 // Does this asm operand have a single letter operand modifier?
530 if (ExtraCode && ExtraCode[0]) {
531 if (ExtraCode[1] != 0) return true; // Unknown modifier.
532
533 const MachineOperand &MO = MI->getOperand(OpNum);
534 switch (ExtraCode[0]) {
535 default:
536 // See if this is a generic print operand
537 return AsmPrinter::PrintAsmOperand(MI, OpNum, ExtraCode, O);
538 case 'X': // hex const int
539 if ((MO.getType()) != MachineOperand::MO_Immediate)
540 return true;
541 O << "0x" << Twine::utohexstr(MO.getImm());
542 return false;
543 case 'x': // hex const int (low 16 bits)
544 if ((MO.getType()) != MachineOperand::MO_Immediate)
545 return true;
546 O << "0x" << Twine::utohexstr(MO.getImm() & 0xffff);
547 return false;
548 case 'd': // decimal const int
549 if ((MO.getType()) != MachineOperand::MO_Immediate)
550 return true;
551 O << MO.getImm();
552 return false;
553 case 'm': // decimal const int minus 1
554 if ((MO.getType()) != MachineOperand::MO_Immediate)
555 return true;
556 O << MO.getImm() - 1;
557 return false;
558 case 'y': // exact log2
559 if ((MO.getType()) != MachineOperand::MO_Immediate)
560 return true;
561 if (!isPowerOf2_64(MO.getImm()))
562 return true;
563 O << Log2_64(MO.getImm());
564 return false;
565 case 'z':
566 // $0 if zero, regular printing otherwise
567 if (MO.getType() == MachineOperand::MO_Immediate && MO.getImm() == 0) {
568 O << "$0";
569 return false;
570 }
571 // If not, call printOperand as normal.
572 break;
573 case 'D': // Second part of a double word register operand
574 case 'L': // Low order register of a double word register operand
575 case 'M': // High order register of a double word register operand
576 {
577 if (OpNum == 0)
578 return true;
579 const MachineOperand &FlagsOP = MI->getOperand(OpNum - 1);
580 if (!FlagsOP.isImm())
581 return true;
582 unsigned Flags = FlagsOP.getImm();
583 unsigned NumVals = InlineAsm::getNumOperandRegisters(Flags);
584 // Number of registers represented by this operand. We are looking
585 // for 2 for 32 bit mode and 1 for 64 bit mode.
586 if (NumVals != 2) {
587 if (Subtarget->isGP64bit() && NumVals == 1 && MO.isReg()) {
588 unsigned Reg = MO.getReg();
589 O << '$' << MipsInstPrinter::getRegisterName(Reg);
590 return false;
591 }
592 return true;
593 }
594
595 unsigned RegOp = OpNum;
596 if (!Subtarget->isGP64bit()){
597 // Endianness reverses which register holds the high or low value
598 // between M and L.
599 switch(ExtraCode[0]) {
600 case 'M':
601 RegOp = (Subtarget->isLittle()) ? OpNum + 1 : OpNum;
602 break;
603 case 'L':
604 RegOp = (Subtarget->isLittle()) ? OpNum : OpNum + 1;
605 break;
606 case 'D': // Always the second part
607 RegOp = OpNum + 1;
608 }
609 if (RegOp >= MI->getNumOperands())
610 return true;
611 const MachineOperand &MO = MI->getOperand(RegOp);
612 if (!MO.isReg())
613 return true;
614 unsigned Reg = MO.getReg();
615 O << '$' << MipsInstPrinter::getRegisterName(Reg);
616 return false;
617 }
618 break;
619 }
620 case 'w':
621 // Print MSA registers for the 'f' constraint
622 // In LLVM, the 'w' modifier doesn't need to do anything.
623 // We can just call printOperand as normal.
624 break;
625 }
626 }
627
628 printOperand(MI, OpNum, O);
629 return false;
630}

632bool MipsAsmPrinter::PrintAsmMemoryOperand(const MachineInstr *MI,
633 unsigned OpNum,
634 const char *ExtraCode,
635 raw_ostream &O) {
636 assert(OpNum + 1 < MI->getNumOperands() && "Insufficient operands");
637 const MachineOperand &BaseMO = MI->getOperand(OpNum);
638 const MachineOperand &OffsetMO = MI->getOperand(OpNum + 1);
639 assert(BaseMO.isReg() && "Unexpected base pointer for inline asm memory operand.");
640 assert(OffsetMO.isImm() && "Unexpected offset for inline asm memory operand.");
641 int Offset = OffsetMO.getImm();
642
643 // Currently we are expecting either no ExtraCode or 'D','M','L'.
644 if (ExtraCode) {
645 switch (ExtraCode[0]) {
646 case 'D':
647 Offset += 4;
648 break;
649 case 'M':
650 if (Subtarget->isLittle())
651 Offset += 4;
652 break;
653 case 'L':
654 if (!Subtarget->isLittle())
655 Offset += 4;
656 break;
657 default:
658 return true; // Unknown modifier.
659 }
660 }
661
662 O << Offset << "($" << MipsInstPrinter::getRegisterName(BaseMO.getReg())
663 << ")";
664
665 return false;
666}

668void MipsAsmPrinter::printOperand(const MachineInstr *MI, int opNum,
669 raw_ostream &O) {
670 const MachineOperand &MO = MI->getOperand(opNum);
671 bool closeP = false;
672
673 if (MO.getTargetFlags())
674 closeP = true;
675
676 switch(MO.getTargetFlags()) {
677 case MipsII::MO_GPREL: O << "%gp_rel("; break;
678 case MipsII::MO_GOT_CALL: O << "%call16("; break;
679 case MipsII::MO_GOT: O << "%got("; break;
680 case MipsII::MO_ABS_HI: O << "%hi("; break;
681 case MipsII::MO_ABS_LO: O << "%lo("; break;
682 case MipsII::MO_HIGHER: O << "%higher("; break;
683 case MipsII::MO_HIGHEST: O << "%highest(("; break;
684 case MipsII::MO_TLSGD: O << "%tlsgd("; break;
685 case MipsII::MO_GOTTPREL: O << "%gottprel("; break;
686 case MipsII::MO_TPREL_HI: O << "%tprel_hi("; break;
687 case MipsII::MO_TPREL_LO: O << "%tprel_lo("; break;
688 case MipsII::MO_GPOFF_HI: O << "%hi(%neg(%gp_rel("; break;
689 case MipsII::MO_GPOFF_LO: O << "%lo(%neg(%gp_rel("; break;
690 case MipsII::MO_CAPTABLE_OFF_HI: O << "%hi(%neg(%captab_rel("; break;
691 case MipsII::MO_CAPTABLE_OFF_LO: O << "%lo(%neg(%captab_rel("; break;
692 case MipsII::MO_GOT_DISP: O << "%got_disp("; break;
693 case MipsII::MO_GOT_PAGE: O << "%got_page("; break;
694 case MipsII::MO_GOT_OFST: O << "%got_ofst("; break;
695
696 case MipsII::MO_CAPTAB11: O << "%captab("; break;
697 case MipsII::MO_CAPTAB_HI16: O << "%captab_hi("; break;
698 case MipsII::MO_CAPTAB_LO16: O << "%captab_lo("; break;
699 case MipsII::MO_CAPTAB_CALL11: O << "%capcall("; break;
700 case MipsII::MO_CAPTAB_CALL_HI16: O << "%capcall_hi("; break;
701 case MipsII::MO_CAPTAB_CALL_LO16: O << "%capcall_lo("; break;
702 }
703
704 switch (MO.getType()) {
705 case MachineOperand::MO_Register:
706 O << '$'
707 << StringRef(MipsInstPrinter::getRegisterName(MO.getReg())).lower();
708 break;
709
710 case MachineOperand::MO_Immediate:
711 O << MO.getImm();
712 break;
713
714 case MachineOperand::MO_MachineBasicBlock:
715 MO.getMBB()->getSymbol()->print(O, MAI);
716 return;
717
718 case MachineOperand::MO_GlobalAddress:
719 PrintSymbolOperand(MO, O);
720 break;
721
722 case MachineOperand::MO_BlockAddress: {
723 MCSymbol *BA = GetBlockAddressSymbol(MO.getBlockAddress());
724 O << BA->getName();
725 break;
726 }
727
728 case MachineOperand::MO_ConstantPoolIndex:
729 O << getDataLayout().getPrivateGlobalPrefix() << "CPI"
730 << getFunctionNumber() << "_" << MO.getIndex();
731 if (MO.getOffset())
732 O << "+" << MO.getOffset();
733 break;
734
735 default:
736 llvm_unreachable("<unknown operand type>");
737 }
738
739 if (closeP) O << ")";
740}

789void MipsAsmPrinter::EmitStartOfAsmFile(Module &M) {
790 MipsTargetStreamer &TS = getTargetStreamer();
791
792 // MipsTargetStreamer has an initialization order problem when emitting an
793 // object file directly (see MipsTargetELFStreamer for full details). Work
794 // around it by re-initializing the PIC state here.
795 TS.setPic(OutContext.getObjectFileInfo()->isPositionIndependent());
796
797 // Compute MIPS architecture attributes based on the default subtarget
798 // that we'd have constructed. Module level directives aren't LTO
799 // clean anyhow.
800 // FIXME: For ifunc related functions we could iterate over and look
801 // for a feature string that doesn't match the default one.
802 const Triple &TT = TM.getTargetTriple();
803 StringRef CPU = MIPS_MC::selectMipsCPU(TT, TM.getTargetCPU());
804 StringRef FS = TM.getTargetFeatureString();
805 const MipsTargetMachine &MTM = static_cast<const MipsTargetMachine &>(TM);
806 const MipsSubtarget STI(TT, CPU, FS, MTM.isLittleEndian(), MTM, 0);
807
808 bool IsABICalls = STI.isABICalls();
809 const MipsABIInfo &ABI = MTM.getABI();
810 if (IsABICalls) {
811 TS.emitDirectiveAbiCalls();
812 // FIXME: This condition should be a lot more complicated that it is here.
813 // Ideally it should test for properties of the ABI and not the ABI
814 // itself.
815 // For the moment, I'm only correcting enough to make MIPS-IV work.
816 if (!isPositionIndependent() && STI.hasSym32())
817 TS.emitDirectiveOptionPic0();
818 }
819
820 // Tell the assembler which ABI we are using
821 std::string SectionName = std::string(".mdebug.") + getCurrentABIString();
822 OutStreamer->SwitchSection(
823 OutContext.getELFSection(SectionName, ELF::SHT_PROGBITS, 0));
824
825 // NaN: At the moment we only support:
826 // 1. .nan legacy (default)
827 // 2. .nan 2008
828 STI.isNaN2008() ? TS.emitDirectiveNaN2008()
829 : TS.emitDirectiveNaNLegacy();
830
831 // TODO: handle O64 ABI
832
833 TS.updateABIInfo(STI);
834
835 // We should always emit a '.module fp=...' but binutils 2.24 does not accept
836 // it. We therefore emit it when it contradicts the ABI defaults (-mfpxx or
837 // -mfp64) and omit it otherwise.
838 if ((ABI.IsO32() && (STI.isABI_FPXX() || STI.isFP64bit())) ||
839 STI.useSoftFloat())
840 TS.emitDirectiveModuleFP();
841
842 // We should always emit a '.module [no]oddspreg' but binutils 2.24 does not
843 // accept it. We therefore emit it when it contradicts the default or an
844 // option has changed the default (i.e. FPXX) and omit it otherwise.
845 if (ABI.IsO32() && (!STI.useOddSPReg() || STI.isABI_FPXX()))
846 TS.emitDirectiveModuleOddSPReg();
847}

849void MipsAsmPrinter::emitInlineAsmStart() const {
850 MipsTargetStreamer &TS = getTargetStreamer();
851
852 // GCC's choice of assembler options for inline assembly code ('at', 'macro'
853 // and 'reorder') is different from LLVM's choice for generated code ('noat',
854 // 'nomacro' and 'noreorder').
855 // In order to maintain compatibility with inline assembly code which depends
856 // on GCC's assembler options being used, we have to switch to those options
857 // for the duration of the inline assembly block and then switch back.
858 TS.emitDirectiveSetPush();
859 TS.emitDirectiveSetAt();
860 TS.emitDirectiveSetMacro();
861 TS.emitDirectiveSetReorder();
862 OutStreamer->AddBlankLine();
863}

867 OutStreamer->AddBlankLine();
868 getTargetStreamer().emitDirectiveSetPop();

887void MipsAsmPrinter::EmitInstrRegReg(const MCSubtargetInfo &STI,
888 unsigned Opcode, unsigned Reg1,

891 //

931void MipsAsmPrinter::EmitSwapFPIntParams(const MCSubtargetInfo &STI,
932 Mips16HardFloatInfo::FPParamVariant PV,
933 bool LE, bool ToFP) {
934 using namespace Mips16HardFloatInfo;
935
936 unsigned MovOpc = ToFP ? Mips::MTC1 : Mips::MFC1;
937 switch (PV) {
938 case FSig:
939 EmitInstrRegReg(STI, MovOpc, Mips::A0, Mips::F12);
940 break;
941 case FFSig:
942 EmitMovFPIntPair(STI, MovOpc, Mips::A0, Mips::A1, Mips::F12, Mips::F14, LE);
943 break;
944 case FDSig:
945 EmitInstrRegReg(STI, MovOpc, Mips::A0, Mips::F12);
946 EmitMovFPIntPair(STI, MovOpc, Mips::A2, Mips::A3, Mips::F14, Mips::F15, LE);
947 break;
948 case DSig:
949 EmitMovFPIntPair(STI, MovOpc, Mips::A0, Mips::A1, Mips::F12, Mips::F13, LE);
950 break;
951 case DDSig:
952 EmitMovFPIntPair(STI, MovOpc, Mips::A0, Mips::A1, Mips::F12, Mips::F13, LE);
953 EmitMovFPIntPair(STI, MovOpc, Mips::A2, Mips::A3, Mips::F14, Mips::F15, LE);
954 break;
955 case DFSig:
956 EmitMovFPIntPair(STI, MovOpc, Mips::A0, Mips::A1, Mips::F12, Mips::F13, LE);
957 EmitInstrRegReg(STI, MovOpc, Mips::A2, Mips::F14);
958 break;
959 case NoSig:
960 return;
961 }
962}

964void MipsAsmPrinter::EmitSwapFPIntRetval(
965 const MCSubtargetInfo &STI, Mips16HardFloatInfo::FPReturnVariant RV,
966 bool LE) {
967 using namespace Mips16HardFloatInfo;
968
969 unsigned MovOpc = Mips::MFC1;
970 switch (RV) {
971 case FRet:
972 EmitInstrRegReg(STI, MovOpc, Mips::V0, Mips::F0);
973 break;
974 case DRet:
975 EmitMovFPIntPair(STI, MovOpc, Mips::V0, Mips::V1, Mips::F0, Mips::F1, LE);
976 break;
977 case CFRet:
978 EmitMovFPIntPair(STI, MovOpc, Mips::V0, Mips::V1, Mips::F0, Mips::F1, LE);
979 break;
980 case CDRet:
981 EmitMovFPIntPair(STI, MovOpc, Mips::V0, Mips::V1, Mips::F0, Mips::F1, LE);
982 EmitMovFPIntPair(STI, MovOpc, Mips::A0, Mips::A1, Mips::F2, Mips::F3, LE);
983 break;
984 case NoFPRet:
985 break;
986 }

989void MipsAsmPrinter::EmitFPCallStub(
990 const char *Symbol, const Mips16HardFloatInfo::FuncSignature *Signature) {
991 using namespace Mips16HardFloatInfo;
992
993 MCSymbol *MSymbol = OutContext.getOrCreateSymbol(StringRef(Symbol));
994 bool LE = getDataLayout().isLittleEndian();
995 // Construct a local MCSubtargetInfo here.
996 // This is because the MachineFunction won't exist (but have not yet been
997 // freed) and since we're at the global level we can use the default
998 // constructed subtarget.
999 std::unique_ptr<MCSubtargetInfo> STI(TM.getTarget().createMCSubtargetInfo(
1000 TM.getTargetTriple().str(), TM.getTargetCPU(),
1001 TM.getTargetFeatureString()));
1002
1003 //
1004 // .global xxxx
1005 //
1006 OutStreamer->EmitSymbolAttribute(MSymbol, MCSA_Global);
1007 const char *RetType;
1008 //
1009 // make the comment field identifying the return and parameter
1010 // types of the floating point stub
1011 // # Stub function to call rettype xxxx (params)
1012 //
1013 switch (Signature->RetSig) {
1014 case FRet:
1015 RetType = "float";
1016 break;
1017 case DRet:
1018 RetType = "double";
1019 break;
1020 case CFRet:
1021 RetType = "complex";
1022 break;
1023 case CDRet:
1024 RetType = "double complex";
1025 break;
1026 case NoFPRet:
1027 RetType = "";
1028 break;
1029 }
1030 const char *Parms;
1031 switch (Signature->ParamSig) {
1032 case FSig:
1033 Parms = "float";
1034 break;
1035 case FFSig:
1036 Parms = "float, float";
1037 break;
1038 case FDSig:
1039 Parms = "float, double";
1040 break;
1041 case DSig:
1042 Parms = "double";
1043 break;
1044 case DDSig:
1045 Parms = "double, double";
1046 break;
1047 case DFSig:
1048 Parms = "double, float";
1049 break;
1050 case NoSig:
1051 Parms = "";
1052 break;
1053 }
1054 OutStreamer->AddComment("\t# Stub function to call " + Twine(RetType) + " " +
1055 Twine(Symbol) + " (" + Twine(Parms) + ")");
1056 //
1057 // probably not necessary but we save and restore the current section state
1058 //
1059 OutStreamer->PushSection();
1060 //
1061 // .section mips16.call.fpxxxx,"ax",@progbits
1062 //
1063 MCSectionELF *M = OutContext.getELFSection(
1064 ".mips16.call.fp." + std::string(Symbol), ELF::SHT_PROGBITS,
1065 ELF::SHF_ALLOC | ELF::SHF_EXECINSTR);
1066 OutStreamer->SwitchSection(M, nullptr);
1067 //
1068 // .align 2
1069 //
1070 OutStreamer->EmitValueToAlignment(4);
1071 MipsTargetStreamer &TS = getTargetStreamer();
1072 //
1073 // .set nomips16
1074 // .set nomicromips
1075 //
1076 TS.emitDirectiveSetNoMips16();
1077 TS.emitDirectiveSetNoMicroMips();
1078 //
1079 // .ent __call_stub_fp_xxxx
1080 // .type __call_stub_fp_xxxx,@function
1081 // __call_stub_fp_xxxx:
1082 //
1083 std::string x = "__call_stub_fp_" + std::string(Symbol);
1084 MCSymbolELF *Stub =
1085 cast<MCSymbolELF>(OutContext.getOrCreateSymbol(StringRef(x)));
1086 TS.emitDirectiveEnt(*Stub);
1087 MCSymbol *MType =
1088 OutContext.getOrCreateSymbol("__call_stub_fp_" + Twine(Symbol));
1089 OutStreamer->EmitSymbolAttribute(MType, MCSA_ELF_TypeFunction);
1090 OutStreamer->EmitLabel(Stub);
1091
1092 // Only handle non-pic for now.
1093 assert(!isPositionIndependent() &&
1094 "should not be here if we are compiling pic");
1095 TS.emitDirectiveSetReorder();
1096 //
1097 // We need to add a MipsMCExpr class to MCTargetDesc to fully implement
1098 // stubs without raw text but this current patch is for compiler generated
1099 // functions and they all return some value.
1100 // The calling sequence for non pic is different in that case and we need
1101 // to implement %lo and %hi in order to handle the case of no return value
1102 // See the corresponding method in Mips16HardFloat for details.
1103 //
1104 // mov the return address to S2.
1105 // we have no stack space to store it and we are about to make another call.
1106 // We need to make sure that the enclosing function knows to save S2
1107 // This should have already been handled.
1108 //
1109 // Mov $18, $31
1110
1111 EmitInstrRegRegReg(*STI, Mips::OR, Mips::S2, Mips::RA, Mips::ZERO);
1112
1113 EmitSwapFPIntParams(*STI, Signature->ParamSig, LE, true);
1114
1115 // Jal xxxx
1116 //
1117 EmitJal(*STI, MSymbol);
1118
1119 // fix return values
1120 EmitSwapFPIntRetval(*STI, Signature->RetSig, LE);
1121 //
1122 // do the return
1123 // if (Signature->RetSig == NoFPRet)
1124 // llvm_unreachable("should not be any stubs here with no return value");
1125 // else
1126 EmitInstrReg(*STI, Mips::JR, Mips::S2);
1127
1128 MCSymbol *Tmp = OutContext.createTempSymbol();
1129 OutStreamer->EmitLabel(Tmp);
1130 const MCSymbolRefExpr *E = MCSymbolRefExpr::create(Stub, OutContext);
1131 const MCSymbolRefExpr *T = MCSymbolRefExpr::create(Tmp, OutContext);
1132 const MCExpr *T_min_E = MCBinaryExpr::createSub(T, E, OutContext);
1133 OutStreamer->emitELFSize(Stub, T_min_E);
1134 TS.emitDirectiveEnd(x);
1135 OutStreamer->PopSection();
1136}

1154void MipsAsmPrinter::EmitSled(const MachineInstr &MI, SledKind Kind) {
1155 const uint8_t NoopsInSledCount = Subtarget->isGP64bit() ? 15 : 11;
1156 // For mips32 we want to emit the following pattern:
1157 //
1158 // .Lxray_sled_N:
1159 // ALIGN
1160 // B .tmpN
1161 // 11 NOP instructions (44 bytes)
1162 // ADDIU T9, T9, 52
1163 // .tmpN
1164 //
1165 // We need the 44 bytes (11 instructions) because at runtime, we'd
1166 // be patching over the full 48 bytes (12 instructions) with the following
1167 // pattern:
1168 //
1169 // ADDIU SP, SP, -8
1170 // NOP
1171 // SW RA, 4(SP)
1172 // SW T9, 0(SP)
1173 // LUI T9, %hi(__xray_FunctionEntry/Exit)
1174 // ORI T9, T9, %lo(__xray_FunctionEntry/Exit)
1175 // LUI T0, %hi(function_id)
1176 // JALR T9
1177 // ORI T0, T0, %lo(function_id)
1178 // LW T9, 0(SP)
1179 // LW RA, 4(SP)
1180 // ADDIU SP, SP, 8
1181 //
1182 // We add 52 bytes to t9 because we want to adjust the function pointer to
1183 // the actual start of function i.e. the address just after the noop sled.
1184 // We do this because gp displacement relocation is emitted at the start of
1185 // of the function i.e after the nop sled and to correctly calculate the
1186 // global offset table address, t9 must hold the address of the instruction
1187 // containing the gp displacement relocation.
1188 // FIXME: Is this correct for the static relocation model?
1189 //
1190 // For mips64 we want to emit the following pattern:
1191 //
1192 // .Lxray_sled_N:
1193 // ALIGN
1194 // B .tmpN
1195 // 15 NOP instructions (60 bytes)
1196 // .tmpN
1197 //
1198 // We need the 60 bytes (15 instructions) because at runtime, we'd
1199 // be patching over the full 64 bytes (16 instructions) with the following
1200 // pattern:
1201 //
1202 // DADDIU SP, SP, -16
1203 // NOP
1204 // SD RA, 8(SP)
1205 // SD T9, 0(SP)
1206 // LUI T9, %highest(__xray_FunctionEntry/Exit)
1207 // ORI T9, T9, %higher(__xray_FunctionEntry/Exit)
1208 // DSLL T9, T9, 16
1209 // ORI T9, T9, %hi(__xray_FunctionEntry/Exit)
1210 // DSLL T9, T9, 16
1211 // ORI T9, T9, %lo(__xray_FunctionEntry/Exit)
1212 // LUI T0, %hi(function_id)
1213 // JALR T9
1214 // ADDIU T0, T0, %lo(function_id)
1215 // LD T9, 0(SP)
1216 // LD RA, 8(SP)
1217 // DADDIU SP, SP, 16
1218 //
1219 OutStreamer->EmitCodeAlignment(4);
1220 auto CurSled = OutContext.createTempSymbol("xray_sled_", true);
1221 OutStreamer->EmitLabel(CurSled);
1222 auto Target = OutContext.createTempSymbol();
1223
1224 // Emit "B .tmpN" instruction, which jumps over the nop sled to the actual
1225 // start of function
1226 const MCExpr *TargetExpr = MCSymbolRefExpr::create(
1227 Target, MCSymbolRefExpr::VariantKind::VK_None, OutContext);
1228 EmitToStreamer(*OutStreamer, MCInstBuilder(Mips::BEQ)
1229 .addReg(Mips::ZERO)
1230 .addReg(Mips::ZERO)
1231 .addExpr(TargetExpr));
1232
1233 for (int8_t I = 0; I < NoopsInSledCount; I++)
1234 EmitToStreamer(*OutStreamer, MCInstBuilder(Mips::SLL)
1235 .addReg(Mips::ZERO)
1236 .addReg(Mips::ZERO)
1237 .addImm(0));
1238
1239 OutStreamer->EmitLabel(Target);
1240
1241 if (!Subtarget->isGP64bit()) {
1242 EmitToStreamer(*OutStreamer,
1243 MCInstBuilder(Mips::ADDiu)
1244 .addReg(Mips::T9)
1245 .addReg(Mips::T9)
1246 .addImm(0x34));
1247 }
1248
1249 recordSled(CurSled, MI, Kind);
1250}

1290// Align all targets of indirect branches on bundle size. Used only if target
1291// is NaCl.
1292void MipsAsmPrinter::NaClAlignIndirectJumpTargets(MachineFunction &MF) {
1293 // Align all blocks that are jumped to through jump table.
1294 if (MachineJumpTableInfo *JtInfo = MF.getJumpTableInfo()) {
1295 const std::vector<MachineJumpTableEntry> &JT = JtInfo->getJumpTables();
1296 for (unsigned I = 0; I < JT.size(); ++I) {
1297 const std::vector<MachineBasicBlock*> &MBBs = JT[I].MBBs;
1298
1299 for (unsigned J = 0; J < MBBs.size(); ++J)
1300 MBBs[J]->setAlignment(MIPS_NACL_BUNDLE_ALIGN);
1301 }
1302 }
1303
1304 // If basic block address is taken, block can be target of indirect branch.
1305 for (auto &MBB : MF) {
1306 if (MBB.hasAddressTaken())
1307 MBB.setAlignment(MIPS_NACL_BUNDLE_ALIGN);
1308 }
1309}
