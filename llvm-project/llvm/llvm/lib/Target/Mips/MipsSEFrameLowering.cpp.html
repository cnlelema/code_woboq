56static std::pair<unsigned, unsigned> getMFHiLoOpc(unsigned Src) {
57 if (Mips::ACC64RegClass.contains(Src))
58 return std::make_pair((unsigned)Mips::PseudoMFHI,
59 (unsigned)Mips::PseudoMFLO);
60
61 if (Mips::ACC64DSPRegClass.contains(Src))
62 return std::make_pair((unsigned)Mips::MFHI_DSP, (unsigned)Mips::MFLO_DSP);
63
64 if (Mips::ACC128RegClass.contains(Src))
65 return std::make_pair((unsigned)Mips::PseudoMFHI64,
66 (unsigned)Mips::PseudoMFLO64);
67
68 return std::make_pair(0, 0);
69}

105ExpandPseudo::ExpandPseudo(MachineFunction &MF_)
106 : MF(MF_), MRI(MF.getRegInfo()),
107 Subtarget(static_cast<const MipsSubtarget &>(MF.getSubtarget())),
108 TII(*static_cast<const MipsSEInstrInfo *>(Subtarget.getInstrInfo())),

111bool ExpandPseudo::expand() {
112 bool Expanded = false;
113
114 for (auto &MBB : MF) {
115 for (Iter I = MBB.begin(), End = MBB.end(); I != End;)
116 Expanded |= expandInstr(MBB, I++);
117 }
118
119 return Expanded;
120}

122bool ExpandPseudo::expandInstr(MachineBasicBlock &MBB, Iter I) {
123 switch(I->getOpcode()) {
124 case Mips::LOAD_CCOND_DSP:
125 expandLoadCCond(MBB, I);
126 break;
127 case Mips::STORE_CCOND_DSP:
128 expandStoreCCond(MBB, I);
129 break;
130 case Mips::LOAD_ACC64:
131 case Mips::LOAD_ACC64DSP:
132 expandLoadACC(MBB, I, 4);
133 break;
134 case Mips::LOAD_ACC128:
135 expandLoadACC(MBB, I, 8);
136 break;
137 case Mips::STORE_ACC64:
138 expandStoreACC(MBB, I, Mips::PseudoMFHI, Mips::PseudoMFLO, 4);
139 break;
140 case Mips::STORE_ACC64DSP:
141 expandStoreACC(MBB, I, Mips::MFHI_DSP, Mips::MFLO_DSP, 4);
142 break;
143 case Mips::STORE_ACC128:
144 expandStoreACC(MBB, I, Mips::PseudoMFHI64, Mips::PseudoMFLO64, 8);
145 break;
146 case Mips::BuildPairF64:
147 if (expandBuildPairF64(MBB, I, false))
148 MBB.erase(I);
149 return false;
150 case Mips::BuildPairF64_64:
151 if (expandBuildPairF64(MBB, I, true))
152 MBB.erase(I);
153 return false;
154 case Mips::ExtractElementF64:
155 if (expandExtractElementF64(MBB, I, false))
156 MBB.erase(I);
157 return false;
158 case Mips::ExtractElementF64_64:
159 if (expandExtractElementF64(MBB, I, true))
160 MBB.erase(I);
161 return false;
162 case TargetOpcode::COPY:
163 if (!expandCopy(MBB, I))
164 return false;
165 break;
166 default:
167 return false;
168 }
169
170 MBB.erase(I);
171 return true;
172}

174void ExpandPseudo::expandLoadCCond(MachineBasicBlock &MBB, Iter I) {
175 // load $vr, FI
176 // copy ccond, $vr
177
178 assert(I->getOperand(0).isReg() && I->getOperand(1).isFI());
179
180 const TargetRegisterClass *RC = RegInfo.intRegClass(4);
181 unsigned VR = MRI.createVirtualRegister(RC);
182 unsigned Dst = I->getOperand(0).getReg(), FI = I->getOperand(1).getIndex();
183
184 TII.loadRegFromStack(MBB, I, VR, FI, RC, &RegInfo, 0);
185 BuildMI(MBB, I, I->getDebugLoc(), TII.get(TargetOpcode::COPY), Dst)
186 .addReg(VR, RegState::Kill);
187}
188
189void ExpandPseudo::expandStoreCCond(MachineBasicBlock &MBB, Iter I) {
190 // copy $vr, ccond
191 // store $vr, FI
192
193 assert(I->getOperand(0).isReg() && I->getOperand(1).isFI());
194
195 const TargetRegisterClass *RC = RegInfo.intRegClass(4);
196 unsigned VR = MRI.createVirtualRegister(RC);
197 unsigned Src = I->getOperand(0).getReg(), FI = I->getOperand(1).getIndex();
198
199 BuildMI(MBB, I, I->getDebugLoc(), TII.get(TargetOpcode::COPY), VR)
200 .addReg(Src, getKillRegState(I->getOperand(0).isKill()));
201 TII.storeRegToStack(MBB, I, VR, true, FI, RC, &RegInfo, 0);
202}

204void ExpandPseudo::expandLoadACC(MachineBasicBlock &MBB, Iter I,
205 unsigned RegSize) {
206 // load $vr0, FI
207 // copy lo, $vr0
208 // load $vr1, FI + 4
209 // copy hi, $vr1
210
211 assert(I->getOperand(0).isReg() && I->getOperand(1).isFI());
212
213 const TargetRegisterClass *RC = RegInfo.intRegClass(RegSize);
214 unsigned VR0 = MRI.createVirtualRegister(RC);
215 unsigned VR1 = MRI.createVirtualRegister(RC);
216 unsigned Dst = I->getOperand(0).getReg(), FI = I->getOperand(1).getIndex();
217 unsigned Lo = RegInfo.getSubReg(Dst, Mips::sub_lo);
218 unsigned Hi = RegInfo.getSubReg(Dst, Mips::sub_hi);
219 DebugLoc DL = I->getDebugLoc();
220 const MCInstrDesc &Desc = TII.get(TargetOpcode::COPY);
221
222 TII.loadRegFromStack(MBB, I, VR0, FI, RC, &RegInfo, 0);
223 BuildMI(MBB, I, DL, Desc, Lo).addReg(VR0, RegState::Kill);
224 TII.loadRegFromStack(MBB, I, VR1, FI, RC, &RegInfo, RegSize);
225 BuildMI(MBB, I, DL, Desc, Hi).addReg(VR1, RegState::Kill);
226}

228void ExpandPseudo::expandStoreACC(MachineBasicBlock &MBB, Iter I,
229 unsigned MFHiOpc, unsigned MFLoOpc,
230 unsigned RegSize) {
231 // mflo $vr0, src
232 // store $vr0, FI
233 // mfhi $vr1, src
234 // store $vr1, FI + 4
235
236 assert(I->getOperand(0).isReg() && I->getOperand(1).isFI());
237
238 const TargetRegisterClass *RC = RegInfo.intRegClass(RegSize);
239 unsigned VR0 = MRI.createVirtualRegister(RC);
240 unsigned VR1 = MRI.createVirtualRegister(RC);
241 unsigned Src = I->getOperand(0).getReg(), FI = I->getOperand(1).getIndex();
242 unsigned SrcKill = getKillRegState(I->getOperand(0).isKill());
243 DebugLoc DL = I->getDebugLoc();
244
245 BuildMI(MBB, I, DL, TII.get(MFLoOpc), VR0).addReg(Src);
246 TII.storeRegToStack(MBB, I, VR0, true, FI, RC, &RegInfo, 0);
247 BuildMI(MBB, I, DL, TII.get(MFHiOpc), VR1).addReg(Src, SrcKill);
248 TII.storeRegToStack(MBB, I, VR1, true, FI, RC, &RegInfo, RegSize);
249}

251bool ExpandPseudo::expandCopy(MachineBasicBlock &MBB, Iter I) {
252 unsigned Src = I->getOperand(1).getReg();
253 std::pair<unsigned, unsigned> Opcodes = getMFHiLoOpc(Src);
254
255 if (!Opcodes.first)
256 return false;
257
258 return expandCopyACC(MBB, I, Opcodes.first, Opcodes.second);
259}

261bool ExpandPseudo::expandCopyACC(MachineBasicBlock &MBB, Iter I,
262 unsigned MFHiOpc, unsigned MFLoOpc) {
263 // mflo $vr0, src
264 // copy dst_lo, $vr0
265 // mfhi $vr1, src
266 // copy dst_hi, $vr1
267
268 unsigned Dst = I->getOperand(0).getReg(), Src = I->getOperand(1).getReg();
269 const TargetRegisterClass *DstRC = RegInfo.getMinimalPhysRegClass(Dst);
270 unsigned VRegSize = RegInfo.getRegSizeInBits(*DstRC) / 16;
271 const TargetRegisterClass *RC = RegInfo.intRegClass(VRegSize);
272 unsigned VR0 = MRI.createVirtualRegister(RC);
273 unsigned VR1 = MRI.createVirtualRegister(RC);
274 unsigned SrcKill = getKillRegState(I->getOperand(1).isKill());
275 unsigned DstLo = RegInfo.getSubReg(Dst, Mips::sub_lo);
276 unsigned DstHi = RegInfo.getSubReg(Dst, Mips::sub_hi);
277 DebugLoc DL = I->getDebugLoc();
278
279 BuildMI(MBB, I, DL, TII.get(MFLoOpc), VR0).addReg(Src);
280 BuildMI(MBB, I, DL, TII.get(TargetOpcode::COPY), DstLo)
281 .addReg(VR0, RegState::Kill);
282 BuildMI(MBB, I, DL, TII.get(MFHiOpc), VR1).addReg(Src, SrcKill);
283 BuildMI(MBB, I, DL, TII.get(TargetOpcode::COPY), DstHi)
284 .addReg(VR1, RegState::Kill);
285 return true;
286}

288/// This method expands the same instruction that MipsSEInstrInfo::
289/// expandBuildPairF64 does, for the case when ABI is fpxx and mthc1 is not
290/// available and the case where the ABI is FP64A. It is implemented here
291/// because frame indexes are eliminated before MipsSEInstrInfo::
292/// expandBuildPairF64 is called.
293bool ExpandPseudo::expandBuildPairF64(MachineBasicBlock &MBB,
294 MachineBasicBlock::iterator I,
295 bool FP64) const {
296 // For fpxx and when mthc1 is not available, use:
297 // spill + reload via ldc1
298 //
299 // The case where dmtc1 is available doesn't need to be handled here
300 // because it never creates a BuildPairF64 node.
301 //
302 // The FP64A ABI (fp64 with nooddspreg) must also use a spill/reload sequence
303 // for odd-numbered double precision values (because the lower 32-bits is
304 // transferred with mtc1 which is redirected to the upper half of the even
305 // register). Unfortunately, we have to make this decision before register
306 // allocation so for now we use a spill/reload sequence for all
307 // double-precision values in regardless of being an odd/even register.
308 //
309 // For the cases that should be covered here MipsSEISelDAGToDAG adds $sp as
310 // implicit operand, so other passes (like ShrinkWrapping) are aware that
311 // stack is used.
312 if (I->getNumOperands() == 4 && I->getOperand(3).isReg()
313 && I->getOperand(3).getReg() == Mips::SP) {
314 unsigned DstReg = I->getOperand(0).getReg();
315 unsigned LoReg = I->getOperand(1).getReg();
316 unsigned HiReg = I->getOperand(2).getReg();
317
318 // It should be impossible to have FGR64 on MIPS-II or MIPS32r1 (which are
319 // the cases where mthc1 is not available). 64-bit architectures and
320 // MIPS32r2 or later can use FGR64 though.
321 assert(Subtarget.isGP64bit() || Subtarget.hasMTHC1() ||
322 !Subtarget.isFP64bit());
323
324 const TargetRegisterClass *RC = &Mips::GPR32RegClass;
325 const TargetRegisterClass *RC2 =
326 FP64 ? &Mips::FGR64RegClass : &Mips::AFGR64RegClass;
327
328 // We re-use the same spill slot each time so that the stack frame doesn't
329 // grow too much in functions with a large number of moves.
330 int FI = MF.getInfo<MipsFunctionInfo>()->getMoveF64ViaSpillFI(RC2);
331 if (!Subtarget.isLittle())
332 std::swap(LoReg, HiReg);
333 TII.storeRegToStack(MBB, I, LoReg, I->getOperand(1).isKill(), FI, RC,
334 &RegInfo, 0);
335 TII.storeRegToStack(MBB, I, HiReg, I->getOperand(2).isKill(), FI, RC,
336 &RegInfo, 4);
337 TII.loadRegFromStack(MBB, I, DstReg, FI, RC2, &RegInfo, 0);
338 return true;
339 }
340
341 return false;
342}

344/// This method expands the same instruction that MipsSEInstrInfo::
345/// expandExtractElementF64 does, for the case when ABI is fpxx and mfhc1 is not
346/// available and the case where the ABI is FP64A. It is implemented here
347/// because frame indexes are eliminated before MipsSEInstrInfo::
348/// expandExtractElementF64 is called.
349bool ExpandPseudo::expandExtractElementF64(MachineBasicBlock &MBB,
350 MachineBasicBlock::iterator I,
351 bool FP64) const {
352 const MachineOperand &Op1 = I->getOperand(1);
353 const MachineOperand &Op2 = I->getOperand(2);
354
355 if ((Op1.isReg() && Op1.isUndef()) || (Op2.isReg() && Op2.isUndef())) {
356 unsigned DstReg = I->getOperand(0).getReg();
357 BuildMI(MBB, I, I->getDebugLoc(), TII.get(Mips::IMPLICIT_DEF), DstReg);
358 return true;
359 }
360
361 // For fpxx and when mfhc1 is not available, use:
362 // spill + reload via ldc1
363 //
364 // The case where dmfc1 is available doesn't need to be handled here
365 // because it never creates a ExtractElementF64 node.
366 //
367 // The FP64A ABI (fp64 with nooddspreg) must also use a spill/reload sequence
368 // for odd-numbered double precision values (because the lower 32-bits is
369 // transferred with mfc1 which is redirected to the upper half of the even
370 // register). Unfortunately, we have to make this decision before register
371 // allocation so for now we use a spill/reload sequence for all
372 // double-precision values in regardless of being an odd/even register.
373 //
374 // For the cases that should be covered here MipsSEISelDAGToDAG adds $sp as
375 // implicit operand, so other passes (like ShrinkWrapping) are aware that
376 // stack is used.
377 if (I->getNumOperands() == 4 && I->getOperand(3).isReg()
378 && I->getOperand(3).getReg() == Mips::SP) {
379 unsigned DstReg = I->getOperand(0).getReg();
380 unsigned SrcReg = Op1.getReg();
381 unsigned N = Op2.getImm();
382 int64_t Offset = 4 * (Subtarget.isLittle() ? N : (1 - N));
383
384 // It should be impossible to have FGR64 on MIPS-II or MIPS32r1 (which are
385 // the cases where mfhc1 is not available). 64-bit architectures and
386 // MIPS32r2 or later can use FGR64 though.
387 assert(Subtarget.isGP64bit() || Subtarget.hasMTHC1() ||
388 !Subtarget.isFP64bit());
389
390 const TargetRegisterClass *RC =
391 FP64 ? &Mips::FGR64RegClass : &Mips::AFGR64RegClass;
392 const TargetRegisterClass *RC2 = &Mips::GPR32RegClass;
393
394 // We re-use the same spill slot each time so that the stack frame doesn't
395 // grow too much in functions with a large number of moves.
396 int FI = MF.getInfo<MipsFunctionInfo>()->getMoveF64ViaSpillFI(RC);
397 TII.storeRegToStack(MBB, I, SrcReg, Op1.isKill(), FI, RC, &RegInfo, 0);
398 TII.loadRegFromStack(MBB, I, DstReg, FI, RC2, &RegInfo, Offset);
399 return true;
400 }
401
402 return false;
403}

408void MipsSEFrameLowering::emitPrologue(MachineFunction &MF,
409 MachineBasicBlock &MBB) const {
410 MachineFrameInfo &MFI = MF.getFrameInfo();
411 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
412
413 const MipsSEInstrInfo &TII =
414 *static_cast<const MipsSEInstrInfo *>(STI.getInstrInfo());
415 const MipsRegisterInfo &RegInfo =
416 *static_cast<const MipsRegisterInfo *>(STI.getRegisterInfo());
417
418 MachineBasicBlock::iterator MBBI = MBB.begin();
419 DebugLoc dl;
420 MipsABIInfo ABI = STI.getABI();
421 unsigned SP = ABI.GetStackPtr();
422 unsigned FP = ABI.GetFramePtr();
423 unsigned ZERO = ABI.GetNullPtr();
424 unsigned MOVE = ABI.GetSPMoveOp();
425 unsigned ADDiu = ABI.GetPtrAddiuOp();
426 unsigned AND = ABI.IsN64() ? Mips::AND64 : Mips::AND;
427
428 // C0 is always live in. In non-CHERI MIPS, C0 is treated as a read-only
429 // register that is implicitly read by loads and stores.
430 MBB.addLiveIn(Mips::DDC);
431
432 const TargetRegisterClass *RC = ABI.ArePtrs64bit() ?
433 &Mips::GPR64RegClass : &Mips::GPR32RegClass;
434
435 // First, compute final stack size.
436 uint64_t StackSize = MFI.getStackSize();
437
438 // No need to allocate space on the stack.
439 if (StackSize == 0 && !MFI.adjustsStack()) return;
440
441 MachineModuleInfo &MMI = MF.getMMI();
442 const MCRegisterInfo *MRI = MMI.getContext().getRegisterInfo();
443
444 // Adjust stack.
445 TII.adjustStackPtr(SP, -StackSize, MBB, MBBI);
446
447 // emit ".cfi_def_cfa_offset StackSize"
448 unsigned CFIIndex = MF.addFrameInst(
449 MCCFIInstruction::createDefCfaOffset(nullptr, -StackSize));
450 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
451 .addCFIIndex(CFIIndex);
452
453 if (MF.getFunction().hasFnAttribute("interrupt"))
454 emitInterruptPrologueStub(MF, MBB);
455
456 bool IsRASpilled = false;
457 const std::vector<CalleeSavedInfo> &CSI = MFI.getCalleeSavedInfo();
458
459 if (!CSI.empty()) {
460 // Find the instruction past the last instruction that saves a callee-saved
461 // register to the stack.
462 for (unsigned i = 0; i < CSI.size(); ++i)
463 ++MBBI;
464
465 // Iterate over list of callee-saved registers and emit .cfi_offset
466 // directives.
467 for (std::vector<CalleeSavedInfo>::const_iterator I = CSI.begin(),
468 E = CSI.end(); I != E; ++I) {
469 int64_t Offset = MFI.getObjectOffset(I->getFrameIdx());
470 unsigned Reg = I->getReg();
471 if (Reg == Mips::RA_64 || Reg == Mips::RA)
472 IsRASpilled = true;
473
474 // If Reg is a double precision register, emit two cfa_offsets,
475 // one for each of the paired single precision registers.
476 if (Mips::AFGR64RegClass.contains(Reg)) {
477 unsigned Reg0 =
478 MRI->getDwarfRegNum(RegInfo.getSubReg(Reg, Mips::sub_lo), true);
479 unsigned Reg1 =
480 MRI->getDwarfRegNum(RegInfo.getSubReg(Reg, Mips::sub_hi), true);
481
482 if (!STI.isLittle())
483 std::swap(Reg0, Reg1);
484
485 unsigned CFIIndex = MF.addFrameInst(
486 MCCFIInstruction::createOffset(nullptr, Reg0, Offset));
487 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
488 .addCFIIndex(CFIIndex);
489
490 CFIIndex = MF.addFrameInst(
491 MCCFIInstruction::createOffset(nullptr, Reg1, Offset + 4));
492 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
493 .addCFIIndex(CFIIndex);
494 } else if (Mips::FGR64RegClass.contains(Reg)) {
495 unsigned Reg0 = MRI->getDwarfRegNum(Reg, true);
496 unsigned Reg1 = MRI->getDwarfRegNum(Reg, true) + 1;
497
498 if (!STI.isLittle())
499 std::swap(Reg0, Reg1);
500
501 unsigned CFIIndex = MF.addFrameInst(
502 MCCFIInstruction::createOffset(nullptr, Reg0, Offset));
503 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
504 .addCFIIndex(CFIIndex);
505
506 CFIIndex = MF.addFrameInst(
507 MCCFIInstruction::createOffset(nullptr, Reg1, Offset + 4));
508 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
509 .addCFIIndex(CFIIndex);
510 } else {
511 // Reg is either in GPR32 or FGR32.
512 unsigned CFIIndex = MF.addFrameInst(MCCFIInstruction::createOffset(
513 nullptr, MRI->getDwarfRegNum(Reg, true), Offset));
514 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
515 .addCFIIndex(CFIIndex);
516 }
517 }
518 }
519
520 if (MipsFI->callsEhReturn()) {
521 // Insert instructions that spill eh data registers.
522 for (int I = 0; I < 4; ++I) {
523 if (!MBB.isLiveIn(ABI.GetEhDataReg(I)))
524 MBB.addLiveIn(ABI.GetEhDataReg(I));
525 TII.storeRegToStackSlot(MBB, MBBI, ABI.GetEhDataReg(I), false,
526 MipsFI->getEhDataRegFI(I), RC, &RegInfo);
527 }
528
529 // Emit .cfi_offset directives for eh data registers.
530 for (int I = 0; I < 4; ++I) {
531 int64_t Offset = MFI.getObjectOffset(MipsFI->getEhDataRegFI(I));
532 unsigned Reg = MRI->getDwarfRegNum(ABI.GetEhDataReg(I), true);
533 unsigned CFIIndex = MF.addFrameInst(
534 MCCFIInstruction::createOffset(nullptr, Reg, Offset));
535 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
536 .addCFIIndex(CFIIndex);
537 }
538 }
539
540 // if framepointer enabled, set it to point to the stack pointer.
541 if (hasFP(MF)) {
542 // Insert instruction "move $fp, $sp" at this location.
543 BuildMI(MBB, MBBI, dl, TII.get(MOVE), FP).addReg(SP).addReg(ZERO)
544 .setMIFlag(MachineInstr::FrameSetup);
545
546 // emit ".cfi_def_cfa_register $fp"
547 unsigned CFIIndex = MF.addFrameInst(MCCFIInstruction::createDefCfaRegister(
548 nullptr, MRI->getDwarfRegNum(FP, true)));
549 BuildMI(MBB, MBBI, dl, TII.get(TargetOpcode::CFI_INSTRUCTION))
550 .addCFIIndex(CFIIndex);
551
552 if (RegInfo.needsStackRealignment(MF)) {
553 // addiu $Reg, $zero, -MaxAlignment
554 // andi $sp, $sp, $Reg
555 unsigned VR = MF.getRegInfo().createVirtualRegister(RC);
556 assert(isInt<16>(MFI.getMaxAlignment()) &&
557 "Function's alignment size requirement is not supported.");
558 int MaxAlign = -(int)MFI.getMaxAlignment();
559 unsigned IntSP = SP;
560 if (ABI.IsCheriPureCap()) {
561 IntSP = MF.getRegInfo().createVirtualRegister(RC);
562 BuildMI(MBB, MBBI, dl, TII.get(Mips::CGetOffset), IntSP).addReg(SP);
563 }
564
565
566 BuildMI(MBB, MBBI, dl, TII.get(ADDiu), VR).addReg(ZERO) .addImm(MaxAlign);
567 BuildMI(MBB, MBBI, dl, TII.get(AND), IntSP).addReg(IntSP).addReg(VR);
568
569 if (ABI.IsCheriPureCap())
570 BuildMI(MBB, MBBI, dl, TII.get(Mips::CSetOffset), SP).addReg(SP)
571 .addReg(IntSP);
572
573 if (hasBP(MF)) {
574 // move $s7, $sp
575 unsigned BP = ABI.GetBasePtr();
576 BuildMI(MBB, MBBI, dl, TII.get(MOVE), BP)
577 .addReg(SP)
578 .addReg(ZERO);
579 }
580 }
581 }
582
583 // If we've spilled RA and we're targeting CHERI, then we want to use
584 // cjr $c16 for return, not jr $ra
585 if (CHERICFI && IsRASpilled && STI.isCheri())
586 for (auto &BB : MF)
587 for (auto &I : BB.terminators())
588 if (I.getOpcode() == Mips::RetRA) {
589 BuildMI(BB, (MachineBasicBlock::iterator)I, I.getDebugLoc(),
590 TII.get(Mips::CJR))
591 .addReg(Mips::C16);
592 I.eraseFromBundle();
593 break;
594 }
595
596}

598void MipsSEFrameLowering::emitInterruptPrologueStub(
599 MachineFunction &MF, MachineBasicBlock &MBB) const {
600 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
601 MachineBasicBlock::iterator MBBI = MBB.begin();
602 DebugLoc DL = MBBI != MBB.end() ? MBBI->getDebugLoc() : DebugLoc();
603
604 // Report an error the target doesn't support Mips32r2 or later.
605 // The epilogue relies on the use of the "ehb" to clear execution
606 // hazards. Pre R2 Mips relies on an implementation defined number
607 // of "ssnop"s to clear the execution hazard. Support for ssnop hazard
608 // clearing is not provided so reject that configuration.
609 if (!STI.hasMips32r2())
610 report_fatal_error(
611 "\"interrupt\" attribute is not supported on pre-MIPS32R2 or "
612 "MIPS16 targets.");
613
614 // The GP register contains the "user" value, so we cannot perform
615 // any gp relative loads until we restore the "kernel" or "system" gp
616 // value. Until support is written we shall only accept the static
617 // relocation model.
618 if ((STI.getRelocationModel() != Reloc::Static))
619 report_fatal_error("\"interrupt\" attribute is only supported for the "
620 "static relocation model on MIPS at the present time.");
621
622 if (!STI.isABI_O32() || STI.hasMips64())
623 report_fatal_error("\"interrupt\" attribute is only supported for the "
624 "O32 ABI on MIPS32R2+ at the present time.");
625
626 // Perform ISR handling like GCC
627 StringRef IntKind =
628 MF.getFunction().getFnAttribute("interrupt").getValueAsString();
629 const TargetRegisterClass *PtrRC = &Mips::GPR32RegClass;
630
631 // EIC interrupt handling needs to read the Cause register to disable
632 // interrupts.
633 if (IntKind == "eic") {
634 // Coprocessor registers are always live per se.
635 MBB.addLiveIn(Mips::COP013);
636 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MFC0), Mips::K0)
637 .addReg(Mips::COP013)
638 .addImm(0)
639 .setMIFlag(MachineInstr::FrameSetup);
640
641 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::EXT), Mips::K0)
642 .addReg(Mips::K0)
643 .addImm(10)
644 .addImm(6)
645 .setMIFlag(MachineInstr::FrameSetup);
646 }
647
648 // Fetch and spill EPC
649 MBB.addLiveIn(Mips::COP014);
650 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MFC0), Mips::K1)
651 .addReg(Mips::COP014)
652 .addImm(0)
653 .setMIFlag(MachineInstr::FrameSetup);
654
655 STI.getInstrInfo()->storeRegToStack(MBB, MBBI, Mips::K1, false,
656 MipsFI->getISRRegFI(0), PtrRC,
657 STI.getRegisterInfo(), 0);
658
659 // Fetch and Spill Status
660 MBB.addLiveIn(Mips::COP012);
661 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MFC0), Mips::K1)
662 .addReg(Mips::COP012)
663 .addImm(0)
664 .setMIFlag(MachineInstr::FrameSetup);
665
666 STI.getInstrInfo()->storeRegToStack(MBB, MBBI, Mips::K1, false,
667 MipsFI->getISRRegFI(1), PtrRC,
668 STI.getRegisterInfo(), 0);
669
670 // Build the configuration for disabling lower priority interrupts. Non EIC
671 // interrupts need to be masked off with zero, EIC from the Cause register.
672 unsigned InsPosition = 8;
673 unsigned InsSize = 0;
674 unsigned SrcReg = Mips::ZERO;
675
676 // If the interrupt we're tied to is the EIC, switch the source for the
677 // masking off interrupts to the cause register.
678 if (IntKind == "eic") {
679 SrcReg = Mips::K0;
680 InsPosition = 10;
681 InsSize = 6;
682 } else
683 InsSize = StringSwitch<unsigned>(IntKind)
684 .Case("sw0", 1)
685 .Case("sw1", 2)
686 .Case("hw0", 3)
687 .Case("hw1", 4)
688 .Case("hw2", 5)
689 .Case("hw3", 6)
690 .Case("hw4", 7)
691 .Case("hw5", 8)
692 .Default(0);
693 assert(InsSize != 0 && "Unknown interrupt type!");
694
695 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::INS), Mips::K1)
696 .addReg(SrcReg)
697 .addImm(InsPosition)
698 .addImm(InsSize)
699 .addReg(Mips::K1)
700 .setMIFlag(MachineInstr::FrameSetup);
701
702 // Mask off KSU, ERL, EXL
703 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::INS), Mips::K1)
704 .addReg(Mips::ZERO)
705 .addImm(1)
706 .addImm(4)
707 .addReg(Mips::K1)
708 .setMIFlag(MachineInstr::FrameSetup);
709
710 // Disable the FPU as we are not spilling those register sets.
711 if (!STI.useSoftFloat())
712 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::INS), Mips::K1)
713 .addReg(Mips::ZERO)
714 .addImm(29)
715 .addImm(1)
716 .addReg(Mips::K1)
717 .setMIFlag(MachineInstr::FrameSetup);
718
719 // Set the new status
720 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MTC0), Mips::COP012)
721 .addReg(Mips::K1)
722 .addImm(0)
723 .setMIFlag(MachineInstr::FrameSetup);
724}

726void MipsSEFrameLowering::emitEpilogue(MachineFunction &MF,
727 MachineBasicBlock &MBB) const {
728 MachineBasicBlock::iterator MBBI = MBB.getFirstTerminator();
729 MachineFrameInfo &MFI = MF.getFrameInfo();
730 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
731
732 const MipsSEInstrInfo &TII =
733 *static_cast<const MipsSEInstrInfo *>(STI.getInstrInfo());
734 const MipsRegisterInfo &RegInfo =
735 *static_cast<const MipsRegisterInfo *>(STI.getRegisterInfo());
736
737 DebugLoc DL = MBBI != MBB.end() ? MBBI->getDebugLoc() : DebugLoc();
738 MipsABIInfo ABI = STI.getABI();
739 unsigned SP = ABI.GetStackPtr();
740 unsigned FP = ABI.GetFramePtr();
741 unsigned ZERO = ABI.GetNullPtr();
742 unsigned MOVE = ABI.GetSPMoveOp();
743
744 // No need to clean up if we didn't allocate space on the stack.
745 if (MFI.getStackSize() == 0 && !MFI.adjustsStack()) return;
746
747 // if framepointer enabled, restore the stack pointer.
748 if (hasFP(MF)) {
749 // Find the first instruction that restores a callee-saved register.
750 MachineBasicBlock::iterator I = MBBI;
751
752 for (unsigned i = 0; i < MFI.getCalleeSavedInfo().size(); ++i)
753 --I;
754
755 // Insert instruction "move $sp, $fp" at this location.
756 BuildMI(MBB, I, DL, TII.get(MOVE), SP).addReg(FP).addReg(ZERO);
757 }
758
759 if (MipsFI->callsEhReturn()) {
760 const TargetRegisterClass *RC =
761 ABI.ArePtrs64bit() ? &Mips::GPR64RegClass : &Mips::GPR32RegClass;
762
763 // Find first instruction that restores a callee-saved register.
764 MachineBasicBlock::iterator I = MBBI;
765 for (unsigned i = 0; i < MFI.getCalleeSavedInfo().size(); ++i)
766 --I;
767
768 // Insert instructions that restore eh data registers.
769 for (int J = 0; J < 4; ++J) {
770 TII.loadRegFromStackSlot(MBB, I, ABI.GetEhDataReg(J),
771 MipsFI->getEhDataRegFI(J), RC, &RegInfo);
772 }
773 }
774
775 if (MF.getFunction().hasFnAttribute("interrupt"))
776 emitInterruptEpilogueStub(MF, MBB);
777
778 // Get the number of bytes from FrameInfo
779 uint64_t StackSize = MFI.getStackSize();
780
781 if (!StackSize)
782 return;
783
784 // Adjust stack.
785 TII.adjustStackPtr(SP, StackSize, MBB, MBBI);
786}

788bool MipsSEFrameLowering::assignCalleeSavedSpillSlots(MachineFunction &MF,
789 const TargetRegisterInfo *TRI, std::vector<CalleeSavedInfo> &CSI) const {
790 // If we're on CHERI and we're compiling for the compatible ABI, then reserve
791 // an extra spill slot for the return capability, if $ra is spilled.
792 if (CHERICFI && STI.isCheri() && !STI.getABI().IsCheriPureCap())
793 for (auto &I : CSI)
794 if (I.getReg() == Mips::RA_64) {
795 CSI.push_back(CalleeSavedInfo(Mips::C16, 0));
796 break;
797 }
798 return false;

801void MipsSEFrameLowering::emitInterruptEpilogueStub(
802 MachineFunction &MF, MachineBasicBlock &MBB) const {
803 MachineBasicBlock::iterator MBBI = MBB.getLastNonDebugInstr();
804 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
805 DebugLoc DL = MBBI != MBB.end() ? MBBI->getDebugLoc() : DebugLoc();
806
807 // Perform ISR handling like GCC
808 const TargetRegisterClass *PtrRC = &Mips::GPR32RegClass;
809
810 // Disable Interrupts.
811 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::DI), Mips::ZERO);
812 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::EHB));
813
814 // Restore EPC
815 STI.getInstrInfo()->loadRegFromStackSlot(MBB, MBBI, Mips::K1,
816 MipsFI->getISRRegFI(0), PtrRC,
817 STI.getRegisterInfo());
818 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MTC0), Mips::COP014)
819 .addReg(Mips::K1)
820 .addImm(0);
821
822 // Restore Status
823 STI.getInstrInfo()->loadRegFromStackSlot(MBB, MBBI, Mips::K1,
824 MipsFI->getISRRegFI(1), PtrRC,
825 STI.getRegisterInfo());
826 BuildMI(MBB, MBBI, DL, STI.getInstrInfo()->get(Mips::MTC0), Mips::COP012)
827 .addReg(Mips::K1)
828 .addImm(0);
829}

831int MipsSEFrameLowering::getFrameIndexReference(const MachineFunction &MF,
832 int FI,
833 unsigned &FrameReg) const {
834 const MachineFrameInfo &MFI = MF.getFrameInfo();
835 MipsABIInfo ABI = STI.getABI();
836
837 if (MFI.isFixedObjectIndex(FI))
838 FrameReg = hasFP(MF) ? ABI.GetFramePtr() : ABI.GetStackPtr();
839 else
840 FrameReg = hasBP(MF) ? ABI.GetBasePtr() : ABI.GetStackPtr();
841
842 return MFI.getObjectOffset(FI) + MFI.getStackSize() -
843 getOffsetOfLocalArea() + MFI.getOffsetAdjustment();

845
846bool MipsSEFrameLowering::
847spillCalleeSavedRegisters(MachineBasicBlock &MBB,
848 MachineBasicBlock::iterator MI,
849 const std::vector<CalleeSavedInfo> &CSI,
850 const TargetRegisterInfo *TRI) const {
851 MachineFunction *MF = MBB.getParent();
852 const TargetInstrInfo &TII = *STI.getInstrInfo();
853
854 bool IsRetAddrTaken = MF->getFrameInfo().isReturnAddressTaken();
855 bool KillRAOnSpill = true;
856 if (STI.isCheri() && !STI.getABI().IsCheriPureCap())
857 KillRAOnSpill = false;
858
859
860 for (unsigned i = 0, e = CSI.size(); i != e; ++i) {
861 // Add the callee-saved register as live-in. Do not add if the register is
862 // RA and return address is taken, because it has already been added in
863 // method MipsTargetLowering::lowerRETURNADDR.
864 // It's killed at the spill, unless the register is RA and return address
865 // is taken.
866 unsigned Reg = CSI[i].getReg();
867 bool IsRA = (Reg == Mips::RA || Reg == Mips::RA_64);
868 if (STI.getABI().IsCheriPureCap())
869 IsRA = (Reg == Mips::C17);
870 bool IsRAAndRetAddrIsTaken = IsRA && IsRetAddrTaken;
871 if (!IsRAAndRetAddrIsTaken)
872 MBB.addLiveIn(Reg);
873
874 // $c16 is not a callee-save register, so if we're being asked to spill it
875 // then we're actually using it to hold the return address as a capability.
876 if (Reg == Mips::C16) {
877 BuildMI(MBB, &*MI, MI->getDebugLoc(), TII.get(Mips::CGetPCC), Mips::C16);
878 BuildMI(MBB, &*MI, MI->getDebugLoc(), TII.get(Mips::CSetOffset), Mips::C16)
879 .addReg(Mips::C16).addReg(Mips::RA_64, RegState::Kill);
880 MachineFrameInfo &MFI = MBB.getParent()->getFrameInfo();
881 MFI.setObjectAlignment(CSI[i].getFrameIdx(), 32);
882 }
883
884 // ISRs require HI/LO to be spilled into kernel registers to be then
885 // spilled to the stack frame.
886 bool IsLOHI = (Reg == Mips::LO0 || Reg == Mips::LO0_64 ||
887 Reg == Mips::HI0 || Reg == Mips::HI0_64);
888 const Function &Func = MBB.getParent()->getFunction();
889 if (IsLOHI && Func.hasFnAttribute("interrupt")) {
890 DebugLoc DL = MI->getDebugLoc();
891
892 unsigned Op = 0;
893 if (!STI.getABI().ArePtrs64bit()) {
894 Op = (Reg == Mips::HI0) ? Mips::MFHI : Mips::MFLO;
895 Reg = Mips::K0;
896 } else {
897 Op = (Reg == Mips::HI0) ? Mips::MFHI64 : Mips::MFLO64;
898 Reg = Mips::K0_64;
899 }
900 BuildMI(MBB, MI, DL, TII.get(Op), Mips::K0)
901 .setMIFlag(MachineInstr::FrameSetup);
902 }
903
904 // Insert the spill to the stack frame.
905 bool IsKill = !IsRAAndRetAddrIsTaken && KillRAOnSpill;
906 const TargetRegisterClass *RC = TRI->getMinimalPhysRegClass(Reg);
907 TII.storeRegToStackSlot(MBB, MI, Reg, IsKill,
908 CSI[i].getFrameIdx(), RC, TRI);
909 }
910
911 return true;
912}

914bool
915MipsSEFrameLowering::hasReservedCallFrame(const MachineFunction &MF) const {
916 const MachineFrameInfo &MFI = MF.getFrameInfo();
917 // Reserve call frame if the size of the maximum call frame fits into 16-bit
918 // immediate field and there are no variable sized objects on the stack.
919 // Make sure the second register scavenger spill slot can be accessed with one
920 // instruction.
921 return isInt<16>(MFI.getMaxCallFrameSize() + getStackAlignment()) &&
922 !MFI.hasVarSizedObjects();
923}

933void MipsSEFrameLowering::determineCalleeSaves(MachineFunction &MF,
934 BitVector &SavedRegs,
935 RegScavenger *RS) const {
936 TargetFrameLowering::determineCalleeSaves(MF, SavedRegs, RS);
937 const TargetRegisterInfo *TRI = MF.getSubtarget().getRegisterInfo();
938 MipsFunctionInfo *MipsFI = MF.getInfo<MipsFunctionInfo>();
939 MipsABIInfo ABI = STI.getABI();
940 unsigned FP = ABI.GetFramePtr();
941 unsigned BP = ABI.GetBasePtr();
942
943 // Mark $fp as used if function has dedicated frame pointer.
944 if (hasFP(MF))
945 setAliasRegs(MF, SavedRegs, FP);
946 // Mark $s7 as used if function has dedicated base pointer.
947 if (hasBP(MF))
948 setAliasRegs(MF, SavedRegs, BP);
949
950 // Create spill slots for eh data registers if function calls eh_return.
951 if (MipsFI->callsEhReturn())
952 MipsFI->createEhDataRegsFI();
953
954 // Create spill slots for Coprocessor 0 registers if function is an ISR.
955 if (MipsFI->isISR())
956 MipsFI->createISRRegFI();
957
958 // Expand pseudo instructions which load, store or copy accumulators.
959 // Add an emergency spill slot if a pseudo was expanded.
960 if (ExpandPseudo(MF).expand()) {
961 // The spill slot should be half the size of the accumulator. If target have
962 // general-purpose registers 64 bits wide, it should be 64-bit, otherwise
963 // it should be 32-bit.
964 const TargetRegisterClass &RC = STI.isGP64bit() ?
965 Mips::GPR64RegClass : Mips::GPR32RegClass;
966 int FI = MF.getFrameInfo().CreateStackObject(TRI->getSpillSize(RC),
967 TRI->getSpillAlignment(RC),
968 false);
969 RS->addScavengingFrameIndex(FI);
970 }
971
972 // Set scavenging frame index if necessary.
973 uint64_t MaxSPOffset = estimateStackSize(MF);
974
975 // We will need the emergency spill slot if we have any stack offsets that
976 // don't fit in an immediate. In normal MIPS code, this means a 16-bit
977 // field, for CHERI code we have an 11-bit immediate (plus a 4-bit shift) for
978 // csc / cld, but only 8-bit immediates (plus a 3-bit shift) for c[ls]x
979 // instructions for other spills. The latter only matters if we're using the
980 // capability-stack ABI.
981 // XXXAR: no longer true if we have the big immediate CLC enabled
982 if (STI.isCheri()) {
983 if (STI.isABI_CheriPureCap()) {
984 if (isInt<10>(MaxSPOffset))
985 return;
986 } else if (isInt<15>(MaxSPOffset))
987 return;
988 }
989 // MSA has a minimum offset of 10 bits signed. If there is a variable
990 // sized object on the stack, the estimation cannot account for it.
991 else if (isIntN(STI.hasMSA() ? 10 : 16, MaxSPOffset) &&
992 !MF.getFrameInfo().hasVarSizedObjects())
993 return;
994
995 const TargetRegisterClass &RC =
996 ABI.ArePtrs64bit() ? Mips::GPR64RegClass : Mips::GPR32RegClass;
997 int FI = MF.getFrameInfo().CreateStackObject(TRI->getSpillSize(RC),
998 TRI->getSpillAlignment(RC),
999 false);
1000 RS->addScavengingFrameIndex(FI);
1001}
