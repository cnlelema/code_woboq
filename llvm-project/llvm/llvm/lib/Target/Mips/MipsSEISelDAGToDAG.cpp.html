51void MipsSEDAGToDAGISel::addDSPCtrlRegOperands(bool IsDef, MachineInstr &MI,
52 MachineFunction &MF) {
53 MachineInstrBuilder MIB(MF, &MI);
54 unsigned Mask = MI.getOperand(1).getImm();
55 unsigned Flag =
56 IsDef ? RegState::ImplicitDefine : RegState::Implicit | RegState::Undef;
57
58 if (Mask & 1)
59 MIB.addReg(Mips::DSPPos, Flag);
60
61 if (Mask & 2)
62 MIB.addReg(Mips::DSPSCount, Flag);
63
64 if (Mask & 4)
65 MIB.addReg(Mips::DSPCarry, Flag);
66
67 if (Mask & 8)
68 MIB.addReg(Mips::DSPOutFlag, Flag);
69
70 if (Mask & 16)
71 MIB.addReg(Mips::DSPCCond, Flag);
72
73 if (Mask & 32)
74 MIB.addReg(Mips::DSPEFI, Flag);
75}

82bool MipsSEDAGToDAGISel::replaceUsesWithCheriNullReg(
83 MachineRegisterInfo *MRI, const MachineInstr &GetNullMI) {
84
85 unsigned SrcReg = 0;
86 if ((GetNullMI.getOpcode() == Mips::CFromPtr &&
87 GetNullMI.getOperand(2).getReg() == Mips::ZERO_64)) {
88 // When zeroing capability registers codegen will often create NULL in a
89 // saved register (e.g. $c18 and then move it to other registers). This is
90 // stupid since it will often spill to the stack and add additional
91 // instructions when we can just use a CGetNull to the destination register
92 LLVM_DEBUG(dbgs() << "Trying to replace uses of CFromPtr $ddc, $zero: ";
93 GetNullMI.dump());
94 SrcReg = GetNullMI.getOperand(0).getReg();
95 }
96 if ((GetNullMI.getOpcode() == Mips::CMove &&
97 GetNullMI.getOperand(1).getReg() == Mips::CNULL)) {
98 LLVM_DEBUG(dbgs() << "Trying to replace uses of CMove $cnull: ";
99 GetNullMI.dump());
100 SrcReg = GetNullMI.getOperand(0).getReg();
101 }
102
103 if (!SrcReg)
104 return false;
105
106 // Cannot replace uses of physical registers (e.g. setting $c13 to null
107 if (TargetRegisterInfo::isPhysicalRegister(SrcReg))
108 return false;
109
110 llvm::SmallVector<MachineInstr *, 4> COPYUses;
111 for (MachineInstr &UseMI : MRI->use_instructions(SrcReg)) {
112 LLVM_DEBUG(dbgs() << "Found use of NULL register:"; UseMI.dump(););
113 if (UseMI.getOpcode() == TargetOpcode::COPY) {
114 COPYUses.push_back(&UseMI);
115 LLVM_DEBUG(dbgs() << "Found COPY use of NULL register:"; UseMI.dump(););
116 }
117 }
118 for (MachineInstr *UseMI : COPYUses) {
119 LLVM_DEBUG(dbgs() << "Replacing COPY of synthesized NULL: "; UseMI->dump());
120 MachineBasicBlock *MBB = UseMI->getParent();
121 auto TargetReg = UseMI->getOperand(0).getReg();
122 // FIXME: this assert only works with virtregs so not for $c13
123 if (TargetRegisterInfo::isVirtualRegister(TargetReg)) {
124#if 0
125 if (!MRI->getRegClass(TargetReg)->hasSuperClassEq(&Mips::CheriGPROrCNullRegClass)) {
126 errs() << "Not a CHERI reg?!"; UseMI->dump();
127 errs() << "Source instr ="; GetNullMI.dump();
128 errs() << "reg class id: " << MRI->getRegClassOrNull(TargetReg)->getID();
129 errs() << " cheri reg class id =" << Mips::CheriGPROrCNullRegClass.getID();
130 errs() << "\n";
131 }
132#endif
133 // The target vreg register must be a CHERI general-purpose register:
134 assert(MRI->getRegClass(TargetReg)->hasSuperClassEq(
135 &Mips::CheriGPRRegClass));
136 } else {
137#if 0
138 if (!Mips::CheriGPROrCNullRegClass.contains(TargetReg)) {
139 errs() << "REG " << (int)TargetReg << " not in CHERI GPRS?\n";
140 errs() << "Use: "; UseMI->dump();
141 errs() << "GetNull: "; UseMI->dump();
142 UseMI->getParent()->dump();
143 llvm_unreachable("Something went wrong");
144 }
145#endif
146 // Check that the physreg is a valid CHERI general-purpose register
147 assert(Mips::CheriGPROrCNullRegClass.contains(TargetReg));
148 }
149 // Remove from parent and replace with the CMove $cnull
150 BuildMI(*MBB, *UseMI, UseMI->getDebugLoc(), TII->get(Mips::CMove),
151 UseMI->getOperand(0).getReg())
152 .addReg(Mips::CNULL);
153 UseMI->removeFromParent();
154 }
155
156 // Replace uses with NullReg.
157 const auto NullReg = Mips::CNULL;
158 for (auto U = MRI->use_begin(SrcReg), E = MRI->use_end(); U != E;) {
159 MachineOperand &MO = *U;
160 unsigned OpNo = U.getOperandNo();
161 MachineInstr *MI = MO.getParent();
162 LLVM_DEBUG(dbgs() << "Found use of NULL register, will try to replace:";
163 MI->dump(););
164 ++U; // increment U now before we potentially change the successors using MO.setReg()
165
166 // Do not replace if it is a phi's operand or is tied to def operand.
167 // Also, since we allow pseudos, unlike for $zero below, we could see an
168 // instruction which doesn't have full operand info, including inline asm.
169 if (MI->isPHI() || MI->isRegTiedToDefOperand(OpNo) || MI->isDebugInstr() ||
170 OpNo >= MI->getDesc().NumOperands) {
171 LLVM_DEBUG(
172 dbgs() << "Cannot replace use of NULL register for PHI or def:";
173 MI->dump(););
174 continue;
175 }
176
177 // Also, we have to check that the register class of the operand
178 // contains the null register.
179 // This can happen for instructions where register zero encodes $ddc.
180 auto OperandType = MI->getDesc().OpInfo[OpNo].OperandType;
181 if (OperandType != Mips::OPERAND_CHERI_GPR_OR_NULL) {
182 if (OperandType == Mips::OPERAND_CHERI_GPR_OR_DDC) {
183 LLVM_DEBUG(dbgs() << "Cannot replace use of NULL register for operand "
184 "where 0 encodes $ddc: ";
185 MO.dump(); dbgs() << " in "; MI->dump(););
186 } else {
187#if !defined(NDEBUG) || defined(LLVM_ENABLE_DUMP)
188 errs() << "Found OPERAND of unknown type " << (int)OperandType
189 << " while attempting to replace uses of NULL\n";
190 MI->dump();
191#endif
192 }
193 continue;
194 }
195
196 MO.setReg(NullReg);
197 LLVM_DEBUG(dbgs() << "Was able to replace use of NULL register:";
198 MI->dump(););
199 }
200
201 // If there are no more uses of the NULL vreg after this pass we can delete it
202 if (MRI->use_empty(SrcReg)) {
203 LLVM_DEBUG(dbgs() << "Was able to remove all uses of GetNull instruction: ";
204 GetNullMI.dump());
205 return true;
206 }
207
208 return false;
209}

211bool MipsSEDAGToDAGISel::replaceUsesWithZeroReg(MachineRegisterInfo *MRI,
212 const MachineInstr& MI) {
213 unsigned DstReg = 0, ZeroReg = 0;
214
215 // Check if MI is "addiu $dst, $zero, 0" or "daddiu $dst, $zero, 0".
216 if ((MI.getOpcode() == Mips::ADDiu) &&
217 (MI.getOperand(1).getReg() == Mips::ZERO) &&
218 (MI.getOperand(2).isImm()) &&
219 (MI.getOperand(2).getImm() == 0)) {
220 DstReg = MI.getOperand(0).getReg();
221 ZeroReg = Mips::ZERO;
222 } else if ((MI.getOpcode() == Mips::DADDiu) &&
223 (MI.getOperand(1).getReg() == Mips::ZERO_64) &&
224 (MI.getOperand(2).isImm()) &&
225 (MI.getOperand(2).getImm() == 0)) {
226 DstReg = MI.getOperand(0).getReg();
227 ZeroReg = Mips::ZERO_64;
228 }
229
230 if (!DstReg) {
231 // Not using MIPS zero reg, try to replace CHERI NULL reg as well
232 return replaceUsesWithCheriNullReg(MRI, MI);
233 }
234
235 // Replace uses with ZeroReg.
236 for (MachineRegisterInfo::use_iterator U = MRI->use_begin(DstReg),
237 E = MRI->use_end(); U != E;) {
238 MachineOperand &MO = *U;
239 unsigned OpNo = U.getOperandNo();
240 MachineInstr *MI = MO.getParent();
241 ++U;
242
243 // CSetPCCOffset is not allowed to have a zero register operand
244 if (MI->getOpcode() == Mips::CSetPCCOffset)
245 continue;
246
247 // Do not replace if it is a phi's operand or is tied to def operand.
248 if (MI->isPHI() || MI->isRegTiedToDefOperand(OpNo) || MI->isPseudo())
249 continue;
250
251 // Also, we have to check that the register class of the operand
252 // contains the zero register.
253 if (!MRI->getRegClass(MO.getReg())->contains(ZeroReg))
254 continue;
255
256 MO.setReg(ZeroReg);
257 }
258
259 if (MRI->use_empty(DstReg)) {
260 LLVM_DEBUG(
261 dbgs() << "Was able to remove all uses of get zero instruction: ";
262 MI.dump());
263 return false; // TODO: return true and fix all the test cases;
264 }
265
266 return false;
267}

269void MipsSEDAGToDAGISel::processFunctionAfterISel(MachineFunction &MF) {
270 MF.getInfo<MipsFunctionInfo>()->initGlobalBaseReg();
271
272 MachineRegisterInfo *MRI = &MF.getRegInfo();
273
274 llvm::SmallVector<MachineInstr*, 8> DeadNullInsts;
275 for (auto &MBB: MF) {
276 for (auto &MI: MBB) {
277 switch (MI.getOpcode()) {
278 case Mips::RDDSP:
279 addDSPCtrlRegOperands(false, MI, MF);
280 break;
281 case Mips::WRDSP:
282 addDSPCtrlRegOperands(true, MI, MF);
283 break;
284 case Mips::BuildPairF64_64:
285 case Mips::ExtractElementF64_64:
286 if (!Subtarget->useOddSPReg()) {
287 MI.addOperand(MachineOperand::CreateReg(Mips::SP, false, true));
288 break;
289 }
290 LLVM_FALLTHROUGH;
291 case Mips::BuildPairF64:
292 case Mips::ExtractElementF64:
293 if (Subtarget->isABI_FPXX() && !Subtarget->hasMTHC1())
294 MI.addOperand(MachineOperand::CreateReg(Mips::SP, false, true));
295 break;
296 default:
297 if (replaceUsesWithZeroReg(MRI, MI))
298 DeadNullInsts.push_back(&MI);
299 }
300 }
301 }
302 // TODO: it seems to me like there should be a pass that removes dead instrs?
303 for (MachineInstr* MI : DeadNullInsts)
304 MI->removeFromParent();
305}

307void MipsSEDAGToDAGISel::selectAddE(SDNode *Node, const SDLoc &DL) const {
308 SDValue InFlag = Node->getOperand(2);
309 unsigned Opc = InFlag.getOpcode();
310 SDValue LHS = Node->getOperand(0), RHS = Node->getOperand(1);
311 EVT VT = LHS.getValueType();
312
313 // In the base case, we can rely on the carry bit from the addsc
314 // instruction.
315 if (Opc == ISD::ADDC) {
316 SDValue Ops[3] = {LHS, RHS, InFlag};
317 CurDAG->SelectNodeTo(Node, Mips::ADDWC, VT, MVT::Glue, Ops);
318 return;
319 }
320
321 assert(Opc == ISD::ADDE && "ISD::ADDE not in a chain of ADDE nodes!");
322
323 // The more complex case is when there is a chain of ISD::ADDE nodes like:
324 // (adde (adde (adde (addc a b) c) d) e).
325 //
326 // The addwc instruction does not write to the carry bit, instead it writes
327 // to bit 20 of the dsp control register. To match this series of nodes, each
328 // intermediate adde node must be expanded to write the carry bit before the
329 // addition.
330
331 // Start by reading the overflow field for addsc and moving the value to the
332 // carry field. The usage of 1 here with MipsISD::RDDSP / Mips::WRDSP
333 // corresponds to reading/writing the entire control register to/from a GPR.
334
335 SDValue CstOne = CurDAG->getTargetConstant(1, DL, MVT::i32);
336
337 SDValue OuFlag = CurDAG->getTargetConstant(20, DL, MVT::i32);
338
339 SDNode *DSPCtrlField =
340 CurDAG->getMachineNode(Mips::RDDSP, DL, MVT::i32, MVT::Glue, CstOne, InFlag);
341
342 SDNode *Carry = CurDAG->getMachineNode(
343 Mips::EXT, DL, MVT::i32, SDValue(DSPCtrlField, 0), OuFlag, CstOne);
344
345 SDValue Ops[4] = {SDValue(DSPCtrlField, 0),
346 CurDAG->getTargetConstant(6, DL, MVT::i32), CstOne,
347 SDValue(Carry, 0)};
348 SDNode *DSPCFWithCarry = CurDAG->getMachineNode(Mips::INS, DL, MVT::i32, Ops);
349
350 // My reading of the MIPS DSP 3.01 specification isn't as clear as I
351 // would like about whether bit 20 always gets overwritten by addwc.
352 // Hence take an extremely conservative view and presume it's sticky. We
353 // therefore need to clear it.
354
355 SDValue Zero = CurDAG->getRegister(Mips::ZERO, MVT::i32);
356
357 SDValue InsOps[4] = {Zero, OuFlag, CstOne, SDValue(DSPCFWithCarry, 0)};
358 SDNode *DSPCtrlFinal = CurDAG->getMachineNode(Mips::INS, DL, MVT::i32, InsOps);
359
360 SDNode *WrDSP = CurDAG->getMachineNode(Mips::WRDSP, DL, MVT::Glue,
361 SDValue(DSPCtrlFinal, 0), CstOne);
362
363 SDValue Operands[3] = {LHS, RHS, SDValue(WrDSP, 0)};
364 CurDAG->SelectNodeTo(Node, Mips::ADDWC, VT, MVT::Glue, Operands);
365}

390/// Match frameindex+offset and frameindex|offset
391bool MipsSEDAGToDAGISel::selectAddrFrameIndexOffset(
392 SDValue Addr, SDValue &Base, SDValue &Offset, unsigned OffsetBits,
393 unsigned ShiftAmount = 0) const {
394 if (CurDAG->isBaseWithConstantOffset(Addr)) {
395 ConstantSDNode *CN = dyn_cast<ConstantSDNode>(Addr.getOperand(1));
396 if (isIntN(OffsetBits + ShiftAmount, CN->getSExtValue())) {
397 EVT ValTy = Addr.getValueType();
398
399 // If the first operand is a FI, get the TargetFI Node
400 if (FrameIndexSDNode *FIN =
401 dyn_cast<FrameIndexSDNode>(Addr.getOperand(0)))
402 Base = CurDAG->getTargetFrameIndex(FIN->getIndex(), ValTy);
403 else {
404 Base = Addr.getOperand(0);
405 // If base is a FI, additional offset calculation is done in
406 // eliminateFrameIndex, otherwise we need to check the alignment
407 if (OffsetToAlignment(CN->getZExtValue(), 1ull << ShiftAmount) != 0)
408 return false;
409 }
410 Offset = CurDAG->getIntPtrConstant(CN->getZExtValue(), SDLoc(Addr), true);
411 return true;
412 }
413 }
414 return false;
415}

417/// ComplexPattern used on MipsInstrInfo
418/// Used on Mips Load/Store instructions
419bool MipsSEDAGToDAGISel::selectAddrRegImm(SDValue Addr, SDValue &Base,
420 SDValue &Offset) const {
421 // if Address is FI, get the TargetFrameIndex.
422 if (selectAddrFrameIndex(Addr, Base, Offset))
423 return true;
424
425 // on PIC code Load GA
426 if (Addr.getOpcode() == MipsISD::Wrapper) {
427 Base = Addr.getOperand(0);
428 Offset = Addr.getOperand(1);
429 return true;
430 }
431
432 if (!TM.isPositionIndependent()) {
433 if ((Addr.getOpcode() == ISD::TargetExternalSymbol ||
434 Addr.getOpcode() == ISD::TargetGlobalAddress))
435 return false;
436 }
437
438 // Addresses of the form FI+const or FI|const
439 if (selectAddrFrameIndexOffset(Addr, Base, Offset, 16))
440 return true;
441
442 // Operand is a result from an ADD.
443 if (Addr.getOpcode() == ISD::ADD) {
444 // When loading from constant pools, load the lower address part in
445 // the instruction itself. Example, instead of:
446 // lui $2, %hi($CPI1_0)
447 // addiu $2, $2, %lo($CPI1_0)
448 // lwc1 $f0, 0($2)
449 // Generate:
450 // lui $2, %hi($CPI1_0)
451 // lwc1 $f0, %lo($CPI1_0)($2)
452 if (Addr.getOperand(1).getOpcode() == MipsISD::Lo ||
453 Addr.getOperand(1).getOpcode() == MipsISD::GPRel) {
454 SDValue Opnd0 = Addr.getOperand(1).getOperand(0);
455 if (isa<ConstantPoolSDNode>(Opnd0) || isa<GlobalAddressSDNode>(Opnd0) ||
456 isa<JumpTableSDNode>(Opnd0)) {
457 Base = Addr.getOperand(0);
458 Offset = Opnd0;
459 return true;
460 }
461 }
462 }
463
464 return false;
465}

613// Select constant vector splats.
614//
615// Returns true and sets Imm if:
616// * MSA is enabled
617// * N is a ISD::BUILD_VECTOR representing a constant splat
618bool MipsSEDAGToDAGISel::selectVSplat(SDNode *N, APInt &Imm,
619 unsigned MinSizeInBits) const {
620 if (!Subtarget->hasMSA())
621 return false;
622
623 BuildVectorSDNode *Node = dyn_cast<BuildVectorSDNode>(N);
624
625 if (!Node)
626 return false;
627
628 APInt SplatValue, SplatUndef;
629 unsigned SplatBitSize;
630 bool HasAnyUndefs;
631
632 if (!Node->isConstantSplat(SplatValue, SplatUndef, SplatBitSize, HasAnyUndefs,
633 MinSizeInBits, !Subtarget->isLittle()))
634 return false;
635
636 Imm = SplatValue;
637
638 return true;
639}

642//
643// In addition to the requirements of selectVSplat(), this function returns
644// true and sets Imm if:
645// * The splat value is the same width as the elements of the vector
646// * The splat value fits in an integer with the specified signed-ness and
647// width.
648//
649// This function looks through ISD::BITCAST nodes.
650// TODO: This might not be appropriate for big-endian MSA since BITCAST is
651// sometimes a shuffle in big-endian mode.
652//
653// It's worth noting that this function is not used as part of the selection
654// of ldi.[bhwd] since it does not permit using the wrong-typed ldi.[bhwd]

668
669 if (( Signed && ImmValue.isSignedIntN(ImmBitSize)) ||
670 (!Signed && ImmValue.isIntN(ImmBitSize))) {
671 Imm = CurDAG->getTargetConstant(ImmValue, SDLoc(N), EltTy);
672 return true;
673 }
674 }
675
676 return false;
677}

756// of left-most bits set (e.g. 0b11...1100...00).
757//
758// In addition to the requirements of selectVSplat(), this function returns
759// true and sets Imm if:
760// * The splat value is the same width as the elements of the vector
761// * The splat value is a consecutive sequence of left-most bits.
762//
763// This function looks through ISD::BITCAST nodes.
764// TODO: This might not be appropriate for big-endian MSA since BITCAST is
765// sometimes a shuffle in big-endian mode.
766bool MipsSEDAGToDAGISel::selectVSplatMaskL(SDValue N, SDValue &Imm) const {
767 APInt ImmValue;
768 EVT EltTy = N->getValueType(0).getVectorElementType();
769
770 if (N->getOpcode() == ISD::BITCAST)
771 N = N->getOperand(0);
772
773 if (selectVSplat(N.getNode(), ImmValue, EltTy.getSizeInBits()) &&
774 ImmValue.getBitWidth() == EltTy.getSizeInBits()) {
775 // Extract the run of set bits starting with bit zero from the bitwise
776 // inverse of ImmValue, and test that the inverse of this is the same
777 // as the original value.
778 if (ImmValue == ~(~ImmValue & ~(~ImmValue + 1))) {
779
780 Imm = CurDAG->getTargetConstant(ImmValue.countPopulation() - 1, SDLoc(N),
781 EltTy);
782 return true;
783 }

786 return false;

841
842bool MipsSEDAGToDAGISel::trySelect(SDNode *Node) {
843 unsigned Opcode = Node->getOpcode();
844 SDLoc DL(Node);
845
846 ///
847 // Instruction Selection not handled by the auto-generated
848 // tablegen selection should be handled here.
849 ///
850 switch(Opcode) {
851 default: break;
852
853 case Mips::PseudoD_SELECT_I:
854 case Mips::PseudoD_SELECT_I64: {
855 MVT VT = Subtarget->isGP64bit() ? MVT::i64 : MVT::i32;
856 SDValue cond = Node->getOperand(0);
857 SDValue Hi1 = Node->getOperand(1);
858 SDValue Lo1 = Node->getOperand(2);
859 SDValue Hi2 = Node->getOperand(3);
860 SDValue Lo2 = Node->getOperand(4);
861
862 SDValue ops[] = {cond, Hi1, Lo1, Hi2, Lo2};
863 EVT NodeTys[] = {VT, VT};
864 ReplaceNode(Node, CurDAG->getMachineNode(Subtarget->isGP64bit()
865 ? Mips::PseudoD_SELECT_I64
866 : Mips::PseudoD_SELECT_I,
867 DL, NodeTys, ops));
868 return true;
869 }
870
871 case ISD::ADDE: {
872 selectAddE(Node, DL);
873 return true;
874 }
875
876 case ISD::ConstantFP: {
877 ConstantFPSDNode *CN = dyn_cast<ConstantFPSDNode>(Node);
878 if (Node->getValueType(0) == MVT::f64 && CN->isExactlyValue(+0.0)) {
879 if (Subtarget->isGP64bit()) {
880 SDValue Zero = CurDAG->getCopyFromReg(CurDAG->getEntryNode(), DL,
881 Mips::ZERO_64, MVT::i64);
882 ReplaceNode(Node,
883 CurDAG->getMachineNode(Mips::DMTC1, DL, MVT::f64, Zero));
884 } else if (Subtarget->isFP64bit()) {
885 SDValue Zero = CurDAG->getCopyFromReg(CurDAG->getEntryNode(), DL,
886 Mips::ZERO, MVT::i32);
887 ReplaceNode(Node, CurDAG->getMachineNode(Mips::BuildPairF64_64, DL,
888 MVT::f64, Zero, Zero));
889 } else {
890 SDValue Zero = CurDAG->getCopyFromReg(CurDAG->getEntryNode(), DL,
891 Mips::ZERO, MVT::i32);
892 ReplaceNode(Node, CurDAG->getMachineNode(Mips::BuildPairF64, DL,
893 MVT::f64, Zero, Zero));
894 }
895 return true;
896 }
897 break;
898 }
899
900 case ISD::Constant: {
901 const ConstantSDNode *CN = dyn_cast<ConstantSDNode>(Node);
902 int64_t Imm = CN->getSExtValue();
903 unsigned Size = CN->getValueSizeInBits(0);
904
905 if (isInt<32>(Imm))
906 break;
907
908 MipsAnalyzeImmediate AnalyzeImm;
909
910 const MipsAnalyzeImmediate::InstSeq &Seq =
911 AnalyzeImm.Analyze(Imm, Size, false);
912
913 MipsAnalyzeImmediate::InstSeq::const_iterator Inst = Seq.begin();
914 SDLoc DL(CN);
915 SDNode *RegOpnd;
916 SDValue ImmOpnd = CurDAG->getTargetConstant(SignExtend64<16>(Inst->ImmOpnd),
917 DL, MVT::i64);
918
919 // The first instruction can be a LUi which is different from other
920 // instructions (ADDiu, ORI and SLL) in that it does not have a register
921 // operand.
922 if (Inst->Opc == Mips::LUi64)
923 RegOpnd = CurDAG->getMachineNode(Inst->Opc, DL, MVT::i64, ImmOpnd);
924 else
925 RegOpnd =
926 CurDAG->getMachineNode(Inst->Opc, DL, MVT::i64,
927 CurDAG->getRegister(Mips::ZERO_64, MVT::i64),
928 ImmOpnd);
929
930 // The remaining instructions in the sequence are handled here.
931 for (++Inst; Inst != Seq.end(); ++Inst) {
932 ImmOpnd = CurDAG->getTargetConstant(SignExtend64<16>(Inst->ImmOpnd), DL,
933 MVT::i64);
934 RegOpnd = CurDAG->getMachineNode(Inst->Opc, DL, MVT::i64,
935 SDValue(RegOpnd, 0), ImmOpnd);
936 }
937
938 ReplaceNode(Node, RegOpnd);
939 return true;
940 }
941
942 case ISD::INTRINSIC_W_CHAIN: {
943 switch (cast<ConstantSDNode>(Node->getOperand(1))->getZExtValue()) {
944 default:
945 break;
946
947 case Intrinsic::mips_cfcmsa: {
948 SDValue ChainIn = Node->getOperand(0);
949 SDValue RegIdx = Node->getOperand(2);
950 SDValue Reg = CurDAG->getCopyFromReg(ChainIn, DL,
951 getMSACtrlReg(RegIdx), MVT::i32);
952 ReplaceNode(Node, Reg.getNode());
953 return true;
954 }
955 }
956 break;
957 }
958
959 case ISD::INTRINSIC_WO_CHAIN: {
960 switch (cast<ConstantSDNode>(Node->getOperand(0))->getZExtValue()) {
961 default:
962 break;
963
964 case Intrinsic::mips_move_v:
965 // Like an assignment but will always produce a move.v even if
966 // unnecessary.
967 ReplaceNode(Node, CurDAG->getMachineNode(Mips::MOVE_V, DL,
968 Node->getValueType(0),
969 Node->getOperand(1)));
970 return true;
971 }
972 break;
973 }
974
975 case ISD::INTRINSIC_VOID: {
976 switch (cast<ConstantSDNode>(Node->getOperand(1))->getZExtValue()) {
977 default:
978 break;
979
980 case Intrinsic::mips_ctcmsa: {
981 SDValue ChainIn = Node->getOperand(0);
982 SDValue RegIdx = Node->getOperand(2);
983 SDValue Value = Node->getOperand(3);
984 SDValue ChainOut = CurDAG->getCopyToReg(ChainIn, DL,
985 getMSACtrlReg(RegIdx), Value);
986 ReplaceNode(Node, ChainOut.getNode());
987 return true;
988 }
989 }
990 break;
991 }
992
993 // Manually match MipsISD::Ins nodes to get the correct instruction. It has
994 // to be done in this fashion so that we respect the differences between
995 // dins and dinsm, as the difference is that the size operand has the range
996 // 0 < size <= 32 for dins while dinsm has the range 2 <= size <= 64 which
997 // means SelectionDAGISel would have to test all the operands at once to
998 // match the instruction.
999 case MipsISD::Ins: {
1000
1001 // Sanity checking for the node operands.
1002 if (Node->getValueType(0) != MVT::i32 && Node->getValueType(0) != MVT::i64)
1003 return false;
1004
1005 if (Node->getNumOperands() != 4)
1006 return false;
1007
1008 if (Node->getOperand(1)->getOpcode() != ISD::Constant ||
1009 Node->getOperand(2)->getOpcode() != ISD::Constant)
1010 return false;
1011
1012 MVT ResTy = Node->getSimpleValueType(0);
1013 uint64_t Pos = Node->getConstantOperandVal(1);
1014 uint64_t Size = Node->getConstantOperandVal(2);
1015
1016 // Size has to be >0 for 'ins', 'dins' and 'dinsu'.
1017 if (!Size)
1018 return false;
1019
1020 if (Pos + Size > 64)
1021 return false;
1022
1023 if (ResTy != MVT::i32 && ResTy != MVT::i64)
1024 return false;
1025
1026 unsigned Opcode = 0;
1027 if (ResTy == MVT::i32) {
1028 if (Pos + Size <= 32)
1029 Opcode = Mips::INS;
1030 } else {
1031 if (Pos + Size <= 32)
1032 Opcode = Mips::DINS;
1033 else if (Pos < 32 && 1 < Size)
1034 Opcode = Mips::DINSM;
1035 else
1036 Opcode = Mips::DINSU;
1037 }
1038
1039 if (Opcode) {
1040 SDValue Ops[4] = {
1041 Node->getOperand(0), CurDAG->getTargetConstant(Pos, DL, MVT::i32),
1042 CurDAG->getTargetConstant(Size, DL, MVT::i32), Node->getOperand(3)};
1043
1044 ReplaceNode(Node, CurDAG->getMachineNode(Opcode, DL, ResTy, Ops));
1045 return true;
1046 }
1047
1048 return false;
1049 }
1050
1051 case MipsISD::ThreadPointer: {
1052 assert(!Subtarget->getABI().UsesCapabilityTls());
1053 EVT PtrVT = getTargetLowering()->getPointerTy(CurDAG->getDataLayout(), 0);
1054 unsigned RdhwrOpc, DestReg;
1055
1056 if (PtrVT == MVT::i32) {
1057 RdhwrOpc = Mips::RDHWR;
1058 DestReg = Mips::V1;
1059 } else {
1060 RdhwrOpc = Mips::RDHWR64;
1061 DestReg = Mips::V1_64;
1062 }
1063
1064 SDNode *Rdhwr =
1065 CurDAG->getMachineNode(RdhwrOpc, DL, Node->getValueType(0),
1066 CurDAG->getRegister(Mips::HWR29, MVT::i32),
1067 CurDAG->getTargetConstant(0, DL, MVT::i32));
1068 SDValue Chain = CurDAG->getCopyToReg(CurDAG->getEntryNode(), DL, DestReg,
1069 SDValue(Rdhwr, 0));
1070 SDValue ResNode = CurDAG->getCopyFromReg(Chain, DL, DestReg, PtrVT);
1071 ReplaceNode(Node, ResNode.getNode());
1072 return true;
1073 }
1074
1075 case MipsISD::CapThreadPointer: {
1076 EVT CapVT = Subtarget->typeForCapabilities();
1077 SDNode *CReadHwr =
1078 CurDAG->getMachineNode(Mips::CReadHwr, DL, CapVT,
1079 CurDAG->getRegister(Mips::CAPHWR1, CapVT));
1080 ReplaceNode(Node, CReadHwr);
1081 return true;
1082 }
1083
1084 case ISD::BUILD_VECTOR: {
1085 // Select appropriate ldi.[bhwd] instructions for constant splats of
1086 // 128-bit when MSA is enabled. Fixup any register class mismatches that
1087 // occur as a result.
1088 //
1089 // This allows the compiler to use a wider range of immediates than would
1090 // otherwise be allowed. If, for example, v4i32 could only use ldi.h then
1091 // it would not be possible to load { 0x01010101, 0x01010101, 0x01010101,
1092 // 0x01010101 } without using a constant pool. This would be sub-optimal
1093 // when // 'ldi.b wd, 1' is capable of producing that bit-pattern in the
1094 // same set/ of registers. Similarly, ldi.h isn't capable of producing {
1095 // 0x00000000, 0x00000001, 0x00000000, 0x00000001 } but 'ldi.d wd, 1' can.
1096
1097 const MipsABIInfo &ABI =
1098 static_cast<const MipsTargetMachine &>(TM).getABI();
1099
1100 BuildVectorSDNode *BVN = cast<BuildVectorSDNode>(Node);
1101 APInt SplatValue, SplatUndef;
1102 unsigned SplatBitSize;
1103 bool HasAnyUndefs;
1104 unsigned LdiOp;
1105 EVT ResVecTy = BVN->getValueType(0);
1106 EVT ViaVecTy;
1107
1108 if (!Subtarget->hasMSA() || !BVN->getValueType(0).is128BitVector())
1109 return false;
1110
1111 if (!BVN->isConstantSplat(SplatValue, SplatUndef, SplatBitSize,
1112 HasAnyUndefs, 8,
1113 !Subtarget->isLittle()))
1114 return false;
1115
1116 switch (SplatBitSize) {
1117 default:
1118 return false;
1119 case 8:
1120 LdiOp = Mips::LDI_B;
1121 ViaVecTy = MVT::v16i8;
1122 break;
1123 case 16:
1124 LdiOp = Mips::LDI_H;
1125 ViaVecTy = MVT::v8i16;
1126 break;
1127 case 32:
1128 LdiOp = Mips::LDI_W;
1129 ViaVecTy = MVT::v4i32;
1130 break;
1131 case 64:
1132 LdiOp = Mips::LDI_D;
1133 ViaVecTy = MVT::v2i64;
1134 break;
1135 }
1136
1137 SDNode *Res;
1138
1139 // If we have a signed 10 bit integer, we can splat it directly.
1140 //
1141 // If we have something bigger we can synthesize the value into a GPR and
1142 // splat from there.
1143 if (SplatValue.isSignedIntN(10)) {
1144 SDValue Imm = CurDAG->getTargetConstant(SplatValue, DL,
1145 ViaVecTy.getVectorElementType());
1146
1147 Res = CurDAG->getMachineNode(LdiOp, DL, ViaVecTy, Imm);
1148 } else if (SplatValue.isSignedIntN(16) &&
1149 ((ABI.IsO32() && SplatBitSize < 64) ||
1150 (ABI.IsN32() || ABI.IsN64()))) {
1151 // Only handle signed 16 bit values when the element size is GPR width.
1152 // MIPS64 can handle all the cases but MIPS32 would need to handle
1153 // negative cases specifically here. Instead, handle those cases as
1154 // 64bit values.
1155
1156 bool Is32BitSplat = ABI.IsO32() || SplatBitSize < 64;
1157 const unsigned ADDiuOp = Is32BitSplat ? Mips::ADDiu : Mips::DADDiu;
1158 const MVT SplatMVT = Is32BitSplat ? MVT::i32 : MVT::i64;
1159 SDValue ZeroVal = CurDAG->getRegister(
1160 Is32BitSplat ? Mips::ZERO : Mips::ZERO_64, SplatMVT);
1161
1162 const unsigned FILLOp =
1163 SplatBitSize == 16
1164 ? Mips::FILL_H
1165 : (SplatBitSize == 32 ? Mips::FILL_W
1166 : (SplatBitSize == 64 ? Mips::FILL_D : 0));
1167
1168 assert(FILLOp != 0 && "Unknown FILL Op for splat synthesis!");
1169 assert((!ABI.IsO32() || (FILLOp != Mips::FILL_D)) &&
1170 "Attempting to use fill.d on MIPS32!");
1171
1172 const unsigned Lo = SplatValue.getLoBits(16).getZExtValue();
1173 SDValue LoVal = CurDAG->getTargetConstant(Lo, DL, SplatMVT);
1174
1175 Res = CurDAG->getMachineNode(ADDiuOp, DL, SplatMVT, ZeroVal, LoVal);
1176 Res = CurDAG->getMachineNode(FILLOp, DL, ViaVecTy, SDValue(Res, 0));
1177
1178 } else if (SplatValue.isSignedIntN(32) && SplatBitSize == 32) {
1179 // Only handle the cases where the splat size agrees with the size
1180 // of the SplatValue here.
1181 const unsigned Lo = SplatValue.getLoBits(16).getZExtValue();
1182 const unsigned Hi = SplatValue.lshr(16).getLoBits(16).getZExtValue();
1183 SDValue ZeroVal = CurDAG->getRegister(Mips::ZERO, MVT::i32);
1184
1185 SDValue LoVal = CurDAG->getTargetConstant(Lo, DL, MVT::i32);
1186 SDValue HiVal = CurDAG->getTargetConstant(Hi, DL, MVT::i32);
1187
1188 if (Hi)
1189 Res = CurDAG->getMachineNode(Mips::LUi, DL, MVT::i32, HiVal);
1190
1191 if (Lo)
1192 Res = CurDAG->getMachineNode(Mips::ORi, DL, MVT::i32,
1193 Hi ? SDValue(Res, 0) : ZeroVal, LoVal);
1194
1195 assert((Hi || Lo) && "Zero case reached 32 bit case splat synthesis!");
1196 Res = CurDAG->getMachineNode(Mips::FILL_W, DL, MVT::v4i32, SDValue(Res, 0));
1197
1198 } else if (SplatValue.isSignedIntN(32) && SplatBitSize == 64 &&
1199 (ABI.IsN32() || ABI.IsN64())) {
1200 // N32 and N64 can perform some tricks that O32 can't for signed 32 bit
1201 // integers due to having 64bit registers. lui will cause the necessary
1202 // zero/sign extension.
1203 const unsigned Lo = SplatValue.getLoBits(16).getZExtValue();
1204 const unsigned Hi = SplatValue.lshr(16).getLoBits(16).getZExtValue();
1205 SDValue ZeroVal = CurDAG->getRegister(Mips::ZERO, MVT::i32);
1206
1207 SDValue LoVal = CurDAG->getTargetConstant(Lo, DL, MVT::i32);
1208 SDValue HiVal = CurDAG->getTargetConstant(Hi, DL, MVT::i32);
1209
1210 if (Hi)
1211 Res = CurDAG->getMachineNode(Mips::LUi, DL, MVT::i32, HiVal);
1212
1213 if (Lo)
1214 Res = CurDAG->getMachineNode(Mips::ORi, DL, MVT::i32,
1215 Hi ? SDValue(Res, 0) : ZeroVal, LoVal);
1216
1217 Res = CurDAG->getMachineNode(
1218 Mips::SUBREG_TO_REG, DL, MVT::i64,
1219 CurDAG->getTargetConstant(((Hi >> 15) & 0x1), DL, MVT::i64),
1220 SDValue(Res, 0),
1221 CurDAG->getTargetConstant(Mips::sub_32, DL, MVT::i64));
1222
1223 Res =
1224 CurDAG->getMachineNode(Mips::FILL_D, DL, MVT::v2i64, SDValue(Res, 0));
1225
1226 } else if (SplatValue.isSignedIntN(64)) {
1227 // If we have a 64 bit Splat value, we perform a similar sequence to the
1228 // above:
1229 //
1230 // MIPS32: MIPS64:
1231 // lui $res, %highest(val) lui $res, %highest(val)
1232 // ori $res, $res, %higher(val) ori $res, $res, %higher(val)
1233 // lui $res2, %hi(val) lui $res2, %hi(val)
1234 // ori $res2, %res2, %lo(val) ori $res2, %res2, %lo(val)
1235 // $res3 = fill $res2 dinsu $res, $res2, 0, 32
1236 // $res4 = insert.w $res3[1], $res fill.d $res
1237 // splat.d $res4, 0
1238 //
1239 // The ability to use dinsu is guaranteed as MSA requires MIPSR5. This saves
1240 // having to materialize the value by shifts and ors.
1241 //
1242 // FIXME: Implement the preferred sequence for MIPS64R6:
1243 //
1244 // MIPS64R6:
1245 // ori $res, $zero, %lo(val)
1246 // daui $res, $res, %hi(val)
1247 // dahi $res, $res, %higher(val)
1248 // dati $res, $res, %highest(cal)
1249 // fill.d $res
1250 //
1251
1252 const unsigned Lo = SplatValue.getLoBits(16).getZExtValue();
1253 const unsigned Hi = SplatValue.lshr(16).getLoBits(16).getZExtValue();
1254 const unsigned Higher = SplatValue.lshr(32).getLoBits(16).getZExtValue();
1255 const unsigned Highest = SplatValue.lshr(48).getLoBits(16).getZExtValue();
1256
1257 SDValue LoVal = CurDAG->getTargetConstant(Lo, DL, MVT::i32);
1258 SDValue HiVal = CurDAG->getTargetConstant(Hi, DL, MVT::i32);
1259 SDValue HigherVal = CurDAG->getTargetConstant(Higher, DL, MVT::i32);
1260 SDValue HighestVal = CurDAG->getTargetConstant(Highest, DL, MVT::i32);
1261 SDValue ZeroVal = CurDAG->getRegister(Mips::ZERO, MVT::i32);
1262
1263 // Independent of whether we're targeting MIPS64 or not, the basic
1264 // operations are the same. Also, directly use the $zero register if
1265 // the 16 bit chunk is zero.
1266 //
1267 // For optimization purposes we always synthesize the splat value as
1268 // an i32 value, then if we're targetting MIPS64, use SUBREG_TO_REG
1269 // just before combining the values with dinsu to produce an i64. This
1270 // enables SelectionDAG to aggressively share components of splat values
1271 // where possible.
1272 //
1273 // FIXME: This is the general constant synthesis problem. This code
1274 // should be factored out into a class shared between all the
1275 // classes that need it. Specifically, for a splat size of 64
1276 // bits that's a negative number we can do better than LUi/ORi
1277 // for the upper 32bits.
1278
1279 if (Hi)
1280 Res = CurDAG->getMachineNode(Mips::LUi, DL, MVT::i32, HiVal);
1281
1282 if (Lo)
1283 Res = CurDAG->getMachineNode(Mips::ORi, DL, MVT::i32,
1284 Hi ? SDValue(Res, 0) : ZeroVal, LoVal);
1285
1286 SDNode *HiRes;
1287 if (Highest)
1288 HiRes = CurDAG->getMachineNode(Mips::LUi, DL, MVT::i32, HighestVal);
1289
1290 if (Higher)
1291 HiRes = CurDAG->getMachineNode(Mips::ORi, DL, MVT::i32,
1292 Highest ? SDValue(HiRes, 0) : ZeroVal,
1293 HigherVal);
1294
1295
1296 if (ABI.IsO32()) {
1297 Res = CurDAG->getMachineNode(Mips::FILL_W, DL, MVT::v4i32,
1298 (Hi || Lo) ? SDValue(Res, 0) : ZeroVal);
1299
1300 Res = CurDAG->getMachineNode(
1301 Mips::INSERT_W, DL, MVT::v4i32, SDValue(Res, 0),
1302 (Highest || Higher) ? SDValue(HiRes, 0) : ZeroVal,
1303 CurDAG->getTargetConstant(1, DL, MVT::i32));
1304
1305 const TargetLowering *TLI = getTargetLowering();
1306 const TargetRegisterClass *RC =
1307 TLI->getRegClassFor(ViaVecTy.getSimpleVT());
1308
1309 Res = CurDAG->getMachineNode(
1310 Mips::COPY_TO_REGCLASS, DL, ViaVecTy, SDValue(Res, 0),
1311 CurDAG->getTargetConstant(RC->getID(), DL, MVT::i32));
1312
1313 Res = CurDAG->getMachineNode(
1314 Mips::SPLATI_D, DL, MVT::v2i64, SDValue(Res, 0),
1315 CurDAG->getTargetConstant(0, DL, MVT::i32));
1316 } else if (ABI.IsN64() || ABI.IsN32()) {
1317
1318 SDValue Zero64Val = CurDAG->getRegister(Mips::ZERO_64, MVT::i64);
1319 const bool HiResNonZero = Highest || Higher;
1320 const bool ResNonZero = Hi || Lo;
1321
1322 if (HiResNonZero)
1323 HiRes = CurDAG->getMachineNode(
1324 Mips::SUBREG_TO_REG, DL, MVT::i64,
1325 CurDAG->getTargetConstant(((Highest >> 15) & 0x1), DL, MVT::i64),
1326 SDValue(HiRes, 0),
1327 CurDAG->getTargetConstant(Mips::sub_32, DL, MVT::i64));
1328
1329 if (ResNonZero)
1330 Res = CurDAG->getMachineNode(
1331 Mips::SUBREG_TO_REG, DL, MVT::i64,
1332 CurDAG->getTargetConstant(((Hi >> 15) & 0x1), DL, MVT::i64),
1333 SDValue(Res, 0),
1334 CurDAG->getTargetConstant(Mips::sub_32, DL, MVT::i64));
1335
1336 // We have 3 cases:
1337 // The HiRes is nonzero but Res is $zero => dsll32 HiRes, 0
1338 // The Res is nonzero but HiRes is $zero => dinsu Res, $zero, 32, 32
1339 // Both are non zero => dinsu Res, HiRes, 32, 32
1340 //
1341 // The obvious "missing" case is when both are zero, but that case is
1342 // handled by the ldi case.
1343 if (ResNonZero) {
1344 IntegerType *Int32Ty =
1345 IntegerType::get(MF->getFunction().getContext(), 32);
1346 const ConstantInt *Const32 = ConstantInt::get(Int32Ty, 32);
1347 SDValue Ops[4] = {HiResNonZero ? SDValue(HiRes, 0) : Zero64Val,
1348 CurDAG->getConstant(*Const32, DL, MVT::i32),
1349 CurDAG->getConstant(*Const32, DL, MVT::i32),
1350 SDValue(Res, 0)};
1351
1352 Res = CurDAG->getMachineNode(Mips::DINSU, DL, MVT::i64, Ops);
1353 } else if (HiResNonZero) {
1354 Res = CurDAG->getMachineNode(
1355 Mips::DSLL32, DL, MVT::i64, SDValue(HiRes, 0),
1356 CurDAG->getTargetConstant(0, DL, MVT::i32));
1357 } else
1358 llvm_unreachable(
1359 "Zero splat value handled by non-zero 64bit splat synthesis!");
1360
1361 Res = CurDAG->getMachineNode(Mips::FILL_D, DL, MVT::v2i64, SDValue(Res, 0));
1362 } else
1363 llvm_unreachable("Unknown ABI in MipsISelDAGToDAG!");
1364
1365 } else
1366 return false;
1367
1368 if (ResVecTy != ViaVecTy) {
1369 // If LdiOp is writing to a different register class to ResVecTy, then
1370 // fix it up here. This COPY_TO_REGCLASS should never cause a move.v
1371 // since the source and destination register sets contain the same
1372 // registers.
1373 const TargetLowering *TLI = getTargetLowering();
1374 MVT ResVecTySimple = ResVecTy.getSimpleVT();
1375 const TargetRegisterClass *RC = TLI->getRegClassFor(ResVecTySimple);
1376 Res = CurDAG->getMachineNode(Mips::COPY_TO_REGCLASS, DL,
1377 ResVecTy, SDValue(Res, 0),
1378 CurDAG->getTargetConstant(RC->getID(), DL,
1379 MVT::i32));
1380 }
1381
1382 ReplaceNode(Node, Res);
1383 return true;
1384 }
1385
1386 }
1387
1388 return false;
1389}

1391bool MipsSEDAGToDAGISel::
1392SelectInlineAsmMemoryOperand(const SDValue &Op, unsigned ConstraintID,
1393 std::vector<SDValue> &OutOps) {
1394 SDValue Base, Offset;
1395
1396 switch(ConstraintID) {
1397 default:
1398 llvm_unreachable("Unexpected asm memory constraint");
1399 // All memory constraints can at least accept raw pointers.
1400 case InlineAsm::Constraint_i:
1401 OutOps.push_back(Op);
1402 OutOps.push_back(CurDAG->getTargetConstant(0, SDLoc(Op), MVT::i32));
1403 return false;
1404 case InlineAsm::Constraint_m:
1405 if (selectAddrRegImm16(Op, Base, Offset)) {
1406 OutOps.push_back(Base);
1407 OutOps.push_back(Offset);
1408 return false;
1409 }
1410 OutOps.push_back(Op);
1411 OutOps.push_back(CurDAG->getTargetConstant(0, SDLoc(Op), MVT::i32));
1412 return false;
1413 case InlineAsm::Constraint_R:
1414 // The 'R' constraint is supposed to be much more complicated than this.
1415 // However, it's becoming less useful due to architectural changes and
1416 // ought to be replaced by other constraints such as 'ZC'.
1417 // For now, support 9-bit signed offsets which is supportable by all
1418 // subtargets for all instructions.
1419 if (selectAddrRegImm9(Op, Base, Offset)) {
1420 OutOps.push_back(Base);
1421 OutOps.push_back(Offset);
1422 return false;
1423 }
1424 OutOps.push_back(Op);
1425 OutOps.push_back(CurDAG->getTargetConstant(0, SDLoc(Op), MVT::i32));
1426 return false;
1427 case InlineAsm::Constraint_ZC:
1428 // ZC matches whatever the pref, ll, and sc instructions can handle for the
1429 // given subtarget.
1430 if (Subtarget->inMicroMipsMode()) {
1431 // On microMIPS, they can handle 12-bit offsets.
1432 if (selectAddrRegImm12(Op, Base, Offset)) {
1433 OutOps.push_back(Base);
1434 OutOps.push_back(Offset);
1435 return false;
1436 }
1437 } else if (Subtarget->hasMips32r6()) {
1438 // On MIPS32r6/MIPS64r6, they can only handle 9-bit offsets.
1439 if (selectAddrRegImm9(Op, Base, Offset)) {
1440 OutOps.push_back(Base);
1441 OutOps.push_back(Offset);
1442 return false;
1443 }
1444 } else if (selectAddrRegImm16(Op, Base, Offset)) {
1445 // Prior to MIPS32r6/MIPS64r6, they can handle 16-bit offsets.
1446 OutOps.push_back(Base);
1447 OutOps.push_back(Offset);
1448 return false;
1449 }
1450 // In all cases, 0-bit offsets are acceptable.
1451 OutOps.push_back(Op);
1452 OutOps.push_back(CurDAG->getTargetConstant(0, SDLoc(Op), MVT::i32));
1453 return false;
1454 }
1455 return true;
1456}
