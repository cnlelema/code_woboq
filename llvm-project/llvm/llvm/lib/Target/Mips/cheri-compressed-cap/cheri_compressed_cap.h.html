414/* Returns the index of the most significant bit set in x */
415static inline uint32_t cc128_idx_MSNZ(uint64_t x) {
416#if defined(__has_builtin)
417#if __has_builtin(__builtin_clzll)
418#define CAP_HAVE_BUILTIN_CLZ
419#endif
420#elif defined(__GNUC__)
421#define CAP_HAVE_BUILTIN_CLZ
422#endif
423
424#ifndef CAP_HAVE_BUILTIN_CLZ
425/* floor(log2(x)) != floor(log2(y)) */
426#warning "__builtin_clzll not supported, using slower path"
427#define ld_neq(x, y) (((x) ^ (y)) > ((x) & (y)))
428 uint32_t r = ld_neq(x, x & 0x5555555555555555ull) + (ld_neq(x, x & 0x3333333333333333ull) << 1) +
429 (ld_neq(x, x & 0x0f0f0f0f0f0f0f0full) << 2) + (ld_neq(x, x & 0x00ff00ff00ff00ffull) << 3) +
430 (ld_neq(x, x & 0x0000ffff0000ffffull) << 4) + (ld_neq(x, x & 0x00000000ffffffffull) << 5);
431#undef ld_neq
432#else
433 assert(x != 0);
434 uint32_t r = 63u - (uint32_t)__builtin_clzll(x);
435 // printf("%s: %016llx = %d (__builtin_clzll(x) = %d, 63 - clz = %d)\n", __func__, x, r, __builtin_clzll(x), 63 -
436 // __builtin_clzll(x));
437#endif
438 return r;
439}

441/*
442 * e = idxMSNZ( (rlength + (rlength >> 6)) >> 19 )
443 * where (rlength + (rlength >> 6)) needs to be a 65 bit integer
444 */
445static inline uint32_t _cc128_compute_e(uint64_t rlength, uint32_t bwidth) {
446 if (rlength < (1u << (bwidth - 1)))
447 return 0;
448
449 return (cc128_idx_MSNZ(rlength) - (bwidth - 2));
450}

493 uint32_t BWidth = seal_mode == CC_SEAL_MODE_SEALED ? CC_L_SEALED_BWIDTH : CC_L_BWIDTH;
494 uint32_t BMask = (1 << BWidth) - 1;
495 uint32_t TMask = BMask >> 2;
496
497 uint8_t IE = (uint8_t)cc128_getbits(pesbt, CC_L_IE_OFF, 1);
498 uint8_t LH = (uint8_t)cc128_getbits(pesbt, CC_L_LH_OFF, 1);
499
500 uint8_t E, L_msb;
501 uint32_t B = (uint32_t)cc128_getbits(pesbt, CC_L_B_OFF, BWidth);
502 uint32_t T = (uint32_t)cc128_getbits(pesbt, CC_L_T_OFF, BWidth - 2);
503
504 if (IE) {
505 // Note: Do not offset by 1. We alsot need to encode E=0 with IE for setbounds!
506 E = ((((LH << CC_L_LOWWIDTH) | (B & CC_L_LOWMASK)) << CC_L_LOWWIDTH) | (T & CC_L_LOWMASK));
507 E = MIN(64 - BWidth + 2, E);
508 B &= ~CC_L_LOWMASK;
509 T &= ~CC_L_LOWMASK;
510 L_msb = 1;
511 } else {
512 E = 0;
513 L_msb = LH;
514 }
515
516 uint32_t type = CAP_OTYPE_UNSEALED;
517
518 if (seal_mode == CC_SEAL_MODE_UNSEALED) {
519 // TODO: remove extra bit for unsealed caps and always store otype
520 type = CAP_OTYPE_UNSEALED;
521 } else if (seal_mode == CC_SEAL_MODE_SEALED) {
522 type = (uint32_t)(cc128_getbits(pesbt, CC_L_OHI_OFF, CC_L_TYPES / 2) << (CC_L_TYPES / 2)) |
523 (uint32_t)cc128_getbits(pesbt, CC_L_OLO_OFF, CC_L_TYPES / 2);
524 } else if (seal_mode == CC_SEAL_MODE_SENTRY) {
525 type = CAP_OTYPE_SENTRY;
526 }
527
528 cdp->cr_otype = type;
529#else
530 cdp->cr_perms = (uint32_t)CC128_EXTRACT_FIELD(pesbt, HWPERMS);
531 cdp->cr_uperms = (uint32_t)CC128_EXTRACT_FIELD(pesbt, UPERMS);
532 uint32_t BWidth = CC128_BOT_WIDTH;
533 uint32_t BMask = (1u << BWidth) - 1;
534 uint32_t TMask = BMask >> 2;
535
536 bool IE = (bool)(uint32_t)CC128_EXTRACT_FIELD(pesbt, INTERNAL_EXPONENT);
537
538 uint8_t E, L_msb;
539 uint32_t B;
540 uint32_t T;
541
542 if (IE) {
543 E = (uint8_t)(CC128_EXTRACT_FIELD(pesbt, EXPONENT_LOW_PART) |
544 (CC128_EXTRACT_FIELD(pesbt, EXPONENT_HIGH_PART) << CC128_FIELD_EXPONENT_LOW_PART_SIZE));
545 // Do not offset by 1! We also need to encode E=0 even with IE
546 // Also allow nonsense values over 64 - BWidth + 2: this is expected by sail-generated tests
547 // E = MIN(64 - BWidth + 2, E);
548 B = (uint32_t)CC128_EXTRACT_FIELD(pesbt, EXP_NONZERO_BOTTOM) << CC128_FIELD_EXPONENT_LOW_PART_SIZE;
549 T = (uint32_t)CC128_EXTRACT_FIELD(pesbt, EXP_NONZERO_TOP) << CC128_FIELD_EXPONENT_HIGH_PART_SIZE;
550 L_msb = 1;
551 } else {
552 E = 0;
553 L_msb = 0;
554 B = (uint32_t)CC128_EXTRACT_FIELD(pesbt, EXP_ZERO_BOTTOM);
555 T = (uint32_t)CC128_EXTRACT_FIELD(pesbt, EXP_ZERO_TOP);
556 }
557 cdp->cr_otype = (uint32_t)CC128_EXTRACT_FIELD(pesbt, OTYPE);
558#endif
559
560 /*
561 Reconstruct top two bits of T given T = B + len and:
562 1) the top two bits of B
563 2) most significant two bits of length derived from format above
564 3) carry out of B[20..0] + len[20..0] that is implied if T[20..0] < B[20..0]
565 */
566 uint8_t L_carry = T < (B & TMask) ? 1 : 0;
567 uint64_t BTop2 = cc128_getbits(B, CC128_MANTISSA_WIDTH - 2, 2);
568 uint8_t T_infer = (BTop2 + L_carry + L_msb) & 0x3;
569 T |= ((uint32_t)T_infer) << (BWidth - 2);
570 E = MIN(CC128_MAX_EXPONENT, E);
571
572 /* Extract bits we need to make the top correction and calculate representable limit */
573 // let a3 = truncate(a >> (E + 11), 3) in
574 // let B3 = truncateLSB(c.B, 3) in
575 // let T3 = truncateLSB(c.T, 3) in
576#if 0
577 unsigned a3 = (unsigned)cc128_truncate64(cursor >> (E + CC128_MANTISSA_WIDTH - 3), 3);
578 unsigned B3 = (unsigned)cc128_truncateLSB_generic(CC128_MANTISSA_WIDTH)(B, 3);
579 unsigned T3 = (unsigned)cc128_truncateLSB_generic(CC128_MANTISSA_WIDTH)(T, 3);
580#else
581 unsigned a3 = (unsigned)cc128_getbits(cursor, E + CC128_MANTISSA_WIDTH - 3, 3);
582 unsigned B3 = (unsigned)cc128_getbits(B, CC128_MANTISSA_WIDTH - 3, 3);
583 unsigned T3 = (unsigned)cc128_getbits(T, CC128_MANTISSA_WIDTH - 3, 3);
584#endif
585 // let R3 = B3 - 0b001 in /* wraps */
586 unsigned R3 = (unsigned)cc128_truncate64(B3 - 1, 3); //B3 == 0 ? 7 : B3 - 1;
587 /* Do address, base and top lie in the R aligned region above the one containing R? */
588 // let aHi = if a3 <_u R3 then 1 else 0 in
589 // let bHi = if B3 <_u R3 then 1 else 0 in
590 // let tHi = if T3 <_u R3 then 1 else 0 in
591 int aHi = a3 < R3 ? 1 : 0;
592 int bHi = B3 < R3 ? 1 : 0;
593 int tHi = T3 < R3 ? 1 : 0;
594
595 /* Compute region corrections for top and base relative to a */
596 // let correction_base = bHi - aHi in
597 // let correction_top = tHi - aHi in
598 int correction_base = bHi - aHi;
599 int correction_top = tHi - aHi;
600 // let a_top = (a >> (E + mantissa_width)) in {
601 // let a_top = (a >> (E + 14)) in
602 // Note: shifting by 64 is a no-op and causes wrong results!
603 const unsigned a_top_shift = E + CC128_MANTISSA_WIDTH;
604 uint64_t a_top = a_top_shift >= CC128_CAP_ADDR_WIDTH ? 0 : cursor >> a_top_shift;
605
606 // base : CapLenBits = truncate((a_top + correction_base) @ c.B @ zeros(E), cap_len_width);
607 unsigned __int128 base = (uint64_t)((int64_t)a_top + correction_base);
608 base <<= CC128_MANTISSA_WIDTH;
609 base |= B;
610 base <<= E;
611 base &= ((unsigned __int128)1 << CC128_CAP_LEN_WIDTH) - 1;
612 assert((uint64_t)(base >> CC128_CAP_ADDR_WIDTH) <= 1); // max 65 bits
613 // top : truncate((a_top + correction_top) @ c.T @ zeros(E), cap_len_width);
614 unsigned __int128 top = (uint64_t)((int64_t)a_top + correction_top);
615 top <<= CC128_MANTISSA_WIDTH;
616 top |= T;
617 top <<= E;
618 top &= ((unsigned __int128)1 << CC128_CAP_LEN_WIDTH) - 1;
619 assert((uint64_t)(top >> CC128_CAP_ADDR_WIDTH) <= 1); // max 65 bits
620
621 // if (base[cap_addr_width] == bitone) then {
622 bool base_high = (base >> CC128_CAP_ADDR_WIDTH);
623 if (base_high) {
624 /* If base[64] is set this indicates under or overflow i.e. a has
625 * wrapped around the address space and been corrected. In this case
626 * we need to correct top[64] because top is not quite modulo 2**64 due
627 * to having max top == 2**64 in one particular case:
628 */
629 // top[cap_addr_width] = if (aHi == 1) & (tHi == 1) then bitone else bitzero;
630 if (aHi && tHi) {
631 top |= ((unsigned __int128)1 << 64);
632 } else {
633 top &= UINT64_MAX; // clear top bit
634 }
635 base = (uint64_t)base; // strip the high bit
636 }
637 assert(base <= CC128_CAP_ADDR_MAX);
638
639 /* The following is apparently equivalent to above and used by hw. */
640 /*
641 let base2 : bits(2) = bitzero @ base[63];
642 let top2 : bits(2) = top[64..63];
643 if (E < (unsigned(resetE) - 1)) & (unsigned(top2 - base2) > 1) then {
644 top[64] = ~(top[64]);
645 };
646 */
647 assert((uint64_t)(top >> 64) <= 1); // should be at most 1 bit over
648
649 // TODO: should really store top and not length!
650 cdp->_cr_length = top - base;
651 if (top < base) {
652 // This can happen for invalid capabilities with arbitrary bit patterns
653 assert((signed __int128)cdp->_cr_length < 0); // must be negative
654 }
655 cdp->cr_offset = cursor - (uint64_t)base;
656 cdp->cr_base = (uint64_t)base;
657}

671/*
672 * Compress a capability to 128 bits.
673 */
674static inline uint64_t compress_128cap_without_xor(const cap_register_t* csp) {
675#ifdef CC128_OLD_FORMAT
676 int seal_mode = csp->cr_otype <= CAP_MAX_SEALED_OTYPE ? CC_SEAL_MODE_SEALED : CC_SEAL_MODE_UNSEALED;
677 if (csp->cr_otype == CAP_OTYPE_SENTRY)
678 seal_mode = CC_SEAL_MODE_SENTRY;
679
680 uint8_t LH;
681 uint32_t BWidth = seal_mode == CC_SEAL_MODE_SEALED ? CC_L_SEALED_BWIDTH : CC_L_BWIDTH;
682 CC128_STATIC_ASSERT(CC128_BOT_WIDTH == 23, "This code assumes 14-bit bot");
683 CC128_STATIC_ASSERT(CC128_BOT_INTERNAL_EXP_WIDTH == 20, "This code assumes 14-bit bot");
684#else
685 CC128_STATIC_ASSERT(CC128_BOT_WIDTH == 14, "This code assumes 14-bit bot");
686 CC128_STATIC_ASSERT(CC128_BOT_INTERNAL_EXP_WIDTH == 11, "This code assumes 14-bit bot");
687#endif
688 uint64_t base = csp->cr_base;
689 const unsigned __int128 top = csp->cr_base + csp->_cr_length;
690 const unsigned __int128 length = csp->_cr_length;
691
692 uint8_t E = (uint8_t)cc128_get_exponent(length);
693 const uint64_t length64 = (uint64_t)length;
694 // from sail: need IE if length bit 12 is set: let ie = (e != 0) | length[12];
695 // 0x1000 (bwidth - 2 bit set) is the first value that cannot be encoded
696 // without the internal exponent:
697 // Note: 12 = BWidth - 2 / BWidth with internal exponent plus one
698 bool IE = E != 0 || cc128_getbits(length64, CC128_BOT_INTERNAL_EXP_WIDTH + 1, 1);
699 uint64_t Be;
700 uint64_t Te;
701#ifdef CC128_OLD_FORMAT
702 uint32_t BMask = (1u << BWidth) - 1;
703 uint32_t TMask = BMask >> 2;
704 if (top > UINT64_MAX) {
705 Te = (UINT64_C(1) << (64 - E)) & TMask;
706 } else {
707 E = (uint8_t)_cc128_compute_e(length64, BWidth);
708 Te = (top64 >> E) & TMask;
709 }
710 Be = (base >> E) & BMask;
711 if (IE) {
712 LH = E >> (2 * CC_L_LOWWIDTH);
713 Be |= (E >> CC_L_LOWWIDTH) & CC_L_LOWMASK;
714 Te |= E & CC_L_LOWMASK;
715 } else {
716 LH = (length64 >> (BWidth - 2)) & 1;
717 }
718 if (seal_mode == CC_SEAL_MODE_SEALED) {
719 uint64_t hi = ((uint64_t)csp->cr_otype >> (CC_L_TYPES / 2)) & ((1 << (CC_L_TYPES / 2)) - 1);
720 uint64_t lo = (uint64_t)csp->cr_otype & ((1 << (CC_L_TYPES / 2)) - 1);
721 Te |= hi << (CC_L_SEALED_BWIDTH - 2);
722 Be |= lo << CC_L_SEALED_BWIDTH;
723 }
724 // assert(seal_mode <= 3);
725 uint64_t perms = ((uint64_t)csp->cr_uperms << CC128_UPERMS_MEM_SHFT) | (uint64_t)csp->cr_perms;
726 uint64_t pesbt =
727 ((((((((((perms << 2) | (uint64_t)seal_mode) << 1) | (uint64_t)IE) << 1) | (uint64_t)LH) << (CC_L_BWIDTH - 2)) |
728 (uint64_t)Te)
729 << CC_L_BWIDTH) |
730 (uint64_t)Be);
731#else
732 if (!IE) {
733 // precisely representable -> just extract the bits
734 assert(top <= UINT64_MAX); // must be 64 bits
735 Be = cc128_truncate64(base, CC128_FIELD_BOTTOM_ENCODED_SIZE);
736 Te = cc128_truncate64((uint64_t)top, CC128_FIELD_TOP_ENCODED_SIZE);
737 } else {
738 uint64_t bot_ie = cc128_truncate64(base >> (E + CC128_EXP_LOW_WIDTH), CC128_BOT_INTERNAL_EXP_WIDTH);
739 uint64_t top_ie = cc128_truncate64((uint64_t)top >> (E + CC128_EXP_LOW_WIDTH), CC128_BOT_INTERNAL_EXP_WIDTH);
740 // /* Find out whether we have lost significant bits of base and top using a
741 // mask of bits that we will lose (including 3 extra for exp). */
742 // maskLo : bits(65) = zero_extend(replicate_bits(0b1, e + 3));
743 // z65 : bits(65) = zeros();
744 // lostSignificantBase = (base65 & maskLo) != z65;
745 // lostSignificantTop = (top & maskLo) != z65;
746 // TODO: stop using unsigned __int128 and just handle bit65 set specially?
747 const unsigned __int128 maskLo = (((unsigned __int128)1u) << (E + CC128_EXP_LOW_WIDTH)) - 1;
748 const unsigned __int128 zero65 = 0;
749 bool lostSignificantBase = (base & maskLo) != zero65;
750 bool lostSignificantTop = (top & maskLo) != zero65;
751 // if lostSignificantTop then {
752 // /* we must increment T to make sure it is still above top even with lost bits.
753 // It might wrap around but if that makes B<T then decoding will compensate. */
754 // T_ie = T_ie + 1;
755 // };
756 if (lostSignificantTop) {
757 top_ie = cc128_truncate64(top_ie + 1, CC128_BOT_INTERNAL_EXP_WIDTH);
758 }
759 // /* Has the length overflowed? We chose e so that the top two bits of len would be 0b01,
760 // but either because of incrementing T or losing bits of base it might have grown. */
761 // len_ie = T_ie - B_ie;
762 // if len_ie[10] then {
763 // /* length overflow -- increment E by one and then recalculate
764 // T, B etc accordingly */
765 // incE = true;
766 //
767 // lostSignificantBase = lostSignificantBase | B_ie[0];
768 // lostSignificantTop = lostSignificantTop | T_ie[0];
769 //
770 // B_ie = truncate(base >> (e + 4), 11);
771 // let incT : range(0,1) = if lostSignificantTop then 1 else 0;
772 // T_ie = truncate(top >> (e + 4), 11) + incT;
773 // };
774 const uint64_t len_ie = cc128_truncate64(top_ie - bot_ie, CC128_BOT_INTERNAL_EXP_WIDTH);
775 bool incE = false;
776 if (cc128_getbits(len_ie, CC128_BOT_INTERNAL_EXP_WIDTH - 1, 1)) {
777 incE = true;
778 lostSignificantBase = lostSignificantBase || cc128_getbits(bot_ie, 0, 1);
779 lostSignificantTop = lostSignificantTop || cc128_getbits(top_ie, 0, 1);
780 bot_ie = cc128_truncate64(base >> (E + CC128_EXP_LOW_WIDTH + 1), CC128_BOT_INTERNAL_EXP_WIDTH);
781 const bool incT = lostSignificantTop;
782 top_ie = cc128_truncate64((uint64_t)(top >> (E + CC128_EXP_LOW_WIDTH + 1)), CC128_BOT_INTERNAL_EXP_WIDTH);
783 if (incT) {
784 top_ie = cc128_truncate64(top_ie + 1, CC128_BOT_INTERNAL_EXP_WIDTH);
785 }
786 }
787 // Bbits = B_ie @ 0b000;
788 // Tbits = T_ie @ 0b000;
789 Be = bot_ie << CC128_FIELD_EXPONENT_LOW_PART_SIZE;
790 Te = top_ie << CC128_FIELD_EXPONENT_LOW_PART_SIZE;
791 const uint8_t newE = E + (incE ? 1 : 0);
792 // Split E between T and B
793 const uint64_t expHighBits = cc128_getbits(newE >> CC128_FIELD_EXPONENT_LOW_PART_SIZE , 0, CC128_FIELD_EXPONENT_HIGH_PART_SIZE);
794 const uint64_t expLowBits = cc128_getbits(newE, 0, CC128_FIELD_EXPONENT_LOW_PART_SIZE);
795 assert(cc128_getbits(Te, 0, CC128_FIELD_EXPONENT_HIGH_PART_SIZE) == 0);
796 assert(cc128_getbits(Be, 0, CC128_FIELD_EXPONENT_LOW_PART_SIZE) == 0);
797 Te |= expHighBits;
798 Be |= expLowBits;
799 }
800 uint64_t pesbt =
801 CC128_ENCODE_FIELD(csp->cr_uperms, UPERMS) |
802 CC128_ENCODE_FIELD(csp->cr_perms, HWPERMS) |
803 CC128_ENCODE_FIELD(csp->cr_otype, OTYPE) |
804 CC128_ENCODE_FIELD(IE, INTERNAL_EXPONENT) |
805 CC128_ENCODE_FIELD(Te, TOP_ENCODED) |
806 CC128_ENCODE_FIELD(Be, BOTTOM_ENCODED);
807 // For untagged values we add the initially loaded reserved bits in the reserved field:
808 if (!csp->cr_tag)
809 pesbt |= CC128_ENCODE_FIELD(CC128_EXTRACT_FIELD(csp->cr_pesbt_xored_for_mem ^ CC128_NULL_XOR_MASK, RESERVED), RESERVED);
810#endif
811 return pesbt;
812}

868/*
869 * Check to see if a memory region is representable by a compressed
870 * capability. It is representable if:
871 *
872 * representable = (inRange && inLimits) || (E >= 44)
873 *
874 * where:
875 *
876 * E = compression exponent (see _cc128_compute_e() above)
877 *
878 * inRange = -s < i < s where i is the increment (or offset)
879 * (In other words, all the bits of i<63, E+20> are the same.)
880 *
881 * inLimits = (i < 0) ? (Imid >= (R - Amid)) && (R != Amid) : (R - Amid - 1)
882 * where Imid = i<E+19, E>, Amid = a<E+19, E>, R = B - 2^12 and a =
883 * base + offset.
884 */
885static inline bool cc128_is_representable(bool sealed, uint64_t base, unsigned __int128 length, uint64_t offset,
886 uint64_t new_offset) {
887#if defined(CC128_USE_FAST_REP_CHECK)
888 const bool slow_representable_check = false;
889#else
890 const bool slow_representable_check = true;
891#endif
892
893#if defined(CC128_OLD_FORMAT)
894 // I change the precision going between unsealed->sealed so the fast check
895 // doesn't work. Instead just compress/decompress.
896 if (slow_representable_check || sealed) {
897#else
898 if (slow_representable_check) {
899#endif // CC128_OLD_FORMAT
900 cap_register_t c;
901 memset(&c, 0, sizeof(c));
902 /* Simply compress and uncompress to check. */
903 c.cr_base = base;
904 c._cr_length = length;
905 c.cr_offset = new_offset;
906 c.cr_otype = sealed ? 42 : CC128_OTYPE_UNSEALED; // important to set as compress assumes this is in bounds
907 return cc128_is_representable_cap_exact(&c);
908 } else {
909 return fast_cc128_is_representable(sealed, base, length, offset, new_offset);
910 }
911}
912
913static bool fast_cc128_is_representable(bool sealed, uint64_t base, unsigned __int128 length, uint64_t offset,
914 uint64_t new_offset) {
915 (void)sealed;
916 uint32_t bwidth = CC128_BOT_WIDTH;
917 uint32_t highest_exp = (64 - bwidth + 2);
918
919 unsigned __int128 top = base + length;
920 // If top is 0xffff... we assume we meant it to be 1 << 64
921 if (top == CAP_MAX_ADDRESS_PLUS_ONE && base == 0) {
922 return true; // 1 << 65 is always representable
923 }
924 if (length == 0) {
925 return true; // length 0 is always representable
926 }
927
928 uint32_t e = cc128_get_exponent(length);
929
930 int64_t b, r, Imid, Amid;
931 bool inRange, inLimits;
932 int64_t inc = (int64_t)(new_offset - offset);
933
934#define MOD_MASK ((UINT64_C(1) << bwidth) - UINT64_C(1))
935
936 /* Check for the boundary cases. */
937
938 b = (int64_t)((base >> e) & MOD_MASK);
939 Imid = (int64_t)((uint64_t)(inc >> e) & MOD_MASK);
940 Amid = (int64_t)(((base + offset) >> e) & MOD_MASK);
941
942 r = (int64_t)(((uint64_t)((b >> (bwidth - 3)) - 1) << (bwidth - 3)) & MOD_MASK);
943
944 /* inRange, test if bits are all the same */
945 inRange = cc128_all_ones((uint64_t)inc, e, bwidth) || cc128_all_zeroes((uint64_t)inc, e, bwidth);
946
947 /* inLimits */
948 if (inc >= 0) {
949 inLimits = ((uint64_t)Imid < (((uint64_t)(r - Amid - 1l)) & MOD_MASK));
950 } else {
951 inLimits = ((uint64_t)Imid >= (((uint64_t)(r - Amid)) & MOD_MASK)) && (r != Amid);
952 }
953#undef MOD_MASK
954
955 return ((inRange && inLimits) || (e >= highest_exp));
956}

958/* @return whether the operation was able to set precise bounds precise or not */
959static inline bool cc128_setbounds_impl(cap_register_t* cap, uint64_t req_base, unsigned __int128 req_top, uint64_t* alignment_mask) {
960 assert(cap->cr_tag && "Cannot be used on untagged capabilities");
961 assert(!cc128_is_cap_sealed(cap) && "Cannot be used on sealed capabilities");
962 assert(req_base <= req_top && "Cannot grow capabilities");
963 /*
964 * With compressed capabilities we may need to increase the range of
965 * memory addresses to be wider than requested so it is
966 * representable.

984 // let e = 52 - CountLeadingZeros(length[64..13]);

1009 // Tbits = truncate(top, 14);
1010 // lostSignificantTop : bool = false;

1101#else

1107 CC128_ENCODE_FIELD(0, HWPERMS) |
1108#ifdef CC128_OLD_FORMAT

1111#else

1113#endif

1137/* @return the mask that needs to be applied to base in order to get a precisely representable capability */
1138static inline uint64_t cc128_get_alignment_mask(uint64_t req_length) {
1139 if (req_length == 0) {
1140 // With a lenght of zero we know it is precise so we can just return an
1141 // all ones mask.
1142 // This avoids undefined behaviour when counting most significant bit later.
1143 return UINT64_MAX;
1144 }
1145 // To compute the mask we set bounds on a maximum permissions capability and
1146 // return the mask that was used to adjust the length
1147 cap_register_t tmpcap;
1148 memset(&tmpcap, 0, sizeof(tmpcap));
1149 tmpcap.cr_tag = 1;
1150 tmpcap.cr_base = 0;
1151 tmpcap._cr_length = CC128_MAX_LENGTH;
1152 tmpcap.cr_otype = CC128_OTYPE_UNSEALED;
1153 uint64_t mask = 0;
1154 // Ensure that the base always needs rounding down by making it all ones until
1155 // one bit before the most significant bit in length
1156 // uint64_t req_base = UINT64_MAX & ~(UINT64_C(1) << cc128_idx_MSNZ(req_length));
1157 uint64_t req_base = UINT64_MAX - req_length;
1158 tmpcap.cr_offset = req_base;
1159 cc128_setbounds_impl(&tmpcap, req_base, req_base + req_length, &mask);
1160 // base should have been rounded down using this mask:
1161 assert((req_base & mask) == tmpcap.cr_base);
1162 return mask;
1163}

1167 // the bits and add one to get a power-of-two

1188static inline void decompress_256cap(inmemory_chericap256 mem, cap_register_t* cdp, bool tagged) {
1189 /* See CHERI ISA: Figure 3.1: 256-bit memory representation of a capability */
1190 cdp->_sbit_for_memory = mem.u64s[0] & 1;
1191 uint32_t hwperms_mask = tagged ? CC256_PERMS_ALL_BITS: CC256_PERMS_ALL_BITS_UNTAGGED;
1192 cdp->cr_perms = (mem.u64s[0] >> CC256_PERMS_MEM_SHFT) & hwperms_mask;
1193 cdp->cr_uperms = (mem.u64s[0] >> CC256_UPERMS_MEM_SHFT) & CC256_UPERMS_ALL_BITS;
1194 cdp->cr_otype = (mem.u64s[0] >> CC256_OTYPE_MEM_SHFT) ^ CC256_OTYPE_UNSEALED;
1195 cdp->cr_base = mem.u64s[2];
1196 /* Length is xor'ed with -1 to ensure that NULL is all zeroes in memory */
1197 cdp->_cr_length = mem.u64s[3] ^ CC256_NULL_LENGTH;
1198 /* TODO: should just have a cr_cursor instead... But that's not the way QEMU works */
1199 cdp->cr_offset = mem.u64s[1] - cdp->cr_base;
1200 cdp->cr_tag = tagged;
1201}

1203static inline void compress_256cap(inmemory_chericap256* buffer, const cap_register_t* csp) {
1204
1205 bool sealed_bit = csp->cr_tag ? cc256_is_cap_sealed(csp) : csp->_sbit_for_memory;
1206 // When writing an untagged value, just write back the bits that were loaded (including the reserved HWPERMS)
1207 uint64_t hwperms_mask = csp->cr_tag ? CC256_PERMS_ALL_BITS: CC256_PERMS_ALL_BITS_UNTAGGED;
1208 buffer->u64s[0] =
1209 sealed_bit |
1210 ((csp->cr_perms & hwperms_mask) << CC256_PERMS_MEM_SHFT) |
1211 ((csp->cr_uperms & CC256_UPERMS_ALL_BITS) << CC256_UPERMS_MEM_SHFT) |
1212 ((uint64_t)(csp->cr_otype ^ CC256_OTYPE_UNSEALED) << CC256_OTYPE_MEM_SHFT);
1213 buffer->u64s[1] = csp->cr_base + csp->cr_offset;
1214 buffer->u64s[2] = csp->cr_base;
1215 uint64_t length64 = csp->_cr_length > UINT64_MAX ? UINT64_MAX : (uint64_t)csp->_cr_length;
1216 buffer->u64s[3] = length64 ^ CC256_NULL_LENGTH;
1217}
