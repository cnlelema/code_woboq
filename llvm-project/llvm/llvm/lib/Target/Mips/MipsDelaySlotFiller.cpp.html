83enum CompactBranchPolicy {
84 CB_Never, ///< The policy 'never' may in some circumstances or for some
85 ///< ISAs not be absolutely adhered to.
86 CB_Optimal, ///< Optimal is the default and will produce compact branches
87 ///< when delay slots cannot be filled.
88 CB_Always ///< 'always' may in some circumstances may not be
89 ///< absolutely adhered to there may not be a corresponding
90 ///< compact form of a branch.

220 bool runOnMachineFunction(MachineFunction &F) override {
221 TM = &F.getTarget();
222 bool Changed = false;
223 for (MachineFunction::iterator FI = F.begin(), FE = F.end();

311/// This function inserts clones of Filler into predecessor blocks.
312static void insertDelayFiller(Iter Filler, const BB2BrMap &BrMap) {
313 MachineFunction *MF = Filler->getParent()->getParent();
314
315 for (BB2BrMap::const_iterator I = BrMap.begin(); I != BrMap.end(); ++I) {
316 if (I->second) {
317 MIBundleBuilder(I->second).append(MF->CloneMachineInstr(&*Filler));
318 ++UsefulSlots;
319 } else {
320 I->first->insert(I->first->end(), MF->CloneMachineInstr(&*Filler));
321 }
322 }
323}

325/// This function adds registers Filler defines to MBB's live-in register list.
326static void addLiveInRegs(Iter Filler, MachineBasicBlock &MBB) {
327 for (unsigned I = 0, E = Filler->getNumOperands(); I != E; ++I) {
328 const MachineOperand &MO = Filler->getOperand(I);
329 unsigned R;
330
331 if (!MO.isReg() || !MO.isDef() || !(R = MO.getReg()))
332 continue;
333
334#ifndef NDEBUG
335 const MachineFunction &MF = *MBB.getParent();
336 assert(MF.getSubtarget().getRegisterInfo()->getAllocatableSet(MF).test(R) &&
337 "Shouldn't move an instruction with unallocatable registers across "
338 "basic block boundaries.");
339#endif
340
341 if (!MBB.isLiveIn(R))
342 MBB.addLiveIn(R);
343 }

349void RegDefsUses::init(const MachineInstr &MI) {
350 // Add all register operands which are explicit and non-variadic.
351 update(MI, 0, MI.getDesc().getNumOperands());
352
353 // If MI is a call, add RA to Defs to prevent users of RA from going into
354 // delay slot.
355 if (MI.isCall()) {
356 assert(MI.getDesc().getNumImplicitDefs() <= 2 &&
357 "Expected one implicit def for call instruction");
358 for (unsigned i = 0; i < MI.getDesc().getNumImplicitDefs(); i++) {
359 MCPhysReg Reg = MI.getDesc().getImplicitDefs()[i];
360 // XXXAR: currently $cgp is marked as a def for cjalr since I don't see
361 // a better way to ensure that $cgp is saved and restored prior to the
362 // call. However, $cgp will only be clobbered after the cjalr instruction
363 // has been executed (and will still have the old value in the delay slot)
364 // so we should not count it as a def for the delay slot filler.
365 // Otherwise we can't move the restore of $cgp prior to the next call into
366 // delay slots.
367 if (Reg == Mips::C26)
368 continue;
369 Defs.set(Reg);
370 }
371 }
372
373 // Add all implicit register operands of branch instructions except
374 // register AT.
375 if (MI.isBranch()) {
376 update(MI, MI.getDesc().getNumOperands(), MI.getNumOperands());
377 Defs.reset(Mips::AT);
378 }
379}

381void RegDefsUses::setCallerSaved(const MachineInstr &MI) {
382 assert(MI.isCall());
383
384 // Add RA/RA_64 to Defs to prevent users of RA/RA_64 from going into
385 // the delay slot. The reason is that RA/RA_64 must not be changed
386 // in the delay slot so that the callee can return to the caller.
387 if (MI.definesRegister(Mips::RA) || MI.definesRegister(Mips::RA_64)) {
388 Defs.set(Mips::RA);
389 Defs.set(Mips::RA_64);
390 }
391
392 // If MI is a call, add all caller-saved registers to Defs.
393 BitVector CallerSavedRegs(TRI.getNumRegs(), true);
394
395 CallerSavedRegs.reset(Mips::ZERO);
396 CallerSavedRegs.reset(Mips::ZERO_64);
397
398 for (const MCPhysReg *R = TRI.getCalleeSavedRegs(MI.getParent()->getParent());
399 *R; ++R)
400 for (MCRegAliasIterator AI(*R, &TRI, true); AI.isValid(); ++AI)
401 CallerSavedRegs.reset(*AI);
402
403 Defs |= CallerSavedRegs;
404}

406void RegDefsUses::setUnallocatableRegs(const MachineFunction &MF) {
407 BitVector AllocSet = TRI.getAllocatableSet(MF);
408 // If we're not a CHERI target, then the C0 register is just a convenient
409 // fiction and shouldn't impede optimisation.
410 if (!MF.getSubtarget<MipsSubtarget>().isCheri())
411 AllocSet[Mips::DDC] = true;
412
413 for (unsigned R : AllocSet.set_bits())
414 for (MCRegAliasIterator AI(R, &TRI, false); AI.isValid(); ++AI)
415 AllocSet.set(*AI);
416
417 AllocSet.set(Mips::ZERO);
418 AllocSet.set(Mips::ZERO_64);
419 AllocSet.set(Mips::CNULL);
420
421 Defs |= AllocSet.flip();
422}

424void RegDefsUses::addLiveOut(const MachineBasicBlock &MBB,
425 const MachineBasicBlock &SuccBB) {
426 for (MachineBasicBlock::const_succ_iterator SI = MBB.succ_begin(),
427 SE = MBB.succ_end(); SI != SE; ++SI)
428 if (*SI != &SuccBB)
429 for (const auto &LI : (*SI)->liveins())
430 Uses.set(LI.PhysReg);
431}

433bool RegDefsUses::update(const MachineInstr &MI, unsigned Begin, unsigned End) {
434 BitVector NewDefs(TRI.getNumRegs()), NewUses(TRI.getNumRegs());
435 bool HasHazard = false;
436
437 for (unsigned I = Begin; I != End; ++I) {
438 const MachineOperand &MO = MI.getOperand(I);
439
440 if (MO.isReg() && MO.getReg())
441 HasHazard |= checkRegDefsUses(NewDefs, NewUses, MO.getReg(), MO.isDef());
442 }
443
444 Defs |= NewDefs;
445 Uses |= NewUses;
446
447 return HasHazard;
448}

450bool RegDefsUses::checkRegDefsUses(BitVector &NewDefs, BitVector &NewUses,
451 unsigned Reg, bool IsDef) const {
452 if (IsDef) {
453 NewDefs.set(Reg);
454 // check whether Reg has already been defined or used.
455 return (isRegInSet(Defs, Reg) || isRegInSet(Uses, Reg));
456 }
457
458 NewUses.set(Reg);
459 // check whether Reg has already been defined.
460 return isRegInSet(Defs, Reg);
461}

463bool RegDefsUses::isRegInSet(const BitVector &RegSet, unsigned Reg) const {
464 // Check Reg and all aliased Registers.
465 for (MCRegAliasIterator AI(Reg, &TRI, true); AI.isValid(); ++AI)
466 if (RegSet.test(*AI))
467 return true;
468 return false;
469}

471bool InspectMemInstr::hasHazard(const MachineInstr &MI) {
472 if (!MI.mayStore() && !MI.mayLoad())
473 return false;
474
475 if (ForbidMemInstr)
476 return true;
477
478 OrigSeenLoad = SeenLoad;
479 OrigSeenStore = SeenStore;
480 SeenLoad |= MI.mayLoad();
481 SeenStore |= MI.mayStore();
482
483 // If MI is an ordered or volatile memory reference, disallow moving
484 // subsequent loads and stores to delay slot.
485 if (MI.hasOrderedMemoryRef() && (OrigSeenLoad || OrigSeenStore)) {
486 ForbidMemInstr = true;
487 return true;
488 }
489
490 return hasHazard_(MI);
491}
492
493bool LoadFromStackOrConst::hasHazard_(const MachineInstr &MI) {
494 if (MI.mayStore())
495 return true;
496
497 if (!MI.hasOneMemOperand() || !(*MI.memoperands_begin())->getPseudoValue())
498 return true;
499
500 if (const PseudoSourceValue *PSV =
501 (*MI.memoperands_begin())->getPseudoValue()) {
502 if (isa<FixedStackPseudoSourceValue>(PSV))
503 return false;
504 return !PSV->isConstant(nullptr) && !PSV->isStack();
505 }
506
507 return true;
508}

513bool MemDefsUses::hasHazard_(const MachineInstr &MI) {
514 bool HasHazard = false;
515
516 // Check underlying object list.
517 SmallVector<ValueType, 4> Objs;
518 if (getUnderlyingObjects(MI, Objs)) {
519 for (ValueType VT : Objs)
520 HasHazard |= updateDefsUses(VT, MI.mayStore());
521 return HasHazard;
522 }
523
524 // No underlying objects found.
525 HasHazard = MI.mayStore() && (OrigSeenLoad || OrigSeenStore);
526 HasHazard |= MI.mayLoad() || OrigSeenStore;
527
528 SeenNoObjLoad |= MI.mayLoad();
529 SeenNoObjStore |= MI.mayStore();
530
531 return HasHazard;
532}
533
534bool MemDefsUses::updateDefsUses(ValueType V, bool MayStore) {
535 if (MayStore)
536 return !Defs.insert(V).second || Uses.count(V) || SeenNoObjStore ||
537 SeenNoObjLoad;
538
539 Uses.insert(V);
540 return Defs.count(V) || SeenNoObjStore;
541}
542
543bool MemDefsUses::
544getUnderlyingObjects(const MachineInstr &MI,
545 SmallVectorImpl<ValueType> &Objects) const {
546 if (!MI.hasOneMemOperand())
547 return false;
548
549 auto & MMO = **MI.memoperands_begin();
550
551 if (const PseudoSourceValue *PSV = MMO.getPseudoValue()) {
552 if (!PSV->isAliased(MFI))
553 return false;
554 Objects.push_back(PSV);
555 return true;
556 }
557
558 if (const Value *V = MMO.getValue()) {
559 SmallVector<const Value *, 4> Objs;
560 GetUnderlyingObjects(V, Objs, DL);
561
562 for (const Value *UValue : Objs) {
563 if (!isIdentifiedObject(V))
564 return false;
565
566 Objects.push_back(UValue);
567 }
568 return true;
569 }
570
571 return false;
572}

574// Replace Branch with the compact branch instruction.
575Iter MipsDelaySlotFiller::replaceWithCompactBranch(MachineBasicBlock &MBB,
576 Iter Branch,
577 const DebugLoc &DL) {
578 const MipsSubtarget &STI = MBB.getParent()->getSubtarget<MipsSubtarget>();
579 const MipsInstrInfo *TII = STI.getInstrInfo();
580
581 unsigned NewOpcode = TII->getEquivalentCompactForm(Branch);
582 Branch = TII->genInstrWithNewOpc(NewOpcode, Branch);
583
584 std::next(Branch)->eraseFromParent();
585 return Branch;
586}

588// For given opcode returns opcode of corresponding instruction with short
589// delay slot.
590// For the pseudo TAILCALL*_MM instructions return the short delay slot
591// form. Unfortunately, TAILCALL<->b16 is denied as b16 has a limited range
592// that is too short to make use of for tail calls.
593static int getEquivalentCallShort(int Opcode) {
594 switch (Opcode) {
595 case Mips::BGEZAL:
596 return Mips::BGEZALS_MM;
597 case Mips::BLTZAL:
598 return Mips::BLTZALS_MM;
599 case Mips::JAL:
600 case Mips::JAL_MM:
601 return Mips::JALS_MM;
602 case Mips::JALR:
603 return Mips::JALRS_MM;
604 case Mips::JALR16_MM:
605 return Mips::JALRS16_MM;
606 case Mips::TAILCALL_MM:
607 llvm_unreachable("Attempting to shorten the TAILCALL_MM pseudo!");
608 case Mips::TAILCALLREG:
609 return Mips::JR16_MM;
610 default:
611 llvm_unreachable("Unexpected call instruction for microMIPS.");
612 }
613}

615/// runOnMachineBasicBlock - Fill in delay slots for the given basic block.
616/// We assume there is only one delay slot per delayed instruction.
617bool MipsDelaySlotFiller::runOnMachineBasicBlock(MachineBasicBlock &MBB) {
618 bool Changed = false;
619 const MipsSubtarget &STI = MBB.getParent()->getSubtarget<MipsSubtarget>();
620 bool InMicroMipsMode = STI.inMicroMipsMode();
621 const MipsInstrInfo *TII = STI.getInstrInfo();
622
623 for (Iter I = MBB.begin(); I != MBB.end(); ++I) {
624 if (!hasUnoccupiedSlot(&*I))
625 continue;
626
627 // Delay slot filling is disabled at -O0, or in microMIPS32R6.
628 if (!DisableDelaySlotFiller && (TM->getOptLevel() != CodeGenOpt::None) &&
629 !(InMicroMipsMode && STI.hasMips32r6())) {
630
631 bool Filled = false;
632
633 if (MipsCompactBranchPolicy.getValue() != CB_Always ||
634 !TII->getEquivalentCompactForm(I)) {
635 if (searchBackward(MBB, *I)) {
636 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": found instruction for delay slot using forward search.\n");
637 Filled = true;
638 } else if (I->isTerminator()) {
639 if (searchSuccBBs(MBB, I)) {
640 Filled = true;
641 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": found instruction for delay slot using successor BB search.\n");
642 }
643 } else if (searchForward(MBB, I)) {
644 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": found instruction for delay slot using forward search.\n");
645 Filled = true;
646 }
647 }
648
649 if (Filled) {
650 // Get instruction with delay slot.
651 MachineBasicBlock::instr_iterator DSI = I.getInstrIterator();
652
653 if (InMicroMipsMode && TII->getInstSizeInBytes(*std::next(DSI)) == 2 &&
654 DSI->isCall()) {
655 // If instruction in delay slot is 16b change opcode to
656 // corresponding instruction with short delay slot.
657
658 // TODO: Implement an instruction mapping table of 16bit opcodes to
659 // 32bit opcodes so that an instruction can be expanded. This would
660 // save 16 bits as a TAILCALL_MM pseudo requires a fullsized nop.
661 // TODO: Permit b16 when branching backwards to the same function
662 // if it is in range.
663 DSI->setDesc(TII->get(getEquivalentCallShort(DSI->getOpcode())));
664 }
665 ++FilledSlots;
666 Changed = true;
667 continue;
668 }
669 }
670
671 // For microMIPS if instruction is BEQ or BNE with one ZERO register, then
672 // instead of adding NOP replace this instruction with the corresponding
673 // compact branch instruction, i.e. BEQZC or BNEZC. Additionally
674 // PseudoReturn and PseudoIndirectBranch are expanded to JR_MM, so they can
675 // be replaced with JRC16_MM.
676
677 // For MIPSR6 attempt to produce the corresponding compact (no delay slot)
678 // form of the CTI. For indirect jumps this will not require inserting a
679 // NOP and for branches will hopefully avoid requiring a NOP.
680 if ((InMicroMipsMode ||
681 (STI.hasMips32r6() && MipsCompactBranchPolicy != CB_Never)) &&
682 TII->getEquivalentCompactForm(I)) {
683 I = replaceWithCompactBranch(MBB, I, I->getDebugLoc());
684 Changed = true;
685 continue;
686 }
687
688 // Bundle the NOP to the instruction with the delay slot.
689 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": could not fill delay slot for";
690 I->dump());
691 BuildMI(MBB, std::next(I), I->getDebugLoc(), TII->get(Mips::NOP));
692 MIBundleBuilder(MBB, I, std::next(I, 2));
693 ++FilledSlots;
694 Changed = true;
695 }
696
697 return Changed;
698}

700template <typename IterTy>
701bool MipsDelaySlotFiller::searchRange(MachineBasicBlock &MBB, IterTy Begin,
702 IterTy End, RegDefsUses &RegDU,
703 InspectMemInstr &IM, Iter Slot,
704 IterTy &Filler) const {
705 for (IterTy I = Begin; I != End;) {
706 IterTy CurrI = I;
707 ++I;
708 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": checking instruction for delay slot: ";
709 CurrI->dump());
710 // skip debug value
711 if (CurrI->isDebugInstr()) {
712 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": ignoring debug instruction for delay slot: ";
713 CurrI->dump());
714 continue;
715 }
716
717 if (terminateSearch(*CurrI)) {
718 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": should terminate search: ";
719 CurrI->dump());
720 break;
721 }
722
723 assert((!CurrI->isCall() && !CurrI->isReturn() && !CurrI->isBranch()) &&
724 "Cannot put calls, returns or branches in delay slot.");
725
726 if (CurrI->isKill()) {
727 CurrI->eraseFromParent();
728 continue;
729 }
730
731 if (delayHasHazard(*CurrI, RegDU, IM))
732 continue;
733
734 const MipsSubtarget &STI = MBB.getParent()->getSubtarget<MipsSubtarget>();
735 if (STI.isTargetNaCl()) {
736 // In NaCl, instructions that must be masked are forbidden in delay slots.
737 // We only check for loads, stores and SP changes. Calls, returns and
738 // branches are not checked because non-NaCl targets never put them in
739 // delay slots.
740 unsigned AddrIdx;
741 if ((isBasePlusOffsetMemoryAccess(CurrI->getOpcode(), &AddrIdx) &&
742 baseRegNeedsLoadStoreMask(CurrI->getOperand(AddrIdx).getReg())) ||
743 CurrI->modifiesRegister(Mips::SP, STI.getRegisterInfo()))
744 continue;
745 }
746
747 bool InMicroMipsMode = STI.inMicroMipsMode();
748 const MipsInstrInfo *TII = STI.getInstrInfo();
749 unsigned Opcode = (*Slot).getOpcode();
750 // This is complicated by the tail call optimization. For non-PIC code
751 // there is only a 32bit sized unconditional branch which can be assumed
752 // to be able to reach the target. b16 only has a range of +/- 1 KB.
753 // It's entirely possible that the target function is reachable with b16
754 // but we don't have enough information to make that decision.
755 if (InMicroMipsMode && TII->getInstSizeInBytes(*CurrI) == 2 &&
756 (Opcode == Mips::JR || Opcode == Mips::PseudoIndirectBranch ||
757 Opcode == Mips::PseudoIndirectBranch_MM ||
758 Opcode == Mips::PseudoReturn || Opcode == Mips::TAILCALL))
759 continue;
760 // Instructions LWP/SWP and MOVEP should not be in a delay slot as that
761 // results in unpredictable behaviour
762 if (InMicroMipsMode && (Opcode == Mips::LWP_MM || Opcode == Mips::SWP_MM ||
763 Opcode == Mips::MOVEP_MM))
764 continue;
765
766 Filler = CurrI;
767 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": found instruction for delay slot:";
768 CurrI->dump());
769
770 return true;
771 }
772
773 return false;
774}

776bool MipsDelaySlotFiller::searchBackward(MachineBasicBlock &MBB,
777 MachineInstr &Slot) const {
778 if (DisableBackwardSearch)
779 return false;
780
781 auto *Fn = MBB.getParent();
782 RegDefsUses RegDU(*Fn->getSubtarget().getRegisterInfo());
783 MemDefsUses MemDU(Fn->getDataLayout(), &Fn->getFrameInfo());
784 ReverseIter Filler;
785
786 RegDU.init(Slot);
787
788 MachineBasicBlock::iterator SlotI = Slot;
789 if (!searchRange(MBB, ++SlotI.getReverse(), MBB.rend(), RegDU, MemDU, Slot,
790 Filler)) {
791 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": could not find instruction for delay slot using backwards search.\n");
792 return false;
793 }
794
795 MBB.splice(std::next(SlotI), &MBB, Filler.getReverse());
796 MIBundleBuilder(MBB, SlotI, std::next(SlotI, 2));
797 ++UsefulSlots;
798 return true;
799}
800
801bool MipsDelaySlotFiller::searchForward(MachineBasicBlock &MBB,
802 Iter Slot) const {
803 // Can handle only calls.
804 if (DisableForwardSearch || !Slot->isCall())
805 return false;
806
807 RegDefsUses RegDU(*MBB.getParent()->getSubtarget().getRegisterInfo());
808 NoMemInstr NM;
809 Iter Filler;
810
811 RegDU.setCallerSaved(*Slot);
812
813 if (!searchRange(MBB, std::next(Slot), MBB.end(), RegDU, NM, Slot, Filler)) {
814 LLVM_DEBUG(dbgs() << DEBUG_TYPE << ": could not find instruction for delay slot using forwards search.\n");
815 return false;
816 }
817
818 MBB.splice(std::next(Slot), &MBB, Filler);
819 MIBundleBuilder(MBB, Slot, std::next(Slot, 2));
820 ++UsefulSlots;
821 return true;
822}

824bool MipsDelaySlotFiller::searchSuccBBs(MachineBasicBlock &MBB,
825 Iter Slot) const {
826 if (DisableSuccBBSearch)
827 return false;
828
829 MachineBasicBlock *SuccBB = selectSuccBB(MBB);
830
831 if (!SuccBB)
832 return false;
833
834 RegDefsUses RegDU(*MBB.getParent()->getSubtarget().getRegisterInfo());
835 bool HasMultipleSuccs = false;
836 BB2BrMap BrMap;
837 std::unique_ptr<InspectMemInstr> IM;
838 Iter Filler;
839 auto *Fn = MBB.getParent();
840
841 // Iterate over SuccBB's predecessor list.
842 for (MachineBasicBlock::pred_iterator PI = SuccBB->pred_begin(),
843 PE = SuccBB->pred_end(); PI != PE; ++PI)
844 if (!examinePred(**PI, *SuccBB, RegDU, HasMultipleSuccs, BrMap))
845 return false;
846
847 // Do not allow moving instructions which have unallocatable register operands
848 // across basic block boundaries.
849 RegDU.setUnallocatableRegs(*Fn);
850
851 // Only allow moving loads from stack or constants if any of the SuccBB's
852 // predecessors have multiple successors.
853 if (HasMultipleSuccs) {
854 IM.reset(new LoadFromStackOrConst());
855 } else {
856 const MachineFrameInfo &MFI = Fn->getFrameInfo();
857 IM.reset(new MemDefsUses(Fn->getDataLayout(), &MFI));
858 }
859
860 if (!searchRange(MBB, SuccBB->begin(), SuccBB->end(), RegDU, *IM, Slot,
861 Filler))
862 return false;
863
864 insertDelayFiller(Filler, BrMap);
865 addLiveInRegs(Filler, *SuccBB);
866 Filler->eraseFromParent();
867
868 return true;
869}

871MachineBasicBlock *
872MipsDelaySlotFiller::selectSuccBB(MachineBasicBlock &B) const {
873 if (B.succ_empty())
874 return nullptr;
875
876 // Select the successor with the larget edge weight.
877 auto &Prob = getAnalysis<MachineBranchProbabilityInfo>();
878 MachineBasicBlock *S = *std::max_element(
879 B.succ_begin(), B.succ_end(),
880 [&](const MachineBasicBlock *Dst0, const MachineBasicBlock *Dst1) {
881 return Prob.getEdgeProbability(&B, Dst0) <
882 Prob.getEdgeProbability(&B, Dst1);
883 });
884 return S->isEHPad() ? nullptr : S;
885}

887std::pair<MipsInstrInfo::BranchType, MachineInstr *>
888MipsDelaySlotFiller::getBranch(MachineBasicBlock &MBB,
889 const MachineBasicBlock &Dst) const {
890 const MipsInstrInfo *TII =
891 MBB.getParent()->getSubtarget<MipsSubtarget>().getInstrInfo();
892 MachineBasicBlock *TrueBB = nullptr, *FalseBB = nullptr;
893 SmallVector<MachineInstr*, 2> BranchInstrs;
894 SmallVector<MachineOperand, 2> Cond;
895
896 MipsInstrInfo::BranchType R =
897 TII->analyzeBranch(MBB, TrueBB, FalseBB, Cond, false, BranchInstrs);
898
899 if ((R == MipsInstrInfo::BT_None) || (R == MipsInstrInfo::BT_NoBranch))
900 return std::make_pair(R, nullptr);
901
902 if (R != MipsInstrInfo::BT_CondUncond) {
903 if (!hasUnoccupiedSlot(BranchInstrs[0]))
904 return std::make_pair(MipsInstrInfo::BT_None, nullptr);
905
906 assert(((R != MipsInstrInfo::BT_Uncond) || (TrueBB == &Dst)));
907
908 return std::make_pair(R, BranchInstrs[0]);
909 }
910
911 assert((TrueBB == &Dst) || (FalseBB == &Dst));
912
913 // Examine the conditional branch. See if its slot is occupied.
914 if (hasUnoccupiedSlot(BranchInstrs[0]))
915 return std::make_pair(MipsInstrInfo::BT_Cond, BranchInstrs[0]);
916
917 // If that fails, try the unconditional branch.
918 if (hasUnoccupiedSlot(BranchInstrs[1]) && (FalseBB == &Dst))
919 return std::make_pair(MipsInstrInfo::BT_Uncond, BranchInstrs[1]);
920
921 return std::make_pair(MipsInstrInfo::BT_None, nullptr);
922}
923
924bool MipsDelaySlotFiller::examinePred(MachineBasicBlock &Pred,
925 const MachineBasicBlock &Succ,
926 RegDefsUses &RegDU,
927 bool &HasMultipleSuccs,
928 BB2BrMap &BrMap) const {
929 std::pair<MipsInstrInfo::BranchType, MachineInstr *> P =
930 getBranch(Pred, Succ);
931
932 // Return if either getBranch wasn't able to analyze the branches or there
933 // were no branches with unoccupied slots.
934 if (P.first == MipsInstrInfo::BT_None)
935 return false;
936
937 if ((P.first != MipsInstrInfo::BT_Uncond) &&
938 (P.first != MipsInstrInfo::BT_NoBranch)) {
939 HasMultipleSuccs = true;
940 RegDU.addLiveOut(Pred, Succ);
941 }
942
943 BrMap[&Pred] = P.second;
944 return true;
945}
946
947bool MipsDelaySlotFiller::delayHasHazard(const MachineInstr &Candidate,
948 RegDefsUses &RegDU,
949 InspectMemInstr &IM) const {
950 assert(!Candidate.isKill() &&
951 "KILL instructions should have been eliminated at this point.");
952
953 bool HasHazard = Candidate.isImplicitDef();
954
955 HasHazard |= IM.hasHazard(Candidate);
956 HasHazard |= RegDU.update(Candidate, 0, Candidate.getNumOperands());
957
958 return HasHazard;
959}
960
961bool MipsDelaySlotFiller::terminateSearch(const MachineInstr &Candidate) const {
962 return (Candidate.isTerminator() || Candidate.isCall() ||
963 Candidate.isPosition() || Candidate.isInlineAsm() ||
964 Candidate.hasUnmodeledSideEffects());
965}
