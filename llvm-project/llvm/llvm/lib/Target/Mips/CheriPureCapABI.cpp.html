98class StackAllocNeedBoundsChecker {
99public:
100 StackAllocNeedBoundsChecker(AllocaInst *AI, const DataLayout &DL)
101 : AI(AI), DL(DL) {
102
103 AllocaSize = AI->getAllocationSizeInBits(DL);
104 }
105 bool check(const Use &U) {
106 if (!AllocaSize) {
107 DBG_MESSAGE("dyanmic size alloca always needs bounds: ";
108 U.getUser()->dump());
109 return true;
110 }
111 APInt CurrentGEPOffset =
112 APInt(DL.getIndexSizeInBits(AI->getType()->getAddressSpace()), 0);
113 bool Result = useNeedsBounds(U, CurrentGEPOffset, 1);
114 if (Result) {
115 DBG_MESSAGE("Found alloca use that needs bounds: "; U.getUser()->dump());
116 }
117 return Result;
118 }
119
120 void findUsesThatNeedBounds(SmallVectorImpl<const Use *> *UsesThatNeedBounds,
121 bool BoundAllUses, bool *MustUseSingleIntrinsic) {
122 // TODO: don't replace load/store instructions with the bounded value
123 for (const Use &U : AI->uses()) {
124 if (BoundAllUses || check(U)) {
125 UsesThatNeedBounds->push_back(&U);
126#if 0 // This should now be fixed
127 // See comment further down as to why we must reuse a single intrinsic
128 // if one of the bounded users is a phi node.
129 if (isa<PHINode>(U.getUser()))
130 *MustUseSingleIntrinsic = true;
131#endif
132 }
133 }
134 }

136private:
137 bool useNeedsBounds(const Use &U, const APInt &CurrentGEPOffset, unsigned Depth) {
138 const Instruction *I = cast<Instruction>(U.getUser());
139 // Value *V = U->get();
140 if (Depth > 10) {
141 DBG_INDENTED("reached max depth, assuming bounds needed.\n");
142 return true;
143 }
144
145 switch (I->getOpcode()) {
146 case Instruction::Call:
147 case Instruction::Invoke: {
148 auto CI = cast<CallBase>(I);
149 switch (CI->getIntrinsicID()) {
150 case Intrinsic::not_intrinsic:
151 // not an intrinsic call -> always need bounds:
152 DBG_INDENTED("Adding stack bounds since it is passed to call: ";
153 I->dump());
154 return true;
155 case Intrinsic::lifetime_start:
156 case Intrinsic::lifetime_end:
157 DBG_INDENTED("No need for stack bounds for lifetime_{start,end}: ";
158 I->dump());
159 return false;
160 case Intrinsic::dbg_declare:
161 case Intrinsic::dbg_value:
162 case Intrinsic::dbg_label:
163 DBG_INDENTED("No need for stack bounds for dbg_{declare,value,label}: ";
164 I->dump());
165 return false;
166 case Intrinsic::memset:
167 case Intrinsic::memset_element_unordered_atomic:
168 case Intrinsic::memcpy:
169 case Intrinsic::memcpy_element_unordered_atomic:
170 case Intrinsic::memmove:
171 case Intrinsic::memmove_element_unordered_atomic:
172 // TODO: an inline-expanded memset/memcpy/memmove doesn't need bounds!
173 DBG_INDENTED("Adding stack bounds for memset/memcpy/memmove: ";
174 I->dump());
175 return true;
176
177 case Intrinsic::vastart:
178 case Intrinsic::vacopy:
179 case Intrinsic::vaend:
180 // TODO: are bounds always needed?
181 DBG_INDENTED("Adding stack bounds for va_start/va_copy/va_end: ";
182 I->dump());
183 return true;
184
185 case Intrinsic::cheri_bounded_stack_cap:
186 case Intrinsic::cheri_cap_address_get:
187 case Intrinsic::cheri_cap_address_set:
188 case Intrinsic::cheri_cap_base_get:
189 case Intrinsic::cheri_cap_bounds_set:
190 case Intrinsic::cheri_cap_bounds_set_exact:
191 case Intrinsic::cheri_cap_build:
192 case Intrinsic::cheri_cap_conditional_seal:
193 case Intrinsic::cheri_cap_diff:
194 case Intrinsic::cheri_cap_from_ddc:
195 case Intrinsic::cheri_cap_from_pointer:
196 case Intrinsic::cheri_cap_length_get:
197 case Intrinsic::cheri_cap_load_tags:
198 case Intrinsic::cheri_cap_offset_get:
199 case Intrinsic::cheri_cap_offset_increment:
200 case Intrinsic::cheri_cap_offset_set:
201 case Intrinsic::cheri_cap_perms_and:
202 case Intrinsic::cheri_cap_perms_check:
203 case Intrinsic::cheri_cap_perms_get:
204 case Intrinsic::cheri_cap_flags_set:
205 case Intrinsic::cheri_cap_flags_get:
206 case Intrinsic::cheri_cap_seal:
207 case Intrinsic::cheri_cap_sealed_get:
208 case Intrinsic::cheri_cap_subset_test:
209 case Intrinsic::cheri_cap_tag_clear:
210 case Intrinsic::cheri_cap_tag_get:
211 case Intrinsic::cheri_cap_to_pointer:
212 case Intrinsic::cheri_cap_type_check:
213 case Intrinsic::cheri_cap_type_copy:
214 case Intrinsic::cheri_cap_type_get:
215 case Intrinsic::cheri_cap_unseal:
216 // CHERI intrinsics might depend on the bounds
217 DBG_INDENTED("Adding stack bounds for alloca used in CHERI intrinsic:";
218 I->dump());
219 return true;
220
221 default:
222 errs() << DEBUG_TYPE << ": Don't know how to handle intrinsic. Assuming bounds needed"; I->dump();
223 DBG_INDENTED("Adding stack bounds for unknown intrinsic call: ";
224 I->dump());
225 return true;
226 }
227 }
228 case Instruction::Load:
229 // Loading from a stack slot does not need bounds if the offset is known to
230 // be within the alloca. We can just use cl* offset($csp) istead of adding
231 // a cincoffset+csetbouds.
232 return canLoadStoreBeOutOfBounds(I, U, CurrentGEPOffset, Depth);
233 case Instruction::Store:
234 // Storing to a stack slot does not need bounds if the offset is known to
235 // be within the alloca. We can just use cs* offset($csp) istead of adding
236 // a cincoffset+csetbouds.
237 return canLoadStoreBeOutOfBounds(I, U, CurrentGEPOffset, Depth);
238
239 case Instruction::AtomicRMW:
240 case Instruction::AtomicCmpXchg:
241 // cmpxchg and rmw are the same as load/store: if it is a fixed stack
242 // slot we can omit the bounds
243 return canLoadStoreBeOutOfBounds(I, U, CurrentGEPOffset, Depth);
244
245 case Instruction::GetElementPtr: {
246 auto GEPI = cast<GetElementPtrInst>(I);
247 if (!GEPI->isInBounds() && !GEPI->hasAllConstantIndices()) {
248 DBG_INDENTED("Adding stack bounds since GEP is not all constants: ";
249 I->dump());
250 return true;
251 }
252 APInt NewGepOffset(CurrentGEPOffset);
253 if (!GEPI->accumulateConstantOffset(DL, NewGepOffset)) {
254 DBG_INDENTED("Could not accumulate constant GEP Offset -> need bounds: ";
255 I->dump());
256 return true;
257 }
258 return anyUserNeedsBounds(GEPI, NewGepOffset, Depth);
259 }
260 case Instruction::PHI:
261 case Instruction::Select:
262 case Instruction::BitCast:
263 case Instruction::AddrSpaceCast:
264 // If any use of the cast/phi/selects needs bounds then we must add bounds
265 // for the initial value.
266 return anyUserNeedsBounds(I, CurrentGEPOffset, Depth);
267 case Instruction::ICmp:
268 // comparisons need bounds since cexeq takes bounds into account.
269 // TODO: when only comparing addresses we might not need them
270 DBG_INDENTED("Adding stack bounds since comparisons need bounds: ";
271 I->dump());
272 return true;
273 case Instruction::Ret:
274 DBG_INDENTED("Adding stack bounds for alloca that is returned: ";
275 I->dump());
276 return true;
277 default:
278 // Something else - be conservative and say it needs bounds.
279 errs() << "DON'T know how to handle "; I->dump();
280 DBG_INDENTED("Adding stack bounds since don't know how to handle: ";
281 I->dump());
282 return true;
283 }
284 }

286 bool anyUserNeedsBounds(const Instruction *I, const APInt &CurrentGEPOffset, unsigned Depth) {
287 DBG_INDENTED("Checking if " << I->getOpcodeName() << " needs stack bounds: ";
288 I->dump());
289 for (const Use &U : I->uses()) {
290 if (useNeedsBounds(U, CurrentGEPOffset, Depth + 1)) {
291 DBG_INDENTED("Adding stack bounds since " << I->getOpcodeName()
292 << " user needs bounds: ";
293 U.getUser()->dump());
294 return true;
295 }
296 }
297 DBG_INDENTED("no " << I->getOpcodeName() << " users need bounds: ";
298 I->dump());
299 return false;
300 }
301
302 bool canLoadStoreBeOutOfBounds(const Instruction *I, const Use& U,
303 const APInt &CurrentGEPOffset, unsigned Depth) {
304 DBG_INDENTED("Checking if load/store needs bounds (GEP offset is "
305 << CurrentGEPOffset << "): ";
306 I->dump());
307 Depth++; // indent the next debug message more
308 assert(AllocaSize &&
309 "dynamic size alloca should have been checked earlier");
310 Type *LoadStoreType = nullptr;
311 if (auto CmpXchg = dyn_cast<AtomicCmpXchgInst>(I)) {
312 if (U.get() != CmpXchg->getPointerOperand()) {
313 DBG_INDENTED("Stack slot used as value and not pointer -> must set bounds\n";);
314 return true;
315 }
316 LoadStoreType = CmpXchg->getNewValOperand()->getType();
317 } else if (auto RMW = dyn_cast<AtomicRMWInst>(I)) {
318 LoadStoreType = RMW->getValOperand()->getType();
319 if (U.get() != RMW->getPointerOperand()) {
320 DBG_INDENTED("Stack slot used as value and not pointer -> must set bounds\n";);
321 return true;
322 }
323 } else if (const auto *Store = dyn_cast<StoreInst>(I)) {
324 if (U.get() != Store->getPointerOperand()) {
325 DBG_INDENTED("Stack slot used as value and not pointer -> must set bounds\n";);
326 return true;
327 }
328 LoadStoreType = Store->getValueOperand()->getType();
329 } else if (const auto *Load = dyn_cast<LoadInst>(I)) {
330 assert(U.get() == Load->getPointerOperand() && "Invalid load?");
331 LoadStoreType = Load->getType();
332 } else {
333 llvm_unreachable("Invalid load/store type");
334 }
335 auto Size = DL.getTypeStoreSize(LoadStoreType);
336 assert((*AllocaSize % 8) == 0 && "AllocaSize must be a multiple of 8 bits");
337 uint64_t AllocaSizeInBytes = (*AllocaSize / 8);
338 DBG_INDENTED("Load/store size="
339 << Size << ", alloca size=" << AllocaSizeInBytes
340 << ", current GEP offset=" << CurrentGEPOffset << " for ";
341 LoadStoreType->dump(););
342 APInt Zero(CurrentGEPOffset.getBitWidth(), 0);
343 APInt Max(CurrentGEPOffset.getBitWidth(), AllocaSizeInBytes);
344 APInt LastAddr = CurrentGEPOffset + Size;
345 if (CurrentGEPOffset.slt(Zero) || LastAddr.sgt(Max)) {
346 DBG_INDENTED(I->getFunction()->getName()
347 << ": stack load/store offset OUT OF BOUNDS -> adding "
348 "csetbounds: ";
349 I->dump());
350 return true;
351 }
352 DBG_INDENTED("Load/store is in bounds -> can reuse $csp for "; I->dump(););
353 return false;
354 }

361class CheriPureCapABI : public ModulePass, public InstVisitor<CheriPureCapABI> {
362 Module *M;
363 llvm::SmallVector<AllocaInst *, 16> Allocas;
364 bool IsCheri128;
365
366
367public:
368 static char ID;
369 CheriPureCapABI() : ModulePass(ID) {}
370 StringRef getPassName() const override { return "CHERI sandbox ABI setup"; }
371 void visitAllocaInst(AllocaInst &AI) { Allocas.push_back(&AI); }
372 bool runOnModule(Module &Mod) override {
373 M = &Mod;
374 // Early abort if we aren't using capabilities on the stack
375 if (Mod.getDataLayout().getAllocaAddrSpace() != 200)
376 return false;
377 IsCheri128 = Mod.getDataLayout().getPointerSizeInBits(200) == 128;
378 bool Modified = false;
379 for (Function &F : Mod)
380 Modified |= runOnFunction(F);
381
382#if 0
383 auto &DL = Mod.getDataLayout();
384 if (IsCheri128) {
385 for (auto &G : Mod.globals()) {
386 // Adjust the alignment for CHERI128 global variable definitions to
387 // ensure that we can precisely represent bounds.
388 // We can increase alignment for exact definitions and common
389 // definitions:
390 if (G.canIncreaseAlignment() || G.hasCommonLinkage()) {
391 auto Size = DL.getTypeAllocSize(G.getValueType());
392 uint64_t Alignment = G.getAlignment();
393 uint64_t ReqAlignment = cc128_get_required_alignment(Size);
394 if (ReqAlignment > Alignment) {
395 ReqAlignment = std::min(ReqAlignment, UINT64_C(1) << 31);
396 LLVM_DEBUG(dbgs() << "\nIncreased alignment for global from "
397 << Alignment << " to " << ReqAlignment << ": ";
398 G.dump(););
399 G.setAlignment(ReqAlignment);
400 }
401 }
402 }
403 }
404#endif
405 return Modified;
406 }
407
408 bool runOnFunction(Function &F) {
409 // always set bounds with optnone
410 bool IsOptNone = F.hasFnAttribute(Attribute::OptimizeNone);
411 // FIXME: should still ignore lifetime-start + lifetime-end intrinsics even at -O0
412
413
414 // TargetTransformInfo& TTI = getAnalysis<TargetTransformInfoWrapperPass>().getTTI(F);
415
416 LLVMContext &C = M->getContext();
417 Allocas.clear();
418 visit(F);
419 // Give up if this function has no allocas
420 if (Allocas.empty())
421 return false;
422
423 LLVM_DEBUG(dbgs() << "\nChecking function " << F.getName() << "\n");
424
425 // TODO: csetboundsexact and round up sizes
426 // Keep around the non-rematerializable cheri_cap_bounds_set code path to
427 // compare how much rematerialization can help
428 Intrinsic::ID BoundedStackCap = UseRematerializableIntrinsic
429 ? Intrinsic::cheri_bounded_stack_cap
430 : Intrinsic::cheri_cap_bounds_set;
431 const DataLayout &DL = F.getParent()->getDataLayout();
432 Type *SizeTy = Type::getIntNTy(C, DL.getIndexSizeInBits(200));
433 Function *BoundedStackFn =
434 Intrinsic::getDeclaration(M, BoundedStackCap, SizeTy);
435 StackBoundsMethod BoundsMode = BoundsSettingMode;
436
437 IRBuilder<> B(C);
438
439 for (AllocaInst *AI : Allocas) {
440 const uint64_t TotalUses = AI->getNumUses();
441 NumProcessed++;
442 Function *SetBoundsIntrin = BoundedStackFn;
443 // Insert immediately after the alloca
444 B.SetInsertPoint(AI);
445 B.SetInsertPoint(&*++B.GetInsertPoint());
446 unsigned ForcedAlignment = 0;
447
448 PointerType *AllocaTy = AI->getType();
449 assert(isCheriPointer(AllocaTy, &DL));
450 Type *AllocationTy = AllocaTy->getElementType();
451 Value *ArraySize = AI->getArraySize();
452
453 // Create a new (AS 0) alloca
454 // For imprecise capabilities, we need to increase the alignment for
455 // on-stack allocations to ensure that we can create precise bounds.
456 if (IsCheri128) {
457 // TODO: do a sane calculation based on capability precision
458 uint64_t AllocaSize = DL.getTypeAllocSize(AllocationTy);
459 if (ConstantInt *CI = dyn_cast<ConstantInt>(ArraySize))
460 AllocaSize *= CI->getValue().getLimitedValue();
461 else
462 AllocaSize *= 1048576;
463 ForcedAlignment = cc128_get_required_alignment(AllocaSize);
464 // MIPS doesn't support stack alignments greater than 2^16
465 ForcedAlignment = std::min(ForcedAlignment, 0x4000U);
466 }
467 AI->setAlignment(std::max(AI->getAlignment(), ForcedAlignment));
468 // Only set bounds for allocas that escape this function
469 bool NeedBounds = true;
470 // Always set bounds if the function has the optnone attribute
471 SmallVector<const Use *, 32> UsesThatNeedBounds;
472 // If one of the bounded alloca users is a PHI we must reuse the single
473 // intrinsic since PHIs must be the first instruction in the basic block
474 // and we can't insert anything before. Theoretically we could still
475 // use separate intrinsics for the other users but if we are already
476 // saving a bounded stack slot we might as well reuse it.
477 bool MustUseSingleIntrinsic = false;
478 if (BoundsMode == StackBoundsMethod::Never) {
479 NeedBounds = false;
480 } else {
481 StackAllocNeedBoundsChecker BoundsChecker(AI, DL);
482 // With -O0 or =always we set bounds on every stack allocation even
483 // if it is not necessary
484 bool BoundAll = IsOptNone || BoundsMode == StackBoundsMethod::AllUses;
485 BoundsChecker.findUsesThatNeedBounds(&UsesThatNeedBounds, BoundAll,
486 &MustUseSingleIntrinsic);
487 NeedBounds = !UsesThatNeedBounds.empty();
488 NumUsesProcessed += TotalUses;
489 DBG_MESSAGE(F.getName()
490 << ": " << UsesThatNeedBounds.size() << " of "
491 << TotalUses << " users need bounds for ";
492 AI->dump());
493 // TODO: remove the all-or-nothing case
494 if (NeedBounds &&
495 BoundsMode == StackBoundsMethod::ForAllUsesIfOneNeedsBounds) {
496 // We are compiling with the all-or-nothing case and found at least
497 // one use that needs bounds -> set bounds on all uses
498 UsesThatNeedBounds.clear();
499 LLVM_DEBUG(dbgs() << "Checking if alloca needs bounds: "; AI->dump());
500
501 BoundsChecker.findUsesThatNeedBounds(&UsesThatNeedBounds,
502 /*BoundAllUses=*/true,
503 &MustUseSingleIntrinsic);
504 }
505 }
506 if (!NeedBounds) {
507 NumCompletelyUnboundedAllocas++;
508 DBG_MESSAGE("No need to set bounds on stack alloca"; AI->dump());
509 continue;
510 }
511
512 if (!AI->isStaticAlloca()) {
513 NumDynamicAllocas++;
514 // TODO: skip bounds on dynamic allocas (maybe add a TLI hook to check
515 // whether the backend already adds bounds to the dynamic_stackalloc)
516 DBG_MESSAGE("Found dynamic alloca: must use single intrinisic and "
517 "bounds.set intrinisic");
518 MustUseSingleIntrinsic = true;
519 SetBoundsIntrin =
520 Intrinsic::getDeclaration(M, Intrinsic::cheri_cap_bounds_set,
521 SizeTy);
522 }
523
524 // Reuse the result of a single csetbounds intrinisic if we are at -O0 or
525 // there are more than N users of this bounded stack capability.
526 const bool ReuseSingleIntrinsicCall =
527 MustUseSingleIntrinsic || IsOptNone ||
528 UsesThatNeedBounds.size() >= SingleIntrinsicThreshold;
529
530 NumUsesWithBounds += UsesThatNeedBounds.size();
531 NumUsesWithoutBounds += TotalUses - UsesThatNeedBounds.size();
532 // Get the size of the alloca
533 unsigned ElementSize = DL.getTypeAllocSize(AllocationTy);
534 Value *Size = ConstantInt::get(Type::getInt64Ty(C), ElementSize);
535 if (AI->isArrayAllocation())
536 Size = B.CreateMul(Size, AI->getArraySize());
537
538 if (cheri::ShouldCollectCSetBoundsStats) {
539 cheri::addSetBoundsStats(AI->getAlignment(), Size, getPassName(),
540 cheri::SetBoundsPointerSource::Stack,
541 "set bounds on " +
542 cheri::inferLocalVariableName(AI),
543 cheri::inferSourceLocation(AI));
544 }
545 LLVM_DEBUG(auto S = cheri::inferConstantValue(Size);
546 dbgs() << AI->getFunction()->getName()
547 << ": setting bounds on stack alloca to "
548 << (S ? Twine(*S) : Twine("<unknown>"));
549 AI->dump());
550
551 Value *SingleBoundedAlloc = nullptr;
552 if (ReuseSingleIntrinsicCall) {
553 NumSingleIntrin++;
554 // We need to convert it to an i8* for the intrinisic:
555 Instruction *AllocaI8 =
556 cast<Instruction>(B.CreateBitCast(AI, Type::getInt8PtrTy(C, 200)));
557 SingleBoundedAlloc = B.CreateCall(SetBoundsIntrin, {AllocaI8, Size});
558 SingleBoundedAlloc = B.CreateBitCast(SingleBoundedAlloc, AllocaTy);
559 }
560 for (const Use *U : UsesThatNeedBounds) {
561 Instruction *I = cast<Instruction>(U->getUser());
562 if (ReuseSingleIntrinsicCall) {
563 const_cast<Use *>(U)->set(SingleBoundedAlloc);
564 } else {
565 if (auto PHI = dyn_cast<PHINode>(I)) {
566 // For PHI nodes we can't insert just before the PHI, instead we
567 // must insert it just before
568 auto BB = PHI->getIncomingBlock(*U);
569 // LLVM_DEBUG(dbgs() << "PHI use coming from"; BB->dump());
570 B.SetInsertPoint(BB->getTerminator());
571 } else {
572 // insert a new intrinsic call before every use. This can avoid
573 // stack spills but will result in additional instructions.
574 // This should avoid spilling registers when using an alloca in a
575 // different basic block.
576 // TODO: there should be a MIR pass to merge unncessary calls
577 B.SetInsertPoint(I);
578 }
579 // We need to convert it to an i8* for the intrinisic. Note: we must
580 // not reuse a bitcast since otherwise that results in spilling the
581 // register that was incremented and doing a setbounds in a different
582 // basic block. This is stupid and we should be either using the
583 // bounded capability everywhere or be doing inc+setoffset in the
584 // other block.
585 Instruction *AllocaI8 =
586 cast<Instruction>(B.CreateBitCast(AI, Type::getInt8PtrTy(C, 200)));
587 auto WithBounds = B.CreateCall(SetBoundsIntrin, {AllocaI8, Size});
588 const_cast<Use *>(U)->set(B.CreateBitCast(WithBounds, AllocaTy));
589 }
590 }
591 }
592 return true;
593 }
