82void MipsInstrInfo::AnalyzeCondBr(const MachineInstr *Inst, unsigned Opc,
83 MachineBasicBlock *&BB,
84 SmallVectorImpl<MachineOperand> &Cond) const {
85 assert(getAnalyzableBrOpc(Opc) && "Not an analyzable branch");
86 int NumOp = Inst->getNumExplicitOperands();
87
88 // for both int and fp branches, the last explicit operand is the
89 // MBB.
90 BB = Inst->getOperand(NumOp-1).getMBB();
91 Cond.push_back(MachineOperand::CreateImm(Opc));
92
93 for (int i = 0; i < NumOp-1; i++)
94 Cond.push_back(Inst->getOperand(i));
95}

97bool MipsInstrInfo::analyzeBranch(MachineBasicBlock &MBB,
98 MachineBasicBlock *&TBB,
99 MachineBasicBlock *&FBB,
100 SmallVectorImpl<MachineOperand> &Cond,
101 bool AllowModify) const {
102 SmallVector<MachineInstr*, 2> BranchInstrs;
103 BranchType BT = analyzeBranch(MBB, TBB, FBB, Cond, AllowModify, BranchInstrs);
104
105 return (BT == BT_None) || (BT == BT_Indirect);
106}
107
108void MipsInstrInfo::BuildCondBr(MachineBasicBlock &MBB, MachineBasicBlock *TBB,
109 const DebugLoc &DL,
110 ArrayRef<MachineOperand> Cond) const {
111 unsigned Opc = Cond[0].getImm();
112 const MCInstrDesc &MCID = get(Opc);
113 MachineInstrBuilder MIB = BuildMI(&MBB, DL, MCID);
114
115 for (unsigned i = 1; i < Cond.size(); ++i) {
116 assert((Cond[i].isImm() || Cond[i].isReg()) &&
117 "Cannot copy operand for conditional branch!");
118 MIB.add(Cond[i]);
119 }
120 MIB.addMBB(TBB);
121}

123unsigned MipsInstrInfo::insertBranch(MachineBasicBlock &MBB,
124 MachineBasicBlock *TBB,
125 MachineBasicBlock *FBB,
126 ArrayRef<MachineOperand> Cond,
127 const DebugLoc &DL,
128 int *BytesAdded) const {
129 // Shouldn't be a fall through.
130 assert(TBB && "insertBranch must not be told to insert a fallthrough");
131 assert(!BytesAdded && "code size not handled");
132
133 // # of condition operands:
134 // Unconditional branches: 0
135 // Floating point branches: 1 (opc)
136 // Int BranchZero: 2 (opc, reg)
137 // Int Branch: 3 (opc, reg0, reg1)
138 assert((Cond.size() <= 3) &&
139 "# of Mips branch conditions must be <= 3!");
140
141 // Two-way Conditional branch.
142 if (FBB) {
143 BuildCondBr(MBB, TBB, DL, Cond);
144 BuildMI(&MBB, DL, get(UncondBrOpc)).addMBB(FBB);
145 return 2;
146 }
147
148 // One way branch.
149 // Unconditional branch.
150 if (Cond.empty())
151 BuildMI(&MBB, DL, get(UncondBrOpc)).addMBB(TBB);
152 else // Conditional branch.
153 BuildCondBr(MBB, TBB, DL, Cond);
154 return 1;
155}

157unsigned MipsInstrInfo::removeBranch(MachineBasicBlock &MBB,
158 int *BytesRemoved) const {
159 assert(!BytesRemoved && "code size not handled");
160
161 MachineBasicBlock::reverse_iterator I = MBB.rbegin(), REnd = MBB.rend();
162 unsigned removed = 0;
163
164 // Up to 2 branches are removed.
165 // Note that indirect branches are not removed.
166 while (I != REnd && removed < 2) {
167 // Skip past debug instructions.
168 if (I->isDebugInstr()) {
169 ++I;
170 continue;
171 }
172 if (!getAnalyzableBrOpc(I->getOpcode()))
173 break;
174 // Remove the branch.
175 I->eraseFromParent();
176 I = MBB.rbegin();
177 ++removed;
178 }
179
180 return removed;
181}

184/// specified Branch instruction.
185bool MipsInstrInfo::reverseBranchCondition(
186 SmallVectorImpl<MachineOperand> &Cond) const {
187 assert( (Cond.size() && Cond.size() <= 3) &&
188 "Invalid Mips branch condition!");

193MipsInstrInfo::BranchType MipsInstrInfo::analyzeBranch(
194 MachineBasicBlock &MBB, MachineBasicBlock *&TBB, MachineBasicBlock *&FBB,
195 SmallVectorImpl<MachineOperand> &Cond, bool AllowModify,
196 SmallVectorImpl<MachineInstr *> &BranchInstrs) const {
197 MachineBasicBlock::reverse_iterator I = MBB.rbegin(), REnd = MBB.rend();
198
199 // Skip all the debug instructions.
200 while (I != REnd && I->isDebugInstr())
201 ++I;
202
203 if (I == REnd || !isUnpredicatedTerminator(*I)) {
204 // This block ends with no branches (it just falls through to its succ).
205 // Leave TBB/FBB null.
206 TBB = FBB = nullptr;
207 return BT_NoBranch;
208 }
209
210 MachineInstr *LastInst = &*I;
211 unsigned LastOpc = LastInst->getOpcode();
212 BranchInstrs.push_back(LastInst);
213
214 // Not an analyzable branch (e.g., indirect jump).
215 if (!getAnalyzableBrOpc(LastOpc))
216 return LastInst->isIndirectBranch() ? BT_Indirect : BT_None;
217
218 // Get the second to last instruction in the block.
219 unsigned SecondLastOpc = 0;
220 MachineInstr *SecondLastInst = nullptr;
221
222 // Skip past any debug instruction to see if the second last actual
223 // is a branch.
224 ++I;
225 while (I != REnd && I->isDebugInstr())
226 ++I;
227
228 if (I != REnd) {
229 SecondLastInst = &*I;
230 SecondLastOpc = getAnalyzableBrOpc(SecondLastInst->getOpcode());
231
232 // Not an analyzable branch (must be an indirect jump).
233 if (isUnpredicatedTerminator(*SecondLastInst) && !SecondLastOpc)
234 return BT_None;
235 }
236
237 // If there is only one terminator instruction, process it.
238 if (!SecondLastOpc) {
239 // Unconditional branch.
240 if (LastInst->isUnconditionalBranch()) {
241 TBB = LastInst->getOperand(0).getMBB();
242 return BT_Uncond;
243 }
244
245 // Conditional branch
246 AnalyzeCondBr(LastInst, LastOpc, TBB, Cond);
247 return BT_Cond;
248 }
249
250 // If we reached here, there are two branches.
251 // If there are three terminators, we don't know what sort of block this is.
252 if (++I != REnd && isUnpredicatedTerminator(*I))
253 return BT_None;
254
255 BranchInstrs.insert(BranchInstrs.begin(), SecondLastInst);
256
257 // If second to last instruction is an unconditional branch,
258 // analyze it and remove the last instruction.
259 if (SecondLastInst->isUnconditionalBranch()) {
260 // Return if the last instruction cannot be removed.
261 if (!AllowModify)
262 return BT_None;
263
264 TBB = SecondLastInst->getOperand(0).getMBB();
265 LastInst->eraseFromParent();
266 BranchInstrs.pop_back();
267 return BT_Uncond;
268 }
269
270 // Conditional branch followed by an unconditional branch.
271 // The last one must be unconditional.
272 if (!LastInst->isUnconditionalBranch())
273 return BT_None;
274
275 AnalyzeCondBr(SecondLastInst, SecondLastOpc, TBB, Cond);
276 FBB = LastInst->getOperand(0).getMBB();
277
278 return BT_CondUncond;
279}

448/// Return the corresponding compact (no delay slot) form of a branch.
449unsigned MipsInstrInfo::getEquivalentCompactForm(
450 const MachineBasicBlock::iterator I) const {
451 unsigned Opcode = I->getOpcode();
452 bool canUseShortMicroMipsCTI = false;
453
454 if (Subtarget.inMicroMipsMode()) {
455 switch (Opcode) {
456 case Mips::BNE:
457 case Mips::BNE_MM:
458 case Mips::BEQ:
459 case Mips::BEQ_MM:
460 // microMIPS has NE,EQ branches that do not have delay slots provided one
461 // of the operands is zero.
462 if (I->getOperand(1).getReg() == Subtarget.getABI().GetZeroReg())
463 canUseShortMicroMipsCTI = true;
464 break;
465 // For microMIPS the PseudoReturn and PseudoIndirectBranch are always
466 // expanded to JR_MM, so they can be replaced with JRC16_MM.
467 case Mips::JR:
468 case Mips::PseudoReturn:
469 case Mips::PseudoIndirectBranch:
470 canUseShortMicroMipsCTI = true;
471 break;
472 }
473 }
474
475 // MIPSR6 forbids both operands being the zero register.
476 if (Subtarget.hasMips32r6() && (I->getNumOperands() > 1) &&
477 (I->getOperand(0).isReg() &&
478 (I->getOperand(0).getReg() == Mips::ZERO ||
479 I->getOperand(0).getReg() == Mips::ZERO_64)) &&
480 (I->getOperand(1).isReg() &&
481 (I->getOperand(1).getReg() == Mips::ZERO ||
482 I->getOperand(1).getReg() == Mips::ZERO_64)))
483 return 0;
484
485 if (Subtarget.hasMips32r6() || canUseShortMicroMipsCTI) {
486 switch (Opcode) {
487 case Mips::B:
488 return Mips::BC;
489 case Mips::BAL:
490 return Mips::BALC;
491 case Mips::BEQ:
492 case Mips::BEQ_MM:
493 if (canUseShortMicroMipsCTI)
494 return Mips::BEQZC_MM;
495 else if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
496 return 0;
497 return Mips::BEQC;
498 case Mips::BNE:
499 case Mips::BNE_MM:
500 if (canUseShortMicroMipsCTI)
501 return Mips::BNEZC_MM;
502 else if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
503 return 0;
504 return Mips::BNEC;
505 case Mips::BGE:
506 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
507 return 0;
508 return Mips::BGEC;
509 case Mips::BGEU:
510 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
511 return 0;
512 return Mips::BGEUC;
513 case Mips::BGEZ:
514 return Mips::BGEZC;
515 case Mips::BGTZ:
516 return Mips::BGTZC;
517 case Mips::BLEZ:
518 return Mips::BLEZC;
519 case Mips::BLT:
520 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
521 return 0;
522 return Mips::BLTC;
523 case Mips::BLTU:
524 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
525 return 0;
526 return Mips::BLTUC;
527 case Mips::BLTZ:
528 return Mips::BLTZC;
529 case Mips::BEQ64:
530 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
531 return 0;
532 return Mips::BEQC64;
533 case Mips::BNE64:
534 if (I->getOperand(0).getReg() == I->getOperand(1).getReg())
535 return 0;
536 return Mips::BNEC64;
537 case Mips::BGTZ64:
538 return Mips::BGTZC64;
539 case Mips::BGEZ64:
540 return Mips::BGEZC64;
541 case Mips::BLTZ64:
542 return Mips::BLTZC64;
543 case Mips::BLEZ64:
544 return Mips::BLEZC64;
545 // For MIPSR6, the instruction 'jic' can be used for these cases. Some
546 // tools will accept 'jrc reg' as an alias for 'jic 0, $reg'.
547 case Mips::JR:
548 case Mips::PseudoIndirectBranchR6:
549 case Mips::PseudoReturn:
550 case Mips::TAILCALLR6REG:
551 if (canUseShortMicroMipsCTI)
552 return Mips::JRC16_MM;
553 return Mips::JIC;
554 case Mips::JALRPseudo:
555 return Mips::JIALC;
556 case Mips::JR64:
557 case Mips::PseudoIndirectBranch64R6:
558 case Mips::PseudoReturn64:
559 case Mips::TAILCALL64R6REG:
560 return Mips::JIC64;
561 case Mips::JALR64Pseudo:
562 return Mips::JIALC64;
563 default:
564 return 0;
565 }
566 }
567
568 return 0;
569}

586/// Return the number of bytes of code the specified instruction may be.
587unsigned MipsInstrInfo::getInstSizeInBytes(const MachineInstr &MI) const {
588 switch (MI.getOpcode()) {
589 default:
590 return MI.getDesc().getSize();
591 case TargetOpcode::INLINEASM:
592 case TargetOpcode::INLINEASM_BR: { // Inline Asm: Variable size.
593 const MachineFunction *MF = MI.getParent()->getParent();
594 const char *AsmStr = MI.getOperand(0).getSymbolName();
595 return getInlineAsmLength(AsmStr, *MF->getTarget().getMCAsmInfo());
596 }
597 case Mips::CONSTPOOL_ENTRY:
598 // If this machine instr is a constant pool entry, its size is recorded as
599 // operand #2.
600 return MI.getOperand(2).getImm();
601 }
602}

604MachineInstrBuilder
605MipsInstrInfo::genInstrWithNewOpc(unsigned NewOpc,
606 MachineBasicBlock::iterator I) const {
607 MachineInstrBuilder MIB;
608
609 // Certain branches have two forms: e.g beq $1, $zero, dest vs beqz $1, dest
610 // Pick the zero form of the branch for readable assembly and for greater
611 // branch distance in non-microMIPS mode.
612 // Additional MIPSR6 does not permit the use of register $zero for compact
613 // branches.
614 // FIXME: Certain atomic sequences on mips64 generate 32bit references to
615 // Mips::ZERO, which is incorrect. This test should be updated to use
616 // Subtarget.getABI().GetZeroReg() when those atomic sequences and others
617 // are fixed.
618 int ZeroOperandPosition = -1;
619 bool BranchWithZeroOperand = false;
620 if (I->isBranch() && !I->isPseudo()) {
621 auto TRI = I->getParent()->getParent()->getSubtarget().getRegisterInfo();
622 ZeroOperandPosition = I->findRegisterUseOperandIdx(Mips::ZERO, false, TRI);
623 BranchWithZeroOperand = ZeroOperandPosition != -1;
624 }
625
626 if (BranchWithZeroOperand) {
627 switch (NewOpc) {
628 case Mips::BEQC:
629 NewOpc = Mips::BEQZC;
630 break;
631 case Mips::BNEC:
632 NewOpc = Mips::BNEZC;
633 break;
634 case Mips::BGEC:
635 NewOpc = Mips::BGEZC;
636 break;
637 case Mips::BLTC:
638 NewOpc = Mips::BLTZC;
639 break;
640 case Mips::BEQC64:
641 NewOpc = Mips::BEQZC64;
642 break;
643 case Mips::BNEC64:
644 NewOpc = Mips::BNEZC64;
645 break;
646 }
647 }
648
649 MIB = BuildMI(*I->getParent(), I, I->getDebugLoc(), get(NewOpc));
650
651 // For MIPSR6 JI*C requires an immediate 0 as an operand, JIALC(64) an
652 // immediate 0 as an operand and requires the removal of it's implicit-def %ra
653 // implicit operand as copying the implicit operations of the instructio we're
654 // looking at will give us the correct flags.
655 if (NewOpc == Mips::JIC || NewOpc == Mips::JIALC || NewOpc == Mips::JIC64 ||
656 NewOpc == Mips::JIALC64) {
657
658 if (NewOpc == Mips::JIALC || NewOpc == Mips::JIALC64)
659 MIB->RemoveOperand(0);
660
661 for (unsigned J = 0, E = I->getDesc().getNumOperands(); J < E; ++J) {
662 MIB.add(I->getOperand(J));
663 }
664
665 MIB.addImm(0);
666
667 // If I has an MCSymbol operand (used by asm printer, to emit R_MIPS_JALR),
668 // add it to the new instruction.
669 for (unsigned J = I->getDesc().getNumOperands(), E = I->getNumOperands();
670 J < E; ++J) {
671 const MachineOperand &MO = I->getOperand(J);
672 if (MO.isMCSymbol() && (MO.getTargetFlags() & MipsII::MO_JALR))
673 MIB.addSym(MO.getMCSymbol(), MipsII::MO_JALR);
674 }
675
676
677 } else {
678 for (unsigned J = 0, E = I->getDesc().getNumOperands(); J < E; ++J) {
679 if (BranchWithZeroOperand && (unsigned)ZeroOperandPosition == J)
680 continue;
681
682 MIB.add(I->getOperand(J));
683 }
684 }
685
686 MIB.copyImplicitOps(*I);
687 MIB.cloneMemRefs(*I);
688 return MIB;
689}

691bool MipsInstrInfo::findCommutedOpIndices(MachineInstr &MI, unsigned &SrcOpIdx1,
692 unsigned &SrcOpIdx2) const {
693 assert(!MI.isBundle() &&
694 "TargetInstrInfo::findCommutedOpIndices() can't handle bundles");
695
696 const MCInstrDesc &MCID = MI.getDesc();
697 if (!MCID.isCommutable())
698 return false;
699
700 switch (MI.getOpcode()) {
701 case Mips::DPADD_U_H:
702 case Mips::DPADD_U_W:
703 case Mips::DPADD_U_D:
704 case Mips::DPADD_S_H:
705 case Mips::DPADD_S_W:
706 case Mips::DPADD_S_D:
707 // The first operand is both input and output, so it should not commute
708 if (!fixCommutedOpIndices(SrcOpIdx1, SrcOpIdx2, 2, 3))
709 return false;
710
711 if (!MI.getOperand(SrcOpIdx1).isReg() || !MI.getOperand(SrcOpIdx2).isReg())
712 return false;
713 return true;
714 }
715 return TargetInstrInfo::findCommutedOpIndices(MI, SrcOpIdx1, SrcOpIdx2);
716}
717
718// ins, ext, dext*, dins have the following constraints:
719// X <= pos < Y
720// X < size <= Y
721// X < pos+size <= Y
722//
723// dinsm and dinsu have the following constraints:
724// X <= pos < Y
725// X <= size <= Y
726// X < pos+size <= Y
727//
728// The callee of verifyInsExtInstruction however gives the bounds of
729// dins[um] like the other (d)ins (d)ext(um) instructions, so that this
730// function doesn't have to vary it's behaviour based on the instruction
731// being checked.
732static bool verifyInsExtInstruction(const MachineInstr &MI, StringRef &ErrInfo,
733 const int64_t PosLow, const int64_t PosHigh,
734 const int64_t SizeLow,
735 const int64_t SizeHigh,
736 const int64_t BothLow,
737 const int64_t BothHigh) {
738 MachineOperand MOPos = MI.getOperand(2);
739 if (!MOPos.isImm()) {
740 ErrInfo = "Position is not an immediate!";
741 return false;
742 }
743 int64_t Pos = MOPos.getImm();
744 if (!((PosLow <= Pos) && (Pos < PosHigh))) {
745 ErrInfo = "Position operand is out of range!";
746 return false;
747 }
748
749 MachineOperand MOSize = MI.getOperand(3);
750 if (!MOSize.isImm()) {
751 ErrInfo = "Size operand is not an immediate!";
752 return false;
753 }
754 int64_t Size = MOSize.getImm();
755 if (!((SizeLow < Size) && (Size <= SizeHigh))) {
756 ErrInfo = "Size operand is out of range!";
757 return false;
758 }
759
760 if (!((BothLow < (Pos + Size)) && ((Pos + Size) <= BothHigh))) {
761 ErrInfo = "Position + Size is out of range!";
762 return false;
763 }
764
765 return true;
766}
767
768// Perform target specific instruction verification.
769template<unsigned Width, unsigned Scale>
770bool checkScaledImmediate(const MachineInstr &MI, StringRef& ErrInfo, unsigned OpndIdx) {
771 assert(MI.getDesc().OpInfo[OpndIdx].OperandType == MCOI::OPERAND_IMMEDIATE);
772 if (MI.getOperand(OpndIdx).isImm() && !isShiftedInt<Width, Scale>(MI.getOperand(OpndIdx).getImm())) {
773 ErrInfo = "Operand immediate is not representable!";
774 return false;
775 }
776 return true;
777}

780bool MipsInstrInfo::verifyInstruction(const MachineInstr &MI,
781 StringRef &ErrInfo) const {
782 // Verify that ins and ext instructions are well formed.
783 switch (MI.getOpcode()) {
784 case Mips::EXT:
785 case Mips::EXT_MM:
786 case Mips::INS:
787 case Mips::INS_MM:
788 case Mips::DINS:
789 return verifyInsExtInstruction(MI, ErrInfo, 0, 32, 0, 32, 0, 32);
790 case Mips::DINSM:
791 // The ISA spec has a subtle difference between dinsm and dextm
792 // in that it says:
793 // 2 <= size <= 64 for 'dinsm' but 'dextm' has 32 < size <= 64.
794 // To make the bounds checks similar, the range 1 < size <= 64 is checked
795 // for 'dinsm'.
796 return verifyInsExtInstruction(MI, ErrInfo, 0, 32, 1, 64, 32, 64);
797 case Mips::DINSU:
798 // The ISA spec has a subtle difference between dinsu and dextu in that
799 // the size range of dinsu is specified as 1 <= size <= 32 whereas size
800 // for dextu is 0 < size <= 32. The range checked for dinsu here is
801 // 0 < size <= 32, which is equivalent and similar to dextu.
802 return verifyInsExtInstruction(MI, ErrInfo, 32, 64, 0, 32, 32, 64);
803 case Mips::DEXT:
804 return verifyInsExtInstruction(MI, ErrInfo, 0, 32, 0, 32, 0, 63);
805 case Mips::DEXTM:
806 return verifyInsExtInstruction(MI, ErrInfo, 0, 32, 32, 64, 32, 64);
807 case Mips::DEXTU:
808 return verifyInsExtInstruction(MI, ErrInfo, 32, 64, 0, 32, 32, 64);
809 case Mips::TAILCALLREG:
810 case Mips::PseudoIndirectBranch:
811 case Mips::JR:
812 case Mips::JR64:
813 case Mips::JALR:
814 case Mips::JALR64:
815 case Mips::JALRPseudo:
816 if (!Subtarget.useIndirectJumpsHazard())
817 return true;
818
819 ErrInfo = "invalid instruction when using jump guards!";
820 return false;
821
822 // Check that we don't use the cjalr output register (usually $c17) in the
823 // delay slot since it will have changed
824 case Mips::CapJumpLinkPseudo:
825 case Mips::CJALR:
826 // errs() << "CAPJUMPLINK: (delay slot: " << MI.hasDelaySlot()
827 // << ", bundle size: " << MI.getBundleSize() << ") "; MI.dump();
828 if (MI.isBundledWithSucc()) {
829 auto &OutputOp =
830 MI.getOpcode() == Mips::CJALR ? MI.getOperand(0) : MI.getOperand(2);
831 if (MI.getOpcode() ==
832 Mips::CapJumpLinkPseudo) // Op2 here is implicitly c17:
833 assert(OutputOp.isReg() && OutputOp.getReg() == Mips::C17);
834 auto DelaySlotInstr = MI.getNextNode();
835 if (DelaySlotInstr->readsRegister(Mips::C17)) {
836 ErrInfo = "Filled CapJumpLinkPseudo delay slot with a read of $c17 "
837 "(which will have been clobbered!)";
838 return false;
839 }
840 }
841 return true;
842 // FIXME: duplicating all this here is silly, tablegen should
843 // be able to generate those checks!
844 case Mips::CAPLOADU8:
845 case Mips::CAPLOADU832:
846 case Mips::CAPLOAD8:
847 case Mips::CAPLOAD832:
848 case Mips::CAPSTORE8:
849 case Mips::CAPSTORE832:
850 return checkScaledImmediate<8, 0>(MI, ErrInfo, 2);
851 case Mips::CAPLOADU16:
852 case Mips::CAPLOADU1632:
853 case Mips::CAPLOAD16:
854 case Mips::CAPLOAD1632:
855 case Mips::CAPSTORE16:
856 case Mips::CAPSTORE1632:
857 return checkScaledImmediate<8, 1>(MI, ErrInfo, 2);
858 case Mips::CAPLOADU32:
859 case Mips::CAPLOAD3264:
860 case Mips::CAPSTORE32:
861 case Mips::CAPSTORE3264:
862 return checkScaledImmediate<8, 2>(MI, ErrInfo, 2);
863 case Mips::CAPLOAD64:
864 case Mips::CAPSTORE64:
865 return checkScaledImmediate<8, 3>(MI, ErrInfo, 2);
866 case Mips::STORECAP:
867 case Mips::LOADCAP:
868 return checkScaledImmediate<11, 4>(MI, ErrInfo, 2);
869 case Mips::LOADCAP_BigImm:
870 return checkScaledImmediate<16, 4>(MI, ErrInfo, 1);
871 case Mips::CIncOffsetImm:
872 return checkScaledImmediate<11, 0>(MI, ErrInfo, 2);
873 case Mips::CSetBoundsImm:
874 // FIXME: actually 11 bit unsigned
875 return checkScaledImmediate<12, 0>(MI, ErrInfo, 2);
876 default:
877 return true;
878 }
879
880 return true;
881}
