8//
9// This file contains a pass that expands pseudo instructions into target
10// instructions to allow proper scheduling, if-conversion, and other late
11// optimizations. This pass should be run after register allocation but before
12// the post-regalloc scheduling pass.
13//
14// This is currently only used for expanding atomic pseudos after register
15// allocation. We do this to avoid the fast register allocator introducing
16// spills between ll and sc. These stores cause some MIPS implementations to
17// abort the atomic RMW sequence.
18//
19//===----------------------------------------------------------------------===//
20

80bool MipsExpandPseudo::expandAtomicCmpSwapSubword(
81 MachineBasicBlock &BB, MachineBasicBlock::iterator I,
82 MachineBasicBlock::iterator &NMBBI) {
83
84 MachineFunction *MF = BB.getParent();
85
86 const bool ArePtrs64bit = STI->getABI().ArePtrs64bit();
87 DebugLoc DL = I->getDebugLoc();
88 unsigned LL, SC;
89
90 unsigned ZERO = Mips::ZERO;
91 unsigned BNE = Mips::BNE;
92 unsigned BEQ = Mips::BEQ;
93 unsigned SEOp =
94 I->getOpcode() == Mips::ATOMIC_CMP_SWAP_I8_POSTRA ? Mips::SEB : Mips::SEH;
95
96 if (STI->inMicroMipsMode()) {
97 LL = STI->hasMips32r6() ? Mips::LL_MMR6 : Mips::LL_MM;
98 SC = STI->hasMips32r6() ? Mips::SC_MMR6 : Mips::SC_MM;
99 BNE = STI->hasMips32r6() ? Mips::BNEC_MMR6 : Mips::BNE_MM;
100 BEQ = STI->hasMips32r6() ? Mips::BEQC_MMR6 : Mips::BEQ_MM;
101 } else {
102 LL = STI->hasMips32r6() ? (ArePtrs64bit ? Mips::LL64_R6 : Mips::LL_R6)
103 : (ArePtrs64bit ? Mips::LL64 : Mips::LL);
104 SC = STI->hasMips32r6() ? (ArePtrs64bit ? Mips::SC64_R6 : Mips::SC_R6)
105 : (ArePtrs64bit ? Mips::SC64 : Mips::SC);
106 }
107
108 unsigned Dest = I->getOperand(0).getReg();
109 unsigned Ptr = I->getOperand(1).getReg();
110 unsigned Mask = I->getOperand(2).getReg();
111 unsigned ShiftCmpVal = I->getOperand(3).getReg();
112 unsigned Mask2 = I->getOperand(4).getReg();
113 unsigned ShiftNewVal = I->getOperand(5).getReg();
114 unsigned ShiftAmnt = I->getOperand(6).getReg();
115 unsigned Scratch = I->getOperand(7).getReg();
116 unsigned Scratch2 = I->getOperand(8).getReg();
117
118 // insert new blocks after the current block
119 const BasicBlock *LLVM_BB = BB.getBasicBlock();
120 MachineBasicBlock *loop1MBB = MF->CreateMachineBasicBlock(LLVM_BB);
121 MachineBasicBlock *loop2MBB = MF->CreateMachineBasicBlock(LLVM_BB);
122 MachineBasicBlock *sinkMBB = MF->CreateMachineBasicBlock(LLVM_BB);
123 MachineBasicBlock *exitMBB = MF->CreateMachineBasicBlock(LLVM_BB);
124 MachineFunction::iterator It = ++BB.getIterator();
125 MF->insert(It, loop1MBB);
126 MF->insert(It, loop2MBB);
127 MF->insert(It, sinkMBB);
128 MF->insert(It, exitMBB);
129
130 // Transfer the remainder of BB and its successor edges to exitMBB.
131 exitMBB->splice(exitMBB->begin(), &BB,
132 std::next(MachineBasicBlock::iterator(I)), BB.end());
133 exitMBB->transferSuccessorsAndUpdatePHIs(&BB);
134
135 // thisMBB:
136 // ...
137 // fallthrough --> loop1MBB
138 BB.addSuccessor(loop1MBB, BranchProbability::getOne());
139 loop1MBB->addSuccessor(sinkMBB);
140 loop1MBB->addSuccessor(loop2MBB);
141 loop1MBB->normalizeSuccProbs();
142 loop2MBB->addSuccessor(loop1MBB);
143 loop2MBB->addSuccessor(sinkMBB);
144 loop2MBB->normalizeSuccProbs();
145 sinkMBB->addSuccessor(exitMBB, BranchProbability::getOne());
146
147 // loop1MBB:
148 // ll dest, 0(ptr)
149 // and Mask', dest, Mask
150 // bne Mask', ShiftCmpVal, exitMBB
151 BuildMI(loop1MBB, DL, TII->get(LL), Scratch).addReg(Ptr).addImm(0);
152 BuildMI(loop1MBB, DL, TII->get(Mips::AND), Scratch2)
153 .addReg(Scratch)
154 .addReg(Mask);
155 BuildMI(loop1MBB, DL, TII->get(BNE))
156 .addReg(Scratch2).addReg(ShiftCmpVal).addMBB(sinkMBB);
157
158 // loop2MBB:
159 // and dest, dest, mask2
160 // or dest, dest, ShiftNewVal
161 // sc dest, dest, 0(ptr)
162 // beq dest, $0, loop1MBB
163 BuildMI(loop2MBB, DL, TII->get(Mips::AND), Scratch)
164 .addReg(Scratch, RegState::Kill)
165 .addReg(Mask2);
166 BuildMI(loop2MBB, DL, TII->get(Mips::OR), Scratch)
167 .addReg(Scratch, RegState::Kill)
168 .addReg(ShiftNewVal);
169 BuildMI(loop2MBB, DL, TII->get(SC), Scratch)
170 .addReg(Scratch, RegState::Kill)
171 .addReg(Ptr)
172 .addImm(0);
173 BuildMI(loop2MBB, DL, TII->get(BEQ))
174 .addReg(Scratch, RegState::Kill)
175 .addReg(ZERO)
176 .addMBB(loop1MBB);
177
178 // sinkMBB:
179 // srl srlres, Mask', shiftamt
180 // sign_extend dest,srlres
181 BuildMI(sinkMBB, DL, TII->get(Mips::SRLV), Dest)
182 .addReg(Scratch2)
183 .addReg(ShiftAmnt);
184 if (STI->hasMips32r2()) {
185 BuildMI(sinkMBB, DL, TII->get(SEOp), Dest).addReg(Dest);
186 } else {
187 const unsigned ShiftImm =
188 I->getOpcode() == Mips::ATOMIC_CMP_SWAP_I16_POSTRA ? 16 : 24;
189 BuildMI(sinkMBB, DL, TII->get(Mips::SLL), Dest)
190 .addReg(Dest, RegState::Kill)
191 .addImm(ShiftImm);
192 BuildMI(sinkMBB, DL, TII->get(Mips::SRA), Dest)
193 .addReg(Dest, RegState::Kill)
194 .addImm(ShiftImm);
195 }
196
197 LivePhysRegs LiveRegs;
198 computeAndAddLiveIns(LiveRegs, *loop1MBB);
199 computeAndAddLiveIns(LiveRegs, *loop2MBB);
200 computeAndAddLiveIns(LiveRegs, *sinkMBB);
201 computeAndAddLiveIns(LiveRegs, *exitMBB);
202
203 NMBBI = BB.end();
204 I->eraseFromParent();
205 return true;
206}

208bool MipsExpandPseudo::expandAtomicCmpSwap(MachineBasicBlock &BB,
209 MachineBasicBlock::iterator I,
210 MachineBasicBlock::iterator &NMBBI,
211 bool IsCapOp) {
212
213 unsigned Size = -1;
214 bool IsCapCmpXchg = false;
215 switch(I->getOpcode()) {
216 case Mips::ATOMIC_CMP_SWAP_I32_POSTRA: Size = 4; break;
217 case Mips::ATOMIC_CMP_SWAP_I64_POSTRA: Size = 8; break;
218 case Mips::CAP_ATOMIC_CMP_SWAP_I8_POSTRA: Size = 1; break;
219 case Mips::CAP_ATOMIC_CMP_SWAP_I16_POSTRA: Size = 2; break;
220 case Mips::CAP_ATOMIC_CMP_SWAP_I32_POSTRA: Size = 4; break;
221 case Mips::CAP_ATOMIC_CMP_SWAP_I64_POSTRA: Size = 8; break;
222 case Mips::CAP_ATOMIC_CMP_SWAP_CAP_POSTRA:
223 Size = CAP_ATOMIC_SIZE;
224 IsCapCmpXchg = true;
225 break;
226 default:
227 llvm_unreachable("Unhandled cmpxchg");
228 }
229 MachineFunction *MF = BB.getParent();
230
231 const bool ArePtrs64bit = STI->getABI().ArePtrs64bit();
232 DebugLoc DL = I->getDebugLoc();
233
234 unsigned LL, SC, ZERO, BNE, BEQ, MOVE;
235
236 if (Size <= 4) {
237 if (STI->inMicroMipsMode()) {
238 LL = STI->hasMips32r6() ? Mips::LL_MMR6 : Mips::LL_MM;
239 SC = STI->hasMips32r6() ? Mips::SC_MMR6 : Mips::SC_MM;
240 BNE = STI->hasMips32r6() ? Mips::BNEC_MMR6 : Mips::BNE_MM;
241 BEQ = STI->hasMips32r6() ? Mips::BEQC_MMR6 : Mips::BEQ_MM;
242 } else {
243 LL = STI->hasMips32r6()
244 ? (ArePtrs64bit ? Mips::LL64_R6 : Mips::LL_R6)
245 : (ArePtrs64bit ? Mips::LL64 : Mips::LL);
246 SC = STI->hasMips32r6()
247 ? (ArePtrs64bit ? Mips::SC64_R6 : Mips::SC_R6)
248 : (ArePtrs64bit ? Mips::SC64 : Mips::SC);
249 BNE = Mips::BNE;
250 BEQ = Mips::BEQ;
251 }
252
253 ZERO = Mips::ZERO;
254 MOVE = Mips::OR;
255 } else {
256 LL = STI->hasMips64r6() ? Mips::LLD_R6 : Mips::LLD;
257 SC = STI->hasMips64r6() ? Mips::SCD_R6 : Mips::SCD;
258 ZERO = Mips::ZERO_64;
259 BNE = Mips::BNE64;
260 BEQ = Mips::BEQ64;
261 MOVE = Mips::OR64;
262 }
263 if (IsCapOp) {
264 switch(Size) {
265 case 1:
266 LL = Mips::CLLB;
267 SC = Mips::CSCB;
268 break;
269 case 2:
270 LL = Mips::CLLH;
271 SC = Mips::CSCH;
272 break;
273 case 4:
274 LL = Mips::CLLW;
275 SC = Mips::CSCW;
276 break;
277 case 8:
278 LL = Mips::CLLD;
279 SC = Mips::CSCD;
280 break;
281 case CAP_ATOMIC_SIZE:
282 assert(IsCapCmpXchg);
283 LL = Mips::CLLC;
284 SC = Mips::CSCC;
285 break;
286 default:
287 llvm_unreachable("Unknown CHERI atomic size!");
288 }
289 }
290
291 unsigned Dest = I->getOperand(0).getReg();
292 unsigned Ptr = I->getOperand(1).getReg();
293 unsigned OldVal = I->getOperand(2).getReg();
294 unsigned NewVal = I->getOperand(3).getReg();
295 unsigned Scratch = I->getOperand(4).getReg();
296
297 // insert new blocks after the current block
298 const BasicBlock *LLVM_BB = BB.getBasicBlock();
299 MachineBasicBlock *loop1MBB = MF->CreateMachineBasicBlock(LLVM_BB);
300 MachineBasicBlock *loop2MBB = MF->CreateMachineBasicBlock(LLVM_BB);
301 MachineBasicBlock *exitMBB = MF->CreateMachineBasicBlock(LLVM_BB);
302 MachineFunction::iterator It = ++BB.getIterator();
303 MF->insert(It, loop1MBB);
304 MF->insert(It, loop2MBB);
305 MF->insert(It, exitMBB);
306
307 // Transfer the remainder of BB and its successor edges to exitMBB.
308 exitMBB->splice(exitMBB->begin(), &BB,
309 std::next(MachineBasicBlock::iterator(I)), BB.end());
310 exitMBB->transferSuccessorsAndUpdatePHIs(&BB);
311
312 // thisMBB:
313 // ...
314 // fallthrough --> loop1MBB
315 BB.addSuccessor(loop1MBB, BranchProbability::getOne());
316 loop1MBB->addSuccessor(exitMBB);
317 loop1MBB->addSuccessor(loop2MBB);
318 loop1MBB->normalizeSuccProbs();
319 loop2MBB->addSuccessor(loop1MBB);
320 loop2MBB->addSuccessor(exitMBB);
321 loop2MBB->normalizeSuccProbs();
322
323 // loop1MBB:
324 // ll dest, 0(ptr)
325 // bne dest, oldval, exitMBB
326 auto LLOp = BuildMI(loop1MBB, DL, TII->get(LL), Dest).addReg(Ptr);
327 if (!IsCapOp)
328 LLOp.addImm(0);
329 if (IsCapCmpXchg) {
330 unsigned CapCmp = STI->useCheriExactEquals() ? Mips::CEXEQ : Mips::CEQ;
331 // load, compare, and exit if not equal
332 // cllc dest, ptr
333 // ceq scratch, dest, oldval,
334 // beqz scratch exitMBB
335 BuildMI(loop1MBB, DL, TII->get(CapCmp), Scratch)
336 .addReg(Dest, RegState::Kill)
337 .addReg(OldVal);
338 BuildMI(loop1MBB, DL, TII->get(BEQ))
339 .addReg(Scratch, RegState::Kill)
340 .addReg(ZERO)
341 .addMBB(exitMBB);
342 } else {
343 BuildMI(loop1MBB, DL, TII->get(BNE))
344 .addReg(Dest, RegState::Kill)
345 .addReg(OldVal)
346 .addMBB(exitMBB);
347 }
348
349 // loop2MBB:
350 // move scratch, NewVal
351 // sc Scratch, Scratch, 0(ptr)
352 // beq Scratch, $0, loop1MBB
353 if (!IsCapOp) {
354 BuildMI(loop2MBB, DL, TII->get(MOVE), Scratch).addReg(NewVal).addReg(ZERO);
355 BuildMI(loop2MBB, DL, TII->get(SC), Scratch).addReg(Scratch).addReg(Ptr).addImm(0);
356 } else {
357 // No need for a move with the CHERI cll*/csc*
358 BuildMI(loop2MBB, DL, TII->get(SC), Scratch).addReg(NewVal).addReg(Ptr);
359 }
360 BuildMI(loop2MBB, DL, TII->get(BEQ))
361 .addReg(Scratch, RegState::Kill).addReg(ZERO).addMBB(loop1MBB);
362
363 LivePhysRegs LiveRegs;
364 computeAndAddLiveIns(LiveRegs, *loop1MBB);
365 computeAndAddLiveIns(LiveRegs, *loop2MBB);
366 computeAndAddLiveIns(LiveRegs, *exitMBB);
367
368 NMBBI = BB.end();
369 I->eraseFromParent();
370 return true;
371}

373bool MipsExpandPseudo::expandAtomicBinOpSubword(
374 MachineBasicBlock &BB, MachineBasicBlock::iterator I,
375 MachineBasicBlock::iterator &NMBBI) {
376
377 MachineFunction *MF = BB.getParent();
378
379 const bool ArePtrs64bit = STI->getABI().ArePtrs64bit();
380 DebugLoc DL = I->getDebugLoc();
381
382 unsigned LL, SC;
383 unsigned BEQ = Mips::BEQ;
384 unsigned SEOp = Mips::SEH;
385
386 if (STI->inMicroMipsMode()) {
387 LL = STI->hasMips32r6() ? Mips::LL_MMR6 : Mips::LL_MM;
388 SC = STI->hasMips32r6() ? Mips::SC_MMR6 : Mips::SC_MM;
389 BEQ = STI->hasMips32r6() ? Mips::BEQC_MMR6 : Mips::BEQ_MM;
390 } else {
391 LL = STI->hasMips32r6() ? (ArePtrs64bit ? Mips::LL64_R6 : Mips::LL_R6)
392 : (ArePtrs64bit ? Mips::LL64 : Mips::LL);
393 SC = STI->hasMips32r6() ? (ArePtrs64bit ? Mips::SC64_R6 : Mips::SC_R6)
394 : (ArePtrs64bit ? Mips::SC64 : Mips::SC);
395 }
396
397 bool IsSwap = false;
398 bool IsNand = false;
399
400 unsigned Opcode = 0;
401 switch (I->getOpcode()) {
402 case Mips::ATOMIC_LOAD_NAND_I8_POSTRA:
403 SEOp = Mips::SEB;
404 LLVM_FALLTHROUGH;
405 case Mips::ATOMIC_LOAD_NAND_I16_POSTRA:
406 IsNand = true;
407 break;
408 case Mips::ATOMIC_SWAP_I8_POSTRA:
409 SEOp = Mips::SEB;
410 LLVM_FALLTHROUGH;
411 case Mips::ATOMIC_SWAP_I16_POSTRA:
412 IsSwap = true;
413 break;
414 case Mips::ATOMIC_LOAD_ADD_I8_POSTRA:
415 SEOp = Mips::SEB;
416 LLVM_FALLTHROUGH;
417 case Mips::ATOMIC_LOAD_ADD_I16_POSTRA:
418 Opcode = Mips::ADDu;
419 break;
420 case Mips::ATOMIC_LOAD_SUB_I8_POSTRA:
421 SEOp = Mips::SEB;
422 LLVM_FALLTHROUGH;
423 case Mips::ATOMIC_LOAD_SUB_I16_POSTRA:
424 Opcode = Mips::SUBu;
425 break;
426 case Mips::ATOMIC_LOAD_AND_I8_POSTRA:
427 SEOp = Mips::SEB;
428 LLVM_FALLTHROUGH;
429 case Mips::ATOMIC_LOAD_AND_I16_POSTRA:
430 Opcode = Mips::AND;
431 break;
432 case Mips::ATOMIC_LOAD_OR_I8_POSTRA:
433 SEOp = Mips::SEB;
434 LLVM_FALLTHROUGH;
435 case Mips::ATOMIC_LOAD_OR_I16_POSTRA:
436 Opcode = Mips::OR;
437 break;
438 case Mips::ATOMIC_LOAD_XOR_I8_POSTRA:
439 SEOp = Mips::SEB;
440 LLVM_FALLTHROUGH;
441 case Mips::ATOMIC_LOAD_XOR_I16_POSTRA:
442 Opcode = Mips::XOR;
443 break;
444 default:
445 llvm_unreachable("Unknown subword atomic pseudo for expansion!");
446 }
447
448 unsigned Dest = I->getOperand(0).getReg();
449 unsigned Ptr = I->getOperand(1).getReg();
450 unsigned Incr = I->getOperand(2).getReg();
451 unsigned Mask = I->getOperand(3).getReg();
452 unsigned Mask2 = I->getOperand(4).getReg();
453 unsigned ShiftAmnt = I->getOperand(5).getReg();
454 unsigned OldVal = I->getOperand(6).getReg();
455 unsigned BinOpRes = I->getOperand(7).getReg();
456 unsigned StoreVal = I->getOperand(8).getReg();
457
458 const BasicBlock *LLVM_BB = BB.getBasicBlock();
459 MachineBasicBlock *loopMBB = MF->CreateMachineBasicBlock(LLVM_BB);
460 MachineBasicBlock *sinkMBB = MF->CreateMachineBasicBlock(LLVM_BB);
461 MachineBasicBlock *exitMBB = MF->CreateMachineBasicBlock(LLVM_BB);
462 MachineFunction::iterator It = ++BB.getIterator();
463 MF->insert(It, loopMBB);
464 MF->insert(It, sinkMBB);
465 MF->insert(It, exitMBB);
466
467 exitMBB->splice(exitMBB->begin(), &BB, std::next(I), BB.end());
468 exitMBB->transferSuccessorsAndUpdatePHIs(&BB);
469
470 BB.addSuccessor(loopMBB, BranchProbability::getOne());
471 loopMBB->addSuccessor(sinkMBB);
472 loopMBB->addSuccessor(loopMBB);
473 loopMBB->normalizeSuccProbs();
474
475 BuildMI(loopMBB, DL, TII->get(LL), OldVal).addReg(Ptr).addImm(0);
476 if (IsNand) {
477 // and andres, oldval, incr2
478 // nor binopres, $0, andres
479 // and newval, binopres, mask
480 BuildMI(loopMBB, DL, TII->get(Mips::AND), BinOpRes)
481 .addReg(OldVal)
482 .addReg(Incr);
483 BuildMI(loopMBB, DL, TII->get(Mips::NOR), BinOpRes)
484 .addReg(Mips::ZERO)
485 .addReg(BinOpRes);
486 BuildMI(loopMBB, DL, TII->get(Mips::AND), BinOpRes)
487 .addReg(BinOpRes)
488 .addReg(Mask);
489 } else if (!IsSwap) {
490 // <binop> binopres, oldval, incr2
491 // and newval, binopres, mask
492 BuildMI(loopMBB, DL, TII->get(Opcode), BinOpRes)
493 .addReg(OldVal)
494 .addReg(Incr);
495 BuildMI(loopMBB, DL, TII->get(Mips::AND), BinOpRes)
496 .addReg(BinOpRes)
497 .addReg(Mask);
498 } else { // atomic.swap
499 // and newval, incr2, mask
500 BuildMI(loopMBB, DL, TII->get(Mips::AND), BinOpRes)
501 .addReg(Incr)
502 .addReg(Mask);
503 }
504
505 // and StoreVal, OlddVal, Mask2
506 // or StoreVal, StoreVal, BinOpRes
507 // StoreVal<tied1> = sc StoreVal, 0(Ptr)
508 // beq StoreVal, zero, loopMBB
509 BuildMI(loopMBB, DL, TII->get(Mips::AND), StoreVal)
510 .addReg(OldVal).addReg(Mask2);
511 BuildMI(loopMBB, DL, TII->get(Mips::OR), StoreVal)
512 .addReg(StoreVal).addReg(BinOpRes);
513 BuildMI(loopMBB, DL, TII->get(SC), StoreVal)
514 .addReg(StoreVal).addReg(Ptr).addImm(0);
515 BuildMI(loopMBB, DL, TII->get(BEQ))
516 .addReg(StoreVal).addReg(Mips::ZERO).addMBB(loopMBB);
517
518 // sinkMBB:
519 // and maskedoldval1,oldval,mask
520 // srl srlres,maskedoldval1,shiftamt
521 // sign_extend dest,srlres
522
523 sinkMBB->addSuccessor(exitMBB, BranchProbability::getOne());
524
525 BuildMI(sinkMBB, DL, TII->get(Mips::AND), Dest)
526 .addReg(OldVal).addReg(Mask);
527 BuildMI(sinkMBB, DL, TII->get(Mips::SRLV), Dest)
528 .addReg(Dest).addReg(ShiftAmnt);
529
530 if (STI->hasMips32r2()) {
531 BuildMI(sinkMBB, DL, TII->get(SEOp), Dest).addReg(Dest);
532 } else {
533 const unsigned ShiftImm = SEOp == Mips::SEH ? 16 : 24;
534 BuildMI(sinkMBB, DL, TII->get(Mips::SLL), Dest)
535 .addReg(Dest, RegState::Kill)
536 .addImm(ShiftImm);
537 BuildMI(sinkMBB, DL, TII->get(Mips::SRA), Dest)
538 .addReg(Dest, RegState::Kill)
539 .addImm(ShiftImm);
540 }
541
542 LivePhysRegs LiveRegs;
543 computeAndAddLiveIns(LiveRegs, *loopMBB);
544 computeAndAddLiveIns(LiveRegs, *sinkMBB);
545 computeAndAddLiveIns(LiveRegs, *exitMBB);
546
547 NMBBI = BB.end();
548 I->eraseFromParent();
549
550 return true;
551}

553bool MipsExpandPseudo::expandPccRelativeAddr(
554 MachineBasicBlock &BB, MachineBasicBlock::iterator I,
555 MachineBasicBlock::iterator &NMBBI) {
556 DebugLoc DL = I->getDebugLoc();
557 // TODO: do we want to handle other kinds of instructions?
558 assert(I->getOperand(1).isBlockAddress() && "Only blockaddr handled for now");
559 auto BlockAddr = I->getOperand(1).getBlockAddress();
560 unsigned ResultReg = I->getOperand(0).getReg();
561 unsigned ScratchReg = I->getOperand(2).getReg();
562 auto BundleStart =
563 BuildMI(BB, I, DL, TII->get(Mips::CGetPCC), ResultReg).getInstr();
564 BuildMI(BB, I, DL, TII->get(Mips::LUi64))
565 .addReg(ScratchReg, RegState::Define)
566 .addBlockAddress(BlockAddr, 4, MipsII::MO_PCREL_HI);
567 BuildMI(BB, I, DL, TII->get(Mips::DADDiu))
568 .addReg(ScratchReg, RegState::Define)
569 .addReg(ScratchReg, RegState::Kill)
570 .addBlockAddress(BlockAddr, 8, MipsII::MO_PCREL_LO);
571 BuildMI(BB, I, DL, TII->get(Mips::CIncOffset), ResultReg)
572 .addReg(ResultReg, RegState::Kill)
573 .addReg(ScratchReg, RegState::Kill);
574
575 MachineBasicBlock::instr_iterator BundleIt(BundleStart);
576 // Note: the final CIncOffset doesn't need to be in the bundle but I think
577 // this makes it clearer when reading the MIR
578 finalizeBundle(BB, BundleIt, std::next(BundleIt, 4));
579 I->eraseFromParent();
580
581 return true;
582}

584bool MipsExpandPseudo::expandAtomicBinOp(MachineBasicBlock &BB,
585 MachineBasicBlock::iterator I,
586 MachineBasicBlock::iterator &NMBBI,
587 unsigned Size, bool IsCapOp) {
588 MachineFunction *MF = BB.getParent();
589
590 const bool ArePtrs64bit = STI->getABI().ArePtrs64bit();
591 DebugLoc DL = I->getDebugLoc();
592
593 unsigned LL, SC, ZERO, BEQ;
594
595 if (Size <= 4) {
596 if (STI->inMicroMipsMode()) {
597 LL = STI->hasMips32r6() ? Mips::LL_MMR6 : Mips::LL_MM;
598 SC = STI->hasMips32r6() ? Mips::SC_MMR6 : Mips::SC_MM;
599 BEQ = STI->hasMips32r6() ? Mips::BEQC_MMR6 : Mips::BEQ_MM;
600 } else {
601 LL = STI->hasMips32r6()
602 ? (ArePtrs64bit ? Mips::LL64_R6 : Mips::LL_R6)
603 : (ArePtrs64bit ? Mips::LL64 : Mips::LL);
604 SC = STI->hasMips32r6()
605 ? (ArePtrs64bit ? Mips::SC64_R6 : Mips::SC_R6)
606 : (ArePtrs64bit ? Mips::SC64 : Mips::SC);
607 BEQ = Mips::BEQ;
608 }
609
610 ZERO = Mips::ZERO;
611 } else {
612 LL = STI->hasMips64r6() ? Mips::LLD_R6 : Mips::LLD;
613 SC = STI->hasMips64r6() ? Mips::SCD_R6 : Mips::SCD;
614 ZERO = Mips::ZERO_64;
615 BEQ = Mips::BEQ64;
616 }
617 if (IsCapOp) {
618 switch(Size) {
619 case 1:
620 LL = Mips::CLLB;
621 SC = Mips::CSCB;
622 break;
623 case 2:
624 LL = Mips::CLLH;
625 SC = Mips::CSCH;
626 break;
627 case 4:
628 LL = Mips::CLLW;
629 SC = Mips::CSCW;
630 break;
631 case 8:
632 LL = Mips::CLLD;
633 SC = Mips::CSCD;
634 break;
635 case CAP_ATOMIC_SIZE:
636 LL = Mips::CLLC;
637 SC = Mips::CSCC;
638 break;
639 default:
640 llvm_unreachable("Unknown CHERI atomic size!");
641 }
642 }
643
644 unsigned OldVal = I->getOperand(0).getReg();
645 unsigned Ptr = I->getOperand(1).getReg();
646 unsigned Incr = I->getOperand(2).getReg();
647 unsigned Scratch = I->getOperand(3).getReg();
648
649 unsigned Opcode = 0;
650 unsigned OR = 0;
651 unsigned AND = 0;
652 unsigned NOR = 0;
653 bool IsNand = false;
654#define CHERI_SMALL_CASES(op) \
655 case Mips::CAP_ATOMIC_##op##_I8_POSTRA: \
656 case Mips::CAP_ATOMIC_##op##_I16_POSTRA: \
657 case Mips::CAP_ATOMIC_##op##_I32_POSTRA:
658 switch (I->getOpcode()) {
659 case Mips::ATOMIC_LOAD_ADD_I32_POSTRA:
660 CHERI_SMALL_CASES(LOAD_ADD)
661 Opcode = Mips::ADDu;
662 break;
663 case Mips::ATOMIC_LOAD_SUB_I32_POSTRA:
664 CHERI_SMALL_CASES(LOAD_SUB)
665 Opcode = Mips::SUBu;
666 break;
667 case Mips::ATOMIC_LOAD_AND_I32_POSTRA:
668 CHERI_SMALL_CASES(LOAD_AND)
669 Opcode = Mips::AND;
670 break;
671 case Mips::ATOMIC_LOAD_OR_I32_POSTRA:
672 CHERI_SMALL_CASES(LOAD_OR)
673 Opcode = Mips::OR;
674 break;
675 case Mips::ATOMIC_LOAD_XOR_I32_POSTRA:
676 CHERI_SMALL_CASES(LOAD_XOR)
677 Opcode = Mips::XOR;
678 break;
679 case Mips::ATOMIC_LOAD_NAND_I32_POSTRA:
680 CHERI_SMALL_CASES(LOAD_NAND)
681 IsNand = true;
682 AND = Mips::AND;
683 NOR = Mips::NOR;
684 break;
685 case Mips::ATOMIC_SWAP_I32_POSTRA:
686 CHERI_SMALL_CASES(SWAP)
687 OR = Mips::OR;
688 break;
689 case Mips::CAP_ATOMIC_LOAD_ADD_I64_POSTRA:
690 case Mips::ATOMIC_LOAD_ADD_I64_POSTRA:
691 Opcode = Mips::DADDu;
692 break;
693 case Mips::CAP_ATOMIC_LOAD_SUB_I64_POSTRA:
694 case Mips::ATOMIC_LOAD_SUB_I64_POSTRA:
695 Opcode = Mips::DSUBu;
696 break;
697 case Mips::CAP_ATOMIC_LOAD_AND_I64_POSTRA:
698 case Mips::ATOMIC_LOAD_AND_I64_POSTRA:
699 Opcode = Mips::AND64;
700 break;
701 case Mips::CAP_ATOMIC_LOAD_OR_I64_POSTRA:
702 case Mips::ATOMIC_LOAD_OR_I64_POSTRA:
703 Opcode = Mips::OR64;
704 break;
705 case Mips::CAP_ATOMIC_LOAD_XOR_I64_POSTRA:
706 case Mips::ATOMIC_LOAD_XOR_I64_POSTRA:
707 Opcode = Mips::XOR64;
708 break;
709 case Mips::CAP_ATOMIC_LOAD_NAND_I64_POSTRA:
710 case Mips::ATOMIC_LOAD_NAND_I64_POSTRA:
711 IsNand = true;
712 AND = Mips::AND64;
713 NOR = Mips::NOR64;
714 break;
715 case Mips::CAP_ATOMIC_SWAP_I64_POSTRA:
716 case Mips::CAP_ATOMIC_SWAP_CAP_POSTRA:
717 case Mips::ATOMIC_SWAP_I64_POSTRA:
718 OR = Mips::OR64;
719 break;
720 default:
721 llvm_unreachable("Unknown pseudo atomic!");
722 }
723
724 const BasicBlock *LLVM_BB = BB.getBasicBlock();
725 MachineBasicBlock *loopMBB = MF->CreateMachineBasicBlock(LLVM_BB);
726 MachineBasicBlock *exitMBB = MF->CreateMachineBasicBlock(LLVM_BB);
727 MachineFunction::iterator It = ++BB.getIterator();
728 MF->insert(It, loopMBB);
729 MF->insert(It, exitMBB);
730
731 exitMBB->splice(exitMBB->begin(), &BB, std::next(I), BB.end());
732 exitMBB->transferSuccessorsAndUpdatePHIs(&BB);
733
734 BB.addSuccessor(loopMBB, BranchProbability::getOne());
735 loopMBB->addSuccessor(exitMBB);
736 loopMBB->addSuccessor(loopMBB);
737 loopMBB->normalizeSuccProbs();
738
739 auto LLOp = BuildMI(loopMBB, DL, TII->get(LL), OldVal).addReg(Ptr);
740 if (!IsCapOp)
741 LLOp.addImm(0);
742 assert((OldVal != Ptr) && "Clobbered the wrong ptr reg!");
743 assert((OldVal != Incr) && "Clobbered the wrong reg!");
744 unsigned SCStoreValue = Scratch;
745 if (Opcode) {
746 BuildMI(loopMBB, DL, TII->get(Opcode), Scratch).addReg(OldVal).addReg(Incr);
747 } else if (IsNand) {
748 assert(AND && NOR &&
749 "Unknown nand instruction for atomic pseudo expansion");
750 BuildMI(loopMBB, DL, TII->get(AND), Scratch).addReg(OldVal).addReg(Incr);
751 BuildMI(loopMBB, DL, TII->get(NOR), Scratch).addReg(ZERO).addReg(Scratch);
752 } else {
753 assert(OR && "Unknown instruction for atomic pseudo expansion!");
754 // CHERI can use different registers for store value and result -> skip move
755 if (IsCapOp)
756 SCStoreValue = Incr;
757 else
758 BuildMI(loopMBB, DL, TII->get(OR), Scratch).addReg(Incr).addReg(ZERO);
759 }
760
761 auto SCOp = BuildMI(loopMBB, DL, TII->get(SC), Scratch).addReg(SCStoreValue).addReg(Ptr);
762 if (!IsCapOp)
763 SCOp.addImm(0);
764 BuildMI(loopMBB, DL, TII->get(BEQ)).addReg(Scratch).addReg(ZERO).addMBB(loopMBB);
765
766 NMBBI = BB.end();
767 I->eraseFromParent();
768
769 LivePhysRegs LiveRegs;
770 computeAndAddLiveIns(LiveRegs, *loopMBB);
771 computeAndAddLiveIns(LiveRegs, *exitMBB);
772
773 return true;
774}

776bool MipsExpandPseudo::expandMI(MachineBasicBlock &MBB,
777 MachineBasicBlock::iterator MBBI,
778 MachineBasicBlock::iterator &NMBB) {
779
780 bool Modified = false;
781
782 switch (MBBI->getOpcode()) {
783 case Mips::ATOMIC_CMP_SWAP_I32_POSTRA:
784 case Mips::ATOMIC_CMP_SWAP_I64_POSTRA:
785 return expandAtomicCmpSwap(MBB, MBBI, NMBB);
786 case Mips::ATOMIC_CMP_SWAP_I8_POSTRA:
787 case Mips::ATOMIC_CMP_SWAP_I16_POSTRA:
788 return expandAtomicCmpSwapSubword(MBB, MBBI, NMBB);
789 case Mips::ATOMIC_SWAP_I8_POSTRA:
790 case Mips::ATOMIC_SWAP_I16_POSTRA:
791 case Mips::ATOMIC_LOAD_NAND_I8_POSTRA:
792 case Mips::ATOMIC_LOAD_NAND_I16_POSTRA:
793 case Mips::ATOMIC_LOAD_ADD_I8_POSTRA:
794 case Mips::ATOMIC_LOAD_ADD_I16_POSTRA:
795 case Mips::ATOMIC_LOAD_SUB_I8_POSTRA:
796 case Mips::ATOMIC_LOAD_SUB_I16_POSTRA:
797 case Mips::ATOMIC_LOAD_AND_I8_POSTRA:
798 case Mips::ATOMIC_LOAD_AND_I16_POSTRA:
799 case Mips::ATOMIC_LOAD_OR_I8_POSTRA:
800 case Mips::ATOMIC_LOAD_OR_I16_POSTRA:
801 case Mips::ATOMIC_LOAD_XOR_I8_POSTRA:
802 case Mips::ATOMIC_LOAD_XOR_I16_POSTRA:
803 return expandAtomicBinOpSubword(MBB, MBBI, NMBB);
804 case Mips::ATOMIC_LOAD_ADD_I32_POSTRA:
805 case Mips::ATOMIC_LOAD_SUB_I32_POSTRA:
806 case Mips::ATOMIC_LOAD_AND_I32_POSTRA:
807 case Mips::ATOMIC_LOAD_OR_I32_POSTRA:
808 case Mips::ATOMIC_LOAD_XOR_I32_POSTRA:
809 case Mips::ATOMIC_LOAD_NAND_I32_POSTRA:
810 case Mips::ATOMIC_SWAP_I32_POSTRA:
811 return expandAtomicBinOp(MBB, MBBI, NMBB, 4);
812 case Mips::ATOMIC_LOAD_ADD_I64_POSTRA:
813 case Mips::ATOMIC_LOAD_SUB_I64_POSTRA:
814 case Mips::ATOMIC_LOAD_AND_I64_POSTRA:
815 case Mips::ATOMIC_LOAD_OR_I64_POSTRA:
816 case Mips::ATOMIC_LOAD_XOR_I64_POSTRA:
817 case Mips::ATOMIC_LOAD_NAND_I64_POSTRA:
818 case Mips::ATOMIC_SWAP_I64_POSTRA:
819 return expandAtomicBinOp(MBB, MBBI, NMBB, 8);
820
821 // CHERI instrs:
822 case Mips::CAP_ATOMIC_SWAP_I8_POSTRA:
823 case Mips::CAP_ATOMIC_LOAD_NAND_I8_POSTRA:
824 case Mips::CAP_ATOMIC_LOAD_ADD_I8_POSTRA:
825 case Mips::CAP_ATOMIC_LOAD_SUB_I8_POSTRA:
826 case Mips::CAP_ATOMIC_LOAD_AND_I8_POSTRA:
827 case Mips::CAP_ATOMIC_LOAD_OR_I8_POSTRA:
828 case Mips::CAP_ATOMIC_LOAD_XOR_I8_POSTRA:
829 return expandAtomicBinOp(MBB, MBBI, NMBB, 1, /*IsCapOp=*/true);
830 case Mips::CAP_ATOMIC_SWAP_I16_POSTRA:
831 case Mips::CAP_ATOMIC_LOAD_NAND_I16_POSTRA:
832 case Mips::CAP_ATOMIC_LOAD_ADD_I16_POSTRA:
833 case Mips::CAP_ATOMIC_LOAD_SUB_I16_POSTRA:
834 case Mips::CAP_ATOMIC_LOAD_AND_I16_POSTRA:
835 case Mips::CAP_ATOMIC_LOAD_OR_I16_POSTRA:
836 case Mips::CAP_ATOMIC_LOAD_XOR_I16_POSTRA:
837 return expandAtomicBinOp(MBB, MBBI, NMBB, 2, /*IsCapOp=*/true);
838 case Mips::CAP_ATOMIC_LOAD_ADD_I32_POSTRA:
839 case Mips::CAP_ATOMIC_LOAD_SUB_I32_POSTRA:
840 case Mips::CAP_ATOMIC_LOAD_AND_I32_POSTRA:
841 case Mips::CAP_ATOMIC_LOAD_OR_I32_POSTRA:
842 case Mips::CAP_ATOMIC_LOAD_XOR_I32_POSTRA:
843 case Mips::CAP_ATOMIC_LOAD_NAND_I32_POSTRA:
844 case Mips::CAP_ATOMIC_SWAP_I32_POSTRA:
845 return expandAtomicBinOp(MBB, MBBI, NMBB, 4, /*IsCapOp=*/true);
846 case Mips::CAP_ATOMIC_LOAD_ADD_I64_POSTRA:
847 case Mips::CAP_ATOMIC_LOAD_SUB_I64_POSTRA:
848 case Mips::CAP_ATOMIC_LOAD_AND_I64_POSTRA:
849 case Mips::CAP_ATOMIC_LOAD_OR_I64_POSTRA:
850 case Mips::CAP_ATOMIC_LOAD_XOR_I64_POSTRA:
851 case Mips::CAP_ATOMIC_LOAD_NAND_I64_POSTRA:
852 case Mips::CAP_ATOMIC_SWAP_I64_POSTRA:
853 return expandAtomicBinOp(MBB, MBBI, NMBB, 8, /*IsCapOp=*/true);
854
855 case Mips::CAP_ATOMIC_SWAP_CAP_POSTRA:
856 // TODO: case Mips::CAP_ATOMIC_LOAD_ADD_CAP_POSTRA:
857 // TODO: case Mips::CAP_ATOMIC_LOAD_SUB_CAP_POSTRA:
858 return expandAtomicBinOp(MBB, MBBI, NMBB, CAP_ATOMIC_SIZE, /*IsCapOp=*/true);
859
860 case Mips::CAP_ATOMIC_CMP_SWAP_I8_POSTRA:
861 case Mips::CAP_ATOMIC_CMP_SWAP_I16_POSTRA:
862 case Mips::CAP_ATOMIC_CMP_SWAP_I32_POSTRA:
863 case Mips::CAP_ATOMIC_CMP_SWAP_I64_POSTRA:
864 case Mips::CAP_ATOMIC_CMP_SWAP_CAP_POSTRA:
865 return expandAtomicCmpSwap(MBB, MBBI, NMBB, /*IsCapOp=*/true);
866 case Mips::PseudoPccRelativeAddressPostRA:
867 return expandPccRelativeAddr(MBB, MBBI, NMBB);
868
869 default:
870 return Modified;
871 }
872}
873
874bool MipsExpandPseudo::expandMBB(MachineBasicBlock &MBB) {
875 bool Modified = false;
876
877 MachineBasicBlock::iterator MBBI = MBB.begin(), E = MBB.end();
878 while (MBBI != E) {
879 MachineBasicBlock::iterator NMBBI = std::next(MBBI);
880 Modified |= expandMI(MBB, MBBI, NMBBI);
881 MBBI = NMBBI;
882 }
883
884 return Modified;
885}

887bool MipsExpandPseudo::runOnMachineFunction(MachineFunction &MF) {
888 STI = &static_cast<const MipsSubtarget &>(MF.getSubtarget());
889 TII = STI->getInstrInfo();
890
891 bool Modified = false;
892 for (MachineFunction::iterator MFI = MF.begin(), E = MF.end(); MFI != E;
893 ++MFI)
894 Modified |= expandMBB(*MFI);
895
896 if (Modified)
897 MF.RenumberBlocks();
898
899 return Modified;
900}
