35MCOperand MipsMCInstLower::LowerSymbolOperand(const MachineOperand &MO,
36 MachineOperandType MOTy,
37 unsigned Offset) const {
38 MCSymbolRefExpr::VariantKind Kind = MCSymbolRefExpr::VK_None;
39 MipsMCExpr::MipsExprKind TargetKind = MipsMCExpr::MEK_None;
40 bool IsGpOff = false;
41 bool IsCapTableOff = false;
42 const MCSymbol *Symbol;
43
44 switch(MO.getTargetFlags()) {
45 default:
46 llvm_unreachable("Invalid target flag!");
47 case MipsII::MO_NO_FLAG:
48 break;
49 case MipsII::MO_GPREL:
50 TargetKind = MipsMCExpr::MEK_GPREL;
51 break;
52 case MipsII::MO_GOT_CALL:
53 TargetKind = MipsMCExpr::MEK_GOT_CALL;
54 break;
55 case MipsII::MO_GOT:
56 TargetKind = MipsMCExpr::MEK_GOT;
57 break;
58 case MipsII::MO_ABS_HI:
59 TargetKind = MipsMCExpr::MEK_HI;
60 break;
61 case MipsII::MO_ABS_LO:
62 TargetKind = MipsMCExpr::MEK_LO;
63 break;
64 case MipsII::MO_TLSGD:
65 TargetKind = MipsMCExpr::MEK_TLSGD;
66 break;
67 case MipsII::MO_TLSLDM:
68 TargetKind = MipsMCExpr::MEK_TLSLDM;
69 break;
70 case MipsII::MO_DTPREL_HI:
71 TargetKind = MipsMCExpr::MEK_DTPREL_HI;
72 break;
73 case MipsII::MO_DTPREL_LO:
74 TargetKind = MipsMCExpr::MEK_DTPREL_LO;
75 break;
76 case MipsII::MO_GOTTPREL:
77 TargetKind = MipsMCExpr::MEK_GOTTPREL;
78 break;
79 case MipsII::MO_TPREL_HI:
80 TargetKind = MipsMCExpr::MEK_TPREL_HI;
81 break;
82 case MipsII::MO_TPREL_LO:
83 TargetKind = MipsMCExpr::MEK_TPREL_LO;
84 break;
85 case MipsII::MO_GPOFF_HI:
86 TargetKind = MipsMCExpr::MEK_HI;
87 IsGpOff = true;
88 break;
89 case MipsII::MO_GPOFF_LO:
90 TargetKind = MipsMCExpr::MEK_LO;
91 IsGpOff = true;
92 break;
93 case MipsII::MO_GOT_DISP:
94 TargetKind = MipsMCExpr::MEK_GOT_DISP;
95 break;
96 case MipsII::MO_GOT_HI16:
97 TargetKind = MipsMCExpr::MEK_GOT_HI16;
98 break;
99 case MipsII::MO_GOT_LO16:
100 TargetKind = MipsMCExpr::MEK_GOT_LO16;
101 break;
102 case MipsII::MO_GOT_PAGE:
103 TargetKind = MipsMCExpr::MEK_GOT_PAGE;
104 break;
105 case MipsII::MO_GOT_OFST:
106 TargetKind = MipsMCExpr::MEK_GOT_OFST;
107 break;
108 case MipsII::MO_HIGHER:
109 TargetKind = MipsMCExpr::MEK_HIGHER;
110 break;
111 case MipsII::MO_HIGHEST:
112 TargetKind = MipsMCExpr::MEK_HIGHEST;
113 break;
114 case MipsII::MO_CALL_HI16:
115 TargetKind = MipsMCExpr::MEK_CALL_HI16;
116 break;
117 case MipsII::MO_CALL_LO16:
118 TargetKind = MipsMCExpr::MEK_CALL_LO16;
119 break;
120 case MipsII::MO_JALR:
121 return MCOperand();
122 case MipsII::MO_PCREL_HI:
123 TargetKind = MipsMCExpr::MEK_PCREL_HI16;
124 break;
125 case MipsII::MO_PCREL_LO:
126 TargetKind = MipsMCExpr::MEK_PCREL_LO16;
127 break;
128 case MipsII::MO_CAPTAB11:
129 TargetKind = MipsMCExpr::MEK_CAPTABLE11;
130 break;
131 case MipsII::MO_CAPTAB20:
132 TargetKind = MipsMCExpr::MEK_CAPTABLE20;
133 break;
134 case MipsII::MO_CAPTAB_HI16:
135 TargetKind = MipsMCExpr::MEK_CAPTABLE_HI16;
136 break;
137 case MipsII::MO_CAPTAB_LO16:
138 TargetKind = MipsMCExpr::MEK_CAPTABLE_LO16;
139 break;
140 case MipsII::MO_CAPTAB_CALL11:
141 TargetKind = MipsMCExpr::MEK_CAPCALL11;
142 break;
143 case MipsII::MO_CAPTAB_CALL20:
144 TargetKind = MipsMCExpr::MEK_CAPCALL20;
145 break;
146 case MipsII::MO_CAPTAB_CALL_HI16:
147 TargetKind = MipsMCExpr::MEK_CAPCALL_HI16;
148 break;
149 case MipsII::MO_CAPTAB_CALL_LO16:
150 TargetKind = MipsMCExpr::MEK_CAPCALL_LO16;
151 break;
152 case MipsII::MO_CAPTABLE_OFF_HI:
153 TargetKind = MipsMCExpr::MEK_HI;
154 IsCapTableOff = true;
155 break;
156 case MipsII::MO_CAPTABLE_OFF_LO:
157 TargetKind = MipsMCExpr::MEK_LO;
158 IsCapTableOff = true;
159 break;
160 case MipsII::MO_CAPTAB_TLSGD_HI16:
161 TargetKind = MipsMCExpr::MEK_CAPTAB_TLSGD_HI16;
162 break;
163 case MipsII::MO_CAPTAB_TLSGD_LO16:
164 TargetKind = MipsMCExpr::MEK_CAPTAB_TLSGD_LO16;
165 break;
166 case MipsII::MO_CAPTAB_TLSLDM_HI16:
167 TargetKind = MipsMCExpr::MEK_CAPTAB_TLSLDM_HI16;
168 break;
169 case MipsII::MO_CAPTAB_TLSLDM_LO16:
170 TargetKind = MipsMCExpr::MEK_CAPTAB_TLSLDM_LO16;
171 break;
172 case MipsII::MO_CAPTAB_TPREL_HI16:
173 TargetKind = MipsMCExpr::MEK_CAPTAB_TPREL_HI16;
174 break;
175 case MipsII::MO_CAPTAB_TPREL_LO16:
176 TargetKind = MipsMCExpr::MEK_CAPTAB_TPREL_LO16;
177 break;
178 }
179
180 switch (MOTy) {
181 case MachineOperand::MO_MachineBasicBlock:
182 Symbol = MO.getMBB()->getSymbol();
183 break;
184
185 case MachineOperand::MO_GlobalAddress:
186 Symbol = AsmPrinter.getSymbol(MO.getGlobal());
187 Offset += MO.getOffset();
188 break;
189
190 case MachineOperand::MO_BlockAddress:
191 Symbol = AsmPrinter.GetBlockAddressSymbol(MO.getBlockAddress());
192 Offset += MO.getOffset();
193 break;
194
195 case MachineOperand::MO_ExternalSymbol:
196 Symbol = AsmPrinter.GetExternalSymbolSymbol(MO.getSymbolName());
197 Offset += MO.getOffset();
198 break;
199
200 case MachineOperand::MO_MCSymbol:
201 Symbol = MO.getMCSymbol();
202 Offset += MO.getOffset();
203 break;
204
205 case MachineOperand::MO_JumpTableIndex:
206 Symbol = AsmPrinter.GetJTISymbol(MO.getIndex());
207 break;
208
209 case MachineOperand::MO_ConstantPoolIndex:
210 Symbol = AsmPrinter.GetCPISymbol(MO.getIndex());
211 Offset += MO.getOffset();
212 break;
213
214 default:
215 llvm_unreachable("<unknown operand type>");
216 }
217
218 const MCExpr *Expr = MCSymbolRefExpr::create(Symbol, Kind, *Ctx);
219
220 if (Offset) {
221 // Assume offset is never negative.
222 assert(Offset > 0);
223
224 Expr = MCBinaryExpr::createAdd(Expr, MCConstantExpr::create(Offset, *Ctx),
225 *Ctx);
226 }
227
228 if (IsGpOff)
229 Expr = MipsMCExpr::createGpOff(TargetKind, Expr, *Ctx);
230 else if (IsCapTableOff)
231 Expr = MipsMCExpr::createCaptableOff(TargetKind, Expr, *Ctx);
232 else if (TargetKind != MipsMCExpr::MEK_None)
233 Expr = MipsMCExpr::create(TargetKind, Expr, *Ctx);
234
235 return MCOperand::createExpr(Expr);
236}

238MCOperand MipsMCInstLower::LowerOperand(const MachineOperand &MO,
239 unsigned offset) const {
240 MachineOperandType MOTy = MO.getType();
241
242 switch (MOTy) {
243 default: llvm_unreachable("unknown operand type");
244 case MachineOperand::MO_Register:
245 // Ignore all implicit register operands.
246 if (MO.isImplicit()) break;
247 return MCOperand::createReg(MO.getReg());
248 case MachineOperand::MO_Immediate:
249 return MCOperand::createImm(MO.getImm() + offset);
250 case MachineOperand::MO_MachineBasicBlock:
251 case MachineOperand::MO_GlobalAddress:
252 case MachineOperand::MO_ExternalSymbol:
253 case MachineOperand::MO_MCSymbol:
254 case MachineOperand::MO_JumpTableIndex:
255 case MachineOperand::MO_ConstantPoolIndex:
256 case MachineOperand::MO_BlockAddress:
257 return LowerSymbolOperand(MO, MOTy, offset);
258 case MachineOperand::MO_RegisterMask:
259 break;
260 }
261
262 return MCOperand();
263}

265MCOperand MipsMCInstLower::createSub(MachineBasicBlock *BB1,
266 MachineBasicBlock *BB2,
267 MipsMCExpr::MipsExprKind Kind) const {
268 const MCSymbolRefExpr *Sym1 = MCSymbolRefExpr::create(BB1->getSymbol(), *Ctx);
269 const MCSymbolRefExpr *Sym2 = MCSymbolRefExpr::create(BB2->getSymbol(), *Ctx);
270 const MCBinaryExpr *Sub = MCBinaryExpr::createSub(Sym1, Sym2, *Ctx);
271
272 return MCOperand::createExpr(MipsMCExpr::create(Kind, Sub, *Ctx));
273}

275void MipsMCInstLower::
276lowerLongBranchLUi(const MachineInstr *MI, MCInst &OutMI) const {
277 OutMI.setOpcode(Mips::LUi);
278
279 // Lower register operand.
280 OutMI.addOperand(LowerOperand(MI->getOperand(0)));
281
282 MipsMCExpr::MipsExprKind Kind;
283 unsigned TargetFlags = MI->getOperand(1).getTargetFlags();
284 switch (TargetFlags) {
285 case MipsII::MO_HIGHEST:
286 Kind = MipsMCExpr::MEK_HIGHEST;
287 break;
288 case MipsII::MO_HIGHER:
289 Kind = MipsMCExpr::MEK_HIGHER;
290 break;
291 case MipsII::MO_ABS_HI:
292 Kind = MipsMCExpr::MEK_HI;
293 break;
294 case MipsII::MO_ABS_LO:
295 Kind = MipsMCExpr::MEK_LO;
296 break;
297 default:
298 report_fatal_error("Unexpected flags for lowerLongBranchLUi");
299 }
300
301 if (MI->getNumOperands() == 2) {
302 const MCExpr *Expr =
303 MCSymbolRefExpr::create(MI->getOperand(1).getMBB()->getSymbol(), *Ctx);
304 const MipsMCExpr *MipsExpr = MipsMCExpr::create(Kind, Expr, *Ctx);
305 OutMI.addOperand(MCOperand::createExpr(MipsExpr));
306 } else if (MI->getNumOperands() == 3) {
307 // Create %hi($tgt-$baltgt).
308 OutMI.addOperand(createSub(MI->getOperand(1).getMBB(),
309 MI->getOperand(2).getMBB(), Kind));
310 }
311}

313void MipsMCInstLower::lowerLongBranchADDiu(const MachineInstr *MI,
314 MCInst &OutMI, int Opcode) const {
315 OutMI.setOpcode(Opcode);
316
317 MipsMCExpr::MipsExprKind Kind;
318 unsigned TargetFlags = MI->getOperand(2).getTargetFlags();
319 switch (TargetFlags) {
320 case MipsII::MO_HIGHEST:
321 Kind = MipsMCExpr::MEK_HIGHEST;
322 break;
323 case MipsII::MO_HIGHER:
324 Kind = MipsMCExpr::MEK_HIGHER;
325 break;
326 case MipsII::MO_ABS_HI:
327 Kind = MipsMCExpr::MEK_HI;
328 break;
329 case MipsII::MO_ABS_LO:
330 Kind = MipsMCExpr::MEK_LO;
331 break;
332 default:
333 report_fatal_error("Unexpected flags for lowerLongBranchADDiu");
334 }
335
336 // Lower two register operands.
337 for (unsigned I = 0, E = 2; I != E; ++I) {
338 const MachineOperand &MO = MI->getOperand(I);
339 OutMI.addOperand(LowerOperand(MO));
340 }
341
342 if (MI->getNumOperands() == 3) {
343 // Lower register operand.
344 const MCExpr *Expr =
345 MCSymbolRefExpr::create(MI->getOperand(2).getMBB()->getSymbol(), *Ctx);
346 const MipsMCExpr *MipsExpr = MipsMCExpr::create(Kind, Expr, *Ctx);
347 OutMI.addOperand(MCOperand::createExpr(MipsExpr));
348 } else if (MI->getNumOperands() == 4) {
349 // Create %lo($tgt-$baltgt) or %hi($tgt-$baltgt).
350 OutMI.addOperand(createSub(MI->getOperand(2).getMBB(),
351 MI->getOperand(3).getMBB(), Kind));
352 }
353}
354
355bool MipsMCInstLower::lowerLongBranch(const MachineInstr *MI,
356 MCInst &OutMI) const {
357 switch (MI->getOpcode()) {
358 default:
359 return false;
360 case Mips::LONG_BRANCH_LUi:
361 case Mips::LONG_BRANCH_LUi2Op:
362 case Mips::LONG_BRANCH_LUi2Op_64:
363 lowerLongBranchLUi(MI, OutMI);
364 return true;
365 case Mips::LONG_BRANCH_ADDiu:
366 case Mips::LONG_BRANCH_ADDiu2Op:
367 lowerLongBranchADDiu(MI, OutMI, Mips::ADDiu);
368 return true;
369 case Mips::LONG_BRANCH_DADDiu:
370 case Mips::LONG_BRANCH_DADDiu2Op:
371 lowerLongBranchADDiu(MI, OutMI, Mips::DADDiu);
372 return true;
373 }
374}
375
376void MipsMCInstLower::Lower(const MachineInstr *MI, MCInst &OutMI) const {
377 if (lowerLongBranch(MI, OutMI))
378 return;
379
380 OutMI.setOpcode(MI->getOpcode());
381
382 for (unsigned i = 0, e = MI->getNumOperands(); i != e; ++i) {
383 const MachineOperand &MO = MI->getOperand(i);
384 MCOperand MCOp = LowerOperand(MO);
385
386 if (MCOp.isValid())
387 OutMI.addOperand(MCOp);
388 }
