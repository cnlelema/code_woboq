811 bool isCallingConvWin64(CallingConv::ID CC) const {
812 switch (CC) {
813 // On Win64, all these conventions just use the default convention.
814 case CallingConv::C:
815 case CallingConv::Fast:
816 case CallingConv::Swift:
817 case CallingConv::X86_FastCall:
818 case CallingConv::X86_StdCall:
819 case CallingConv::X86_ThisCall:
820 case CallingConv::X86_VectorCall:
821 case CallingConv::Intel_OCL_BI:
822 return isTargetWin64();
823 // This convention allows using the Win64 convention on other targets.
824 case CallingConv::Win64:
825 return true;
826 // This convention allows using the SysV convention on Windows targets.
827 case CallingConv::X86_64_SysV:
828 return false;
829 // Otherwise, who knows what this is.
830 default:
831 return false;
