10// number stored in an exception registration stack object. The registration

144bool WinEHStatePass::runOnFunction(Function &F) {
145 // Don't insert state stores or exception handler thunks for
146 // available_externally functions. The handler needs to reference the LSDA,
147 // which will not be emitted in this case.
148 if (F.hasAvailableExternallyLinkage())
149 return false;
150
151 // Check the personality. Do nothing if this personality doesn't use funclets.
152 if (!F.hasPersonalityFn())
153 return false;
154 PersonalityFn =
155 dyn_cast<Function>(F.getPersonalityFn()->stripPointerCasts());
156 if (!PersonalityFn)
157 return false;
158 Personality = classifyEHPersonality(PersonalityFn);
159 if (!isFuncletEHPersonality(Personality))
160 return false;
161
162 // Skip this function if there are no EH pads and we aren't using IR-level
163 // outlining.
164 bool HasPads = false;
165 for (BasicBlock &BB : F) {
166 if (BB.isEHPad()) {
167 HasPads = true;
168 break;
169 }
170 }
171 if (!HasPads)
172 return false;
173
174 Type *Int8PtrType = Type::getInt8PtrTy(TheModule->getContext());
175 SetJmp3 = TheModule->getOrInsertFunction(
176 "_setjmp3", FunctionType::get(
177 Type::getInt32Ty(TheModule->getContext()),
178 {Int8PtrType, Type::getInt32Ty(TheModule->getContext())},
179 /*isVarArg=*/true));
180
181 // Disable frame pointer elimination in this function.
182 // FIXME: Do the nested handlers need to keep the parent ebp in ebp, or can we
183 // use an arbitrary register?
184 F.addFnAttr("no-frame-pointer-elim", "true");
185
186 emitExceptionRegistrationRecord(&F);
187
188 // The state numbers calculated here in IR must agree with what we calculate
189 // later on for the MachineFunction. In particular, if an IR pass deletes an
190 // unreachable EH pad after this point before machine CFG construction, we
191 // will be in trouble. If this assumption is ever broken, we should turn the
192 // numbers into an immutable analysis pass.
193 WinEHFuncInfo FuncInfo;
194 addStateStores(F, FuncInfo);
195
196 // Reset per-function state.
197 PersonalityFn = nullptr;
198 Personality = EHPersonality::Unknown;
199 UseStackGuard = false;
200 RegNode = nullptr;
201 EHGuardNode = nullptr;
202
203 return true;
204}

206/// Get the common EH registration subobject:
207/// typedef _EXCEPTION_DISPOSITION (*PEXCEPTION_ROUTINE)(
208/// _EXCEPTION_RECORD *, void *, _CONTEXT *, void *);
209/// struct EHRegistrationNode {
210/// EHRegistrationNode *Next;
211/// PEXCEPTION_ROUTINE Handler;
212/// };
213Type *WinEHStatePass::getEHLinkRegistrationType() {
214 if (EHLinkRegistrationTy)
215 return EHLinkRegistrationTy;
216 LLVMContext &Context = TheModule->getContext();
217 EHLinkRegistrationTy = StructType::create(Context, "EHRegistrationNode");
218 Type *FieldTys[] = {
219 EHLinkRegistrationTy->getPointerTo(0), // EHRegistrationNode *Next
220 Type::getInt8PtrTy(Context) // EXCEPTION_DISPOSITION (*Handler)(...)
221 };
222 EHLinkRegistrationTy->setBody(FieldTys, false);
223 return EHLinkRegistrationTy;
224}

269// Emit an exception registration record. These are stack allocations with the
270// common subobject of two pointers: the previous registration record (the old
271// fs:00) and the personality function for the current frame. The data before
272// and after that is personality function specific.
273void WinEHStatePass::emitExceptionRegistrationRecord(Function *F) {
274 assert(Personality == EHPersonality::MSVC_CXX ||
275 Personality == EHPersonality::MSVC_X86SEH);
276
277 // Struct type of RegNode. Used for GEPing.
278 Type *RegNodeTy;
279
280 IRBuilder<> Builder(&F->getEntryBlock(), F->getEntryBlock().begin());
281 Type *Int8PtrType = Builder.getInt8PtrTy();
282 Type *Int32Ty = Builder.getInt32Ty();
283 Type *VoidTy = Builder.getVoidTy();
284
285 if (Personality == EHPersonality::MSVC_CXX) {
286 RegNodeTy = getCXXEHRegistrationType();
287 RegNode = Builder.CreateAlloca(RegNodeTy);
288 // SavedESP = llvm.stacksave()
289 Value *SP = Builder.CreateCall(
290 Intrinsic::getDeclaration(TheModule, Intrinsic::stacksave,
291 { Type::getInt8PtrTy(F->getContext()) }), {});
292 Builder.CreateStore(SP, Builder.CreateStructGEP(RegNodeTy, RegNode, 0));
293 // TryLevel = -1
294 StateFieldIndex = 2;
295 ParentBaseState = -1;
296 insertStateNumberStore(&*Builder.GetInsertPoint(), ParentBaseState);
297 // Handler = __ehhandler$F
298 Function *Trampoline = generateLSDAInEAXThunk(F);
299 Link = Builder.CreateStructGEP(RegNodeTy, RegNode, 1);
300 linkExceptionRegistration(Builder, Trampoline);
301
302 CxxLongjmpUnwind = TheModule->getOrInsertFunction(
303 "__CxxLongjmpUnwind",
304 FunctionType::get(VoidTy, Int8PtrType, /*isVarArg=*/false));
305 cast<Function>(CxxLongjmpUnwind.getCallee()->stripPointerCasts())
306 ->setCallingConv(CallingConv::X86_StdCall);
307 } else if (Personality == EHPersonality::MSVC_X86SEH) {
308 // If _except_handler4 is in use, some additional guard checks and prologue
309 // stuff is required.
310 StringRef PersonalityName = PersonalityFn->getName();
311 UseStackGuard = (PersonalityName == "_except_handler4");
312
313 // Allocate local structures.
314 RegNodeTy = getSEHRegistrationType();
315 RegNode = Builder.CreateAlloca(RegNodeTy);
316 if (UseStackGuard)
317 EHGuardNode = Builder.CreateAlloca(Int32Ty);
318
319 // SavedESP = llvm.stacksave()
320 Value *SP = Builder.CreateCall(
321 Intrinsic::getDeclaration(TheModule, Intrinsic::stacksave,
322 { Type::getInt8PtrTy(F->getContext()) }), {});
323 Builder.CreateStore(SP, Builder.CreateStructGEP(RegNodeTy, RegNode, 0));
324 // TryLevel = -2 / -1
325 StateFieldIndex = 4;
326 ParentBaseState = UseStackGuard ? -2 : -1;
327 insertStateNumberStore(&*Builder.GetInsertPoint(), ParentBaseState);
328 // ScopeTable = llvm.x86.seh.lsda(F)
329 Value *LSDA = emitEHLSDA(Builder, F);
330 LSDA = Builder.CreatePtrToInt(LSDA, Int32Ty);
331 // If using _except_handler4, xor the address of the table with
332 // __security_cookie.
333 if (UseStackGuard) {
334 Cookie = TheModule->getOrInsertGlobal("__security_cookie", Int32Ty);
335 Value *Val = Builder.CreateLoad(Int32Ty, Cookie, "cookie");
336 LSDA = Builder.CreateXor(LSDA, Val);
337 }
338 Builder.CreateStore(LSDA, Builder.CreateStructGEP(RegNodeTy, RegNode, 3));
339
340 // If using _except_handler4, the EHGuard contains: FramePtr xor Cookie.
341 if (UseStackGuard) {
342 Value *Val = Builder.CreateLoad(Int32Ty, Cookie);
343 Value *FrameAddr = Builder.CreateCall(
344 Intrinsic::getDeclaration(
345 TheModule, Intrinsic::frameaddress,
346 Builder.getInt8PtrTy(
347 F->getParent()->getDataLayout().getProgramAddressSpace())),
348 Builder.getInt32(0), "frameaddr");
349 Value *FrameAddrI32 = Builder.CreatePtrToInt(FrameAddr, Int32Ty);
350 FrameAddrI32 = Builder.CreateXor(FrameAddrI32, Val);
351 Builder.CreateStore(FrameAddrI32, EHGuardNode);
352 }
353
354 // Register the exception handler.
355 Link = Builder.CreateStructGEP(RegNodeTy, RegNode, 2);
356 linkExceptionRegistration(Builder, PersonalityFn);
357
358 SehLongjmpUnwind = TheModule->getOrInsertFunction(
359 UseStackGuard ? "_seh_longjmp_unwind4" : "_seh_longjmp_unwind",
360 FunctionType::get(Type::getVoidTy(TheModule->getContext()), Int8PtrType,
361 /*isVarArg=*/false));
362 cast<Function>(SehLongjmpUnwind.getCallee()->stripPointerCasts())
363 ->setCallingConv(CallingConv::X86_StdCall);
364 } else {
365 llvm_unreachable("unexpected personality function");
366 }
367
368 // Insert an unlink before all returns.
369 for (BasicBlock &BB : *F) {
370 Instruction *T = BB.getTerminator();
371 if (!isa<ReturnInst>(T))
372 continue;
373 Builder.SetInsertPoint(T);
374 unlinkExceptionRegistration(Builder);
375 }
376}

378Value *WinEHStatePass::emitEHLSDA(IRBuilder<> &Builder, Function *F) {
379 Value *FI8 = Builder.CreateBitCast(F, Type::getInt8PtrTy(F->getContext()));
380 return Builder.CreateCall(
381 Intrinsic::getDeclaration(TheModule, Intrinsic::x86_seh_lsda), FI8);
382}

384/// Generate a thunk that puts the LSDA of ParentFunc in EAX and then calls
385/// PersonalityFn, forwarding the parameters passed to PEXCEPTION_ROUTINE:
386/// typedef _EXCEPTION_DISPOSITION (*PEXCEPTION_ROUTINE)(
387/// _EXCEPTION_RECORD *, void *, _CONTEXT *, void *);
388/// We essentially want this code:
389/// movl $lsda, %eax
390/// jmpl ___CxxFrameHandler3
391Function *WinEHStatePass::generateLSDAInEAXThunk(Function *ParentFunc) {
392 LLVMContext &Context = ParentFunc->getContext();
393 Type *Int32Ty = Type::getInt32Ty(Context);
394 Type *Int8PtrType = Type::getInt8PtrTy(Context);
395 Type *ArgTys[5] = {Int8PtrType, Int8PtrType, Int8PtrType, Int8PtrType,
396 Int8PtrType};
397 FunctionType *TrampolineTy =
398 FunctionType::get(Int32Ty, makeArrayRef(&ArgTys[0], 4),
399 /*isVarArg=*/false);
400 FunctionType *TargetFuncTy =
401 FunctionType::get(Int32Ty, makeArrayRef(&ArgTys[0], 5),
402 /*isVarArg=*/false);
403 Function *Trampoline =
404 Function::Create(TrampolineTy, GlobalValue::InternalLinkage,
405 Twine("__ehhandler$") + GlobalValue::dropLLVMManglingEscape(
406 ParentFunc->getName()),
407 TheModule);
408 if (auto *C = ParentFunc->getComdat())
409 Trampoline->setComdat(C);
410 BasicBlock *EntryBB = BasicBlock::Create(Context, "entry", Trampoline);
411 IRBuilder<> Builder(EntryBB);
412 Value *LSDA = emitEHLSDA(Builder, ParentFunc);
413 Value *CastPersonality =
414 Builder.CreateBitCast(PersonalityFn, TargetFuncTy->getPointerTo());
415 auto AI = Trampoline->arg_begin();
416 Value *Args[5] = {LSDA, &*AI++, &*AI++, &*AI++, &*AI++};
417 CallInst *Call = Builder.CreateCall(TargetFuncTy, CastPersonality, Args);
418 // Can't use musttail due to prototype mismatch, but we can use tail.
419 Call->setTailCall(true);
420 // Set inreg so we pass it in EAX.
421 Call->addParamAttr(0, Attribute::InReg);
422 Builder.CreateRet(Call);
423 return Trampoline;
424}

426void WinEHStatePass::linkExceptionRegistration(IRBuilder<> &Builder,
427 Function *Handler) {
428 // Emit the .safeseh directive for this function.
429 Handler->addFnAttr("safeseh");
430
431 Type *LinkTy = getEHLinkRegistrationType();
432 // Handler = Handler
433 Value *HandlerI8 = Builder.CreateBitCast(Handler, Builder.getInt8PtrTy());
434 Builder.CreateStore(HandlerI8, Builder.CreateStructGEP(LinkTy, Link, 1));
435 // Next = [fs:00]
436 Constant *FSZero =
437 Constant::getNullValue(LinkTy->getPointerTo()->getPointerTo(257));
438 Value *Next = Builder.CreateLoad(LinkTy->getPointerTo(), FSZero);
439 Builder.CreateStore(Next, Builder.CreateStructGEP(LinkTy, Link, 0));
440 // [fs:00] = Link
441 Builder.CreateStore(Link, FSZero);
442}
443
444void WinEHStatePass::unlinkExceptionRegistration(IRBuilder<> &Builder) {
445 // Clone Link into the current BB for better address mode folding.
446 if (auto *GEP = dyn_cast<GetElementPtrInst>(Link)) {
447 GEP = cast<GetElementPtrInst>(GEP->clone());
448 Builder.Insert(GEP);
449 Link = GEP;
450 }
451 Type *LinkTy = getEHLinkRegistrationType();
452 // [fs:00] = Link->Next
453 Value *Next = Builder.CreateLoad(LinkTy->getPointerTo(),
454 Builder.CreateStructGEP(LinkTy, Link, 0));
455 Constant *FSZero =
456 Constant::getNullValue(LinkTy->getPointerTo()->getPointerTo(257));
457 Builder.CreateStore(Next, FSZero);
458}

460// Calls to setjmp(p) are lowered to _setjmp3(p, 0) by the frontend.
461// The idea behind _setjmp3 is that it takes an optional number of personality
462// specific parameters to indicate how to restore the personality-specific frame
463// state when longjmp is initiated. Typically, the current TryLevel is saved.
464void WinEHStatePass::rewriteSetJmpCallSite(IRBuilder<> &Builder, Function &F,
465 CallSite CS, Value *State) {
466 // Don't rewrite calls with a weird number of arguments.
467 if (CS.getNumArgOperands() != 2)
468 return;
469
470 Instruction *Inst = CS.getInstruction();
471
472 SmallVector<OperandBundleDef, 1> OpBundles;
473 CS.getOperandBundlesAsDefs(OpBundles);
474
475 SmallVector<Value *, 3> OptionalArgs;
476 if (Personality == EHPersonality::MSVC_CXX) {
477 OptionalArgs.push_back(CxxLongjmpUnwind.getCallee());
478 OptionalArgs.push_back(State);
479 OptionalArgs.push_back(emitEHLSDA(Builder, &F));
480 } else if (Personality == EHPersonality::MSVC_X86SEH) {
481 OptionalArgs.push_back(SehLongjmpUnwind.getCallee());
482 OptionalArgs.push_back(State);
483 if (UseStackGuard)
484 OptionalArgs.push_back(Cookie);
485 } else {
486 llvm_unreachable("unhandled personality!");
487 }
488
489 SmallVector<Value *, 5> Args;
490 Args.push_back(
491 Builder.CreateBitCast(CS.getArgOperand(0), Builder.getInt8PtrTy()));
492 Args.push_back(Builder.getInt32(OptionalArgs.size()));
493 Args.append(OptionalArgs.begin(), OptionalArgs.end());
494
495 CallSite NewCS;
496 if (CS.isCall()) {
497 auto *CI = cast<CallInst>(Inst);
498 CallInst *NewCI = Builder.CreateCall(SetJmp3, Args, OpBundles);
499 NewCI->setTailCallKind(CI->getTailCallKind());
500 NewCS = NewCI;
501 } else {
502 auto *II = cast<InvokeInst>(Inst);
503 NewCS = Builder.CreateInvoke(
504 SetJmp3, II->getNormalDest(), II->getUnwindDest(), Args, OpBundles);
505 }
506 NewCS.setCallingConv(CS.getCallingConv());
507 NewCS.setAttributes(CS.getAttributes());
508 NewCS->setDebugLoc(CS->getDebugLoc());
509
510 Instruction *NewInst = NewCS.getInstruction();
511 NewInst->takeName(Inst);
512 Inst->replaceAllUsesWith(NewInst);
513 Inst->eraseFromParent();
514}

516// Figure out what state we should assign calls in this block.
517int WinEHStatePass::getBaseStateForBB(
518 DenseMap<BasicBlock *, ColorVector> &BlockColors, WinEHFuncInfo &FuncInfo,
519 BasicBlock *BB) {
520 int BaseState = ParentBaseState;
521 auto &BBColors = BlockColors[BB];
522
523 assert(BBColors.size() == 1 && "multi-color BB not removed by preparation");
524 BasicBlock *FuncletEntryBB = BBColors.front();
525 if (auto *FuncletPad =
526 dyn_cast<FuncletPadInst>(FuncletEntryBB->getFirstNonPHI())) {
527 auto BaseStateI = FuncInfo.FuncletBaseStateMap.find(FuncletPad);
528 if (BaseStateI != FuncInfo.FuncletBaseStateMap.end())
529 BaseState = BaseStateI->second;
530 }
531
532 return BaseState;
533}

535// Calculate the state a call-site is in.
536int WinEHStatePass::getStateForCallSite(
537 DenseMap<BasicBlock *, ColorVector> &BlockColors, WinEHFuncInfo &FuncInfo,
538 CallSite CS) {
539 if (auto *II = dyn_cast<InvokeInst>(CS.getInstruction())) {
540 // Look up the state number of the EH pad this unwinds to.
541 assert(FuncInfo.InvokeStateMap.count(II) && "invoke has no state!");
542 return FuncInfo.InvokeStateMap[II];
543 }
544 // Possibly throwing call instructions have no actions to take after
545 // an unwind. Ensure they are in the -1 state.
546 return getBaseStateForBB(BlockColors, FuncInfo, CS.getParent());
547}

549// Calculate the intersection of all the FinalStates for a BasicBlock's
550// predecessors.
551static int getPredState(DenseMap<BasicBlock *, int> &FinalStates, Function &F,
552 int ParentBaseState, BasicBlock *BB) {
553 // The entry block has no predecessors but we know that the prologue always
554 // sets us up with a fixed state.
555 if (&F.getEntryBlock() == BB)
556 return ParentBaseState;
557
558 // This is an EH Pad, conservatively report this basic block as overdefined.
559 if (BB->isEHPad())
560 return OverdefinedState;
561
562 int CommonState = OverdefinedState;
563 for (BasicBlock *PredBB : predecessors(BB)) {
564 // We didn't manage to get a state for one of these predecessors,
565 // conservatively report this basic block as overdefined.
566 auto PredEndState = FinalStates.find(PredBB);
567 if (PredEndState == FinalStates.end())
568 return OverdefinedState;
569
570 // This code is reachable via exceptional control flow,
571 // conservatively report this basic block as overdefined.
572 if (isa<CatchReturnInst>(PredBB->getTerminator()))
573 return OverdefinedState;
574
575 int PredState = PredEndState->second;
576 assert(PredState != OverdefinedState &&
577 "overdefined BBs shouldn't be in FinalStates");
578 if (CommonState == OverdefinedState)
579 CommonState = PredState;
580
581 // At least two predecessors have different FinalStates,
582 // conservatively report this basic block as overdefined.
583 if (CommonState != PredState)
584 return OverdefinedState;
585 }
586
587 return CommonState;
588}

590// Calculate the intersection of all the InitialStates for a BasicBlock's
591// successors.
592static int getSuccState(DenseMap<BasicBlock *, int> &InitialStates, Function &F,
593 int ParentBaseState, BasicBlock *BB) {
594 // This block rejoins normal control flow,
595 // conservatively report this basic block as overdefined.
596 if (isa<CatchReturnInst>(BB->getTerminator()))
597 return OverdefinedState;
598
599 int CommonState = OverdefinedState;
600 for (BasicBlock *SuccBB : successors(BB)) {
601 // We didn't manage to get a state for one of these predecessors,
602 // conservatively report this basic block as overdefined.
603 auto SuccStartState = InitialStates.find(SuccBB);
604 if (SuccStartState == InitialStates.end())
605 return OverdefinedState;
606
607 // This is an EH Pad, conservatively report this basic block as overdefined.
608 if (SuccBB->isEHPad())
609 return OverdefinedState;
610
611 int SuccState = SuccStartState->second;
612 assert(SuccState != OverdefinedState &&
613 "overdefined BBs shouldn't be in FinalStates");
614 if (CommonState == OverdefinedState)
615 CommonState = SuccState;
616
617 // At least two successors have different InitialStates,
618 // conservatively report this basic block as overdefined.
619 if (CommonState != SuccState)
620 return OverdefinedState;
621 }
622
623 return CommonState;
624}

626bool WinEHStatePass::isStateStoreNeeded(EHPersonality Personality,
627 CallSite CS) {
628 if (!CS)
629 return false;
630
631 // If the function touches memory, it needs a state store.
632 if (isAsynchronousEHPersonality(Personality))
633 return !CS.doesNotAccessMemory();
634
635 // If the function throws, it needs a state store.
636 return !CS.doesNotThrow();
637}

639void WinEHStatePass::addStateStores(Function &F, WinEHFuncInfo &FuncInfo) {
640 // Mark the registration node. The backend needs to know which alloca it is so
641 // that it can recover the original frame pointer.
642 IRBuilder<> Builder(RegNode->getNextNode());
643 Value *RegNodeI8 = Builder.CreateBitCast(RegNode, Builder.getInt8PtrTy());
644 Builder.CreateCall(
645 Intrinsic::getDeclaration(TheModule, Intrinsic::x86_seh_ehregnode),
646 {RegNodeI8});
647
648 if (EHGuardNode) {
649 IRBuilder<> Builder(EHGuardNode->getNextNode());
650 Value *EHGuardNodeI8 =
651 Builder.CreateBitCast(EHGuardNode, Builder.getInt8PtrTy());
652 Builder.CreateCall(
653 Intrinsic::getDeclaration(TheModule, Intrinsic::x86_seh_ehguard),
654 {EHGuardNodeI8});
655 }
656
657 // Calculate state numbers.
658 if (isAsynchronousEHPersonality(Personality))
659 calculateSEHStateNumbers(&F, FuncInfo);
660 else
661 calculateWinCXXEHStateNumbers(&F, FuncInfo);
662
663 // Iterate all the instructions and emit state number stores.
664 DenseMap<BasicBlock *, ColorVector> BlockColors = colorEHFunclets(F);
665 ReversePostOrderTraversal<Function *> RPOT(&F);
666
667 // InitialStates yields the state of the first call-site for a BasicBlock.
668 DenseMap<BasicBlock *, int> InitialStates;
669 // FinalStates yields the state of the last call-site for a BasicBlock.
670 DenseMap<BasicBlock *, int> FinalStates;
671 // Worklist used to revisit BasicBlocks with indeterminate
672 // Initial/Final-States.
673 std::deque<BasicBlock *> Worklist;
674 // Fill in InitialStates and FinalStates for BasicBlocks with call-sites.
675 for (BasicBlock *BB : RPOT) {
676 int InitialState = OverdefinedState;
677 int FinalState;
678 if (&F.getEntryBlock() == BB)
679 InitialState = FinalState = ParentBaseState;
680 for (Instruction &I : *BB) {
681 CallSite CS(&I);
682 if (!isStateStoreNeeded(Personality, CS))
683 continue;
684
685 int State = getStateForCallSite(BlockColors, FuncInfo, CS);
686 if (InitialState == OverdefinedState)
687 InitialState = State;
688 FinalState = State;
689 }
690 // No call-sites in this basic block? That's OK, we will come back to these
691 // in a later pass.
692 if (InitialState == OverdefinedState) {
693 Worklist.push_back(BB);
694 continue;
695 }
696 LLVM_DEBUG(dbgs() << "X86WinEHState: " << BB->getName()
697 << " InitialState=" << InitialState << '\n');
698 LLVM_DEBUG(dbgs() << "X86WinEHState: " << BB->getName()
699 << " FinalState=" << FinalState << '\n');
700 InitialStates.insert({BB, InitialState});
701 FinalStates.insert({BB, FinalState});
702 }
703
704 // Try to fill-in InitialStates and FinalStates which have no call-sites.
705 while (!Worklist.empty()) {
706 BasicBlock *BB = Worklist.front();
707 Worklist.pop_front();
708 // This BasicBlock has already been figured out, nothing more we can do.
709 if (InitialStates.count(BB) != 0)
710 continue;
711
712 int PredState = getPredState(FinalStates, F, ParentBaseState, BB);
713 if (PredState == OverdefinedState)
714 continue;
715
716 // We successfully inferred this BasicBlock's state via it's predecessors;
717 // enqueue it's successors to see if we can infer their states.
718 InitialStates.insert({BB, PredState});
719 FinalStates.insert({BB, PredState});
720 for (BasicBlock *SuccBB : successors(BB))
721 Worklist.push_back(SuccBB);
722 }
723
724 // Try to hoist stores from successors.
725 for (BasicBlock *BB : RPOT) {
726 int SuccState = getSuccState(InitialStates, F, ParentBaseState, BB);
727 if (SuccState == OverdefinedState)
728 continue;
729
730 // Update our FinalState to reflect the common InitialState of our
731 // successors.
732 FinalStates.insert({BB, SuccState});
733 }
734
735 // Finally, insert state stores before call-sites which transition us to a new
736 // state.
737 for (BasicBlock *BB : RPOT) {
738 auto &BBColors = BlockColors[BB];
739 BasicBlock *FuncletEntryBB = BBColors.front();
740 if (isa<CleanupPadInst>(FuncletEntryBB->getFirstNonPHI()))
741 continue;
742
743 int PrevState = getPredState(FinalStates, F, ParentBaseState, BB);
744 LLVM_DEBUG(dbgs() << "X86WinEHState: " << BB->getName()
745 << " PrevState=" << PrevState << '\n');
746
747 for (Instruction &I : *BB) {
748 CallSite CS(&I);
749 if (!isStateStoreNeeded(Personality, CS))
750 continue;
751
752 int State = getStateForCallSite(BlockColors, FuncInfo, CS);
753 if (State != PrevState)
754 insertStateNumberStore(&I, State);
755 PrevState = State;
756 }
757
758 // We might have hoisted a state store into this block, emit it now.
759 auto EndState = FinalStates.find(BB);
760 if (EndState != FinalStates.end())
761 if (EndState->second != PrevState)
762 insertStateNumberStore(BB->getTerminator(), EndState->second);
763 }
764
765 SmallVector<CallSite, 1> SetJmp3CallSites;
766 for (BasicBlock *BB : RPOT) {
767 for (Instruction &I : *BB) {
768 CallSite CS(&I);
769 if (!CS)
770 continue;
771 if (CS.getCalledValue()->stripPointerCasts() !=
772 SetJmp3.getCallee()->stripPointerCasts())
773 continue;
774
775 SetJmp3CallSites.push_back(CS);
776 }
777 }
778
779 for (CallSite CS : SetJmp3CallSites) {
780 auto &BBColors = BlockColors[CS->getParent()];
781 BasicBlock *FuncletEntryBB = BBColors.front();
782 bool InCleanup = isa<CleanupPadInst>(FuncletEntryBB->getFirstNonPHI());
783
784 IRBuilder<> Builder(CS.getInstruction());
785 Value *State;
786 if (InCleanup) {
787 Value *StateField = Builder.CreateStructGEP(RegNode->getAllocatedType(),
788 RegNode, StateFieldIndex);
789 State = Builder.CreateLoad(Builder.getInt32Ty(), StateField);
790 } else {
791 State = Builder.getInt32(getStateForCallSite(BlockColors, FuncInfo, CS));
792 }
793 rewriteSetJmpCallSite(Builder, F, CS, State);
794 }
795}
796
797void WinEHStatePass::insertStateNumberStore(Instruction *IP, int State) {
798 IRBuilder<> Builder(IP);
799 Value *StateField = Builder.CreateStructGEP(RegNode->getAllocatedType(),
800 RegNode, StateFieldIndex);
801 Builder.CreateStore(Builder.getInt32(State), StateField);
802}
