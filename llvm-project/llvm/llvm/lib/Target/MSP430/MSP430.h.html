21 // MSP430 specific condition code.
22 enum CondCodes {
23 COND_E = 0, // aka COND_Z
24 COND_NE = 1, // aka COND_NZ
25 COND_HS = 2, // aka COND_C
26 COND_LO = 3, // aka COND_NC
27 COND_GE = 4,
28 COND_L = 5,
29 COND_N = 6, // jump if negative
30 COND_NONE, // unconditional
31
32 COND_INVALID = -1
33 };
34}
