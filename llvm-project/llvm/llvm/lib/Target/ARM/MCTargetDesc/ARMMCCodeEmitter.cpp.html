294 unsigned getModImmOpValue(const MCInst &MI, unsigned Op,
295 SmallVectorImpl<MCFixup> &Fixups,
296 const MCSubtargetInfo &ST) const {
297 const MCOperand &MO = MI.getOperand(Op);
298
299 // Support for fixups (MCFixup)
300 if (MO.isExpr()) {
301 const MCExpr *Expr = MO.getExpr();
302 // Fixups resolve to plain values that need to be encoded.
303 MCFixupKind Kind = MCFixupKind(ARM::fixup_arm_mod_imm);
304 Fixups.push_back(MCFixup::create(0, Expr, Kind, MI.getLoc()));
305 return 0;
306 }
307
308 // Immediate is already in its encoded format
309 return MO.getImm();
310 }

441/// NEONThumb2DataIPostEncoder - Post-process encoded NEON data-processing
442/// instructions, and rewrite them to their Thumb2 form if we are currently in
443/// Thumb2 mode.
444unsigned ARMMCCodeEmitter::NEONThumb2DataIPostEncoder(const MCInst &MI,
445 unsigned EncodedValue,
446 const MCSubtargetInfo &STI) const {
447 if (isThumb2(STI)) {
448 // NEON Thumb2 data-processsing encodings are very simple: bit 24 is moved
449 // to bit 12 of the high half-word (i.e. bit 28), and bits 27-24 are
450 // set to 1111.
451 unsigned Bit24 = EncodedValue & 0x01000000;
452 unsigned Bit28 = Bit24 << 4;
453 EncodedValue &= 0xEFFFFFFF;
454 EncodedValue |= Bit28;
455 EncodedValue |= 0x0F000000;
456 }
457
458 return EncodedValue;
459}

513/// getMachineOpValue - Return binary encoding of operand. If the machine
514/// operand requires relocation, record the relocation and return zero.
515unsigned ARMMCCodeEmitter::
516getMachineOpValue(const MCInst &MI, const MCOperand &MO,
517 SmallVectorImpl<MCFixup> &Fixups,
518 const MCSubtargetInfo &STI) const {
519 if (MO.isReg()) {
520 unsigned Reg = MO.getReg();
521 unsigned RegNo = CTX.getRegisterInfo()->getEncodingValue(Reg);
522
523 // Q registers are encoded as 2x their register number.
524 switch (Reg) {
525 default:
526 return RegNo;
527 case ARM::Q0: case ARM::Q1: case ARM::Q2: case ARM::Q3:
528 case ARM::Q4: case ARM::Q5: case ARM::Q6: case ARM::Q7:
529 case ARM::Q8: case ARM::Q9: case ARM::Q10: case ARM::Q11:
530 case ARM::Q12: case ARM::Q13: case ARM::Q14: case ARM::Q15:
531 return 2 * RegNo;
532 }
533 } else if (MO.isImm()) {
534 return static_cast<unsigned>(MO.getImm());
535 } else if (MO.isFPImm()) {
536 return static_cast<unsigned>(APFloat(MO.getFPImm())
537 .bitcastToAPInt().getHiBits(32).getLimitedValue());
538 }
539
540 llvm_unreachable("Unable to encode MCOperand!");
541}

543/// getAddrModeImmOpValue - Return encoding info for 'reg +/- imm' operand.
544bool ARMMCCodeEmitter::
545EncodeAddrModeOpValues(const MCInst &MI, unsigned OpIdx, unsigned &Reg,
546 unsigned &Imm, SmallVectorImpl<MCFixup> &Fixups,
547 const MCSubtargetInfo &STI) const {
548 const MCOperand &MO = MI.getOperand(OpIdx);
549 const MCOperand &MO1 = MI.getOperand(OpIdx + 1);
550
551 Reg = CTX.getRegisterInfo()->getEncodingValue(MO.getReg());
552
553 int32_t SImm = MO1.getImm();
554 bool isAdd = true;
555
556 // Special value for #-0
557 if (SImm == INT32_MIN) {
558 SImm = 0;
559 isAdd = false;
560 }
561
562 // Immediate is always encoded as positive. The 'U' bit controls add vs sub.
563 if (SImm < 0) {
564 SImm = -SImm;
565 isAdd = false;
566 }
567
568 Imm = SImm;
569 return isAdd;
570}

572/// getBranchTargetOpValue - Helper function to get the branch target operand,
573/// which is either an immediate or requires a fixup.
574static uint32_t getBranchTargetOpValue(const MCInst &MI, unsigned OpIdx,
575 unsigned FixupKind,
576 SmallVectorImpl<MCFixup> &Fixups,
577 const MCSubtargetInfo &STI) {
578 const MCOperand &MO = MI.getOperand(OpIdx);
579
580 // If the destination is an immediate, we have nothing to do.
581 if (MO.isImm()) return MO.getImm();
582 assert(MO.isExpr() && "Unexpected branch target type!");
583 const MCExpr *Expr = MO.getExpr();
584 MCFixupKind Kind = MCFixupKind(FixupKind);
585 Fixups.push_back(MCFixup::create(0, Expr, Kind, MI.getLoc()));
586
587 // All of the information is in the fixup.
588 return 0;
589}

591// Thumb BL and BLX use a strange offset encoding where bits 22 and 21 are
592// determined by negating them and XOR'ing them with bit 23.
593static int32_t encodeThumbBLOffset(int32_t offset) {
594 offset >>= 1;
595 uint32_t S = (offset & 0x800000) >> 23;
596 uint32_t J1 = (offset & 0x400000) >> 22;
597 uint32_t J2 = (offset & 0x200000) >> 21;
598 J1 = (~J1 & 0x1);
599 J2 = (~J2 & 0x1);
600 J1 ^= S;
601 J2 ^= S;
602
603 offset &= ~0x600000;
604 offset |= J1 << 22;
605 offset |= J2 << 21;
606
607 return offset;
608}

835/// getITMaskOpValue - Return the architectural encoding of an IT
836/// predication mask, given the MCOperand format.
837uint32_t ARMMCCodeEmitter::
838getITMaskOpValue(const MCInst &MI, unsigned OpIdx,
839 SmallVectorImpl<MCFixup> &Fixups,
840 const MCSubtargetInfo &STI) const {
841 const MCOperand MaskMO = MI.getOperand(OpIdx);
842 assert(MaskMO.isImm() && "Unexpected operand type!");
843
844 unsigned Mask = MaskMO.getImm();
845
846 // IT masks are encoded as a sequence of replacement low-order bits
847 // for the condition code. So if the low bit of the starting
848 // condition code is 1, then we have to flip all the bits above the
849 // terminating bit (which is the lowest 1 bit).
850 assert(OpIdx > 0 && "IT mask appears first!");
851 const MCOperand CondMO = MI.getOperand(OpIdx-1);
852 assert(CondMO.isImm() && "Unexpected operand type!");
853 if (CondMO.getImm() & 1) {
854 unsigned LowBit = Mask & -Mask;
855 unsigned BitsAboveLowBit = 0xF & (-LowBit << 1);
856 Mask ^= BitsAboveLowBit;
857 }
858
859 return Mask;
860}

891/// getAddrModeImm12OpValue - Return encoding info for 'reg +/- imm12' operand.
892uint32_t ARMMCCodeEmitter::
893getAddrModeImm12OpValue(const MCInst &MI, unsigned OpIdx,
894 SmallVectorImpl<MCFixup> &Fixups,
895 const MCSubtargetInfo &STI) const {
896 // {17-13} = reg
897 // {12} = (U)nsigned (add == '1', sub == '0')
898 // {11-0} = imm12
899 unsigned Reg, Imm12;
900 bool isAdd = true;
901 // If The first operand isn't a register, we have a label reference.
902 const MCOperand &MO = MI.getOperand(OpIdx);
903 if (!MO.isReg()) {
904 Reg = CTX.getRegisterInfo()->getEncodingValue(ARM::PC); // Rn is PC.
905 Imm12 = 0;
906
907 if (MO.isExpr()) {
908 const MCExpr *Expr = MO.getExpr();
909 isAdd = false ; // 'U' bit is set as part of the fixup.
910
911 MCFixupKind Kind;
912 if (isThumb2(STI))
913 Kind = MCFixupKind(ARM::fixup_t2_ldst_pcrel_12);
914 else
915 Kind = MCFixupKind(ARM::fixup_arm_ldst_pcrel_12);
916 Fixups.push_back(MCFixup::create(0, Expr, Kind, MI.getLoc()));
917
918 ++MCNumCPRelocations;
919 } else {
920 Reg = ARM::PC;
921 int32_t Offset = MO.getImm();
922 if (Offset == INT32_MIN) {
923 Offset = 0;
924 isAdd = false;
925 } else if (Offset < 0) {
926 Offset *= -1;
927 isAdd = false;
928 }
929 Imm12 = Offset;
930 }
931 } else
932 isAdd = EncodeAddrModeOpValues(MI, OpIdx, Reg, Imm12, Fixups, STI);
933
934 uint32_t Binary = Imm12 & 0xfff;
935 // Immediate is always encoded as positive. The 'U' bit controls add vs sub.
936 if (isAdd)
937 Binary |= (1 << 12);
938 Binary |= (Reg << 13);
939 return Binary;
940}

942template<unsigned Bits, unsigned Shift>
943uint32_t ARMMCCodeEmitter::
944getT2ScaledImmOpValue(const MCInst &MI, unsigned OpIdx,
945 SmallVectorImpl<MCFixup> &Fixups,
946 const MCSubtargetInfo &STI) const {
947 // FIXME: The immediate operand should have already been encoded like this
948 // before ever getting here. The encoder method should just need to combine
949 // the MI operands for the register and the offset into a single
950 // representation for the complex operand in the .td file. This isn't just
951 // style, unfortunately. As-is, we can't represent the distinct encoding
952 // for #-0.
953
954 // {Bits} = (U)nsigned (add == '1', sub == '0')
955 // {(Bits-1)-0} = immediate
956 int32_t Imm = MI.getOperand(OpIdx).getImm();
957 bool isAdd = Imm >= 0;
958
959 // Immediate is always encoded as positive. The 'U' bit controls add vs sub.
960 if (Imm < 0)
961 Imm = -(uint32_t)Imm;
962
963 Imm >>= Shift;
964
965 uint32_t Binary = Imm & ((1U << Bits) - 1);
966 // Immediate is always encoded as positive. The 'U' bit controls add vs sub.
967 if (isAdd)
968 Binary |= (1U << Bits);
969 return Binary;
970}

1054uint32_t
1055ARMMCCodeEmitter::getHiLo16ImmOpValue(const MCInst &MI, unsigned OpIdx,
1056 SmallVectorImpl<MCFixup> &Fixups,
1057 const MCSubtargetInfo &STI) const {
1058 // {20-16} = imm{15-12}
1059 // {11-0} = imm{11-0}
1060 const MCOperand &MO = MI.getOperand(OpIdx);
1061 if (MO.isImm())
1062 // Hi / lo 16 bits already extracted during earlier passes.
1063 return static_cast<unsigned>(MO.getImm());
1064
1065 // Handle :upper16: and :lower16: assembly prefixes.
1066 const MCExpr *E = MO.getExpr();
1067 MCFixupKind Kind;
1068 if (E->getKind() == MCExpr::Target) {
1069 const ARMMCExpr *ARM16Expr = cast<ARMMCExpr>(E);
1070 E = ARM16Expr->getSubExpr();
1071
1072 if (const MCConstantExpr *MCE = dyn_cast<MCConstantExpr>(E)) {
1073 const int64_t Value = MCE->getValue();
1074 if (Value > UINT32_MAX)
1075 report_fatal_error("constant value truncated (limited to 32-bit)");
1076
1077 switch (ARM16Expr->getKind()) {
1078 case ARMMCExpr::VK_ARM_HI16:
1079 return (int32_t(Value) & 0xffff0000) >> 16;
1080 case ARMMCExpr::VK_ARM_LO16:
1081 return (int32_t(Value) & 0x0000ffff);
1082 default: llvm_unreachable("Unsupported ARMFixup");
1083 }
1084 }
1085
1086 switch (ARM16Expr->getKind()) {
1087 default: llvm_unreachable("Unsupported ARMFixup");
1088 case ARMMCExpr::VK_ARM_HI16:
1089 Kind = MCFixupKind(isThumb(STI) ? ARM::fixup_t2_movt_hi16
1090 : ARM::fixup_arm_movt_hi16);
1091 break;
1092 case ARMMCExpr::VK_ARM_LO16:
1093 Kind = MCFixupKind(isThumb(STI) ? ARM::fixup_t2_movw_lo16
1094 : ARM::fixup_arm_movw_lo16);
1095 break;
1096 }
1097
1098 Fixups.push_back(MCFixup::create(0, E, Kind, MI.getLoc()));
1099 return 0;
1100 }
1101 // If the expression doesn't have :upper16: or :lower16: on it,
1102 // it's just a plain immediate expression, previously those evaluated to
1103 // the lower 16 bits of the expression regardless of whether
1104 // we have a movt or a movw, but that led to misleadingly results.
1105 // This is disallowed in the AsmParser in validateInstruction()
1106 // so this should never happen.
1107 llvm_unreachable("expression without :upper16: or :lower16:");
1108}

1110uint32_t ARMMCCodeEmitter::
1111getLdStSORegOpValue(const MCInst &MI, unsigned OpIdx,
1112 SmallVectorImpl<MCFixup> &Fixups,
1113 const MCSubtargetInfo &STI) const {
1114 const MCOperand &MO = MI.getOperand(OpIdx);
1115 const MCOperand &MO1 = MI.getOperand(OpIdx+1);
1116 const MCOperand &MO2 = MI.getOperand(OpIdx+2);
1117 unsigned Rn = CTX.getRegisterInfo()->getEncodingValue(MO.getReg());
1118 unsigned Rm = CTX.getRegisterInfo()->getEncodingValue(MO1.getReg());
1119 unsigned ShImm = ARM_AM::getAM2Offset(MO2.getImm());
1120 bool isAdd = ARM_AM::getAM2Op(MO2.getImm()) == ARM_AM::add;
1121 ARM_AM::ShiftOpc ShOp = ARM_AM::getAM2ShiftOpc(MO2.getImm());
1122 unsigned SBits = getShiftOp(ShOp);
1123
1124 // While "lsr #32" and "asr #32" exist, they are encoded with a 0 in the shift
1125 // amount. However, it would be an easy mistake to make so check here.
1126 assert((ShImm & ~0x1f) == 0 && "Out of range shift amount");
1127
1128 // {16-13} = Rn
1129 // {12} = isAdd
1130 // {11-0} = shifter
1131 // {3-0} = Rm
1132 // {4} = 0
1133 // {6-5} = type
1134 // {11-7} = imm
1135 uint32_t Binary = Rm;
1136 Binary |= Rn << 13;
1137 Binary |= SBits << 5;
1138 Binary |= ShImm << 7;
1139 if (isAdd)
1140 Binary |= 1 << 12;
1141 return Binary;

1144uint32_t ARMMCCodeEmitter::
1145getAddrMode2OffsetOpValue(const MCInst &MI, unsigned OpIdx,
1146 SmallVectorImpl<MCFixup> &Fixups,
1147 const MCSubtargetInfo &STI) const {
1148 // {13} 1 == imm12, 0 == Rm
1149 // {12} isAdd
1150 // {11-0} imm12/Rm
1151 const MCOperand &MO = MI.getOperand(OpIdx);
1152 const MCOperand &MO1 = MI.getOperand(OpIdx+1);
1153 unsigned Imm = MO1.getImm();
1154 bool isAdd = ARM_AM::getAM2Op(Imm) == ARM_AM::add;
1155 bool isReg = MO.getReg() != 0;
1156 uint32_t Binary = ARM_AM::getAM2Offset(Imm);
1157 // if reg +/- reg, Rm will be non-zero. Otherwise, we have reg +/- imm12
1158 if (isReg) {
1159 ARM_AM::ShiftOpc ShOp = ARM_AM::getAM2ShiftOpc(Imm);

1161 Binary |= getShiftOp(ShOp) << 5; // Shift type is bits [6:5]
1162 Binary |= CTX.getRegisterInfo()->getEncodingValue(MO.getReg()); // Rm is bits [3:0]
1163 }
1164 return Binary | (isAdd << 12) | (isReg << 13);
1165}

1357unsigned ARMMCCodeEmitter::
1358getSORegRegOpValue(const MCInst &MI, unsigned OpIdx,
1359 SmallVectorImpl<MCFixup> &Fixups,
1360 const MCSubtargetInfo &STI) const {
1361 // Sub-operands are [reg, reg, imm]. The first register is Rm, the reg to be
1362 // shifted. The second is Rs, the amount to shift by, and the third specifies
1363 // the type of the shift.
1364 //
1365 // {3-0} = Rm.
1366 // {4} = 1
1367 // {6-5} = type
1368 // {11-8} = Rs
1369 // {7} = 0
1370
1371 const MCOperand &MO = MI.getOperand(OpIdx);
1372 const MCOperand &MO1 = MI.getOperand(OpIdx + 1);
1373 const MCOperand &MO2 = MI.getOperand(OpIdx + 2);
1374 ARM_AM::ShiftOpc SOpc = ARM_AM::getSORegShOp(MO2.getImm());
1375
1376 // Encode Rm.
1377 unsigned Binary = CTX.getRegisterInfo()->getEncodingValue(MO.getReg());
1378
1379 // Encode the shift opcode.
1380 unsigned SBits = 0;
1381 unsigned Rs = MO1.getReg();
1382 if (Rs) {
1383 // Set shift operand (bit[7:4]).
1384 // LSL - 0001
1385 // LSR - 0011
1386 // ASR - 0101
1387 // ROR - 0111
1388 switch (SOpc) {
1389 default: llvm_unreachable("Unknown shift opc!");
1390 case ARM_AM::lsl: SBits = 0x1; break;
1391 case ARM_AM::lsr: SBits = 0x3; break;
1392 case ARM_AM::asr: SBits = 0x5; break;
1393 case ARM_AM::ror: SBits = 0x7; break;
1394 }
1395 }
1396
1397 Binary |= SBits << 4;
1398
1399 // Encode the shift operation Rs.
1400 // Encode Rs bit[11:8].
1401 assert(ARM_AM::getSORegOffset(MO2.getImm()) == 0);
1402 return Binary | (CTX.getRegisterInfo()->getEncodingValue(Rs) << ARMII::RegRsShift);
1403}

1405unsigned ARMMCCodeEmitter::
1406getSORegImmOpValue(const MCInst &MI, unsigned OpIdx,
1407 SmallVectorImpl<MCFixup> &Fixups,
1408 const MCSubtargetInfo &STI) const {
1409 // Sub-operands are [reg, imm]. The first register is Rm, the reg to be
1410 // shifted. The second is the amount to shift by.
1411 //
1412 // {3-0} = Rm.
1413 // {4} = 0
1414 // {6-5} = type
1415 // {11-7} = imm
1416
1417 const MCOperand &MO = MI.getOperand(OpIdx);
1418 const MCOperand &MO1 = MI.getOperand(OpIdx + 1);
1419 ARM_AM::ShiftOpc SOpc = ARM_AM::getSORegShOp(MO1.getImm());
1420
1421 // Encode Rm.
1422 unsigned Binary = CTX.getRegisterInfo()->getEncodingValue(MO.getReg());
1423
1424 // Encode the shift opcode.
1425 unsigned SBits = 0;
1426
1427 // Set shift operand (bit[6:4]).
1428 // LSL - 000
1429 // LSR - 010
1430 // ASR - 100
1431 // ROR - 110
1432 // RRX - 110 and bit[11:8] clear.
1433 switch (SOpc) {
1434 default: llvm_unreachable("Unknown shift opc!");
1435 case ARM_AM::lsl: SBits = 0x0; break;
1436 case ARM_AM::lsr: SBits = 0x2; break;
1437 case ARM_AM::asr: SBits = 0x4; break;
1438 case ARM_AM::ror: SBits = 0x6; break;
1439 case ARM_AM::rrx:
1440 Binary |= 0x60;
1441 return Binary;
1442 }
1443
1444 // Encode shift_imm bit[11:7].
1445 Binary |= SBits << 4;
1446 unsigned Offset = ARM_AM::getSORegOffset(MO1.getImm());
1447 assert(Offset < 32 && "Offset must be in range 0-31!");
1448 return Binary | (Offset << 7);
1449}

1553unsigned ARMMCCodeEmitter::
1554getBitfieldInvertedMaskOpValue(const MCInst &MI, unsigned Op,
1555 SmallVectorImpl<MCFixup> &Fixups,
1556 const MCSubtargetInfo &STI) const {
1557 // 10 bits. lower 5 bits are the lsb of the mask, high five bits are the
1558 // msb of the mask.
1559 const MCOperand &MO = MI.getOperand(Op);
1560 uint32_t v = ~MO.getImm();
1561 uint32_t lsb = countTrailingZeros(v);
1562 uint32_t msb = (32 - countLeadingZeros (v)) - 1;
1563 assert(v != 0 && lsb < 32 && msb < 32 && "Illegal bitfield mask!");
1564 return lsb | (msb << 5);
1565}

1567unsigned ARMMCCodeEmitter::
1568getRegisterListOpValue(const MCInst &MI, unsigned Op,
1569 SmallVectorImpl<MCFixup> &Fixups,
1570 const MCSubtargetInfo &STI) const {
1571 // VLDM/VSTM/VSCCLRM:
1572 // {12-8} = Vd
1573 // {7-0} = Number of registers
1574 //
1575 // LDM/STM:
1576 // {15-0} = Bitfield of GPRs.
1577 unsigned Reg = MI.getOperand(Op).getReg();
1578 bool SPRRegs = ARMMCRegisterClasses[ARM::SPRRegClassID].contains(Reg);
1579 bool DPRRegs = ARMMCRegisterClasses[ARM::DPRRegClassID].contains(Reg);
1580 bool CLRMRegs = MI.getOpcode() == ARM::t2CLRM;
1581
1582 unsigned Binary = 0;
1583
1584 if (SPRRegs || DPRRegs) {
1585 // VLDM/VSTM/VSCCLRM
1586 unsigned RegNo = CTX.getRegisterInfo()->getEncodingValue(Reg);
1587 unsigned NumRegs = (MI.getNumOperands() - Op) & 0xff;
1588 Binary |= (RegNo & 0x1f) << 8;
1589
1590 // Ignore VPR
1591 if (MI.getOpcode() == ARM::VSCCLRMD || MI.getOpcode() == ARM::VSCCLRMS)
1592 --NumRegs;
1593 if (SPRRegs)
1594 Binary |= NumRegs;
1595 else
1596 Binary |= NumRegs * 2;
1597 } else {
1598 const MCRegisterInfo &MRI = *CTX.getRegisterInfo();
1599 if (!CLRMRegs) {
1600 assert(std::is_sorted(MI.begin() + Op, MI.end(),
1601 [&](const MCOperand &LHS, const MCOperand &RHS) {
1602 return MRI.getEncodingValue(LHS.getReg()) <
1603 MRI.getEncodingValue(RHS.getReg());
1604 }));
1605 }
1606
1607 for (unsigned I = Op, E = MI.getNumOperands(); I < E; ++I) {
1608 unsigned RegNo;
1609 if (CLRMRegs && MI.getOperand(I).getReg() == ARM::APSR) {
1610 RegNo = 15;
1611 } else {
1612 RegNo = MRI.getEncodingValue(MI.getOperand(I).getReg());
1613 }
1614 Binary |= 1 << RegNo;
1615 }
1616 }
1617
1618 return Binary;
1619}

1732void ARMMCCodeEmitter::
1733encodeInstruction(const MCInst &MI, raw_ostream &OS,
1734 SmallVectorImpl<MCFixup> &Fixups,
1735 const MCSubtargetInfo &STI) const {
1736 // Pseudo instructions don't get encoded.
1737 const MCInstrDesc &Desc = MCII.get(MI.getOpcode());
1738 uint64_t TSFlags = Desc.TSFlags;
1739 if ((TSFlags & ARMII::FormMask) == ARMII::Pseudo)
1740 return;
1741
1742 int Size;
1743 if (Desc.getSize() == 2 || Desc.getSize() == 4)
1744 Size = Desc.getSize();
1745 else
1746 llvm_unreachable("Unexpected instruction size!");
1747
1748 uint32_t Binary = getBinaryCodeForInstr(MI, Fixups, STI);
1749 // Thumb 32-bit wide instructions need to emit the high order halfword
1750 // first.
1751 if (isThumb(STI) && Size == 4) {
1752 EmitConstant(Binary >> 16, 2, OS);
1753 EmitConstant(Binary & 0xffff, 2, OS);
1754 } else
1755 EmitConstant(Binary, Size, OS);
1756 ++MCNumEmitted; // Keep track of the # of mi's emitted.
1757}

1770uint32_t
1771ARMMCCodeEmitter::getBFAfterTargetOpValue(const MCInst &MI, unsigned OpIdx,
1772 SmallVectorImpl<MCFixup> &Fixups,
1773 const MCSubtargetInfo &STI) const {
1774 const MCOperand MO = MI.getOperand(OpIdx);
1775 const MCOperand BranchMO = MI.getOperand(0);
1776
1777 if (MO.isExpr()) {
1778 assert(BranchMO.isExpr());
1779 const MCExpr *DiffExpr = MCBinaryExpr::createSub(
1780 MO.getExpr(), BranchMO.getExpr(), CTX);
1781 MCFixupKind Kind = MCFixupKind(ARM::fixup_bfcsel_else_target);
1782 Fixups.push_back(llvm::MCFixup::create(0, DiffExpr, Kind, MI.getLoc()));
1783 return 0;
1784 }
1785
1786 assert(MO.isImm() && BranchMO.isImm());
1787 int Diff = MO.getImm() - BranchMO.getImm();
1788 assert(Diff == 4 || Diff == 2);
1789
1790 return Diff == 4;
1791}
