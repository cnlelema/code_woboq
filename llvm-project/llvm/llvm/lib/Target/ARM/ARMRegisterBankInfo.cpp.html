31enum PartialMappingIdx {
32 PMI_GPR,
33 PMI_SPR,
34 PMI_DPR,
35 PMI_Min = PMI_GPR,
36};
37
38RegisterBankInfo::PartialMapping PartMappings[]{
39 // GPR Partial Mapping
40 {0, 32, GPRRegBank},
41 // SPR Partial Mapping
42 {0, 32, FPRRegBank},
43 // DPR Partial Mapping
44 {0, 64, FPRRegBank},
45};
46
47#ifndef NDEBUG
48static bool checkPartMapping(const RegisterBankInfo::PartialMapping &PM,
49 unsigned Start, unsigned Length,
50 unsigned RegBankID) {
51 return PM.StartIdx == Start && PM.Length == Length &&
52 PM.RegBank->getID() == RegBankID;
53}
54
55static void checkPartialMappings() {
56 assert(
57 checkPartMapping(PartMappings[PMI_GPR - PMI_Min], 0, 32, GPRRegBankID) &&
58 "Wrong mapping for GPR");
59 assert(
60 checkPartMapping(PartMappings[PMI_SPR - PMI_Min], 0, 32, FPRRegBankID) &&
61 "Wrong mapping for SPR");
62 assert(
63 checkPartMapping(PartMappings[PMI_DPR - PMI_Min], 0, 64, FPRRegBankID) &&
64 "Wrong mapping for DPR");
65}
66#endif

68enum ValueMappingIdx {
69 InvalidIdx = 0,
70 GPR3OpsIdx = 1,
71 SPR3OpsIdx = 4,
72 DPR3OpsIdx = 7,
73};
74
75RegisterBankInfo::ValueMapping ValueMappings[] = {
76 // invalid
77 {nullptr, 0},
78 // 3 ops in GPRs
79 {&PartMappings[PMI_GPR - PMI_Min], 1},
80 {&PartMappings[PMI_GPR - PMI_Min], 1},
81 {&PartMappings[PMI_GPR - PMI_Min], 1},
82 // 3 ops in SPRs
83 {&PartMappings[PMI_SPR - PMI_Min], 1},
84 {&PartMappings[PMI_SPR - PMI_Min], 1},
85 {&PartMappings[PMI_SPR - PMI_Min], 1},
86 // 3 ops in DPRs
87 {&PartMappings[PMI_DPR - PMI_Min], 1},
88 {&PartMappings[PMI_DPR - PMI_Min], 1},
89 {&PartMappings[PMI_DPR - PMI_Min], 1}};
90
91#ifndef NDEBUG
92static bool checkValueMapping(const RegisterBankInfo::ValueMapping &VM,
93 RegisterBankInfo::PartialMapping *BreakDown) {
94 return VM.NumBreakDowns == 1 && VM.BreakDown == BreakDown;
95}
96
97static void checkValueMappings() {
98 assert(checkValueMapping(ValueMappings[GPR3OpsIdx],
99 &PartMappings[PMI_GPR - PMI_Min]) &&
100 "Wrong value mapping for 3 GPR ops instruction");
101 assert(checkValueMapping(ValueMappings[GPR3OpsIdx + 1],
102 &PartMappings[PMI_GPR - PMI_Min]) &&
103 "Wrong value mapping for 3 GPR ops instruction");
104 assert(checkValueMapping(ValueMappings[GPR3OpsIdx + 2],
105 &PartMappings[PMI_GPR - PMI_Min]) &&
106 "Wrong value mapping for 3 GPR ops instruction");
107
108 assert(checkValueMapping(ValueMappings[SPR3OpsIdx],
109 &PartMappings[PMI_SPR - PMI_Min]) &&
110 "Wrong value mapping for 3 SPR ops instruction");
111 assert(checkValueMapping(ValueMappings[SPR3OpsIdx + 1],
112 &PartMappings[PMI_SPR - PMI_Min]) &&
113 "Wrong value mapping for 3 SPR ops instruction");
114 assert(checkValueMapping(ValueMappings[SPR3OpsIdx + 2],
115 &PartMappings[PMI_SPR - PMI_Min]) &&
116 "Wrong value mapping for 3 SPR ops instruction");
117
118 assert(checkValueMapping(ValueMappings[DPR3OpsIdx],
119 &PartMappings[PMI_DPR - PMI_Min]) &&
120 "Wrong value mapping for 3 DPR ops instruction");
121 assert(checkValueMapping(ValueMappings[DPR3OpsIdx + 1],
122 &PartMappings[PMI_DPR - PMI_Min]) &&
123 "Wrong value mapping for 3 DPR ops instruction");
124 assert(checkValueMapping(ValueMappings[DPR3OpsIdx + 2],
125 &PartMappings[PMI_DPR - PMI_Min]) &&
126 "Wrong value mapping for 3 DPR ops instruction");
127}
128#endif

132ARMRegisterBankInfo::ARMRegisterBankInfo(const TargetRegisterInfo &TRI)
133 : ARMGenRegisterBankInfo() {
134 static bool AlreadyInit = false;
135 // We have only one set of register banks, whatever the subtarget
136 // is. Therefore, the initialization of the RegBanks table should be
137 // done only once. Indeed the table of all register banks
138 // (ARM::RegBanks) is unique in the compiler. At some point, it
139 // will get tablegen'ed and the whole constructor becomes empty.
140 if (AlreadyInit)
141 return;
142 AlreadyInit = true;
143
144 const RegisterBank &RBGPR = getRegBank(ARM::GPRRegBankID);
145 (void)RBGPR;
146 assert(&ARM::GPRRegBank == &RBGPR && "The order in RegBanks is messed up");
147
148 // Initialize the GPR bank.
149 assert(RBGPR.covers(*TRI.getRegClass(ARM::GPRRegClassID)) &&
150 "Subclass not added?");
151 assert(RBGPR.covers(*TRI.getRegClass(ARM::GPRwithAPSRRegClassID)) &&
152 "Subclass not added?");
153 assert(RBGPR.covers(*TRI.getRegClass(ARM::GPRnopcRegClassID)) &&
154 "Subclass not added?");
155 assert(RBGPR.covers(*TRI.getRegClass(ARM::rGPRRegClassID)) &&
156 "Subclass not added?");
157 assert(RBGPR.covers(*TRI.getRegClass(ARM::tGPRRegClassID)) &&
158 "Subclass not added?");
159 assert(RBGPR.covers(*TRI.getRegClass(ARM::tcGPRRegClassID)) &&
160 "Subclass not added?");
161 assert(RBGPR.covers(*TRI.getRegClass(ARM::tGPR_and_tcGPRRegClassID)) &&
162 "Subclass not added?");
163 assert(RBGPR.getSize() == 32 && "GPRs should hold up to 32-bit");
164
165#ifndef NDEBUG
166 ARM::checkPartialMappings();
167 ARM::checkValueMappings();
168#endif
169}

171const RegisterBank &ARMRegisterBankInfo::getRegBankFromRegClass(
172 const TargetRegisterClass &RC) const {
173 using namespace ARM;
174
175 switch (RC.getID()) {
176 case GPRRegClassID:
177 case GPRwithAPSRRegClassID:
178 case GPRnopcRegClassID:
179 case rGPRRegClassID:
180 case GPRspRegClassID:
181 case tGPR_and_tcGPRRegClassID:
182 case tcGPRRegClassID:
183 case tGPRRegClassID:
184 case tGPREvenRegClassID:
185 case tGPROddRegClassID:
186 case tGPR_and_tGPREvenRegClassID:
187 case tGPR_and_tGPROddRegClassID:
188 case tGPREven_and_tcGPRRegClassID:
189 case tGPREven_and_tGPR_and_tcGPRRegClassID:
190 case tGPROdd_and_tcGPRRegClassID:
191 return getRegBank(ARM::GPRRegBankID);
192 case HPRRegClassID:
193 case SPR_8RegClassID:
194 case SPRRegClassID:
195 case DPR_8RegClassID:
196 case DPRRegClassID:
197 case QPRRegClassID:
198 return getRegBank(ARM::FPRRegBankID);
199 default:
200 llvm_unreachable("Unsupported register kind");
201 }
202
203 llvm_unreachable("Switch should handle all register classes");
204}

206const RegisterBankInfo::InstructionMapping &
207ARMRegisterBankInfo::getInstrMapping(const MachineInstr &MI) const {
208 auto Opc = MI.getOpcode();
209
210 // Try the default logic for non-generic instructions that are either copies
211 // or already have some operands assigned to banks.
212 if (!isPreISelGenericOpcode(Opc) || Opc == TargetOpcode::G_PHI) {
213 const InstructionMapping &Mapping = getInstrMappingImpl(MI);
214 if (Mapping.isValid())
215 return Mapping;
216 }
217
218 using namespace TargetOpcode;
219
220 const MachineFunction &MF = *MI.getParent()->getParent();
221 const MachineRegisterInfo &MRI = MF.getRegInfo();
222 unsigned NumOperands = MI.getNumOperands();
223 const ValueMapping *OperandsMapping = &ARM::ValueMappings[ARM::GPR3OpsIdx];
224
225 switch (Opc) {
226 case G_ADD:
227 case G_SUB:
228 case G_MUL:
229 case G_AND:
230 case G_OR:
231 case G_XOR:
232 case G_LSHR:
233 case G_ASHR:
234 case G_SHL:
235 case G_SDIV:
236 case G_UDIV:
237 case G_SEXT:
238 case G_ZEXT:
239 case G_ANYEXT:
240 case G_GEP:
241 case G_INTTOPTR:
242 case G_PTRTOINT:
243 case G_CTLZ:
244 // FIXME: We're abusing the fact that everything lives in a GPR for now; in
245 // the real world we would use different mappings.
246 OperandsMapping = &ARM::ValueMappings[ARM::GPR3OpsIdx];
247 break;
248 case G_TRUNC: {
249 // In some cases we may end up with a G_TRUNC from a 64-bit value to a
250 // 32-bit value. This isn't a real floating point trunc (that would be a
251 // G_FPTRUNC). Instead it is an integer trunc in disguise, which can appear
252 // because the legalizer doesn't distinguish between integer and floating
253 // point values so it may leave some 64-bit integers un-narrowed. Until we
254 // have a more principled solution that doesn't let such things sneak all
255 // the way to this point, just map the source to a DPR and the destination
256 // to a GPR.
257 LLT LargeTy = MRI.getType(MI.getOperand(1).getReg());
258 OperandsMapping =
259 LargeTy.getSizeInBits() <= 32
260 ? &ARM::ValueMappings[ARM::GPR3OpsIdx]
261 : getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx],
262 &ARM::ValueMappings[ARM::DPR3OpsIdx]});
263 break;
264 }
265 case G_LOAD:
266 case G_STORE: {
267 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
268 OperandsMapping =
269 Ty.getSizeInBits() == 64
270 ? getOperandsMapping({&ARM::ValueMappings[ARM::DPR3OpsIdx],
271 &ARM::ValueMappings[ARM::GPR3OpsIdx]})
272 : &ARM::ValueMappings[ARM::GPR3OpsIdx];
273 break;
274 }
275 case G_FADD:
276 case G_FSUB:
277 case G_FMUL:
278 case G_FDIV:
279 case G_FNEG: {
280 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
281 OperandsMapping =Ty.getSizeInBits() == 64
282 ? &ARM::ValueMappings[ARM::DPR3OpsIdx]
283 : &ARM::ValueMappings[ARM::SPR3OpsIdx];
284 break;
285 }
286 case G_FMA: {
287 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
288 OperandsMapping =
289 Ty.getSizeInBits() == 64
290 ? getOperandsMapping({&ARM::ValueMappings[ARM::DPR3OpsIdx],
291 &ARM::ValueMappings[ARM::DPR3OpsIdx],
292 &ARM::ValueMappings[ARM::DPR3OpsIdx],
293 &ARM::ValueMappings[ARM::DPR3OpsIdx]})
294 : getOperandsMapping({&ARM::ValueMappings[ARM::SPR3OpsIdx],
295 &ARM::ValueMappings[ARM::SPR3OpsIdx],
296 &ARM::ValueMappings[ARM::SPR3OpsIdx],
297 &ARM::ValueMappings[ARM::SPR3OpsIdx]});
298 break;
299 }
300 case G_FPEXT: {
301 LLT ToTy = MRI.getType(MI.getOperand(0).getReg());
302 LLT FromTy = MRI.getType(MI.getOperand(1).getReg());
303 if (ToTy.getSizeInBits() == 64 && FromTy.getSizeInBits() == 32)
304 OperandsMapping =
305 getOperandsMapping({&ARM::ValueMappings[ARM::DPR3OpsIdx],
306 &ARM::ValueMappings[ARM::SPR3OpsIdx]});
307 break;
308 }
309 case G_FPTRUNC: {
310 LLT ToTy = MRI.getType(MI.getOperand(0).getReg());
311 LLT FromTy = MRI.getType(MI.getOperand(1).getReg());
312 if (ToTy.getSizeInBits() == 32 && FromTy.getSizeInBits() == 64)
313 OperandsMapping =
314 getOperandsMapping({&ARM::ValueMappings[ARM::SPR3OpsIdx],
315 &ARM::ValueMappings[ARM::DPR3OpsIdx]});
316 break;
317 }
318 case G_FPTOSI:
319 case G_FPTOUI: {
320 LLT ToTy = MRI.getType(MI.getOperand(0).getReg());
321 LLT FromTy = MRI.getType(MI.getOperand(1).getReg());
322 if ((FromTy.getSizeInBits() == 32 || FromTy.getSizeInBits() == 64) &&
323 ToTy.getSizeInBits() == 32)
324 OperandsMapping =
325 FromTy.getSizeInBits() == 64
326 ? getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx],
327 &ARM::ValueMappings[ARM::DPR3OpsIdx]})
328 : getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx],
329 &ARM::ValueMappings[ARM::SPR3OpsIdx]});
330 break;
331 }
332 case G_SITOFP:
333 case G_UITOFP: {
334 LLT ToTy = MRI.getType(MI.getOperand(0).getReg());
335 LLT FromTy = MRI.getType(MI.getOperand(1).getReg());
336 if (FromTy.getSizeInBits() == 32 &&
337 (ToTy.getSizeInBits() == 32 || ToTy.getSizeInBits() == 64))
338 OperandsMapping =
339 ToTy.getSizeInBits() == 64
340 ? getOperandsMapping({&ARM::ValueMappings[ARM::DPR3OpsIdx],
341 &ARM::ValueMappings[ARM::GPR3OpsIdx]})
342 : getOperandsMapping({&ARM::ValueMappings[ARM::SPR3OpsIdx],
343 &ARM::ValueMappings[ARM::GPR3OpsIdx]});
344 break;
345 }
346 case G_FCONSTANT: {
347 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
348 OperandsMapping = getOperandsMapping(
349 {Ty.getSizeInBits() == 64 ? &ARM::ValueMappings[ARM::DPR3OpsIdx]
350 : &ARM::ValueMappings[ARM::SPR3OpsIdx],
351 nullptr});
352 break;
353 }
354 case G_CONSTANT:
355 case G_FRAME_INDEX:
356 case G_GLOBAL_VALUE:
357 OperandsMapping =
358 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx], nullptr});
359 break;
360 case G_SELECT: {
361 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
362 (void)Ty;
363 LLT Ty2 = MRI.getType(MI.getOperand(1).getReg());
364 (void)Ty2;
365 assert(Ty.getSizeInBits() == 32 && "Unsupported size for G_SELECT");
366 assert(Ty2.getSizeInBits() == 1 && "Unsupported size for G_SELECT");
367 OperandsMapping =
368 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx],
369 &ARM::ValueMappings[ARM::GPR3OpsIdx],
370 &ARM::ValueMappings[ARM::GPR3OpsIdx],
371 &ARM::ValueMappings[ARM::GPR3OpsIdx]});
372 break;
373 }
374 case G_ICMP: {
375 LLT Ty2 = MRI.getType(MI.getOperand(2).getReg());
376 (void)Ty2;
377 assert(Ty2.getSizeInBits() == 32 && "Unsupported size for G_ICMP");
378 OperandsMapping =
379 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx], nullptr,
380 &ARM::ValueMappings[ARM::GPR3OpsIdx],
381 &ARM::ValueMappings[ARM::GPR3OpsIdx]});
382 break;
383 }
384 case G_FCMP: {
385 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
386 (void)Ty;
387 LLT Ty1 = MRI.getType(MI.getOperand(2).getReg());
388 LLT Ty2 = MRI.getType(MI.getOperand(3).getReg());
389 (void)Ty2;
390 assert(Ty.getSizeInBits() == 1 && "Unsupported size for G_FCMP");
391 assert(Ty1.getSizeInBits() == Ty2.getSizeInBits() &&
392 "Mismatched operand sizes for G_FCMP");
393
394 unsigned Size = Ty1.getSizeInBits();
395 assert((Size == 32 || Size == 64) && "Unsupported size for G_FCMP");
396
397 auto FPRValueMapping = Size == 32 ? &ARM::ValueMappings[ARM::SPR3OpsIdx]
398 : &ARM::ValueMappings[ARM::DPR3OpsIdx];
399 OperandsMapping =
400 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx], nullptr,
401 FPRValueMapping, FPRValueMapping});
402 break;
403 }
404 case G_MERGE_VALUES: {
405 // We only support G_MERGE_VALUES for creating a double precision floating
406 // point value out of two GPRs.
407 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
408 LLT Ty1 = MRI.getType(MI.getOperand(1).getReg());
409 LLT Ty2 = MRI.getType(MI.getOperand(2).getReg());
410 if (Ty.getSizeInBits() != 64 || Ty1.getSizeInBits() != 32 ||
411 Ty2.getSizeInBits() != 32)
412 return getInvalidInstructionMapping();
413 OperandsMapping =
414 getOperandsMapping({&ARM::ValueMappings[ARM::DPR3OpsIdx],
415 &ARM::ValueMappings[ARM::GPR3OpsIdx],
416 &ARM::ValueMappings[ARM::GPR3OpsIdx]});
417 break;
418 }
419 case G_UNMERGE_VALUES: {
420 // We only support G_UNMERGE_VALUES for splitting a double precision
421 // floating point value into two GPRs.
422 LLT Ty = MRI.getType(MI.getOperand(0).getReg());
423 LLT Ty1 = MRI.getType(MI.getOperand(1).getReg());
424 LLT Ty2 = MRI.getType(MI.getOperand(2).getReg());
425 if (Ty.getSizeInBits() != 32 || Ty1.getSizeInBits() != 32 ||
426 Ty2.getSizeInBits() != 64)
427 return getInvalidInstructionMapping();
428 OperandsMapping =
429 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx],
430 &ARM::ValueMappings[ARM::GPR3OpsIdx],
431 &ARM::ValueMappings[ARM::DPR3OpsIdx]});
432 break;
433 }
434 case G_BR:
435 OperandsMapping = getOperandsMapping({nullptr});
436 break;
437 case G_BRCOND:
438 OperandsMapping =
439 getOperandsMapping({&ARM::ValueMappings[ARM::GPR3OpsIdx], nullptr});
440 break;
441 case DBG_VALUE: {
442 SmallVector<const ValueMapping *, 4> OperandBanks(NumOperands);
443 const MachineOperand &MaybeReg = MI.getOperand(0);
444 if (MaybeReg.isReg() && MaybeReg.getReg()) {
445 unsigned Size = MRI.getType(MaybeReg.getReg()).getSizeInBits();
446 if (Size > 32 && Size != 64)
447 return getInvalidInstructionMapping();
448 OperandBanks[0] = Size == 64 ? &ARM::ValueMappings[ARM::DPR3OpsIdx]
449 : &ARM::ValueMappings[ARM::GPR3OpsIdx];
450 }
451 OperandsMapping = getOperandsMapping(OperandBanks);
452 break;
453 }
454 default:
455 return getInvalidInstructionMapping();
456 }
457
458#ifndef NDEBUG
459 for (unsigned i = 0; i < NumOperands; i++) {
460 for (const auto &Mapping : OperandsMapping[i]) {
461 assert(
462 (Mapping.RegBank->getID() != ARM::FPRRegBankID ||
463 MF.getSubtarget<ARMSubtarget>().hasVFP2Base()) &&
464 "Trying to use floating point register bank on target without vfp");
465 }
466 }
467#endif
468
469 return getInstructionMapping(DefaultMappingID, /*Cost=*/1, OperandsMapping,
470 NumOperands);
471}
