45 /// Build the equivalent inputs of a REG_SEQUENCE for the given \p MI
46 /// and \p DefIdx.
47 /// \p [out] InputRegs of the equivalent REG_SEQUENCE. Each element of
48 /// the list is modeled as <Reg:SubReg, SubIdx>.
49 /// E.g., REG_SEQUENCE %1:sub1, sub0, %2, sub1 would produce
50 /// two elements:
51 /// - %1:sub1, sub0
52 /// - %2<:0>, sub1
53 ///

55 /// with the pair \p MI, \p DefIdx. False otherwise.

249 /// shouldScheduleLoadsNear - This is a used by the pre-regalloc scheduler to
250 /// determine (in conjunction with areLoadsFromSameBasePtr) if two loads
251 /// should be scheduled togther. On some targets if two loads are loading from
252 /// addresses in the same cache line, it's better if they are scheduled
253 /// together. This function takes two integers that represent the load offsets

255 /// to schedule the two loads together. "NumLoads" is the number of loads that

402 /// Modeling special VFP / NEON fp MLA / MLS hazards.
403
404 /// MLxEntryMap - Map fp MLA / MLS to the corresponding entry in the internal
405 /// MLx table.
406 DenseMap<unsigned, unsigned> MLxEntryMap;
407
408 /// MLxHazardOpcodes - Set of add / sub and multiply opcodes that would cause
409 /// stalls when scheduled together with fp MLA / MLS opcodes.
410 SmallSet<unsigned, 16> MLxHazardOpcodes;

419 /// isFpMLxInstruction - This version also returns the multiply opcode and the

423 unsigned &AddSubOpc, bool &NegAcc,
424 bool &HasLane) const;

426 /// canCauseFpMLxStall - Return true if an instruction of the specified opcode

437 /// Returns predicate register associated with the given frame instruction.
438 unsigned getFramePred(const MachineInstr &MI) const {
439 assert(isFrameInstr(MI));
440 // Operands of ADJCALLSTACKDOWN/ADJCALLSTACKUP:
441 // - argument declared in the pattern:
442 // 0 - frame size
443 // 1 - arg of CALLSEQ_START/CALLSEQ_END
444 // 2 - predicate code (like ARMCC::AL)
445 // - added by predOps:
446 // 3 - predicate reg
447 return MI.getOperand(3).getReg();
448 }

451/// Get the operands corresponding to the given \p Pred value. By default, the
452/// predicate register is assumed to be 0 (no register), but you can pass in a
453/// \p PredReg if that is not the case.
454static inline std::array<MachineOperand, 2> predOps(ARMCC::CondCodes Pred,
455 unsigned PredReg = 0) {
456 return {{MachineOperand::CreateImm(static_cast<int64_t>(Pred)),
457 MachineOperand::CreateReg(PredReg, false)}};
458}
459
460/// Get the operand corresponding to the conditional code result. By default,
461/// this is 0 (no register).
462static inline MachineOperand condCodeOp(unsigned CCReg = 0) {
463 return MachineOperand::CreateReg(CCReg, false);
464}

466/// Get the operand corresponding to the conditional code result for Thumb1.
467/// This operand will always refer to CPSR and it will have the Define flag set.
468/// You can optionally set the Dead flag by means of \p isDead.
469static inline MachineOperand t1CondCodeOp(bool isDead = false) {
470 return MachineOperand::CreateReg(ARM::CPSR,
471 /*Define*/ true, /*Implicit*/ false,
472 /*Kill*/ false, isDead);
473}

496static inline bool isPopOpcode(int Opc) {
497 return Opc == ARM::tPOP_RET || Opc == ARM::LDMIA_RET ||
498 Opc == ARM::t2LDMIA_RET || Opc == ARM::tPOP || Opc == ARM::LDMIA_UPD ||
499 Opc == ARM::t2LDMIA_UPD || Opc == ARM::VLDMDIA_UPD;
500}

502static inline bool isPushOpcode(int Opc) {
