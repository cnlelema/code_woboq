43bool ARMTTIImpl::areInlineCompatible(const Function *Caller,
44 const Function *Callee) const {
45 const TargetMachine &TM = getTLI()->getTargetMachine();
46 const FeatureBitset &CallerBits =
47 TM.getSubtargetImpl(*Caller)->getFeatureBits();
48 const FeatureBitset &CalleeBits =
49 TM.getSubtargetImpl(*Callee)->getFeatureBits();
50
51 // To inline a callee, all features not in the whitelist must match exactly.
52 bool MatchExact = (CallerBits & ~InlineFeatureWhitelist) ==
53 (CalleeBits & ~InlineFeatureWhitelist);
54 // For features in the whitelist, the callee's features must be a subset of
55 // the callers'.
56 bool MatchSubset = ((CallerBits & CalleeBits) & InlineFeatureWhitelist) ==
57 (CalleeBits & InlineFeatureWhitelist);
58 return MatchExact && MatchSubset;
59}

61int ARMTTIImpl::getIntImmCost(const APInt &Imm, Type *Ty) {
62 assert(Ty->isIntegerTy());
63
64 unsigned Bits = Ty->getPrimitiveSizeInBits();
65 if (Bits == 0 || Imm.getActiveBits() >= 64)
66 return 4;
67
68 int64_t SImmVal = Imm.getSExtValue();
69 uint64_t ZImmVal = Imm.getZExtValue();
70 if (!ST->isThumb()) {
71 if ((SImmVal >= 0 && SImmVal < 65536) ||
72 (ARM_AM::getSOImmVal(ZImmVal) != -1) ||
73 (ARM_AM::getSOImmVal(~ZImmVal) != -1))
74 return 1;
75 return ST->hasV6T2Ops() ? 2 : 3;
76 }
77 if (ST->isThumb2()) {
78 if ((SImmVal >= 0 && SImmVal < 65536) ||
79 (ARM_AM::getT2SOImmVal(ZImmVal) != -1) ||
80 (ARM_AM::getT2SOImmVal(~ZImmVal) != -1))
81 return 1;
82 return ST->hasV6T2Ops() ? 2 : 3;
83 }
84 // Thumb1, any i8 imm cost 1.
85 if (Bits == 8 || (SImmVal >= 0 && SImmVal < 256))
86 return 1;
87 if ((~SImmVal < 256) || ARM_AM::isThumbImmShiftedVal(ZImmVal))
88 return 2;
89 // Load from constantpool.
90 return 3;
91}

93// Constants smaller than 256 fit in the immediate field of
94// Thumb1 instructions so we return a zero cost and 1 otherwise.
95int ARMTTIImpl::getIntImmCodeSizeCost(unsigned Opcode, unsigned Idx,
96 const APInt &Imm, Type *Ty) {
97 if (Imm.isNonNegative() && Imm.getLimitedValue() < 256)
98 return 0;
99
100 return 1;
101}

103int ARMTTIImpl::getIntImmCost(unsigned Opcode, unsigned Idx, const APInt &Imm,
104 Type *Ty) {
105 // Division by a constant can be turned into multiplication, but only if we
106 // know it's constant. So it's not so much that the immediate is cheap (it's
107 // not), but that the alternative is worse.
108 // FIXME: this is probably unneeded with GlobalISel.
109 if ((Opcode == Instruction::SDiv || Opcode == Instruction::UDiv ||
110 Opcode == Instruction::SRem || Opcode == Instruction::URem) &&
111 Idx == 1)
112 return 0;
113
114 if (Opcode == Instruction::And) {
115 // UXTB/UXTH
116 if (Imm == 255 || Imm == 65535)
117 return 0;
118 // Conversion to BIC is free, and means we can use ~Imm instead.
119 return std::min(getIntImmCost(Imm, Ty), getIntImmCost(~Imm, Ty));
120 }
121
122 if (Opcode == Instruction::Add)
123 // Conversion to SUB is free, and means we can use -Imm instead.
124 return std::min(getIntImmCost(Imm, Ty), getIntImmCost(-Imm, Ty));
125
126 if (Opcode == Instruction::ICmp && Imm.isNegative() &&
127 Ty->getIntegerBitWidth() == 32) {
128 int64_t NegImm = -Imm.getSExtValue();
129 if (ST->isThumb2() && NegImm < 1<<12)
130 // icmp X, #-C -> cmn X, #C
131 return 0;
132 if (ST->isThumb() && NegImm < 1<<8)
133 // icmp X, #-C -> adds X, #C
134 return 0;
135 }
136
137 // xor a, -1 can always be folded to MVN
138 if (Opcode == Instruction::Xor && Imm.isAllOnesValue())
139 return 0;
140
141 return getIntImmCost(Imm, Ty);
142}

338int ARMTTIImpl::getVectorInstrCost(unsigned Opcode, Type *ValTy,
339 unsigned Index) {
340 // Penalize inserting into an D-subregister. We end up with a three times
341 // lower estimated throughput on swift.
342 if (ST->hasSlowLoadDSubregister() && Opcode == Instruction::InsertElement &&
343 ValTy->isVectorTy() && ValTy->getScalarSizeInBits() <= 32)
344 return 3;
345
346 if ((Opcode == Instruction::InsertElement ||
347 Opcode == Instruction::ExtractElement)) {
348 // Cross-class copies are expensive on many microarchitectures,
349 // so assume they are expensive by default.
350 if (ValTy->getVectorElementType()->isIntegerTy())
351 return 3;
352
353 // Even if it's not a cross class copy, this likely leads to mixing
354 // of NEON and VFP code and should be therefore penalized.
355 if (ValTy->isVectorTy() &&
356 ValTy->getScalarSizeInBits() <= 32)
357 return std::max(BaseT::getVectorInstrCost(Opcode, ValTy, Index), 2U);
358 }
359
360 return BaseT::getVectorInstrCost(Opcode, ValTy, Index);
361}

363int ARMTTIImpl::getCmpSelInstrCost(unsigned Opcode, Type *ValTy, Type *CondTy,
364 const Instruction *I) {
365 int ISD = TLI->InstructionOpcodeToISD(Opcode);
366 // On NEON a vector select gets lowered to vbsl.
367 if (ST->hasNEON() && ValTy->isVectorTy() && ISD == ISD::SELECT) {
368 // Lowering of some vector selects is currently far from perfect.
369 static const TypeConversionCostTblEntry NEONVectorSelectTbl[] = {
370 { ISD::SELECT, MVT::v4i1, MVT::v4i64, 4*4 + 1*2 + 1 },
371 { ISD::SELECT, MVT::v8i1, MVT::v8i64, 50 },
372 { ISD::SELECT, MVT::v16i1, MVT::v16i64, 100 }
373 };
374
375 EVT SelCondTy = TLI->getValueType(DL, CondTy);
376 EVT SelValTy = TLI->getValueType(DL, ValTy);
377 if (SelCondTy.isSimple() && SelValTy.isSimple()) {
378 if (const auto *Entry = ConvertCostTableLookup(NEONVectorSelectTbl, ISD,
379 SelCondTy.getSimpleVT(),
380 SelValTy.getSimpleVT()))
381 return Entry->Cost;
382 }
383
384 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, ValTy);
385 return LT.first;
386 }
387
388 return BaseT::getCmpSelInstrCost(Opcode, ValTy, CondTy, I);
389}

391int ARMTTIImpl::getAddressComputationCost(Type *Ty, ScalarEvolution *SE,
392 const SCEV *Ptr) {
393 // Address computations in vectorized code with non-consecutive addresses will
394 // likely result in more instructions compared to scalar code where the
395 // computation can more often be merged into the index mode. The resulting
396 // extra micro-ops can significantly decrease throughput.
397 unsigned NumVectorInstToHideOverhead = 10;
398 int MaxMergeDistance = 64;
399
400 if (Ty->isVectorTy() && SE &&
401 !BaseT::isConstantStridedAccessLessThan(SE, Ptr, MaxMergeDistance + 1))
402 return NumVectorInstToHideOverhead;
403
404 // In many cases the address computation is not merged into the instruction
405 // addressing mode.
406 return 1;
407}

409int ARMTTIImpl::getMemcpyCost(const Instruction *I) {
410 const MemCpyInst *MI = dyn_cast<MemCpyInst>(I);
411 assert(MI && "MemcpyInst expected");
412 ConstantInt *C = dyn_cast<ConstantInt>(MI->getLength());
413
414 // To model the cost of a library call, we assume 1 for the call, and
415 // 3 for the argument setup.
416 const unsigned LibCallCost = 4;
417
418 // If 'size' is not a constant, a library call will be generated.
419 if (!C)
420 return LibCallCost;
421
422 const unsigned Size = C->getValue().getZExtValue();
423 const unsigned DstAlign = MI->getDestAlignment();
424 const unsigned SrcAlign = MI->getSourceAlignment();
425 const Function *F = I->getParent()->getParent();
426 const unsigned Limit = TLI->getMaxStoresPerMemmove(F->hasMinSize());
427 std::vector<EVT> MemOps;
428
429 // MemOps will be poplulated with a list of data types that needs to be
430 // loaded and stored. That's why we multiply the number of elements by 2 to
431 // get the cost for this memcpy.
432 if (getTLI()->findOptimalMemOpLowering(
433 MemOps, Limit, Size, DstAlign, SrcAlign, false /*IsMemset*/,
434 false /*ZeroMemset*/, false /*MemcpyStrSrc*/, false /*AllowOverlap*/,
435 false /*MustPreserveCheriCaps*/,
436 MI->getDestAddressSpace(), MI->getSourceAddressSpace(),
437 F->getAttributes()))
438 return MemOps.size() * 2;
439
440 // If we can't find an optimal memop lowering, return the default cost
441 return LibCallCost;
442}

444int ARMTTIImpl::getShuffleCost(TTI::ShuffleKind Kind, Type *Tp, int Index,
445 Type *SubTp) {
446 if (Kind == TTI::SK_Broadcast) {
447 static const CostTblEntry NEONDupTbl[] = {
448 // VDUP handles these cases.
449 {ISD::VECTOR_SHUFFLE, MVT::v2i32, 1},
450 {ISD::VECTOR_SHUFFLE, MVT::v2f32, 1},
451 {ISD::VECTOR_SHUFFLE, MVT::v2i64, 1},
452 {ISD::VECTOR_SHUFFLE, MVT::v2f64, 1},
453 {ISD::VECTOR_SHUFFLE, MVT::v4i16, 1},
454 {ISD::VECTOR_SHUFFLE, MVT::v8i8, 1},
455
456 {ISD::VECTOR_SHUFFLE, MVT::v4i32, 1},
457 {ISD::VECTOR_SHUFFLE, MVT::v4f32, 1},
458 {ISD::VECTOR_SHUFFLE, MVT::v8i16, 1},
459 {ISD::VECTOR_SHUFFLE, MVT::v16i8, 1}};
460
461 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, Tp);
462
463 if (const auto *Entry = CostTableLookup(NEONDupTbl, ISD::VECTOR_SHUFFLE,
464 LT.second))
465 return LT.first * Entry->Cost;
466
467 return BaseT::getShuffleCost(Kind, Tp, Index, SubTp);
468 }
469 if (Kind == TTI::SK_Reverse) {
470 static const CostTblEntry NEONShuffleTbl[] = {
471 // Reverse shuffle cost one instruction if we are shuffling within a
472 // double word (vrev) or two if we shuffle a quad word (vrev, vext).
473 {ISD::VECTOR_SHUFFLE, MVT::v2i32, 1},
474 {ISD::VECTOR_SHUFFLE, MVT::v2f32, 1},
475 {ISD::VECTOR_SHUFFLE, MVT::v2i64, 1},
476 {ISD::VECTOR_SHUFFLE, MVT::v2f64, 1},
477 {ISD::VECTOR_SHUFFLE, MVT::v4i16, 1},
478 {ISD::VECTOR_SHUFFLE, MVT::v8i8, 1},
479
480 {ISD::VECTOR_SHUFFLE, MVT::v4i32, 2},
481 {ISD::VECTOR_SHUFFLE, MVT::v4f32, 2},
482 {ISD::VECTOR_SHUFFLE, MVT::v8i16, 2},
483 {ISD::VECTOR_SHUFFLE, MVT::v16i8, 2}};
484
485 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, Tp);
486
487 if (const auto *Entry = CostTableLookup(NEONShuffleTbl, ISD::VECTOR_SHUFFLE,
488 LT.second))
489 return LT.first * Entry->Cost;
490
491 return BaseT::getShuffleCost(Kind, Tp, Index, SubTp);
492 }
493 if (Kind == TTI::SK_Select) {
494 static const CostTblEntry NEONSelShuffleTbl[] = {
495 // Select shuffle cost table for ARM. Cost is the number of instructions
496 // required to create the shuffled vector.
497
498 {ISD::VECTOR_SHUFFLE, MVT::v2f32, 1},
499 {ISD::VECTOR_SHUFFLE, MVT::v2i64, 1},
500 {ISD::VECTOR_SHUFFLE, MVT::v2f64, 1},
501 {ISD::VECTOR_SHUFFLE, MVT::v2i32, 1},
502
503 {ISD::VECTOR_SHUFFLE, MVT::v4i32, 2},
504 {ISD::VECTOR_SHUFFLE, MVT::v4f32, 2},
505 {ISD::VECTOR_SHUFFLE, MVT::v4i16, 2},
506
507 {ISD::VECTOR_SHUFFLE, MVT::v8i16, 16},
508
509 {ISD::VECTOR_SHUFFLE, MVT::v16i8, 32}};
510
511 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, Tp);
512 if (const auto *Entry = CostTableLookup(NEONSelShuffleTbl,
513 ISD::VECTOR_SHUFFLE, LT.second))
514 return LT.first * Entry->Cost;
515 return BaseT::getShuffleCost(Kind, Tp, Index, SubTp);
516 }
517 return BaseT::getShuffleCost(Kind, Tp, Index, SubTp);
518}

520int ARMTTIImpl::getArithmeticInstrCost(
521 unsigned Opcode, Type *Ty, TTI::OperandValueKind Op1Info,
522 TTI::OperandValueKind Op2Info, TTI::OperandValueProperties Opd1PropInfo,
523 TTI::OperandValueProperties Opd2PropInfo,
524 ArrayRef<const Value *> Args) {
525 int ISDOpcode = TLI->InstructionOpcodeToISD(Opcode);
526 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, Ty);
527
528 const unsigned FunctionCallDivCost = 20;
529 const unsigned ReciprocalDivCost = 10;
530 static const CostTblEntry CostTbl[] = {
531 // Division.
532 // These costs are somewhat random. Choose a cost of 20 to indicate that
533 // vectorizing devision (added function call) is going to be very expensive.
534 // Double registers types.
535 { ISD::SDIV, MVT::v1i64, 1 * FunctionCallDivCost},
536 { ISD::UDIV, MVT::v1i64, 1 * FunctionCallDivCost},
537 { ISD::SREM, MVT::v1i64, 1 * FunctionCallDivCost},
538 { ISD::UREM, MVT::v1i64, 1 * FunctionCallDivCost},
539 { ISD::SDIV, MVT::v2i32, 2 * FunctionCallDivCost},
540 { ISD::UDIV, MVT::v2i32, 2 * FunctionCallDivCost},
541 { ISD::SREM, MVT::v2i32, 2 * FunctionCallDivCost},
542 { ISD::UREM, MVT::v2i32, 2 * FunctionCallDivCost},
543 { ISD::SDIV, MVT::v4i16, ReciprocalDivCost},
544 { ISD::UDIV, MVT::v4i16, ReciprocalDivCost},
545 { ISD::SREM, MVT::v4i16, 4 * FunctionCallDivCost},
546 { ISD::UREM, MVT::v4i16, 4 * FunctionCallDivCost},
547 { ISD::SDIV, MVT::v8i8, ReciprocalDivCost},
548 { ISD::UDIV, MVT::v8i8, ReciprocalDivCost},
549 { ISD::SREM, MVT::v8i8, 8 * FunctionCallDivCost},
550 { ISD::UREM, MVT::v8i8, 8 * FunctionCallDivCost},
551 // Quad register types.
552 { ISD::SDIV, MVT::v2i64, 2 * FunctionCallDivCost},
553 { ISD::UDIV, MVT::v2i64, 2 * FunctionCallDivCost},
554 { ISD::SREM, MVT::v2i64, 2 * FunctionCallDivCost},
555 { ISD::UREM, MVT::v2i64, 2 * FunctionCallDivCost},
556 { ISD::SDIV, MVT::v4i32, 4 * FunctionCallDivCost},
557 { ISD::UDIV, MVT::v4i32, 4 * FunctionCallDivCost},
558 { ISD::SREM, MVT::v4i32, 4 * FunctionCallDivCost},
559 { ISD::UREM, MVT::v4i32, 4 * FunctionCallDivCost},
560 { ISD::SDIV, MVT::v8i16, 8 * FunctionCallDivCost},
561 { ISD::UDIV, MVT::v8i16, 8 * FunctionCallDivCost},
562 { ISD::SREM, MVT::v8i16, 8 * FunctionCallDivCost},
563 { ISD::UREM, MVT::v8i16, 8 * FunctionCallDivCost},
564 { ISD::SDIV, MVT::v16i8, 16 * FunctionCallDivCost},
565 { ISD::UDIV, MVT::v16i8, 16 * FunctionCallDivCost},
566 { ISD::SREM, MVT::v16i8, 16 * FunctionCallDivCost},
567 { ISD::UREM, MVT::v16i8, 16 * FunctionCallDivCost},
568 // Multiplication.
569 };
570
571 if (ST->hasNEON())
572 if (const auto *Entry = CostTableLookup(CostTbl, ISDOpcode, LT.second))
573 return LT.first * Entry->Cost;
574
575 int Cost = BaseT::getArithmeticInstrCost(Opcode, Ty, Op1Info, Op2Info,
576 Opd1PropInfo, Opd2PropInfo);
577
578 // This is somewhat of a hack. The problem that we are facing is that SROA
579 // creates a sequence of shift, and, or instructions to construct values.
580 // These sequences are recognized by the ISel and have zero-cost. Not so for
581 // the vectorized code. Because we have support for v2i64 but not i64 those
582 // sequences look particularly beneficial to vectorize.
583 // To work around this we increase the cost of v2i64 operations to make them
584 // seem less beneficial.
585 if (LT.second == MVT::v2i64 &&
586 Op2Info == TargetTransformInfo::OK_UniformConstantValue)
587 Cost += 4;
588
589 return Cost;
590}

592int ARMTTIImpl::getMemoryOpCost(unsigned Opcode, Type *Src, unsigned Alignment,
593 unsigned AddressSpace, const Instruction *I) {
594 std::pair<int, MVT> LT = TLI->getTypeLegalizationCost(DL, Src);
595
596 if (Src->isVectorTy() && Alignment != 16 &&
597 Src->getVectorElementType()->isDoubleTy()) {
598 // Unaligned loads/stores are extremely inefficient.
599 // We need 4 uops for vst.1/vld.1 vs 1uop for vldr/vstr.
600 return LT.first * 4;
601 }
602 return LT.first;
603}

605int ARMTTIImpl::getInterleavedMemoryOpCost(unsigned Opcode, Type *VecTy,
606 unsigned Factor,
607 ArrayRef<unsigned> Indices,
608 unsigned Alignment,
609 unsigned AddressSpace,
610 bool UseMaskForCond,
611 bool UseMaskForGaps) {
612 assert(Factor >= 2 && "Invalid interleave factor");
613 assert(isa<VectorType>(VecTy) && "Expect a vector type");
614
615 // vldN/vstN doesn't support vector types of i64/f64 element.
616 bool EltIs64Bits = DL.getTypeSizeInBits(VecTy->getScalarType()) == 64;
617
618 if (Factor <= TLI->getMaxSupportedInterleaveFactor() && !EltIs64Bits &&
619 !UseMaskForCond && !UseMaskForGaps) {
620 unsigned NumElts = VecTy->getVectorNumElements();
621 auto *SubVecTy = VectorType::get(VecTy->getScalarType(), NumElts / Factor);
622
623 // vldN/vstN only support legal vector types of size 64 or 128 in bits.
624 // Accesses having vector types that are a multiple of 128 bits can be
625 // matched to more than one vldN/vstN instruction.
626 if (NumElts % Factor == 0 &&
627 TLI->isLegalInterleavedAccessType(SubVecTy, DL))
628 return Factor * TLI->getNumInterleavedAccesses(SubVecTy, DL);
629 }
630
631 return BaseT::getInterleavedMemoryOpCost(Opcode, VecTy, Factor, Indices,
632 Alignment, AddressSpace,
633 UseMaskForCond, UseMaskForGaps);
634}

636bool ARMTTIImpl::isLoweredToCall(const Function *F) {
637 if (!F->isIntrinsic())
638 BaseT::isLoweredToCall(F);
639
640 // Assume all Arm-specific intrinsics map to an instruction.
641 if (F->getName().startswith("llvm.arm"))
642 return false;
643
644 switch (F->getIntrinsicID()) {
645 default: break;
646 case Intrinsic::powi:
647 case Intrinsic::sin:
648 case Intrinsic::cos:
649 case Intrinsic::pow:
650 case Intrinsic::log:
651 case Intrinsic::log10:
652 case Intrinsic::log2:
653 case Intrinsic::exp:
654 case Intrinsic::exp2:
655 return true;
656 case Intrinsic::sqrt:
657 case Intrinsic::fabs:
658 case Intrinsic::copysign:
659 case Intrinsic::floor:
660 case Intrinsic::ceil:
661 case Intrinsic::trunc:
662 case Intrinsic::rint:
663 case Intrinsic::nearbyint:
664 case Intrinsic::round:
665 case Intrinsic::canonicalize:
666 case Intrinsic::lround:
667 case Intrinsic::llround:
668 case Intrinsic::lrint:
669 case Intrinsic::llrint:
670 if (F->getReturnType()->isDoubleTy() && !ST->hasFP64())
671 return true;
672 if (F->getReturnType()->isHalfTy() && !ST->hasFullFP16())
673 return true;
674 // Some operations can be handled by vector instructions and assume
675 // unsupported vectors will be expanded into supported scalar ones.
676 // TODO Handle scalar operations properly.
677 return !ST->hasFPARMv8Base() && !ST->hasVFP2Base();
678 case Intrinsic::masked_store:
679 case Intrinsic::masked_load:
680 case Intrinsic::masked_gather:
681 case Intrinsic::masked_scatter:
682 return !ST->hasMVEIntegerOps();
683 case Intrinsic::sadd_with_overflow:
684 case Intrinsic::uadd_with_overflow:
685 case Intrinsic::ssub_with_overflow:
686 case Intrinsic::usub_with_overflow:
687 case Intrinsic::sadd_sat:
688 case Intrinsic::uadd_sat:
689 case Intrinsic::ssub_sat:
690 case Intrinsic::usub_sat:
691 return false;
692 }
693
694 return BaseT::isLoweredToCall(F);
695}

697bool ARMTTIImpl::isHardwareLoopProfitable(Loop *L, ScalarEvolution &SE,
698 AssumptionCache &AC,
699 TargetLibraryInfo *LibInfo,
700 TTI::HardwareLoopInfo &HWLoopInfo) {
701 // Low-overhead branches are only supported in the 'low-overhead branch'
702 // extension of v8.1-m.
703 if (!ST->hasLOB() || DisableLowOverheadLoops)
704 return false;
705
706 // For now, for simplicity, only support loops with one exit block.
707 if (!L->getExitBlock())
708 return false;
709
710 if (!SE.hasLoopInvariantBackedgeTakenCount(L))
711 return false;
712
713 const SCEV *BackedgeTakenCount = SE.getBackedgeTakenCount(L);
714 if (isa<SCEVCouldNotCompute>(BackedgeTakenCount))
715 return false;
716
717 const SCEV *TripCountSCEV =
718 SE.getAddExpr(BackedgeTakenCount,
719 SE.getOne(BackedgeTakenCount->getType()));
720
721 // We need to store the trip count in LR, a 32-bit register.
722 if (SE.getUnsignedRangeMax(TripCountSCEV).getBitWidth() > 32)
723 return false;
724
725 // Making a call will trash LR and clear LO_BRANCH_INFO, so there's little
726 // point in generating a hardware loop if that's going to happen.
727 auto MaybeCall = [this](Instruction &I) {
728 const ARMTargetLowering *TLI = getTLI();
729 unsigned ISD = TLI->InstructionOpcodeToISD(I.getOpcode());
730 EVT VT = TLI->getValueType(DL, I.getType(), true);
731 if (TLI->getOperationAction(ISD, VT) == TargetLowering::LibCall)
732 return true;
733
734 // Check if an intrinsic will be lowered to a call and assume that any
735 // other CallInst will generate a bl.
736 if (auto *Call = dyn_cast<CallInst>(&I)) {
737 if (isa<IntrinsicInst>(Call)) {
738 if (const Function *F = Call->getCalledFunction())
739 return isLoweredToCall(F);
740 }
741 return true;
742 }
743
744 // FPv5 provides conversions between integer, double-precision,
745 // single-precision, and half-precision formats.
746 switch (I.getOpcode()) {
747 default:
748 break;
749 case Instruction::FPToSI:
750 case Instruction::FPToUI:
751 case Instruction::SIToFP:
752 case Instruction::UIToFP:
753 case Instruction::FPTrunc:
754 case Instruction::FPExt:
755 return !ST->hasFPARMv8Base();
756 }
757
758 // FIXME: Unfortunately the approach of checking the Operation Action does
759 // not catch all cases of Legalization that use library calls. Our
760 // Legalization step categorizes some transformations into library calls as
761 // Custom, Expand or even Legal when doing type legalization. So for now
762 // we have to special case for instance the SDIV of 64bit integers and the
763 // use of floating point emulation.
764 if (VT.isInteger() && VT.getSizeInBits() >= 64) {
765 switch (ISD) {
766 default:
767 break;
768 case ISD::SDIV:
769 case ISD::UDIV:
770 case ISD::SREM:
771 case ISD::UREM:
772 case ISD::SDIVREM:
773 case ISD::UDIVREM:
774 return true;
775 }
776 }
777
778 // Assume all other non-float operations are supported.
779 if (!VT.isFloatingPoint())
780 return false;
781
782 // We'll need a library call to handle most floats when using soft.
783 if (TLI->useSoftFloat()) {
784 switch (I.getOpcode()) {
785 default:
786 return true;
787 case Instruction::Alloca:
788 case Instruction::Load:
789 case Instruction::Store:
790 case Instruction::Select:
791 case Instruction::PHI:
792 return false;
793 }
794 }
795
796 // We'll need a libcall to perform double precision operations on a single
797 // precision only FPU.
798 if (I.getType()->isDoubleTy() && !ST->hasFP64())
799 return true;
800
801 // Likewise for half precision arithmetic.
802 if (I.getType()->isHalfTy() && !ST->hasFullFP16())
803 return true;
804
805 return false;
806 };
807
808 auto IsHardwareLoopIntrinsic = [](Instruction &I) {
809 if (auto *Call = dyn_cast<IntrinsicInst>(&I)) {
810 switch (Call->getIntrinsicID()) {
811 default:
812 break;
813 case Intrinsic::set_loop_iterations:
814 case Intrinsic::loop_decrement:
815 case Intrinsic::loop_decrement_reg:
816 return true;
817 }
818 }
819 return false;
820 };
821
822 // Scan the instructions to see if there's any that we know will turn into a
823 // call or if this loop is already a low-overhead loop.
824 auto ScanLoop = [&](Loop *L) {
825 for (auto *BB : L->getBlocks()) {
826 for (auto &I : *BB) {
827 if (MaybeCall(I) || IsHardwareLoopIntrinsic(I))
828 return false;
829 }
830 }
831 return true;
832 };
833
834 // Visit inner loops.
835 for (auto Inner : *L)
836 if (!ScanLoop(Inner))
837 return false;
838
839 if (!ScanLoop(L))
840 return false;
841
842 // TODO: Check whether the trip count calculation is expensive. If L is the
843 // inner loop but we know it has a low trip count, calculating that trip
844 // count (in the parent loop) may be detrimental.
845
846 LLVMContext &C = L->getHeader()->getContext();
847 HWLoopInfo.CounterInReg = true;
848 HWLoopInfo.IsNestingLegal = false;
849 HWLoopInfo.CountType = Type::getInt32Ty(C);
850 HWLoopInfo.LoopDecrement = ConstantInt::get(HWLoopInfo.CountType, 1);
851 return true;
852}

854void ARMTTIImpl::getUnrollingPreferences(Loop *L, ScalarEvolution &SE,
855 TTI::UnrollingPreferences &UP) {
856 // Only currently enable these preferences for M-Class cores.
857 if (!ST->isMClass())
858 return BasicTTIImplBase::getUnrollingPreferences(L, SE, UP);
859
860 // Disable loop unrolling for Oz and Os.
861 UP.OptSizeThreshold = 0;
862 UP.PartialOptSizeThreshold = 0;
863 if (L->getHeader()->getParent()->hasOptSize())
864 return;
865
866 // Only enable on Thumb-2 targets.
867 if (!ST->isThumb2())
868 return;
869
870 SmallVector<BasicBlock*, 4> ExitingBlocks;
871 L->getExitingBlocks(ExitingBlocks);
872 LLVM_DEBUG(dbgs() << "Loop has:\n"
873 << "Blocks: " << L->getNumBlocks() << "\n"
874 << "Exit blocks: " << ExitingBlocks.size() << "\n");
875
876 // Only allow another exit other than the latch. This acts as an early exit
877 // as it mirrors the profitability calculation of the runtime unroller.
878 if (ExitingBlocks.size() > 2)
879 return;
880
881 // Limit the CFG of the loop body for targets with a branch predictor.
882 // Allowing 4 blocks permits if-then-else diamonds in the body.
883 if (ST->hasBranchPredictor() && L->getNumBlocks() > 4)
884 return;
885
886 // Scan the loop: don't unroll loops with calls as this could prevent
887 // inlining.
888 unsigned Cost = 0;
889 for (auto *BB : L->getBlocks()) {
890 for (auto &I : *BB) {
891 if (isa<CallInst>(I) || isa<InvokeInst>(I)) {
892 ImmutableCallSite CS(&I);
893 if (const Function *F = CS.getCalledFunction()) {
894 if (!isLoweredToCall(F))
895 continue;
896 }
897 return;
898 }
899 SmallVector<const Value*, 4> Operands(I.value_op_begin(),
900 I.value_op_end());
901 Cost += getUserCost(&I, Operands);
902 }
903 }
904
905 LLVM_DEBUG(dbgs() << "Cost of loop: " << Cost << "\n");
906
907 UP.Partial = true;
908 UP.Runtime = true;
909 UP.UpperBound = true;
910 UP.UnrollRemainder = true;
911 UP.DefaultUnrollRuntimeCount = 4;
912 UP.UnrollAndJam = true;
913 UP.UnrollAndJamInnerLoopThreshold = 60;
914
915 // Force unrolling small loops can be very useful because of the branch
916 // taken cost of the backedge.
917 if (Cost < 12)
918 UP.Force = true;
919}
