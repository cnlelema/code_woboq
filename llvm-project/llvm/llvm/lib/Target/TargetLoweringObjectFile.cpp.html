36/// Initialize - this method must be called before any actual lowering is
37/// done. This specifies the current context for codegen, and gives the
38/// lowering implementations a chance to set up their default sections.
39void TargetLoweringObjectFile::Initialize(MCContext &ctx,
40 const TargetMachine &TM) {
41 Ctx = &ctx;
42 // `Initialize` can be called more than once.
43 delete Mang;
44 Mang = new Mangler();
45 InitMCObjectFileInfo(TM.getTargetTriple(), TM.isPositionIndependent(), *Ctx,
46 TM.getCodeModel() == CodeModel::Large);
47
48 // Reset various EH DWARF encodings.
49 PersonalityEncoding = LSDAEncoding = TTypeEncoding = dwarf::DW_EH_PE_absptr;
50}

56static bool isNullOrUndef(const Constant *C) {
57 // Check that the constant isn't all zeros or undefs.
58 if (C->isNullValue() || isa<UndefValue>(C))
59 return true;
60 if (!isa<ConstantAggregate>(C))
61 return false;
62 for (auto Operand : C->operand_values()) {
63 if (!isNullOrUndef(cast<Constant>(Operand)))
64 return false;
65 }
66 return true;
67}
68
69static bool isSuitableForBSS(const GlobalVariable *GV) {
70 const Constant *C = GV->getInitializer();
71
72 // Must have zero initializer.
73 if (!isNullOrUndef(C))
74 return false;
75
76 // Leave constant zeros in readonly constant sections, so they can be shared.
77 if (GV->isConstant())
78 return false;
79
80 // If the global has an explicit section specified, don't put it in BSS.
81 if (GV->hasSection())
82 return false;
83
84 // Otherwise, put it in BSS!
85 return true;
86}

88/// IsNullTerminatedString - Return true if the specified constant (which is
89/// known to have a type that is an array of 1/2/4 byte elements) ends with a
90/// nul value and contains no other nuls in it. Note that this is more general
91/// than ConstantDataSequential::isString because we allow 2 & 4 byte strings.
92static bool IsNullTerminatedString(const Constant *C) {
93 // First check: is we have constant array terminated with zero
94 if (const ConstantDataSequential *CDS = dyn_cast<ConstantDataSequential>(C)) {
95 unsigned NumElts = CDS->getNumElements();
96 assert(NumElts != 0 && "Can't have an empty CDS");
97
98 if (CDS->getElementAsInteger(NumElts-1) != 0)
99 return false; // Not null terminated.
100
101 // Verify that the null doesn't occur anywhere else in the string.
102 for (unsigned i = 0; i != NumElts-1; ++i)
103 if (CDS->getElementAsInteger(i) == 0)
104 return false;
105 return true;
106 }
107
108 // Another possibility: [1 x i8] zeroinitializer
109 if (isa<ConstantAggregateZero>(C))
110 return cast<ArrayType>(C->getType())->getNumElements() == 1;
111
112 return false;
113}

115MCSymbol *TargetLoweringObjectFile::getSymbolWithGlobalValueBase(
116 const GlobalValue *GV, StringRef Suffix, const TargetMachine &TM) const {
117 assert(!Suffix.empty());
118
119 SmallString<60> NameStr;
120 NameStr += GV->getParent()->getDataLayout().getPrivateGlobalPrefix();
121 TM.getNameWithPrefix(NameStr, GV, *Mang);
122 NameStr.append(Suffix.begin(), Suffix.end());
123 return Ctx->getOrCreateSymbol(NameStr);
124}

138static bool isCapabilityType(const Type* T, const DataLayout& DL) {
139 if (T->isPointerTy())
140 return DL.isFatPointer(T);
141 for (const Type* SubTy : T->subtypes()) {
142 if (isCapabilityType(SubTy, DL))
143 return true;
144 }
145 return false;
146}

148/// Check whether the global variable needs a capability relocation (such as
149/// setting the tag bit on startup). This is needed to check whether the
150/// variable can be placed into the read only section
151static bool needsCapabilityRelocation(const GlobalVariable* GV,
152 const DataLayout& DL) {
153 // TODO: make it dependent on the initializer
154 return isCapabilityType(GV->getValueType(), DL);
155}

157/// getKindForGlobal - This is a top-level target-independent classifier for
158/// a global object. Given a global variable and information from the TM, this
159/// function classifies the global in a target independent manner. This function
160/// may be overridden by the target implementation.
161SectionKind TargetLoweringObjectFile::getKindForGlobal(const GlobalObject *GO,
162 const TargetMachine &TM){
163 assert(!GO->isDeclaration() && !GO->hasAvailableExternallyLinkage() &&
164 "Can only be used for global definitions");
165
166 // Functions are classified as text sections.
167 if (isa<Function>(GO))
168 return SectionKind::getText();
169
170 // Global variables require more detailed analysis.
171 const auto *GVar = cast<GlobalVariable>(GO);
172
173 // Handle thread-local data first.
174 if (GVar->isThreadLocal()) {
175 if (isSuitableForBSS(GVar) && !TM.Options.NoZerosInBSS)
176 return SectionKind::getThreadBSS();
177 return SectionKind::getThreadData();
178 }
179
180 // Variables with common linkage always get classified as common.
181 if (GVar->hasCommonLinkage())
182 return SectionKind::getCommon();
183
184 // Most non-mergeable zero data can be put in the BSS section unless otherwise
185 // specified.
186 if (isSuitableForBSS(GVar) && !TM.Options.NoZerosInBSS) {
187 if (GVar->hasLocalLinkage())
188 return SectionKind::getBSSLocal();
189 else if (GVar->hasExternalLinkage())
190 return SectionKind::getBSSExtern();
191 return SectionKind::getBSS();
192 }
193
194 // If the global is marked constant, we can put it into a mergable section,
195 // a mergable string section, or general .data if it contains relocations.
196 if (GVar->isConstant()) {
197 // If the initializer for the global contains something that requires a
198 // relocation, then we may have to drop this into a writable data section
199 // even though it is marked const.
200 const Constant *C = GVar->getInitializer();
201 if (!C->needsRelocation()) {
202 // If the global is required to have a unique address, it can't be put
203 // into a mergable section: just drop it into the general read-only
204 // section instead.
205 if (!GVar->hasGlobalUnnamedAddr())
206 return SectionKind::getReadOnly();
207
208 // If initializer is a null-terminated string, put it in a "cstring"
209 // section of the right width.
210 if (ArrayType *ATy = dyn_cast<ArrayType>(C->getType())) {
211 if (IntegerType *ITy =
212 dyn_cast<IntegerType>(ATy->getElementType())) {
213 if ((ITy->getBitWidth() == 8 || ITy->getBitWidth() == 16 ||
214 ITy->getBitWidth() == 32) &&
215 IsNullTerminatedString(C)) {
216 if (ITy->getBitWidth() == 8)
217 return SectionKind::getMergeable1ByteCString();
218 if (ITy->getBitWidth() == 16)
219 return SectionKind::getMergeable2ByteCString();
220
221 assert(ITy->getBitWidth() == 32 && "Unknown width");
222 return SectionKind::getMergeable4ByteCString();
223 }
224 }
225 }
226
227 // Otherwise, just drop it into a mergable constant section. If we have
228 // a section for this size, use it, otherwise use the arbitrary sized
229 // mergable section.
230 switch (
231 GVar->getParent()->getDataLayout().getTypeAllocSize(C->getType())) {
232 case 4: return SectionKind::getMergeableConst4();
233 case 8: return SectionKind::getMergeableConst8();
234 case 16: return SectionKind::getMergeableConst16();
235 case 32: return SectionKind::getMergeableConst32();
236 default:
237 return SectionKind::getReadOnly();
238 }
239
240 } else {
241 // Even with a static relocation model the startup code will have to
242 // modify capabilities to initialize them because we can't store tag bits
243 // in the ELF file
244 if (needsCapabilityRelocation(GVar, GVar->getParent()->getDataLayout()))
245 return SectionKind::getReadOnlyWithRel();
246 // In static, ROPI and RWPI relocation models, the linker will resolve
247 // all addresses, so the relocation entries will actually be constants by
248 // the time the app starts up. However, we can't put this into a
249 // mergable section, because the linker doesn't take relocations into
250 // consideration when it tries to merge entries in the section.
251 Reloc::Model ReloModel = TM.getRelocationModel();
252 if (ReloModel == Reloc::Static || ReloModel == Reloc::ROPI ||
253 ReloModel == Reloc::RWPI || ReloModel == Reloc::ROPI_RWPI)
254 return SectionKind::getReadOnly();
255
256 // Otherwise, the dynamic linker needs to fix it up, put it in the
257 // writable data.rel section.
258 return SectionKind::getReadOnlyWithRel();
259 }
260 }
261
262 // Okay, this isn't a constant.
263 return SectionKind::getData();
264}

266/// This method computes the appropriate section to emit the specified global
267/// variable or function definition. This should not be passed external (or
268/// available externally) globals.
269MCSection *TargetLoweringObjectFile::SectionForGlobal(
270 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
271 // Select section name.
272 if (GO->hasSection())
273 return getExplicitSectionGlobal(GO, Kind, TM);
274
275 if (auto *GVar = dyn_cast<GlobalVariable>(GO)) {
276 auto Attrs = GVar->getAttributes();
277 if ((Attrs.hasAttribute("bss-section") && Kind.isBSS()) ||
278 (Attrs.hasAttribute("data-section") && Kind.isData()) ||
279 (Attrs.hasAttribute("rodata-section") && Kind.isReadOnly())) {
280 return getExplicitSectionGlobal(GO, Kind, TM);
281 }
282 }
283
284 if (auto *F = dyn_cast<Function>(GO)) {
285 if (F->hasFnAttribute("implicit-section-name"))
286 return getExplicitSectionGlobal(GO, Kind, TM);
287 }
288
289 // Use default section depending on the 'type' of global
290 return SelectSectionForGlobal(GO, Kind, TM);
291}

301bool TargetLoweringObjectFile::shouldPutJumpTableInFunctionSection(
302 bool UsesLabelDifference, const Function &F) const {
303 // In PIC mode, we need to emit the jump table to the same section as the
304 // function body itself, otherwise the label differences won't make sense.
305 // FIXME: Need a better predicate for this: what about custom entries?
306 if (UsesLabelDifference)
307 return true;
308
309 // We should also do if the section name is NULL or function is declared
310 // in discardable section
311 // FIXME: this isn't the right predicate, should be based on the MCSection
312 // for the function.
313 return F.isWeakForLinker();
314}

339const MCExpr *TargetLoweringObjectFile::
340getTTypeReference(const MCSymbolRefExpr *Sym, unsigned Encoding,
341 MCStreamer &Streamer) const {
342 switch (Encoding & 0x70) {
343 default:
344 report_fatal_error("We do not support this DWARF encoding yet!");
345 case dwarf::DW_EH_PE_absptr:
346 // Do nothing special
347 return Sym;
348 case dwarf::DW_EH_PE_pcrel: {
349 // Emit a label to the streamer for the current position. This gives us
350 // .-foo addressing.
351 MCSymbol *PCSym = getContext().createTempSymbol();
352 Streamer.EmitLabel(PCSym);
353 const MCExpr *PC = MCSymbolRefExpr::create(PCSym, getContext());
354 return MCBinaryExpr::createSub(Sym, PC, getContext());
355 }
356 }
357}
