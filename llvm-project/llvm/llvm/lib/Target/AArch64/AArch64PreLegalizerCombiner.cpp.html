28namespace {
29class AArch64PreLegalizerCombinerInfo : public CombinerInfo {
30public:
31 AArch64PreLegalizerCombinerInfo()
32 : CombinerInfo(/*AllowIllegalOps*/ true, /*ShouldLegalizeIllegal*/ false,
33 /*LegalizerInfo*/ nullptr) {}
34 virtual bool combine(GISelChangeObserver &Observer, MachineInstr &MI,
35 MachineIRBuilder &B) const override;
36};
37
38bool AArch64PreLegalizerCombinerInfo::combine(GISelChangeObserver &Observer,
39 MachineInstr &MI,
40 MachineIRBuilder &B) const {
41 CombinerHelper Helper(Observer, B);
42
43 switch (MI.getOpcode()) {
44 default:
45 return false;
46 case TargetOpcode::COPY:
47 return Helper.tryCombineCopy(MI);
48 case TargetOpcode::G_LOAD:
49 case TargetOpcode::G_SEXTLOAD:
50 case TargetOpcode::G_ZEXTLOAD:
51 return Helper.tryCombineExtendingLoads(MI);
52 }
53
54 return false;
55}

60class AArch64PreLegalizerCombiner : public MachineFunctionPass {
61public:
62 static char ID;
63
64 AArch64PreLegalizerCombiner();
65
66 StringRef getPassName() const override { return "AArch64PreLegalizerCombiner"; }
67
68 bool runOnMachineFunction(MachineFunction &MF) override;
69
70 void getAnalysisUsage(AnalysisUsage &AU) const override;
71};
72}
73
74void AArch64PreLegalizerCombiner::getAnalysisUsage(AnalysisUsage &AU) const {
75 AU.addRequired<TargetPassConfig>();
76 AU.setPreservesCFG();
77 getSelectionDAGFallbackAnalysisUsage(AU);
78 MachineFunctionPass::getAnalysisUsage(AU);
79}
80
81AArch64PreLegalizerCombiner::AArch64PreLegalizerCombiner() : MachineFunctionPass(ID) {
82 initializeAArch64PreLegalizerCombinerPass(*PassRegistry::getPassRegistry());
83}
84
85bool AArch64PreLegalizerCombiner::runOnMachineFunction(MachineFunction &MF) {
86 if (MF.getProperties().hasProperty(
87 MachineFunctionProperties::Property::FailedISel))
88 return false;
89 auto *TPC = &getAnalysis<TargetPassConfig>();
90 AArch64PreLegalizerCombinerInfo PCInfo;
91 Combiner C(PCInfo, TPC);
92 return C.combineMachineInstrs(MF, /*CSEInfo*/ nullptr);
93}
