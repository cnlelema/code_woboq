115bool AArch64GenRegisterBankInfo::checkPartialMap(unsigned Idx,
116 unsigned ValStartIdx,
117 unsigned ValLength,
118 const RegisterBank &RB) {
119 const PartialMapping &Map = PartMappings[Idx - PartialMappingIdx::PMI_Min];
120 return Map.StartIdx == ValStartIdx && Map.Length == ValLength &&
121 Map.RegBank == &RB;
122}

124bool AArch64GenRegisterBankInfo::checkValueMapImpl(unsigned Idx,
125 unsigned FirstInBank,
126 unsigned Size,
127 unsigned Offset) {
128 unsigned PartialMapBaseIdx = Idx - PartialMappingIdx::PMI_Min;
129 const ValueMapping &Map =
130 AArch64GenRegisterBankInfo::getValueMapping((PartialMappingIdx)FirstInBank, Size)[Offset];
131 return Map.BreakDown == &PartMappings[PartialMapBaseIdx] &&
132 Map.NumBreakDowns == 1;
133}

135bool AArch64GenRegisterBankInfo::checkPartialMappingIdx(
136 PartialMappingIdx FirstAlias, PartialMappingIdx LastAlias,
137 ArrayRef<PartialMappingIdx> Order) {
138 if (Order.front() != FirstAlias)
139 return false;
140 if (Order.back() != LastAlias)
141 return false;
142 if (Order.front() > Order.back())
143 return false;
144
145 PartialMappingIdx Previous = Order.front();
146 bool First = true;
147 for (const auto &Current : Order) {
148 if (First) {
149 First = false;
150 continue;
151 }
152 if (Previous + 1 != Current)
153 return false;
154 Previous = Current;
155 }
156 return true;
157}

159unsigned AArch64GenRegisterBankInfo::getRegBankBaseIdxOffset(unsigned RBIdx,
160 unsigned Size) {
161 if (RBIdx == PMI_FirstGPR) {
162 if (Size <= 32)
163 return 0;
164 if (Size <= 64)
165 return 1;
166 return -1;
167 }
168 if (RBIdx == PMI_FirstFPR) {
169 if (Size <= 16)
170 return 0;
171 if (Size <= 32)
172 return 1;
173 if (Size <= 64)
174 return 2;
175 if (Size <= 128)
176 return 3;
177 if (Size <= 256)
178 return 4;
179 if (Size <= 512)
180 return 5;
181 return -1;
182 }
183 return -1;
184}

186const RegisterBankInfo::ValueMapping *
187AArch64GenRegisterBankInfo::getValueMapping(PartialMappingIdx RBIdx,
188 unsigned Size) {
189 assert(RBIdx != PartialMappingIdx::PMI_None && "No mapping needed for that");
190 unsigned BaseIdxOffset = getRegBankBaseIdxOffset(RBIdx, Size);
191 if (BaseIdxOffset == -1u)
192 return &ValMappings[InvalidIdx];
193
194 unsigned ValMappingIdx =
195 First3OpsIdx + (RBIdx - PartialMappingIdx::PMI_Min + BaseIdxOffset) *
196 ValueMappingIdx::DistanceBetweenRegBanks;
197 assert(ValMappingIdx >= First3OpsIdx && ValMappingIdx <= Last3OpsIdx &&
198 "Mapping out of bound");
199
200 return &ValMappings[ValMappingIdx];
201}

203AArch64GenRegisterBankInfo::PartialMappingIdx
204 AArch64GenRegisterBankInfo::BankIDToCopyMapIdx[]{
205 PMI_None, // CCR
206 PMI_FirstFPR, // FPR
207 PMI_FirstGPR, // GPR
208 };
209
210const RegisterBankInfo::ValueMapping *
211AArch64GenRegisterBankInfo::getCopyMapping(unsigned DstBankID,
212 unsigned SrcBankID, unsigned Size) {
213 assert(DstBankID < AArch64::NumRegisterBanks && "Invalid bank ID");
214 assert(SrcBankID < AArch64::NumRegisterBanks && "Invalid bank ID");
215 PartialMappingIdx DstRBIdx = BankIDToCopyMapIdx[DstBankID];
216 PartialMappingIdx SrcRBIdx = BankIDToCopyMapIdx[SrcBankID];
217 assert(DstRBIdx != PMI_None && "No such mapping");
218 assert(SrcRBIdx != PMI_None && "No such mapping");
219
220 if (DstRBIdx == SrcRBIdx)
221 return getValueMapping(DstRBIdx, Size);
222
223 assert(Size <= 64 && "GPR cannot handle that size");
224 unsigned ValMappingIdx =
225 FirstCrossRegCpyIdx +
226 (DstRBIdx - PMI_Min + getRegBankBaseIdxOffset(DstRBIdx, Size)) *
227 ValueMappingIdx::DistanceBetweenCrossRegCpy;
228 assert(ValMappingIdx >= FirstCrossRegCpyIdx &&
229 ValMappingIdx <= LastCrossRegCpyIdx && "Mapping out of bound");
230 return &ValMappings[ValMappingIdx];
231}

233const RegisterBankInfo::ValueMapping *
234AArch64GenRegisterBankInfo::getFPExtMapping(unsigned DstSize,
235 unsigned SrcSize) {
236 // We support:
237 // - For Scalar:
238 // - 16 to 32.
239 // - 16 to 64.
240 // - 32 to 64.
241 // => FPR 16 to FPR 32|64
242 // => FPR 32 to FPR 64
243 // - For vectors:
244 // - v4f16 to v4f32
245 // - v2f32 to v2f64
246 // => FPR 64 to FPR 128
247
248 // Check that we have been asked sensible sizes.
249 if (SrcSize == 16) {
250 assert((DstSize == 32 || DstSize == 64) && "Unexpected half extension");
251 if (DstSize == 32)
252 return &ValMappings[FPExt16To32Idx];
253 return &ValMappings[FPExt16To64Idx];
254 }
255
256 if (SrcSize == 32) {
257 assert(DstSize == 64 && "Unexpected float extension");
258 return &ValMappings[FPExt32To64Idx];
259 }
260 assert((SrcSize == 64 || DstSize == 128) && "Unexpected vector extension");
261 return &ValMappings[FPExt64To128Idx];
262}
