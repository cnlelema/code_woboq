31StringRef AArch64MCExpr::getVariantKindName() const {
32 switch (static_cast<uint32_t>(getKind())) {
33 case VK_CALL: return "";
34 case VK_LO12: return ":lo12:";
35 case VK_ABS_G3: return ":abs_g3:";
36 case VK_ABS_G2: return ":abs_g2:";
37 case VK_ABS_G2_S: return ":abs_g2_s:";
38 case VK_ABS_G2_NC: return ":abs_g2_nc:";
39 case VK_ABS_G1: return ":abs_g1:";
40 case VK_ABS_G1_S: return ":abs_g1_s:";
41 case VK_ABS_G1_NC: return ":abs_g1_nc:";
42 case VK_ABS_G0: return ":abs_g0:";
43 case VK_ABS_G0_S: return ":abs_g0_s:";
44 case VK_ABS_G0_NC: return ":abs_g0_nc:";
45 case VK_DTPREL_G2: return ":dtprel_g2:";
46 case VK_DTPREL_G1: return ":dtprel_g1:";
47 case VK_DTPREL_G1_NC: return ":dtprel_g1_nc:";
48 case VK_DTPREL_G0: return ":dtprel_g0:";
49 case VK_DTPREL_G0_NC: return ":dtprel_g0_nc:";
50 case VK_DTPREL_HI12: return ":dtprel_hi12:";
51 case VK_DTPREL_LO12: return ":dtprel_lo12:";
52 case VK_DTPREL_LO12_NC: return ":dtprel_lo12_nc:";
53 case VK_TPREL_G2: return ":tprel_g2:";
54 case VK_TPREL_G1: return ":tprel_g1:";
55 case VK_TPREL_G1_NC: return ":tprel_g1_nc:";
56 case VK_TPREL_G0: return ":tprel_g0:";
57 case VK_TPREL_G0_NC: return ":tprel_g0_nc:";
58 case VK_TPREL_HI12: return ":tprel_hi12:";
59 case VK_TPREL_LO12: return ":tprel_lo12:";
60 case VK_TPREL_LO12_NC: return ":tprel_lo12_nc:";
61 case VK_TLSDESC_LO12: return ":tlsdesc_lo12:";
62 case VK_ABS_PAGE: return "";
63 case VK_ABS_PAGE_NC: return ":pg_hi21_nc:";
64 case VK_GOT: return ":got:";
65 case VK_GOT_PAGE: return ":got:";
66 case VK_GOT_LO12: return ":got_lo12:";
67 case VK_GOTTPREL: return ":gottprel:";
68 case VK_GOTTPREL_PAGE: return ":gottprel:";
69 case VK_GOTTPREL_LO12_NC: return ":gottprel_lo12:";
70 case VK_GOTTPREL_G1: return ":gottprel_g1:";
71 case VK_GOTTPREL_G0_NC: return ":gottprel_g0_nc:";
72 case VK_TLSDESC: return "";
73 case VK_TLSDESC_PAGE: return ":tlsdesc:";
74 case VK_SECREL_LO12: return ":secrel_lo12:";
75 case VK_SECREL_HI12: return ":secrel_hi12:";
76 default:
77 llvm_unreachable("Invalid ELF symbol kind");
78 }
79}

94bool AArch64MCExpr::evaluateAsRelocatableImpl(MCValue &Res,
95 const MCAsmLayout *Layout,
96 const MCFixup *Fixup) const {
97 if (!getSubExpr()->evaluateAsRelocatable(Res, Layout, Fixup))
98 return false;
99
100 Res =
101 MCValue::get(Res.getSymA(), Res.getSymB(), Res.getConstant(), getKind());
102
103 return true;
104}

106static void fixELFSymbolsInTLSFixupsImpl(const MCExpr *Expr, MCAssembler &Asm) {
107 switch (Expr->getKind()) {
108 case MCExpr::Target:
109 llvm_unreachable("Can't handle nested target expression");
110 break;
111 case MCExpr::Constant:
112 break;
113
114 case MCExpr::Binary: {
115 const MCBinaryExpr *BE = cast<MCBinaryExpr>(Expr);
116 fixELFSymbolsInTLSFixupsImpl(BE->getLHS(), Asm);
117 fixELFSymbolsInTLSFixupsImpl(BE->getRHS(), Asm);
118 break;
119 }
120
121 case MCExpr::SymbolRef: {
122 // We're known to be under a TLS fixup, so any symbol should be
123 // modified. There should be only one.
124 const MCSymbolRefExpr &SymRef = *cast<MCSymbolRefExpr>(Expr);
125 cast<MCSymbolELF>(SymRef.getSymbol()).setType(ELF::STT_TLS);
126 break;
127 }
128
129 case MCExpr::Unary:
130 fixELFSymbolsInTLSFixupsImpl(cast<MCUnaryExpr>(Expr)->getSubExpr(), Asm);
131 break;
132 }
133}
134
135void AArch64MCExpr::fixELFSymbolsInTLSFixups(MCAssembler &Asm) const {
136 switch (getSymbolLoc(Kind)) {
137 default:
138 return;
139 case VK_DTPREL:
140 case VK_GOTTPREL:
141 case VK_TPREL:
142 case VK_TLSDESC:
143 break;
144 }
145
146 fixELFSymbolsInTLSFixupsImpl(getSubExpr(), Asm);
147}
