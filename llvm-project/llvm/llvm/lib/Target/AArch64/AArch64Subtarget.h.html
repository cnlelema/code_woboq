428 bool useSmallAddressing() const {
429 switch (TLInfo.getTargetMachine().getCodeModel()) {
430 case CodeModel::Kernel:
431 // Kernel is currently allowed only for Fuchsia targets,
432 // where it is the same as Small for almost all purposes.
433 case CodeModel::Small:
434 return true;
435 default:
436 return false;
437 }
438 }

461 case CallingConv::C:
462 case CallingConv::Fast:
463 case CallingConv::Swift:
