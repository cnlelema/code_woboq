14#ifndef LLVM_LIB_TARGET_AARCH64_AARCH64CALLLOWERING_H
15#define LLVM_LIB_TARGET_AARCH64_AARCH64CALLLOWERING_H
16
17#include "llvm/ADT/ArrayRef.h"
18#include "llvm/CodeGen/GlobalISel/CallLowering.h"
19#include "llvm/IR/CallingConv.h"
20#include <cstdint>
21#include <functional>
22
23namespace llvm {
24
25class AArch64TargetLowering;
26class CCValAssign;
27class DataLayout;
28class MachineIRBuilder;
29class MachineRegisterInfo;
30class Type;
31
32class AArch64CallLowering: public CallLowering {
33public:
34 AArch64CallLowering(const AArch64TargetLowering &TLI);
35
36 bool lowerReturn(MachineIRBuilder &MIRBuilder, const Value *Val,
37 ArrayRef<unsigned> VRegs,
38 unsigned SwiftErrorVReg) const override;
39
40 bool lowerFormalArguments(MachineIRBuilder &MIRBuilder, const Function &F,
41 ArrayRef<unsigned> VRegs) const override;
42
43 bool lowerCall(MachineIRBuilder &MIRBuilder, CallingConv::ID CallConv,
44 const MachineOperand &Callee, const ArgInfo &OrigRet,
45 ArrayRef<ArgInfo> OrigArgs,
46 unsigned SwiftErrorVReg) const override;
47
48 bool lowerCall(MachineIRBuilder &MIRBuilder, CallingConv::ID CallConv,
49 const MachineOperand &Callee, const ArgInfo &OrigRet,
50 ArrayRef<ArgInfo> OrigArgs) const override {
51 return lowerCall(MIRBuilder, CallConv, Callee, OrigRet, OrigArgs, 0);
52 }
53
54 bool supportSwiftError() const override { return true; }
55
56private:
57 using RegHandler = std::function<void(MachineIRBuilder &, Type *, unsigned,
58 CCValAssign &)>;
59
60 using MemHandler =
61 std::function<void(MachineIRBuilder &, int, CCValAssign &)>;
62
63 using SplitArgTy = std::function<void(unsigned, uint64_t)>;
64
65 void splitToValueTypes(const ArgInfo &OrigArgInfo,
66 SmallVectorImpl<ArgInfo> &SplitArgs,
67 const DataLayout &DL, MachineRegisterInfo &MRI,
68 CallingConv::ID CallConv,
69 const SplitArgTy &SplitArg) const;
70};
71
72} // end namespace llvm
73
74#endif // LLVM_LIB_TARGET_AARCH64_AARCH64CALLLOWERING_H
