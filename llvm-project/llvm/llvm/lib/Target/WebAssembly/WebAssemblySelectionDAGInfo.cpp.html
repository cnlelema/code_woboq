21SDValue WebAssemblySelectionDAGInfo::EmitTargetCodeForMemcpy(
22 SelectionDAG &DAG, const SDLoc &DL, SDValue Chain, SDValue Dst, SDValue Src,
23 SDValue Size, unsigned Align, bool IsVolatile, bool AlwaysInline,
24 bool MustPreserveCheriCapabilities, MachinePointerInfo DstPtrInfo,
25 MachinePointerInfo SrcPtrInfo) const {
26 if (!DAG.getMachineFunction()
27 .getSubtarget<WebAssemblySubtarget>()
28 .hasBulkMemory())
29 return SDValue();
30
31 SDValue MemIdx = DAG.getConstant(0, DL, MVT::i32);
32 return DAG.getNode(WebAssemblyISD::MEMORY_COPY, DL, MVT::Other,
33 {Chain, MemIdx, MemIdx, Dst, Src,
34 DAG.getZExtOrTrunc(Size, DL, MVT::i32)});
35}

47SDValue WebAssemblySelectionDAGInfo::EmitTargetCodeForMemset(
48 SelectionDAG &DAG, const SDLoc &DL, SDValue Chain, SDValue Dst, SDValue Val,
49 SDValue Size, unsigned Align, bool IsVolatile,
50 MachinePointerInfo DstPtrInfo) const {
51 if (!DAG.getMachineFunction()
52 .getSubtarget<WebAssemblySubtarget>()
53 .hasBulkMemory())
54 return SDValue();
55
56 SDValue MemIdx = DAG.getConstant(0, DL, MVT::i32);
57 // Only low byte matters for val argument, so anyext the i8
58 return DAG.getNode(WebAssemblyISD::MEMORY_FILL, DL, MVT::Other, Chain, MemIdx,
59 Dst, DAG.getAnyExtOrTrunc(Val, DL, MVT::i32),
60 DAG.getZExtOrTrunc(Size, DL, MVT::i32));
61}
