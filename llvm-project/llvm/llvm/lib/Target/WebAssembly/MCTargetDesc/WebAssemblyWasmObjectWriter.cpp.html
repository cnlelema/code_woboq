42WebAssemblyWasmObjectWriter::WebAssemblyWasmObjectWriter(bool Is64Bit)
43 : MCWasmObjectTargetWriter(Is64Bit) {}
44
45static const MCSection *getFixupSection(const MCExpr *Expr) {
46 if (auto SyExp = dyn_cast<MCSymbolRefExpr>(Expr)) {
47 if (SyExp->getSymbol().isInSection())
48 return &SyExp->getSymbol().getSection();
49 return nullptr;
50 }
51
52 if (auto BinOp = dyn_cast<MCBinaryExpr>(Expr)) {
53 auto SectionLHS = getFixupSection(BinOp->getLHS());
54 auto SectionRHS = getFixupSection(BinOp->getRHS());
55 return SectionLHS == SectionRHS ? nullptr : SectionLHS;
56 }
57
58 if (auto UnOp = dyn_cast<MCUnaryExpr>(Expr))
59 return getFixupSection(UnOp->getSubExpr());
60
61 return nullptr;
62}

64unsigned WebAssemblyWasmObjectWriter::getRelocType(const MCValue &Target,
65 const MCFixup &Fixup) const {
66 const MCSymbolRefExpr *RefA = Target.getSymA();
67 assert(RefA);
68 auto& SymA = cast<MCSymbolWasm>(RefA->getSymbol());
69
70 MCSymbolRefExpr::VariantKind Modifier = Target.getAccessVariant();
71
72 switch (Modifier) {
73 case MCSymbolRefExpr::VK_GOT:
74 return wasm::R_WASM_GLOBAL_INDEX_LEB;
75 case MCSymbolRefExpr::VK_WASM_TBREL:
76 assert(SymA.isFunction());
77 return wasm::R_WASM_TABLE_INDEX_REL_SLEB;
78 case MCSymbolRefExpr::VK_WASM_MBREL:
79 assert(SymA.isData());
80 return wasm::R_WASM_MEMORY_ADDR_REL_SLEB;
81 case MCSymbolRefExpr::VK_WASM_TYPEINDEX:
82 return wasm::R_WASM_TYPE_INDEX_LEB;
83 default:
84 break;
85 }
86
87 switch (unsigned(Fixup.getKind())) {
88 case WebAssembly::fixup_sleb128_i32:
89 if (SymA.isFunction())
90 return wasm::R_WASM_TABLE_INDEX_SLEB;
91 return wasm::R_WASM_MEMORY_ADDR_SLEB;
92 case WebAssembly::fixup_sleb128_i64:
93 llvm_unreachable("fixup_sleb128_i64 not implemented yet");
94 case WebAssembly::fixup_uleb128_i32:
95 if (SymA.isGlobal())
96 return wasm::R_WASM_GLOBAL_INDEX_LEB;
97 if (SymA.isFunction())
98 return wasm::R_WASM_FUNCTION_INDEX_LEB;
99 if (SymA.isEvent())
100 return wasm::R_WASM_EVENT_INDEX_LEB;
101 return wasm::R_WASM_MEMORY_ADDR_LEB;
102 case FK_Data_4:
103 if (SymA.isFunction())
104 return wasm::R_WASM_TABLE_INDEX_I32;
105 if (auto Section = static_cast<const MCSectionWasm *>(
106 getFixupSection(Fixup.getValue()))) {
107 if (Section->getKind().isText())
108 return wasm::R_WASM_FUNCTION_OFFSET_I32;
109 else if (!Section->isWasmData())
110 return wasm::R_WASM_SECTION_OFFSET_I32;
111 }
112 return wasm::R_WASM_MEMORY_ADDR_I32;
113 default:
114 llvm_unreachable("unimplemented fixup kind");
115 }
116}
