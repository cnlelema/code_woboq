14#include "TargetInfo/WebAssemblyTargetInfo.h"
15#include "llvm/Support/TargetRegistry.h"
16using namespace llvm;

18#define DEBUG_TYPE "wasm-target-info"
19
20Target &llvm::getTheWebAssemblyTarget32() {
21 static Target TheWebAssemblyTarget32;
22 return TheWebAssemblyTarget32;
23}
24Target &llvm::getTheWebAssemblyTarget64() {
25 static Target TheWebAssemblyTarget64;
26 return TheWebAssemblyTarget64;
27}
28
29extern "C" void LLVMInitializeWebAssemblyTargetInfo() {
30 RegisterTarget<Triple::wasm32> X(getTheWebAssemblyTarget32(), "wasm32",
31 "WebAssembly 32-bit", "WebAssembly");
32 RegisterTarget<Triple::wasm64> Y(getTheWebAssemblyTarget64(), "wasm64",
33 "WebAssembly 64-bit", "WebAssembly");
34}
