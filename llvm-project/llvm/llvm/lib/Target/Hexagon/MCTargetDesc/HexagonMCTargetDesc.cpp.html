97static StringRef HexagonGetArchVariant() {
98 if (MV5)
99 return "hexagonv5";
100 if (MV55)
101 return "hexagonv55";
102 if (MV60)
103 return "hexagonv60";

113StringRef Hexagon_MC::selectHexagonCPU(StringRef CPU) {
114 StringRef ArchV = HexagonGetArchVariant();
115 if (!ArchV.empty() && !CPU.empty()) {
116 if (ArchV != CPU)
117 report_fatal_error("conflicting architectures specified.");
118 return CPU;
119 }
120 if (ArchV.empty()) {
121 if (CPU.empty())
122 CPU = DefaultArch;
123 return CPU;
124 }
125 return ArchV;
126}

132class HexagonTargetAsmStreamer : public HexagonTargetStreamer {
133public:
134 HexagonTargetAsmStreamer(MCStreamer &S,
135 formatted_raw_ostream &OS,
136 bool isVerboseAsm,
137 MCInstPrinter &IP)
138 : HexagonTargetStreamer(S) {}
139
140 void prettyPrintAsm(MCInstPrinter &InstPrinter, raw_ostream &OS,
141 const MCInst &Inst, const MCSubtargetInfo &STI) override {
142 assert(HexagonMCInstrInfo::isBundle(Inst));
143 assert(HexagonMCInstrInfo::bundleSize(Inst) <= HEXAGON_PACKET_SIZE);
144 std::string Buffer;
145 {
146 raw_string_ostream TempStream(Buffer);
147 InstPrinter.printInst(&Inst, TempStream, "", STI);
148 }
149 StringRef Contents(Buffer);
150 auto PacketBundle = Contents.rsplit('\n');
151 auto HeadTail = PacketBundle.first.split('\n');
152 StringRef Separator = "\n";
153 StringRef Indent = "\t";
154 OS << "\t{\n";
155 while (!HeadTail.first.empty()) {
156 StringRef InstTxt;
157 auto Duplex = HeadTail.first.split('\v');
158 if (!Duplex.second.empty()) {
159 OS << Indent << Duplex.first << Separator;
160 InstTxt = Duplex.second;
161 } else if (!HeadTail.first.trim().startswith("immext")) {
162 InstTxt = Duplex.first;
163 }
164 if (!InstTxt.empty())
165 OS << Indent << InstTxt << Separator;
166 HeadTail = HeadTail.second.split('\n');
167 }
168
169 if (HexagonMCInstrInfo::isMemReorderDisabled(Inst))
170 OS << "\n\t} :mem_noshuf" << PacketBundle.second;
171 else
172 OS << "\t}" << PacketBundle.second;
173 }
174};

221static MCAsmInfo *createHexagonMCAsmInfo(const MCRegisterInfo &MRI,
222 const Triple &TT) {
223 MCAsmInfo *MAI = new HexagonMCAsmInfo(TT);
224
225 // VirtualFP = (R30 + #0).
226 MCCFIInstruction Inst =
227 MCCFIInstruction::createDefCfa(nullptr,
228 MRI.getDwarfRegNum(Hexagon::R30, true), 0);
229 MAI->addInitialFrameState(Inst);
230
231 return MAI;
232}

269static void LLVM_ATTRIBUTE_UNUSED clearFeature(MCSubtargetInfo* STI, uint64_t F) {
270 uint64_t FB = STI->getFeatureBits().to_ullong();
271 if (FB & (1ULL << F))

281std::string selectHexagonFS(StringRef CPU, StringRef FS) {
282 SmallVector<StringRef, 3> Result;
283 if (!FS.empty())
284 Result.push_back(FS);
285
286 switch (EnableHVX) {
287 case Hexagon::ArchEnum::V5:
288 case Hexagon::ArchEnum::V55:
289 break;
290 case Hexagon::ArchEnum::V60:
291 Result.push_back("+hvxv60");
292 break;
293 case Hexagon::ArchEnum::V62:
294 Result.push_back("+hvxv62");
295 break;
296 case Hexagon::ArchEnum::V65:
297 Result.push_back("+hvxv65");
298 break;
299 case Hexagon::ArchEnum::V66:
300 Result.push_back("+hvxv66");
301 break;
302 case Hexagon::ArchEnum::Generic:{
303 Result.push_back(StringSwitch<StringRef>(CPU)
304 .Case("hexagonv60", "+hvxv60")
305 .Case("hexagonv62", "+hvxv62")
306 .Case("hexagonv65", "+hvxv65")
307 .Case("hexagonv66", "+hvxv66"));
308 break;
309 }
310 case Hexagon::ArchEnum::NoArch:
311 // Sentinal if -mhvx isn't specified
312 break;
313 }
314 return join(Result.begin(), Result.end(), ",");
315}
316}

318static bool isCPUValid(std::string CPU)
319{
320 std::vector<std::string> table {
321 "generic", "hexagonv5", "hexagonv55", "hexagonv60",
322 "hexagonv62", "hexagonv65", "hexagonv66",
323 };
324
325 return std::find(table.begin(), table.end(), CPU) != table.end();
326}

338FeatureBitset Hexagon_MC::completeHVXFeatures(const FeatureBitset &S) {
339 using namespace Hexagon;
340 // Make sure that +hvx-length turns hvx on, and that "hvx" alone
341 // turns on hvxvNN, corresponding to the existing ArchVNN.
342 FeatureBitset FB = S;
343 unsigned CpuArch = ArchV5;
344 for (unsigned F : {ArchV66, ArchV65, ArchV62, ArchV60, ArchV55, ArchV5}) {
345 if (!FB.test(F))
346 continue;
347 CpuArch = F;
348 break;
349 }
350 bool UseHvx = false;
351 for (unsigned F : {ExtensionHVX, ExtensionHVX64B, ExtensionHVX128B}) {
352 if (!FB.test(F))
353 continue;
354 UseHvx = true;
355 break;
356 }
357 bool HasHvxVer = false;
358 for (unsigned F : {ExtensionHVXV60, ExtensionHVXV62, ExtensionHVXV65,
359 ExtensionHVXV66}) {
360 if (!FB.test(F))
361 continue;
362 HasHvxVer = true;
363 UseHvx = true;
364 break;
365 }
366
367 if (!UseHvx || HasHvxVer)
368 return FB;
369
370 // HasHvxVer is false, and UseHvx is true.
371 switch (CpuArch) {
372 case ArchV66:
373 FB.set(ExtensionHVXV66);
374 LLVM_FALLTHROUGH;
375 case ArchV65:
376 FB.set(ExtensionHVXV65);
377 LLVM_FALLTHROUGH;
378 case ArchV62:
379 FB.set(ExtensionHVXV62);
380 LLVM_FALLTHROUGH;
381 case ArchV60:
382 FB.set(ExtensionHVXV60);
383 break;
384 }
385 return FB;
386}

388MCSubtargetInfo *Hexagon_MC::createHexagonMCSubtargetInfo(const Triple &TT,
389 StringRef CPU,
390 StringRef FS) {
391 std::pair<std::string, std::string> Features = selectCPUAndFS(CPU, FS);
392 StringRef CPUName = Features.first;
393 StringRef ArchFS = Features.second;
394
395 if (!isCPUValid(CPUName.str())) {
396 errs() << "error: invalid CPU \"" << CPUName.str().c_str()
397 << "\" specified\n";
398 return nullptr;
399 }
400
401 MCSubtargetInfo *X = createHexagonMCSubtargetInfoImpl(TT, CPUName, ArchFS);
402 if (HexagonDisableDuplex) {
403 llvm::FeatureBitset Features = X->getFeatureBits();
404 X->setFeatureBits(Features.set(Hexagon::FeatureDuplex, false));
405 }
406
407 X->setFeatureBits(completeHVXFeatures(X->getFeatureBits()));
408 return X;
409}

411unsigned Hexagon_MC::GetELFFlags(const MCSubtargetInfo &STI) {
412 static std::map<StringRef,unsigned> ElfFlags = {
413 {"hexagonv5", ELF::EF_HEXAGON_MACH_V5},
414 {"hexagonv55", ELF::EF_HEXAGON_MACH_V55},
415 {"hexagonv60", ELF::EF_HEXAGON_MACH_V60},
416 {"hexagonv62", ELF::EF_HEXAGON_MACH_V62},
417 {"hexagonv65", ELF::EF_HEXAGON_MACH_V65},
418 {"hexagonv66", ELF::EF_HEXAGON_MACH_V66},
419 };
420
421 auto F = ElfFlags.find(STI.getCPU());
422 assert(F != ElfFlags.end() && "Unrecognized Architecture");
423 return F->second;
424}

427class HexagonMCInstrAnalysis : public MCInstrAnalysis {
428public:
429 HexagonMCInstrAnalysis(MCInstrInfo const *Info) : MCInstrAnalysis(Info) {}
430
431 bool isUnconditionalBranch(MCInst const &Inst) const override {
432 //assert(!HexagonMCInstrInfo::isBundle(Inst));
433 return MCInstrAnalysis::isUnconditionalBranch(Inst);
434 }
435
436 bool isConditionalBranch(MCInst const &Inst) const override {
437 //assert(!HexagonMCInstrInfo::isBundle(Inst));
438 return MCInstrAnalysis::isConditionalBranch(Inst);
439 }
440
441 bool evaluateBranch(MCInst const &Inst, uint64_t Addr,
442 uint64_t Size, uint64_t &Target) const override {
443 //assert(!HexagonMCInstrInfo::isBundle(Inst));
444 if(!HexagonMCInstrInfo::isExtendable(*Info, Inst))
445 return false;
446 auto const &Extended(HexagonMCInstrInfo::getExtendableOperand(*Info, Inst));
447 assert(Extended.isExpr());
448 int64_t Value;
449 if(!Extended.getExpr()->evaluateAsAbsolute(Value))
450 return false;
451 Target = Value;
452 return true;
453 }
454};
455}
