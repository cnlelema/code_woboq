95static bool opcodeEmitsNoInsts(const MachineInstr &MI) {
96 if (MI.isMetaInstruction())
97 return true;
98
99 // Handle target specific opcodes.
100 switch (MI.getOpcode()) {
101 case AMDGPU::SI_MASK_BRANCH:
102 return true;
103 default:
104 return false;
105 }
106}

108bool SIInsertSkips::shouldSkip(const MachineBasicBlock &From,
109 const MachineBasicBlock &To) const {
110 unsigned NumInstr = 0;
111 const MachineFunction *MF = From.getParent();
112
113 for (MachineFunction::const_iterator MBBI(&From), ToI(&To), End = MF->end();
114 MBBI != End && MBBI != ToI; ++MBBI) {
115 const MachineBasicBlock &MBB = *MBBI;
116
117 for (MachineBasicBlock::const_iterator I = MBB.begin(), E = MBB.end();
118 NumInstr < SkipThreshold && I != E; ++I) {
119 if (opcodeEmitsNoInsts(*I))
120 continue;
121
122 // FIXME: Since this is required for correctness, this should be inserted
123 // during SILowerControlFlow.
124
125 // When a uniform loop is inside non-uniform control flow, the branch
126 // leaving the loop might be an S_CBRANCH_VCCNZ, which is never taken
127 // when EXEC = 0. We should skip the loop lest it becomes infinite.
128 if (I->getOpcode() == AMDGPU::S_CBRANCH_VCCNZ ||
129 I->getOpcode() == AMDGPU::S_CBRANCH_VCCZ)
130 return true;
131
132 if (TII->hasUnwantedEffectsWhenEXECEmpty(*I))
133 return true;
134
135 // These instructions are potentially expensive even if EXEC = 0.
136 if (TII->isSMRD(*I) || TII->isVMEM(*I) || TII->isFLAT(*I) ||
137 I->getOpcode() == AMDGPU::S_WAITCNT)
138 return true;
139
140 ++NumInstr;
141 if (NumInstr >= SkipThreshold)
142 return true;
143 }
144 }
145
146 return false;
147}

149bool SIInsertSkips::skipIfDead(MachineInstr &MI, MachineBasicBlock &NextBB) {
150 MachineBasicBlock &MBB = *MI.getParent();
151 MachineFunction *MF = MBB.getParent();
152
153 if (MF->getFunction().getCallingConv() != CallingConv::AMDGPU_PS ||
154 !shouldSkip(MBB, MBB.getParent()->back()))
155 return false;
156
157 MachineBasicBlock *SkipBB = insertSkipBlock(MBB, MI.getIterator());
158
159 const DebugLoc &DL = MI.getDebugLoc();
160
161 // If the exec mask is non-zero, skip the next two instructions
162 BuildMI(&MBB, DL, TII->get(AMDGPU::S_CBRANCH_EXECNZ))
163 .addMBB(&NextBB);
164
165 MachineBasicBlock::iterator Insert = SkipBB->begin();
166
167 // Exec mask is zero: Export to NULL target...
168 BuildMI(*SkipBB, Insert, DL, TII->get(AMDGPU::EXP_DONE))
169 .addImm(0x09) // V_008DFC_SQ_EXP_NULL
170 .addReg(AMDGPU::VGPR0, RegState::Undef)
171 .addReg(AMDGPU::VGPR0, RegState::Undef)
172 .addReg(AMDGPU::VGPR0, RegState::Undef)
173 .addReg(AMDGPU::VGPR0, RegState::Undef)
174 .addImm(1) // vm
175 .addImm(0) // compr
176 .addImm(0); // en
177
178 // ... and terminate wavefront.
179 BuildMI(*SkipBB, Insert, DL, TII->get(AMDGPU::S_ENDPGM)).addImm(0);
180
181 return true;
182}

184void SIInsertSkips::kill(MachineInstr &MI) {
185 MachineBasicBlock &MBB = *MI.getParent();
186 DebugLoc DL = MI.getDebugLoc();
187
188 switch (MI.getOpcode()) {
189 case AMDGPU::SI_KILL_F32_COND_IMM_TERMINATOR: {
190 unsigned Opcode = 0;
191
192 // The opcodes are inverted because the inline immediate has to be
193 // the first operand, e.g. from "x < imm" to "imm > x"
194 switch (MI.getOperand(2).getImm()) {
195 case ISD::SETOEQ:
196 case ISD::SETEQ:
197 Opcode = AMDGPU::V_CMPX_EQ_F32_e64;
198 break;
199 case ISD::SETOGT:
200 case ISD::SETGT:
201 Opcode = AMDGPU::V_CMPX_LT_F32_e64;
202 break;
203 case ISD::SETOGE:
204 case ISD::SETGE:
205 Opcode = AMDGPU::V_CMPX_LE_F32_e64;
206 break;
207 case ISD::SETOLT:
208 case ISD::SETLT:
209 Opcode = AMDGPU::V_CMPX_GT_F32_e64;
210 break;
211 case ISD::SETOLE:
212 case ISD::SETLE:
213 Opcode = AMDGPU::V_CMPX_GE_F32_e64;
214 break;
215 case ISD::SETONE:
216 case ISD::SETNE:
217 Opcode = AMDGPU::V_CMPX_LG_F32_e64;
218 break;
219 case ISD::SETO:
220 Opcode = AMDGPU::V_CMPX_O_F32_e64;
221 break;
222 case ISD::SETUO:
223 Opcode = AMDGPU::V_CMPX_U_F32_e64;
224 break;
225 case ISD::SETUEQ:
226 Opcode = AMDGPU::V_CMPX_NLG_F32_e64;
227 break;
228 case ISD::SETUGT:
229 Opcode = AMDGPU::V_CMPX_NGE_F32_e64;
230 break;
231 case ISD::SETUGE:
232 Opcode = AMDGPU::V_CMPX_NGT_F32_e64;
233 break;
234 case ISD::SETULT:
235 Opcode = AMDGPU::V_CMPX_NLE_F32_e64;
236 break;
237 case ISD::SETULE:
238 Opcode = AMDGPU::V_CMPX_NLT_F32_e64;
239 break;
240 case ISD::SETUNE:
241 Opcode = AMDGPU::V_CMPX_NEQ_F32_e64;
242 break;
243 default:
244 llvm_unreachable("invalid ISD:SET cond code");
245 }
246
247 const GCNSubtarget &ST = MBB.getParent()->getSubtarget<GCNSubtarget>();
248 if (ST.hasNoSdstCMPX())
249 Opcode = AMDGPU::getVCMPXNoSDstOp(Opcode);
250
251 assert(MI.getOperand(0).isReg());
252
253 if (TRI->isVGPR(MBB.getParent()->getRegInfo(),
254 MI.getOperand(0).getReg())) {
255 Opcode = AMDGPU::getVOPe32(Opcode);
256 BuildMI(MBB, &MI, DL, TII->get(Opcode))
257 .add(MI.getOperand(1))
258 .add(MI.getOperand(0));
259 } else {
260 auto I = BuildMI(MBB, &MI, DL, TII->get(Opcode));
261 if (!ST.hasNoSdstCMPX())
262 I.addReg(AMDGPU::VCC, RegState::Define);
263
264 I.addImm(0) // src0 modifiers
265 .add(MI.getOperand(1))
266 .addImm(0) // src1 modifiers
267 .add(MI.getOperand(0));
268
269 I.addImm(0); // omod
270 }
271 break;
272 }
273 case AMDGPU::SI_KILL_I1_TERMINATOR: {
274 const MachineOperand &Op = MI.getOperand(0);
275 int64_t KillVal = MI.getOperand(1).getImm();
276 assert(KillVal == 0 || KillVal == -1);
277
278 // Kill all threads if Op0 is an immediate and equal to the Kill value.
279 if (Op.isImm()) {
280 int64_t Imm = Op.getImm();
281 assert(Imm == 0 || Imm == -1);
282
283 if (Imm == KillVal)
284 BuildMI(MBB, &MI, DL, TII->get(AMDGPU::S_MOV_B64), AMDGPU::EXEC)
285 .addImm(0);
286 break;
287 }
288
289 unsigned Opcode = KillVal ? AMDGPU::S_ANDN2_B64 : AMDGPU::S_AND_B64;
290 BuildMI(MBB, &MI, DL, TII->get(Opcode), AMDGPU::EXEC)
291 .addReg(AMDGPU::EXEC)
292 .add(Op);
293 break;
294 }
295 default:
296 llvm_unreachable("invalid opcode, expected SI_KILL_*_TERMINATOR");
297 }
298}

300MachineBasicBlock *SIInsertSkips::insertSkipBlock(
301 MachineBasicBlock &MBB, MachineBasicBlock::iterator I) const {
302 MachineFunction *MF = MBB.getParent();
303
304 MachineBasicBlock *SkipBB = MF->CreateMachineBasicBlock();
305 MachineFunction::iterator MBBI(MBB);
306 ++MBBI;
307
308 MF->insert(MBBI, SkipBB);
309 MBB.addSuccessor(SkipBB);
310
311 return SkipBB;
312}

314// Returns true if a branch over the block was inserted.
315bool SIInsertSkips::skipMaskBranch(MachineInstr &MI,
316 MachineBasicBlock &SrcMBB) {
317 MachineBasicBlock *DestBB = MI.getOperand(0).getMBB();
318
319 if (!shouldSkip(**SrcMBB.succ_begin(), *DestBB))
320 return false;
321
322 const DebugLoc &DL = MI.getDebugLoc();
323 MachineBasicBlock::iterator InsPt = std::next(MI.getIterator());
324
325 BuildMI(SrcMBB, InsPt, DL, TII->get(AMDGPU::S_CBRANCH_EXECZ))
326 .addMBB(DestBB);
327
328 return true;
329}

331bool SIInsertSkips::optimizeVccBranch(MachineInstr &MI) const {
332 // Match:
333 // sreg = -1
334 // vcc = S_AND_B64 exec, sreg
335 // S_CBRANCH_VCC[N]Z
336 // =>
337 // S_CBRANCH_EXEC[N]Z
338 bool Changed = false;
339 MachineBasicBlock &MBB = *MI.getParent();
340 const unsigned CondReg = AMDGPU::VCC;
341 const unsigned ExecReg = AMDGPU::EXEC;
342 const unsigned And = AMDGPU::S_AND_B64;
343
344 MachineBasicBlock::reverse_iterator A = MI.getReverseIterator(),
345 E = MBB.rend();
346 bool ReadsCond = false;
347 unsigned Threshold = 5;
348 for (++A ; A != E ; ++A) {
349 if (!--Threshold)
350 return false;
351 if (A->modifiesRegister(ExecReg, TRI))
352 return false;
353 if (A->modifiesRegister(CondReg, TRI)) {
354 if (!A->definesRegister(CondReg, TRI) || A->getOpcode() != And)
355 return false;
356 break;
357 }
358 ReadsCond |= A->readsRegister(CondReg, TRI);
359 }
360 if (A == E)
361 return false;
362
363 MachineOperand &Op1 = A->getOperand(1);
364 MachineOperand &Op2 = A->getOperand(2);
365 if (Op1.getReg() != ExecReg && Op2.isReg() && Op2.getReg() == ExecReg) {
366 TII->commuteInstruction(*A);
367 Changed = true;
368 }
369 if (Op1.getReg() != ExecReg)
370 return Changed;
371 if (Op2.isImm() && Op2.getImm() != -1)
372 return Changed;
373
374 unsigned SReg = AMDGPU::NoRegister;
375 if (Op2.isReg()) {
376 SReg = Op2.getReg();
377 auto M = std::next(A);
378 bool ReadsSreg = false;
379 for ( ; M != E ; ++M) {
380 if (M->definesRegister(SReg, TRI))
381 break;
382 if (M->modifiesRegister(SReg, TRI))
383 return Changed;
384 ReadsSreg |= M->readsRegister(SReg, TRI);
385 }
386 if (M == E ||
387 !M->isMoveImmediate() ||
388 !M->getOperand(1).isImm() ||
389 M->getOperand(1).getImm() != -1)
390 return Changed;
391 // First if sreg is only used in and instruction fold the immediate
392 // into that and.
393 if (!ReadsSreg && Op2.isKill()) {
394 A->getOperand(2).ChangeToImmediate(-1);
395 M->eraseFromParent();
396 }
397 }
398
399 if (!ReadsCond && A->registerDefIsDead(AMDGPU::SCC) &&
400 MI.killsRegister(CondReg, TRI))
401 A->eraseFromParent();
402
403 bool IsVCCZ = MI.getOpcode() == AMDGPU::S_CBRANCH_VCCZ;
404 if (SReg == ExecReg) {
405 if (IsVCCZ) {
406 MI.eraseFromParent();
407 return true;
408 }
409 MI.setDesc(TII->get(AMDGPU::S_BRANCH));
410 } else {
411 MI.setDesc(TII->get(IsVCCZ ? AMDGPU::S_CBRANCH_EXECZ
412 : AMDGPU::S_CBRANCH_EXECNZ));
413 }
414
415 MI.RemoveOperand(MI.findRegisterUseOperandIdx(CondReg, false /*Kill*/, TRI));
416 MI.addImplicitDefUseOperands(*MBB.getParent());
417
418 return true;
419}

421bool SIInsertSkips::runOnMachineFunction(MachineFunction &MF) {
422 const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
423 TII = ST.getInstrInfo();
424 TRI = &TII->getRegisterInfo();
425 SkipThreshold = SkipThresholdFlag;
426
427 bool HaveKill = false;
428 bool MadeChange = false;
429
430 // Track depth of exec mask, divergent branches.
431 SmallVector<MachineBasicBlock *, 16> ExecBranchStack;
432
433 MachineFunction::iterator NextBB;
434
435 MachineBasicBlock *EmptyMBBAtEnd = nullptr;
436
437 for (MachineFunction::iterator BI = MF.begin(), BE = MF.end();
438 BI != BE; BI = NextBB) {
439 NextBB = std::next(BI);
440 MachineBasicBlock &MBB = *BI;
441 bool HaveSkipBlock = false;
442
443 if (!ExecBranchStack.empty() && ExecBranchStack.back() == &MBB) {
444 // Reached convergence point for last divergent branch.
445 ExecBranchStack.pop_back();
446 }
447
448 if (HaveKill && ExecBranchStack.empty()) {
449 HaveKill = false;
450
451 // TODO: Insert skip if exec is 0?
452 }
453
454 MachineBasicBlock::iterator I, Next;
455 for (I = MBB.begin(); I != MBB.end(); I = Next) {
456 Next = std::next(I);
457
458 MachineInstr &MI = *I;
459
460 switch (MI.getOpcode()) {
461 case AMDGPU::SI_MASK_BRANCH:
462 ExecBranchStack.push_back(MI.getOperand(0).getMBB());
463 MadeChange |= skipMaskBranch(MI, MBB);
464 break;
465
466 case AMDGPU::S_BRANCH:
467 // Optimize out branches to the next block.
468 // FIXME: Shouldn't this be handled by BranchFolding?
469 if (MBB.isLayoutSuccessor(MI.getOperand(0).getMBB())) {
470 MI.eraseFromParent();
471 } else if (HaveSkipBlock) {
472 // Remove the given unconditional branch when a skip block has been
473 // inserted after the current one and let skip the two instructions
474 // performing the kill if the exec mask is non-zero.
475 MI.eraseFromParent();
476 }
477 break;
478
479 case AMDGPU::SI_KILL_F32_COND_IMM_TERMINATOR:
480 case AMDGPU::SI_KILL_I1_TERMINATOR:
481 MadeChange = true;
482 kill(MI);
483
484 if (ExecBranchStack.empty()) {
485 if (NextBB != BE && skipIfDead(MI, *NextBB)) {
486 HaveSkipBlock = true;
487 NextBB = std::next(BI);
488 BE = MF.end();
489 }
490 } else {
491 HaveKill = true;
492 }
493
494 MI.eraseFromParent();
495 break;
496
497 case AMDGPU::SI_RETURN_TO_EPILOG:
498 // FIXME: Should move somewhere else
499 assert(!MF.getInfo<SIMachineFunctionInfo>()->returnsVoid());
500
501 // Graphics shaders returning non-void shouldn't contain S_ENDPGM,
502 // because external bytecode will be appended at the end.
503 if (BI != --MF.end() || I != MBB.getFirstTerminator()) {
504 // SI_RETURN_TO_EPILOG is not the last instruction. Add an empty block at
505 // the end and jump there.
506 if (!EmptyMBBAtEnd) {
507 EmptyMBBAtEnd = MF.CreateMachineBasicBlock();
508 MF.insert(MF.end(), EmptyMBBAtEnd);
509 }
510
511 MBB.addSuccessor(EmptyMBBAtEnd);
512 BuildMI(*BI, I, MI.getDebugLoc(), TII->get(AMDGPU::S_BRANCH))
513 .addMBB(EmptyMBBAtEnd);
514 I->eraseFromParent();
515 }
516 break;
517
518 case AMDGPU::S_CBRANCH_VCCZ:
519 case AMDGPU::S_CBRANCH_VCCNZ:
520 MadeChange |= optimizeVccBranch(MI);
521 break;
522
523 default:
524 break;
525 }
526 }
527 }
528
529 return MadeChange;
