47void AMDGPUInstPrinter::printU16ImmOperand(const MCInst *MI, unsigned OpNo,
48 const MCSubtargetInfo &STI,
49 raw_ostream &O) {
50 // It's possible to end up with a 32-bit literal used with a 16-bit operand
51 // with ignored high bits. Print as 32-bit anyway in that case.
52 int64_t Imm = MI->getOperand(OpNo).getImm();
53 if (isInt<16>(Imm) || isUInt<16>(Imm))
54 O << formatHex(static_cast<uint64_t>(Imm & 0xffff));
55 else
56 printU32ImmOperand(MI, OpNo, STI, O);
57}

278void AMDGPUInstPrinter::printRegOperand(unsigned RegNo, raw_ostream &O,
279 const MCRegisterInfo &MRI) {
280 switch (RegNo) {
281 case AMDGPU::VCC:
282 O << "vcc";
283 return;
284 case AMDGPU::SRC_VCCZ:
285 O << "src_vccz";
286 return;
287 case AMDGPU::SRC_EXECZ:
288 O << "src_execz";
289 return;
290 case AMDGPU::SRC_SCC:
291 O << "src_scc";
292 return;
293 case AMDGPU::EXEC:
294 O << "exec";
295 return;
296 case AMDGPU::M0:
297 O << "m0";
298 return;
299 case AMDGPU::SGPR_NULL:
300 O << "null";
301 return;
302 case AMDGPU::FLAT_SCR:
303 O << "flat_scratch";
304 return;
305 case AMDGPU::XNACK_MASK:
306 O << "xnack_mask";
307 return;
308 case AMDGPU::SRC_SHARED_BASE:
309 O << "src_shared_base";
310 return;
311 case AMDGPU::SRC_SHARED_LIMIT:
312 O << "src_shared_limit";
313 return;
314 case AMDGPU::SRC_PRIVATE_BASE:
315 O << "src_private_base";
316 return;
317 case AMDGPU::SRC_PRIVATE_LIMIT:
318 O << "src_private_limit";
319 return;
320 case AMDGPU::SRC_POPS_EXITING_WAVE_ID:
321 O << "src_pops_exiting_wave_id";
322 return;
323 case AMDGPU::LDS_DIRECT:
324 O << "src_lds_direct";
325 return;
326 case AMDGPU::VCC_LO:
327 O << "vcc_lo";
328 return;
329 case AMDGPU::VCC_HI:
330 O << "vcc_hi";
331 return;
332 case AMDGPU::TBA_LO:
333 O << "tba_lo";
334 return;
335 case AMDGPU::TBA_HI:
336 O << "tba_hi";
337 return;
338 case AMDGPU::TMA_LO:
339 O << "tma_lo";
340 return;
341 case AMDGPU::TMA_HI:
342 O << "tma_hi";
343 return;
344 case AMDGPU::EXEC_LO:
345 O << "exec_lo";
346 return;
347 case AMDGPU::EXEC_HI:
348 O << "exec_hi";
349 return;
350 case AMDGPU::FLAT_SCR_LO:
351 O << "flat_scratch_lo";
352 return;
353 case AMDGPU::FLAT_SCR_HI:
354 O << "flat_scratch_hi";
355 return;
356 case AMDGPU::XNACK_MASK_LO:
357 O << "xnack_mask_lo";
358 return;
359 case AMDGPU::XNACK_MASK_HI:
360 O << "xnack_mask_hi";
361 return;
362 case AMDGPU::FP_REG:
363 case AMDGPU::SP_REG:
364 case AMDGPU::SCRATCH_WAVE_OFFSET_REG:
365 case AMDGPU::PRIVATE_RSRC_REG:
366 llvm_unreachable("pseudo-register should not ever be emitted");
367 case AMDGPU::SCC:
368 llvm_unreachable("pseudo scc should not ever be emitted");
369 default:
370 break;
371 }
372
373 // The low 8 bits of the encoding value is the register index, for both VGPRs
374 // and SGPRs.
375 unsigned RegIdx = MRI.getEncodingValue(RegNo) & ((1 << 8) - 1);
376
377 unsigned NumRegs;
378 if (MRI.getRegClass(AMDGPU::VGPR_32RegClassID).contains(RegNo)) {
379 O << 'v';
380 NumRegs = 1;
381 } else if (MRI.getRegClass(AMDGPU::SGPR_32RegClassID).contains(RegNo)) {
382 O << 's';
383 NumRegs = 1;
384 } else if (MRI.getRegClass(AMDGPU::VReg_64RegClassID).contains(RegNo)) {
385 O <<'v';
386 NumRegs = 2;
387 } else if (MRI.getRegClass(AMDGPU::SGPR_64RegClassID).contains(RegNo)) {
388 O << 's';
389 NumRegs = 2;
390 } else if (MRI.getRegClass(AMDGPU::VReg_128RegClassID).contains(RegNo)) {
391 O << 'v';
392 NumRegs = 4;
393 } else if (MRI.getRegClass(AMDGPU::SGPR_128RegClassID).contains(RegNo)) {
394 O << 's';
395 NumRegs = 4;
396 } else if (MRI.getRegClass(AMDGPU::VReg_96RegClassID).contains(RegNo)) {
397 O << 'v';
398 NumRegs = 3;
399 } else if (MRI.getRegClass(AMDGPU::SReg_96RegClassID).contains(RegNo)) {
400 O << 's';
401 NumRegs = 3;
402 } else if (MRI.getRegClass(AMDGPU::VReg_160RegClassID).contains(RegNo)) {
403 O << 'v';
404 NumRegs = 5;
405 } else if (MRI.getRegClass(AMDGPU::VReg_256RegClassID).contains(RegNo)) {
406 O << 'v';
407 NumRegs = 8;
408 } else if (MRI.getRegClass(AMDGPU::SGPR_256RegClassID).contains(RegNo)) {
409 O << 's';
410 NumRegs = 8;
411 } else if (MRI.getRegClass(AMDGPU::VReg_512RegClassID).contains(RegNo)) {
412 O << 'v';
413 NumRegs = 16;
414 } else if (MRI.getRegClass(AMDGPU::SGPR_512RegClassID).contains(RegNo)) {
415 O << 's';
416 NumRegs = 16;
417 } else {
418 O << getRegisterName(RegNo);
419 return;
420 }
421
422 if (NumRegs == 1) {
423 O << RegIdx;
424 return;
425 }
426
427 O << '[' << RegIdx << ':' << (RegIdx + NumRegs - 1) << ']';
428}

430void AMDGPUInstPrinter::printVOPDst(const MCInst *MI, unsigned OpNo,
431 const MCSubtargetInfo &STI, raw_ostream &O) {
432 if (OpNo == 0) {
433 if (MII.get(MI->getOpcode()).TSFlags & SIInstrFlags::VOP3)
434 O << "_e64 ";
435 else if (MII.get(MI->getOpcode()).TSFlags & SIInstrFlags::DPP)
436 O << "_dpp ";
437 else if (MII.get(MI->getOpcode()).TSFlags & SIInstrFlags::SDWA)
438 O << "_sdwa ";
439 else
440 O << "_e32 ";
441 }
442
443 printOperand(MI, OpNo, STI, O);
444
445 switch (MI->getOpcode()) {
446 default: break;
447
448 case AMDGPU::V_ADD_CO_CI_U32_e32_gfx10:
449 case AMDGPU::V_SUB_CO_CI_U32_e32_gfx10:
450 case AMDGPU::V_SUBREV_CO_CI_U32_e32_gfx10:
451 case AMDGPU::V_ADD_CO_CI_U32_sdwa_gfx10:
452 case AMDGPU::V_SUB_CO_CI_U32_sdwa_gfx10:
453 case AMDGPU::V_SUBREV_CO_CI_U32_sdwa_gfx10:
454 case AMDGPU::V_ADD_CO_CI_U32_dpp_gfx10:
455 case AMDGPU::V_SUB_CO_CI_U32_dpp_gfx10:
456 case AMDGPU::V_SUBREV_CO_CI_U32_dpp_gfx10:
457 case AMDGPU::V_ADD_CO_CI_U32_dpp8_gfx10:
458 case AMDGPU::V_SUB_CO_CI_U32_dpp8_gfx10:
459 case AMDGPU::V_SUBREV_CO_CI_U32_dpp8_gfx10:
460 printDefaultVccOperand(1, STI, O);
461 break;
462 }
463}

475void AMDGPUInstPrinter::printImmediate16(uint32_t Imm,
476 const MCSubtargetInfo &STI,
477 raw_ostream &O) {
478 int16_t SImm = static_cast<int16_t>(Imm);
479 if (SImm >= -16 && SImm <= 64) {
480 O << SImm;
481 return;
482 }
483
484 if (Imm == 0x3C00)
485 O<< "1.0";
486 else if (Imm == 0xBC00)
487 O<< "-1.0";
488 else if (Imm == 0x3800)
489 O<< "0.5";
490 else if (Imm == 0xB800)
491 O<< "-0.5";
492 else if (Imm == 0x4000)
493 O<< "2.0";
494 else if (Imm == 0xC000)
495 O<< "-2.0";
496 else if (Imm == 0x4400)
497 O<< "4.0";
498 else if (Imm == 0xC400)
499 O<< "-4.0";
500 else if (Imm == 0x3118) {
501 assert(STI.getFeatureBits()[AMDGPU::FeatureInv2PiInlineImm]);
502 O << "0.15915494";
503 } else
504 O << formatHex(static_cast<uint64_t>(Imm));
505}

514void AMDGPUInstPrinter::printImmediate32(uint32_t Imm,
515 const MCSubtargetInfo &STI,
516 raw_ostream &O) {
517 int32_t SImm = static_cast<int32_t>(Imm);
518 if (SImm >= -16 && SImm <= 64) {
519 O << SImm;
520 return;
521 }
522
523 if (Imm == FloatToBits(0.0f))
524 O << "0.0";
525 else if (Imm == FloatToBits(1.0f))
526 O << "1.0";
527 else if (Imm == FloatToBits(-1.0f))
528 O << "-1.0";
529 else if (Imm == FloatToBits(0.5f))
530 O << "0.5";
531 else if (Imm == FloatToBits(-0.5f))
532 O << "-0.5";
533 else if (Imm == FloatToBits(2.0f))
534 O << "2.0";
535 else if (Imm == FloatToBits(-2.0f))
536 O << "-2.0";
537 else if (Imm == FloatToBits(4.0f))
538 O << "4.0";
539 else if (Imm == FloatToBits(-4.0f))
540 O << "-4.0";
541 else if (Imm == 0x3e22f983 &&
542 STI.getFeatureBits()[AMDGPU::FeatureInv2PiInlineImm])
543 O << "0.15915494";
544 else
545 O << formatHex(static_cast<uint64_t>(Imm));
546}

548void AMDGPUInstPrinter::printImmediate64(uint64_t Imm,
549 const MCSubtargetInfo &STI,
550 raw_ostream &O) {
551 int64_t SImm = static_cast<int64_t>(Imm);
552 if (SImm >= -16 && SImm <= 64) {
553 O << SImm;
554 return;
555 }
556
557 if (Imm == DoubleToBits(0.0))
558 O << "0.0";
559 else if (Imm == DoubleToBits(1.0))
560 O << "1.0";
561 else if (Imm == DoubleToBits(-1.0))
562 O << "-1.0";
563 else if (Imm == DoubleToBits(0.5))
564 O << "0.5";
565 else if (Imm == DoubleToBits(-0.5))
566 O << "-0.5";
567 else if (Imm == DoubleToBits(2.0))
568 O << "2.0";
569 else if (Imm == DoubleToBits(-2.0))
570 O << "-2.0";
571 else if (Imm == DoubleToBits(4.0))
572 O << "4.0";
573 else if (Imm == DoubleToBits(-4.0))
574 O << "-4.0";
575 else if (Imm == 0x3fc45f306dc9c882 &&
576 STI.getFeatureBits()[AMDGPU::FeatureInv2PiInlineImm])
577 O << "0.15915494309189532";
578 else {
579 assert(isUInt<32>(Imm) || Imm == 0x3fc45f306dc9c882);
580
581 // In rare situations, we will have a 32-bit literal in a 64-bit
582 // operand. This is technically allowed for the encoding of s_mov_b64.
583 O << formatHex(static_cast<uint64_t>(Imm));
584 }

597void AMDGPUInstPrinter::printOperand(const MCInst *MI, unsigned OpNo,
598 const MCSubtargetInfo &STI,
599 raw_ostream &O) {
600 const MCInstrDesc &Desc = MII.get(MI->getOpcode());
601 if (OpNo == 0 && (Desc.TSFlags & SIInstrFlags::VOPC) &&
602 (Desc.hasImplicitDefOfPhysReg(AMDGPU::VCC) ||
603 Desc.hasImplicitDefOfPhysReg(AMDGPU::VCC_LO)))
604 printDefaultVccOperand(OpNo, STI, O);
605
606 if (OpNo >= MI->getNumOperands()) {
607 O << "/*Missing OP" << OpNo << "*/";
608 return;
609 }
610
611 const MCOperand &Op = MI->getOperand(OpNo);
612 if (Op.isReg()) {
613 printRegOperand(Op.getReg(), O, MRI);
614 } else if (Op.isImm()) {
615 switch (Desc.OpInfo[OpNo].OperandType) {
616 case AMDGPU::OPERAND_REG_IMM_INT32:
617 case AMDGPU::OPERAND_REG_IMM_FP32:
618 case AMDGPU::OPERAND_REG_INLINE_C_INT32:
619 case AMDGPU::OPERAND_REG_INLINE_C_FP32:
620 case MCOI::OPERAND_IMMEDIATE:
621 printImmediate32(Op.getImm(), STI, O);
622 break;
623 case AMDGPU::OPERAND_REG_IMM_INT64:
624 case AMDGPU::OPERAND_REG_IMM_FP64:
625 case AMDGPU::OPERAND_REG_INLINE_C_INT64:
626 case AMDGPU::OPERAND_REG_INLINE_C_FP64:
627 printImmediate64(Op.getImm(), STI, O);
628 break;
629 case AMDGPU::OPERAND_REG_INLINE_C_INT16:
630 case AMDGPU::OPERAND_REG_INLINE_C_FP16:
631 case AMDGPU::OPERAND_REG_IMM_INT16:
632 case AMDGPU::OPERAND_REG_IMM_FP16:
633 printImmediate16(Op.getImm(), STI, O);
634 break;
635 case AMDGPU::OPERAND_REG_IMM_V2INT16:
636 case AMDGPU::OPERAND_REG_IMM_V2FP16:
637 if (!isUInt<16>(Op.getImm()) &&
638 STI.getFeatureBits()[AMDGPU::FeatureVOP3Literal]) {
639 printImmediate32(Op.getImm(), STI, O);
640 break;
641 }
642 LLVM_FALLTHROUGH;
643 case AMDGPU::OPERAND_REG_INLINE_C_V2FP16:
644 case AMDGPU::OPERAND_REG_INLINE_C_V2INT16:
645 printImmediateV216(Op.getImm(), STI, O);
646 break;
647 case MCOI::OPERAND_UNKNOWN:
648 case MCOI::OPERAND_PCREL:
649 O << formatDec(Op.getImm());
650 break;
651 case MCOI::OPERAND_REGISTER:
652 // FIXME: This should be removed and handled somewhere else. Seems to come
653 // from a disassembler bug.
654 O << "/*invalid immediate*/";
655 break;
656 default:
657 // We hit this for the immediate instruction bits that don't yet have a
658 // custom printer.
659 llvm_unreachable("unexpected immediate operand type");
660 }
661 } else if (Op.isFPImm()) {
662 // We special case 0.0 because otherwise it will be printed as an integer.
663 if (Op.getFPImm() == 0.0)
664 O << "0.0";
665 else {
666 const MCInstrDesc &Desc = MII.get(MI->getOpcode());
667 int RCID = Desc.OpInfo[OpNo].RegClass;
668 unsigned RCBits = AMDGPU::getRegBitWidth(MRI.getRegClass(RCID));
669 if (RCBits == 32)
670 printImmediate32(FloatToBits(Op.getFPImm()), STI, O);
671 else if (RCBits == 64)
672 printImmediate64(DoubleToBits(Op.getFPImm()), STI, O);
673 else
674 llvm_unreachable("Invalid register class size");
675 }
676 } else if (Op.isExpr()) {
677 const MCExpr *Exp = Op.getExpr();
678 Exp->print(O, &MAI);
679 } else {
680 O << "/*INV_OP*/";
681 }
682
683 switch (MI->getOpcode()) {
684 default: break;
685
686 case AMDGPU::V_CNDMASK_B32_e32_gfx10:
687 case AMDGPU::V_ADD_CO_CI_U32_e32_gfx10:
688 case AMDGPU::V_SUB_CO_CI_U32_e32_gfx10:
689 case AMDGPU::V_SUBREV_CO_CI_U32_e32_gfx10:
690 case AMDGPU::V_ADD_CO_CI_U32_dpp_gfx10:
691 case AMDGPU::V_SUB_CO_CI_U32_dpp_gfx10:
692 case AMDGPU::V_SUBREV_CO_CI_U32_dpp_gfx10:
693 case AMDGPU::V_ADD_CO_CI_U32_dpp8_gfx10:
694 case AMDGPU::V_SUB_CO_CI_U32_dpp8_gfx10:
695 case AMDGPU::V_SUBREV_CO_CI_U32_dpp8_gfx10:
696
697 case AMDGPU::V_CNDMASK_B32_e32_gfx6_gfx7:
698 case AMDGPU::V_CNDMASK_B32_e32_vi:
699 if ((int)OpNo == AMDGPU::getNamedOperandIdx(MI->getOpcode(),
700 AMDGPU::OpName::src1))
701 printDefaultVccOperand(OpNo, STI, O);
702 break;
703 }
704}

1022static bool allOpsDefaultValue(const int* Ops, int NumOps, int Mod,
1023 bool IsPacked, bool HasDstSel) {
1024 int DefaultValue = IsPacked && (Mod == SISrcMods::OP_SEL_1);
1025
1026 for (int I = 0; I < NumOps; ++I) {
1027 if (!!(Ops[I] & Mod) != DefaultValue)
1028 return false;
1029 }
1030
1031 if (HasDstSel && (Ops[0] & SISrcMods::DST_OP_SEL) != 0)
1032 return false;
1033
1034 return true;
1035}

1037void AMDGPUInstPrinter::printPackedModifier(const MCInst *MI,
1038 StringRef Name,
1039 unsigned Mod,
1040 raw_ostream &O) {
1041 unsigned Opc = MI->getOpcode();
1042 int NumOps = 0;
1043 int Ops[3];
1044
1045 for (int OpName : { AMDGPU::OpName::src0_modifiers,
1046 AMDGPU::OpName::src1_modifiers,
1047 AMDGPU::OpName::src2_modifiers }) {
1048 int Idx = AMDGPU::getNamedOperandIdx(Opc, OpName);
1049 if (Idx == -1)
1050 break;
1051
1052 Ops[NumOps++] = MI->getOperand(Idx).getImm();
1053 }
1054
1055 const bool HasDstSel =
1056 NumOps > 0 &&
1057 Mod == SISrcMods::OP_SEL_0 &&
1058 MII.get(MI->getOpcode()).TSFlags & SIInstrFlags::VOP3_OPSEL;
1059
1060 const bool IsPacked =
1061 MII.get(MI->getOpcode()).TSFlags & SIInstrFlags::IsPacked;
1062
1063 if (allOpsDefaultValue(Ops, NumOps, Mod, IsPacked, HasDstSel))
1064 return;
1065
1066 O << Name;
1067 for (int I = 0; I < NumOps; ++I) {
1068 if (I != 0)
1069 O << ',';
1070
1071 O << !!(Ops[I] & Mod);
1072 }
1073
1074 if (HasDstSel) {
1075 O << ',' << !!(Ops[0] & SISrcMods::DST_OP_SEL);
1076 }
1077
1078 O << ']';
1079}

1227void AMDGPUInstPrinter::printSendMsg(const MCInst *MI, unsigned OpNo,
1228 const MCSubtargetInfo &STI,

1233 const unsigned Id = SImm16 & ID_MASK_;
1234 do {
1235 if (Id == ID_INTERRUPT ||
1236 (Id == ID_GS_ALLOC_REQ && !AMDGPU::isSI(STI) && !AMDGPU::isCI(STI) &&
1237 !AMDGPU::isVI(STI))) {
1238 if ((SImm16 & ~ID_MASK_) != 0) // Unused/unknown bits must be 0.
1239 break;
1240 O << "sendmsg(" << IdSymbolic[Id] << ')';
1241 return;
1242 }
1243 if (Id == ID_GS || Id == ID_GS_DONE) {
1244 if ((SImm16 & ~(ID_MASK_|OP_GS_MASK_|STREAM_ID_MASK_)) != 0) // Unused/unknown bits must be 0.
1245 break;
1246 const unsigned OpGs = (SImm16 & OP_GS_MASK_) >> OP_SHIFT_;
1247 const unsigned StreamId = (SImm16 & STREAM_ID_MASK_) >> STREAM_ID_SHIFT_;
1248 if (OpGs == OP_GS_NOP && Id != ID_GS_DONE) // NOP to be used for GS_DONE only.
1249 break;
1250 if (OpGs == OP_GS_NOP && StreamId != 0) // NOP does not use/define stream id bits.
1251 break;
1252 O << "sendmsg(" << IdSymbolic[Id] << ", " << OpGsSymbolic[OpGs];
1253 if (OpGs != OP_GS_NOP) { O << ", " << StreamId; }
1254 O << ')';
1255 return;
1256 }
1257 if (Id == ID_SYSMSG) {
1258 if ((SImm16 & ~(ID_MASK_|OP_SYS_MASK_)) != 0) // Unused/unknown bits must be 0.
1259 break;
1260 const unsigned OpSys = (SImm16 & OP_SYS_MASK_) >> OP_SHIFT_;
1261 if (! (OP_SYS_FIRST_ <= OpSys && OpSys < OP_SYS_LAST_)) // Unused/unknown.
1262 break;
1263 O << "sendmsg(" << IdSymbolic[Id] << ", " << OpSysSymbolic[OpSys] << ')';
1264 return;
1265 }
1266 } while (false);
1267 O << SImm16; // Unknown simm16 code.
1268}

1270static void printSwizzleBitmask(const uint16_t AndMask,
1271 const uint16_t OrMask,
1272 const uint16_t XorMask,
1273 raw_ostream &O) {
1274 using namespace llvm::AMDGPU::Swizzle;
1275
1276 uint16_t Probe0 = ((0 & AndMask) | OrMask) ^ XorMask;
1277 uint16_t Probe1 = ((BITMASK_MASK & AndMask) | OrMask) ^ XorMask;
1278
1279 O << "\"";
1280
1281 for (unsigned Mask = 1 << (BITMASK_WIDTH - 1); Mask > 0; Mask >>= 1) {
1282 uint16_t p0 = Probe0 & Mask;
1283 uint16_t p1 = Probe1 & Mask;
1284
1285 if (p0 == p1) {
1286 if (p0 == 0) {
1287 O << "0";
1288 } else {
1289 O << "1";
1290 }
1291 } else {
1292 if (p0 == 0) {
1293 O << "p";
1294 } else {
1295 O << "i";
1296 }
1297 }
1298 }
1299
1300 O << "\"";
1301}

1303void AMDGPUInstPrinter::printSwizzle(const MCInst *MI, unsigned OpNo,
1304 const MCSubtargetInfo &STI,
1305 raw_ostream &O) {
1306 using namespace llvm::AMDGPU::Swizzle;
1307
1308 uint16_t Imm = MI->getOperand(OpNo).getImm();
1309 if (Imm == 0) {
1310 return;
1311 }
1312
1313 O << " offset:";
1314
1315 if ((Imm & QUAD_PERM_ENC_MASK) == QUAD_PERM_ENC) {
1316
1317 O << "swizzle(" << IdSymbolic[ID_QUAD_PERM];
1318 for (unsigned I = 0; I < LANE_NUM; ++I) {
1319 O << ",";
1320 O << formatDec(Imm & LANE_MASK);
1321 Imm >>= LANE_SHIFT;
1322 }
1323 O << ")";
1324
1325 } else if ((Imm & BITMASK_PERM_ENC_MASK) == BITMASK_PERM_ENC) {
1326
1327 uint16_t AndMask = (Imm >> BITMASK_AND_SHIFT) & BITMASK_MASK;
1328 uint16_t OrMask = (Imm >> BITMASK_OR_SHIFT) & BITMASK_MASK;
1329 uint16_t XorMask = (Imm >> BITMASK_XOR_SHIFT) & BITMASK_MASK;
1330
1331 if (AndMask == BITMASK_MAX &&
1332 OrMask == 0 &&
1333 countPopulation(XorMask) == 1) {
1334
1335 O << "swizzle(" << IdSymbolic[ID_SWAP];
1336 O << ",";
1337 O << formatDec(XorMask);
1338 O << ")";
1339
1340 } else if (AndMask == BITMASK_MAX &&
1341 OrMask == 0 && XorMask > 0 &&
1342 isPowerOf2_64(XorMask + 1)) {
1343
1344 O << "swizzle(" << IdSymbolic[ID_REVERSE];
1345 O << ",";
1346 O << formatDec(XorMask + 1);
1347 O << ")";
1348
1349 } else {
1350
1351 uint16_t GroupSize = BITMASK_MAX - AndMask + 1;
1352 if (GroupSize > 1 &&
1353 isPowerOf2_64(GroupSize) &&
1354 OrMask < GroupSize &&
1355 XorMask == 0) {
1356
1357 O << "swizzle(" << IdSymbolic[ID_BROADCAST];
1358 O << ",";
1359 O << formatDec(GroupSize);
1360 O << ",";
1361 O << formatDec(OrMask);
1362 O << ")";
1363
1364 } else {
1365 O << "swizzle(" << IdSymbolic[ID_BITMASK_PERM];
1366 O << ",";
1367 printSwizzleBitmask(AndMask, OrMask, XorMask, O);
1368 O << ")";
1369 }
1370 }
1371 } else {
1372 printU16ImmDecOperand(MI, OpNo, O);
1373 }
1374}
