46R600Subtarget::initializeSubtargetDependencies(const Triple &TT,
47 StringRef GPU, StringRef FS) {
48 SmallString<256> FullFS("+promote-alloca,");
49 FullFS += FS;
50 ParseSubtargetFeatures(GPU, FullFS);
51
52 // FIXME: I don't think think Evergreen has any useful support for
53 // denormals, but should be checked. Should we issue a warning somewhere
54 // if someone tries to enable these?

56 FP32Denormals = false;

58
59 HasMulU24 = getGeneration() >= EVERGREEN;
60 HasMulI24 = hasCaymanISA();
61
62 return *this;
63}

65GCNSubtarget &
66GCNSubtarget::initializeSubtargetDependencies(const Triple &TT,
67 StringRef GPU, StringRef FS) {
68 // Determine default and user-specified characteristics
69 // On SI+, we want FP64 denormals to be on by default. FP32 denormals can be
70 // enabled, but some instructions do not respect them and they run at the
71 // double precision rate, so don't enable by default.
72 //
73 // We want to be able to turn these off, but making this a subtarget feature
74 // for SI has the unhelpful behavior that it unsets everything else if you
75 // disable it.
76 //
77 // Similarly we want enable-prt-strict-null to be on by default and not to
78 // unset everything else if it is disabled
79
80 // Assuming ECC is enabled is the conservative default.
81 SmallString<256> FullFS("+promote-alloca,+load-store-opt,+sram-ecc,+xnack,");
82
83 if (isAmdHsaOS()) // Turn on FlatForGlobal for HSA.
84 FullFS += "+flat-for-global,+unaligned-buffer-access,+trap-handler,";
85
86 // FIXME: I don't think think Evergreen has any useful support for
87 // denormals, but should be checked. Should we issue a warning somewhere
88 // if someone tries to enable these?
89 if (getGeneration() >= AMDGPUSubtarget::SOUTHERN_ISLANDS) {
90 FullFS += "+fp64-fp16-denormals,";
91 } else {
92 FullFS += "-fp32-denormals,";
93 }
94
95 FullFS += "+enable-prt-strict-null,"; // This is overridden by a disable in FS
96
97 FullFS += FS;
98
99 ParseSubtargetFeatures(GPU, FullFS);
100
101 // We don't support FP64 for EG/NI atm.
102 assert(!hasFP64() || (getGeneration() >= AMDGPUSubtarget::SOUTHERN_ISLANDS));
103
104 // Unless +-flat-for-global is specified, turn on FlatForGlobal for all OS-es
105 // on VI and newer hardware to avoid assertion failures due to missing ADDR64
106 // variants of MUBUF instructions.
107 if (!hasAddr64() && !FS.contains("flat-for-global")) {
108 FlatForGlobal = true;
109 }
110
111 // Set defaults if needed.
112 if (MaxPrivateElementSize == 0)
113 MaxPrivateElementSize = 4;
114
115 if (LDSBankCount == 0)
116 LDSBankCount = 32;
117
118 if (TT.getArch() == Triple::amdgcn) {
119 if (LocalMemorySize == 0)
120 LocalMemorySize = 32768;
121
122 // Do something sensible for unspecified target.
123 if (!HasMovrel && !HasVGPRIndexMode)
124 HasMovrel = true;
125 }
126
127 // Don't crash on invalid devices.
128 if (WavefrontSize == 0)
129 WavefrontSize = 64;
130
131 HasFminFmaxLegacy = getGeneration() < AMDGPUSubtarget::VOLCANIC_ISLANDS;
132
133 if (DoesNotSupportXNACK && EnableXNACK) {
134 ToggleFeature(AMDGPU::FeatureXNACK);
135 EnableXNACK = false;
136 }
137
138 // ECC is on by default, but turn it off if the hardware doesn't support it
139 // anyway. This matters for the gfx9 targets with d16 loads, but don't support
140 // ECC.
141 if (DoesNotSupportSRAMECC && EnableSRAMECC) {
142 ToggleFeature(AMDGPU::FeatureSRAMECC);
143 EnableSRAMECC = false;
144 }
145
146 return *this;
147}

285unsigned AMDGPUSubtarget::getMaxLocalMemSizeWithWaveCount(unsigned NWaves,
286 const Function &F) const {
287 if (NWaves == 1)
288 return getLocalMemorySize();
289 unsigned WorkGroupSize = getFlatWorkGroupSizes(F).second;
290 unsigned WorkGroupsPerCu = getMaxWorkGroupsPerCU(WorkGroupSize);
291 if (!WorkGroupsPerCu)
292 return 0;
293 unsigned MaxWaves = getMaxWavesPerEU();
294 return getLocalMemorySize() * MaxWaves / WorkGroupsPerCu / NWaves;
295}
296
297unsigned AMDGPUSubtarget::getOccupancyWithLocalMemSize(uint32_t Bytes,
298 const Function &F) const {
299 unsigned WorkGroupSize = getFlatWorkGroupSizes(F).second;
300 unsigned WorkGroupsPerCu = getMaxWorkGroupsPerCU(WorkGroupSize);
301 if (!WorkGroupsPerCu)
302 return 0;
303 unsigned MaxWaves = getMaxWavesPerEU();
304 unsigned Limit = getLocalMemorySize() * MaxWaves / WorkGroupsPerCu;
305 unsigned NumWaves = Limit / (Bytes ? Bytes : 1u);
306 NumWaves = std::min(NumWaves, MaxWaves);
307 NumWaves = std::max(NumWaves, 1u);
308 return NumWaves;
309}

317std::pair<unsigned, unsigned>
318AMDGPUSubtarget::getDefaultFlatWorkGroupSize(CallingConv::ID CC) const {
319 switch (CC) {
320 case CallingConv::AMDGPU_CS:
321 case CallingConv::AMDGPU_KERNEL:
322 case CallingConv::SPIR_KERNEL:
323 return std::make_pair(getWavefrontSize() * 2,
324 std::max(getWavefrontSize() * 4, 256u));
325 case CallingConv::AMDGPU_VS:
326 case CallingConv::AMDGPU_LS:
327 case CallingConv::AMDGPU_HS:
328 case CallingConv::AMDGPU_ES:
329 case CallingConv::AMDGPU_GS:
330 case CallingConv::AMDGPU_PS:
331 return std::make_pair(1, getWavefrontSize());
332 default:
333 return std::make_pair(1, 16 * getWavefrontSize());
334 }
335}

337std::pair<unsigned, unsigned> AMDGPUSubtarget::getFlatWorkGroupSizes(
338 const Function &F) const {
339 // FIXME: 1024 if function.
340 // Default minimum/maximum flat work group sizes.
341 std::pair<unsigned, unsigned> Default =
342 getDefaultFlatWorkGroupSize(F.getCallingConv());
343
344 // Requested minimum/maximum flat work group sizes.
345 std::pair<unsigned, unsigned> Requested = AMDGPU::getIntegerPairAttribute(
346 F, "amdgpu-flat-work-group-size", Default);
347
348 // Make sure requested minimum is less than requested maximum.
349 if (Requested.first > Requested.second)
350 return Default;
351
352 // Make sure requested values do not violate subtarget's specifications.
353 if (Requested.first < getMinFlatWorkGroupSize())
354 return Default;
355 if (Requested.second > getMaxFlatWorkGroupSize())
356 return Default;
357
358 return Requested;
359}

361std::pair<unsigned, unsigned> AMDGPUSubtarget::getWavesPerEU(
362 const Function &F) const {
363 // Default minimum/maximum number of waves per execution unit.
364 std::pair<unsigned, unsigned> Default(1, getMaxWavesPerEU());
365
366 // Default/requested minimum/maximum flat work group sizes.
367 std::pair<unsigned, unsigned> FlatWorkGroupSizes = getFlatWorkGroupSizes(F);
368
369 // If minimum/maximum flat work group sizes were explicitly requested using
370 // "amdgpu-flat-work-group-size" attribute, then set default minimum/maximum
371 // number of waves per execution unit to values implied by requested
372 // minimum/maximum flat work group sizes.
373 unsigned MinImpliedByFlatWorkGroupSize =
374 getMaxWavesPerEU(FlatWorkGroupSizes.second);
375 bool RequestedFlatWorkGroupSize = false;
376
377 if (F.hasFnAttribute("amdgpu-flat-work-group-size")) {
378 Default.first = MinImpliedByFlatWorkGroupSize;
379 RequestedFlatWorkGroupSize = true;
380 }
381
382 // Requested minimum/maximum number of waves per execution unit.
383 std::pair<unsigned, unsigned> Requested = AMDGPU::getIntegerPairAttribute(
384 F, "amdgpu-waves-per-eu", Default, true);
385
386 // Make sure requested minimum is less than requested maximum.
387 if (Requested.second && Requested.first > Requested.second)
388 return Default;
389
390 // Make sure requested values do not violate subtarget's specifications.
391 if (Requested.first < getMinWavesPerEU() ||
392 Requested.first > getMaxWavesPerEU())
393 return Default;
394 if (Requested.second > getMaxWavesPerEU())
395 return Default;
396
397 // Make sure requested values are compatible with values implied by requested
398 // minimum/maximum flat work group sizes.
399 if (RequestedFlatWorkGroupSize &&
400 Requested.first < MinImpliedByFlatWorkGroupSize)
401 return Default;
402
403 return Requested;
404}

406bool AMDGPUSubtarget::makeLIDRangeMetadata(Instruction *I) const {
407 Function *Kernel = I->getParent()->getParent();
408 unsigned MinSize = 0;
409 unsigned MaxSize = getFlatWorkGroupSizes(*Kernel).second;
410 bool IdQuery = false;
411
412 // If reqd_work_group_size is present it narrows value down.
413 if (auto *CI = dyn_cast<CallInst>(I)) {
414 const Function *F = CI->getCalledFunction();
415 if (F) {
416 unsigned Dim = UINT_MAX;
417 switch (F->getIntrinsicID()) {
418 case Intrinsic::amdgcn_workitem_id_x:
419 case Intrinsic::r600_read_tidig_x:
420 IdQuery = true;
421 LLVM_FALLTHROUGH;
422 case Intrinsic::r600_read_local_size_x:
423 Dim = 0;
424 break;
425 case Intrinsic::amdgcn_workitem_id_y:
426 case Intrinsic::r600_read_tidig_y:
427 IdQuery = true;
428 LLVM_FALLTHROUGH;
429 case Intrinsic::r600_read_local_size_y:
430 Dim = 1;
431 break;
432 case Intrinsic::amdgcn_workitem_id_z:
433 case Intrinsic::r600_read_tidig_z:
434 IdQuery = true;
435 LLVM_FALLTHROUGH;
436 case Intrinsic::r600_read_local_size_z:
437 Dim = 2;
438 break;
439 default:
440 break;
441 }
442 if (Dim <= 3) {
443 if (auto Node = Kernel->getMetadata("reqd_work_group_size"))
444 if (Node->getNumOperands() == 3)
445 MinSize = MaxSize = mdconst::extract<ConstantInt>(
446 Node->getOperand(Dim))->getZExtValue();
447 }
448 }
449 }
450
451 if (!MaxSize)
452 return false;
453
454 // Range metadata is [Lo, Hi). For ID query we need to pass max size
455 // as Hi. For size query we need to pass Hi + 1.
456 if (IdQuery)
457 MinSize = 0;
458 else
459 ++MaxSize;
460
461 MDBuilder MDB(I->getContext());
462 MDNode *MaxWorkGroupSizeRange = MDB.createRange(APInt(32, MinSize),
463 APInt(32, MaxSize));
464 I->setMetadata(LLVMContext::MD_range, MaxWorkGroupSizeRange);
465 return true;
466}

468uint64_t AMDGPUSubtarget::getExplicitKernArgSize(const Function &F,
469 unsigned &MaxAlign) const {
470 assert(F.getCallingConv() == CallingConv::AMDGPU_KERNEL ||
471 F.getCallingConv() == CallingConv::SPIR_KERNEL);
472
473 const DataLayout &DL = F.getParent()->getDataLayout();
474 uint64_t ExplicitArgBytes = 0;
475 MaxAlign = 1;
476
477 for (const Argument &Arg : F.args()) {
478 Type *ArgTy = Arg.getType();
479
480 unsigned Align = DL.getABITypeAlignment(ArgTy);
481 uint64_t AllocSize = DL.getTypeAllocSize(ArgTy);
482 ExplicitArgBytes = alignTo(ExplicitArgBytes, Align) + AllocSize;
483 MaxAlign = std::max(MaxAlign, Align);
484 }
485
486 return ExplicitArgBytes;
487}
488
489unsigned AMDGPUSubtarget::getKernArgSegmentSize(const Function &F,
490 unsigned &MaxAlign) const {
491 uint64_t ExplicitArgBytes = getExplicitKernArgSize(F, MaxAlign);
492
493 unsigned ExplicitOffset = getExplicitKernelArgOffset(F);
494
495 uint64_t TotalSize = ExplicitOffset + ExplicitArgBytes;
496 unsigned ImplicitBytes = getImplicitArgNumBytes(F);
497 if (ImplicitBytes != 0) {
498 unsigned Alignment = getAlignmentForImplicitArgPtr();
499 TotalSize = alignTo(ExplicitArgBytes, Alignment) + ImplicitBytes;
500 }
501
502 // Being able to dereference past the end is useful for emitting scalar loads.
503 return alignTo(TotalSize, 4);
504}

523void GCNSubtarget::overrideSchedPolicy(MachineSchedPolicy &Policy,
524 unsigned NumRegionInstrs) const {
525 // Track register pressure so the scheduler can try to decrease
526 // pressure once register usage is above the threshold defined by
527 // SIRegisterInfo::getRegPressureSetLimit()
528 Policy.ShouldTrackPressure = true;
529
530 // Enabling both top down and bottom up scheduling seems to give us less
531 // register spills than just using one of these approaches on its own.
532 Policy.OnlyTopDown = false;
533 Policy.OnlyBottomUp = false;
534
535 // Enabling ShouldTrackLaneMasks crashes the SI Machine Scheduler.
536 if (!enableSIScheduler())
537 Policy.ShouldTrackLaneMasks = true;
538}

592unsigned GCNSubtarget::getReservedNumSGPRs(const MachineFunction &MF) const {
593 const SIMachineFunctionInfo &MFI = *MF.getInfo<SIMachineFunctionInfo>();
594 if (getGeneration() >= AMDGPUSubtarget::GFX10)
595 return 2; // VCC. FLAT_SCRATCH and XNACK are no longer in SGPRs.
596
597 if (MFI.hasFlatScratchInit()) {
598 if (getGeneration() >= AMDGPUSubtarget::VOLCANIC_ISLANDS)
599 return 6; // FLAT_SCRATCH, XNACK, VCC (in that order).
600 if (getGeneration() == AMDGPUSubtarget::SEA_ISLANDS)
601 return 4; // FLAT_SCRATCH, VCC (in that order).
602 }
603
604 if (isXNACKEnabled())
605 return 4; // XNACK, VCC (in that order).
606 return 2; // VCC.
607}

609unsigned GCNSubtarget::getMaxNumSGPRs(const MachineFunction &MF) const {
610 const Function &F = MF.getFunction();
611 const SIMachineFunctionInfo &MFI = *MF.getInfo<SIMachineFunctionInfo>();
612
613 // Compute maximum number of SGPRs function can use using default/requested
614 // minimum number of waves per execution unit.
615 std::pair<unsigned, unsigned> WavesPerEU = MFI.getWavesPerEU();
616 unsigned MaxNumSGPRs = getMaxNumSGPRs(WavesPerEU.first, false);
617 unsigned MaxAddressableNumSGPRs = getMaxNumSGPRs(WavesPerEU.first, true);
618
619 // Check if maximum number of SGPRs was explicitly requested using
620 // "amdgpu-num-sgpr" attribute.
621 if (F.hasFnAttribute("amdgpu-num-sgpr")) {
622 unsigned Requested = AMDGPU::getIntegerAttribute(
623 F, "amdgpu-num-sgpr", MaxNumSGPRs);
624
625 // Make sure requested value does not violate subtarget's specifications.
626 if (Requested && (Requested <= getReservedNumSGPRs(MF)))
627 Requested = 0;
628
629 // If more SGPRs are required to support the input user/system SGPRs,
630 // increase to accommodate them.
631 //
632 // FIXME: This really ends up using the requested number of SGPRs + number
633 // of reserved special registers in total. Theoretically you could re-use
634 // the last input registers for these special registers, but this would
635 // require a lot of complexity to deal with the weird aliasing.
636 unsigned InputNumSGPRs = MFI.getNumPreloadedSGPRs();
637 if (Requested && Requested < InputNumSGPRs)
638 Requested = InputNumSGPRs;
639
640 // Make sure requested value is compatible with values implied by
641 // default/requested minimum/maximum number of waves per execution unit.
642 if (Requested && Requested > getMaxNumSGPRs(WavesPerEU.first, false))
643 Requested = 0;
644 if (WavesPerEU.second &&
645 Requested && Requested < getMinNumSGPRs(WavesPerEU.second))
646 Requested = 0;
647
648 if (Requested)
649 MaxNumSGPRs = Requested;
650 }
651
652 if (hasSGPRInitBug())
653 MaxNumSGPRs = AMDGPU::IsaInfo::FIXED_NUM_SGPRS_FOR_INIT_BUG;
654
655 return std::min(MaxNumSGPRs - getReservedNumSGPRs(MF),
656 MaxAddressableNumSGPRs);
657}

659unsigned GCNSubtarget::getMaxNumVGPRs(const MachineFunction &MF) const {
660 const Function &F = MF.getFunction();
661 const SIMachineFunctionInfo &MFI = *MF.getInfo<SIMachineFunctionInfo>();
662
663 // Compute maximum number of VGPRs function can use using default/requested
664 // minimum number of waves per execution unit.
665 std::pair<unsigned, unsigned> WavesPerEU = MFI.getWavesPerEU();
666 unsigned MaxNumVGPRs = getMaxNumVGPRs(WavesPerEU.first);
667
668 // Check if maximum number of VGPRs was explicitly requested using
669 // "amdgpu-num-vgpr" attribute.
670 if (F.hasFnAttribute("amdgpu-num-vgpr")) {
671 unsigned Requested = AMDGPU::getIntegerAttribute(
672 F, "amdgpu-num-vgpr", MaxNumVGPRs);
673
674 // Make sure requested value is compatible with values implied by
675 // default/requested minimum/maximum number of waves per execution unit.
676 if (Requested && Requested > getMaxNumVGPRs(WavesPerEU.first))
677 Requested = 0;
678 if (WavesPerEU.second &&
679 Requested && Requested < getMinNumVGPRs(WavesPerEU.second))
680 Requested = 0;
681
682 if (Requested)
683 MaxNumVGPRs = Requested;
684 }
685
686 return MaxNumVGPRs;
687}

689namespace {
690struct MemOpClusterMutation : ScheduleDAGMutation {
691 const SIInstrInfo *TII;
692
693 MemOpClusterMutation(const SIInstrInfo *tii) : TII(tii) {}
694
695 void apply(ScheduleDAGInstrs *DAG) override {
696 SUnit *SUa = nullptr;
697 // Search for two consequent memory operations and link them
698 // to prevent scheduler from moving them apart.
699 // In DAG pre-process SUnits are in the original order of
700 // the instructions before scheduling.
701 for (SUnit &SU : DAG->SUnits) {
702 MachineInstr &MI2 = *SU.getInstr();
703 if (!MI2.mayLoad() && !MI2.mayStore()) {
704 SUa = nullptr;
705 continue;
706 }
707 if (!SUa) {
708 SUa = &SU;
709 continue;
710 }
711
712 MachineInstr &MI1 = *SUa->getInstr();
713 if ((TII->isVMEM(MI1) && TII->isVMEM(MI2)) ||
714 (TII->isFLAT(MI1) && TII->isFLAT(MI2)) ||
715 (TII->isSMRD(MI1) && TII->isSMRD(MI2)) ||
716 (TII->isDS(MI1) && TII->isDS(MI2))) {
717 SU.addPredBarrier(SUa);
718
719 for (const SDep &SI : SU.Preds) {
720 if (SI.getSUnit() != SUa)
721 SUa->addPred(SDep(SI.getSUnit(), SDep::Artificial));
722 }
723
724 if (&SU != &DAG->ExitSU) {
725 for (const SDep &SI : SUa->Succs) {
726 if (SI.getSUnit() != &SU)
727 SI.getSUnit()->addPred(SDep(&SU, SDep::Artificial));
728 }
729 }
730 }
731
732 SUa = &SU;
733 }
734 }
735};
736} // namespace

738void GCNSubtarget::getPostRAMutations(
739 std::vector<std::unique_ptr<ScheduleDAGMutation>> &Mutations) const {
740 Mutations.push_back(llvm::make_unique<MemOpClusterMutation>(&InstrInfo));
741}
