308/// Represents the counter values to wait for in an s_waitcnt instruction.
309///
310/// Large values (including the maximum possible integer) can be used to
311/// represent "don't care" waits.
312struct Waitcnt {
313 unsigned VmCnt = ~0u;
314 unsigned ExpCnt = ~0u;
315 unsigned LgkmCnt = ~0u;
316 unsigned VsCnt = ~0u;
317
318 Waitcnt() {}
319 Waitcnt(unsigned VmCnt, unsigned ExpCnt, unsigned LgkmCnt, unsigned VsCnt)
320 : VmCnt(VmCnt), ExpCnt(ExpCnt), LgkmCnt(LgkmCnt), VsCnt(VsCnt) {}
321
322 static Waitcnt allZero(const IsaVersion &Version) {
323 return Waitcnt(0, 0, 0, Version.Major >= 10 ? 0 : ~0u);
324 }
325 static Waitcnt allZeroExceptVsCnt() { return Waitcnt(0, 0, 0, ~0u); }
326
327 bool hasWait() const {
328 return VmCnt != ~0u || ExpCnt != ~0u || LgkmCnt != ~0u || VsCnt != ~0u;
329 }
330
331 bool dominates(const Waitcnt &Other) const {
332 return VmCnt <= Other.VmCnt && ExpCnt <= Other.ExpCnt &&
333 LgkmCnt <= Other.LgkmCnt && VsCnt <= Other.VsCnt;
334 }
335
336 Waitcnt combined(const Waitcnt &Other) const {
337 return Waitcnt(std::min(VmCnt, Other.VmCnt), std::min(ExpCnt, Other.ExpCnt),
338 std::min(LgkmCnt, Other.LgkmCnt),
339 std::min(VsCnt, Other.VsCnt));
340 }
341};

503LLVM_READNONE
504inline unsigned getOperandSize(const MCOperandInfo &OpInfo) {
505 switch (OpInfo.OperandType) {
506 case AMDGPU::OPERAND_REG_IMM_INT32:
507 case AMDGPU::OPERAND_REG_IMM_FP32:
508 case AMDGPU::OPERAND_REG_INLINE_C_INT32:
509 case AMDGPU::OPERAND_REG_INLINE_C_FP32:
510 return 4;
511
512 case AMDGPU::OPERAND_REG_IMM_INT64:
513 case AMDGPU::OPERAND_REG_IMM_FP64:
514 case AMDGPU::OPERAND_REG_INLINE_C_INT64:
515 case AMDGPU::OPERAND_REG_INLINE_C_FP64:
516 return 8;
517
518 case AMDGPU::OPERAND_REG_IMM_INT16:
519 case AMDGPU::OPERAND_REG_IMM_FP16:
520 case AMDGPU::OPERAND_REG_INLINE_C_INT16:
521 case AMDGPU::OPERAND_REG_INLINE_C_FP16:
522 case AMDGPU::OPERAND_REG_INLINE_C_V2INT16:
523 case AMDGPU::OPERAND_REG_INLINE_C_V2FP16:
524 case AMDGPU::OPERAND_REG_IMM_V2INT16:
525 case AMDGPU::OPERAND_REG_IMM_V2FP16:
526 return 2;
527
528 default:
529 llvm_unreachable("unhandled operand type");
530 }
531}

569// Track defaults for fields in the MODE registser.
570struct SIModeRegisterDefaults {
571 /// Floating point opcodes that support exception flag gathering quiet and
572 /// propagate signaling NaN inputs per IEEE 754-2008. Min_dx10 and max_dx10
573 /// become IEEE 754- 2008 compliant due to signaling NaN propagation and
574 /// quieting.
575 bool IEEE : 1;
576
577 /// Used by the vector ALU to force DX10-style treatment of NaNs: when set,
578 /// clamp NaN to zero; otherwise, pass NaN through.
579 bool DX10Clamp : 1;
580
581 // TODO: FP mode fields
582
583 SIModeRegisterDefaults() :
584 IEEE(true),
585 DX10Clamp(true) {}
586
587 SIModeRegisterDefaults(const Function &F);
588
589 static SIModeRegisterDefaults getDefaultForCallingConv(CallingConv::ID CC) {
590 SIModeRegisterDefaults Mode;
591 Mode.DX10Clamp = true;
592 Mode.IEEE = AMDGPU::isCompute(CC);
593 return Mode;
594 }
595
596 bool operator ==(const SIModeRegisterDefaults Other) const {
597 return IEEE == Other.IEEE && DX10Clamp == Other.DX10Clamp;
598 }
599
600 // FIXME: Inlining should be OK for dx10-clamp, since the caller's mode should
601 // be able to override.
602 bool isInlineCompatible(SIModeRegisterDefaults CalleeMode) const {
603 return *this == CalleeMode;
604 }
605};
