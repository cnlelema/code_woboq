30// Read the PAL metadata from IR metadata, where it was put by the frontend.
31void AMDGPUPALMetadata::readFromIR(Module &M) {
32 auto NamedMD = M.getNamedMetadata("amdgpu.pal.metadata.msgpack");
33 if (NamedMD && NamedMD->getNumOperands()) {
34 // This is the new msgpack format for metadata. It is a NamedMD containing
35 // an MDTuple containing an MDString containing the msgpack data.
36 BlobType = ELF::NT_AMDGPU_METADATA;
37 auto MDN = dyn_cast<MDTuple>(NamedMD->getOperand(0));
38 if (MDN && MDN->getNumOperands()) {
39 if (auto MDS = dyn_cast<MDString>(MDN->getOperand(0)))
40 setFromMsgPackBlob(MDS->getString());
41 }
42 return;
43 }
44 BlobType = ELF::NT_AMD_AMDGPU_PAL_METADATA;
45 NamedMD = M.getNamedMetadata("amdgpu.pal.metadata");
46 if (!NamedMD || !NamedMD->getNumOperands())
47 return;
48 // This is the old reg=value pair format for metadata. It is a NamedMD
49 // containing an MDTuple containing a number of MDNodes each of which is an
50 // integer value, and each two integer values forms a key=value pair that we
51 // store as Registers[key]=value in the map.
52 auto Tuple = dyn_cast<MDTuple>(NamedMD->getOperand(0));
53 if (!Tuple)
54 return;
55 for (unsigned I = 0, E = Tuple->getNumOperands() & -2; I != E; I += 2) {
56 auto Key = mdconst::dyn_extract<ConstantInt>(Tuple->getOperand(I));
57 auto Val = mdconst::dyn_extract<ConstantInt>(Tuple->getOperand(I + 1));
58 if (!Key || !Val)
59 continue;
60 setRegister(Key->getZExtValue(), Val->getZExtValue());
61 }
62}

64// Set PAL metadata from a binary blob from the applicable .note record.
65// Returns false if bad format. Blob must remain valid for the lifetime of the
66// Metadata.
67bool AMDGPUPALMetadata::setFromBlob(unsigned Type, StringRef Blob) {
68 BlobType = Type;
69 if (Type == ELF::NT_AMD_AMDGPU_PAL_METADATA)
70 return setFromLegacyBlob(Blob);
71 return setFromMsgPackBlob(Blob);
72}
73
74// Set PAL metadata from legacy (array of key=value pairs) blob.
75bool AMDGPUPALMetadata::setFromLegacyBlob(StringRef Blob) {
76 auto Data = reinterpret_cast<const uint32_t *>(Blob.data());
77 for (unsigned I = 0; I != Blob.size() / sizeof(uint32_t) / 2; ++I)
78 setRegister(Data[I * 2], Data[I * 2 + 1]);
79 return true;
80}

82// Set PAL metadata from msgpack blob.
83bool AMDGPUPALMetadata::setFromMsgPackBlob(StringRef Blob) {
84 msgpack::Reader Reader(Blob);
85 return MsgPackDoc.readFromBlob(Blob, /*Multi=*/false);
86}

88// Given the calling convention, calculate the register number for rsrc1. In
89// principle the register number could change in future hardware, but we know
90// it is the same for gfx6-9 (except that LS and ES don't exist on gfx9), so
91// we can use fixed values.
92static unsigned getRsrc1Reg(CallingConv::ID CC) {
93 switch (CC) {
94 default:
95 return PALMD::R_2E12_COMPUTE_PGM_RSRC1;
96 case CallingConv::AMDGPU_LS:
97 return PALMD::R_2D4A_SPI_SHADER_PGM_RSRC1_LS;
98 case CallingConv::AMDGPU_HS:
99 return PALMD::R_2D0A_SPI_SHADER_PGM_RSRC1_HS;
100 case CallingConv::AMDGPU_ES:
101 return PALMD::R_2CCA_SPI_SHADER_PGM_RSRC1_ES;
102 case CallingConv::AMDGPU_GS:
103 return PALMD::R_2C8A_SPI_SHADER_PGM_RSRC1_GS;
104 case CallingConv::AMDGPU_VS:
105 return PALMD::R_2C4A_SPI_SHADER_PGM_RSRC1_VS;
106 case CallingConv::AMDGPU_PS:
107 return PALMD::R_2C0A_SPI_SHADER_PGM_RSRC1_PS;
108 }
109}

111// Calculate the PAL metadata key for *S_SCRATCH_SIZE. It can be used
112// with a constant offset to access any non-register shader-specific PAL
113// metadata key.
114static unsigned getScratchSizeKey(CallingConv::ID CC) {
115 switch (CC) {
116 case CallingConv::AMDGPU_PS:
117 return PALMD::Key::PS_SCRATCH_SIZE;
118 case CallingConv::AMDGPU_VS:
119 return PALMD::Key::VS_SCRATCH_SIZE;
120 case CallingConv::AMDGPU_GS:
121 return PALMD::Key::GS_SCRATCH_SIZE;
122 case CallingConv::AMDGPU_ES:
123 return PALMD::Key::ES_SCRATCH_SIZE;
124 case CallingConv::AMDGPU_HS:
125 return PALMD::Key::HS_SCRATCH_SIZE;
126 case CallingConv::AMDGPU_LS:
127 return PALMD::Key::LS_SCRATCH_SIZE;
128 default:
129 return PALMD::Key::CS_SCRATCH_SIZE;
130 }
131}

157// Get a register from the metadata, or 0 if not currently set.
158unsigned AMDGPUPALMetadata::getRegister(unsigned Reg) {
159 auto Regs = getRegisters();
160 auto It = Regs.find(MsgPackDoc.getNode(Reg));
161 if (It == Regs.end())
162 return 0;
163 auto N = It->second;
164 if (N.getKind() != msgpack::Type::UInt)
165 return 0;
166 return N.getUInt();
167}
168
169// Set a register in the metadata.
170// In fact this ORs the value into any previous setting of the register.
171void AMDGPUPALMetadata::setRegister(unsigned Reg, unsigned Val) {
172 if (!isLegacy()) {
173 // In the new MsgPack format, ignore register numbered >= 0x10000000. It
174 // is a PAL ABI pseudo-register in the old non-MsgPack format.
175 if (Reg >= 0x10000000)
176 return;
177 }
178 auto &N = getRegisters()[MsgPackDoc.getNode(Reg)];
179 if (N.getKind() == msgpack::Type::UInt)
180 Val |= N.getUInt();
181 N = N.getDocument()->getNode(Val);
182}

185void AMDGPUPALMetadata::setEntryPoint(unsigned CC, StringRef Name) {
186 if (isLegacy())
187 return;
188 // Msgpack format.
189 getHwStage(CC)[".entry_point"] = MsgPackDoc.getNode(Name, /*Copy=*/true);
190}

192// Set the number of used vgprs in the metadata. This is an optional
193// advisory record for logging etc; wave dispatch actually uses the rsrc1
194// register for the shader stage to determine the number of vgprs to
195// allocate.
196void AMDGPUPALMetadata::setNumUsedVgprs(CallingConv::ID CC, unsigned Val) {
197 if (isLegacy()) {
198 // Old non-msgpack format.
199 unsigned NumUsedVgprsKey = getScratchSizeKey(CC) +
200 PALMD::Key::VS_NUM_USED_VGPRS -
201 PALMD::Key::VS_SCRATCH_SIZE;
202 setRegister(NumUsedVgprsKey, Val);
203 return;
204 }
205 // Msgpack format.
206 getHwStage(CC)[".vgpr_count"] = MsgPackDoc.getNode(Val);
207}

519// Convert the accumulated PAL metadata into an asm directive.
520void AMDGPUPALMetadata::toString(std::string &String) {
521 String.clear();
522 if (!BlobType)
523 return;
524 raw_string_ostream Stream(String);
525 if (isLegacy()) {
526 if (MsgPackDoc.getRoot().getKind() == msgpack::Type::Nil)
527 return;
528 // Old linear reg=val format.
529 Stream << '\t' << AMDGPU::PALMD::AssemblerDirective << ' ';
530 auto Regs = getRegisters();
531 for (auto I = Regs.begin(), E = Regs.end(); I != E; ++I) {
532 if (I != Regs.begin())
533 Stream << ',';
534 unsigned Reg = I->first.getUInt();
535 unsigned Val = I->second.getUInt();
536 Stream << "0x" << Twine::utohexstr(Reg) << ",0x" << Twine::utohexstr(Val);
537 }
538 Stream << '\n';
539 return;
540 }
541
542 // New msgpack-based format -- output as YAML (with unsigned numbers in hex),
543 // but first change the registers map to use names.
544 MsgPackDoc.setHexMode();
545 auto &RegsObj = refRegisters();
546 auto OrigRegs = RegsObj.getMap();
547 RegsObj = MsgPackDoc.getMapNode();
548 for (auto I : OrigRegs) {
549 auto Key = I.first;
550 if (const char *RegName = getRegisterName(Key.getUInt())) {
551 std::string KeyName = Key.toString();
552 KeyName += " (";
553 KeyName += RegName;
554 KeyName += ')';
555 Key = MsgPackDoc.getNode(KeyName, /*Copy=*/true);
556 }
557 RegsObj.getMap()[Key] = I.second;
558 }
559
560 // Output as YAML.
561 Stream << '\t' << AMDGPU::PALMD::AssemblerDirectiveBegin << '\n';
562 MsgPackDoc.toYAML(Stream);
563 Stream << '\t' << AMDGPU::PALMD::AssemblerDirectiveEnd << '\n';
564
565 // Restore original registers map.
566 RegsObj = OrigRegs;
567}

569// Convert the accumulated PAL metadata into a binary blob for writing as
570// a .note record of the specified AMD type. Returns an empty blob if
571// there is no PAL metadata,
572void AMDGPUPALMetadata::toBlob(unsigned Type, std::string &Blob) {
573 if (Type == ELF::NT_AMD_AMDGPU_PAL_METADATA)
574 toLegacyBlob(Blob);
575 else if (Type)
576 toMsgPackBlob(Blob);
577}

579void AMDGPUPALMetadata::toLegacyBlob(std::string &Blob) {
580 Blob.clear();
581 auto Registers = getRegisters();
582 if (Registers.getMap().empty())
583 return;
584 raw_string_ostream OS(Blob);
585 support::endian::Writer EW(OS, support::endianness::little);
586 for (auto I : Registers.getMap()) {
587 EW.write(uint32_t(I.first.getUInt()));
588 EW.write(uint32_t(I.second.getUInt()));
589 }
590}

592void AMDGPUPALMetadata::toMsgPackBlob(std::string &Blob) {
593 Blob.clear();
594 MsgPackDoc.writeToBlob(Blob);
595}

597// Set PAL metadata from YAML text. Returns false if failed.
598bool AMDGPUPALMetadata::setFromString(StringRef S) {
599 BlobType = ELF::NT_AMDGPU_METADATA;
600 if (!MsgPackDoc.fromYAML(S))
601 return false;
602
603 // In the registers map, some keys may be of the form "0xa191
604 // (SPI_PS_INPUT_CNTL_0)", in which case the YAML input code made it a
605 // string. We need to turn it into a number.
606 auto &RegsObj = refRegisters();
607 auto OrigRegs = RegsObj;
608 RegsObj = MsgPackDoc.getMapNode();
609 Registers = RegsObj.getMap();
610 bool Ok = true;
611 for (auto I : OrigRegs.getMap()) {
612 auto Key = I.first;
613 if (Key.getKind() == msgpack::Type::String) {
614 StringRef S = Key.getString();
615 uint64_t Val;
616 if (S.consumeInteger(0, Val)) {
617 Ok = false;
618 errs() << "Unrecognized PAL metadata register key '" << S << "'\n";
619 continue;
620 }
621 Key = MsgPackDoc.getNode(uint64_t(Val));
622 }
623 Registers.getMap()[Key] = I.second;
624 }
625 return Ok;
626}

628// Reference (create if necessary) the node for the registers map.
629msgpack::DocNode &AMDGPUPALMetadata::refRegisters() {
630 auto &N =
631 MsgPackDoc.getRoot()
632 .getMap(/*Convert=*/true)[MsgPackDoc.getNode("amdpal.pipelines")]
633 .getArray(/*Convert=*/true)[0]
634 .getMap(/*Convert=*/true)[MsgPackDoc.getNode(".registers")];
635 N.getMap(/*Convert=*/true);
636 return N;
637}
