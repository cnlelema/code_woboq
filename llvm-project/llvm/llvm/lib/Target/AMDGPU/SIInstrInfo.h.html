655 /// \p returns true if \p UseMO is substituted with \p DefMO in \p MI it would
656 /// be an inline immediate.
657 bool isInlineConstant(const MachineInstr &MI,
658 const MachineOperand &UseMO,
659 const MachineOperand &DefMO) const {
660 assert(UseMO.getParent() == &MI);
661 int OpIdx = MI.getOperandNo(&UseMO);
662 if (!MI.getDesc().OpInfo || OpIdx >= MI.getDesc().NumOperands) {
663 return false;
664 }
665
666 return isInlineConstant(DefMO, MI.getDesc().OpInfo[OpIdx]);
667 }
668
669 /// \p returns true if the operand \p OpIdx in \p MI is a valid inline
670 /// immediate.
671 bool isInlineConstant(const MachineInstr &MI, unsigned OpIdx) const {
672 const MachineOperand &MO = MI.getOperand(OpIdx);
673 return isInlineConstant(MO, MI.getDesc().OpInfo[OpIdx].OperandType);
674 }
675
676 bool isInlineConstant(const MachineInstr &MI, unsigned OpIdx,
677 const MachineOperand &MO) const {
678 if (!MI.getDesc().OpInfo || OpIdx >= MI.getDesc().NumOperands)
679 return false;
680
681 if (MI.isCopy()) {
682 unsigned Size = getOpSize(MI, OpIdx);
683 assert(Size == 8 || Size == 4);
684
685 uint8_t OpType = (Size == 8) ?
686 AMDGPU::OPERAND_REG_IMM_INT64 : AMDGPU::OPERAND_REG_IMM_INT32;
687 return isInlineConstant(MO, OpType);
688 }
689
690 return isInlineConstant(MO, MI.getDesc().OpInfo[OpIdx].OperandType);
691 }

754 /// Return the size in bytes of the operand OpNo on the given
755 // instruction opcode.
756 unsigned getOpSize(uint16_t Opcode, unsigned OpNo) const {
757 const MCOperandInfo &OpInfo = get(Opcode).OpInfo[OpNo];
758
759 if (OpInfo.RegClass == -1) {
760 // If this is an immediate operand, this must be a 32-bit literal.
761 assert(OpInfo.OperandType == MCOI::OPERAND_IMMEDIATE);
762 return 4;
763 }
764
765 return RI.getRegSizeInBits(*RI.getRegClass(OpInfo.RegClass)) / 8;
766 }
767
768 /// This form should usually be preferred since it handles operands
769 /// with unknown register classes.
770 unsigned getOpSize(const MachineInstr &MI, unsigned OpNo) const {
771 const MachineOperand &MO = MI.getOperand(OpNo);
772 if (MO.isReg()) {
773 if (unsigned SubReg = MO.getSubReg()) {
774 assert(RI.getRegSizeInBits(*RI.getSubClassWithSubReg(
775 MI.getParent()->getParent()->getRegInfo().
776 getRegClass(MO.getReg()), SubReg)) >= 32 &&
777 "Sub-dword subregs are not supported");
778 return RI.getSubRegIndexLaneMask(SubReg).getNumLanes() * 4;
779 }
780 }
781 return RI.getRegSizeInBits(*getOpRegClass(MI, OpNo)) / 8;
782 }
