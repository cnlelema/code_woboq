31static bool hasPressureSet(const int *PSets, unsigned PSetID) {
32 for (unsigned i = 0; PSets[i] != -1; ++i) {
33 if (PSets[i] == (int)PSetID)
34 return true;
35 }
36 return false;
37}

39void SIRegisterInfo::classifyPressureSet(unsigned PSetID, unsigned Reg,
40 BitVector &PressureSets) const {
41 for (MCRegUnitIterator U(Reg, this); U.isValid(); ++U) {
42 const int *PSets = getRegUnitPressureSets(*U);
43 if (hasPressureSet(PSets, PSetID)) {
44 PressureSets.set(PSetID);
45 break;
46 }
47 }
48}

61SIRegisterInfo::SIRegisterInfo(const GCNSubtarget &ST) :
62 AMDGPURegisterInfo(),
63 SGPRPressureSets(getNumRegPressureSets()),
64 VGPRPressureSets(getNumRegPressureSets()),
65 SpillSGPRToVGPR(false),
66 SpillSGPRToSMEM(false) {
67 if (EnableSpillSGPRToSMEM && ST.hasScalarStores())
68 SpillSGPRToSMEM = true;
69 else if (EnableSpillSGPRToVGPR)
70 SpillSGPRToVGPR = true;
71
72 unsigned NumRegPressureSets = getNumRegPressureSets();
73
74 SGPRSetID = NumRegPressureSets;
75 VGPRSetID = NumRegPressureSets;
76
77 for (unsigned i = 0; i < NumRegPressureSets; ++i) {
78 classifyPressureSet(i, AMDGPU::SGPR0, SGPRPressureSets);
79 classifyPressureSet(i, AMDGPU::VGPR0, VGPRPressureSets);
80 }
81
82 // Determine the number of reg units for each pressure set.
83 std::vector<unsigned> PressureSetRegUnits(NumRegPressureSets, 0);
84 for (unsigned i = 0, e = getNumRegUnits(); i != e; ++i) {
85 const int *PSets = getRegUnitPressureSets(i);
86 for (unsigned j = 0; PSets[j] != -1; ++j) {
87 ++PressureSetRegUnits[PSets[j]];
88 }
89 }
90
91 unsigned VGPRMax = 0, SGPRMax = 0;
92 for (unsigned i = 0; i < NumRegPressureSets; ++i) {
93 if (isVGPRPressureSet(i) && PressureSetRegUnits[i] > VGPRMax) {
94 VGPRSetID = i;
95 VGPRMax = PressureSetRegUnits[i];
96 continue;
97 }
98 if (isSGPRPressureSet(i) && PressureSetRegUnits[i] > SGPRMax) {
99 SGPRSetID = i;
100 SGPRMax = PressureSetRegUnits[i];
101 }
102 }
103
104 assert(SGPRSetID < NumRegPressureSets &&
105 VGPRSetID < NumRegPressureSets);
106}

117static unsigned findPrivateSegmentWaveByteOffsetRegIndex(unsigned RegCount) {
118 unsigned Reg;
119
120 // Try to place it in a hole after PrivateSegmentBufferReg.
121 if (RegCount & 3) {
122 // We cannot put the segment buffer in (Idx - 4) ... (Idx - 1) due to
123 // alignment constraints, so we have a hole where can put the wave offset.
124 Reg = RegCount - 1;
125 } else {
126 // We can put the segment buffer in (Idx - 4) ... (Idx - 1) and put the
127 // wave offset before it.
128 Reg = RegCount - 5;
129 }
130
131 return Reg;
132}

138 return AMDGPU::SGPR_32RegClass.getRegister(Reg);

141BitVector SIRegisterInfo::getReservedRegs(const MachineFunction &MF) const {
142 BitVector Reserved(getNumRegs());
143
144 // EXEC_LO and EXEC_HI could be allocated and used as regular register, but
145 // this seems likely to result in bugs, so I'm marking them as reserved.
146 reserveRegisterTuples(Reserved, AMDGPU::EXEC);
147 reserveRegisterTuples(Reserved, AMDGPU::FLAT_SCR);
148
149 // M0 has to be reserved so that llvm accepts it as a live-in into a block.
150 reserveRegisterTuples(Reserved, AMDGPU::M0);
151
152 // Reserve src_vccz, src_execz, src_scc.
153 reserveRegisterTuples(Reserved, AMDGPU::SRC_VCCZ);
154 reserveRegisterTuples(Reserved, AMDGPU::SRC_EXECZ);
155 reserveRegisterTuples(Reserved, AMDGPU::SRC_SCC);
156
157 // Reserve the memory aperture registers.
158 reserveRegisterTuples(Reserved, AMDGPU::SRC_SHARED_BASE);
159 reserveRegisterTuples(Reserved, AMDGPU::SRC_SHARED_LIMIT);
160 reserveRegisterTuples(Reserved, AMDGPU::SRC_PRIVATE_BASE);
161 reserveRegisterTuples(Reserved, AMDGPU::SRC_PRIVATE_LIMIT);
162
163 // Reserve src_pops_exiting_wave_id - support is not implemented in Codegen.
164 reserveRegisterTuples(Reserved, AMDGPU::SRC_POPS_EXITING_WAVE_ID);
165
166 // Reserve xnack_mask registers - support is not implemented in Codegen.
167 reserveRegisterTuples(Reserved, AMDGPU::XNACK_MASK);
168
169 // Reserve lds_direct register - support is not implemented in Codegen.
170 reserveRegisterTuples(Reserved, AMDGPU::LDS_DIRECT);
171
172 // Reserve Trap Handler registers - support is not implemented in Codegen.
173 reserveRegisterTuples(Reserved, AMDGPU::TBA);
174 reserveRegisterTuples(Reserved, AMDGPU::TMA);
175 reserveRegisterTuples(Reserved, AMDGPU::TTMP0_TTMP1);
176 reserveRegisterTuples(Reserved, AMDGPU::TTMP2_TTMP3);
177 reserveRegisterTuples(Reserved, AMDGPU::TTMP4_TTMP5);
178 reserveRegisterTuples(Reserved, AMDGPU::TTMP6_TTMP7);
179 reserveRegisterTuples(Reserved, AMDGPU::TTMP8_TTMP9);
180 reserveRegisterTuples(Reserved, AMDGPU::TTMP10_TTMP11);
181 reserveRegisterTuples(Reserved, AMDGPU::TTMP12_TTMP13);
182 reserveRegisterTuples(Reserved, AMDGPU::TTMP14_TTMP15);
183
184 // Reserve null register - it shall never be allocated
185 reserveRegisterTuples(Reserved, AMDGPU::SGPR_NULL);
186
187 const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
188
189 unsigned MaxNumSGPRs = ST.getMaxNumSGPRs(MF);
190 unsigned TotalNumSGPRs = AMDGPU::SGPR_32RegClass.getNumRegs();
191 for (unsigned i = MaxNumSGPRs; i < TotalNumSGPRs; ++i) {
192 unsigned Reg = AMDGPU::SGPR_32RegClass.getRegister(i);
193 reserveRegisterTuples(Reserved, Reg);
194 }
195
196 unsigned MaxNumVGPRs = ST.getMaxNumVGPRs(MF);
197 unsigned TotalNumVGPRs = AMDGPU::VGPR_32RegClass.getNumRegs();
198 for (unsigned i = MaxNumVGPRs; i < TotalNumVGPRs; ++i) {
199 unsigned Reg = AMDGPU::VGPR_32RegClass.getRegister(i);
200 reserveRegisterTuples(Reserved, Reg);
201 }
202
203 const SIMachineFunctionInfo *MFI = MF.getInfo<SIMachineFunctionInfo>();
204
205 unsigned ScratchWaveOffsetReg = MFI->getScratchWaveOffsetReg();
206 if (ScratchWaveOffsetReg != AMDGPU::NoRegister) {
207 // Reserve 1 SGPR for scratch wave offset in case we need to spill.
208 reserveRegisterTuples(Reserved, ScratchWaveOffsetReg);
209 }
210
211 unsigned ScratchRSrcReg = MFI->getScratchRSrcReg();
212 if (ScratchRSrcReg != AMDGPU::NoRegister) {
213 // Reserve 4 SGPRs for the scratch buffer resource descriptor in case we need
214 // to spill.
215 // TODO: May need to reserve a VGPR if doing LDS spilling.
216 reserveRegisterTuples(Reserved, ScratchRSrcReg);
217 assert(!isSubRegister(ScratchRSrcReg, ScratchWaveOffsetReg));
218 }
219
220 // We have to assume the SP is needed in case there are calls in the function,
221 // which is detected after the function is lowered. If we aren't really going
222 // to need SP, don't bother reserving it.
223 unsigned StackPtrReg = MFI->getStackPtrOffsetReg();
224
225 if (StackPtrReg != AMDGPU::NoRegister) {
226 reserveRegisterTuples(Reserved, StackPtrReg);
227 assert(!isSubRegister(ScratchRSrcReg, StackPtrReg));
228 }
229
230 unsigned FrameReg = MFI->getFrameOffsetReg();
231 if (FrameReg != AMDGPU::NoRegister) {
232 reserveRegisterTuples(Reserved, FrameReg);
233 assert(!isSubRegister(ScratchRSrcReg, FrameReg));
234 }
235
236 for (unsigned Reg : MFI->WWMReservedRegs) {
237 reserveRegisterTuples(Reserved, Reg);
238 }
239
240 return Reserved;
241}

278bool SIRegisterInfo::requiresFrameIndexReplacementScavenging(
279 const MachineFunction &MF) const {
280 const MachineFrameInfo &MFI = MF.getFrameInfo();
281 if (!MFI.hasStackObjects())
282 return false;
283
284 // The scavenger is used for large frames which may require finding a free
285 // register for large offsets.
286 if (!isUInt<12>(MFI.getStackSize()))
287 return true;
288
289 // If using scalar stores, for spills, m0 is needed for the scalar store
290 // offset (pre-GFX9). m0 is unallocatable, so we can't create a virtual
291 // register for it during frame index elimination, so the scavenger is
292 // directly needed.
293 return MF.getSubtarget<GCNSubtarget>().hasScalarStores() &&
294 MF.getInfo<SIMachineFunctionInfo>()->hasSpilledSGPRs();
295}

337void SIRegisterInfo::materializeFrameBaseRegister(MachineBasicBlock *MBB,
338 unsigned BaseReg,
339 int FrameIdx,
340 int64_t Offset) const {
341 MachineBasicBlock::iterator Ins = MBB->begin();
342 DebugLoc DL; // Defaults to "unknown"
343
344 if (Ins != MBB->end())
345 DL = Ins->getDebugLoc();
346
347 MachineFunction *MF = MBB->getParent();
348 const GCNSubtarget &Subtarget = MF->getSubtarget<GCNSubtarget>();
349 const SIInstrInfo *TII = Subtarget.getInstrInfo();
350
351 if (Offset == 0) {
352 BuildMI(*MBB, Ins, DL, TII->get(AMDGPU::V_MOV_B32_e32), BaseReg)
353 .addFrameIndex(FrameIdx);
354 return;
355 }
356
357 MachineRegisterInfo &MRI = MF->getRegInfo();
358 unsigned OffsetReg = MRI.createVirtualRegister(&AMDGPU::SReg_32_XM0RegClass);
359
360 unsigned FIReg = MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
361
362 BuildMI(*MBB, Ins, DL, TII->get(AMDGPU::S_MOV_B32), OffsetReg)
363 .addImm(Offset);
364 BuildMI(*MBB, Ins, DL, TII->get(AMDGPU::V_MOV_B32_e32), FIReg)
365 .addFrameIndex(FrameIdx);
366
367 TII->getAddNoCarry(*MBB, Ins, DL, BaseReg)
368 .addReg(OffsetReg, RegState::Kill)
369 .addReg(FIReg)
370 .addImm(0); // clamp bit
371}

373void SIRegisterInfo::resolveFrameIndex(MachineInstr &MI, unsigned BaseReg,
374 int64_t Offset) const {
375
376 MachineBasicBlock *MBB = MI.getParent();
377 MachineFunction *MF = MBB->getParent();
378 const GCNSubtarget &Subtarget = MF->getSubtarget<GCNSubtarget>();
379 const SIInstrInfo *TII = Subtarget.getInstrInfo();
380
381#ifndef NDEBUG
382 // FIXME: Is it possible to be storing a frame index to itself?
383 bool SeenFI = false;
384 for (const MachineOperand &MO: MI.operands()) {
385 if (MO.isFI()) {
386 if (SeenFI)
387 llvm_unreachable("should not see multiple frame indices");
388
389 SeenFI = true;
390 }
391 }
392#endif
393
394 MachineOperand *FIOp = TII->getNamedOperand(MI, AMDGPU::OpName::vaddr);
395 assert(FIOp && FIOp->isFI() && "frame index must be address operand");
396 assert(TII->isMUBUF(MI));
397 assert(TII->getNamedOperand(MI, AMDGPU::OpName::soffset)->getReg() ==
398 MF->getInfo<SIMachineFunctionInfo>()->getFrameOffsetReg() &&
399 "should only be seeing frame offset relative FrameIndex");
400
401
402 MachineOperand *OffsetOp = TII->getNamedOperand(MI, AMDGPU::OpName::offset);
403 int64_t NewOffset = OffsetOp->getImm() + Offset;
404 assert(isUInt<12>(NewOffset) && "offset should be legal");
405
406 FIOp->ChangeToRegister(BaseReg, false);
407 OffsetOp->setImm(NewOffset);
408}

525// This differs from buildSpillLoadStore by only scavenging a VGPR. It does not
526// need to handle the case where an SGPR may need to be spilled while spilling.
527static bool buildMUBUFOffsetLoadStore(const SIInstrInfo *TII,
528 MachineFrameInfo &MFI,
529 MachineBasicBlock::iterator MI,
530 int Index,
531 int64_t Offset) {
532 MachineBasicBlock *MBB = MI->getParent();
533 const DebugLoc &DL = MI->getDebugLoc();
534 bool IsStore = MI->mayStore();
535
536 unsigned Opc = MI->getOpcode();
537 int LoadStoreOp = IsStore ?
538 getOffsetMUBUFStore(Opc) : getOffsetMUBUFLoad(Opc);
539 if (LoadStoreOp == -1)
540 return false;
541
542 const MachineOperand *Reg = TII->getNamedOperand(*MI, AMDGPU::OpName::vdata);
543 MachineInstrBuilder NewMI =
544 BuildMI(*MBB, MI, DL, TII->get(LoadStoreOp))
545 .add(*Reg)
546 .add(*TII->getNamedOperand(*MI, AMDGPU::OpName::srsrc))
547 .add(*TII->getNamedOperand(*MI, AMDGPU::OpName::soffset))
548 .addImm(Offset)
549 .addImm(0) // glc
550 .addImm(0) // slc
551 .addImm(0) // tfe
552 .addImm(0) // dlc
553 .cloneMemRefs(*MI);
554
555 const MachineOperand *VDataIn = TII->getNamedOperand(*MI,
556 AMDGPU::OpName::vdata_in);
557 if (VDataIn)
558 NewMI.add(*VDataIn);
559 return true;
560}

562void SIRegisterInfo::buildSpillLoadStore(MachineBasicBlock::iterator MI,
563 unsigned LoadStoreOp,
564 int Index,
565 unsigned ValueReg,
566 bool IsKill,
567 unsigned ScratchRsrcReg,
568 unsigned ScratchOffsetReg,
569 int64_t InstOffset,
570 MachineMemOperand *MMO,
571 RegScavenger *RS) const {
572 MachineBasicBlock *MBB = MI->getParent();
573 MachineFunction *MF = MI->getParent()->getParent();
574 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
575 const SIInstrInfo *TII = ST.getInstrInfo();
576 const MachineFrameInfo &MFI = MF->getFrameInfo();
577
578 const MCInstrDesc &Desc = TII->get(LoadStoreOp);
579 const DebugLoc &DL = MI->getDebugLoc();
580 bool IsStore = Desc.mayStore();
581
582 bool Scavenged = false;
583 unsigned SOffset = ScratchOffsetReg;
584
585 const unsigned EltSize = 4;
586 const TargetRegisterClass *RC = getRegClassForReg(MF->getRegInfo(), ValueReg);
587 unsigned NumSubRegs = AMDGPU::getRegBitWidth(RC->getID()) / (EltSize * CHAR_BIT);
588 unsigned Size = NumSubRegs * EltSize;
589 int64_t Offset = InstOffset + MFI.getObjectOffset(Index);
590 int64_t ScratchOffsetRegDelta = 0;
591
592 unsigned Align = MFI.getObjectAlignment(Index);
593 const MachinePointerInfo &BasePtrInfo = MMO->getPointerInfo();
594
595 assert((Offset % EltSize) == 0 && "unexpected VGPR spill offset");
596
597 if (!isUInt<12>(Offset + Size - EltSize)) {
598 SOffset = AMDGPU::NoRegister;
599
600 // We currently only support spilling VGPRs to EltSize boundaries, meaning
601 // we can simplify the adjustment of Offset here to just scale with
602 // WavefrontSize.
603 Offset *= ST.getWavefrontSize();
604
605 // We don't have access to the register scavenger if this function is called
606 // during PEI::scavengeFrameVirtualRegs().
607 if (RS)
608 SOffset = RS->scavengeRegister(&AMDGPU::SGPR_32RegClass, MI, 0, false);
609
610 if (SOffset == AMDGPU::NoRegister) {
611 // There are no free SGPRs, and since we are in the process of spilling
612 // VGPRs too. Since we need a VGPR in order to spill SGPRs (this is true
613 // on SI/CI and on VI it is true until we implement spilling using scalar
614 // stores), we have no way to free up an SGPR. Our solution here is to
615 // add the offset directly to the ScratchOffset register, and then
616 // subtract the offset after the spill to return ScratchOffset to it's
617 // original value.
618 SOffset = ScratchOffsetReg;
619 ScratchOffsetRegDelta = Offset;
620 } else {
621 Scavenged = true;
622 }
623
624 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_ADD_U32), SOffset)
625 .addReg(ScratchOffsetReg)
626 .addImm(Offset);
627
628 Offset = 0;
629 }
630
631 for (unsigned i = 0, e = NumSubRegs; i != e; ++i, Offset += EltSize) {
632 unsigned SubReg = NumSubRegs == 1 ?
633 ValueReg : getSubReg(ValueReg, getSubRegFromChannel(i));
634
635 unsigned SOffsetRegState = 0;
636 unsigned SrcDstRegState = getDefRegState(!IsStore);
637 if (i + 1 == e) {
638 SOffsetRegState |= getKillRegState(Scavenged);
639 // The last implicit use carries the "Kill" flag.
640 SrcDstRegState |= getKillRegState(IsKill);
641 }
642
643 MachinePointerInfo PInfo = BasePtrInfo.getWithOffset(EltSize * i);
644 MachineMemOperand *NewMMO
645 = MF->getMachineMemOperand(PInfo, MMO->getFlags(),
646 EltSize, MinAlign(Align, EltSize * i));
647
648 auto MIB = BuildMI(*MBB, MI, DL, Desc)
649 .addReg(SubReg, getDefRegState(!IsStore) | getKillRegState(IsKill))
650 .addReg(ScratchRsrcReg)
651 .addReg(SOffset, SOffsetRegState)
652 .addImm(Offset)
653 .addImm(0) // glc
654 .addImm(0) // slc
655 .addImm(0) // tfe
656 .addImm(0) // dlc
657 .addMemOperand(NewMMO);
658
659 if (NumSubRegs > 1)
660 MIB.addReg(ValueReg, RegState::Implicit | SrcDstRegState);
661 }
662
663 if (ScratchOffsetRegDelta != 0) {
664 // Subtract the offset we added to the ScratchOffset register.
665 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_SUB_U32), ScratchOffsetReg)
666 .addReg(ScratchOffsetReg)
667 .addImm(ScratchOffsetRegDelta);
668 }
669}

671static std::pair<unsigned, unsigned> getSpillEltSize(unsigned SuperRegSize,
672 bool Store) {
673 if (SuperRegSize % 16 == 0) {
674 return { 16, Store ? AMDGPU::S_BUFFER_STORE_DWORDX4_SGPR :
675 AMDGPU::S_BUFFER_LOAD_DWORDX4_SGPR };
676 }
677
678 if (SuperRegSize % 8 == 0) {
679 return { 8, Store ? AMDGPU::S_BUFFER_STORE_DWORDX2_SGPR :
680 AMDGPU::S_BUFFER_LOAD_DWORDX2_SGPR };
681 }
682
683 return { 4, Store ? AMDGPU::S_BUFFER_STORE_DWORD_SGPR :
684 AMDGPU::S_BUFFER_LOAD_DWORD_SGPR};
685}

687bool SIRegisterInfo::spillSGPR(MachineBasicBlock::iterator MI,
688 int Index,
689 RegScavenger *RS,
690 bool OnlyToVGPR) const {
691 MachineBasicBlock *MBB = MI->getParent();
692 MachineFunction *MF = MBB->getParent();
693 SIMachineFunctionInfo *MFI = MF->getInfo<SIMachineFunctionInfo>();
694 DenseSet<unsigned> SGPRSpillVGPRDefinedSet;
695
696 ArrayRef<SIMachineFunctionInfo::SpilledReg> VGPRSpills
697 = MFI->getSGPRToVGPRSpills(Index);
698 bool SpillToVGPR = !VGPRSpills.empty();
699 if (OnlyToVGPR && !SpillToVGPR)
700 return false;
701
702 MachineRegisterInfo &MRI = MF->getRegInfo();
703 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
704 const SIInstrInfo *TII = ST.getInstrInfo();
705
706 unsigned SuperReg = MI->getOperand(0).getReg();
707 bool IsKill = MI->getOperand(0).isKill();
708 const DebugLoc &DL = MI->getDebugLoc();
709
710 MachineFrameInfo &FrameInfo = MF->getFrameInfo();
711
712 bool SpillToSMEM = spillSGPRToSMEM();
713 if (SpillToSMEM && OnlyToVGPR)
714 return false;
715
716 unsigned FrameReg = getFrameRegister(*MF);
717
718 assert(SpillToVGPR || (SuperReg != MFI->getStackPtrOffsetReg() &&
719 SuperReg != MFI->getFrameOffsetReg() &&
720 SuperReg != MFI->getScratchWaveOffsetReg()));
721
722 assert(SuperReg != AMDGPU::M0 && "m0 should never spill");
723
724 unsigned OffsetReg = AMDGPU::M0;
725 unsigned M0CopyReg = AMDGPU::NoRegister;
726
727 if (SpillToSMEM) {
728 if (RS->isRegUsed(AMDGPU::M0)) {
729 M0CopyReg = MRI.createVirtualRegister(&AMDGPU::SReg_32_XM0RegClass);
730 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::COPY), M0CopyReg)
731 .addReg(AMDGPU::M0);
732 }
733 }
734
735 unsigned ScalarStoreOp;
736 unsigned EltSize = 4;
737 const TargetRegisterClass *RC = getPhysRegClass(SuperReg);
738 if (SpillToSMEM && isSGPRClass(RC)) {
739 // XXX - if private_element_size is larger than 4 it might be useful to be
740 // able to spill wider vmem spills.
741 std::tie(EltSize, ScalarStoreOp) =
742 getSpillEltSize(getRegSizeInBits(*RC) / 8, true);
743 }
744
745 ArrayRef<int16_t> SplitParts = getRegSplitParts(RC, EltSize);
746 unsigned NumSubRegs = SplitParts.empty() ? 1 : SplitParts.size();
747
748 // SubReg carries the "Kill" flag when SubReg == SuperReg.
749 unsigned SubKillState = getKillRegState((NumSubRegs == 1) && IsKill);
750 for (unsigned i = 0, e = NumSubRegs; i < e; ++i) {
751 unsigned SubReg = NumSubRegs == 1 ?
752 SuperReg : getSubReg(SuperReg, SplitParts[i]);
753
754 if (SpillToSMEM) {
755 int64_t FrOffset = FrameInfo.getObjectOffset(Index);
756
757 // The allocated memory size is really the wavefront size * the frame
758 // index size. The widest register class is 64 bytes, so a 4-byte scratch
759 // allocation is enough to spill this in a single stack object.
760 //
761 // FIXME: Frame size/offsets are computed earlier than this, so the extra
762 // space is still unnecessarily allocated.
763
764 unsigned Align = FrameInfo.getObjectAlignment(Index);
765 MachinePointerInfo PtrInfo
766 = MachinePointerInfo::getFixedStack(*MF, Index, EltSize * i);
767 MachineMemOperand *MMO
768 = MF->getMachineMemOperand(PtrInfo, MachineMemOperand::MOStore,
769 EltSize, MinAlign(Align, EltSize * i));
770
771 // SMEM instructions only support a single offset, so increment the wave
772 // offset.
773
774 int64_t Offset = (ST.getWavefrontSize() * FrOffset) + (EltSize * i);
775 if (Offset != 0) {
776 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_ADD_U32), OffsetReg)
777 .addReg(FrameReg)
778 .addImm(Offset);
779 } else {
780 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_MOV_B32), OffsetReg)
781 .addReg(FrameReg);
782 }
783
784 BuildMI(*MBB, MI, DL, TII->get(ScalarStoreOp))
785 .addReg(SubReg, getKillRegState(IsKill)) // sdata
786 .addReg(MFI->getScratchRSrcReg()) // sbase
787 .addReg(OffsetReg, RegState::Kill) // soff
788 .addImm(0) // glc
789 .addImm(0) // dlc
790 .addMemOperand(MMO);
791
792 continue;
793 }
794
795 if (SpillToVGPR) {
796 SIMachineFunctionInfo::SpilledReg Spill = VGPRSpills[i];
797
798 // During SGPR spilling to VGPR, determine if the VGPR is defined. The
799 // only circumstance in which we say it is undefined is when it is the
800 // first spill to this VGPR in the first basic block.
801 bool VGPRDefined = true;
802 if (MBB == &MF->front())
803 VGPRDefined = !SGPRSpillVGPRDefinedSet.insert(Spill.VGPR).second;
804
805 // Mark the "old value of vgpr" input undef only if this is the first sgpr
806 // spill to this specific vgpr in the first basic block.
807 BuildMI(*MBB, MI, DL,
808 TII->getMCOpcodeFromPseudo(AMDGPU::V_WRITELANE_B32),
809 Spill.VGPR)
810 .addReg(SubReg, getKillRegState(IsKill))
811 .addImm(Spill.Lane)
812 .addReg(Spill.VGPR, VGPRDefined ? 0 : RegState::Undef);
813
814 // FIXME: Since this spills to another register instead of an actual
815 // frame index, we should delete the frame index when all references to
816 // it are fixed.
817 } else {
818 // XXX - Can to VGPR spill fail for some subregisters but not others?
819 if (OnlyToVGPR)
820 return false;
821
822 // Spill SGPR to a frame index.
823 // TODO: Should VI try to spill to VGPR and then spill to SMEM?
824 unsigned TmpReg = MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
825 // TODO: Should VI try to spill to VGPR and then spill to SMEM?
826
827 MachineInstrBuilder Mov
828 = BuildMI(*MBB, MI, DL, TII->get(AMDGPU::V_MOV_B32_e32), TmpReg)
829 .addReg(SubReg, SubKillState);
830
831
832 // There could be undef components of a spilled super register.
833 // TODO: Can we detect this and skip the spill?
834 if (NumSubRegs > 1) {
835 // The last implicit use of the SuperReg carries the "Kill" flag.
836 unsigned SuperKillState = 0;
837 if (i + 1 == e)
838 SuperKillState |= getKillRegState(IsKill);
839 Mov.addReg(SuperReg, RegState::Implicit | SuperKillState);
840 }
841
842 unsigned Align = FrameInfo.getObjectAlignment(Index);
843 MachinePointerInfo PtrInfo
844 = MachinePointerInfo::getFixedStack(*MF, Index, EltSize * i);
845 MachineMemOperand *MMO
846 = MF->getMachineMemOperand(PtrInfo, MachineMemOperand::MOStore,
847 EltSize, MinAlign(Align, EltSize * i));
848 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::SI_SPILL_V32_SAVE))
849 .addReg(TmpReg, RegState::Kill) // src
850 .addFrameIndex(Index) // vaddr
851 .addReg(MFI->getScratchRSrcReg()) // srrsrc
852 .addReg(MFI->getStackPtrOffsetReg()) // soffset
853 .addImm(i * 4) // offset
854 .addMemOperand(MMO);
855 }
856 }
857
858 if (M0CopyReg != AMDGPU::NoRegister) {
859 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::COPY), AMDGPU::M0)
860 .addReg(M0CopyReg, RegState::Kill);
861 }
862
863 MI->eraseFromParent();
864 MFI->addToSpilledSGPRs(NumSubRegs);
865 return true;
866}

868bool SIRegisterInfo::restoreSGPR(MachineBasicBlock::iterator MI,
869 int Index,
870 RegScavenger *RS,
871 bool OnlyToVGPR) const {
872 MachineFunction *MF = MI->getParent()->getParent();
873 MachineRegisterInfo &MRI = MF->getRegInfo();
874 MachineBasicBlock *MBB = MI->getParent();
875 SIMachineFunctionInfo *MFI = MF->getInfo<SIMachineFunctionInfo>();
876
877 ArrayRef<SIMachineFunctionInfo::SpilledReg> VGPRSpills
878 = MFI->getSGPRToVGPRSpills(Index);
879 bool SpillToVGPR = !VGPRSpills.empty();
880 if (OnlyToVGPR && !SpillToVGPR)
881 return false;
882
883 MachineFrameInfo &FrameInfo = MF->getFrameInfo();
884 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
885 const SIInstrInfo *TII = ST.getInstrInfo();
886 const DebugLoc &DL = MI->getDebugLoc();
887
888 unsigned SuperReg = MI->getOperand(0).getReg();
889 bool SpillToSMEM = spillSGPRToSMEM();
890 if (SpillToSMEM && OnlyToVGPR)
891 return false;
892
893 assert(SuperReg != AMDGPU::M0 && "m0 should never spill");
894
895 unsigned OffsetReg = AMDGPU::M0;
896 unsigned M0CopyReg = AMDGPU::NoRegister;
897
898 if (SpillToSMEM) {
899 if (RS->isRegUsed(AMDGPU::M0)) {
900 M0CopyReg = MRI.createVirtualRegister(&AMDGPU::SReg_32_XM0RegClass);
901 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::COPY), M0CopyReg)
902 .addReg(AMDGPU::M0);
903 }
904 }
905
906 unsigned EltSize = 4;
907 unsigned ScalarLoadOp;
908
909 unsigned FrameReg = getFrameRegister(*MF);
910
911 const TargetRegisterClass *RC = getPhysRegClass(SuperReg);
912 if (SpillToSMEM && isSGPRClass(RC)) {
913 // XXX - if private_element_size is larger than 4 it might be useful to be
914 // able to spill wider vmem spills.
915 std::tie(EltSize, ScalarLoadOp) =
916 getSpillEltSize(getRegSizeInBits(*RC) / 8, false);
917 }
918
919 ArrayRef<int16_t> SplitParts = getRegSplitParts(RC, EltSize);
920 unsigned NumSubRegs = SplitParts.empty() ? 1 : SplitParts.size();
921
922 // SubReg carries the "Kill" flag when SubReg == SuperReg.
923 int64_t FrOffset = FrameInfo.getObjectOffset(Index);
924
925 for (unsigned i = 0, e = NumSubRegs; i < e; ++i) {
926 unsigned SubReg = NumSubRegs == 1 ?
927 SuperReg : getSubReg(SuperReg, SplitParts[i]);
928
929 if (SpillToSMEM) {
930 // FIXME: Size may be > 4 but extra bytes wasted.
931 unsigned Align = FrameInfo.getObjectAlignment(Index);
932 MachinePointerInfo PtrInfo
933 = MachinePointerInfo::getFixedStack(*MF, Index, EltSize * i);
934 MachineMemOperand *MMO
935 = MF->getMachineMemOperand(PtrInfo, MachineMemOperand::MOLoad,
936 EltSize, MinAlign(Align, EltSize * i));
937
938 // Add i * 4 offset
939 int64_t Offset = (ST.getWavefrontSize() * FrOffset) + (EltSize * i);
940 if (Offset != 0) {
941 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_ADD_U32), OffsetReg)
942 .addReg(FrameReg)
943 .addImm(Offset);
944 } else {
945 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_MOV_B32), OffsetReg)
946 .addReg(FrameReg);
947 }
948
949 auto MIB =
950 BuildMI(*MBB, MI, DL, TII->get(ScalarLoadOp), SubReg)
951 .addReg(MFI->getScratchRSrcReg()) // sbase
952 .addReg(OffsetReg, RegState::Kill) // soff
953 .addImm(0) // glc
954 .addImm(0) // dlc
955 .addMemOperand(MMO);
956
957 if (NumSubRegs > 1 && i == 0)
958 MIB.addReg(SuperReg, RegState::ImplicitDefine);
959
960 continue;
961 }
962
963 if (SpillToVGPR) {
964 SIMachineFunctionInfo::SpilledReg Spill = VGPRSpills[i];
965 auto MIB =
966 BuildMI(*MBB, MI, DL, TII->getMCOpcodeFromPseudo(AMDGPU::V_READLANE_B32),
967 SubReg)
968 .addReg(Spill.VGPR)
969 .addImm(Spill.Lane);
970
971 if (NumSubRegs > 1 && i == 0)
972 MIB.addReg(SuperReg, RegState::ImplicitDefine);
973 } else {
974 if (OnlyToVGPR)
975 return false;
976
977 // Restore SGPR from a stack slot.
978 // FIXME: We should use S_LOAD_DWORD here for VI.
979 unsigned TmpReg = MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
980 unsigned Align = FrameInfo.getObjectAlignment(Index);
981
982 MachinePointerInfo PtrInfo
983 = MachinePointerInfo::getFixedStack(*MF, Index, EltSize * i);
984
985 MachineMemOperand *MMO = MF->getMachineMemOperand(PtrInfo,
986 MachineMemOperand::MOLoad, EltSize,
987 MinAlign(Align, EltSize * i));
988
989 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::SI_SPILL_V32_RESTORE), TmpReg)
990 .addFrameIndex(Index) // vaddr
991 .addReg(MFI->getScratchRSrcReg()) // srsrc
992 .addReg(MFI->getStackPtrOffsetReg()) // soffset
993 .addImm(i * 4) // offset
994 .addMemOperand(MMO);
995
996 auto MIB =
997 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::V_READFIRSTLANE_B32), SubReg)
998 .addReg(TmpReg, RegState::Kill);
999
1000 if (NumSubRegs > 1)
1001 MIB.addReg(MI->getOperand(0).getReg(), RegState::ImplicitDefine);
1002 }
1003 }
1004
1005 if (M0CopyReg != AMDGPU::NoRegister) {
1006 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::COPY), AMDGPU::M0)
1007 .addReg(M0CopyReg, RegState::Kill);
1008 }
1009
1010 MI->eraseFromParent();
1011 return true;
1012}

1043void SIRegisterInfo::eliminateFrameIndex(MachineBasicBlock::iterator MI,
1044 int SPAdj, unsigned FIOperandNum,
1045 RegScavenger *RS) const {
1046 MachineFunction *MF = MI->getParent()->getParent();
1047 MachineRegisterInfo &MRI = MF->getRegInfo();
1048 MachineBasicBlock *MBB = MI->getParent();
1049 SIMachineFunctionInfo *MFI = MF->getInfo<SIMachineFunctionInfo>();
1050 MachineFrameInfo &FrameInfo = MF->getFrameInfo();
1051 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
1052 const SIInstrInfo *TII = ST.getInstrInfo();
1053 DebugLoc DL = MI->getDebugLoc();
1054
1055 MachineOperand &FIOp = MI->getOperand(FIOperandNum);
1056 int Index = MI->getOperand(FIOperandNum).getIndex();
1057
1058 unsigned FrameReg = getFrameRegister(*MF);
1059
1060 switch (MI->getOpcode()) {
1061 // SGPR register spill
1062 case AMDGPU::SI_SPILL_S512_SAVE:
1063 case AMDGPU::SI_SPILL_S256_SAVE:
1064 case AMDGPU::SI_SPILL_S160_SAVE:
1065 case AMDGPU::SI_SPILL_S128_SAVE:
1066 case AMDGPU::SI_SPILL_S96_SAVE:
1067 case AMDGPU::SI_SPILL_S64_SAVE:
1068 case AMDGPU::SI_SPILL_S32_SAVE: {
1069 spillSGPR(MI, Index, RS);
1070 break;
1071 }
1072
1073 // SGPR register restore
1074 case AMDGPU::SI_SPILL_S512_RESTORE:
1075 case AMDGPU::SI_SPILL_S256_RESTORE:
1076 case AMDGPU::SI_SPILL_S160_RESTORE:
1077 case AMDGPU::SI_SPILL_S128_RESTORE:
1078 case AMDGPU::SI_SPILL_S96_RESTORE:
1079 case AMDGPU::SI_SPILL_S64_RESTORE:
1080 case AMDGPU::SI_SPILL_S32_RESTORE: {
1081 restoreSGPR(MI, Index, RS);
1082 break;
1083 }
1084
1085 // VGPR register spill
1086 case AMDGPU::SI_SPILL_V512_SAVE:
1087 case AMDGPU::SI_SPILL_V256_SAVE:
1088 case AMDGPU::SI_SPILL_V160_SAVE:
1089 case AMDGPU::SI_SPILL_V128_SAVE:
1090 case AMDGPU::SI_SPILL_V96_SAVE:
1091 case AMDGPU::SI_SPILL_V64_SAVE:
1092 case AMDGPU::SI_SPILL_V32_SAVE: {
1093 const MachineOperand *VData = TII->getNamedOperand(*MI,
1094 AMDGPU::OpName::vdata);
1095 assert(TII->getNamedOperand(*MI, AMDGPU::OpName::soffset)->getReg() ==
1096 MFI->getStackPtrOffsetReg());
1097
1098 buildSpillLoadStore(MI, AMDGPU::BUFFER_STORE_DWORD_OFFSET,
1099 Index,
1100 VData->getReg(), VData->isKill(),
1101 TII->getNamedOperand(*MI, AMDGPU::OpName::srsrc)->getReg(),
1102 FrameReg,
1103 TII->getNamedOperand(*MI, AMDGPU::OpName::offset)->getImm(),
1104 *MI->memoperands_begin(),
1105 RS);
1106 MFI->addToSpilledVGPRs(getNumSubRegsForSpillOp(MI->getOpcode()));
1107 MI->eraseFromParent();
1108 break;
1109 }
1110 case AMDGPU::SI_SPILL_V32_RESTORE:
1111 case AMDGPU::SI_SPILL_V64_RESTORE:
1112 case AMDGPU::SI_SPILL_V96_RESTORE:
1113 case AMDGPU::SI_SPILL_V128_RESTORE:
1114 case AMDGPU::SI_SPILL_V160_RESTORE:
1115 case AMDGPU::SI_SPILL_V256_RESTORE:
1116 case AMDGPU::SI_SPILL_V512_RESTORE: {
1117 const MachineOperand *VData = TII->getNamedOperand(*MI,
1118 AMDGPU::OpName::vdata);
1119 assert(TII->getNamedOperand(*MI, AMDGPU::OpName::soffset)->getReg() ==
1120 MFI->getStackPtrOffsetReg());
1121
1122 buildSpillLoadStore(MI, AMDGPU::BUFFER_LOAD_DWORD_OFFSET,
1123 Index,
1124 VData->getReg(), VData->isKill(),
1125 TII->getNamedOperand(*MI, AMDGPU::OpName::srsrc)->getReg(),
1126 FrameReg,
1127 TII->getNamedOperand(*MI, AMDGPU::OpName::offset)->getImm(),
1128 *MI->memoperands_begin(),
1129 RS);
1130 MI->eraseFromParent();
1131 break;
1132 }
1133
1134 default: {
1135 const DebugLoc &DL = MI->getDebugLoc();
1136 bool IsMUBUF = TII->isMUBUF(*MI);
1137
1138 if (!IsMUBUF && !MFI->isEntryFunction()) {
1139 // Convert to an absolute stack address by finding the offset from the
1140 // scratch wave base and scaling by the wave size.
1141 //
1142 // In an entry function/kernel the offset is already the absolute
1143 // address relative to the frame register.
1144
1145 unsigned DiffReg
1146 = MRI.createVirtualRegister(&AMDGPU::SReg_32_XM0RegClass);
1147
1148 bool IsCopy = MI->getOpcode() == AMDGPU::V_MOV_B32_e32;
1149 unsigned ResultReg = IsCopy ?
1150 MI->getOperand(0).getReg() :
1151 MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
1152
1153 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_SUB_U32), DiffReg)
1154 .addReg(FrameReg)
1155 .addReg(MFI->getScratchWaveOffsetReg());
1156
1157 int64_t Offset = FrameInfo.getObjectOffset(Index);
1158 if (Offset == 0) {
1159 // XXX - This never happens because of emergency scavenging slot at 0?
1160 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::V_LSHRREV_B32_e64), ResultReg)
1161 .addImm(Log2_32(ST.getWavefrontSize()))
1162 .addReg(DiffReg);
1163 } else {
1164 unsigned ScaledReg
1165 = MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
1166
1167 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::V_LSHRREV_B32_e64), ScaledReg)
1168 .addImm(Log2_32(ST.getWavefrontSize()))
1169 .addReg(DiffReg, RegState::Kill);
1170
1171 // TODO: Fold if use instruction is another add of a constant.
1172 if (AMDGPU::isInlinableLiteral32(Offset, ST.hasInv2PiInlineImm())) {
1173 TII->getAddNoCarry(*MBB, MI, DL, ResultReg)
1174 .addImm(Offset)
1175 .addReg(ScaledReg, RegState::Kill)
1176 .addImm(0); // clamp bit
1177 } else {
1178 unsigned ConstOffsetReg
1179 = MRI.createVirtualRegister(&AMDGPU::SReg_32_XM0RegClass);
1180
1181 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::S_MOV_B32), ConstOffsetReg)
1182 .addImm(Offset);
1183 TII->getAddNoCarry(*MBB, MI, DL, ResultReg)
1184 .addReg(ConstOffsetReg, RegState::Kill)
1185 .addReg(ScaledReg, RegState::Kill)
1186 .addImm(0); // clamp bit
1187 }
1188 }
1189
1190 // Don't introduce an extra copy if we're just materializing in a mov.
1191 if (IsCopy)
1192 MI->eraseFromParent();
1193 else
1194 FIOp.ChangeToRegister(ResultReg, false, false, true);
1195 return;
1196 }
1197
1198 if (IsMUBUF) {
1199 // Disable offen so we don't need a 0 vgpr base.
1200 assert(static_cast<int>(FIOperandNum) ==
1201 AMDGPU::getNamedOperandIdx(MI->getOpcode(),
1202 AMDGPU::OpName::vaddr));
1203
1204 assert(TII->getNamedOperand(*MI, AMDGPU::OpName::soffset)->getReg() ==
1205 MFI->getStackPtrOffsetReg());
1206
1207 TII->getNamedOperand(*MI, AMDGPU::OpName::soffset)->setReg(FrameReg);
1208
1209 int64_t Offset = FrameInfo.getObjectOffset(Index);
1210 int64_t OldImm
1211 = TII->getNamedOperand(*MI, AMDGPU::OpName::offset)->getImm();
1212 int64_t NewOffset = OldImm + Offset;
1213
1214 if (isUInt<12>(NewOffset) &&
1215 buildMUBUFOffsetLoadStore(TII, FrameInfo, MI, Index, NewOffset)) {
1216 MI->eraseFromParent();
1217 return;
1218 }
1219 }
1220
1221 // If the offset is simply too big, don't convert to a scratch wave offset
1222 // relative index.
1223
1224 int64_t Offset = FrameInfo.getObjectOffset(Index);
1225 FIOp.ChangeToImmediate(Offset);
1226 if (!TII->isImmOperandLegal(*MI, FIOperandNum, FIOp)) {
1227 unsigned TmpReg = MRI.createVirtualRegister(&AMDGPU::VGPR_32RegClass);
1228 BuildMI(*MBB, MI, DL, TII->get(AMDGPU::V_MOV_B32_e32), TmpReg)
1229 .addImm(Offset);
1230 FIOp.ChangeToRegister(TmpReg, false, false, true);
1231 }
1232 }
1233 }

1445bool SIRegisterInfo::shouldRewriteCopySrc(
1446 const TargetRegisterClass *DefRC,
1447 unsigned DefSubReg,
1448 const TargetRegisterClass *SrcRC,
1449 unsigned SrcSubReg) const {
1450 // We want to prefer the smallest register class possible, so we don't want to
1451 // stop and rewrite on anything that looks like a subregister
1452 // extract. Operations mostly don't care about the super register class, so we
1453 // only want to stop on the most basic of copies between the same register
1454 // class.
1455 //
1456 // e.g. if we have something like
1457 // %0 = ...
1458 // %1 = ...
1459 // %2 = REG_SEQUENCE %0, sub0, %1, sub1, %2, sub2
1460 // %3 = COPY %2, sub0
1461 //
1462 // We want to look through the COPY to find:
1463 // => %3 = COPY %0
1464
1465 // Plain copy.
1466 return getCommonSubClass(DefRC, SrcRC) != nullptr;
1467}

1709// Find reaching register definition
1710MachineInstr *SIRegisterInfo::findReachingDef(unsigned Reg, unsigned SubReg,
1711 MachineInstr &Use,
1712 MachineRegisterInfo &MRI,
1713 LiveIntervals *LIS) const {
1714 auto &MDT = LIS->getAnalysis<MachineDominatorTree>();
1715 SlotIndex UseIdx = LIS->getInstructionIndex(Use);
1716 SlotIndex DefIdx;
1717
1718 if (TargetRegisterInfo::isVirtualRegister(Reg)) {
1719 if (!LIS->hasInterval(Reg))
1720 return nullptr;
1721 LiveInterval &LI = LIS->getInterval(Reg);
1722 LaneBitmask SubLanes = SubReg ? getSubRegIndexLaneMask(SubReg)
1723 : MRI.getMaxLaneMaskForVReg(Reg);
1724 VNInfo *V = nullptr;
1725 if (LI.hasSubRanges()) {
1726 for (auto &S : LI.subranges()) {
1727 if ((S.LaneMask & SubLanes) == SubLanes) {
1728 V = S.getVNInfoAt(UseIdx);
1729 break;
1730 }
1731 }
1732 } else {
1733 V = LI.getVNInfoAt(UseIdx);
1734 }
1735 if (!V)
1736 return nullptr;
1737 DefIdx = V->def;
1738 } else {
1739 // Find last def.
1740 for (MCRegUnitIterator Units(Reg, this); Units.isValid(); ++Units) {
1741 LiveRange &LR = LIS->getRegUnit(*Units);
1742 if (VNInfo *V = LR.getVNInfoAt(UseIdx)) {
1743 if (!DefIdx.isValid() ||
1744 MDT.dominates(LIS->getInstructionFromIndex(DefIdx),
1745 LIS->getInstructionFromIndex(V->def)))
1746 DefIdx = V->def;
1747 } else {
1748 return nullptr;
1749 }
1750 }
1751 }
1752
1753 MachineInstr *Def = LIS->getInstructionFromIndex(DefIdx);
1754
1755 if (!Def || !MDT.dominates(Def, &Use))
1756 return nullptr;
1757
1758 assert(Def->modifiesRegister(Reg, this));
1759
1760 return Def;
1761}
