112struct llvm::GVN::Expression {
113 uint32_t opcode;
114 Type *type;
115 bool commutative = false;
116 SmallVector<uint32_t, 4> varargs;
117
118 Expression(uint32_t o = ~2U) : opcode(o) {}
119
120 bool operator==(const Expression &other) const {
121 if (opcode != other.opcode)
122 return false;
123 if (opcode == ~0U || opcode == ~1U)
124 return true;
125 if (type != other.type)
126 return false;
127 if (varargs != other.varargs)
128 return false;
129 return true;
130 }
131
132 friend hash_code hash_value(const Expression &Value) {
133 return hash_combine(
134 Value.opcode, Value.type,
135 hash_combine_range(Value.varargs.begin(), Value.varargs.end()));
136 }
137};

158/// Represents a particular available value that we know how to materialize.
159/// Materialization of an AvailableValue never fails. An AvailableValue is
160/// implicitly associated with a rematerialization point which is the
161/// location of the instruction from which it was formed.
162struct llvm::gvn::AvailableValue {
163 enum ValType {
164 SimpleVal, // A simple offsetted value that is accessed.
165 LoadVal, // A value produced by a load.
166 MemIntrin, // A memory intrinsic which is loaded from.
167 UndefVal // A UndefValue representing a value from dead block (which
168 // is not yet physically removed from the CFG).
169 };
170
171 /// V - The value that is live out of the block.
172 PointerIntPair<Value *, 2, ValType> Val;
173
174 /// Offset - The byte offset in Val that is interesting for the load query.
175 unsigned Offset;
176
177 static AvailableValue get(Value *V, unsigned Offset = 0) {
178 AvailableValue Res;
179 Res.Val.setPointer(V);
180 Res.Val.setInt(SimpleVal);
181 Res.Offset = Offset;
182 return Res;
183 }
184
185 static AvailableValue getMI(MemIntrinsic *MI, unsigned Offset = 0) {
186 AvailableValue Res;
187 Res.Val.setPointer(MI);
188 Res.Val.setInt(MemIntrin);
189 Res.Offset = Offset;
190 return Res;
191 }
192
193 static AvailableValue getLoad(LoadInst *LI, unsigned Offset = 0) {
194 AvailableValue Res;
195 Res.Val.setPointer(LI);
196 Res.Val.setInt(LoadVal);
197 Res.Offset = Offset;
198 return Res;
199 }
200
201 static AvailableValue getUndef() {
202 AvailableValue Res;
203 Res.Val.setPointer(nullptr);
204 Res.Val.setInt(UndefVal);
205 Res.Offset = 0;
206 return Res;
207 }
208
209 bool isSimpleValue() const { return Val.getInt() == SimpleVal; }
210 bool isCoercedLoadValue() const { return Val.getInt() == LoadVal; }
211 bool isMemIntrinValue() const { return Val.getInt() == MemIntrin; }
212 bool isUndefValue() const { return Val.getInt() == UndefVal; }
213
214 Value *getSimpleValue() const {
215 assert(isSimpleValue() && "Wrong accessor");
216 return Val.getPointer();
217 }
218
219 LoadInst *getCoercedLoadValue() const {
220 assert(isCoercedLoadValue() && "Wrong accessor");
221 return cast<LoadInst>(Val.getPointer());
222 }
223
224 MemIntrinsic *getMemIntrinValue() const {
225 assert(isMemIntrinValue() && "Wrong accessor");
226 return cast<MemIntrinsic>(Val.getPointer());
227 }

230 /// here to the specified type. This handles various coercion cases.

232 GVN &gvn) const;

271GVN::Expression GVN::ValueTable::createExpr(Instruction *I) {
272 Expression e;
273 e.type = I->getType();
274 e.opcode = I->getOpcode();
275 for (Instruction::op_iterator OI = I->op_begin(), OE = I->op_end();
276 OI != OE; ++OI)
277 e.varargs.push_back(lookupOrAdd(*OI));
278 if (I->isCommutative()) {
279 // Ensure that commutative instructions that only differ by a permutation
280 // of their operands get the same value number by sorting the operand value
281 // numbers. Since all commutative instructions have two operands it is more
282 // efficient to sort by hand rather than using, say, std::sort.
283 assert(I->getNumOperands() == 2 && "Unsupported commutative instruction!");
284 if (e.varargs[0] > e.varargs[1])
285 std::swap(e.varargs[0], e.varargs[1]);
286 e.commutative = true;
287 }
288
289 if (CmpInst *C = dyn_cast<CmpInst>(I)) {
290 // Sort the operand value numbers so x<y and y>x get the same value number.
291 CmpInst::Predicate Predicate = C->getPredicate();
292 if (e.varargs[0] > e.varargs[1]) {
293 std::swap(e.varargs[0], e.varargs[1]);
294 Predicate = CmpInst::getSwappedPredicate(Predicate);
295 }
296 e.opcode = (C->getOpcode() << 8) | Predicate;
297 e.commutative = true;
298 } else if (InsertValueInst *E = dyn_cast<InsertValueInst>(I)) {
299 for (InsertValueInst::idx_iterator II = E->idx_begin(), IE = E->idx_end();
300 II != IE; ++II)
301 e.varargs.push_back(*II);
302 }
303
304 return e;
305}

307GVN::Expression GVN::ValueTable::createCmpExpr(unsigned Opcode,
308 CmpInst::Predicate Predicate,
309 Value *LHS, Value *RHS) {
310 assert((Opcode == Instruction::ICmp || Opcode == Instruction::FCmp) &&
311 "Not a comparison!");
312 Expression e;
313 e.type = CmpInst::makeCmpResultType(LHS->getType());
314 e.varargs.push_back(lookupOrAdd(LHS));
315 e.varargs.push_back(lookupOrAdd(RHS));
316
317 // Sort the operand value numbers so x<y and y>x get the same value number.
318 if (e.varargs[0] > e.varargs[1]) {
319 std::swap(e.varargs[0], e.varargs[1]);
320 Predicate = CmpInst::getSwappedPredicate(Predicate);
321 }
322 e.opcode = (Opcode << 8) | Predicate;
323 e.commutative = true;
324 return e;
325}

327GVN::Expression GVN::ValueTable::createExtractvalueExpr(ExtractValueInst *EI) {
328 assert(EI && "Not an ExtractValueInst?");
329 Expression e;
330 e.type = EI->getType();
331 e.opcode = 0;
332
333 WithOverflowInst *WO = dyn_cast<WithOverflowInst>(EI->getAggregateOperand());
334 if (WO != nullptr && EI->getNumIndices() == 1 && *EI->idx_begin() == 0) {
335 // EI is an extract from one of our with.overflow intrinsics. Synthesize
336 // a semantically equivalent expression instead of an extract value
337 // expression.
338 e.opcode = WO->getBinaryOp();
339 e.varargs.push_back(lookupOrAdd(WO->getLHS()));
340 e.varargs.push_back(lookupOrAdd(WO->getRHS()));
341 return e;
342 }
343
344 // Not a recognised intrinsic. Fall back to producing an extract value
345 // expression.
346 e.opcode = EI->getOpcode();
347 for (Instruction::op_iterator OI = EI->op_begin(), OE = EI->op_end();
348 OI != OE; ++OI)
349 e.varargs.push_back(lookupOrAdd(*OI));
350
351 for (ExtractValueInst::idx_iterator II = EI->idx_begin(), IE = EI->idx_end();
352 II != IE; ++II)
353 e.varargs.push_back(*II);
354
355 return e;
356}

374uint32_t GVN::ValueTable::lookupOrAddCall(CallInst *C) {
375 if (AA->doesNotAccessMemory(C)) {
376 Expression exp = createExpr(C);
377 uint32_t e = assignExpNewValueNum(exp).first;
378 valueNumbering[C] = e;
379 return e;
380 } else if (MD && AA->onlyReadsMemory(C)) {
381 Expression exp = createExpr(C);
382 auto ValNum = assignExpNewValueNum(exp);
383 if (ValNum.second) {
384 valueNumbering[C] = ValNum.first;
385 return ValNum.first;
386 }
387
388 MemDepResult local_dep = MD->getDependency(C);
389
390 if (!local_dep.isDef() && !local_dep.isNonLocal()) {
391 valueNumbering[C] = nextValueNumber;
392 return nextValueNumber++;
393 }
394
395 if (local_dep.isDef()) {
396 CallInst* local_cdep = cast<CallInst>(local_dep.getInst());
397
398 if (local_cdep->getNumArgOperands() != C->getNumArgOperands()) {
399 valueNumbering[C] = nextValueNumber;
400 return nextValueNumber++;
401 }
402
403 for (unsigned i = 0, e = C->getNumArgOperands(); i < e; ++i) {
404 uint32_t c_vn = lookupOrAdd(C->getArgOperand(i));
405 uint32_t cd_vn = lookupOrAdd(local_cdep->getArgOperand(i));
406 if (c_vn != cd_vn) {
407 valueNumbering[C] = nextValueNumber;
408 return nextValueNumber++;
409 }
410 }
411
412 uint32_t v = lookupOrAdd(local_cdep);
413 valueNumbering[C] = v;
414 return v;
415 }
416
417 // Non-local case.
418 const MemoryDependenceResults::NonLocalDepInfo &deps =
419 MD->getNonLocalCallDependency(C);
420 // FIXME: Move the checking logic to MemDep!
421 CallInst* cdep = nullptr;
422
423 // Check to see if we have a single dominating call instruction that is
424 // identical to C.
425 for (unsigned i = 0, e = deps.size(); i != e; ++i) {
426 const NonLocalDepEntry *I = &deps[i];
427 if (I->getResult().isNonLocal())
428 continue;
429
430 // We don't handle non-definitions. If we already have a call, reject
431 // instruction dependencies.
432 if (!I->getResult().isDef() || cdep != nullptr) {
433 cdep = nullptr;
434 break;
435 }
436
437 CallInst *NonLocalDepCall = dyn_cast<CallInst>(I->getResult().getInst());
438 // FIXME: All duplicated with non-local case.
439 if (NonLocalDepCall && DT->properlyDominates(I->getBB(), C->getParent())){
440 cdep = NonLocalDepCall;
441 continue;
442 }
443
444 cdep = nullptr;
445 break;
446 }
447
448 if (!cdep) {
449 valueNumbering[C] = nextValueNumber;
450 return nextValueNumber++;
451 }
452
453 if (cdep->getNumArgOperands() != C->getNumArgOperands()) {
454 valueNumbering[C] = nextValueNumber;
455 return nextValueNumber++;
456 }
457 for (unsigned i = 0, e = C->getNumArgOperands(); i < e; ++i) {
458 uint32_t c_vn = lookupOrAdd(C->getArgOperand(i));
459 uint32_t cd_vn = lookupOrAdd(cdep->getArgOperand(i));
460 if (c_vn != cd_vn) {
461 valueNumbering[C] = nextValueNumber;
462 return nextValueNumber++;
463 }
464 }
465
466 uint32_t v = lookupOrAdd(cdep);
467 valueNumbering[C] = v;
468 return v;
469 } else {
470 valueNumbering[C] = nextValueNumber;
471 return nextValueNumber++;
472 }
473}

478/// lookup_or_add - Returns the value number for the specified value, assigning
479/// it a new number if it did not have one before.
480uint32_t GVN::ValueTable::lookupOrAdd(Value *V) {
481 DenseMap<Value*, uint32_t>::iterator VI = valueNumbering.find(V);
482 if (VI != valueNumbering.end())
483 return VI->second;
484
485 if (!isa<Instruction>(V)) {
486 valueNumbering[V] = nextValueNumber;
487 return nextValueNumber++;
488 }
489
490 Instruction* I = cast<Instruction>(V);
491 Expression exp;
492 switch (I->getOpcode()) {
493 case Instruction::Call:
494 return lookupOrAddCall(cast<CallInst>(I));
495 case Instruction::Add:
496 case Instruction::FAdd:
497 case Instruction::Sub:
498 case Instruction::FSub:
499 case Instruction::Mul:
500 case Instruction::FMul:
501 case Instruction::UDiv:
502 case Instruction::SDiv:
503 case Instruction::FDiv:
504 case Instruction::URem:
505 case Instruction::SRem:
506 case Instruction::FRem:
507 case Instruction::Shl:
508 case Instruction::LShr:
509 case Instruction::AShr:
510 case Instruction::And:
511 case Instruction::Or:
512 case Instruction::Xor:
513 case Instruction::ICmp:
514 case Instruction::FCmp:
515 case Instruction::Trunc:
516 case Instruction::ZExt:
517 case Instruction::SExt:
518 case Instruction::FPToUI:
519 case Instruction::FPToSI:
520 case Instruction::UIToFP:
521 case Instruction::SIToFP:
522 case Instruction::FPTrunc:
523 case Instruction::FPExt:
524 case Instruction::PtrToInt:
525 case Instruction::IntToPtr:
526 case Instruction::AddrSpaceCast:
527 case Instruction::BitCast:
528 case Instruction::Select:
529 case Instruction::ExtractElement:
530 case Instruction::InsertElement:
531 case Instruction::ShuffleVector:
532 case Instruction::InsertValue:
533 case Instruction::GetElementPtr:
534 exp = createExpr(I);
535 break;
536 case Instruction::ExtractValue:
537 exp = createExtractvalueExpr(cast<ExtractValueInst>(I));
538 break;
539 case Instruction::PHI:
540 valueNumbering[V] = nextValueNumber;
541 NumberingPhi[nextValueNumber] = cast<PHINode>(V);
542 return nextValueNumber++;
543 default:
544 valueNumbering[V] = nextValueNumber;
545 return nextValueNumber++;
546 }
547
548 uint32_t e = assignExpNewValueNum(exp).first;
549 valueNumbering[V] = e;
550 return e;
551}

553/// Returns the value number of the specified value. Fails if
554/// the value has not yet been numbered.
555uint32_t GVN::ValueTable::lookup(Value *V, bool Verify) const {
556 DenseMap<Value*, uint32_t>::const_iterator VI = valueNumbering.find(V);
557 if (Verify) {
558 assert(VI != valueNumbering.end() && "Value not numbered?");
559 return VI->second;
560 }
561 return (VI != valueNumbering.end()) ? VI->second : 0;
562}

609PreservedAnalyses GVN::run(Function &F, FunctionAnalysisManager &AM) {
610 // FIXME: The order of evaluation of these 'getResult' calls is very
611 // significant! Re-ordering these variables will cause GVN when run alone to
612 // be less effective! We should fix memdep and basic-aa to not exhibit this
613 // behavior, but until then don't change the order here.
614 auto &AC = AM.getResult<AssumptionAnalysis>(F);
615 auto &DT = AM.getResult<DominatorTreeAnalysis>(F);
616 auto &TLI = AM.getResult<TargetLibraryAnalysis>(F);
617 auto &AA = AM.getResult<AAManager>(F);
618 auto &MemDep = AM.getResult<MemoryDependenceAnalysis>(F);
619 auto *LI = AM.getCachedResult<LoopAnalysis>(F);
620 auto &ORE = AM.getResult<OptimizationRemarkEmitterAnalysis>(F);
621 bool Changed = runImpl(F, AC, DT, TLI, AA, &MemDep, LI, &ORE);
622 if (!Changed)
623 return PreservedAnalyses::all();
624 PreservedAnalyses PA;
625 PA.preserve<DominatorTreeAnalysis>();
626 PA.preserve<GlobalsAA>();
627 PA.preserve<TargetLibraryAnalysis>();
628 return PA;
629}

643/// Return true if we can prove that the value
644/// we're analyzing is fully available in the specified block. As we go, keep
645/// track of which blocks we know are fully alive in FullyAvailableBlocks. This
646/// map is actually a tri-state map with the following values:
647/// 0) we know the block *is not* fully available.
648/// 1) we know the block *is* fully available.
649/// 2) we do not know whether the block is fully available or not, but we are
650/// currently speculating that it will be.
651/// 3) we are speculating for this block and have used that to speculate for
652/// other blocks.
653static bool IsValueFullyAvailableInBlock(BasicBlock *BB,
654 DenseMap<BasicBlock*, char> &FullyAvailableBlocks,
655 uint32_t RecurseDepth) {
656 if (RecurseDepth > MaxRecurseDepth)
657 return false;
658
659 // Optimistically assume that the block is fully available and check to see
660 // if we already know about this block in one lookup.
661 std::pair<DenseMap<BasicBlock*, char>::iterator, bool> IV =
662 FullyAvailableBlocks.insert(std::make_pair(BB, 2));
663
664 // If the entry already existed for this block, return the precomputed value.
665 if (!IV.second) {
666 // If this is a speculative "available" value, mark it as being used for
667 // speculation of other blocks.
668 if (IV.first->second == 2)
669 IV.first->second = 3;
670 return IV.first->second != 0;
671 }
672
673 // Otherwise, see if it is fully available in all predecessors.
674 pred_iterator PI = pred_begin(BB), PE = pred_end(BB);
675
676 // If this block has no predecessors, it isn't live-in here.
677 if (PI == PE)
678 goto SpeculationFailure;
679
680 for (; PI != PE; ++PI)
681 // If the value isn't fully available in one of our predecessors, then it
682 // isn't fully available in this block either. Undo our previous
683 // optimistic assumption and bail out.
684 if (!IsValueFullyAvailableInBlock(*PI, FullyAvailableBlocks,RecurseDepth+1))
685 goto SpeculationFailure;
686
687 return true;
688
689// If we get here, we found out that this is not, after
690// all, a fully-available block. We have a problem if we speculated on this and
691// used the speculation to mark other blocks as available.
692SpeculationFailure:
693 char &BBVal = FullyAvailableBlocks[BB];
694
695 // If we didn't speculate on this, just return with it set to false.
696 if (BBVal == 2) {
697 BBVal = 0;
698 return false;
699 }
700
701 // If we did speculate on this value, we could have blocks set to 1 that are
702 // incorrect. Walk the (transitive) successors of this block and mark them as
703 // 0 if set to one.
704 SmallVector<BasicBlock*, 32> BBWorklist;
705 BBWorklist.push_back(BB);
706
707 do {
708 BasicBlock *Entry = BBWorklist.pop_back_val();
709 // Note that this sets blocks to 0 (unavailable) if they happen to not
710 // already be in FullyAvailableBlocks. This is safe.
711 char &EntryVal = FullyAvailableBlocks[Entry];
712 if (EntryVal == 0) continue; // Already unavailable.
713
714 // Mark as unavailable.
715 EntryVal = 0;
716
717 BBWorklist.append(succ_begin(Entry), succ_end(Entry));
718 } while (!BBWorklist.empty());
719
720 return false;
721}

723/// Given a set of loads specified by ValuesPerBlock,
724/// construct SSA form, allowing us to eliminate LI. This returns the value
725/// that should be used at LI's definition site.
726static Value *ConstructSSAForLoadSet(LoadInst *LI,
727 SmallVectorImpl<AvailableValueInBlock> &ValuesPerBlock,
728 GVN &gvn) {
729 // Check for the fully redundant, dominating load case. In this case, we can
730 // just use the dominating value directly.
731 if (ValuesPerBlock.size() == 1 &&
732 gvn.getDominatorTree().properlyDominates(ValuesPerBlock[0].BB,
733 LI->getParent())) {
734 assert(!ValuesPerBlock[0].AV.isUndefValue() &&
735 "Dead BB dominate this block");
736 return ValuesPerBlock[0].MaterializeAdjustedValue(LI, gvn);
737 }
738
739 // Otherwise, we have to construct SSA form.
740 SmallVector<PHINode*, 8> NewPHIs;
741 SSAUpdater SSAUpdate(&NewPHIs);
742 SSAUpdate.Initialize(LI->getType(), LI->getName());
743
744 for (const AvailableValueInBlock &AV : ValuesPerBlock) {
745 BasicBlock *BB = AV.BB;
746
747 if (SSAUpdate.HasValueForBlock(BB))
748 continue;
749
750 // If the value is the load that we will be eliminating, and the block it's
751 // available in is the block that the load is in, then don't add it as
752 // SSAUpdater will resolve the value to the relevant phi which may let it
753 // avoid phi construction entirely if there's actually only one value.
754 if (BB == LI->getParent() &&
755 ((AV.AV.isSimpleValue() && AV.AV.getSimpleValue() == LI) ||
756 (AV.AV.isCoercedLoadValue() && AV.AV.getCoercedLoadValue() == LI)))
757 continue;
758
759 SSAUpdate.AddAvailableValue(BB, AV.MaterializeAdjustedValue(LI, gvn));
760 }
761
762 // Perform PHI construction.
763 return SSAUpdate.GetValueInMiddleOfBlock(LI->getParent());
764}

766Value *AvailableValue::MaterializeAdjustedValue(LoadInst *LI,
767 Instruction *InsertPt,
768 GVN &gvn) const {
769 Value *Res;
770 Type *LoadTy = LI->getType();
771 const DataLayout &DL = LI->getModule()->getDataLayout();
772 if (isSimpleValue()) {
773 Res = getSimpleValue();
774 if (Res->getType() != LoadTy) {
775 Res = getStoreValueForLoad(Res, Offset, LoadTy, InsertPt, DL);
776
777 LLVM_DEBUG(dbgs() << "GVN COERCED NONLOCAL VAL:\nOffset: " << Offset
778 << " " << *getSimpleValue() << '\n'
779 << *Res << '\n'
780 << "\n\n\n");
781 }
782 } else if (isCoercedLoadValue()) {
783 LoadInst *Load = getCoercedLoadValue();
784 if (Load->getType() == LoadTy && Offset == 0) {
785 Res = Load;
786 } else {
787 Res = getLoadValueForLoad(Load, Offset, LoadTy, InsertPt, DL, gvn);
788 // We would like to use gvn.markInstructionForDeletion here, but we can't
789 // because the load is already memoized into the leader map table that GVN
790 // tracks. It is potentially possible to remove the load from the table,
791 // but then there all of the operations based on it would need to be
792 // rehashed. Just leave the dead load around.
793 gvn.getMemDep().removeInstruction(Load);
794 LLVM_DEBUG(dbgs() << "GVN COERCED NONLOCAL LOAD:\nOffset: " << Offset
795 << " " << *getCoercedLoadValue() << '\n'
796 << *Res << '\n'
797 << "\n\n\n");
798 }
799 } else if (isMemIntrinValue()) {
800 Res = getMemInstValueForLoad(getMemIntrinValue(), Offset, LoadTy,
801 InsertPt, DL);
802 LLVM_DEBUG(dbgs() << "GVN COERCED NONLOCAL MEM INTRIN:\nOffset: " << Offset
803 << " " << *getMemIntrinValue() << '\n'
804 << *Res << '\n'
805 << "\n\n\n");
806 } else {
807 assert(isUndefValue() && "Should be UndefVal");
808 LLVM_DEBUG(dbgs() << "GVN COERCED NONLOCAL Undef:\n";);
809 return UndefValue::get(LoadTy);
810 }
811 assert(Res && "failed to materialize?");
812 return Res;
813}

854static bool canExtractBitsFromCapability(Type *Src, Type *Dst, int Offset,
855 const DataLayout &DL) {
856 // At worst capability to capability
857 if (Src == Dst && Offset == 0)
858 return true;
859 // Compute the offset in bits.
860 Offset = Offset * 8;
861 // If we're not extracting bits from a capability then we're ok.
862 if (!isa<PointerType>(Src))
863 return true;
864 if (!DL.isFatPointer(Src))
865 return true;
866 if (DL.getPointerBaseSizeInBits(Src) < Offset + DL.getTypeSizeInBits(Dst)) {
867 return false;
868 }
869 return true;
870}

872bool GVN::AnalyzeLoadAvailability(LoadInst *LI, MemDepResult DepInfo,
873 Value *Address, AvailableValue &Res) {
874 assert((DepInfo.isDef() || DepInfo.isClobber()) &&
875 "expected a local dependence");
876 assert(LI->isUnordered() && "rules below are incorrect for ordered access");
877
878 const DataLayout &DL = LI->getModule()->getDataLayout();
879
880 Instruction *DepInst = DepInfo.getInst();
881 if (DepInfo.isClobber()) {
882 // If the dependence is to a store that writes to a superset of the bits
883 // read by the load, we can extract the bits we need for the load from the
884 // stored value.
885 if (StoreInst *DepSI = dyn_cast<StoreInst>(DepInst)) {
886 // Can't forward from non-atomic to atomic without violating memory model.
887 if (Address && LI->isAtomic() <= DepSI->isAtomic()) {
888 int Offset =
889 analyzeLoadFromClobberingStore(LI->getType(), Address, DepSI, DL, TLI);
890 if (Offset != -1) {
891 if (canExtractBitsFromCapability(DepSI->getValueOperand()->getType(),
892 LI->getType(), Offset, DL)) {
893 Res = AvailableValue::get(DepSI->getValueOperand(), Offset);
894 return true;
895 }
896 return false;
897 }
898 }
899 }
900
901 // Check to see if we have something like this:
902 // load i32* P
903 // load i8* (P+1)
904 // if we have this, replace the later with an extraction from the former.
905 if (LoadInst *DepLI = dyn_cast<LoadInst>(DepInst)) {
906 // If this is a clobber and L is the first instruction in its block, then
907 // we have the first instruction in the entry block.
908 // Can't forward from non-atomic to atomic without violating memory model.
909 if (DepLI != LI && Address && LI->isAtomic() <= DepLI->isAtomic()) {
910 int Offset =
911 analyzeLoadFromClobberingLoad(LI->getType(), Address, DepLI, DL, TLI);
912
913 if (Offset != -1 &&
914 canExtractBitsFromCapability(DepLI->getType(), LI->getType(),
915 Offset, DL)) {
916 Res = AvailableValue::getLoad(DepLI, Offset);
917 return true;
918 }
919 }
920 }
921
922 // If the clobbering value is a memset/memcpy/memmove, see if we can
923 // forward a value on from it.
924 if (MemIntrinsic *DepMI = dyn_cast<MemIntrinsic>(DepInst)) {
925 if (Address && !LI->isAtomic()) {
926 int Offset = analyzeLoadFromClobberingMemInst(LI->getType(), Address,
927 DepMI, DL, TLI);
928 if (Offset != -1) {
929 Res = AvailableValue::getMI(DepMI, Offset);
930 return true;
931 }
932 }
933 }
934 // Nothing known about this clobber, have to be conservative
935 LLVM_DEBUG(
936 // fast print dep, using operator<< on instruction is too slow.
937 dbgs() << "GVN: load "; LI->printAsOperand(dbgs());
938 dbgs() << " is clobbered by " << *DepInst << '\n';);
939 if (ORE->allowExtraAnalysis(DEBUG_TYPE))
940 reportMayClobberedLoad(LI, DepInfo, DT, ORE);
941
942 return false;
943 }
944 assert(DepInfo.isDef() && "follows from above");
945
946 // Loading the allocation -> undef.
947 if (isa<AllocaInst>(DepInst) || isMallocLikeFn(DepInst, TLI) ||
948 // Loading immediately after lifetime begin -> undef.
949 isLifetimeStart(DepInst)) {
950 Res = AvailableValue::get(UndefValue::get(LI->getType()));
951 return true;
952 }
953
954 // Loading from calloc (which zero initializes memory) -> zero
955 if (isCallocLikeFn(DepInst, TLI)) {
956 Res = AvailableValue::get(Constant::getNullValue(LI->getType()));
957 return true;
958 }
959
960 if (StoreInst *S = dyn_cast<StoreInst>(DepInst)) {
961 // Reject loads and stores that are to the same address but are of
962 // different types if we have to. If the stored value is larger or equal to
963 // the loaded value, we can reuse it.
964 if (!canCoerceMustAliasedValueToLoad(S->getValueOperand(), LI->getType(),
965 DL))
966 return false;
967
968 // Can't forward from non-atomic to atomic without violating memory model.
969 if (S->isAtomic() < LI->isAtomic())
970 return false;
971
972 Res = AvailableValue::get(S->getValueOperand());
973 return true;
974 }
975
976 if (LoadInst *LD = dyn_cast<LoadInst>(DepInst)) {
977 // If the types mismatch and we can't handle it, reject reuse of the load.
978 // If the stored value is larger or equal to the loaded value, we can reuse
979 // it.
980 if (!canCoerceMustAliasedValueToLoad(LD, LI->getType(), DL))
981 return false;
982
983 // Can't forward from non-atomic to atomic without violating memory model.
984 if (LD->isAtomic() < LI->isAtomic())
985 return false;
986
987 Res = AvailableValue::getLoad(LD);
988 return true;
989 }
990
991 // Unknown def - must be conservative
992 LLVM_DEBUG(
993 // fast print dep, using operator<< on instruction is too slow.
994 dbgs() << "GVN: load "; LI->printAsOperand(dbgs());
995 dbgs() << " has unknown def " << *DepInst << '\n';);
996 return false;
997}

999void GVN::AnalyzeLoadAvailability(LoadInst *LI, LoadDepVect &Deps,
1000 AvailValInBlkVect &ValuesPerBlock,
1001 UnavailBlkVect &UnavailableBlocks) {
1002 // Filter out useless results (non-locals, etc). Keep track of the blocks
1003 // where we have a value available in repl, also keep track of whether we see
1004 // dependencies that produce an unknown value for the load (such as a call
1005 // that could potentially clobber the load).
1006 unsigned NumDeps = Deps.size();
1007 for (unsigned i = 0, e = NumDeps; i != e; ++i) {
1008 BasicBlock *DepBB = Deps[i].getBB();
1009 MemDepResult DepInfo = Deps[i].getResult();
1010
1011 if (DeadBlocks.count(DepBB)) {
1012 // Dead dependent mem-op disguise as a load evaluating the same value
1013 // as the load in question.
1014 ValuesPerBlock.push_back(AvailableValueInBlock::getUndef(DepBB));
1015 continue;
1016 }
1017
1018 if (!DepInfo.isDef() && !DepInfo.isClobber()) {
1019 UnavailableBlocks.push_back(DepBB);
1020 continue;
1021 }
1022
1023 // The address being loaded in this non-local block may not be the same as
1024 // the pointer operand of the load if PHI translation occurs. Make sure
1025 // to consider the right address.
1026 Value *Address = Deps[i].getAddress();
1027
1028 AvailableValue AV;
1029 if (AnalyzeLoadAvailability(LI, DepInfo, Address, AV)) {
1030 // subtlety: because we know this was a non-local dependency, we know
1031 // it's safe to materialize anywhere between the instruction within
1032 // DepInfo and the end of it's block.
1033 ValuesPerBlock.push_back(AvailableValueInBlock::get(DepBB,
1034 std::move(AV)));
1035 } else {
1036 UnavailableBlocks.push_back(DepBB);
1037 }
1038 }
1039
1040 assert(NumDeps == ValuesPerBlock.size() + UnavailableBlocks.size() &&
1041 "post condition violation");
1042}

1044bool GVN::PerformLoadPRE(LoadInst *LI, AvailValInBlkVect &ValuesPerBlock,
1045 UnavailBlkVect &UnavailableBlocks) {
1046 // Okay, we have *some* definitions of the value. This means that the value
1047 // is available in some of our (transitive) predecessors. Lets think about
1048 // doing PRE of this load. This will involve inserting a new load into the
1049 // predecessor when it's not available. We could do this in general, but
1050 // prefer to not increase code size. As such, we only do this when we know
1051 // that we only have to insert *one* load (which means we're basically moving
1052 // the load, not inserting a new one).
1053
1054 SmallPtrSet<BasicBlock *, 4> Blockers(UnavailableBlocks.begin(),
1055 UnavailableBlocks.end());
1056
1057 // Let's find the first basic block with more than one predecessor. Walk
1058 // backwards through predecessors if needed.
1059 BasicBlock *LoadBB = LI->getParent();
1060 BasicBlock *TmpBB = LoadBB;
1061 bool IsSafeToSpeculativelyExecute = isSafeToSpeculativelyExecute(LI);
1062
1063 // Check that there is no implicit control flow instructions above our load in
1064 // its block. If there is an instruction that doesn't always pass the
1065 // execution to the following instruction, then moving through it may become
1066 // invalid. For example:
1067 //
1068 // int arr[LEN];
1069 // int index = ???;
1070 // ...
1071 // guard(0 <= index && index < LEN);
1072 // use(arr[index]);
1073 //
1074 // It is illegal to move the array access to any point above the guard,
1075 // because if the index is out of bounds we should deoptimize rather than
1076 // access the array.
1077 // Check that there is no guard in this block above our instruction.
1078 if (!IsSafeToSpeculativelyExecute && ICF->isDominatedByICFIFromSameBlock(LI))
1079 return false;
1080 while (TmpBB->getSinglePredecessor()) {
1081 TmpBB = TmpBB->getSinglePredecessor();
1082 if (TmpBB == LoadBB) // Infinite (unreachable) loop.
1083 return false;
1084 if (Blockers.count(TmpBB))
1085 return false;
1086
1087 // If any of these blocks has more than one successor (i.e. if the edge we
1088 // just traversed was critical), then there are other paths through this
1089 // block along which the load may not be anticipated. Hoisting the load
1090 // above this block would be adding the load to execution paths along
1091 // which it was not previously executed.
1092 if (TmpBB->getTerminator()->getNumSuccessors() != 1)
1093 return false;
1094
1095 // Check that there is no implicit control flow in a block above.
1096 if (!IsSafeToSpeculativelyExecute && ICF->hasICF(TmpBB))
1097 return false;
1098 }
1099
1100 assert(TmpBB);
1101 LoadBB = TmpBB;
1102
1103 // Check to see how many predecessors have the loaded value fully
1104 // available.
1105 MapVector<BasicBlock *, Value *> PredLoads;
1106 DenseMap<BasicBlock*, char> FullyAvailableBlocks;
1107 for (const AvailableValueInBlock &AV : ValuesPerBlock)
1108 FullyAvailableBlocks[AV.BB] = true;
1109 for (BasicBlock *UnavailableBB : UnavailableBlocks)
1110 FullyAvailableBlocks[UnavailableBB] = false;
1111
1112 SmallVector<BasicBlock *, 4> CriticalEdgePred;
1113 for (BasicBlock *Pred : predecessors(LoadBB)) {
1114 // If any predecessor block is an EH pad that does not allow non-PHI
1115 // instructions before the terminator, we can't PRE the load.
1116 if (Pred->getTerminator()->isEHPad()) {
1117 LLVM_DEBUG(
1118 dbgs() << "COULD NOT PRE LOAD BECAUSE OF AN EH PAD PREDECESSOR '"
1119 << Pred->getName() << "': " << *LI << '\n');
1120 return false;
1121 }
1122
1123 if (IsValueFullyAvailableInBlock(Pred, FullyAvailableBlocks, 0)) {
1124 continue;
1125 }
1126
1127 if (Pred->getTerminator()->getNumSuccessors() != 1) {
1128 if (isa<IndirectBrInst>(Pred->getTerminator())) {
1129 LLVM_DEBUG(
1130 dbgs() << "COULD NOT PRE LOAD BECAUSE OF INDBR CRITICAL EDGE '"
1131 << Pred->getName() << "': " << *LI << '\n');
1132 return false;
1133 }
1134
1135 // FIXME: Can we support the fallthrough edge?
1136 if (isa<CallBrInst>(Pred->getTerminator())) {
1137 LLVM_DEBUG(
1138 dbgs() << "COULD NOT PRE LOAD BECAUSE OF CALLBR CRITICAL EDGE '"
1139 << Pred->getName() << "': " << *LI << '\n');
1140 return false;
1141 }
1142
1143 if (LoadBB->isEHPad()) {
1144 LLVM_DEBUG(
1145 dbgs() << "COULD NOT PRE LOAD BECAUSE OF AN EH PAD CRITICAL EDGE '"
1146 << Pred->getName() << "': " << *LI << '\n');
1147 return false;
1148 }
1149
1150 CriticalEdgePred.push_back(Pred);
1151 } else {
1152 // Only add the predecessors that will not be split for now.
1153 PredLoads[Pred] = nullptr;
1154 }
1155 }
1156
1157 // Decide whether PRE is profitable for this load.
1158 unsigned NumUnavailablePreds = PredLoads.size() + CriticalEdgePred.size();
1159 assert(NumUnavailablePreds != 0 &&
1160 "Fully available value should already be eliminated!");
1161
1162 // If this load is unavailable in multiple predecessors, reject it.
1163 // FIXME: If we could restructure the CFG, we could make a common pred with
1164 // all the preds that don't have an available LI and insert a new load into
1165 // that one block.
1166 if (NumUnavailablePreds != 1)
1167 return false;
1168
1169 // Split critical edges, and update the unavailable predecessors accordingly.
1170 for (BasicBlock *OrigPred : CriticalEdgePred) {
1171 BasicBlock *NewPred = splitCriticalEdges(OrigPred, LoadBB);
1172 assert(!PredLoads.count(OrigPred) && "Split edges shouldn't be in map!");
1173 PredLoads[NewPred] = nullptr;
1174 LLVM_DEBUG(dbgs() << "Split critical edge " << OrigPred->getName() << "->"
1175 << LoadBB->getName() << '\n');
1176 }
1177
1178 // Check if the load can safely be moved to all the unavailable predecessors.
1179 bool CanDoPRE = true;
1180 const DataLayout &DL = LI->getModule()->getDataLayout();
1181 SmallVector<Instruction*, 8> NewInsts;
1182 for (auto &PredLoad : PredLoads) {
1183 BasicBlock *UnavailablePred = PredLoad.first;
1184
1185 // Do PHI translation to get its value in the predecessor if necessary. The
1186 // returned pointer (if non-null) is guaranteed to dominate UnavailablePred.
1187
1188 // If all preds have a single successor, then we know it is safe to insert
1189 // the load on the pred (?!?), so we can insert code to materialize the
1190 // pointer if it is not available.
1191 PHITransAddr Address(LI->getPointerOperand(), DL, AC);
1192 Value *LoadPtr = nullptr;
1193 LoadPtr = Address.PHITranslateWithInsertion(LoadBB, UnavailablePred,
1194 *DT, NewInsts);
1195
1196 // If we couldn't find or insert a computation of this phi translated value,
1197 // we fail PRE.
1198 if (!LoadPtr) {
1199 LLVM_DEBUG(dbgs() << "COULDN'T INSERT PHI TRANSLATED VALUE OF: "
1200 << *LI->getPointerOperand() << "\n");
1201 CanDoPRE = false;
1202 break;
1203 }
1204
1205 PredLoad.second = LoadPtr;
1206 }
1207
1208 if (!CanDoPRE) {
1209 while (!NewInsts.empty()) {
1210 Instruction *I = NewInsts.pop_back_val();
1211 markInstructionForDeletion(I);
1212 }
1213 // HINT: Don't revert the edge-splitting as following transformation may
1214 // also need to split these critical edges.
1215 return !CriticalEdgePred.empty();
1216 }
1217
1218 // Okay, we can eliminate this load by inserting a reload in the predecessor
1219 // and using PHI construction to get the value in the other predecessors, do
1220 // it.
1221 LLVM_DEBUG(dbgs() << "GVN REMOVING PRE LOAD: " << *LI << '\n');
1222 LLVM_DEBUG(if (!NewInsts.empty()) dbgs()
1223 << "INSERTED " << NewInsts.size() << " INSTS: " << *NewInsts.back()
1224 << '\n');
1225
1226 // Assign value numbers to the new instructions.
1227 for (Instruction *I : NewInsts) {
1228 // Instructions that have been inserted in predecessor(s) to materialize
1229 // the load address do not retain their original debug locations. Doing
1230 // so could lead to confusing (but correct) source attributions.
1231 if (const DebugLoc &DL = I->getDebugLoc())
1232 I->setDebugLoc(DebugLoc::get(0, 0, DL.getScope(), DL.getInlinedAt()));
1233
1234 // FIXME: We really _ought_ to insert these value numbers into their
1235 // parent's availability map. However, in doing so, we risk getting into
1236 // ordering issues. If a block hasn't been processed yet, we would be
1237 // marking a value as AVAIL-IN, which isn't what we intend.
1238 VN.lookupOrAdd(I);
1239 }
1240
1241 for (const auto &PredLoad : PredLoads) {
1242 BasicBlock *UnavailablePred = PredLoad.first;
1243 Value *LoadPtr = PredLoad.second;
1244
1245 auto *NewLoad =
1246 new LoadInst(LI->getType(), LoadPtr, LI->getName() + ".pre",
1247 LI->isVolatile(), LI->getAlignment(), LI->getOrdering(),
1248 LI->getSyncScopeID(), UnavailablePred->getTerminator());
1249 NewLoad->setDebugLoc(LI->getDebugLoc());
1250
1251 // Transfer the old load's AA tags to the new load.
1252 AAMDNodes Tags;
1253 LI->getAAMetadata(Tags);
1254 if (Tags)
1255 NewLoad->setAAMetadata(Tags);
1256
1257 if (auto *MD = LI->getMetadata(LLVMContext::MD_invariant_load))
1258 NewLoad->setMetadata(LLVMContext::MD_invariant_load, MD);
1259 if (auto *InvGroupMD = LI->getMetadata(LLVMContext::MD_invariant_group))
1260 NewLoad->setMetadata(LLVMContext::MD_invariant_group, InvGroupMD);
1261 if (auto *RangeMD = LI->getMetadata(LLVMContext::MD_range))
1262 NewLoad->setMetadata(LLVMContext::MD_range, RangeMD);
1263
1264 // We do not propagate the old load's debug location, because the new
1265 // load now lives in a different BB, and we want to avoid a jumpy line
1266 // table.
1267 // FIXME: How do we retain source locations without causing poor debugging
1268 // behavior?
1269
1270 // Add the newly created load.
1271 ValuesPerBlock.push_back(AvailableValueInBlock::get(UnavailablePred,
1272 NewLoad));
1273 MD->invalidateCachedPointerInfo(LoadPtr);
1274 LLVM_DEBUG(dbgs() << "GVN INSERTED " << *NewLoad << '\n');
1275 }
1276
1277 // Perform PHI construction.
1278 Value *V = ConstructSSAForLoadSet(LI, ValuesPerBlock, *this);
1279 if (V == LI)
1280 return true;
1281 LI->replaceAllUsesWith(V);
1282 if (isa<PHINode>(V))
1283 V->takeName(LI);
1284 if (Instruction *I = dyn_cast<Instruction>(V))
1285 I->setDebugLoc(LI->getDebugLoc());
1286 if (V->getType()->isPtrOrPtrVectorTy())
1287 MD->invalidateCachedPointerInfo(V);
1288 markInstructionForDeletion(LI);
1289 ORE->emit([&]() {
1290 return OptimizationRemark(DEBUG_TYPE, "LoadPRE", LI)
1291 << "load eliminated by PRE";
1292 });
1293 ++NumPRELoad;
1294 return true;
1295}

1300

1306 });

1309/// Attempt to eliminate a load whose dependencies are
1310/// non-local by performing PHI construction.
1311bool GVN::processNonLocalLoad(LoadInst *LI) {
1312 // non-local speculations are not allowed under asan.
1313 if (LI->getParent()->getParent()->hasFnAttribute(
1314 Attribute::SanitizeAddress) ||
1315 LI->getParent()->getParent()->hasFnAttribute(
1316 Attribute::SanitizeHWAddress))
1317 return false;
1318
1319 // Step 1: Find the non-local dependencies of the load.
1320 LoadDepVect Deps;
1321 MD->getNonLocalPointerDependency(LI, Deps);
1322
1323 // If we had to process more than one hundred blocks to find the
1324 // dependencies, this load isn't worth worrying about. Optimizing
1325 // it will be too expensive.
1326 unsigned NumDeps = Deps.size();
1327 if (NumDeps > MaxNumDeps)
1328 return false;
1329
1330 // If we had a phi translation failure, we'll have a single entry which is a
1331 // clobber in the current block. Reject this early.
1332 if (NumDeps == 1 &&
1333 !Deps[0].getResult().isDef() && !Deps[0].getResult().isClobber()) {
1334 LLVM_DEBUG(dbgs() << "GVN: non-local load "; LI->printAsOperand(dbgs());
1335 dbgs() << " has unknown dependencies\n";);
1336 return false;
1337 }
1338
1339 // If this load follows a GEP, see if we can PRE the indices before analyzing.
1340 if (GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(LI->getOperand(0))) {
1341 for (GetElementPtrInst::op_iterator OI = GEP->idx_begin(),
1342 OE = GEP->idx_end();
1343 OI != OE; ++OI)
1344 if (Instruction *I = dyn_cast<Instruction>(OI->get()))
1345 performScalarPRE(I);
1346 }
1347
1348 // Step 2: Analyze the availability of the load
1349 AvailValInBlkVect ValuesPerBlock;
1350 UnavailBlkVect UnavailableBlocks;
1351 AnalyzeLoadAvailability(LI, Deps, ValuesPerBlock, UnavailableBlocks);
1352
1353 // If we have no predecessors that produce a known value for this load, exit
1354 // early.
1355 if (ValuesPerBlock.empty())
1356 return false;
1357
1358 // Step 3: Eliminate fully redundancy.
1359 //
1360 // If all of the instructions we depend on produce a known value for this
1361 // load, then it is fully redundant and we can use PHI insertion to compute
1362 // its value. Insert PHIs and remove the fully redundant value now.
1363 if (UnavailableBlocks.empty()) {
1364 LLVM_DEBUG(dbgs() << "GVN REMOVING NONLOCAL LOAD: " << *LI << '\n');
1365
1366 // Perform PHI construction.
1367 Value *V = ConstructSSAForLoadSet(LI, ValuesPerBlock, *this);
1368 if (V == LI)
1369 return false;
1370 LI->replaceAllUsesWith(V);
1371
1372 if (isa<PHINode>(V))
1373 V->takeName(LI);
1374 if (Instruction *I = dyn_cast<Instruction>(V))
1375 // If instruction I has debug info, then we should not update it.
1376 // Also, if I has a null DebugLoc, then it is still potentially incorrect
1377 // to propagate LI's DebugLoc because LI may not post-dominate I.
1378 if (LI->getDebugLoc() && LI->getParent() == I->getParent())
1379 I->setDebugLoc(LI->getDebugLoc());
1380 if (V->getType()->isPtrOrPtrVectorTy())
1381 MD->invalidateCachedPointerInfo(V);
1382 markInstructionForDeletion(LI);
1383 ++NumGVNLoad;
1384 reportLoadElim(LI, V, ORE);
1385 return true;
1386 }
1387
1388 // Step 4: Eliminate partial redundancy.
1389 if (!EnablePRE || !EnableLoadPRE)
1390 return false;
1391
1392 return PerformLoadPRE(LI, ValuesPerBlock, UnavailableBlocks);
1393}

1395bool GVN::processAssumeIntrinsic(IntrinsicInst *IntrinsicI) {
1396 assert(IntrinsicI->getIntrinsicID() == Intrinsic::assume &&
1397 "This function can only be called with llvm.assume intrinsic");
1398 Value *V = IntrinsicI->getArgOperand(0);
1399
1400 if (ConstantInt *Cond = dyn_cast<ConstantInt>(V)) {
1401 if (Cond->isZero()) {
1402 Type *Int8Ty = Type::getInt8Ty(V->getContext());
1403 // Insert a new store to null instruction before the load to indicate that
1404 // this code is not reachable. FIXME: We could insert unreachable
1405 // instruction directly because we can modify the CFG.
1406 new StoreInst(UndefValue::get(Int8Ty),
1407 Constant::getNullValue(Int8Ty->getPointerTo()),
1408 IntrinsicI);
1409 }
1410 markInstructionForDeletion(IntrinsicI);
1411 return false;
1412 } else if (isa<Constant>(V)) {
1413 // If it's not false, and constant, it must evaluate to true. This means our
1414 // assume is assume(true), and thus, pointless, and we don't want to do
1415 // anything more here.
1416 return false;
1417 }
1418
1419 Constant *True = ConstantInt::getTrue(V->getContext());
1420 bool Changed = false;
1421
1422 for (BasicBlock *Successor : successors(IntrinsicI->getParent())) {
1423 BasicBlockEdge Edge(IntrinsicI->getParent(), Successor);
1424
1425 // This property is only true in dominated successors, propagateEquality
1426 // will check dominance for us.
1427 Changed |= propagateEquality(V, True, Edge, false);
1428 }
1429
1430 // We can replace assume value with true, which covers cases like this:
1431 // call void @llvm.assume(i1 %cmp)
1432 // br i1 %cmp, label %bb1, label %bb2 ; will change %cmp to true
1433 ReplaceWithConstMap[V] = True;
1434
1435 // If one of *cmp *eq operand is const, adding it to map will cover this:
1436 // %cmp = fcmp oeq float 3.000000e+00, %0 ; const on lhs could happen
1437 // call void @llvm.assume(i1 %cmp)
1438 // ret float %0 ; will change it to ret float 3.000000e+00
1439 if (auto *CmpI = dyn_cast<CmpInst>(V)) {
1440 if (CmpI->getPredicate() == CmpInst::Predicate::ICMP_EQ ||
1441 CmpI->getPredicate() == CmpInst::Predicate::FCMP_OEQ ||
1442 (CmpI->getPredicate() == CmpInst::Predicate::FCMP_UEQ &&
1443 CmpI->getFastMathFlags().noNaNs())) {
1444 Value *CmpLHS = CmpI->getOperand(0);
1445 Value *CmpRHS = CmpI->getOperand(1);
1446 if (isa<Constant>(CmpLHS))
1447 std::swap(CmpLHS, CmpRHS);
1448 auto *RHSConst = dyn_cast<Constant>(CmpRHS);
1449
1450 // If only one operand is constant.
1451 if (RHSConst != nullptr && !isa<Constant>(CmpLHS))
1452 ReplaceWithConstMap[CmpLHS] = RHSConst;
1453 }
1454 }
1455 return Changed;
1456}

1459 patchReplacementInstruction(I, Repl);
1460 I->replaceAllUsesWith(Repl);

1463/// Attempt to eliminate a load, first by eliminating it
1464/// locally, and then attempting non-local elimination if that fails.
1465bool GVN::processLoad(LoadInst *L) {
1466 if (!MD)
1467 return false;
1468
1469 // This code hasn't been audited for ordered or volatile memory access
1470 if (!L->isUnordered())
1471 return false;
1472
1473 if (L->use_empty()) {
1474 markInstructionForDeletion(L);
1475 return true;
1476 }
1477
1478 // ... to a pointer that has been loaded from before...
1479 MemDepResult Dep = MD->getDependency(L);
1480
1481 // If it is defined in another block, try harder.
1482 if (Dep.isNonLocal())
1483 return processNonLocalLoad(L);
1484
1485 // Only handle the local case below
1486 if (!Dep.isDef() && !Dep.isClobber()) {
1487 // This might be a NonFuncLocal or an Unknown
1488 LLVM_DEBUG(
1489 // fast print dep, using operator<< on instruction is too slow.
1490 dbgs() << "GVN: load "; L->printAsOperand(dbgs());
1491 dbgs() << " has unknown dependence\n";);
1492 return false;
1493 }
1494
1495 AvailableValue AV;
1496 if (AnalyzeLoadAvailability(L, Dep, L->getPointerOperand(), AV)) {
1497 Value *AvailableValue = AV.MaterializeAdjustedValue(L, L, *this);
1498
1499 // Replace the load!
1500 patchAndReplaceAllUsesWith(L, AvailableValue);
1501 markInstructionForDeletion(L);
1502 ++NumGVNLoad;
1503 reportLoadElim(L, AvailableValue, ORE);
1504 // Tell MDA to rexamine the reused pointer since we might have more
1505 // information after forwarding it.
1506 if (MD && AvailableValue->getType()->isPtrOrPtrVectorTy())
1507 MD->invalidateCachedPointerInfo(AvailableValue);
1508 return true;
1509 }
1510
1511 return false;
1512}

1540/// Wrap phiTranslateImpl to provide caching functionality.
1541uint32_t GVN::ValueTable::phiTranslate(const BasicBlock *Pred,
1542 const BasicBlock *PhiBlock, uint32_t Num,
1543 GVN &Gvn) {

1545 if (FindRes != PhiTranslateTable.end())
1546 return FindRes->second;

1552/// Translate value number \p Num using phis, so that it has the values of
1553/// the phis in BB.
1554uint32_t GVN::ValueTable::phiTranslateImpl(const BasicBlock *Pred,
1555 const BasicBlock *PhiBlock,
1556 uint32_t Num, GVN &Gvn) {
1557 if (PHINode *PN = NumberingPhi[Num]) {
1558 for (unsigned i = 0; i != PN->getNumIncomingValues(); ++i) {
1559 if (PN->getParent() == PhiBlock && PN->getIncomingBlock(i) == Pred)
1560 if (uint32_t TransVal = lookup(PN->getIncomingValue(i), false))
1561 return TransVal;
1562 }
1563 return Num;
1564 }
1565
1566 // If there is any value related with Num is defined in a BB other than
1567 // PhiBlock, it cannot depend on a phi in PhiBlock without going through
1568 // a backedge. We can do an early exit in that case to save compile time.
1569 if (!areAllValsInBB(Num, PhiBlock, Gvn))
1570 return Num;
1571
1572 if (Num >= ExprIdx.size() || ExprIdx[Num] == 0)
1573 return Num;
1574 Expression Exp = Expressions[ExprIdx[Num]];
1575
1576 for (unsigned i = 0; i < Exp.varargs.size(); i++) {
1577 // For InsertValue and ExtractValue, some varargs are index numbers
1578 // instead of value numbers. Those index numbers should not be
1579 // translated.
1580 if ((i > 1 && Exp.opcode == Instruction::InsertValue) ||
1581 (i > 0 && Exp.opcode == Instruction::ExtractValue))
1582 continue;
1583 Exp.varargs[i] = phiTranslate(Pred, PhiBlock, Exp.varargs[i], Gvn);
1584 }
1585
1586 if (Exp.commutative) {
1587 assert(Exp.varargs.size() == 2 && "Unsupported commutative expression!");
1588 if (Exp.varargs[0] > Exp.varargs[1]) {
1589 std::swap(Exp.varargs[0], Exp.varargs[1]);
1590 uint32_t Opcode = Exp.opcode >> 8;
1591 if (Opcode == Instruction::ICmp || Opcode == Instruction::FCmp)
1592 Exp.opcode = (Opcode << 8) |
1593 CmpInst::getSwappedPredicate(
1594 static_cast<CmpInst::Predicate>(Exp.opcode & 255));
1595 }
1596 }
1597
1598 if (uint32_t NewNum = expressionNumbering[Exp])
1599 return NewNum;
1600 return Num;
1601}

1614// In order to find a leader for a given value number at a
1615// specific basic block, we first obtain the list of all Values for that number,
1616// and then scan the list to find one whose block dominates the block in
1617// question. This is fast because dominator tree queries consist of only
1618// a few comparisons of DFS numbers.
1619Value *GVN::findLeader(const BasicBlock *BB, uint32_t num) {
1620 LeaderTableEntry Vals = LeaderTable[num];
1621 if (!Vals.Val) return nullptr;
1622
1623 Value *Val = nullptr;
1624 if (DT->dominates(Vals.BB, BB)) {
1625 Val = Vals.Val;
1626 if (isa<Constant>(Val)) return Val;
1627 }
1628
1629 LeaderTableEntry* Next = Vals.Next;
1630 while (Next) {
1631 if (DT->dominates(Next->BB, BB)) {
1632 if (isa<Constant>(Next->Val)) return Next->Val;
1633 if (!Val) Val = Next->Val;
1634 }
1635
1636 Next = Next->Next;
1637 }
1638
1639 return Val;
1640}

1642/// There is an edge from 'Src' to 'Dst'. Return
1643/// true if every path from the entry block to 'Dst' passes via this edge. In
1644/// particular 'Dst' must not be reachable via another edge from 'Src'.
1645static bool isOnlyReachableViaThisEdge(const BasicBlockEdge &E,
1646 DominatorTree *DT) {
1647 // While in theory it is interesting to consider the case in which Dst has
1648 // more than one predecessor, because Dst might be part of a loop which is
1649 // only reachable from Src, in practice it is pointless since at the time
1650 // GVN runs all such loops have preheaders, which means that Dst will have
1651 // been changed to have only one predecessor, namely Src.
1652 const BasicBlock *Pred = E.getEnd()->getSinglePredecessor();
1653 assert((!Pred || Pred == E.getStart()) &&
1654 "No edge between these basic blocks!");
1655 return Pred != nullptr;
1656}

1686/// The given values are known to be equal in every block
1687/// dominated by 'Root'. Exploit this, for example by replacing 'LHS' with
1688/// 'RHS' everywhere in the scope. Returns whether a change was made.
1689/// If DominatesByEdge is false, then it means that we will propagate the RHS
1690/// value starting from the end of Root.Start.
1691bool GVN::propagateEquality(Value *LHS, Value *RHS, const BasicBlockEdge &Root,
1692 bool DominatesByEdge) {
1693 SmallVector<std::pair<Value*, Value*>, 4> Worklist;
1694 Worklist.push_back(std::make_pair(LHS, RHS));
1695 bool Changed = false;
1696 // For speed, compute a conservative fast approximation to
1697 // DT->dominates(Root, Root.getEnd());
1698 const bool RootDominatesEnd = isOnlyReachableViaThisEdge(Root, DT);
1699
1700 while (!Worklist.empty()) {
1701 std::pair<Value*, Value*> Item = Worklist.pop_back_val();
1702 LHS = Item.first; RHS = Item.second;
1703
1704 if (LHS == RHS)
1705 continue;
1706 assert(LHS->getType() == RHS->getType() && "Equality but unequal types!");
1707
1708 // Don't try to propagate equalities between constants.
1709 if (isa<Constant>(LHS) && isa<Constant>(RHS))
1710 continue;
1711
1712 // Prefer a constant on the right-hand side, or an Argument if no constants.
1713 if (isa<Constant>(LHS) || (isa<Argument>(LHS) && !isa<Constant>(RHS)))
1714 std::swap(LHS, RHS);
1715 assert((isa<Argument>(LHS) || isa<Instruction>(LHS)) && "Unexpected value!");
1716
1717 // If there is no obvious reason to prefer the left-hand side over the
1718 // right-hand side, ensure the longest lived term is on the right-hand side,
1719 // so the shortest lived term will be replaced by the longest lived.
1720 // This tends to expose more simplifications.
1721 uint32_t LVN = VN.lookupOrAdd(LHS);
1722 if ((isa<Argument>(LHS) && isa<Argument>(RHS)) ||
1723 (isa<Instruction>(LHS) && isa<Instruction>(RHS))) {
1724 // Move the 'oldest' value to the right-hand side, using the value number
1725 // as a proxy for age.
1726 uint32_t RVN = VN.lookupOrAdd(RHS);
1727 if (LVN < RVN) {
1728 std::swap(LHS, RHS);
1729 LVN = RVN;
1730 }
1731 }
1732
1733 // If value numbering later sees that an instruction in the scope is equal
1734 // to 'LHS' then ensure it will be turned into 'RHS'. In order to preserve
1735 // the invariant that instructions only occur in the leader table for their
1736 // own value number (this is used by removeFromLeaderTable), do not do this
1737 // if RHS is an instruction (if an instruction in the scope is morphed into
1738 // LHS then it will be turned into RHS by the next GVN iteration anyway, so
1739 // using the leader table is about compiling faster, not optimizing better).
1740 // The leader table only tracks basic blocks, not edges. Only add to if we
1741 // have the simple case where the edge dominates the end.
1742 if (RootDominatesEnd && !isa<Instruction>(RHS))
1743 addToLeaderTable(LVN, RHS, Root.getEnd());
1744
1745 // Replace all occurrences of 'LHS' with 'RHS' everywhere in the scope. As
1746 // LHS always has at least one use that is not dominated by Root, this will
1747 // never do anything if LHS has only one use.
1748 if (!LHS->hasOneUse()) {
1749 unsigned NumReplacements =
1750 DominatesByEdge
1751 ? replaceDominatedUsesWith(LHS, RHS, *DT, Root)
1752 : replaceDominatedUsesWith(LHS, RHS, *DT, Root.getStart());
1753
1754 Changed |= NumReplacements > 0;
1755 NumGVNEqProp += NumReplacements;
1756 // Cached information for anything that uses LHS will be invalid.
1757 if (MD)
1758 MD->invalidateCachedPointerInfo(LHS);
1759 }
1760
1761 // Now try to deduce additional equalities from this one. For example, if
1762 // the known equality was "(A != B)" == "false" then it follows that A and B
1763 // are equal in the scope. Only boolean equalities with an explicit true or
1764 // false RHS are currently supported.
1765 if (!RHS->getType()->isIntegerTy(1))
1766 // Not a boolean equality - bail out.
1767 continue;
1768 ConstantInt *CI = dyn_cast<ConstantInt>(RHS);
1769 if (!CI)
1770 // RHS neither 'true' nor 'false' - bail out.
1771 continue;
1772 // Whether RHS equals 'true'. Otherwise it equals 'false'.
1773 bool isKnownTrue = CI->isMinusOne();
1774 bool isKnownFalse = !isKnownTrue;
1775
1776 // If "A && B" is known true then both A and B are known true. If "A || B"
1777 // is known false then both A and B are known false.
1778 Value *A, *B;
1779 if ((isKnownTrue && match(LHS, m_And(m_Value(A), m_Value(B)))) ||
1780 (isKnownFalse && match(LHS, m_Or(m_Value(A), m_Value(B))))) {
1781 Worklist.push_back(std::make_pair(A, RHS));
1782 Worklist.push_back(std::make_pair(B, RHS));
1783 continue;
1784 }
1785
1786 // If we are propagating an equality like "(A == B)" == "true" then also
1787 // propagate the equality A == B. When propagating a comparison such as
1788 // "(A >= B)" == "true", replace all instances of "A < B" with "false".
1789 if (CmpInst *Cmp = dyn_cast<CmpInst>(LHS)) {
1790 Value *Op0 = Cmp->getOperand(0), *Op1 = Cmp->getOperand(1);
1791
1792 // If "A == B" is known true, or "A != B" is known false, then replace
1793 // A with B everywhere in the scope.
1794 if ((isKnownTrue && Cmp->getPredicate() == CmpInst::ICMP_EQ) ||
1795 (isKnownFalse && Cmp->getPredicate() == CmpInst::ICMP_NE))
1796 Worklist.push_back(std::make_pair(Op0, Op1));
1797
1798 // Handle the floating point versions of equality comparisons too.
1799 if ((isKnownTrue && Cmp->getPredicate() == CmpInst::FCMP_OEQ) ||
1800 (isKnownFalse && Cmp->getPredicate() == CmpInst::FCMP_UNE)) {
1801
1802 // Floating point -0.0 and 0.0 compare equal, so we can only
1803 // propagate values if we know that we have a constant and that
1804 // its value is non-zero.
1805
1806 // FIXME: We should do this optimization if 'no signed zeros' is
1807 // applicable via an instruction-level fast-math-flag or some other
1808 // indicator that relaxed FP semantics are being used.
1809
1810 if (isa<ConstantFP>(Op1) && !cast<ConstantFP>(Op1)->isZero())
1811 Worklist.push_back(std::make_pair(Op0, Op1));
1812 }
1813
1814 // If "A >= B" is known true, replace "A < B" with false everywhere.
1815 CmpInst::Predicate NotPred = Cmp->getInversePredicate();
1816 Constant *NotVal = ConstantInt::get(Cmp->getType(), isKnownFalse);
1817 // Since we don't have the instruction "A < B" immediately to hand, work
1818 // out the value number that it would have and use that to find an
1819 // appropriate instruction (if any).
1820 uint32_t NextNum = VN.getNextUnusedValueNumber();
1821 uint32_t Num = VN.lookupOrAddCmp(Cmp->getOpcode(), NotPred, Op0, Op1);
1822 // If the number we were assigned was brand new then there is no point in
1823 // looking for an instruction realizing it: there cannot be one!
1824 if (Num < NextNum) {
1825 Value *NotCmp = findLeader(Root.getEnd(), Num);
1826 if (NotCmp && isa<Instruction>(NotCmp)) {
1827 unsigned NumReplacements =
1828 DominatesByEdge
1829 ? replaceDominatedUsesWith(NotCmp, NotVal, *DT, Root)
1830 : replaceDominatedUsesWith(NotCmp, NotVal, *DT,
1831 Root.getStart());
1832 Changed |= NumReplacements > 0;
1833 NumGVNEqProp += NumReplacements;
1834 // Cached information for anything that uses NotCmp will be invalid.
1835 if (MD)
1836 MD->invalidateCachedPointerInfo(NotCmp);
1837 }
1838 }
1839 // Ensure that any instruction in scope that gets the "A < B" value number
1840 // is replaced with false.
1841 // The leader table only tracks basic blocks, not edges. Only add to if we
1842 // have the simple case where the edge dominates the end.
1843 if (RootDominatesEnd)
1844 addToLeaderTable(Num, NotVal, Root.getEnd());
1845
1846 continue;
1847 }
1848 }
1849
1850 return Changed;
1851}

1853/// When calculating availability, handle an instruction
1854/// by inserting it into the appropriate sets
1855bool GVN::processInstruction(Instruction *I) {
1856 // Ignore dbg info intrinsics.
1857 if (isa<DbgInfoIntrinsic>(I))
1858 return false;
1859
1860 // If the instruction can be easily simplified then do so now in preference
1861 // to value numbering it. Value numbering often exposes redundancies, for
1862 // example if it determines that %y is equal to %x then the instruction
1863 // "%z = and i32 %x, %y" becomes "%z = and i32 %x, %x" which we now simplify.
1864 const DataLayout &DL = I->getModule()->getDataLayout();
1865 if (Value *V = SimplifyInstruction(I, {DL, TLI, DT, AC})) {
1866 bool Changed = false;
1867 if (!I->use_empty()) {
1868 I->replaceAllUsesWith(V);
1869 Changed = true;
1870 }
1871 if (isInstructionTriviallyDead(I, TLI)) {
1872 markInstructionForDeletion(I);
1873 Changed = true;
1874 }
1875 if (Changed) {
1876 if (MD && V->getType()->isPtrOrPtrVectorTy())
1877 MD->invalidateCachedPointerInfo(V);
1878 ++NumGVNSimpl;
1879 return true;
1880 }
1881 }
1882
1883 if (IntrinsicInst *IntrinsicI = dyn_cast<IntrinsicInst>(I))
1884 if (IntrinsicI->getIntrinsicID() == Intrinsic::assume)
1885 return processAssumeIntrinsic(IntrinsicI);
1886
1887 if (LoadInst *LI = dyn_cast<LoadInst>(I)) {
1888 if (processLoad(LI))
1889 return true;
1890
1891 unsigned Num = VN.lookupOrAdd(LI);
1892 addToLeaderTable(Num, LI, LI->getParent());
1893 return false;
1894 }
1895
1896 // For conditional branches, we can perform simple conditional propagation on
1897 // the condition value itself.
1898 if (BranchInst *BI = dyn_cast<BranchInst>(I)) {
1899 if (!BI->isConditional())
1900 return false;
1901
1902 if (isa<Constant>(BI->getCondition()))
1903 return processFoldableCondBr(BI);
1904
1905 Value *BranchCond = BI->getCondition();
1906 BasicBlock *TrueSucc = BI->getSuccessor(0);
1907 BasicBlock *FalseSucc = BI->getSuccessor(1);
1908 // Avoid multiple edges early.
1909 if (TrueSucc == FalseSucc)
1910 return false;
1911
1912 BasicBlock *Parent = BI->getParent();
1913 bool Changed = false;
1914
1915 Value *TrueVal = ConstantInt::getTrue(TrueSucc->getContext());
1916 BasicBlockEdge TrueE(Parent, TrueSucc);
1917 Changed |= propagateEquality(BranchCond, TrueVal, TrueE, true);
1918
1919 Value *FalseVal = ConstantInt::getFalse(FalseSucc->getContext());
1920 BasicBlockEdge FalseE(Parent, FalseSucc);
1921 Changed |= propagateEquality(BranchCond, FalseVal, FalseE, true);
1922
1923 return Changed;
1924 }
1925
1926 // For switches, propagate the case values into the case destinations.
1927 if (SwitchInst *SI = dyn_cast<SwitchInst>(I)) {
1928 Value *SwitchCond = SI->getCondition();
1929 BasicBlock *Parent = SI->getParent();
1930 bool Changed = false;
1931
1932 // Remember how many outgoing edges there are to every successor.
1933 SmallDenseMap<BasicBlock *, unsigned, 16> SwitchEdges;
1934 for (unsigned i = 0, n = SI->getNumSuccessors(); i != n; ++i)
1935 ++SwitchEdges[SI->getSuccessor(i)];
1936
1937 for (SwitchInst::CaseIt i = SI->case_begin(), e = SI->case_end();
1938 i != e; ++i) {
1939 BasicBlock *Dst = i->getCaseSuccessor();
1940 // If there is only a single edge, propagate the case value into it.
1941 if (SwitchEdges.lookup(Dst) == 1) {
1942 BasicBlockEdge E(Parent, Dst);
1943 Changed |= propagateEquality(SwitchCond, i->getCaseValue(), E, true);
1944 }
1945 }
1946 return Changed;
1947 }
1948
1949 // Instructions with void type don't return a value, so there's
1950 // no point in trying to find redundancies in them.
1951 if (I->getType()->isVoidTy())
1952 return false;
1953
1954 uint32_t NextNum = VN.getNextUnusedValueNumber();
1955 unsigned Num = VN.lookupOrAdd(I);
1956
1957 // Allocations are always uniquely numbered, so we can save time and memory
1958 // by fast failing them.
1959 if (isa<AllocaInst>(I) || I->isTerminator() || isa<PHINode>(I)) {
1960 addToLeaderTable(Num, I, I->getParent());
1961 return false;
1962 }
1963
1964 // If the number we were assigned was a brand new VN, then we don't
1965 // need to do a lookup to see if the number already exists
1966 // somewhere in the domtree: it can't!
1967 if (Num >= NextNum) {
1968 addToLeaderTable(Num, I, I->getParent());
1969 return false;
1970 }
1971
1972 // Perform fast-path value-number based elimination of values inherited from
1973 // dominators.
1974 Value *Repl = findLeader(I->getParent(), Num);
1975 if (!Repl) {
1976 // Failure, just remember this instance for future use.
1977 addToLeaderTable(Num, I, I->getParent());
1978 return false;
1979 } else if (Repl == I) {
1980 // If I was the result of a shortcut PRE, it might already be in the table
1981 // and the best replacement for itself. Nothing to do.
1982 return false;
1983 }
1984
1985 // Remove it!
1986 patchAndReplaceAllUsesWith(I, Repl);
1987 if (MD && Repl->getType()->isPtrOrPtrVectorTy())
1988 MD->invalidateCachedPointerInfo(Repl);
1989 markInstructionForDeletion(I);
1990 return true;
1991}

1993/// runOnFunction - This is the main transformation entry point for a function.
1994bool GVN::runImpl(Function &F, AssumptionCache &RunAC, DominatorTree &RunDT,
1995 const TargetLibraryInfo &RunTLI, AAResults &RunAA,
1996 MemoryDependenceResults *RunMD, LoopInfo *LI,
1997 OptimizationRemarkEmitter *RunORE) {
1998 AC = &RunAC;
1999 DT = &RunDT;
2000 VN.setDomTree(DT);
2001 TLI = &RunTLI;
2002 VN.setAliasAnalysis(&RunAA);
2003 MD = RunMD;
2004 ImplicitControlFlowTracking ImplicitCFT(DT);
2005 ICF = &ImplicitCFT;
2006 VN.setMemDep(MD);
2007 ORE = RunORE;
2008 InvalidBlockRPONumbers = true;
2009
2010 bool Changed = false;
2011 bool ShouldContinue = true;
2012
2013 DomTreeUpdater DTU(DT, DomTreeUpdater::UpdateStrategy::Eager);
2014 // Merge unconditional branches, allowing PRE to catch more
2015 // optimization opportunities.
2016 for (Function::iterator FI = F.begin(), FE = F.end(); FI != FE; ) {
2017 BasicBlock *BB = &*FI++;
2018
2019 bool removedBlock = MergeBlockIntoPredecessor(BB, &DTU, LI, nullptr, MD);
2020 if (removedBlock)
2021 ++NumGVNBlocks;
2022
2023 Changed |= removedBlock;
2024 }
2025
2026 unsigned Iteration = 0;
2027 while (ShouldContinue) {
2028 LLVM_DEBUG(dbgs() << "GVN iteration: " << Iteration << "\n");
2029 ShouldContinue = iterateOnFunction(F);
2030 Changed |= ShouldContinue;
2031 ++Iteration;
2032 }
2033
2034 if (EnablePRE) {
2035 // Fabricate val-num for dead-code in order to suppress assertion in
2036 // performPRE().
2037 assignValNumForDeadCode();
2038 bool PREChanged = true;
2039 while (PREChanged) {
2040 PREChanged = performPRE(F);
2041 Changed |= PREChanged;
2042 }
2043 }
2044
2045 // FIXME: Should perform GVN again after PRE does something. PRE can move
2046 // computations into blocks where they become fully redundant. Note that
2047 // we can't do this until PRE's critical edge splitting updates memdep.
2048 // Actually, when this happens, we should just fully integrate PRE into GVN.
2049
2050 cleanupGlobalSets();
2051 // Do not cleanup DeadBlocks in cleanupGlobalSets() as it's called for each
2052 // iteration.
2053 DeadBlocks.clear();
2054
2055 return Changed;
2056}

2058bool GVN::processBlock(BasicBlock *BB) {
2059 // FIXME: Kill off InstrsToErase by doing erasing eagerly in a helper function
2060 // (and incrementing BI before processing an instruction).
2061 assert(InstrsToErase.empty() &&
2062 "We expect InstrsToErase to be empty across iterations");
2063 if (DeadBlocks.count(BB))
2064 return false;
2065
2066 // Clearing map before every BB because it can be used only for single BB.
2067 ReplaceWithConstMap.clear();
2068 bool ChangedFunction = false;
2069
2070 for (BasicBlock::iterator BI = BB->begin(), BE = BB->end();
2071 BI != BE;) {
2072 if (!ReplaceWithConstMap.empty())
2073 ChangedFunction |= replaceOperandsWithConsts(&*BI);
2074 ChangedFunction |= processInstruction(&*BI);
2075
2076 if (InstrsToErase.empty()) {
2077 ++BI;
2078 continue;
2079 }
2080
2081 // If we need some instructions deleted, do it now.
2082 NumGVNInstr += InstrsToErase.size();
2083
2084 // Avoid iterator invalidation.
2085 bool AtStart = BI == BB->begin();
2086 if (!AtStart)
2087 --BI;
2088
2089 for (auto *I : InstrsToErase) {
2090 assert(I->getParent() == BB && "Removing instruction from wrong block?");
2091 LLVM_DEBUG(dbgs() << "GVN removed: " << *I << '\n');
2092 salvageDebugInfo(*I);
2093 if (MD) MD->removeInstruction(I);
2094 LLVM_DEBUG(verifyRemoved(I));
2095 ICF->removeInstruction(I);
2096 I->eraseFromParent();
2097 }
2098 InstrsToErase.clear();
2099
2100 if (AtStart)
2101 BI = BB->begin();
2102 else
2103 ++BI;
2104 }
2105
2106 return ChangedFunction;
2107}

2109// Instantiate an expression in a predecessor that lacked it.
2110bool GVN::performScalarPREInsertion(Instruction *Instr, BasicBlock *Pred,
2111 BasicBlock *Curr, unsigned int ValNo) {
2112 // Because we are going top-down through the block, all value numbers
2113 // will be available in the predecessor by the time we need them. Any
2114 // that weren't originally present will have been instantiated earlier
2115 // in this loop.
2116 bool success = true;
2117 for (unsigned i = 0, e = Instr->getNumOperands(); i != e; ++i) {
2118 Value *Op = Instr->getOperand(i);
2119 if (isa<Argument>(Op) || isa<Constant>(Op) || isa<GlobalValue>(Op))
2120 continue;
2121 // This could be a newly inserted instruction, in which case, we won't
2122 // find a value number, and should give up before we hurt ourselves.
2123 // FIXME: Rewrite the infrastructure to let it easier to value number
2124 // and process newly inserted instructions.
2125 if (!VN.exists(Op)) {
2126 success = false;
2127 break;
2128 }
2129 uint32_t TValNo =
2130 VN.phiTranslate(Pred, Curr, VN.lookup(Op), *this);
2131 if (Value *V = findLeader(Pred, TValNo)) {
2132 Instr->setOperand(i, V);
2133 } else {
2134 success = false;
2135 break;
2136 }
2137 }
2138
2139 // Fail out if we encounter an operand that is not available in
2140 // the PRE predecessor. This is typically because of loads which
2141 // are not value numbered precisely.
2142 if (!success)
2143 return false;
2144
2145 Instr->insertBefore(Pred->getTerminator());
2146 Instr->setName(Instr->getName() + ".pre");
2147 Instr->setDebugLoc(Instr->getDebugLoc());
2148
2149 unsigned Num = VN.lookupOrAdd(Instr);
2150 VN.add(Instr, Num);
2151
2152 // Update the availability map to include the new instruction.
2153 addToLeaderTable(Num, Instr, Pred);
2154 return true;
2155}

2157bool GVN::performScalarPRE(Instruction *CurInst) {
2158 if (isa<AllocaInst>(CurInst) || CurInst->isTerminator() ||
2159 isa<PHINode>(CurInst) || CurInst->getType()->isVoidTy() ||
2160 CurInst->mayReadFromMemory() || CurInst->mayHaveSideEffects() ||
2161 isa<DbgInfoIntrinsic>(CurInst))
2162 return false;
2163
2164 // Don't do PRE on compares. The PHI would prevent CodeGenPrepare from
2165 // sinking the compare again, and it would force the code generator to
2166 // move the i1 from processor flags or predicate registers into a general
2167 // purpose register.
2168 if (isa<CmpInst>(CurInst))
2169 return false;
2170
2171 // Don't do PRE on GEPs. The inserted PHI would prevent CodeGenPrepare from
2172 // sinking the addressing mode computation back to its uses. Extending the
2173 // GEP's live range increases the register pressure, and therefore it can
2174 // introduce unnecessary spills.
2175 //
2176 // This doesn't prevent Load PRE. PHI translation will make the GEP available
2177 // to the load by moving it to the predecessor block if necessary.
2178 if (isa<GetElementPtrInst>(CurInst))
2179 return false;
2180
2181 // We don't currently value number ANY inline asm calls.
2182 if (auto *CallB = dyn_cast<CallBase>(CurInst))
2183 if (CallB->isInlineAsm())
2184 return false;
2185
2186 uint32_t ValNo = VN.lookup(CurInst);
2187
2188 // Look for the predecessors for PRE opportunities. We're
2189 // only trying to solve the basic diamond case, where
2190 // a value is computed in the successor and one predecessor,
2191 // but not the other. We also explicitly disallow cases
2192 // where the successor is its own predecessor, because they're
2193 // more complicated to get right.
2194 unsigned NumWith = 0;
2195 unsigned NumWithout = 0;
2196 BasicBlock *PREPred = nullptr;
2197 BasicBlock *CurrentBlock = CurInst->getParent();
2198
2199 // Update the RPO numbers for this function.
2200 if (InvalidBlockRPONumbers)
2201 assignBlockRPONumber(*CurrentBlock->getParent());
2202
2203 SmallVector<std::pair<Value *, BasicBlock *>, 8> predMap;
2204 for (BasicBlock *P : predecessors(CurrentBlock)) {
2205 // We're not interested in PRE where blocks with predecessors that are
2206 // not reachable.
2207 if (!DT->isReachableFromEntry(P)) {
2208 NumWithout = 2;
2209 break;
2210 }
2211 // It is not safe to do PRE when P->CurrentBlock is a loop backedge, and
2212 // when CurInst has operand defined in CurrentBlock (so it may be defined
2213 // by phi in the loop header).
2214 assert(BlockRPONumber.count(P) && BlockRPONumber.count(CurrentBlock) &&
2215 "Invalid BlockRPONumber map.");
2216 if (BlockRPONumber[P] >= BlockRPONumber[CurrentBlock] &&
2217 llvm::any_of(CurInst->operands(), [&](const Use &U) {
2218 if (auto *Inst = dyn_cast<Instruction>(U.get()))
2219 return Inst->getParent() == CurrentBlock;
2220 return false;
2221 })) {
2222 NumWithout = 2;
2223 break;
2224 }
2225
2226 uint32_t TValNo = VN.phiTranslate(P, CurrentBlock, ValNo, *this);
2227 Value *predV = findLeader(P, TValNo);
2228 if (!predV) {
2229 predMap.push_back(std::make_pair(static_cast<Value *>(nullptr), P));
2230 PREPred = P;
2231 ++NumWithout;
2232 } else if (predV == CurInst) {
2233 /* CurInst dominates this predecessor. */
2234 NumWithout = 2;
2235 break;
2236 } else {
2237 predMap.push_back(std::make_pair(predV, P));
2238 ++NumWith;
2239 }
2240 }
2241
2242 // Don't do PRE when it might increase code size, i.e. when
2243 // we would need to insert instructions in more than one pred.
2244 if (NumWithout > 1 || NumWith == 0)
2245 return false;
2246
2247 // We may have a case where all predecessors have the instruction,
2248 // and we just need to insert a phi node. Otherwise, perform
2249 // insertion.
2250 Instruction *PREInstr = nullptr;
2251
2252 if (NumWithout != 0) {
2253 if (!isSafeToSpeculativelyExecute(CurInst)) {
2254 // It is only valid to insert a new instruction if the current instruction
2255 // is always executed. An instruction with implicit control flow could
2256 // prevent us from doing it. If we cannot speculate the execution, then
2257 // PRE should be prohibited.
2258 if (ICF->isDominatedByICFIFromSameBlock(CurInst))
2259 return false;
2260 }
2261
2262 // Don't do PRE across indirect branch.
2263 if (isa<IndirectBrInst>(PREPred->getTerminator()))
2264 return false;
2265
2266 // Don't do PRE across callbr.
2267 // FIXME: Can we do this across the fallthrough edge?
2268 if (isa<CallBrInst>(PREPred->getTerminator()))
2269 return false;
2270
2271 // We can't do PRE safely on a critical edge, so instead we schedule
2272 // the edge to be split and perform the PRE the next time we iterate
2273 // on the function.
2274 unsigned SuccNum = GetSuccessorNumber(PREPred, CurrentBlock);
2275 if (isCriticalEdge(PREPred->getTerminator(), SuccNum)) {
2276 toSplit.push_back(std::make_pair(PREPred->getTerminator(), SuccNum));
2277 return false;
2278 }
2279 // We need to insert somewhere, so let's give it a shot
2280 PREInstr = CurInst->clone();
2281 if (!performScalarPREInsertion(PREInstr, PREPred, CurrentBlock, ValNo)) {
2282 // If we failed insertion, make sure we remove the instruction.
2283 LLVM_DEBUG(verifyRemoved(PREInstr));
2284 PREInstr->deleteValue();
2285 return false;
2286 }
2287 }
2288
2289 // Either we should have filled in the PRE instruction, or we should
2290 // not have needed insertions.
2291 assert(PREInstr != nullptr || NumWithout == 0);
2292
2293 ++NumGVNPRE;
2294
2295 // Create a PHI to make the value available in this block.
2296 PHINode *Phi =
2297 PHINode::Create(CurInst->getType(), predMap.size(),
2298 CurInst->getName() + ".pre-phi", &CurrentBlock->front());
2299 for (unsigned i = 0, e = predMap.size(); i != e; ++i) {
2300 if (Value *V = predMap[i].first) {
2301 // If we use an existing value in this phi, we have to patch the original
2302 // value because the phi will be used to replace a later value.
2303 patchReplacementInstruction(CurInst, V);
2304 Phi->addIncoming(V, predMap[i].second);
2305 } else
2306 Phi->addIncoming(PREInstr, PREPred);
2307 }
2308
2309 VN.add(Phi, ValNo);
2310 // After creating a new PHI for ValNo, the phi translate result for ValNo will
2311 // be changed, so erase the related stale entries in phi translate cache.
2312 VN.eraseTranslateCacheEntry(ValNo, *CurrentBlock);
2313 addToLeaderTable(ValNo, Phi, CurrentBlock);
2314 Phi->setDebugLoc(CurInst->getDebugLoc());
2315 CurInst->replaceAllUsesWith(Phi);
2316 if (MD && Phi->getType()->isPtrOrPtrVectorTy())
2317 MD->invalidateCachedPointerInfo(Phi);
2318 VN.erase(CurInst);
2319 removeFromLeaderTable(ValNo, CurInst, CurrentBlock);
2320
2321 LLVM_DEBUG(dbgs() << "GVN PRE removed: " << *CurInst << '\n');
2322 if (MD)
2323 MD->removeInstruction(CurInst);
2324 LLVM_DEBUG(verifyRemoved(CurInst));
2325 // FIXME: Intended to be markInstructionForDeletion(CurInst), but it causes
2326 // some assertion failures.
2327 ICF->removeInstruction(CurInst);
2328 CurInst->eraseFromParent();
2329 ++NumGVNInstr;
2330
2331 return true;
2332}

2334/// Perform a purely local form of PRE that looks for diamond
2335/// control flow patterns and attempts to perform simple PRE at the join point.
2336bool GVN::performPRE(Function &F) {
2337 bool Changed = false;
2338 for (BasicBlock *CurrentBlock : depth_first(&F.getEntryBlock())) {
2339 // Nothing to PRE in the entry block.
2340 if (CurrentBlock == &F.getEntryBlock())
2341 continue;
2342
2343 // Don't perform PRE on an EH pad.
2344 if (CurrentBlock->isEHPad())
2345 continue;
2346
2347 for (BasicBlock::iterator BI = CurrentBlock->begin(),
2348 BE = CurrentBlock->end();
2349 BI != BE;) {
2350 Instruction *CurInst = &*BI++;
2351 Changed |= performScalarPRE(CurInst);
2352 }
2353 }
2354
2355 if (splitCriticalEdges())
2356 Changed = true;
2357
2358 return Changed;
2359}

2413/// Verify that the specified instruction does not occur in our
2414/// internal data structures.
2415void GVN::verifyRemoved(const Instruction *Inst) const {
2416 VN.verifyRemoved(Inst);
2417
2418 // Walk through the value number scope to make sure the instruction isn't
2419 // ferreted away in it.
2420 for (DenseMap<uint32_t, LeaderTableEntry>::const_iterator
2421 I = LeaderTable.begin(), E = LeaderTable.end(); I != E; ++I) {

2423 assert(Node->Val != Inst && "Inst still in value numbering scope!");

2427 assert(Node->Val != Inst && "Inst still in value numbering scope!");

2432/// BB is declared dead, which implied other blocks become dead as well. This
2433/// function is to add all these blocks to "DeadBlocks". For the dead blocks'
2434/// live successors, update their phi nodes by replacing the operands
2435/// corresponding to dead blocks with UndefVal.
2436void GVN::addDeadBlock(BasicBlock *BB) {
2437 SmallVector<BasicBlock *, 4> NewDead;
2438 SmallSetVector<BasicBlock *, 4> DF;
2439
2440 NewDead.push_back(BB);
2441 while (!NewDead.empty()) {
2442 BasicBlock *D = NewDead.pop_back_val();
2443 if (DeadBlocks.count(D))
2444 continue;
2445
2446 // All blocks dominated by D are dead.
2447 SmallVector<BasicBlock *, 8> Dom;
2448 DT->getDescendants(D, Dom);
2449 DeadBlocks.insert(Dom.begin(), Dom.end());
2450
2451 // Figure out the dominance-frontier(D).
2452 for (BasicBlock *B : Dom) {
2453 for (BasicBlock *S : successors(B)) {
2454 if (DeadBlocks.count(S))
2455 continue;
2456
2457 bool AllPredDead = true;
2458 for (BasicBlock *P : predecessors(S))
2459 if (!DeadBlocks.count(P)) {
2460 AllPredDead = false;
2461 break;
2462 }
2463
2464 if (!AllPredDead) {
2465 // S could be proved dead later on. That is why we don't update phi
2466 // operands at this moment.
2467 DF.insert(S);
2468 } else {
2469 // While S is not dominated by D, it is dead by now. This could take
2470 // place if S already have a dead predecessor before D is declared
2471 // dead.
2472 NewDead.push_back(S);
2473 }
2474 }
2475 }
2476 }
2477
2478 // For the dead blocks' live successors, update their phi nodes by replacing
2479 // the operands corresponding to dead blocks with UndefVal.
2480 for(SmallSetVector<BasicBlock *, 4>::iterator I = DF.begin(), E = DF.end();
2481 I != E; I++) {
2482 BasicBlock *B = *I;
2483 if (DeadBlocks.count(B))
2484 continue;
2485
2486 SmallVector<BasicBlock *, 4> Preds(pred_begin(B), pred_end(B));
2487 for (BasicBlock *P : Preds) {
2488 if (!DeadBlocks.count(P))
2489 continue;
2490
2491 if (isCriticalEdge(P->getTerminator(), GetSuccessorNumber(P, B))) {
2492 if (BasicBlock *S = splitCriticalEdges(P, B))
2493 DeadBlocks.insert(P = S);
2494 }
2495
2496 for (BasicBlock::iterator II = B->begin(); isa<PHINode>(II); ++II) {
2497 PHINode &Phi = cast<PHINode>(*II);
2498 Phi.setIncomingValue(Phi.getBasicBlockIndex(P),
2499 UndefValue::get(Phi.getType()));
2500 if (MD)
2501 MD->invalidateCachedPointerInfo(&Phi);

2507// If the given branch is recognized as a foldable branch (i.e. conditional
2508// branch with constant condition), it will perform following analyses and
2509// transformation.
2510// 1) If the dead out-coming edge is a critical-edge, split it. Let
2511// R be the target of the dead out-coming edge.
2512// 1) Identify the set of dead blocks implied by the branch's dead outcoming
2513// edge. The result of this step will be {X| X is dominated by R}
2514// 2) Identify those blocks which haves at least one dead predecessor. The
2515// result of this step will be dominance-frontier(R).
2516// 3) Update the PHIs in DF(R) by replacing the operands corresponding to
2517// dead blocks with "UndefVal" in an hope these PHIs will optimized away.
2518//
2519// Return true iff *NEW* dead code are found.
2520bool GVN::processFoldableCondBr(BranchInst *BI) {
2521 if (!BI || BI->isUnconditional())
2522 return false;
2523
2524 // If a branch has two identical successors, we cannot declare either dead.
2525 if (BI->getSuccessor(0) == BI->getSuccessor(1))
2526 return false;
2527
2528 ConstantInt *Cond = dyn_cast<ConstantInt>(BI->getCondition());
2529 if (!Cond)
2530 return false;
2531
2532 BasicBlock *DeadRoot =
2533 Cond->getZExtValue() ? BI->getSuccessor(1) : BI->getSuccessor(0);
2534 if (DeadBlocks.count(DeadRoot))
2535 return false;
2536
2537 if (!DeadRoot->getSinglePredecessor())
2538 DeadRoot = splitCriticalEdges(BI->getParent(), DeadRoot);
2539
2540 addDeadBlock(DeadRoot);
2541 return true;
2542}

2544// performPRE() will trigger assert if it comes across an instruction without
2545// associated val-num. As it normally has far more live instructions than dead
2546// instructions, it makes more sense just to "fabricate" a val-number for the
2547// dead code than checking if instruction involved is dead or not.
2548void GVN::assignValNumForDeadCode() {
2549 for (BasicBlock *BB : DeadBlocks) {
2550 for (Instruction &Inst : *BB) {
2551 unsigned ValNum = VN.lookupOrAdd(&Inst);
2552 addToLeaderTable(ValNum, &Inst, BB);
2553 }
2554 }
