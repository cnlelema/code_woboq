138SanitizerCoverageOptions getOptions(int LegacyCoverageLevel) {
139 SanitizerCoverageOptions Res;
140 switch (LegacyCoverageLevel) {
141 case 0:
142 Res.CoverageType = SanitizerCoverageOptions::SCK_None;
143 break;
144 case 1:
145 Res.CoverageType = SanitizerCoverageOptions::SCK_Function;
146 break;
147 case 2:
148 Res.CoverageType = SanitizerCoverageOptions::SCK_BB;
149 break;
150 case 3:
151 Res.CoverageType = SanitizerCoverageOptions::SCK_Edge;
152 break;
153 case 4:
154 Res.CoverageType = SanitizerCoverageOptions::SCK_Edge;
155 Res.IndirectCalls = true;
156 break;
157 }
158 return Res;
159}
160
161SanitizerCoverageOptions OverrideFromCL(SanitizerCoverageOptions Options) {
162 // Sets CoverageType and IndirectCalls.
163 SanitizerCoverageOptions CLOpts = getOptions(ClCoverageLevel);
164 Options.CoverageType = std::max(Options.CoverageType, CLOpts.CoverageType);
165 Options.IndirectCalls |= CLOpts.IndirectCalls;
166 Options.TraceCmp |= ClCMPTracing;
167 Options.TraceDiv |= ClDIVTracing;
168 Options.TraceGep |= ClGEPTracing;
169 Options.TracePC |= ClTracePC;
170 Options.TracePCGuard |= ClTracePCGuard;
171 Options.Inline8bitCounters |= ClInline8bitCounters;
172 Options.PCTable |= ClCreatePCTable;
173 Options.NoPrune |= !ClPruneBlocks;
174 Options.StackDepth |= ClStackDepth;
175 if (!Options.TracePCGuard && !Options.TracePC &&
176 !Options.Inline8bitCounters && !Options.StackDepth)
177 Options.TracePCGuard = true; // TracePCGuard is default.
178 return Options;
179}

259std::pair<Value *, Value *>
260SanitizerCoverageModule::CreateSecStartEnd(Module &M, const char *Section,
261 Type *Ty) {
262 GlobalVariable *SecStart =
263 new GlobalVariable(M, Ty, false, GlobalVariable::ExternalLinkage, nullptr,
264 getSectionStart(Section));
265 SecStart->setVisibility(GlobalValue::HiddenVisibility);
266 GlobalVariable *SecEnd =
267 new GlobalVariable(M, Ty, false, GlobalVariable::ExternalLinkage,
268 nullptr, getSectionEnd(Section));
269 SecEnd->setVisibility(GlobalValue::HiddenVisibility);
270 IRBuilder<> IRB(M.getContext());
271 Value *SecEndPtr = IRB.CreatePointerCast(SecEnd, Ty);
272 if (!TargetTriple.isOSBinFormatCOFF())
273 return std::make_pair(IRB.CreatePointerCast(SecStart, Ty), SecEndPtr);
274
275 // Account for the fact that on windows-msvc __start_* symbols actually
276 // point to a uint64_t before the start of the array.
277 auto SecStartI8Ptr = IRB.CreatePointerCast(SecStart, GlobalsInt8PtrTy);
278 auto GEP = IRB.CreateGEP(Int8Ty, SecStartI8Ptr,
279 ConstantInt::get(IntptrTy, sizeof(uint64_t)));
280 return std::make_pair(IRB.CreatePointerCast(GEP, Ty), SecEndPtr);
281}

283Function *SanitizerCoverageModule::CreateInitCallsForSections(
284 Module &M, const char *CtorName, const char *InitFunctionName, Type *Ty,
285 const char *Section) {
286 auto SecStartEnd = CreateSecStartEnd(M, Section, Ty);
287 auto SecStart = SecStartEnd.first;
288 auto SecEnd = SecStartEnd.second;
289 Function *CtorFunc;
290 std::tie(CtorFunc, std::ignore) = createSanitizerCtorAndInitFunctions(
291 M, CtorName, InitFunctionName, {Ty, Ty}, {SecStart, SecEnd});
292 assert(CtorFunc->getName() == CtorName);
293
294 if (TargetTriple.supportsCOMDAT()) {
295 // Use comdat to dedup CtorFunc.
296 CtorFunc->setComdat(M.getOrInsertComdat(CtorName));
297 appendToGlobalCtors(M, CtorFunc, SanCtorAndDtorPriority, CtorFunc);
298 } else {
299 appendToGlobalCtors(M, CtorFunc, SanCtorAndDtorPriority);
300 }
301
302 if (TargetTriple.isOSBinFormatCOFF()) {
303 // In COFF files, if the contructors are set as COMDAT (they are because
304 // COFF supports COMDAT) and the linker flag /OPT:REF (strip unreferenced
305 // functions and data) is used, the constructors get stripped. To prevent
306 // this, give the constructors weak ODR linkage and ensure the linker knows
307 // to include the sancov constructor. This way the linker can deduplicate
308 // the constructors but always leave one copy.
309 CtorFunc->setLinkage(GlobalValue::WeakODRLinkage);
310 appendToUsed(M, CtorFunc);
311 }
312 return CtorFunc;
313}

315bool SanitizerCoverageModule::runOnModule(Module &M) {
316 if (Options.CoverageType == SanitizerCoverageOptions::SCK_None)
317 return false;
318 C = &(M.getContext());
319 DL = &M.getDataLayout();
320 CurModule = &M;
321 CurModuleUniqueId = getUniqueModuleId(CurModule);
322 TargetTriple = Triple(M.getTargetTriple());
323 FunctionGuardArray = nullptr;
324 Function8bitCounterArray = nullptr;
325 FunctionPCsArray = nullptr;
326 // XXXAR: assuming Address space zero pointer -> range of any pointer
327 IntptrTy = Type::getIntNTy(*C, DL->getPointerSizeInBits(0));
328
329 PcAddrTy = Type::getIntNTy(*C, DL->getIndexSizeInBits(DL->getProgramAddressSpace()));
330 PcAddrPtrTy = PcAddrTy->getPointerTo(DL->getGlobalsAddressSpace());
331 Type *VoidTy = Type::getVoidTy(*C);
332 IRBuilder<> IRB(*C);
333 GlobalsInt64PtrTy =
334 PointerType::get(IRB.getInt64Ty(), DL->getGlobalsAddressSpace());
335 GlobalsInt32PtrTy =
336 PointerType::get(IRB.getInt32Ty(), DL->getGlobalsAddressSpace());
337 GlobalsInt8PtrTy =
338 PointerType::get(IRB.getInt8Ty(), DL->getGlobalsAddressSpace());
339 Int64Ty = IRB.getInt64Ty();
340 Int32Ty = IRB.getInt32Ty();
341 Int16Ty = IRB.getInt16Ty();
342 Int8Ty = IRB.getInt8Ty();
343
344 SanCovTracePCIndir =
345 M.getOrInsertFunction(SanCovTracePCIndirName, VoidTy, IntptrTy);
346 // Make sure smaller parameters are zero-extended to i64 as required by the
347 // x86_64 ABI.
348 AttributeList SanCovTraceCmpZeroExtAL;
349 if (TargetTriple.getArch() == Triple::x86_64) {
350 SanCovTraceCmpZeroExtAL =
351 SanCovTraceCmpZeroExtAL.addParamAttribute(*C, 0, Attribute::ZExt);
352 SanCovTraceCmpZeroExtAL =
353 SanCovTraceCmpZeroExtAL.addParamAttribute(*C, 1, Attribute::ZExt);
354 }
355
356 SanCovTraceCmpFunction[0] =
357 M.getOrInsertFunction(SanCovTraceCmp1, SanCovTraceCmpZeroExtAL, VoidTy,
358 IRB.getInt8Ty(), IRB.getInt8Ty());
359 SanCovTraceCmpFunction[1] =
360 M.getOrInsertFunction(SanCovTraceCmp2, SanCovTraceCmpZeroExtAL, VoidTy,
361 IRB.getInt16Ty(), IRB.getInt16Ty());
362 SanCovTraceCmpFunction[2] =
363 M.getOrInsertFunction(SanCovTraceCmp4, SanCovTraceCmpZeroExtAL, VoidTy,
364 IRB.getInt32Ty(), IRB.getInt32Ty());
365 SanCovTraceCmpFunction[3] =
366 M.getOrInsertFunction(SanCovTraceCmp8, VoidTy, Int64Ty, Int64Ty);
367
368 SanCovTraceConstCmpFunction[0] = M.getOrInsertFunction(
369 SanCovTraceConstCmp1, SanCovTraceCmpZeroExtAL, VoidTy, Int8Ty, Int8Ty);
370 SanCovTraceConstCmpFunction[1] = M.getOrInsertFunction(
371 SanCovTraceConstCmp2, SanCovTraceCmpZeroExtAL, VoidTy, Int16Ty, Int16Ty);
372 SanCovTraceConstCmpFunction[2] = M.getOrInsertFunction(
373 SanCovTraceConstCmp4, SanCovTraceCmpZeroExtAL, VoidTy, Int32Ty, Int32Ty);
374 SanCovTraceConstCmpFunction[3] =
375 M.getOrInsertFunction(SanCovTraceConstCmp8, VoidTy, Int64Ty, Int64Ty);
376
377 {
378 AttributeList AL;
379 if (TargetTriple.getArch() == Triple::x86_64)
380 AL = AL.addParamAttribute(*C, 0, Attribute::ZExt);
381 SanCovTraceDivFunction[0] =
382 M.getOrInsertFunction(SanCovTraceDiv4, AL, VoidTy, IRB.getInt32Ty());
383 }
384 SanCovTraceDivFunction[1] =
385 M.getOrInsertFunction(SanCovTraceDiv8, VoidTy, Int64Ty);
386 SanCovTraceGepFunction =
387 M.getOrInsertFunction(SanCovTraceGep, VoidTy, IntptrTy);
388 SanCovTraceSwitchFunction = M.getOrInsertFunction(
389 SanCovTraceSwitchName, VoidTy, Int64Ty, GlobalsInt64PtrTy);
390
391 Constant *SanCovLowestStackConstant =
392 M.getOrInsertGlobal(SanCovLowestStackName, IntptrTy);
393 SanCovLowestStack = dyn_cast<GlobalVariable>(SanCovLowestStackConstant);
394 if (!SanCovLowestStack) {
395 C->emitError(StringRef("'") + SanCovLowestStackName +
396 "' should not be declared by the user");
397 return true;
398 }
399 SanCovLowestStack->setThreadLocalMode(
400 GlobalValue::ThreadLocalMode::InitialExecTLSModel);
401 if (Options.StackDepth && !SanCovLowestStack->isDeclaration())
402 SanCovLowestStack->setInitializer(Constant::getAllOnesValue(IntptrTy));
403
404 // We insert an empty inline asm after cov callbacks to avoid callback merge.
405 EmptyAsm = InlineAsm::get(FunctionType::get(IRB.getVoidTy(), false),
406 StringRef(""), StringRef(""),
407 /*hasSideEffects=*/true);
408
409 SanCovTracePC = M.getOrInsertFunction(SanCovTracePCName, VoidTy);
410 SanCovTracePCGuard =
411 M.getOrInsertFunction(SanCovTracePCGuardName, VoidTy, GlobalsInt32PtrTy);
412
413 for (auto &F : M)
414 runOnFunction(F);
415
416 Function *Ctor = nullptr;
417
418 if (FunctionGuardArray)
419 Ctor = CreateInitCallsForSections(M, SanCovModuleCtorTracePcGuardName,
420 SanCovTracePCGuardInitName, GlobalsInt32PtrTy,
421 SanCovGuardsSectionName);
422 if (Function8bitCounterArray)
423 Ctor = CreateInitCallsForSections(M, SanCovModuleCtor8bitCountersName,
424 SanCov8bitCountersInitName, GlobalsInt8PtrTy,
425 SanCovCountersSectionName);
426 if (Ctor && Options.PCTable) {
427 auto SecStartEnd = CreateSecStartEnd(M, SanCovPCsSectionName, PcAddrPtrTy);
428 FunctionCallee InitFunction = declareSanitizerInitFunction(
429 M, SanCovPCsInitName, {PcAddrPtrTy, PcAddrPtrTy});
430 IRBuilder<> IRBCtor(Ctor->getEntryBlock().getTerminator());
431 IRBCtor.CreateCall(InitFunction, {SecStartEnd.first, SecStartEnd.second});
432 }
433 // We don't reference these arrays directly in any of our runtime functions,
434 // so we need to prevent them from being dead stripped.
435 if (TargetTriple.isOSBinFormatMachO())
436 appendToUsed(M, GlobalsToAppendToUsed);
437 appendToCompilerUsed(M, GlobalsToAppendToCompilerUsed);
438 return true;
439}

441// True if block has successors and it dominates all of them.
442static bool isFullDominator(const BasicBlock *BB, const DominatorTree *DT) {
443 if (succ_begin(BB) == succ_end(BB))
444 return false;
445
446 for (const BasicBlock *SUCC : make_range(succ_begin(BB), succ_end(BB))) {
447 if (!DT->dominates(BB, SUCC))
448 return false;
449 }
450
451 return true;
452}
453
454// True if block has predecessors and it postdominates all of them.
455static bool isFullPostDominator(const BasicBlock *BB,
456 const PostDominatorTree *PDT) {
457 if (pred_begin(BB) == pred_end(BB))
458 return false;
459
460 for (const BasicBlock *PRED : make_range(pred_begin(BB), pred_end(BB))) {
461 if (!PDT->dominates(BB, PRED))
462 return false;
463 }
464
465 return true;
466}

468static bool shouldInstrumentBlock(const Function &F, const BasicBlock *BB,
469 const DominatorTree *DT,
470 const PostDominatorTree *PDT,
471 const SanitizerCoverageOptions &Options) {
472 // Don't insert coverage for blocks containing nothing but unreachable: we
473 // will never call __sanitizer_cov() for them, so counting them in
474 // NumberOfInstrumentedBlocks() might complicate calculation of code coverage
475 // percentage. Also, unreachable instructions frequently have no debug
476 // locations.
477 if (isa<UnreachableInst>(BB->getFirstNonPHIOrDbgOrLifetime()))
478 return false;
479
480 // Don't insert coverage into blocks without a valid insertion point
481 // (catchswitch blocks).
482 if (BB->getFirstInsertionPt() == BB->end())
483 return false;
484
485 if (Options.NoPrune || &F.getEntryBlock() == BB)
486 return true;
487
488 if (Options.CoverageType == SanitizerCoverageOptions::SCK_Function &&
489 &F.getEntryBlock() != BB)
490 return false;
491
492 // Do not instrument full dominators, or full post-dominators with multiple
493 // predecessors.
494 return !isFullDominator(BB, DT)
495 && !(isFullPostDominator(BB, PDT) && !BB->getSinglePredecessor());
496}

499// Returns true iff From->To is a backedge.
500// A twist here is that we treat From->To as a backedge if
501// * To dominates From or
502// * To->UniqueSuccessor dominates From
503static bool IsBackEdge(BasicBlock *From, BasicBlock *To,
504 const DominatorTree *DT) {
505 if (DT->dominates(To, From))
506 return true;
507 if (auto Next = To->getUniqueSuccessor())
508 if (DT->dominates(Next, From))
509 return true;
510 return false;
511}
512
513// Prunes uninteresting Cmp instrumentation:
514// * CMP instructions that feed into loop backedge branch.
515//
516// Note that Cmp pruning is controlled by the same flag as the
517// BB pruning.
518static bool IsInterestingCmp(ICmpInst *CMP, const DominatorTree *DT,
519 const SanitizerCoverageOptions &Options) {
520 if (!Options.NoPrune)
521 if (CMP->hasOneUse())
522 if (auto BR = dyn_cast<BranchInst>(CMP->user_back()))
523 for (BasicBlock *B : BR->successors())
524 if (IsBackEdge(BR->getParent(), B, DT))
525 return false;
526 return true;
527}

529bool SanitizerCoverageModule::runOnFunction(Function &F) {
530 if (F.empty())
531 return false;
532 if (F.getName().find(".module_ctor") != std::string::npos)
533 return false; // Should not instrument sanitizer init functions.
534 if (F.getName().startswith("__sanitizer_"))
535 return false; // Don't instrument __sanitizer_* callbacks.
536 // Don't touch available_externally functions, their actual body is elewhere.
537 if (F.getLinkage() == GlobalValue::AvailableExternallyLinkage)
538 return false;
539 // Don't instrument MSVC CRT configuration helpers. They may run before normal
540 // initialization.
541 if (F.getName() == "__local_stdio_printf_options" ||
542 F.getName() == "__local_stdio_scanf_options")
543 return false;
544 if (isa<UnreachableInst>(F.getEntryBlock().getTerminator()))
545 return false;
546 // Don't instrument functions using SEH for now. Splitting basic blocks like
547 // we do for coverage breaks WinEHPrepare.
548 // FIXME: Remove this when SEH no longer uses landingpad pattern matching.
549 if (F.hasPersonalityFn() &&
550 isAsynchronousEHPersonality(classifyEHPersonality(F.getPersonalityFn())))
551 return false;
552 if (Options.CoverageType >= SanitizerCoverageOptions::SCK_Edge)
553 SplitAllCriticalEdges(F, CriticalEdgeSplittingOptions().setIgnoreUnreachableDests());
554 SmallVector<Instruction *, 8> IndirCalls;
555 SmallVector<BasicBlock *, 16> BlocksToInstrument;
556 SmallVector<Instruction *, 8> CmpTraceTargets;
557 SmallVector<Instruction *, 8> SwitchTraceTargets;
558 SmallVector<BinaryOperator *, 8> DivTraceTargets;
559 SmallVector<GetElementPtrInst *, 8> GepTraceTargets;
560
561 const DominatorTree *DT =
562 &getAnalysis<DominatorTreeWrapperPass>(F).getDomTree();
563 const PostDominatorTree *PDT =
564 &getAnalysis<PostDominatorTreeWrapperPass>(F).getPostDomTree();
565 bool IsLeafFunc = true;
566
567 for (auto &BB : F) {
568 if (shouldInstrumentBlock(F, &BB, DT, PDT, Options))
569 BlocksToInstrument.push_back(&BB);
570 for (auto &Inst : BB) {
571 if (Options.IndirectCalls) {
572 CallSite CS(&Inst);
573 if (CS && !CS.getCalledFunction())
574 IndirCalls.push_back(&Inst);
575 }
576 if (Options.TraceCmp) {
577 if (ICmpInst *CMP = dyn_cast<ICmpInst>(&Inst))
578 if (IsInterestingCmp(CMP, DT, Options))
579 CmpTraceTargets.push_back(&Inst);
580 if (isa<SwitchInst>(&Inst))
581 SwitchTraceTargets.push_back(&Inst);
582 }
583 if (Options.TraceDiv)
584 if (BinaryOperator *BO = dyn_cast<BinaryOperator>(&Inst))
585 if (BO->getOpcode() == Instruction::SDiv ||
586 BO->getOpcode() == Instruction::UDiv)
587 DivTraceTargets.push_back(BO);
588 if (Options.TraceGep)
589 if (GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(&Inst))
590 GepTraceTargets.push_back(GEP);
591 if (Options.StackDepth)
592 if (isa<InvokeInst>(Inst) ||
593 (isa<CallInst>(Inst) && !isa<IntrinsicInst>(Inst)))
594 IsLeafFunc = false;
595 }
596 }
597
598 InjectCoverage(F, BlocksToInstrument, IsLeafFunc);
599 InjectCoverageForIndirectCalls(F, IndirCalls);
600 InjectTraceForCmp(F, CmpTraceTargets);
601 InjectTraceForSwitch(F, SwitchTraceTargets);
602 InjectTraceForDiv(F, DivTraceTargets);
603 InjectTraceForGep(F, GepTraceTargets);
604 return true;
605}

608 size_t NumElements, Function &F, Type *Ty, const char *Section) {
609 ArrayType *ArrayTy = ArrayType::get(Ty, NumElements);
610 auto Array = new GlobalVariable(
611 *CurModule, ArrayTy, false, GlobalVariable::PrivateLinkage,
612 Constant::getNullValue(ArrayTy), "__sancov_gen_");

630GlobalVariable *
631SanitizerCoverageModule::CreatePCArray(Function &F,
632 ArrayRef<BasicBlock *> AllBlocks) {
633 size_t N = AllBlocks.size();
634 assert(N);
635 SmallVector<Constant *, 32> PCs;
636 IRBuilder<> IRB(&*F.getEntryBlock().getFirstInsertionPt());
637 for (size_t i = 0; i < N; i++) {
638 if (&F.getEntryBlock() == AllBlocks[i]) {
639 PCs.push_back(cast<Constant>(IRB.CreatePtrToInt(&F, PcAddrTy)));
640 PCs.push_back(ConstantInt::get(PcAddrTy, 1));
641 } else {
642 PCs.push_back(cast<Constant>(IRB.CreatePtrToInt(
643 BlockAddress::get(AllBlocks[i]), PcAddrTy)));
644 PCs.push_back(ConstantInt::get(PcAddrTy, 0));
645 }
646 }
647 auto *PCArray = CreateFunctionLocalArrayInSection(N * 2, F, PcAddrTy,
648 SanCovPCsSectionName);
649 PCArray->setInitializer(
650 ConstantArray::get(ArrayType::get(PcAddrTy, N * 2), PCs));
651 PCArray->setConstant(true);
652
653 return PCArray;
654}

670bool SanitizerCoverageModule::InjectCoverage(Function &F,
671 ArrayRef<BasicBlock *> AllBlocks,
672 bool IsLeafFunc) {
673 if (AllBlocks.empty()) return false;
674 CreateFunctionLocalArrays(F, AllBlocks);
675 for (size_t i = 0, N = AllBlocks.size(); i < N; i++)
676 InjectCoverageAtBlock(F, *AllBlocks[i], i, IsLeafFunc);
677 return true;
678}
679
680// On every indirect call we call a run-time function
681// __sanitizer_cov_indir_call* with two parameters:
682// - callee address,
683// - global cache array that contains CacheSize pointers (zero-initialized).
684// The cache is used to speed up recording the caller-callee pairs.
685// The address of the caller is passed implicitly via caller PC.
686// CacheSize is encoded in the name of the run-time function.
687void SanitizerCoverageModule::InjectCoverageForIndirectCalls(
688 Function &F, ArrayRef<Instruction *> IndirCalls) {
689 if (IndirCalls.empty())
690 return;
691 assert(Options.TracePC || Options.TracePCGuard || Options.Inline8bitCounters);
692 for (auto I : IndirCalls) {
693 IRBuilder<> IRB(I);
694 CallSite CS(I);
695 Value *Callee = CS.getCalledValue();
696 if (isa<InlineAsm>(Callee))
697 continue;
698 IRB.CreateCall(SanCovTracePCIndir, IRB.CreatePointerCast(Callee, IntptrTy));
699 }
700}
701
702// For every switch statement we insert a call:
703// __sanitizer_cov_trace_switch(CondValue,
704// {NumCases, ValueSizeInBits, Case0Value, Case1Value, Case2Value, ... })
705
706void SanitizerCoverageModule::InjectTraceForSwitch(
707 Function &, ArrayRef<Instruction *> SwitchTraceTargets) {
708 for (auto I : SwitchTraceTargets) {
709 if (SwitchInst *SI = dyn_cast<SwitchInst>(I)) {
710 IRBuilder<> IRB(I);
711 SmallVector<Constant *, 16> Initializers;
712 Value *Cond = SI->getCondition();
713 if (Cond->getType()->getScalarSizeInBits() >
714 Int64Ty->getScalarSizeInBits())
715 continue;
716 Initializers.push_back(ConstantInt::get(Int64Ty, SI->getNumCases()));
717 Initializers.push_back(
718 ConstantInt::get(Int64Ty, Cond->getType()->getScalarSizeInBits()));
719 if (Cond->getType()->getScalarSizeInBits() <
720 Int64Ty->getScalarSizeInBits())
721 Cond = IRB.CreateIntCast(Cond, Int64Ty, false);
722 for (auto It : SI->cases()) {
723 Constant *C = It.getCaseValue();
724 if (C->getType()->getScalarSizeInBits() <
725 Int64Ty->getScalarSizeInBits())
726 C = ConstantExpr::getCast(CastInst::ZExt, It.getCaseValue(), Int64Ty);
727 Initializers.push_back(C);
728 }
729 llvm::sort(Initializers.begin() + 2, Initializers.end(),
730 [](const Constant *A, const Constant *B) {
731 return cast<ConstantInt>(A)->getLimitedValue() <
732 cast<ConstantInt>(B)->getLimitedValue();
733 });
734 ArrayType *ArrayOfInt64Ty = ArrayType::get(Int64Ty, Initializers.size());
735 GlobalVariable *GV = new GlobalVariable(
736 *CurModule, ArrayOfInt64Ty, false, GlobalVariable::InternalLinkage,
737 ConstantArray::get(ArrayOfInt64Ty, Initializers),
738 "__sancov_gen_cov_switch_values");
739 IRB.CreateCall(SanCovTraceSwitchFunction,
740 {Cond, IRB.CreatePointerCast(GV, GlobalsInt64PtrTy)});
741 }
742 }
743}
744
745void SanitizerCoverageModule::InjectTraceForDiv(
746 Function &, ArrayRef<BinaryOperator *> DivTraceTargets) {
747 for (auto BO : DivTraceTargets) {
748 IRBuilder<> IRB(BO);
749 Value *A1 = BO->getOperand(1);
750 if (isa<ConstantInt>(A1)) continue;
751 if (!A1->getType()->isIntegerTy())
752 continue;
753 uint64_t TypeSize = DL->getTypeStoreSizeInBits(A1->getType());
754 int CallbackIdx = TypeSize == 32 ? 0 :
755 TypeSize == 64 ? 1 : -1;
756 if (CallbackIdx < 0) continue;
757 auto Ty = Type::getIntNTy(*C, TypeSize);
758 IRB.CreateCall(SanCovTraceDivFunction[CallbackIdx],
759 {IRB.CreateIntCast(A1, Ty, true)});
760 }
761}
762
763void SanitizerCoverageModule::InjectTraceForGep(
764 Function &, ArrayRef<GetElementPtrInst *> GepTraceTargets) {
765 for (auto GEP : GepTraceTargets) {
766 IRBuilder<> IRB(GEP);
767 for (auto I = GEP->idx_begin(); I != GEP->idx_end(); ++I)
768 if (!isa<ConstantInt>(*I) && (*I)->getType()->isIntegerTy())
769 IRB.CreateCall(SanCovTraceGepFunction,
770 {IRB.CreateIntCast(*I, IntptrTy, true)});
771 }
772}
773
774void SanitizerCoverageModule::InjectTraceForCmp(
775 Function &, ArrayRef<Instruction *> CmpTraceTargets) {
776 for (auto I : CmpTraceTargets) {
777 if (ICmpInst *ICMP = dyn_cast<ICmpInst>(I)) {
778 IRBuilder<> IRB(ICMP);
779 Value *A0 = ICMP->getOperand(0);
780 Value *A1 = ICMP->getOperand(1);
781 if (!A0->getType()->isIntegerTy())
782 continue;
783 uint64_t TypeSize = DL->getTypeStoreSizeInBits(A0->getType());
784 int CallbackIdx = TypeSize == 8 ? 0 :
785 TypeSize == 16 ? 1 :
786 TypeSize == 32 ? 2 :
787 TypeSize == 64 ? 3 : -1;
788 if (CallbackIdx < 0) continue;
789 // __sanitizer_cov_trace_cmp((type_size << 32) | predicate, A0, A1);
790 auto CallbackFunc = SanCovTraceCmpFunction[CallbackIdx];
791 bool FirstIsConst = isa<ConstantInt>(A0);
792 bool SecondIsConst = isa<ConstantInt>(A1);
793 // If both are const, then we don't need such a comparison.
794 if (FirstIsConst && SecondIsConst) continue;
795 // If only one is const, then make it the first callback argument.
796 if (FirstIsConst || SecondIsConst) {
797 CallbackFunc = SanCovTraceConstCmpFunction[CallbackIdx];
798 if (SecondIsConst)
799 std::swap(A0, A1);
800 }
801
802 auto Ty = Type::getIntNTy(*C, TypeSize);
803 IRB.CreateCall(CallbackFunc, {IRB.CreateIntCast(A0, Ty, true),
804 IRB.CreateIntCast(A1, Ty, true)});
805 }
806 }
807}

809void SanitizerCoverageModule::InjectCoverageAtBlock(Function &F, BasicBlock &BB,
810 size_t Idx,
811 bool IsLeafFunc) {
812 BasicBlock::iterator IP = BB.getFirstInsertionPt();
813 bool IsEntryBB = &BB == &F.getEntryBlock();
814 DebugLoc EntryLoc;
815 if (IsEntryBB) {
816 if (auto SP = F.getSubprogram())
817 EntryLoc = DebugLoc::get(SP->getScopeLine(), 0, SP);
818 // Keep static allocas and llvm.localescape calls in the entry block. Even
819 // if we aren't splitting the block, it's nice for allocas to be before
820 // calls.
821 IP = PrepareToSplitEntryBlock(BB, IP);
822 } else {
823 EntryLoc = IP->getDebugLoc();
824 }
825
826 IRBuilder<> IRB(&*IP);
827 IRB.SetCurrentDebugLocation(EntryLoc);
828 if (Options.TracePC) {
829 IRB.CreateCall(SanCovTracePC); // gets the PC using GET_CALLER_PC.
830 IRB.CreateCall(EmptyAsm, {}); // Avoids callback merge.
831 }
832 if (Options.TracePCGuard) {
833#if 0 // WTF is this using inttoptr instead of a GEP?
834 auto GuardPtr = IRB.CreateIntToPtr(
835 IRB.CreateAdd(IRB.CreatePointerCast(FunctionGuardArray, IntptrTy),
836 ConstantInt::get(IntptrTy, Idx * 4)),
837 GlobalsInt32PtrTy);
838#else
839 auto GuardPtr = IRB.CreateGEP(
840 FunctionGuardArray->getValueType(), FunctionGuardArray,
841 {ConstantInt::get(Int32Ty, 0), ConstantInt::get(IntptrTy, Idx)});
842#endif
843 IRB.CreateCall(SanCovTracePCGuard, GuardPtr);
844 IRB.CreateCall(EmptyAsm, {}); // Avoids callback merge.
845 }
846 if (Options.Inline8bitCounters) {
847 auto CounterPtr = IRB.CreateGEP(
848 Function8bitCounterArray->getValueType(), Function8bitCounterArray,
849 {ConstantInt::get(IntptrTy, 0), ConstantInt::get(IntptrTy, Idx)});
850 auto Load = IRB.CreateLoad(Int8Ty, CounterPtr);
851 auto Inc = IRB.CreateAdd(Load, ConstantInt::get(Int8Ty, 1));
852 auto Store = IRB.CreateStore(Inc, CounterPtr);
853 SetNoSanitizeMetadata(Load);
854 SetNoSanitizeMetadata(Store);
855 }
856 if (Options.StackDepth && IsEntryBB && !IsLeafFunc) {
857 // Check stack depth. If it's the deepest so far, record it.
858 Function *GetFrameAddr = Intrinsic::getDeclaration(
859 F.getParent(), Intrinsic::frameaddress,
860 Int8Ty->getPointerTo(
861 F.getParent()->getDataLayout().getProgramAddressSpace()));
862 auto FrameAddrPtr =
863 IRB.CreateCall(GetFrameAddr, {Constant::getNullValue(Int32Ty)});
864 auto FrameAddrInt = IRB.CreatePtrToInt(FrameAddrPtr, IntptrTy);
865 auto LowestStack = IRB.CreateLoad(IntptrTy, SanCovLowestStack);
866 auto IsStackLower = IRB.CreateICmpULT(FrameAddrInt, LowestStack);
867 auto ThenTerm = SplitBlockAndInsertIfThen(IsStackLower, &*IP, false);
868 IRBuilder<> ThenIRB(ThenTerm);
869 auto Store = ThenIRB.CreateStore(FrameAddrInt, SanCovLowestStack);
870 SetNoSanitizeMetadata(LowestStack);
871 SetNoSanitizeMetadata(Store);
872 }
873}
