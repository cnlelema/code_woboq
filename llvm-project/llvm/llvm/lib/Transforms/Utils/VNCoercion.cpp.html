16/// Return true if coerceAvailableValueToLoadType will succeed.
17bool canCoerceMustAliasedValueToLoad(Value *StoredVal, Type *LoadTy,
18 const DataLayout &DL) {
19 Type *StoredTy = StoredVal->getType();
20 if (StoredTy == LoadTy)
21 return true;
22
23 // If the loaded or stored value is an first class array or struct, don't try
24 // to transform them. We need to be able to bitcast to integer.
25 if (LoadTy->isStructTy() || LoadTy->isArrayTy() || StoredTy->isStructTy() ||
26 StoredTy->isArrayTy())
27 return false;
28
29 uint64_t StoreSize = DL.getTypeSizeInBits(StoredTy);
30
31 // The store size must be byte-aligned to support future type casts.
32 if (llvm::alignTo(StoreSize, 8) != StoreSize)
33 return false;
34
35 // The store has to be at least as big as the load.
36 if (StoreSize < DL.getTypeSizeInBits(LoadTy))
37 return false;
38
39 // Don't coerce non-integral pointers to integers or vice versa.
40 if (DL.isNonIntegralPointerType(StoredVal->getType()->getScalarType()) !=
41 DL.isNonIntegralPointerType(LoadTy->getScalarType())) {
42 // As a special case, allow coercion of memset used to initialize
43 // an array w/null. Despite non-integral pointers not generally having a
44 // specific bit pattern, we do assume null is zero.
45 if (auto *CI = dyn_cast<Constant>(StoredVal))
46 return CI->isNullValue();
47 return false;
48 }
49
50 // We can't coerce a store of a fat pointer to a load of anything that isn't
51 // a fat pointer. We also can't coerce a store of a non fat pointer to
52 // anything that is a fat pointer.
53 if (DL.isFatPointer(StoredVal->getType()) != DL.isFatPointer(LoadTy))
54 return false;
55
56 return true;
57}

59template <class T, class HelperClass>
60static T *coerceAvailableValueToLoadTypeHelper(T *StoredVal, Type *LoadedTy,
61 HelperClass &Helper,
62 const DataLayout &DL) {
63 assert(canCoerceMustAliasedValueToLoad(StoredVal, LoadedTy, DL) &&
64 "precondition violation - materialization can't fail");
65 if (auto *C = dyn_cast<Constant>(StoredVal))
66 if (auto *FoldedStoredVal = ConstantFoldConstant(C, DL))
67 StoredVal = FoldedStoredVal;
68
69 // If this is already the right type, just return it.
70 Type *StoredValTy = StoredVal->getType();
71
72 uint64_t StoredValSize = DL.getTypeSizeInBits(StoredValTy);
73 uint64_t LoadedValSize = DL.getTypeSizeInBits(LoadedTy);
74
75 // If the store and reload are the same size, we can always reuse it.
76 if (StoredValSize == LoadedValSize) {
77 // Pointer to Pointer -> use bitcast.
78 if (StoredValTy->isPtrOrPtrVectorTy() && LoadedTy->isPtrOrPtrVectorTy()) {
79 StoredVal = Helper.CreateBitCast(StoredVal, LoadedTy);
80 } else {
81 // Convert source pointers to integers, which can be bitcast.
82 if (StoredValTy->isPtrOrPtrVectorTy()) {
83 StoredValTy = DL.getIntPtrType(StoredValTy);
84 StoredVal = Helper.CreatePtrToInt(StoredVal, StoredValTy);
85 }
86
87 Type *TypeToCastTo = LoadedTy;
88 if (TypeToCastTo->isPtrOrPtrVectorTy())
89 TypeToCastTo = DL.getIntPtrType(TypeToCastTo);
90
91 if (StoredValTy != TypeToCastTo)
92 StoredVal = Helper.CreateBitCast(StoredVal, TypeToCastTo);
93
94 // Cast to pointer if the load needs a pointer type.
95 if (LoadedTy->isPtrOrPtrVectorTy())
96 StoredVal = Helper.CreateIntToPtr(StoredVal, LoadedTy);
97 }
98
99 if (auto *C = dyn_cast<ConstantExpr>(StoredVal))
100 if (auto *FoldedStoredVal = ConstantFoldConstant(C, DL))
101 StoredVal = FoldedStoredVal;
102
103 return StoredVal;
104 }
105 // If the loaded value is smaller than the available value, then we can
106 // extract out a piece from it. If the available value is too small, then we
107 // can't do anything.
108 assert(StoredValSize >= LoadedValSize &&
109 "canCoerceMustAliasedValueToLoad fail");
110
111 // Convert source pointers to integers, which can be manipulated.
112 if (StoredValTy->isPtrOrPtrVectorTy()) {
113 if (DL.isFatPointer(StoredValTy->getScalarType()->getPointerAddressSpace()))
114 return nullptr;
115 StoredValTy = DL.getIntPtrType(StoredValTy);
116 StoredVal = Helper.CreatePtrToInt(StoredVal, StoredValTy);
117 }
118
119 // Convert vectors and fp to integer, which can be manipulated.
120 if (!StoredValTy->isIntegerTy()) {
121 StoredValTy = IntegerType::get(StoredValTy->getContext(), StoredValSize);
122 StoredVal = Helper.CreateBitCast(StoredVal, StoredValTy);
123 }
124
125 // If this is a big-endian system, we need to shift the value down to the low
126 // bits so that a truncate will work.
127 if (DL.isBigEndian()) {
128 uint64_t ShiftAmt = DL.getTypeStoreSizeInBits(StoredValTy) -
129 DL.getTypeStoreSizeInBits(LoadedTy);
130 StoredVal = Helper.CreateLShr(
131 StoredVal, ConstantInt::get(StoredVal->getType(), ShiftAmt));
132 }
133
134 // Truncate the integer to the right size now.
135 Type *NewIntTy = IntegerType::get(StoredValTy->getContext(), LoadedValSize);
136 StoredVal = Helper.CreateTruncOrBitCast(StoredVal, NewIntTy);
137
138 if (LoadedTy != NewIntTy) {
139 // If the result is a pointer, inttoptr.
140 if (LoadedTy->isPtrOrPtrVectorTy())
141 StoredVal = Helper.CreateIntToPtr(StoredVal, LoadedTy);
142 else
143 // Otherwise, bitcast.
144 StoredVal = Helper.CreateBitCast(StoredVal, LoadedTy);
145 }
146
147 if (auto *C = dyn_cast<Constant>(StoredVal))
148 if (auto *FoldedStoredVal = ConstantFoldConstant(C, DL))
149 StoredVal = FoldedStoredVal;
150
151 return StoredVal;
152}

154/// If we saw a store of a value to memory, and
155/// then a load from a must-aliased pointer of a different type, try to coerce
156/// the stored value. LoadedTy is the type of the load we want to replace.
157/// IRB is IRBuilder used to insert new instructions.
158///
159/// If we can't do it, return null.
160Value *coerceAvailableValueToLoadType(Value *StoredVal, Type *LoadedTy,
161 IRBuilder<> &IRB, const DataLayout &DL) {
162 return coerceAvailableValueToLoadTypeHelper(StoredVal, LoadedTy, IRB, DL);
163}

165/// This function is called when we have a memdep query of a load that ends up
166/// being a clobbering memory write (store, memset, memcpy, memmove). This
167/// means that the write *may* provide bits used by the load but we can't be
168/// sure because the pointers don't must-alias.
169///
170/// Check this case to see if there is anything more we can do before we give
171/// up. This returns -1 if we have to give up, or a byte number in the stored
172/// value of the piece that feeds the load.
173static int analyzeLoadFromClobberingWrite(Type *LoadTy, Value *LoadPtr,
174 Value *WritePtr,
175 uint64_t WriteSizeInBits,
176 const DataLayout &DL,
177 const TargetLibraryInfo *TLI) {
178 // If the loaded or stored value is a first class array or struct, don't try
179 // to transform them. We need to be able to bitcast to integer.
180 if (LoadTy->isStructTy() || LoadTy->isArrayTy())
181 return -1;
182
183 int64_t StoreOffset = 0, LoadOffset = 0;
184 Value *StoreBase =
185 GetPointerBaseWithConstantOffset(WritePtr, StoreOffset, DL);
186 Value *LoadBase = GetPointerBaseWithConstantOffset(LoadPtr, LoadOffset, DL);
187 if (StoreBase != LoadBase)
188 return -1;
189
190 // If the load and store are to the exact same address, they should have been
191 // a must alias. AA must have gotten confused.
192 // FIXME: Study to see if/when this happens. One case is forwarding a memset
193 // to a load from the base of the memset.
194
195 // If the load and store don't overlap at all, the store doesn't provide
196 // anything to the load. In this case, they really don't alias at all, AA
197 // must have gotten confused.
198 uint64_t LoadSize = DL.getTypeSizeInBits(LoadTy);
199
200 if ((WriteSizeInBits & 7) | (LoadSize & 7))
201 return -1;
202 uint64_t StoreSize = WriteSizeInBits / 8; // Convert to bytes.
203 LoadSize /= 8;
204
205 bool isAAFailure = false;
206 if (StoreOffset < LoadOffset)
207 isAAFailure = StoreOffset + int64_t(StoreSize) <= LoadOffset;
208 else
209 isAAFailure = LoadOffset + int64_t(LoadSize) <= StoreOffset;
210
211 if (isAAFailure)
212 return -1;
213
214 // If the Load isn't completely contained within the stored bits, we don't
215 // have all the bits to feed it. We could do something crazy in the future
216 // (issue a smaller load then merge the bits in) but this seems unlikely to be
217 // valuable.
218 if (StoreOffset > LoadOffset ||
219 StoreOffset + StoreSize < LoadOffset + LoadSize)
220 return -1;
221
222 // If we're dealing with a bounds-checked pointer, then we have stricter
223 // overlap requirements.
224 if (DL.isFatPointer(LoadPtr->getType()->getPointerAddressSpace())) {
225 uint64_t Size = 0;
226 bool KnownSize = getObjectSize(StoreBase, Size, DL, TLI);
227 // If we don't know the size of the underlying object then we can't
228 // assume that we can look through this pointer.
229 unsigned SrcValSize = DL.getTypeStoreSize(LoadTy);
230 if (!KnownSize)
231 return -1;
232 if ((SrcValSize+LoadSize > SrcValSize) && (NextPowerOf2(SrcValSize)+LoadOffset > Size))
233 return -1;
234 }
235
236 // Okay, we can do this transformation. Return the number of bytes into the
237 // store that the load is.
238 return LoadOffset - StoreOffset;
239}

241/// This function is called when we have a
242/// memdep query of a load that ends up being a clobbering store.
243int analyzeLoadFromClobberingStore(Type *LoadTy, Value *LoadPtr,
244 StoreInst *DepSI, const DataLayout &DL,
245 const TargetLibraryInfo *TLI) {
246 auto *StoredVal = DepSI->getValueOperand();
247
248 // Cannot handle reading from store of first-class aggregate yet.
249 if (StoredVal->getType()->isStructTy() ||
250 StoredVal->getType()->isArrayTy())
251 return -1;
252
253 // Don't coerce non-integral pointers to integers or vice versa.
254 if (DL.isNonIntegralPointerType(StoredVal->getType()->getScalarType()) !=
255 DL.isNonIntegralPointerType(LoadTy->getScalarType())) {
256 // Allow casts of zero values to null as a special case
257 auto *CI = dyn_cast<Constant>(StoredVal);
258 if (!CI || !CI->isNullValue())
259 return -1;
260 }
261
262 Value *StorePtr = DepSI->getPointerOperand();
263 uint64_t StoreSize =
264 DL.getTypeSizeInBits(DepSI->getValueOperand()->getType());
265 return analyzeLoadFromClobberingWrite(LoadTy, LoadPtr, StorePtr, StoreSize,
266 DL, TLI);
267}

269/// This function is called when we have a
270/// memdep query of a load that ends up being clobbered by another load. See if
271/// the other load can feed into the second load.
272int analyzeLoadFromClobberingLoad(Type *LoadTy, Value *LoadPtr, LoadInst *DepLI,
273 const DataLayout &DL,
274 const TargetLibraryInfo *TLI) {
275 // Cannot handle reading from store of first-class aggregate yet.
276 if (DepLI->getType()->isStructTy() || DepLI->getType()->isArrayTy())
277 return -1;
278
279 // Don't coerce non-integral pointers to integers or vice versa.
280 if (DL.isNonIntegralPointerType(DepLI->getType()->getScalarType()) !=
281 DL.isNonIntegralPointerType(LoadTy->getScalarType()))
282 return -1;
283
284 Value *DepPtr = DepLI->getPointerOperand();
285 uint64_t DepSize = DL.getTypeSizeInBits(DepLI->getType());
286 int R = analyzeLoadFromClobberingWrite(LoadTy, LoadPtr, DepPtr, DepSize, DL,
287 TLI);
288 if (R != -1)
289 return R;
290
291 // If we have a load/load clobber an DepLI can be widened to cover this load,
292 // then we should widen it!
293 int64_t LoadOffs = 0;
294 const Value *LoadBase =
295 GetPointerBaseWithConstantOffset(LoadPtr, LoadOffs, DL);
296 unsigned LoadSize = DL.getTypeStoreSize(LoadTy);
297
298 unsigned Size = MemoryDependenceResults::getLoadLoadClobberFullWidthSize(
299 LoadBase, LoadOffs, LoadSize, DepLI);
300 if (Size == 0)
301 return -1;
302
303 // Check non-obvious conditions enforced by MDA which we rely on for being
304 // able to materialize this potentially available value
305 assert(DepLI->isSimple() && "Cannot widen volatile/atomic load!");
306 assert(DepLI->getType()->isIntegerTy() && "Can't widen non-integer load");
307
308 return analyzeLoadFromClobberingWrite(LoadTy, LoadPtr, DepPtr, Size * 8, DL, TLI);
309}

313int analyzeLoadFromClobberingMemInst(Type *LoadTy, Value *LoadPtr,
314 MemIntrinsic *MI, const DataLayout &DL,
315 const TargetLibraryInfo *TLI) {
316 // If the mem operation is a non-constant size, we can't handle it.
317 ConstantInt *SizeCst = dyn_cast<ConstantInt>(MI->getLength());
318 if (!SizeCst)
319 return -1;
320 // If this is a pointer type that's larger than the largest integer that we
321 // support, then ignore it.
322 if (LoadTy->isPointerTy() &&
323 DL.getTypeSizeInBits(LoadTy) > DL.getLargestLegalIntTypeSizeInBits())
324 return -1;
325 uint64_t MemSizeInBits = SizeCst->getZExtValue() * 8;
326
327 // If this is memset, we just need to see if the offset is valid in the size
328 // of the memset..
329 if (MI->getIntrinsicID() == Intrinsic::memset) {
330 if (DL.isNonIntegralPointerType(LoadTy->getScalarType())) {
331 auto *CI = dyn_cast<ConstantInt>(cast<MemSetInst>(MI)->getValue());
332 if (!CI || !CI->isZero())
333 return -1;
334 }
335 return analyzeLoadFromClobberingWrite(LoadTy, LoadPtr, MI->getDest(),
336 MemSizeInBits, DL, TLI);
337 }
338
339 // If we have a memcpy/memmove, the only case we can handle is if this is a
340 // copy from constant memory. In that case, we can read directly from the
341 // constant memory.
342 MemTransferInst *MTI = cast<MemTransferInst>(MI);
343
344 Constant *Src = dyn_cast<Constant>(MTI->getSource());
345 if (!Src)
346 return -1;
347
348 GlobalVariable *GV = dyn_cast<GlobalVariable>(GetUnderlyingObject(Src, DL));
349 if (!GV || !GV->isConstant() || !GV->hasDefinitiveInitializer())
350 return -1;
351
352 // See if the access is within the bounds of the transfer.
353 int Offset = analyzeLoadFromClobberingWrite(LoadTy, LoadPtr, MI->getDest(),
354 MemSizeInBits, DL, TLI);
355 if (Offset == -1)
356 return Offset;
357
358 // Don't coerce non-integral pointers to integers or vice versa, and the
359 // memtransfer is implicitly a raw byte code
360 if (DL.isNonIntegralPointerType(LoadTy->getScalarType()))
361 // TODO: Can allow nullptrs from constant zeros
362 return -1;
363
364 unsigned AS = Src->getType()->getPointerAddressSpace();
365 // Otherwise, see if we can constant fold a load from the constant with the
366 // offset applied as appropriate.
367 Src =
368 ConstantExpr::getBitCast(Src, Type::getInt8PtrTy(Src->getContext(), AS));
369 Constant *OffsetCst =
370 ConstantInt::get(Type::getInt64Ty(Src->getContext()), (unsigned)Offset);
371 Src = ConstantExpr::getGetElementPtr(Type::getInt8Ty(Src->getContext()), Src,
372 OffsetCst);
373 Src = ConstantExpr::getBitCast(Src, PointerType::get(LoadTy, AS));
374 if (ConstantFoldLoadFromConstPtr(Src, LoadTy, DL))
375 return Offset;
376 return -1;
377}

379template <class T, class HelperClass>
380static T *getStoreValueForLoadHelper(T *SrcVal, unsigned Offset, Type *LoadTy,
381 HelperClass &Helper,
382 const DataLayout &DL) {
383 LLVMContext &Ctx = SrcVal->getType()->getContext();
384
385 // If two pointers are in the same address space, they have the same size,
386 // so we don't need to do any truncation, etc. This avoids introducing
387 // ptrtoint instructions for pointers that may be non-integral.
388 if (SrcVal->getType()->isPointerTy() && LoadTy->isPointerTy() &&
389 cast<PointerType>(SrcVal->getType())->getAddressSpace() ==
390 cast<PointerType>(LoadTy)->getAddressSpace()) {
391 return SrcVal;
392 }
393
394 uint64_t StoreSize = (DL.getTypeSizeInBits(SrcVal->getType()) + 7) / 8;
395 uint64_t LoadSize = (DL.getTypeSizeInBits(LoadTy) + 7) / 8;
396
397 if ((LoadSize == StoreSize) && Offset == 0 &&
398 SrcVal->getType()->isPointerTy() && LoadTy->isPointerTy())
399 return coerceAvailableValueToLoadTypeHelper(
400 Helper.CreateBitCast(SrcVal, LoadTy), LoadTy, Helper, DL);
401
402 // Compute which bits of the stored value are being used by the load. Convert
403 // to an integer type to start with.
404 if (SrcVal->getType()->isPtrOrPtrVectorTy())
405 SrcVal = Helper.CreatePtrToInt(SrcVal, DL.getIntPtrType(SrcVal->getType()));
406 if (!SrcVal->getType()->isIntegerTy())
407 SrcVal = Helper.CreateBitCast(SrcVal, IntegerType::get(Ctx, StoreSize * 8));
408
409 // Shift the bits to the least significant depending on endianness.
410 unsigned ShiftAmt;
411 if (DL.isLittleEndian())
412 ShiftAmt = Offset * 8;
413 else
414 ShiftAmt = (StoreSize - LoadSize - Offset) * 8;
415 if (ShiftAmt)
416 SrcVal = Helper.CreateLShr(SrcVal,
417 ConstantInt::get(SrcVal->getType(), ShiftAmt));
418
419 if (LoadSize != StoreSize)
420 SrcVal = Helper.CreateTruncOrBitCast(SrcVal,
421 IntegerType::get(Ctx, LoadSize * 8));
422 return SrcVal;
423}

429Value *getStoreValueForLoad(Value *SrcVal, unsigned Offset, Type *LoadTy,

444/// This function is called when we have a memdep query of a load that ends up
445/// being a clobbering load. This means that the load *may* provide bits used
446/// by the load but we can't be sure because the pointers don't must-alias.
447/// Check this case to see if there is anything more we can do before we give
448/// up.
449Value *getLoadValueForLoad(LoadInst *SrcVal, unsigned Offset, Type *LoadTy,
450 Instruction *InsertPt, const DataLayout &DL,
451 GVN &gvn) {
452 // If Offset+LoadTy exceeds the size of SrcVal, then we must be wanting to
453 // widen SrcVal out to a larger load.
454 unsigned SrcValStoreSize = DL.getTypeStoreSize(SrcVal->getType());
455 unsigned LoadSize = DL.getTypeStoreSize(LoadTy);
456 if (Offset + LoadSize > SrcValStoreSize) {
457 assert(SrcVal->isSimple() && "Cannot widen volatile/atomic load!");
458 assert(SrcVal->getType()->isIntegerTy() && "Can't widen non-integer load");
459 // If we have a load/load clobber an DepLI can be widened to cover this
460 // load, then we should widen it to the next power of 2 size big enough!
461 unsigned NewLoadSize = Offset + LoadSize;
462 if (!isPowerOf2_32(NewLoadSize))
463 NewLoadSize = NextPowerOf2(NewLoadSize);
464
465 Value *PtrVal = SrcVal->getPointerOperand();
466 // Insert the new load after the old load. This ensures that subsequent
467 // memdep queries will find the new load. We can't easily remove the old
468 // load completely because it is already in the value numbering table.
469 IRBuilder<> Builder(SrcVal->getParent(), ++BasicBlock::iterator(SrcVal));
470 Type *DestTy = IntegerType::get(LoadTy->getContext(), NewLoadSize * 8);
471 Type *DestPTy =
472 PointerType::get(DestTy, PtrVal->getType()->getPointerAddressSpace());
473 Builder.SetCurrentDebugLocation(SrcVal->getDebugLoc());
474 PtrVal = Builder.CreateBitCast(PtrVal, DestPTy);
475 LoadInst *NewLoad = Builder.CreateLoad(DestTy, PtrVal);
476 NewLoad->takeName(SrcVal);
477 NewLoad->setAlignment(SrcVal->getAlignment());
478
479 LLVM_DEBUG(dbgs() << "GVN WIDENED LOAD: " << *SrcVal << "\n");
480 LLVM_DEBUG(dbgs() << "TO: " << *NewLoad << "\n");
481
482 // Replace uses of the original load with the wider load. On a big endian
483 // system, we need to shift down to get the relevant bits.
484 Value *RV = NewLoad;
485 if (DL.isBigEndian())
486 RV = Builder.CreateLShr(RV, (NewLoadSize - SrcValStoreSize) * 8);
487 RV = Builder.CreateTrunc(RV, SrcVal->getType());
488 SrcVal->replaceAllUsesWith(RV);
489
490 // We would like to use gvn.markInstructionForDeletion here, but we can't
491 // because the load is already memoized into the leader map table that GVN
492 // tracks. It is potentially possible to remove the load from the table,
493 // but then there all of the operations based on it would need to be
494 // rehashed. Just leave the dead load around.
495 gvn.getMemDep().removeInstruction(SrcVal);
496 SrcVal = NewLoad;
497 }
498
499 return getStoreValueForLoad(SrcVal, Offset, LoadTy, InsertPt, DL);
500}

511template <class T, class HelperClass>
512T *getMemInstValueForLoadHelper(MemIntrinsic *SrcInst, unsigned Offset,
513 Type *LoadTy, HelperClass &Helper,
514 const DataLayout &DL) {
515 LLVMContext &Ctx = LoadTy->getContext();
516 uint64_t LoadSize = DL.getTypeSizeInBits(LoadTy) / 8;
517
518 // We know that this method is only called when the mem transfer fully
519 // provides the bits for the load.
520 if (MemSetInst *MSI = dyn_cast<MemSetInst>(SrcInst)) {
521 // memset(P, 'x', 1234) -> splat('x'), even if x is a variable, and
522 // independently of what the offset is.
523 T *Val = cast<T>(MSI->getValue());
524 if (LoadSize != 1)
525 Val =
526 Helper.CreateZExtOrBitCast(Val, IntegerType::get(Ctx, LoadSize * 8));
527 T *OneElt = Val;
528
529 // Splat the value out to the right number of bits.
530 for (unsigned NumBytesSet = 1; NumBytesSet != LoadSize;) {
531 // If we can double the number of bytes set, do it.
532 if (NumBytesSet * 2 <= LoadSize) {
533 T *ShVal = Helper.CreateShl(
534 Val, ConstantInt::get(Val->getType(), NumBytesSet * 8));
535 Val = Helper.CreateOr(Val, ShVal);
536 NumBytesSet <<= 1;
537 continue;
538 }
539
540 // Otherwise insert one byte at a time.
541 T *ShVal = Helper.CreateShl(Val, ConstantInt::get(Val->getType(), 1 * 8));
542 Val = Helper.CreateOr(OneElt, ShVal);
543 ++NumBytesSet;
544 }
545
546 return coerceAvailableValueToLoadTypeHelper(Val, LoadTy, Helper, DL);
547 }
548
549 // Otherwise, this is a memcpy/memmove from a constant global.
550 MemTransferInst *MTI = cast<MemTransferInst>(SrcInst);
551 Constant *Src = cast<Constant>(MTI->getSource());
552 unsigned AS = Src->getType()->getPointerAddressSpace();
553
554 // Otherwise, see if we can constant fold a load from the constant with the
555 // offset applied as appropriate.
556 Src =
557 ConstantExpr::getBitCast(Src, Type::getInt8PtrTy(Src->getContext(), AS));
558 Constant *OffsetCst =
559 ConstantInt::get(Type::getInt64Ty(Src->getContext()), (unsigned)Offset);
560 Src = ConstantExpr::getGetElementPtr(Type::getInt8Ty(Src->getContext()), Src,
561 OffsetCst);
562 Src = ConstantExpr::getBitCast(Src, PointerType::get(LoadTy, AS));
563 return ConstantFoldLoadFromConstPtr(Src, LoadTy, DL);
564}

576Constant *getConstantMemInstValueForLoad(MemIntrinsic *SrcInst, unsigned Offset,
577 Type *LoadTy, const DataLayout &DL) {
578 // The only case analyzeLoadFromClobberingMemInst cannot be converted to a
579 // constant is when it's a memset of a non-constant.
580 if (auto *MSI = dyn_cast<MemSetInst>(SrcInst))
581 if (!isa<Constant>(MSI->getValue()))
582 return nullptr;
583 ConstantFolder F;
584 return getMemInstValueForLoadHelper<Constant, ConstantFolder>(SrcInst, Offset,
585 LoadTy, F, DL);
586}
