23static void appendToGlobalArray(const char *Array, Module &M, Function *F,
24 int Priority, Constant *Data) {
25 IRBuilder<> IRB(M.getContext());
26 FunctionType *FnTy = FunctionType::get(IRB.getVoidTy(), false);
27
28 unsigned CtorPtrAS = M.getDataLayout().getProgramAddressSpace();
29 // And the pointers to the ctors are always AS0 for CHERI. We could
30 // theoretically make them AS200 but right now RTLD/csu expects that
31 // .init_array contains virtual addresses and not capabilities
32 if (llvm::Triple(M.getTargetTriple()).isMIPS())
33 CtorPtrAS = 0;
34 llvm::Type *CtorPFTy = llvm::PointerType::get(FnTy, CtorPtrAS);
35 llvm::Type *ArgTy =
36 IRB.getInt8PtrTy(M.getDataLayout().getGlobalsAddressSpace());
37
38 // Get the current set of static global constructors and add the new ctor
39 // to the list.
40 SmallVector<Constant *, 16> CurrentCtors;
41 StructType *EltTy = StructType::get(IRB.getInt32Ty(), CtorPFTy, ArgTy);
42 if (GlobalVariable *GVCtor = M.getNamedGlobal(Array)) {
43 if (Constant *Init = GVCtor->getInitializer()) {
44 unsigned n = Init->getNumOperands();
45 CurrentCtors.reserve(n + 1);
46 for (unsigned i = 0; i != n; ++i)
47 CurrentCtors.push_back(cast<Constant>(Init->getOperand(i)));
48 }
49 GVCtor->eraseFromParent();
50 }
51
52 // Build a 3 field global_ctor entry. We don't take a comdat key.
53 Constant *CSVals[3];
54 CSVals[0] = IRB.getInt32(Priority);
55 CSVals[1] = ConstantExpr::getPointerBitCastOrAddrSpaceCast(F, CtorPFTy);
56 CSVals[2] = Data ? ConstantExpr::getPointerCast(Data, ArgTy)
57 : Constant::getNullValue(ArgTy);
58 Constant *RuntimeCtorInit =
59 ConstantStruct::get(EltTy, makeArrayRef(CSVals, EltTy->getNumElements()));
60
61 CurrentCtors.push_back(RuntimeCtorInit);
62
63 // Create a new initializer.
64 ArrayType *AT = ArrayType::get(EltTy, CurrentCtors.size());
65 Constant *NewInit = ConstantArray::get(AT, CurrentCtors);
66
67 // Create the new global variable and replace all uses of
68 // the old global variable with the new one.
69 (void)new GlobalVariable(M, NewInit->getType(), false,
70 GlobalValue::AppendingLinkage, NewInit, Array);
71}

81static void appendToUsedList(Module &M, StringRef Name, ArrayRef<GlobalValue *> Values) {
82 GlobalVariable *GV = M.getGlobalVariable(Name);
83 SmallPtrSet<Constant *, 16> InitAsSet;
84 SmallVector<Constant *, 16> Init;
85 if (GV) {
86 ConstantArray *CA = dyn_cast<ConstantArray>(GV->getInitializer());
87 for (auto &Op : CA->operands()) {
88 Constant *C = cast_or_null<Constant>(Op);
89 if (InitAsSet.insert(C).second)
90 Init.push_back(C);
91 }
92 GV->eraseFromParent();
93 }
94
95 Type *Int8PtrTy = llvm::Type::getInt8PtrTy(M.getContext(), 0);
96 for (auto *V : Values) {
97 Constant *C = ConstantExpr::getPointerBitCastOrAddrSpaceCast(V, Int8PtrTy);
98 if (InitAsSet.insert(C).second)
99 Init.push_back(C);
100 }
101
102 if (Init.empty())
103 return;
104
105 ArrayType *ATy = ArrayType::get(Int8PtrTy, Init.size());
106 GV = new llvm::GlobalVariable(M, ATy, false, GlobalValue::AppendingLinkage,
107 ConstantArray::get(ATy, Init), Name);
108 GV->setSection("llvm.metadata");
109}

119FunctionCallee
120llvm::declareSanitizerInitFunction(Module &M, StringRef InitName,
121 ArrayRef<Type *> InitArgTypes) {
122 assert(!InitName.empty() && "Expected init function name");
123 return M.getOrInsertFunction(
124 InitName,
125 FunctionType::get(Type::getVoidTy(M.getContext()), InitArgTypes, false),
126 AttributeList());
127}

129std::pair<Function *, FunctionCallee> llvm::createSanitizerCtorAndInitFunctions(
130 Module &M, StringRef CtorName, StringRef InitName,
131 ArrayRef<Type *> InitArgTypes, ArrayRef<Value *> InitArgs,
132 StringRef VersionCheckName) {
133 assert(!InitName.empty() && "Expected init function name");
134 assert(InitArgs.size() == InitArgTypes.size() &&
135 "Sanitizer's init function expects different number of arguments");
136 FunctionCallee InitFunction =
137 declareSanitizerInitFunction(M, InitName, InitArgTypes);
138 Function *Ctor = Function::Create(
139 FunctionType::get(Type::getVoidTy(M.getContext()), false),
140 GlobalValue::InternalLinkage, CtorName, &M);
141 BasicBlock *CtorBB = BasicBlock::Create(M.getContext(), "", Ctor);
142 IRBuilder<> IRB(ReturnInst::Create(M.getContext(), CtorBB));
143 IRB.CreateCall(InitFunction, InitArgs);
144 if (!VersionCheckName.empty()) {
145 FunctionCallee VersionCheckFunction = M.getOrInsertFunction(
146 VersionCheckName, FunctionType::get(IRB.getVoidTy(), {}, false),
147 AttributeList());
148 IRB.CreateCall(VersionCheckFunction, {});
149 }
150 return std::make_pair(Ctor, InitFunction);
151}

153std::pair<Function *, FunctionCallee>
154llvm::getOrCreateSanitizerCtorAndInitFunctions(
155 Module &M, StringRef CtorName, StringRef InitName,
156 ArrayRef<Type *> InitArgTypes, ArrayRef<Value *> InitArgs,
157 function_ref<void(Function *, FunctionCallee)> FunctionsCreatedCallback,
158 StringRef VersionCheckName) {
159 assert(!CtorName.empty() && "Expected ctor function name");
160
161 if (Function *Ctor = M.getFunction(CtorName))
162 // FIXME: Sink this logic into the module, similar to the handling of
163 // globals. This will make moving to a concurrent model much easier.
164 if (Ctor->arg_size() == 0 ||
165 Ctor->getReturnType() == Type::getVoidTy(M.getContext()))
166 return {Ctor, declareSanitizerInitFunction(M, InitName, InitArgTypes)};
167
168 Function *Ctor;
169 FunctionCallee InitFunction;
170 std::tie(Ctor, InitFunction) = llvm::createSanitizerCtorAndInitFunctions(
171 M, CtorName, InitName, InitArgTypes, InitArgs, VersionCheckName);
172 FunctionsCreatedCallback(Ctor, InitFunction);
173 return std::make_pair(Ctor, InitFunction);
174}

176Function *llvm::getOrCreateInitFunction(Module &M, StringRef Name) {
177 assert(!Name.empty() && "Expected init function name");
178 if (Function *F = M.getFunction(Name)) {
179 if (F->arg_size() != 0 ||
180 F->getReturnType() != Type::getVoidTy(M.getContext())) {
181 std::string Err;
182 raw_string_ostream Stream(Err);
183 Stream << "Sanitizer interface function defined with wrong type: " << *F;
184 report_fatal_error(Err);
185 }
186 return F;
187 }
188 Function *F =
189 cast<Function>(M.getOrInsertFunction(Name, AttributeList(),
190 Type::getVoidTy(M.getContext()))
191 .getCallee());
192
193 appendToGlobalCtors(M, F, 0);
194
195 return F;
196}

198void llvm::filterDeadComdatFunctions(
199 Module &M, SmallVectorImpl<Function *> &DeadComdatFunctions) {
200 // Build a map from the comdat to the number of entries in that comdat we
201 // think are dead. If this fully covers the comdat group, then the entire
202 // group is dead. If we find another entry in the comdat group though, we'll
203 // have to preserve the whole group.
204 SmallDenseMap<Comdat *, int, 16> ComdatEntriesCovered;
205 for (Function *F : DeadComdatFunctions) {
206 Comdat *C = F->getComdat();
207 assert(C && "Expected all input GVs to be in a comdat!");
208 ComdatEntriesCovered[C] += 1;
209 }
210
211 auto CheckComdat = [&](Comdat &C) {
212 auto CI = ComdatEntriesCovered.find(&C);
213 if (CI == ComdatEntriesCovered.end())
214 return;
215
216 // If this could have been covered by a dead entry, just subtract one to
217 // account for it.
218 if (CI->second > 0) {
219 CI->second -= 1;
220 return;
221 }
222
223 // If we've already accounted for all the entries that were dead, the
224 // entire comdat is alive so remove it from the map.
225 ComdatEntriesCovered.erase(CI);
226 };
227
228 auto CheckAllComdats = [&] {
229 for (Function &F : M.functions())
230 if (Comdat *C = F.getComdat()) {
231 CheckComdat(*C);
232 if (ComdatEntriesCovered.empty())
233 return;
234 }
235 for (GlobalVariable &GV : M.globals())
236 if (Comdat *C = GV.getComdat()) {
237 CheckComdat(*C);
238 if (ComdatEntriesCovered.empty())
239 return;
240 }
241 for (GlobalAlias &GA : M.aliases())
242 if (Comdat *C = GA.getComdat()) {
243 CheckComdat(*C);
244 if (ComdatEntriesCovered.empty())
245 return;
246 }
247 };
248 CheckAllComdats();
249
250 if (ComdatEntriesCovered.empty()) {
251 DeadComdatFunctions.clear();
252 return;
253 }
254
255 // Remove the entries that were not covering.
256 erase_if(DeadComdatFunctions, [&](GlobalValue *GV) {
257 return ComdatEntriesCovered.find(GV->getComdat()) ==
258 ComdatEntriesCovered.end();
259 });
260}

262std::string llvm::getUniqueModuleId(Module *M) {
263 MD5 Md5;
264 bool ExportsSymbols = false;
265 auto AddGlobal = [&](GlobalValue &GV) {
266 if (GV.isDeclaration() || GV.getName().startswith("llvm.") ||
267 !GV.hasExternalLinkage() || GV.hasComdat())
268 return;
269 ExportsSymbols = true;
270 Md5.update(GV.getName());
271 Md5.update(ArrayRef<uint8_t>{0});
272 };
273
274 for (auto &F : *M)
275 AddGlobal(F);
276 for (auto &GV : M->globals())
277 AddGlobal(GV);
278 for (auto &GA : M->aliases())
279 AddGlobal(GA);
280 for (auto &IF : M->ifuncs())
281 AddGlobal(IF);
282
283 if (!ExportsSymbols)
284 return "";
285
286 MD5::MD5Result R;
287 Md5.final(R);
288
289 SmallString<32> Str;
290 MD5::stringifyResult(R, Str);
291 return ("$" + Str).str();
292}
