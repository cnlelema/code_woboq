22std::string llvm::cheri::inferSourceLocation(const DebugLoc &DL,
23 StringRef FunctionName) {
24 std::string Result;
25 raw_string_ostream OS(Result);
26 if (DL) {
27 DL.print(OS);
28 OS.flush();
29 }
30 if (Result.empty() && !FunctionName.empty()) {
31 OS << "<somewhere in " << FunctionName << ">";
32 OS.flush();
33 }
34 return Result;
35}

37std::string llvm::cheri::inferSourceLocation(Instruction *AI) {
38 assert(AI);
39 std::string Result = inferSourceLocation(AI->getDebugLoc(), StringRef());
40 if (Result.empty()) {
41 // some instructions such as alloca instruction don't have a debug loc
42 // attached directly so we need to look for calls to llvm.dbg.declare()
43 SmallVector<DbgVariableIntrinsic *, 2> DbgVars;
44 findDbgUsers(DbgVars, AI);
45 for (auto &DbgV : DbgVars) {
46 Result = inferSourceLocation(DbgV->getDebugLoc(), StringRef());
47 if (!Result.empty())
48 break;
49 }
50 }
51 // No debug instructions found -> just fall back to function name
52 if (Result.empty()) {
53 Result = ("<somewhere in " + AI->getFunction()->getName() + ">").str();
54 }
55 return Result;
56}

58std::string llvm::cheri::inferLocalVariableName(AllocaInst *AI) {
59 assert(AI);
60 // try to find a
61 // some instructions such as alloca instruction don't have a debug loc
62 // attached directly so we need to look for calls to llvm.dbg.declare()
63 SmallVector<DbgVariableIntrinsic *, 2> DbgVars;
64 findDbgUsers(DbgVars, AI);
65 for (auto &DbgV : DbgVars) {
66 if (DbgV->isAddressOfVariable()) {
67 return ("local variable " + DbgV->getVariable()->getName()).str();
68 }
69 }
70 // No debug instructions found -> just fall back to alloca name
71 if (AI->hasName())
72 return ("AllocaInst " + AI->getName()).str();
73 std::string Result;
74 llvm::raw_string_ostream OS(Result);
75 OS << "anonymous AllocaInst of type ";
76 AI->getType()->print(OS);
77 OS.flush();
78 return Result;
79}
