21/// This pass frees the MachineFunction object associated with a Function.
22class LogCheriSetBounds : public FunctionPass {
23public:
24 static char ID;
25
26 LogCheriSetBounds() : FunctionPass(ID) {
27 assert(cheri::ShouldCollectCSetBoundsStats &&
28 "Pass should only be created when we are logging bouds");
29 }
30
31 void getAnalysisUsage(AnalysisUsage &AU) const override {
32 AU.addRequired<AssumptionCacheTracker>();
33 AU.addRequired<DominatorTreeWrapperPass>();
34 }
35
36 bool runOnFunction(Function &F) override {
37 // errs() << "Logging bounds for " << F.getName() << "\n";
38 DominatorTree *DT = &getAnalysis<DominatorTreeWrapperPass>().getDomTree();
39 AssumptionCache *AC =
40 &getAnalysis<AssumptionCacheTracker>().getAssumptionCache(F);
41 // For MIPS we can guess the size here by multiplying by 4:
42 const DataLayout &DL = F.getParent()->getDataLayout();
43 for (auto &BB : F) {
44 for (Instruction &I : BB) {
45 // TODO: CallBase
46 Function *CalledFunc = nullptr;
47 AttributeList AL;
48 // TODO: use CallBase when we merge upstream
49 if (const CallInst *CI = dyn_cast<CallInst>(&I)) {
50 CalledFunc = CI->getCalledFunction();
51 AL = CI->getAttributes();
52 } else if (const InvokeInst *II = dyn_cast<InvokeInst>(&I)) {
53 CalledFunc = II->getCalledFunction();
54 AL = II->getAttributes();
55 }
56 const auto LogAllocSize =
57 [&](std::pair<unsigned, Optional<unsigned>> AllocSize) {
58 // TODO: should we also log this for non-capabilities?
59 if (!DL.isFatPointer(I.getType()))
60 return;
61 // Without the assumption cache we don't get any benefit from
62 // assume_aligned attributes and getKnownAlignment will return 1
63 uint64_t Alignment = getKnownAlignment(&I, DL, &I, AC, DT);
64 Optional<uint64_t> KnownSize;
65 Optional<uint64_t> SizeMultipleOf;
66
67 Optional<uint64_t> FirstConstant;
68 Optional<uint64_t> SecondConstant;
69 if (I.getNumOperands() > AllocSize.first)
70 FirstConstant =
71 cheri::inferConstantValue(I.getOperand(AllocSize.first));
72 if (AllocSize.second && I.getNumOperands() > *AllocSize.second)
73 SecondConstant =
74 cheri::inferConstantValue(I.getOperand(*AllocSize.second));
75 if (FirstConstant) {
76 if (!AllocSize.second) {
77 // If we used the one-argument version of allocsize this is
78 // the total known size
79 KnownSize = *FirstConstant;
80 } else {
81 // If there is a second argument that also needs to be a
82 // constant so that we can infer the size
83 if (SecondConstant) {
84 KnownSize = *FirstConstant * *SecondConstant;
85 } else {
86 // Otherwise can only infer that it is a multiple of N
87 SizeMultipleOf = FirstConstant;
88 }
89 }
90 } else if (SecondConstant) {
91 // First argument is not a constant, but if the second one is we
92 // can at least infer the known multiple
93 SizeMultipleOf = SecondConstant;
94 }
95 // Assume zero for now, parse assume_aligned bits?
96 // Assume that all alloc_size functions allocate on the heap.
97 // This may not be quite true since some might use shared memory
98 // but shouldn't really matter for analysis purposes
99 cheri::CSetBoundsStats->add(
100 Alignment, KnownSize, "function with alloc_size",
101 cheri::SetBoundsPointerSource::Heap,
102 "call to " +
103 (CalledFunc ? CalledFunc->getName() : "function pointer"),
104 cheri::inferSourceLocation(&I), SizeMultipleOf);
105 };
106 // We can only do this analysis on calls that return pointers
107 if (!I.getType()->isPointerTy())
108 continue;
109
110 // Check both the target function and the CallInst for alloc_size attrs:
111 if (CalledFunc) {
112 if (CalledFunc->hasFnAttribute(Attribute::AttrKind::AllocSize)) {
113 Attribute Attr =
114 CalledFunc->getFnAttribute(Attribute::AttrKind::AllocSize);
115 LogAllocSize(Attr.getAllocSizeArgs());
116 continue;
117 }
118 }
119 // no attr on the called func, try call site too
120 if (AL.hasFnAttribute(Attribute::AttrKind::AllocSize)) {
121 LogAllocSize(AL.getFnAttributes().getAllocSizeArgs());
122 }
123 }
124 }
125 // TODO: get machine function and count instructions?
126 // Probably better to get the code bounds from the ELF writer instead
127 // MF.getInstructionCount();
128 return false;
129 }
130
131 StringRef getPassName() const override { return "Log CHERI SetBounds Stats"; }
132};
133
134} // end anonymous namespace
135
136char LogCheriSetBounds::ID;
137
138FunctionPass *llvm::createLogCheriSetBoundsPass() {
139 static bool created = false;
140 assert(!created && "Should only be created once");
141 created = true;
142 return new LogCheriSetBounds();
143}
