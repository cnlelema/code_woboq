35namespace {
36class CHERICapDirectCalls : public FunctionPass,
37 public InstVisitor<CHERICapDirectCalls> {
38 bool Modified = false;
39 SmallVector<Value*,32> DeadInstructions;
40 StringRef getPassName() const override {
41 return "Contract CHERI pure-capability calls";
42 }
43public:
44 static char ID;
45 CHERICapDirectCalls() : FunctionPass(ID) {}
46 void visitCallSite(CallSite CS) {
47 Value *Callee = CS.getCalledValue()->stripPointerCasts();
48 if (Callee->getType()->getPointerAddressSpace() != 200)
49 return;
50 auto *SetOffset = dyn_cast<IntrinsicInst>(Callee);
51 if (!SetOffset)
52 return;
53 if (SetOffset->getIntrinsicID() != llvm::Intrinsic::cheri_cap_offset_set)
54 return;
55 auto *PtrToInt = dyn_cast<PtrToIntOperator>(SetOffset->getOperand(1));
56 if (!PtrToInt)
57 return;
58 auto *Func =
59 dyn_cast<Function>(PtrToInt->getOperand(0)->stripPointerCasts());
60 if (!Func)
61 return;
62 auto *CalledValue = CS.getCalledValue();
63 // insert a bitcast if the type of CalledValue and Func are different
64 if (CalledValue->getType() != Func->getType()) {
65 if (auto *CalledValPtrTy = dyn_cast<PointerType>(CalledValue->getType())) {
66 IRBuilder<> B(CS.getInstruction());
67 // XXXAR: the legacy ABI expects all functions to be in AS0 even though program AS is 200!
68 auto *NewCalledValue = B.CreateBitCast(Func, PointerType::get(CalledValPtrTy->getElementType(), 0));
69 CS.setCalledFunction(NewCalledValue);
70 }
71 }
72 else {
73 CS.setCalledFunction(Func);
74 CS.mutateFunctionType(cast<FunctionType>(Func->getType()->getElementType()));
75 }
76 if (CalledValue->use_begin() == CalledValue->use_end()) {
77 DeadInstructions.push_back(CalledValue);
78 } else {
79#if !defined(NDEBUG) || defined(LLVM_ENABLE_DUMP)
80 llvm::errs() << "CALLEE: "; CalledValue->dump();
81 llvm::errs() << "Func: "; Func->dump();
82 llvm::errs() << "NUM USES: " << CalledValue->getNumUses() << "\n";
83#endif
84 assert(CalledValue->getNumUses() == 0 && "Unexpected uses");
85 }
86 }
87 bool runOnFunction(Function &F) override {
88 Modified = false;
89 DeadInstructions.clear();
90 visit(F);
91 for (auto *V : DeadInstructions)
92 RecursivelyDeleteTriviallyDeadInstructions(V);
93 return Modified;
94 }
95 void getAnalysisUsage(AnalysisUsage &AU) const override {
96 AU.setPreservesCFG();
97 }
98};
99}
100
101char CHERICapDirectCalls::ID = 0;
102INITIALIZE_PASS(CHERICapDirectCalls, "cheri-direct-calls",
103 "Eliminate PCC-relative calls", false, false)
104
105Pass *llvm::createCHERICapDirectCallsPass() { return new CHERICapDirectCalls(); }
