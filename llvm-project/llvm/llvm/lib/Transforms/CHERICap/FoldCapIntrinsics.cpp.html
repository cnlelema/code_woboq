57 template <typename Infer>
58 void foldGet(Module *M, Intrinsic::ID ID, ArrayRef<Type*> Tys, Infer infer,
59 int NullValue = 0) {
60 Function *Func = M->getFunction(Intrinsic::getName(ID, Tys));
61 if (!Func)
62 return;
63 // Calling eraseFromParent() inside the following loop causes iterators
64 // to be invalidated and crashes -> collect and erase instead
65 SmallVector<Instruction *, 8> ToErase;
66 for (Value *Use : Func->users()) {
67 CallInst *CI = cast<CallInst>(Use);
68 if (Value *Replacement = infer(CI->getOperand(0), CI, NullValue)) {
69 LLVM_DEBUG(dbgs() << "Replacing all uses: "; CI->dump();
70 dbgs() << "New replacement: "; Replacement->dump(););
71 CI->replaceAllUsesWith(Replacement);
72 // CI->eraseFromParent();
73 ToErase.push_back(CI);
74 Modified = true;
75 }
76 }
77 RecursivelyDeleteTriviallyDeadInstructions(ToErase);
78 }

80 void foldGetIntrinisics(Module *M) {
81 foldGet(M, Intrinsic::cheri_cap_offset_get, {CapSizeTy},
82 [this](Value *V, CallInst *CI, int) {
83 return inferCapabilityOffset(V, CI, CI->getType());
84 });
85 foldGet(M, Intrinsic::cheri_cap_address_get, {CapAddrTy},
86 [this](Value *V, CallInst *CI, int) {
87 return inferCapabilityAddress(V, CI, CI->getType());
88 });
89 // For all other capability fields we can only infer a value when the
90 // argument is a null capability (potentially with an offset)
91 auto inferOther = [this](Value *V, CallInst *CI, int NullValue) {
92 // TODO: Can we ignore capabilities becoming unrepresentable for any of
93 // these intrinsics?
94 return inferCapabilityNonOffsetField(V, CI, NullValue);
95 };
96 foldGet(M, Intrinsic::cheri_cap_base_get, {CapAddrTy}, inferOther);
97 foldGet(M, Intrinsic::cheri_cap_perms_get, {CapSizeTy}, inferOther);
98 foldGet(M, Intrinsic::cheri_cap_flags_get, {CapSizeTy}, inferOther);
99 foldGet(M, Intrinsic::cheri_cap_tag_get, {},
100 [this](Value *V, CallInst *CI, int) -> Value * {
101 if (cheri::isKnownUntaggedCapability(V, DL))
102 return Constant::getNullValue(CI->getType());
103 return nullptr;
104 });
105 foldGet(M, Intrinsic::cheri_cap_sealed_get, {}, inferOther);
106 // CGetType and CGetLen on a null capability now return -1
107 foldGet(M, Intrinsic::cheri_cap_length_get, {CapSizeTy}, inferOther, -1);
108 foldGet(M, Intrinsic::cheri_cap_type_get, {CapSizeTy}, inferOther, -1);
109 }

111 static Constant* getIntToPtrSourceValue(Value* V) {
112 if (ConstantExpr* CE = dyn_cast<ConstantExpr>(V)) {
113 if (CE->isCast() && CE->getOpcode() == Instruction::IntToPtr) {
114 assert(CE->getNumOperands() == 1);
115 Constant *Src = CE->getOperand(0);
116 return Src;
117 }
118 }
119 return nullptr;
120 }

122 Value *inferCapabilityNonOffsetField(Value *V, CallInst *Call, int NullValue,
123 bool OnlyIfRootWasNull = false) {
124 // Calling an llvm.cheri.cap.$INTRIN.get() on a null value or
125 // an integer stored in a capability always returns 0 or -1 (for CGetLen and
126 // CGetType) unless $INTRIN is offset/address (in which case it returns the
127 // int value)
128 if (isa<ConstantPointerNull>(V) || getIntToPtrSourceValue(V)) {
129 Type *Ty = Call->getType();
130 return NullValue == 0 ? llvm::Constant::getNullValue(Ty)
131 : llvm::Constant::getAllOnesValue(Ty);
132 }
133 Value *Arg = nullptr;
134 // We can ignore all setoffset/incoffset/setaddr operations if called on
135 // null -> set OnlyIfRootWasNull=true when recursing. This currently doesn't
136 // change anything but if we were to also infer for non-null values we
137 // would need to abort any further analysis after a setoffset/incoffset or
138 // setaddr since the intrinsic could cause the capability to become
139 // unrepresentable (which is not an issue if it is already null)
140 if (match(V, m_Intrinsic<Intrinsic::cheri_cap_offset_set>(m_Value(Arg)))) {
141 return inferCapabilityNonOffsetField(Arg, Call, NullValue, true);
142 } else if (match(V, m_Intrinsic<Intrinsic::cheri_cap_offset_increment>(
143 m_Value(Arg)))) {
144 return inferCapabilityNonOffsetField(Arg, Call, NullValue, true);
145 } else if (match(V, m_Intrinsic<Intrinsic::cheri_cap_address_set>(
146 m_Value(Arg)))) {
147 return inferCapabilityNonOffsetField(Arg, Call, NullValue, true);
148 }
149 // TODO: is there anything else we can infer?
150 return nullptr;
151 }
152
153 // Returns the offset increment if V is a GEP instruction of
154 Value *getOffsetIncrement(Value *V, Value **Arg) {
155 if (!V)
156 return nullptr;
157 Value *Offset = nullptr;
158 if (match(V, m_Intrinsic<Intrinsic::cheri_cap_offset_increment>(
159 m_Value(*Arg), m_Value(Offset)))) {
160 return Offset;
161 } else if (GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(V)) {
162 *Arg = GEP->getOperand(0);
163
164 // XXXAR: do I need the inbounds check here?
165 if (GEP->isInBounds() && GEP->getType() == I8CapTy) {
166 APInt Offset(CapSizeTy->getIntegerBitWidth(), 0, true);
167 if (GEP->accumulateConstantOffset(*DL, Offset)) {
168 return ConstantInt::get(CapSizeTy, Offset);
169 } else if (GEP->getOperand(1)->getType() == CapSizeTy) {
170 // also handle non-constant GEPs:
171 // XXXAR: not sure this is always profitable, sometimes a CIncOffset
172 // might be is better
173 return GEP->getOperand(1);
174 }
175 }
176 }
177 return nullptr;
178 }

180 template <Intrinsic::ID Intrin, typename InferFunc>
181 Value *inferCapabilityOffsetOrAddr(InferFunc &&InferOffsetOrAddr,
182 Value *V, CallInst *Call,
183 Type *IntTy,
184 Value **BaseCap = nullptr) {
185 // For null values both getoffset and getaddr return zero
186 if (isa<ConstantPointerNull>(V)) {
187 if (BaseCap)
188 *BaseCap = V;
189 return llvm::Constant::getNullValue(IntTy);
190 }
191 // For inttoptr values both getoffset and getaddr return the integer value
192 if (Constant* IntToPtr = getIntToPtrSourceValue(V)) {
193 if (BaseCap)
194 *BaseCap = V;
195 return IntToPtr;
196 }
197 Value *Arg = nullptr;
198 Value *IntrinArg = nullptr;
199 // For a setoffset we can infer the offset (and the address for setoffset on
200 // NULL)
201 if (match(V, m_Intrinsic<Intrinsic::cheri_cap_offset_set>(
202 m_Value(Arg), m_Value(IntrinArg)))) {
203 // If we are inferring the offset a setoffset can always be used. For
204 // the address we can only infer the value if the setoffset call is on
205 // NULL
206 if (Intrin == Intrinsic::cheri_cap_offset_get ||
207 isa<ConstantPointerNull>(Arg)) {
208 if (BaseCap)
209 *BaseCap = Arg;
210 return IntrinArg;
211 }
212 }
213
214 if (match(V, m_Intrinsic<Intrinsic::cheri_cap_address_set>(
215 m_Value(Arg), m_Value(IntrinArg)))) {
216 // If we are inferring the offset a we can only infer the value of the
217 // setaddr intrinsic was used on NULL. For getaddr the value will always
218 // be the argumennt of the setaddr intrinsic.
219 if (Intrin == Intrinsic::cheri_cap_address_get ||
220 isa<ConstantPointerNull>(Arg)) {
221 if (BaseCap)
222 *BaseCap = Arg;
223 return IntrinArg;
224 }
225 }
226 // Finally we can fold inc-offset calls into the result as adds
227 if (Value *Increment = getOffsetIncrement(V, &Arg)) {
228 if (Value *LHS = (this->*InferOffsetOrAddr)(Arg, Call, IntTy, BaseCap)) {
229 IRBuilder<> B(cast<Instruction>(V));
230 return B.CreateAdd(LHS, Increment);
231 }
232 }
233 // TODO: is there anything else we can infer?
234 return nullptr;
235 }

237 Value *inferCapabilityOffset(Value *V, CallInst *Call, Type *IntTy,
238 Value **BaseCap = nullptr) {
239 return inferCapabilityOffsetOrAddr<Intrinsic::cheri_cap_offset_get>(
240 &CHERICapFoldIntrinsics::inferCapabilityOffset, V, Call, IntTy, BaseCap);
241 }

243 Value *inferCapabilityAddress(Value *V, CallInst *Call, Type *IntTy,
244 Value **BaseCap = nullptr) {
245 return inferCapabilityOffsetOrAddr<Intrinsic::cheri_cap_address_get>(
246 &CHERICapFoldIntrinsics::inferCapabilityAddress, V, Call, IntTy, BaseCap);
247 }

249 void foldIncOffsetSetOffsetSetAddrOnlyUserIncrement(
250 CallInst *CI, SmallPtrSet<Instruction *, 8> &ToErase) {
251 // errs() << __func__ << ": num users=" << CI->getNumUses() << ": "; CI->dump();
252 User* OnlyUser = CI->hasOneUse() ? *CI->user_begin() : nullptr;
253 while (OnlyUser) {
254 // If there is only a single use of the setoffset result, we might be
255 // able to fold it into a single setoffset (for a GEP or incoffset)
256 Value* Arg = nullptr;
257 if (Value *Increment = getOffsetIncrement(OnlyUser, &Arg)) {
258#if 0
259 errs() << "CI before folding: "; CI->dump();
260 errs() << "Block before folding: "; CI->getParent()->dump();
261 errs() << "OnlyUser: "; OnlyUser->dump();
262 errs() << "Increment: "; Increment->dump();
263#endif
264 assert(Arg == CI);
265 Instruction *ReplacedInstr = cast<Instruction>(OnlyUser);
266 IRBuilder<> B(ReplacedInstr);
267 Value *NewOffset = B.CreateAdd(CI->getOperand(1), Increment);
268 LLVM_DEBUG(dbgs() << "CI->setOperand(1) -> "; NewOffset->dump());
269 CI->setOperand(1, NewOffset);
270 LLVM_DEBUG(dbgs() << "New CI: "; CI->dump());
271 // We have to move this instruction after the offset instruction
272 // because otherwise we use a value that has not yet been defined
273 CI->moveAfter(ReplacedInstr);
274 // Keep doing this transformation for incoffset chains with only a
275 // single user:
276 OnlyUser = ReplacedInstr->hasOneUse() ? *ReplacedInstr->user_begin() : nullptr;
277 LLVM_DEBUG(dbgs() << "Replacing all uses: "; ReplacedInstr->dump();
278 dbgs() << "New replacement: "; CI->dump(););
279 ReplacedInstr->replaceAllUsesWith(CI);
280#if 0
281 errs() << "ReplacedInstr (" << ReplacedInstr->getNumUses() << " uses): "; ReplacedInstr->dump();
282 errs() << "ReplacedInstr: "; ReplacedInstr->dump();
283 errs() << "New OnlyUser: "; if (OnlyUser) OnlyUser->dump(); else errs() << "nullptr\n";
284 errs() << "New CI (" << CI->getNumUses() << " uses): "; CI->dump();
285 errs() << "New Block: "; CI->getParent()->dump();
286#endif
287 // erasing here can cause a crash -> add to list so that caller can remove it
288 // ReplacedInstr->eraseFromParent();
289 ToErase.insert(ReplacedInstr);
290 assert(OnlyUser != ReplacedInstr && "Should not cause an infinite loop!");
291 Modified = true;
292 if (!OnlyUser)
293 break;
294 } else {
295 break;
296 }
297 }
298 }

300 /// Replace get-offset, add, set-offset sequences with inc-offset
301 void foldSetOffset(Module *M) {
302 foldSetOffsetOrSetAddress<Intrinsic::cheri_cap_offset_get,
303 Intrinsic::cheri_cap_offset_set>(M, SetOffset);
304 }
305
306 /// Replace get-address, add, set-address sequences with inc-offset
307 void foldSetAddress(Module *M) {
308 foldSetOffsetOrSetAddress<Intrinsic::cheri_cap_address_get,
309 Intrinsic::cheri_cap_address_set>(M, SetAddr);
310 }

312 template <Intrinsic::ID GetIntrinsic, Intrinsic::ID SetIntrinsic>
313 void foldSetOffsetOrSetAddress(Module *M, Function *SetFunc) {
314 assert(IncOffset && "Not initialized correclty");
315 SmallVector<CallInst *, 16> SetCalls;
316 SmallPtrSet<Instruction*, 8> ToErase;
317 for (Value *V : SetFunc->users())
318 SetCalls.push_back(cast<CallInst>(V));
319 for (CallInst *CI : SetCalls) {
320 if (ToErase.count(CI)) {
321 assert(CI->hasNUses(0));
322 continue;
323 }
324
325 Value *LHS, *RHS;
326 // Replace set{offset,addr}(set{offset,addr}(C, A), B) with
327 // set{offset,addr}(C, B). Must come before the setoffset to
328 // incoffset transformation in case C is an add.
329 while (match(CI->getOperand(0),
330 m_Intrinsic<SetIntrinsic>(m_Value(LHS), m_Value(RHS)))) {
331 Instruction *NestedInstr = cast<Instruction>(CI->getOperand(0));
332 if (NestedInstr->hasOneUse())
333 ToErase.insert(NestedInstr);
334 LLVM_DEBUG(dbgs() << "CI->setOperand(0) -> "; LHS->dump());
335 CI->setOperand(0, LHS);
336 if (isa<ConstantPointerNull>(LHS))
337 CI->removeParamAttr(0, Attribute::NonNull);
338 LLVM_DEBUG(dbgs() << "New CI: "; CI->dump());
339 Modified = true;
340 }
341 Value *BaseCap = CI->getOperand(0);
342
343 // fold chains of set{offset,addr}, (inc-offset/GEP)+ into a single set-{offset,addr}
344 foldIncOffsetSetOffsetSetAddrOnlyUserIncrement(CI, ToErase);
345
346 if (match(CI->getOperand(1), m_Add(m_Value(LHS), m_Value(RHS)))) {
347 Value *Add = nullptr;
348 if (match(LHS, m_Intrinsic<GetIntrinsic>(m_Specific(BaseCap))))
349 Add = RHS;
350 else if (match(RHS, m_Intrinsic<GetIntrinsic>(m_Specific(BaseCap))))
351 Add = LHS;
352 if (Add) {
353 IRBuilder<> B(CI);
354 CallInst *Replacement = B.CreateCall(IncOffset, {BaseCap, Add});
355 Replacement->setTailCall(true);
356 LLVM_DEBUG(dbgs() << "Replacing all uses: "; CI->dump();
357 dbgs() << "New replacement: "; Replacement->dump(););
358 CI->replaceAllUsesWith(Replacement);
359 ToErase.insert(CI);
360 Modified = true;
361 }
362 }
363
364 // Fold set{offset,addr}(A, get{offset,addr}(A)) -> A
365 if (match(CI->getOperand(1),
366 m_Intrinsic<GetIntrinsic>(m_Specific(BaseCap)))) {
367 LLVM_DEBUG(dbgs() << "Replacing all uses: "; CI->dump();
368 dbgs() << "New replacement: "; BaseCap->dump(););
369 CI->replaceAllUsesWith(BaseCap);
370 ToErase.insert(CI);
371 Modified = true;
372 }
373
374 // Fold set{offset,addr}(NULL, 0) -> NULL
375 if (match(BaseCap, m_Zero()) && match(CI->getOperand(1), m_Zero())) {
376 assert(isa<ConstantPointerNull>(BaseCap));
377 LLVM_DEBUG(dbgs() << "Replacing all uses: "; CI->dump();
378 dbgs() << "New replacement: "; BaseCap->dump(););
379 CI->replaceAllUsesWith(BaseCap);
380 ToErase.insert(CI);
381 Modified = true;
382 }
383
384 }
385 for (Instruction* I : ToErase)
386 RecursivelyDeleteTriviallyDeadInstructions(I, TLI);
387 }

389 /// Replace set-offset, inc-offset sequences with a single set-offset
390 /// Also fold multiple inc-offsets into a single on if possible
391 void foldIncOffset() {
392 SmallVector<CallInst *, 16> IncOffsets;
393 SmallPtrSet<Instruction*, 8> ToErase;
394 for (Value *V : IncOffset->users())
395 IncOffsets.push_back(cast<CallInst>(V));
396 for (CallInst *CI : IncOffsets) {
397 if (ToErase.count(CI)) {
398 assert(CI->hasNUses(0));
399 continue;
400 }
401 // fold chains of inc-offset, (inc-offset/GEP)+ into a single inc-offset
402 foldIncOffsetSetOffsetSetAddrOnlyUserIncrement(CI, ToErase);
403
404 // Fold incoffset(A, 0) -> A since incrementing by zero doesn't do anything
405 if (match(CI->getOperand(1), m_Zero())) {
406 LLVM_DEBUG(dbgs() << "Replacing all uses: "; CI->dump();
407 dbgs() << "New replacement: "; CI->getOperand(0)->dump(););
408 CI->replaceAllUsesWith(CI->getOperand(0));
409 ToErase.insert(CI);
410 Modified = true;
411 continue;
412 }
413 }
414 for (Instruction* I : ToErase)
415 RecursivelyDeleteTriviallyDeadInstructions(I);
416 }

418public:
419 static char ID;
420 CHERICapFoldIntrinsics() : ModulePass(ID) {}
421 StringRef getPassName() const override {
422 return "CHERI fold capability intrinsics";
423 }
424 bool runOnModule(Module &M) override {
425 if (skipModule(M))
426 return false;
427
428 TLI = &getAnalysis<TargetLibraryInfoWrapperPass>().getTLI();
429
430 Modified = false;
431 DL = &M.getDataLayout();
432 LLVMContext &C = M.getContext();
433 I8CapTy = Type::getInt8PtrTy(C, 200);
434 CapAddrTy = Type::getIntNTy(C, DL->getPointerBaseSizeInBits(200));
435 CapSizeTy = Type::getIntNTy(C, DL->getIndexSizeInBits(200));
436 // Don't add these intrinsics to the module if none of them are used:
437 IncOffset = M.getFunction(
438 Intrinsic::getName(Intrinsic::cheri_cap_offset_increment, CapSizeTy));
439 SetOffset =
440 M.getFunction(Intrinsic::getName(Intrinsic::cheri_cap_offset_set,
441 CapSizeTy));
442 GetOffset =
443 M.getFunction(Intrinsic::getName(Intrinsic::cheri_cap_offset_get,
444 CapSizeTy));
445 SetAddr =
446 M.getFunction(Intrinsic::getName(Intrinsic::cheri_cap_address_set,
447 CapAddrTy));
448 GetAddr =
449 M.getFunction(Intrinsic::getName(Intrinsic::cheri_cap_address_get,
450 CapAddrTy));
451 // at least one intrinsic was used -> we need to run the fold
452 // setoffset/incoffset pass
453
454 if (IncOffset || SetOffset || GetOffset) {
455 // Ensure that all the intrinsics exist in the module
456 if (!IncOffset)
457 IncOffset = Intrinsic::getDeclaration(
458 &M, Intrinsic::cheri_cap_offset_increment, CapSizeTy);
459 if (!SetOffset)
460 SetOffset = Intrinsic::getDeclaration(
461 &M, Intrinsic::cheri_cap_offset_set, CapSizeTy);
462 if (!GetOffset)
463 GetOffset = Intrinsic::getDeclaration(
464 &M, Intrinsic::cheri_cap_offset_get, CapSizeTy);
465 // TODO: does the order here matter?
466 foldIncOffset();
467 foldSetOffset(&M);
468 }
469 if (SetAddr || GetAddr) {
470 // Ensure that all the intrinsics exist in the module
471 if (!SetAddr)
472 SetAddr = Intrinsic::getDeclaration(
473 &M, Intrinsic::cheri_cap_address_set, CapAddrTy);
474 if (!GetAddr)
475 GetAddr = Intrinsic::getDeclaration(
476 &M, Intrinsic::cheri_cap_address_get, CapAddrTy);
477 if (!IncOffset)
478 IncOffset = Intrinsic::getDeclaration(
479 &M, Intrinsic::cheri_cap_offset_increment, CapSizeTy);
480 // TODO: does the order here matter?
481 foldSetAddress(&M);
482 }
483
484 foldGetIntrinisics(&M);
485 return Modified;
486 }
487 void getAnalysisUsage(AnalysisUsage &AU) const override {
488 AU.addRequired<TargetLibraryInfoWrapperPass>();
489 AU.setPreservesCFG();
490 }
