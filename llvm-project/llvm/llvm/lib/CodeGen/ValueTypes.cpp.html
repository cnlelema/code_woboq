17EVT EVT::changeExtendedTypeToInteger() const {
18 LLVMContext &Context = LLVMTy->getContext();
19 return getIntegerVT(Context, getSizeInBits());
20}
21
22EVT EVT::changeExtendedVectorElementTypeToInteger() const {
23 LLVMContext &Context = LLVMTy->getContext();
24 EVT IntTy = getIntegerVT(Context, getScalarSizeInBits());
25 return getVectorVT(Context, IntTy, getVectorNumElements());
26}

28EVT EVT::getExtendedIntegerVT(LLVMContext &Context, unsigned BitWidth) {
29 EVT VT;
30 VT.LLVMTy = IntegerType::get(Context, BitWidth);
31 assert(VT.isExtended() && "Type is not extended!");
32 return VT;
33}

35EVT EVT::getExtendedVectorVT(LLVMContext &Context, EVT VT,
36 unsigned NumElements) {
37 EVT ResultVT;
38 ResultVT.LLVMTy = VectorType::get(VT.getTypeForEVT(Context), NumElements);
39 assert(ResultVT.isExtended() && "Type is not extended!");
40 return ResultVT;
41}

105unsigned EVT::getExtendedSizeInBits() const {
106 assert(isExtended() && "Type is not extended!");
107 if (IntegerType *ITy = dyn_cast<IntegerType>(LLVMTy))
108 return ITy->getBitWidth();
109 if (VectorType *VTy = dyn_cast<VectorType>(LLVMTy))
110 return VTy->getBitWidth();
111 llvm_unreachable("Unrecognized extended type!");
112}

208/// getTypeForEVT - This method returns an LLVM type corresponding to the
209/// specified EVT. For integer types, this returns an unsigned type. Note
210/// that this will abort for types that cannot be represented.
211Type *EVT::getTypeForEVT(LLVMContext &Context) const {
212 switch (V.SimpleTy) {
213 default:
214 assert(isExtended() && "Type is not extended!");
215 return LLVMTy;
216 case MVT::isVoid: return Type::getVoidTy(Context);
217 case MVT::i1: return Type::getInt1Ty(Context);
218 case MVT::i8: return Type::getInt8Ty(Context);
219 case MVT::i16: return Type::getInt16Ty(Context);
220 case MVT::i32: return Type::getInt32Ty(Context);
221 case MVT::i64: return Type::getInt64Ty(Context);
222 case MVT::i128: return IntegerType::get(Context, 128);
223 case MVT::f16: return Type::getHalfTy(Context);
224 case MVT::f32: return Type::getFloatTy(Context);
225 case MVT::f64: return Type::getDoubleTy(Context);
226 case MVT::f80: return Type::getX86_FP80Ty(Context);
227 case MVT::f128: return Type::getFP128Ty(Context);
228 case MVT::ppcf128: return Type::getPPC_FP128Ty(Context);
229 case MVT::x86mmx: return Type::getX86_MMXTy(Context);
230 case MVT::v1i1: return VectorType::get(Type::getInt1Ty(Context), 1);
231 case MVT::v2i1: return VectorType::get(Type::getInt1Ty(Context), 2);
232 case MVT::v4i1: return VectorType::get(Type::getInt1Ty(Context), 4);
233 case MVT::v8i1: return VectorType::get(Type::getInt1Ty(Context), 8);
234 case MVT::v16i1: return VectorType::get(Type::getInt1Ty(Context), 16);
235 case MVT::v32i1: return VectorType::get(Type::getInt1Ty(Context), 32);
236 case MVT::v64i1: return VectorType::get(Type::getInt1Ty(Context), 64);
237 case MVT::v128i1: return VectorType::get(Type::getInt1Ty(Context), 128);
238 case MVT::v512i1: return VectorType::get(Type::getInt1Ty(Context), 512);
239 case MVT::v1024i1: return VectorType::get(Type::getInt1Ty(Context), 1024);
240 case MVT::v1i8: return VectorType::get(Type::getInt8Ty(Context), 1);
241 case MVT::v2i8: return VectorType::get(Type::getInt8Ty(Context), 2);
242 case MVT::v4i8: return VectorType::get(Type::getInt8Ty(Context), 4);
243 case MVT::v8i8: return VectorType::get(Type::getInt8Ty(Context), 8);
244 case MVT::v16i8: return VectorType::get(Type::getInt8Ty(Context), 16);
245 case MVT::v32i8: return VectorType::get(Type::getInt8Ty(Context), 32);
246 case MVT::v64i8: return VectorType::get(Type::getInt8Ty(Context), 64);
247 case MVT::v128i8: return VectorType::get(Type::getInt8Ty(Context), 128);
248 case MVT::v256i8: return VectorType::get(Type::getInt8Ty(Context), 256);
249 case MVT::v1i16: return VectorType::get(Type::getInt16Ty(Context), 1);
250 case MVT::v2i16: return VectorType::get(Type::getInt16Ty(Context), 2);
251 case MVT::v4i16: return VectorType::get(Type::getInt16Ty(Context), 4);
252 case MVT::v8i16: return VectorType::get(Type::getInt16Ty(Context), 8);
253 case MVT::v16i16: return VectorType::get(Type::getInt16Ty(Context), 16);
254 case MVT::v32i16: return VectorType::get(Type::getInt16Ty(Context), 32);
255 case MVT::v64i16: return VectorType::get(Type::getInt16Ty(Context), 64);
256 case MVT::v128i16: return VectorType::get(Type::getInt16Ty(Context), 128);
257 case MVT::v1i32: return VectorType::get(Type::getInt32Ty(Context), 1);
258 case MVT::v2i32: return VectorType::get(Type::getInt32Ty(Context), 2);
259 case MVT::v3i32: return VectorType::get(Type::getInt32Ty(Context), 3);
260 case MVT::v4i32: return VectorType::get(Type::getInt32Ty(Context), 4);
261 case MVT::v5i32: return VectorType::get(Type::getInt32Ty(Context), 5);
262 case MVT::v8i32: return VectorType::get(Type::getInt32Ty(Context), 8);
263 case MVT::v16i32: return VectorType::get(Type::getInt32Ty(Context), 16);
264 case MVT::v32i32: return VectorType::get(Type::getInt32Ty(Context), 32);
265 case MVT::v64i32: return VectorType::get(Type::getInt32Ty(Context), 64);
266 case MVT::v1i64: return VectorType::get(Type::getInt64Ty(Context), 1);
267 case MVT::v2i64: return VectorType::get(Type::getInt64Ty(Context), 2);
268 case MVT::v4i64: return VectorType::get(Type::getInt64Ty(Context), 4);
269 case MVT::v8i64: return VectorType::get(Type::getInt64Ty(Context), 8);
270 case MVT::v16i64: return VectorType::get(Type::getInt64Ty(Context), 16);
271 case MVT::v32i64: return VectorType::get(Type::getInt64Ty(Context), 32);
272 case MVT::v1i128: return VectorType::get(Type::getInt128Ty(Context), 1);
273 case MVT::v2f16: return VectorType::get(Type::getHalfTy(Context), 2);
274 case MVT::v4f16: return VectorType::get(Type::getHalfTy(Context), 4);
275 case MVT::v8f16: return VectorType::get(Type::getHalfTy(Context), 8);
276 case MVT::v1f32: return VectorType::get(Type::getFloatTy(Context), 1);
277 case MVT::v2f32: return VectorType::get(Type::getFloatTy(Context), 2);
278 case MVT::v3f32: return VectorType::get(Type::getFloatTy(Context), 3);
279 case MVT::v4f32: return VectorType::get(Type::getFloatTy(Context), 4);
280 case MVT::v5f32: return VectorType::get(Type::getFloatTy(Context), 5);
281 case MVT::v8f32: return VectorType::get(Type::getFloatTy(Context), 8);
282 case MVT::v16f32: return VectorType::get(Type::getFloatTy(Context), 16);
283 case MVT::v1f64: return VectorType::get(Type::getDoubleTy(Context), 1);
284 case MVT::v2f64: return VectorType::get(Type::getDoubleTy(Context), 2);
285 case MVT::v4f64: return VectorType::get(Type::getDoubleTy(Context), 4);
286 case MVT::v8f64: return VectorType::get(Type::getDoubleTy(Context), 8);
287 case MVT::Metadata: return Type::getMetadataTy(Context);
288 case MVT::iFATPTR64:
289 case MVT::iFATPTR128:
290 case MVT::iFATPTR256:
291 case MVT::iFATPTR512:
292 case MVT::iFATPTRAny:
293 return PointerType::get(Type::getInt8Ty(Context), 200);
294 }
295}

297/// Return the value type corresponding to the specified type. This returns all
298/// pointers as MVT::iPTR. If HandleUnknown is true, unknown types are returned
299/// as Other, otherwise they are invalid.
300MVT MVT::getVT(Type *Ty, bool HandleUnknown){
301 switch (Ty->getTypeID()) {
302 default:
303 if (HandleUnknown) return MVT(MVT::Other);
304 llvm_unreachable("Unknown type!");
305 case Type::VoidTyID:
306 return MVT::isVoid;
307 case Type::IntegerTyID:
308 return getIntegerVT(cast<IntegerType>(Ty)->getBitWidth());
309 case Type::HalfTyID: return MVT(MVT::f16);
310 case Type::FloatTyID: return MVT(MVT::f32);
311 case Type::DoubleTyID: return MVT(MVT::f64);
312 case Type::X86_FP80TyID: return MVT(MVT::f80);
313 case Type::X86_MMXTyID: return MVT(MVT::x86mmx);
314 case Type::FP128TyID: return MVT(MVT::f128);
315 case Type::PPC_FP128TyID: return MVT(MVT::ppcf128);
316 case Type::PointerTyID: {
317 // FIXME: require a DataLayout here!
318 if (isCheriPointer(Ty, nullptr))
319 return MVT(MVT::iFATPTRAny);
320 return MVT(MVT::iPTR);
321 }
322 case Type::VectorTyID: {
323 VectorType *VTy = cast<VectorType>(Ty);
324 return getVectorVT(
325 getVT(VTy->getElementType(), false), VTy->getNumElements());
326 }
327 }
328}

330/// getEVT - Return the value type corresponding to the specified type. This
331/// returns all pointers as MVT::iPTR. If HandleUnknown is true, unknown types
332/// are returned as Other, otherwise they are invalid.
333EVT EVT::getEVT(Type *Ty, bool HandleUnknown){
334 switch (Ty->getTypeID()) {
335 default:
336 return MVT::getVT(Ty, HandleUnknown);
337 case Type::IntegerTyID:
338 return getIntegerVT(Ty->getContext(), cast<IntegerType>(Ty)->getBitWidth());
339 case Type::VectorTyID: {
340 VectorType *VTy = cast<VectorType>(Ty);
341 return getVectorVT(Ty->getContext(), getEVT(VTy->getElementType(), false),
342 VTy->getNumElements());
343 }
344 }
