71// If @MI is a DBG_VALUE with debug value described by a defined
72// register, returns the number of this register. In the other case, returns 0.
73static unsigned isDbgValueDescribedByReg(const MachineInstr &MI) {
74 assert(MI.isDebugValue() && "expected a DBG_VALUE");
75 assert(MI.getNumOperands() == 4 && "malformed DBG_VALUE");
76 // If location of variable is described using a register (directly
77 // or indirectly), this register is always a first operand.
78 return MI.getOperand(0).isReg() ? MI.getOperand(0).getReg() : 0;
79}

93 /// Keeps track of lexical scopes associated with a user value's source
94 /// location.
95 class UserValueScopes {
96 DebugLoc DL;
97 LexicalScopes &LS;
98 SmallPtrSet<const MachineBasicBlock *, 4> LBlocks;
99
100 public:
101 UserValueScopes(DebugLoc D, LexicalScopes &L) : DL(std::move(D)), LS(L) {}
102
103 /// Return true if current scope dominates at least one machine
104 /// instruction in a given machine basic block.
105 bool dominates(MachineBasicBlock *MBB) {
106 if (LBlocks.empty())
107 LS.getMachineBasicBlocks(DL, LBlocks);
108 return LBlocks.count(MBB) != 0 || LS.dominates(DL, MBB);
109 }
110 };

112 using FragmentInfo = DIExpression::FragmentInfo;

115 /// Storage for identifying a potentially inlined instance of a variable,
116 /// or a fragment thereof.
117 class DebugVariable {
118 const DILocalVariable *Variable;
119 OptFragmentInfo Fragment;
120 const DILocation *InlinedAt;
121
122 /// Fragment that will overlap all other fragments. Used as default when
123 /// caller demands a fragment.
124 static const FragmentInfo DefaultFragment;
125
126 public:
127 DebugVariable(const DILocalVariable *Var, OptFragmentInfo &&FragmentInfo,
128 const DILocation *InlinedAt)
129 : Variable(Var), Fragment(FragmentInfo), InlinedAt(InlinedAt) {}
130
131 DebugVariable(const DILocalVariable *Var, OptFragmentInfo &FragmentInfo,
132 const DILocation *InlinedAt)
133 : Variable(Var), Fragment(FragmentInfo), InlinedAt(InlinedAt) {}
134
135 DebugVariable(const DILocalVariable *Var, const DIExpression *DIExpr,
136 const DILocation *InlinedAt)
137 : DebugVariable(Var, DIExpr->getFragmentInfo(), InlinedAt) {}
138
139 DebugVariable(const MachineInstr &MI)
140 : DebugVariable(MI.getDebugVariable(),
141 MI.getDebugExpression()->getFragmentInfo(),
142 MI.getDebugLoc()->getInlinedAt()) {}
143
144 const DILocalVariable *getVar() const { return Variable; }
145 const OptFragmentInfo &getFragment() const { return Fragment; }
146 const DILocation *getInlinedAt() const { return InlinedAt; }
147
148 const FragmentInfo getFragmentDefault() const {
149 return Fragment.getValueOr(DefaultFragment);
150 }
151
152 static bool isFragmentDefault(FragmentInfo &F) {
153 return F == DefaultFragment;
154 }
155
156 bool operator==(const DebugVariable &Other) const {
157 return std::tie(Variable, Fragment, InlinedAt) ==
158 std::tie(Other.Variable, Other.Fragment, Other.InlinedAt);
159 }
160
161 bool operator<(const DebugVariable &Other) const {
162 return std::tie(Variable, Fragment, InlinedAt) <
163 std::tie(Other.Variable, Other.Fragment, Other.InlinedAt);
164 }
165 };

169 /// A pair of debug variable and value location.
170 struct VarLoc {
171 // The location at which a spilled variable resides. It consists of a
172 // register and an offset.
173 struct SpillLoc {
174 unsigned SpillBase;
175 int SpillOffset;
176 bool operator==(const SpillLoc &Other) const {
177 return SpillBase == Other.SpillBase && SpillOffset == Other.SpillOffset;
178 }
179 };
180
181 const DebugVariable Var;
182 const MachineInstr &MI; ///< Only used for cloning a new DBG_VALUE.
183 mutable UserValueScopes UVS;
184 enum VarLocKind {
185 InvalidKind = 0,
186 RegisterKind,
187 SpillLocKind,
188 ImmediateKind
189 } Kind = InvalidKind;
190
191 /// The value location. Stored separately to avoid repeatedly
192 /// extracting it from MI.
193 union {
194 uint64_t RegNo;
195 SpillLoc SpillLocation;
196 uint64_t Hash;
197 int64_t Immediate;
198 const ConstantFP *FPImm;
199 const ConstantInt *CImm;
200 } Loc;
201
202 VarLoc(const MachineInstr &MI, LexicalScopes &LS)
203 : Var(MI), MI(MI), UVS(MI.getDebugLoc(), LS) {
204 static_assert((sizeof(Loc) == sizeof(uint64_t)),
205 "hash does not cover all members of Loc");
206 assert(MI.isDebugValue() && "not a DBG_VALUE");
207 assert(MI.getNumOperands() == 4 && "malformed DBG_VALUE");
208 if (int RegNo = isDbgValueDescribedByReg(MI)) {
209 Kind = RegisterKind;
210 Loc.RegNo = RegNo;
211 } else if (MI.getOperand(0).isImm()) {
212 Kind = ImmediateKind;
213 Loc.Immediate = MI.getOperand(0).getImm();
214 } else if (MI.getOperand(0).isFPImm()) {
215 Kind = ImmediateKind;
216 Loc.FPImm = MI.getOperand(0).getFPImm();
217 } else if (MI.getOperand(0).isCImm()) {
218 Kind = ImmediateKind;
219 Loc.CImm = MI.getOperand(0).getCImm();
220 }
221 }
222
223 /// The constructor for spill locations.
224 VarLoc(const MachineInstr &MI, unsigned SpillBase, int SpillOffset,
225 LexicalScopes &LS)
226 : Var(MI), MI(MI), UVS(MI.getDebugLoc(), LS) {
227 assert(MI.isDebugValue() && "not a DBG_VALUE");
228 assert(MI.getNumOperands() == 4 && "malformed DBG_VALUE");
229 Kind = SpillLocKind;
230 Loc.SpillLocation = {SpillBase, SpillOffset};
231 }
232
233 // Is the Loc field a constant or constant object?
234 bool isConstant() const { return Kind == ImmediateKind; }
235
236 /// If this variable is described by a register, return it,
237 /// otherwise return 0.
238 unsigned isDescribedByReg() const {
239 if (Kind == RegisterKind)
240 return Loc.RegNo;
241 return 0;
242 }
243
244 /// Determine whether the lexical scope of this value's debug location
245 /// dominates MBB.
246 bool dominates(MachineBasicBlock &MBB) const { return UVS.dominates(&MBB); }
247
248#if !defined(NDEBUG) || defined(LLVM_ENABLE_DUMP)
249 LLVM_DUMP_METHOD void dump() const { MI.dump(); }
250#endif
251
252 bool operator==(const VarLoc &Other) const {
253 return Kind == Other.Kind && Var == Other.Var &&
254 Loc.Hash == Other.Loc.Hash;
255 }
256
257 /// This operator guarantees that VarLocs are sorted by Variable first.
258 bool operator<(const VarLoc &Other) const {
259 if (Var == Other.Var)
260 return Loc.Hash < Other.Loc.Hash;
261 return Var < Other.Var;
262 }
263 };

287 /// This holds the working set of currently open ranges. For fast
288 /// access, this is done both as a set of VarLocIDs, and a map of
289 /// DebugVariable to recent VarLocID. Note that a DBG_VALUE ends all
290 /// previous open ranges for the same variable.
291 class OpenRangesSet {
292 VarLocSet VarLocs;
293 SmallDenseMap<DebugVariable, unsigned, 8> Vars;
294 OverlapMap &OverlappingFragments;
295
296 public:
297 OpenRangesSet(OverlapMap &_OLapMap) : OverlappingFragments(_OLapMap) {}
298
299 const VarLocSet &getVarLocs() const { return VarLocs; }
300
301 /// Terminate all open ranges for Var by removing it from the set.
302 void erase(DebugVariable Var);
303
304 /// Terminate all open ranges listed in \c KillSet by removing
305 /// them from the set.
306 void erase(const VarLocSet &KillSet, const VarLocMap &VarLocIDs) {
307 VarLocs.intersectWithComplement(KillSet);
308 for (unsigned ID : KillSet)
309 Vars.erase(VarLocIDs[ID].Var);
310 }
311
312 /// Insert a new range into the set.
313 void insert(unsigned VarLocID, DebugVariable Var) {
314 VarLocs.set(VarLocID);
315 Vars.insert({Var, VarLocID});
316 }
317
318 /// Empty the set.
319 void clear() {
320 VarLocs.clear();
321 Vars.clear();
322 }
323
324 /// Return whether the set is empty or not.
325 bool empty() const {
326 assert(Vars.empty() == VarLocs.empty() && "open ranges are inconsistent");
327 return VarLocs.empty();
328 }
329 };
330
331 bool isSpillInstruction(const MachineInstr &MI, MachineFunction *MF,
332 unsigned &Reg);
333 /// If a given instruction is identified as a spill, return the spill location

335 Optional<VarLoc::SpillLoc> isRestoreInstruction(const MachineInstr &MI,
336 MachineFunction *MF,
337 unsigned &Reg);
338 /// Given a spill instruction, extract the register and offset used to

340 VarLoc::SpillLoc extractSpillBaseRegAndOffset(const MachineInstr &MI);
341 void insertTransferDebugPair(MachineInstr &MI, OpenRangesSet &OpenRanges,

353 const VarLocMap &VarLocIDs);
354 bool transferTerminatorInst(MachineInstr &MI, OpenRangesSet &OpenRanges,
355 VarLocInMBB &OutLocs, const VarLocMap &VarLocIDs);
356
357 bool process(MachineInstr &MI, OpenRangesSet &OpenRanges,
358 VarLocInMBB &OutLocs, VarLocMap &VarLocIDs,
359 TransferMap &Transfers, bool transferChanges);
360
361 void accumulateFragmentMap(MachineInstr &MI, VarToFragments &SeenFragments,
362 OverlapMap &OLapMap);
363
364 bool join(MachineBasicBlock &MBB, VarLocInMBB &OutLocs, VarLocInMBB &InLocs,
365 const VarLocMap &VarLocIDs,
366 SmallPtrSet<const MachineBasicBlock *, 16> &Visited,
367 SmallPtrSetImpl<const MachineBasicBlock *> &ArtificialBlocks);
368
369 bool ExtendRanges(MachineFunction &MF);

371public:
372 static char ID;
373
374 /// Default construct and initialize the pass.
375 LiveDebugValues();
376
377 /// Tell the pass manager which passes we depend on and what
378 /// information we preserve.
379 void getAnalysisUsage(AnalysisUsage &AU) const override;

400 using DV = LiveDebugValues::DebugVariable;
401 using OptFragmentInfo = LiveDebugValues::OptFragmentInfo;
402 using FragmentInfo = LiveDebugValues::FragmentInfo;

404 // Empty key: no key should be generated that has no DILocalVariable.
405 static inline DV getEmptyKey() {
406 return DV(nullptr, OptFragmentInfo(), nullptr);
407 }
408
409 // Difference in tombstone is that the Optional is meaningful

415 unsigned HV = 0;
416 const OptFragmentInfo &Fragment = D.getFragment();
417 if (Fragment)
418 HV = DenseMapInfo<FragmentInfo>::getHashValue(*Fragment);
419
420 return hash_combine(D.getVar(), HV, D.getInlinedAt());
421 }
422
423 static bool isEqual(const DV &A, const DV &B) { return A == B; }
424};

456/// Erase a variable from the set of open ranges, and additionally erase any
457/// fragments that may overlap it.
458void LiveDebugValues::OpenRangesSet::erase(DebugVariable Var) {
459 // Erasure helper.
460 auto DoErase = [this](DebugVariable VarToErase) {
461 auto It = Vars.find(VarToErase);
462 if (It != Vars.end()) {
463 unsigned ID = It->second;
464 VarLocs.reset(ID);
465 Vars.erase(It);
466 }
467 };
468
469 // Erase the variable/fragment that ends here.
470 DoErase(Var);
471
472 // Extract the fragment. Interpret an empty fragment as one that covers all
473 // possible bits.
474 FragmentInfo ThisFragment = Var.getFragmentDefault();
475
476 // There may be fragments that overlap the designated fragment. Look them up
477 // in the pre-computed overlap map, and erase them too.
478 auto MapIt = OverlappingFragments.find({Var.getVar(), ThisFragment});
479 if (MapIt != OverlappingFragments.end()) {
480 for (auto Fragment : MapIt->second) {
481 LiveDebugValues::OptFragmentInfo FragmentHolder;
482 if (!DebugVariable::isFragmentDefault(Fragment))
483 FragmentHolder = LiveDebugValues::OptFragmentInfo(Fragment);
484 DoErase({Var.getVar(), FragmentHolder, Var.getInlinedAt()});
485 }
486 }
487}

516LiveDebugValues::VarLoc::SpillLoc
517LiveDebugValues::extractSpillBaseRegAndOffset(const MachineInstr &MI) {
518 assert(MI.hasOneMemOperand() &&
519 "Spill instruction does not have exactly one memory operand?");
520 auto MMOI = MI.memoperands_begin();
521 const PseudoSourceValue *PVal = (*MMOI)->getPseudoValue();
522 assert(PVal->kind() == PseudoSourceValue::FixedStack &&
523 "Inconsistent memory operand in spill instruction");
524 int FI = cast<FixedStackPseudoSourceValue>(PVal)->getFrameIndex();
525 const MachineBasicBlock *MBB = MI.getParent();
526 unsigned Reg;
527 int Offset = TFI->getFrameIndexReference(*MBB->getParent(), FI, Reg);
528 return {Reg, Offset};
529}

531/// End all previous ranges related to @MI and start a new range from @MI
532/// if it is a DBG_VALUE instr.
533void LiveDebugValues::transferDebugValue(const MachineInstr &MI,
534 OpenRangesSet &OpenRanges,
535 VarLocMap &VarLocIDs) {
536 if (!MI.isDebugValue())
537 return;
538 const DILocalVariable *Var = MI.getDebugVariable();
539 const DIExpression *Expr = MI.getDebugExpression();
540 const DILocation *DebugLoc = MI.getDebugLoc();
541 const DILocation *InlinedAt = DebugLoc->getInlinedAt();
542 assert(Var->isValidLocationForIntrinsic(DebugLoc) &&
543 "Expected inlined-at fields to agree");
544
545 // End all previous ranges of Var.
546 DebugVariable V(Var, Expr, InlinedAt);
547 OpenRanges.erase(V);
548
549 // Add the VarLoc to OpenRanges from this DBG_VALUE.
550 unsigned ID;
551 if (isDbgValueDescribedByReg(MI) || MI.getOperand(0).isImm() ||
552 MI.getOperand(0).isFPImm() || MI.getOperand(0).isCImm()) {
553 // Use normal VarLoc constructor for registers and immediates.
554 VarLoc VL(MI, LS);
555 ID = VarLocIDs.insert(VL);
556 OpenRanges.insert(ID, VL.Var);
557 } else if (MI.hasOneMemOperand()) {
558 // It's a stack spill -- fetch spill base and offset.
559 VarLoc::SpillLoc SpillLocation = extractSpillBaseRegAndOffset(MI);
560 VarLoc VL(MI, SpillLocation.SpillBase, SpillLocation.SpillOffset, LS);
561 ID = VarLocIDs.insert(VL);
562 OpenRanges.insert(ID, VL.Var);
563 } else {
564 // This must be an undefined location. We should leave OpenRanges closed.
565 assert(MI.getOperand(0).isReg() && MI.getOperand(0).getReg() == 0 &&
566 "Unexpected non-undef DBG_VALUE encountered");
567 }
568}
569
570/// Create new TransferDebugPair and insert it in \p Transfers. The VarLoc
571/// with \p OldVarID should be deleted form \p OpenRanges and replaced with
572/// new VarLoc. If \p NewReg is different than default zero value then the
573/// new location will be register location created by the copy like instruction,
574/// otherwise it is variable's location on the stack.
575void LiveDebugValues::insertTransferDebugPair(
576 MachineInstr &MI, OpenRangesSet &OpenRanges, TransferMap &Transfers,
577 VarLocMap &VarLocIDs, unsigned OldVarID, TransferKind Kind,
578 unsigned NewReg) {
579 const MachineInstr *DebugInstr = &VarLocIDs[OldVarID].MI;
580 MachineFunction *MF = MI.getParent()->getParent();
581 MachineInstr *NewDebugInstr;
582
583 auto ProcessVarLoc = [&MI, &OpenRanges, &Transfers, &DebugInstr,
584 &VarLocIDs](VarLoc &VL, MachineInstr *NewDebugInstr) {
585 unsigned LocId = VarLocIDs.insert(VL);
586
587 // Close this variable's previous location range.
588 DebugVariable V(*DebugInstr);
589 OpenRanges.erase(V);
590
591 OpenRanges.insert(LocId, VL.Var);
592 // The newly created DBG_VALUE instruction NewDebugInstr must be inserted
593 // after MI. Keep track of the pairing.
594 TransferDebugPair MIP = {&MI, NewDebugInstr};
595 Transfers.push_back(MIP);
596 };
597
598 // End all previous ranges of Var.
599 OpenRanges.erase(VarLocIDs[OldVarID].Var);
600 switch (Kind) {
601 case TransferKind::TransferCopy: {
602 assert(NewReg &&
603 "No register supplied when handling a copy of a debug value");
604 // Create a DBG_VALUE instruction to describe the Var in its new
605 // register location.
606 NewDebugInstr = BuildMI(
607 *MF, DebugInstr->getDebugLoc(), DebugInstr->getDesc(),
608 DebugInstr->isIndirectDebugValue(), NewReg,
609 DebugInstr->getDebugVariable(), DebugInstr->getDebugExpression());
610 if (DebugInstr->isIndirectDebugValue())
611 NewDebugInstr->getOperand(1).setImm(DebugInstr->getOperand(1).getImm());
612 VarLoc VL(*NewDebugInstr, LS);
613 ProcessVarLoc(VL, NewDebugInstr);
614 LLVM_DEBUG(dbgs() << "Creating DBG_VALUE inst for register copy: ";
615 NewDebugInstr->print(dbgs(), /*IsStandalone*/false,
616 /*SkipOpers*/false, /*SkipDebugLoc*/false,
617 /*AddNewLine*/true, TII));
618 return;
619 }
620 case TransferKind::TransferSpill: {
621 // Create a DBG_VALUE instruction to describe the Var in its spilled
622 // location.
623 VarLoc::SpillLoc SpillLocation = extractSpillBaseRegAndOffset(MI);
624 auto *SpillExpr = DIExpression::prepend(DebugInstr->getDebugExpression(),
625 DIExpression::ApplyOffset,
626 SpillLocation.SpillOffset);
627 NewDebugInstr = BuildMI(
628 *MF, DebugInstr->getDebugLoc(), DebugInstr->getDesc(), true,
629 SpillLocation.SpillBase, DebugInstr->getDebugVariable(), SpillExpr);
630 VarLoc VL(*NewDebugInstr, SpillLocation.SpillBase,
631 SpillLocation.SpillOffset, LS);
632 ProcessVarLoc(VL, NewDebugInstr);
633 LLVM_DEBUG(dbgs() << "Creating DBG_VALUE inst for spill: ";
634 NewDebugInstr->print(dbgs(), /*IsStandalone*/false,
635 /*SkipOpers*/false, /*SkipDebugLoc*/false,
636 /*AddNewLine*/true, TII));
637 return;
638 }
639 case TransferKind::TransferRestore: {
640 assert(NewReg &&
641 "No register supplied when handling a restore of a debug value");
642 MachineFunction *MF = MI.getMF();
643 DIBuilder DIB(*const_cast<Function &>(MF->getFunction()).getParent());
644 NewDebugInstr =
645 BuildMI(*MF, DebugInstr->getDebugLoc(), DebugInstr->getDesc(), false,
646 NewReg, DebugInstr->getDebugVariable(), DIB.createExpression());
647 VarLoc VL(*NewDebugInstr, LS);
648 ProcessVarLoc(VL, NewDebugInstr);
649 LLVM_DEBUG(dbgs() << "Creating DBG_VALUE inst for register restore: ";
650 NewDebugInstr->print(dbgs(), /*IsStandalone*/false,
651 /*SkipOpers*/false, /*SkipDebugLoc*/false,
652 /*AddNewLine*/true, TII));
653 return;
654 }
655 }
656 llvm_unreachable("Invalid transfer kind");
657}

659/// A definition of a register may mark the end of a range.
660void LiveDebugValues::transferRegisterDef(MachineInstr &MI,
661 OpenRangesSet &OpenRanges,
662 const VarLocMap &VarLocIDs) {
663 MachineFunction *MF = MI.getMF();
664 const TargetLowering *TLI = MF->getSubtarget().getTargetLowering();
665 unsigned SP = TLI->getStackPointerRegisterToSaveRestore();
666 SparseBitVector<> KillSet;
667 for (const MachineOperand &MO : MI.operands()) {
668 // Determine whether the operand is a register def. Assume that call
669 // instructions never clobber SP, because some backends (e.g., AArch64)
670 // never list SP in the regmask.
671 if (MO.isReg() && MO.isDef() && MO.getReg() &&
672 TRI->isPhysicalRegister(MO.getReg()) &&
673 !(MI.isCall() && MO.getReg() == SP)) {
674 // Remove ranges of all aliased registers.
675 for (MCRegAliasIterator RAI(MO.getReg(), TRI, true); RAI.isValid(); ++RAI)
676 for (unsigned ID : OpenRanges.getVarLocs())
677 if (VarLocIDs[ID].isDescribedByReg() == *RAI)
678 KillSet.set(ID);
679 } else if (MO.isRegMask()) {
680 // Remove ranges of all clobbered registers. Register masks don't usually
681 // list SP as preserved. While the debug info may be off for an
682 // instruction or two around callee-cleanup calls, transferring the
683 // DEBUG_VALUE across the call is still a better user experience.
684 for (unsigned ID : OpenRanges.getVarLocs()) {
685 unsigned Reg = VarLocIDs[ID].isDescribedByReg();
686 if (Reg && Reg != SP && MO.clobbersPhysReg(Reg))
687 KillSet.set(ID);
688 }
689 }
690 }
691 OpenRanges.erase(KillSet, VarLocIDs);
692}
693
694/// Decide if @MI is a spill instruction and return true if it is. We use 2
695/// criteria to make this decision:
696/// - Is this instruction a store to a spill slot?
697/// - Is there a register operand that is both used and killed?
698/// TODO: Store optimization can fold spills into other stores (including
699/// other spills). We do not handle this yet (more than one memory operand).
700bool LiveDebugValues::isSpillInstruction(const MachineInstr &MI,
701 MachineFunction *MF, unsigned &Reg) {
702 SmallVector<const MachineMemOperand*, 1> Accesses;
703
704 // TODO: Handle multiple stores folded into one.
705 if (!MI.hasOneMemOperand())
706 return false;
707
708 if (!MI.getSpillSize(TII) && !MI.getFoldedSpillSize(TII))
709 return false; // This is not a spill instruction, since no valid size was
710 // returned from either function.
711
712 auto isKilledReg = [&](const MachineOperand MO, unsigned &Reg) {
713 if (!MO.isReg() || !MO.isUse()) {
714 Reg = 0;
715 return false;
716 }
717 Reg = MO.getReg();
718 return MO.isKill();
719 };
720
721 for (const MachineOperand &MO : MI.operands()) {
722 // In a spill instruction generated by the InlineSpiller the spilled
723 // register has its kill flag set.
724 if (isKilledReg(MO, Reg))
725 return true;
726 if (Reg != 0) {
727 // Check whether next instruction kills the spilled register.
728 // FIXME: Current solution does not cover search for killed register in
729 // bundles and instructions further down the chain.
730 auto NextI = std::next(MI.getIterator());
731 // Skip next instruction that points to basic block end iterator.
732 if (MI.getParent()->end() == NextI)
733 continue;
734 unsigned RegNext;
735 for (const MachineOperand &MONext : NextI->operands()) {
736 // Return true if we came across the register from the
737 // previous spill instruction that is killed in NextI.
738 if (isKilledReg(MONext, RegNext) && RegNext == Reg)
739 return true;
740 }
741 }
742 }
743 // Return false if we didn't find spilled register.
744 return false;
745}

747Optional<LiveDebugValues::VarLoc::SpillLoc>
748LiveDebugValues::isRestoreInstruction(const MachineInstr &MI,
749 MachineFunction *MF, unsigned &Reg) {
750 if (!MI.hasOneMemOperand())
751 return None;
752
753 // FIXME: Handle folded restore instructions with more than one memory
754 // operand.
755 if (MI.getRestoreSize(TII)) {
756 Reg = MI.getOperand(0).getReg();
757 return extractSpillBaseRegAndOffset(MI);
758 }
759 return None;
760}
761
762/// A spilled register may indicate that we have to end the current range of
763/// a variable and create a new one for the spill location.
764/// A restored register may indicate the reverse situation.
765/// We don't want to insert any instructions in process(), so we just create
766/// the DBG_VALUE without inserting it and keep track of it in \p Transfers.
767/// It will be inserted into the BB when we're done iterating over the
768/// instructions.
769void LiveDebugValues::transferSpillOrRestoreInst(MachineInstr &MI,
770 OpenRangesSet &OpenRanges,
771 VarLocMap &VarLocIDs,
772 TransferMap &Transfers) {
773 MachineFunction *MF = MI.getMF();
774 TransferKind TKind;
775 unsigned Reg;
776 Optional<VarLoc::SpillLoc> Loc;
777
778 LLVM_DEBUG(dbgs() << "Examining instruction: "; MI.dump(););
779
780 if (isSpillInstruction(MI, MF, Reg)) {
781 TKind = TransferKind::TransferSpill;
782 LLVM_DEBUG(dbgs() << "Recognized as spill: "; MI.dump(););
783 LLVM_DEBUG(dbgs() << "Register: " << Reg << " " << printReg(Reg, TRI)
784 << "\n");
785 } else {
786 if (!(Loc = isRestoreInstruction(MI, MF, Reg)))
787 return;
788 TKind = TransferKind::TransferRestore;
789 LLVM_DEBUG(dbgs() << "Recognized as restore: "; MI.dump(););
790 LLVM_DEBUG(dbgs() << "Register: " << Reg << " " << printReg(Reg, TRI)
791 << "\n");
792 }
793 // Check if the register or spill location is the location of a debug value.
794 for (unsigned ID : OpenRanges.getVarLocs()) {
795 if (TKind == TransferKind::TransferSpill &&
796 VarLocIDs[ID].isDescribedByReg() == Reg) {
797 LLVM_DEBUG(dbgs() << "Spilling Register " << printReg(Reg, TRI) << '('
798 << VarLocIDs[ID].Var.getVar()->getName() << ")\n");
799 } else if (TKind == TransferKind::TransferRestore &&
800 VarLocIDs[ID].Loc.SpillLocation == *Loc) {
801 LLVM_DEBUG(dbgs() << "Restoring Register " << printReg(Reg, TRI) << '('
802 << VarLocIDs[ID].Var.getVar()->getName() << ")\n");
803 } else
804 continue;
805 insertTransferDebugPair(MI, OpenRanges, Transfers, VarLocIDs, ID, TKind,
806 Reg);
807 return;
808 }
809}

811/// If \p MI is a register copy instruction, that copies a previously tracked
812/// value from one register to another register that is callee saved, we
813/// create new DBG_VALUE instruction described with copy destination register.
814void LiveDebugValues::transferRegisterCopy(MachineInstr &MI,
815 OpenRangesSet &OpenRanges,
816 VarLocMap &VarLocIDs,
817 TransferMap &Transfers) {
818 const MachineOperand *SrcRegOp, *DestRegOp;
819
820 if (!TII->isCopyInstr(MI, SrcRegOp, DestRegOp) || !SrcRegOp->isKill() ||
821 !DestRegOp->isDef())
822 return;
823
824 auto isCalleSavedReg = [&](unsigned Reg) {
825 for (MCRegAliasIterator RAI(Reg, TRI, true); RAI.isValid(); ++RAI)
826 if (CalleeSavedRegs.test(*RAI))
827 return true;
828 return false;
829 };
830
831 unsigned SrcReg = SrcRegOp->getReg();
832 unsigned DestReg = DestRegOp->getReg();
833
834 // We want to recognize instructions where destination register is callee
835 // saved register. If register that could be clobbered by the call is
836 // included, there would be a great chance that it is going to be clobbered
837 // soon. It is more likely that previous register location, which is callee
838 // saved, is going to stay unclobbered longer, even if it is killed.
839 if (!isCalleSavedReg(DestReg))
840 return;
841
842 for (unsigned ID : OpenRanges.getVarLocs()) {
843 if (VarLocIDs[ID].isDescribedByReg() == SrcReg) {
844 insertTransferDebugPair(MI, OpenRanges, Transfers, VarLocIDs, ID,
845 TransferKind::TransferCopy, DestReg);
846 return;
847 }
848 }
849}

851/// Terminate all open ranges at the end of the current basic block.
852bool LiveDebugValues::transferTerminatorInst(MachineInstr &MI,
853 OpenRangesSet &OpenRanges,
854 VarLocInMBB &OutLocs,
855 const VarLocMap &VarLocIDs) {
856 bool Changed = false;
857 const MachineBasicBlock *CurMBB = MI.getParent();
858 if (!(MI.isTerminator() || (&MI == &CurMBB->back())))
859 return false;
860
861 if (OpenRanges.empty())
862 return false;
863
864 LLVM_DEBUG(for (unsigned ID
865 : OpenRanges.getVarLocs()) {
866 // Copy OpenRanges to OutLocs, if not already present.
867 dbgs() << "Add to OutLocs in MBB #" << CurMBB->getNumber() << ": ";
868 VarLocIDs[ID].dump();
869 });
870 VarLocSet &VLS = OutLocs[CurMBB];
871 Changed = VLS |= OpenRanges.getVarLocs();
872 // New OutLocs set may be different due to spill, restore or register
873 // copy instruction processing.
874 if (Changed)
875 VLS = OpenRanges.getVarLocs();
876 OpenRanges.clear();
877 return Changed;
878}

880/// Accumulate a mapping between each DILocalVariable fragment and other
881/// fragments of that DILocalVariable which overlap. This reduces work during
882/// the data-flow stage from "Find any overlapping fragments" to "Check if the
883/// known-to-overlap fragments are present".
884/// \param MI A previously unprocessed DEBUG_VALUE instruction to analyze for
885/// fragment usage.
886/// \param SeenFragments Map from DILocalVariable to all fragments of that
887/// Variable which are known to exist.
888/// \param OverlappingFragments The overlap map being constructed, from one
889/// Var/Fragment pair to a vector of fragments known to overlap.
890void LiveDebugValues::accumulateFragmentMap(MachineInstr &MI,
891 VarToFragments &SeenFragments,
892 OverlapMap &OverlappingFragments) {
893 DebugVariable MIVar(MI);
894 FragmentInfo ThisFragment = MIVar.getFragmentDefault();
895
896 // If this is the first sighting of this variable, then we are guaranteed
897 // there are currently no overlapping fragments either. Initialize the set
898 // of seen fragments, record no overlaps for the current one, and return.
899 auto SeenIt = SeenFragments.find(MIVar.getVar());
900 if (SeenIt == SeenFragments.end()) {
901 SmallSet<FragmentInfo, 4> OneFragment;
902 OneFragment.insert(ThisFragment);
903 SeenFragments.insert({MIVar.getVar(), OneFragment});
904
905 OverlappingFragments.insert({{MIVar.getVar(), ThisFragment}, {}});
906 return;
907 }
908
909 // If this particular Variable/Fragment pair already exists in the overlap
910 // map, it has already been accounted for.
911 auto IsInOLapMap =
912 OverlappingFragments.insert({{MIVar.getVar(), ThisFragment}, {}});
913 if (!IsInOLapMap.second)
914 return;
915
916 auto &ThisFragmentsOverlaps = IsInOLapMap.first->second;
917 auto &AllSeenFragments = SeenIt->second;
918
919 // Otherwise, examine all other seen fragments for this variable, with "this"
920 // fragment being a previously unseen fragment. Record any pair of
921 // overlapping fragments.
922 for (auto &ASeenFragment : AllSeenFragments) {
923 // Does this previously seen fragment overlap?
924 if (DIExpression::fragmentsOverlap(ThisFragment, ASeenFragment)) {
925 // Yes: Mark the current fragment as being overlapped.
926 ThisFragmentsOverlaps.push_back(ASeenFragment);
927 // Mark the previously seen fragment as being overlapped by the current
928 // one.
929 auto ASeenFragmentsOverlaps =
930 OverlappingFragments.find({MIVar.getVar(), ASeenFragment});
931 assert(ASeenFragmentsOverlaps != OverlappingFragments.end() &&
932 "Previously seen var fragment has no vector of overlaps");
933 ASeenFragmentsOverlaps->second.push_back(ThisFragment);
934 }
935 }
936
937 AllSeenFragments.insert(ThisFragment);
938}

940/// This routine creates OpenRanges and OutLocs.
941bool LiveDebugValues::process(MachineInstr &MI, OpenRangesSet &OpenRanges,
942 VarLocInMBB &OutLocs, VarLocMap &VarLocIDs,
943 TransferMap &Transfers, bool transferChanges) {
944 bool Changed = false;
945 transferDebugValue(MI, OpenRanges, VarLocIDs);
946 transferRegisterDef(MI, OpenRanges, VarLocIDs);
947 if (transferChanges) {
948 transferRegisterCopy(MI, OpenRanges, VarLocIDs, Transfers);
949 transferSpillOrRestoreInst(MI, OpenRanges, VarLocIDs, Transfers);
950 }
951 Changed = transferTerminatorInst(MI, OpenRanges, OutLocs, VarLocIDs);
952 return Changed;
953}

955/// This routine joins the analysis results of all incoming edges in @MBB by
956/// inserting a new DBG_VALUE instruction at the start of the @MBB - if the same
957/// source variable in all the predecessors of @MBB reside in the same location.
958bool LiveDebugValues::join(
959 MachineBasicBlock &MBB, VarLocInMBB &OutLocs, VarLocInMBB &InLocs,
960 const VarLocMap &VarLocIDs,
961 SmallPtrSet<const MachineBasicBlock *, 16> &Visited,
962 SmallPtrSetImpl<const MachineBasicBlock *> &ArtificialBlocks) {
963 LLVM_DEBUG(dbgs() << "join MBB: " << MBB.getNumber() << "\n");
964 bool Changed = false;
965
966 VarLocSet InLocsT; // Temporary incoming locations.
967
968 // For all predecessors of this MBB, find the set of VarLocs that
969 // can be joined.
970 int NumVisited = 0;
971 for (auto p : MBB.predecessors()) {
972 // Ignore unvisited predecessor blocks. As we are processing
973 // the blocks in reverse post-order any unvisited block can
974 // be considered to not remove any incoming values.
975 if (!Visited.count(p)) {
976 LLVM_DEBUG(dbgs() << " ignoring unvisited pred MBB: " << p->getNumber()
977 << "\n");
978 continue;
979 }
980 auto OL = OutLocs.find(p);
981 // Join is null in case of empty OutLocs from any of the pred.
982 if (OL == OutLocs.end())
983 return false;
984
985 // Just copy over the Out locs to incoming locs for the first visited
986 // predecessor, and for all other predecessors join the Out locs.
987 if (!NumVisited)
988 InLocsT = OL->second;
989 else
990 InLocsT &= OL->second;
991
992 LLVM_DEBUG({
993 if (!InLocsT.empty()) {
994 for (auto ID : InLocsT)
995 dbgs() << " gathered candidate incoming var: "
996 << VarLocIDs[ID].Var.getVar()->getName() << "\n";
997 }
998 });
999
1000 NumVisited++;
1001 }
1002
1003 // Filter out DBG_VALUES that are out of scope.
1004 VarLocSet KillSet;
1005 bool IsArtificial = ArtificialBlocks.count(&MBB);
1006 if (!IsArtificial) {
1007 for (auto ID : InLocsT) {
1008 if (!VarLocIDs[ID].dominates(MBB)) {
1009 KillSet.set(ID);
1010 LLVM_DEBUG({
1011 auto Name = VarLocIDs[ID].Var.getVar()->getName();
1012 dbgs() << " killing " << Name << ", it doesn't dominate MBB\n";
1013 });
1014 }
1015 }
1016 }
1017 InLocsT.intersectWithComplement(KillSet);
1018
1019 // As we are processing blocks in reverse post-order we
1020 // should have processed at least one predecessor, unless it
1021 // is the entry block which has no predecessor.
1022 assert((NumVisited || MBB.pred_empty()) &&
1023 "Should have processed at least one predecessor");
1024 if (InLocsT.empty())
1025 return false;
1026
1027 VarLocSet &ILS = InLocs[&MBB];
1028
1029 // Insert DBG_VALUE instructions, if not already inserted.
1030 VarLocSet Diff = InLocsT;
1031 Diff.intersectWithComplement(ILS);
1032 for (auto ID : Diff) {
1033 // This VarLoc is not found in InLocs i.e. it is not yet inserted. So, a
1034 // new range is started for the var from the mbb's beginning by inserting
1035 // a new DBG_VALUE. process() will end this range however appropriate.
1036 const VarLoc &DiffIt = VarLocIDs[ID];
1037 const MachineInstr *DebugInstr = &DiffIt.MI;
1038 MachineInstr *MI = nullptr;
1039 if (DiffIt.isConstant()) {
1040 MachineOperand MO(DebugInstr->getOperand(0));
1041 MI = BuildMI(MBB, MBB.instr_begin(), DebugInstr->getDebugLoc(),
1042 DebugInstr->getDesc(), false, MO,
1043 DebugInstr->getDebugVariable(),
1044 DebugInstr->getDebugExpression());
1045 } else {
1046 MI = BuildMI(MBB, MBB.instr_begin(), DebugInstr->getDebugLoc(),
1047 DebugInstr->getDesc(), DebugInstr->isIndirectDebugValue(),
1048 DebugInstr->getOperand(0).getReg(),
1049 DebugInstr->getDebugVariable(),
1050 DebugInstr->getDebugExpression());
1051 if (DebugInstr->isIndirectDebugValue())
1052 MI->getOperand(1).setImm(DebugInstr->getOperand(1).getImm());
1053 }
1054 LLVM_DEBUG(dbgs() << "Inserted: "; MI->dump(););
1055 ILS.set(ID);
1056 ++NumInserted;
1057 Changed = true;
1058 }
1059 return Changed;
1060}

1062/// Calculate the liveness information for the given machine function and
1063/// extend ranges across basic blocks.
1064bool LiveDebugValues::ExtendRanges(MachineFunction &MF) {
1065 LLVM_DEBUG(dbgs() << "\nDebug Range Extension\n");
1066
1067 bool Changed = false;
1068 bool OLChanged = false;
1069 bool MBBJoined = false;
1070
1071 VarLocMap VarLocIDs; // Map VarLoc<>unique ID for use in bitvectors.
1072 OverlapMap OverlapFragments; // Map of overlapping variable fragments
1073 OpenRangesSet OpenRanges(OverlapFragments);
1074 // Ranges that are open until end of bb.
1075 VarLocInMBB OutLocs; // Ranges that exist beyond bb.
1076 VarLocInMBB InLocs; // Ranges that are incoming after joining.
1077 TransferMap Transfers; // DBG_VALUEs associated with spills.
1078
1079 VarToFragments SeenFragments;
1080
1081 // Blocks which are artificial, i.e. blocks which exclusively contain
1082 // instructions without locations, or with line 0 locations.
1083 SmallPtrSet<const MachineBasicBlock *, 16> ArtificialBlocks;
1084
1085 DenseMap<unsigned int, MachineBasicBlock *> OrderToBB;
1086 DenseMap<MachineBasicBlock *, unsigned int> BBToOrder;
1087 std::priority_queue<unsigned int, std::vector<unsigned int>,
1088 std::greater<unsigned int>>
1089 Worklist;
1090 std::priority_queue<unsigned int, std::vector<unsigned int>,
1091 std::greater<unsigned int>>
1092 Pending;
1093
1094 enum : bool { dontTransferChanges = false, transferChanges = true };
1095
1096 // Initialize every mbb with OutLocs.
1097 // We are not looking at any spill instructions during the initial pass
1098 // over the BBs. The LiveDebugVariables pass has already created DBG_VALUE
1099 // instructions for spills of registers that are known to be user variables
1100 // within the BB in which the spill occurs.
1101 for (auto &MBB : MF) {
1102 for (auto &MI : MBB) {
1103 process(MI, OpenRanges, OutLocs, VarLocIDs, Transfers,
1104 dontTransferChanges);
1105 if (MI.isDebugValue())
1106 accumulateFragmentMap(MI, SeenFragments, OverlapFragments);
1107 }
1108 }
1109
1110 auto hasNonArtificialLocation = [](const MachineInstr &MI) -> bool {
1111 if (const DebugLoc &DL = MI.getDebugLoc())
1112 return DL.getLine() != 0;
1113 return false;
1114 };
1115 for (auto &MBB : MF)
1116 if (none_of(MBB.instrs(), hasNonArtificialLocation))
1117 ArtificialBlocks.insert(&MBB);
1118
1119 LLVM_DEBUG(printVarLocInMBB(MF, OutLocs, VarLocIDs,
1120 "OutLocs after initialization", dbgs()));
1121
1122 ReversePostOrderTraversal<MachineFunction *> RPOT(&MF);
1123 unsigned int RPONumber = 0;
1124 for (auto RI = RPOT.begin(), RE = RPOT.end(); RI != RE; ++RI) {
1125 OrderToBB[RPONumber] = *RI;
1126 BBToOrder[*RI] = RPONumber;
1127 Worklist.push(RPONumber);
1128 ++RPONumber;
1129 }
1130 // This is a standard "union of predecessor outs" dataflow problem.
1131 // To solve it, we perform join() and process() using the two worklist method
1132 // until the ranges converge.
1133 // Ranges have converged when both worklists are empty.
1134 SmallPtrSet<const MachineBasicBlock *, 16> Visited;
1135 while (!Worklist.empty() || !Pending.empty()) {
1136 // We track what is on the pending worklist to avoid inserting the same
1137 // thing twice. We could avoid this with a custom priority queue, but this
1138 // is probably not worth it.
1139 SmallPtrSet<MachineBasicBlock *, 16> OnPending;
1140 LLVM_DEBUG(dbgs() << "Processing Worklist\n");
1141 while (!Worklist.empty()) {
1142 MachineBasicBlock *MBB = OrderToBB[Worklist.top()];
1143 Worklist.pop();
1144 MBBJoined =
1145 join(*MBB, OutLocs, InLocs, VarLocIDs, Visited, ArtificialBlocks);
1146 Visited.insert(MBB);
1147 if (MBBJoined) {
1148 MBBJoined = false;
1149 Changed = true;
1150 // Now that we have started to extend ranges across BBs we need to
1151 // examine spill instructions to see whether they spill registers that
1152 // correspond to user variables.
1153 for (auto &MI : *MBB)
1154 OLChanged |= process(MI, OpenRanges, OutLocs, VarLocIDs, Transfers,
1155 transferChanges);
1156
1157 // Add any DBG_VALUE instructions necessitated by spills.
1158 for (auto &TR : Transfers)
1159 MBB->insertAfter(MachineBasicBlock::iterator(*TR.TransferInst),
1160 TR.DebugInst);
1161 Transfers.clear();
1162
1163 LLVM_DEBUG(printVarLocInMBB(MF, OutLocs, VarLocIDs,
1164 "OutLocs after propagating", dbgs()));
1165 LLVM_DEBUG(printVarLocInMBB(MF, InLocs, VarLocIDs,
1166 "InLocs after propagating", dbgs()));
1167
1168 if (OLChanged) {
1169 OLChanged = false;
1170 for (auto s : MBB->successors())
1171 if (OnPending.insert(s).second) {
1172 Pending.push(BBToOrder[s]);
1173 }
1174 }
1175 }
1176 }
1177 Worklist.swap(Pending);
1178 // At this point, pending must be empty, since it was just the empty
1179 // worklist
1180 assert(Pending.empty() && "Pending should be empty");
1181 }
1182
1183 LLVM_DEBUG(printVarLocInMBB(MF, OutLocs, VarLocIDs, "Final OutLocs", dbgs()));
1184 LLVM_DEBUG(printVarLocInMBB(MF, InLocs, VarLocIDs, "Final InLocs", dbgs()));
1185 return Changed;
1186}
1187
1188bool LiveDebugValues::runOnMachineFunction(MachineFunction &MF) {
1189 if (!MF.getFunction().getSubprogram())
1190 // LiveDebugValues will already have removed all DBG_VALUEs.
1191 return false;
1192
1193 // Skip functions from NoDebug compilation units.
1194 if (MF.getFunction().getSubprogram()->getUnit()->getEmissionKind() ==
1195 DICompileUnit::NoDebug)
1196 return false;
1197
1198 TRI = MF.getSubtarget().getRegisterInfo();
1199 TII = MF.getSubtarget().getInstrInfo();
1200 TFI = MF.getSubtarget().getFrameLowering();
1201 TFI->determineCalleeSaves(MF, CalleeSavedRegs,
1202 make_unique<RegScavenger>().get());
1203 LS.initialize(MF);
1204
1205 bool Changed = ExtendRanges(MF);
1206 return Changed;
1207}
