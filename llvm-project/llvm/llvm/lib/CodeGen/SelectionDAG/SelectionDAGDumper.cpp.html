461// Print the MMO with more information from the SelectionDAG.
462static void printMemOperand(raw_ostream &OS, const MachineMemOperand &MMO,
463 const MachineFunction *MF, const Module *M,
464 const MachineFrameInfo *MFI,
465 const TargetInstrInfo *TII, LLVMContext &Ctx) {
466 ModuleSlotTracker MST(M);
467 if (MF)
468 MST.incorporateFunction(MF->getFunction());
469 SmallVector<StringRef, 0> SSNs;
470 MMO.print(OS, MST, SSNs, Ctx, MFI, TII);
471}
472
473static void printMemOperand(raw_ostream &OS, const MachineMemOperand &MMO,
474 const SelectionDAG *G) {
475 if (G) {
476 const MachineFunction *MF = &G->getMachineFunction();
477 return printMemOperand(OS, MMO, MF, MF->getFunction().getParent(),
478 &MF->getFrameInfo(), G->getSubtarget().getInstrInfo(),
479 *G->getContext());
480 } else {
481 LLVMContext Ctx;
482 return printMemOperand(OS, MMO, /*MF=*/nullptr, /*M=*/nullptr,
483 /*MFI=*/nullptr, /*TII=*/nullptr, Ctx);
484 }
485}

496void SDNode::print_types(raw_ostream &OS, const SelectionDAG *G) const {
497 for (unsigned i = 0, e = getNumValues(); i != e; ++i) {
498 if (i) OS << ",";
499 if (getValueType(i) == MVT::Other)
500 OS << "ch";
501 else
502 OS << getValueType(i).getEVTString();
503 }

786/// Return true if this node is so simple that we should just print it inline
787/// if it appears as an operand.
788static bool shouldPrintInline(const SDNode &Node, const SelectionDAG *G) {
789 // Avoid lots of cluttering when inline printing nodes with associated
790 // DbgValues in verbose mode.
791 if (VerboseDAGDumping && G && !G->GetDbgValues(&Node).empty())
792 return false;
793 if (Node.getOpcode() == ISD::EntryToken)
794 return false;
795 return Node.getNumOperands() == 0;
796}

799static void DumpNodes(const SDNode *N, unsigned indent, const SelectionDAG *G) {
800 for (const SDValue &Op : N->op_values()) {
801 if (shouldPrintInline(*Op.getNode(), G))
802 continue;
803 if (Op.getNode()->hasOneUse())
804 DumpNodes(Op.getNode(), indent+2, G);
805 }
806
807 dbgs().indent(indent);
808 N->dump(G);
809}

811LLVM_DUMP_METHOD void SelectionDAG::dump() const {
812 dbgs() << "SelectionDAG has " << AllNodes.size() << " nodes:\n";
813
814 for (allnodes_const_iterator I = allnodes_begin(), E = allnodes_end();
815 I != E; ++I) {
816 const SDNode *N = &*I;
817 if (!N->hasOneUse() && N != getRoot().getNode() &&
818 (!shouldPrintInline(*N, this) || N->use_empty()))
819 DumpNodes(N, 2, this);
820 }
821
822 if (getRoot().getNode()) DumpNodes(getRoot().getNode(), 2, this);
823 dbgs() << "\n";
824
825 if (VerboseDAGDumping) {
826 if (DbgBegin() != DbgEnd())
827 dbgs() << "SDDbgValues:\n";
828 for (auto *Dbg : make_range(DbgBegin(), DbgEnd()))
829 Dbg->dump();
830 if (ByvalParmDbgBegin() != ByvalParmDbgEnd())
831 dbgs() << "Byval SDDbgValues:\n";
832 for (auto *Dbg : make_range(ByvalParmDbgBegin(), ByvalParmDbgEnd()))
833 Dbg->dump();
834 }
835 dbgs() << "\n";
836}

846static bool printOperand(raw_ostream &OS, const SelectionDAG *G,
847 const SDValue Value) {
848 if (!Value.getNode()) {
849 OS << "<null>";
850 return false;
851 } else if (shouldPrintInline(*Value.getNode(), G)) {
852 OS << Value->getOperationName(G) << ':';
853 Value->print_types(OS, G);
854 Value->print_details(OS, G);
855 return true;
856 } else {
857 OS << PrintNodeId(*Value.getNode());
858 if (unsigned RN = Value.getResNo())
859 OS << ':' << RN;
860 return false;
861 }
862}

864#if !defined(NDEBUG) || defined(LLVM_ENABLE_DUMP)
865using VisitedSDNodeSet = SmallPtrSet<const SDNode *, 32>;
866
867static void DumpNodesr(raw_ostream &OS, const SDNode *N, unsigned indent,
868 const SelectionDAG *G, VisitedSDNodeSet &once) {
869 if (!once.insert(N).second) // If we've been here before, return now.
870 return;
871
872 // Dump the current SDNode, but don't end the line yet.
873 OS.indent(indent);
874 N->printr(OS, G);
875
876 // Having printed this SDNode, walk the children:
877 for (unsigned i = 0, e = N->getNumOperands(); i != e; ++i) {
878 if (i) OS << ",";
879 OS << " ";
880
881 const SDValue Op = N->getOperand(i);
882 bool printedInline = printOperand(OS, G, Op);
883 if (printedInline)
884 once.insert(Op.getNode());
885 }
886
887 OS << "\n";
888
889 // Dump children that have grandchildren on their own line(s).
890 for (const SDValue &Op : N->op_values())
891 DumpNodesr(OS, Op.getNode(), indent+2, G, once);
892}

905static void printrWithDepthHelper(raw_ostream &OS, const SDNode *N,
906 const SelectionDAG *G, unsigned depth,
907 unsigned indent) {
908 if (depth == 0)
909 return;
910
911 OS.indent(indent);
912
913 N->print(OS, G);
914
915 if (depth < 1)
916 return;
917
918 for (const SDValue &Op : N->op_values()) {
919 // Don't follow chain operands.
920 if (Op.getValueType() == MVT::Other)
921 continue;
922 OS << '\n';
923 printrWithDepthHelper(OS, Op.getNode(), G, depth-1, indent+2);
924 }
925}
