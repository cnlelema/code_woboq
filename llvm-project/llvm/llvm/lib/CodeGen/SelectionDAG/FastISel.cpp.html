9// This file contains the implementation of the FastISel class.
10//
11// "Fast" instruction selection is designed to emit very poor code quickly.
12// Also, it is not designed to be able to do much lowering, so most illegal
13// types (e.g. i64 on 32-bit targets) and operations are not supported. It is
14// also not intended to be able to do much optimization, except in a few cases
15// where doing optimizations reduces overall compile time. For example, folding
16// constants into immediate fields is often done, because it's cheap and it
17// reduces the number of instructions later phases have to examine.
18//
19// "Fast" instruction selection is able to fail gracefully and transfer
20// control to the SelectionDAG selector for operations that it doesn't
21// support. In many cases, this allows us to avoid duplicating a lot of
22// the complicated lowering logic that SelectionDAG currently has.
23//
24// The intended use for "fast" instruction selection is "-O0" mode
25// compilation, where the quality of the generated code is irrelevant when
26// weighed against the speed at which the code can be generated. Also,
27// at -O0, the LLVM optimizers are not running, and this makes the
28// compile time of codegen a much higher portion of the overall compile
29// time. Despite its limitations, "fast" instruction selection is able to
30// handle enough code on its own to provide noticeable overall speedups
31// in -O0 compiles.
32//
33// Basic operations are supported in a target-independent way, by reading
34// the same instruction descriptions that the SelectionDAG selector reads,
35// and identifying simple arithmetic operations that can be directly selected
36// from simple operators. More complicated operations currently require
37// target-specific code.
38//

131 assert(LocalValueMap.empty() &&
132 "local values should be cleared after finishing a BB");

135 // contains labels or copies, use the last instruction as the last local

146bool FastISel::lowerArguments() {
147 if (!FuncInfo.CanLowerReturn)
148 // Fallback to SDISel argument lowering code to deal with sret pointer
149 // parameter.
150 return false;
151
152 if (!fastLowerArguments())
153 return false;
154
155 // Enter arguments into ValueMap for uses in non-entry BBs.
156 for (Function::const_arg_iterator I = FuncInfo.Fn->arg_begin(),
157 E = FuncInfo.Fn->arg_end();
158 I != E; ++I) {
159 DenseMap<const Value *, unsigned>::iterator VI = LocalValueMap.find(&*I);
160 assert(VI != LocalValueMap.end() && "Missed an argument?");
161 FuncInfo.ValueMap[&*I] = VI->second;
162 }
163 return true;
164}

166/// Return the defined register if this instruction defines exactly one
167/// virtual register and uses no other virtual registers. Otherwise return 0.
168static unsigned findSinkableLocalRegDef(MachineInstr &MI) {
169 unsigned RegDef = 0;
170 for (const MachineOperand &MO : MI.operands()) {
171 if (!MO.isReg())
172 continue;
173 if (MO.isDef()) {
174 if (RegDef)
175 return 0;
176 RegDef = MO.getReg();
177 } else if (TargetRegisterInfo::isVirtualRegister(MO.getReg())) {
178 // This is another use of a vreg. Don't try to sink it.
179 return 0;
180 }
181 }
182 return RegDef;
183}

185void FastISel::flushLocalValueMap() {
186 // Try to sink local values down to their first use so that we can give them a
187 // better debug location. This has the side effect of shrinking local value
188 // live ranges, which helps out fast regalloc.
189 if (SinkLocalValues && LastLocalValue != EmitStartPt) {
190 // Sink local value materialization instructions between EmitStartPt and
191 // LastLocalValue. Visit them bottom-up, starting from LastLocalValue, to
192 // avoid inserting into the range that we're iterating over.
193 MachineBasicBlock::reverse_iterator RE =
194 EmitStartPt ? MachineBasicBlock::reverse_iterator(EmitStartPt)
195 : FuncInfo.MBB->rend();
196 MachineBasicBlock::reverse_iterator RI(LastLocalValue);
197
198 InstOrderMap OrderMap;
199 for (; RI != RE;) {
200 MachineInstr &LocalMI = *RI;
201 ++RI;
202 bool Store = true;
203 if (!LocalMI.isSafeToMove(nullptr, Store))
204 continue;
205 unsigned DefReg = findSinkableLocalRegDef(LocalMI);
206 if (DefReg == 0)
207 continue;
208
209 sinkLocalValueMaterialization(LocalMI, DefReg, OrderMap);
210 }
211 }
212
213 LocalValueMap.clear();
214 LastLocalValue = EmitStartPt;
215 recomputeInsertPt();
216 SavedInsertPt = FuncInfo.InsertPt;
217 LastFlushPoint = FuncInfo.InsertPt;
218}

220static bool isRegUsedByPhiNodes(unsigned DefReg,
221 FunctionLoweringInfo &FuncInfo) {
222 for (auto &P : FuncInfo.PHINodesToUpdate)
223 if (P.second == DefReg)
224 return true;
225 return false;

228/// Build a map of instruction orders. Return the first terminator and its
229/// order. Consider EH_LABEL instructions to be terminators as well, since local
230/// values for phis after invokes must be materialized before the call.
231void FastISel::InstOrderMap::initialize(
232 MachineBasicBlock *MBB, MachineBasicBlock::iterator LastFlushPoint) {
233 unsigned Order = 0;
234 for (MachineInstr &I : *MBB) {
235 if (!FirstTerminator &&
236 (I.isTerminator() || (I.isEHLabel() && &I != &MBB->front()))) {
237 FirstTerminator = &I;
238 FirstTerminatorOrder = Order;
239 }
240 Orders[&I] = Order++;
241
242 // We don't need to order instructions past the last flush point.
243 if (I.getIterator() == LastFlushPoint)
244 break;
245 }
246}

248void FastISel::sinkLocalValueMaterialization(MachineInstr &LocalMI,
249 unsigned DefReg,
250 InstOrderMap &OrderMap) {
251 // If this register is used by a register fixup, MRI will not contain all
252 // the uses until after register fixups, so don't attempt to sink or DCE
253 // this instruction. Register fixups typically come from no-op cast
254 // instructions, which replace the cast instruction vreg with the local
255 // value vreg.
256 if (FuncInfo.RegsWithFixups.count(DefReg))
257 return;
258
259 // We can DCE this instruction if there are no uses and it wasn't a
260 // materialized for a successor PHI node.
261 bool UsedByPHI = isRegUsedByPhiNodes(DefReg, FuncInfo);
262 if (!UsedByPHI && MRI.use_nodbg_empty(DefReg)) {
263 if (EmitStartPt == &LocalMI)
264 EmitStartPt = EmitStartPt->getPrevNode();
265 LLVM_DEBUG(dbgs() << "removing dead local value materialization "
266 << LocalMI);
267 OrderMap.Orders.erase(&LocalMI);
268 LocalMI.eraseFromParent();
269 return;
270 }
271
272 // Number the instructions if we haven't yet so we can efficiently find the
273 // earliest use.
274 if (OrderMap.Orders.empty())
275 OrderMap.initialize(FuncInfo.MBB, LastFlushPoint);
276
277 // Find the first user in the BB.
278 MachineInstr *FirstUser = nullptr;
279 unsigned FirstOrder = std::numeric_limits<unsigned>::max();
280 for (MachineInstr &UseInst : MRI.use_nodbg_instructions(DefReg)) {
281 auto I = OrderMap.Orders.find(&UseInst);
282 assert(I != OrderMap.Orders.end() &&
283 "local value used by instruction outside local region");
284 unsigned UseOrder = I->second;
285 if (UseOrder < FirstOrder) {
286 FirstOrder = UseOrder;
287 FirstUser = &UseInst;
288 }
289 }
290
291 // The insertion point will be the first terminator or the first user,
292 // whichever came first. If there was no terminator, this must be a
293 // fallthrough block and the insertion point is the end of the block.
294 MachineBasicBlock::instr_iterator SinkPos;
295 if (UsedByPHI && OrderMap.FirstTerminatorOrder < FirstOrder) {
296 FirstOrder = OrderMap.FirstTerminatorOrder;
297 SinkPos = OrderMap.FirstTerminator->getIterator();
298 } else if (FirstUser) {
299 SinkPos = FirstUser->getIterator();
300 } else {
301 assert(UsedByPHI && "must be users if not used by a phi");
302 SinkPos = FuncInfo.MBB->instr_end();
303 }
304
305 // Collect all DBG_VALUEs before the new insertion position so that we can
306 // sink them.
307 SmallVector<MachineInstr *, 1> DbgValues;
308 for (MachineInstr &DbgVal : MRI.use_instructions(DefReg)) {
309 if (!DbgVal.isDebugValue())
310 continue;
311 unsigned UseOrder = OrderMap.Orders[&DbgVal];
312 if (UseOrder < FirstOrder)
313 DbgValues.push_back(&DbgVal);
314 }
315
316 // Sink LocalMI before SinkPos and assign it the same DebugLoc.
317 LLVM_DEBUG(dbgs() << "sinking local value to first use " << LocalMI);
318 FuncInfo.MBB->remove(&LocalMI);
319 FuncInfo.MBB->insert(SinkPos, &LocalMI);
320 if (SinkPos != FuncInfo.MBB->end())
321 LocalMI.setDebugLoc(SinkPos->getDebugLoc());
322
323 // Sink any debug values that we've collected.
324 for (MachineInstr *DI : DbgValues) {
325 FuncInfo.MBB->remove(DI);
326 FuncInfo.MBB->insert(SinkPos, DI);
327 }
328}

330bool FastISel::hasTrivialKill(const Value *V) {
331 // Don't consider constants or arguments to have trivial kills.
332 const Instruction *I = dyn_cast<Instruction>(V);
333 if (!I)
334 return false;
335
336 // No-op casts are trivially coalesced by fast-isel.
337 if (const auto *Cast = dyn_cast<CastInst>(I))
338 if (Cast->isNoopCast(DL) && !hasTrivialKill(Cast->getOperand(0)))
339 return false;
340
341 // Even the value might have only one use in the LLVM IR, it is possible that
342 // FastISel might fold the use into another instruction and now there is more
343 // than one use at the Machine Instruction level.
344 unsigned Reg = lookUpRegForValue(V);
345 if (Reg && !MRI.use_empty(Reg))
346 return false;
347
348 // GEPs with all zero indices are trivially coalesced by fast-isel.
349 if (const auto *GEP = dyn_cast<GetElementPtrInst>(I))
350 if (GEP->hasAllZeroIndices() && !hasTrivialKill(GEP->getOperand(0)))
351 return false;
352
353 // Only instructions with a single use in the same basic block are considered
354 // to have trivial kills.
355 return I->hasOneUse() &&
356 !(I->getOpcode() == Instruction::BitCast ||
357 I->getOpcode() == Instruction::PtrToInt ||
358 I->getOpcode() == Instruction::IntToPtr) &&
359 cast<Instruction>(*I->user_begin())->getParent() == I->getParent();
360}

362unsigned FastISel::getRegForValue(const Value *V) {
363 EVT RealVT = TLI.getValueType(DL, V->getType(), /*AllowUnknown=*/true);
364 // Don't handle non-simple values in FastISel.
365 if (!RealVT.isSimple())
366 return 0;
367
368 // Ignore illegal types. We must do this before looking up the value
369 // in ValueMap because Arguments are given virtual registers regardless
370 // of whether FastISel can handle them.
371 MVT VT = RealVT.getSimpleVT();
372 if (!TLI.isTypeLegal(VT)) {
373 // Handle integer promotions, though, because they're common and easy.
374 if (VT == MVT::i1 || VT == MVT::i8 || VT == MVT::i16)
375 VT = TLI.getTypeToTransformTo(V->getContext(), VT).getSimpleVT();
376 else
377 return 0;
378 }
379
380 // Look up the value to see if we already have a register for it.
381 unsigned Reg = lookUpRegForValue(V);
382 if (Reg)
383 return Reg;
384
385 // In bottom-up mode, just create the virtual register which will be used
386 // to hold the value. It will be materialized later.
387 if (isa<Instruction>(V) &&
388 (!isa<AllocaInst>(V) ||
389 !FuncInfo.StaticAllocaMap.count(cast<AllocaInst>(V))))
390 return FuncInfo.InitializeRegForValue(V);
391
392 SavePoint SaveInsertPt = enterLocalValueArea();
393
394 // Materialize the value in a register. Emit any instructions in the
395 // local value area.
396 Reg = materializeRegForValue(V, VT);
397
398 leaveLocalValueArea(SaveInsertPt);
399
400 return Reg;
401}

403unsigned FastISel::materializeConstant(const Value *V, MVT VT) {
404 unsigned Reg = 0;
405 if (const auto *CI = dyn_cast<ConstantInt>(V)) {
406 if (CI->getValue().getActiveBits() <= 64)
407 Reg = fastEmit_i(VT, VT, ISD::Constant, CI->getZExtValue());
408 } else if (isa<AllocaInst>(V))
409 Reg = fastMaterializeAlloca(cast<AllocaInst>(V));
410 else if (isa<ConstantPointerNull>(V))
411 // Translate this as an integer zero so that it can be
412 // local-CSE'd with actual integer zeros.
413 Reg = getRegForValue(
414 Constant::getNullValue(DL.getIntPtrType(V->getContext())));
415 else if (const auto *CF = dyn_cast<ConstantFP>(V)) {
416 if (CF->isNullValue())
417 Reg = fastMaterializeFloatZero(CF);
418 else
419 // Try to emit the constant directly.
420 Reg = fastEmit_f(VT, VT, ISD::ConstantFP, CF);
421
422 if (!Reg) {
423 // Try to emit the constant by using an integer constant with a cast.
424 const APFloat &Flt = CF->getValueAPF();
425 EVT IntVT = TLI.getPointerRangeTy(DL);
426 uint32_t IntBitWidth = IntVT.getSizeInBits();
427 APSInt SIntVal(IntBitWidth, /*isUnsigned=*/false);
428 bool isExact;
429 (void)Flt.convertToInteger(SIntVal, APFloat::rmTowardZero, &isExact);
430 if (isExact) {
431 unsigned IntegerReg =
432 getRegForValue(ConstantInt::get(V->getContext(), SIntVal));
433 if (IntegerReg != 0)
434 Reg = fastEmit_r(IntVT.getSimpleVT(), VT, ISD::SINT_TO_FP, IntegerReg,
435 /*Kill=*/false);
436 }
437 }
438 } else if (const auto *Op = dyn_cast<Operator>(V)) {
439 if (!selectOperator(Op, Op->getOpcode()))
440 if (!isa<Instruction>(Op) ||
441 !fastSelectInstruction(cast<Instruction>(Op)))
442 return 0;
443 Reg = lookUpRegForValue(Op);
444 } else if (isa<UndefValue>(V)) {
445 Reg = createResultReg(TLI.getRegClassFor(VT));
446 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
447 TII.get(TargetOpcode::IMPLICIT_DEF), Reg);
448 }
449 return Reg;
450}

452/// Helper for getRegForValue. This function is called when the value isn't
453/// already available in a register and must be materialized with new
454/// instructions.
455unsigned FastISel::materializeRegForValue(const Value *V, MVT VT) {
456 unsigned Reg = 0;
457 // Give the target-specific code a try first.
458 if (isa<Constant>(V))
459 Reg = fastMaterializeConstant(cast<Constant>(V));
460
461 // If target-specific code couldn't or didn't want to handle the value, then
462 // give target-independent code a try.
463 if (!Reg)
464 Reg = materializeConstant(V, VT);
465
466 // Don't cache constant materializations in the general ValueMap.
467 // To do so would require tracking what uses they dominate.
468 if (Reg) {
469 LocalValueMap[V] = Reg;
470 LastLocalValue = MRI.getVRegDef(Reg);
471 }
472 return Reg;
473}

475unsigned FastISel::lookUpRegForValue(const Value *V) {
476 // Look up the value to see if we already have a register for it. We
477 // cache values defined by Instructions across blocks, and other values
478 // only locally. This is because Instructions already have the SSA
479 // def-dominates-use requirement enforced.
480 DenseMap<const Value *, unsigned>::iterator I = FuncInfo.ValueMap.find(V);
481 if (I != FuncInfo.ValueMap.end())
482 return I->second;
483 return LocalValueMap[V];
484}

486void FastISel::updateValueMap(const Value *I, unsigned Reg, unsigned NumRegs) {
487 if (!isa<Instruction>(I)) {
488 LocalValueMap[I] = Reg;
489 return;
490 }
491
492 unsigned &AssignedReg = FuncInfo.ValueMap[I];
493 if (AssignedReg == 0)
494 // Use the new register.
495 AssignedReg = Reg;
496 else if (Reg != AssignedReg) {
497 // Arrange for uses of AssignedReg to be replaced by uses of Reg.
498 for (unsigned i = 0; i < NumRegs; i++) {
499 FuncInfo.RegFixups[AssignedReg + i] = Reg + i;
500 FuncInfo.RegsWithFixups.insert(Reg + i);
501 }
502
503 AssignedReg = Reg;
504 }
505}

507std::pair<unsigned, bool> FastISel::getRegForGEPIndex(const Value *Idx) {
508 unsigned IdxN = getRegForValue(Idx);
509 if (IdxN == 0)
510 // Unhandled operand. Halt "fast" selection and bail.
511 return std::pair<unsigned, bool>(0, false);
512
513 bool IdxNIsKill = hasTrivialKill(Idx);
514
515 // If the index is smaller or larger than intptr_t, truncate or extend it.
516 MVT PtrVT = TLI.getPointerRangeTy(DL);
517 EVT IdxVT = EVT::getEVT(Idx->getType(), /*HandleUnknown=*/false);
518 if (IdxVT.bitsLT(PtrVT)) {
519 IdxN = fastEmit_r(IdxVT.getSimpleVT(), PtrVT, ISD::SIGN_EXTEND, IdxN,
520 IdxNIsKill);
521 IdxNIsKill = true;
522 } else if (IdxVT.bitsGT(PtrVT)) {
523 IdxN =
524 fastEmit_r(IdxVT.getSimpleVT(), PtrVT, ISD::TRUNCATE, IdxN, IdxNIsKill);
525 IdxNIsKill = true;
526 }
527 return std::pair<unsigned, bool>(IdxN, IdxNIsKill);
528}

530void FastISel::recomputeInsertPt() {
531 if (getLastLocalValue()) {
532 FuncInfo.InsertPt = getLastLocalValue();
533 FuncInfo.MBB = FuncInfo.InsertPt->getParent();
534 ++FuncInfo.InsertPt;
535 } else
536 FuncInfo.InsertPt = FuncInfo.MBB->getFirstNonPHI();
537
538 // Now skip past any EH_LABELs, which must remain at the beginning.
539 while (FuncInfo.InsertPt != FuncInfo.MBB->end() &&
540 FuncInfo.InsertPt->getOpcode() == TargetOpcode::EH_LABEL)
541 ++FuncInfo.InsertPt;
542}

544void FastISel::removeDeadCode(MachineBasicBlock::iterator I,
545 MachineBasicBlock::iterator E) {
546 assert(I.isValid() && E.isValid() && std::distance(I, E) > 0 &&
547 "Invalid iterator!");
548 while (I != E) {
549 if (LastFlushPoint == I)
550 LastFlushPoint = E;
551 if (SavedInsertPt == I)
552 SavedInsertPt = E;
553 if (EmitStartPt == I)
554 EmitStartPt = E.isValid() ? &*E : nullptr;
555 if (LastLocalValue == I)
556 LastLocalValue = E.isValid() ? &*E : nullptr;
557
558 MachineInstr *Dead = &*I;
559 ++I;
560 Dead->eraseFromParent();
561 ++NumFastIselDead;
562 }
563 recomputeInsertPt();
564}

566FastISel::SavePoint FastISel::enterLocalValueArea() {
567 MachineBasicBlock::iterator OldInsertPt = FuncInfo.InsertPt;
568 DebugLoc OldDL = DbgLoc;
569 recomputeInsertPt();
570 DbgLoc = DebugLoc();
571 SavePoint SP = {OldInsertPt, OldDL};
572 return SP;
573}

575void FastISel::leaveLocalValueArea(SavePoint OldInsertPt) {
576 if (FuncInfo.InsertPt != FuncInfo.MBB->begin())
577 LastLocalValue = &*std::prev(FuncInfo.InsertPt);
578
579 // Restore the previous insert position.
580 FuncInfo.InsertPt = OldInsertPt.InsertPt;
581 DbgLoc = OldInsertPt.DL;
582}

584bool FastISel::selectBinaryOp(const User *I, unsigned ISDOpcode) {
585 EVT VT = EVT::getEVT(I->getType(), /*HandleUnknown=*/true);
586 if (VT == MVT::Other || !VT.isSimple())
587 // Unhandled type. Halt "fast" selection and bail.
588 return false;
589
590 // We only handle legal types. For example, on x86-32 the instruction
591 // selector contains all of the 64-bit instructions from x86-64,
592 // under the assumption that i64 won't be used if the target doesn't
593 // support it.
594 if (!TLI.isTypeLegal(VT)) {
595 // MVT::i1 is special. Allow AND, OR, or XOR because they
596 // don't require additional zeroing, which makes them easy.
597 if (VT == MVT::i1 && (ISDOpcode == ISD::AND || ISDOpcode == ISD::OR ||
598 ISDOpcode == ISD::XOR))
599 VT = TLI.getTypeToTransformTo(I->getContext(), VT);
600 else
601 return false;
602 }
603
604 // Check if the first operand is a constant, and handle it as "ri". At -O0,
605 // we don't have anything that canonicalizes operand order.
606 if (const auto *CI = dyn_cast<ConstantInt>(I->getOperand(0)))
607 if (isa<Instruction>(I) && cast<Instruction>(I)->isCommutative()) {
608 unsigned Op1 = getRegForValue(I->getOperand(1));
609 if (!Op1)
610 return false;
611 bool Op1IsKill = hasTrivialKill(I->getOperand(1));
612
613 unsigned ResultReg =
614 fastEmit_ri_(VT.getSimpleVT(), ISDOpcode, Op1, Op1IsKill,
615 CI->getZExtValue(), VT.getSimpleVT());
616 if (!ResultReg)
617 return false;
618
619 // We successfully emitted code for the given LLVM Instruction.
620 updateValueMap(I, ResultReg);
621 return true;
622 }
623
624 unsigned Op0 = getRegForValue(I->getOperand(0));
625 if (!Op0) // Unhandled operand. Halt "fast" selection and bail.
626 return false;
627 bool Op0IsKill = hasTrivialKill(I->getOperand(0));
628
629 // Check if the second operand is a constant and handle it appropriately.
630 if (const auto *CI = dyn_cast<ConstantInt>(I->getOperand(1))) {
631 uint64_t Imm = CI->getSExtValue();
632
633 // Transform "sdiv exact X, 8" -> "sra X, 3".
634 if (ISDOpcode == ISD::SDIV && isa<BinaryOperator>(I) &&
635 cast<BinaryOperator>(I)->isExact() && isPowerOf2_64(Imm)) {
636 Imm = Log2_64(Imm);
637 ISDOpcode = ISD::SRA;
638 }
639
640 // Transform "urem x, pow2" -> "and x, pow2-1".
641 if (ISDOpcode == ISD::UREM && isa<BinaryOperator>(I) &&
642 isPowerOf2_64(Imm)) {
643 --Imm;
644 ISDOpcode = ISD::AND;
645 }
646
647 unsigned ResultReg = fastEmit_ri_(VT.getSimpleVT(), ISDOpcode, Op0,
648 Op0IsKill, Imm, VT.getSimpleVT());
649 if (!ResultReg)
650 return false;
651
652 // We successfully emitted code for the given LLVM Instruction.
653 updateValueMap(I, ResultReg);
654 return true;
655 }
656
657 unsigned Op1 = getRegForValue(I->getOperand(1));
658 if (!Op1) // Unhandled operand. Halt "fast" selection and bail.
659 return false;
660 bool Op1IsKill = hasTrivialKill(I->getOperand(1));
661
662 // Now we have both operands in registers. Emit the instruction.
663 unsigned ResultReg = fastEmit_rr(VT.getSimpleVT(), VT.getSimpleVT(),
664 ISDOpcode, Op0, Op0IsKill, Op1, Op1IsKill);
665 if (!ResultReg)
666 // Target-specific code wasn't able to find a machine opcode for
667 // the given ISD opcode and type. Halt "fast" selection and bail.
668 return false;
669
670 // We successfully emitted code for the given LLVM Instruction.
671 updateValueMap(I, ResultReg);
672 return true;
673}

675bool FastISel::selectGetElementPtr(const User *I) {
676 unsigned N = getRegForValue(I->getOperand(0));
677 if (!N) // Unhandled operand. Halt "fast" selection and bail.
678 return false;
679 if (I->getOperand(0)->getType()->getPointerAddressSpace() != 0)
680 return false;
681 bool NIsKill = hasTrivialKill(I->getOperand(0));
682
683 // Keep a running tab of the total offset to coalesce multiple N = N + Offset
684 // into a single N = N + TotalOffset.
685 uint64_t TotalOffs = 0;
686 // FIXME: What's a good SWAG number for MaxOffs?
687 uint64_t MaxOffs = 2048;
688 MVT VT = TLI.getPointerRangeTy(DL);
689 for (gep_type_iterator GTI = gep_type_begin(I), E = gep_type_end(I);
690 GTI != E; ++GTI) {
691 const Value *Idx = GTI.getOperand();
692 if (StructType *StTy = GTI.getStructTypeOrNull()) {
693 uint64_t Field = cast<ConstantInt>(Idx)->getZExtValue();
694 if (Field) {
695 // N = N + Offset
696 TotalOffs += DL.getStructLayout(StTy)->getElementOffset(Field);
697 if (TotalOffs >= MaxOffs) {
698 N = fastEmit_ri_(VT, ISD::ADD, N, NIsKill, TotalOffs, VT);
699 if (!N) // Unhandled operand. Halt "fast" selection and bail.
700 return false;
701 NIsKill = true;
702 TotalOffs = 0;
703 }
704 }
705 } else {
706 Type *Ty = GTI.getIndexedType();
707
708 // If this is a constant subscript, handle it quickly.
709 if (const auto *CI = dyn_cast<ConstantInt>(Idx)) {
710 if (CI->isZero())
711 continue;
712 // N = N + Offset
713 uint64_t IdxN = CI->getValue().sextOrTrunc(64).getSExtValue();
714 TotalOffs += DL.getTypeAllocSize(Ty) * IdxN;
715 if (TotalOffs >= MaxOffs) {
716 N = fastEmit_ri_(VT, ISD::ADD, N, NIsKill, TotalOffs, VT);
717 if (!N) // Unhandled operand. Halt "fast" selection and bail.
718 return false;
719 NIsKill = true;
720 TotalOffs = 0;
721 }
722 continue;
723 }
724 if (TotalOffs) {
725 N = fastEmit_ri_(VT, ISD::ADD, N, NIsKill, TotalOffs, VT);
726 if (!N) // Unhandled operand. Halt "fast" selection and bail.
727 return false;
728 NIsKill = true;
729 TotalOffs = 0;
730 }
731
732 // N = N + Idx * ElementSize;
733 uint64_t ElementSize = DL.getTypeAllocSize(Ty);
734 std::pair<unsigned, bool> Pair = getRegForGEPIndex(Idx);
735 unsigned IdxN = Pair.first;
736 bool IdxNIsKill = Pair.second;
737 if (!IdxN) // Unhandled operand. Halt "fast" selection and bail.
738 return false;
739
740 if (ElementSize != 1) {
741 IdxN = fastEmit_ri_(VT, ISD::MUL, IdxN, IdxNIsKill, ElementSize, VT);
742 if (!IdxN) // Unhandled operand. Halt "fast" selection and bail.
743 return false;
744 IdxNIsKill = true;
745 }
746 N = fastEmit_rr(VT, VT, ISD::ADD, N, NIsKill, IdxN, IdxNIsKill);
747 if (!N) // Unhandled operand. Halt "fast" selection and bail.
748 return false;
749 }
750 }
751 if (TotalOffs) {
752 N = fastEmit_ri_(VT, ISD::ADD, N, NIsKill, TotalOffs, VT);
753 if (!N) // Unhandled operand. Halt "fast" selection and bail.
754 return false;
755 }
756
757 // We successfully emitted code for the given LLVM Instruction.
758 updateValueMap(I, N);
759 return true;
760}

762bool FastISel::addStackMapLiveVars(SmallVectorImpl<MachineOperand> &Ops,
763 const CallInst *CI, unsigned StartIdx) {
764 for (unsigned i = StartIdx, e = CI->getNumArgOperands(); i != e; ++i) {
765 Value *Val = CI->getArgOperand(i);
766 // Check for constants and encode them with a StackMaps::ConstantOp prefix.
767 if (const auto *C = dyn_cast<ConstantInt>(Val)) {
768 Ops.push_back(MachineOperand::CreateImm(StackMaps::ConstantOp));
769 Ops.push_back(MachineOperand::CreateImm(C->getSExtValue()));
770 } else if (isa<ConstantPointerNull>(Val)) {
771 Ops.push_back(MachineOperand::CreateImm(StackMaps::ConstantOp));
772 Ops.push_back(MachineOperand::CreateImm(0));
773 } else if (auto *AI = dyn_cast<AllocaInst>(Val)) {
774 // Values coming from a stack location also require a special encoding,
775 // but that is added later on by the target specific frame index
776 // elimination implementation.
777 auto SI = FuncInfo.StaticAllocaMap.find(AI);
778 if (SI != FuncInfo.StaticAllocaMap.end())
779 Ops.push_back(MachineOperand::CreateFI(SI->second));
780 else
781 return false;
782 } else {
783 unsigned Reg = getRegForValue(Val);
784 if (!Reg)
785 return false;
786 Ops.push_back(MachineOperand::CreateReg(Reg, /*IsDef=*/false));
787 }
788 }
789 return true;
790}

792bool FastISel::selectStackmap(const CallInst *I) {
793 // void @llvm.experimental.stackmap(i64 <id>, i32 <numShadowBytes>,
794 // [live variables...])
795 assert(I->getCalledFunction()->getReturnType()->isVoidTy() &&
796 "Stackmap cannot return a value.");
797
798 // The stackmap intrinsic only records the live variables (the arguments
799 // passed to it) and emits NOPS (if requested). Unlike the patchpoint
800 // intrinsic, this won't be lowered to a function call. This means we don't
801 // have to worry about calling conventions and target-specific lowering code.
802 // Instead we perform the call lowering right here.
803 //
804 // CALLSEQ_START(0, 0...)
805 // STACKMAP(id, nbytes, ...)
806 // CALLSEQ_END(0, 0)
807 //
808 SmallVector<MachineOperand, 32> Ops;
809
810 // Add the <id> and <numBytes> constants.
811 assert(isa<ConstantInt>(I->getOperand(PatchPointOpers::IDPos)) &&
812 "Expected a constant integer.");
813 const auto *ID = cast<ConstantInt>(I->getOperand(PatchPointOpers::IDPos));
814 Ops.push_back(MachineOperand::CreateImm(ID->getZExtValue()));
815
816 assert(isa<ConstantInt>(I->getOperand(PatchPointOpers::NBytesPos)) &&
817 "Expected a constant integer.");
818 const auto *NumBytes =
819 cast<ConstantInt>(I->getOperand(PatchPointOpers::NBytesPos));
820 Ops.push_back(MachineOperand::CreateImm(NumBytes->getZExtValue()));
821
822 // Push live variables for the stack map (skipping the first two arguments
823 // <id> and <numBytes>).
824 if (!addStackMapLiveVars(Ops, I, 2))
825 return false;
826
827 // We are not adding any register mask info here, because the stackmap doesn't
828 // clobber anything.
829
830 // Add scratch registers as implicit def and early clobber.
831 CallingConv::ID CC = I->getCallingConv();
832 const MCPhysReg *ScratchRegs = TLI.getScratchRegisters(CC);
833 for (unsigned i = 0; ScratchRegs[i]; ++i)
834 Ops.push_back(MachineOperand::CreateReg(
835 ScratchRegs[i], /*IsDef=*/true, /*IsImp=*/true, /*IsKill=*/false,
836 /*IsDead=*/false, /*IsUndef=*/false, /*IsEarlyClobber=*/true));
837
838 // Issue CALLSEQ_START
839 unsigned AdjStackDown = TII.getCallFrameSetupOpcode();
840 auto Builder =
841 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, TII.get(AdjStackDown));
842 const MCInstrDesc &MCID = Builder.getInstr()->getDesc();
843 for (unsigned I = 0, E = MCID.getNumOperands(); I < E; ++I)
844 Builder.addImm(0);
845
846 // Issue STACKMAP.
847 MachineInstrBuilder MIB = BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
848 TII.get(TargetOpcode::STACKMAP));
849 for (auto const &MO : Ops)
850 MIB.add(MO);
851
852 // Issue CALLSEQ_END
853 unsigned AdjStackUp = TII.getCallFrameDestroyOpcode();
854 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, TII.get(AdjStackUp))
855 .addImm(0)
856 .addImm(0);
857
858 // Inform the Frame Information that we have a stackmap in this function.
859 FuncInfo.MF->getFrameInfo().setHasStackMap();
860
861 return true;
862}

905bool FastISel::selectPatchpoint(const CallInst *I) {
906 // void|i64 @llvm.experimental.patchpoint.void|i64(i64 <id>,
907 // i32 <numBytes>,
908 // i8* <target>,
909 // i32 <numArgs>,
910 // [Args...],
911 // [live variables...])
912 CallingConv::ID CC = I->getCallingConv();
913 bool IsAnyRegCC = CC == CallingConv::AnyReg;
914 bool HasDef = !I->getType()->isVoidTy();
915 Value *Callee = I->getOperand(PatchPointOpers::TargetPos)->stripPointerCasts();
916
917 // Get the real number of arguments participating in the call <numArgs>
918 assert(isa<ConstantInt>(I->getOperand(PatchPointOpers::NArgPos)) &&
919 "Expected a constant integer.");
920 const auto *NumArgsVal =
921 cast<ConstantInt>(I->getOperand(PatchPointOpers::NArgPos));
922 unsigned NumArgs = NumArgsVal->getZExtValue();
923
924 // Skip the four meta args: <id>, <numNopBytes>, <target>, <numArgs>
925 // This includes all meta-operands up to but not including CC.
926 unsigned NumMetaOpers = PatchPointOpers::CCPos;
927 assert(I->getNumArgOperands() >= NumMetaOpers + NumArgs &&
928 "Not enough arguments provided to the patchpoint intrinsic");
929
930 // For AnyRegCC the arguments are lowered later on manually.
931 unsigned NumCallArgs = IsAnyRegCC ? 0 : NumArgs;
932 CallLoweringInfo CLI;
933 CLI.setIsPatchPoint();
934 if (!lowerCallOperands(I, NumMetaOpers, NumCallArgs, Callee, IsAnyRegCC, CLI))
935 return false;
936
937 assert(CLI.Call && "No call instruction specified.");
938
939 SmallVector<MachineOperand, 32> Ops;
940
941 // Add an explicit result reg if we use the anyreg calling convention.
942 if (IsAnyRegCC && HasDef) {
943 assert(CLI.NumResultRegs == 0 && "Unexpected result register.");
944 CLI.ResultReg = createResultReg(TLI.getRegClassFor(MVT::i64));
945 CLI.NumResultRegs = 1;
946 Ops.push_back(MachineOperand::CreateReg(CLI.ResultReg, /*IsDef=*/true));
947 }
948
949 // Add the <id> and <numBytes> constants.
950 assert(isa<ConstantInt>(I->getOperand(PatchPointOpers::IDPos)) &&
951 "Expected a constant integer.");
952 const auto *ID = cast<ConstantInt>(I->getOperand(PatchPointOpers::IDPos));
953 Ops.push_back(MachineOperand::CreateImm(ID->getZExtValue()));
954
955 assert(isa<ConstantInt>(I->getOperand(PatchPointOpers::NBytesPos)) &&
956 "Expected a constant integer.");
957 const auto *NumBytes =
958 cast<ConstantInt>(I->getOperand(PatchPointOpers::NBytesPos));
959 Ops.push_back(MachineOperand::CreateImm(NumBytes->getZExtValue()));
960
961 // Add the call target.
962 if (const auto *C = dyn_cast<IntToPtrInst>(Callee)) {
963 uint64_t CalleeConstAddr =
964 cast<ConstantInt>(C->getOperand(0))->getZExtValue();
965 Ops.push_back(MachineOperand::CreateImm(CalleeConstAddr));
966 } else if (const auto *C = dyn_cast<ConstantExpr>(Callee)) {
967 if (C->getOpcode() == Instruction::IntToPtr) {
968 uint64_t CalleeConstAddr =
969 cast<ConstantInt>(C->getOperand(0))->getZExtValue();
970 Ops.push_back(MachineOperand::CreateImm(CalleeConstAddr));
971 } else
972 llvm_unreachable("Unsupported ConstantExpr.");
973 } else if (const auto *GV = dyn_cast<GlobalValue>(Callee)) {
974 Ops.push_back(MachineOperand::CreateGA(GV, 0));
975 } else if (isa<ConstantPointerNull>(Callee))
976 Ops.push_back(MachineOperand::CreateImm(0));
977 else
978 llvm_unreachable("Unsupported callee address.");
979
980 // Adjust <numArgs> to account for any arguments that have been passed on
981 // the stack instead.
982 unsigned NumCallRegArgs = IsAnyRegCC ? NumArgs : CLI.OutRegs.size();
983 Ops.push_back(MachineOperand::CreateImm(NumCallRegArgs));
984
985 // Add the calling convention
986 Ops.push_back(MachineOperand::CreateImm((unsigned)CC));
987
988 // Add the arguments we omitted previously. The register allocator should
989 // place these in any free register.
990 if (IsAnyRegCC) {
991 for (unsigned i = NumMetaOpers, e = NumMetaOpers + NumArgs; i != e; ++i) {
992 unsigned Reg = getRegForValue(I->getArgOperand(i));
993 if (!Reg)
994 return false;
995 Ops.push_back(MachineOperand::CreateReg(Reg, /*IsDef=*/false));
996 }
997 }
998
999 // Push the arguments from the call instruction.
1000 for (auto Reg : CLI.OutRegs)
1001 Ops.push_back(MachineOperand::CreateReg(Reg, /*IsDef=*/false));
1002
1003 // Push live variables for the stack map.
1004 if (!addStackMapLiveVars(Ops, I, NumMetaOpers + NumArgs))
1005 return false;
1006
1007 // Push the register mask info.
1008 Ops.push_back(MachineOperand::CreateRegMask(
1009 TRI.getCallPreservedMask(*FuncInfo.MF, CC)));
1010
1011 // Add scratch registers as implicit def and early clobber.
1012 const MCPhysReg *ScratchRegs = TLI.getScratchRegisters(CC);
1013 for (unsigned i = 0; ScratchRegs[i]; ++i)
1014 Ops.push_back(MachineOperand::CreateReg(
1015 ScratchRegs[i], /*IsDef=*/true, /*IsImp=*/true, /*IsKill=*/false,
1016 /*IsDead=*/false, /*IsUndef=*/false, /*IsEarlyClobber=*/true));
1017
1018 // Add implicit defs (return values).
1019 for (auto Reg : CLI.InRegs)
1020 Ops.push_back(MachineOperand::CreateReg(Reg, /*IsDef=*/true,
1021 /*IsImpl=*/true));
1022
1023 // Insert the patchpoint instruction before the call generated by the target.
1024 MachineInstrBuilder MIB = BuildMI(*FuncInfo.MBB, CLI.Call, DbgLoc,
1025 TII.get(TargetOpcode::PATCHPOINT));
1026
1027 for (auto &MO : Ops)
1028 MIB.add(MO);
1029
1030 MIB->setPhysRegsDeadExcept(CLI.InRegs, TRI);
1031
1032 // Delete the original call instruction.
1033 CLI.Call->eraseFromParent();
1034
1035 // Inform the Frame Information that we have a patchpoint in this function.
1036 FuncInfo.MF->getFrameInfo().setHasPatchPoint();
1037
1038 if (CLI.NumResultRegs)
1039 updateValueMap(I, CLI.ResultReg, CLI.NumResultRegs);
1040 return true;
1041}

1138bool FastISel::lowerCallTo(CallLoweringInfo &CLI) {
1139 // Handle the incoming return values from the call.
1140 CLI.clearIns();
1141 SmallVector<EVT, 4> RetTys;
1142 ComputeValueVTs(TLI, DL, CLI.RetTy, RetTys);
1143
1144 SmallVector<ISD::OutputArg, 4> Outs;
1145 GetReturnInfo(CLI.CallConv, CLI.RetTy, getReturnAttrs(CLI), Outs, TLI, DL);
1146
1147 bool CanLowerReturn = TLI.CanLowerReturn(
1148 CLI.CallConv, *FuncInfo.MF, CLI.IsVarArg, Outs, CLI.RetTy->getContext());
1149
1150 // FIXME: sret demotion isn't supported yet - bail out.
1151 if (!CanLowerReturn)
1152 return false;
1153
1154 for (unsigned I = 0, E = RetTys.size(); I != E; ++I) {
1155 EVT VT = RetTys[I];
1156 MVT RegisterVT = TLI.getRegisterType(CLI.RetTy->getContext(), VT);
1157 unsigned NumRegs = TLI.getNumRegisters(CLI.RetTy->getContext(), VT);
1158 for (unsigned i = 0; i != NumRegs; ++i) {
1159 ISD::InputArg MyFlags;
1160 MyFlags.VT = RegisterVT;
1161 MyFlags.ArgVT = VT;
1162 MyFlags.Used = CLI.IsReturnValueUsed;
1163 if (CLI.RetSExt)
1164 MyFlags.Flags.setSExt();
1165 if (CLI.RetZExt)
1166 MyFlags.Flags.setZExt();
1167 if (CLI.IsInReg)
1168 MyFlags.Flags.setInReg();
1169 CLI.Ins.push_back(MyFlags);
1170 }
1171 }
1172
1173 // Handle all of the outgoing arguments.
1174 CLI.clearOuts();
1175 for (auto &Arg : CLI.getArgs()) {
1176 Type *FinalType = Arg.Ty;
1177 if (Arg.IsByVal)
1178 FinalType = cast<PointerType>(Arg.Ty)->getElementType();
1179 bool NeedsRegBlock = TLI.functionArgumentNeedsConsecutiveRegisters(
1180 FinalType, CLI.CallConv, CLI.IsVarArg);
1181
1182 ISD::ArgFlagsTy Flags;
1183 if (Arg.IsZExt)
1184 Flags.setZExt();
1185 if (Arg.IsSExt)
1186 Flags.setSExt();
1187 if (Arg.IsInReg)
1188 Flags.setInReg();
1189 if (Arg.IsSRet)
1190 Flags.setSRet();
1191 if (Arg.IsSwiftSelf)
1192 Flags.setSwiftSelf();
1193 if (Arg.IsSwiftError)
1194 Flags.setSwiftError();
1195 if (Arg.IsByVal)
1196 Flags.setByVal();
1197 if (Arg.IsInAlloca) {
1198 Flags.setInAlloca();
1199 // Set the byval flag for CCAssignFn callbacks that don't know about
1200 // inalloca. This way we can know how many bytes we should've allocated
1201 // and how many bytes a callee cleanup function will pop. If we port
1202 // inalloca to more targets, we'll have to add custom inalloca handling in
1203 // the various CC lowering callbacks.
1204 Flags.setByVal();
1205 }
1206 if (Arg.IsByVal || Arg.IsInAlloca) {
1207 PointerType *Ty = cast<PointerType>(Arg.Ty);
1208 Type *ElementTy = Ty->getElementType();
1209 unsigned FrameSize =
1210 DL.getTypeAllocSize(Arg.ByValType ? Arg.ByValType : ElementTy);
1211
1212 // For ByVal, alignment should come from FE. BE will guess if this info
1213 // is not there, but there are cases it cannot get right.
1214 unsigned FrameAlign = Arg.Alignment;
1215 if (!FrameAlign)
1216 FrameAlign = TLI.getByValTypeAlignment(ElementTy, DL);
1217 Flags.setByValSize(FrameSize);
1218 Flags.setByValAlign(FrameAlign);
1219 }
1220 if (Arg.IsNest)
1221 Flags.setNest();
1222 if (NeedsRegBlock)
1223 Flags.setInConsecutiveRegs();
1224 unsigned OriginalAlignment = DL.getABITypeAlignment(Arg.Ty);
1225 Flags.setOrigAlign(OriginalAlignment);
1226
1227 CLI.OutVals.push_back(Arg.Val);
1228 CLI.OutFlags.push_back(Flags);
1229 }
1230
1231 if (!fastLowerCall(CLI))
1232 return false;
1233
1234 // Set all unused physreg defs as dead.
1235 assert(CLI.Call && "No call instruction specified.");
1236 CLI.Call->setPhysRegsDeadExcept(CLI.InRegs, TRI);
1237
1238 if (CLI.NumResultRegs && CLI.CS)
1239 updateValueMap(CLI.CS->getInstruction(), CLI.ResultReg, CLI.NumResultRegs);
1240
1241 // Set labels for heapallocsite call.
1242 if (CLI.CS && CLI.CS->getInstruction()->getMetadata("heapallocsite")) {
1243 MDNode *MD = CLI.CS->getInstruction()->getMetadata("heapallocsite");
1244 MF->addCodeViewHeapAllocSite(CLI.Call, MD);
1245 }
1246
1247 return true;
1248}

1250bool FastISel::lowerCall(const CallInst *CI) {
1251 ImmutableCallSite CS(CI);
1252
1253 FunctionType *FuncTy = CS.getFunctionType();
1254 Type *RetTy = CS.getType();
1255
1256 ArgListTy Args;
1257 ArgListEntry Entry;
1258 Args.reserve(CS.arg_size());
1259
1260 for (ImmutableCallSite::arg_iterator i = CS.arg_begin(), e = CS.arg_end();
1261 i != e; ++i) {
1262 Value *V = *i;
1263
1264 // Skip empty types
1265 if (V->getType()->isEmptyTy())
1266 continue;
1267
1268 Entry.Val = V;
1269 Entry.Ty = V->getType();
1270
1271 // Skip the first return-type Attribute to get to params.
1272 Entry.setAttributes(&CS, i - CS.arg_begin());
1273 Args.push_back(Entry);
1274 }
1275
1276 // Check if target-independent constraints permit a tail call here.
1277 // Target-dependent constraints are checked within fastLowerCall.
1278 bool IsTailCall = CI->isTailCall();
1279 if (IsTailCall && !isInTailCallPosition(CS, TM))
1280 IsTailCall = false;
1281
1282 CallLoweringInfo CLI;
1283 CLI.setCallee(RetTy, FuncTy, CI->getCalledValue(), std::move(Args), CS)
1284 .setTailCall(IsTailCall);
1285
1286 return lowerCallTo(CLI);
1287}

1289bool FastISel::selectCall(const User *I) {
1290 const CallInst *Call = cast<CallInst>(I);
1291
1292 // Handle simple inline asms.
1293 if (const InlineAsm *IA = dyn_cast<InlineAsm>(Call->getCalledValue())) {
1294 // If the inline asm has side effects, then make sure that no local value
1295 // lives across by flushing the local value map.
1296 if (IA->hasSideEffects())
1297 flushLocalValueMap();
1298
1299 // Don't attempt to handle constraints.
1300 if (!IA->getConstraintString().empty())
1301 return false;
1302
1303 unsigned ExtraInfo = 0;
1304 if (IA->hasSideEffects())
1305 ExtraInfo |= InlineAsm::Extra_HasSideEffects;
1306 if (IA->isAlignStack())
1307 ExtraInfo |= InlineAsm::Extra_IsAlignStack;
1308
1309 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
1310 TII.get(TargetOpcode::INLINEASM))
1311 .addExternalSymbol(IA->getAsmString().c_str())
1312 .addImm(ExtraInfo);
1313 return true;
1314 }
1315
1316 // Handle intrinsic function calls.
1317 if (const auto *II = dyn_cast<IntrinsicInst>(Call))
1318 return selectIntrinsicCall(II);
1319
1320 // Usually, it does not make sense to initialize a value,
1321 // make an unrelated function call and use the value, because
1322 // it tends to be spilled on the stack. So, we move the pointer
1323 // to the last local value to the beginning of the block, so that
1324 // all the values which have already been materialized,
1325 // appear after the call. It also makes sense to skip intrinsics
1326 // since they tend to be inlined.
1327 flushLocalValueMap();
1328
1329 return lowerCall(Call);
1330}

1332bool FastISel::selectIntrinsicCall(const IntrinsicInst *II) {
1333 switch (II->getIntrinsicID()) {
1334 default:
1335 break;
1336 // At -O0 we don't care about the lifetime intrinsics.
1337 case Intrinsic::lifetime_start:
1338 case Intrinsic::lifetime_end:
1339 // The donothing intrinsic does, well, nothing.
1340 case Intrinsic::donothing:
1341 // Neither does the sideeffect intrinsic.
1342 case Intrinsic::sideeffect:
1343 // Neither does the assume intrinsic; it's also OK not to codegen its operand.
1344 case Intrinsic::assume:
1345 return true;
1346 case Intrinsic::dbg_declare: {
1347 const DbgDeclareInst *DI = cast<DbgDeclareInst>(II);
1348 assert(DI->getVariable() && "Missing variable");
1349 if (!FuncInfo.MF->getMMI().hasDebugInfo()) {
1350 LLVM_DEBUG(dbgs() << "Dropping debug info for " << *DI << "\n");
1351 return true;
1352 }
1353
1354 const Value *Address = DI->getAddress();
1355 if (!Address || isa<UndefValue>(Address)) {
1356 LLVM_DEBUG(dbgs() << "Dropping debug info for " << *DI << "\n");
1357 return true;
1358 }
1359
1360 // Byval arguments with frame indices were already handled after argument
1361 // lowering and before isel.
1362 const auto *Arg =
1363 dyn_cast<Argument>(Address->stripInBoundsConstantOffsets());
1364 if (Arg && FuncInfo.getArgumentFrameIndex(Arg) != INT_MAX)
1365 return true;
1366
1367 Optional<MachineOperand> Op;
1368 if (unsigned Reg = lookUpRegForValue(Address))
1369 Op = MachineOperand::CreateReg(Reg, false);
1370
1371 // If we have a VLA that has a "use" in a metadata node that's then used
1372 // here but it has no other uses, then we have a problem. E.g.,
1373 //
1374 // int foo (const int *x) {
1375 // char a[*x];
1376 // return 0;
1377 // }
1378 //
1379 // If we assign 'a' a vreg and fast isel later on has to use the selection
1380 // DAG isel, it will want to copy the value to the vreg. However, there are
1381 // no uses, which goes counter to what selection DAG isel expects.
1382 if (!Op && !Address->use_empty() && isa<Instruction>(Address) &&
1383 (!isa<AllocaInst>(Address) ||
1384 !FuncInfo.StaticAllocaMap.count(cast<AllocaInst>(Address))))
1385 Op = MachineOperand::CreateReg(FuncInfo.InitializeRegForValue(Address),
1386 false);
1387
1388 if (Op) {
1389 assert(DI->getVariable()->isValidLocationForIntrinsic(DbgLoc) &&
1390 "Expected inlined-at fields to agree");
1391 // A dbg.declare describes the address of a source variable, so lower it
1392 // into an indirect DBG_VALUE.
1393 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
1394 TII.get(TargetOpcode::DBG_VALUE), /*IsIndirect*/ true,
1395 *Op, DI->getVariable(), DI->getExpression());
1396 } else {
1397 // We can't yet handle anything else here because it would require
1398 // generating code, thus altering codegen because of debug info.
1399 LLVM_DEBUG(dbgs() << "Dropping debug info for " << *DI << "\n");
1400 }
1401 return true;
1402 }
1403 case Intrinsic::dbg_value: {
1404 // This form of DBG_VALUE is target-independent.
1405 const DbgValueInst *DI = cast<DbgValueInst>(II);
1406 const MCInstrDesc &II = TII.get(TargetOpcode::DBG_VALUE);
1407 const Value *V = DI->getValue();
1408 assert(DI->getVariable()->isValidLocationForIntrinsic(DbgLoc) &&
1409 "Expected inlined-at fields to agree");
1410 if (!V) {
1411 // Currently the optimizer can produce this; insert an undef to
1412 // help debugging. Probably the optimizer should not do this.
1413 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, II, false, 0U,
1414 DI->getVariable(), DI->getExpression());
1415 } else if (const auto *CI = dyn_cast<ConstantInt>(V)) {
1416 if (CI->getBitWidth() > 64)
1417 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, II)
1418 .addCImm(CI)
1419 .addImm(0U)
1420 .addMetadata(DI->getVariable())
1421 .addMetadata(DI->getExpression());
1422 else
1423 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, II)
1424 .addImm(CI->getZExtValue())
1425 .addImm(0U)
1426 .addMetadata(DI->getVariable())
1427 .addMetadata(DI->getExpression());
1428 } else if (const auto *CF = dyn_cast<ConstantFP>(V)) {
1429 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, II)
1430 .addFPImm(CF)
1431 .addImm(0U)
1432 .addMetadata(DI->getVariable())
1433 .addMetadata(DI->getExpression());
1434 } else if (unsigned Reg = lookUpRegForValue(V)) {
1435 // FIXME: This does not handle register-indirect values at offset 0.
1436 bool IsIndirect = false;
1437 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc, II, IsIndirect, Reg,
1438 DI->getVariable(), DI->getExpression());
1439 } else {
1440 // We can't yet handle anything else here because it would require
1441 // generating code, thus altering codegen because of debug info.
1442 LLVM_DEBUG(dbgs() << "Dropping debug info for " << *DI << "\n");
1443 }
1444 return true;
1445 }
1446 case Intrinsic::dbg_label: {
1447 const DbgLabelInst *DI = cast<DbgLabelInst>(II);
1448 assert(DI->getLabel() && "Missing label");
1449 if (!FuncInfo.MF->getMMI().hasDebugInfo()) {
1450 LLVM_DEBUG(dbgs() << "Dropping debug info for " << *DI << "\n");
1451 return true;
1452 }
1453
1454 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
1455 TII.get(TargetOpcode::DBG_LABEL)).addMetadata(DI->getLabel());
1456 return true;
1457 }
1458 case Intrinsic::objectsize: {
1459 ConstantInt *CI = cast<ConstantInt>(II->getArgOperand(1));
1460 unsigned long long Res = CI->isZero() ? -1ULL : 0;
1461 Constant *ResCI = ConstantInt::get(II->getType(), Res);
1462 unsigned ResultReg = getRegForValue(ResCI);
1463 if (!ResultReg)
1464 return false;
1465 updateValueMap(II, ResultReg);
1466 return true;
1467 }
1468 case Intrinsic::is_constant: {
1469 Constant *ResCI = ConstantInt::get(II->getType(), 0);
1470 unsigned ResultReg = getRegForValue(ResCI);
1471 if (!ResultReg)
1472 return false;
1473 updateValueMap(II, ResultReg);
1474 return true;
1475 }
1476 case Intrinsic::launder_invariant_group:
1477 case Intrinsic::strip_invariant_group:
1478 case Intrinsic::expect: {
1479 unsigned ResultReg = getRegForValue(II->getArgOperand(0));
1480 if (!ResultReg)
1481 return false;
1482 updateValueMap(II, ResultReg);
1483 return true;
1484 }
1485 case Intrinsic::experimental_stackmap:
1486 return selectStackmap(II);
1487 case Intrinsic::experimental_patchpoint_void:
1488 case Intrinsic::experimental_patchpoint_i64:
1489 return selectPatchpoint(II);
1490
1491 case Intrinsic::xray_customevent:
1492 return selectXRayCustomEvent(II);
1493 case Intrinsic::xray_typedevent:
1494 return selectXRayTypedEvent(II);
1495 }
1496
1497 return fastLowerIntrinsicCall(II);
1498}

1500bool FastISel::selectCast(const User *I, unsigned Opcode) {
1501 EVT SrcVT = TLI.getValueType(DL, I->getOperand(0)->getType());
1502 EVT DstVT = TLI.getValueType(DL, I->getType());
1503
1504 if (SrcVT == MVT::Other || !SrcVT.isSimple() || DstVT == MVT::Other ||
1505 !DstVT.isSimple())
1506 // Unhandled type. Halt "fast" selection and bail.
1507 return false;
1508
1509 // Check if the destination type is legal.
1510 if (!TLI.isTypeLegal(DstVT))
1511 return false;
1512
1513 // Check if the source operand is legal.
1514 if (!TLI.isTypeLegal(SrcVT))
1515 return false;
1516
1517 unsigned InputReg = getRegForValue(I->getOperand(0));
1518 if (!InputReg)
1519 // Unhandled operand. Halt "fast" selection and bail.
1520 return false;
1521
1522 bool InputRegIsKill = hasTrivialKill(I->getOperand(0));
1523
1524 unsigned ResultReg = fastEmit_r(SrcVT.getSimpleVT(), DstVT.getSimpleVT(),
1525 Opcode, InputReg, InputRegIsKill);
1526 if (!ResultReg)
1527 return false;
1528
1529 updateValueMap(I, ResultReg);
1530 return true;
1531}

1533bool FastISel::selectBitCast(const User *I) {
1534 // If the bitcast doesn't change the type, just use the operand value.
1535 if (I->getType() == I->getOperand(0)->getType()) {
1536 unsigned Reg = getRegForValue(I->getOperand(0));
1537 if (!Reg)
1538 return false;
1539 updateValueMap(I, Reg);
1540 return true;
1541 }
1542
1543 // Bitcasts of other values become reg-reg copies or BITCAST operators.
1544 EVT SrcEVT = TLI.getValueType(DL, I->getOperand(0)->getType());
1545 EVT DstEVT = TLI.getValueType(DL, I->getType());
1546 if (SrcEVT == MVT::Other || DstEVT == MVT::Other ||
1547 !TLI.isTypeLegal(SrcEVT) || !TLI.isTypeLegal(DstEVT))
1548 // Unhandled type. Halt "fast" selection and bail.
1549 return false;
1550
1551 MVT SrcVT = SrcEVT.getSimpleVT();
1552 MVT DstVT = DstEVT.getSimpleVT();
1553 unsigned Op0 = getRegForValue(I->getOperand(0));
1554 if (!Op0) // Unhandled operand. Halt "fast" selection and bail.
1555 return false;
1556 bool Op0IsKill = hasTrivialKill(I->getOperand(0));
1557
1558 // First, try to perform the bitcast by inserting a reg-reg copy.
1559 unsigned ResultReg = 0;
1560 if (SrcVT == DstVT) {
1561 const TargetRegisterClass *SrcClass = TLI.getRegClassFor(SrcVT);
1562 const TargetRegisterClass *DstClass = TLI.getRegClassFor(DstVT);
1563 // Don't attempt a cross-class copy. It will likely fail.
1564 if (SrcClass == DstClass) {
1565 ResultReg = createResultReg(DstClass);
1566 BuildMI(*FuncInfo.MBB, FuncInfo.InsertPt, DbgLoc,
1567 TII.get(TargetOpcode::COPY), ResultReg).addReg(Op0);
1568 }
1569 }
1570
1571 // If the reg-reg copy failed, select a BITCAST opcode.
1572 if (!ResultReg)
1573 ResultReg = fastEmit_r(SrcVT, DstVT, ISD::BITCAST, Op0, Op0IsKill);
1574
1575 if (!ResultReg)
1576 return false;
1577
1578 updateValueMap(I, ResultReg);
1579 return true;
1580}

1601bool FastISel::selectInstruction(const Instruction *I) {
1602 MachineInstr *SavedLastLocalValue = getLastLocalValue();
1603 // Just before the terminator instruction, insert instructions to
1604 // feed PHI nodes in successor blocks.
1605 if (I->isTerminator()) {
1606 if (!handlePHINodesInSuccessorBlocks(I->getParent())) {
1607 // PHI node handling may have generated local value instructions,
1608 // even though it failed to handle all PHI nodes.
1609 // We remove these instructions because SelectionDAGISel will generate
1610 // them again.
1611 removeDeadLocalValueCode(SavedLastLocalValue);
1612 return false;
1613 }
1614 }
1615
1616 // FastISel does not handle any operand bundles except OB_funclet.
1617 if (ImmutableCallSite CS = ImmutableCallSite(I))
1618 for (unsigned i = 0, e = CS.getNumOperandBundles(); i != e; ++i)
1619 if (CS.getOperandBundleAt(i).getTagID() != LLVMContext::OB_funclet)
1620 return false;
1621
1622 DbgLoc = I->getDebugLoc();
1623
1624 SavedInsertPt = FuncInfo.InsertPt;
1625
1626 if (const auto *Call = dyn_cast<CallInst>(I)) {
1627 const Function *F = Call->getCalledFunction();
1628 LibFunc Func;
1629
1630 // As a special case, don't handle calls to builtin library functions that
1631 // may be translated directly to target instructions.
1632 if (F && !F->hasLocalLinkage() && F->hasName() &&
1633 LibInfo->getLibFunc(F->getName(), Func) &&
1634 LibInfo->hasOptimizedCodeGen(Func))
1635 return false;
1636
1637 // Don't handle Intrinsic::trap if a trap function is specified.
1638 if (F && F->getIntrinsicID() == Intrinsic::trap &&
1639 Call->hasFnAttr("trap-func-name"))
1640 return false;
1641 }
1642
1643 // First, try doing target-independent selection.
1644 if (!SkipTargetIndependentISel) {
1645 if (selectOperator(I, I->getOpcode())) {
1646 ++NumFastIselSuccessIndependent;
1647 DbgLoc = DebugLoc();
1648 return true;
1649 }
1650 // Remove dead code.
1651 recomputeInsertPt();
1652 if (SavedInsertPt != FuncInfo.InsertPt)
1653 removeDeadCode(FuncInfo.InsertPt, SavedInsertPt);
1654 SavedInsertPt = FuncInfo.InsertPt;
1655 }
1656 // Next, try calling the target to attempt to handle the instruction.
1657 if (fastSelectInstruction(I)) {
1658 ++NumFastIselSuccessTarget;
1659 DbgLoc = DebugLoc();
1660 return true;
1661 }
1662 // Remove dead code.
1663 recomputeInsertPt();
1664 if (SavedInsertPt != FuncInfo.InsertPt)
1665 removeDeadCode(FuncInfo.InsertPt, SavedInsertPt);
1666
1667 DbgLoc = DebugLoc();
1668 // Undo phi node updates, because they will be added again by SelectionDAG.
1669 if (I->isTerminator()) {
1670 // PHI node handling may have generated local value instructions.
1671 // We remove them because SelectionDAGISel will generate them again.
1672 removeDeadLocalValueCode(SavedLastLocalValue);
1673 FuncInfo.PHINodesToUpdate.resize(FuncInfo.OrigNumPHINodesToUpdate);
1674 }
1675 return false;
1676}

1678/// Emit an unconditional branch to the given block, unless it is the immediate
1679/// (fall-through) successor, and update the CFG.
1680void FastISel::fastEmitBranch(MachineBasicBlock *MSucc,
1681 const DebugLoc &DbgLoc) {
1682 if (FuncInfo.MBB->getBasicBlock()->size() > 1 &&
1683 FuncInfo.MBB->isLayoutSuccessor(MSucc)) {
1684 // For more accurate line information if this is the only instruction
1685 // in the block then emit it, otherwise we have the unconditional
1686 // fall-through case, which needs no instructions.
1687 } else {
1688 // The unconditional branch case.
1689 TII.insertBranch(*FuncInfo.MBB, MSucc, nullptr,
1690 SmallVector<MachineOperand, 0>(), DbgLoc);
1691 }
1692 if (FuncInfo.BPI) {
1693 auto BranchProbability = FuncInfo.BPI->getEdgeProbability(
1694 FuncInfo.MBB->getBasicBlock(), MSucc->getBasicBlock());
1695 FuncInfo.MBB->addSuccessor(MSucc, BranchProbability);
1696 } else
1697 FuncInfo.MBB->addSuccessorWithoutProb(MSucc);
1698}

1700void FastISel::finishCondBranch(const BasicBlock *BranchBB,
1701 MachineBasicBlock *TrueMBB,
1702 MachineBasicBlock *FalseMBB) {
1703 // Add TrueMBB as successor unless it is equal to the FalseMBB: This can
1704 // happen in degenerate IR and MachineIR forbids to have a block twice in the
1705 // successor/predecessor lists.
1706 if (TrueMBB != FalseMBB) {
1707 if (FuncInfo.BPI) {
1708 auto BranchProbability =
1709 FuncInfo.BPI->getEdgeProbability(BranchBB, TrueMBB->getBasicBlock());
1710 FuncInfo.MBB->addSuccessor(TrueMBB, BranchProbability);
1711 } else
1712 FuncInfo.MBB->addSuccessorWithoutProb(TrueMBB);
1713 }
1714
1715 fastEmitBranch(FalseMBB, DbgLoc);
1716}

1718/// Emit an FNeg operation.
1719bool FastISel::selectFNeg(const User *I, const Value *In) {
1720 unsigned OpReg = getRegForValue(In);
1721 if (!OpReg)
1722 return false;
1723 bool OpRegIsKill = hasTrivialKill(In);
1724
1725 // If the target has ISD::FNEG, use it.
1726 EVT VT = TLI.getValueType(DL, I->getType());
1727 unsigned ResultReg = fastEmit_r(VT.getSimpleVT(), VT.getSimpleVT(), ISD::FNEG,
1728 OpReg, OpRegIsKill);
1729 if (ResultReg) {
1730 updateValueMap(I, ResultReg);
1731 return true;
1732 }
1733
1734 // Bitcast the value to integer, twiddle the sign bit with xor,
1735 // and then bitcast it back to floating-point.
1736 if (VT.getSizeInBits() > 64)
1737 return false;
1738 EVT IntVT = EVT::getIntegerVT(I->getContext(), VT.getSizeInBits());
1739 if (!TLI.isTypeLegal(IntVT))
1740 return false;
1741
1742 unsigned IntReg = fastEmit_r(VT.getSimpleVT(), IntVT.getSimpleVT(),
1743 ISD::BITCAST, OpReg, OpRegIsKill);
1744 if (!IntReg)
1745 return false;
1746
1747 unsigned IntResultReg = fastEmit_ri_(
1748 IntVT.getSimpleVT(), ISD::XOR, IntReg, /*IsKill=*/true,
1749 UINT64_C(1) << (VT.getSizeInBits() - 1), IntVT.getSimpleVT());
1750 if (!IntResultReg)
1751 return false;
1752
1753 ResultReg = fastEmit_r(IntVT.getSimpleVT(), VT.getSimpleVT(), ISD::BITCAST,
1754 IntResultReg, /*IsKill=*/true);
1755 if (!ResultReg)
1756 return false;
1757
1758 updateValueMap(I, ResultReg);
1759 return true;
1760}

1762bool FastISel::selectExtractValue(const User *U) {
1763 const ExtractValueInst *EVI = dyn_cast<ExtractValueInst>(U);
1764 if (!EVI)
1765 return false;
1766
1767 // Make sure we only try to handle extracts with a legal result. But also
1768 // allow i1 because it's easy.
1769 EVT RealVT = TLI.getValueType(DL, EVI->getType(), /*AllowUnknown=*/true);
1770 if (!RealVT.isSimple())
1771 return false;
1772 MVT VT = RealVT.getSimpleVT();
1773 if (!TLI.isTypeLegal(VT) && VT != MVT::i1)
1774 return false;
1775
1776 const Value *Op0 = EVI->getOperand(0);
1777 Type *AggTy = Op0->getType();
1778
1779 // Get the base result register.
1780 unsigned ResultReg;
1781 DenseMap<const Value *, unsigned>::iterator I = FuncInfo.ValueMap.find(Op0);
1782 if (I != FuncInfo.ValueMap.end())
1783 ResultReg = I->second;
1784 else if (isa<Instruction>(Op0))
1785 ResultReg = FuncInfo.InitializeRegForValue(Op0);
1786 else
1787 return false; // fast-isel can't handle aggregate constants at the moment
1788
1789 // Get the actual result register, which is an offset from the base register.
1790 unsigned VTIndex = ComputeLinearIndex(AggTy, EVI->getIndices());
1791
1792 SmallVector<EVT, 4> AggValueVTs;
1793 ComputeValueVTs(TLI, DL, AggTy, AggValueVTs);
1794
1795 for (unsigned i = 0; i < VTIndex; i++)
1796 ResultReg += TLI.getNumRegisters(FuncInfo.Fn->getContext(), AggValueVTs[i]);
1797
1798 updateValueMap(EVI, ResultReg);
1799 return true;

1802bool FastISel::selectOperator(const User *I, unsigned Opcode) {
1803 switch (Opcode) {
1804 case Instruction::Add:
1805 return selectBinaryOp(I, ISD::ADD);
1806 case Instruction::FAdd:
1807 return selectBinaryOp(I, ISD::FADD);
1808 case Instruction::Sub:
1809 return selectBinaryOp(I, ISD::SUB);
1810 case Instruction::FSub: {
1811 // FNeg is currently represented in LLVM IR as a special case of FSub.
1812 Value *X;
1813 if (match(I, m_FNeg(m_Value(X))))
1814 return selectFNeg(I, X);
1815 return selectBinaryOp(I, ISD::FSUB);
1816 }
1817 case Instruction::Mul:
1818 return selectBinaryOp(I, ISD::MUL);
1819 case Instruction::FMul:
1820 return selectBinaryOp(I, ISD::FMUL);
1821 case Instruction::SDiv:
1822 return selectBinaryOp(I, ISD::SDIV);
1823 case Instruction::UDiv:
1824 return selectBinaryOp(I, ISD::UDIV);
1825 case Instruction::FDiv:
1826 return selectBinaryOp(I, ISD::FDIV);
1827 case Instruction::SRem:
1828 return selectBinaryOp(I, ISD::SREM);
1829 case Instruction::URem:
1830 return selectBinaryOp(I, ISD::UREM);
1831 case Instruction::FRem:
1832 return selectBinaryOp(I, ISD::FREM);
1833 case Instruction::Shl:
1834 return selectBinaryOp(I, ISD::SHL);
1835 case Instruction::LShr:
1836 return selectBinaryOp(I, ISD::SRL);
1837 case Instruction::AShr:
1838 return selectBinaryOp(I, ISD::SRA);
1839 case Instruction::And:
1840 return selectBinaryOp(I, ISD::AND);
1841 case Instruction::Or:
1842 return selectBinaryOp(I, ISD::OR);
1843 case Instruction::Xor:
1844 return selectBinaryOp(I, ISD::XOR);
1845
1846 case Instruction::FNeg:
1847 return selectFNeg(I, I->getOperand(0));
1848
1849 case Instruction::GetElementPtr:
1850 return selectGetElementPtr(I);
1851
1852 case Instruction::Br: {
1853 const BranchInst *BI = cast<BranchInst>(I);
1854
1855 if (BI->isUnconditional()) {
1856 const BasicBlock *LLVMSucc = BI->getSuccessor(0);
1857 MachineBasicBlock *MSucc = FuncInfo.MBBMap[LLVMSucc];
1858 fastEmitBranch(MSucc, BI->getDebugLoc());
1859 return true;
1860 }
1861
1862 // Conditional branches are not handed yet.
1863 // Halt "fast" selection and bail.
1864 return false;
1865 }
1866
1867 case Instruction::Unreachable:
1868 if (TM.Options.TrapUnreachable)
1869 return fastEmit_(MVT::Other, MVT::Other, ISD::TRAP) != 0;
1870 else
1871 return true;
1872
1873 case Instruction::Alloca:
1874 // FunctionLowering has the static-sized case covered.
1875 if (FuncInfo.StaticAllocaMap.count(cast<AllocaInst>(I)))
1876 return true;
1877
1878 // Dynamic-sized alloca is not handled yet.
1879 return false;
1880
1881 case Instruction::Call:
1882 // On AIX, call lowering uses the DAG-ISEL path currently so that the
1883 // callee of the direct function call instruction will be mapped to the
1884 // symbol for the function's entry point, which is distinct from the
1885 // function descriptor symbol. The latter is the symbol whose XCOFF symbol
1886 // name is the C-linkage name of the source level function.
1887 if (TM.getTargetTriple().isOSAIX())
1888 return false;
1889 return selectCall(I);
1890
1891 case Instruction::BitCast:
1892 return selectBitCast(I);
1893
1894 case Instruction::FPToSI:
1895 return selectCast(I, ISD::FP_TO_SINT);
1896 case Instruction::ZExt:
1897 return selectCast(I, ISD::ZERO_EXTEND);
1898 case Instruction::SExt:
1899 return selectCast(I, ISD::SIGN_EXTEND);
1900 case Instruction::Trunc:
1901 return selectCast(I, ISD::TRUNCATE);
1902 case Instruction::SIToFP:
1903 return selectCast(I, ISD::SINT_TO_FP);
1904
1905 case Instruction::IntToPtr: // Deliberate fall-through.
1906 case Instruction::PtrToInt: {
1907 EVT SrcVT = TLI.getValueType(DL, I->getOperand(0)->getType());
1908 EVT DstVT = TLI.getValueType(DL, I->getType());
1909 if (DstVT.bitsGT(SrcVT))
1910 return selectCast(I, ISD::ZERO_EXTEND);
1911 if (DstVT.bitsLT(SrcVT))
1912 return selectCast(I, ISD::TRUNCATE);
1913 unsigned Reg = getRegForValue(I->getOperand(0));
1914 if (!Reg)
1915 return false;
1916 updateValueMap(I, Reg);
1917 return true;
1918 }
1919
1920 case Instruction::ExtractValue:
1921 return selectExtractValue(I);
1922
1923 case Instruction::PHI:
1924 llvm_unreachable("FastISel shouldn't visit PHI nodes!");
1925
1926 default:
1927 // Unhandled instruction. Halt "fast" selection and bail.
1928 return false;
1929 }

1980/// This method is a wrapper of fastEmit_ri. It first tries to emit an
1981/// instruction with an immediate operand using fastEmit_ri.
1982/// If that fails, it materializes the immediate into a register and try
1983/// fastEmit_rr instead.
1984unsigned FastISel::fastEmit_ri_(MVT VT, unsigned Opcode, unsigned Op0,
1985 bool Op0IsKill, uint64_t Imm, MVT ImmType) {
1986 // If this is a multiply by a power of two, emit this as a shift left.
1987 if (Opcode == ISD::MUL && isPowerOf2_64(Imm)) {
1988 Opcode = ISD::SHL;
1989 Imm = Log2_64(Imm);
1990 } else if (Opcode == ISD::UDIV && isPowerOf2_64(Imm)) {
1991 // div x, 8 -> srl x, 3
1992 Opcode = ISD::SRL;
1993 Imm = Log2_64(Imm);
1994 }
1995
1996 // Horrible hack (to be removed), check to make sure shift amounts are
1997 // in-range.
1998 if ((Opcode == ISD::SHL || Opcode == ISD::SRA || Opcode == ISD::SRL) &&
1999 Imm >= VT.getSizeInBits())
2000 return 0;
2001
2002 // First check if immediate type is legal. If not, we can't use the ri form.
2003 unsigned ResultReg = fastEmit_ri(VT, VT, Opcode, Op0, Op0IsKill, Imm);
2004 if (ResultReg)
2005 return ResultReg;
2006 unsigned MaterialReg = fastEmit_i(ImmType, ImmType, ISD::Constant, Imm);
2007 bool IsImmKill = true;
2008 if (!MaterialReg) {
2009 // This is a bit ugly/slow, but failing here means falling out of
2010 // fast-isel, which would be very slow.
2011 IntegerType *ITy =
2012 IntegerType::get(FuncInfo.Fn->getContext(), VT.getSizeInBits());
2013 MaterialReg = getRegForValue(ConstantInt::get(ITy, Imm));
2014 if (!MaterialReg)
2015 return 0;
2016 // FIXME: If the materialized register here has no uses yet then this
2017 // will be the first use and we should be able to mark it as killed.
2018 // However, the local value area for materialising constant expressions
2019 // grows down, not up, which means that any constant expressions we generate
2020 // later which also use 'Imm' could be after this instruction and therefore
2021 // after this kill.
2022 IsImmKill = false;
2023 }
2024 return fastEmit_rr(VT, VT, Opcode, Op0, Op0IsKill, MaterialReg, IsImmKill);
2025}

2256/// HandlePHINodesInSuccessorBlocks - Handle PHI nodes in successor blocks.
2257/// Emit code to ensure constants are copied into registers when needed.
2258/// Remember the virtual registers that need to be added to the Machine PHI
2259/// nodes as input. We cannot just directly add them, because expansion
2260/// might result in multiple MBB's for one BB. As such, the start of the
2261/// BB might correspond to a different MBB than the end.
2262bool FastISel::handlePHINodesInSuccessorBlocks(const BasicBlock *LLVMBB) {
2263 const Instruction *TI = LLVMBB->getTerminator();
2264
2265 SmallPtrSet<MachineBasicBlock *, 4> SuccsHandled;
2266 FuncInfo.OrigNumPHINodesToUpdate = FuncInfo.PHINodesToUpdate.size();
2267
2268 // Check successor nodes' PHI nodes that expect a constant to be available
2269 // from this block.
2270 for (unsigned succ = 0, e = TI->getNumSuccessors(); succ != e; ++succ) {
2271 const BasicBlock *SuccBB = TI->getSuccessor(succ);
2272 if (!isa<PHINode>(SuccBB->begin()))
2273 continue;
2274 MachineBasicBlock *SuccMBB = FuncInfo.MBBMap[SuccBB];
2275
2276 // If this terminator has multiple identical successors (common for
2277 // switches), only handle each succ once.
2278 if (!SuccsHandled.insert(SuccMBB).second)
2279 continue;
2280
2281 MachineBasicBlock::iterator MBBI = SuccMBB->begin();
2282
2283 // At this point we know that there is a 1-1 correspondence between LLVM PHI
2284 // nodes and Machine PHI nodes, but the incoming operands have not been
2285 // emitted yet.
2286 for (const PHINode &PN : SuccBB->phis()) {
2287 // Ignore dead phi's.
2288 if (PN.use_empty())
2289 continue;
2290
2291 // Only handle legal types. Two interesting things to note here. First,
2292 // by bailing out early, we may leave behind some dead instructions,
2293 // since SelectionDAG's HandlePHINodesInSuccessorBlocks will insert its
2294 // own moves. Second, this check is necessary because FastISel doesn't
2295 // use CreateRegs to create registers, so it always creates
2296 // exactly one register for each non-void instruction.
2297 EVT VT = TLI.getValueType(DL, PN.getType(), /*AllowUnknown=*/true);
2298 if (VT == MVT::Other || !TLI.isTypeLegal(VT)) {
2299 // Handle integer promotions, though, because they're common and easy.
2300 if (!(VT == MVT::i1 || VT == MVT::i8 || VT == MVT::i16)) {
2301 FuncInfo.PHINodesToUpdate.resize(FuncInfo.OrigNumPHINodesToUpdate);
2302 return false;
2303 }
2304 }
2305
2306 const Value *PHIOp = PN.getIncomingValueForBlock(LLVMBB);
2307
2308 // Set the DebugLoc for the copy. Prefer the location of the operand
2309 // if there is one; use the location of the PHI otherwise.
2310 DbgLoc = PN.getDebugLoc();
2311 if (const auto *Inst = dyn_cast<Instruction>(PHIOp))
2312 DbgLoc = Inst->getDebugLoc();
2313
2314 unsigned Reg = getRegForValue(PHIOp);
2315 if (!Reg) {
2316 FuncInfo.PHINodesToUpdate.resize(FuncInfo.OrigNumPHINodesToUpdate);
2317 return false;
2318 }
2319 FuncInfo.PHINodesToUpdate.push_back(std::make_pair(&*MBBI++, Reg));
2320 DbgLoc = DebugLoc();
2321 }
2322 }
2323
2324 return true;
2325}

2327bool FastISel::tryToFoldLoad(const LoadInst *LI, const Instruction *FoldInst) {
2328 assert(LI->hasOneUse() &&
2329 "tryToFoldLoad expected a LoadInst with a single use");
2330 // We know that the load has a single use, but don't know what it is. If it
2331 // isn't one of the folded instructions, then we can't succeed here. Handle
2332 // this by scanning the single-use users of the load until we get to FoldInst.
2333 unsigned MaxUsers = 6; // Don't scan down huge single-use chains of instrs.
2334
2335 const Instruction *TheUser = LI->user_back();
2336 while (TheUser != FoldInst && // Scan up until we find FoldInst.
2337 // Stay in the right block.
2338 TheUser->getParent() == FoldInst->getParent() &&
2339 --MaxUsers) { // Don't scan too far.
2340 // If there are multiple or no uses of this instruction, then bail out.
2341 if (!TheUser->hasOneUse())
2342 return false;
2343
2344 TheUser = TheUser->user_back();
2345 }
2346
2347 // If we didn't find the fold instruction, then we failed to collapse the
2348 // sequence.
2349 if (TheUser != FoldInst)
2350 return false;
2351
2352 // Don't try to fold volatile loads. Target has to deal with alignment
2353 // constraints.
2354 if (LI->isVolatile())
2355 return false;
2356
2357 // Figure out which vreg this is going into. If there is no assigned vreg yet
2358 // then there actually was no reference to it. Perhaps the load is referenced
2359 // by a dead instruction.
2360 unsigned LoadReg = getRegForValue(LI);
2361 if (!LoadReg)
2362 return false;
2363
2364 // We can't fold if this vreg has no uses or more than one use. Multiple uses
2365 // may mean that the instruction got lowered to multiple MIs, or the use of
2366 // the loaded value ended up being multiple operands of the result.
2367 if (!MRI.hasOneUse(LoadReg))
2368 return false;
2369
2370 MachineRegisterInfo::reg_iterator RI = MRI.reg_begin(LoadReg);
2371 MachineInstr *User = RI->getParent();
2372
2373 // Set the insertion point properly. Folding the load can cause generation of
2374 // other random instructions (like sign extends) for addressing modes; make
2375 // sure they get inserted in a logical place before the new instruction.
2376 FuncInfo.InsertPt = User;
2377 FuncInfo.MBB = User->getParent();
2378
2379 // Ask the target to try folding the load.
2380 return tryToFoldLoadIntoMI(User, RI.getOperandNo(), LI);
2381}

2383bool FastISel::canFoldAddIntoGEP(const User *GEP, const Value *Add) {
2384 // Must be an add.
2385 if (!isa<AddOperator>(Add))
2386 return false;
2387 // Type size needs to match.
2388 if (DL.getTypeSizeInBits(GEP->getType()) !=
2389 DL.getTypeSizeInBits(Add->getType()))
2390 return false;
2391 // Must be in the same basic block.
2392 if (isa<Instruction>(Add) &&
2393 FuncInfo.MBBMap[cast<Instruction>(Add)->getParent()] != FuncInfo.MBB)
2394 return false;
2395 // Must have a constant operand.
2396 return isa<ConstantInt>(cast<AddOperator>(Add)->getOperand(1));
2397}

2399MachineMemOperand *
2400FastISel::createMachineMemOperandFor(const Instruction *I) const {
2401 const Value *Ptr;
2402 Type *ValTy;
2403 unsigned Alignment;
2404 MachineMemOperand::Flags Flags;
2405 bool IsVolatile;
2406
2407 if (const auto *LI = dyn_cast<LoadInst>(I)) {
2408 Alignment = LI->getAlignment();
2409 IsVolatile = LI->isVolatile();
2410 Flags = MachineMemOperand::MOLoad;
2411 Ptr = LI->getPointerOperand();
2412 ValTy = LI->getType();
2413 } else if (const auto *SI = dyn_cast<StoreInst>(I)) {
2414 Alignment = SI->getAlignment();
2415 IsVolatile = SI->isVolatile();
2416 Flags = MachineMemOperand::MOStore;
2417 Ptr = SI->getPointerOperand();
2418 ValTy = SI->getValueOperand()->getType();
2419 } else
2420 return nullptr;
2421
2422 bool IsNonTemporal = I->getMetadata(LLVMContext::MD_nontemporal) != nullptr;
2423 bool IsInvariant = I->getMetadata(LLVMContext::MD_invariant_load) != nullptr;
2424 bool IsDereferenceable =
2425 I->getMetadata(LLVMContext::MD_dereferenceable) != nullptr;
2426 const MDNode *Ranges = I->getMetadata(LLVMContext::MD_range);
2427
2428 AAMDNodes AAInfo;
2429 I->getAAMetadata(AAInfo);
2430
2431 if (Alignment == 0) // Ensure that codegen never sees alignment 0.
2432 Alignment = DL.getABITypeAlignment(ValTy);
2433
2434 unsigned Size = DL.getTypeStoreSize(ValTy);
2435
2436 if (IsVolatile)
2437 Flags |= MachineMemOperand::MOVolatile;
2438 if (IsNonTemporal)
2439 Flags |= MachineMemOperand::MONonTemporal;
2440 if (IsDereferenceable)
2441 Flags |= MachineMemOperand::MODereferenceable;
2442 if (IsInvariant)
2443 Flags |= MachineMemOperand::MOInvariant;
2444
2445 return FuncInfo.MF->getMachineMemOperand(MachinePointerInfo(Ptr), Flags, Size,
2446 Alignment, AAInfo, Ranges);
2447}

2449CmpInst::Predicate FastISel::optimizeCmpPredicate(const CmpInst *CI) const {
2450 // If both operands are the same, then try to optimize or fold the cmp.
2451 CmpInst::Predicate Predicate = CI->getPredicate();
2452 if (CI->getOperand(0) != CI->getOperand(1))
2453 return Predicate;
2454
2455 switch (Predicate) {
2456 default: llvm_unreachable("Invalid predicate!");
2457 case CmpInst::FCMP_FALSE: Predicate = CmpInst::FCMP_FALSE; break;
2458 case CmpInst::FCMP_OEQ: Predicate = CmpInst::FCMP_ORD; break;
2459 case CmpInst::FCMP_OGT: Predicate = CmpInst::FCMP_FALSE; break;
2460 case CmpInst::FCMP_OGE: Predicate = CmpInst::FCMP_ORD; break;
2461 case CmpInst::FCMP_OLT: Predicate = CmpInst::FCMP_FALSE; break;
2462 case CmpInst::FCMP_OLE: Predicate = CmpInst::FCMP_ORD; break;
2463 case CmpInst::FCMP_ONE: Predicate = CmpInst::FCMP_FALSE; break;
2464 case CmpInst::FCMP_ORD: Predicate = CmpInst::FCMP_ORD; break;
2465 case CmpInst::FCMP_UNO: Predicate = CmpInst::FCMP_UNO; break;
2466 case CmpInst::FCMP_UEQ: Predicate = CmpInst::FCMP_TRUE; break;
2467 case CmpInst::FCMP_UGT: Predicate = CmpInst::FCMP_UNO; break;
2468 case CmpInst::FCMP_UGE: Predicate = CmpInst::FCMP_TRUE; break;
2469 case CmpInst::FCMP_ULT: Predicate = CmpInst::FCMP_UNO; break;
2470 case CmpInst::FCMP_ULE: Predicate = CmpInst::FCMP_TRUE; break;
2471 case CmpInst::FCMP_UNE: Predicate = CmpInst::FCMP_UNO; break;
2472 case CmpInst::FCMP_TRUE: Predicate = CmpInst::FCMP_TRUE; break;
2473
2474 case CmpInst::ICMP_EQ: Predicate = CmpInst::FCMP_TRUE; break;
2475 case CmpInst::ICMP_NE: Predicate = CmpInst::FCMP_FALSE; break;
2476 case CmpInst::ICMP_UGT: Predicate = CmpInst::FCMP_FALSE; break;
2477 case CmpInst::ICMP_UGE: Predicate = CmpInst::FCMP_TRUE; break;
2478 case CmpInst::ICMP_ULT: Predicate = CmpInst::FCMP_FALSE; break;
2479 case CmpInst::ICMP_ULE: Predicate = CmpInst::FCMP_TRUE; break;
2480 case CmpInst::ICMP_SGT: Predicate = CmpInst::FCMP_FALSE; break;
2481 case CmpInst::ICMP_SGE: Predicate = CmpInst::FCMP_TRUE; break;
2482 case CmpInst::ICMP_SLT: Predicate = CmpInst::FCMP_FALSE; break;
