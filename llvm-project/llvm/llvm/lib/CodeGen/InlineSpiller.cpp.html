251// When spilling a virtual register, we also spill any snippets it is connected
252// to. The snippets are small live ranges that only have a single real use,
253// leftovers from live range splitting. Spilling them enables memory operand
254// folding or tightens the live range around the single use.
255//
256// This minimizes register pressure and maximizes the store-to-load distance for
257// spill slots which can be important in tight loops.
258
259/// isFullCopyOf - If MI is a COPY to or from Reg, return the other register,
260/// otherwise return 0.
261static unsigned isFullCopyOf(const MachineInstr &MI, unsigned Reg) {
262 if (!MI.isFullCopy())
263 return 0;
264 if (MI.getOperand(0).getReg() == Reg)
265 return MI.getOperand(1).getReg();
266 if (MI.getOperand(1).getReg() == Reg)
267 return MI.getOperand(0).getReg();
268 return 0;
269}

271/// isSnippet - Identify if a live interval is a snippet that should be spilled.
272/// It is assumed that SnipLI is a virtual register with the same original as
273/// Edit->getReg().
274bool InlineSpiller::isSnippet(const LiveInterval &SnipLI) {
275 unsigned Reg = Edit->getReg();
276
277 // A snippet is a tiny live range with only a single instruction using it
278 // besides copies to/from Reg or spills/fills. We accept:
279 //
280 // %snip = COPY %Reg / FILL fi#
281 // %snip = USE %snip
282 // %Reg = COPY %snip / SPILL %snip, fi#
283 //
284 if (SnipLI.getNumValNums() > 2 || !LIS.intervalIsInOneMBB(SnipLI))
285 return false;
286
287 MachineInstr *UseMI = nullptr;
288
289 // Check that all uses satisfy our criteria.
290 for (MachineRegisterInfo::reg_instr_nodbg_iterator
291 RI = MRI.reg_instr_nodbg_begin(SnipLI.reg),
292 E = MRI.reg_instr_nodbg_end(); RI != E; ) {
293 MachineInstr &MI = *RI++;
294
295 // Allow copies to/from Reg.
296 if (isFullCopyOf(MI, Reg))
297 continue;
298
299 // Allow stack slot loads.
300 int FI;
301 if (SnipLI.reg == TII.isLoadFromStackSlot(MI, FI) && FI == StackSlot)
302 continue;
303
304 // Allow stack slot stores.
305 if (SnipLI.reg == TII.isStoreToStackSlot(MI, FI) && FI == StackSlot)
306 continue;
307
308 // Allow a single additional instruction.
309 if (UseMI && &MI != UseMI)
310 return false;
311 UseMI = &MI;
312 }
313 return true;
314}

316/// collectRegsToSpill - Collect live range snippets that only have a single
317/// real use.
318void InlineSpiller::collectRegsToSpill() {
319 unsigned Reg = Edit->getReg();
320
321 // Main register always spills.
322 RegsToSpill.assign(1, Reg);
323 SnippetCopies.clear();
324
325 // Snippets all have the same original, so there can't be any for an original
326 // register.
327 if (Original == Reg)
328 return;
329
330 for (MachineRegisterInfo::reg_instr_iterator
331 RI = MRI.reg_instr_begin(Reg), E = MRI.reg_instr_end(); RI != E; ) {
332 MachineInstr &MI = *RI++;
333 unsigned SnipReg = isFullCopyOf(MI, Reg);
334 if (!isSibling(SnipReg))
335 continue;
336 LiveInterval &SnipLI = LIS.getInterval(SnipReg);
337 if (!isSnippet(SnipLI))
338 continue;
339 SnippetCopies.insert(&MI);
340 if (isRegToSpill(SnipReg))
341 continue;
342 RegsToSpill.push_back(SnipReg);
343 LLVM_DEBUG(dbgs() << "\talso spill snippet " << SnipLI << '\n');
344 ++NumSnippets;
345 }
346}

353/// It is beneficial to spill to earlier place in the same BB in case
354/// as follows:
355/// There is an alternative def earlier in the same MBB.
356/// Hoist the spill as far as possible in SpillMBB. This can ease
357/// register pressure:
358///
359/// x = def
360/// y = use x
361/// s = copy x
362///
363/// Hoisting the spill of s to immediately after the def removes the
364/// interference between x and y:
365///
366/// x = def
367/// spill x
368/// y = use killed x
369///
370/// This hoist only helps when the copy kills its source.
371///
372bool InlineSpiller::hoistSpillInsideBB(LiveInterval &SpillLI,
373 MachineInstr &CopyMI) {
374 SlotIndex Idx = LIS.getInstructionIndex(CopyMI);
375#ifndef NDEBUG
376 VNInfo *VNI = SpillLI.getVNInfoAt(Idx.getRegSlot());
377 assert(VNI && VNI->def == Idx.getRegSlot() && "Not defined by copy");
378#endif
379
380 unsigned SrcReg = CopyMI.getOperand(1).getReg();
381 LiveInterval &SrcLI = LIS.getInterval(SrcReg);
382 VNInfo *SrcVNI = SrcLI.getVNInfoAt(Idx);
383 LiveQueryResult SrcQ = SrcLI.Query(Idx);
384 MachineBasicBlock *DefMBB = LIS.getMBBFromIndex(SrcVNI->def);
385 if (DefMBB != CopyMI.getParent() || !SrcQ.isKill())
386 return false;
387
388 // Conservatively extend the stack slot range to the range of the original
389 // value. We may be able to do better with stack slot coloring by being more
390 // careful here.
391 assert(StackInt && "No stack slot assigned yet.");
392 LiveInterval &OrigLI = LIS.getInterval(Original);
393 VNInfo *OrigVNI = OrigLI.getVNInfoAt(Idx);
394 StackInt->MergeValueInAsValue(OrigLI, OrigVNI, StackInt->getValNumInfo(0));
395 LLVM_DEBUG(dbgs() << "\tmerged orig valno " << OrigVNI->id << ": "
396 << *StackInt << '\n');
397
398 // We are going to spill SrcVNI immediately after its def, so clear out
399 // any later spills of the same value.
400 eliminateRedundantSpills(SrcLI, SrcVNI);
401
402 MachineBasicBlock *MBB = LIS.getMBBFromIndex(SrcVNI->def);
403 MachineBasicBlock::iterator MII;
404 if (SrcVNI->isPHIDef())
405 MII = MBB->SkipPHIsLabelsAndDebug(MBB->begin());
406 else {
407 MachineInstr *DefMI = LIS.getInstructionFromIndex(SrcVNI->def);
408 assert(DefMI && "Defining instruction disappeared");
409 MII = DefMI;
410 ++MII;
411 }
412 // Insert spill without kill flag immediately after def.
413 TII.storeRegToStackSlot(*MBB, MII, SrcReg, false, StackSlot,
414 MRI.getRegClass(SrcReg), &TRI);
415 --MII; // Point to store instruction.
416 LIS.InsertMachineInstrInMaps(*MII);
417#if 0
418 // CHERI, with hardware floating point, lacks instructions for storing
419 // floating point via a capability. As such, the storeRegToStackSlot call
420 // will insert two instructions and we must update the maps for both of them.
421 if (MII != MBB->begin()) {
422 --MII; // Point to the instruction before the store instruction.
423 if (LIS.isNotInMIMap(*MII))
424 LIS.InsertMachineInstrInMaps(*MII);
425 }
426#endif
427 LLVM_DEBUG(dbgs() << "\thoisted: " << SrcVNI->def << '\t' << *MII);
428
429 HSpiller.addToMergeableSpills(*MII, StackSlot, Original);
430 ++NumSpills;
431 return true;
432}

434/// eliminateRedundantSpills - SLI:VNI is known to be on the stack. Remove any
435/// redundant spills of this value in SLI.reg and sibling copies.
436void InlineSpiller::eliminateRedundantSpills(LiveInterval &SLI, VNInfo *VNI) {
437 assert(VNI && "Missing value");
438 SmallVector<std::pair<LiveInterval*, VNInfo*>, 8> WorkList;
439 WorkList.push_back(std::make_pair(&SLI, VNI));
440 assert(StackInt && "No stack slot assigned yet.");
441
442 do {
443 LiveInterval *LI;
444 std::tie(LI, VNI) = WorkList.pop_back_val();
445 unsigned Reg = LI->reg;
446 LLVM_DEBUG(dbgs() << "Checking redundant spills for " << VNI->id << '@'
447 << VNI->def << " in " << *LI << '\n');
448
449 // Regs to spill are taken care of.
450 if (isRegToSpill(Reg))
451 continue;
452
453 // Add all of VNI's live range to StackInt.
454 StackInt->MergeValueInAsValue(*LI, VNI, StackInt->getValNumInfo(0));
455 LLVM_DEBUG(dbgs() << "Merged to stack int: " << *StackInt << '\n');
456
457 // Find all spills and copies of VNI.
458 for (MachineRegisterInfo::use_instr_nodbg_iterator
459 UI = MRI.use_instr_nodbg_begin(Reg), E = MRI.use_instr_nodbg_end();
460 UI != E; ) {
461 MachineInstr &MI = *UI++;
462 if (!MI.isCopy() && !MI.mayStore())
463 continue;
464 SlotIndex Idx = LIS.getInstructionIndex(MI);
465 if (LI->getVNInfoAt(Idx) != VNI)
466 continue;
467
468 // Follow sibling copies down the dominator tree.
469 if (unsigned DstReg = isFullCopyOf(MI, Reg)) {
470 if (isSibling(DstReg)) {
471 LiveInterval &DstLI = LIS.getInterval(DstReg);
472 VNInfo *DstVNI = DstLI.getVNInfoAt(Idx.getRegSlot());
473 assert(DstVNI && "Missing defined value");
474 assert(DstVNI->def == Idx.getRegSlot() && "Wrong copy def slot");
475 WorkList.push_back(std::make_pair(&DstLI, DstVNI));
476 }
477 continue;
478 }
479
480 // Erase spills.
481 int FI;
482 if (Reg == TII.isStoreToStackSlot(MI, FI) && FI == StackSlot) {
483 LLVM_DEBUG(dbgs() << "Redundant spill " << Idx << '\t' << MI);
484 // eliminateDeadDefs won't normally remove stores, so switch opcode.
485 MI.setDesc(TII.get(TargetOpcode::KILL));
486 DeadDefs.push_back(&MI);
487 ++NumSpillsRemoved;
488 if (HSpiller.rmFromMergeableSpills(MI, StackSlot))
489 --NumSpills;
490 }
491 }
492 } while (!WorkList.empty());
493}

499/// markValueUsed - Remember that VNI failed to rematerialize, so its defining
500/// instruction cannot be eliminated. See through snippet copies
501void InlineSpiller::markValueUsed(LiveInterval *LI, VNInfo *VNI) {
502 SmallVector<std::pair<LiveInterval*, VNInfo*>, 8> WorkList;
503 WorkList.push_back(std::make_pair(LI, VNI));
504 do {
505 std::tie(LI, VNI) = WorkList.pop_back_val();
506 if (!UsedValues.insert(VNI).second)
507 continue;
508
509 if (VNI->isPHIDef()) {
510 MachineBasicBlock *MBB = LIS.getMBBFromIndex(VNI->def);
511 for (MachineBasicBlock *P : MBB->predecessors()) {
512 VNInfo *PVNI = LI->getVNInfoBefore(LIS.getMBBEndIdx(P));
513 if (PVNI)
514 WorkList.push_back(std::make_pair(LI, PVNI));
515 }
516 continue;
517 }
518
519 // Follow snippet copies.
520 MachineInstr *MI = LIS.getInstructionFromIndex(VNI->def);
521 if (!SnippetCopies.count(MI))
522 continue;
523 LiveInterval &SnipLI = LIS.getInterval(MI->getOperand(1).getReg());
524 assert(isRegToSpill(SnipLI.reg) && "Unexpected register in copy");
525 VNInfo *SnipVNI = SnipLI.getVNInfoAt(VNI->def.getRegSlot(true));
526 assert(SnipVNI && "Snippet undefined before copy");
527 WorkList.push_back(std::make_pair(&SnipLI, SnipVNI));
528 } while (!WorkList.empty());
529}

531bool InlineSpiller::canGuaranteeAssignmentAfterRemat(unsigned VReg,
532 MachineInstr &MI) {
533 if (!RestrictStatepointRemat)
534 return true;
535 // Here's a quick explanation of the problem we're trying to handle here:
536 // * There are some pseudo instructions with more vreg uses than there are
537 // physical registers on the machine.
538 // * This is normally handled by spilling the vreg, and folding the reload
539 // into the user instruction. (Thus decreasing the number of used vregs
540 // until the remainder can be assigned to physregs.)
541 // * However, since we may try to spill vregs in any order, we can end up
542 // trying to spill each operand to the instruction, and then rematting it
543 // instead. When that happens, the new live intervals (for the remats) are
544 // expected to be trivially assignable (i.e. RS_Done). However, since we
545 // may have more remats than physregs, we're guaranteed to fail to assign
546 // one.
547 // At the moment, we only handle this for STATEPOINTs since they're the only
548 // psuedo op where we've seen this. If we start seeing other instructions
549 // with the same problem, we need to revisit this.
550 return (MI.getOpcode() != TargetOpcode::STATEPOINT);
551}

553/// reMaterializeFor - Attempt to rematerialize before MI instead of reloading.
554bool InlineSpiller::reMaterializeFor(LiveInterval &VirtReg, MachineInstr &MI) {
555 // Analyze instruction
556 SmallVector<std::pair<MachineInstr *, unsigned>, 8> Ops;
557 MIBundleOperands::VirtRegInfo RI =
558 MIBundleOperands(MI).analyzeVirtReg(VirtReg.reg, &Ops);
559
560 if (!RI.Reads)
561 return false;
562
563 SlotIndex UseIdx = LIS.getInstructionIndex(MI).getRegSlot(true);
564 VNInfo *ParentVNI = VirtReg.getVNInfoAt(UseIdx.getBaseIndex());
565
566 if (!ParentVNI) {
567 LLVM_DEBUG(dbgs() << "\tadding <undef> flags: ");
568 for (unsigned i = 0, e = MI.getNumOperands(); i != e; ++i) {
569 MachineOperand &MO = MI.getOperand(i);
570 if (MO.isReg() && MO.isUse() && MO.getReg() == VirtReg.reg)
571 MO.setIsUndef();
572 }
573 LLVM_DEBUG(dbgs() << UseIdx << '\t' << MI);
574 return true;
575 }
576
577 if (SnippetCopies.count(&MI))
578 return false;
579
580 LiveInterval &OrigLI = LIS.getInterval(Original);
581 VNInfo *OrigVNI = OrigLI.getVNInfoAt(UseIdx);
582 LiveRangeEdit::Remat RM(ParentVNI);
583 RM.OrigMI = LIS.getInstructionFromIndex(OrigVNI->def);
584
585 if (!Edit->canRematerializeAt(RM, OrigVNI, UseIdx, false)) {
586 markValueUsed(&VirtReg, ParentVNI);
587 LLVM_DEBUG(dbgs() << "\tcannot remat for " << UseIdx << '\t' << MI);
588 return false;
589 }
590
591 // If the instruction also writes VirtReg.reg, it had better not require the
592 // same register for uses and defs.
593 if (RI.Tied) {
594 markValueUsed(&VirtReg, ParentVNI);
595 LLVM_DEBUG(dbgs() << "\tcannot remat tied reg: " << UseIdx << '\t' << MI);
596 return false;
597 }
598
599 // Before rematerializing into a register for a single instruction, try to
600 // fold a load into the instruction. That avoids allocating a new register.
601 if (RM.OrigMI->canFoldAsLoad() &&
602 foldMemoryOperand(Ops, RM.OrigMI)) {
603 Edit->markRematerialized(RM.ParentVNI);
604 ++NumFoldedLoads;
605 return true;
606 }
607
608 // If we can't guarantee that we'll be able to actually assign the new vreg,
609 // we can't remat.
610 if (!canGuaranteeAssignmentAfterRemat(VirtReg.reg, MI)) {
611 markValueUsed(&VirtReg, ParentVNI);
612 LLVM_DEBUG(dbgs() << "\tcannot remat for " << UseIdx << '\t' << MI);
613 return false;
614 }
615
616 // Allocate a new register for the remat.
617 unsigned NewVReg = Edit->createFrom(Original);
618
619 // Finally we can rematerialize OrigMI before MI.
620 SlotIndex DefIdx =
621 Edit->rematerializeAt(*MI.getParent(), MI, NewVReg, RM, TRI);
622
623 // We take the DebugLoc from MI, since OrigMI may be attributed to a
624 // different source location.
625 auto *NewMI = LIS.getInstructionFromIndex(DefIdx);
626 NewMI->setDebugLoc(MI.getDebugLoc());
627
628 (void)DefIdx;
629 LLVM_DEBUG(dbgs() << "\tremat: " << DefIdx << '\t'
630 << *LIS.getInstructionFromIndex(DefIdx));
631
632 // Replace operands
633 for (const auto &OpPair : Ops) {
634 MachineOperand &MO = OpPair.first->getOperand(OpPair.second);
635 if (MO.isReg() && MO.isUse() && MO.getReg() == VirtReg.reg) {
636 MO.setReg(NewVReg);
637 MO.setIsKill();
638 }
639 }
640 LLVM_DEBUG(dbgs() << "\t " << UseIdx << '\t' << MI << '\n');
641
642 ++NumRemats;
643 return true;
644}

646/// reMaterializeAll - Try to rematerialize as many uses as possible,
647/// and trim the live ranges after.
648void InlineSpiller::reMaterializeAll() {
649 if (!Edit->anyRematerializable(AA))
650 return;
651
652 UsedValues.clear();
653
654 // Try to remat before all uses of snippets.
655 bool anyRemat = false;
656 for (unsigned Reg : RegsToSpill) {
657 LiveInterval &LI = LIS.getInterval(Reg);
658 for (MachineRegisterInfo::reg_bundle_iterator
659 RegI = MRI.reg_bundle_begin(Reg), E = MRI.reg_bundle_end();
660 RegI != E; ) {
661 MachineInstr &MI = *RegI++;
662
663 // Debug values are not allowed to affect codegen.
664 if (MI.isDebugValue())
665 continue;
666
667 assert(!MI.isDebugInstr() && "Did not expect to find a use in debug "
668 "instruction that isn't a DBG_VALUE");
669
670 anyRemat |= reMaterializeFor(LI, MI);
671 }
672 }
673 if (!anyRemat)
674 return;
675
676 // Remove any values that were completely rematted.
677 for (unsigned Reg : RegsToSpill) {
678 LiveInterval &LI = LIS.getInterval(Reg);
679 for (LiveInterval::vni_iterator I = LI.vni_begin(), E = LI.vni_end();
680 I != E; ++I) {
681 VNInfo *VNI = *I;
682 if (VNI->isUnused() || VNI->isPHIDef() || UsedValues.count(VNI))
683 continue;
684 MachineInstr *MI = LIS.getInstructionFromIndex(VNI->def);
685 MI->addRegisterDead(Reg, &TRI);
686 if (!MI->allDefsAreDead())
687 continue;
688 LLVM_DEBUG(dbgs() << "All defs dead: " << *MI);
689 DeadDefs.push_back(MI);
690 }
691 }
692
693 // Eliminate dead code after remat. Note that some snippet copies may be
694 // deleted here.
695 if (DeadDefs.empty())
696 return;
697 LLVM_DEBUG(dbgs() << "Remat created " << DeadDefs.size() << " dead defs.\n");
698 Edit->eliminateDeadDefs(DeadDefs, RegsToSpill, AA);
699
700 // LiveRangeEdit::eliminateDeadDef is used to remove dead define instructions
701 // after rematerialization. To remove a VNI for a vreg from its LiveInterval,
702 // LiveIntervals::removeVRegDefAt is used. However, after non-PHI VNIs are all
703 // removed, PHI VNI are still left in the LiveInterval.
704 // So to get rid of unused reg, we need to check whether it has non-dbg
705 // reference instead of whether it has non-empty interval.
706 unsigned ResultPos = 0;
707 for (unsigned Reg : RegsToSpill) {
708 if (MRI.reg_nodbg_empty(Reg)) {
709 Edit->eraseVirtReg(Reg);
710 continue;
711 }
712
713 assert(LIS.hasInterval(Reg) &&
714 (!LIS.getInterval(Reg).empty() || !MRI.reg_nodbg_empty(Reg)) &&
715 "Empty and not used live-range?!");
716
717 RegsToSpill[ResultPos++] = Reg;
718 }
719 RegsToSpill.erase(RegsToSpill.begin() + ResultPos, RegsToSpill.end());
720 LLVM_DEBUG(dbgs() << RegsToSpill.size()
721 << " registers to spill after remat.\n");
722}

728/// If MI is a load or store of StackSlot, it can be removed.
729bool InlineSpiller::coalesceStackAccess(MachineInstr *MI, unsigned Reg) {
730 int FI = 0;
731 unsigned InstrReg = TII.isLoadFromStackSlot(*MI, FI);
732 bool IsLoad = InstrReg;
733 if (!IsLoad)
734 InstrReg = TII.isStoreToStackSlot(*MI, FI);
735
736 // We have a stack access. Is it the right register and slot?
737 if (InstrReg != Reg || FI != StackSlot)
738 return false;
739
740 if (!IsLoad)
741 HSpiller.rmFromMergeableSpills(*MI, StackSlot);
742
743 LLVM_DEBUG(dbgs() << "Coalescing stack access: " << *MI);
744 LIS.RemoveMachineInstrFromMaps(*MI);
745 MI->eraseFromParent();
746
747 if (IsLoad) {
748 ++NumReloadsRemoved;
749 --NumReloads;
750 } else {
751 ++NumSpillsRemoved;
752 --NumSpills;
753 }
754
755 return true;
756}

793/// foldMemoryOperand - Try folding stack slot references in Ops into their
794/// instructions.
795///
796/// @param Ops Operand indices from analyzeVirtReg().
797/// @param LoadMI Load instruction to use instead of stack slot when non-null.
798/// @return True on success.
799bool InlineSpiller::
800foldMemoryOperand(ArrayRef<std::pair<MachineInstr *, unsigned>> Ops,
801 MachineInstr *LoadMI) {
802 if (Ops.empty())
803 return false;
804 // Don't attempt folding in bundles.
805 MachineInstr *MI = Ops.front().first;
806 if (Ops.back().first != MI || MI->isBundled())
807 return false;
808
809 bool WasCopy = MI->isCopy();
810 unsigned ImpReg = 0;
811
812 // Spill subregs if the target allows it.
813 // We always want to spill subregs for stackmap/patchpoint pseudos.
814 bool SpillSubRegs = TII.isSubregFoldable() ||
815 MI->getOpcode() == TargetOpcode::STATEPOINT ||
816 MI->getOpcode() == TargetOpcode::PATCHPOINT ||
817 MI->getOpcode() == TargetOpcode::STACKMAP;
818
819 // TargetInstrInfo::foldMemoryOperand only expects explicit, non-tied
820 // operands.
821 SmallVector<unsigned, 8> FoldOps;
822 for (const auto &OpPair : Ops) {
823 unsigned Idx = OpPair.second;
824 assert(MI == OpPair.first && "Instruction conflict during operand folding");
825 MachineOperand &MO = MI->getOperand(Idx);
826 if (MO.isImplicit()) {
827 ImpReg = MO.getReg();
828 continue;
829 }
830
831 if (!SpillSubRegs && MO.getSubReg())
832 return false;
833 // We cannot fold a load instruction into a def.
834 if (LoadMI && MO.isDef())
835 return false;
836 // Tied use operands should not be passed to foldMemoryOperand.
837 if (!MI->isRegTiedToDefOperand(Idx))
838 FoldOps.push_back(Idx);
839 }
840
841 // If we only have implicit uses, we won't be able to fold that.
842 // Moreover, TargetInstrInfo::foldMemoryOperand will assert if we try!
843 if (FoldOps.empty())
844 return false;
845
846 MachineInstrSpan MIS(MI);
847
848 MachineInstr *FoldMI =
849 LoadMI ? TII.foldMemoryOperand(*MI, FoldOps, *LoadMI, &LIS)
850 : TII.foldMemoryOperand(*MI, FoldOps, StackSlot, &LIS, &VRM);
851 if (!FoldMI)
852 return false;
853
854 // Remove LIS for any dead defs in the original MI not in FoldMI.
855 for (MIBundleOperands MO(*MI); MO.isValid(); ++MO) {
856 if (!MO->isReg())
857 continue;
858 unsigned Reg = MO->getReg();
859 if (!Reg || TargetRegisterInfo::isVirtualRegister(Reg) ||
860 MRI.isReserved(Reg)) {
861 continue;
862 }
863 // Skip non-Defs, including undef uses and internal reads.
864 if (MO->isUse())
865 continue;
866 MIBundleOperands::PhysRegInfo RI =
867 MIBundleOperands(*FoldMI).analyzePhysReg(Reg, &TRI);
868 if (RI.FullyDefined)
869 continue;
870 // FoldMI does not define this physreg. Remove the LI segment.
871 assert(MO->isDead() && "Cannot fold physreg def");
872 SlotIndex Idx = LIS.getInstructionIndex(*MI).getRegSlot();
873 LIS.removePhysRegDefAt(Reg, Idx);
874 }
875
876 int FI;
877 if (TII.isStoreToStackSlot(*MI, FI) &&
878 HSpiller.rmFromMergeableSpills(*MI, FI))
879 --NumSpills;
880 LIS.ReplaceMachineInstrInMaps(*MI, *FoldMI);
881 MI->eraseFromParent();
882
883 // Insert any new instructions other than FoldMI into the LIS maps.
884 assert(!MIS.empty() && "Unexpected empty span of instructions!");
885 for (MachineInstr &MI : MIS)
886 if (&MI != FoldMI)
887 LIS.InsertMachineInstrInMaps(MI);
888
889 // TII.foldMemoryOperand may have left some implicit operands on the
890 // instruction. Strip them.
891 if (ImpReg)
892 for (unsigned i = FoldMI->getNumOperands(); i; --i) {
893 MachineOperand &MO = FoldMI->getOperand(i - 1);
894 if (!MO.isReg() || !MO.isImplicit())
895 break;
896 if (MO.getReg() == ImpReg)
897 FoldMI->RemoveOperand(i - 1);
898 }
899
900 LLVM_DEBUG(dumpMachineInstrRangeWithSlotIndex(MIS.begin(), MIS.end(), LIS,
901 "folded"));
902
903 if (!WasCopy)
904 ++NumFolded;
905 else if (Ops.front().second == 0) {
906 ++NumSpills;
907 HSpiller.addToMergeableSpills(*FoldMI, StackSlot, Original);
908 } else
909 ++NumReloads;
910 return true;
911}

929/// Check if \p Def fully defines a VReg with an undefined value.
930/// If that's the case, that means the value of VReg is actually
931/// not relevant.
932static bool isFullUndefDef(const MachineInstr &Def) {
933 if (!Def.isImplicitDef())
934 return false;
935 assert(Def.getNumOperands() == 1 &&
936 "Implicit def with more than one definition");
937 // We can say that the VReg defined by Def is undef, only if it is
938 // fully defined by Def. Otherwise, some of the lanes may not be
939 // undef and the value of the VReg matters.
940 return !Def.getOperand(0).getSubReg();
941}

943/// insertSpill - Insert a spill of NewVReg after MI.
944void InlineSpiller::insertSpill(unsigned NewVReg, bool isKill,
945 MachineBasicBlock::iterator MI) {
946 MachineBasicBlock &MBB = *MI->getParent();
947
948 MachineInstrSpan MIS(MI);
949 bool IsRealSpill = true;
950 if (isFullUndefDef(*MI)) {
951 // Don't spill undef value.
952 // Anything works for undef, in particular keeping the memory
953 // uninitialized is a viable option and it saves code size and
954 // run time.
955 BuildMI(MBB, std::next(MI), MI->getDebugLoc(), TII.get(TargetOpcode::KILL))
956 .addReg(NewVReg, getKillRegState(isKill));
957 IsRealSpill = false;
958 } else
959 TII.storeRegToStackSlot(MBB, std::next(MI), NewVReg, isKill, StackSlot,
960 MRI.getRegClass(NewVReg), &TRI);
961
962 LIS.InsertMachineInstrRangeInMaps(std::next(MI), MIS.end());
963
964 LLVM_DEBUG(dumpMachineInstrRangeWithSlotIndex(std::next(MI), MIS.end(), LIS,
965 "spill"));
966 ++NumSpills;
967 if (IsRealSpill)
968 HSpiller.addToMergeableSpills(*std::next(MI), StackSlot, Original);
969}

971/// spillAroundUses - insert spill code around each use of Reg.
972void InlineSpiller::spillAroundUses(unsigned Reg) {
973 LLVM_DEBUG(dbgs() << "spillAroundUses " << printReg(Reg) << '\n');
974 LiveInterval &OldLI = LIS.getInterval(Reg);
975
976 // Iterate over instructions using Reg.
977 for (MachineRegisterInfo::reg_bundle_iterator
978 RegI = MRI.reg_bundle_begin(Reg), E = MRI.reg_bundle_end();
979 RegI != E; ) {
980 MachineInstr *MI = &*(RegI++);
981
982 // Debug values are not allowed to affect codegen.
983 if (MI->isDebugValue()) {
984 // Modify DBG_VALUE now that the value is in a spill slot.
985 MachineBasicBlock *MBB = MI->getParent();
986 LLVM_DEBUG(dbgs() << "Modifying debug info due to spill:\t" << *MI);
987 buildDbgValueForSpill(*MBB, MI, *MI, StackSlot);
988 MBB->erase(MI);
989 continue;
990 }
991
992 assert(!MI->isDebugInstr() && "Did not expect to find a use in debug "
993 "instruction that isn't a DBG_VALUE");
994
995 // Ignore copies to/from snippets. We'll delete them.
996 if (SnippetCopies.count(MI))
997 continue;
998
999 // Stack slot accesses may coalesce away.
1000 if (coalesceStackAccess(MI, Reg))
1001 continue;
1002
1003 // Analyze instruction.
1004 SmallVector<std::pair<MachineInstr*, unsigned>, 8> Ops;
1005 MIBundleOperands::VirtRegInfo RI =
1006 MIBundleOperands(*MI).analyzeVirtReg(Reg, &Ops);
1007
1008 // Find the slot index where this instruction reads and writes OldLI.
1009 // This is usually the def slot, except for tied early clobbers.
1010 SlotIndex Idx = LIS.getInstructionIndex(*MI).getRegSlot();
1011 if (VNInfo *VNI = OldLI.getVNInfoAt(Idx.getRegSlot(true)))
1012 if (SlotIndex::isSameInstr(Idx, VNI->def))
1013 Idx = VNI->def;
1014
1015 // Check for a sibling copy.
1016 unsigned SibReg = isFullCopyOf(*MI, Reg);
1017 if (SibReg && isSibling(SibReg)) {
1018 // This may actually be a copy between snippets.
1019 if (isRegToSpill(SibReg)) {
1020 LLVM_DEBUG(dbgs() << "Found new snippet copy: " << *MI);
1021 SnippetCopies.insert(MI);
1022 continue;
1023 }
1024 if (RI.Writes) {
1025 if (hoistSpillInsideBB(OldLI, *MI)) {
1026 // This COPY is now dead, the value is already in the stack slot.
1027 MI->getOperand(0).setIsDead();
1028 DeadDefs.push_back(MI);
1029 continue;
1030 }
1031 } else {
1032 // This is a reload for a sib-reg copy. Drop spills downstream.
1033 LiveInterval &SibLI = LIS.getInterval(SibReg);
1034 eliminateRedundantSpills(SibLI, SibLI.getVNInfoAt(Idx));
1035 // The COPY will fold to a reload below.
1036 }
1037 }
1038
1039 // Attempt to fold memory ops.
1040 if (foldMemoryOperand(Ops))
1041 continue;
1042
1043 // Create a new virtual register for spill/fill.
1044 // FIXME: Infer regclass from instruction alone.
1045 unsigned NewVReg = Edit->createFrom(Reg);
1046
1047 if (RI.Reads)
1048 insertReload(NewVReg, Idx, MI);
1049
1050 // Rewrite instruction operands.
1051 bool hasLiveDef = false;
1052 for (const auto &OpPair : Ops) {
1053 MachineOperand &MO = OpPair.first->getOperand(OpPair.second);
1054 MO.setReg(NewVReg);
1055 if (MO.isUse()) {
1056 if (!OpPair.first->isRegTiedToDefOperand(OpPair.second))
1057 MO.setIsKill();
1058 } else {
1059 if (!MO.isDead())
1060 hasLiveDef = true;
1061 }
1062 }
1063 LLVM_DEBUG(dbgs() << "\trewrite: " << Idx << '\t' << *MI << '\n');
1064
1065 // FIXME: Use a second vreg if instruction has no tied ops.
1066 if (RI.Writes)
1067 if (hasLiveDef)
1068 insertSpill(NewVReg, true, MI);
1069 }
1070}

1072/// spillAll - Spill all registers remaining after rematerialization.
1073void InlineSpiller::spillAll() {
1074 // Update LiveStacks now that we are committed to spilling.
1075 if (StackSlot == VirtRegMap::NO_STACK_SLOT) {
1076 StackSlot = VRM.assignVirt2StackSlot(Original);
1077 StackInt = &LSS.getOrCreateInterval(StackSlot, MRI.getRegClass(Original));
1078 StackInt->getNextValue(SlotIndex(), LSS.getVNInfoAllocator());
1079 } else
1080 StackInt = &LSS.getInterval(StackSlot);
1081
1082 if (Original != Edit->getReg())
1083 VRM.assignVirt2StackSlot(Edit->getReg(), StackSlot);
1084
1085 assert(StackInt->getNumValNums() == 1 && "Bad stack interval values");
1086 for (unsigned Reg : RegsToSpill)
1087 StackInt->MergeSegmentsInAsValue(LIS.getInterval(Reg),
1088 StackInt->getValNumInfo(0));
1089 LLVM_DEBUG(dbgs() << "Merged spilled regs: " << *StackInt << '\n');
1090
1091 // Spill around uses of all RegsToSpill.
1092 for (unsigned Reg : RegsToSpill)
1093 spillAroundUses(Reg);
1094
1095 // Hoisted spills may cause dead code.
1096 if (!DeadDefs.empty()) {
1097 LLVM_DEBUG(dbgs() << "Eliminating " << DeadDefs.size() << " dead defs\n");
1098 Edit->eliminateDeadDefs(DeadDefs, RegsToSpill, AA);
1099 }
1100
1101 // Finally delete the SnippetCopies.
1102 for (unsigned Reg : RegsToSpill) {
1103 for (MachineRegisterInfo::reg_instr_iterator
1104 RI = MRI.reg_instr_begin(Reg), E = MRI.reg_instr_end();
1105 RI != E; ) {
1106 MachineInstr &MI = *(RI++);
1107 assert(SnippetCopies.count(&MI) && "Remaining use wasn't a snippet copy");
1108 // FIXME: Do this with a LiveRangeEdit callback.
1109 LIS.RemoveMachineInstrFromMaps(MI);
1110 MI.eraseFromParent();
1111 }
1112 }
1113
1114 // Delete all spilled registers.
1115 for (unsigned Reg : RegsToSpill)
1116 Edit->eraseVirtReg(Reg);
1117}

1119void InlineSpiller::spill(LiveRangeEdit &edit) {
1120 ++NumSpilledRanges;
1121 Edit = &edit;
1122 assert(!TargetRegisterInfo::isStackSlot(edit.getReg())
1123 && "Trying to spill a stack slot.");
1124 // Share a stack slot among all descendants of Original.
1125 Original = VRM.getOriginal(edit.getReg());
1126 StackSlot = VRM.getStackSlot(Original);
1127 StackInt = nullptr;
1128
1129 LLVM_DEBUG(dbgs() << "Inline spilling "
1130 << TRI.getRegClassName(MRI.getRegClass(edit.getReg()))
1131 << ':' << edit.getParent() << "\nFrom original "
1132 << printReg(Original) << '\n');
1133 assert(edit.getParent().isSpillable() &&
1134 "Attempting to spill already spilled value.");
1135 assert(DeadDefs.empty() && "Previous spill didn't remove dead defs");
1136
1137 collectRegsToSpill();
1138 reMaterializeAll();
1139
1140 // Remat may handle everything.
1141 if (!RegsToSpill.empty())
1142 spillAll();
1143
1144 Edit->calculateRegClassAndHint(MF, Loops, MBFI);
1145}

1150/// When a spill is inserted, add the spill to MergeableSpills map.
1151void HoistSpillHelper::addToMergeableSpills(MachineInstr &Spill, int StackSlot,
1152 unsigned Original) {
1153 BumpPtrAllocator &Allocator = LIS.getVNInfoAllocator();
1154 LiveInterval &OrigLI = LIS.getInterval(Original);
1155 // save a copy of LiveInterval in StackSlotToOrigLI because the original
1156 // LiveInterval may be cleared after all its references are spilled.
1157 if (StackSlotToOrigLI.find(StackSlot) == StackSlotToOrigLI.end()) {
1158 auto LI = llvm::make_unique<LiveInterval>(OrigLI.reg, OrigLI.weight);
1159 LI->assign(OrigLI, Allocator);
1160 StackSlotToOrigLI[StackSlot] = std::move(LI);
1161 }
1162 SlotIndex Idx = LIS.getInstructionIndex(Spill);
1163 VNInfo *OrigVNI = StackSlotToOrigLI[StackSlot]->getVNInfoAt(Idx.getRegSlot());
1164 std::pair<int, VNInfo *> MIdx = std::make_pair(StackSlot, OrigVNI);
1165 MergeableSpills[MIdx].insert(&Spill);
1166}

1168/// When a spill is removed, remove the spill from MergeableSpills map.
1169/// Return true if the spill is removed successfully.
1170bool HoistSpillHelper::rmFromMergeableSpills(MachineInstr &Spill,
1171 int StackSlot) {
1172 auto It = StackSlotToOrigLI.find(StackSlot);
1173 if (It == StackSlotToOrigLI.end())
1174 return false;

1181/// Check BB to see if it is a possible target BB to place a hoisted spill,
1182/// i.e., there should be a living sibling of OrigReg at the insert point.
1183bool HoistSpillHelper::isSpillCandBB(LiveInterval &OrigLI, VNInfo &OrigVNI,
1184 MachineBasicBlock &BB, unsigned &LiveReg) {

1186 unsigned OrigReg = OrigLI.reg;

1188 if (MI != BB.end())
1189 Idx = LIS.getInstructionIndex(*MI);
1190 else
1191 Idx = LIS.getMBBEndIdx(&BB).getPrevSlot();
1192 SmallSetVector<unsigned, 16> &Siblings = Virt2SiblingsMap[OrigReg];
1193 assert(OrigLI.getVNInfoAt(Idx) == &OrigVNI && "Unexpected VNI");

1206/// Remove redundant spills in the same BB. Save those redundant spills in
1207/// SpillsToRm, and save the spill to keep and its BB in SpillBBToSpill map.
1208void HoistSpillHelper::rmRedundantSpills(
1209 SmallPtrSet<MachineInstr *, 16> &Spills,
1210 SmallVectorImpl<MachineInstr *> &SpillsToRm,
1211 DenseMap<MachineDomTreeNode *, MachineInstr *> &SpillBBToSpill) {
1212 // For each spill saw, check SpillBBToSpill[] and see if its BB already has
1213 // another spill inside. If a BB contains more than one spill, only keep the
1214 // earlier spill with smaller SlotIndex.
1215 for (const auto CurrentSpill : Spills) {
1216 MachineBasicBlock *Block = CurrentSpill->getParent();
1217 MachineDomTreeNode *Node = MDT.getBase().getNode(Block);
1218 MachineInstr *PrevSpill = SpillBBToSpill[Node];
1219 if (PrevSpill) {
1220 SlotIndex PIdx = LIS.getInstructionIndex(*PrevSpill);
1221 SlotIndex CIdx = LIS.getInstructionIndex(*CurrentSpill);
1222 MachineInstr *SpillToRm = (CIdx > PIdx) ? CurrentSpill : PrevSpill;
1223 MachineInstr *SpillToKeep = (CIdx > PIdx) ? PrevSpill : CurrentSpill;
1224 SpillsToRm.push_back(SpillToRm);
1225 SpillBBToSpill[MDT.getBase().getNode(Block)] = SpillToKeep;
1226 } else {
1227 SpillBBToSpill[MDT.getBase().getNode(Block)] = CurrentSpill;
1228 }
1229 }
1230 for (const auto SpillToRm : SpillsToRm)
1231 Spills.erase(SpillToRm);
1232}

1234/// Starting from \p Root find a top-down traversal order of the dominator
1235/// tree to visit all basic blocks containing the elements of \p Spills.
1236/// Redundant spills will be found and put into \p SpillsToRm at the same
1237/// time. \p SpillBBToSpill will be populated as part of the process and
1238/// maps a basic block to the first store occurring in the basic block.
1239/// \post SpillsToRm.union(Spills\@post) == Spills\@pre
1240void HoistSpillHelper::getVisitOrders(
1241 MachineBasicBlock *Root, SmallPtrSet<MachineInstr *, 16> &Spills,
1242 SmallVectorImpl<MachineDomTreeNode *> &Orders,
1243 SmallVectorImpl<MachineInstr *> &SpillsToRm,
1244 DenseMap<MachineDomTreeNode *, unsigned> &SpillsToKeep,
1245 DenseMap<MachineDomTreeNode *, MachineInstr *> &SpillBBToSpill) {
1246 // The set contains all the possible BB nodes to which we may hoist
1247 // original spills.
1248 SmallPtrSet<MachineDomTreeNode *, 8> WorkSet;
1249 // Save the BB nodes on the path from the first BB node containing
1250 // non-redundant spill to the Root node.
1251 SmallPtrSet<MachineDomTreeNode *, 8> NodesOnPath;
1252 // All the spills to be hoisted must originate from a single def instruction
1253 // to the OrigReg. It means the def instruction should dominate all the spills
1254 // to be hoisted. We choose the BB where the def instruction is located as
1255 // the Root.
1256 MachineDomTreeNode *RootIDomNode = MDT[Root]->getIDom();
1257 // For every node on the dominator tree with spill, walk up on the dominator
1258 // tree towards the Root node until it is reached. If there is other node
1259 // containing spill in the middle of the path, the previous spill saw will
1260 // be redundant and the node containing it will be removed. All the nodes on
1261 // the path starting from the first node with non-redundant spill to the Root
1262 // node will be added to the WorkSet, which will contain all the possible
1263 // locations where spills may be hoisted to after the loop below is done.
1264 for (const auto Spill : Spills) {
1265 MachineBasicBlock *Block = Spill->getParent();
1266 MachineDomTreeNode *Node = MDT[Block];
1267 MachineInstr *SpillToRm = nullptr;
1268 while (Node != RootIDomNode) {
1269 // If Node dominates Block, and it already contains a spill, the spill in
1270 // Block will be redundant.
1271 if (Node != MDT[Block] && SpillBBToSpill[Node]) {
1272 SpillToRm = SpillBBToSpill[MDT[Block]];
1273 break;
1274 /// If we see the Node already in WorkSet, the path from the Node to
1275 /// the Root node must already be traversed by another spill.
1276 /// Then no need to repeat.
1277 } else if (WorkSet.count(Node)) {
1278 break;
1279 } else {
1280 NodesOnPath.insert(Node);
1281 }
1282 Node = Node->getIDom();
1283 }
1284 if (SpillToRm) {
1285 SpillsToRm.push_back(SpillToRm);
1286 } else {
1287 // Add a BB containing the original spills to SpillsToKeep -- i.e.,
1288 // set the initial status before hoisting start. The value of BBs
1289 // containing original spills is set to 0, in order to descriminate
1290 // with BBs containing hoisted spills which will be inserted to
1291 // SpillsToKeep later during hoisting.
1292 SpillsToKeep[MDT[Block]] = 0;
1293 WorkSet.insert(NodesOnPath.begin(), NodesOnPath.end());
1294 }
1295 NodesOnPath.clear();
1296 }
1297
1298 // Sort the nodes in WorkSet in top-down order and save the nodes
1299 // in Orders. Orders will be used for hoisting in runHoistSpills.
1300 unsigned idx = 0;
1301 Orders.push_back(MDT.getBase().getNode(Root));
1302 do {
1303 MachineDomTreeNode *Node = Orders[idx++];
1304 const std::vector<MachineDomTreeNode *> &Children = Node->getChildren();
1305 unsigned NumChildren = Children.size();
1306 for (unsigned i = 0; i != NumChildren; ++i) {
1307 MachineDomTreeNode *Child = Children[i];
1308 if (WorkSet.count(Child))
1309 Orders.push_back(Child);
1310 }
1311 } while (idx != Orders.size());
1312 assert(Orders.size() == WorkSet.size() &&
1313 "Orders have different size with WorkSet");
1314
1315#ifndef NDEBUG
1316 LLVM_DEBUG(dbgs() << "Orders size is " << Orders.size() << "\n");
1317 SmallVector<MachineDomTreeNode *, 32>::reverse_iterator RIt = Orders.rbegin();
1318 for (; RIt != Orders.rend(); RIt++)
1319 LLVM_DEBUG(dbgs() << "BB" << (*RIt)->getBlock()->getNumber() << ",");
1320 LLVM_DEBUG(dbgs() << "\n");
1321#endif
1322}

1324/// Try to hoist spills according to BB hotness. The spills to removed will
1325/// be saved in \p SpillsToRm. The spills to be inserted will be saved in
1326/// \p SpillsToIns.
1327void HoistSpillHelper::runHoistSpills(
1328 LiveInterval &OrigLI, VNInfo &OrigVNI,
1329 SmallPtrSet<MachineInstr *, 16> &Spills,
1330 SmallVectorImpl<MachineInstr *> &SpillsToRm,
1331 DenseMap<MachineBasicBlock *, unsigned> &SpillsToIns) {
1332 // Visit order of dominator tree nodes.
1333 SmallVector<MachineDomTreeNode *, 32> Orders;
1334 // SpillsToKeep contains all the nodes where spills are to be inserted
1335 // during hoisting. If the spill to be inserted is an original spill
1336 // (not a hoisted one), the value of the map entry is 0. If the spill
1337 // is a hoisted spill, the value of the map entry is the VReg to be used
1338 // as the source of the spill.
1339 DenseMap<MachineDomTreeNode *, unsigned> SpillsToKeep;
1340 // Map from BB to the first spill inside of it.
1341 DenseMap<MachineDomTreeNode *, MachineInstr *> SpillBBToSpill;
1342
1343 rmRedundantSpills(Spills, SpillsToRm, SpillBBToSpill);
1344
1345 MachineBasicBlock *Root = LIS.getMBBFromIndex(OrigVNI.def);
1346 getVisitOrders(Root, Spills, Orders, SpillsToRm, SpillsToKeep,
1347 SpillBBToSpill);
1348
1349 // SpillsInSubTreeMap keeps the map from a dom tree node to a pair of
1350 // nodes set and the cost of all the spills inside those nodes.
1351 // The nodes set are the locations where spills are to be inserted
1352 // in the subtree of current node.
1353 using NodesCostPair =
1354 std::pair<SmallPtrSet<MachineDomTreeNode *, 16>, BlockFrequency>;
1355 DenseMap<MachineDomTreeNode *, NodesCostPair> SpillsInSubTreeMap;
1356
1357 // Iterate Orders set in reverse order, which will be a bottom-up order
1358 // in the dominator tree. Once we visit a dom tree node, we know its
1359 // children have already been visited and the spill locations in the
1360 // subtrees of all the children have been determined.
1361 SmallVector<MachineDomTreeNode *, 32>::reverse_iterator RIt = Orders.rbegin();
1362 for (; RIt != Orders.rend(); RIt++) {
1363 MachineBasicBlock *Block = (*RIt)->getBlock();
1364
1365 // If Block contains an original spill, simply continue.
1366 if (SpillsToKeep.find(*RIt) != SpillsToKeep.end() && !SpillsToKeep[*RIt]) {
1367 SpillsInSubTreeMap[*RIt].first.insert(*RIt);
1368 // SpillsInSubTreeMap[*RIt].second contains the cost of spill.
1369 SpillsInSubTreeMap[*RIt].second = MBFI.getBlockFreq(Block);
1370 continue;
1371 }
1372
1373 // Collect spills in subtree of current node (*RIt) to
1374 // SpillsInSubTreeMap[*RIt].first.
1375 const std::vector<MachineDomTreeNode *> &Children = (*RIt)->getChildren();
1376 unsigned NumChildren = Children.size();
1377 for (unsigned i = 0; i != NumChildren; ++i) {
1378 MachineDomTreeNode *Child = Children[i];
1379 if (SpillsInSubTreeMap.find(Child) == SpillsInSubTreeMap.end())
1380 continue;
1381 // The stmt "SpillsInSubTree = SpillsInSubTreeMap[*RIt].first" below
1382 // should be placed before getting the begin and end iterators of
1383 // SpillsInSubTreeMap[Child].first, or else the iterators may be
1384 // invalidated when SpillsInSubTreeMap[*RIt] is seen the first time
1385 // and the map grows and then the original buckets in the map are moved.
1386 SmallPtrSet<MachineDomTreeNode *, 16> &SpillsInSubTree =
1387 SpillsInSubTreeMap[*RIt].first;
1388 BlockFrequency &SubTreeCost = SpillsInSubTreeMap[*RIt].second;
1389 SubTreeCost += SpillsInSubTreeMap[Child].second;
1390 auto BI = SpillsInSubTreeMap[Child].first.begin();
1391 auto EI = SpillsInSubTreeMap[Child].first.end();
1392 SpillsInSubTree.insert(BI, EI);
1393 SpillsInSubTreeMap.erase(Child);
1394 }
1395
1396 SmallPtrSet<MachineDomTreeNode *, 16> &SpillsInSubTree =
1397 SpillsInSubTreeMap[*RIt].first;
1398 BlockFrequency &SubTreeCost = SpillsInSubTreeMap[*RIt].second;
1399 // No spills in subtree, simply continue.
1400 if (SpillsInSubTree.empty())
1401 continue;
1402
1403 // Check whether Block is a possible candidate to insert spill.
1404 unsigned LiveReg = 0;
1405 if (!isSpillCandBB(OrigLI, OrigVNI, *Block, LiveReg))
1406 continue;
1407
1408 // If there are multiple spills that could be merged, bias a little
1409 // to hoist the spill.
1410 BranchProbability MarginProb = (SpillsInSubTree.size() > 1)
1411 ? BranchProbability(9, 10)
1412 : BranchProbability(1, 1);
1413 if (SubTreeCost > MBFI.getBlockFreq(Block) * MarginProb) {
1414 // Hoist: Move spills to current Block.
1415 for (const auto SpillBB : SpillsInSubTree) {
1416 // When SpillBB is a BB contains original spill, insert the spill
1417 // to SpillsToRm.
1418 if (SpillsToKeep.find(SpillBB) != SpillsToKeep.end() &&
1419 !SpillsToKeep[SpillBB]) {
1420 MachineInstr *SpillToRm = SpillBBToSpill[SpillBB];
1421 SpillsToRm.push_back(SpillToRm);
1422 }
1423 // SpillBB will not contain spill anymore, remove it from SpillsToKeep.
1424 SpillsToKeep.erase(SpillBB);
1425 }
1426 // Current Block is the BB containing the new hoisted spill. Add it to
1427 // SpillsToKeep. LiveReg is the source of the new spill.
1428 SpillsToKeep[*RIt] = LiveReg;
1429 LLVM_DEBUG({
1430 dbgs() << "spills in BB: ";
1431 for (const auto Rspill : SpillsInSubTree)
1432 dbgs() << Rspill->getBlock()->getNumber() << " ";
1433 dbgs() << "were promoted to BB" << (*RIt)->getBlock()->getNumber()
1434 << "\n";
1435 });
1436 SpillsInSubTree.clear();
1437 SpillsInSubTree.insert(*RIt);
1438 SubTreeCost = MBFI.getBlockFreq(Block);
1439 }
1440 }
1441 // For spills in SpillsToKeep with LiveReg set (i.e., not original spill),
1442 // save them to SpillsToIns.
1443 for (const auto Ent : SpillsToKeep) {
1444 if (Ent.second)
1445 SpillsToIns[Ent.first->getBlock()] = Ent.second;
1446 }
1447}
1448
1449/// For spills with equal values, remove redundant spills and hoist those left
1450/// to less hot spots.
1451///
1452/// Spills with equal values will be collected into the same set in
1453/// MergeableSpills when spill is inserted. These equal spills are originated
1454/// from the same defining instruction and are dominated by the instruction.
1455/// Before hoisting all the equal spills, redundant spills inside in the same
1456/// BB are first marked to be deleted. Then starting from the spills left, walk
1457/// up on the dominator tree towards the Root node where the define instruction
1458/// is located, mark the dominated spills to be deleted along the way and
1459/// collect the BB nodes on the path from non-dominated spills to the define
1460/// instruction into a WorkSet. The nodes in WorkSet are the candidate places
1461/// where we are considering to hoist the spills. We iterate the WorkSet in
1462/// bottom-up order, and for each node, we will decide whether to hoist spills
1463/// inside its subtree to that node. In this way, we can get benefit locally
1464/// even if hoisting all the equal spills to one cold place is impossible.
1465void HoistSpillHelper::hoistAllSpills() {
1466 SmallVector<unsigned, 4> NewVRegs;
1467 LiveRangeEdit Edit(nullptr, NewVRegs, MF, LIS, &VRM, this);
1468
1469 for (unsigned i = 0, e = MRI.getNumVirtRegs(); i != e; ++i) {
1470 unsigned Reg = TargetRegisterInfo::index2VirtReg(i);
1471 unsigned Original = VRM.getPreSplitReg(Reg);
1472 if (!MRI.def_empty(Reg))
1473 Virt2SiblingsMap[Original].insert(Reg);
1474 }
1475
1476 // Each entry in MergeableSpills contains a spill set with equal values.
1477 for (auto &Ent : MergeableSpills) {
1478 int Slot = Ent.first.first;
1479 LiveInterval &OrigLI = *StackSlotToOrigLI[Slot];
1480 VNInfo *OrigVNI = Ent.first.second;
1481 SmallPtrSet<MachineInstr *, 16> &EqValSpills = Ent.second;
1482 if (Ent.second.empty())
1483 continue;
1484
1485 LLVM_DEBUG({
1486 dbgs() << "\nFor Slot" << Slot << " and VN" << OrigVNI->id << ":\n"
1487 << "Equal spills in BB: ";
1488 for (const auto spill : EqValSpills)
1489 dbgs() << spill->getParent()->getNumber() << " ";
1490 dbgs() << "\n";
1491 });
1492
1493 // SpillsToRm is the spill set to be removed from EqValSpills.
1494 SmallVector<MachineInstr *, 16> SpillsToRm;
1495 // SpillsToIns is the spill set to be newly inserted after hoisting.
1496 DenseMap<MachineBasicBlock *, unsigned> SpillsToIns;
1497
1498 runHoistSpills(OrigLI, *OrigVNI, EqValSpills, SpillsToRm, SpillsToIns);
1499
1500 LLVM_DEBUG({
1501 dbgs() << "Finally inserted spills in BB: ";
1502 for (const auto Ispill : SpillsToIns)
1503 dbgs() << Ispill.first->getNumber() << " ";
1504 dbgs() << "\nFinally removed spills in BB: ";
1505 for (const auto Rspill : SpillsToRm)
1506 dbgs() << Rspill->getParent()->getNumber() << " ";
1507 dbgs() << "\n";
1508 });
1509
1510 // Stack live range update.
1511 LiveInterval &StackIntvl = LSS.getInterval(Slot);
1512 if (!SpillsToIns.empty() || !SpillsToRm.empty())
1513 StackIntvl.MergeValueInAsValue(OrigLI, OrigVNI,
1514 StackIntvl.getValNumInfo(0));
1515
1516 // Insert hoisted spills.
1517 for (auto const Insert : SpillsToIns) {
1518 MachineBasicBlock *BB = Insert.first;
1519 unsigned LiveReg = Insert.second;
1520 MachineBasicBlock::iterator MI = IPA.getLastInsertPointIter(OrigLI, *BB);
1521 TII.storeRegToStackSlot(*BB, MI, LiveReg, false, Slot,
1522 MRI.getRegClass(LiveReg), &TRI);
1523 LIS.InsertMachineInstrRangeInMaps(std::prev(MI), MI);
1524 ++NumSpills;
1525 }
1526
1527 // Remove redundant spills or change them to dead instructions.
1528 NumSpills -= SpillsToRm.size();
1529 for (auto const RMEnt : SpillsToRm) {
1530 RMEnt->setDesc(TII.get(TargetOpcode::KILL));
1531 for (unsigned i = RMEnt->getNumOperands(); i; --i) {
1532 MachineOperand &MO = RMEnt->getOperand(i - 1);
1533 if (MO.isReg() && MO.isImplicit() && MO.isDef() && !MO.isDead())
1534 RMEnt->RemoveOperand(i - 1);
1535 }
1536 }
1537 Edit.eliminateDeadDefs(SpillsToRm, None, AA);
1538 }
1539}
