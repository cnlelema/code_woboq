22namespace llvm {
23
24class Twine;
25
26/// A token produced by the machine instruction lexer.
27struct MIToken {
28 enum TokenKind {
29 // Markers
30 Eof,
31 Error,
32 Newline,
33
34 // Tokens with no info.
35 comma,
36 equal,
37 underscore,
38 colon,
39 coloncolon,
40 dot,
41 exclaim,
42 lparen,
43 rparen,
44 lbrace,
45 rbrace,
46 plus,
47 minus,
48 less,
49 greater,
50
51 // Keywords
52 kw_implicit,
53 kw_implicit_define,
54 kw_def,
55 kw_dead,
56 kw_dereferenceable,
57 kw_killed,
58 kw_undef,
59 kw_internal,
60 kw_early_clobber,
61 kw_debug_use,
62 kw_renamable,
63 kw_tied_def,
64 kw_frame_setup,
65 kw_frame_destroy,
66 kw_nnan,
67 kw_ninf,
68 kw_nsz,
69 kw_arcp,
70 kw_contract,
71 kw_afn,
72 kw_reassoc,
73 kw_nuw,
74 kw_nsw,
75 kw_exact,
76 kw_fpexcept,
77 kw_debug_location,
78 kw_cfi_same_value,
79 kw_cfi_offset,
80 kw_cfi_rel_offset,
81 kw_cfi_def_cfa_register,
82 kw_cfi_def_cfa_offset,
83 kw_cfi_adjust_cfa_offset,
84 kw_cfi_escape,
85 kw_cfi_def_cfa,
86 kw_cfi_register,
87 kw_cfi_remember_state,
88 kw_cfi_restore,
89 kw_cfi_restore_state,
90 kw_cfi_undefined,
91 kw_cfi_window_save,
92 kw_cfi_aarch64_negate_ra_sign_state,
93 kw_blockaddress,
94 kw_intrinsic,
95 kw_target_index,
96 kw_half,
97 kw_float,
98 kw_double,
99 kw_x86_fp80,
100 kw_fp128,
101 kw_ppc_fp128,
102 kw_target_flags,
103 kw_volatile,
104 kw_non_temporal,
105 kw_invariant,
106 kw_align,
107 kw_addrspace,
108 kw_stack,
109 kw_got,
110 kw_cap_table,
111 kw_jump_table,
112 kw_constant_pool,
113 kw_call_entry,
114 kw_liveout,
115 kw_address_taken,
116 kw_landing_pad,
117 kw_liveins,
118 kw_successors,
119 kw_floatpred,
120 kw_intpred,
121 kw_pre_instr_symbol,
122 kw_post_instr_symbol,
123 kw_unknown_size,
124
125 // Named metadata keywords
126 md_tbaa,
127 md_alias_scope,
128 md_noalias,
129 md_range,
130 md_diexpr,
131 md_dilocation,
132
133 // Identifier tokens
134 Identifier,
135 NamedRegister,
136 NamedVirtualRegister,
137 MachineBasicBlockLabel,
138 MachineBasicBlock,
139 StackObject,
140 FixedStackObject,
141 NamedGlobalValue,
142 GlobalValue,
143 ExternalSymbol,
144 MCSymbol,
145
146 // Other tokens
147 IntegerLiteral,
148 FloatingPointLiteral,
149 HexLiteral,
150 VirtualRegister,
151 ConstantPoolItem,
152 JumpTableIndex,
153 NamedIRBlock,
154 IRBlock,
155 NamedIRValue,
156 IRValue,
157 QuotedIRValue, // `<constant value>`
158 SubRegisterIndex,
159 StringConstant
160 };
161
162private:
163 TokenKind Kind = Error;
164 StringRef Range;
165 StringRef StringValue;
166 std::string StringValueStorage;
167 APSInt IntVal;
168
169public:
170 MIToken() = default;
171
172 MIToken &reset(TokenKind Kind, StringRef Range);
173
174 MIToken &setStringValue(StringRef StrVal);
175 MIToken &setOwnedStringValue(std::string StrVal);
176 MIToken &setIntegerValue(APSInt IntVal);
177
178 TokenKind kind() const { return Kind; }
179
180 bool isError() const { return Kind == Error; }
181
182 bool isNewlineOrEOF() const { return Kind == Newline || Kind == Eof; }
183
184 bool isErrorOrEOF() const { return Kind == Error || Kind == Eof; }
185
186 bool isRegister() const {
187 return Kind == NamedRegister || Kind == underscore ||
188 Kind == NamedVirtualRegister || Kind == VirtualRegister;
189 }
190
191 bool isRegisterFlag() const {
192 return Kind == kw_implicit || Kind == kw_implicit_define ||
193 Kind == kw_def || Kind == kw_dead || Kind == kw_killed ||
194 Kind == kw_undef || Kind == kw_internal ||
195 Kind == kw_early_clobber || Kind == kw_debug_use ||
196 Kind == kw_renamable;
197 }
198
199 bool isMemoryOperandFlag() const {
200 return Kind == kw_volatile || Kind == kw_non_temporal ||
201 Kind == kw_dereferenceable || Kind == kw_invariant ||
202 Kind == StringConstant;
203 }
204
205 bool is(TokenKind K) const { return Kind == K; }
206
207 bool isNot(TokenKind K) const { return Kind != K; }
208
209 StringRef::iterator location() const { return Range.begin(); }
210
211 StringRef range() const { return Range; }
212
213 /// Return the token's string value.
214 StringRef stringValue() const { return StringValue; }
215
216 const APSInt &integerValue() const { return IntVal; }
217
218 bool hasIntegerValue() const {
219 return Kind == IntegerLiteral || Kind == MachineBasicBlock ||
220 Kind == MachineBasicBlockLabel || Kind == StackObject ||
221 Kind == FixedStackObject || Kind == GlobalValue ||
222 Kind == VirtualRegister || Kind == ConstantPoolItem ||
223 Kind == JumpTableIndex || Kind == IRBlock || Kind == IRValue;
224 }
225};
226
227/// Consume a single machine instruction token in the given source and return
228/// the remaining source string.
229StringRef lexMIToken(
230 StringRef Source, MIToken &Token,
231 function_ref<void(StringRef::iterator, const Twine &)> ErrorCallback);
232
233} // end namespace llvm
