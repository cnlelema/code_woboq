29using PairTy = std::pair<MCSymbol *, MachineModuleInfoImpl::StubValueTy>;
30static int SortSymbolPair(const PairTy *LHS, const PairTy *RHS) {
31 return LHS->first->getName().compare(RHS->first->getName());
32}
33
34MachineModuleInfoImpl::SymbolListTy MachineModuleInfoImpl::getSortedStubs(
35 DenseMap<MCSymbol *, MachineModuleInfoImpl::StubValueTy> &Map) {
36 MachineModuleInfoImpl::SymbolListTy List(Map.begin(), Map.end());
37
38 array_pod_sort(List.begin(), List.end(), SortSymbolPair);
39
40 Map.clear();
41 return List;
42}
