92static void reportTranslationError(MachineFunction &MF,
93 const TargetPassConfig &TPC,
94 OptimizationRemarkEmitter &ORE,
95 OptimizationRemarkMissed &R) {
96 MF.getProperties().set(MachineFunctionProperties::Property::FailedISel);
97
98 // Print the function name explicitly if we don't have a debug location (which
99 // makes the diagnostic less useful) or if we're going to emit a raw error.
100 if (!R.getLocation().isValid() || TPC.isGlobalISelAbortEnabled())
101 R << (" (in function: " + MF.getName() + ")").str();
102
103 if (TPC.isGlobalISelAbortEnabled())
104 report_fatal_error(R.getMsg());
105 else
106 ORE.emit(R);
107}

155IRTranslator::ValueToVRegInfo::VRegListT &
156IRTranslator::allocateVRegs(const Value &Val) {
157 assert(!VMap.contains(Val) && "Value already allocated in VMap");
158 auto *Regs = VMap.getVRegs(Val);
159 auto *Offsets = VMap.getOffsets(Val);
160 SmallVector<LLT, 4> SplitTys;
161 computeValueLLTs(*DL, *Val.getType(), SplitTys,
162 Offsets->empty() ? Offsets : nullptr);
163 for (unsigned i = 0; i < SplitTys.size(); ++i)
164 Regs->push_back(0);
165 return *Regs;
166}

168ArrayRef<unsigned> IRTranslator::getOrCreateVRegs(const Value &Val) {
169 auto VRegsIt = VMap.findVRegs(Val);
170 if (VRegsIt != VMap.vregs_end())
171 return *VRegsIt->second;
172
173 if (Val.getType()->isVoidTy())
174 return *VMap.getVRegs(Val);
175
176 // Create entry for this type.
177 auto *VRegs = VMap.getVRegs(Val);
178 auto *Offsets = VMap.getOffsets(Val);
179
180 assert(Val.getType()->isSized() &&
181 "Don't know how to create an empty vreg");
182
183 SmallVector<LLT, 4> SplitTys;
184 computeValueLLTs(*DL, *Val.getType(), SplitTys,
185 Offsets->empty() ? Offsets : nullptr);
186
187 if (!isa<Constant>(Val)) {
188 for (auto Ty : SplitTys)
189 VRegs->push_back(MRI->createGenericVirtualRegister(Ty));
190 return *VRegs;
191 }
192
193 if (Val.getType()->isAggregateType()) {
194 // UndefValue, ConstantAggregateZero
195 auto &C = cast<Constant>(Val);
196 unsigned Idx = 0;
197 while (auto Elt = C.getAggregateElement(Idx++)) {
198 auto EltRegs = getOrCreateVRegs(*Elt);
199 llvm::copy(EltRegs, std::back_inserter(*VRegs));
200 }
201 } else {
202 assert(SplitTys.size() == 1 && "unexpectedly split LLT");
203 VRegs->push_back(MRI->createGenericVirtualRegister(SplitTys[0]));
204 bool Success = translate(cast<Constant>(Val), VRegs->front());
205 if (!Success) {
206 OptimizationRemarkMissed R("gisel-irtranslator", "GISelFailure",
207 MF->getFunction().getSubprogram(),
208 &MF->getFunction().getEntryBlock());
209 R << "unable to translate constant: " << ore::NV("Type", Val.getType());
210 reportTranslationError(*MF, *TPC, *ORE, R);
211 return *VRegs;
212 }
213 }
214
215 return *VRegs;
216}

218int IRTranslator::getOrCreateFrameIndex(const AllocaInst &AI) {
219 if (FrameIndices.find(&AI) != FrameIndices.end())
220 return FrameIndices[&AI];
221
222 unsigned ElementSize = DL->getTypeAllocSize(AI.getAllocatedType());
223 unsigned Size =
224 ElementSize * cast<ConstantInt>(AI.getArraySize())->getZExtValue();
225
226 // Always allocate at least one byte.
227 Size = std::max(Size, 1u);
228
229 unsigned Alignment = AI.getAlignment();
230 if (!Alignment)
231 Alignment = DL->getABITypeAlignment(AI.getAllocatedType());
232
233 int &FI = FrameIndices[&AI];
234 FI = MF->getFrameInfo().CreateStackObject(Size, Alignment, false, &AI);
235 return FI;
236}

238unsigned IRTranslator::getMemOpAlignment(const Instruction &I) {
239 unsigned Alignment = 0;
240 Type *ValTy = nullptr;
241 if (const StoreInst *SI = dyn_cast<StoreInst>(&I)) {
242 Alignment = SI->getAlignment();
243 ValTy = SI->getValueOperand()->getType();
244 } else if (const LoadInst *LI = dyn_cast<LoadInst>(&I)) {
245 Alignment = LI->getAlignment();
246 ValTy = LI->getType();
247 } else if (const AtomicCmpXchgInst *AI = dyn_cast<AtomicCmpXchgInst>(&I)) {
248 // TODO(PR27168): This instruction has no alignment attribute, but unlike
249 // the default alignment for load/store, the default here is to assume
250 // it has NATURAL alignment, not DataLayout-specified alignment.
251 const DataLayout &DL = AI->getModule()->getDataLayout();
252 Alignment = DL.getTypeStoreSize(AI->getCompareOperand()->getType());
253 ValTy = AI->getCompareOperand()->getType();
254 } else if (const AtomicRMWInst *AI = dyn_cast<AtomicRMWInst>(&I)) {
255 // TODO(PR27168): This instruction has no alignment attribute, but unlike
256 // the default alignment for load/store, the default here is to assume
257 // it has NATURAL alignment, not DataLayout-specified alignment.
258 const DataLayout &DL = AI->getModule()->getDataLayout();
259 Alignment = DL.getTypeStoreSize(AI->getValOperand()->getType());
260 ValTy = AI->getType();
261 } else {
262 OptimizationRemarkMissed R("gisel-irtranslator", "", &I);
263 R << "unable to translate memop: " << ore::NV("Opcode", &I);
264 reportTranslationError(*MF, *TPC, *ORE, R);
265 return 1;
266 }
267
268 return Alignment ? Alignment : DL->getABITypeAlignment(ValTy);
269}

282bool IRTranslator::translateBinaryOp(unsigned Opcode, const User &U,
283 MachineIRBuilder &MIRBuilder) {
284 // FIXME: handle signed/unsigned wrapping flags.
285
286 // Get or create a virtual register for each value.
287 // Unless the value is a Constant => loadimm cst?
288 // or inline constant each time?
289 // Creation of a virtual register needs to have a size.
290 unsigned Op0 = getOrCreateVReg(*U.getOperand(0));
291 unsigned Op1 = getOrCreateVReg(*U.getOperand(1));
292 unsigned Res = getOrCreateVReg(U);
293 uint16_t Flags = 0;
294 if (isa<Instruction>(U)) {
295 const Instruction &I = cast<Instruction>(U);
296 Flags = MachineInstr::copyFlagsFromInstruction(I);
297 }
298
299 MIRBuilder.buildInstr(Opcode, {Res}, {Op0, Op1}, Flags);
300 return true;
301}

322bool IRTranslator::translateCompare(const User &U,
323 MachineIRBuilder &MIRBuilder) {
324 const CmpInst *CI = dyn_cast<CmpInst>(&U);
325 unsigned Op0 = getOrCreateVReg(*U.getOperand(0));
326 unsigned Op1 = getOrCreateVReg(*U.getOperand(1));
327 unsigned Res = getOrCreateVReg(U);
328 CmpInst::Predicate Pred =
329 CI ? CI->getPredicate() : static_cast<CmpInst::Predicate>(
330 cast<ConstantExpr>(U).getPredicate());
331 if (CmpInst::isIntPredicate(Pred))
332 MIRBuilder.buildICmp(Pred, Res, Op0, Op1);
333 else if (Pred == CmpInst::FCMP_FALSE)
334 MIRBuilder.buildCopy(
335 Res, getOrCreateVReg(*Constant::getNullValue(CI->getType())));
336 else if (Pred == CmpInst::FCMP_TRUE)
337 MIRBuilder.buildCopy(
338 Res, getOrCreateVReg(*Constant::getAllOnesValue(CI->getType())));
339 else {
340 MIRBuilder.buildInstr(TargetOpcode::G_FCMP, {Res}, {Pred, Op0, Op1},
341 MachineInstr::copyFlagsFromInstruction(*CI));
342 }
343
344 return true;
345}

347bool IRTranslator::translateRet(const User &U, MachineIRBuilder &MIRBuilder) {
348 const ReturnInst &RI = cast<ReturnInst>(U);
349 const Value *Ret = RI.getReturnValue();
350 if (Ret && DL->getTypeStoreSize(Ret->getType()) == 0)
351 Ret = nullptr;
352
353 ArrayRef<unsigned> VRegs;
354 if (Ret)
355 VRegs = getOrCreateVRegs(*Ret);
356
357 unsigned SwiftErrorVReg = 0;
358 if (CLI->supportSwiftError() && SwiftError.getFunctionArg()) {
359 SwiftErrorVReg = SwiftError.getOrCreateVRegUseAt(
360 &RI, &MIRBuilder.getMBB(), SwiftError.getFunctionArg());
361 }
362
363 // The target may mess up with the insertion point, but
364 // this is not important as a return is the last instruction
365 // of the block anyway.
366 return CLI->lowerReturn(MIRBuilder, Ret, VRegs, SwiftErrorVReg);
367}

369bool IRTranslator::translateBr(const User &U, MachineIRBuilder &MIRBuilder) {
370 const BranchInst &BrInst = cast<BranchInst>(U);
371 unsigned Succ = 0;
372 if (!BrInst.isUnconditional()) {
373 // We want a G_BRCOND to the true BB followed by an unconditional branch.
374 unsigned Tst = getOrCreateVReg(*BrInst.getCondition());
375 const BasicBlock &TrueTgt = *cast<BasicBlock>(BrInst.getSuccessor(Succ++));
376 MachineBasicBlock &TrueBB = getMBB(TrueTgt);
377 MIRBuilder.buildBrCond(Tst, TrueBB);
378 }
379
380 const BasicBlock &BrTgt = *cast<BasicBlock>(BrInst.getSuccessor(Succ));
381 MachineBasicBlock &TgtBB = getMBB(BrTgt);
382 MachineBasicBlock &CurBB = MIRBuilder.getMBB();
383
384 // If the unconditional target is the layout successor, fallthrough.
385 if (!CurBB.isLayoutSuccessor(&TgtBB))
386 MIRBuilder.buildBr(TgtBB);
387
388 // Link successors.
389 for (const BasicBlock *Succ : successors(&BrInst))
390 CurBB.addSuccessor(&getMBB(*Succ));
391 return true;
392}

394bool IRTranslator::translateSwitch(const User &U,
395 MachineIRBuilder &MIRBuilder) {
396 // For now, just translate as a chain of conditional branches.
397 // FIXME: could we share most of the logic/code in
398 // SelectionDAGBuilder::visitSwitch between SelectionDAG and GlobalISel?
399 // At first sight, it seems most of the logic in there is independent of
400 // SelectionDAG-specifics and a lot of work went in to optimize switch
401 // lowering in there.
402
403 const SwitchInst &SwInst = cast<SwitchInst>(U);
404 const unsigned SwCondValue = getOrCreateVReg(*SwInst.getCondition());
405 const BasicBlock *OrigBB = SwInst.getParent();
406
407 LLT LLTi1 = getLLTForType(*Type::getInt1Ty(U.getContext()), *DL);
408 for (auto &CaseIt : SwInst.cases()) {
409 const unsigned CaseValueReg = getOrCreateVReg(*CaseIt.getCaseValue());
410 const unsigned Tst = MRI->createGenericVirtualRegister(LLTi1);
411 MIRBuilder.buildICmp(CmpInst::ICMP_EQ, Tst, CaseValueReg, SwCondValue);
412 MachineBasicBlock &CurMBB = MIRBuilder.getMBB();
413 const BasicBlock *TrueBB = CaseIt.getCaseSuccessor();
414 MachineBasicBlock &TrueMBB = getMBB(*TrueBB);
415
416 MIRBuilder.buildBrCond(Tst, TrueMBB);
417 CurMBB.addSuccessor(&TrueMBB);
418 addMachineCFGPred({OrigBB, TrueBB}, &CurMBB);
419
420 MachineBasicBlock *FalseMBB =
421 MF->CreateMachineBasicBlock(SwInst.getParent());
422 // Insert the comparison blocks one after the other.
423 MF->insert(std::next(CurMBB.getIterator()), FalseMBB);
424 MIRBuilder.buildBr(*FalseMBB);
425 CurMBB.addSuccessor(FalseMBB);
426
427 MIRBuilder.setMBB(*FalseMBB);
428 }
429 // handle default case
430 const BasicBlock *DefaultBB = SwInst.getDefaultDest();
431 MachineBasicBlock &DefaultMBB = getMBB(*DefaultBB);
432 MIRBuilder.buildBr(DefaultMBB);
433 MachineBasicBlock &CurMBB = MIRBuilder.getMBB();
434 CurMBB.addSuccessor(&DefaultMBB);
435 addMachineCFGPred({OrigBB, DefaultBB}, &CurMBB);
436
437 return true;
438}

463bool IRTranslator::translateLoad(const User &U, MachineIRBuilder &MIRBuilder) {
464 const LoadInst &LI = cast<LoadInst>(U);
465
466 auto Flags = LI.isVolatile() ? MachineMemOperand::MOVolatile
467 : MachineMemOperand::MONone;
468 Flags |= MachineMemOperand::MOLoad;
469
470 if (DL->getTypeStoreSize(LI.getType()) == 0)
471 return true;
472
473 ArrayRef<unsigned> Regs = getOrCreateVRegs(LI);
474 ArrayRef<uint64_t> Offsets = *VMap.getOffsets(LI);
475 unsigned Base = getOrCreateVReg(*LI.getPointerOperand());
476
477 Type *OffsetIRTy = DL->getIntPtrType(LI.getPointerOperandType());
478 LLT OffsetTy = getLLTForType(*OffsetIRTy, *DL);
479
480 if (CLI->supportSwiftError() && isSwiftError(LI.getPointerOperand())) {
481 assert(Regs.size() == 1 && "swifterror should be single pointer");
482 unsigned VReg = SwiftError.getOrCreateVRegUseAt(&LI, &MIRBuilder.getMBB(),
483 LI.getPointerOperand());
484 MIRBuilder.buildCopy(Regs[0], VReg);
485 return true;
486 }
487
488
489 for (unsigned i = 0; i < Regs.size(); ++i) {
490 unsigned Addr = 0;
491 MIRBuilder.materializeGEP(Addr, Base, OffsetTy, Offsets[i] / 8);
492
493 MachinePointerInfo Ptr(LI.getPointerOperand(), Offsets[i] / 8);
494 unsigned BaseAlign = getMemOpAlignment(LI);
495 auto MMO = MF->getMachineMemOperand(
496 Ptr, Flags, (MRI->getType(Regs[i]).getSizeInBits() + 7) / 8,
497 MinAlign(BaseAlign, Offsets[i] / 8), AAMDNodes(), nullptr,
498 LI.getSyncScopeID(), LI.getOrdering());
499 MIRBuilder.buildLoad(Regs[i], Addr, *MMO);
500 }
501
502 return true;
503}

505bool IRTranslator::translateStore(const User &U, MachineIRBuilder &MIRBuilder) {
506 const StoreInst &SI = cast<StoreInst>(U);
507 auto Flags = SI.isVolatile() ? MachineMemOperand::MOVolatile
508 : MachineMemOperand::MONone;
509 Flags |= MachineMemOperand::MOStore;
510
511 if (DL->getTypeStoreSize(SI.getValueOperand()->getType()) == 0)
512 return true;
513
514 ArrayRef<unsigned> Vals = getOrCreateVRegs(*SI.getValueOperand());
515 ArrayRef<uint64_t> Offsets = *VMap.getOffsets(*SI.getValueOperand());
516 unsigned Base = getOrCreateVReg(*SI.getPointerOperand());
517
518 Type *OffsetIRTy = DL->getIntPtrType(SI.getPointerOperandType());
519 LLT OffsetTy = getLLTForType(*OffsetIRTy, *DL);
520
521 if (CLI->supportSwiftError() && isSwiftError(SI.getPointerOperand())) {
522 assert(Vals.size() == 1 && "swifterror should be single pointer");
523
524 unsigned VReg = SwiftError.getOrCreateVRegDefAt(&SI, &MIRBuilder.getMBB(),
525 SI.getPointerOperand());
526 MIRBuilder.buildCopy(VReg, Vals[0]);
527 return true;
528 }
529
530 for (unsigned i = 0; i < Vals.size(); ++i) {
531 unsigned Addr = 0;
532 MIRBuilder.materializeGEP(Addr, Base, OffsetTy, Offsets[i] / 8);
533
534 MachinePointerInfo Ptr(SI.getPointerOperand(), Offsets[i] / 8);
535 unsigned BaseAlign = getMemOpAlignment(SI);
536 auto MMO = MF->getMachineMemOperand(
537 Ptr, Flags, (MRI->getType(Vals[i]).getSizeInBits() + 7) / 8,
538 MinAlign(BaseAlign, Offsets[i] / 8), AAMDNodes(), nullptr,
539 SI.getSyncScopeID(), SI.getOrdering());
540 MIRBuilder.buildStore(Vals[i], Addr, *MMO);
541 }
542 return true;
543}

545static uint64_t getOffsetFromIndices(const User &U, const DataLayout &DL) {
546 const Value *Src = U.getOperand(0);
547 Type *Int32Ty = Type::getInt32Ty(U.getContext());
548
549 // getIndexedOffsetInType is designed for GEPs, so the first index is the
550 // usual array element rather than looking into the actual aggregate.
551 SmallVector<Value *, 1> Indices;
552 Indices.push_back(ConstantInt::get(Int32Ty, 0));
553
554 if (const ExtractValueInst *EVI = dyn_cast<ExtractValueInst>(&U)) {
555 for (auto Idx : EVI->indices())
556 Indices.push_back(ConstantInt::get(Int32Ty, Idx));
557 } else if (const InsertValueInst *IVI = dyn_cast<InsertValueInst>(&U)) {
558 for (auto Idx : IVI->indices())
559 Indices.push_back(ConstantInt::get(Int32Ty, Idx));
560 } else {
561 for (unsigned i = 1; i < U.getNumOperands(); ++i)
562 Indices.push_back(U.getOperand(i));
563 }
564
565 return 8 * static_cast<uint64_t>(
566 DL.getIndexedOffsetInType(Src->getType(), Indices));
567}

569bool IRTranslator::translateExtractValue(const User &U,
570 MachineIRBuilder &MIRBuilder) {
571 const Value *Src = U.getOperand(0);
572 uint64_t Offset = getOffsetFromIndices(U, *DL);
573 ArrayRef<unsigned> SrcRegs = getOrCreateVRegs(*Src);
574 ArrayRef<uint64_t> Offsets = *VMap.getOffsets(*Src);
575 unsigned Idx = llvm::lower_bound(Offsets, Offset) - Offsets.begin();
576 auto &DstRegs = allocateVRegs(U);
577
578 for (unsigned i = 0; i < DstRegs.size(); ++i)
579 DstRegs[i] = SrcRegs[Idx++];
580
581 return true;
582}

584bool IRTranslator::translateInsertValue(const User &U,
585 MachineIRBuilder &MIRBuilder) {
586 const Value *Src = U.getOperand(0);
587 uint64_t Offset = getOffsetFromIndices(U, *DL);
588 auto &DstRegs = allocateVRegs(U);
589 ArrayRef<uint64_t> DstOffsets = *VMap.getOffsets(U);
590 ArrayRef<unsigned> SrcRegs = getOrCreateVRegs(*Src);
591 ArrayRef<unsigned> InsertedRegs = getOrCreateVRegs(*U.getOperand(1));
592 auto InsertedIt = InsertedRegs.begin();
593
594 for (unsigned i = 0; i < DstRegs.size(); ++i) {
595 if (DstOffsets[i] >= Offset && InsertedIt != InsertedRegs.end())
596 DstRegs[i] = *InsertedIt++;
597 else
598 DstRegs[i] = SrcRegs[i];
599 }
600
601 return true;
602}

624bool IRTranslator::translateBitCast(const User &U,
625 MachineIRBuilder &MIRBuilder) {
626 // If we're bitcasting to the source type, we can reuse the source vreg.
627 if (getLLTForType(*U.getOperand(0)->getType(), *DL) ==
628 getLLTForType(*U.getType(), *DL)) {
629 unsigned SrcReg = getOrCreateVReg(*U.getOperand(0));

632 // Emit a copy to satisfy the users we already emitted.
633 if (!Regs.empty())
634 MIRBuilder.buildCopy(Regs[0], SrcReg);
635 else {
636 Regs.push_back(SrcReg);
637 VMap.getOffsets(U)->push_back(0);
638 }
639 return true;
640 }
641 return translateCast(TargetOpcode::G_BITCAST, U, MIRBuilder);

652bool IRTranslator::translateGetElementPtr(const User &U,
653 MachineIRBuilder &MIRBuilder) {
654 // FIXME: support vector GEPs.
655 if (U.getType()->isVectorTy())
656 return false;
657
658 Value &Op0 = *U.getOperand(0);
659 unsigned BaseReg = getOrCreateVReg(Op0);
660 Type *PtrIRTy = Op0.getType();
661 LLT PtrTy = getLLTForType(*PtrIRTy, *DL);
662 Type *OffsetIRTy = DL->getIntPtrType(PtrIRTy);
663 LLT OffsetTy = getLLTForType(*OffsetIRTy, *DL);
664
665 int64_t Offset = 0;
666 for (gep_type_iterator GTI = gep_type_begin(&U), E = gep_type_end(&U);
667 GTI != E; ++GTI) {
668 const Value *Idx = GTI.getOperand();
669 if (StructType *StTy = GTI.getStructTypeOrNull()) {
670 unsigned Field = cast<Constant>(Idx)->getUniqueInteger().getZExtValue();
671 Offset += DL->getStructLayout(StTy)->getElementOffset(Field);
672 continue;
673 } else {
674 uint64_t ElementSize = DL->getTypeAllocSize(GTI.getIndexedType());
675
676 // If this is a scalar constant or a splat vector of constants,
677 // handle it quickly.
678 if (const auto *CI = dyn_cast<ConstantInt>(Idx)) {
679 Offset += ElementSize * CI->getSExtValue();
680 continue;
681 }
682
683 if (Offset != 0) {
684 unsigned NewBaseReg = MRI->createGenericVirtualRegister(PtrTy);
685 LLT OffsetTy = getLLTForType(*OffsetIRTy, *DL);
686 auto OffsetMIB = MIRBuilder.buildConstant({OffsetTy}, Offset);
687 MIRBuilder.buildGEP(NewBaseReg, BaseReg, OffsetMIB.getReg(0));
688
689 BaseReg = NewBaseReg;
690 Offset = 0;
691 }
692
693 unsigned IdxReg = getOrCreateVReg(*Idx);
694 if (MRI->getType(IdxReg) != OffsetTy) {
695 unsigned NewIdxReg = MRI->createGenericVirtualRegister(OffsetTy);
696 MIRBuilder.buildSExtOrTrunc(NewIdxReg, IdxReg);
697 IdxReg = NewIdxReg;
698 }
699
700 // N = N + Idx * ElementSize;
701 // Avoid doing it for ElementSize of 1.
702 unsigned GepOffsetReg;
703 if (ElementSize != 1) {
704 GepOffsetReg = MRI->createGenericVirtualRegister(OffsetTy);
705 auto ElementSizeMIB = MIRBuilder.buildConstant(
706 getLLTForType(*OffsetIRTy, *DL), ElementSize);
707 MIRBuilder.buildMul(GepOffsetReg, ElementSizeMIB.getReg(0), IdxReg);
708 } else
709 GepOffsetReg = IdxReg;
710
711 unsigned NewBaseReg = MRI->createGenericVirtualRegister(PtrTy);
712 MIRBuilder.buildGEP(NewBaseReg, BaseReg, GepOffsetReg);
713 BaseReg = NewBaseReg;
714 }
715 }
716
717 if (Offset != 0) {
718 auto OffsetMIB =
719 MIRBuilder.buildConstant(getLLTForType(*OffsetIRTy, *DL), Offset);
720 MIRBuilder.buildGEP(getOrCreateVReg(U), BaseReg, OffsetMIB.getReg(0));
721 return true;
722 }
723
724 MIRBuilder.buildCopy(getOrCreateVReg(U), BaseReg);
725 return true;
726}

728bool IRTranslator::translateMemfunc(const CallInst &CI,
729 MachineIRBuilder &MIRBuilder,
730 unsigned ID) {
731
732 // If the source is undef, then just emit a nop.
733 if (isa<UndefValue>(CI.getArgOperand(1))) {
734 switch (ID) {
735 case Intrinsic::memmove:
736 case Intrinsic::memcpy:
737 case Intrinsic::memset:
738 return true;
739 default:
740 break;
741 }
742 }
743
744 LLT SizeTy = getLLTForType(*CI.getArgOperand(2)->getType(), *DL);
745 Type *DstTy = CI.getArgOperand(0)->getType();
746 if (cast<PointerType>(DstTy)->getAddressSpace() != 0 ||
747 SizeTy.getSizeInBits() != DL->getPointerSizeInBits(0))
748 return false;
749
750 SmallVector<CallLowering::ArgInfo, 8> Args;
751 for (int i = 0; i < 3; ++i) {
752 const auto &Arg = CI.getArgOperand(i);
753 Args.emplace_back(getOrCreateVReg(*Arg), Arg->getType());
754 }
755
756 const char *Callee;
757 switch (ID) {
758 case Intrinsic::memmove:
759 case Intrinsic::memcpy: {
760 Type *SrcTy = CI.getArgOperand(1)->getType();
761 if(cast<PointerType>(SrcTy)->getAddressSpace() != 0)
762 return false;
763 Callee = ID == Intrinsic::memcpy ? "memcpy" : "memmove";
764 break;
765 }
766 case Intrinsic::memset:
767 Callee = "memset";
768 break;
769 default:
770 return false;
771 }
772
773 return CLI->lowerCall(MIRBuilder, CI.getCallingConv(),
774 MachineOperand::CreateES(Callee),
775 CallLowering::ArgInfo(0, CI.getType()), Args);
776}

880
881bool IRTranslator::translateKnownIntrinsic(const CallInst &CI, Intrinsic::ID ID,
882 MachineIRBuilder &MIRBuilder) {
883
884 // If this is a simple intrinsic (that is, we just need to add a def of
885 // a vreg, and uses for each arg operand, then translate it.
886 if (translateSimpleIntrinsic(CI, ID, MIRBuilder))
887 return true;
888
889 switch (ID) {
890 default:
891 break;
892 case Intrinsic::lifetime_start:
893 case Intrinsic::lifetime_end: {
894 // No stack colouring in O0, discard region information.
895 if (MF->getTarget().getOptLevel() == CodeGenOpt::None)
896 return true;
897
898 unsigned Op = ID == Intrinsic::lifetime_start ? TargetOpcode::LIFETIME_START
899 : TargetOpcode::LIFETIME_END;
900
901 // Get the underlying objects for the location passed on the lifetime
902 // marker.
903 SmallVector<const Value *, 4> Allocas;
904 GetUnderlyingObjects(CI.getArgOperand(1), Allocas, *DL);
905
906 // Iterate over each underlying object, creating lifetime markers for each
907 // static alloca. Quit if we find a non-static alloca.
908 for (const Value *V : Allocas) {
909 const AllocaInst *AI = dyn_cast<AllocaInst>(V);
910 if (!AI)
911 continue;
912
913 if (!AI->isStaticAlloca())
914 return true;
915
916 MIRBuilder.buildInstr(Op).addFrameIndex(getOrCreateFrameIndex(*AI));
917 }
918 return true;
919 }
920 case Intrinsic::dbg_declare: {
921 const DbgDeclareInst &DI = cast<DbgDeclareInst>(CI);
922 assert(DI.getVariable() && "Missing variable");
923
924 const Value *Address = DI.getAddress();
925 if (!Address || isa<UndefValue>(Address)) {
926 LLVM_DEBUG(dbgs() << "Dropping debug info for " << DI << "\n");
927 return true;
928 }
929
930 assert(DI.getVariable()->isValidLocationForIntrinsic(
931 MIRBuilder.getDebugLoc()) &&
932 "Expected inlined-at fields to agree");
933 auto AI = dyn_cast<AllocaInst>(Address);
934 if (AI && AI->isStaticAlloca()) {
935 // Static allocas are tracked at the MF level, no need for DBG_VALUE
936 // instructions (in fact, they get ignored if they *do* exist).
937 MF->setVariableDbgInfo(DI.getVariable(), DI.getExpression(),
938 getOrCreateFrameIndex(*AI), DI.getDebugLoc());
939 } else {
940 // A dbg.declare describes the address of a source variable, so lower it
941 // into an indirect DBG_VALUE.
942 MIRBuilder.buildIndirectDbgValue(getOrCreateVReg(*Address),
943 DI.getVariable(), DI.getExpression());
944 }
945 return true;
946 }
947 case Intrinsic::dbg_label: {
948 const DbgLabelInst &DI = cast<DbgLabelInst>(CI);
949 assert(DI.getLabel() && "Missing label");
950
951 assert(DI.getLabel()->isValidLocationForIntrinsic(
952 MIRBuilder.getDebugLoc()) &&
953 "Expected inlined-at fields to agree");
954
955 MIRBuilder.buildDbgLabel(DI.getLabel());
956 return true;
957 }
958 case Intrinsic::vaend:
959 // No target I know of cares about va_end. Certainly no in-tree target
960 // does. Simplest intrinsic ever!
961 return true;
962 case Intrinsic::vastart: {
963 auto &TLI = *MF->getSubtarget().getTargetLowering();
964 Value *Ptr = CI.getArgOperand(0);
965 unsigned ListSize =
966 TLI.getVaListSizeInBits(*DL, Ptr->getType()->getPointerAddressSpace()) /
967 8;
968
969 // FIXME: Get alignment
970 MIRBuilder.buildInstr(TargetOpcode::G_VASTART)
971 .addUse(getOrCreateVReg(*Ptr))
972 .addMemOperand(MF->getMachineMemOperand(
973 MachinePointerInfo(Ptr), MachineMemOperand::MOStore, ListSize, 1));
974 return true;
975 }
976 case Intrinsic::dbg_value: {
977 // This form of DBG_VALUE is target-independent.
978 const DbgValueInst &DI = cast<DbgValueInst>(CI);
979 const Value *V = DI.getValue();
980 assert(DI.getVariable()->isValidLocationForIntrinsic(
981 MIRBuilder.getDebugLoc()) &&
982 "Expected inlined-at fields to agree");
983 if (!V) {
984 // Currently the optimizer can produce this; insert an undef to
985 // help debugging. Probably the optimizer should not do this.
986 MIRBuilder.buildIndirectDbgValue(0, DI.getVariable(), DI.getExpression());
987 } else if (const auto *CI = dyn_cast<Constant>(V)) {
988 MIRBuilder.buildConstDbgValue(*CI, DI.getVariable(), DI.getExpression());
989 } else {
990 unsigned Reg = getOrCreateVReg(*V);
991 // FIXME: This does not handle register-indirect values at offset 0. The
992 // direct/indirect thing shouldn't really be handled by something as
993 // implicit as reg+noreg vs reg+imm in the first palce, but it seems
994 // pretty baked in right now.
995 MIRBuilder.buildDirectDbgValue(Reg, DI.getVariable(), DI.getExpression());
996 }
997 return true;
998 }
999 case Intrinsic::uadd_with_overflow:
1000 return translateOverflowIntrinsic(CI, TargetOpcode::G_UADDO, MIRBuilder);
1001 case Intrinsic::sadd_with_overflow:
1002 return translateOverflowIntrinsic(CI, TargetOpcode::G_SADDO, MIRBuilder);
1003 case Intrinsic::usub_with_overflow:
1004 return translateOverflowIntrinsic(CI, TargetOpcode::G_USUBO, MIRBuilder);
1005 case Intrinsic::ssub_with_overflow:
1006 return translateOverflowIntrinsic(CI, TargetOpcode::G_SSUBO, MIRBuilder);
1007 case Intrinsic::umul_with_overflow:
1008 return translateOverflowIntrinsic(CI, TargetOpcode::G_UMULO, MIRBuilder);
1009 case Intrinsic::smul_with_overflow:
1010 return translateOverflowIntrinsic(CI, TargetOpcode::G_SMULO, MIRBuilder);
1011 case Intrinsic::fmuladd: {
1012 const TargetMachine &TM = MF->getTarget();
1013 const TargetLowering &TLI = *MF->getSubtarget().getTargetLowering();
1014 unsigned Dst = getOrCreateVReg(CI);
1015 unsigned Op0 = getOrCreateVReg(*CI.getArgOperand(0));
1016 unsigned Op1 = getOrCreateVReg(*CI.getArgOperand(1));
1017 unsigned Op2 = getOrCreateVReg(*CI.getArgOperand(2));
1018 if (TM.Options.AllowFPOpFusion != FPOpFusion::Strict &&
1019 TLI.isFMAFasterThanFMulAndFAdd(TLI.getValueType(*DL, CI.getType()))) {
1020 // TODO: Revisit this to see if we should move this part of the
1021 // lowering to the combiner.
1022 MIRBuilder.buildInstr(TargetOpcode::G_FMA, {Dst}, {Op0, Op1, Op2},
1023 MachineInstr::copyFlagsFromInstruction(CI));
1024 } else {
1025 LLT Ty = getLLTForType(*CI.getType(), *DL);
1026 auto FMul = MIRBuilder.buildInstr(TargetOpcode::G_FMUL, {Ty}, {Op0, Op1},
1027 MachineInstr::copyFlagsFromInstruction(CI));
1028 MIRBuilder.buildInstr(TargetOpcode::G_FADD, {Dst}, {FMul, Op2},
1029 MachineInstr::copyFlagsFromInstruction(CI));
1030 }
1031 return true;
1032 }
1033 case Intrinsic::memcpy:
1034 case Intrinsic::memmove:
1035 case Intrinsic::memset:
1036 return translateMemfunc(CI, MIRBuilder, ID);
1037 case Intrinsic::eh_typeid_for: {
1038 GlobalValue *GV = ExtractTypeInfo(CI.getArgOperand(0));
1039 unsigned Reg = getOrCreateVReg(CI);
1040 unsigned TypeID = MF->getTypeIDFor(GV);
1041 MIRBuilder.buildConstant(Reg, TypeID);
1042 return true;
1043 }
1044 case Intrinsic::objectsize: {
1045 // If we don't know by now, we're never going to know.
1046 const ConstantInt *Min = cast<ConstantInt>(CI.getArgOperand(1));
1047
1048 MIRBuilder.buildConstant(getOrCreateVReg(CI), Min->isZero() ? -1ULL : 0);
1049 return true;
1050 }
1051 case Intrinsic::is_constant:
1052 // If this wasn't constant-folded away by now, then it's not a
1053 // constant.
1054 MIRBuilder.buildConstant(getOrCreateVReg(CI), 0);
1055 return true;
1056 case Intrinsic::stackguard:
1057 getStackGuard(getOrCreateVReg(CI), MIRBuilder);
1058 return true;
1059 case Intrinsic::stackprotector: {
1060 LLT PtrTy = getLLTForType(*CI.getArgOperand(0)->getType(), *DL);
1061 unsigned GuardVal = MRI->createGenericVirtualRegister(PtrTy);
1062 getStackGuard(GuardVal, MIRBuilder);
1063
1064 AllocaInst *Slot = cast<AllocaInst>(CI.getArgOperand(1));
1065 int FI = getOrCreateFrameIndex(*Slot);
1066 MF->getFrameInfo().setStackProtectorIndex(FI);
1067
1068 MIRBuilder.buildStore(
1069 GuardVal, getOrCreateVReg(*Slot),
1070 *MF->getMachineMemOperand(MachinePointerInfo::getFixedStack(*MF, FI),
1071 MachineMemOperand::MOStore |
1072 MachineMemOperand::MOVolatile,
1073 PtrTy.getSizeInBits() / 8, 8));
1074 return true;
1075 }
1076 case Intrinsic::stacksave: {
1077 // Save the stack pointer to the location provided by the intrinsic.
1078 unsigned Reg = getOrCreateVReg(CI);
1079 unsigned StackPtr = MF->getSubtarget()
1080 .getTargetLowering()
1081 ->getStackPointerRegisterToSaveRestore();
1082
1083 // If the target doesn't specify a stack pointer, then fall back.
1084 if (!StackPtr)
1085 return false;
1086
1087 MIRBuilder.buildCopy(Reg, StackPtr);
1088 return true;
1089 }
1090 case Intrinsic::stackrestore: {
1091 // Restore the stack pointer from the location provided by the intrinsic.
1092 unsigned Reg = getOrCreateVReg(*CI.getArgOperand(0));
1093 unsigned StackPtr = MF->getSubtarget()
1094 .getTargetLowering()
1095 ->getStackPointerRegisterToSaveRestore();
1096
1097 // If the target doesn't specify a stack pointer, then fall back.
1098 if (!StackPtr)
1099 return false;
1100
1101 MIRBuilder.buildCopy(StackPtr, Reg);
1102 return true;
1103 }
1104 case Intrinsic::cttz:
1105 case Intrinsic::ctlz: {
1106 ConstantInt *Cst = cast<ConstantInt>(CI.getArgOperand(1));
1107 bool isTrailing = ID == Intrinsic::cttz;
1108 unsigned Opcode = isTrailing
1109 ? Cst->isZero() ? TargetOpcode::G_CTTZ
1110 : TargetOpcode::G_CTTZ_ZERO_UNDEF
1111 : Cst->isZero() ? TargetOpcode::G_CTLZ
1112 : TargetOpcode::G_CTLZ_ZERO_UNDEF;
1113 MIRBuilder.buildInstr(Opcode)
1114 .addDef(getOrCreateVReg(CI))
1115 .addUse(getOrCreateVReg(*CI.getArgOperand(0)));
1116 return true;
1117 }
1118 case Intrinsic::invariant_start: {
1119 LLT PtrTy = getLLTForType(*CI.getArgOperand(0)->getType(), *DL);
1120 unsigned Undef = MRI->createGenericVirtualRegister(PtrTy);
1121 MIRBuilder.buildUndef(Undef);
1122 return true;
1123 }
1124 case Intrinsic::invariant_end:
1125 return true;
1126 case Intrinsic::assume:
1127 case Intrinsic::var_annotation:
1128 case Intrinsic::sideeffect:
1129 // Discard annotate attributes, assumptions, and artificial side-effects.
1130 return true;
1131 }
1132 return false;
1133}

1182bool IRTranslator::translateCall(const User &U, MachineIRBuilder &MIRBuilder) {
1183 const CallInst &CI = cast<CallInst>(U);
1184 auto TII = MF->getTarget().getIntrinsicInfo();
1185 const Function *F = CI.getCalledFunction();
1186
1187 // FIXME: support Windows dllimport function calls.
1188 if (F && F->hasDLLImportStorageClass())
1189 return false;
1190
1191 if (CI.isInlineAsm())
1192 return translateInlineAsm(CI, MIRBuilder);
1193
1194 Intrinsic::ID ID = Intrinsic::not_intrinsic;
1195 if (F && F->isIntrinsic()) {
1196 ID = F->getIntrinsicID();
1197 if (TII && ID == Intrinsic::not_intrinsic)
1198 ID = static_cast<Intrinsic::ID>(TII->getIntrinsicID(F));
1199 }
1200
1201 if (!F || !F->isIntrinsic() || ID == Intrinsic::not_intrinsic) {
1202 bool IsSplitType = valueIsSplit(CI);
1203 unsigned Res = IsSplitType ? MRI->createGenericVirtualRegister(
1204 getLLTForType(*CI.getType(), *DL))
1205 : getOrCreateVReg(CI);
1206
1207 SmallVector<unsigned, 8> Args;
1208 unsigned SwiftErrorVReg = 0;
1209 for (auto &Arg: CI.arg_operands()) {
1210 if (CLI->supportSwiftError() && isSwiftError(Arg)) {
1211 LLT Ty = getLLTForType(*Arg->getType(), *DL);
1212 unsigned InVReg = MRI->createGenericVirtualRegister(Ty);
1213 MIRBuilder.buildCopy(InVReg, SwiftError.getOrCreateVRegUseAt(
1214 &CI, &MIRBuilder.getMBB(), Arg));
1215 Args.push_back(InVReg);
1216 SwiftErrorVReg =
1217 SwiftError.getOrCreateVRegDefAt(&CI, &MIRBuilder.getMBB(), Arg);
1218 continue;
1219 }
1220 Args.push_back(packRegs(*Arg, MIRBuilder));
1221 }
1222
1223 MF->getFrameInfo().setHasCalls(true);
1224 bool Success =
1225 CLI->lowerCall(MIRBuilder, &CI, Res, Args, SwiftErrorVReg,
1226 [&]() { return getOrCreateVReg(*CI.getCalledValue()); });
1227
1228 if (IsSplitType)
1229 unpackRegs(CI, Res, MIRBuilder);
1230
1231 return Success;
1232 }
1233
1234 assert(ID != Intrinsic::not_intrinsic && "unknown intrinsic");
1235
1236 if (translateKnownIntrinsic(CI, ID, MIRBuilder))
1237 return true;
1238
1239 ArrayRef<unsigned> ResultRegs;
1240 if (!CI.getType()->isVoidTy())
1241 ResultRegs = getOrCreateVRegs(CI);
1242
1243 MachineInstrBuilder MIB =
1244 MIRBuilder.buildIntrinsic(ID, ResultRegs, !CI.doesNotAccessMemory());
1245 if (isa<FPMathOperator>(CI))
1246 MIB->copyIRFlags(CI);
1247
1248 for (auto &Arg : CI.arg_operands()) {
1249 // Some intrinsics take metadata parameters. Reject them.
1250 if (isa<MetadataAsValue>(Arg))
1251 return false;
1252 MIB.addUse(packRegs(*Arg, MIRBuilder));
1253 }
1254
1255 // Add a MachineMemOperand if it is a target mem intrinsic.
1256 const TargetLowering &TLI = *MF->getSubtarget().getTargetLowering();
1257 TargetLowering::IntrinsicInfo Info;
1258 // TODO: Add a GlobalISel version of getTgtMemIntrinsic.
1259 if (TLI.getTgtMemIntrinsic(Info, CI, *MF, ID)) {
1260 unsigned Align = Info.align;
1261 if (Align == 0)
1262 Align = DL->getABITypeAlignment(Info.memVT.getTypeForEVT(F->getContext()));
1263
1264 uint64_t Size = Info.memVT.getStoreSize();
1265 MIB.addMemOperand(MF->getMachineMemOperand(MachinePointerInfo(Info.ptrVal),
1266 Info.flags, Size, Align));
1267 }
1268
1269 return true;
1270}

1348bool IRTranslator::translateLandingPad(const User &U,
1349 MachineIRBuilder &MIRBuilder) {
1350 const LandingPadInst &LP = cast<LandingPadInst>(U);
1351
1352 MachineBasicBlock &MBB = MIRBuilder.getMBB();
1353
1354 MBB.setIsEHPad();
1355
1356 // If there aren't registers to copy the values into (e.g., during SjLj
1357 // exceptions), then don't bother.
1358 auto &TLI = *MF->getSubtarget().getTargetLowering();
1359 const Constant *PersonalityFn = MF->getFunction().getPersonalityFn();
1360 if (TLI.getExceptionPointerRegister(PersonalityFn) == 0 &&
1361 TLI.getExceptionSelectorRegister(PersonalityFn) == 0)
1362 return true;
1363
1364 // If landingpad's return type is token type, we don't create DAG nodes
1365 // for its exception pointer and selector value. The extraction of exception
1366 // pointer or selector value from token type landingpads is not currently
1367 // supported.
1368 if (LP.getType()->isTokenTy())
1369 return true;
1370
1371 // Add a label to mark the beginning of the landing pad. Deletion of the
1372 // landing pad can thus be detected via the MachineModuleInfo.
1373 MIRBuilder.buildInstr(TargetOpcode::EH_LABEL)
1374 .addSym(MF->addLandingPad(&MBB));
1375
1376 LLT Ty = getLLTForType(*LP.getType(), *DL);
1377 unsigned Undef = MRI->createGenericVirtualRegister(Ty);
1378 MIRBuilder.buildUndef(Undef);
1379
1380 SmallVector<LLT, 2> Tys;
1381 for (Type *Ty : cast<StructType>(LP.getType())->elements())
1382 Tys.push_back(getLLTForType(*Ty, *DL));
1383 assert(Tys.size() == 2 && "Only two-valued landingpads are supported");
1384
1385 // Mark exception register as live in.
1386 unsigned ExceptionReg = TLI.getExceptionPointerRegister(PersonalityFn);
1387 if (!ExceptionReg)
1388 return false;
1389
1390 MBB.addLiveIn(ExceptionReg);
1391 ArrayRef<unsigned> ResRegs = getOrCreateVRegs(LP);
1392 MIRBuilder.buildCopy(ResRegs[0], ExceptionReg);
1393
1394 unsigned SelectorReg = TLI.getExceptionSelectorRegister(PersonalityFn);
1395 if (!SelectorReg)
1396 return false;
1397
1398 MBB.addLiveIn(SelectorReg);
1399 unsigned PtrVReg = MRI->createGenericVirtualRegister(Tys[0]);
1400 MIRBuilder.buildCopy(PtrVReg, SelectorReg);
1401 MIRBuilder.buildCast(ResRegs[1], PtrVReg);
1402
1403 return true;
1404}

1406bool IRTranslator::translateAlloca(const User &U,
1407 MachineIRBuilder &MIRBuilder) {
1408 auto &AI = cast<AllocaInst>(U);
1409
1410 if (AI.isSwiftError())
1411 return true;
1412
1413 if (AI.isStaticAlloca()) {
1414 unsigned Res = getOrCreateVReg(AI);
1415 int FI = getOrCreateFrameIndex(AI);
1416 MIRBuilder.buildFrameIndex(Res, FI);
1417 return true;
1418 }
1419
1420 // FIXME: support stack probing for Windows.
1421 if (MF->getTarget().getTargetTriple().isOSWindows())
1422 return false;
1423
1424 // Now we're in the harder dynamic case.
1425 Type *Ty = AI.getAllocatedType();
1426 unsigned Align =
1427 std::max((unsigned)DL->getPrefTypeAlignment(Ty), AI.getAlignment());
1428
1429 unsigned NumElts = getOrCreateVReg(*AI.getArraySize());
1430
1431 Type *IntPtrIRTy = DL->getIntPtrType(AI.getType());
1432 LLT IntPtrTy = getLLTForType(*IntPtrIRTy, *DL);
1433 if (MRI->getType(NumElts) != IntPtrTy) {
1434 unsigned ExtElts = MRI->createGenericVirtualRegister(IntPtrTy);
1435 MIRBuilder.buildZExtOrTrunc(ExtElts, NumElts);
1436 NumElts = ExtElts;
1437 }
1438
1439 unsigned AllocSize = MRI->createGenericVirtualRegister(IntPtrTy);
1440 unsigned TySize =
1441 getOrCreateVReg(*ConstantInt::get(IntPtrIRTy, -DL->getTypeAllocSize(Ty)));
1442 MIRBuilder.buildMul(AllocSize, NumElts, TySize);
1443
1444 LLT PtrTy = getLLTForType(*AI.getType(), *DL);
1445 auto &TLI = *MF->getSubtarget().getTargetLowering();
1446 unsigned SPReg = TLI.getStackPointerRegisterToSaveRestore();
1447
1448 unsigned SPTmp = MRI->createGenericVirtualRegister(PtrTy);
1449 MIRBuilder.buildCopy(SPTmp, SPReg);
1450
1451 unsigned AllocTmp = MRI->createGenericVirtualRegister(PtrTy);
1452 MIRBuilder.buildGEP(AllocTmp, SPTmp, AllocSize);
1453
1454 // Handle alignment. We have to realign if the allocation granule was smaller
1455 // than stack alignment, or the specific alloca requires more than stack
1456 // alignment.
1457 unsigned StackAlign =
1458 MF->getSubtarget().getFrameLowering()->getStackAlignment();
1459 Align = std::max(Align, StackAlign);
1460 if (Align > StackAlign || DL->getTypeAllocSize(Ty) % StackAlign != 0) {
1461 // Round the size of the allocation up to the stack alignment size
1462 // by add SA-1 to the size. This doesn't overflow because we're computing
1463 // an address inside an alloca.
1464 unsigned AlignedAlloc = MRI->createGenericVirtualRegister(PtrTy);
1465 MIRBuilder.buildPtrMask(AlignedAlloc, AllocTmp, Log2_32(Align));
1466 AllocTmp = AlignedAlloc;
1467 }
1468
1469 MIRBuilder.buildCopy(SPReg, AllocTmp);
1470 MIRBuilder.buildCopy(getOrCreateVReg(AI), AllocTmp);
1471
1472 MF->getFrameInfo().CreateVariableSizedObject(Align ? Align : 1, &AI);
1473 assert(MF->getFrameInfo().hasVarSizedObjects());
1474 return true;
1475}

1729bool IRTranslator::translate(const Constant &C, unsigned Reg) {
1730 if (auto CI = dyn_cast<ConstantInt>(&C))
1731 EntryBuilder->buildConstant(Reg, *CI);
1732 else if (auto CF = dyn_cast<ConstantFP>(&C))
1733 EntryBuilder->buildFConstant(Reg, *CF);
1734 else if (isa<UndefValue>(C))
1735 EntryBuilder->buildUndef(Reg);
1736 else if (isa<ConstantPointerNull>(C)) {
1737 // As we are trying to build a constant val of 0 into a pointer,
1738 // insert a cast to make them correct with respect to types.
1739 unsigned NullSize = DL->getTypeSizeInBits(C.getType());
1740 auto *ZeroTy = Type::getIntNTy(C.getContext(), NullSize);
1741 auto *ZeroVal = ConstantInt::get(ZeroTy, 0);
1742 unsigned ZeroReg = getOrCreateVReg(*ZeroVal);
1743 EntryBuilder->buildCast(Reg, ZeroReg);
1744 } else if (auto GV = dyn_cast<GlobalValue>(&C))
1745 EntryBuilder->buildGlobalValue(Reg, GV);
1746 else if (auto CAZ = dyn_cast<ConstantAggregateZero>(&C)) {
1747 if (!CAZ->getType()->isVectorTy())
1748 return false;
1749 // Return the scalar if it is a <1 x Ty> vector.
1750 if (CAZ->getNumElements() == 1)
1751 return translate(*CAZ->getElementValue(0u), Reg);
1752 SmallVector<unsigned, 4> Ops;
1753 for (unsigned i = 0; i < CAZ->getNumElements(); ++i) {
1754 Constant &Elt = *CAZ->getElementValue(i);
1755 Ops.push_back(getOrCreateVReg(Elt));
1756 }
1757 EntryBuilder->buildBuildVector(Reg, Ops);
1758 } else if (auto CV = dyn_cast<ConstantDataVector>(&C)) {
1759 // Return the scalar if it is a <1 x Ty> vector.
1760 if (CV->getNumElements() == 1)
1761 return translate(*CV->getElementAsConstant(0), Reg);
1762 SmallVector<unsigned, 4> Ops;
1763 for (unsigned i = 0; i < CV->getNumElements(); ++i) {
1764 Constant &Elt = *CV->getElementAsConstant(i);
1765 Ops.push_back(getOrCreateVReg(Elt));
1766 }
1767 EntryBuilder->buildBuildVector(Reg, Ops);
1768 } else if (auto CE = dyn_cast<ConstantExpr>(&C)) {
1769 switch(CE->getOpcode()) {
1770#define HANDLE_INST(NUM, OPCODE, CLASS) \
1771 case Instruction::OPCODE: \
1772 return translate##OPCODE(*CE, *EntryBuilder.get());
1773#include "llvm/IR/Instruction.def"
1774 default:
1775 return false;
1776 }
1777 } else if (auto CV = dyn_cast<ConstantVector>(&C)) {
1778 if (CV->getNumOperands() == 1)
1779 return translate(*CV->getOperand(0), Reg);
1780 SmallVector<unsigned, 4> Ops;
1781 for (unsigned i = 0; i < CV->getNumOperands(); ++i) {
1782 Ops.push_back(getOrCreateVReg(*CV->getOperand(i)));
1783 }
1784 EntryBuilder->buildBuildVector(Reg, Ops);
1785 } else if (auto *BA = dyn_cast<BlockAddress>(&C)) {
1786 EntryBuilder->buildBlockAddress(Reg, BA);
1787 } else
1788 return false;
1789
1790 return true;
1791}

1793void IRTranslator::finalizeFunction() {
1794 // Release the memory used by the different maps we
1795 // needed during the translation.
1796 PendingPHIs.clear();
1797 VMap.reset();
1798 FrameIndices.clear();
1799 MachinePreds.clear();
1800 // MachineIRBuilder::DebugLoc can outlive the DILocation it holds. Clear it
1801 // to avoid accessing free’d memory (in runOnMachineFunction) and to avoid
1802 // destroying it twice (in ~IRTranslator() and ~LLVMContext())
1803 EntryBuilder.reset();
1804 CurBuilder.reset();
1805}

1807bool IRTranslator::runOnMachineFunction(MachineFunction &CurMF) {
1808 MF = &CurMF;
1809 const Function &F = MF->getFunction();
1810 if (F.empty())
1811 return false;
1812 GISelCSEAnalysisWrapper &Wrapper =
1813 getAnalysis<GISelCSEAnalysisWrapperPass>().getCSEWrapper();
1814 // Set the CSEConfig and run the analysis.
1815 GISelCSEInfo *CSEInfo = nullptr;
1816 TPC = &getAnalysis<TargetPassConfig>();
1817 bool EnableCSE = EnableCSEInIRTranslator.getNumOccurrences()
1818 ? EnableCSEInIRTranslator
1819 : TPC->isGISelCSEEnabled();
1820
1821 if (EnableCSE) {
1822 EntryBuilder = make_unique<CSEMIRBuilder>(CurMF);
1823 CSEInfo = &Wrapper.get(TPC->getCSEConfig());
1824 EntryBuilder->setCSEInfo(CSEInfo);
1825 CurBuilder = make_unique<CSEMIRBuilder>(CurMF);
1826 CurBuilder->setCSEInfo(CSEInfo);
1827 } else {
1828 EntryBuilder = make_unique<MachineIRBuilder>();
1829 CurBuilder = make_unique<MachineIRBuilder>();
1830 }
1831 CLI = MF->getSubtarget().getCallLowering();
1832 CurBuilder->setMF(*MF);
1833 EntryBuilder->setMF(*MF);
1834 MRI = &MF->getRegInfo();
1835 DL = &F.getParent()->getDataLayout();
1836 ORE = llvm::make_unique<OptimizationRemarkEmitter>(&F);
1837
1838 assert(PendingPHIs.empty() && "stale PHIs");
1839
1840 if (!DL->isLittleEndian()) {
1841 // Currently we don't properly handle big endian code.
1842 OptimizationRemarkMissed R("gisel-irtranslator", "GISelFailure",
1843 F.getSubprogram(), &F.getEntryBlock());
1844 R << "unable to translate in big endian mode";
1845 reportTranslationError(*MF, *TPC, *ORE, R);
1846 }
1847
1848 // Release the per-function state when we return, whether we succeeded or not.
1849 auto FinalizeOnReturn = make_scope_exit([this]() { finalizeFunction(); });
1850
1851 // Setup a separate basic-block for the arguments and constants
1852 MachineBasicBlock *EntryBB = MF->CreateMachineBasicBlock();
1853 MF->push_back(EntryBB);
1854 EntryBuilder->setMBB(*EntryBB);
1855
1856 DebugLoc DbgLoc = F.getEntryBlock().getFirstNonPHI()->getDebugLoc();
1857 SwiftError.setFunction(CurMF);
1858 SwiftError.createEntriesInEntryBlock(DbgLoc);
1859
1860 // Create all blocks, in IR order, to preserve the layout.
1861 for (const BasicBlock &BB: F) {
1862 auto *&MBB = BBToMBB[&BB];
1863
1864 MBB = MF->CreateMachineBasicBlock(&BB);
1865 MF->push_back(MBB);
1866
1867 if (BB.hasAddressTaken())
1868 MBB->setHasAddressTaken();
1869 }
1870
1871 // Make our arguments/constants entry block fallthrough to the IR entry block.
1872 EntryBB->addSuccessor(&getMBB(F.front()));
1873
1874 // Lower the actual args into this basic block.
1875 SmallVector<unsigned, 8> VRegArgs;
1876 for (const Argument &Arg: F.args()) {
1877 if (DL->getTypeStoreSize(Arg.getType()) == 0)
1878 continue; // Don't handle zero sized types.
1879 VRegArgs.push_back(
1880 MRI->createGenericVirtualRegister(getLLTForType(*Arg.getType(), *DL)));
1881
1882 if (Arg.hasSwiftErrorAttr())
1883 SwiftError.setCurrentVReg(EntryBB, SwiftError.getFunctionArg(),
1884 VRegArgs.back());
1885 }
1886
1887 // We don't currently support translating swifterror or swiftself functions.
1888 for (auto &Arg : F.args()) {
1889 if (Arg.hasSwiftSelfAttr()) {
1890 OptimizationRemarkMissed R("gisel-irtranslator", "GISelFailure",
1891 F.getSubprogram(), &F.getEntryBlock());
1892 R << "unable to lower arguments due to swiftself: "
1893 << ore::NV("Prototype", F.getType());
1894 reportTranslationError(*MF, *TPC, *ORE, R);
1895 return false;
1896 }
1897 }
1898
1899 if (!CLI->lowerFormalArguments(*EntryBuilder.get(), F, VRegArgs)) {
1900 OptimizationRemarkMissed R("gisel-irtranslator", "GISelFailure",
1901 F.getSubprogram(), &F.getEntryBlock());
1902 R << "unable to lower arguments: " << ore::NV("Prototype", F.getType());
1903 reportTranslationError(*MF, *TPC, *ORE, R);
1904 return false;
1905 }
1906
1907 auto ArgIt = F.arg_begin();
1908 for (auto &VArg : VRegArgs) {
1909 // If the argument is an unsplit scalar then don't use unpackRegs to avoid
1910 // creating redundant copies.
1911 if (!valueIsSplit(*ArgIt, VMap.getOffsets(*ArgIt))) {
1912 auto &VRegs = *VMap.getVRegs(cast<Value>(*ArgIt));
1913 assert(VRegs.empty() && "VRegs already populated?");
1914 VRegs.push_back(VArg);
1915 } else {
1916 unpackRegs(*ArgIt, VArg, *EntryBuilder.get());
1917 }
1918 ArgIt++;
1919 }
1920
1921 // Need to visit defs before uses when translating instructions.
1922 GISelObserverWrapper WrapperObserver;
1923 if (EnableCSE && CSEInfo)
1924 WrapperObserver.addObserver(CSEInfo);
1925 {
1926 ReversePostOrderTraversal<const Function *> RPOT(&F);
1927#ifndef NDEBUG
1928 DILocationVerifier Verifier;
1929 WrapperObserver.addObserver(&Verifier);
1930#endif // ifndef NDEBUG
1931 RAIIDelegateInstaller DelInstall(*MF, &WrapperObserver);
1932 for (const BasicBlock *BB : RPOT) {
1933 MachineBasicBlock &MBB = getMBB(*BB);
1934 // Set the insertion point of all the following translations to
1935 // the end of this basic block.
1936 CurBuilder->setMBB(MBB);
1937
1938 for (const Instruction &Inst : *BB) {
1939#ifndef NDEBUG
1940 Verifier.setCurrentInst(&Inst);
1941#endif // ifndef NDEBUG
1942 if (translate(Inst))
1943 continue;
1944
1945 OptimizationRemarkMissed R("gisel-irtranslator", "GISelFailure",
1946 Inst.getDebugLoc(), BB);
1947 R << "unable to translate instruction: " << ore::NV("Opcode", &Inst);
1948
1949 if (ORE->allowExtraAnalysis("gisel-irtranslator")) {
1950 std::string InstStrStorage;
1951 raw_string_ostream InstStr(InstStrStorage);
1952 InstStr << Inst;
1953
1954 R << ": '" << InstStr.str() << "'";
1955 }
1956
1957 reportTranslationError(*MF, *TPC, *ORE, R);
1958 return false;
1959 }
1960 }
1961#ifndef NDEBUG
1962 WrapperObserver.removeObserver(&Verifier);
1963#endif
1964 }
1965
1966 finishPendingPhis();
1967
1968 SwiftError.propagateVRegs();
1969
1970 // Merge the argument lowering and constants block with its single
1971 // successor, the LLVM-IR entry block. We want the basic block to
1972 // be maximal.
1973 assert(EntryBB->succ_size() == 1 &&
1974 "Custom BB used for lowering should have only one successor");
1975 // Get the successor of the current entry block.
1976 MachineBasicBlock &NewEntryBB = **EntryBB->succ_begin();
1977 assert(NewEntryBB.pred_size() == 1 &&
1978 "LLVM-IR entry block has a predecessor!?");
1979 // Move all the instruction from the current entry block to the
1980 // new entry block.
1981 NewEntryBB.splice(NewEntryBB.begin(), EntryBB, EntryBB->begin(),
1982 EntryBB->end());
1983
1984 // Update the live-in information for the new entry block.
1985 for (const MachineBasicBlock::RegisterMaskPair &LiveIn : EntryBB->liveins())
1986 NewEntryBB.addLiveIn(LiveIn);
1987 NewEntryBB.sortUniqueLiveIns();
1988
1989 // Get rid of the now empty basic block.
1990 EntryBB->removeSuccessor(&NewEntryBB);
1991 MF->remove(EntryBB);
1992 MF->DeleteMachineBasicBlock(EntryBB);
1993
1994 assert(&MF->front() == &NewEntryBB &&
1995 "New entry wasn't next in the list of basic block!");
1996
1997 // Initialize stack protector information.
1998 StackProtector &SP = getAnalysis<StackProtector>();
1999 SP.copyToMachineFrameInfo(MF->getFrameInfo());
2000
2001 return false;
2002}
