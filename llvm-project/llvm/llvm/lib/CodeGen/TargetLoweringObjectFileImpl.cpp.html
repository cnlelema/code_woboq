64static void GetObjCImageInfo(Module &M, unsigned &Version, unsigned &Flags,
65 StringRef &Section) {
66 SmallVector<Module::ModuleFlagEntry, 8> ModuleFlags;
67 M.getModuleFlagsMetadata(ModuleFlags);
68
69 for (const auto &MFE: ModuleFlags) {
70 // Ignore flags with 'Require' behaviour.
71 if (MFE.Behavior == Module::Require)
72 continue;
73
74 StringRef Key = MFE.Key->getString();
75 if (Key == "Objective-C Image Info Version") {
76 Version = mdconst::extract<ConstantInt>(MFE.Val)->getZExtValue();
77 } else if (Key == "Objective-C Garbage Collection" ||
78 Key == "Objective-C GC Only" ||
79 Key == "Objective-C Is Simulated" ||
80 Key == "Objective-C Class Properties" ||
81 Key == "Objective-C Image Swift Version") {
82 Flags |= mdconst::extract<ConstantInt>(MFE.Val)->getZExtValue();
83 } else if (Key == "Objective-C Image Info Section") {
84 Section = cast<MDString>(MFE.Val)->getString();
85 }
86 }
87}

93void TargetLoweringObjectFileELF::Initialize(MCContext &Ctx,
94 const TargetMachine &TgtM) {
95 TargetLoweringObjectFile::Initialize(Ctx, TgtM);
96 TM = &TgtM;
97
98 CodeModel::Model CM = TgtM.getCodeModel();
99
100 switch (TgtM.getTargetTriple().getArch()) {
101 case Triple::arm:
102 case Triple::armeb:
103 case Triple::thumb:
104 case Triple::thumbeb:
105 if (Ctx.getAsmInfo()->getExceptionHandlingType() == ExceptionHandling::ARM)
106 break;
107 // Fallthrough if not using EHABI
108 LLVM_FALLTHROUGH;
109 case Triple::ppc:
110 case Triple::x86:
111 PersonalityEncoding = isPositionIndependent()
112 ? dwarf::DW_EH_PE_indirect |
113 dwarf::DW_EH_PE_pcrel |
114 dwarf::DW_EH_PE_sdata4
115 : dwarf::DW_EH_PE_absptr;
116 LSDAEncoding = isPositionIndependent()
117 ? dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4
118 : dwarf::DW_EH_PE_absptr;
119 TTypeEncoding = isPositionIndependent()
120 ? dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
121 dwarf::DW_EH_PE_sdata4
122 : dwarf::DW_EH_PE_absptr;
123 break;
124 case Triple::x86_64:
125 if (isPositionIndependent()) {
126 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
127 ((CM == CodeModel::Small || CM == CodeModel::Medium)
128 ? dwarf::DW_EH_PE_sdata4 : dwarf::DW_EH_PE_sdata8);
129 LSDAEncoding = dwarf::DW_EH_PE_pcrel |
130 (CM == CodeModel::Small
131 ? dwarf::DW_EH_PE_sdata4 : dwarf::DW_EH_PE_sdata8);
132 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
133 ((CM == CodeModel::Small || CM == CodeModel::Medium)
134 ? dwarf::DW_EH_PE_sdata8 : dwarf::DW_EH_PE_sdata4);
135 } else {
136 PersonalityEncoding =
137 (CM == CodeModel::Small || CM == CodeModel::Medium)
138 ? dwarf::DW_EH_PE_udata4 : dwarf::DW_EH_PE_absptr;
139 LSDAEncoding = (CM == CodeModel::Small)
140 ? dwarf::DW_EH_PE_udata4 : dwarf::DW_EH_PE_absptr;
141 TTypeEncoding = (CM == CodeModel::Small)
142 ? dwarf::DW_EH_PE_udata4 : dwarf::DW_EH_PE_absptr;
143 }
144 break;
145 case Triple::hexagon:
146 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
147 LSDAEncoding = dwarf::DW_EH_PE_absptr;
148 TTypeEncoding = dwarf::DW_EH_PE_absptr;
149 if (isPositionIndependent()) {
150 PersonalityEncoding |= dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel;
151 LSDAEncoding |= dwarf::DW_EH_PE_pcrel;
152 TTypeEncoding |= dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel;
153 }
154 break;
155 case Triple::aarch64:
156 case Triple::aarch64_be:
157 // The small model guarantees static code/data size < 4GB, but not where it
158 // will be in memory. Most of these could end up >2GB away so even a signed
159 // pc-relative 32-bit address is insufficient, theoretically.
160 if (isPositionIndependent()) {
161 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
162 dwarf::DW_EH_PE_sdata8;
163 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata8;
164 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
165 dwarf::DW_EH_PE_sdata8;
166 } else {
167 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
168 LSDAEncoding = dwarf::DW_EH_PE_absptr;
169 TTypeEncoding = dwarf::DW_EH_PE_absptr;
170 }
171 break;
172 case Triple::lanai:
173 LSDAEncoding = dwarf::DW_EH_PE_absptr;
174 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
175 TTypeEncoding = dwarf::DW_EH_PE_absptr;
176 break;
177 case Triple::mips:
178 case Triple::mipsel:
179 case Triple::mips64:
180 case Triple::mips64el:
181 case Triple::cheri:
182 // MIPS uses indirect pointer to refer personality functions and types, so
183 // that the eh_frame section can be read-only. DW.ref.personality will be
184 // generated for relocation.
185 PersonalityEncoding = dwarf::DW_EH_PE_indirect;
186 // FIXME: The N64 ABI probably ought to use DW_EH_PE_sdata8 but we can't
187 // identify N64 from just a triple.
188 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
189 dwarf::DW_EH_PE_sdata4;
190 // We don't support PC-relative LSDA references in GAS so we use the default
191 // DW_EH_PE_absptr for those.
192
193 // FreeBSD must be explicit about the data size and using pcrel since it's
194 // assembler/linker won't do the automatic conversion that the Linux tools
195 // do.
196 if (TgtM.getTargetTriple().isOSFreeBSD()) {
197 PersonalityEncoding |= dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
198 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
199 }
200 break;
201 case Triple::ppc64:
202 case Triple::ppc64le:
203 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
204 dwarf::DW_EH_PE_udata8;
205 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_udata8;
206 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
207 dwarf::DW_EH_PE_udata8;
208 break;
209 case Triple::sparcel:
210 case Triple::sparc:
211 if (isPositionIndependent()) {
212 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
213 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
214 dwarf::DW_EH_PE_sdata4;
215 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
216 dwarf::DW_EH_PE_sdata4;
217 } else {
218 LSDAEncoding = dwarf::DW_EH_PE_absptr;
219 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
220 TTypeEncoding = dwarf::DW_EH_PE_absptr;
221 }
222 break;
223 case Triple::sparcv9:
224 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
225 if (isPositionIndependent()) {
226 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
227 dwarf::DW_EH_PE_sdata4;
228 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
229 dwarf::DW_EH_PE_sdata4;
230 } else {
231 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
232 TTypeEncoding = dwarf::DW_EH_PE_absptr;
233 }
234 break;
235 case Triple::systemz:
236 // All currently-defined code models guarantee that 4-byte PC-relative
237 // values will be in range.
238 if (isPositionIndependent()) {
239 PersonalityEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
240 dwarf::DW_EH_PE_sdata4;
241 LSDAEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;
242 TTypeEncoding = dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel |
243 dwarf::DW_EH_PE_sdata4;
244 } else {
245 PersonalityEncoding = dwarf::DW_EH_PE_absptr;
246 LSDAEncoding = dwarf::DW_EH_PE_absptr;
247 TTypeEncoding = dwarf::DW_EH_PE_absptr;
248 }
249 break;
250 default:
251 break;
252 }
253}

255void TargetLoweringObjectFileELF::emitModuleMetadata(MCStreamer &Streamer,
256 Module &M) const {
257 auto &C = getContext();
258
259 if (NamedMDNode *LinkerOptions = M.getNamedMetadata("llvm.linker.options")) {
260 auto *S = C.getELFSection(".linker-options", ELF::SHT_LLVM_LINKER_OPTIONS,
261 ELF::SHF_EXCLUDE);
262
263 Streamer.SwitchSection(S);
264
265 for (const auto &Operand : LinkerOptions->operands()) {
266 if (cast<MDNode>(Operand)->getNumOperands() != 2)
267 report_fatal_error("invalid llvm.linker.options");
268 for (const auto &Option : cast<MDNode>(Operand)->operands()) {
269 Streamer.EmitBytes(cast<MDString>(Option)->getString());
270 Streamer.EmitIntValue(0, 1);
271 }
272 }
273 }
274
275 if (NamedMDNode *DependentLibraries = M.getNamedMetadata("llvm.dependent-libraries")) {
276 auto *S = C.getELFSection(".deplibs", ELF::SHT_LLVM_DEPENDENT_LIBRARIES,
277 ELF::SHF_MERGE | ELF::SHF_STRINGS, 1, "");
278
279 Streamer.SwitchSection(S);
280
281 for (const auto &Operand : DependentLibraries->operands()) {
282 Streamer.EmitBytes(
283 cast<MDString>(cast<MDNode>(Operand)->getOperand(0))->getString());
284 Streamer.EmitIntValue(0, 1);
285 }
286 }
287
288 unsigned Version = 0;
289 unsigned Flags = 0;
290 StringRef Section;
291
292 GetObjCImageInfo(M, Version, Flags, Section);
293 if (!Section.empty()) {
294 auto *S = C.getELFSection(Section, ELF::SHT_PROGBITS, ELF::SHF_ALLOC);
295 Streamer.SwitchSection(S);
296 Streamer.EmitLabel(C.getOrCreateSymbol(StringRef("OBJC_IMAGE_INFO")));
297 Streamer.EmitIntValue(Version, 4);
298 Streamer.EmitIntValue(Flags, 4);
299 Streamer.AddBlankLine();
300 }
301
302 SmallVector<Module::ModuleFlagEntry, 8> ModuleFlags;
303 M.getModuleFlagsMetadata(ModuleFlags);
304
305 MDNode *CFGProfile = nullptr;
306
307 for (const auto &MFE : ModuleFlags) {
308 StringRef Key = MFE.Key->getString();
309 if (Key == "CG Profile") {
310 CFGProfile = cast<MDNode>(MFE.Val);
311 break;
312 }
313 }
314
315 if (!CFGProfile)
316 return;
317
318 auto GetSym = [this](const MDOperand &MDO) -> MCSymbol * {
319 if (!MDO)
320 return nullptr;
321 auto V = cast<ValueAsMetadata>(MDO);
322 const Function *F = cast<Function>(V->getValue());
323 return TM->getSymbol(F);
324 };
325
326 for (const auto &Edge : CFGProfile->operands()) {
327 MDNode *E = cast<MDNode>(Edge);
328 const MCSymbol *From = GetSym(E->getOperand(0));
329 const MCSymbol *To = GetSym(E->getOperand(1));
330 // Skip null functions. This can happen if functions are dead stripped after
331 // the CGProfile pass has been run.
332 if (!From || !To)
333 continue;
334 uint64_t Count = cast<ConstantAsMetadata>(E->getOperand(2))
335 ->getValue()
336 ->getUniqueInteger()
337 .getZExtValue();
338 Streamer.emitCGProfileEntry(
339 MCSymbolRefExpr::create(From, MCSymbolRefExpr::VK_None, C),
340 MCSymbolRefExpr::create(To, MCSymbolRefExpr::VK_None, C), Count);
341 }
342}

344MCSymbol *TargetLoweringObjectFileELF::getCFIPersonalitySymbol(
345 const GlobalValue *GV, const TargetMachine &TM,
346 MachineModuleInfo *MMI) const {
347 unsigned Encoding = getPersonalityEncoding();
348 if ((Encoding & 0x80) == DW_EH_PE_indirect)
349 return getContext().getOrCreateSymbol(StringRef("DW.ref.") +
350 TM.getSymbol(GV)->getName());
351 if ((Encoding & 0x70) == DW_EH_PE_absptr)
352 return TM.getSymbol(GV);
353 report_fatal_error("We do not support this DWARF encoding yet!");
354}

356void TargetLoweringObjectFileELF::emitPersonalityValue(
357 MCStreamer &Streamer, const DataLayout &DL, const MCSymbol *Sym) const {
358 SmallString<64> NameData("DW.ref.");
359 NameData += Sym->getName();
360 MCSymbolELF *Label =
361 cast<MCSymbolELF>(getContext().getOrCreateSymbol(NameData));
362 Streamer.EmitSymbolAttribute(Label, MCSA_Hidden);
363 Streamer.EmitSymbolAttribute(Label, MCSA_Weak);
364 unsigned Flags = ELF::SHF_ALLOC | ELF::SHF_WRITE | ELF::SHF_GROUP;
365 MCSection *Sec = getContext().getELFNamedSection(".data", Label->getName(),
366 ELF::SHT_PROGBITS, Flags, 0);
367 unsigned AS = DL.getProgramAddressSpace();
368 unsigned Size = DL.getPointerSize(AS);
369 Streamer.SwitchSection(Sec);
370 Streamer.EmitValueToAlignment(DL.getPointerABIAlignment(AS));
371 Streamer.EmitSymbolAttribute(Label, MCSA_ELF_TypeObject);
372 const MCExpr *E = MCConstantExpr::create(Size, getContext());
373 Streamer.emitELFSize(Label, E);
374 Streamer.EmitLabel(Label);
375
376 if (DL.isFatPointer(AS)) {
377 Streamer.EmitCheriCapability(Sym, 0, Size);
378 } else {
379 Streamer.EmitSymbolValue(Sym, Size);
380 }

383const MCExpr *TargetLoweringObjectFileELF::getTTypeGlobalReference(
384 const GlobalValue *GV, unsigned Encoding, const TargetMachine &TM,
385 MachineModuleInfo *MMI, MCStreamer &Streamer) const {
386 if (Encoding & DW_EH_PE_indirect) {
387 MachineModuleInfoELF &ELFMMI = MMI->getObjFileInfo<MachineModuleInfoELF>();
388
389 MCSymbol *SSym = getSymbolWithGlobalValueBase(GV, ".DW.stub", TM);
390
391 // Add information about the stub reference to ELFMMI so that the stub
392 // gets emitted by the asmprinter.
393 MachineModuleInfoImpl::StubValueTy &StubSym = ELFMMI.getGVStubEntry(SSym);
394 if (!StubSym.getPointer()) {
395 MCSymbol *Sym = TM.getSymbol(GV);
396 StubSym = MachineModuleInfoImpl::StubValueTy(Sym, !GV->hasLocalLinkage());
397 }
398
399 return TargetLoweringObjectFile::
400 getTTypeReference(MCSymbolRefExpr::create(SSym, getContext()),
401 Encoding & ~DW_EH_PE_indirect, Streamer);
402 }
403
404 return TargetLoweringObjectFile::getTTypeGlobalReference(GV, Encoding, TM,
405 MMI, Streamer);
406}

408static SectionKind getELFKindForNamedSection(StringRef Name, SectionKind K) {
409 // N.B.: The defaults used in here are not the same ones used in MC.
410 // We follow gcc, MC follows gas. For example, given ".section .eh_frame",
411 // both gas and MC will produce a section with no flags. Given
412 // section(".eh_frame") gcc will produce:
413 //
414 // .section .eh_frame,"a",@progbits
415
416 if (Name == getInstrProfSectionName(IPSK_covmap, Triple::ELF,
417 /*AddSegmentInfo=*/false))
418 return SectionKind::getMetadata();
419
420 if (Name.empty() || Name[0] != '.') return K;
421
422 // Default implementation based on some magic section names.
423 if (Name == ".bss" ||
424 Name.startswith(".bss.") ||
425 Name.startswith(".gnu.linkonce.b.") ||
426 Name.startswith(".llvm.linkonce.b.") ||
427 Name == ".sbss" ||
428 Name.startswith(".sbss.") ||
429 Name.startswith(".gnu.linkonce.sb.") ||
430 Name.startswith(".llvm.linkonce.sb."))
431 return SectionKind::getBSS();
432
433 if (Name == ".tdata" ||
434 Name.startswith(".tdata.") ||
435 Name.startswith(".gnu.linkonce.td.") ||
436 Name.startswith(".llvm.linkonce.td."))
437 return SectionKind::getThreadData();
438
439 if (Name == ".tbss" ||
440 Name.startswith(".tbss.") ||
441 Name.startswith(".gnu.linkonce.tb.") ||
442 Name.startswith(".llvm.linkonce.tb."))
443 return SectionKind::getThreadBSS();
444
445 return K;
446}

448static unsigned getELFSectionType(StringRef Name, SectionKind K) {
449 // Use SHT_NOTE for section whose name starts with ".note" to allow
450 // emitting ELF notes from C variable declaration.
451 // See https://gcc.gnu.org/bugzilla/show_bug.cgi?id=77609
452 if (Name.startswith(".note"))
453 return ELF::SHT_NOTE;
454
455 if (Name == ".init_array")
456 return ELF::SHT_INIT_ARRAY;
457
458 if (Name == ".fini_array")
459 return ELF::SHT_FINI_ARRAY;
460
461 if (Name == ".preinit_array")
462 return ELF::SHT_PREINIT_ARRAY;
463
464 if (K.isBSS() || K.isThreadBSS())
465 return ELF::SHT_NOBITS;
466
467 return ELF::SHT_PROGBITS;
468}

470static unsigned getELFSectionFlags(SectionKind K) {
471 unsigned Flags = 0;
472
473 if (!K.isMetadata())
474 Flags |= ELF::SHF_ALLOC;
475
476 if (K.isText())
477 Flags |= ELF::SHF_EXECINSTR;
478
479 if (K.isExecuteOnly())
480 Flags |= ELF::SHF_ARM_PURECODE;
481
482 if (K.isWriteable())
483 Flags |= ELF::SHF_WRITE;
484
485 if (K.isThreadLocal())
486 Flags |= ELF::SHF_TLS;
487
488 if (K.isMergeableCString() || K.isMergeableConst())
489 Flags |= ELF::SHF_MERGE;
490
491 if (K.isMergeableCString())
492 Flags |= ELF::SHF_STRINGS;
493
494 return Flags;
495}

509static const MCSymbolELF *getAssociatedSymbol(const GlobalObject *GO,
510 const TargetMachine &TM) {
511 MDNode *MD = GO->getMetadata(LLVMContext::MD_associated);
512 if (!MD)
513 return nullptr;
514
515 const MDOperand &Op = MD->getOperand(0);
516 if (!Op.get())
517 return nullptr;
518
519 auto *VM = dyn_cast<ValueAsMetadata>(Op);
520 if (!VM)
521 report_fatal_error("MD_associated operand is not ValueAsMetadata");
522
523 GlobalObject *OtherGO = dyn_cast<GlobalObject>(VM->getValue());
524 return OtherGO ? dyn_cast<MCSymbolELF>(TM.getSymbol(OtherGO)) : nullptr;
525}

527static unsigned getEntrySizeForKind(SectionKind Kind) {
528 if (Kind.isMergeable1ByteCString())
529 return 1;
530 else if (Kind.isMergeable2ByteCString())
531 return 2;
532 else if (Kind.isMergeable4ByteCString())
533 return 4;
534 else if (Kind.isMergeableConst4())
535 return 4;
536 else if (Kind.isMergeableConst8())
537 return 8;
538 else if (Kind.isMergeableConst16())
539 return 16;
540 else if (Kind.isMergeableConst32())
541 return 32;
542 else {
543 // We shouldn't have mergeable C strings or mergeable constants that we
544 // didn't handle above.
545 assert(!Kind.isMergeableCString() && "unknown string width");
546 assert(!Kind.isMergeableConst() && "unknown data width");
547 return 0;
548 }

551MCSection *TargetLoweringObjectFileELF::getExplicitSectionGlobal(
552 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
553 StringRef SectionName = GO->getSection();
554
555 // Check if '#pragma clang section' name is applicable.
556 // Note that pragma directive overrides -ffunction-section, -fdata-section
557 // and so section name is exactly as user specified and not uniqued.
558 const GlobalVariable *GV = dyn_cast<GlobalVariable>(GO);
559 if (GV && GV->hasImplicitSection()) {
560 auto Attrs = GV->getAttributes();
561 if (Attrs.hasAttribute("bss-section") && Kind.isBSS()) {
562 SectionName = Attrs.getAttribute("bss-section").getValueAsString();
563 } else if (Attrs.hasAttribute("rodata-section") && Kind.isReadOnly()) {
564 SectionName = Attrs.getAttribute("rodata-section").getValueAsString();
565 } else if (Attrs.hasAttribute("data-section") && Kind.isData()) {
566 SectionName = Attrs.getAttribute("data-section").getValueAsString();
567 }
568 }
569 const Function *F = dyn_cast<Function>(GO);
570 if (F && F->hasFnAttribute("implicit-section-name")) {
571 SectionName = F->getFnAttribute("implicit-section-name").getValueAsString();
572 }
573
574 // Infer section flags from the section name if we can.
575 Kind = getELFKindForNamedSection(SectionName, Kind);
576
577 StringRef Group = "";
578 unsigned Flags = getELFSectionFlags(Kind);
579 if (const Comdat *C = getELFComdat(GO)) {
580 Group = C->getName();
581 Flags |= ELF::SHF_GROUP;
582 }
583
584 // A section can have at most one associated section. Put each global with
585 // MD_associated in a unique section.
586 unsigned UniqueID = MCContext::GenericSectionID;
587 const MCSymbolELF *AssociatedSymbol = getAssociatedSymbol(GO, TM);
588 if (AssociatedSymbol) {
589 UniqueID = NextUniqueID++;
590 Flags |= ELF::SHF_LINK_ORDER;
591 }
592
593 MCSectionELF *Section = getContext().getELFSection(
594 SectionName, getELFSectionType(SectionName, Kind), Flags,
595 getEntrySizeForKind(Kind), Group, UniqueID, AssociatedSymbol);
596 // Make sure that we did not get some other section with incompatible sh_link.
597 // This should not be possible due to UniqueID code above.
598 assert(Section->getAssociatedSymbol() == AssociatedSymbol &&
599 "Associated symbol mismatch between sections");
600 return Section;
601}

603/// Return the section prefix name used by options FunctionsSections and
604/// DataSections.
605static StringRef getSectionPrefixForGlobal(SectionKind Kind) {
606 if (Kind.isText())
607 return ".text";
608 if (Kind.isReadOnly())
609 return ".rodata";
610 if (Kind.isBSS())
611 return ".bss";
612 if (Kind.isThreadData())
613 return ".tdata";
614 if (Kind.isThreadBSS())
615 return ".tbss";
616 if (Kind.isData())
617 return ".data";
618 assert(Kind.isReadOnlyWithRel() && "Unknown section kind");
619 return ".data.rel.ro";
620}

622static MCSectionELF *selectELFSectionForGlobal(
623 MCContext &Ctx, const GlobalObject *GO, SectionKind Kind, Mangler &Mang,
624 const TargetMachine &TM, bool EmitUniqueSection, unsigned Flags,
625 unsigned *NextUniqueID, const MCSymbolELF *AssociatedSymbol) {
626
627 StringRef Group = "";
628 if (const Comdat *C = getELFComdat(GO)) {
629 Flags |= ELF::SHF_GROUP;
630 Group = C->getName();
631 }
632
633 // Get the section entry size based on the kind.
634 unsigned EntrySize = getEntrySizeForKind(Kind);
635
636 SmallString<128> Name;
637 if (Kind.isMergeableCString()) {
638 // We also need alignment here.
639 // FIXME: this is getting the alignment of the character, not the
640 // alignment of the global!
641 unsigned Align = GO->getParent()->getDataLayout().getPreferredAlignment(
642 cast<GlobalVariable>(GO));
643
644 std::string SizeSpec = ".rodata.str" + utostr(EntrySize) + ".";
645 Name = SizeSpec + utostr(Align);
646 } else if (Kind.isMergeableConst()) {
647 Name = ".rodata.cst";
648 Name += utostr(EntrySize);
649 } else {
650 Name = getSectionPrefixForGlobal(Kind);
651 }
652
653 if (const auto *F = dyn_cast<Function>(GO)) {
654 const auto &OptionalPrefix = F->getSectionPrefix();
655 if (OptionalPrefix)
656 Name += *OptionalPrefix;
657 }
658
659 unsigned UniqueID = MCContext::GenericSectionID;
660 if (EmitUniqueSection) {
661 if (TM.getUniqueSectionNames()) {
662 Name.push_back('.');
663 TM.getNameWithPrefix(Name, GO, Mang, true /*MayAlwaysUsePrivate*/);
664 } else {
665 UniqueID = *NextUniqueID;
666 (*NextUniqueID)++;
667 }
668 }
669 // Use 0 as the unique ID for execute-only text.
670 if (Kind.isExecuteOnly())
671 UniqueID = 0;
672 return Ctx.getELFSection(Name, getELFSectionType(Name, Kind), Flags,
673 EntrySize, Group, UniqueID, AssociatedSymbol);
674}
675
676MCSection *TargetLoweringObjectFileELF::SelectSectionForGlobal(
677 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
678 unsigned Flags = getELFSectionFlags(Kind);
679
680 // If we have -ffunction-section or -fdata-section then we should emit the
681 // global value to a uniqued section specifically for it.
682 bool EmitUniqueSection = false;
683 if (!(Flags & ELF::SHF_MERGE) && !Kind.isCommon()) {
684 if (Kind.isText())
685 EmitUniqueSection = TM.getFunctionSections();
686 else
687 EmitUniqueSection = TM.getDataSections();
688 }
689 EmitUniqueSection |= GO->hasComdat();
690
691 const MCSymbolELF *AssociatedSymbol = getAssociatedSymbol(GO, TM);
692 if (AssociatedSymbol) {
693 EmitUniqueSection = true;
694 Flags |= ELF::SHF_LINK_ORDER;
695 }
696
697 MCSectionELF *Section = selectELFSectionForGlobal(
698 getContext(), GO, Kind, getMangler(), TM, EmitUniqueSection, Flags,
699 &NextUniqueID, AssociatedSymbol);
700 assert(Section->getAssociatedSymbol() == AssociatedSymbol);
701 return Section;
702}

746static MCSectionELF *getStaticStructorSection(MCContext &Ctx, bool UseInitArray,
747 bool IsCtor, unsigned Priority,
748 const MCSymbol *KeySym) {
749 std::string Name;
750 unsigned Type;
751 unsigned Flags = ELF::SHF_ALLOC | ELF::SHF_WRITE;
752 StringRef COMDAT = KeySym ? KeySym->getName() : "";
753
754 if (KeySym)
755 Flags |= ELF::SHF_GROUP;
756
757 if (UseInitArray) {
758 if (IsCtor) {
759 Type = ELF::SHT_INIT_ARRAY;
760 Name = ".init_array";

777 raw_string_ostream(Name) << format(".%05u", 65535 - Priority);
778 Type = ELF::SHT_PROGBITS;
779 }
780
781 return Ctx.getELFSection(Name, Type, Flags, 0, COMDAT);
782}

877 Module &M) const {
878 // Emit the linker options if present.
879 if (auto *LinkerOptions = M.getNamedMetadata("llvm.linker.options")) {
880 for (const auto &Option : LinkerOptions->operands()) {
881 SmallVector<std::string, 4> StrOptions;

891

929MCSection *TargetLoweringObjectFileMachO::getExplicitSectionGlobal(
930 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
931 // Parse the section specifier and create it if valid.
932 StringRef Segment, Section;
933 unsigned TAA = 0, StubSize = 0;
934 bool TAAParsed;
935
936 checkMachOComdat(GO);
937
938 std::string ErrorCode =
939 MCSectionMachO::ParseSectionSpecifier(GO->getSection(), Segment, Section,
940 TAA, TAAParsed, StubSize);
941 if (!ErrorCode.empty()) {
942 // If invalid, report the error with report_fatal_error.
943 report_fatal_error("Global variable '" + GO->getName() +
944 "' has an invalid section specifier '" +
945 GO->getSection() + "': " + ErrorCode + ".");
946 }
947
948 // Get the section.
949 MCSectionMachO *S =
950 getContext().getMachOSection(Segment, Section, TAA, StubSize, Kind);
951
952 // If TAA wasn't set by ParseSectionSpecifier() above,
953 // use the value returned by getMachOSection() as a default.
954 if (!TAAParsed)
955 TAA = S->getTypeAndAttributes();
956
957 // Okay, now that we got the section, verify that the TAA & StubSize agree.
958 // If the user declared multiple globals with different section flags, we need
959 // to reject it here.
960 if (S->getTypeAndAttributes() != TAA || S->getStubSize() != StubSize) {
961 // If invalid, report the error with report_fatal_error.
962 report_fatal_error("Global variable '" + GO->getName() +
963 "' section type or attributes does not match previous"
964 " section specifier");
965 }
966
967 return S;
968}
969
970MCSection *TargetLoweringObjectFileMachO::SelectSectionForGlobal(
971 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
972 checkMachOComdat(GO);
973
974 // Handle thread local data.
975 if (Kind.isThreadBSS()) return TLSBSSSection;
976 if (Kind.isThreadData()) return TLSDataSection;
977
978 if (Kind.isText())
979 return GO->isWeakForLinker() ? TextCoalSection : TextSection;
980
981 // If this is weak/linkonce, put this in a coalescable section, either in text
982 // or data depending on if it is writable.
983 if (GO->isWeakForLinker()) {
984 if (Kind.isReadOnly())
985 return ConstTextCoalSection;
986 if (Kind.isReadOnlyWithRel())
987 return ConstDataCoalSection;
988 return DataCoalSection;
989 }
990
991 // FIXME: Alignment check should be handled by section classifier.
992 if (Kind.isMergeable1ByteCString() &&
993 GO->getParent()->getDataLayout().getPreferredAlignment(
994 cast<GlobalVariable>(GO)) < 32)
995 return CStringSection;
996
997 // Do not put 16-bit arrays in the UString section if they have an
998 // externally visible label, this runs into issues with certain linker
999 // versions.
1000 if (Kind.isMergeable2ByteCString() && !GO->hasExternalLinkage() &&
1001 GO->getParent()->getDataLayout().getPreferredAlignment(
1002 cast<GlobalVariable>(GO)) < 32)
1003 return UStringSection;
1004
1005 // With MachO only variables whose corresponding symbol starts with 'l' or
1006 // 'L' can be merged, so we only try merging GVs with private linkage.
1007 if (GO->hasPrivateLinkage() && Kind.isMergeableConst()) {
1008 if (Kind.isMergeableConst4())
1009 return FourByteConstantSection;
1010 if (Kind.isMergeableConst8())
1011 return EightByteConstantSection;
1012 if (Kind.isMergeableConst16())
1013 return SixteenByteConstantSection;
1014 }
1015
1016 // Otherwise, if it is readonly, but not something we can specially optimize,
1017 // just drop it in .const.
1018 if (Kind.isReadOnly())
1019 return ReadOnlySection;
1020
1021 // If this is marked const, put it into a const section. But if the dynamic
1022 // linker needs to write to it, put it in the data segment.
1023 if (Kind.isReadOnlyWithRel())
1024 return ConstDataSection;
1025
1026 // Put zero initialized globals with strong external linkage in the
1027 // DATA, __common section with the .zerofill directive.
1028 if (Kind.isBSSExtern())
1029 return DataCommonSection;
1030
1031 // Put zero initialized globals with local linkage in __DATA,__bss directive
1032 // with the .zerofill directive (aka .lcomm).
1033 if (Kind.isBSSLocal())
1034 return DataBSSSection;
1035
1036 // Otherwise, just drop the variable in the normal data section.
1037 return DataSection;
1038}

1104
1105const MCExpr *TargetLoweringObjectFileMachO::getIndirectSymViaGOTPCRel(
1106 const MCSymbol *Sym, const MCValue &MV, int64_t Offset,
1107 MachineModuleInfo *MMI, MCStreamer &Streamer) const {
1108 // Although MachO 32-bit targets do not explicitly have a GOTPCREL relocation
1109 // as 64-bit do, we replace the GOT equivalent by accessing the final symbol
1110 // through a non_lazy_ptr stub instead. One advantage is that it allows the
1111 // computation of deltas to final external symbols. Example:
1112 //
1113 // _extgotequiv:
1114 // .long _extfoo
1115 //
1116 // _delta:
1117 // .long _extgotequiv-_delta
1118 //
1119 // is transformed to:
1120 //
1121 // _delta:
1122 // .long L_extfoo$non_lazy_ptr-(_delta+0)
1123 //
1124 // .section __IMPORT,__pointers,non_lazy_symbol_pointers
1125 // L_extfoo$non_lazy_ptr:
1126 // .indirect_symbol _extfoo
1127 // .long 0
1128 //
1129 // The indirect symbol table (and sections of non_lazy_symbol_pointers type)
1130 // may point to both local (same translation unit) and global (other
1131 // translation units) symbols. Example:
1132 //
1133 // .section __DATA,__pointers,non_lazy_symbol_pointers
1134 // L1:
1135 // .indirect_symbol _myGlobal
1136 // .long 0
1137 // L2:
1138 // .indirect_symbol _myLocal
1139 // .long _myLocal
1140 //
1141 // If the symbol is local, instead of the symbol's index, the assembler
1142 // places the constant INDIRECT_SYMBOL_LOCAL into the indirect symbol table.
1143 // Then the linker will notice the constant in the table and will look at the
1144 // content of the symbol.
1145 MachineModuleInfoMachO &MachOMMI =
1146 MMI->getObjFileInfo<MachineModuleInfoMachO>();
1147 MCContext &Ctx = getContext();
1148
1149 // The offset must consider the original displacement from the base symbol
1150 // since 32-bit targets don't have a GOTPCREL to fold the PC displacement.
1151 Offset = -MV.getConstant();
1152 const MCSymbol *BaseSym = &MV.getSymB()->getSymbol();
1153
1154 // Access the final symbol via sym$non_lazy_ptr and generate the appropriated
1155 // non_lazy_ptr stubs.
1156 SmallString<128> Name;
1157 StringRef Suffix = "$non_lazy_ptr";
1158 Name += MMI->getModule()->getDataLayout().getPrivateGlobalPrefix();
1159 Name += Sym->getName();
1160 Name += Suffix;
1161 MCSymbol *Stub = Ctx.getOrCreateSymbol(Name);
1162
1163 MachineModuleInfoImpl::StubValueTy &StubSym = MachOMMI.getGVStubEntry(Stub);
1164 if (!StubSym.getPointer()) {
1165 bool IsIndirectLocal = Sym->isDefined() && !Sym->isExternal();
1166 // With the assumption that IsIndirectLocal == GV->hasLocalLinkage().
1167 StubSym = MachineModuleInfoImpl::StubValueTy(const_cast<MCSymbol *>(Sym),
1168 !IsIndirectLocal);
1169 }
1170
1171 const MCExpr *BSymExpr =
1172 MCSymbolRefExpr::create(BaseSym, MCSymbolRefExpr::VK_None, Ctx);
1173 const MCExpr *LHS =
1174 MCSymbolRefExpr::create(Stub, MCSymbolRefExpr::VK_None, Ctx);
1175
1176 if (!Offset)
1177 return MCBinaryExpr::createSub(LHS, BSymExpr, Ctx);
1178
1179 const MCExpr *RHS =
1180 MCBinaryExpr::createAdd(BSymExpr, MCConstantExpr::create(Offset, Ctx), Ctx);
1181 return MCBinaryExpr::createSub(LHS, RHS, Ctx);
1182}

1217 bool isThumb = TM.getTargetTriple().getArch() == Triple::thumb;
1218
1219 if (K.isMetadata())
1220 Flags |=
1221 COFF::IMAGE_SCN_MEM_DISCARDABLE;

1223 Flags |=
1224 COFF::IMAGE_SCN_MEM_EXECUTE |
1225 COFF::IMAGE_SCN_MEM_READ |

1227 (isThumb ? COFF::IMAGE_SCN_MEM_16BIT : (COFF::SectionCharacteristics)0);

1229 Flags |=
1230 COFF::IMAGE_SCN_CNT_UNINITIALIZED_DATA |
1231 COFF::IMAGE_SCN_MEM_READ |
1232 COFF::IMAGE_SCN_MEM_WRITE;
1233 else if (K.isThreadLocal())

1332MCSection *TargetLoweringObjectFileCOFF::SelectSectionForGlobal(
1333 const GlobalObject *GO, SectionKind Kind, const TargetMachine &TM) const {
1334 // If we have -ffunction-sections then we should emit the global value to a
1335 // uniqued section specifically for it.
1336 bool EmitUniquedSection;
1337 if (Kind.isText())
1338 EmitUniquedSection = TM.getFunctionSections();
1339 else
1340 EmitUniquedSection = TM.getDataSections();
1341
1342 if ((EmitUniquedSection && !Kind.isCommon()) || GO->hasComdat()) {
1343 SmallString<256> Name = getCOFFSectionNameForUniqueGlobal(Kind);
1344
1345 unsigned Characteristics = getCOFFSectionFlags(Kind, TM);
1346
1347 Characteristics |= COFF::IMAGE_SCN_LNK_COMDAT;
1348 int Selection = getSelectionForCOFF(GO);
1349 if (!Selection)
1350 Selection = COFF::IMAGE_COMDAT_SELECT_NODUPLICATES;
1351 const GlobalValue *ComdatGV;
1352 if (GO->hasComdat())
1353 ComdatGV = getComdatGVForCOFF(GO);
1354 else
1355 ComdatGV = GO;
1356
1357 unsigned UniqueID = MCContext::GenericSectionID;
1358 if (EmitUniquedSection)
1359 UniqueID = NextUniqueID++;
1360
1361 if (!ComdatGV->hasPrivateLinkage()) {
1362 MCSymbol *Sym = TM.getSymbol(ComdatGV);
1363 StringRef COMDATSymName = Sym->getName();
1364
1365 // Append "$symbol" to the section name *before* IR-level mangling is
1366 // applied when targetting mingw. This is what GCC does, and the ld.bfd
1367 // COFF linker will not properly handle comdats otherwise.
1368 if (getTargetTriple().isWindowsGNUEnvironment())
1369 raw_svector_ostream(Name) << '$' << ComdatGV->getName();
1370
1371 return getContext().getCOFFSection(Name, Characteristics, Kind,
1372 COMDATSymName, Selection, UniqueID);
1373 } else {
1374 SmallString<256> TmpData;
1375 getMangler().getNameWithPrefix(TmpData, GO, /*CannotUsePrivateLabel=*/true);
1376 return getContext().getCOFFSection(Name, Characteristics, Kind, TmpData,
1377 Selection, UniqueID);
1378 }
1379 }
1380
1381 if (Kind.isText())
1382 return TextSection;
1383
1384 if (Kind.isThreadLocal())
1385 return TLSDataSection;
1386
1387 if (Kind.isReadOnly() || Kind.isReadOnlyWithRel())
1388 return ReadOnlySection;
1389
1390 // Note: we claim that common symbols are put in BSSSection, but they are
1391 // really emitted with the magic .comm directive, which creates a symbol table
1392 // entry but not a section.
1393 if (Kind.isBSS() || Kind.isCommon())
1394 return BSSSection;
1395
1396 return DataSection;
1397}

1439 Module &M) const {
1440 if (NamedMDNode *LinkerOptions = M.getNamedMetadata("llvm.linker.options")) {

1446 for (const auto &Option : LinkerOptions->operands()) {
1447 for (const auto &Piece : cast<MDNode>(Option)->operands()) {

1449 std::string Directive(" ");
1450 Directive.append(cast<MDString>(Piece)->getString());
1451 Streamer.EmitBytes(Directive);
1452 }
1453 }
1454 }
1455
1456 unsigned Version = 0;
1457 unsigned Flags = 0;
1458 StringRef Section;
1459
1460 GetObjCImageInfo(M, Version, Flags, Section);
1461 if (Section.empty())
1462 return;

1464 auto &C = getContext();

1466 Section, COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ,

1469 Streamer.EmitLabel(C.getOrCreateSymbol(StringRef("OBJC_IMAGE_INFO")));

1500static MCSectionCOFF *getCOFFStaticStructorSection(MCContext &Ctx,
1501 const Triple &T, bool IsCtor,
1502 unsigned Priority,
1503 const MCSymbol *KeySym,
1504 MCSectionCOFF *Default) {
1505 if (T.isKnownWindowsMSVCEnvironment() || T.isWindowsItaniumEnvironment()) {
1506 // If the priority is the default, use .CRT$XCU, possibly associative.
1507 if (Priority == 65535)
1508 return Ctx.getAssociativeCOFFSection(Default, KeySym, 0);
1509
1510 // Otherwise, we need to compute a new section name. Low priorities should
1511 // run earlier. The linker will sort sections ASCII-betically, and we need a
1512 // string that sorts between .CRT$XCA and .CRT$XCU. In the general case, we
1513 // make a name like ".CRT$XCT12345", since that runs before .CRT$XCU. Really
1514 // low priorities need to sort before 'L', since the CRT uses that
1515 // internally, so we use ".CRT$XCA00001" for them.
1516 SmallString<24> Name;
1517 raw_svector_ostream OS(Name);
1518 OS << ".CRT$XC" << (Priority < 200 ? 'A' : 'T') << format("%05u", Priority);
1519 MCSectionCOFF *Sec = Ctx.getCOFFSection(
1520 Name, COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ,
1521 SectionKind::getReadOnly());
1522 return Ctx.getAssociativeCOFFSection(Sec, KeySym, 0);
1523 }
1524
1525 std::string Name = IsCtor ? ".ctors" : ".dtors";
1526 if (Priority != 65535)
1527 raw_string_ostream(Name) << format(".%05u", 65535 - Priority);
1528
1529 return Ctx.getAssociativeCOFFSection(
1530 Ctx.getCOFFSection(Name, COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
1531 COFF::IMAGE_SCN_MEM_READ |
1532 COFF::IMAGE_SCN_MEM_WRITE,
1533 SectionKind::getData()),
1534 KeySym, 0);
1535}

1561const MCExpr *TargetLoweringObjectFileCOFF::lowerRelativeReference(
1562 const GlobalValue *LHS, const GlobalValue *RHS,
1563 const TargetMachine &TM) const {
1564 const Triple &T = TM.getTargetTriple();
1565 if (!T.isKnownWindowsMSVCEnvironment() &&
1566 !T.isWindowsItaniumEnvironment() &&
1567 !T.isWindowsCoreCLREnvironment())
1568 return nullptr;
1569
1570 // Our symbols should exist in address space zero, cowardly no-op if
1571 // otherwise.
1572 if (LHS->getType()->getPointerAddressSpace() != 0 ||
1573 RHS->getType()->getPointerAddressSpace() != 0)
1574 return nullptr;
1575
1576 // Both ptrtoint instructions must wrap global objects:
1577 // - Only global variables are eligible for image relative relocations.
1578 // - The subtrahend refers to the special symbol __ImageBase, a GlobalVariable.
1579 // We expect __ImageBase to be a global variable without a section, externally
1580 // defined.
1581 //
1582 // It should look something like this: @__ImageBase = external constant i8
1583 if (!isa<GlobalObject>(LHS) || !isa<GlobalVariable>(RHS) ||
1584 LHS->isThreadLocal() || RHS->isThreadLocal() ||
1585 RHS->getName() != "__ImageBase" || !RHS->hasExternalLinkage() ||
1586 cast<GlobalVariable>(RHS)->hasInitializer() || RHS->hasSection())
1587 return nullptr;
1588
1589 return MCSymbolRefExpr::create(TM.getSymbol(LHS),
1590 MCSymbolRefExpr::VK_COFF_IMGREL32,
1591 getContext());

1604static std::string scalarConstantToHexString(const Constant *C) {
1605 Type *Ty = C->getType();
1606 if (isa<UndefValue>(C)) {
1607 return APIntToHexString(APInt::getNullValue(Ty->getPrimitiveSizeInBits()));
1608 } else if (const auto *CFP = dyn_cast<ConstantFP>(C)) {
1609 return APIntToHexString(CFP->getValueAPF().bitcastToAPInt());
1610 } else if (const auto *CI = dyn_cast<ConstantInt>(C)) {
1611 return APIntToHexString(CI->getValue());
1612 } else {
1613 unsigned NumElements;
1614 if (isa<VectorType>(Ty))
1615 NumElements = Ty->getVectorNumElements();
1616 else
1617 NumElements = Ty->getArrayNumElements();
1618 std::string HexString;
1619 for (int I = NumElements - 1, E = -1; I != E; --I)
1620 HexString += scalarConstantToHexString(C->getAggregateElement(I));
