38/// EmitSLEB128 - emit the specified signed leb128 value.
39void AsmPrinter::EmitSLEB128(int64_t Value, const char *Desc) const {
40 if (isVerbose() && Desc)
41 OutStreamer->AddComment(Desc);
42
43 OutStreamer->EmitSLEB128IntValue(Value);
44}
45
46void AsmPrinter::EmitULEB128(uint64_t Value, const char *Desc, unsigned PadTo) const {
47 if (isVerbose() && Desc)
48 OutStreamer->AddComment(Desc);
49
50 OutStreamer->EmitULEB128IntValue(Value, PadTo);
51}

53/// Emit something like ".uleb128 Hi-Lo".
54void AsmPrinter::EmitLabelDifferenceAsULEB128(const MCSymbol *Hi,
55 const MCSymbol *Lo) const {
56 OutStreamer->emitAbsoluteSymbolDiffAsULEB128(Hi, Lo);
57}

59static const char *DecodeDWARFEncoding(unsigned Encoding) {
60 switch (Encoding) {
61 case dwarf::DW_EH_PE_absptr:
62 return "absptr";
63 case dwarf::DW_EH_PE_omit:
64 return "omit";
65 case dwarf::DW_EH_PE_pcrel:
66 return "pcrel";
67 case dwarf::DW_EH_PE_uleb128:
68 return "uleb128";
69 case dwarf::DW_EH_PE_sleb128:
70 return "sleb128";
71 case dwarf::DW_EH_PE_udata4:
72 return "udata4";
73 case dwarf::DW_EH_PE_udata8:
74 return "udata8";
75 case dwarf::DW_EH_PE_sdata4:
76 return "sdata4";
77 case dwarf::DW_EH_PE_sdata8:
78 return "sdata8";
79 case dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_udata4:
80 return "pcrel udata4";
81 case dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4:
82 return "pcrel sdata4";
83 case dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_udata8:
84 return "pcrel udata8";
85 case dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata8:
86 return "pcrel sdata8";
87 case dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_udata4
88 :
89 return "indirect pcrel udata4";
90 case dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4
91 :
92 return "indirect pcrel sdata4";
93 case dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_udata8
94 :
95 return "indirect pcrel udata8";
96 case dwarf::DW_EH_PE_indirect | dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata8
97 :
98 return "indirect pcrel sdata8";
99 }
100
101 return "<unknown encoding>";
102}

104/// EmitEncodingByte - Emit a .byte 42 directive that corresponds to an
105/// encoding. If verbose assembly output is enabled, we output comments
106/// describing the encoding. Desc is an optional string saying what the
107/// encoding is specifying (e.g. "LSDA").
108void AsmPrinter::EmitEncodingByte(unsigned Val, const char *Desc) const {
109 if (isVerbose()) {
110 if (Desc)
111 OutStreamer->AddComment(Twine(Desc) + " Encoding = " +
112 Twine(DecodeDWARFEncoding(Val)));
113 else
114 OutStreamer->AddComment(Twine("Encoding = ") + DecodeDWARFEncoding(Val));
115 }
116
117 OutStreamer->EmitIntValue(Val, 1);
118}

120/// GetSizeOfEncodedValue - Return the size of the encoding in bytes.
121unsigned AsmPrinter::GetSizeOfEncodedValue(unsigned Encoding) const {
122 if (Encoding == dwarf::DW_EH_PE_omit)
123 return 0;
124
125 switch (Encoding & 0x07) {
126 default:
127 llvm_unreachable("Invalid encoded value.");
128 case dwarf::DW_EH_PE_absptr:
129 assert(!MF->getDataLayout().isFatPointer(0u) &&
130 "Should not use GetSizeOfEncodedValue when emitting capabilities");
131 return MF->getDataLayout().getPointerSize();
132 case dwarf::DW_EH_PE_udata2:
133 return 2;
134 case dwarf::DW_EH_PE_udata4:
135 return 4;
136 case dwarf::DW_EH_PE_udata8:
137 return 8;
138 }
139}

141void AsmPrinter::EmitTTypeReference(const GlobalValue *GV,
142 unsigned Encoding) const {
143 if (GV) {
144 const TargetLoweringObjectFile &TLOF = getObjFileLowering();
145
146 const MCExpr *Exp =
147 TLOF.getTTypeGlobalReference(GV, Encoding, TM, MMI, *OutStreamer);
148 OutStreamer->EmitValue(Exp, GetSizeOfEncodedValue(Encoding));
149 } else
150 OutStreamer->EmitIntValue(0, GetSizeOfEncodedValue(Encoding));
151}

153void AsmPrinter::emitDwarfSymbolReference(const MCSymbol *Label,
154 bool ForceOffset) const {
155 if (!ForceOffset) {
156 // On COFF targets, we have to emit the special .secrel32 directive.
157 if (MAI->needsDwarfSectionOffsetDirective()) {
158 OutStreamer->EmitCOFFSecRel32(Label, /*Offset=*/0);
159 return;
160 }
161
162 // If the format uses relocations with dwarf, refer to the symbol directly.
163 if (MAI->doesDwarfUseRelocationsAcrossSections()) {
164 OutStreamer->EmitSymbolValue(Label, 4);
165 return;
166 }
167 }
168
169 // Otherwise, emit it as a label difference from the start of the section.
170 EmitLabelDifference(Label, Label->getSection().getBeginSymbol(), 4);
171}
172
173void AsmPrinter::emitDwarfStringOffset(DwarfStringPoolEntry S) const {
174 if (MAI->doesDwarfUseRelocationsAcrossSections()) {
175 assert(S.Symbol && "No symbol available");
176 emitDwarfSymbolReference(S.Symbol);
177 return;
178 }
179
180 // Just emit the offset directly; no need for symbol math.
181 emitInt32(S.Offset);
182}
183
184void AsmPrinter::EmitDwarfOffset(const MCSymbol *Label, uint64_t Offset) const {
185 EmitLabelPlusOffset(Label, Offset, MAI->getCodePointerSize());
186}

192void AsmPrinter::emitCFIInstruction(const MCCFIInstruction &Inst) const {
193 switch (Inst.getOperation()) {
194 default:
195 llvm_unreachable("Unexpected instruction");
196 case MCCFIInstruction::OpDefCfaOffset:
197 OutStreamer->EmitCFIDefCfaOffset(Inst.getOffset());
198 break;
199 case MCCFIInstruction::OpAdjustCfaOffset:
200 OutStreamer->EmitCFIAdjustCfaOffset(Inst.getOffset());
201 break;
202 case MCCFIInstruction::OpDefCfa:
203 OutStreamer->EmitCFIDefCfa(Inst.getRegister(), Inst.getOffset());
204 break;
205 case MCCFIInstruction::OpDefCfaRegister:
206 OutStreamer->EmitCFIDefCfaRegister(Inst.getRegister());
207 break;
208 case MCCFIInstruction::OpOffset:
209 OutStreamer->EmitCFIOffset(Inst.getRegister(), Inst.getOffset());
210 break;
211 case MCCFIInstruction::OpRegister:
212 OutStreamer->EmitCFIRegister(Inst.getRegister(), Inst.getRegister2());
213 break;
214 case MCCFIInstruction::OpWindowSave:
215 OutStreamer->EmitCFIWindowSave();
216 break;
217 case MCCFIInstruction::OpNegateRAState:
218 OutStreamer->EmitCFINegateRAState();
219 break;
220 case MCCFIInstruction::OpSameValue:
221 OutStreamer->EmitCFISameValue(Inst.getRegister());
222 break;
223 case MCCFIInstruction::OpGnuArgsSize:
224 OutStreamer->EmitCFIGnuArgsSize(Inst.getOffset());
225 break;
226 case MCCFIInstruction::OpEscape:
227 OutStreamer->EmitCFIEscape(Inst.getValues());
228 break;
229 case MCCFIInstruction::OpRestore:
230 OutStreamer->EmitCFIRestore(Inst.getRegister());
231 break;
232 }
233}

235void AsmPrinter::emitDwarfDIE(const DIE &Die) const {
236 // Emit the code (index) for the abbreviation.
237 if (isVerbose())
238 OutStreamer->AddComment("Abbrev [" + Twine(Die.getAbbrevNumber()) + "] 0x" +
239 Twine::utohexstr(Die.getOffset()) + ":0x" +
240 Twine::utohexstr(Die.getSize()) + " " +
241 dwarf::TagString(Die.getTag()));
242 EmitULEB128(Die.getAbbrevNumber());
243
244 // Emit the DIE attribute values.
245 for (const auto &V : Die.values()) {
246 dwarf::Attribute Attr = V.getAttribute();
247 assert(V.getForm() && "Too many attributes for DIE (check abbreviation)");
248
249 if (isVerbose()) {
250 OutStreamer->AddComment(dwarf::AttributeString(Attr));
251 if (Attr == dwarf::DW_AT_accessibility)
252 OutStreamer->AddComment(
253 dwarf::AccessibilityString(V.getDIEInteger().getValue()));
254 }
255
256 // Emit an attribute using the defined form.
257 V.EmitValue(this);
258 }
259
260 // Emit the DIE children if any.
261 if (Die.hasChildren()) {
262 for (auto &Child : Die.children())
263 emitDwarfDIE(Child);
264
265 OutStreamer->AddComment("End Of Children Mark");
266 emitInt8(0);
267 }
268}
269
270void AsmPrinter::emitDwarfAbbrev(const DIEAbbrev &Abbrev) const {
271 // Emit the abbreviations code (base 1 index.)
272 EmitULEB128(Abbrev.getNumber(), "Abbreviation Code");
273
274 // Emit the abbreviations data.
275 Abbrev.Emit(this);
