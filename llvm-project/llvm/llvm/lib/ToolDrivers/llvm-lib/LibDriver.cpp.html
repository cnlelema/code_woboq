61static std::string getOutputPath(opt::InputArgList *Args,
62 const NewArchiveMember &FirstMember) {
63 if (auto *Arg = Args->getLastArg(OPT_out))
64 return Arg->getValue();
65 SmallString<128> Val = StringRef(FirstMember.Buf->getBufferIdentifier());
66 sys::path::replace_extension(Val, ".lib");
67 return Val.str();
68}

70static std::vector<StringRef> getSearchPaths(opt::InputArgList *Args,
71 StringSaver &Saver) {
72 std::vector<StringRef> Ret;
73 // Add current directory as first item of the search path.
74 Ret.push_back("");
75
76 // Add /libpath flags.
77 for (auto *Arg : Args->filtered(OPT_libpath))
78 Ret.push_back(Arg->getValue());
79
80 // Add $LIB.
81 Optional<std::string> EnvOpt = sys::Process::GetEnv("LIB");
82 if (!EnvOpt.hasValue())
83 return Ret;
84 StringRef Env = Saver.save(*EnvOpt);
85 while (!Env.empty()) {
86 StringRef Path;
87 std::tie(Path, Env) = Env.split(';');
88 Ret.push_back(Path);
89 }
90 return Ret;
91}

93static std::string findInputFile(StringRef File, ArrayRef<StringRef> Paths) {
94 for (StringRef Dir : Paths) {
95 SmallString<128> Path = Dir;
96 sys::path::append(Path, File);
97 if (sys::fs::exists(Path))
98 return Path.str().str();
99 }
100 return "";
101}

103static void fatalOpenError(llvm::Error E, Twine File) {
104 if (!E)
105 return;
106 handleAllErrors(std::move(E), [&](const llvm::ErrorInfoBase &EIB) {
107 llvm::errs() << "error opening '" << File << "': " << EIB.message() << '\n';
108 exit(1);
109 });
110}

112static void doList(opt::InputArgList& Args) {
113 // lib.exe prints the contents of the first archive file.
114 std::unique_ptr<MemoryBuffer> B;
115 for (auto *Arg : Args.filtered(OPT_INPUT)) {
116 // Create or open the archive object.
117 ErrorOr<std::unique_ptr<MemoryBuffer>> MaybeBuf =
118 MemoryBuffer::getFile(Arg->getValue(), -1, false);
119 fatalOpenError(errorCodeToError(MaybeBuf.getError()), Arg->getValue());
120
121 if (identify_magic(MaybeBuf.get()->getBuffer()) == file_magic::archive) {
122 B = std::move(MaybeBuf.get());
123 break;
124 }
125 }
126
127 // lib.exe doesn't print an error if no .lib files are passed.
128 if (!B)
129 return;
130
131 Error Err = Error::success();
132 object::Archive Archive(B.get()->getMemBufferRef(), Err);
133 fatalOpenError(std::move(Err), B->getBufferIdentifier());
134
135 for (auto &C : Archive.children(Err)) {
136 Expected<StringRef> NameOrErr = C.getName();
137 fatalOpenError(NameOrErr.takeError(), B->getBufferIdentifier());
138 StringRef Name = NameOrErr.get();
139 llvm::outs() << Name << '\n';
140 }
141 fatalOpenError(std::move(Err), B->getBufferIdentifier());
142}

144int llvm::libDriverMain(ArrayRef<const char *> ArgsArr) {
145 BumpPtrAllocator Alloc;
146 StringSaver Saver(Alloc);
147
148 // Parse command line arguments.
149 SmallVector<const char *, 20> NewArgs(ArgsArr.begin(), ArgsArr.end());
150 cl::ExpandResponseFiles(Saver, cl::TokenizeWindowsCommandLine, NewArgs);
151 ArgsArr = NewArgs;
152
153 LibOptTable Table;
154 unsigned MissingIndex;
155 unsigned MissingCount;
156 opt::InputArgList Args =
157 Table.ParseArgs(ArgsArr.slice(1), MissingIndex, MissingCount);
158 if (MissingCount) {
159 llvm::errs() << "missing arg value for \""
160 << Args.getArgString(MissingIndex) << "\", expected "
161 << MissingCount
162 << (MissingCount == 1 ? " argument.\n" : " arguments.\n");
163 return 1;
164 }
165 for (auto *Arg : Args.filtered(OPT_UNKNOWN))
166 llvm::errs() << "ignoring unknown argument: " << Arg->getSpelling() << "\n";
167
168 // Handle /help
169 if (Args.hasArg(OPT_help)) {
170 Table.PrintHelp(outs(), "llvm-lib [options] file...", "LLVM Lib");
171 return 0;
172 }
173
174 // If no input files, silently do nothing to match lib.exe.
175 if (!Args.hasArgNoClaim(OPT_INPUT))
176 return 0;
177
178 if (Args.hasArg(OPT_lst)) {
179 doList(Args);
180 return 0;
181 }
182
183 std::vector<StringRef> SearchPaths = getSearchPaths(&Args, Saver);
184
185 COFF::MachineTypes LibMachine = COFF::IMAGE_FILE_MACHINE_UNKNOWN;
186 std::string LibMachineSource;
187 if (auto *Arg = Args.getLastArg(OPT_machine)) {
188 LibMachine = getMachineType(Arg->getValue());
189 if (LibMachine == COFF::IMAGE_FILE_MACHINE_UNKNOWN) {
190 llvm::errs() << "unknown /machine: arg " << Arg->getValue() << '\n';
191 return 1;
192 }
193 LibMachineSource =
194 std::string(" (from '/machine:") + Arg->getValue() + "' flag)";
195 }
196
197 // Create a NewArchiveMember for each input file.
198 std::vector<NewArchiveMember> Members;
199 for (auto *Arg : Args.filtered(OPT_INPUT)) {
200 std::string Path = findInputFile(Arg->getValue(), SearchPaths);
201 if (Path.empty()) {
202 llvm::errs() << Arg->getValue() << ": no such file or directory\n";
203 return 1;
204 }
205
206 Expected<NewArchiveMember> MOrErr =
207 NewArchiveMember::getFile(Saver.save(Path), /*Deterministic=*/true);
208 if (!MOrErr) {
209 handleAllErrors(MOrErr.takeError(), [&](const ErrorInfoBase &EIB) {
210 llvm::errs() << Arg->getValue() << ": " << EIB.message() << "\n";
211 });
212 return 1;
213 }
214
215 file_magic Magic = identify_magic(MOrErr->Buf->getBuffer());
216 if (Magic != file_magic::coff_object && Magic != file_magic::bitcode &&
217 Magic != file_magic::windows_resource) {
218 llvm::errs() << Arg->getValue()
219 << ": not a COFF object, bitcode or resource file\n";
220 return 1;
221 }
222
223 // Check that all input files have the same machine type.
224 // Mixing normal objects and LTO bitcode files is fine as long as they
225 // have the same machine type.
226 // Doing this here duplicates the header parsing work that writeArchive()
227 // below does, but it's not a lot of work and it's a bit awkward to do
228 // in writeArchive() which needs to support many tools, can't assume the
229 // input is COFF, and doesn't have a good way to report errors.
230 COFF::MachineTypes FileMachine = COFF::IMAGE_FILE_MACHINE_UNKNOWN;
231 if (Magic == file_magic::coff_object) {
232 std::error_code EC;
233 object::COFFObjectFile Obj(*MOrErr->Buf, EC);
234 if (EC) {
235 llvm::errs() << Arg->getValue() << ": failed to open: " << EC.message()
236 << '\n';
237 return 1;
238 }
239 uint16_t Machine = Obj.getMachine();
240 if (Machine != COFF::IMAGE_FILE_MACHINE_I386 &&
241 Machine != COFF::IMAGE_FILE_MACHINE_AMD64 &&
242 Machine != COFF::IMAGE_FILE_MACHINE_ARMNT &&
243 Machine != COFF::IMAGE_FILE_MACHINE_ARM64) {
244 llvm::errs() << Arg->getValue() << ": unknown machine: " << Machine
245 << '\n';
246 return 1;
247 }
248 FileMachine = static_cast<COFF::MachineTypes>(Machine);
249 } else if (Magic == file_magic::bitcode) {
250 Expected<std::string> TripleStr = getBitcodeTargetTriple(*MOrErr->Buf);
251 if (!TripleStr) {
252 llvm::errs() << Arg->getValue()
253 << ": failed to get target triple from bitcode\n";
254 return 1;
255 }
256 switch (Triple(*TripleStr).getArch()) {
257 case Triple::x86:
258 FileMachine = COFF::IMAGE_FILE_MACHINE_I386;
259 break;
260 case Triple::x86_64:
261 FileMachine = COFF::IMAGE_FILE_MACHINE_AMD64;
262 break;
263 case Triple::arm:
264 FileMachine = COFF::IMAGE_FILE_MACHINE_ARMNT;
265 break;
266 case Triple::aarch64:
267 FileMachine = COFF::IMAGE_FILE_MACHINE_ARM64;
268 break;
269 default:
270 llvm::errs() << Arg->getValue() << ": unknown arch in target triple "
271 << *TripleStr << '\n';
272 return 1;
273 }
274 }
275
276 // FIXME: Once lld-link rejects multiple resource .obj files:
277 // Call convertResToCOFF() on .res files and add the resulting
278 // COFF file to the .lib output instead of adding the .res file, and remove
279 // this check. See PR42180.
280 if (FileMachine != COFF::IMAGE_FILE_MACHINE_UNKNOWN) {
281 if (LibMachine == COFF::IMAGE_FILE_MACHINE_UNKNOWN) {
282 LibMachine = FileMachine;
283 LibMachineSource = std::string(" (inferred from earlier file '") +
284 Arg->getValue() + "')";
285 } else if (LibMachine != FileMachine) {
286 llvm::errs() << Arg->getValue() << ": file machine type "
287 << machineToStr(FileMachine)
288 << " conflicts with library machine type "
289 << machineToStr(LibMachine) << LibMachineSource << '\n';
290 return 1;
291 }
292 }
293
294 Members.emplace_back(std::move(*MOrErr));
295 }
296
297 // Create an archive file.
298 std::string OutputPath = getOutputPath(&Args, Members[0]);
299 // llvm-lib uses relative paths for both regular and thin archives, unlike
300 // standard GNU ar, which only uses relative paths for thin archives and
301 // basenames for regular archives.
302 for (NewArchiveMember &Member : Members) {
303 if (sys::path::is_relative(Member.MemberName)) {
304 Expected<std::string> PathOrErr =
305 computeArchiveRelativePath(OutputPath, Member.MemberName);
306 if (PathOrErr)
307 Member.MemberName = Saver.save(*PathOrErr);
308 }
309 }
310
311 if (Error E =
312 writeArchive(OutputPath, Members,
313 /*WriteSymtab=*/true, object::Archive::K_GNU,
314 /*Deterministic*/ true, Args.hasArg(OPT_llvmlibthin))) {
315 handleAllErrors(std::move(E), [&](const ErrorInfoBase &EI) {
316 llvm::errs() << OutputPath << ": " << EI.message() << "\n";
317 });
318 return 1;
319 }
320
321 return 0;
322}
