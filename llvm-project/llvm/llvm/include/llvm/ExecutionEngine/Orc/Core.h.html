163/// Tracks responsibility for materialization, and mediates interactions between
164/// MaterializationUnits and JDs.
165///
166/// An instance of this class is passed to MaterializationUnits when their
167/// materialize method is called. It allows MaterializationUnits to resolve and
168/// emit symbols, or abandon materialization by notifying any unmaterialized
169/// symbols of an error.
170class MaterializationResponsibility {
171 friend class MaterializationUnit;
172public:
173 MaterializationResponsibility(MaterializationResponsibility &&) = default;
174 MaterializationResponsibility &
175 operator=(MaterializationResponsibility &&) = delete;
176
177 /// Destruct a MaterializationResponsibility instance. In debug mode
178 /// this asserts that all symbols being tracked have been either
179 /// emitted or notified of an error.
180 ~MaterializationResponsibility();
181
182 /// Returns the target JITDylib that these symbols are being materialized
183 /// into.
184 JITDylib &getTargetJITDylib() const { return JD; }
185
186 /// Returns the VModuleKey for this instance.
187 VModuleKey getVModuleKey() const { return K; }
188
189 /// Returns the symbol flags map for this responsibility instance.
190 /// Note: The returned flags may have transient flags (Lazy, Materializing)
191 /// set. These should be stripped with JITSymbolFlags::stripTransientFlags
192 /// before using.
193 const SymbolFlagsMap &getSymbols() const { return SymbolFlags; }
194
195 /// Returns the names of any symbols covered by this
196 /// MaterializationResponsibility object that have queries pending. This
197 /// information can be used to return responsibility for unrequested symbols
198 /// back to the JITDylib via the delegate method.
199 SymbolNameSet getRequestedSymbols() const;
200
201 /// Notifies the target JITDylib that the given symbols have been resolved.
202 /// This will update the given symbols' addresses in the JITDylib, and notify
203 /// any pending queries on the given symbols of their resolution. The given
204 /// symbols must be ones covered by this MaterializationResponsibility
205 /// instance. Individual calls to this method may resolve a subset of the
206 /// symbols, but all symbols must have been resolved prior to calling emit.
207 void resolve(const SymbolMap &Symbols);
208
209 /// Notifies the target JITDylib (and any pending queries on that JITDylib)
210 /// that all symbols covered by this MaterializationResponsibility instance
211 /// have been emitted.
212 void emit();
213
214 /// Adds new symbols to the JITDylib and this responsibility instance.
215 /// JITDylib entries start out in the materializing state.
216 ///
217 /// This method can be used by materialization units that want to add
218 /// additional symbols at materialization time (e.g. stubs, compile
219 /// callbacks, metadata).
220 Error defineMaterializing(const SymbolFlagsMap &SymbolFlags);
221
222 /// Notify all not-yet-emitted covered by this MaterializationResponsibility
223 /// instance that an error has occurred.
224 /// This will remove all symbols covered by this MaterializationResponsibilty
225 /// from the target JITDylib, and send an error to any queries waiting on
226 /// these symbols.
227 void failMaterialization();
228
229 /// Transfers responsibility to the given MaterializationUnit for all
230 /// symbols defined by that MaterializationUnit. This allows
231 /// materializers to break up work based on run-time information (e.g.
232 /// by introspecting which symbols have actually been looked up and
233 /// materializing only those).
234 void replace(std::unique_ptr<MaterializationUnit> MU);
235
236 /// Delegates responsibility for the given symbols to the returned
237 /// materialization responsibility. Useful for breaking up work between
238 /// threads, or different kinds of materialization processes.
239 MaterializationResponsibility delegate(const SymbolNameSet &Symbols,
240 VModuleKey NewKey = VModuleKey());
241
242 void addDependencies(const SymbolStringPtr &Name,
243 const SymbolDependenceMap &Dependencies);
244
245 /// Add dependencies that apply to all symbols covered by this instance.
246 void addDependenciesForAll(const SymbolDependenceMap &Dependencies);
247
248private:
249 /// Create a MaterializationResponsibility for the given JITDylib and
250 /// initial symbols.
251 MaterializationResponsibility(JITDylib &JD, SymbolFlagsMap SymbolFlags,
252 VModuleKey K);
253
254 JITDylib &JD;
255 SymbolFlagsMap SymbolFlags;
256 VModuleKey K;

259/// A MaterializationUnit represents a set of symbol definitions that can
260/// be materialized as a group, or individually discarded (when
261/// overriding definitions are encountered).
262///
263/// MaterializationUnits are used when providing lazy definitions of symbols to
264/// JITDylibs. The JITDylib will call materialize when the address of a symbol
265/// is requested via the lookup method. The JITDylib will call discard if a
266/// stronger definition is added or already present.
267class MaterializationUnit {
268public:
269 MaterializationUnit(SymbolFlagsMap InitalSymbolFlags, VModuleKey K)
270 : SymbolFlags(std::move(InitalSymbolFlags)), K(std::move(K)) {}
271
272 virtual ~MaterializationUnit() {}
273
274 /// Return the name of this materialization unit. Useful for debugging
275 /// output.
276 virtual StringRef getName() const = 0;
277
278 /// Return the set of symbols that this source provides.
279 const SymbolFlagsMap &getSymbols() const { return SymbolFlags; }
280
281 /// Called by materialization dispatchers (see
282 /// ExecutionSession::DispatchMaterializationFunction) to trigger
283 /// materialization of this MaterializationUnit.
284 void doMaterialize(JITDylib &JD) {
285 materialize(MaterializationResponsibility(JD, std::move(SymbolFlags),
286 std::move(K)));
287 }
288
289 /// Called by JITDylibs to notify MaterializationUnits that the given symbol
290 /// has been overridden.
291 void doDiscard(const JITDylib &JD, const SymbolStringPtr &Name) {
292 SymbolFlags.erase(Name);
293 discard(JD, std::move(Name));
294 }
295
296protected:
297 SymbolFlagsMap SymbolFlags;
298 VModuleKey K;
299
300private:
301 virtual void anchor();
302
303 /// Implementations of this method should materialize all symbols
304 /// in the materialzation unit, except for those that have been
305 /// previously discarded.
306 virtual void materialize(MaterializationResponsibility R) = 0;
307
308 /// Implementations of this method should discard the given symbol
309 /// from the source (e.g. if the source is an LLVM IR Module and the
310 /// symbol is a function, delete the function body or mark it available
311 /// externally).
312 virtual void discard(const JITDylib &JD, const SymbolStringPtr &Name) = 0;
313};

529 /// Set the search order to be used when fixing up definitions in JITDylib.
530 /// This will replace the previous search order, and apply to any symbol
531 /// resolutions made for definitions in this JITDylib after the call to
532 /// setSearchOrder (even if the definition itself was added before the
533 /// call).
534 ///
535 /// If SearchThisJITDylibFirst is set, which by default it is, then this
536 /// JITDylib will add itself to the beginning of the SearchOrder (Clients
537 /// should *not* put this JITDylib in the list in this case, to avoid
538 /// redundant lookups).
539 ///
540 /// If SearchThisJITDylibFirst is false then the search order will be used as
541 /// given. The main motivation for this feature is to support deliberate
542 /// shadowing of symbols in this JITDylib by a facade JITDylib. For example,
543 /// the facade may resolve function names to stubs, and the stubs may compile
544 /// lazily by looking up symbols in this dylib. Adding the facade dylib
545 /// as the first in the search order (instead of this dylib) ensures that
546 /// definitions within this dylib resolve to the lazy-compiling stubs,
547 /// rather than immediately materializing the definitions in this dylib.

941template <typename MaterializationUnitType>
942Error JITDylib::define(std::unique_ptr<MaterializationUnitType> &&MU) {
943 assert(MU && "Can not define with a null MU");
944 return ES.runSessionLocked([&, this]() -> Error {
945 if (auto Err = defineImpl(*MU))
946 return Err;
947
948 /// defineImpl succeeded.
949 auto UMI = std::make_shared<UnmaterializedInfo>(std::move(MU));
950 for (auto &KV : UMI->MU->getSymbols())
951 UnmaterializedInfos[KV.first] = UMI;
952
953 return Error::success();
954 });
955}
