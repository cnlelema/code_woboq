13#ifndef LLVM_EXECUTIONENGINE_JITLINK_EHFRAMESUPPORT_H
14#define LLVM_EXECUTIONENGINE_JITLINK_EHFRAMESUPPORT_H

16#include "llvm/ADT/Triple.h"
17#include "llvm/ExecutionEngine/JITLink/JITLink.h"
18#include "llvm/ExecutionEngine/JITSymbol.h"
19#include "llvm/Support/Error.h"
20
21namespace llvm {
22namespace jitlink {
23
24/// Registers all FDEs in the given eh-frame section with the current process.
25Error registerEHFrameSection(const void *EHFrameSectionAddr);
26
27/// Deregisters all FDEs in the given eh-frame section with the current process.
28Error deregisterEHFrameSection(const void *EHFrameSectionAddr);
29
30using StoreFrameAddressFunction = std::function<void(JITTargetAddress)>;
31
32/// Creates a pass that records the address of the EH frame section. If no
33/// eh-frame section is found, it will set EHFrameAddr to zero.
34///
35/// Authors of JITLinkContexts can use this function to register a post-fixup
36/// pass that records the address of the eh-frame section. This address can
37/// be used after finalization to register and deregister the frame.
38AtomGraphPassFunction
39createEHFrameRecorderPass(const Triple &TT,
40 StoreFrameAddressFunction StoreFrameAddress);
41
42} // end namespace jitlink
43} // end namespace llvm

45#endif // LLVM_EXECUTIONENGINE_JITLINK_EHFRAMESUPPORT_H
