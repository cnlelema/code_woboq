120/// Edge class. Represents both object file relocations, as well as layout and
121/// keep-alive constraints.
122class Edge {
123public:
124 using Kind = uint8_t;
125
126 using GenericEdgeKind = enum : Kind {
127 Invalid, // Invalid edge value.
128 FirstKeepAlive, // Keeps target alive. Offset/addend zero.
129 KeepAlive = FirstKeepAlive, // Tag first edge kind that preserves liveness.
130 LayoutNext, // Layout constraint. Offset/Addend zero.
131 FirstRelocation // First architecture specific relocation.
132 };
133
134 using OffsetT = uint32_t;
135 using AddendT = int64_t;
136
137 Edge(Kind K, OffsetT Offset, Atom &Target, AddendT Addend)
138 : Target(&Target), Offset(Offset), Addend(Addend), K(K) {}
139
140 OffsetT getOffset() const { return Offset; }
141 Kind getKind() const { return K; }
142 void setKind(Kind K) { this->K = K; }
143 bool isRelocation() const { return K >= FirstRelocation; }
144 Kind getRelocation() const {
145 assert(isRelocation() && "Not a relocation edge");
146 return K - FirstRelocation;
147 }
148 bool isKeepAlive() const { return K >= FirstKeepAlive; }
149 Atom &getTarget() const { return *Target; }
150 void setTarget(Atom &Target) { this->Target = &Target; }
151 AddendT getAddend() const { return Addend; }
152 void setAddend(AddendT Addend) { this->Addend = Addend; }
153
154private:
155 Atom *Target;
156 OffsetT Offset;
157 AddendT Addend;
158 Kind K = 0;
159};

165/// Base Atom class. Used by absolute and undefined atoms.
166class Atom {
167 friend class AtomGraph;
168
169protected:
170 /// Create a named (as yet unresolved) atom.
171 Atom(StringRef Name)
172 : Name(Name), IsDefined(false), IsLive(false), ShouldDiscard(false),
173 IsGlobal(false), IsAbsolute(false), IsCallable(false),
174 IsExported(false), IsWeak(false), HasLayoutNext(false),
175 IsCommon(false) {}
176
177 /// Create an absolute symbol atom.
178 Atom(StringRef Name, JITTargetAddress Address)
179 : Name(Name), Address(Address), IsDefined(true), IsLive(false),
180 ShouldDiscard(false), IsGlobal(false), IsAbsolute(false),
181 IsCallable(false), IsExported(false), IsWeak(false),
182 HasLayoutNext(false), IsCommon(false) {}
183
184public:
185 /// Returns true if this atom has a name.
186 bool hasName() const { return Name != StringRef(); }
187
188 /// Returns the name of this atom.
189 StringRef getName() const { return Name; }
190
191 /// Returns the current target address of this atom.
192 /// The initial target address (for atoms that have one) will be taken from
193 /// the input object file's virtual address space. During the layout phase
194 /// of JIT linking the atom's address will be updated to point to its final
195 /// address in the JIT'd process.
196 JITTargetAddress getAddress() const { return Address; }
197
198 /// Set the current target address of this atom.
199 void setAddress(JITTargetAddress Address) { this->Address = Address; }
200
201 /// Returns true if this is a defined atom.
202 bool isDefined() const { return IsDefined; }
203
204 /// Returns true if this atom is marked as live.
205 bool isLive() const { return IsLive; }
206
207 /// Mark this atom as live.
208 ///
209 /// Note: Only defined and absolute atoms can be marked live.
210 void setLive(bool IsLive) {
211 assert((IsDefined || IsAbsolute || !IsLive) &&
212 "Only defined and absolute atoms can be marked live");
213 this->IsLive = IsLive;
214 }
215
216 /// Returns true if this atom should be discarded during pruning.
217 bool shouldDiscard() const { return ShouldDiscard; }
218
219 /// Mark this atom to be discarded.
220 ///
221 /// Note: Only defined and absolute atoms can be marked live.
222 void setShouldDiscard(bool ShouldDiscard) {
223 assert((IsDefined || IsAbsolute || !ShouldDiscard) &&
224 "Only defined and absolute atoms can be marked live");
225 this->ShouldDiscard = ShouldDiscard;
226 }
227
228 /// Returns true if this definition is global (i.e. visible outside this
229 /// linkage unit).
230 ///
231 /// Note: This is distict from Exported, which means visibile outside the
232 /// JITDylib that this graph is being linked in to.
233 bool isGlobal() const { return IsGlobal; }
234
235 /// Mark this atom as global.
236 void setGlobal(bool IsGlobal) { this->IsGlobal = IsGlobal; }
237
238 /// Returns true if this atom represents an absolute symbol.
239 bool isAbsolute() const { return IsAbsolute; }
240
241 /// Returns true if this atom is known to be callable.
242 ///
243 /// Primarily provided for easy interoperability with ORC, which uses the
244 /// JITSymbolFlags::Common flag to identify symbols that can be interposed
245 /// with stubs.
246 bool isCallable() const { return IsCallable; }
247
248 /// Mark this atom as callable.
249 void setCallable(bool IsCallable) {
250 assert((IsDefined || IsAbsolute || !IsCallable) &&
251 "Callable atoms must be defined or absolute");
252 this->IsCallable = IsCallable;
253 }
254
255 /// Returns true if this atom should appear in the symbol table of a final
256 /// linked image.
257 bool isExported() const { return IsExported; }
258
259 /// Mark this atom as exported.
260 void setExported(bool IsExported) {
261 assert((!IsExported || ((IsDefined || IsAbsolute) && hasName())) &&
262 "Exported atoms must have names");
263 this->IsExported = IsExported;
264 }
265
266 /// Returns true if this is a weak symbol.
267 bool isWeak() const { return IsWeak; }
268
269 /// Mark this atom as weak.
270 void setWeak(bool IsWeak) { this->IsWeak = IsWeak; }
271
272private:
273 StringRef Name;
274 JITTargetAddress Address = 0;
275
276 bool IsDefined : 1;
277 bool IsLive : 1;
278 bool ShouldDiscard : 1;
279
280 bool IsGlobal : 1;
281 bool IsAbsolute : 1;
282 bool IsCallable : 1;
283 bool IsExported : 1;
284 bool IsWeak : 1;
285
286protected:
287 // These flags only make sense for DefinedAtom, but we can minimize the size
288 // of DefinedAtom by defining them here.
289 bool HasLayoutNext : 1;
290 bool IsCommon : 1;
291};

300/// Represents a section address range via a pair of DefinedAtom pointers to
301/// the first and last atoms in the section.
302class SectionRange {
303public:
304 SectionRange() = default;
305 SectionRange(DefinedAtom *First, DefinedAtom *Last)
306 : First(First), Last(Last) {}
307 DefinedAtom *getFirstAtom() const {
308 assert((!Last || First) && "First can not be null if end is non-null");
309 return First;
310 }
311 DefinedAtom *getLastAtom() const {
312 assert((First || !Last) && "Last can not be null if start is non-null");
313 return Last;
314 }

316 assert((First || !Last) && "Last can not be null if start is non-null");
317 return !First;
318 }
319 JITTargetAddress getStart() const;
320 JITTargetAddress getEnd() const;
321 uint64_t getSize() const;
322
323private:
324 DefinedAtom *First = nullptr;
325 DefinedAtom *Last = nullptr;
326};

328/// Represents an object file section.
329class Section {
330 friend class AtomGraph;
331
332private:
333 Section(StringRef Name, uint32_t Alignment, sys::Memory::ProtectionFlags Prot,
334 unsigned Ordinal, bool IsZeroFill)
335 : Name(Name), Alignment(Alignment), Prot(Prot), Ordinal(Ordinal),
336 IsZeroFill(IsZeroFill) {
337 assert(isPowerOf2_32(Alignment) && "Alignments must be a power of 2");
338 }
339
340 using DefinedAtomSet = DenseSet<DefinedAtom *>;
341
342public:
343 using atom_iterator = DefinedAtomSet::iterator;
344 using const_atom_iterator = DefinedAtomSet::const_iterator;
345
346 ~Section();
347 StringRef getName() const { return Name; }
348 uint32_t getAlignment() const { return Alignment; }
349 sys::Memory::ProtectionFlags getProtectionFlags() const { return Prot; }
350 unsigned getSectionOrdinal() const { return Ordinal; }
351 size_t getNextAtomOrdinal() { return ++NextAtomOrdinal; }
352
353 bool isZeroFill() const { return IsZeroFill; }
354
355 /// Returns an iterator over the atoms in the section (in no particular
356 /// order).
357 iterator_range<atom_iterator> atoms() {
358 return make_range(DefinedAtoms.begin(), DefinedAtoms.end());
359 }
360
361 /// Returns an iterator over the atoms in the section (in no particular
362 /// order).
363 iterator_range<const_atom_iterator> atoms() const {
364 return make_range(DefinedAtoms.begin(), DefinedAtoms.end());
365 }
366
367 /// Return the number of atoms in this section.
368 DefinedAtomSet::size_type atoms_size() { return DefinedAtoms.size(); }
369
370 /// Return true if this section contains no atoms.
371 bool atoms_empty() const { return DefinedAtoms.empty(); }
372
373 /// Returns the range of this section as the pair of atoms with the lowest
374 /// and highest target address. This operation is expensive, as it
375 /// must traverse all atoms in the section.
376 ///
377 /// Note: If the section is empty, both values will be null. The section
378 /// address will evaluate to null, and the size to zero. If the section
379 /// contains a single atom both values will point to it, the address will
380 /// evaluate to the address of that atom, and the size will be the size of
381 /// that atom.
382 SectionRange getRange() const;
383
384private:
385 void addAtom(DefinedAtom &DA) {
386 assert(!DefinedAtoms.count(&DA) && "Atom is already in this section");
387 DefinedAtoms.insert(&DA);
388 }
389
390 void removeAtom(DefinedAtom &DA) {
391 assert(DefinedAtoms.count(&DA) && "Atom is not in this section");
392 DefinedAtoms.erase(&DA);
393 }
394
395 StringRef Name;
396 uint32_t Alignment = 0;
397 sys::Memory::ProtectionFlags Prot;
398 unsigned Ordinal = 0;
399 unsigned NextAtomOrdinal = 0;
400 bool IsZeroFill = false;
401 DefinedAtomSet DefinedAtoms;
402};

404/// Defined atom class. Suitable for use by defined named and anonymous
405/// atoms.
406class DefinedAtom : public Atom {
407 friend class AtomGraph;
408
409private:
410 DefinedAtom(Section &Parent, JITTargetAddress Address, uint32_t Alignment)
411 : Atom("", Address), Parent(Parent), Ordinal(Parent.getNextAtomOrdinal()),
412 Alignment(Alignment) {
413 assert(isPowerOf2_32(Alignment) && "Alignments must be a power of two");
414 }
415
416 DefinedAtom(Section &Parent, StringRef Name, JITTargetAddress Address,
417 uint32_t Alignment)
418 : Atom(Name, Address), Parent(Parent),
419 Ordinal(Parent.getNextAtomOrdinal()), Alignment(Alignment) {
420 assert(isPowerOf2_32(Alignment) && "Alignments must be a power of two");
421 }
422
423public:
424 using edge_iterator = EdgeVector::iterator;
425
426 Section &getSection() const { return Parent; }
427
428 uint64_t getSize() const { return Size; }
429
430 StringRef getContent() const {
431 assert(!Parent.isZeroFill() && "Trying to get content for zero-fill atom");
432 assert(Size <= std::numeric_limits<size_t>::max() &&
433 "Content size too large");
434 return {ContentPtr, static_cast<size_t>(Size)};
435 }
436 void setContent(StringRef Content) {
437 assert(!Parent.isZeroFill() && "Calling setContent on zero-fill atom?");
438 ContentPtr = Content.data();
439 Size = Content.size();
440 }
441
442 bool isZeroFill() const { return Parent.isZeroFill(); }
443
444 void setZeroFill(uint64_t Size) {
445 assert(Parent.isZeroFill() && !ContentPtr &&
446 "Can't set zero-fill length of a non zero-fill atom");
447 this->Size = Size;
448 }
449
450 uint64_t getZeroFillSize() const {
451 assert(Parent.isZeroFill() &&
452 "Can't get zero-fill length of a non zero-fill atom");
453 return Size;
454 }
455
456 uint32_t getAlignment() const { return Alignment; }
457
458 bool hasLayoutNext() const { return HasLayoutNext; }
459 void setLayoutNext(DefinedAtom &Next) {
460 assert(!HasLayoutNext && "Atom already has layout-next constraint");
461 HasLayoutNext = true;
462 Edges.push_back(Edge(Edge::LayoutNext, 0, Next, 0));
463 }
464 DefinedAtom &getLayoutNext() {
465 assert(HasLayoutNext && "Atom does not have a layout-next constraint");
466 DefinedAtom *Next = nullptr;
467 for (auto &E : edges())
468 if (E.getKind() == Edge::LayoutNext) {
469 assert(E.getTarget().isDefined() &&
470 "layout-next target atom must be a defined atom");
471 Next = static_cast<DefinedAtom *>(&E.getTarget());
472 break;
473 }
474 assert(Next && "Missing LayoutNext edge");
475 return *Next;
476 }
477
478 bool isCommon() const { return IsCommon; }
479
480 void addEdge(Edge::Kind K, Edge::OffsetT Offset, Atom &Target,
481 Edge::AddendT Addend) {
482 assert(K != Edge::LayoutNext &&
483 "Layout edges should be added via setLayoutNext");
484 Edges.push_back(Edge(K, Offset, Target, Addend));
485 }
486
487 iterator_range<edge_iterator> edges() {
488 return make_range(Edges.begin(), Edges.end());
489 }
490 size_t edges_size() const { return Edges.size(); }
491 bool edges_empty() const { return Edges.empty(); }
492
493 unsigned getOrdinal() const { return Ordinal; }
494
495private:
496 void setCommon(uint64_t Size) {
497 assert(ContentPtr == 0 && "Atom already has content?");
498 IsCommon = true;
499 setZeroFill(Size);
500 }
501
502 EdgeVector Edges;
503 uint64_t Size = 0;
504 Section &Parent;
505 const char *ContentPtr = nullptr;
506 unsigned Ordinal = 0;
507 uint32_t Alignment = 0;
508};

510inline JITTargetAddress SectionRange::getStart() const {
511 return First ? First->getAddress() : 0;
512}
513
514inline JITTargetAddress SectionRange::getEnd() const {
515 return Last ? Last->getAddress() + Last->getSize() : 0;
516}
517
518inline uint64_t SectionRange::getSize() const { return getEnd() - getStart(); }
519
520inline SectionRange Section::getRange() const {
521 if (atoms_empty())
522 return SectionRange();
523 DefinedAtom *First = *DefinedAtoms.begin(), *Last = *DefinedAtoms.begin();
524 for (auto *DA : atoms()) {
525 if (DA->getAddress() < First->getAddress())
526 First = DA;
527 if (DA->getAddress() > Last->getAddress())
528 Last = DA;
529 }
530 return SectionRange(First, Last);
531}

533class AtomGraph {
534private:
535 using SectionList = std::vector<std::unique_ptr<Section>>;
536 using AddressToAtomMap = std::map<JITTargetAddress, DefinedAtom *>;
537 using NamedAtomMap = DenseMap<StringRef, Atom *>;

540public:
541 using external_atom_iterator = ExternalAtomSet::iterator;

543 using section_iterator = pointee_iterator<SectionList::iterator>;
544 using const_section_iterator = pointee_iterator<SectionList::const_iterator>;

546 template <typename SecItrT, typename AtomItrT, typename T>
547 class defined_atom_iterator_impl
548 : public iterator_facade_base<
549 defined_atom_iterator_impl<SecItrT, AtomItrT, T>,
550 std::forward_iterator_tag, T> {
551 public:
552 defined_atom_iterator_impl() = default;
553
554 defined_atom_iterator_impl(SecItrT SI, SecItrT SE)
555 : SI(SI), SE(SE),
556 AI(SI != SE ? SI->atoms().begin() : Section::atom_iterator()) {
557 moveToNextAtomOrEnd();
558 }
559
560 bool operator==(const defined_atom_iterator_impl &RHS) const {
561 return (SI == RHS.SI) && (AI == RHS.AI);
562 }
563
564 T operator*() const {
565 assert(AI != SI->atoms().end() && "Dereferencing end?");
566 return *AI;
567 }
568
569 defined_atom_iterator_impl operator++() {
570 ++AI;
571 moveToNextAtomOrEnd();
572 return *this;
573 }
574
575 private:
576 void moveToNextAtomOrEnd() {
577 while (SI != SE && AI == SI->atoms().end()) {
578 ++SI;
579 if (SI == SE)
580 AI = Section::atom_iterator();
581 else
582 AI = SI->atoms().begin();
583 }
584 }
585
586 SecItrT SI, SE;
587 AtomItrT AI;
588 };

645 /// Add an anonymous defined atom to the graph.
646 ///
647 /// Anonymous atoms have content but no name. They must have an address.
648 DefinedAtom &addAnonymousAtom(Section &Parent, JITTargetAddress Address,

718 iterator_range<const_defined_atom_iterator> defined_atoms() const {
719 return make_range(
720 const_defined_atom_iterator(Sections.begin(), Sections.end()),
721 const_defined_atom_iterator(Sections.end(), Sections.end()));
722 }

724 /// Returns the atom with the given name, which must exist in this graph.
725 Atom &getAtomByName(StringRef Name) {
726 auto I = NamedAtoms.find(Name);
727 assert(I != NamedAtoms.end() && "Name not in NamedAtoms map");
728 return *I->second;
729 }

731 /// Returns the atom with the given name, which must exist in this graph and
732 /// be a DefinedAtom.
733 DefinedAtom &getDefinedAtomByName(StringRef Name) {
734 auto &A = getAtomByName(Name);
735 assert(A.isDefined() && "Atom is not a defined atom");
736 return static_cast<DefinedAtom &>(A);

739 /// Search for the given atom by name.
740 /// Returns the atom (if found) or an error (if no atom with this name
741 /// exists).
742 Expected<Atom &> findAtomByName(StringRef Name) {
743 auto I = NamedAtoms.find(Name);
744 if (I == NamedAtoms.end())
745 return make_error<JITLinkError>("No atom named " + Name);
746 return *I->second;
747 }
748
749 /// Search for the given defined atom by name.
750 /// Returns the defined atom (if found) or an error (if no atom with this
751 /// name exists, or if one exists but is not a defined atom).
752 Expected<DefinedAtom &> findDefinedAtomByName(StringRef Name) {
753 auto I = NamedAtoms.find(Name);
754 if (I == NamedAtoms.end())
755 return make_error<JITLinkError>("No atom named " + Name);
756 if (!I->second->isDefined())
757 return make_error<JITLinkError>("Atom " + Name +
758 " exists but is not a "
759 "defined atom");
760 return static_cast<DefinedAtom &>(*I->second);
761 }
762
763 /// Returns the atom covering the given address, or an error if no such atom
764 /// exists.
765 ///
766 /// Returns null if no atom exists at the given address.
767 DefinedAtom *getAtomByAddress(JITTargetAddress Address) {
768 refreshAddrToAtomCache();
769
770 // If there are no defined atoms, bail out early.
771 if (AddrToAtomCache->empty())
772 return nullptr;
773
774 // Find the atom *after* the given address.
775 auto I = AddrToAtomCache->upper_bound(Address);
776
777 // If this address falls before any known atom, bail out.
778 if (I == AddrToAtomCache->begin())
779 return nullptr;
780
781 // The atom we're looking for is the one before the atom we found.
782 --I;
783
784 // Otherwise range check the atom that was found.
785 assert(!I->second->getContent().empty() && "Atom content not set");
786 if (Address >= I->second->getAddress() + I->second->getContent().size())
787 return nullptr;
788
789 return I->second;
790 }
791
792 /// Like getAtomByAddress, but returns an Error if the given address is not
793 /// covered by an atom, rather than a null pointer.
794 Expected<DefinedAtom &> findAtomByAddress(JITTargetAddress Address) {
795 if (auto *DA = getAtomByAddress(Address))
796 return *DA;
797 return make_error<JITLinkError>("No atom at address " +
798 formatv("{0:x16}", Address));
799 }

801 // Remove the given external atom from the graph.
802 void removeExternalAtom(Atom &A) {
803 assert(!A.isDefined() && !A.isAbsolute() && "A is not an external atom");
804 assert(ExternalAtoms.count(&A) && "A is not in the external atoms set");
805 ExternalAtoms.erase(&A);
806 A.~Atom();
807 }
808
809 /// Remove the given absolute atom from the graph.
810 void removeAbsoluteAtom(Atom &A) {
811 assert(A.isAbsolute() && "A is not an absolute atom");
812 assert(AbsoluteAtoms.count(&A) && "A is not in the absolute atoms set");
813 AbsoluteAtoms.erase(&A);
814 A.~Atom();
815 }
816
817 /// Remove the given defined atom from the graph.
818 void removeDefinedAtom(DefinedAtom &DA) {
819 if (AddrToAtomCache) {
820 assert(AddrToAtomCache->count(DA.getAddress()) &&
821 "Cache exists, but does not contain atom");
822 AddrToAtomCache->erase(DA.getAddress());
823 }
824 if (DA.hasName()) {
825 assert(NamedAtoms.count(DA.getName()) && "Named atom not in map");
826 NamedAtoms.erase(DA.getName());
827 }
828 DA.getSection().removeAtom(DA);
829 DA.~DefinedAtom();
830 }

840 void dump(raw_ostream &OS,
841 std::function<StringRef(Edge::Kind)> EdegKindToName =

844private:
845 AddressToAtomMap &getAddrToAtomMap() {
846 refreshAddrToAtomCache();
847 return *AddrToAtomCache;

855 void refreshAddrToAtomCache() const {
856 if (!AddrToAtomCache) {
857 AddrToAtomCache = AddressToAtomMap();
858 for (auto *DA : defined_atoms())
859 (*AddrToAtomCache)[DA->getAddress()] = const_cast<DefinedAtom *>(DA);

883/// An atom graph pass configuration, consisting of a list of pre-prune,
884/// post-prune, and post-fixup passes.
885struct PassConfiguration {
886
887 /// Pre-prune passes.
888 ///
889 /// These passes are called on the graph after it is built, and before any
890 /// atoms have been pruned.
891 ///
892 /// Notable use cases: Marking atoms live or should-discard.
893 AtomGraphPassList PrePrunePasses;
894
895 /// Post-prune passes.
896 ///
897 /// These passes are called on the graph after dead and should-discard atoms
898 /// have been removed, but before fixups are applied.
899 ///
900 /// Notable use cases: Building GOT, stub, and TLV atoms.
901 AtomGraphPassList PostPrunePasses;
902
903 /// Post-fixup passes.
904 ///
905 /// These passes are called on the graph after atom contents has been copied
906 /// to working memory, and fixups applied.
907 ///
908 /// Notable use cases: Testing and validation.
909 AtomGraphPassList PostFixupPasses;
910};
