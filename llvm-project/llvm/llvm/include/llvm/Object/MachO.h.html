412 // Given a SegIndex, SegOffset, and PointerSize, verify a valid section exists
413 // that fully contains a pointer at that location. Multiple fixups in a bind
414 // (such as with the BIND_OPCODE_DO_BIND_ULEB_TIMES_SKIPPING_ULEB opcode) can
415 // be tested via the Count and Skip parameters.
416 //
417 // This is used by MachOBindEntry::moveNext() to validate a MachOBindEntry.
418 const char *BindEntryCheckSegAndOffsets(int32_t SegIndex, uint64_t SegOffset,
419 uint8_t PointerSize, uint32_t Count=1,
420 uint32_t Skip=0) const {
421 return BindRebaseSectionTable->checkSegAndOffsets(SegIndex, SegOffset,

604 static std::string getBuildPlatform(uint32_t platform) {
605 switch (platform) {
606 case MachO::PLATFORM_MACOS: return "macos";
607 case MachO::PLATFORM_IOS: return "ios";
608 case MachO::PLATFORM_TVOS: return "tvos";
609 case MachO::PLATFORM_WATCHOS: return "watchos";
610 case MachO::PLATFORM_BRIDGEOS: return "bridgeos";
611 case MachO::PLATFORM_IOSSIMULATOR: return "iossimulator";
612 case MachO::PLATFORM_TVOSSIMULATOR: return "tvossimulator";
613 case MachO::PLATFORM_WATCHOSSIMULATOR: return "watchossimulator";
614 default:
615 std::string ret;
616 raw_string_ostream ss(ret);
617 ss << format_hex(platform, 8, true);
618 return ss.str();
619 }
620 }

635 static std::string getVersionString(uint32_t version) {
636 uint32_t major = (version >> 16) & 0xffff;
637 uint32_t minor = (version >> 8) & 0xff;
638 uint32_t update = version & 0xff;
639
640 SmallString<32> Version;
641 Version = utostr(major) + "." + utostr(minor);
642 if (update != 0)
643 Version += "." + utostr(update);
644 return Version.str();
645 }
