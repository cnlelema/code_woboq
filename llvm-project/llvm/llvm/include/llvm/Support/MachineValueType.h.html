254 // A class to represent the number of elements in a vector
255 //
256 // For fixed-length vectors, the total number of elements is equal to 'Min'
257 // For scalable vectors, the total number of elements is a multiple of 'Min'
258 class ElementCount {
259 public:
260 unsigned Min;
261 bool Scalable;
262
263 ElementCount(unsigned Min, bool Scalable)
264 : Min(Min), Scalable(Scalable) {}
265
266 ElementCount operator*(unsigned RHS) {
267 return { Min * RHS, Scalable };
268 }
269
270 ElementCount& operator*=(unsigned RHS) {
271 Min *= RHS;
272 return *this;
273 }
274
275 ElementCount operator/(unsigned RHS) {
276 return { Min / RHS, Scalable };
277 }
278
279 ElementCount& operator/=(unsigned RHS) {
280 Min /= RHS;
281 return *this;
282 }
283
284 bool operator==(const ElementCount& RHS) {
285 return Min == RHS.Min && Scalable == RHS.Scalable;
286 }
287 };

292 bool operator>(const MVT& S) const { return SimpleTy > S.SimpleTy; }
293 bool operator<(const MVT& S) const { return SimpleTy < S.SimpleTy; }
294 bool operator==(const MVT& S) const { return SimpleTy == S.SimpleTy; }
295 bool operator!=(const MVT& S) const { return SimpleTy != S.SimpleTy; }

297 bool operator<=(const MVT& S) const { return SimpleTy <= S.SimpleTy; }

299 /// Return true if this is a valid simple valuetype.
300 bool isValid() const {
301 return (SimpleTy >= MVT::FIRST_VALUETYPE &&
302 SimpleTy < MVT::LAST_VALUETYPE);
303 }
304
305 /// Return true if this is a FP or a vector FP type.
306 bool isFloatingPoint() const {
307 return ((SimpleTy >= MVT::FIRST_FP_VALUETYPE &&
308 SimpleTy <= MVT::LAST_FP_VALUETYPE) ||
309 (SimpleTy >= MVT::FIRST_FP_VECTOR_VALUETYPE &&
310 SimpleTy <= MVT::LAST_FP_VECTOR_VALUETYPE));
311 }

313 /// Return true if this is an integer or a vector integer type.
314 bool isInteger() const {
315 return ((SimpleTy >= MVT::FIRST_INTEGER_VALUETYPE &&
316 SimpleTy <= MVT::LAST_INTEGER_VALUETYPE) ||
317 (SimpleTy >= MVT::FIRST_INTEGER_VECTOR_VALUETYPE &&
318 SimpleTy <= MVT::LAST_INTEGER_VECTOR_VALUETYPE));
319 }

321 /// Return true if this is a fat pointer type.

333 /// Return true if this is a vector value type.

414 /// Returns true if the given vector is a power of 2.
415 bool isPow2VectorType() const {
416 unsigned NElts = getVectorNumElements();
417 return !(NElts & (NElts - 1));
418 }
419
420 /// Widens the length of the given vector MVT up to the nearest power of 2
421 /// and returns that type.
422 MVT getPow2VectorType() const {
423 if (isPow2VectorType())
424 return *this;
425
426 unsigned NElts = getVectorNumElements();
427 unsigned Pow2NElts = 1 << Log2_32_Ceil(NElts);
428 return MVT::getVectorVT(getVectorElementType(), Pow2NElts);
429 }

654 default:
