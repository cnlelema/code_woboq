31class StatsOutputFile {
32 std::unique_ptr<llvm::raw_fd_ostream> Stream;
33 int FD;
34 explicit StatsOutputFile(int FD, std::unique_ptr<llvm::raw_fd_ostream> S)
35 : Stream(std::move(S)), FD(FD) {}
36
37public:
38 using ErrorCallback = llvm::unique_function<void(StringRef, const std::error_code &)>;
39 llvm::raw_fd_ostream &stream() { return *Stream; }
40 uint64_t size();
41 ~StatsOutputFile();
42 static std::unique_ptr<StatsOutputFile>
43 open(StringRef File, ErrorCallback OnOpenError, ErrorCallback OnLockError);
44};
45
46enum class SetBoundsPointerSource {
47 Unknown,
48 Heap,
49 Stack,
50 GlobalVar,
51 CodePointer,
52 SubObject,
53};
54
55class CSetBoundsStatistics {
56public:
57 struct Entry {
58 uint64_t KnownAlignment;
59 Optional<uint64_t> RequestedSize;
60 Optional<uint64_t> RequestedSizeMultipleOf;
61 SetBoundsPointerSource PointerKind;
62 std::string SourceLocation;
63 std::string Pass;
64 std::string Details;
65 };
66 CSetBoundsStatistics();
67 ~CSetBoundsStatistics();
68
69 void add(unsigned KnownAlignment, Optional<uint64_t> Length, StringRef Pass,
70 SetBoundsPointerSource Kind, const Twine &Details,
71 std::string SourceLoc, Optional<uint64_t> SizeMultipleOf = None);
72 void print(llvm::raw_ostream &OS, StringRef MainFile, bool PrintHeader);
73 void print(StatsOutputFile &S, StringRef MainFile);
74 static StringRef outputFile();
75
76private:
77 SmallVector<Entry, 32> Entries;
78};
