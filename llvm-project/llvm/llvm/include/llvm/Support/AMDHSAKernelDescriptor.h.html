21// Gets offset of specified member in specified type.

26// Creates enumeration entries used for packing bits into integers. Enumeration
27// entries include bit shift amount, bit width, and bit mask.
28#ifndef AMDHSA_BITS_ENUM_ENTRY
29#define AMDHSA_BITS_ENUM_ENTRY(NAME, SHIFT, WIDTH) \
30 NAME ## _SHIFT = (SHIFT), \
31 NAME ## _WIDTH = (WIDTH), \
32 NAME = (((1 << (WIDTH)) - 1) << (SHIFT))
33#endif // AMDHSA_BITS_ENUM_ENTRY
34
35// Gets bits for specified bit mask from specified source.
36#ifndef AMDHSA_BITS_GET
37#define AMDHSA_BITS_GET(SRC, MSK) ((SRC & MSK) >> MSK ## _SHIFT)
38#endif // AMDHSA_BITS_GET
39
40// Sets bits for specified bit mask in specified destination.
41#ifndef AMDHSA_BITS_SET
42#define AMDHSA_BITS_SET(DST, MSK, VAL) \
43 DST &= ~MSK; \
44 DST |= ((VAL << MSK ## _SHIFT) & MSK)
45#endif // AMDHSA_BITS_SET

150// Kernel descriptor. Must be kept backwards compatible.
151struct kernel_descriptor_t {
152 uint32_t group_segment_fixed_size;
153 uint32_t private_segment_fixed_size;
154 uint8_t reserved0[8];
155 int64_t kernel_code_entry_byte_offset;
156 uint8_t reserved1[20];
157 uint32_t compute_pgm_rsrc3; // GFX10+
158 uint32_t compute_pgm_rsrc1;
159 uint32_t compute_pgm_rsrc2;
160 uint16_t kernel_code_properties;
161 uint8_t reserved2[6];
162};
163
164static_assert(
165 sizeof(kernel_descriptor_t) == 64,
166 "invalid size for kernel_descriptor_t");
167static_assert(
168 offsetof(kernel_descriptor_t, group_segment_fixed_size) == 0,
169 "invalid offset for group_segment_fixed_size");
170static_assert(
171 offsetof(kernel_descriptor_t, private_segment_fixed_size) == 4,
172 "invalid offset for private_segment_fixed_size");
173static_assert(
174 offsetof(kernel_descriptor_t, reserved0) == 8,
175 "invalid offset for reserved0");
176static_assert(
177 offsetof(kernel_descriptor_t, kernel_code_entry_byte_offset) == 16,
178 "invalid offset for kernel_code_entry_byte_offset");
179static_assert(
180 offsetof(kernel_descriptor_t, reserved1) == 24,
181 "invalid offset for reserved1");
182static_assert(
183 offsetof(kernel_descriptor_t, compute_pgm_rsrc3) == 44,
184 "invalid offset for compute_pgm_rsrc3");
185static_assert(
186 offsetof(kernel_descriptor_t, compute_pgm_rsrc1) == 48,
187 "invalid offset for compute_pgm_rsrc1");
188static_assert(
189 offsetof(kernel_descriptor_t, compute_pgm_rsrc2) == 52,
190 "invalid offset for compute_pgm_rsrc2");
191static_assert(
192 offsetof(kernel_descriptor_t, kernel_code_properties) == 56,
193 "invalid offset for kernel_code_properties");
194static_assert(
195 offsetof(kernel_descriptor_t, reserved2) == 58,
196 "invalid offset for reserved2");
