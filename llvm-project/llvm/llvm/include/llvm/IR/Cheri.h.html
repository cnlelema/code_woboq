27/// Try to get a DataLayout from an Instr/BB/Func/Module:
28inline const DataLayout *getDataLayoutOrNull(const Module *M) {
29 return M ? &M->getDataLayout() : nullptr;
30}
31inline const DataLayout *getDataLayoutOrNull(const Function *FuncForDL) {
32 return FuncForDL ? getDataLayoutOrNull(FuncForDL->getParent()) : nullptr;
33}
34inline const DataLayout *getDataLayoutOrNull(const BasicBlock *BBForDL) {
35 return BBForDL ? getDataLayoutOrNull(BBForDL->getParent()) : nullptr;
36}
37inline const DataLayout *getDataLayoutOrNull(const Instruction *InstForDL) {
38 return InstForDL ? getDataLayoutOrNull(InstForDL->getParent()) : nullptr;
39}
40
41/// This function can be used if there is no DataLayout available and will
42/// in that case assume that AS200 is only used for CHERI
43inline bool isCheriPointer(unsigned AddrSpace, const DataLayout *DL) {
44 // FIXME: this assumes no other backend uses AS200/201
45 if (DL)
46 return DL->isFatPointer(AddrSpace);
47 return AddrSpace == 200 /* TODO: || AddrSpace == 201 */;
48}
49
50inline bool isCheriPointer(Type *Ty, const DataLayout *DL) {
51 return Ty->isPointerTy() && isCheriPointer(Ty->getPointerAddressSpace(), DL);
52}
