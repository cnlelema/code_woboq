9// This file defines classes that make it really easy to deal with intrinsic
10// functions with the isa/dyncast family of functions. In particular, this
11// allows you to do things like:
12//
13// if (MemCpyInst *MCI = dyn_cast<MemCpyInst>(Inst))
14// ... MCI->getDest() ... MCI->getSource() ...
15//
16// All intrinsic function calls are instances of the call instruction, so these
17// are all subclasses of the CallInst class. Note that none of these classes
18// has state or virtual methods, which is an important part of this gross/neat
19// hack working.
20//
21//===----------------------------------------------------------------------===//
22

40 /// A wrapper class for inspecting calls to intrinsic functions.
41 /// This allows the standard isa/dyncast/cast functionality to work with calls
42 /// to intrinsic functions.
43 class IntrinsicInst : public CallInst {
44 public:
45 IntrinsicInst() = delete;
46 IntrinsicInst(const IntrinsicInst &) = delete;
47 IntrinsicInst &operator=(const IntrinsicInst &) = delete;
48
49 /// Return the intrinsic ID of this intrinsic.
50 Intrinsic::ID getIntrinsicID() const {
51 return getCalledFunction()->getIntrinsicID();
52 }
53
54 // Methods for support type inquiry through isa, cast, and dyn_cast:
55 static bool classof(const CallInst *I) {
56 if (const Function *CF = I->getCalledFunction())
57 return CF->isIntrinsic();
58 return false;
59 }
60 static bool classof(const Value *V) {
61 return isa<CallInst>(V) && classof(cast<CallInst>(V));
62 }
63 };

93
94 /// Does this describe the address of a local variable. True for dbg.addr
95 /// and dbg.declare, but not dbg.value, which describes its value.

97 return getIntrinsicID() != Intrinsic::dbg_value;
98 }
99

103

110 }
111
112 Metadata *getRawExpression() const {
113 return cast<MetadataAsValue>(getArgOperand(2))->getMetadata();

115
116 /// Get the size (in bits) of the variable, or fragment of the variable that
117 /// is described.

348 /// Common base class for all memory intrinsics. Simply provides
349 /// common methods.
350 /// Written as CRTP to avoid a common base class amongst the
351 /// three atomicity hierarchies.
352 template <typename Derived> class MemIntrinsicBase : public IntrinsicInst {
353 private:
354 enum { ARG_DEST = 0, ARG_LENGTH = 2 };
355
356 public:
357 Value *getRawDest() const {
358 return const_cast<Value *>(getArgOperand(ARG_DEST));
359 }
360 const Use &getRawDestUse() const { return getArgOperandUse(ARG_DEST); }
361 Use &getRawDestUse() { return getArgOperandUse(ARG_DEST); }
362
363 Value *getLength() const {
364 return const_cast<Value *>(getArgOperand(ARG_LENGTH));
365 }
366 const Use &getLengthUse() const { return getArgOperandUse(ARG_LENGTH); }
367 Use &getLengthUse() { return getArgOperandUse(ARG_LENGTH); }
368
369 /// This is just like getRawDest, but it strips off any cast
370 /// instructions (including addrspacecast) that feed it, giving the
371 /// original input. The returned value is guaranteed to be a pointer.
372 Value *getDest() const { return getRawDest()->stripPointerCasts(); }
373
374 unsigned getDestAddressSpace() const {
375 return cast<PointerType>(getRawDest()->getType())->getAddressSpace();
376 }
377
378 unsigned getDestAlignment() const { return getParamAlignment(ARG_DEST); }
379
380 /// Set the specified arguments of the instruction.
381 void setDest(Value *Ptr) {
382 assert(getRawDest()->getType() == Ptr->getType() &&
383 "setDest called with pointer of wrong type!");
384 setArgOperand(ARG_DEST, Ptr);
385 }
386
387 void setDestAlignment(unsigned Align) {
388 removeParamAttr(ARG_DEST, Attribute::Alignment);
389 if (Align > 0)
390 addParamAttr(ARG_DEST,
391 Attribute::getWithAlignment(getContext(), Align));
392 }
393
394 void setLength(Value *L) {
395 assert(getLength()->getType() == L->getType() &&
396 "setLength called with value of wrong type!");
397 setArgOperand(ARG_LENGTH, L);
398 }
399 };

401 /// Common base class for all memory transfer intrinsics. Simply provides
402 /// common methods.
403 template <class BaseCL> class MemTransferBase : public BaseCL {
404 private:
405 enum { ARG_SOURCE = 1 };
406
407 public:
408 /// Return the arguments to the instruction.
409 Value *getRawSource() const {
410 return const_cast<Value *>(BaseCL::getArgOperand(ARG_SOURCE));
411 }
412 const Use &getRawSourceUse() const {
413 return BaseCL::getArgOperandUse(ARG_SOURCE);
414 }
415 Use &getRawSourceUse() { return BaseCL::getArgOperandUse(ARG_SOURCE); }
416
417 /// This is just like getRawSource, but it strips off any cast
418 /// instructions that feed it, giving the original input. The returned
419 /// value is guaranteed to be a pointer.
420 Value *getSource() const { return getRawSource()->stripPointerCasts(); }
421
422 unsigned getSourceAddressSpace() const {
423 return cast<PointerType>(getRawSource()->getType())->getAddressSpace();
424 }
425
426 unsigned getSourceAlignment() const {
427 return BaseCL::getParamAlignment(ARG_SOURCE);
428 }
429
430 void setSource(Value *Ptr) {
431 assert(getRawSource()->getType() == Ptr->getType() &&
432 "setSource called with pointer of wrong type!");
433 BaseCL::setArgOperand(ARG_SOURCE, Ptr);
434 }
435
436 void setSourceAlignment(unsigned Align) {
437 BaseCL::removeParamAttr(ARG_SOURCE, Attribute::Alignment);
438 if (Align > 0)
439 BaseCL::addParamAttr(ARG_SOURCE, Attribute::getWithAlignment(
440 BaseCL::getContext(), Align));
441 }
442 };
