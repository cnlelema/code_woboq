9// This file defines the CallSite class, which is a handy wrapper for code that
10// wants to treat Call, Invoke and CallBr instructions in a generic way. When
11// in non-mutation context (e.g. an analysis) ImmutableCallSite should be used.
12// Finally, when some degree of customization is necessary between these two
13// extremes, CallSiteBase<> can be supplied with fine-tuned parameters.
14//
15// NOTE: These classes are supposed to have "value semantics". So they should be
16// passed by value, not by reference; they should not be "new"ed or "delete"d.
17// They are efficiently copyable, assignable and constructable, with cost
18// equivalent to copying a pointer (notice that they have only a single data
19// member). The internal representation carries a flag which indicates which of
20// the three variants is enclosed. This allows for cheaper checks when various
21// accessors of CallSite are employed.
22//
23//===----------------------------------------------------------------------===//

51template <typename FunTy = const Function, typename BBTy = const BasicBlock,
52 typename ValTy = const Value, typename UserTy = const User,
53 typename UseTy = const Use, typename InstrTy = const Instruction,
54 typename CallTy = const CallInst,
55 typename InvokeTy = const InvokeInst,
56 typename CallBrTy = const CallBrInst,
57 typename IterTy = User::const_op_iterator>
58class CallSiteBase {
59protected:
60 PointerIntPair<InstrTy *, 2, int> I;
61
62 CallSiteBase() = default;
63 CallSiteBase(CallTy *CI) : I(CI, 1) { assert(CI); }
64 CallSiteBase(InvokeTy *II) : I(II, 0) { assert(II); }
65 CallSiteBase(CallBrTy *CBI) : I(CBI, 2) { assert(CBI); }
66 explicit CallSiteBase(ValTy *II) { *this = get(II); }
67
68private:
69 /// This static method is like a constructor. It will create an appropriate
70 /// call site for a Call, Invoke or CallBr instruction, but it can also create
71 /// a null initialized CallSiteBase object for something which is NOT a call
72 /// site.
73 static CallSiteBase get(ValTy *V) {
74 if (InstrTy *II = dyn_cast<InstrTy>(V)) {
75 if (II->getOpcode() == Instruction::Call)
76 return CallSiteBase(static_cast<CallTy*>(II));
77 if (II->getOpcode() == Instruction::Invoke)
78 return CallSiteBase(static_cast<InvokeTy*>(II));
79 if (II->getOpcode() == Instruction::CallBr)
80 return CallSiteBase(static_cast<CallBrTy *>(II));
81 }
82 return CallSiteBase();
83 }
84
85public:
86 /// Return true if a CallInst is enclosed.
87 bool isCall() const { return I.getInt() == 1; }
88
89 /// Return true if a InvokeInst is enclosed. !I.getInt() may also signify a
90 /// NULL instruction pointer, so check that.
91 bool isInvoke() const { return getInstruction() && I.getInt() == 0; }
92
93 /// Return true if a CallBrInst is enclosed.
94 bool isCallBr() const { return I.getInt() == 2; }
95
96 InstrTy *getInstruction() const { return I.getPointer(); }
97 InstrTy *operator->() const { return I.getPointer(); }
98 explicit operator bool() const { return I.getPointer(); }

100 /// Get the basic block containing the call site.
101 BBTy* getParent() const { return getInstruction()->getParent(); }

103 /// Return the pointer to function that is being called.
104 ValTy *getCalledValue() const {
105 assert(getInstruction() && "Not a call, invoke or callbr instruction!");
106 return *getCallee();
107 }

109 /// Return the function being called if this is a direct call, otherwise
110 /// return null (if it's an indirect call).
111 FunTy *getCalledFunction() const {
112 return dyn_cast<FunTy>(getCalledValue());
113 }
114
115 /// Return true if the callsite is an indirect call.
116 bool isIndirectCall() const {
117 const Value *V = getCalledValue();
118 if (!V)
119 return false;
120 if (isa<FunTy>(V) || isa<Constant>(V))
121 return false;
122 if (const CallBase *CB = dyn_cast<CallBase>(getInstruction()))
123 if (CB->isInlineAsm())
124 return false;
125 return true;
126 }

128 /// Set the callee to the specified value. Unlike the function of the same
129 /// name on CallBase, does not modify the type!
130 void setCalledFunction(Value *V) {
131 assert(getInstruction() && "Not a call, callbr, or invoke instruction!");
132 assert(cast<PointerType>(V->getType())->getElementType() ==
133 cast<CallBase>(getInstruction())->getFunctionType() &&
134 "New callee type does not match FunctionType on call");
135 *getCallee() = V;
136 }

246 /// data_operands_begin/data_operands_end - Return iterators iterating over
247 /// the call / invoke / callbr argument list and bundle operands. For invokes,
248 /// this is the set of instruction operands except the invoke target and the
249 /// two successor blocks; for calls this is the set of instruction operands
250 /// except the call target; for callbrs the number of labels to skip must be
251 /// determined first.
252
253 IterTy data_operands_begin() const {
254 assert(getInstruction() && "Not a call or invoke instruction!");
255 return cast<CallBase>(getInstruction())->data_operands_begin();
256 }
257 IterTy data_operands_end() const {
258 assert(getInstruction() && "Not a call or invoke instruction!");
259 return cast<CallBase>(getInstruction())->data_operands_end();
260 }
261 iterator_range<IterTy> data_ops() const {
262 return make_range(data_operands_begin(), data_operands_end());
263 }
264 bool data_operands_empty() const {
265 return data_operands_end() == data_operands_begin();
266 }
267 unsigned data_operands_size() const {
268 return std::distance(data_operands_begin(), data_operands_end());
269 }

288#define CALLSITE_DELEGATE_GETTER(METHOD) \
289 InstrTy *II = getInstruction(); \
290 return isCall() ? cast<CallInst>(II)->METHOD \
291 : isCallBr() ? cast<CallBrInst>(II)->METHOD \
292 : cast<InvokeInst>(II)->METHOD
293
294#define CALLSITE_DELEGATE_SETTER(METHOD) \
295 InstrTy *II = getInstruction(); \
296 if (isCall()) \
297 cast<CallInst>(II)->METHOD; \
298 else if (isCallBr()) \
299 cast<CallBrInst>(II)->METHOD; \
300 else \
301 cast<InvokeInst>(II)->METHOD

650 /// Return true if the return value is known to be not null.
651 /// This may be because it has the nonnull attribute, or because at least
652 /// one byte is dereferenceable and the pointer is in addrspace(0).
653 bool isReturnNonNull() const {
654 if (hasRetAttr(Attribute::NonNull))
655 return true;
656 else if (getDereferenceableBytes(AttributeList::ReturnIndex) > 0 &&
657 !NullPointerIsDefined(getCaller(),
658 getType()->getPointerAddressSpace()))
659 return true;
660
661 return false;
662 }

664 /// Returns true if this CallSite passes the given Value* as an argument to
665 /// the called function.
666 bool hasArgument(const Value *Arg) const {
667 for (arg_iterator AI = this->arg_begin(), E = this->arg_end(); AI != E;
668 ++AI)
669 if (AI->get() == Arg)
670 return true;
671 return false;
672 }

675 IterTy getCallee() const {

704/// AbstractCallSite
705///
706/// An abstract call site is a wrapper that allows to treat direct,
707/// indirect, and callback calls the same. If an abstract call site
708/// represents a direct or indirect call site it behaves like a stripped
709/// down version of a normal call site object. The abstract call site can
710/// also represent a callback call, thus the fact that the initially
711/// called function (=broker) may invoke a third one (=callback callee).
712/// In this case, the abstract call site hides the middle man, hence the
713/// broker function. The result is a representation of the callback call,
714/// inside the broker, but in the context of the original call to the broker.
715///
716/// There are up to three functions involved when we talk about callback call
717/// sites. The caller (1), which invokes the broker function. The broker
718/// function (2), that will invoke the callee zero or more times. And finally
719/// the callee (3), which is the target of the callback call.
720///
721/// The abstract call site will handle the mapping from parameters to arguments
722/// depending on the semantic of the broker function. However, it is important
723/// to note that the mapping is often partial. Thus, some arguments of the
724/// call/invoke instruction are mapped to parameters of the callee while others
725/// are not.
726class AbstractCallSite {
727public:
728
729 /// The encoding of a callback with regards to the underlying instruction.
730 struct CallbackInfo {
731
732 /// For direct/indirect calls the parameter encoding is empty. If it is not,
733 /// the abstract call site represents a callback. In that case, the first
734 /// element of the encoding vector represents which argument of the call
735 /// site CS is the callback callee. The remaining elements map parameters
736 /// (identified by their position) to the arguments that will be passed
737 /// through (also identified by position but in the call site instruction).
738 ///
739 /// NOTE that we use LLVM argument numbers (starting at 0) and not
740 /// clang/source argument numbers (starting at 1). The -1 entries represent
741 /// unknown values that are passed to the callee.
742 using ParameterEncodingTy = SmallVector<int, 0>;
743 ParameterEncodingTy ParameterEncoding;
744
745 };
746
747private:
748
749 /// The underlying call site:
750 /// caller -> callee, if this is a direct or indirect call site
751 /// caller -> broker function, if this is a callback call site
752 CallSite CS;
753
754 /// The encoding of a callback with regards to the underlying instruction.
755 CallbackInfo CI;
756
757public:
758 /// Sole constructor for abstract call sites (ACS).
759 ///
760 /// An abstract call site can only be constructed through a llvm::Use because
761 /// each operand (=use) of an instruction could potentially be a different
762 /// abstract call site. Furthermore, even if the value of the llvm::Use is the
763 /// same, and the user is as well, the abstract call sites might not be.
764 ///
765 /// If a use is not associated with an abstract call site the constructed ACS
766 /// will evaluate to false if converted to a boolean.
767 ///
768 /// If the use is the callee use of a call or invoke instruction, the
769 /// constructed abstract call site will behave as a llvm::CallSite would.
770 ///
771 /// If the use is not a callee use of a call or invoke instruction, the
772 /// callback metadata is used to determine the argument <-> parameter mapping
773 /// as well as the callee of the abstract call site.
774 AbstractCallSite(const Use *U);
775
776 /// Conversion operator to conveniently check for a valid/initialized ACS.
777 explicit operator bool() const { return (bool)CS; }
778
779 /// Return the underlying instruction.
780 Instruction *getInstruction() const { return CS.getInstruction(); }
781
782 /// Return the call site abstraction for the underlying instruction.
783 CallSite getCallSite() const { return CS; }
784
785 /// Return true if this ACS represents a direct call.
786 bool isDirectCall() const {
787 return !isCallbackCall() && !CS.isIndirectCall();
788 }
789
790 /// Return true if this ACS represents an indirect call.
791 bool isIndirectCall() const {
792 return !isCallbackCall() && CS.isIndirectCall();
793 }
794
795 /// Return true if this ACS represents a callback call.
796 bool isCallbackCall() const {
797 // For a callback call site the callee is ALWAYS stored first in the
798 // transitive values vector. Thus, a non-empty vector indicates a callback.
799 return !CI.ParameterEncoding.empty();
800 }
801
802 /// Return true if @p UI is the use that defines the callee of this ACS.
803 bool isCallee(Value::const_user_iterator UI) const {
804 return isCallee(&UI.getUse());
805 }
806
807 /// Return true if @p U is the use that defines the callee of this ACS.
808 bool isCallee(const Use *U) const {
809 if (isDirectCall())
810 return CS.isCallee(U);
811
812 assert(!CI.ParameterEncoding.empty() &&
813 "Callback without parameter encoding!");
814
815 return (int)CS.getArgumentNo(U) == CI.ParameterEncoding[0];
816 }
817
818 /// Return the number of parameters of the callee.
819 unsigned getNumArgOperands() const {
820 if (isDirectCall())
821 return CS.getNumArgOperands();
822 // Subtract 1 for the callee encoding.
823 return CI.ParameterEncoding.size() - 1;
824 }
825
826 /// Return the operand index of the underlying instruction associated with @p
827 /// Arg.
828 int getCallArgOperandNo(Argument &Arg) const {
829 return getCallArgOperandNo(Arg.getArgNo());
830 }
831
832 /// Return the operand index of the underlying instruction associated with
833 /// the function parameter number @p ArgNo or -1 if there is none.
834 int getCallArgOperandNo(unsigned ArgNo) const {
835 if (isDirectCall())
836 return ArgNo;
837 // Add 1 for the callee encoding.
838 return CI.ParameterEncoding[ArgNo + 1];
839 }
840
841 /// Return the operand of the underlying instruction associated with @p Arg.
842 Value *getCallArgOperand(Argument &Arg) const {
843 return getCallArgOperand(Arg.getArgNo());
844 }
845
846 /// Return the operand of the underlying instruction associated with the
847 /// function parameter number @p ArgNo or nullptr if there is none.
848 Value *getCallArgOperand(unsigned ArgNo) const {
849 if (isDirectCall())
850 return CS.getArgOperand(ArgNo);
851 // Add 1 for the callee encoding.
852 return CI.ParameterEncoding[ArgNo + 1] >= 0
853 ? CS.getArgOperand(CI.ParameterEncoding[ArgNo + 1])
854 : nullptr;
855 }
856
857 /// Return the operand index of the underlying instruction associated with the
858 /// callee of this ACS. Only valid for callback calls!
859 int getCallArgOperandNoForCallee() const {
860 assert(isCallbackCall());
861 assert(CI.ParameterEncoding.size() && CI.ParameterEncoding[0] > 0);
862 return CI.ParameterEncoding[0];
863 }
864
865 /// Return the pointer to function that is being called.
866 Value *getCalledValue() const {
867 if (isDirectCall())
868 return CS.getCalledValue();
869 return CS.getArgOperand(getCallArgOperandNoForCallee());
870 }
871
872 /// Return the function being called if this is a direct call, otherwise
873 /// return null (if it's an indirect call).
874 Function *getCalledFunction() const {
875 Value *V = getCalledValue();
876 return V ? dyn_cast<Function>(V->stripPointerCasts()) : nullptr;
877 }
878};
