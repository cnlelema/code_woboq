57class UnaryInstruction : public Instruction {
58protected:
59 UnaryInstruction(Type *Ty, unsigned iType, Value *V,
60 Instruction *IB = nullptr)
61 : Instruction(Ty, iType, &Op<0>(), 1, IB) {
62 Op<0>() = V;
63 }
64 UnaryInstruction(Type *Ty, unsigned iType, Value *V, BasicBlock *IAE)
65 : Instruction(Ty, iType, &Op<0>(), 1, IAE) {
66 Op<0>() = V;
67 }
68
69public:
70 // allocate space for exactly one operand
71 void *operator new(size_t s) {
72 return User::operator new(s, 1);
73 }
74
75 /// Transparently provide more efficient getOperand methods.
76 DECLARE_TRANSPARENT_OPERAND_ACCESSORS(Value);
77
78 // Methods for support type inquiry through isa, cast, and dyn_cast:
79 static bool classof(const Instruction *I) {
80 return I->isUnaryOp() ||
81 I->getOpcode() == Instruction::Alloca ||
82 I->getOpcode() == Instruction::Load ||
83 I->getOpcode() == Instruction::VAArg ||
84 I->getOpcode() == Instruction::ExtractValue ||
85 (I->getOpcode() >= CastOpsBegin && I->getOpcode() < CastOpsEnd);
86 }
87 static bool classof(const Value *V) {
88 return isa<Instruction>(V) && classof(cast<Instruction>(V));
89 }
90};

93struct OperandTraits<UnaryInstruction> :
94 public FixedNumOperandTraits<UnaryInstruction, 1> {
95};

1486 /// A.
1487 ///
1488 /// Data operands include call arguments and values used in operand bundles,
1489 /// but does not include the callee operand. This routine dispatches to the
1490 /// underlying AttributeList or the OperandBundleUser as appropriate.
1491 ///
1492 /// The index \p i is interpreted as
1493 ///
1494 /// \p i == Attribute::ReturnIndex -> the return value
1495 /// \p i in [1, arg_size + 1) -> argument number (\p i - 1)
1496 /// \p i in [arg_size + 1, data_operand_size + 1) -> bundle operand at index
1497 /// (\p i - 1) in the operand list.

1501 "Data operand index out of bounds!");

1504 // question is a call argument; or be indirectly implied by the kind of its
1505 // containing operand bundle, if the operand is a bundle operand.

1509
1510 // FIXME: Avoid these i - 1 calculations and update the API to use
1511 // zero-based indices.

1986 /// Return the start of the list of BundleOpInfo instances associated
1987 /// with this OperandBundleUser.
1988 ///
1989 /// OperandBundleUser uses the descriptor area co-allocated with the host User
1990 /// to store some meta information about which operands are "normal" operands,
1991 /// and which ones belong to some operand bundle.
1992 ///
1993 /// The layout of an operand bundle user is
1994 ///
1995 /// +-----------uint32_t End-------------------------------------+
1996 /// | |
1997 /// | +--------uint32_t Begin--------------------+ |
1998 /// | | | |
1999 /// ^ ^ v v
2000 /// |------|------|----|----|----|----|----|---------|----|---------|----|-----
2001 /// | BOI0 | BOI1 | .. | DU | U0 | U1 | .. | BOI0_U0 | .. | BOI1_U0 | .. | Un
2002 /// |------|------|----|----|----|----|----|---------|----|---------|----|-----
2003 /// v v ^ ^
2004 /// | | | |
2005 /// | +--------uint32_t Begin------------+ |
2006 /// | |
2007 /// +-----------uint32_t End-----------------------------+
2008 ///
2009 ///
2010 /// BOI0, BOI1 ... are descriptions of operand bundles in this User's use
2011 /// list. These descriptions are installed and managed by this class, and
2012 /// they're all instances of OperandBundleUser<T>::BundleOpInfo.
2013 ///
2014 /// DU is an additional descriptor installed by User's 'operator new' to keep
2015 /// track of the 'BOI0 ... BOIN' co-allocation. OperandBundleUser does not
2016 /// access or modify DU in any way, it's an implementation detail private to
2017 /// User.
2018 ///
2019 /// The regular Use& vector for the User starts at U0. The operand bundle
2020 /// uses are part of the Use& vector, just like normal uses. In the diagram
2021 /// above, the operand bundle uses start at BOI0_U0. Each instance of
2022 /// BundleOpInfo has information about a contiguous set of uses constituting
2023 /// an operand bundle, and the total set of operand bundle uses themselves
2024 /// form a contiguous set of uses (i.e. there are no gaps between uses
2025 /// corresponding to individual operand bundles).
2026 ///
2027 /// This class does not know the location of the set of operand bundle uses
2028 /// within the use list -- that is decided by the User using this class via
2029 /// the BeginIdx argument in populateBundleOperandInfos.
2030 ///
2031 /// Currently operand bundle users with hung-off operands are not supported.
2032 bundle_op_iterator bundle_op_info_begin() {
2033 if (!hasDescriptor())
2034 return nullptr;
2035
2036 uint8_t *BytesBegin = getDescriptor().begin();
2037 return reinterpret_cast<bundle_op_iterator>(BytesBegin);
2038 }
2039
2040 /// Return the start of the list of BundleOpInfo instances associated
2041 /// with this OperandBundleUser.
2042 const_bundle_op_iterator bundle_op_info_begin() const {
2043 auto *NonConstThis = const_cast<CallBase *>(this);
2044 return NonConstThis->bundle_op_info_begin();
2045 }
2046
2047 /// Return the end of the list of BundleOpInfo instances associated
2048 /// with this OperandBundleUser.
2049 bundle_op_iterator bundle_op_info_end() {
2050 if (!hasDescriptor())
2051 return nullptr;
2052
2053 uint8_t *BytesEnd = getDescriptor().end();
2054 return reinterpret_cast<bundle_op_iterator>(BytesEnd);
2055 }
2056
2057 /// Return the end of the list of BundleOpInfo instances associated
2058 /// with this OperandBundleUser.
2059 const_bundle_op_iterator bundle_op_info_end() const {
2060 auto *NonConstThis = const_cast<CallBase *>(this);
2061 return NonConstThis->bundle_op_info_end();
2062 }

2065 iterator_range<bundle_op_iterator> bundle_op_infos() {
2066 return make_range(bundle_op_info_begin(), bundle_op_info_end());

2072 }

2074 /// Populate the BundleOpInfo instances and the Use& vector from \p
2075 /// Bundles. Return the op_iterator pointing to the Use& one past the last
2076 /// last bundle operand use.
2077 ///
2078 /// Each \p OperandBundleDef instance is tracked by a OperandBundleInfo
2079 /// instance allocated in this User's descriptor.
2080 op_iterator populateBundleOperandInfos(ArrayRef<OperandBundleDef> Bundles,
2081 const unsigned BeginIndex);

2083 /// Return the BundleOpInfo for the operand at index OpIdx.
2084 ///
2085 /// It is an error to call this with an OpIdx that does not correspond to an
2086 /// bundle operand.
2087 const BundleOpInfo &getBundleOpInfoForOperand(unsigned OpIdx) const {
2088 for (auto &BOI : bundle_op_infos())
2089 if (BOI.Begin <= OpIdx && OpIdx < BOI.End)
2090 return BOI;
2091
2092 llvm_unreachable("Did not find operand bundle for operand!");
2093 }
