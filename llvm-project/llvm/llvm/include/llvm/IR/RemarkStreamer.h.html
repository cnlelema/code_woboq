13#ifndef LLVM_IR_REMARKSTREAMER_H
14#define LLVM_IR_REMARKSTREAMER_H
15
16#include "llvm/IR/DiagnosticInfo.h"
17#include "llvm/Remarks/RemarkSerializer.h"
18#include "llvm/Support/Error.h"
19#include "llvm/Support/Regex.h"
20#include "llvm/Support/raw_ostream.h"
21#include <string>
22#include <vector>
23
24namespace llvm {
25/// Streamer for remarks.
26class RemarkStreamer {
27 /// The filename that the remark diagnostics are emitted to.
28 const std::string Filename;
29 /// The regex used to filter remarks based on the passes that emit them.
30 Optional<Regex> PassFilter;
31 /// The object used to serialize the remarks to a specific format.
32 std::unique_ptr<remarks::Serializer> Serializer;
33
34 /// Temporary buffer for converting diagnostics into remark objects. This is
35 /// used for the remark arguments that are converted from a vector of
36 /// diagnostic arguments to a vector of remark arguments.
37 SmallVector<remarks::Argument, 8> TmpArgs;
38 /// Convert diagnostics into remark objects. The result uses \p TmpArgs as a
39 /// temporary buffer for the remark arguments, and relies on all the strings
40 /// to be kept in memory until the next call to `toRemark`.
41 /// The lifetime of the members of the result is bound to the lifetime of both
42 /// the remark streamer and the LLVM diagnostics.
43 remarks::Remark toRemark(const DiagnosticInfoOptimizationBase &Diag);
44
45public:
46 RemarkStreamer(StringRef Filename,
47 std::unique_ptr<remarks::Serializer> Serializer);
48 /// Return the filename that the remark diagnostics are emitted to.
49 StringRef getFilename() const { return Filename; }
50 /// Return stream that the remark diagnostics are emitted to.
51 raw_ostream &getStream() { return Serializer->OS; }
52 /// Return the serializer used for this stream.
53 remarks::Serializer &getSerializer() { return *Serializer; }
54 /// Set a pass filter based on a regex \p Filter.
55 /// Returns an error if the regex is invalid.
56 Error setFilter(StringRef Filter);
57 /// Emit a diagnostic through the streamer.
58 void emit(const DiagnosticInfoOptimizationBase &Diag);
59};
60} // end namespace llvm
61
62#endif // LLVM_IR_REMARKSTREAMER_H
