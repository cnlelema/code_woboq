8/// \file / This file provides routines used by LLVM's value numbering passes to
9/// perform various forms of value extraction from memory when the types are not
10/// identical. For example, given
11///
12/// store i32 8, i32 *%foo
13/// %a = bitcast i32 *%foo to i16
14/// %val = load i16, i16 *%a
15///
16/// It possible to extract the value of the load of %a from the store to %foo.
17/// These routines know how to tell whether they can do that (the analyze*
18/// routines), and can also insert the necessary IR to do it (the get*
19/// routines).
20
21#ifndef LLVM_TRANSFORMS_UTILS_VNCOERCION_H
22#define LLVM_TRANSFORMS_UTILS_VNCOERCION_H
23#include "llvm/IR/IRBuilder.h"
24
25namespace llvm {
26class Function;
27class StoreInst;
28class LoadInst;
29class MemIntrinsic;
30class Instruction;
31class Value;
32class Type;
33class DataLayout;
34class TargetLibraryInfo;
35class GVN;
36
37namespace VNCoercion {
38/// Return true if CoerceAvailableValueToLoadType would succeed if it was
39/// called.
40bool canCoerceMustAliasedValueToLoad(Value *StoredVal, Type *LoadTy,
41 const DataLayout &DL);
42
43/// If we saw a store of a value to memory, and then a load from a must-aliased
44/// pointer of a different type, try to coerce the stored value to the loaded
45/// type. LoadedTy is the type of the load we want to replace. IRB is
46/// IRBuilder used to insert new instructions.
47///
48/// If we can't do it, return null.
49Value *coerceAvailableValueToLoadType(Value *StoredVal, Type *LoadedTy,
50 IRBuilder<> &IRB, const DataLayout &DL);
51
52/// This function determines whether a value for the pointer LoadPtr can be
53/// extracted from the store at DepSI.
54///
55/// On success, it returns the offset into DepSI that extraction would start.
56/// On failure, it returns -1.
57int analyzeLoadFromClobberingStore(Type *LoadTy, Value *LoadPtr,
58 StoreInst *DepSI, const DataLayout &DL,
59 const TargetLibraryInfo *TLI);
60
61/// This function determines whether a value for the pointer LoadPtr can be
62/// extracted from the load at DepLI.
63///
64/// On success, it returns the offset into DepLI that extraction would start.
65/// On failure, it returns -1.
66int analyzeLoadFromClobberingLoad(Type *LoadTy, Value *LoadPtr, LoadInst *DepLI,
67 const DataLayout &DL,
68 const TargetLibraryInfo *TLI);
69
70/// This function determines whether a value for the pointer LoadPtr can be
71/// extracted from the memory intrinsic at DepMI.
72///
73/// On success, it returns the offset into DepMI that extraction would start.
74/// On failure, it returns -1.
75int analyzeLoadFromClobberingMemInst(Type *LoadTy, Value *LoadPtr,
76 MemIntrinsic *DepMI, const DataLayout &DL,
77 const TargetLibraryInfo *TLI);
78
79/// If analyzeLoadFromClobberingStore returned an offset, this function can be
80/// used to actually perform the extraction of the bits from the store. It
81/// inserts instructions to do so at InsertPt, and returns the extracted value.
82Value *getStoreValueForLoad(Value *SrcVal, unsigned Offset, Type *LoadTy,
83 Instruction *InsertPt, const DataLayout &DL);
84// This is the same as getStoreValueForLoad, except it performs no insertion
85// It only allows constant inputs.
86Constant *getConstantStoreValueForLoad(Constant *SrcVal, unsigned Offset,
87 Type *LoadTy, const DataLayout &DL);
88
89/// If analyzeLoadFromClobberingLoad returned an offset, this function can be
90/// used to actually perform the extraction of the bits from the load, including
91/// any necessary load widening. It inserts instructions to do so at InsertPt,
92/// and returns the extracted value.
93Value *getLoadValueForLoad(LoadInst *SrcVal, unsigned Offset, Type *LoadTy,
94 Instruction *InsertPt, const DataLayout &DL,
95 GVN &gvn);
96// This is the same as getLoadValueForLoad, except it is given the load value as
97// a constant. It returns nullptr if it would require widening the load.
98Constant *getConstantLoadValueForLoad(Constant *SrcVal, unsigned Offset,
99 Type *LoadTy, const DataLayout &DL);
100
101/// If analyzeLoadFromClobberingMemInst returned an offset, this function can be
102/// used to actually perform the extraction of the bits from the memory
103/// intrinsic. It inserts instructions to do so at InsertPt, and returns the
104/// extracted value.
105Value *getMemInstValueForLoad(MemIntrinsic *SrcInst, unsigned Offset,
106 Type *LoadTy, Instruction *InsertPt,
107 const DataLayout &DL);
108// This is the same as getStoreValueForLoad, except it performs no insertion.
109// It returns nullptr if it cannot produce a constant.
110Constant *getConstantMemInstValueForLoad(MemIntrinsic *SrcInst, unsigned Offset,
111 Type *LoadTy, const DataLayout &DL);
112}
113}
114#endif
