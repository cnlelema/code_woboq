56/// Ensure that all exit blocks of the loop are dedicated exits.
57///
58/// For any loop exit block with non-loop predecessors, we split the loop
59/// predecessors to use a dedicated loop exit block. We update the dominator
60/// tree and loop info if provided, and will preserve LCSSA if requested.
61bool formDedicatedExitBlocks(Loop *L, DominatorTree *DT, LoopInfo *LI,
62 MemorySSAUpdater *MSSAU, bool PreserveLCSSA);

64/// Ensures LCSSA form for every instruction from the Worklist in the scope of
65/// innermost containing loop.
66///
67/// For the given instruction which have uses outside of the loop, an LCSSA PHI
68/// node is inserted and the uses outside the loop are rewritten to use this
69/// node.
70///
71/// LoopInfo and DominatorTree are required and, since the routine makes no
72/// changes to CFG, preserved.
73///
74/// Returns true if any modifications are made.
75bool formLCSSAForInstructions(SmallVectorImpl<Instruction *> &Worklist,
76 DominatorTree &DT, LoopInfo &LI);
77
78/// Put loop into LCSSA form.
79///
80/// Looks at all instructions in the loop which have uses outside of the
81/// current loop. For each, an LCSSA PHI node is inserted and the uses outside
82/// the loop are rewritten to use this node. Sub-loops must be in LCSSA form
83/// already.
84///
85/// LoopInfo and DominatorTree are required and preserved.
86///
87/// If ScalarEvolution is passed in, it will be preserved.
88///
89/// Returns true if any modifications are made to the loop.
90bool formLCSSA(Loop &L, DominatorTree &DT, LoopInfo *LI, ScalarEvolution *SE);
91
92/// Put a loop nest into LCSSA form.
93///
94/// This recursively forms LCSSA for a loop nest.
95///
96/// LoopInfo and DominatorTree are required and preserved.
97///
98/// If ScalarEvolution is passed in, it will be preserved.
99///
100/// Returns true if any modifications are made to the loop.
101bool formLCSSARecursively(Loop &L, DominatorTree &DT, LoopInfo *LI,
102 ScalarEvolution *SE);

111/// Walk the specified region of the CFG (defined by all blocks
112/// dominated by the specified block, and that are in the current loop) in
113/// reverse depth first order w.r.t the DominatorTree. This allows us to visit
114/// uses before definitions, allowing us to sink a loop body in one pass without
115/// iteration. Takes DomTreeNode, AliasAnalysis, LoopInfo, DominatorTree,
116/// DataLayout, TargetLibraryInfo, Loop, AliasSet information for all
117/// instructions of the loop and loop safety information as
118/// arguments. Diagnostics is emitted via \p ORE. It returns changed status.
119bool sinkRegion(DomTreeNode *, AliasAnalysis *, LoopInfo *, DominatorTree *,
120 TargetLibraryInfo *, TargetTransformInfo *, Loop *,
121 AliasSetTracker *, MemorySSAUpdater *, ICFLoopSafetyInfo *,
122 SinkAndHoistLICMFlags &, OptimizationRemarkEmitter *);
123
124/// Walk the specified region of the CFG (defined by all blocks
125/// dominated by the specified block, and that are in the current loop) in depth
126/// first order w.r.t the DominatorTree. This allows us to visit definitions
127/// before uses, allowing us to hoist a loop body in one pass without iteration.
128/// Takes DomTreeNode, AliasAnalysis, LoopInfo, DominatorTree, DataLayout,
129/// TargetLibraryInfo, Loop, AliasSet information for all instructions of the
130/// loop and loop safety information as arguments. Diagnostics is emitted via \p
131/// ORE. It returns changed status.
132bool hoistRegion(DomTreeNode *, AliasAnalysis *, LoopInfo *, DominatorTree *,
133 TargetLibraryInfo *, Loop *, AliasSetTracker *,
134 MemorySSAUpdater *, ICFLoopSafetyInfo *,
135 SinkAndHoistLICMFlags &, OptimizationRemarkEmitter *);

151/// Try to promote memory values to scalars by sinking stores out of
152/// the loop and moving loads to before the loop. We do this by looping over
153/// the stores in the loop, looking for stores to Must pointers which are
154/// loop invariant. It takes a set of must-alias values, Loop exit blocks
155/// vector, loop exit blocks insertion point vector, PredIteratorCache,
156/// LoopInfo, DominatorTree, Loop, AliasSet information for all instructions
157/// of the loop and loop safety information as arguments.
158/// Diagnostics is emitted via \p ORE. It returns changed status.
159bool promoteLoopAccessesToScalars(
160 const SmallSetVector<Value *, 8> &, SmallVectorImpl<BasicBlock *> &,
161 SmallVectorImpl<Instruction *> &, SmallVectorImpl<MemoryAccess *> &,
162 PredIteratorCache &, LoopInfo *, DominatorTree *, const TargetLibraryInfo *,
163 Loop *, AliasSetTracker *, MemorySSAUpdater *, ICFLoopSafetyInfo *,
164 OptimizationRemarkEmitter *);

185/// Create a new loop identifier for a loop created from a loop transformation.
186///
187/// @param OrigLoopID The loop ID of the loop before the transformation.
188/// @param FollowupAttrs List of attribute names that contain attributes to be
189/// added to the new loop ID.
190/// @param InheritOptionsAttrsPrefix Selects which attributes should be inherited
191/// from the original loop. The following values
192/// are considered:
193/// nullptr : Inherit all attributes from @p OrigLoopID.
194/// "" : Do not inherit any attribute from @p OrigLoopID; only use
195/// those specified by a followup attribute.
196/// "<prefix>": Inherit all attributes except those which start with
197/// <prefix>; commonly used to remove metadata for the
198/// applied transformation.
199/// @param AlwaysNew If true, do not try to reuse OrigLoopID and never return
200/// None.
201///
202/// @return The loop ID for the after-transformation loop. The following values
203/// can be returned:
204/// None : No followup attribute was found; it is up to the
205/// transformation to choose attributes that make sense.
206/// @p OrigLoopID: The original identifier can be reused.
207/// nullptr : The new loop has no attributes.
208/// MDNode* : A new unique loop identifier.
209Optional<MDNode *>
210makeFollowupLoopID(MDNode *OrigLoopID, ArrayRef<StringRef> FollowupAttrs,
211 const char *InheritOptionsAttrsPrefix = "",
212 bool AlwaysNew = false);

217/// The mode sets how eager a transformation should be applied.
218enum TransformationMode {
219 /// The pass can use heuristics to determine whether a transformation should
220 /// be applied.
221 TM_Unspecified,
222
223 /// The transformation should be applied without considering a cost model.
224 TM_Enable,
225
226 /// The transformation should not be applied.
227 TM_Disable,
228
229 /// Force is a flag and should not be used alone.
230 TM_Force = 0x04,
231
232 /// The transformation was directed by the user, e.g. by a #pragma in
233 /// the source code. If the transformation could not be applied, a
234 /// warning should be emitted.
235 TM_ForcedByUser = TM_Enable | TM_Force,
236
237 /// The transformation must not be applied. For instance, `#pragma clang loop
238 /// unroll(disable)` explicitly forbids any unrolling to take place. Unlike
239 /// general loop metadata, it must not be dropped. Most passes should not
240 /// behave differently under TM_Disable and TM_SuppressedByUser.
241 TM_SuppressedByUser = TM_Disable | TM_Force
242};

274/// Returns true if is legal to hoist or sink this instruction disregarding the
275/// possible introduction of faults. Reasoning about potential faulting
276/// instructions is the responsibility of the caller since it is challenging to
277/// do efficiently from within this routine.
278/// \p TargetExecutesOncePerLoop is true only when it is guaranteed that the
279/// target executes at most once per execution of the loop body. This is used
280/// to assess the legality of duplicating atomic loads. Generally, this is
281/// true when moving out of loop and not true when moving into loops.
282/// If \p ORE is set use it to emit optimization remarks.
283bool canSinkOrHoistInst(Instruction &I, AAResults *AA, DominatorTree *DT,
284 Loop *CurLoop, AliasSetTracker *CurAST,
285 MemorySSAUpdater *MSSAU, bool TargetExecutesOncePerLoop,
286 SinkAndHoistLICMFlags *LICMFlags = nullptr,
