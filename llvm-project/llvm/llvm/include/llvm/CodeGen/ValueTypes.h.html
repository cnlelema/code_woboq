27 class Type;

29 /// Extended Value Type. Capable of holding value types which are not native
30 /// for any processor (such as the i12345 type), as well as the types an MVT
31 /// can represent.
32 struct EVT {
33 //private:
34 MVT V = MVT::INVALID_SIMPLE_VALUE_TYPE;
35 Type *LLVMTy = nullptr;
36
37 public:
38 constexpr EVT() = default;
39 constexpr EVT(MVT::SimpleValueType SVT) : V(SVT) {}
40 constexpr EVT(MVT S) : V(S) {}
41
42 bool operator==(EVT VT) const {
43 return !(*this != VT);
44 }
45 bool operator!=(EVT VT) const {
46 if (V.SimpleTy != VT.V.SimpleTy)
47 return true;
48 if (V.SimpleTy == MVT::INVALID_SIMPLE_VALUE_TYPE)
49 return LLVMTy != VT.LLVMTy;
50 return false;
51 }
52
53 /// Returns the EVT that represents a floating-point type with the given
54 /// number of bits. There are two floating-point types with 128 bits - this
55 /// returns f128 rather than ppcf128.
56 static EVT getFloatingPointVT(unsigned BitWidth) {
57 return MVT::getFloatingPointVT(BitWidth);
58 }
59
60 /// Returns the EVT that represents an integer with the given number of
61 /// bits.
62 static EVT getIntegerVT(LLVMContext &Context, unsigned BitWidth) {
63 MVT M = MVT::getIntegerVT(BitWidth);
64 if (M.SimpleTy != MVT::INVALID_SIMPLE_VALUE_TYPE)
65 return M;
66 return getExtendedIntegerVT(Context, BitWidth);
67 }
68
69 /// Returns the EVT that represents a vector NumElements in length, where
70 /// each element is of type VT.
71 static EVT getVectorVT(LLVMContext &Context, EVT VT, unsigned NumElements,
72 bool IsScalable = false) {
73 MVT M = MVT::getVectorVT(VT.V, NumElements, IsScalable);
74 if (M.SimpleTy != MVT::INVALID_SIMPLE_VALUE_TYPE)
75 return M;
76
77 assert(!IsScalable && "We don't support extended scalable types yet");
78 return getExtendedVectorVT(Context, VT, NumElements);
79 }
80
81 /// Returns the EVT that represents a vector EC.Min elements in length,
82 /// where each element is of type VT.
83 static EVT getVectorVT(LLVMContext &Context, EVT VT, MVT::ElementCount EC) {
84 MVT M = MVT::getVectorVT(VT.V, EC);
85 if (M.SimpleTy != MVT::INVALID_SIMPLE_VALUE_TYPE)
86 return M;
87 assert (!EC.Scalable && "We don't support extended scalable types yet");
88 return getExtendedVectorVT(Context, VT, EC.Min);
89 }
90
91 /// Return a vector with the same number of elements as this vector, but
92 /// with the element type converted to an integer type with the same
93 /// bitwidth.
94 EVT changeVectorElementTypeToInteger() const {
95 if (!isSimple()) {
96 assert (!isScalableVector() &&
97 "We don't support extended scalable types yet");
98 return changeExtendedVectorElementTypeToInteger();
99 }
100 MVT EltTy = getSimpleVT().getVectorElementType();
101 unsigned BitWidth = EltTy.getSizeInBits();
102 MVT IntTy = MVT::getIntegerVT(BitWidth);
103 MVT VecTy = MVT::getVectorVT(IntTy, getVectorNumElements(),
104 isScalableVector());
105 assert(VecTy.SimpleTy != MVT::INVALID_SIMPLE_VALUE_TYPE &&
106 "Simple vector VT not representable by simple integer vector VT!");
107 return VecTy;
108 }
109
110 /// Return the type converted to an equivalently sized integer or vector
111 /// with integer element type. Similar to changeVectorElementTypeToInteger,
112 /// but also handles scalars.
113 EVT changeTypeToInteger() {
114 if (isVector())
115 return changeVectorElementTypeToInteger();
116
117 if (isSimple())
118 return MVT::getIntegerVT(getSizeInBits());
119
120 return changeExtendedTypeToInteger();
121 }
122
123 /// Test if the given EVT is simple (as opposed to being extended).
124 bool isSimple() const {
125 return V.SimpleTy != MVT::INVALID_SIMPLE_VALUE_TYPE;
126 }
127
128 /// Test if the given EVT is extended (as opposed to being simple).
129 bool isExtended() const {
130 return !isSimple();
131 }
132
133 /// Return true if this is a FP or a vector FP type.
134 bool isFloatingPoint() const {
135 return isSimple() ? V.isFloatingPoint() : isExtendedFloatingPoint();
136 }
137
138 /// Return true if this is an integer or a vector integer type.
139 bool isInteger() const {
140 return isSimple() ? V.isInteger() : isExtendedInteger();
141 }
142
143 /// Return true if this is a fat pointer type.
144 bool isFatPointer() const {
145 return isSimple() ? V.isFatPointer() : false;
146 }
147
148 /// Return true if this is an integer, but not a vector.
149 bool isScalarInteger() const {
150 return isSimple() ? V.isScalarInteger() : isExtendedScalarInteger();
151 }
152
153 /// Return true if this is a vector value type.
154 bool isVector() const {
155 return isSimple() ? V.isVector() : isExtendedVector();
156 }
157
158 /// Return true if this is a vector type where the runtime
159 /// length is machine dependent
160 bool isScalableVector() const {
161 // FIXME: We don't support extended scalable types yet, because the
162 // matching IR type doesn't exist. Once it has been added, this can
163 // be changed to call isExtendedScalableVector.
164 if (!isSimple())
165 return false;
166 return V.isScalableVector();
167 }
168
169 /// Return true if this is a 16-bit vector type.
170 bool is16BitVector() const {
171 return isSimple() ? V.is16BitVector() : isExtended16BitVector();
172 }
173
174 /// Return true if this is a 32-bit vector type.
175 bool is32BitVector() const {
176 return isSimple() ? V.is32BitVector() : isExtended32BitVector();
177 }
178
179 /// Return true if this is a 64-bit vector type.
180 bool is64BitVector() const {
181 return isSimple() ? V.is64BitVector() : isExtended64BitVector();
182 }
183
184 /// Return true if this is a 128-bit vector type.
185 bool is128BitVector() const {
186 return isSimple() ? V.is128BitVector() : isExtended128BitVector();
187 }
188
189 /// Return true if this is a 256-bit vector type.
190 bool is256BitVector() const {
191 return isSimple() ? V.is256BitVector() : isExtended256BitVector();
192 }
193
194 /// Return true if this is a 512-bit vector type.
195 bool is512BitVector() const {
196 return isSimple() ? V.is512BitVector() : isExtended512BitVector();
197 }
198
199 /// Return true if this is a 1024-bit vector type.
200 bool is1024BitVector() const {
201 return isSimple() ? V.is1024BitVector() : isExtended1024BitVector();
202 }
203
204 /// Return true if this is a 2048-bit vector type.
205 bool is2048BitVector() const {
206 return isSimple() ? V.is2048BitVector() : isExtended2048BitVector();
207 }
208
209 /// Return true if this is an overloaded type for TableGen.
210 bool isOverloaded() const {
211 return (V==MVT::iAny || V==MVT::fAny || V==MVT::vAny || V==MVT::iPTRAny);
212 }
213
214 /// Return true if the bit size is a multiple of 8.
215 bool isByteSized() const {
216 return (getSizeInBits() & 7) == 0;
217 }
218
219 /// Return true if the size is a power-of-two number of bytes.
220 bool isRound() const {
221 unsigned BitSize = getSizeInBits();
222 return BitSize >= 8 && !(BitSize & (BitSize - 1));
223 }
224
225 /// Return true if this has the same number of bits as VT.
226 bool bitsEq(EVT VT) const {
227 if (EVT::operator==(VT)) return true;
228 return getSizeInBits() == VT.getSizeInBits();
229 }
230
231 /// Return true if this has more bits than VT.
232 bool bitsGT(EVT VT) const {
233 if (EVT::operator==(VT)) return false;
234 return getSizeInBits() > VT.getSizeInBits();
235 }
236
237 /// Return true if this has no less bits than VT.
238 bool bitsGE(EVT VT) const {
239 if (EVT::operator==(VT)) return true;
240 return getSizeInBits() >= VT.getSizeInBits();
241 }
242
243 /// Return true if this has less bits than VT.
244 bool bitsLT(EVT VT) const {
245 if (EVT::operator==(VT)) return false;
246 return getSizeInBits() < VT.getSizeInBits();
247 }
248
249 /// Return true if this has no more bits than VT.
250 bool bitsLE(EVT VT) const {
251 if (EVT::operator==(VT)) return true;
252 return getSizeInBits() <= VT.getSizeInBits();
253 }
254
255 /// Return the SimpleValueType held in the specified simple EVT.
256 MVT getSimpleVT() const {
257 assert(isSimple() && "Expected a SimpleValueType!");
258 return V;
259 }
260
261 /// If this is a vector type, return the element type, otherwise return
262 /// this.
263 EVT getScalarType() const {
264 return isVector() ? getVectorElementType() : *this;
265 }
266
267 /// Given a vector type, return the type of each element.
268 EVT getVectorElementType() const {
269 assert(isVector() && "Invalid vector type!");
270 if (isSimple())
271 return V.getVectorElementType();
272 return getExtendedVectorElementType();
273 }
274
275 /// Given a vector type, return the number of elements it contains.
276 unsigned getVectorNumElements() const {
277 assert(isVector() && "Invalid vector type!");
278 if (isSimple())
279 return V.getVectorNumElements();
280 return getExtendedVectorNumElements();
281 }
282
283 // Given a (possibly scalable) vector type, return the ElementCount
284 MVT::ElementCount getVectorElementCount() const {
285 assert((isVector()) && "Invalid vector type!");
286 if (isSimple())
287 return V.getVectorElementCount();
288
289 assert(!isScalableVector() &&
290 "We don't support extended scalable types yet");
291 return {getExtendedVectorNumElements(), false};
292 }
293
294 /// Return the size of the specified value type in bits.
295 unsigned getSizeInBits() const {
296 if (isSimple())
297 return V.getSizeInBits();
298 return getExtendedSizeInBits();
299 }
300
301 unsigned getScalarSizeInBits() const {
302 return getScalarType().getSizeInBits();
303 }
304
305 /// Return the number of bytes overwritten by a store of the specified value
306 /// type.
307 unsigned getStoreSize() const {
308 return (getSizeInBits() + 7) / 8;
309 }
310
311 /// Return the number of bits overwritten by a store of the specified value
312 /// type.
313 unsigned getStoreSizeInBits() const {
314 return getStoreSize() * 8;
315 }
316
317 /// Rounds the bit-width of the given integer EVT up to the nearest power of
318 /// two (and at least to eight), and returns the integer EVT with that
319 /// number of bits.
320 EVT getRoundIntegerType(LLVMContext &Context) const {
321 assert(isInteger() && !isVector() && "Invalid integer type!");
322 unsigned BitWidth = getSizeInBits();
323 if (BitWidth <= 8)
324 return EVT(MVT::i8);
325 return getIntegerVT(Context, 1 << Log2_32_Ceil(BitWidth));
326 }
327
328 /// Finds the smallest simple value type that is greater than or equal to
329 /// half the width of this EVT. If no simple value type can be found, an
330 /// extended integer value type of half the size (rounded up) is returned.
331 EVT getHalfSizedIntegerVT(LLVMContext &Context) const {
332 assert(isInteger() && !isVector() && "Invalid integer type!");
333 unsigned EVTSize = getSizeInBits();
334 for (unsigned IntVT = MVT::FIRST_INTEGER_VALUETYPE;
335 IntVT <= MVT::LAST_INTEGER_VALUETYPE; ++IntVT) {
336 EVT HalfVT = EVT((MVT::SimpleValueType)IntVT);
337 if (HalfVT.getSizeInBits() * 2 >= EVTSize)
338 return HalfVT;
339 }
340 return getIntegerVT(Context, (EVTSize + 1) / 2);
341 }
342
343 /// Return a VT for an integer vector type with the size of the
344 /// elements doubled. The typed returned may be an extended type.
345 EVT widenIntegerVectorElementType(LLVMContext &Context) const {
346 EVT EltVT = getVectorElementType();
347 EltVT = EVT::getIntegerVT(Context, 2 * EltVT.getSizeInBits());
348 return EVT::getVectorVT(Context, EltVT, getVectorElementCount());
349 }
350
351 // Return a VT for a vector type with the same element type but
352 // half the number of elements. The type returned may be an
353 // extended type.
354 EVT getHalfNumVectorElementsVT(LLVMContext &Context) const {
355 EVT EltVT = getVectorElementType();
356 auto EltCnt = getVectorElementCount();
357 assert(!(EltCnt.Min & 1) && "Splitting vector, but not in half!");
358 return EVT::getVectorVT(Context, EltVT, EltCnt / 2);
359 }
360
361 /// Returns true if the given vector is a power of 2.
362 bool isPow2VectorType() const {
363 unsigned NElts = getVectorNumElements();
364 return !(NElts & (NElts - 1));
365 }
366
367 /// Widens the length of the given vector EVT up to the nearest power of 2
368 /// and returns that type.
369 EVT getPow2VectorType(LLVMContext &Context) const {
370 if (!isPow2VectorType()) {
371 unsigned NElts = getVectorNumElements();
372 unsigned Pow2NElts = 1 << Log2_32_Ceil(NElts);
373 return EVT::getVectorVT(Context, getVectorElementType(), Pow2NElts,
374 isScalableVector());
375 }
376 else {
377 return *this;
378 }
379 }

381 /// This function returns value type as a string, e.g. "i32".
382 std::string getEVTString() const;
383
384 /// This method returns an LLVM type corresponding to the specified EVT.
385 /// For integer types, this returns an unsigned type. Note that this will
386 /// abort for types that cannot be represented.
387 Type *getTypeForEVT(LLVMContext &Context) const;
388
389 /// Return the value type corresponding to the specified type.
390 /// This returns all pointers as iPTR. If HandleUnknown is true, unknown
391 /// types are returned as Other, otherwise they are invalid.
392 static EVT getEVT(Type *Ty, bool HandleUnknown = false);
393
394 intptr_t getRawBits() const {
395 if (isSimple())
396 return V.SimpleTy;
397 else
398 return (intptr_t)(LLVMTy);
399 }
400
401 /// A meaningless but well-behaved order, useful for constructing
402 /// containers.
403 struct compareRawBits {
404 bool operator()(EVT L, EVT R) const {
405 if (L.V.SimpleTy == R.V.SimpleTy)
406 return L.LLVMTy < R.LLVMTy;
407 else
408 return L.V.SimpleTy < R.V.SimpleTy;
409 }
410 };
411
412 private:
413 // Methods for handling the Extended-type case in functions above.
414 // These are all out-of-line to prevent users of this header file
415 // from having a dependency on Type.h.
416 EVT changeExtendedTypeToInteger() const;
417 EVT changeExtendedVectorElementTypeToInteger() const;
418 static EVT getExtendedIntegerVT(LLVMContext &C, unsigned BitWidth);
419 static EVT getExtendedVectorVT(LLVMContext &C, EVT VT,
420 unsigned NumElements);
421 bool isExtendedFloatingPoint() const LLVM_READONLY;
422 bool isExtendedInteger() const LLVM_READONLY;
423 bool isExtendedScalarInteger() const LLVM_READONLY;
424 bool isExtendedVector() const LLVM_READONLY;
425 bool isExtended16BitVector() const LLVM_READONLY;
426 bool isExtended32BitVector() const LLVM_READONLY;
427 bool isExtended64BitVector() const LLVM_READONLY;
428 bool isExtended128BitVector() const LLVM_READONLY;
429 bool isExtended256BitVector() const LLVM_READONLY;
430 bool isExtended512BitVector() const LLVM_READONLY;
431 bool isExtended1024BitVector() const LLVM_READONLY;
432 bool isExtended2048BitVector() const LLVM_READONLY;
433 EVT getExtendedVectorElementType() const;
434 unsigned getExtendedVectorNumElements() const LLVM_READONLY;
435 unsigned getExtendedSizeInBits() const LLVM_READONLY;
436 };
