23namespace llvm {

25class SelectionDAG;

27//===----------------------------------------------------------------------===//
28/// Targets can subclass this to parameterize the
29/// SelectionDAG lowering and instruction selection process.
30///
31class SelectionDAGTargetInfo {
32public:
33 explicit SelectionDAGTargetInfo() = default;
34 SelectionDAGTargetInfo(const SelectionDAGTargetInfo &) = delete;
35 SelectionDAGTargetInfo &operator=(const SelectionDAGTargetInfo &) = delete;
36 virtual ~SelectionDAGTargetInfo();
37
38 /// Emit target-specific code that performs a memcpy.
39 /// This can be used by targets to provide code sequences for cases
40 /// that don't fit the target's parameters for simple loads/stores and can be
41 /// more efficient than using a library call. This function can return a null
42 /// SDValue if the target declines to use custom code and a different
43 /// lowering strategy should be used.
44 ///
45 /// If AlwaysInline is true, the size is constant and the target should not
46 /// emit any calls and is strongly encouraged to attempt to emit inline code
47 /// even if it is beyond the usual threshold because this intrinsic is being
48 /// expanded in a place where calls are not feasible (e.g. within the prologue
49 /// for another call). If the target chooses to decline an AlwaysInline
50 /// request here, legalize will resort to using simple loads and stores.
51 virtual SDValue EmitTargetCodeForMemcpy(
52 SelectionDAG &DAG, const SDLoc &dl, SDValue Chain, SDValue Op1,
53 SDValue Op2, SDValue Op3, unsigned Align, bool isVolatile,
54 bool AlwaysInline, bool MustPreserveCheriCapabilities,
55 MachinePointerInfo DstPtrInfo, MachinePointerInfo SrcPtrInfo) const {
56 return SDValue();
57 }
58
59 /// Emit target-specific code that performs a memmove.
60 /// This can be used by targets to provide code sequences for cases
61 /// that don't fit the target's parameters for simple loads/stores and can be
62 /// more efficient than using a library call. This function can return a null
63 /// SDValue if the target declines to use custom code and a different
64 /// lowering strategy should be used.
65 virtual SDValue EmitTargetCodeForMemmove(
66 SelectionDAG &DAG, const SDLoc &dl, SDValue Chain, SDValue Op1,
67 SDValue Op2, SDValue Op3, unsigned Align, bool isVolatile,
68 bool MustPreserveCheriCapabilities, MachinePointerInfo DstPtrInfo,
69 MachinePointerInfo SrcPtrInfo) const {
70 return SDValue();
71 }
72
73 /// Emit target-specific code that performs a memset.
74 /// This can be used by targets to provide code sequences for cases
75 /// that don't fit the target's parameters for simple stores and can be more
76 /// efficient than using a library call. This function can return a null
77 /// SDValue if the target declines to use custom code and a different
78 /// lowering strategy should be used.
79 virtual SDValue EmitTargetCodeForMemset(SelectionDAG &DAG, const SDLoc &dl,
80 SDValue Chain, SDValue Op1,
81 SDValue Op2, SDValue Op3,
82 unsigned Align, bool isVolatile,
83 MachinePointerInfo DstPtrInfo) const {
84 return SDValue();
85 }
86
87 /// Emit target-specific code that performs a memcmp, in cases where that is
88 /// faster than a libcall. The first returned SDValue is the result of the
89 /// memcmp and the second is the chain. Both SDValues can be null if a normal
90 /// libcall should be used.
91 virtual std::pair<SDValue, SDValue>
92 EmitTargetCodeForMemcmp(SelectionDAG &DAG, const SDLoc &dl, SDValue Chain,
93 SDValue Op1, SDValue Op2, SDValue Op3,
94 MachinePointerInfo Op1PtrInfo,
95 MachinePointerInfo Op2PtrInfo) const {
96 return std::make_pair(SDValue(), SDValue());
97 }
98
99 /// Emit target-specific code that performs a memchr, in cases where that is
100 /// faster than a libcall. The first returned SDValue is the result of the
101 /// memchr and the second is the chain. Both SDValues can be null if a normal
102 /// libcall should be used.
103 virtual std::pair<SDValue, SDValue>
104 EmitTargetCodeForMemchr(SelectionDAG &DAG, const SDLoc &dl, SDValue Chain,
105 SDValue Src, SDValue Char, SDValue Length,
106 MachinePointerInfo SrcPtrInfo) const {
107 return std::make_pair(SDValue(), SDValue());
108 }
109
110 /// Emit target-specific code that performs a strcpy or stpcpy, in cases
111 /// where that is faster than a libcall.
112 /// The first returned SDValue is the result of the copy (the start
113 /// of the destination string for strcpy, a pointer to the null terminator
114 /// for stpcpy) and the second is the chain. Both SDValues can be null
115 /// if a normal libcall should be used.
116 virtual std::pair<SDValue, SDValue>
117 EmitTargetCodeForStrcpy(SelectionDAG &DAG, const SDLoc &DL, SDValue Chain,
118 SDValue Dest, SDValue Src,
119 MachinePointerInfo DestPtrInfo,
120 MachinePointerInfo SrcPtrInfo, bool isStpcpy) const {
121 return std::make_pair(SDValue(), SDValue());
122 }
123
124 /// Emit target-specific code that performs a strcmp, in cases where that is
125 /// faster than a libcall.
126 /// The first returned SDValue is the result of the strcmp and the second is
127 /// the chain. Both SDValues can be null if a normal libcall should be used.
128 virtual std::pair<SDValue, SDValue>
129 EmitTargetCodeForStrcmp(SelectionDAG &DAG, const SDLoc &dl, SDValue Chain,
130 SDValue Op1, SDValue Op2,
131 MachinePointerInfo Op1PtrInfo,
132 MachinePointerInfo Op2PtrInfo) const {
133 return std::make_pair(SDValue(), SDValue());
134 }
135
136 virtual std::pair<SDValue, SDValue>
137 EmitTargetCodeForStrlen(SelectionDAG &DAG, const SDLoc &DL, SDValue Chain,
138 SDValue Src, MachinePointerInfo SrcPtrInfo) const {
139 return std::make_pair(SDValue(), SDValue());
140 }
141
142 virtual std::pair<SDValue, SDValue>
143 EmitTargetCodeForStrnlen(SelectionDAG &DAG, const SDLoc &DL, SDValue Chain,
144 SDValue Src, SDValue MaxLength,
145 MachinePointerInfo SrcPtrInfo) const {
146 return std::make_pair(SDValue(), SDValue());
147 }
148
149 // Return true when the decision to generate FMA's (or FMS, FMLA etc) rather
150 // than FMUL and ADD is delegated to the machine combiner.
151 virtual bool generateFMAsInMachineCombiner(CodeGenOpt::Level OptLevel) const {
152 return false;
153 }
154};

156} // end namespace llvm
