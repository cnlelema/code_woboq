25enum CaseClusterKind {
26 /// A cluster of adjacent case labels with the same destination, or just one
27 /// case.
28 CC_Range,
29 /// A cluster of cases suitable for jump table lowering.
30 CC_JumpTable,
31 /// A cluster of cases suitable for bit test lowering.
32 CC_BitTests
33};
34
35/// A cluster of case labels.
36struct CaseCluster {
37 CaseClusterKind Kind;
38 const ConstantInt *Low, *High;
39 union {
40 MachineBasicBlock *MBB;
41 unsigned JTCasesIndex;
42 unsigned BTCasesIndex;
43 };
44 BranchProbability Prob;
45
46 static CaseCluster range(const ConstantInt *Low, const ConstantInt *High,
47 MachineBasicBlock *MBB, BranchProbability Prob) {
48 CaseCluster C;
49 C.Kind = CC_Range;
50 C.Low = Low;
51 C.High = High;
52 C.MBB = MBB;
53 C.Prob = Prob;
54 return C;
55 }
56
57 static CaseCluster jumpTable(const ConstantInt *Low, const ConstantInt *High,
58 unsigned JTCasesIndex, BranchProbability Prob) {
59 CaseCluster C;
60 C.Kind = CC_JumpTable;
61 C.Low = Low;
62 C.High = High;
63 C.JTCasesIndex = JTCasesIndex;
64 C.Prob = Prob;
65 return C;
66 }
67
68 static CaseCluster bitTests(const ConstantInt *Low, const ConstantInt *High,
69 unsigned BTCasesIndex, BranchProbability Prob) {
70 CaseCluster C;
71 C.Kind = CC_BitTests;
72 C.Low = Low;
73 C.High = High;
74 C.BTCasesIndex = BTCasesIndex;
75 C.Prob = Prob;
76 return C;
77 }
78};

100/// This structure is used to communicate between SelectionDAGBuilder and
101/// SDISel for the code generation of additional basic blocks needed by
102/// multi-case switch statements.
103struct CaseBlock {
104 // The condition code to use for the case block's setcc node.
105 // Besides the integer condition codes, this can also be SETTRUE, in which
106 // case no comparison gets emitted.
107 ISD::CondCode CC;
108
109 // The LHS/MHS/RHS of the comparison to emit.
110 // Emit by default LHS op RHS. MHS is used for range comparisons:
111 // If MHS is not null: (LHS <= MHS) and (MHS <= RHS).
112 const Value *CmpLHS, *CmpMHS, *CmpRHS;
113
114 // The block to branch to if the setcc is true/false.
115 MachineBasicBlock *TrueBB, *FalseBB;
116
117 // The block into which to emit the code for the setcc and branches.
118 MachineBasicBlock *ThisBB;
119
120 /// The debug location of the instruction this CaseBlock was
121 /// produced from.
122 SDLoc DL;
123
124 // Branch weights.
125 BranchProbability TrueProb, FalseProb;
126
127 CaseBlock(ISD::CondCode cc, const Value *cmplhs, const Value *cmprhs,
128 const Value *cmpmiddle, MachineBasicBlock *truebb,
129 MachineBasicBlock *falsebb, MachineBasicBlock *me, SDLoc dl,
130 BranchProbability trueprob = BranchProbability::getUnknown(),
131 BranchProbability falseprob = BranchProbability::getUnknown())
132 : CC(cc), CmpLHS(cmplhs), CmpMHS(cmpmiddle), CmpRHS(cmprhs),
133 TrueBB(truebb), FalseBB(falsebb), ThisBB(me), DL(dl),
134 TrueProb(trueprob), FalseProb(falseprob) {}
135};

137struct JumpTable {
138 /// The virtual register containing the index of the jump table entry
139 /// to jump to.
140 unsigned Reg;
141 /// The JumpTableIndex for this jump table in the function.
142 unsigned JTI;
143 /// The MBB into which to emit the code for the indirect jump.
144 MachineBasicBlock *MBB;
145 /// The MBB of the default bb, which is a successor of the range
146 /// check MBB. This is when updating PHI nodes in successors.
147 MachineBasicBlock *Default;
148
149 JumpTable(unsigned R, unsigned J, MachineBasicBlock *M, MachineBasicBlock *D)
150 : Reg(R), JTI(J), MBB(M), Default(D) {}
151};
152struct JumpTableHeader {
153 APInt First;
154 APInt Last;
155 const Value *SValue;
156 MachineBasicBlock *HeaderBB;
157 bool Emitted;
158 bool OmitRangeCheck;
159
160 JumpTableHeader(APInt F, APInt L, const Value *SV, MachineBasicBlock *H,
161 bool E = false)
162 : First(std::move(F)), Last(std::move(L)), SValue(SV), HeaderBB(H),
163 Emitted(E), OmitRangeCheck(false) {}
164};
165using JumpTableBlock = std::pair<JumpTableHeader, JumpTable>;
