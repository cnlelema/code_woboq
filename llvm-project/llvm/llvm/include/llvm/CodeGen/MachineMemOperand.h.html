33/// This class contains a discriminated union of information about pointers in
34/// memory operands, relating them back to LLVM IR or to virtual locations (such
35/// as frame indices) that are exposed during codegen.
36struct MachinePointerInfo {
37 /// This is the IR pointer value for the access, or it is null if unknown.
38 /// If this is null, then the access is to a pointer in the default address
39 /// space.
40 PointerUnion<const Value *, const PseudoSourceValue *> V;
41
42 /// Offset - This is an offset from the base Value*.
43 int64_t Offset;
44
45 uint8_t StackID;
46
47 unsigned AddrSpace = 0;
48
49 explicit MachinePointerInfo(const Value *v, int64_t offset = 0,
50 uint8_t ID = 0)
51 : V(v), Offset(offset), StackID(ID) {
52 AddrSpace = v ? v->getType()->getPointerAddressSpace() : 0;
53 }
54
55 explicit MachinePointerInfo(const PseudoSourceValue *v, int64_t offset = 0,
56 uint8_t ID = 0)
57 : V(v), Offset(offset), StackID(ID) {
58 AddrSpace = v ? v->getAddressSpace() : 0;
59 }
60
61 explicit MachinePointerInfo(unsigned AddressSpace = 0)
62 : V((const Value *)nullptr), Offset(0), StackID(0),
63 AddrSpace(AddressSpace) {}
64
65 explicit MachinePointerInfo(
66 PointerUnion<const Value *, const PseudoSourceValue *> v,
67 int64_t offset = 0,
68 uint8_t ID = 0)
69 : V(v), Offset(offset), StackID(ID) {
70 if (V) {
71 if (const auto *ValPtr = V.dyn_cast<const Value*>())
72 AddrSpace = ValPtr->getType()->getPointerAddressSpace();
73 else
74 AddrSpace = V.get<const PseudoSourceValue*>()->getAddressSpace();
75 }
76 }
77
78 MachinePointerInfo getWithOffset(int64_t O) const {
79 if (V.isNull())
80 return MachinePointerInfo(AddrSpace);
81 if (V.is<const Value*>())
82 return MachinePointerInfo(V.get<const Value*>(), Offset+O, StackID);
83 return MachinePointerInfo(V.get<const PseudoSourceValue*>(), Offset+O,
84 StackID);
85 }
86
87 /// Return true if memory region [V, V+Offset+Size) is known to be
88 /// dereferenceable.
89 bool isDereferenceable(unsigned Size, LLVMContext &C,
90 const DataLayout &DL) const;
91
92 /// Return the LLVM IR address space number that this pointer points into.
93 unsigned getAddrSpace() const;
94
95 /// Return a MachinePointerInfo record that refers to the constant pool.
96 static MachinePointerInfo getConstantPool(MachineFunction &MF);
97
98 /// Return a MachinePointerInfo record that refers to the specified
99 /// FrameIndex.
100 static MachinePointerInfo getFixedStack(MachineFunction &MF, int FI,
101 int64_t Offset = 0);
102
103 /// Return a MachinePointerInfo record that refers to a jump table entry.
104 static MachinePointerInfo getJumpTable(MachineFunction &MF);
105
106 /// Return a MachinePointerInfo record that refers to a GOT entry.
107 static MachinePointerInfo getGOT(MachineFunction &MF);
108
109 /// Return a MachinePointerInfo record that refers to a MCT entry.
110 static MachinePointerInfo getCapTable(MachineFunction &MF);
111
112 /// Stack pointer relative access.
113 static MachinePointerInfo getStack(MachineFunction &MF, int64_t Offset,
114 uint8_t ID = 0);
115
116 /// Stack memory without other information.
117 static MachinePointerInfo getUnknownStack(MachineFunction &MF);
118};

122/// A description of a memory reference used in the backend.
123/// Instead of holding a StoreInst or LoadInst, this class holds the address
124/// Value of the reference along with a byte size and offset. This allows it
125/// to describe lowered loads and stores. Also, the special PseudoSourceValue
126/// objects can be used to represent loads and stores to memory locations
127/// that aren't explicit in the regular LLVM IR.
128///
129class MachineMemOperand {
130public:
131 /// Flags values. These may be or'd together.
132 enum Flags : uint16_t {
133 // No flags set.
134 MONone = 0,
135 /// The memory access reads data.
136 MOLoad = 1u << 0,
137 /// The memory access writes data.
138 MOStore = 1u << 1,
139 /// The memory access is volatile.
140 MOVolatile = 1u << 2,
141 /// The memory access is non-temporal.
142 MONonTemporal = 1u << 3,
143 /// The memory access is dereferenceable (i.e., doesn't trap).
144 MODereferenceable = 1u << 4,
145 /// The memory access always returns the same value (or traps).
146 MOInvariant = 1u << 5,
147
148 // Reserved for use by target-specific passes.
149 // Targets may override getSerializableMachineMemOperandTargetFlags() to
150 // enable MIR serialization/parsing of these flags. If more of these flags
151 // are added, the MIR printing/parsing code will need to be updated as well.
152 MOTargetFlag1 = 1u << 6,
153 MOTargetFlag2 = 1u << 7,
154 MOTargetFlag3 = 1u << 8,
155
156 LLVM_MARK_AS_BITMASK_ENUM(/* LargestFlag = */ MOTargetFlag3)
157 };
158
159private:
160 /// Atomic information for this memory operation.
161 struct MachineAtomicInfo {
162 /// Synchronization scope ID for this memory operation.
163 unsigned SSID : 8; // SyncScope::ID
164 /// Atomic ordering requirements for this memory operation. For cmpxchg
165 /// atomic operations, atomic ordering requirements when store occurs.
166 unsigned Ordering : 4; // enum AtomicOrdering
167 /// For cmpxchg atomic operations, atomic ordering requirements when store
168 /// does not occur.
169 unsigned FailureOrdering : 4; // enum AtomicOrdering
170 };
171
172 MachinePointerInfo PtrInfo;
173 uint64_t Size;
174 Flags FlagVals;
175 uint16_t BaseAlignLog2; // log_2(base_alignment) + 1
176 MachineAtomicInfo AtomicInfo;
177 AAMDNodes AAInfo;
178 const MDNode *Ranges;
179
180public:
181 /// Construct a MachineMemOperand object with the specified PtrInfo, flags,
182 /// size, and base alignment. For atomic operations the synchronization scope
183 /// and atomic ordering requirements must also be specified. For cmpxchg
184 /// atomic operations the atomic ordering requirements when store does not
185 /// occur must also be specified.
186 MachineMemOperand(MachinePointerInfo PtrInfo, Flags flags, uint64_t s,
187 uint64_t a,
188 const AAMDNodes &AAInfo = AAMDNodes(),
189 const MDNode *Ranges = nullptr,
190 SyncScope::ID SSID = SyncScope::System,
191 AtomicOrdering Ordering = AtomicOrdering::NotAtomic,
192 AtomicOrdering FailureOrdering = AtomicOrdering::NotAtomic);
193
194 const MachinePointerInfo &getPointerInfo() const { return PtrInfo; }
195
196 /// Return the base address of the memory access. This may either be a normal
197 /// LLVM IR Value, or one of the special values used in CodeGen.
198 /// Special values are those obtained via
199 /// PseudoSourceValue::getFixedStack(int), PseudoSourceValue::getStack, and
200 /// other PseudoSourceValue member functions which return objects which stand
201 /// for frame/stack pointer relative references and other special references
202 /// which are not representable in the high-level IR.
203 const Value *getValue() const { return PtrInfo.V.dyn_cast<const Value*>(); }
204
205 const PseudoSourceValue *getPseudoValue() const {
206 return PtrInfo.V.dyn_cast<const PseudoSourceValue*>();
207 }
208
209 const void *getOpaqueValue() const { return PtrInfo.V.getOpaqueValue(); }
210
211 /// Return the raw flags of the source value, \see Flags.
212 Flags getFlags() const { return FlagVals; }
213
214 /// Bitwise OR the current flags with the given flags.
215 void setFlags(Flags f) { FlagVals |= f; }
216
217 /// For normal values, this is a byte offset added to the base address.
218 /// For PseudoSourceValue::FPRel values, this is the FrameIndex number.
219 int64_t getOffset() const { return PtrInfo.Offset; }
220
221 unsigned getAddrSpace() const { return PtrInfo.getAddrSpace(); }
222
223 /// Return the size in bytes of the memory reference.
224 uint64_t getSize() const { return Size; }
225
226 /// Return the size in bits of the memory reference.
227 uint64_t getSizeInBits() const { return Size * 8; }
228
229 /// Return the minimum known alignment in bytes of the actual memory
230 /// reference.
231 uint64_t getAlignment() const;
232
233 /// Return the minimum known alignment in bytes of the base address, without
234 /// the offset.
235 uint64_t getBaseAlignment() const { return (1u << BaseAlignLog2) >> 1; }
236
237 /// Return the AA tags for the memory reference.
238 AAMDNodes getAAInfo() const { return AAInfo; }
239
240 /// Return the range tag for the memory reference.
241 const MDNode *getRanges() const { return Ranges; }
242
243 /// Returns the synchronization scope ID for this memory operation.
244 SyncScope::ID getSyncScopeID() const {
245 return static_cast<SyncScope::ID>(AtomicInfo.SSID);
246 }
247
248 /// Return the atomic ordering requirements for this memory operation. For
249 /// cmpxchg atomic operations, return the atomic ordering requirements when
250 /// store occurs.
251 AtomicOrdering getOrdering() const {
252 return static_cast<AtomicOrdering>(AtomicInfo.Ordering);
253 }
254
255 /// For cmpxchg atomic operations, return the atomic ordering requirements
256 /// when store does not occur.
257 AtomicOrdering getFailureOrdering() const {
258 return static_cast<AtomicOrdering>(AtomicInfo.FailureOrdering);
259 }
260
261 bool isLoad() const { return FlagVals & MOLoad; }
262 bool isStore() const { return FlagVals & MOStore; }
263 bool isVolatile() const { return FlagVals & MOVolatile; }
264 bool isNonTemporal() const { return FlagVals & MONonTemporal; }
265 bool isDereferenceable() const { return FlagVals & MODereferenceable; }
266 bool isInvariant() const { return FlagVals & MOInvariant; }
267
268 /// Returns true if this operation has an atomic ordering requirement of
269 /// unordered or higher, false otherwise.
270 bool isAtomic() const { return getOrdering() != AtomicOrdering::NotAtomic; }
271
272 /// Returns true if this memory operation doesn't have any ordering
273 /// constraints other than normal aliasing. Volatile and (ordered) atomic
274 /// memory operations can't be reordered.
275 bool isUnordered() const {
276 return (getOrdering() == AtomicOrdering::NotAtomic ||
277 getOrdering() == AtomicOrdering::Unordered) &&
278 !isVolatile();
279 }
280
281 /// Update this MachineMemOperand to reflect the alignment of MMO, if it has a
282 /// greater alignment. This must only be used when the new alignment applies
283 /// to all users of this MachineMemOperand.
284 void refineAlignment(const MachineMemOperand *MMO);
285
286 /// Change the SourceValue for this MachineMemOperand. This should only be
287 /// used when an object is being relocated and all references to it are being
288 /// updated.
289 void setValue(const Value *NewSV) { PtrInfo.V = NewSV; }
290 void setValue(const PseudoSourceValue *NewSV) { PtrInfo.V = NewSV; }
291 void setOffset(int64_t NewOffset) { PtrInfo.Offset = NewOffset; }
292
293 /// Profile - Gather unique data for the object.
294 ///
295 void Profile(FoldingSetNodeID &ID) const;
296
297 /// Support for operator<<.
298 /// @{
299 void print(raw_ostream &OS) const;
300 void print(raw_ostream &OS, ModuleSlotTracker &MST) const;
301 void print(raw_ostream &OS, ModuleSlotTracker &MST,
302 SmallVectorImpl<StringRef> &SSNs, const LLVMContext &Context,
303 const MachineFrameInfo *MFI, const TargetInstrInfo *TII) const;
304 /// @}
305
306 friend bool operator==(const MachineMemOperand &LHS,
307 const MachineMemOperand &RHS) {
308 return LHS.getValue() == RHS.getValue() &&
309 LHS.getPseudoValue() == RHS.getPseudoValue() &&
310 LHS.getSize() == RHS.getSize() &&
311 LHS.getOffset() == RHS.getOffset() &&
312 LHS.getFlags() == RHS.getFlags() &&
313 LHS.getAAInfo() == RHS.getAAInfo() &&
314 LHS.getRanges() == RHS.getRanges() &&
315 LHS.getAlignment() == RHS.getAlignment() &&
316 LHS.getAddrSpace() == RHS.getAddrSpace();
317 }
318
319 friend bool operator!=(const MachineMemOperand &LHS,
320 const MachineMemOperand &RHS) {
321 return !(LHS == RHS);
322 }
323};
324
325inline raw_ostream &operator<<(raw_ostream &OS, const MachineMemOperand &MRO) {
326 MRO.print(OS);
327 return OS;
328}
