32 MULSUBW_OP2,
33 MULADDWI_OP1,
34 MULSUBWI_OP1,

36 MULADDX_OP2,
37 MULSUBX_OP1,
38 MULSUBX_OP2,
39 MULADDXI_OP1,
40 MULSUBXI_OP1,
41 // Floating Point
42 FMULADDS_OP1,
43 FMULADDS_OP2,
44 FMULSUBS_OP1,
45 FMULSUBS_OP2,
46 FMULADDD_OP1,
47 FMULADDD_OP2,
48 FMULSUBD_OP1,
49 FMULSUBD_OP2,
50 FNMULSUBS_OP1,
51 FNMULSUBD_OP1,
