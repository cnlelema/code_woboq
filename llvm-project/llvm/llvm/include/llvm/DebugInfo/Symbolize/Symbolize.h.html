13#ifndef LLVM_DEBUGINFO_SYMBOLIZE_SYMBOLIZE_H
14#define LLVM_DEBUGINFO_SYMBOLIZE_SYMBOLIZE_H
15
16#include "llvm/DebugInfo/Symbolize/SymbolizableModule.h"
17#include "llvm/Object/Binary.h"
18#include "llvm/Object/ObjectFile.h"
19#include "llvm/Support/Error.h"
20#include <algorithm>
21#include <cstdint>
22#include <map>
23#include <memory>
24#include <string>
25#include <utility>
26#include <vector>
27
28namespace llvm {
29namespace symbolize {
30
31using namespace object;
32
33using FunctionNameKind = DILineInfoSpecifier::FunctionNameKind;
34
35class LLVMSymbolizer {
36public:
37 struct Options {
38 FunctionNameKind PrintFunctions = FunctionNameKind::LinkageName;
39 bool UseSymbolTable = true;
40 bool Demangle = true;
41 bool RelativeAddresses = false;
42 std::string DefaultArch;
43 std::vector<std::string> DsymHints;
44 std::string FallbackDebugPath;
45 std::string DWPName;
46 };
47
48 LLVMSymbolizer() = default;
49 LLVMSymbolizer(const Options &Opts) : Opts(Opts) {}
50
51 ~LLVMSymbolizer() {
52 flush();
53 }
54
55 Expected<DILineInfo> symbolizeCode(const std::string &ModuleName,
56 object::SectionedAddress ModuleOffset);
57 Expected<DIInliningInfo>
58 symbolizeInlinedCode(const std::string &ModuleName,
59 object::SectionedAddress ModuleOffset);
60 Expected<DIGlobal> symbolizeData(const std::string &ModuleName,
61 object::SectionedAddress ModuleOffset);
62 void flush();
63
64 static std::string
65 DemangleName(const std::string &Name,
66 const SymbolizableModule *DbiModuleDescriptor);
67
68private:
69 // Bundles together object file with code/data and object file with
70 // corresponding debug info. These objects can be the same.
71 using ObjectPair = std::pair<ObjectFile *, ObjectFile *>;
72
73 /// Returns a SymbolizableModule or an error if loading debug info failed.
74 /// Only one attempt is made to load a module, and errors during loading are
75 /// only reported once. Subsequent calls to get module info for a module that
76 /// failed to load will return nullptr.
77 Expected<SymbolizableModule *>
78 getOrCreateModuleInfo(const std::string &ModuleName);
79
80 ObjectFile *lookUpDsymFile(const std::string &Path,
81 const MachOObjectFile *ExeObj,
82 const std::string &ArchName);
83 ObjectFile *lookUpDebuglinkObject(const std::string &Path,
84 const ObjectFile *Obj,
85 const std::string &ArchName);
86
87 /// Returns pair of pointers to object and debug object.
88 Expected<ObjectPair> getOrCreateObjectPair(const std::string &Path,
89 const std::string &ArchName);
90
91 /// Return a pointer to object file at specified path, for a specified
92 /// architecture (e.g. if path refers to a Mach-O universal binary, only one
93 /// object file from it will be returned).
94 Expected<ObjectFile *> getOrCreateObject(const std::string &Path,
95 const std::string &ArchName);
96
97 std::map<std::string, std::unique_ptr<SymbolizableModule>> Modules;
98
99 /// Contains cached results of getOrCreateObjectPair().
100 std::map<std::pair<std::string, std::string>, ObjectPair>
101 ObjectPairForPathArch;
102
103 /// Contains parsed binary for each path, or parsing error.
104 std::map<std::string, OwningBinary<Binary>> BinaryForPath;
105
106 /// Parsed object file for path/architecture pair, where "path" refers
107 /// to Mach-O universal binary.
108 std::map<std::pair<std::string, std::string>, std::unique_ptr<ObjectFile>>
109 ObjectForUBPathAndArch;
110
111 Options Opts;
112};
113
114} // end namespace symbolize
115} // end namespace llvm
116
117#endif // LLVM_DEBUGINFO_SYMBOLIZE_SYMBOLIZE_H
