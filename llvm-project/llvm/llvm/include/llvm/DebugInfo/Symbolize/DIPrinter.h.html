14#ifndef LLVM_DEBUGINFO_SYMBOLIZE_DIPRINTER_H
15#define LLVM_DEBUGINFO_SYMBOLIZE_DIPRINTER_H
16
17#include "llvm/Support/raw_ostream.h"
18
19namespace llvm {
20struct DILineInfo;
21class DIInliningInfo;
22struct DIGlobal;
23
24namespace symbolize {
25
26class DIPrinter {
27public:
28 enum class OutputStyle { LLVM, GNU };
29
30private:
31 raw_ostream &OS;
32 bool PrintFunctionNames;
33 bool PrintPretty;
34 int PrintSourceContext;
35 bool Verbose;
36 bool Basenames;
37 OutputStyle Style;
38
39 void print(const DILineInfo &Info, bool Inlined);
40 void printContext(const std::string &FileName, int64_t Line);
41
42public:
43 DIPrinter(raw_ostream &OS, bool PrintFunctionNames = true,
44 bool PrintPretty = false, int PrintSourceContext = 0,
45 bool Verbose = false, bool Basenames = false,
46 OutputStyle Style = OutputStyle::LLVM)
47 : OS(OS), PrintFunctionNames(PrintFunctionNames),
48 PrintPretty(PrintPretty), PrintSourceContext(PrintSourceContext),
49 Verbose(Verbose), Basenames(Basenames), Style(Style) {}
50
51 DIPrinter &operator<<(const DILineInfo &Info);
52 DIPrinter &operator<<(const DIInliningInfo &Info);
53 DIPrinter &operator<<(const DIGlobal &Global);
54};
55}
56}
57
58#endif
