9#ifndef LLVM_DEBUGINFO_PDB_ERROR_H

12#include "llvm/ADT/StringRef.h"
13#include "llvm/Support/Error.h"

15namespace llvm {
16namespace pdb {
17
18enum class pdb_error_code {
19 invalid_utf8_path = 1,
20 dia_sdk_not_present,
21 dia_failed_loading,
22 signature_out_of_date,
23 external_cmdline_ref,
24 unspecified,
25};
26} // namespace pdb
27} // namespace llvm
28
29namespace std {
30template <>
31struct is_error_code_enum<llvm::pdb::pdb_error_code> : std::true_type {};
32} // namespace std
33
34namespace llvm {
35namespace pdb {
36const std::error_category &PDBErrCategory();
37
38inline std::error_code make_error_code(pdb_error_code E) {
39 return std::error_code(static_cast<int>(E), PDBErrCategory());
40}
41
42/// Base class for errors originating when parsing raw PDB files
43class PDBError : public ErrorInfo<PDBError, StringError> {
44public:
45 using ErrorInfo<PDBError, StringError>::ErrorInfo; // inherit constructors
46 PDBError(const Twine &S) : ErrorInfo(S, pdb_error_code::unspecified) {}
47 static char ID;
48};
49} // namespace pdb
50} // namespace llvm
51#endif
