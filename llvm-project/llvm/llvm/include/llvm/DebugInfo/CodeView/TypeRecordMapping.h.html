12#include "llvm/ADT/Optional.h"

15#include "llvm/Support/Error.h"

17namespace llvm {
18class BinaryStreamReader;
19class BinaryStreamWriter;
20
21namespace codeview {
22class TypeRecordMapping : public TypeVisitorCallbacks {
23public:
24 explicit TypeRecordMapping(BinaryStreamReader &Reader) : IO(Reader) {}
25 explicit TypeRecordMapping(BinaryStreamWriter &Writer) : IO(Writer) {}
26
27 using TypeVisitorCallbacks::visitTypeBegin;
28 Error visitTypeBegin(CVType &Record) override;
29 Error visitTypeEnd(CVType &Record) override;
30
31 Error visitMemberBegin(CVMemberRecord &Record) override;
32 Error visitMemberEnd(CVMemberRecord &Record) override;
33
34#define TYPE_RECORD(EnumName, EnumVal, Name) \
35 Error visitKnownRecord(CVType &CVR, Name##Record &Record) override;
36#define MEMBER_RECORD(EnumName, EnumVal, Name) \
37 Error visitKnownMember(CVMemberRecord &CVR, Name##Record &Record) override;
38#define TYPE_RECORD_ALIAS(EnumName, EnumVal, Name, AliasName)
39#define MEMBER_RECORD_ALIAS(EnumName, EnumVal, Name, AliasName)
40#include "llvm/DebugInfo/CodeView/CodeViewTypes.def"
41
42private:
43 Optional<TypeLeafKind> TypeKind;
44 Optional<TypeLeafKind> MemberKind;
45
46 CodeViewRecordIO IO;
47};
48}
49}
