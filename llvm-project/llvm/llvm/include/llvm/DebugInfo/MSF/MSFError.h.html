10#define LLVM_DEBUGINFO_MSF_MSFERROR_H

12#include "llvm/Support/Error.h"

14#include <string>

16namespace llvm {
17namespace msf {
18enum class msf_error_code {
19 unspecified = 1,
20 insufficient_buffer,
21 not_writable,
22 no_stream,
23 invalid_format,
24 block_in_use
25};
26} // namespace msf
27} // namespace llvm
28
29namespace std {
30template <>
31struct is_error_code_enum<llvm::msf::msf_error_code> : std::true_type {};
32} // namespace std
33
34namespace llvm {
35namespace msf {
36const std::error_category &MSFErrCategory();
37
38inline std::error_code make_error_code(msf_error_code E) {
39 return std::error_code(static_cast<int>(E), MSFErrCategory());
40}
41
42/// Base class for errors originating when parsing raw PDB files
43class MSFError : public ErrorInfo<MSFError, StringError> {
44public:
45 using ErrorInfo<MSFError, StringError>::ErrorInfo; // inherit constructors
46 MSFError(const Twine &S) : ErrorInfo(S, msf_error_code::unspecified) {}
47 static char ID;
48};
49} // namespace msf
50} // namespace llvm

52#endif // LLVM_DEBUGINFO_MSF_MSFERROR_H
