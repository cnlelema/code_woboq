8#ifndef LLVM_MC_MCSYMBOLXCOFF_H
9#define LLVM_MC_MCSYMBOLXCOFF_H
10
11#include "llvm/BinaryFormat/XCOFF.h"
12#include "llvm/MC/MCSymbol.h"
13
14namespace llvm {
15
16class GlobalValue;
17
18class MCSymbolXCOFF : public MCSymbol {
19 // The IR symbol this MCSymbolXCOFF is based on. It is set on function
20 // entry point symbols when they are the callee operand of a direct call
21 // SDNode.
22 const GlobalValue *GV = nullptr;
23
24public:
25 MCSymbolXCOFF(const StringMapEntry<bool> *Name, bool isTemporary)
26 : MCSymbol(SymbolKindXCOFF, Name, isTemporary) {}
27
28 void setGlobalValue(const GlobalValue *G) { GV = G; }
29 const GlobalValue *getGlobalValue() const { return GV; }
30
31 static bool classof(const MCSymbol *S) { return S->isXCOFF(); }
32};
33
34} // end namespace llvm
35
36#endif // LLVM_MC_MCSYMBOLXCOFF_H
