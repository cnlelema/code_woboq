60/// Target specific streamer interface. This is used so that targets can
61/// implement support for target specific assembly directives.
62///
63/// If target foo wants to use this, it should implement 3 classes:
64/// * FooTargetStreamer : public MCTargetStreamer
65/// * FooTargetAsmStreamer : public FooTargetStreamer
66/// * FooTargetELFStreamer : public FooTargetStreamer
67///
68/// FooTargetStreamer should have a pure virtual method for each directive. For
69/// example, for a ".bar symbol_name" directive, it should have
70/// virtual emitBar(const MCSymbol &Symbol) = 0;
71///
72/// The FooTargetAsmStreamer and FooTargetELFStreamer classes implement the
73/// method. The assembly streamer just prints ".bar symbol_name". The object
74/// streamer does whatever is needed to implement .bar in the object file.
75///
76/// In the assembly printer and parser the target streamer can be used by
77/// calling getTargetStreamer and casting it to FooTargetStreamer:
78///
79/// MCTargetStreamer &TS = OutStreamer.getTargetStreamer();
80/// FooTargetStreamer &ATS = static_cast<FooTargetStreamer &>(TS);
81///
82/// The base classes FooTargetAsmStreamer and FooTargetELFStreamer should
83/// *never* be treated differently. Callers should always talk to a
84/// FooTargetStreamer.
85class MCTargetStreamer {
86protected:
87 MCStreamer &Streamer;
88
89public:
90 MCTargetStreamer(MCStreamer &S);
91 virtual ~MCTargetStreamer();
92
93 MCStreamer &getStreamer() { return Streamer; }
94

99
100 virtual void prettyPrintAsm(MCInstPrinter &InstPrinter, raw_ostream &OS,
101 const MCInst &Inst, const MCSubtargetInfo &STI);
102
103 virtual void emitDwarfFileDirective(StringRef Directive);
104
105 /// Update streamer for a new active section.
106 ///
107 /// This is called by PopSection and SwitchSection, if the current
108 /// section changes.
109 virtual void changeSection(const MCSection *CurSection, MCSection *Section,
110 const MCExpr *SubSection, raw_ostream &OS);
111
112 virtual void emitValue(const MCExpr *Value);
113
114 /// Emit the bytes in \p Data into the output.
115 ///
116 /// This is used to emit bytes in \p Data as sequence of .byte directives.
117 virtual void emitRawBytes(StringRef Data);
118
119 virtual void finish();
120
121 /// Whether to use the __cap_relocs hack (if the target supports capabilities)
122 virtual bool useLegacyCapRelocs() const;
123
124 /// CHERI128 uses compressed capabilities. If we would like to guarantee
125 /// non-overlapping bounds for all global symbols we must over-align the
126 /// symbol if the size is no precisely representable. We also add padding at
127 /// the end to ensure that we cannot access another variable that happens to
128 /// be located in the bytes that are accessible after the end of the object
129 /// due to the bounds having been rounded up.
130 virtual TailPaddingAmount getTailPaddingForPreciseBounds(unsigned Size) {
131 return TailPaddingAmount::None;
132 };
133 virtual unsigned getAlignmentForPreciseBounds(unsigned Size) { return 0; };
134};

378 /// Update streamer for a new active section.
379 ///
380 /// This is called by PopSection and SwitchSection, if the current
381 /// section changes.
382 virtual void ChangeSection(MCSection *, const MCExpr *);
383
384 /// Save the current and previous section on the section stack.
385 void PushSection() {
386 SectionStack.push_back(
387 std::make_pair(getCurrentSection(), getPreviousSection()));
388 }
389
390 /// Restore the current and previous section from the section stack.
391 /// Calls ChangeSection as needed.
392 ///
393 /// Returns false if the stack was empty.
394 bool PopSection() {
395 if (SectionStack.size() <= 1)
396 return false;
397 auto I = SectionStack.end();
398 --I;
399 MCSectionSubPair OldSection = I->first;
400 --I;
401 MCSectionSubPair NewSection = I->first;
402
403 if (OldSection != NewSection)
404 ChangeSection(NewSection.first, NewSection.second);
405 SectionStack.pop_back();
406 return true;
407 }
408
409 bool SubSection(const MCExpr *Subsection) {
410 if (SectionStack.empty())
411 return false;
412
413 SwitchSection(SectionStack.back().first.first, Subsection);
414 return true;
415 }

424 /// Set the current section where code is being emitted to \p Section.
425 /// This is required to update CurSection. This version does not call
426 /// ChangeSection.
427 void SwitchSectionNoChange(MCSection *Section,
428 const MCExpr *Subsection = nullptr) {
429 assert(Section && "Cannot switch to a null section!");
430 MCSectionSubPair curSection = SectionStack.back().first;
431 SectionStack.back().second = curSection;

433 SectionStack.back().first = MCSectionSubPair(Section, Subsection);
434 }
