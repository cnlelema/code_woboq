9#ifndef LLVM_MC_MCTARGETOPTIONS_H
10#define LLVM_MC_MCTARGETOPTIONS_H
11
12#include <string>
13#include <vector>
14
15namespace llvm {
16
17enum class ExceptionHandling {
18 None, ///< No exception support
19 DwarfCFI, ///< DWARF-like instruction based exceptions
20 SjLj, ///< setjmp/longjmp based exceptions
21 ARM, ///< ARM EHABI
22 WinEH, ///< Windows Exception Handling
23 Wasm, ///< WebAssembly Exception Handling
24};
25
26enum class DebugCompressionType {
27 None, ///< No compression
28 GNU, ///< zlib-gnu style compression
29 Z, ///< zlib style complession
30};
31
32enum class CheriCapabilityTableABI {
33 Legacy, /// Use legacy ABI instead (load from got and use .size.foo to get the
34 /// size)
35 PLT, /// Use PLT stubs to set reserved register $cgp (functions assume $cgp is
36 /// set correctly)
37 Pcrel, /// Derive register $cgp from $pcc (does not need to be set on function
38 /// entry)
39 FunctionDescriptor /// Use function descriptors to get $cgp (functions assume
40 /// $cgp is set correctly) (TODO: different approaches
41 /// possible here)
42};
43
44enum class CheriCapabilityTlsABI {
45 Legacy, /// Use legacy ABI instead (thread pointer is normal integer register
46 /// and capability is derived from DDC)
47 CapEquiv /// Use an equivalent setup to MIPS, but with pointers replaced by
48 /// capabilities and using a capability thread pointer register
49};
50
51class StringRef;
52
53class MCTargetOptions {
54public:
55 enum AsmInstrumentation {
56 AsmInstrumentationNone,
57 AsmInstrumentationAddress
58 };
59
60 bool MCRelaxAll : 1;
61 bool MCNoExecStack : 1;
62 bool MCFatalWarnings : 1;
63 bool MCNoWarn : 1;
64 bool MCNoDeprecatedWarn : 1;
65 bool MCSaveTempLabels : 1;
66 bool MCUseDwarfDirectory : 1;
67 bool MCIncrementalLinkerCompatible : 1;
68 bool MCPIECopyRelocations : 1;
69 bool ShowMCEncoding : 1;
70 bool ShowMCInst : 1;
71 bool AsmVerbose : 1;
72
73 /// Preserve Comments in Assembly.
74 bool PreserveAsmComments : 1;
75
76 int DwarfVersion = 0;
77
78 std::string ABIName;
79 std::string SplitDwarfFile;
80
81 /// Additional paths to search for `.include` directives when using the
82 /// integrated assembler.
83 std::vector<std::string> IASSearchPaths;
84
85 MCTargetOptions();
86
87 /// getABIName - If this returns a non-empty string this represents the
88 /// textual name of the ABI that we want the backend to use, e.g. o32, or
89 /// aapcs-linux.
90 StringRef getABIName() const;
91
92 // HACK to make the cheri cap table setting visible to clang:
93 static bool cheriUsesCapabilityTable();
94 static CheriCapabilityTableABI cheriCapabilityTableABI();
95
96 // Ditto for cheri cap tls setting
97 static bool cheriUsesCapabilityTls();
98 static CheriCapabilityTlsABI cheriCapabilityTlsABI();
99};
100
101} // end namespace llvm

103#endif // LLVM_MC_MCTARGETOPTIONS_H
