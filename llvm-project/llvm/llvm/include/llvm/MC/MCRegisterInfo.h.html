27/// An unsigned integer type large enough to represent all physical registers,
28/// but not necessarily virtual registers.
29using MCPhysReg = uint16_t;

33/// MCRegisterClass - Base class of TargetRegisterClass.
34class MCRegisterClass {
35public:
36 using iterator = const MCPhysReg*;
37 using const_iterator = const MCPhysReg*;
38
39 const iterator RegsBegin;
40 const uint8_t *const RegSet;
41 const uint32_t NameIdx;
42 const uint16_t RegsSize;
43 const uint16_t RegSetSize;
44 const uint16_t ID;
45 const int8_t CopyCost;
46 const bool Allocatable;
47
48 /// getID() - Return the register class ID number.
49 ///
50 unsigned getID() const { return ID; }
51
52 /// begin/end - Return all of the registers in this class.
53 ///
54 iterator begin() const { return RegsBegin; }
55 iterator end() const { return RegsBegin + RegsSize; }
56
57 /// getNumRegs - Return the number of registers in this class.
58 ///
59 unsigned getNumRegs() const { return RegsSize; }
60
61 /// getRegister - Return the specified register in the class.
62 ///
63 unsigned getRegister(unsigned i) const {
64 assert(i < getNumRegs() && "Register number out of range!");
65 return RegsBegin[i];
66 }
67
68 /// contains - Return true if the specified register is included in this
69 /// register class. This does not include virtual registers.
70 bool contains(unsigned Reg) const {
71 unsigned InByte = Reg % 8;
72 unsigned Byte = Reg / 8;
73 if (Byte >= RegSetSize)
74 return false;
75 return (RegSet[Byte] & (1 << InByte)) != 0;
76 }
77
78 /// contains - Return true if both registers are in this class.
79 bool contains(unsigned Reg1, unsigned Reg2) const {
80 return contains(Reg1) && contains(Reg2);
81 }
82
83 /// getCopyCost - Return the cost of copying a value between two registers in
84 /// this class. A negative number means the register class is very expensive
85 /// to copy e.g. status flag register classes.
86 int getCopyCost() const { return CopyCost; }
87
88 /// isAllocatable - Return true if this register class may be used to create
89 /// virtual registers.
90 bool isAllocatable() const { return Allocatable; }
91};

93/// MCRegisterDesc - This record contains information about a particular
94/// register. The SubRegs field is a zero terminated array of registers that
95/// are sub-registers of the specific register, e.g. AL, AH are sub-registers
96/// of AX. The SuperRegs field is a zero terminated array of registers that are
97/// super-registers of the specific register, e.g. RAX, EAX, are
98/// super-registers of AX.
99///
100struct MCRegisterDesc {
101 uint32_t Name; // Printable name for the reg (for debugging)
102 uint32_t SubRegs; // Sub-register set, described above
103 uint32_t SuperRegs; // Super-register set, described above
104
105 // Offset into MCRI::SubRegIndices of a list of sub-register indices for each
106 // sub-register in SubRegs.
107 uint32_t SubRegIndices;
108
109 // RegUnits - Points to the list of register units. The low 4 bits holds the
110 // Scale, the high bits hold an offset into DiffLists. See MCRegUnitIterator.
111 uint32_t RegUnits;
112
113 /// Index into list with lane mask sequences. The sequence contains a lanemask
114 /// for every register unit.
115 uint16_t RegUnitLaneMasks;
116};

118/// MCRegisterInfo base class - We assume that the target defines a static
119/// array of MCRegisterDesc objects that represent all of the machine
120/// registers that the target has. As such, we simply have to track a pointer
121/// to this array so that we can turn register number into a register
122/// descriptor.
123///
124/// Note this class is designed to be a base class of TargetRegisterInfo, which
125/// is the interface used by codegen. However, specific targets *should never*

184 /// DiffListIterator - Base iterator class that can traverse the
185 /// differentially encoded register and regunit lists in DiffLists.
186 /// Don't use this class directly, use one of the specialized sub-classes
187 /// defined below.
188 class DiffListIterator {
189 uint16_t Val = 0;
190 const MCPhysReg *List = nullptr;
191
192 protected:
193 /// Create an invalid iterator. Call init() to point to something useful.
194 DiffListIterator() = default;
195
196 /// init - Point the iterator to InitVal, decoding subsequent values from
197 /// DiffList. The iterator will initially point to InitVal, sub-classes are
198 /// responsible for skipping the seed value if it is not part of the list.
199 void init(MCPhysReg InitVal, const MCPhysReg *DiffList) {
200 Val = InitVal;
201 List = DiffList;
202 }
203
204 /// advance - Move to the next list position, return the applied
205 /// differential. This function does not detect the end of the list, that
206 /// is the caller's responsibility (by checking for a 0 return value).
207 unsigned advance() {
208 assert(isValid() && "Cannot move off the end of the list.");
209 MCPhysReg D = *List++;
210 Val += D;
211 return D;
212 }
213
214 public:
215 /// isValid - returns true if this iterator is not yet at the end.
216 bool isValid() const { return List; }
217
218 /// Dereference the iterator to get the value at the current position.
219 unsigned operator*() const { return Val; }
220
221 /// Pre-increment to move to the next position.
222 void operator++() {
223 // The end of the list is encoded as a 0 differential.
224 if (!advance())
225 List = nullptr;
226 }
227 };

546// Definition for isSuperRegister. Put it down here since it needs the
547// iterator defined above in addition to the MCRegisterInfo class itself.
548inline bool MCRegisterInfo::isSuperRegister(unsigned RegA, unsigned RegB) const{
549 for (MCSuperRegIterator I(RegA, this); I.isValid(); ++I)

555//===----------------------------------------------------------------------===//
556// Register Units
557//===----------------------------------------------------------------------===//
558
559// Register units are used to compute register aliasing. Every register has at
560// least one register unit, but it can have more. Two registers overlap if and
561// only if they have a common register unit.
562//
563// A target with a complicated sub-register structure will typically have many
564// fewer register units than actual registers. MCRI::getNumRegUnits() returns
565// the number of register units in the target.
566
567// MCRegUnitIterator enumerates a list of register units for Reg. The list is
568// in ascending numerical order.
569class MCRegUnitIterator : public MCRegisterInfo::DiffListIterator {
570public:
571 /// MCRegUnitIterator - Create an iterator that traverses the register units
572 /// in Reg.
573 MCRegUnitIterator() = default;
574
575 MCRegUnitIterator(unsigned Reg, const MCRegisterInfo *MCRI) {
576 assert(Reg && "Null register has no regunits");
577 // Decode the RegUnits MCRegisterDesc field.
578 unsigned RU = MCRI->get(Reg).RegUnits;
579 unsigned Scale = RU & 15;
580 unsigned Offset = RU >> 4;
581
582 // Initialize the iterator to Reg * Scale, and the List pointer to
583 // DiffLists + Offset.
584 init(Reg * Scale, MCRI->DiffLists + Offset);
585
586 // That may not be a valid unit, we need to advance by one to get the real
587 // unit number. The first differential can be 0 which would normally
588 // terminate the list, but since we know every register has at least one
589 // unit, we can allow a 0 differential here.
590 advance();
591 }
592};

627// Each register unit has one or two root registers. The complete set of
628// registers containing a register unit is the union of the roots and their
629// super-registers. All registers aliasing Unit can be visited like this:
630//
631// for (MCRegUnitRootIterator RI(Unit, MCRI); RI.isValid(); ++RI) {
632// for (MCSuperRegIterator SI(*RI, MCRI, true); SI.isValid(); ++SI)
633// visit(*SI);
634// }
635
636/// MCRegUnitRootIterator enumerates the root registers of a register unit.
637class MCRegUnitRootIterator {
638 uint16_t Reg0 = 0;
639 uint16_t Reg1 = 0;
640
641public:
642 MCRegUnitRootIterator() = default;
643
644 MCRegUnitRootIterator(unsigned RegUnit, const MCRegisterInfo *MCRI) {
645 assert(RegUnit < MCRI->getNumRegUnits() && "Invalid register unit");
646 Reg0 = MCRI->RegUnitRoots[RegUnit][0];
647 Reg1 = MCRI->RegUnitRoots[RegUnit][1];
648 }
649
650 /// Dereference to get the current root register.
651 unsigned operator*() const {
652 return Reg0;
653 }
654
655 /// Check if the iterator is at the end of the list.
656 bool isValid() const {
657 return Reg0;
658 }
659
660 /// Preincrement to move to the next root register.
661 void operator++() {
662 assert(isValid() && "Cannot move off the end of the list.");
663 Reg0 = Reg1;
664 Reg1 = 0;
665 }
666};

668/// MCRegAliasIterator enumerates all registers aliasing Reg. If IncludeSelf is
669/// set, Reg itself is included in the list. This iterator does not guarantee
670/// any ordering or that entries are unique.
671class MCRegAliasIterator {
672private:
673 unsigned Reg;
674 const MCRegisterInfo *MCRI;
675 bool IncludeSelf;
676
677 MCRegUnitIterator RI;
678 MCRegUnitRootIterator RRI;
679 MCSuperRegIterator SI;
680
681public:
682 MCRegAliasIterator(unsigned Reg, const MCRegisterInfo *MCRI,
683 bool IncludeSelf)
684 : Reg(Reg), MCRI(MCRI), IncludeSelf(IncludeSelf) {
685 // Initialize the iterators.
686 for (RI = MCRegUnitIterator(Reg, MCRI); RI.isValid(); ++RI) {
687 for (RRI = MCRegUnitRootIterator(*RI, MCRI); RRI.isValid(); ++RRI) {
688 for (SI = MCSuperRegIterator(*RRI, MCRI, true); SI.isValid(); ++SI) {
689 if (!(!IncludeSelf && Reg == *SI))
690 return;
691 }
692 }
693 }
694 }
695
696 bool isValid() const { return RI.isValid(); }
697
698 unsigned operator*() const {
699 assert(SI.isValid() && "Cannot dereference an invalid iterator.");
700 return *SI;
701 }
702
703 void advance() {
704 // Assuming SI is valid.
705 ++SI;
706 if (SI.isValid()) return;
707
708 ++RRI;
709 if (RRI.isValid()) {
710 SI = MCSuperRegIterator(*RRI, MCRI, true);
711 return;
712 }
713
714 ++RI;
715 if (RI.isValid()) {
716 RRI = MCRegUnitRootIterator(*RI, MCRI);
717 SI = MCSuperRegIterator(*RRI, MCRI, true);
718 }
719 }
720
721 void operator++() {
722 assert(isValid() && "Cannot move off the end of the list.");
723 do advance();
724 while (!IncludeSelf && isValid() && *SI == Reg);
725 }
726};
