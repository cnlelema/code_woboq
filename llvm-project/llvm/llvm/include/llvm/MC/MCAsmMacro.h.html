18namespace llvm {

20/// Target independent representation for an assembler token.
21class AsmToken {
22public:
23 enum TokenKind {
24 // Markers
25 Eof, Error,
26
27 // String values.
28 Identifier,
29 String,
30
31 // Integer values.
32 Integer,
33 BigNum, // larger than 64 bits
34
35 // Real values.
36 Real,
37
38 // Comments
39 Comment,
40 HashDirective,
41 // No-value.
42 EndOfStatement,
43 Colon,
44 Space,
45 Plus, Minus, Tilde,
46 Slash, // '/'
47 BackSlash, // '\'
48 LParen, RParen, LBrac, RBrac, LCurly, RCurly,
49 Star, Dot, Comma, Dollar, Equal, EqualEqual,
50
51 Pipe, PipePipe, Caret,
52 Amp, AmpAmp, Exclaim, ExclaimEqual, Percent, Hash,
53 Less, LessEqual, LessLess, LessGreater,
54 Greater, GreaterEqual, GreaterGreater, At, MinusGreater,
55
56 // MIPS unary expression operators such as %neg.
57 PercentCall16, PercentCall_Hi, PercentCall_Lo, PercentDtprel_Hi,
58 PercentDtprel_Lo, PercentGot, PercentGot_Disp, PercentGot_Hi, PercentGot_Lo,
59 PercentGot_Ofst, PercentGot_Page, PercentGottprel, PercentGp_Rel, PercentHi,
60 PercentHigher, PercentHighest, PercentLo, PercentNeg, PercentPcrel_Hi,
61 PercentPcrel_Lo, PercentTlsgd, PercentTlsldm, PercentTprel_Hi,
62 PercentTprel_Lo,
63
64 // CHERI capability relocation expressions:
65 PercentCapTabCall11, PercentCapTabCall_Hi, PercentCapTabCall_Lo,
66 PercentCapTab11, PercentCapTab_Hi, PercentCapTab_Lo,
67 PercentCapTab20, PercentCapTabCall20, PercentCapTab_Rel,
68 PercentCapTabTlsgd_Hi, PercentCapTabTlsgd_Lo,
69 PercentCapTabTlsldm_Hi, PercentCapTabTlsldm_Lo,
70 PercentCapTabTprel_Hi, PercentCapTabTprel_Lo
71 };
72
73private:
74 TokenKind Kind;
75
76 /// A reference to the entire token contents; this is always a pointer into
77 /// a memory buffer owned by the source manager.
78 StringRef Str;
79
80 APInt IntVal;
81
82public:
83 AsmToken() = default;
84 AsmToken(TokenKind Kind, StringRef Str, APInt IntVal)
85 : Kind(Kind), Str(Str), IntVal(std::move(IntVal)) {}
86 AsmToken(TokenKind Kind, StringRef Str, int64_t IntVal = 0)
87 : Kind(Kind), Str(Str), IntVal(64, IntVal, true) {}
88
89 TokenKind getKind() const { return Kind; }
90 bool is(TokenKind K) const { return Kind == K; }
91 bool isNot(TokenKind K) const { return Kind != K; }
92
93 SMLoc getLoc() const;
94 SMLoc getEndLoc() const;
95 SMRange getLocRange() const;
96
97 /// Get the contents of a string token (without quotes).
98 StringRef getStringContents() const {
99 assert(Kind == String && "This token isn't a string!");
100 return Str.slice(1, Str.size() - 1);
101 }
102
103 /// Get the identifier string for the current token, which should be an
104 /// identifier or a string. This gets the portion of the string which should
105 /// be used as the identifier, e.g., it does not include the quotes on
106 /// strings.
107 StringRef getIdentifier() const {
108 if (Kind == Identifier)
109 return getString();
110 return getStringContents();
111 }
112
113 /// Get the string for the current token, this includes all characters (for
114 /// example, the quotes on strings) in the token.
115 ///
116 /// The returned StringRef points into the source manager's memory buffer, and
117 /// is safe to store across calls to Lex().
118 StringRef getString() const { return Str; }
119
120 // FIXME: Don't compute this in advance, it makes every token larger, and is
121 // also not generally what we want (it is nicer for recovery etc. to lex 123br
122 // as a single token, then diagnose as an invalid number).
123 int64_t getIntVal() const {
124 assert(Kind == Integer && "This token isn't an integer!");
125 return IntVal.getZExtValue();
126 }
127
128 APInt getAPIntVal() const {
129 assert((Kind == Integer || Kind == BigNum) &&
130 "This token isn't an integer!");
131 return IntVal;
132 }
133
134 void dump(raw_ostream &OS) const;
135 void dump() const { dump(dbgs()); }
136};

138struct MCAsmMacroParameter {
139 StringRef Name;
140 std::vector<AsmToken> Value;
141 bool Required = false;
142 bool Vararg = false;
143
144 MCAsmMacroParameter() = default;
145
146 void dump() const { dump(dbgs()); }
147 void dump(raw_ostream &OS) const;
148};
149
150typedef std::vector<MCAsmMacroParameter> MCAsmMacroParameters;
151struct MCAsmMacro {
152 StringRef Name;
153 StringRef Body;
154 MCAsmMacroParameters Parameters;
155
156public:
157 MCAsmMacro(StringRef N, StringRef B, MCAsmMacroParameters P)
158 : Name(N), Body(B), Parameters(std::move(P)) {}
159
160 void dump() const { dump(dbgs()); }
161 void dump(raw_ostream &OS) const;
162};
