13#ifndef LLVM_REMARKS_REMARK_PARSER_H
14#define LLVM_REMARKS_REMARK_PARSER_H
15
16#include "llvm/ADT/SmallVector.h"
17#include "llvm/ADT/StringRef.h"
18#include "llvm/Remarks/Remark.h"
19#include "llvm/Support/Error.h"
20#include <memory>
21
22namespace llvm {
23namespace remarks {
24
25struct ParserImpl;
26
27/// Parser used to parse a raw buffer to remarks::Remark objects.
28struct Parser {
29 /// The hidden implementation of the parser.
30 std::unique_ptr<ParserImpl> Impl;
31
32 /// Create a parser parsing \p Buffer to Remark objects.
33 /// This constructor should be only used for parsing YAML remarks.
34 Parser(StringRef Buffer);
35
36 /// Create a parser parsing \p Buffer to Remark objects, using \p StrTabBuf as
37 /// string table.
38 /// This constructor should be only used for parsing YAML remarks.
39 Parser(StringRef Buffer, StringRef StrTabBuf);
40
41 // Needed because ParserImpl is an incomplete type.
42 ~Parser();
43
44 /// Returns an empty Optional if it reached the end.
45 /// Returns a valid remark otherwise.
46 Expected<const Remark *> getNext() const;
47};
48
49/// In-memory representation of the string table parsed from a buffer (e.g. the
50/// remarks section).
51struct ParsedStringTable {
52 /// The buffer mapped from the section contents.
53 StringRef Buffer;
54 /// Collection of offsets in the buffer for each string entry.
55 SmallVector<size_t, 8> Offsets;
56
57 Expected<StringRef> operator[](size_t Index);
58 ParsedStringTable(StringRef Buffer);
59};
60
61} // end namespace remarks
62} // end namespace llvm

64#endif /* LLVM_REMARKS_REMARK_PARSER_H */
