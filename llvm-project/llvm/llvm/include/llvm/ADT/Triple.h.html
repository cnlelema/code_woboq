22/// Triple - Helper class for working with autoconf configuration names. For
23/// historical reasons, we also call these 'triples' (they used to contain
24/// exactly three fields).
25///
26/// Configuration names are strings in the canonical form:
27/// ARCHITECTURE-VENDOR-OPERATING_SYSTEM
28/// or
29/// ARCHITECTURE-VENDOR-OPERATING_SYSTEM-ENVIRONMENT
30///
31/// This class is used for clients which want to support arbitrary
32/// configuration names, but also want to implement certain special
33/// behavior for particular configurations. This class isolates the mapping
34/// from the components of the configuration name to well known IDs.
35///
36/// At its core the Triple class is designed to be a wrapper for a triple
37/// string; the constructor does not change or normalize the triple string.
38/// Clients that need to handle the non-canonical triples that users often
39/// specify should use the normalize method.
40///
41/// See autoconf/config.guess for a glimpse into what configuration names
42/// look like in practice.

398 /// Test whether the architecture is 64-bit
399 ///
400 /// Note that this tests for 64-bit pointer width, and nothing else. Note
401 /// that we intentionally expose only three predicates, 64-bit, 32-bit, and

403 /// is not summed up in the triple, and so only a coarse grained predicate
404 /// system is provided.

417 /// isOSVersionLT - Helper function for doing comparisons against version
418 /// numbers included in the target triple.
419 bool isOSVersionLT(unsigned Major, unsigned Minor = 0,
420 unsigned Micro = 0) const {
421 unsigned LHS[3];
422 getOSVersion(LHS[0], LHS[1], LHS[2]);
423
424 if (LHS[0] != Major)
425 return LHS[0] < Major;
426 if (LHS[1] != Minor)
427 return LHS[1] < Minor;
428 if (LHS[2] != Micro)
429 return LHS[2] < Micro;
430
431 return false;
432 }
433
434 bool isOSVersionLT(const Triple &Other) const {
435 unsigned RHS[3];
436 Other.getOSVersion(RHS[0], RHS[1], RHS[2]);
437 return isOSVersionLT(RHS[0], RHS[1], RHS[2]);
438 }
439
440 /// isMacOSXVersionLT - Comparison function for checking OS X version
441 /// compatibility, which handles supporting skewed version numbering schemes
442 /// used by the "darwin" triples.
443 bool isMacOSXVersionLT(unsigned Major, unsigned Minor = 0,
444 unsigned Micro = 0) const {
445 assert(isMacOSX() && "Not an OS X triple!");
446
447 // If this is OS X, expect a sane version number.
448 if (getOS() == Triple::MacOSX)
449 return isOSVersionLT(Major, Minor, Micro);
450
451 // Otherwise, compare to the "Darwin" number.
452 assert(Major == 10 && "Unexpected major version");
453 return isOSVersionLT(Minor + 4, Micro, 0);
454 }
