9// This header provides a C API to use the LLVM link time optimization
10// library. This is intended to be used by linkers which are C-only in
11// their implementation for performing LTO.
12//
13//===----------------------------------------------------------------------===//
14
15#ifndef LLVM_C_LINKTIMEOPTIMIZER_H
16#define LLVM_C_LINKTIMEOPTIMIZER_H
17
18#ifdef __cplusplus
19extern "C" {
20#endif
21
22/**
23 * @defgroup LLVMCLinkTimeOptimizer Link Time Optimization
24 * @ingroup LLVMC
25 *
26 * @{
27 */
28
29 /// This provides a dummy type for pointers to the LTO object.
30 typedef void* llvm_lto_t;
31
32 /// This provides a C-visible enumerator to manage status codes.
33 /// This should map exactly onto the C++ enumerator LTOStatus.
34 typedef enum llvm_lto_status {
35 LLVM_LTO_UNKNOWN,
36 LLVM_LTO_OPT_SUCCESS,
37 LLVM_LTO_READ_SUCCESS,
38 LLVM_LTO_READ_FAILURE,
39 LLVM_LTO_WRITE_FAILURE,
40 LLVM_LTO_NO_TARGET,
41 LLVM_LTO_NO_WORK,
42 LLVM_LTO_MODULE_MERGE_FAILURE,
43 LLVM_LTO_ASM_FAILURE,
44
45 // Added C-specific error codes
46 LLVM_LTO_NULL_OBJECT
47 } llvm_lto_status_t;
48
49 /// This provides C interface to initialize link time optimizer. This allows
50 /// linker to use dlopen() interface to dynamically load LinkTimeOptimizer.
51 /// extern "C" helps, because dlopen() interface uses name to find the symbol.
52 extern llvm_lto_t llvm_create_optimizer(void);
53 extern void llvm_destroy_optimizer(llvm_lto_t lto);
54
55 extern llvm_lto_status_t llvm_read_object_file
56 (llvm_lto_t lto, const char* input_filename);
57 extern llvm_lto_status_t llvm_optimize_modules
58 (llvm_lto_t lto, const char* output_filename);
59
60/**
61 * @}
62 */
63
64#ifdef __cplusplus
65}
66#endif
67
68#endif
