23 // Various helper functions.
24 LLVM_ATTRIBUTE_NORETURN void reportError(Twine Msg);
25 void reportWarning(Twine Msg);
26 void warn(llvm::Error Err);
27 void error(std::error_code EC);
28 void error(llvm::Error EC);
29 template <typename T> T error(llvm::Expected<T> &&E) {
30 error(E.takeError());
31 return std::move(*E);
32 }
33
34 template <class T> T unwrapOrError(ErrorOr<T> EO) {
35 if (EO)
36 return *EO;
37 reportError(EO.getError().message());
38 }
39 template <class T> T unwrapOrError(Expected<T> EO) {
40 if (EO)
41 return *EO;
42 std::string Buf;
43 raw_string_ostream OS(Buf);
44 logAllUnhandledErrors(EO.takeError(), OS);
45 OS.flush();
46 reportError(Buf);
47 }
