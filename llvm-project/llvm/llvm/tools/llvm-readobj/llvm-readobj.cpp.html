9// This is a tool similar to readelf, except it works on multiple object file
10// formats. The main purpose of this tool is to provide detailed output suitable
11// for FileCheck.
12//
13// Flags should be similar to readelf where supported, but the output format
14// does not need to be identical. The point is to not make users learn yet
15// another set of flags.
16//
17// Output should be specialized for each format where appropriate.
18//
19//===----------------------------------------------------------------------===//

385LLVM_ATTRIBUTE_NORETURN void reportError(Twine Msg) {
386 errs() << "\n";
387 WithColor::error(errs()) << Msg << "\n";
388 exit(1);
389}
390
391void reportWarning(Twine Msg) {
392 errs() << "\n";
393 WithColor::warning(errs()) << Msg << "\n";
394}
395
396void warn(Error Err) {
397 handleAllErrors(std::move(Err), [&](const ErrorInfoBase &EI) {
398 reportWarning(EI.message());
399 });
400}
401
402void error(Error EC) {
403 if (!EC)
404 return;
405 handleAllErrors(std::move(EC),
406 [&](const ErrorInfoBase &EI) { reportError(EI.message()); });
407}
408
409void error(std::error_code EC) {
410 if (!EC)
411 return;
412 reportError(EC.message());
413}

417static void reportError(StringRef Input, Error Err) {
418 if (Input == "-")
419 Input = "<stdin>";
420 error(createFileError(Input, std::move(Err)));
421}

423static void reportError(StringRef Input, std::error_code EC) {
424 reportError(Input, errorCodeToError(EC));
425}

427static bool isMipsArch(unsigned Arch) {
428 switch (Arch) {
429 case llvm::Triple::mips:
430 case llvm::Triple::mipsel:
431 case llvm::Triple::mips64:
432 case llvm::Triple::mips64el:
433 case llvm::Triple::cheri:
434 return true;
435 default:
436 return false;
437 }
438}

455/// Creates an format-specific object file dumper.
456static std::error_code createDumper(const ObjectFile *Obj,
457 ScopedPrinter &Writer,
458 std::unique_ptr<ObjDumper> &Result) {
459 if (!Obj)
460 return readobj_error::unsupported_file_format;
461
462 if (Obj->isCOFF())
463 return createCOFFDumper(Obj, Writer, Result);
464 if (Obj->isELF())
465 return createELFDumper(Obj, Writer, Result);
466 if (Obj->isMachO())
467 return createMachODumper(Obj, Writer, Result);
468 if (Obj->isWasm())
469 return createWasmDumper(Obj, Writer, Result);
470 if (Obj->isXCOFF())
471 return createXCOFFDumper(Obj, Writer, Result);
472
473 return readobj_error::unsupported_obj_file_format;
474}

476/// Dumps the specified object file.
477static void dumpObject(const ObjectFile *Obj, ScopedPrinter &Writer) {
478 std::unique_ptr<ObjDumper> Dumper;
479 if (std::error_code EC = createDumper(Obj, Writer, Dumper))
480 reportError(Obj->getFileName(), EC);
481
482 if (opts::Output == opts::LLVM) {
483 Writer.startLine() << "\n";
484 Writer.printString("File", Obj->getFileName());
485 Writer.printString("Format", Obj->getFileFormatName());
486 Writer.printString("Arch", Triple::getArchTypeName(
487 (llvm::Triple::ArchType)Obj->getArch()));
488 Writer.printString("AddressSize",
489 formatv("{0}bit", 8 * Obj->getBytesInAddress()));
490 Dumper->printLoadName();
491 }
492
493 if (opts::FileHeaders)
494 Dumper->printFileHeaders();
495 if (opts::SectionHeaders)
496 Dumper->printSectionHeaders();
497 if (opts::Relocations)
498 Dumper->printRelocations();
499 if (opts::DynRelocs)
500 Dumper->printDynamicRelocations();
501 if (opts::Symbols || opts::DynamicSymbols)
502 Dumper->printSymbols(opts::Symbols, opts::DynamicSymbols);
503 if (opts::HashSymbols)
504 Dumper->printHashSymbols();
505 if (opts::UnwindInfo)
506 Dumper->printUnwindInfo();
507 if (opts::DynamicTable)
508 Dumper->printDynamicTable();
509 if (opts::NeededLibraries)
510 Dumper->printNeededLibraries();
511 if (opts::ProgramHeaders || opts::SectionMapping == cl::BOU_TRUE)
512 Dumper->printProgramHeaders(opts::ProgramHeaders, opts::SectionMapping);
513 if (!opts::StringDump.empty())
514 llvm::for_each(opts::StringDump, [&Dumper, Obj](StringRef SectionName) {
515 Dumper->printSectionAsString(Obj, SectionName);
516 });
517 if (!opts::HexDump.empty())
518 llvm::for_each(opts::HexDump, [&Dumper, Obj](StringRef SectionName) {
519 Dumper->printSectionAsHex(Obj, SectionName);
520 });
521 if (opts::HashTable)
522 Dumper->printHashTable();
523 if (opts::GnuHashTable)
524 Dumper->printGnuHashTable();
525 if (opts::VersionInfo)
526 Dumper->printVersionInfo();
527 if (Obj->isELF()) {
528 if (opts::ELFLinkerOptions)
529 Dumper->printELFLinkerOptions();
530 if (Obj->getArch() == llvm::Triple::arm)
531 if (opts::ARMAttributes)
532 Dumper->printAttributes();
533 if (isMipsArch(Obj->getArch())) {
534 if (opts::MipsPLTGOT)
535 Dumper->printMipsPLTGOT();
536 if (opts::MipsABIFlags)
537 Dumper->printMipsABIFlags();
538 if (opts::MipsReginfo)
539 Dumper->printMipsReginfo();
540 if (opts::MipsOptions)
541 Dumper->printMipsOptions();
542 if (opts::CheriCapRelocs)
543 Dumper->printCheriCapRelocs();
544 if (opts::CheriCapTable)
545 Dumper->printCheriCapTable();
546 if (opts::CheriCapTableMapping)
547 Dumper->printCheriCapTableMapping();
548 }
549 if (opts::SectionGroups)
550 Dumper->printGroupSections();
551 if (opts::HashHistogram)
552 Dumper->printHashHistogram();
553 if (opts::CGProfile)
554 Dumper->printCGProfile();
555 if (opts::Addrsig)
556 Dumper->printAddrsig();
557 if (opts::Notes)
558 Dumper->printNotes();
559 }
560 if (Obj->isCOFF()) {
561 if (opts::COFFImports)
562 Dumper->printCOFFImports();
563 if (opts::COFFExports)
564 Dumper->printCOFFExports();
565 if (opts::COFFDirectives)
566 Dumper->printCOFFDirectives();
567 if (opts::COFFBaseRelocs)
568 Dumper->printCOFFBaseReloc();
569 if (opts::COFFDebugDirectory)
570 Dumper->printCOFFDebugDirectory();
571 if (opts::COFFResources)
572 Dumper->printCOFFResources();
573 if (opts::COFFLoadConfig)
574 Dumper->printCOFFLoadConfig();
575 if (opts::Addrsig)
576 Dumper->printAddrsig();
577 if (opts::CodeView)
578 Dumper->printCodeViewDebugInfo();
579 if (opts::CodeViewMergedTypes)
580 Dumper->mergeCodeViewTypes(CVTypes.IDTable, CVTypes.TypeTable,
581 CVTypes.GlobalIDTable, CVTypes.GlobalTypeTable,
582 opts::CodeViewEnableGHash);
583 }
584 if (Obj->isMachO()) {
585 if (opts::MachODataInCode)
586 Dumper->printMachODataInCode();
587 if (opts::MachOIndirectSymbols)
588 Dumper->printMachOIndirectSymbols();
589 if (opts::MachOLinkerOptions)
590 Dumper->printMachOLinkerOptions();
591 if (opts::MachOSegment)
592 Dumper->printMachOSegment();
593 if (opts::MachOVersionMin)
594 Dumper->printMachOVersionMin();
595 if (opts::MachODysymtab)
596 Dumper->printMachODysymtab();
597 }
598 if (opts::PrintStackMap)
599 Dumper->printStackMap();
600}

602/// Dumps each object file in \a Arc;
603static void dumpArchive(const Archive *Arc, ScopedPrinter &Writer) {
604 Error Err = Error::success();
605 for (auto &Child : Arc->children(Err)) {
606 Expected<std::unique_ptr<Binary>> ChildOrErr = Child.getAsBinary();
607 if (!ChildOrErr) {
608 if (auto E = isNotObjectErrorInvalidFileType(ChildOrErr.takeError())) {
609 reportError(Arc->getFileName(), std::move(E));
610 }
611 continue;
612 }
613 if (ObjectFile *Obj = dyn_cast<ObjectFile>(&*ChildOrErr.get()))
614 dumpObject(Obj, Writer);
615 else if (COFFImportFile *Imp = dyn_cast<COFFImportFile>(&*ChildOrErr.get()))
616 dumpCOFFImportFile(Imp, Writer);
617 else
618 reportError(Arc->getFileName(), readobj_error::unrecognized_file_format);
619 }
620 if (Err)
621 reportError(Arc->getFileName(), std::move(Err));
622}

624/// Dumps each object file in \a MachO Universal Binary;
625static void dumpMachOUniversalBinary(const MachOUniversalBinary *UBinary,
626 ScopedPrinter &Writer) {
627 for (const MachOUniversalBinary::ObjectForArch &Obj : UBinary->objects()) {
628 Expected<std::unique_ptr<MachOObjectFile>> ObjOrErr = Obj.getAsObjectFile();
629 if (ObjOrErr)
630 dumpObject(&*ObjOrErr.get(), Writer);
631 else if (auto E = isNotObjectErrorInvalidFileType(ObjOrErr.takeError())) {
632 reportError(UBinary->getFileName(), ObjOrErr.takeError());
633 }
634 else if (Expected<std::unique_ptr<Archive>> AOrErr = Obj.getAsArchive())
635 dumpArchive(&*AOrErr.get(), Writer);
636 }
637}

648/// Opens \a File and dumps it.
649static void dumpInput(StringRef File, ScopedPrinter &Writer) {
650 // Attempt to open the binary.
651 Expected<OwningBinary<Binary>> BinaryOrErr = createBinary(File);
652 if (!BinaryOrErr)
653 reportError(File, BinaryOrErr.takeError());
654 Binary &Binary = *BinaryOrErr.get().getBinary();
655
656 if (Archive *Arc = dyn_cast<Archive>(&Binary))
657 dumpArchive(Arc, Writer);
658 else if (MachOUniversalBinary *UBinary =
659 dyn_cast<MachOUniversalBinary>(&Binary))
660 dumpMachOUniversalBinary(UBinary, Writer);
661 else if (ObjectFile *Obj = dyn_cast<ObjectFile>(&Binary))
662 dumpObject(Obj, Writer);
663 else if (COFFImportFile *Import = dyn_cast<COFFImportFile>(&Binary))
664 dumpCOFFImportFile(Import, Writer);
665 else if (WindowsResource *WinRes = dyn_cast<WindowsResource>(&Binary))
666 dumpWindowsResourceFile(WinRes, Writer);
667 else
668 reportError(File, readobj_error::unrecognized_file_format);
669
670 CVTypes.Binaries.push_back(std::move(*BinaryOrErr));
671}

673/// Registers aliases that should only be allowed by readobj.
674static void registerReadobjAliases() {
675 // -s has meant --sections for a very long time in llvm-readobj despite
676 // meaning --symbols in readelf.
677 static cl::alias SectionsShort("s", cl::desc("Alias for --section-headers"),
678 cl::aliasopt(opts::SectionHeaders),
679 cl::NotHidden);
680
681 // Only register -t in llvm-readobj, as readelf reserves it for
682 // --section-details (not implemented yet).
683 static cl::alias SymbolsShort("t", cl::desc("Alias for --symbols"),
684 cl::aliasopt(opts::Symbols), cl::NotHidden);
685
686 // The following two-letter aliases are only provided for readobj, as readelf
687 // allows single-letter args to be grouped together.
688 static cl::alias SectionRelocationsShort(
689 "sr", cl::desc("Alias for --section-relocations"),
690 cl::aliasopt(opts::SectionRelocations));
691 static cl::alias SectionDataShort("sd", cl::desc("Alias for --section-data"),
692 cl::aliasopt(opts::SectionData));
693 static cl::alias SectionSymbolsShort("st",
694 cl::desc("Alias for --section-symbols"),
695 cl::aliasopt(opts::SectionSymbols));
696 static cl::alias DynamicSymbolsShort("dt",
697 cl::desc("Alias for --dyn-symbols"),
698 cl::aliasopt(opts::DynamicSymbols));
699}
700
701/// Registers aliases that should only be allowed by readelf.
702static void registerReadelfAliases() {
703 // -s is here because for readobj it means --sections.
704 static cl::alias SymbolsShort("s", cl::desc("Alias for --symbols"),
705 cl::aliasopt(opts::Symbols), cl::NotHidden,
706 cl::Grouping);
707
708 // Allow all single letter flags to be grouped together.
709 for (auto &OptEntry : cl::getRegisteredOptions()) {
710 StringRef ArgName = OptEntry.getKey();
711 cl::Option *Option = OptEntry.getValue();
712 if (ArgName.size() == 1)
713 apply(Option, cl::Grouping);
714 }
715}

717int main(int argc, const char *argv[]) {
718 InitLLVM X(argc, argv);
719
720 // Register the target printer for --version.
721 cl::AddExtraVersionPrinter(TargetRegistry::printRegisteredTargetsForVersion);
722
723 if (sys::path::stem(argv[0]).contains("readelf")) {
724 opts::Output = opts::GNU;
725 registerReadelfAliases();
726 } else {
727 registerReadobjAliases();
728 }
729
730 cl::ParseCommandLineOptions(argc, argv, "LLVM Object Reader\n");
731
732 if (opts::All) {
733 opts::FileHeaders = true;
734 opts::ProgramHeaders = true;
735 opts::SectionHeaders = true;
736 opts::Symbols = true;
737 opts::Relocations = true;
738 opts::DynamicTable = true;
739 opts::Notes = true;
740 opts::VersionInfo = true;
741 opts::UnwindInfo = true;
742 opts::SectionGroups = true;
743 opts::HashHistogram = true;
744 opts::CheriCapRelocs = true;
745 opts::CheriCapTable = true;
746 }
747
748 if (opts::Headers) {
749 opts::FileHeaders = true;
750 opts::ProgramHeaders = true;
751 opts::SectionHeaders = true;
752 }
753
754 // Default to stdin if no filename is specified.
755 if (opts::InputFilenames.empty())
756 opts::InputFilenames.push_back("-");
757
758 ScopedPrinter Writer(fouts());
759 for (const std::string &I : opts::InputFilenames)
760 dumpInput(I, Writer);
761
762 if (opts::CodeViewMergedTypes) {
763 if (opts::CodeViewEnableGHash)
764 dumpCodeViewMergedTypes(Writer, CVTypes.GlobalIDTable.records(),
765 CVTypes.GlobalTypeTable.records());
766 else
767 dumpCodeViewMergedTypes(Writer, CVTypes.IDTable.records(),
768 CVTypes.TypeTable.records());
769 }
770
771 return 0;
772}
