193class COFFObjectDumpDelegate : public SymbolDumpDelegate {
194public:
195 COFFObjectDumpDelegate(COFFDumper &CD, const SectionRef &SR,
196 const COFFObjectFile *Obj, StringRef SectionContents)
197 : CD(CD), SR(SR), SectionContents(SectionContents) {
198 Sec = Obj->getCOFFSection(SR);
199 }
200
201 uint32_t getRecordOffset(BinaryStreamReader Reader) override {
202 ArrayRef<uint8_t> Data;
203 if (auto EC = Reader.readLongestContiguousChunk(Data)) {
204 llvm::consumeError(std::move(EC));
205 return 0;
206 }
207 return Data.data() - SectionContents.bytes_begin();
208 }
209
210 void printRelocatedField(StringRef Label, uint32_t RelocOffset,
211 uint32_t Offset, StringRef *RelocSym) override {
212 CD.printRelocatedField(Label, Sec, RelocOffset, Offset, RelocSym);
213 }
214
215 void printBinaryBlockWithRelocs(StringRef Label,
216 ArrayRef<uint8_t> Block) override {
217 StringRef SBlock(reinterpret_cast<const char *>(Block.data()),
218 Block.size());
219 if (opts::CodeViewSubsectionBytes)
220 CD.printBinaryBlockWithRelocs(Label, SR, SectionContents, SBlock);
221 }
222
223 StringRef getFileNameForFileOffset(uint32_t FileOffset) override {
224 return CD.getFileNameForFileOffset(FileOffset);
225 }
226
227 DebugStringTableSubsectionRef getStringTable() override {
228 return CD.CVStringTable;
229 }

231private:

242std::error_code createCOFFDumper(const object::ObjectFile *Obj,
243 ScopedPrinter &Writer,
244 std::unique_ptr<ObjDumper> &Result) {
245 const COFFObjectFile *COFFObj = dyn_cast<COFFObjectFile>(Obj);
246 if (!COFFObj)
247 return readobj_error::unsupported_obj_file_format;
248
249 Result.reset(new COFFDumper(COFFObj, Writer));
250 return readobj_error::success;
251}

255// Given a section and an offset into this section the function returns the
256// symbol used for the relocation at the offset.
257std::error_code COFFDumper::resolveSymbol(const coff_section *Section,
258 uint64_t Offset, SymbolRef &Sym) {
259 cacheRelocations();
260 const auto &Relocations = RelocMap[Section];
261 auto SymI = Obj->symbol_end();
262 for (const auto &Relocation : Relocations) {
263 uint64_t RelocationOffset = Relocation.getOffset();
264
265 if (RelocationOffset == Offset) {
266 SymI = Relocation.getSymbol();
267 break;
268 }
269 }
270 if (SymI == Obj->symbol_end())
271 return readobj_error::unknown_symbol;
272 Sym = *SymI;
273 return readobj_error::success;
274}

276// Given a section and an offset into this section the function returns the name
277// of the symbol used for the relocation at the offset.
278std::error_code COFFDumper::resolveSymbolName(const coff_section *Section,
279 uint64_t Offset,
280 StringRef &Name) {
281 SymbolRef Symbol;
282 if (std::error_code EC = resolveSymbol(Section, Offset, Symbol))
283 return EC;
284 Expected<StringRef> NameOrErr = Symbol.getName();
285 if (!NameOrErr)
286 return errorToErrorCode(NameOrErr.takeError());
287 Name = *NameOrErr;
288 return std::error_code();
289}

291// Helper for when you have a pointer to real data and you want to know about
292// relocations against it.
293std::error_code COFFDumper::resolveSymbolName(const coff_section *Section,
294 StringRef SectionContents,
295 const void *RelocPtr,
296 StringRef &Name) {
297 assert(SectionContents.data() < RelocPtr &&
298 RelocPtr < SectionContents.data() + SectionContents.size() &&
299 "pointer to relocated object is not in section");
300 uint64_t Offset = ptrdiff_t(reinterpret_cast<const char *>(RelocPtr) -
301 SectionContents.data());
302 return resolveSymbolName(Section, Offset, Name);
303}

305void COFFDumper::printRelocatedField(StringRef Label, const coff_section *Sec,
306 uint32_t RelocOffset, uint32_t Offset,
307 StringRef *RelocSym) {
308 StringRef SymStorage;
309 StringRef &Symbol = RelocSym ? *RelocSym : SymStorage;
310 if (!resolveSymbolName(Sec, RelocOffset, Symbol))
311 W.printSymbolOffset(Label, Symbol, Offset);
312 else
313 W.printHex(Label, RelocOffset);
314}

316void COFFDumper::printBinaryBlockWithRelocs(StringRef Label,
317 const SectionRef &Sec,
318 StringRef SectionContents,
319 StringRef Block) {
320 W.printBinaryBlock(Label, Block);
321
322 assert(SectionContents.begin() < Block.begin() &&
323 SectionContents.end() >= Block.end() &&
324 "Block is not contained in SectionContents");
325 uint64_t OffsetStart = Block.data() - SectionContents.data();
326 uint64_t OffsetEnd = OffsetStart + Block.size();
327
328 W.flush();
329 cacheRelocations();
330 ListScope D(W, "BlockRelocations");
331 const coff_section *Section = Obj->getCOFFSection(Sec);
332 const auto &Relocations = RelocMap[Section];
333 for (const auto &Relocation : Relocations) {
334 uint64_t RelocationOffset = Relocation.getOffset();
335 if (OffsetStart <= RelocationOffset && RelocationOffset < OffsetEnd)
336 printRelocation(Sec, Relocation, OffsetStart);
337 }
338}

905void COFFDumper::initializeFileAndStringTables(BinaryStreamReader &Reader) {
906 while (Reader.bytesRemaining() > 0 &&
907 (!CVFileChecksumTable.valid() || !CVStringTable.valid())) {
908 // The section consists of a number of subsection in the following format:
909 // |SubSectionType|SubSectionSize|Contents...|
910 uint32_t SubType, SubSectionSize;
911 error(Reader.readInteger(SubType));
912 error(Reader.readInteger(SubSectionSize));
913
914 StringRef Contents;
915 error(Reader.readFixedString(Contents, SubSectionSize));
916
917 BinaryStreamRef ST(Contents, support::little);
918 switch (DebugSubsectionKind(SubType)) {
919 case DebugSubsectionKind::FileChecksums:
920 error(CVFileChecksumTable.initialize(ST));
921 break;
922 case DebugSubsectionKind::StringTable:
923 error(CVStringTable.initialize(ST));
924 break;
925 default:
926 break;
927 }
928
929 uint32_t PaddedSize = alignTo(SubSectionSize, 4);
930 error(Reader.skip(PaddedSize - SubSectionSize));
931 }

934void COFFDumper::printCodeViewSymbolSection(StringRef SectionName,
935 const SectionRef &Section) {
936 StringRef SectionContents = unwrapOrError(Section.getContents());
937 StringRef Data = SectionContents;
938
939 SmallVector<StringRef, 10> FunctionNames;
940 StringMap<StringRef> FunctionLineTables;
941
942 ListScope D(W, "CodeViewDebugInfo");
943 // Print the section to allow correlation with printSectionHeaders.
944 W.printNumber("Section", SectionName, Obj->getSectionID(Section));
945
946 uint32_t Magic;
947 error(consume(Data, Magic));
948 W.printHex("Magic", Magic);
949 if (Magic != COFF::DEBUG_SECTION_MAGIC)
950 return error(object_error::parse_failed);
951
952 BinaryStreamReader FSReader(Data, support::little);
953 initializeFileAndStringTables(FSReader);
954
955 // TODO: Convert this over to using ModuleSubstreamVisitor.
956 while (!Data.empty()) {
957 // The section consists of a number of subsection in the following format:
958 // |SubSectionType|SubSectionSize|Contents...|
959 uint32_t SubType, SubSectionSize;
960 error(consume(Data, SubType));
961 error(consume(Data, SubSectionSize));
962
963 ListScope S(W, "Subsection");
964 W.printEnum("SubSectionType", SubType, makeArrayRef(SubSectionTypes));
965 W.printHex("SubSectionSize", SubSectionSize);
966
967 // Get the contents of the subsection.
968 if (SubSectionSize > Data.size())
969 return error(object_error::parse_failed);
970 StringRef Contents = Data.substr(0, SubSectionSize);
971
972 // Add SubSectionSize to the current offset and align that offset to find
973 // the next subsection.
974 size_t SectionOffset = Data.data() - SectionContents.data();
975 size_t NextOffset = SectionOffset + SubSectionSize;
976 NextOffset = alignTo(NextOffset, 4);
977 if (NextOffset > SectionContents.size())
978 return error(object_error::parse_failed);
979 Data = SectionContents.drop_front(NextOffset);
980
981 // Optionally print the subsection bytes in case our parsing gets confused
982 // later.
983 if (opts::CodeViewSubsectionBytes)
984 printBinaryBlockWithRelocs("SubSectionContents", Section, SectionContents,
985 Contents);
986
987 switch (DebugSubsectionKind(SubType)) {
988 case DebugSubsectionKind::Symbols:
989 printCodeViewSymbolsSubsection(Contents, Section, SectionContents);
990 break;
991
992 case DebugSubsectionKind::InlineeLines:
993 printCodeViewInlineeLines(Contents);
994 break;
995
996 case DebugSubsectionKind::FileChecksums:
997 printCodeViewFileChecksums(Contents);
998 break;
999
1000 case DebugSubsectionKind::Lines: {
1001 // Holds a PC to file:line table. Some data to parse this subsection is
1002 // stored in the other subsections, so just check sanity and store the
1003 // pointers for deferred processing.
1004
1005 if (SubSectionSize < 12) {
1006 // There should be at least three words to store two function
1007 // relocations and size of the code.
1008 error(object_error::parse_failed);
1009 return;
1010 }
1011
1012 StringRef LinkageName;
1013 error(resolveSymbolName(Obj->getCOFFSection(Section), SectionOffset,
1014 LinkageName));
1015 W.printString("LinkageName", LinkageName);
1016 if (FunctionLineTables.count(LinkageName) != 0) {
1017 // Saw debug info for this function already?
1018 error(object_error::parse_failed);
1019 return;
1020 }
1021
1022 FunctionLineTables[LinkageName] = Contents;
1023 FunctionNames.push_back(LinkageName);
1024 break;
1025 }
1026 case DebugSubsectionKind::FrameData: {
1027 // First four bytes is a relocation against the function.
1028 BinaryStreamReader SR(Contents, llvm::support::little);
1029
1030 DebugFrameDataSubsectionRef FrameData;
1031 error(FrameData.initialize(SR));
1032
1033 StringRef LinkageName;
1034 error(resolveSymbolName(Obj->getCOFFSection(Section), SectionContents,
1035 FrameData.getRelocPtr(), LinkageName));
1036 W.printString("LinkageName", LinkageName);
1037
1038 // To find the active frame description, search this array for the
1039 // smallest PC range that includes the current PC.
1040 for (const auto &FD : FrameData) {
1041 StringRef FrameFunc = error(CVStringTable.getString(FD.FrameFunc));
1042
1043 DictScope S(W, "FrameData");
1044 W.printHex("RvaStart", FD.RvaStart);
1045 W.printHex("CodeSize", FD.CodeSize);
1046 W.printHex("LocalSize", FD.LocalSize);
1047 W.printHex("ParamsSize", FD.ParamsSize);
1048 W.printHex("MaxStackSize", FD.MaxStackSize);
1049 W.printHex("PrologSize", FD.PrologSize);
1050 W.printHex("SavedRegsSize", FD.SavedRegsSize);
1051 W.printFlags("Flags", FD.Flags, makeArrayRef(FrameDataFlags));
1052
1053 // The FrameFunc string is a small RPN program. It can be broken up into
1054 // statements that end in the '=' operator, which assigns the value on
1055 // the top of the stack to the previously pushed variable. Variables can
1056 // be temporary values ($T0) or physical registers ($esp). Print each
1057 // assignment on its own line to make these programs easier to read.
1058 {
1059 ListScope FFS(W, "FrameFunc");
1060 while (!FrameFunc.empty()) {
1061 size_t EqOrEnd = FrameFunc.find('=');
1062 if (EqOrEnd == StringRef::npos)
1063 EqOrEnd = FrameFunc.size();
1064 else
1065 ++EqOrEnd;
1066 StringRef Stmt = FrameFunc.substr(0, EqOrEnd);
1067 W.printString(Stmt);
1068 FrameFunc = FrameFunc.drop_front(EqOrEnd).trim();
1069 }
1070 }
1071 }
1072 break;
1073 }
1074
1075 // Do nothing for unrecognized subsections.
1076 default:
1077 break;
1078 }
1079 W.flush();
1080 }
1081
1082 // Dump the line tables now that we've read all the subsections and know all
1083 // the required information.
1084 for (unsigned I = 0, E = FunctionNames.size(); I != E; ++I) {
1085 StringRef Name = FunctionNames[I];
1086 ListScope S(W, "FunctionLineTable");
1087 W.printString("LinkageName", Name);
1088
1089 BinaryStreamReader Reader(FunctionLineTables[Name], support::little);
1090
1091 DebugLinesSubsectionRef LineInfo;
1092 error(LineInfo.initialize(Reader));
1093
1094 W.printHex("Flags", LineInfo.header()->Flags);
1095 W.printHex("CodeSize", LineInfo.header()->CodeSize);
1096 for (const auto &Entry : LineInfo) {
1097
1098 ListScope S(W, "FilenameSegment");
1099 printFileNameForOffset("Filename", Entry.NameIndex);
1100 uint32_t ColumnIndex = 0;
1101 for (const auto &Line : Entry.LineNumbers) {
1102 if (Line.Offset >= LineInfo.header()->CodeSize) {
1103 error(object_error::parse_failed);
1104 return;
1105 }
1106
1107 std::string PC = formatv("+{0:X}", uint32_t(Line.Offset));
1108 ListScope PCScope(W, PC);
1109 codeview::LineInfo LI(Line.Flags);
1110
1111 if (LI.isAlwaysStepInto())
1112 W.printString("StepInto", StringRef("Always"));
1113 else if (LI.isNeverStepInto())
1114 W.printString("StepInto", StringRef("Never"));
1115 else
1116 W.printNumber("LineNumberStart", LI.getStartLine());
1117 W.printNumber("LineNumberEndDelta", LI.getLineDelta());
1118 W.printBoolean("IsStatement", LI.isStatement());
1119 if (LineInfo.hasColumnInfo()) {
1120 W.printNumber("ColStart", Entry.Columns[ColumnIndex].StartColumn);
1121 W.printNumber("ColEnd", Entry.Columns[ColumnIndex].EndColumn);
1122 ++ColumnIndex;
1123 }
1124 }
1125 }
1126 }

1404void COFFDumper::printSymbol(const SymbolRef &Sym) {
1405 DictScope D(W, "Symbol");
1406
1407 COFFSymbolRef Symbol = Obj->getCOFFSymbol(Sym);
1408 const coff_section *Section;
1409 if (std::error_code EC = Obj->getSection(Symbol.getSectionNumber(), Section)) {
1410 W.startLine() << "Invalid section number: " << EC.message() << "\n";
1411 W.flush();
1412 return;
1413 }
1414
1415 StringRef SymbolName;
1416 if (Obj->getSymbolName(Symbol, SymbolName))
1417 SymbolName = "";
1418
1419 StringRef SectionName;
1420 if (Expected<StringRef> NameOrErr =
1421 getSectionName(Obj, Symbol.getSectionNumber(), Section))
1422 SectionName = *NameOrErr;
1423
1424 W.printString("Name", SymbolName);
1425 W.printNumber("Value", Symbol.getValue());
1426 W.printNumber("Section", SectionName, Symbol.getSectionNumber());
1427 W.printEnum ("BaseType", Symbol.getBaseType(), makeArrayRef(ImageSymType));
1428 W.printEnum ("ComplexType", Symbol.getComplexType(),
1429 makeArrayRef(ImageSymDType));
1430 W.printEnum ("StorageClass", Symbol.getStorageClass(),
1431 makeArrayRef(ImageSymClass));
1432 W.printNumber("AuxSymbolCount", Symbol.getNumberOfAuxSymbols());
1433
1434 for (uint8_t I = 0; I < Symbol.getNumberOfAuxSymbols(); ++I) {
1435 if (Symbol.isFunctionDefinition()) {
1436 const coff_aux_function_definition *Aux;
1437 error(getSymbolAuxData(Obj, Symbol, I, Aux));
1438
1439 DictScope AS(W, "AuxFunctionDef");
1440 W.printNumber("TagIndex", Aux->TagIndex);
1441 W.printNumber("TotalSize", Aux->TotalSize);
1442 W.printHex("PointerToLineNumber", Aux->PointerToLinenumber);
1443 W.printHex("PointerToNextFunction", Aux->PointerToNextFunction);
1444
1445 } else if (Symbol.isAnyUndefined()) {
1446 const coff_aux_weak_external *Aux;
1447 error(getSymbolAuxData(Obj, Symbol, I, Aux));
1448
1449 Expected<COFFSymbolRef> Linked = Obj->getSymbol(Aux->TagIndex);
1450 StringRef LinkedName;
1451 std::error_code EC = errorToErrorCode(Linked.takeError());
1452 if (EC || (EC = Obj->getSymbolName(*Linked, LinkedName))) {
1453 LinkedName = "";
1454 error(EC);
1455 }
1456
1457 DictScope AS(W, "AuxWeakExternal");
1458 W.printNumber("Linked", LinkedName, Aux->TagIndex);
1459 W.printEnum ("Search", Aux->Characteristics,
1460 makeArrayRef(WeakExternalCharacteristics));
1461
1462 } else if (Symbol.isFileRecord()) {
1463 const char *FileName;
1464 error(getSymbolAuxData(Obj, Symbol, I, FileName));
1465
1466 DictScope AS(W, "AuxFileRecord");
1467
1468 StringRef Name(FileName, Symbol.getNumberOfAuxSymbols() *
1469 Obj->getSymbolTableEntrySize());
1470 W.printString("FileName", Name.rtrim(StringRef("\0", 1)));
1471 break;
1472 } else if (Symbol.isSectionDefinition()) {
1473 const coff_aux_section_definition *Aux;
1474 error(getSymbolAuxData(Obj, Symbol, I, Aux));
1475
1476 int32_t AuxNumber = Aux->getNumber(Symbol.isBigObj());
1477
1478 DictScope AS(W, "AuxSectionDef");
1479 W.printNumber("Length", Aux->Length);
1480 W.printNumber("RelocationCount", Aux->NumberOfRelocations);
1481 W.printNumber("LineNumberCount", Aux->NumberOfLinenumbers);
1482 W.printHex("Checksum", Aux->CheckSum);
1483 W.printNumber("Number", AuxNumber);
1484 W.printEnum("Selection", Aux->Selection, makeArrayRef(ImageCOMDATSelect));
1485
1486 if (Section && Section->Characteristics & COFF::IMAGE_SCN_LNK_COMDAT
1487 && Aux->Selection == COFF::IMAGE_COMDAT_SELECT_ASSOCIATIVE) {
1488 const coff_section *Assoc;
1489 StringRef AssocName = "";
1490 if (std::error_code EC = Obj->getSection(AuxNumber, Assoc))
1491 error(EC);
1492 Expected<StringRef> Res = getSectionName(Obj, AuxNumber, Assoc);
1493 if (!Res)
1494 error(Res.takeError());
1495 AssocName = *Res;
1496
1497 W.printNumber("AssocSection", AssocName, AuxNumber);
1498 }
1499 } else if (Symbol.isCLRToken()) {
1500 const coff_aux_clr_token *Aux;
1501 error(getSymbolAuxData(Obj, Symbol, I, Aux));
1502
1503 Expected<COFFSymbolRef> ReferredSym =
1504 Obj->getSymbol(Aux->SymbolTableIndex);
1505 StringRef ReferredName;
1506 std::error_code EC = errorToErrorCode(ReferredSym.takeError());
1507 if (EC || (EC = Obj->getSymbolName(*ReferredSym, ReferredName))) {
1508 ReferredName = "";
1509 error(EC);
1510 }
1511
1512 DictScope AS(W, "AuxCLRToken");
1513 W.printNumber("AuxType", Aux->AuxType);
1514 W.printNumber("Reserved", Aux->Reserved);
1515 W.printNumber("SymbolTableIndex", ReferredName, Aux->SymbolTableIndex);
1516
1517 } else {
1518 W.startLine() << "<unhandled auxiliary record>\n";
1519 }
1520 }
1521}

1721COFFDumper::countTotalTableEntries(ResourceSectionRef RSF,
1722 const coff_resource_dir_table &Table,
1723 StringRef Level) {

1726 i++) {

1728 if (Entry.Offset.isSubDir()) {

1731 NextLevel = "Language";
1732 else
1733 NextLevel = "Name";

1743void COFFDumper::printResourceDirectoryTable(
1744 ResourceSectionRef RSF, const coff_resource_dir_table &Table,
1745 StringRef Level) {
1746
1747 W.printNumber("Number of String Entries", Table.NumberOfNameEntries);
1748 W.printNumber("Number of ID Entries", Table.NumberOfIDEntries);
1749
1750 // Iterate through level in resource directory tree.
1751 for (int i = 0; i < Table.NumberOfNameEntries + Table.NumberOfIDEntries;
1752 i++) {
1753 auto Entry = unwrapOrError(getResourceDirectoryTableEntry(Table, i));
1754 StringRef Name;
1755 SmallString<20> IDStr;
1756 raw_svector_ostream OS(IDStr);
1757 if (i < Table.NumberOfNameEntries) {
1758 ArrayRef<UTF16> RawEntryNameString =
1759 unwrapOrError(RSF.getEntryNameString(Entry));
1760 std::vector<UTF16> EndianCorrectedNameString;
1761 if (llvm::sys::IsBigEndianHost) {
1762 EndianCorrectedNameString.resize(RawEntryNameString.size() + 1);
1763 std::copy(RawEntryNameString.begin(), RawEntryNameString.end(),
1764 EndianCorrectedNameString.begin() + 1);
1765 EndianCorrectedNameString[0] = UNI_UTF16_BYTE_ORDER_MARK_SWAPPED;
1766 RawEntryNameString = makeArrayRef(EndianCorrectedNameString);
1767 }
1768 std::string EntryNameString;
1769 if (!llvm::convertUTF16ToUTF8String(RawEntryNameString, EntryNameString))
1770 error(object_error::parse_failed);
1771 OS << ": ";
1772 OS << EntryNameString;
1773 } else {
1774 if (Level == "Type") {
1775 OS << ": ";
1776 printResourceTypeName(Entry.Identifier.ID, OS);
1777 IDStr = IDStr.slice(0, IDStr.find_first_of(")", 0) + 1);
1778 } else {
1779 OS << ": (ID " << Entry.Identifier.ID << ")";
1780 }
1781 }
1782 Name = StringRef(IDStr);
1783 ListScope ResourceType(W, Level.str() + Name.str());
1784 if (Entry.Offset.isSubDir()) {
1785 W.printHex("Table Offset", Entry.Offset.value());
1786 StringRef NextLevel;
1787 if (Level == "Name")
1788 NextLevel = "Language";
1789 else
1790 NextLevel = "Name";
1791 auto &NextTable = unwrapOrError(RSF.getEntrySubDir(Entry));
1792 printResourceDirectoryTable(RSF, NextTable, NextLevel);
1793 } else {
1794 W.printHex("Entry Offset", Entry.Offset.value());
1795 char FormattedTime[20] = {};
1796 time_t TDS = time_t(Table.TimeDateStamp);
1797 strftime(FormattedTime, 20, "%Y-%m-%d %H:%M:%S", gmtime(&TDS));
1798 W.printHex("Time/Date Stamp", FormattedTime, Table.TimeDateStamp);
1799 W.printNumber("Major Version", Table.MajorVersion);
1800 W.printNumber("Minor Version", Table.MinorVersion);
1801 W.printNumber("Characteristics", Table.Characteristics);
1802 }
1803 }

1841void COFFDumper::printAddrsig() {
1842 object::SectionRef AddrsigSection;
1843 for (auto Sec : Obj->sections()) {
1844 StringRef Name;
1845 Sec.getName(Name);
1846 if (Name == ".llvm_addrsig") {
1847 AddrsigSection = Sec;
1848 break;
1849 }
1850 }
1851
1852 if (AddrsigSection == object::SectionRef())
1853 return;
1854
1855 StringRef AddrsigContents = unwrapOrError(AddrsigSection.getContents());
1856 ArrayRef<uint8_t> AddrsigContentsArray(AddrsigContents.bytes_begin(),
1857 AddrsigContents.size());
1858
1859 ListScope L(W, "Addrsig");
1860 const uint8_t *Cur = AddrsigContents.bytes_begin();
1861 const uint8_t *End = AddrsigContents.bytes_end();
1862 while (Cur != End) {
1863 unsigned Size;
1864 const char *Err;
1865 uint64_t SymIndex = decodeULEB128(Cur, &Size, End, &Err);
1866 if (Err)
1867 reportError(Err);
1868
1869 Expected<COFFSymbolRef> Sym = Obj->getSymbol(SymIndex);
1870 StringRef SymName;
1871 std::error_code EC = errorToErrorCode(Sym.takeError());
1872 if (EC || (EC = Obj->getSymbolName(*Sym, SymName))) {
1873 SymName = "";
1874 error(EC);
1875 }
1876
1877 W.printNumber("Sym", SymName, SymIndex);
1878 Cur += Size;
1879 }
