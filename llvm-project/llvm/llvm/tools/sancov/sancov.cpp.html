218// ----------- Coverage I/O ----------
219template <typename T>
220static void readInts(const char *Start, const char *End,
221 std::set<uint64_t> *Ints) {
222 const T *S = reinterpret_cast<const T *>(Start);
223 const T *E = reinterpret_cast<const T *>(End);
224 std::copy(S, E, std::inserter(*Ints, Ints->end()));
225}

227ErrorOr<std::unique_ptr<RawCoverage>>
228RawCoverage::read(const std::string &FileName) {
229 ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
230 MemoryBuffer::getFile(FileName);
231 if (!BufOrErr)
232 return BufOrErr.getError();
233 std::unique_ptr<MemoryBuffer> Buf = std::move(BufOrErr.get());
234 if (Buf->getBufferSize() < 8) {
235 errs() << "File too small (<8): " << Buf->getBufferSize() << '\n';
236 return make_error_code(errc::illegal_byte_sequence);
237 }
238 const FileHeader *Header =
239 reinterpret_cast<const FileHeader *>(Buf->getBufferStart());
240
241 if (Header->Magic != BinCoverageMagic) {
242 errs() << "Wrong magic: " << Header->Magic << '\n';
243 return make_error_code(errc::illegal_byte_sequence);
244 }
245
246 auto Addrs = llvm::make_unique<std::set<uint64_t>>();
247
248 switch (Header->Bitness) {
249 case Bitness64:
250 readInts<uint64_t>(Buf->getBufferStart() + 8, Buf->getBufferEnd(),
251 Addrs.get());
252 break;
253 case Bitness32:
254 readInts<uint32_t>(Buf->getBufferStart() + 8, Buf->getBufferEnd(),
255 Addrs.get());
256 break;
257 default:
258 errs() << "Unsupported bitness: " << Header->Bitness << '\n';
259 return make_error_code(errc::illegal_byte_sequence);
260 }
261
262 return std::unique_ptr<RawCoverage>(new RawCoverage(std::move(Addrs)));
263}

293 // Helper RAII class to output JSON objects.
294 class Object {
295 public:
296 Object(JSONWriter *W, raw_ostream &OS) : W(W), OS(OS) {
297 OS << "{";
298 W->Indent++;
299 }
300 Object(const Object &) = delete;
301 ~Object() {
302 W->Indent--;
303 OS << "\n";
304 W->indent();
305 OS << "}";
306 }
307
308 void key(StringRef Key) {
309 Index++;
310 if (Index > 0)
311 OS << ",";
312 OS << "\n";
313 W->indent();
314 printJSONStringLiteral(Key, OS);
315 OS << " : ";
316 }
317
318 private:
319 JSONWriter *W;
320 raw_ostream &OS;
321 int Index = -1;
322 };
323
324 std::unique_ptr<Object> object() { return make_unique<Object>(this, OS); }

326 // Helper RAII class to output JSON arrays.

328 public:

334 if (Index > 0)
335 OS << ", ";
336 }
337
338 private:

341 };

343 std::unique_ptr<Array> array() { return make_unique<Array>(OS); }

347
348 static void printJSONStringLiteral(StringRef S, raw_ostream &OS) {
349 if (S.find('"') == std::string::npos) {
350 OS << "\"" << S << "\"";
351 return;
352 }
353 OS << "\"";
354 for (char Ch : S.bytes()) {
355 if (Ch == '"')
356 OS << "\\";
357 OS << Ch;
358 }
359 OS << "\"";
360 }
361
362 raw_ostream &OS;
363 int Indent = 0;
364};

366// Output symbolized information for coverage points in JSON.
367// Format:
368// {
369// '<file_name>' : {
370// '<function_name>' : {
371// '<point_id'> : '<line_number>:'<column_number'.
372// ....
373// }
374// }
375// }
376static void operator<<(JSONWriter &W,
377 const std::vector<CoveragePoint> &Points) {
378 // Group points by file.
379 auto ByFile(W.object());
380 std::map<std::string, std::vector<const CoveragePoint *>> PointsByFile;
381 for (const auto &Point : Points) {
382 for (const DILineInfo &Loc : Point.Locs) {
383 PointsByFile[Loc.FileName].push_back(&Point);
384 }
385 }
386
387 for (const auto &P : PointsByFile) {
388 std::string FileName = P.first;
389 ByFile->key(FileName);
390
391 // Group points by function.
392 auto ByFn(W.object());
393 std::map<std::string, std::vector<const CoveragePoint *>> PointsByFn;
394 for (auto PointPtr : P.second) {
395 for (const DILineInfo &Loc : PointPtr->Locs) {
396 PointsByFn[Loc.FunctionName].push_back(PointPtr);
397 }
398 }
399
400 for (const auto &P : PointsByFn) {
401 std::string FunctionName = P.first;
402 std::set<std::string> WrittenIds;
403
404 ByFn->key(FunctionName);
405
406 // Output <point_id> : "<line>:<col>".
407 auto ById(W.object());
408 for (const CoveragePoint *Point : P.second) {
409 for (const auto &Loc : Point->Locs) {
410 if (Loc.FileName != FileName || Loc.FunctionName != FunctionName)
411 continue;
412 if (WrittenIds.find(Point->Id) != WrittenIds.end())
413 continue;
414
415 WrittenIds.insert(Point->Id);
416 ById->key(Point->Id);
417 W << (utostr(Loc.Line) + ":" + utostr(Loc.Column));
418 }
419 }
420 }
421 }
422}

450static std::string parseScalarString(yaml::Node *N) {
451 SmallString<64> StringStorage;
452 yaml::ScalarNode *S = dyn_cast<yaml::ScalarNode>(N);
453 failIf(!S, "expected string");
454 return S->getValue(StringStorage);

457std::unique_ptr<SymbolizedCoverage>
458SymbolizedCoverage::read(const std::string &InputFile) {
459 auto Coverage(make_unique<SymbolizedCoverage>());
460
461 std::map<std::string, CoveragePoint> Points;
462 ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
463 MemoryBuffer::getFile(InputFile);
464 failIfError(BufOrErr);
465
466 SourceMgr SM;
467 yaml::Stream S(**BufOrErr, SM);
468
469 yaml::document_iterator DI = S.begin();
470 failIf(DI == S.end(), "empty document: " + InputFile);
471 yaml::Node *Root = DI->getRoot();
472 failIf(!Root, "expecting root node: " + InputFile);
473 yaml::MappingNode *Top = dyn_cast<yaml::MappingNode>(Root);
474 failIf(!Top, "expecting mapping node: " + InputFile);
475
476 for (auto &KVNode : *Top) {
477 auto Key = parseScalarString(KVNode.getKey());
478
479 if (Key == "covered-points") {
480 yaml::SequenceNode *Points =
481 dyn_cast<yaml::SequenceNode>(KVNode.getValue());
482 failIf(!Points, "expected array: " + InputFile);
483
484 for (auto I = Points->begin(), E = Points->end(); I != E; ++I) {
485 Coverage->CoveredIds.insert(parseScalarString(&*I));
486 }
487 } else if (Key == "binary-hash") {
488 Coverage->BinaryHash = parseScalarString(KVNode.getValue());
489 } else if (Key == "point-symbol-info") {
490 yaml::MappingNode *PointSymbolInfo =
491 dyn_cast<yaml::MappingNode>(KVNode.getValue());
492 failIf(!PointSymbolInfo, "expected mapping node: " + InputFile);
493
494 for (auto &FileKVNode : *PointSymbolInfo) {
495 auto Filename = parseScalarString(FileKVNode.getKey());
496
497 yaml::MappingNode *FileInfo =
498 dyn_cast<yaml::MappingNode>(FileKVNode.getValue());
499 failIf(!FileInfo, "expected mapping node: " + InputFile);
500
501 for (auto &FunctionKVNode : *FileInfo) {
502 auto FunctionName = parseScalarString(FunctionKVNode.getKey());
503
504 yaml::MappingNode *FunctionInfo =
505 dyn_cast<yaml::MappingNode>(FunctionKVNode.getValue());
506 failIf(!FunctionInfo, "expected mapping node: " + InputFile);
507
508 for (auto &PointKVNode : *FunctionInfo) {
509 auto PointId = parseScalarString(PointKVNode.getKey());
510 auto Loc = parseScalarString(PointKVNode.getValue());
511
512 size_t ColonPos = Loc.find(':');
513 failIf(ColonPos == std::string::npos, "expected ':': " + InputFile);
514
515 auto LineStr = Loc.substr(0, ColonPos);
516 auto ColStr = Loc.substr(ColonPos + 1, Loc.size());
517
518 if (Points.find(PointId) == Points.end())
519 Points.insert(std::make_pair(PointId, CoveragePoint(PointId)));
520
521 DILineInfo LineInfo;
522 LineInfo.FileName = Filename;
523 LineInfo.FunctionName = FunctionName;
524 char *End;
525 LineInfo.Line = std::strtoul(LineStr.c_str(), &End, 10);
526 LineInfo.Column = std::strtoul(ColStr.c_str(), &End, 10);
527
528 CoveragePoint *CoveragePoint = &Points.find(PointId)->second;
529 CoveragePoint->Locs.push_back(LineInfo);
530 }
531 }
532 }
533 } else {
534 errs() << "Ignoring unknown key: " << Key << "\n";
535 }
536 }
537
538 for (auto &KV : Points) {
539 Coverage->Points.push_back(KV.second);
540 }
541
542 return Coverage;
543}

570class Blacklists {
571public:
572 Blacklists()
573 : DefaultBlacklist(createDefaultBlacklist()),
574 UserBlacklist(createUserBlacklist()) {}
575
576 bool isBlacklisted(const DILineInfo &I) {
577 if (DefaultBlacklist &&
578 DefaultBlacklist->inSection("sancov", "fun", I.FunctionName))
579 return true;
580 if (DefaultBlacklist &&
581 DefaultBlacklist->inSection("sancov", "src", I.FileName))
582 return true;
583 if (UserBlacklist &&
584 UserBlacklist->inSection("sancov", "fun", I.FunctionName))
585 return true;
586 if (UserBlacklist && UserBlacklist->inSection("sancov", "src", I.FileName))
587 return true;
588 return false;
589 }
590
591private:
592 static std::unique_ptr<SpecialCaseList> createDefaultBlacklist() {
593 if (!ClUseDefaultBlacklist)
594 return std::unique_ptr<SpecialCaseList>();
595 std::unique_ptr<MemoryBuffer> MB =
596 MemoryBuffer::getMemBuffer(DefaultBlacklistStr);
597 std::string Error;
598 auto Blacklist = SpecialCaseList::create(MB.get(), Error);
599 failIfNotEmpty(Error);
600 return Blacklist;
601 }
602
603 static std::unique_ptr<SpecialCaseList> createUserBlacklist() {
604 if (ClBlacklist.empty())
605 return std::unique_ptr<SpecialCaseList>();
606
607 return SpecialCaseList::createOrDie({{ClBlacklist}});
608 }
609 std::unique_ptr<SpecialCaseList> DefaultBlacklist;
610 std::unique_ptr<SpecialCaseList> UserBlacklist;
611};

613static std::vector<CoveragePoint>
614getCoveragePoints(const std::string &ObjectFile,
615 const std::set<uint64_t> &Addrs,
616 const std::set<uint64_t> &CoveredAddrs) {
617 std::vector<CoveragePoint> Result;
618 auto Symbolizer(createSymbolizer());
619 Blacklists B;
620
621 std::set<std::string> CoveredFiles;
622 if (ClSkipDeadFiles) {
623 for (auto Addr : CoveredAddrs) {
624 // TODO: it would be neccessary to set proper section index here.
625 // object::SectionedAddress::UndefSection works for only absolute
626 // addresses.
627 object::SectionedAddress ModuleAddress = {
628 Addr, object::SectionedAddress::UndefSection};
629
630 auto LineInfo = Symbolizer->symbolizeCode(ObjectFile, ModuleAddress);
631 failIfError(LineInfo);
632 CoveredFiles.insert(LineInfo->FileName);
633 auto InliningInfo =
634 Symbolizer->symbolizeInlinedCode(ObjectFile, ModuleAddress);
635 failIfError(InliningInfo);
636 for (uint32_t I = 0; I < InliningInfo->getNumberOfFrames(); ++I) {
637 auto FrameInfo = InliningInfo->getFrame(I);
638 CoveredFiles.insert(FrameInfo.FileName);
639 }
640 }
641 }
642
643 for (auto Addr : Addrs) {
644 std::set<DILineInfo> Infos; // deduplicate debug info.
645
646 // TODO: it would be neccessary to set proper section index here.
647 // object::SectionedAddress::UndefSection works for only absolute addresses.
648 object::SectionedAddress ModuleAddress = {
649 Addr, object::SectionedAddress::UndefSection};
650
651 auto LineInfo = Symbolizer->symbolizeCode(ObjectFile, ModuleAddress);
652 failIfError(LineInfo);
653 if (ClSkipDeadFiles &&
654 CoveredFiles.find(LineInfo->FileName) == CoveredFiles.end())
655 continue;
656 LineInfo->FileName = normalizeFilename(LineInfo->FileName);
657 if (B.isBlacklisted(*LineInfo))
658 continue;
659
660 auto Id = utohexstr(Addr, true);
661 auto Point = CoveragePoint(Id);
662 Infos.insert(*LineInfo);
663 Point.Locs.push_back(*LineInfo);
664
665 auto InliningInfo =
666 Symbolizer->symbolizeInlinedCode(ObjectFile, ModuleAddress);
667 failIfError(InliningInfo);
668 for (uint32_t I = 0; I < InliningInfo->getNumberOfFrames(); ++I) {
669 auto FrameInfo = InliningInfo->getFrame(I);
670 if (ClSkipDeadFiles &&
671 CoveredFiles.find(FrameInfo.FileName) == CoveredFiles.end())
672 continue;
673 FrameInfo.FileName = normalizeFilename(FrameInfo.FileName);
674 if (B.isBlacklisted(FrameInfo))
675 continue;
676 if (Infos.find(FrameInfo) == Infos.end()) {
677 Infos.insert(FrameInfo);
678 Point.Locs.push_back(FrameInfo);
679 }
680 }
681
682 Result.push_back(Point);
683 }
684
685 return Result;
686}

698// Locate __sanitizer_cov* function addresses inside the stubs table on MachO.
699static void findMachOIndirectCovFunctions(const object::MachOObjectFile &O,
700 std::set<uint64_t> *Result) {
701 MachO::dysymtab_command Dysymtab = O.getDysymtabLoadCommand();
702 MachO::symtab_command Symtab = O.getSymtabLoadCommand();
703
704 for (const auto &Load : O.load_commands()) {
705 if (Load.C.cmd == MachO::LC_SEGMENT_64) {
706 MachO::segment_command_64 Seg = O.getSegment64LoadCommand(Load);
707 for (unsigned J = 0; J < Seg.nsects; ++J) {
708 MachO::section_64 Sec = O.getSection64(Load, J);
709
710 uint32_t SectionType = Sec.flags & MachO::SECTION_TYPE;
711 if (SectionType == MachO::S_SYMBOL_STUBS) {
712 uint32_t Stride = Sec.reserved2;
713 uint32_t Cnt = Sec.size / Stride;
714 uint32_t N = Sec.reserved1;
715 for (uint32_t J = 0; J < Cnt && N + J < Dysymtab.nindirectsyms; J++) {
716 uint32_t IndirectSymbol =
717 O.getIndirectSymbolTableEntry(Dysymtab, N + J);
718 uint64_t Addr = Sec.addr + J * Stride;
719 if (IndirectSymbol < Symtab.nsyms) {
720 object::SymbolRef Symbol = *(O.getSymbolByIndex(IndirectSymbol));
721 Expected<StringRef> Name = Symbol.getName();
722 failIfError(Name);
723 if (isCoveragePointSymbol(Name.get())) {
724 Result->insert(Addr);
725 }
726 }
727 }
728 }
729 }
730 }
731 if (Load.C.cmd == MachO::LC_SEGMENT) {
732 errs() << "ERROR: 32 bit MachO binaries not supported\n";
733 }
734 }
735}

737// Locate __sanitizer_cov* function addresses that are used for coverage
738// reporting.
739static std::set<uint64_t>
740findSanitizerCovFunctions(const object::ObjectFile &O) {
741 std::set<uint64_t> Result;
742
743 for (const object::SymbolRef &Symbol : O.symbols()) {
744 Expected<uint64_t> AddressOrErr = Symbol.getAddress();
745 failIfError(AddressOrErr);
746 uint64_t Address = AddressOrErr.get();
747
748 Expected<StringRef> NameOrErr = Symbol.getName();
749 failIfError(NameOrErr);
750 StringRef Name = NameOrErr.get();
751
752 if (!(Symbol.getFlags() & object::BasicSymbolRef::SF_Undefined) &&
753 isCoveragePointSymbol(Name)) {
754 Result.insert(Address);
755 }
756 }
757
758 if (const auto *CO = dyn_cast<object::COFFObjectFile>(&O)) {
759 for (const object::ExportDirectoryEntryRef &Export :
760 CO->export_directories()) {
761 uint32_t RVA;
762 std::error_code EC = Export.getExportRVA(RVA);
763 failIfError(EC);
764
765 StringRef Name;
766 EC = Export.getSymbolName(Name);
767 failIfError(EC);
768
769 if (isCoveragePointSymbol(Name))
770 Result.insert(CO->getImageBase() + RVA);
771 }
772 }
773
774 if (const auto *MO = dyn_cast<object::MachOObjectFile>(&O)) {
775 findMachOIndirectCovFunctions(*MO, &Result);
776 }
777
778 return Result;
779}

781static uint64_t getPreviousInstructionPc(uint64_t PC,
782 Triple TheTriple) {
783 if (TheTriple.isARM()) {
784 return (PC - 3) & (~1);
785 } else if (TheTriple.isAArch64()) {
786 return PC - 4;
787 } else if (TheTriple.isMIPS()) {
788 return PC - 8;
789 } else {
790 return PC - 1;
791 }
792}

794// Locate addresses of all coverage points in a file. Coverage point
795// is defined as the 'address of instruction following __sanitizer_cov
796// call - 1'.
797static void getObjectCoveragePoints(const object::ObjectFile &O,
798 std::set<uint64_t> *Addrs) {
799 Triple TheTriple("unknown-unknown-unknown");
800 TheTriple.setArch(Triple::ArchType(O.getArch()));
801 auto TripleName = TheTriple.getTriple();
802
803 std::string Error;
804 const Target *TheTarget = TargetRegistry::lookupTarget(TripleName, Error);
805 failIfNotEmpty(Error);
806
807 std::unique_ptr<const MCSubtargetInfo> STI(
808 TheTarget->createMCSubtargetInfo(TripleName, "", ""));
809 failIfEmpty(STI, "no subtarget info for target " + TripleName);
810
811 std::unique_ptr<const MCRegisterInfo> MRI(
812 TheTarget->createMCRegInfo(TripleName));
813 failIfEmpty(MRI, "no register info for target " + TripleName);
814
815 std::unique_ptr<const MCAsmInfo> AsmInfo(
816 TheTarget->createMCAsmInfo(*MRI, TripleName));
817 failIfEmpty(AsmInfo, "no asm info for target " + TripleName);
818
819 std::unique_ptr<const MCObjectFileInfo> MOFI(new MCObjectFileInfo);
820 MCContext Ctx(AsmInfo.get(), MRI.get(), MOFI.get());
821 std::unique_ptr<MCDisassembler> DisAsm(
822 TheTarget->createMCDisassembler(*STI, Ctx));
823 failIfEmpty(DisAsm, "no disassembler info for target " + TripleName);
824
825 std::unique_ptr<const MCInstrInfo> MII(TheTarget->createMCInstrInfo());
826 failIfEmpty(MII, "no instruction info for target " + TripleName);
827
828 std::unique_ptr<const MCInstrAnalysis> MIA(
829 TheTarget->createMCInstrAnalysis(MII.get()));
830 failIfEmpty(MIA, "no instruction analysis info for target " + TripleName);
831
832 auto SanCovAddrs = findSanitizerCovFunctions(O);
833 if (SanCovAddrs.empty())
834 fail("__sanitizer_cov* functions not found");
835
836 for (object::SectionRef Section : O.sections()) {
837 if (Section.isVirtual() || !Section.isText()) // llvm-objdump does the same.
838 continue;
839 uint64_t SectionAddr = Section.getAddress();
840 uint64_t SectSize = Section.getSize();
841 if (!SectSize)
842 continue;
843
844 Expected<StringRef> BytesStr = Section.getContents();
845 failIfError(BytesStr);
846 ArrayRef<uint8_t> Bytes = arrayRefFromStringRef(*BytesStr);
847
848 for (uint64_t Index = 0, Size = 0; Index < Section.getSize();
849 Index += Size) {
850 MCInst Inst;
851 if (!DisAsm->getInstruction(Inst, Size, Bytes.slice(Index),
852 SectionAddr + Index, nulls(), nulls())) {
853 if (Size == 0)
854 Size = 1;
855 continue;
856 }
857 uint64_t Addr = Index + SectionAddr;
858 // Sanitizer coverage uses the address of the next instruction - 1.
859 uint64_t CovPoint = getPreviousInstructionPc(Addr + Size, TheTriple);
860 uint64_t Target;
861 if (MIA->isCall(Inst) &&
862 MIA->evaluateBranch(Inst, SectionAddr + Index, Size, Target) &&
863 SanCovAddrs.find(Target) != SanCovAddrs.end())
864 Addrs->insert(CovPoint);
865 }
866 }
867}

869static void
870visitObjectFiles(const object::Archive &A,
871 function_ref<void(const object::ObjectFile &)> Fn) {
872 Error Err = Error::success();
873 for (auto &C : A.children(Err)) {
874 Expected<std::unique_ptr<object::Binary>> ChildOrErr = C.getAsBinary();
875 failIfError(ChildOrErr);
876 if (auto *O = dyn_cast<object::ObjectFile>(&*ChildOrErr.get()))
877 Fn(*O);
878 else
879 failIfError(object::object_error::invalid_file_type);
880 }
881 failIfError(std::move(Err));
882}

884static void
885visitObjectFiles(const std::string &FileName,
886 function_ref<void(const object::ObjectFile &)> Fn) {
887 Expected<object::OwningBinary<object::Binary>> BinaryOrErr =
888 object::createBinary(FileName);
889 if (!BinaryOrErr)
890 failIfError(BinaryOrErr);
891
892 object::Binary &Binary = *BinaryOrErr.get().getBinary();
893 if (object::Archive *A = dyn_cast<object::Archive>(&Binary))
894 visitObjectFiles(*A, Fn);
895 else if (object::ObjectFile *O = dyn_cast<object::ObjectFile>(&Binary))
896 Fn(*O);
897 else
898 failIfError(object::object_error::invalid_file_type);
899}

930static ErrorOr<bool> isCoverageFile(const std::string &FileName) {
931 auto ShortFileName = llvm::sys::path::filename(FileName);
932 if (!SancovFileRegex.match(ShortFileName))
933 return false;
934
935 ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
936 MemoryBuffer::getFile(FileName);
937 if (!BufOrErr) {
938 errs() << "Warning: " << BufOrErr.getError().message() << "("
939 << BufOrErr.getError().value()
940 << "), filename: " << llvm::sys::path::filename(FileName) << "\n";
941 return BufOrErr.getError();
942 }
943 std::unique_ptr<MemoryBuffer> Buf = std::move(BufOrErr.get());
944 if (Buf->getBufferSize() < 8) {
945 return false;
946 }
947 const FileHeader *Header =
948 reinterpret_cast<const FileHeader *>(Buf->getBufferStart());
949 return Header->Magic == BinCoverageMagic;
950}

957static std::unique_ptr<SymbolizedCoverage>
958symbolize(const RawCoverage &Data, const std::string ObjectFile) {
959 auto Coverage = make_unique<SymbolizedCoverage>();
960
961 ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
962 MemoryBuffer::getFile(ObjectFile);
963 failIfError(BufOrErr);
964 SHA1 Hasher;
965 Hasher.update((*BufOrErr)->getBuffer());
966 Coverage->BinaryHash = toHex(Hasher.final());
967
968 Blacklists B;
969 auto Symbolizer(createSymbolizer());
970
971 for (uint64_t Addr : *Data.Addrs) {
972 // TODO: it would be neccessary to set proper section index here.
973 // object::SectionedAddress::UndefSection works for only absolute addresses.
974 auto LineInfo = Symbolizer->symbolizeCode(
975 ObjectFile, {Addr, object::SectionedAddress::UndefSection});
976 failIfError(LineInfo);
977 if (B.isBlacklisted(*LineInfo))
978 continue;
979
980 Coverage->CoveredIds.insert(utohexstr(Addr, true));
981 }
982
983 std::set<uint64_t> AllAddrs = findCoveragePointAddrs(ObjectFile);
984 if (!std::includes(AllAddrs.begin(), AllAddrs.end(), Data.Addrs->begin(),
985 Data.Addrs->end())) {
986 fail("Coverage points in binary and .sancov file do not match.");
987 }
988 Coverage->Points = getCoveragePoints(ObjectFile, AllAddrs, *Data.Addrs);
989 return Coverage;
990}

992struct FileFn {
993 bool operator<(const FileFn &RHS) const {
994 return std::tie(FileName, FunctionName) <
995 std::tie(RHS.FileName, RHS.FunctionName);

998 std::string FileName;
999 std::string FunctionName;

1002static std::set<FileFn>
1003computeFunctions(const std::vector<CoveragePoint> &Points) {
1004 std::set<FileFn> Fns;
1005 for (const auto &Point : Points) {
1006 for (const auto &Loc : Point.Locs) {
1007 Fns.insert(FileFn{Loc.FileName, Loc.FunctionName});
1008 }
1009 }
1010 return Fns;

1013static std::set<FileFn>
1014computeNotCoveredFunctions(const SymbolizedCoverage &Coverage) {
1015 auto Fns = computeFunctions(Coverage.Points);
1016
1017 for (const auto &Point : Coverage.Points) {
1018 if (Coverage.CoveredIds.find(Point.Id) == Coverage.CoveredIds.end())
1019 continue;
1020
1021 for (const auto &Loc : Point.Locs) {
1022 Fns.erase(FileFn{Loc.FileName, Loc.FunctionName});
1023 }
1024 }
1025
1026 return Fns;
1027}

1029static std::set<FileFn>
1030computeCoveredFunctions(const SymbolizedCoverage &Coverage) {
1031 auto AllFns = computeFunctions(Coverage.Points);
1032 std::set<FileFn> Result;
1033
1034 for (const auto &Point : Coverage.Points) {
1035 if (Coverage.CoveredIds.find(Point.Id) == Coverage.CoveredIds.end())
1036 continue;
1037
1038 for (const auto &Loc : Point.Locs) {
1039 Result.insert(FileFn{Loc.FileName, Loc.FunctionName});
1040 }
1041 }
1042
1043 return Result;
1044}

1046typedef std::map<FileFn, std::pair<uint32_t, uint32_t>> FunctionLocs;
1047// finds first location in a file for each function.
1048static FunctionLocs resolveFunctions(const SymbolizedCoverage &Coverage,
1049 const std::set<FileFn> &Fns) {
1050 FunctionLocs Result;
1051 for (const auto &Point : Coverage.Points) {
1052 for (const auto &Loc : Point.Locs) {
1053 FileFn Fn = FileFn{Loc.FileName, Loc.FunctionName};
1054 if (Fns.find(Fn) == Fns.end())
1055 continue;
1056
1057 auto P = std::make_pair(Loc.Line, Loc.Column);
1058 auto I = Result.find(Fn);
1059 if (I == Result.end() || I->second > P) {
1060 Result[Fn] = P;
1061 }
1062 }
1063 }
1064 return Result;
1065}

1067static void printFunctionLocs(const FunctionLocs &FnLocs, raw_ostream &OS) {
1068 for (const auto &P : FnLocs) {
1069 OS << stripPathPrefix(P.first.FileName) << ":" << P.second.first << " "
1070 << P.first.FunctionName << "\n";
1071 }

1073CoverageStats computeStats(const SymbolizedCoverage &Coverage) {
1074 CoverageStats Stats = {Coverage.Points.size(), Coverage.CoveredIds.size(),
1075 computeFunctions(Coverage.Points).size(),
1076 computeCoveredFunctions(Coverage).size()};
1077 return Stats;
1078}

1108static std::unique_ptr<SymbolizedCoverage>
1109merge(const std::vector<std::unique_ptr<SymbolizedCoverage>> &Coverages) {
1110 if (Coverages.empty())
1111 return nullptr;
1112
1113 auto Result = make_unique<SymbolizedCoverage>();
1114
1115 for (size_t I = 0; I < Coverages.size(); ++I) {
1116 const SymbolizedCoverage &Coverage = *Coverages[I];
1117 std::string Prefix;
1118 if (Coverages.size() > 1) {
1119 // prefix is not needed when there's only one file.
1120 Prefix = utostr(I);
1121 }
1122
1123 for (const auto &Id : Coverage.CoveredIds) {
1124 Result->CoveredIds.insert(Prefix + Id);
1125 }
1126
1127 for (const auto &CovPoint : Coverage.Points) {
1128 CoveragePoint NewPoint(CovPoint);
1129 NewPoint.Id = Prefix + CovPoint.Id;
1130 Result->Points.push_back(NewPoint);
1131 }
1132 }
1133
1134 if (Coverages.size() == 1) {
1135 Result->BinaryHash = Coverages[0]->BinaryHash;
1136 }
1137
1138 return Result;
1139}

1141static std::unique_ptr<SymbolizedCoverage>
1142readSymbolizeAndMergeCmdArguments(std::vector<std::string> FileNames) {
1143 std::vector<std::unique_ptr<SymbolizedCoverage>> Coverages;
1144
1145 {
1146 // Short name => file name.
1147 std::map<std::string, std::string> ObjFiles;
1148 std::string FirstObjFile;
1149 std::set<std::string> CovFiles;
1150
1151 // Partition input values into coverage/object files.
1152 for (const auto &FileName : FileNames) {
1153 if (isSymbolizedCoverageFile(FileName)) {
1154 Coverages.push_back(SymbolizedCoverage::read(FileName));
1155 }
1156
1157 auto ErrorOrIsCoverage = isCoverageFile(FileName);
1158 if (!ErrorOrIsCoverage)
1159 continue;
1160 if (ErrorOrIsCoverage.get()) {
1161 CovFiles.insert(FileName);
1162 } else {
1163 auto ShortFileName = llvm::sys::path::filename(FileName);
1164 if (ObjFiles.find(ShortFileName) != ObjFiles.end()) {
1165 fail("Duplicate binary file with a short name: " + ShortFileName);
1166 }
1167
1168 ObjFiles[ShortFileName] = FileName;
1169 if (FirstObjFile.empty())
1170 FirstObjFile = FileName;
1171 }
1172 }
1173
1174 SmallVector<StringRef, 2> Components;
1175
1176 // Object file => list of corresponding coverage file names.
1177 std::map<std::string, std::vector<std::string>> CoverageByObjFile;
1178 for (const auto &FileName : CovFiles) {
1179 auto ShortFileName = llvm::sys::path::filename(FileName);
1180 auto Ok = SancovFileRegex.match(ShortFileName, &Components);
1181 if (!Ok) {
1182 fail("Can't match coverage file name against "
1183 "<module_name>.<pid>.sancov pattern: " +
1184 FileName);
1185 }
1186
1187 auto Iter = ObjFiles.find(Components[1]);
1188 if (Iter == ObjFiles.end()) {
1189 fail("Object file for coverage not found: " + FileName);
1190 }
1191
1192 CoverageByObjFile[Iter->second].push_back(FileName);
1193 };
1194
1195 for (const auto &Pair : ObjFiles) {
1196 auto FileName = Pair.second;
1197 if (CoverageByObjFile.find(FileName) == CoverageByObjFile.end())
1198 errs() << "WARNING: No coverage file for " << FileName << "\n";
1199 }
1200
1201 // Read raw coverage and symbolize it.
1202 for (const auto &Pair : CoverageByObjFile) {
1203 if (findSanitizerCovFunctions(Pair.first).empty()) {
1204 errs()
1205 << "WARNING: Ignoring " << Pair.first
1206 << " and its coverage because __sanitizer_cov* functions were not "
1207 "found.\n";
1208 continue;
1209 }
1210
1211 for (const std::string &CoverageFile : Pair.second) {
1212 auto DataOrError = RawCoverage::read(CoverageFile);
1213 failIfError(DataOrError);
1214 Coverages.push_back(symbolize(*DataOrError.get(), Pair.first));
1215 }
1216 }
1217 }
1218
1219 return merge(Coverages);
1220}

1224int main(int Argc, char **Argv) {
1225 // Print stack trace if we signal out.
1226 sys::PrintStackTraceOnErrorSignal(Argv[0]);
1227 PrettyStackTraceProgram X(Argc, Argv);
1228 llvm_shutdown_obj Y; // Call llvm_shutdown() on exit.
1229
1230 llvm::InitializeAllTargetInfos();
1231 llvm::InitializeAllTargetMCs();
1232 llvm::InitializeAllDisassemblers();
1233
1234 cl::ParseCommandLineOptions(Argc, Argv,
1235 "Sanitizer Coverage Processing Tool (sancov)\n\n"
1236 " This tool can extract various coverage-related information from: \n"
1237 " coverage-instrumented binary files, raw .sancov files and their "
1238 "symbolized .symcov version.\n"
1239 " Depending on chosen action the tool expects different input files:\n"
1240 " -print-coverage-pcs - coverage-instrumented binary files\n"
1241 " -print-coverage - .sancov files\n"
1242 " <other actions> - .sancov files & corresponding binary "
1243 "files, .symcov files\n"
1244 );
1245
1246 // -print doesn't need object files.
1247 if (Action == PrintAction) {
1248 readAndPrintRawCoverage(ClInputFiles, outs());
1249 return 0;
1250 } else if (Action == PrintCovPointsAction) {
1251 // -print-coverage-points doesn't need coverage files.
1252 for (const std::string &ObjFile : ClInputFiles) {
1253 printCovPoints(ObjFile, outs());
1254 }
1255 return 0;
1256 }
1257
1258 auto Coverage = readSymbolizeAndMergeCmdArguments(ClInputFiles);
1259 failIf(!Coverage, "No valid coverage files given.");
1260
1261 switch (Action) {
1262 case CoveredFunctionsAction: {
1263 printCoveredFunctions(*Coverage, outs());
1264 return 0;
1265 }
1266 case NotCoveredFunctionsAction: {
1267 printNotCoveredFunctions(*Coverage, outs());
1268 return 0;
1269 }
1270 case StatsAction: {
1271 outs() << computeStats(*Coverage);
1272 return 0;
1273 }
1274 case MergeAction:
1275 case SymbolizeAction: { // merge & symbolize are synonims.
1276 JSONWriter W(outs());
1277 W << *Coverage;
1278 return 0;
1279 }
1280 case HtmlReportAction:
1281 errs() << "-html-report option is removed: "
1282 "use -symbolize & coverage-report-server.py instead\n";
1283 return 1;
1284 case PrintAction:
1285 case PrintCovPointsAction:
1286 llvm_unreachable("unsupported action");
1287 }
