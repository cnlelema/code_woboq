155template <class T> static T checkedGetHex(StringRef S) {
156 T Value;
157 bool Fail = S.getAsInteger(16, Value);
158 assert(!Fail);
159 (void)Fail;
160 return Value;
161}

163// Fills exactly Len bytes of buffer with hexadecimal characters
164// representing value 'X'
165template <class T, class Iterator>
166static Iterator utohexstr(T X, Iterator It, size_t Len) {
167 // Fill range with '0'
168 std::fill(It, It + Len, '0');
169
170 for (long I = Len - 1; I >= 0; --I) {
171 unsigned char Mod = static_cast<unsigned char>(X) & 15;
172 *(It + I) = hexdigit(Mod, false);
173 X >>= 4;
174 }
175 assert(X == 0);
176 return It + Len;
177}

179uint8_t IHexRecord::getChecksum(StringRef S) {
180 assert((S.size() & 1) == 0);
181 uint8_t Checksum = 0;
182 while (!S.empty()) {
183 Checksum += checkedGetHex<uint8_t>(S.take_front(2));
184 S = S.drop_front(2);
185 }
186 return -Checksum;
187}

189IHexLineData IHexRecord::getLine(uint8_t Type, uint16_t Addr,
190 ArrayRef<uint8_t> Data) {
191 IHexLineData Line(getLineLength(Data.size()));
192 assert(Line.size());
193 auto Iter = Line.begin();
194 *Iter++ = ':';
195 Iter = utohexstr(Data.size(), Iter, 2);
196 Iter = utohexstr(Addr, Iter, 4);
197 Iter = utohexstr(Type, Iter, 2);
198 for (uint8_t X : Data)
199 Iter = utohexstr(X, Iter, 2);
200 StringRef S(Line.data() + 1, std::distance(Line.begin() + 1, Iter));
201 Iter = utohexstr(getChecksum(S), Iter, 2);
202 *Iter++ = '\r';
203 *Iter++ = '\n';
204 assert(Iter == Line.end());
205 return Line;
206}

208static Error checkRecord(const IHexRecord &R) {
209 switch (R.Type) {
210 case IHexRecord::Data:
211 if (R.HexData.size() == 0)
212 return createStringError(
213 errc::invalid_argument,
214 "zero data length is not allowed for data records");
215 break;
216 case IHexRecord::EndOfFile:
217 break;
218 case IHexRecord::SegmentAddr:
219 // 20-bit segment address. Data length must be 2 bytes
220 // (4 bytes in hex)
221 if (R.HexData.size() != 4)
222 return createStringError(
223 errc::invalid_argument,
224 "segment address data should be 2 bytes in size");
225 break;
226 case IHexRecord::StartAddr80x86:
227 case IHexRecord::StartAddr:
228 if (R.HexData.size() != 8)
229 return createStringError(errc::invalid_argument,
230 "start address data should be 4 bytes in size");
231 // According to Intel HEX specification '03' record
232 // only specifies the code address within the 20-bit
233 // segmented address space of the 8086/80186. This
234 // means 12 high order bits should be zeroes.
235 if (R.Type == IHexRecord::StartAddr80x86 &&
236 R.HexData.take_front(3) != "000")
237 return createStringError(errc::invalid_argument,
238 "start address exceeds 20 bit for 80x86");
239 break;
240 case IHexRecord::ExtendedAddr:
241 // 16-31 bits of linear base address
242 if (R.HexData.size() != 4)
243 return createStringError(
244 errc::invalid_argument,
245 "extended address data should be 2 bytes in size");
246 break;
247 default:
248 // Unknown record type
249 return createStringError(errc::invalid_argument, "unknown record type: %u",
250 static_cast<unsigned>(R.Type));
251 }
252 return Error::success();
253}

255// Checks that IHEX line contains valid characters.
256// This allows converting hexadecimal data to integers
257// without extra verification.
258static Error checkChars(StringRef Line) {
259 assert(!Line.empty());
260 if (Line[0] != ':')
261 return createStringError(errc::invalid_argument,
262 "missing ':' in the beginning of line.");
263
264 for (size_t Pos = 1; Pos < Line.size(); ++Pos)
265 if (hexDigitValue(Line[Pos]) == -1U)
266 return createStringError(errc::invalid_argument,
267 "invalid character at position %zu.", Pos + 1);
268 return Error::success();
269}

271Expected<IHexRecord> IHexRecord::parse(StringRef Line) {
272 assert(!Line.empty());
273
274 // ':' + Length + Address + Type + Checksum with empty data ':LLAAAATTCC'
275 if (Line.size() < 11)
276 return createStringError(errc::invalid_argument,
277 "line is too short: %zu chars.", Line.size());
278
279 if (Error E = checkChars(Line))
280 return std::move(E);
281
282 IHexRecord Rec;
283 size_t DataLen = checkedGetHex<uint8_t>(Line.substr(1, 2));
284 if (Line.size() != getLength(DataLen))
285 return createStringError(errc::invalid_argument,
286 "invalid line length %zu (should be %zu)",
287 Line.size(), getLength(DataLen));
288
289 Rec.Addr = checkedGetHex<uint16_t>(Line.substr(3, 4));
290 Rec.Type = checkedGetHex<uint8_t>(Line.substr(7, 2));
291 Rec.HexData = Line.substr(9, DataLen * 2);
292
293 if (getChecksum(Line.drop_front(1)) != 0)
294 return createStringError(errc::invalid_argument, "incorrect checksum.");
295 if (Error E = checkRecord(Rec))
296 return std::move(E);
297 return Rec;
298}

300static uint64_t sectionPhysicalAddr(const SectionBase *Sec) {
301 Segment *Seg = Sec->ParentSegment;
302 if (Seg && Seg->Type != ELF::PT_LOAD)
303 Seg = nullptr;
304 return Seg ? Seg->PAddr + Sec->OriginalOffset - Seg->OriginalOffset
305 : Sec->Addr;

308void IHexSectionWriterBase::writeSection(const SectionBase *Sec,
309 ArrayRef<uint8_t> Data) {
310 assert(Data.size() == Sec->Size);
311 const uint32_t ChunkSize = 16;
312 uint32_t Addr = sectionPhysicalAddr(Sec) & 0xFFFFFFFFU;
313 while (!Data.empty()) {
314 uint64_t DataSize = std::min<uint64_t>(Data.size(), ChunkSize);
315 if (Addr > SegmentAddr + BaseAddr + 0xFFFFU) {
316 if (Addr > 0xFFFFFU) {
317 // Write extended address record, zeroing segment address
318 // if needed.
319 if (SegmentAddr != 0)
320 SegmentAddr = writeSegmentAddr(0U);
321 BaseAddr = writeBaseAddr(Addr);
322 } else {
323 // We can still remain 16-bit
324 SegmentAddr = writeSegmentAddr(Addr);
325 }
326 }
327 uint64_t SegOffset = Addr - BaseAddr - SegmentAddr;
328 assert(SegOffset <= 0xFFFFU);
329 DataSize = std::min(DataSize, 0x10000U - SegOffset);
330 writeData(0, SegOffset, Data.take_front(DataSize));
331 Addr += DataSize;
332 Data = Data.drop_front(DataSize);
333 }
334}

336uint64_t IHexSectionWriterBase::writeSegmentAddr(uint64_t Addr) {
337 assert(Addr <= 0xFFFFFU);
338 uint8_t Data[] = {static_cast<uint8_t>((Addr & 0xF0000U) >> 12), 0};
339 writeData(2, 0, Data);
340 return Addr & 0xF0000U;
341}

343uint64_t IHexSectionWriterBase::writeBaseAddr(uint64_t Addr) {
344 assert(Addr <= 0xFFFFFFFFU);
345 uint64_t Base = Addr & 0xFFFF0000U;
346 uint8_t Data[] = {static_cast<uint8_t>(Base >> 24),
347 static_cast<uint8_t>((Base >> 16) & 0xFF)};
348 writeData(4, 0, Data);
349 return Base;
350}

400static const std::vector<uint8_t> ZlibGnuMagic = {'Z', 'L', 'I', 'B'};

402static bool isDataGnuCompressed(ArrayRef<uint8_t> Data) {
403 return Data.size() > ZlibGnuMagic.size() &&
404 std::equal(ZlibGnuMagic.begin(), ZlibGnuMagic.end(), Data.data());
405}

407template <class ELFT>
408static std::tuple<uint64_t, uint64_t>
409getDecompressedSizeAndAlignment(ArrayRef<uint8_t> Data) {
410 const bool IsGnuDebug = isDataGnuCompressed(Data);
411 const uint64_t DecompressedSize =
412 IsGnuDebug
413 ? support::endian::read64be(Data.data() + ZlibGnuMagic.size())
414 : reinterpret_cast<const Elf_Chdr_Impl<ELFT> *>(Data.data())->ch_size;
415 const uint64_t DecompressedAlign =
416 IsGnuDebug ? 1
417 : reinterpret_cast<const Elf_Chdr_Impl<ELFT> *>(Data.data())
418 ->ch_addralign;
419
420 return std::make_tuple(DecompressedSize, DecompressedAlign);
421}

423template <class ELFT>
424void ELFSectionWriter<ELFT>::visit(const DecompressedSection &Sec) {
425 const size_t DataOffset = isDataGnuCompressed(Sec.OriginalData)
426 ? (ZlibGnuMagic.size() + sizeof(Sec.Size))
427 : sizeof(Elf_Chdr_Impl<ELFT>);
428
429 StringRef CompressedContent(
430 reinterpret_cast<const char *>(Sec.OriginalData.data()) + DataOffset,
431 Sec.OriginalData.size() - DataOffset);
432
433 SmallVector<char, 128> DecompressedContent;
434 if (Error E = zlib::uncompress(CompressedContent, DecompressedContent,
435 static_cast<size_t>(Sec.Size)))
436 reportError(Sec.Name, std::move(E));
437
438 uint8_t *Buf = Out.getBufferStart() + Sec.Offset;
439 std::copy(DecompressedContent.begin(), DecompressedContent.end(), Buf);
440}

475template <class ELFT>
476void ELFSectionWriter<ELFT>::visit(const CompressedSection &Sec) {
477 uint8_t *Buf = Out.getBufferStart() + Sec.Offset;
478 if (Sec.CompressionType == DebugCompressionType::None) {
479 std::copy(Sec.OriginalData.begin(), Sec.OriginalData.end(), Buf);
480 return;
481 }
482
483 if (Sec.CompressionType == DebugCompressionType::GNU) {
484 const char *Magic = "ZLIB";
485 memcpy(Buf, Magic, strlen(Magic));
486 Buf += strlen(Magic);
487 const uint64_t DecompressedSize =
488 support::endian::read64be(&Sec.DecompressedSize);
489 memcpy(Buf, &DecompressedSize, sizeof(DecompressedSize));
490 Buf += sizeof(DecompressedSize);
491 } else {
492 Elf_Chdr_Impl<ELFT> Chdr;
493 Chdr.ch_type = ELF::ELFCOMPRESS_ZLIB;
494 Chdr.ch_size = Sec.DecompressedSize;
495 Chdr.ch_addralign = Sec.DecompressedAlign;
496 memcpy(Buf, &Chdr, sizeof(Chdr));
497 Buf += sizeof(Chdr);
498 }
499
500 std::copy(Sec.CompressedData.begin(), Sec.CompressedData.end(), Buf);
501}

503CompressedSection::CompressedSection(const SectionBase &Sec,
504 DebugCompressionType CompressionType)
505 : SectionBase(Sec), CompressionType(CompressionType),
506 DecompressedSize(Sec.OriginalData.size()), DecompressedAlign(Sec.Align) {
507 if (Error E = zlib::compress(
508 StringRef(reinterpret_cast<const char *>(OriginalData.data()),
509 OriginalData.size()),
510 CompressedData))
511 reportError(Name, std::move(E));
512
513 size_t ChdrSize;
514 if (CompressionType == DebugCompressionType::GNU) {
515 Name = ".z" + Sec.Name.substr(1);
516 ChdrSize = sizeof("ZLIB") - 1 + sizeof(uint64_t);
517 } else {
518 Flags |= ELF::SHF_COMPRESSED;
519 ChdrSize =
520 std::max(std::max(sizeof(object::Elf_Chdr_Impl<object::ELF64LE>),
521 sizeof(object::Elf_Chdr_Impl<object::ELF64BE>)),
522 std::max(sizeof(object::Elf_Chdr_Impl<object::ELF32LE>),
523 sizeof(object::Elf_Chdr_Impl<object::ELF32BE>)));
524 }
525 Size = ChdrSize + CompressedData.size();
526 Align = 8;
527}

594static bool isValidReservedSectionIndex(uint16_t Index, uint16_t Machine) {
595 switch (Index) {
596 case SHN_ABS:
597 case SHN_COMMON:
598 return true;
599 }
600 if (Machine == EM_HEXAGON) {
601 switch (Index) {
602 case SHN_HEXAGON_SCOMMON:
603 case SHN_HEXAGON_SCOMMON_2:
604 case SHN_HEXAGON_SCOMMON_4:
605 case SHN_HEXAGON_SCOMMON_8:
606 return true;
607 }
608 }
609 return false;
610}

612// Large indexes force us to clarify exactly what this function should do. This
613// function should return the value that will appear in st_shndx when written
614// out.
615uint16_t Symbol::getShndx() const {
616 if (DefinedIn != nullptr) {
617 if (DefinedIn->Index >= SHN_LORESERVE)
618 return SHN_XINDEX;
619 return DefinedIn->Index;
620 }
621 switch (ShndxType) {
622 // This means that we don't have a defined section but we do need to
623 // output a legitimate section index.
624 case SYMBOL_SIMPLE_INDEX:
625 return SHN_UNDEF;
626 case SYMBOL_ABS:
627 case SYMBOL_COMMON:
628 case SYMBOL_HEXAGON_SCOMMON:
629 case SYMBOL_HEXAGON_SCOMMON_2:
630 case SYMBOL_HEXAGON_SCOMMON_4:
631 case SYMBOL_HEXAGON_SCOMMON_8:
632 case SYMBOL_XINDEX:
633 return static_cast<uint16_t>(ShndxType);
634 }
635 llvm_unreachable("Symbol with invalid ShndxType encountered");
636}

646void SymbolTableSection::addSymbol(Twine Name, uint8_t Bind, uint8_t Type,
647 SectionBase *DefinedIn, uint64_t Value,
648 uint8_t Visibility, uint16_t Shndx,
649 uint64_t SymbolSize) {
650 Symbol Sym;
651 Sym.Name = Name.str();
652 Sym.Binding = Bind;
653 Sym.Type = Type;
654 Sym.DefinedIn = DefinedIn;
655 if (DefinedIn != nullptr)
656 DefinedIn->HasSymbol = true;
657 if (DefinedIn == nullptr) {
658 if (Shndx >= SHN_LORESERVE)
659 Sym.ShndxType = static_cast<SymbolShndxType>(Shndx);

1056// Returns true IFF a section is wholly inside the range of a segment
1057static bool sectionWithinSegment(const SectionBase &Section,
1058 const Segment &Segment) {
1059 // If a section is empty it should be treated like it has a size of 1. This is
1060 // to clarify the case when an empty section lies on a boundary between two
1061 // segments and ensures that the section "belongs" to the second segment and
1062 // not the first.
1063 uint64_t SecSize = Section.Size ? Section.Size : 1;
1064
1065 if (Section.Type == SHT_NOBITS) {
1066 if (!(Section.Flags & SHF_ALLOC))
1067 return false;
1068
1069 bool SectionIsTLS = Section.Flags & SHF_TLS;
1070 bool SegmentIsTLS = Segment.Type == PT_TLS;
1071 if (SectionIsTLS != SegmentIsTLS)
1072 return false;
1073
1074 return Segment.VAddr <= Section.Addr &&
1075 Segment.VAddr + Segment.MemSize >= Section.Addr + SecSize;
1076 }
1077
1078 return Segment.Offset <= Section.OriginalOffset &&
1079 Segment.Offset + Segment.FileSize >= Section.OriginalOffset + SecSize;
1080}
1081
1082// Returns true IFF a segment's original offset is inside of another segment's
1083// range.
1084static bool segmentOverlapsSegment(const Segment &Child,
1085 const Segment &Parent) {
1086
1087 return Parent.OriginalOffset <= Child.OriginalOffset &&
1088 Parent.OriginalOffset + Parent.FileSize > Child.OriginalOffset;
1089}
1090
1091static bool compareSegmentsByOffset(const Segment *A, const Segment *B) {
1092 // Any segment without a parent segment should come before a segment
1093 // that has a parent segment.
1094 if (A->OriginalOffset < B->OriginalOffset)
1095 return true;
1096 if (A->OriginalOffset > B->OriginalOffset)
1097 return false;
1098 return A->Index < B->Index;
1099}
1100
1101static bool compareSegmentsByPAddr(const Segment *A, const Segment *B) {
1102 if (A->PAddr < B->PAddr)
1103 return true;
1104 if (A->PAddr > B->PAddr)
1105 return false;
1106 return A->Index < B->Index;
1107}

1181// Adds sections from IHEX data file. Data should have been
1182// fully validated by this time.
1183void IHexELFBuilder::addDataSections() {
1184 OwnedDataSection *Section = nullptr;
1185 uint64_t SegmentAddr = 0, BaseAddr = 0;
1186 uint32_t SecNo = 1;
1187
1188 for (const IHexRecord &R : Records) {
1189 uint64_t RecAddr;
1190 switch (R.Type) {
1191 case IHexRecord::Data:
1192 // Ignore empty data records
1193 if (R.HexData.empty())
1194 continue;
1195 RecAddr = R.Addr + SegmentAddr + BaseAddr;
1196 if (!Section || Section->Addr + Section->Size != RecAddr)
1197 // OriginalOffset field is only used to sort section properly, so
1198 // instead of keeping track of real offset in IHEX file, we use
1199 // section number.
1200 Section = &Obj->addSection<OwnedDataSection>(
1201 ".sec" + std::to_string(SecNo++), RecAddr,
1202 ELF::SHF_ALLOC | ELF::SHF_WRITE, SecNo);
1203 Section->appendHexData(R.HexData);
1204 break;
1205 case IHexRecord::EndOfFile:
1206 break;
1207 case IHexRecord::SegmentAddr:
1208 // 20-bit segment address.
1209 SegmentAddr = checkedGetHex<uint16_t>(R.HexData) << 4;
1210 break;
1211 case IHexRecord::StartAddr80x86:
1212 case IHexRecord::StartAddr:
1213 Obj->Entry = checkedGetHex<uint32_t>(R.HexData);
1214 assert(Obj->Entry <= 0xFFFFFU);
1215 break;
1216 case IHexRecord::ExtendedAddr:
1217 // 16-31 bits of linear base address
1218 BaseAddr = checkedGetHex<uint16_t>(R.HexData) << 16;
1219 break;
1220 default:
1221 llvm_unreachable("unknown record type");
1222 }
1223 }
1224}

1237template <class ELFT> void ELFBuilder<ELFT>::setParentSegment(Segment &Child) {
1238 for (Segment &Parent : Obj.segments()) {
1239 // Every segment will overlap with itself but we don't want a segment to
1240 // be it's own parent so we avoid that situation.
1241 if (&Child != &Parent && segmentOverlapsSegment(Child, Parent)) {
1242 // We want a canonical "most parental" segment but this requires
1243 // inspecting the ParentSegment.
1244 if (compareSegmentsByOffset(&Parent, &Child))
1245 if (Child.ParentSegment == nullptr ||
1246 compareSegmentsByOffset(&Parent, Child.ParentSegment)) {
1247 Child.ParentSegment = &Parent;
1248 }
1249 }
1250 }

1267template <class ELFT>
1268void ELFBuilder<ELFT>::readProgramHeaders(const ELFFile<ELFT> &HeadersFile) {
1269 uint32_t Index = 0;
1270 for (const auto &Phdr : unwrapOrError(HeadersFile.program_headers())) {
1271 if (Phdr.p_offset + Phdr.p_filesz > HeadersFile.getBufSize())
1272 error("program header with offset 0x" + Twine::utohexstr(Phdr.p_offset) +
1273 " and file size 0x" + Twine::utohexstr(Phdr.p_filesz) +
1274 " goes past the end of the file");
1275
1276 ArrayRef<uint8_t> Data{HeadersFile.base() + Phdr.p_offset,
1277 (size_t)Phdr.p_filesz};
1278 Segment &Seg = Obj.addSegment(Data);
1279 Seg.Type = Phdr.p_type;
1280 Seg.Flags = Phdr.p_flags;
1281 Seg.OriginalOffset = Phdr.p_offset + EhdrOffset;
1282 Seg.Offset = Phdr.p_offset + EhdrOffset;
1283 Seg.VAddr = Phdr.p_vaddr;
1284 Seg.PAddr = Phdr.p_paddr;
1285 Seg.FileSize = Phdr.p_filesz;
1286 Seg.MemSize = Phdr.p_memsz;
1287 Seg.Align = Phdr.p_align;
1288 Seg.Index = Index++;
1289 for (SectionBase &Section : Obj.sections()) {
1290 if (sectionWithinSegment(Section, Seg)) {
1291 Seg.addSection(&Section);
1292 if (!Section.ParentSegment ||
1293 Section.ParentSegment->Offset > Seg.Offset) {
1294 Section.ParentSegment = &Seg;
1295 }
1296 }
1297 }
1298 }
1299
1300 auto &ElfHdr = Obj.ElfHdrSegment;
1301 ElfHdr.Index = Index++;
1302 ElfHdr.OriginalOffset = ElfHdr.Offset = EhdrOffset;
1303
1304 const auto &Ehdr = *HeadersFile.getHeader();
1305 auto &PrHdr = Obj.ProgramHdrSegment;
1306 PrHdr.Type = PT_PHDR;
1307 PrHdr.Flags = 0;
1308 // The spec requires us to have p_vaddr % p_align == p_offset % p_align.
1309 // Whereas this works automatically for ElfHdr, here OriginalOffset is
1310 // always non-zero and to ensure the equation we assign the same value to
1311 // VAddr as well.
1312 PrHdr.OriginalOffset = PrHdr.Offset = PrHdr.VAddr = EhdrOffset + Ehdr.e_phoff;
1313 PrHdr.PAddr = 0;
1314 PrHdr.FileSize = PrHdr.MemSize = Ehdr.e_phentsize * Ehdr.e_phnum;
1315 // The spec requires us to naturally align all the fields.
1316 PrHdr.Align = sizeof(Elf_Addr);
1317 PrHdr.Index = Index++;
1318
1319 // Now we do an O(n^2) loop through the segments in order to match up
1320 // segments.
1321 for (Segment &Child : Obj.segments())
1322 setParentSegment(Child);
1323 setParentSegment(ElfHdr);
1324 setParentSegment(PrHdr);
1325}

1327template <class ELFT>
1328void ELFBuilder<ELFT>::initGroupSection(GroupSection *GroupSec) {
1329 if (GroupSec->Align % sizeof(ELF::Elf32_Word) != 0)
1330 error("invalid alignment " + Twine(GroupSec->Align) + " of group section '" +
1331 GroupSec->Name + "'");
1332 SectionTableRef SecTable = Obj.sections();
1333 auto SymTab = SecTable.template getSectionOfType<SymbolTableSection>(
1334 GroupSec->Link,
1335 "link field value '" + Twine(GroupSec->Link) + "' in section '" +
1336 GroupSec->Name + "' is invalid",
1337 "link field value '" + Twine(GroupSec->Link) + "' in section '" +
1338 GroupSec->Name + "' is not a symbol table");
1339 Symbol *Sym = SymTab->getSymbolByIndex(GroupSec->Info);
1340 if (!Sym)
1341 error("info field value '" + Twine(GroupSec->Info) + "' in section '" +
1342 GroupSec->Name + "' is not a valid symbol index");
1343 GroupSec->setSymTab(SymTab);
1344 GroupSec->setSymbol(Sym);
1345 if (GroupSec->Contents.size() % sizeof(ELF::Elf32_Word) ||
1346 GroupSec->Contents.empty())
1347 error("the content of the section " + GroupSec->Name + " is malformed");
1348 const ELF::Elf32_Word *Word =
1349 reinterpret_cast<const ELF::Elf32_Word *>(GroupSec->Contents.data());
1350 const ELF::Elf32_Word *End =
1351 Word + GroupSec->Contents.size() / sizeof(ELF::Elf32_Word);
1352 GroupSec->setFlagWord(*Word++);
1353 for (; Word != End; ++Word) {
1354 uint32_t Index = support::endian::read32<ELFT::TargetEndianness>(Word);
1355 GroupSec->addMember(SecTable.getSection(
1356 Index, "group member index " + Twine(Index) + " in section '" +
1357 GroupSec->Name + "' is invalid"));
1358 }

1361template <class ELFT>
1362void ELFBuilder<ELFT>::initSymbolTable(SymbolTableSection *SymTab) {
1363 const Elf_Shdr &Shdr = *unwrapOrError(ElfFile.getSection(SymTab->Index));
1364 StringRef StrTabData = unwrapOrError(ElfFile.getStringTableForSymtab(Shdr));
1365 ArrayRef<Elf_Word> ShndxData;
1366
1367 auto Symbols = unwrapOrError(ElfFile.symbols(&Shdr));
1368 for (const auto &Sym : Symbols) {
1369 SectionBase *DefSection = nullptr;
1370 StringRef Name = unwrapOrError(Sym.getName(StrTabData));
1371
1372 if (Sym.st_shndx == SHN_XINDEX) {
1373 if (SymTab->getShndxTable() == nullptr)
1374 error("symbol '" + Name +
1375 "' has index SHN_XINDEX but no SHT_SYMTAB_SHNDX section exists");
1376 if (ShndxData.data() == nullptr) {
1377 const Elf_Shdr &ShndxSec =
1378 *unwrapOrError(ElfFile.getSection(SymTab->getShndxTable()->Index));
1379 ShndxData = unwrapOrError(
1380 ElfFile.template getSectionContentsAsArray<Elf_Word>(&ShndxSec));
1381 if (ShndxData.size() != Symbols.size())
1382 error("symbol section index table does not have the same number of "
1383 "entries as the symbol table");
1384 }
1385 Elf_Word Index = ShndxData[&Sym - Symbols.begin()];
1386 DefSection = Obj.sections().getSection(
1387 Index,
1388 "symbol '" + Name + "' has invalid section index " + Twine(Index));
1389 } else if (Sym.st_shndx >= SHN_LORESERVE) {
1390 if (!isValidReservedSectionIndex(Sym.st_shndx, Obj.Machine)) {
1391 error(
1392 "symbol '" + Name +
1393 "' has unsupported value greater than or equal to SHN_LORESERVE: " +
1394 Twine(Sym.st_shndx));
1395 }
1396 } else if (Sym.st_shndx != SHN_UNDEF) {
1397 DefSection = Obj.sections().getSection(
1398 Sym.st_shndx, "symbol '" + Name +
1399 "' is defined has invalid section index " +
1400 Twine(Sym.st_shndx));
1401 }
1402
1403 SymTab->addSymbol(Name, Sym.getBinding(), Sym.getType(), DefSection,
1404 Sym.getValue(), Sym.st_other, Sym.st_shndx, Sym.st_size);
1405 }

1443template <class ELFT>
1444SectionBase &ELFBuilder<ELFT>::makeSection(const Elf_Shdr &Shdr) {
1445 ArrayRef<uint8_t> Data;
1446 switch (Shdr.sh_type) {
1447 case SHT_REL:
1448 case SHT_RELA:
1449 if (Shdr.sh_flags & SHF_ALLOC) {
1450 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1451 return Obj.addSection<DynamicRelocationSection>(Data);
1452 }
1453 return Obj.addSection<RelocationSection>();
1454 case SHT_STRTAB:
1455 // If a string table is allocated we don't want to mess with it. That would
1456 // mean altering the memory image. There are no special link types or
1457 // anything so we can just use a Section.
1458 if (Shdr.sh_flags & SHF_ALLOC) {
1459 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1460 return Obj.addSection<Section>(Data);
1461 }
1462 return Obj.addSection<StringTableSection>();
1463 case SHT_HASH:
1464 case SHT_GNU_HASH:
1465 // Hash tables should refer to SHT_DYNSYM which we're not going to change.
1466 // Because of this we don't need to mess with the hash tables either.
1467 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1468 return Obj.addSection<Section>(Data);
1469 case SHT_GROUP:
1470 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1471 return Obj.addSection<GroupSection>(Data);
1472 case SHT_DYNSYM:
1473 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1474 return Obj.addSection<DynamicSymbolTableSection>(Data);
1475 case SHT_DYNAMIC:
1476 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1477 return Obj.addSection<DynamicSection>(Data);
1478 case SHT_SYMTAB: {
1479 auto &SymTab = Obj.addSection<SymbolTableSection>();
1480 Obj.SymbolTable = &SymTab;
1481 return SymTab;
1482 }
1483 case SHT_SYMTAB_SHNDX: {
1484 auto &ShndxSection = Obj.addSection<SectionIndexSection>();
1485 Obj.SectionIndexTable = &ShndxSection;
1486 return ShndxSection;
1487 }
1488 case SHT_NOBITS:
1489 return Obj.addSection<Section>(Data);
1490 default: {
1491 Data = unwrapOrError(ElfFile.getSectionContents(&Shdr));
1492
1493 StringRef Name = unwrapOrError(ElfFile.getSectionName(&Shdr));
1494 if (Name.startswith(".zdebug") || (Shdr.sh_flags & ELF::SHF_COMPRESSED)) {
1495 uint64_t DecompressedSize, DecompressedAlign;
1496 std::tie(DecompressedSize, DecompressedAlign) =
1497 getDecompressedSizeAndAlignment<ELFT>(Data);
1498 return Obj.addSection<CompressedSection>(Data, DecompressedSize,
1499 DecompressedAlign);
1500 }
1501
1502 return Obj.addSection<Section>(Data);
1503 }
1504 }

1507template <class ELFT> void ELFBuilder<ELFT>::readSectionHeaders() {
1508 uint32_t Index = 0;
1509 for (const auto &Shdr : unwrapOrError(ElfFile.sections())) {
1510 if (Index == 0) {
1511 ++Index;
1512 continue;
1513 }
1514 auto &Sec = makeSection(Shdr);
1515 Sec.Name = unwrapOrError(ElfFile.getSectionName(&Shdr));
1516 Sec.Type = Shdr.sh_type;
1517 Sec.Flags = Shdr.sh_flags;
1518 Sec.Addr = Shdr.sh_addr;
1519 Sec.Offset = Shdr.sh_offset;
1520 Sec.OriginalOffset = Shdr.sh_offset;
1521 Sec.Size = Shdr.sh_size;
1522 Sec.Link = Shdr.sh_link;
1523 Sec.Info = Shdr.sh_info;
1524 Sec.Align = Shdr.sh_addralign;
1525 Sec.EntrySize = Shdr.sh_entsize;
1526 Sec.Index = Index++;
1527 Sec.OriginalData =
1528 ArrayRef<uint8_t>(ElfFile.base() + Shdr.sh_offset,
1529 (Shdr.sh_type == SHT_NOBITS) ? 0 : Shdr.sh_size);
1530 }

1533template <class ELFT> void ELFBuilder<ELFT>::readSections() {
1534 // If a section index table exists we'll need to initialize it before we
1535 // initialize the symbol table because the symbol table might need to
1536 // reference it.
1537 if (Obj.SectionIndexTable)
1538 Obj.SectionIndexTable->initialize(Obj.sections());
1539
1540 // Now that all of the sections have been added we can fill out some extra
1541 // details about symbol tables. We need the symbol table filled out before
1542 // any relocations.
1543 if (Obj.SymbolTable) {
1544 Obj.SymbolTable->initialize(Obj.sections());
1545 initSymbolTable(Obj.SymbolTable);
1546 }
1547
1548 // Now that all sections and symbols have been added we can add
1549 // relocations that reference symbols and set the link and info fields for
1550 // relocation sections.
1551 for (auto &Section : Obj.sections()) {
1552 if (&Section == Obj.SymbolTable)
1553 continue;
1554 Section.initialize(Obj.sections());
1555 if (auto RelSec = dyn_cast<RelocationSection>(&Section)) {
1556 auto Shdr = unwrapOrError(ElfFile.sections()).begin() + RelSec->Index;
1557 if (RelSec->Type == SHT_REL)
1558 initRelocations(RelSec, Obj.SymbolTable,
1559 unwrapOrError(ElfFile.rels(Shdr)));
1560 else
1561 initRelocations(RelSec, Obj.SymbolTable,
1562 unwrapOrError(ElfFile.relas(Shdr)));
1563 } else if (auto GroupSec = dyn_cast<GroupSection>(&Section)) {
1564 initGroupSection(GroupSec);
1565 }
1566 }
1567
1568 uint32_t ShstrIndex = ElfFile.getHeader()->e_shstrndx;
1569 if (ShstrIndex == SHN_XINDEX)
1570 ShstrIndex = unwrapOrError(ElfFile.getSection(0))->sh_link;
1571
1572 if (ShstrIndex == SHN_UNDEF)
1573 Obj.HadShdrs = false;
1574 else
1575 Obj.SectionNames =
1576 Obj.sections().template getSectionOfType<StringTableSection>(
1577 ShstrIndex,
1578 "e_shstrndx field value " + Twine(ShstrIndex) + " in elf header " +
1579 " is invalid",
1580 "e_shstrndx field value " + Twine(ShstrIndex) + " in elf header " +
1581 " is not a string table");
1582}

1586 findEhdrOffset();
1587
1588 // The ELFFile whose ELF headers and program headers are copied into the
1589 // output file. Normally the same as ElfFile, but if we're extracting a
1590 // loadable partition it will point to the partition's headers.
1591 ELFFile<ELFT> HeadersFile = unwrapOrError(ELFFile<ELFT>::create(toStringRef(
1592 {ElfFile.base() + EhdrOffset, ElfFile.getBufSize() - EhdrOffset})));

1598 Obj.Machine = Ehdr.e_machine;
1599 Obj.Version = Ehdr.e_version;
1600 Obj.Entry = Ehdr.e_entry;
1601 Obj.Flags = Ehdr.e_flags;
1602
1603 readSections();

1605}

1615Expected<std::vector<IHexRecord>> IHexReader::parse() const {
1616 SmallVector<StringRef, 16> Lines;
1617 std::vector<IHexRecord> Records;
1618 bool HasSections = false;
1619
1620 MemBuf->getBuffer().split(Lines, '\n');
1621 Records.reserve(Lines.size());
1622 for (size_t LineNo = 1; LineNo <= Lines.size(); ++LineNo) {
1623 StringRef Line = Lines[LineNo - 1].trim();
1624 if (Line.empty())
1625 continue;
1626
1627 Expected<IHexRecord> R = IHexRecord::parse(Line);
1628 if (!R)
1629 return parseError(LineNo, R.takeError());
1630 if (R->Type == IHexRecord::EndOfFile)
1631 break;
1632 HasSections |= (R->Type == IHexRecord::Data);
1633 Records.push_back(*R);
1634 }
1635 if (!HasSections)
1636 return parseError(-1U, "no sections");
1637
1638 return std::move(Records);
1639}

1668template <class ELFT> void ELFWriter<ELFT>::writeEhdr() {
1669 Elf_Ehdr &Ehdr = *reinterpret_cast<Elf_Ehdr *>(Buf.getBufferStart());
1670 std::fill(Ehdr.e_ident, Ehdr.e_ident + 16, 0);
1671 Ehdr.e_ident[EI_MAG0] = 0x7f;
1672 Ehdr.e_ident[EI_MAG1] = 'E';
1673 Ehdr.e_ident[EI_MAG2] = 'L';
1674 Ehdr.e_ident[EI_MAG3] = 'F';
1675 Ehdr.e_ident[EI_CLASS] = ELFT::Is64Bits ? ELFCLASS64 : ELFCLASS32;
1676 Ehdr.e_ident[EI_DATA] =
1677 ELFT::TargetEndianness == support::big ? ELFDATA2MSB : ELFDATA2LSB;
1678 Ehdr.e_ident[EI_VERSION] = EV_CURRENT;
1679 Ehdr.e_ident[EI_OSABI] = Obj.OSABI;
1680 Ehdr.e_ident[EI_ABIVERSION] = Obj.ABIVersion;
1681
1682 Ehdr.e_type = Obj.Type;
1683 Ehdr.e_machine = Obj.Machine;
1684 Ehdr.e_version = Obj.Version;
1685 Ehdr.e_entry = Obj.Entry;
1686 // We have to use the fully-qualified name llvm::size
1687 // since some compilers complain on ambiguous resolution.
1688 Ehdr.e_phnum = llvm::size(Obj.segments());
1689 Ehdr.e_phoff = (Ehdr.e_phnum != 0) ? Obj.ProgramHdrSegment.Offset : 0;
1690 Ehdr.e_phentsize = (Ehdr.e_phnum != 0) ? sizeof(Elf_Phdr) : 0;
1691 Ehdr.e_flags = Obj.Flags;
1692 Ehdr.e_ehsize = sizeof(Elf_Ehdr);
1693 if (WriteSectionHeaders && Obj.sections().size() != 0) {
1694 Ehdr.e_shentsize = sizeof(Elf_Shdr);
1695 Ehdr.e_shoff = Obj.SHOffset;
1696 // """
1697 // If the number of sections is greater than or equal to
1698 // SHN_LORESERVE (0xff00), this member has the value zero and the actual
1699 // number of section header table entries is contained in the sh_size field
1700 // of the section header at index 0.
1701 // """
1702 auto Shnum = Obj.sections().size() + 1;
1703 if (Shnum >= SHN_LORESERVE)
1704 Ehdr.e_shnum = 0;
1705 else
1706 Ehdr.e_shnum = Shnum;
1707 // """
1708 // If the section name string table section index is greater than or equal
1709 // to SHN_LORESERVE (0xff00), this member has the value SHN_XINDEX (0xffff)
1710 // and the actual index of the section name string table section is
1711 // contained in the sh_link field of the section header at index 0.
1712 // """
1713 if (Obj.SectionNames->Index >= SHN_LORESERVE)
1714 Ehdr.e_shstrndx = SHN_XINDEX;
1715 else
1716 Ehdr.e_shstrndx = Obj.SectionNames->Index;
1717 } else {
1718 Ehdr.e_shentsize = 0;
1719 Ehdr.e_shoff = 0;
1720 Ehdr.e_shnum = 0;
1721 Ehdr.e_shstrndx = 0;
1722 }
1723}

1760 for (SectionBase &Sec : Obj.sections())

1762 // section data if the section is not in a segment. Note that this renders

1768template <class ELFT> void ELFWriter<ELFT>::writeSegmentData() {
1769 for (Segment &Seg : Obj.segments()) {
1770 uint8_t *B = Buf.getBufferStart() + Seg.Offset;
1771 assert(Seg.FileSize == Seg.getContents().size() &&
1772 "Segment size must match contents size");
1773 std::memcpy(B, Seg.getContents().data(), Seg.FileSize);
1774 }
1775
1776 // Iterate over removed sections and overwrite their old data with zeroes.
1777 for (auto &Sec : Obj.removedSections()) {
1778 Segment *Parent = Sec.ParentSegment;
1779 if (Parent == nullptr || Sec.Type == SHT_NOBITS || Sec.Size == 0)
1780 continue;
1781 uint64_t Offset =
1782 Sec.OriginalOffset - Parent->OriginalOffset + Parent->Offset;
1783 std::memset(Buf.getBufferStart() + Offset, 0, Sec.Size);
1784 }

1791Error Object::removeSections(bool AllowBrokenLinks,
1792 std::function<bool(const SectionBase &)> ToRemove) {
1793
1794 auto Iter = std::stable_partition(
1795 std::begin(Sections), std::end(Sections), [=](const SecPtr &Sec) {
1796 if (ToRemove(*Sec))
1797 return false;
1798 if (auto RelSec = dyn_cast<RelocationSectionBase>(Sec.get())) {
1799 if (auto ToRelSec = RelSec->getSection())
1800 return !ToRemove(*ToRelSec);
1801 }
1802 return true;
1803 });
1804 if (SymbolTable != nullptr && ToRemove(*SymbolTable))
1805 SymbolTable = nullptr;
1806 if (SectionNames != nullptr && ToRemove(*SectionNames))
1807 SectionNames = nullptr;
1808 if (SectionIndexTable != nullptr && ToRemove(*SectionIndexTable))
1809 SectionIndexTable = nullptr;
1810 // Now make sure there are no remaining references to the sections that will
1811 // be removed. Sometimes it is impossible to remove a reference so we emit
1812 // an error here instead.
1813 std::unordered_set<const SectionBase *> RemoveSections;
1814 RemoveSections.reserve(std::distance(Iter, std::end(Sections)));
1815 for (auto &RemoveSec : make_range(Iter, std::end(Sections))) {
1816 for (auto &Segment : Segments)
1817 Segment->removeSection(RemoveSec.get());
1818 RemoveSections.insert(RemoveSec.get());
1819 }
1820
1821 // For each section that remains alive, we want to remove the dead references.
1822 // This either might update the content of the section (e.g. remove symbols
1823 // from symbol table that belongs to removed section) or trigger an error if
1824 // a live section critically depends on a section being removed somehow
1825 // (e.g. the removed section is referenced by a relocation).
1826 for (auto &KeepSec : make_range(std::begin(Sections), Iter)) {
1827 if (Error E = KeepSec->removeSectionReferences(AllowBrokenLinks,
1828 [&RemoveSections](const SectionBase *Sec) {
1829 return RemoveSections.find(Sec) != RemoveSections.end();
1830 }))
1831 return E;
1832 }
1833
1834 // Transfer removed sections into the Object RemovedSections container for use
1835 // later.
1836 std::move(Iter, Sections.end(), std::back_inserter(RemovedSections));
1837 // Now finally get rid of them all together.
1838 Sections.erase(Iter, std::end(Sections));
1839 return Error::success();
1840}

1850void Object::sortSections() {
1851 // Use stable_sort to maintain the original ordering as closely as possible.
1852 llvm::stable_sort(Sections, [](const SecPtr &A, const SecPtr &B) {
1853 // Put SHT_GROUP sections first, since group section headers must come
1854 // before the sections they contain. This also matches what GNU objcopy
1855 // does.
1856 if (A->Type != B->Type &&
1857 (A->Type == ELF::SHT_GROUP || B->Type == ELF::SHT_GROUP))
1858 return A->Type == ELF::SHT_GROUP;
1859 // For all other sections, sort by offset order.
1860 return A->OriginalOffset < B->OriginalOffset;
1861 });
1862}

1864static uint64_t alignToAddr(uint64_t Offset, uint64_t Addr, uint64_t Align) {
1865 // Calculate Diff such that (Offset + Diff) & -Align == Addr & -Align.
1866 if (Align == 0)
1867 Align = 1;
1868 auto Diff =
1869 static_cast<int64_t>(Addr % Align) - static_cast<int64_t>(Offset % Align);
1870 // We only want to add to Offset, however, so if Diff < 0 we can add Align and
1871 // (Offset + Diff) & -Align == Addr & -Align will still hold.
1872 if (Diff < 0)
1873 Diff += Align;
1874 return Offset + Diff;
1875}

1877// Orders segments such that if x = y->ParentSegment then y comes before x.
1878static void orderSegments(std::vector<Segment *> &Segments) {
1879 llvm::stable_sort(Segments, compareSegmentsByOffset);

1882// This function finds a consistent layout for a list of segments starting from
1883// an Offset. It assumes that Segments have been sorted by OrderSegments and
1884// returns an Offset one past the end of the last segment.
1885static uint64_t layoutSegments(std::vector<Segment *> &Segments,
1886 uint64_t Offset) {
1887 assert(std::is_sorted(std::begin(Segments), std::end(Segments),
1888 compareSegmentsByOffset));
1889 // The only way a segment should move is if a section was between two
1890 // segments and that section was removed. If that section isn't in a segment
1891 // then it's acceptable, but not ideal, to simply move it to after the
1892 // segments. So we can simply layout segments one after the other accounting
1893 // for alignment.
1894 for (Segment *Seg : Segments) {
1895 // We assume that segments have been ordered by OriginalOffset and Index
1896 // such that a parent segment will always come before a child segment in
1897 // OrderedSegments. This means that the Offset of the ParentSegment should
1898 // already be set and we can set our offset relative to it.
1899 if (Seg->ParentSegment != nullptr) {
1900 Segment *Parent = Seg->ParentSegment;
1901 Seg->Offset =
1902 Parent->Offset + Seg->OriginalOffset - Parent->OriginalOffset;
1903 } else {
1904 Offset = alignToAddr(Offset, Seg->VAddr, Seg->Align);
1905 Seg->Offset = Offset;
1906 }
1907 Offset = std::max(Offset, Seg->Offset + Seg->FileSize);
1908 }
1909 return Offset;
1910}

1912// This function finds a consistent layout for a list of sections. It assumes
1913// that the ->ParentSegment of each section has already been laid out. The
1914// supplied starting Offset is used for the starting offset of any section that
1915// does not have a ParentSegment. It returns either the offset given if all
1916// sections had a ParentSegment or an offset one past the last section if there
1917// was a section that didn't have a ParentSegment.
1918template <class Range>
1919static uint64_t layoutSections(Range Sections, uint64_t Offset) {
1920 // Now the offset of every segment has been set we can assign the offsets
1921 // of each section. For sections that are covered by a segment we should use
1922 // the segment's original offset and the section's original offset to compute
1923 // the offset from the start of the segment. Using the offset from the start
1924 // of the segment we can assign a new offset to the section. For sections not
1925 // covered by segments we can just bump Offset to the next valid location.
1926 uint32_t Index = 1;
1927 for (auto &Section : Sections) {
1928 Section.Index = Index++;
1929 if (Section.ParentSegment != nullptr) {
1930 auto Segment = *Section.ParentSegment;
1931 Section.Offset =
1932 Segment.Offset + (Section.OriginalOffset - Segment.OriginalOffset);
1933 } else {
1934 Offset = alignTo(Offset, Section.Align == 0 ? 1 : Section.Align);
1935 Section.Offset = Offset;
1936 if (Section.Type != SHT_NOBITS)
1937 Offset += Section.Size;
1938 }
1939 }
1940 return Offset;
1941}

1953template <class ELFT> void ELFWriter<ELFT>::assignOffsets() {
1954 // We need a temporary list of segments that has a special order to it
1955 // so that we know that anytime ->ParentSegment is set that segment has
1956 // already had its offset properly set.
1957 std::vector<Segment *> OrderedSegments;
1958 for (Segment &Segment : Obj.segments())
1959 OrderedSegments.push_back(&Segment);
1960 OrderedSegments.push_back(&Obj.ElfHdrSegment);
1961 OrderedSegments.push_back(&Obj.ProgramHdrSegment);
1962 orderSegments(OrderedSegments);
1963 // Offset is used as the start offset of the first segment to be laid out.
1964 // Since the ELF Header (ElfHdrSegment) must be at the start of the file,
1965 // we start at offset 0.
1966 uint64_t Offset = 0;
1967 Offset = layoutSegments(OrderedSegments, Offset);
1968 Offset = layoutSections(Obj.sections(), Offset);
1969 // If we need to write the section header table out then we need to align the
1970 // Offset so that SHOffset is valid.
1971 if (WriteSectionHeaders)
1972 Offset = alignTo(Offset, sizeof(Elf_Addr));
1973 Obj.SHOffset = Offset;
1974}

1997template <class ELFT> Error ELFWriter<ELFT>::finalize() {
1998 // It could happen that SectionNames has been removed and yet the user wants
1999 // a section header table output. We need to throw an error if a user tries
2000 // to do that.
2001 if (Obj.SectionNames == nullptr && WriteSectionHeaders)
2002 return createStringError(llvm::errc::invalid_argument,
2003 "cannot write section header table because "
2004 "section header string table was removed");
2005
2006 Obj.sortSections();
2007
2008 // We need to assign indexes before we perform layout because we need to know
2009 // if we need large indexes or not. We can assign indexes first and check as
2010 // we go to see if we will actully need large indexes.
2011 bool NeedsLargeIndexes = false;
2012 if (Obj.sections().size() >= SHN_LORESERVE) {
2013 SectionTableRef Sections = Obj.sections();
2014 NeedsLargeIndexes =
2015 std::any_of(Sections.begin() + SHN_LORESERVE, Sections.end(),
2016 [](const SectionBase &Sec) { return Sec.HasSymbol; });
2017 // TODO: handle case where only one section needs the large index table but
2018 // only needs it because the large index table hasn't been removed yet.
2019 }
2020
2021 if (NeedsLargeIndexes) {
2022 // This means we definitely need to have a section index table but if we
2023 // already have one then we should use it instead of making a new one.
2024 if (Obj.SymbolTable != nullptr && Obj.SectionIndexTable == nullptr) {
2025 // Addition of a section to the end does not invalidate the indexes of
2026 // other sections and assigns the correct index to the new section.
2027 auto &Shndx = Obj.addSection<SectionIndexSection>();
2028 Obj.SymbolTable->setShndxTable(&Shndx);
2029 Shndx.setSymTab(Obj.SymbolTable);
2030 }
2031 } else {
2032 // Since we don't need SectionIndexTable we should remove it and all
2033 // references to it.
2034 if (Obj.SectionIndexTable != nullptr) {
2035 // We do not support sections referring to the section index table.
2036 if (Error E = Obj.removeSections(false /*AllowBrokenLinks*/,
2037 [this](const SectionBase &Sec) {
2038 return &Sec == Obj.SectionIndexTable;
2039 }))
2040 return E;
2041 }
2042 }
2043
2044 // Make sure we add the names of all the sections. Importantly this must be
2045 // done after we decide to add or remove SectionIndexes.
2046 if (Obj.SectionNames != nullptr)
2047 for (const auto &Section : Obj.sections()) {
2048 Obj.SectionNames->addString(Section.Name);
2049 }
2050
2051 initEhdrSegment();
2052
2053 // Before we can prepare for layout the indexes need to be finalized.
2054 // Also, the output arch may not be the same as the input arch, so fix up
2055 // size-related fields before doing layout calculations.
2056 uint64_t Index = 0;
2057 auto SecSizer = llvm::make_unique<ELFSectionSizer<ELFT>>();
2058 for (auto &Sec : Obj.sections()) {
2059 Sec.Index = Index++;
2060 Sec.accept(*SecSizer);
2061 }
2062
2063 // The symbol table does not update all other sections on update. For
2064 // instance, symbol names are not added as new symbols are added. This means
2065 // that some sections, like .strtab, don't yet have their final size.
2066 if (Obj.SymbolTable != nullptr)
2067 Obj.SymbolTable->prepareForLayout();
2068
2069 // Now that all strings are added we want to finalize string table builders,
2070 // because that affects section sizes which in turn affects section offsets.
2071 for (SectionBase &Sec : Obj.sections())
2072 if (auto StrTab = dyn_cast<StringTableSection>(&Sec))
2073 StrTab->prepareForLayout();
2074
2075 assignOffsets();
2076
2077 // layoutSections could have modified section indexes, so we need
2078 // to fill the index table after assignOffsets.
2079 if (Obj.SymbolTable != nullptr)
2080 Obj.SymbolTable->fillShndxTable();
2081
2082 // Finally now that all offsets and indexes have been set we can finalize any
2083 // remaining issues.
2084 uint64_t Offset = Obj.SHOffset + sizeof(Elf_Shdr);
2085 for (SectionBase &Section : Obj.sections()) {
2086 Section.HeaderOffset = Offset;
2087 Offset += sizeof(Elf_Shdr);
2088 if (WriteSectionHeaders)
2089 Section.NameIndex = Obj.SectionNames->findIndex(Section.Name);
2090 Section.finalize();
2091 }
2092
2093 if (Error E = Buf.allocate(totalSize()))
2094 return E;
2095 SecWriter = llvm::make_unique<ELFSectionWriter<ELFT>>(Buf);
2096 return Error::success();
2097}

2106Error BinaryWriter::finalize() {
2107 // TODO: Create a filter range to construct OrderedSegments from so that this
2108 // code can be deduped with assignOffsets above. This should also solve the
2109 // todo below for LayoutSections.
2110 // We need a temporary list of segments that has a special order to it
2111 // so that we know that anytime ->ParentSegment is set that segment has
2112 // already had it's offset properly set. We only want to consider the segments
2113 // that will affect layout of allocated sections so we only add those.
2114 std::vector<Segment *> OrderedSegments;
2115 for (SectionBase &Section : Obj.sections())
2116 if ((Section.Flags & SHF_ALLOC) != 0 && Section.ParentSegment != nullptr)
2117 OrderedSegments.push_back(Section.ParentSegment);
2118
2119 // For binary output, we're going to use physical addresses instead of
2120 // virtual addresses, since a binary output is used for cases like ROM
2121 // loading and physical addresses are intended for ROM loading.
2122 // However, if no segment has a physical address, we'll fallback to using
2123 // virtual addresses for all.
2124 if (all_of(OrderedSegments,
2125 [](const Segment *Seg) { return Seg->PAddr == 0; }))
2126 for (Segment *Seg : OrderedSegments)
2127 Seg->PAddr = Seg->VAddr;
2128
2129 llvm::stable_sort(OrderedSegments, compareSegmentsByPAddr);
2130
2131 // Because we add a ParentSegment for each section we might have duplicate
2132 // segments in OrderedSegments. If there were duplicates then LayoutSegments
2133 // would do very strange things.
2134 auto End =
2135 std::unique(std::begin(OrderedSegments), std::end(OrderedSegments));
2136 OrderedSegments.erase(End, std::end(OrderedSegments));
2137
2138 uint64_t Offset = 0;
2139
2140 // Modify the first segment so that there is no gap at the start. This allows
2141 // our layout algorithm to proceed as expected while not writing out the gap
2142 // at the start.
2143 if (!OrderedSegments.empty()) {
2144 Segment *Seg = OrderedSegments[0];
2145 const SectionBase *Sec = Seg->firstSection();
2146 auto Diff = Sec->OriginalOffset - Seg->OriginalOffset;
2147 Seg->OriginalOffset += Diff;
2148 // The size needs to be shrunk as well.
2149 Seg->FileSize -= Diff;
2150 // The PAddr needs to be increased to remove the gap before the first
2151 // section.
2152 Seg->PAddr += Diff;
2153 uint64_t LowestPAddr = Seg->PAddr;
2154 for (Segment *Segment : OrderedSegments) {
2155 Segment->Offset = Segment->PAddr - LowestPAddr;
2156 Offset = std::max(Offset, Segment->Offset + Segment->FileSize);
2157 }
2158 }
2159
2160 // TODO: generalize LayoutSections to take a range. Pass a special range
2161 // constructed from an iterator that skips values for which a predicate does
2162 // not hold. Then pass such a range to LayoutSections instead of constructing
2163 // AllocatedSections here.
2164 std::vector<SectionBase *> AllocatedSections;
2165 for (SectionBase &Section : Obj.sections())
2166 if (Section.Flags & SHF_ALLOC)
2167 AllocatedSections.push_back(&Section);
2168 layoutSections(make_pointee_range(AllocatedSections), Offset);
2169
2170 // Now that every section has been laid out we just need to compute the total
2171 // file size. This might not be the same as the offset returned by
2172 // LayoutSections, because we want to truncate the last segment to the end of
2173 // its last section, to match GNU objcopy's behaviour.
2174 TotalSize = 0;
2175 for (SectionBase *Section : AllocatedSections)
2176 if (Section->Type != SHT_NOBITS)
2177 TotalSize = std::max(TotalSize, Section->Offset + Section->Size);
2178
2179 if (Error E = Buf.allocate(TotalSize))
2180 return E;
2181 SecWriter = llvm::make_unique<BinarySectionWriter>(Buf);
2182 return Error::success();
2183}

2243Error IHexWriter::finalize() {
2244 bool UseSegments = false;
2245 auto ShouldWrite = [](const SectionBase &Sec) {
2246 return (Sec.Flags & ELF::SHF_ALLOC) && (Sec.Type != ELF::SHT_NOBITS);
2247 };
2248 auto IsInPtLoad = [](const SectionBase &Sec) {
2249 return Sec.ParentSegment && Sec.ParentSegment->Type == ELF::PT_LOAD;
2250 };
2251
2252 // We can't write 64-bit addresses.
2253 if (addressOverflows32bit(Obj.Entry))
2254 return createStringError(errc::invalid_argument,
2255 "Entry point address 0x%llx overflows 32 bits.",
2256 Obj.Entry);
2257
2258 // If any section we're to write has segment then we
2259 // switch to using physical addresses. Otherwise we
2260 // use section virtual address.
2261 for (auto &Section : Obj.sections())
2262 if (ShouldWrite(Section) && IsInPtLoad(Section)) {
2263 UseSegments = true;
2264 break;
2265 }
2266
2267 for (auto &Section : Obj.sections())
2268 if (ShouldWrite(Section) && (!UseSegments || IsInPtLoad(Section))) {
2269 if (Error E = checkSection(Section))
2270 return E;
2271 Sections.insert(&Section);
2272 }
2273
2274 IHexSectionWriterBase LengthCalc(Buf);
2275 for (const SectionBase *Sec : Sections)
2276 Sec->accept(LengthCalc);
2277
2278 // We need space to write section records + StartAddress record
2279 // (if start adress is not zero) + EndOfFile record.
2280 TotalSize = LengthCalc.getBufferOffset() +
2281 (Obj.Entry ? IHexRecord::getLineLength(4) : 0) +
2282 IHexRecord::getLineLength(0);
