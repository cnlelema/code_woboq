56static bool isDebugSection(const SectionBase &Sec) {
57 return StringRef(Sec.Name).startswith(".debug") ||
58 StringRef(Sec.Name).startswith(".zdebug") || Sec.Name == ".gdb_index";
59}

61static bool isDWOSection(const SectionBase &Sec) {
62 return StringRef(Sec.Name).endswith(".dwo");
63}

65static bool onlyKeepDWOPred(const Object &Obj, const SectionBase &Sec) {
66 // We can't remove the section header string table.
67 if (&Sec == Obj.SectionNames)
68 return false;
69 // Short of keeping the string table we want to keep everything that is a DWO
70 // section and remove everything else.
71 return !isDWOSection(Sec);
72}

74uint64_t getNewShfFlags(SectionFlag AllFlags) {
75 uint64_t NewFlags = 0;
76 if (AllFlags & SectionFlag::SecAlloc)
77 NewFlags |= ELF::SHF_ALLOC;
78 if (!(AllFlags & SectionFlag::SecReadonly))
79 NewFlags |= ELF::SHF_WRITE;
80 if (AllFlags & SectionFlag::SecCode)
81 NewFlags |= ELF::SHF_EXECINSTR;
82 if (AllFlags & SectionFlag::SecMerge)
83 NewFlags |= ELF::SHF_MERGE;
84 if (AllFlags & SectionFlag::SecStrings)
85 NewFlags |= ELF::SHF_STRINGS;
86 return NewFlags;
87}
88
89static uint64_t getSectionFlagsPreserveMask(uint64_t OldFlags,
90 uint64_t NewFlags) {
91 // Preserve some flags which should not be dropped when setting flags.
92 // Also, preserve anything OS/processor dependant.
93 const uint64_t PreserveMask = ELF::SHF_COMPRESSED | ELF::SHF_EXCLUDE |
94 ELF::SHF_GROUP | ELF::SHF_LINK_ORDER |
95 ELF::SHF_MASKOS | ELF::SHF_MASKPROC |
96 ELF::SHF_TLS | ELF::SHF_INFO_LINK;
97 return (OldFlags & PreserveMask) | (NewFlags & ~PreserveMask);
98}
99
100static void setSectionFlagsAndType(SectionBase &Sec, SectionFlag Flags) {
101 Sec.Flags = getSectionFlagsPreserveMask(Sec.Flags, getNewShfFlags(Flags));
102
103 // In GNU objcopy, certain flags promote SHT_NOBITS to SHT_PROGBITS. This rule
104 // may promote more non-ALLOC sections than GNU objcopy, but it is fine as
105 // non-ALLOC SHT_NOBITS sections do not make much sense.
106 if (Sec.Type == SHT_NOBITS &&
107 (!(Sec.Flags & ELF::SHF_ALLOC) ||
108 Flags & (SectionFlag::SecContents | SectionFlag::SecLoad)))
109 Sec.Type = SHT_PROGBITS;
110}

112static ElfType getOutputElfType(const Binary &Bin) {
113 // Infer output ELF type from the input ELF object
114 if (isa<ELFObjectFile<ELF32LE>>(Bin))
115 return ELFT_ELF32LE;
116 if (isa<ELFObjectFile<ELF64LE>>(Bin))
117 return ELFT_ELF64LE;
118 if (isa<ELFObjectFile<ELF32BE>>(Bin))
119 return ELFT_ELF32BE;
120 if (isa<ELFObjectFile<ELF64BE>>(Bin))
121 return ELFT_ELF64BE;
122 llvm_unreachable("Invalid ELFType");
123}

125static ElfType getOutputElfType(const MachineInfo &MI) {
126 // Infer output ELF type from the binary arch specified
127 if (MI.Is64Bit)
128 return MI.IsLittleEndian ? ELFT_ELF64LE : ELFT_ELF64BE;
129 else
130 return MI.IsLittleEndian ? ELFT_ELF32LE : ELFT_ELF32BE;
131}

133static std::unique_ptr<Writer> createELFWriter(const CopyConfig &Config,
134 Object &Obj, Buffer &Buf,
135 ElfType OutputElfType) {
136 // Depending on the initial ELFT and OutputFormat we need a different Writer.
137 switch (OutputElfType) {
138 case ELFT_ELF32LE:
139 return llvm::make_unique<ELFWriter<ELF32LE>>(Obj, Buf,
140 !Config.StripSections);
141 case ELFT_ELF64LE:
142 return llvm::make_unique<ELFWriter<ELF64LE>>(Obj, Buf,
143 !Config.StripSections);
144 case ELFT_ELF32BE:
145 return llvm::make_unique<ELFWriter<ELF32BE>>(Obj, Buf,
146 !Config.StripSections);
147 case ELFT_ELF64BE:
148 return llvm::make_unique<ELFWriter<ELF64BE>>(Obj, Buf,
149 !Config.StripSections);
150 }
151 llvm_unreachable("Invalid output format");
152}

154static std::unique_ptr<Writer> createWriter(const CopyConfig &Config,
155 Object &Obj, Buffer &Buf,
156 ElfType OutputElfType) {
157 using Functor = std::function<std::unique_ptr<Writer>()>;
158 return StringSwitch<Functor>(Config.OutputFormat)
159 .Case("binary", [&] { return llvm::make_unique<BinaryWriter>(Obj, Buf); })
160 .Case("ihex", [&] { return llvm::make_unique<IHexWriter>(Obj, Buf); })
161 .Default(
162 [&] { return createELFWriter(Config, Obj, Buf, OutputElfType); })();
163}

165template <class ELFT>
166static Expected<ArrayRef<uint8_t>>
167findBuildID(const CopyConfig &Config, const object::ELFFile<ELFT> &In) {
168 auto PhdrsOrErr = In.program_headers();
169 if (auto Err = PhdrsOrErr.takeError())
170 return createFileError(Config.InputFilename, std::move(Err));
171
172 for (const auto &Phdr : *PhdrsOrErr) {
173 if (Phdr.p_type != PT_NOTE)
174 continue;
175 Error Err = Error::success();
176 for (const auto &Note : In.notes(Phdr, Err))
177 if (Note.getType() == NT_GNU_BUILD_ID && Note.getName() == ELF_NOTE_GNU)
178 return Note.getDesc();
179 if (Err)
180 return createFileError(Config.InputFilename, std::move(Err));
181 }
182
183 return createFileError(
184 Config.InputFilename,
185 createStringError(llvm::errc::invalid_argument,
186 "could not find build ID"));
187}

189static Expected<ArrayRef<uint8_t>>
190findBuildID(const CopyConfig &Config, const object::ELFObjectFileBase &In) {
191 if (auto *O = dyn_cast<ELFObjectFile<ELF32LE>>(&In))
192 return findBuildID(Config, *O->getELFFile());
193 else if (auto *O = dyn_cast<ELFObjectFile<ELF64LE>>(&In))
194 return findBuildID(Config, *O->getELFFile());
195 else if (auto *O = dyn_cast<ELFObjectFile<ELF32BE>>(&In))
196 return findBuildID(Config, *O->getELFFile());
197 else if (auto *O = dyn_cast<ELFObjectFile<ELF64BE>>(&In))
198 return findBuildID(Config, *O->getELFFile());
199
200 llvm_unreachable("Bad file format");
201}

204static Error makeStringError(std::error_code EC, const Twine &Msg, Ts &&... Args) {
205 std::string FullMsg = (EC.message() + ": " + Msg).str();
206 return createStringError(EC, FullMsg.c_str(), std::forward<Ts>(Args)...);

213static Error linkToBuildIdDir(const CopyConfig &Config, StringRef ToLink,
214 StringRef Suffix,
215 ArrayRef<uint8_t> BuildIdBytes) {
216 SmallString<128> Path = Config.BuildIdLinkDir;
217 sys::path::append(Path, llvm::toHex(BuildIdBytes[0], /*LowerCase*/ true));
218 if (auto EC = sys::fs::create_directories(Path))
219 return createFileError(
220 Path.str(),
221 makeStringError(EC, "cannot create build ID link directory"));
222
223 sys::path::append(Path,
224 llvm::toHex(BuildIdBytes.slice(1), /*LowerCase*/ true));
225 Path += Suffix;
226 SmallString<128> TmpPath;
227 // create_hard_link races so we need to link to a temporary path but
228 // we want to make sure that we choose a filename that does not exist.
229 // By using 32 model characters we get 128-bits of entropy. It is
230 // unlikely that this string has ever existed before much less exists
231 // on this disk or in the current working directory.
232 // Additionally we prepend the original Path for debugging but also
233 // because it ensures that we're linking within a directory on the same
234 // partition on the same device which is critical. It has the added
235 // win of yet further decreasing the odds of a conflict.
236 sys::fs::createUniquePath(Twine(Path) + "-" + MODEL_32 + ".tmp", TmpPath,
237 /*MakeAbsolute*/ false);
238 if (auto EC = sys::fs::create_hard_link(ToLink, TmpPath)) {
239 Path.push_back('\0');
240 return makeStringError(EC, "cannot link '%s' to '%s'", ToLink.data(),
241 Path.data());
242 }
243 // We then atomically rename the link into place which will just move the
244 // link. If rename fails something is more seriously wrong so just return
245 // an error.
246 if (auto EC = sys::fs::rename(TmpPath, Path)) {
247 Path.push_back('\0');
248 return makeStringError(EC, "cannot link '%s' to '%s'", ToLink.data(),
249 Path.data());
250 }
251 // If `Path` was already a hard-link to the same underlying file then the
252 // temp file will be left so we need to remove it. Remove will not cause
253 // an error by default if the file is already gone so just blindly remove
254 // it rather than checking.
255 if (auto EC = sys::fs::remove(TmpPath)) {
256 TmpPath.push_back('\0');
257 return makeStringError(EC, "could not remove '%s'", TmpPath.data());
258 }
259 return Error::success();
260}

262static Error splitDWOToFile(const CopyConfig &Config, const Reader &Reader,
263 StringRef File, ElfType OutputElfType) {
264 auto DWOFile = Reader.create();
265 auto OnlyKeepDWOPred = [&DWOFile](const SectionBase &Sec) {
266 return onlyKeepDWOPred(*DWOFile, Sec);
267 };
268 if (Error E = DWOFile->removeSections(Config.AllowBrokenLinks,
269 OnlyKeepDWOPred))
270 return E;
271 if (Config.OutputArch) {
272 DWOFile->Machine = Config.OutputArch.getValue().EMachine;
273 DWOFile->OSABI = Config.OutputArch.getValue().OSABI;
274 }
275 FileBuffer FB(File);
276 auto Writer = createWriter(Config, *DWOFile, FB, OutputElfType);
277 if (Error E = Writer->finalize())
278 return E;
279 return Writer->write();
280}

282static Error dumpSectionToFile(StringRef SecName, StringRef Filename,
283 Object &Obj) {
284 for (auto &Sec : Obj.sections()) {
285 if (Sec.Name == SecName) {
286 if (Sec.OriginalData.empty())
287 return createStringError(object_error::parse_failed,
288 "cannot dump section '%s': it has no contents",
289 SecName.str().c_str());
290 Expected<std::unique_ptr<FileOutputBuffer>> BufferOrErr =
291 FileOutputBuffer::create(Filename, Sec.OriginalData.size());
292 if (!BufferOrErr)
293 return BufferOrErr.takeError();
294 std::unique_ptr<FileOutputBuffer> Buf = std::move(*BufferOrErr);
295 std::copy(Sec.OriginalData.begin(), Sec.OriginalData.end(),
296 Buf->getBufferStart());
297 if (Error E = Buf->commit())
298 return E;
299 return Error::success();
300 }
301 }
302 return createStringError(object_error::parse_failed, "section '%s' not found",
303 SecName.str().c_str());
304}

311static void replaceDebugSections(
312 Object &Obj, SectionPred &RemovePred,
313 function_ref<bool(const SectionBase &)> shouldReplace,
314 function_ref<SectionBase *(const SectionBase *)> addSection) {
315 // Build a list of the debug sections we are going to replace.
316 // We can't call `addSection` while iterating over sections,
317 // because it would mutate the sections array.
318 SmallVector<SectionBase *, 13> ToReplace;
319 for (auto &Sec : Obj.sections())
320 if (shouldReplace(Sec))
321 ToReplace.push_back(&Sec);
322
323 // Build a mapping from original section to a new one.
324 DenseMap<SectionBase *, SectionBase *> FromTo;
325 for (SectionBase *S : ToReplace)
326 FromTo[S] = addSection(S);
327
328 // Now we want to update the target sections of relocation
329 // sections. Also we will update the relocations themselves
330 // to update the symbol references.
331 for (auto &Sec : Obj.sections())
332 Sec.replaceSectionReferences(FromTo);
333
334 RemovePred = [shouldReplace, RemovePred](const SectionBase &Sec) {
335 return shouldReplace(Sec) || RemovePred(Sec);
336 };
337}

339static bool isUnneededSymbol(const Symbol &Sym) {
340 return !Sym.Referenced &&
341 (Sym.Binding == STB_LOCAL || Sym.getShndx() == SHN_UNDEF) &&
342 Sym.Type != STT_SECTION;
343}

345static Error updateAndRemoveSymbols(const CopyConfig &Config, Object &Obj) {
346 // TODO: update or remove symbols only if there is an option that affects
347 // them.
348 if (!Obj.SymbolTable)
349 return Error::success();
350
351 Obj.SymbolTable->updateSymbols([&](Symbol &Sym) {
352 // Common and undefined symbols don't make sense as local symbols, and can
353 // even cause crashes if we localize those, so skip them.
354 if (!Sym.isCommon() && Sym.getShndx() != SHN_UNDEF &&
355 ((Config.LocalizeHidden &&
356 (Sym.Visibility == STV_HIDDEN || Sym.Visibility == STV_INTERNAL)) ||
357 is_contained(Config.SymbolsToLocalize, Sym.Name)))
358 Sym.Binding = STB_LOCAL;
359
360 // Note: these two globalize flags have very similar names but different
361 // meanings:
362 //
363 // --globalize-symbol: promote a symbol to global
364 // --keep-global-symbol: all symbols except for these should be made local
365 //
366 // If --globalize-symbol is specified for a given symbol, it will be
367 // global in the output file even if it is not included via
368 // --keep-global-symbol. Because of that, make sure to check
369 // --globalize-symbol second.
370 if (!Config.SymbolsToKeepGlobal.empty() &&
371 !is_contained(Config.SymbolsToKeepGlobal, Sym.Name) &&
372 Sym.getShndx() != SHN_UNDEF)
373 Sym.Binding = STB_LOCAL;
374
375 if (is_contained(Config.SymbolsToGlobalize, Sym.Name) &&
376 Sym.getShndx() != SHN_UNDEF)
377 Sym.Binding = STB_GLOBAL;
378
379 if (is_contained(Config.SymbolsToWeaken, Sym.Name) &&
380 Sym.Binding == STB_GLOBAL)
381 Sym.Binding = STB_WEAK;
382
383 if (Config.Weaken && Sym.Binding == STB_GLOBAL &&
384 Sym.getShndx() != SHN_UNDEF)
385 Sym.Binding = STB_WEAK;
386
387 const auto I = Config.SymbolsToRename.find(Sym.Name);
388 if (I != Config.SymbolsToRename.end())
389 Sym.Name = I->getValue();
390
391 if (!Config.SymbolsPrefix.empty() && Sym.Type != STT_SECTION)
392 Sym.Name = (Config.SymbolsPrefix + Sym.Name).str();
393 });
394
395 // The purpose of this loop is to mark symbols referenced by sections
396 // (like GroupSection or RelocationSection). This way, we know which
397 // symbols are still 'needed' and which are not.
398 if (Config.StripUnneeded || !Config.UnneededSymbolsToRemove.empty() ||
399 !Config.OnlySection.empty()) {
400 for (auto &Section : Obj.sections())
401 Section.markSymbols();
402 }
403
404 auto RemoveSymbolsPred = [&](const Symbol &Sym) {
405 if (is_contained(Config.SymbolsToKeep, Sym.Name) ||
406 (Config.KeepFileSymbols && Sym.Type == STT_FILE))
407 return false;
408
409 if ((Config.DiscardMode == DiscardType::All ||
410 (Config.DiscardMode == DiscardType::Locals &&
411 StringRef(Sym.Name).startswith(".L"))) &&
412 Sym.Binding == STB_LOCAL && Sym.getShndx() != SHN_UNDEF &&
413 Sym.Type != STT_FILE && Sym.Type != STT_SECTION)
414 return true;
415
416 if (Config.StripAll || Config.StripAllGNU)
417 return true;
418
419 if (is_contained(Config.SymbolsToRemove, Sym.Name))
420 return true;
421
422 if ((Config.StripUnneeded ||
423 is_contained(Config.UnneededSymbolsToRemove, Sym.Name)) &&
424 isUnneededSymbol(Sym))
425 return true;
426
427 // We want to remove undefined symbols if all references have been stripped.
428 if (!Config.OnlySection.empty() && !Sym.Referenced &&
429 Sym.getShndx() == SHN_UNDEF)
430 return true;
431
432 return false;
433 };
434
435 return Obj.removeSymbols(RemoveSymbolsPred);
436}

438static Error replaceAndRemoveSections(const CopyConfig &Config, Object &Obj) {
439 SectionPred RemovePred = [](const SectionBase &) { return false; };
440
441 // Removes:
442 if (!Config.ToRemove.empty()) {
443 RemovePred = [&Config](const SectionBase &Sec) {
444 return is_contained(Config.ToRemove, Sec.Name);
445 };
446 }
447
448 if (Config.StripDWO || !Config.SplitDWO.empty())
449 RemovePred = [RemovePred](const SectionBase &Sec) {
450 return isDWOSection(Sec) || RemovePred(Sec);
451 };
452
453 if (Config.ExtractDWO)
454 RemovePred = [RemovePred, &Obj](const SectionBase &Sec) {
455 return onlyKeepDWOPred(Obj, Sec) || RemovePred(Sec);
456 };
457
458 if (Config.StripAllGNU)
459 RemovePred = [RemovePred, &Obj](const SectionBase &Sec) {
460 if (RemovePred(Sec))
461 return true;
462 if ((Sec.Flags & SHF_ALLOC) != 0)
463 return false;
464 if (&Sec == Obj.SectionNames)
465 return false;
466 switch (Sec.Type) {
467 case SHT_SYMTAB:
468 case SHT_REL:
469 case SHT_RELA:
470 case SHT_STRTAB:
471 return true;
472 }
473 return isDebugSection(Sec);
474 };
475
476 if (Config.StripSections) {
477 RemovePred = [RemovePred](const SectionBase &Sec) {
478 return RemovePred(Sec) || Sec.ParentSegment == nullptr;
479 };
480 }
481
482 if (Config.StripDebug) {
483 RemovePred = [RemovePred](const SectionBase &Sec) {
484 return RemovePred(Sec) || isDebugSection(Sec);
485 };
486 }
487
488 if (Config.StripNonAlloc)
489 RemovePred = [RemovePred, &Obj](const SectionBase &Sec) {
490 if (RemovePred(Sec))
491 return true;
492 if (&Sec == Obj.SectionNames)
493 return false;
494 return (Sec.Flags & SHF_ALLOC) == 0 && Sec.ParentSegment == nullptr;
495 };
496
497 if (Config.StripAll)
498 RemovePred = [RemovePred, &Obj](const SectionBase &Sec) {
499 if (RemovePred(Sec))
500 return true;
501 if (&Sec == Obj.SectionNames)
502 return false;
503 if (StringRef(Sec.Name).startswith(".gnu.warning"))
504 return false;
505 if (Sec.ParentSegment != nullptr)
506 return false;
507 return (Sec.Flags & SHF_ALLOC) == 0;
508 };
509
510 if (Config.ExtractPartition || Config.ExtractMainPartition) {
511 RemovePred = [RemovePred](const SectionBase &Sec) {
512 if (RemovePred(Sec))
513 return true;
514 if (Sec.Type == SHT_LLVM_PART_EHDR || Sec.Type == SHT_LLVM_PART_PHDR)
515 return true;
516 return (Sec.Flags & SHF_ALLOC) != 0 && !Sec.ParentSegment;
517 };
518 }
519
520 // Explicit copies:
521 if (!Config.OnlySection.empty()) {
522 RemovePred = [&Config, RemovePred, &Obj](const SectionBase &Sec) {
523 // Explicitly keep these sections regardless of previous removes.
524 if (is_contained(Config.OnlySection, Sec.Name))
525 return false;
526
527 // Allow all implicit removes.
528 if (RemovePred(Sec))
529 return true;
530
531 // Keep special sections.
532 if (Obj.SectionNames == &Sec)
533 return false;
534 if (Obj.SymbolTable == &Sec ||
535 (Obj.SymbolTable && Obj.SymbolTable->getStrTab() == &Sec))
536 return false;
537
538 // Remove everything else.
539 return true;
540 };
541 }
542
543 if (!Config.KeepSection.empty()) {
544 RemovePred = [&Config, RemovePred](const SectionBase &Sec) {
545 // Explicitly keep these sections regardless of previous removes.
546 if (is_contained(Config.KeepSection, Sec.Name))
547 return false;
548 // Otherwise defer to RemovePred.
549 return RemovePred(Sec);
550 };
551 }
552
553 // This has to be the last predicate assignment.
554 // If the option --keep-symbol has been specified
555 // and at least one of those symbols is present
556 // (equivalently, the updated symbol table is not empty)
557 // the symbol table and the string table should not be removed.
558 if ((!Config.SymbolsToKeep.empty() || Config.KeepFileSymbols) &&
559 Obj.SymbolTable && !Obj.SymbolTable->empty()) {
560 RemovePred = [&Obj, RemovePred](const SectionBase &Sec) {
561 if (&Sec == Obj.SymbolTable || &Sec == Obj.SymbolTable->getStrTab())
562 return false;
563 return RemovePred(Sec);
564 };
565 }
566
567 if (Config.CompressionType != DebugCompressionType::None)
568 replaceDebugSections(Obj, RemovePred, isCompressable,
569 [&Config, &Obj](const SectionBase *S) {
570 return &Obj.addSection<CompressedSection>(
571 *S, Config.CompressionType);
572 });
573 else if (Config.DecompressDebugSections)
574 replaceDebugSections(
575 Obj, RemovePred,
576 [](const SectionBase &S) { return isa<CompressedSection>(&S); },
577 [&Obj](const SectionBase *S) {
578 auto CS = cast<CompressedSection>(S);
579 return &Obj.addSection<DecompressedSection>(*CS);
580 });
581
582 return Obj.removeSections(Config.AllowBrokenLinks, RemovePred);
583}

585// This function handles the high level operations of GNU objcopy including
586// handling command line options. It's important to outline certain properties
587// we expect to hold of the command line operations. Any operation that "keeps"
588// should keep regardless of a remove. Additionally any removal should respect
589// any previous removals. Lastly whether or not something is removed shouldn't
590// depend a) on the order the options occur in or b) on some opaque priority
591// system. The only priority is that keeps/copies overrule removes.
592static Error handleArgs(const CopyConfig &Config, Object &Obj,
593 const Reader &Reader, ElfType OutputElfType) {
594
595 if (!Config.SplitDWO.empty())
596 if (Error E =
597 splitDWOToFile(Config, Reader, Config.SplitDWO, OutputElfType))
598 return E;
599
600 if (Config.OutputArch) {
601 Obj.Machine = Config.OutputArch.getValue().EMachine;
602 Obj.OSABI = Config.OutputArch.getValue().OSABI;
603 }
604
605 // It is important to remove the sections first. For example, we want to
606 // remove the relocation sections before removing the symbols. That allows
607 // us to avoid reporting the inappropriate errors about removing symbols
608 // named in relocations.
609 if (Error E = replaceAndRemoveSections(Config, Obj))
610 return E;
611
612 if (Error E = updateAndRemoveSymbols(Config, Obj))
613 return E;
614
615 if (!Config.SectionsToRename.empty() || !Config.AllocSectionsPrefix.empty()) {
616 DenseSet<SectionBase *> PrefixedSections;
617 for (auto &Sec : Obj.sections()) {
618 const auto Iter = Config.SectionsToRename.find(Sec.Name);
619 if (Iter != Config.SectionsToRename.end()) {
620 const SectionRename &SR = Iter->second;
621 Sec.Name = SR.NewName;
622 if (SR.NewFlags.hasValue())
623 setSectionFlagsAndType(Sec, SR.NewFlags.getValue());
624 }
625
626 // Add a prefix to allocated sections and their relocation sections. This
627 // should be done after renaming the section by Config.SectionToRename to
628 // imitate the GNU objcopy behavior.
629 if (!Config.AllocSectionsPrefix.empty()) {
630 if (Sec.Flags & SHF_ALLOC) {
631 Sec.Name = (Config.AllocSectionsPrefix + Sec.Name).str();
632 PrefixedSections.insert(&Sec);
633
634 // Rename relocation sections associated to the allocated sections.
635 // For example, if we rename .text to .prefix.text, we also rename
636 // .rel.text to .rel.prefix.text.
637 //
638 // Dynamic relocation sections (SHT_REL[A] with SHF_ALLOC) are handled
639 // above, e.g., .rela.plt is renamed to .prefix.rela.plt, not
640 // .rela.prefix.plt since GNU objcopy does so.
641 } else if (auto *RelocSec = dyn_cast<RelocationSectionBase>(&Sec)) {
642 auto *TargetSec = RelocSec->getSection();
643 if (TargetSec && (TargetSec->Flags & SHF_ALLOC)) {
644 StringRef prefix;
645 switch (Sec.Type) {
646 case SHT_REL:
647 prefix = ".rel";
648 break;
649 case SHT_RELA:
650 prefix = ".rela";
651 break;
652 default:
653 continue;
654 }
655
656 // If the relocation section comes *after* the target section, we
657 // don't add Config.AllocSectionsPrefix because we've already added
658 // the prefix to TargetSec->Name. Otherwise, if the relocation
659 // section comes *before* the target section, we add the prefix.
660 if (PrefixedSections.count(TargetSec)) {
661 Sec.Name = (prefix + TargetSec->Name).str();
662 } else {
663 const auto Iter = Config.SectionsToRename.find(TargetSec->Name);
664 if (Iter != Config.SectionsToRename.end()) {
665 // Both `--rename-section` and `--prefix-alloc-sections` are
666 // given but the target section is not yet renamed.
667 Sec.Name =
668 (prefix + Config.AllocSectionsPrefix + Iter->second.NewName)
669 .str();
670 } else {
671 Sec.Name =
672 (prefix + Config.AllocSectionsPrefix + TargetSec->Name)
673 .str();
674 }
675 }
676 }
677 }
678 }
679 }
680 }
681
682 if (!Config.SetSectionFlags.empty()) {
683 for (auto &Sec : Obj.sections()) {
684 const auto Iter = Config.SetSectionFlags.find(Sec.Name);
685 if (Iter != Config.SetSectionFlags.end()) {
686 const SectionFlagsUpdate &SFU = Iter->second;
687 setSectionFlagsAndType(Sec, SFU.NewFlags);
688 }
689 }
690 }
691
692 for (const auto &Flag : Config.AddSection) {
693 std::pair<StringRef, StringRef> SecPair = Flag.split("=");
694 StringRef SecName = SecPair.first;
695 StringRef File = SecPair.second;
696 ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
697 MemoryBuffer::getFile(File);
698 if (!BufOrErr)
699 return createFileError(File, errorCodeToError(BufOrErr.getError()));
700 std::unique_ptr<MemoryBuffer> Buf = std::move(*BufOrErr);
701 ArrayRef<uint8_t> Data(
702 reinterpret_cast<const uint8_t *>(Buf->getBufferStart()),
703 Buf->getBufferSize());
704 OwnedDataSection &NewSection =
705 Obj.addSection<OwnedDataSection>(SecName, Data);
706 if (SecName.startswith(".note") && SecName != ".note.GNU-stack")
707 NewSection.Type = SHT_NOTE;
708 }
709
710 for (const auto &Flag : Config.DumpSection) {
711 std::pair<StringRef, StringRef> SecPair = Flag.split("=");
712 StringRef SecName = SecPair.first;
713 StringRef File = SecPair.second;
714 if (Error E = dumpSectionToFile(SecName, File, Obj))
715 return E;
716 }
717
718 if (!Config.AddGnuDebugLink.empty())
719 Obj.addSection<GnuDebugLinkSection>(Config.AddGnuDebugLink,
720 Config.GnuDebugLinkCRC32);
721
722 for (const NewSymbolInfo &SI : Config.SymbolsToAdd) {
723 SectionBase *Sec = Obj.findSection(SI.SectionName);
724 uint64_t Value = Sec ? Sec->Addr + SI.Value : SI.Value;
725 Obj.SymbolTable->addSymbol(
726 SI.SymbolName, SI.Bind, SI.Type, Sec, Value, SI.Visibility,
727 Sec ? (uint16_t)SYMBOL_SIMPLE_INDEX : (uint16_t)SHN_ABS, 0);
728 }
729
730 if (Config.EntryExpr)
731 Obj.Entry = Config.EntryExpr(Obj.Entry);
732 return Error::success();
733}

735static Error writeOutput(const CopyConfig &Config, Object &Obj, Buffer &Out,
736 ElfType OutputElfType) {
737 std::unique_ptr<Writer> Writer =
738 createWriter(Config, Obj, Out, OutputElfType);
739 if (Error E = Writer->finalize())
740 return E;
741 return Writer->write();
742}

768
769Error executeObjcopyOnBinary(const CopyConfig &Config,
770 object::ELFObjectFileBase &In, Buffer &Out) {
771 ELFReader Reader(&In, Config.ExtractPartition);
772 std::unique_ptr<Object> Obj = Reader.create();
773 // Prefer OutputArch (-O<format>) if set, otherwise infer it from the input.
774 const ElfType OutputElfType =
775 Config.OutputArch ? getOutputElfType(Config.OutputArch.getValue())
776 : getOutputElfType(In);
777 ArrayRef<uint8_t> BuildIdBytes;
778
779 if (!Config.BuildIdLinkDir.empty()) {
780 auto BuildIdBytesOrErr = findBuildID(Config, In);
781 if (auto E = BuildIdBytesOrErr.takeError())
782 return E;
783 BuildIdBytes = *BuildIdBytesOrErr;
784
785 if (BuildIdBytes.size() < 2)
786 return createFileError(
787 Config.InputFilename,
788 createStringError(object_error::parse_failed,
789 "build ID is smaller than two bytes"));
790 }
791
792 if (!Config.BuildIdLinkDir.empty() && Config.BuildIdLinkInput)
793 if (Error E =
794 linkToBuildIdDir(Config, Config.InputFilename,
795 Config.BuildIdLinkInput.getValue(), BuildIdBytes))
796 return E;
797
798 if (Error E = handleArgs(Config, *Obj, Reader, OutputElfType))
799 return createFileError(Config.InputFilename, std::move(E));
800
801 if (Error E = writeOutput(Config, *Obj, Out, OutputElfType))
802 return createFileError(Config.InputFilename, std::move(E));
803 if (!Config.BuildIdLinkDir.empty() && Config.BuildIdLinkOutput)
804 if (Error E =
805 linkToBuildIdDir(Config, Config.OutputFilename,
806 Config.BuildIdLinkOutput.getValue(), BuildIdBytes))
807 return createFileError(Config.OutputFilename, std::move(E));
808
809 return Error::success();
810}
