25struct MachHeader {
26 uint32_t Magic;
27 uint32_t CPUType;
28 uint32_t CPUSubType;
29 uint32_t FileType;
30 uint32_t NCmds;
31 uint32_t SizeOfCmds;
32 uint32_t Flags;
33 uint32_t Reserved = 0;
34};

36struct Section {
37 std::string Sectname;
38 std::string Segname;
39 uint64_t Addr;
40 uint64_t Size;
41 uint32_t Offset;
42 uint32_t Align;
43 uint32_t RelOff;
44 uint32_t NReloc;
45 uint32_t Flags;
46 uint32_t Reserved1;
47 uint32_t Reserved2;
48 uint32_t Reserved3;
49
50 StringRef Content;
51 std::vector<MachO::any_relocation_info> Relocations;
52
53 MachO::SectionType getType() const {
54 return static_cast<MachO::SectionType>(Flags & MachO::SECTION_TYPE);
55 }
56
57 bool isVirtualSection() const {
58 return (getType() == MachO::S_ZEROFILL ||
59 getType() == MachO::S_GB_ZEROFILL ||
60 getType() == MachO::S_THREAD_LOCAL_ZEROFILL);
61 }
62};

64struct LoadCommand {
65 // The type MachO::macho_load_command is defined in llvm/BinaryFormat/MachO.h
66 // and it is a union of all the structs corresponding to various load
67 // commands.
68 MachO::macho_load_command MachOLoadCommand;
69
70 // The raw content of the payload of the load command (located right after the
71 // corresponding struct). In some cases it is either empty or can be
72 // copied-over without digging into its structure.
73 ArrayRef<uint8_t> Payload;
74
75 // Some load commands can contain (inside the payload) an array of sections,
76 // though the contents of the sections are stored separately. The struct
77 // Section describes only sections' metadata and where to find the
78 // corresponding content inside the binary.
79 std::vector<Section> Sections;
80};

82struct NListEntry {
83 uint32_t n_strx;
84 uint8_t n_type;
85 uint8_t n_sect;
86 uint16_t n_desc;
87 uint64_t n_value;
88};

90/// The location of the symbol table inside the binary is described by LC_SYMTAB
91/// load command.
92struct SymbolTable {
93 std::vector<NListEntry> NameList;
94};
95
96/// The location of the string table inside the binary is described by LC_SYMTAB
97/// load command.
98struct StringTable {
99 std::vector<std::string> Strings;
100};
101
102/// The location of the rebase info inside the binary is described by
103/// LC_DYLD_INFO load command. Dyld rebases an image whenever dyld loads it at
104/// an address different from its preferred address. The rebase information is
105/// a stream of byte sized opcodes whose symbolic names start with
106/// REBASE_OPCODE_. Conceptually the rebase information is a table of tuples:
107/// <seg-index, seg-offset, type>
108/// The opcodes are a compressed way to encode the table by only
109/// encoding when a column changes. In addition simple patterns
110/// like "every n'th offset for m times" can be encoded in a few
111/// bytes.
112struct RebaseInfo {
113 // At the moment we do not parse this info (and it is simply copied over),
114 // but the proper support will be added later.
115 ArrayRef<uint8_t> Opcodes;
116};
117
118/// The location of the bind info inside the binary is described by
119/// LC_DYLD_INFO load command. Dyld binds an image during the loading process,
120/// if the image requires any pointers to be initialized to symbols in other
121/// images. The bind information is a stream of byte sized opcodes whose
122/// symbolic names start with BIND_OPCODE_. Conceptually the bind information is
123/// a table of tuples: <seg-index, seg-offset, type, symbol-library-ordinal,
124/// symbol-name, addend> The opcodes are a compressed way to encode the table by
125/// only encoding when a column changes. In addition simple patterns like for
126/// runs of pointers initialized to the same value can be encoded in a few
127/// bytes.
128struct BindInfo {
129 // At the moment we do not parse this info (and it is simply copied over),
130 // but the proper support will be added later.
131 ArrayRef<uint8_t> Opcodes;
132};
133
134/// The location of the weak bind info inside the binary is described by
135/// LC_DYLD_INFO load command. Some C++ programs require dyld to unique symbols
136/// so that all images in the process use the same copy of some code/data. This
137/// step is done after binding. The content of the weak_bind info is an opcode
138/// stream like the bind_info. But it is sorted alphabetically by symbol name.
139/// This enable dyld to walk all images with weak binding information in order
140/// and look for collisions. If there are no collisions, dyld does no updating.
141/// That means that some fixups are also encoded in the bind_info. For
142/// instance, all calls to "operator new" are first bound to libstdc++.dylib
143/// using the information in bind_info. Then if some image overrides operator
144/// new that is detected when the weak_bind information is processed and the
145/// call to operator new is then rebound.
146struct WeakBindInfo {
147 // At the moment we do not parse this info (and it is simply copied over),
148 // but the proper support will be added later.
149 ArrayRef<uint8_t> Opcodes;
150};
151
152/// The location of the lazy bind info inside the binary is described by
153/// LC_DYLD_INFO load command. Some uses of external symbols do not need to be
154/// bound immediately. Instead they can be lazily bound on first use. The
155/// lazy_bind contains a stream of BIND opcodes to bind all lazy symbols. Normal
156/// use is that dyld ignores the lazy_bind section when loading an image.
157/// Instead the static linker arranged for the lazy pointer to initially point
158/// to a helper function which pushes the offset into the lazy_bind area for the
159/// symbol needing to be bound, then jumps to dyld which simply adds the offset
160/// to lazy_bind_off to get the information on what to bind.
161struct LazyBindInfo {
162 ArrayRef<uint8_t> Opcodes;
163};
164
165/// The location of the export info inside the binary is described by
166/// LC_DYLD_INFO load command. The symbols exported by a dylib are encoded in a
167/// trie. This is a compact representation that factors out common prefixes. It
168/// also reduces LINKEDIT pages in RAM because it encodes all information (name,
169/// address, flags) in one small, contiguous range. The export area is a stream
170/// of nodes. The first node sequentially is the start node for the trie. Nodes
171/// for a symbol start with a uleb128 that is the length of the exported symbol
172/// information for the string so far. If there is no exported symbol, the node
173/// starts with a zero byte. If there is exported info, it follows the length.
174/// First is a uleb128 containing flags. Normally, it is followed by
175/// a uleb128 encoded offset which is location of the content named
176/// by the symbol from the mach_header for the image. If the flags
177/// is EXPORT_SYMBOL_FLAGS_REEXPORT, then following the flags is
178/// a uleb128 encoded library ordinal, then a zero terminated
179/// UTF8 string. If the string is zero length, then the symbol
180/// is re-export from the specified dylib with the same name.
181/// If the flags is EXPORT_SYMBOL_FLAGS_STUB_AND_RESOLVER, then following
182/// the flags is two uleb128s: the stub offset and the resolver offset.
183/// The stub is used by non-lazy pointers. The resolver is used
184/// by lazy pointers and must be called to get the actual address to use.
185/// After the optional exported symbol information is a byte of
186/// how many edges (0-255) that this node has leaving it,
187/// followed by each edge.
188/// Each edge is a zero terminated UTF8 of the addition chars
189/// in the symbol, followed by a uleb128 offset for the node that
190/// edge points to.
191struct ExportInfo {
192 ArrayRef<uint8_t> Trie;
193};

195struct Object {
196 MachHeader Header;
197 std::vector<LoadCommand> LoadCommands;
198
199 SymbolTable SymTable;
200 StringTable StrTable;
201
202 RebaseInfo Rebases;
203 BindInfo Binds;
204 WeakBindInfo WeakBinds;
205 LazyBindInfo LazyBinds;
206 ExportInfo Exports;
207
208 /// The index of LC_SYMTAB load command if present.
209 Optional<size_t> SymTabCommandIndex;
210 /// The index of LC_DYLD_INFO or LC_DYLD_INFO_ONLY load command if present.
211 Optional<size_t> DyLdInfoCommandIndex;
212};
