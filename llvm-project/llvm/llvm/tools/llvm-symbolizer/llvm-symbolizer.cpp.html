8//
9// This utility works much like "addr2line". It is able of transforming
10// tuples (module name, module offset) to code locations (function name,
11// file, line number, column number). It is targeted for compiler-rt tools
12// (especially AddressSanitizer and ThreadSanitizer) that can use it
13// to symbolize stack traces in their error reports.
14//
15//===----------------------------------------------------------------------===//

157template<typename T>
158static bool error(Expected<T> &ResOrErr) {
159 if (ResOrErr)
160 return false;
161 logAllUnhandledErrors(ResOrErr.takeError(), errs(),
162 "LLVMSymbolizer: error reading file: ");
163 return true;
164}
165
166static bool parseCommand(StringRef InputString, bool &IsData,
167 std::string &ModuleName, uint64_t &ModuleOffset) {
168 const char kDelimiters[] = " \n\r";
169 ModuleName = "";
170 if (InputString.consume_front("CODE ")) {
171 IsData = false;
172 } else if (InputString.consume_front("DATA ")) {
173 IsData = true;
174 } else {
175 // If no cmd, assume it's CODE.
176 IsData = false;
177 }
178 const char *pos = InputString.data();
179 // Skip delimiters and parse input filename (if needed).
180 if (ClBinaryName.empty()) {
181 pos += strspn(pos, kDelimiters);
182 if (*pos == '"' || *pos == '\'') {
183 char quote = *pos;
184 pos++;
185 const char *end = strchr(pos, quote);
186 if (!end)
187 return false;
188 ModuleName = std::string(pos, end - pos);
189 pos = end + 1;
190 } else {
191 int name_length = strcspn(pos, kDelimiters);
192 ModuleName = std::string(pos, name_length);
193 pos += name_length;
194 }
195 } else {
196 ModuleName = ClBinaryName;
197 }
198 // Skip delimiters and parse module offset.
199 pos += strspn(pos, kDelimiters);
200 int offset_length = strcspn(pos, kDelimiters);
201 return !StringRef(pos, offset_length).getAsInteger(0, ModuleOffset);
202}

204static void symbolizeInput(StringRef InputString, LLVMSymbolizer &Symbolizer,
205 DIPrinter &Printer) {
206 bool IsData = false;
207 std::string ModuleName;
208 uint64_t Offset = 0;
209 if (!parseCommand(StringRef(InputString), IsData, ModuleName, Offset)) {
210 outs() << InputString;
211 return;
212 }
213
214 if (ClPrintAddress) {
215 outs() << "0x";
216 outs().write_hex(Offset);
217 StringRef Delimiter = ClPrettyPrint ? ": " : "\n";
218 outs() << Delimiter;
219 }
220 Offset -= ClAdjustVMA;
221 if (IsData) {
222 auto ResOrErr = Symbolizer.symbolizeData(
223 ModuleName, {Offset, object::SectionedAddress::UndefSection});
224 Printer << (error(ResOrErr) ? DIGlobal() : ResOrErr.get());
225 } else if (ClPrintInlining) {
226 auto ResOrErr = Symbolizer.symbolizeInlinedCode(
227 ModuleName, {Offset, object::SectionedAddress::UndefSection});
228 Printer << (error(ResOrErr) ? DIInliningInfo() : ResOrErr.get());
229 } else if (ClOutputStyle == DIPrinter::OutputStyle::GNU) {
230 // With ClPrintFunctions == FunctionNameKind::LinkageName (default)
231 // and ClUseSymbolTable == true (also default), Symbolizer.symbolizeCode()
232 // may override the name of an inlined function with the name of the topmost
233 // caller function in the inlining chain. This contradicts the existing
234 // behavior of addr2line. Symbolizer.symbolizeInlinedCode() overrides only
235 // the topmost function, which suits our needs better.
236 auto ResOrErr = Symbolizer.symbolizeInlinedCode(
237 ModuleName, {Offset, object::SectionedAddress::UndefSection});
238 Printer << (error(ResOrErr) ? DILineInfo() : ResOrErr.get().getFrame(0));
239 } else {
240 auto ResOrErr = Symbolizer.symbolizeCode(
241 ModuleName, {Offset, object::SectionedAddress::UndefSection});
242 Printer << (error(ResOrErr) ? DILineInfo() : ResOrErr.get());
243 }
244 if (ClOutputStyle == DIPrinter::OutputStyle::LLVM)
245 outs() << "\n";
246}
247
248int main(int argc, char **argv) {
249 InitLLVM X(argc, argv);
250
251 bool IsAddr2Line = sys::path::stem(argv[0]).contains("addr2line");
252
253 if (IsAddr2Line) {
254 ClDemangle.setInitialValue(false);
255 ClPrintFunctions.setInitialValue(FunctionNameKind::None);
256 ClPrintInlining.setInitialValue(false);
257 ClOutputStyle.setInitialValue(DIPrinter::OutputStyle::GNU);
258 }
259
260 llvm::sys::InitializeCOMRAII COM(llvm::sys::COMThreadingMode::MultiThreaded);
261 cl::ParseCommandLineOptions(argc, argv, IsAddr2Line ? "llvm-addr2line\n"
262 : "llvm-symbolizer\n");
263
264 // If both --demangle and --no-demangle are specified then pick the last one.
265 if (ClNoDemangle.getPosition() > ClDemangle.getPosition())
266 ClDemangle = !ClNoDemangle;
267
268 LLVMSymbolizer::Options Opts;
269 Opts.PrintFunctions = ClPrintFunctions;
270 Opts.UseSymbolTable = ClUseSymbolTable;
271 Opts.Demangle = ClDemangle;
272 Opts.RelativeAddresses = ClUseRelativeAddress;
273 Opts.DefaultArch = ClDefaultArch;
274 Opts.FallbackDebugPath = ClFallbackDebugPath;
275 Opts.DWPName = ClDwpName;
276
277 for (const auto &hint : ClDsymHint) {
278 if (sys::path::extension(hint) == ".dSYM") {
279 Opts.DsymHints.push_back(hint);
280 } else {
281 errs() << "Warning: invalid dSYM hint: \"" << hint <<
282 "\" (must have the '.dSYM' extension).\n";
283 }
284 }
285 LLVMSymbolizer Symbolizer(Opts);
286
287 DIPrinter Printer(outs(), ClPrintFunctions != FunctionNameKind::None,
288 ClPrettyPrint, ClPrintSourceContextLines, ClVerbose,
289 ClBasenames, ClOutputStyle);
290
291 if (ClInputAddresses.empty()) {
292 const int kMaxInputStringLength = 1024;
293 char InputString[kMaxInputStringLength];
294
295 while (fgets(InputString, sizeof(InputString), stdin)) {
296 symbolizeInput(InputString, Symbolizer, Printer);
297 outs().flush();
298 }
299 } else {
300 for (StringRef Address : ClInputAddresses)
301 symbolizeInput(Address, Symbolizer, Printer);
302 }
303
304 return 0;
305}
