9#include "obj2yaml.h"
10#include "llvm/Object/XCOFFObjectFile.h"
11#include "llvm/ObjectYAML/XCOFFYAML.h"
12#include "llvm/Support/YAMLTraits.h"
13
14using namespace llvm;
15using namespace llvm::object;
16namespace {
17
18class XCOFFDumper {
19 const object::XCOFFObjectFile &Obj;
20 XCOFFYAML::Object YAMLObj;
21 void dumpHeader();
22 std::error_code dumpSymbols();
23
24public:
25 XCOFFDumper(const object::XCOFFObjectFile &obj) : Obj(obj) {}
26 std::error_code dump();
27 XCOFFYAML::Object &getYAMLObj() { return YAMLObj; }
28};
29} // namespace
30
31std::error_code XCOFFDumper::dump() {
32 std::error_code EC;
33 dumpHeader();
34 EC = dumpSymbols();
35 return EC;
36}
37
38void XCOFFDumper::dumpHeader() {
39 const XCOFFFileHeader *FileHdrPtr = Obj.getFileHeader();
40
41 YAMLObj.Header.Magic = FileHdrPtr->Magic;
42 YAMLObj.Header.NumberOfSections = FileHdrPtr->NumberOfSections;
43 YAMLObj.Header.TimeStamp = FileHdrPtr->TimeStamp;
44 YAMLObj.Header.SymbolTableOffset = FileHdrPtr->SymbolTableOffset;
45 YAMLObj.Header.NumberOfSymTableEntries = FileHdrPtr->NumberOfSymTableEntries;
46 YAMLObj.Header.AuxHeaderSize = FileHdrPtr->AuxHeaderSize;
47 YAMLObj.Header.Flags = FileHdrPtr->Flags;
48}
49
50std::error_code XCOFFDumper::dumpSymbols() {
51 std::vector<XCOFFYAML::Symbol> &Symbols = YAMLObj.Symbols;
52
53 for (const SymbolRef &S : Obj.symbols()) {
54 DataRefImpl SymbolDRI = S.getRawDataRefImpl();
55 const XCOFFSymbolEntry *SymbolEntPtr = Obj.toSymbolEntry(SymbolDRI);
56 XCOFFYAML::Symbol Sym;
57
58 Expected<StringRef> SymNameRefOrErr = Obj.getSymbolName(SymbolDRI);
59 if (!SymNameRefOrErr) {
60 return errorToErrorCode(SymNameRefOrErr.takeError());
61 }
62 Sym.SymbolName = SymNameRefOrErr.get();
63
64 Sym.Value = SymbolEntPtr->Value;
65
66 Expected<StringRef> SectionNameRefOrErr =
67 Obj.getSymbolSectionName(SymbolEntPtr);
68 if (!SectionNameRefOrErr)
69 return errorToErrorCode(SectionNameRefOrErr.takeError());
70
71 Sym.SectionName = SectionNameRefOrErr.get();
72
73 Sym.Type = SymbolEntPtr->SymbolType;
74 Sym.StorageClass = SymbolEntPtr->StorageClass;
75 Sym.NumberOfAuxEntries = SymbolEntPtr->NumberOfAuxEntries;
76 Symbols.push_back(Sym);
77 }
78
79 return std::error_code();
80}
81
82std::error_code xcoff2yaml(raw_ostream &Out,
83 const object::XCOFFObjectFile &Obj) {
84 XCOFFDumper Dumper(Obj);
85
86 if (std::error_code EC = Dumper.dump())
87 return EC;
88
89 yaml::Output Yout(Out);
90 Yout << Dumper.getYAMLObj();
91
92 return std::error_code();
93}
