9// This program takes in a list of bitcode files, links them and performs
10// link-time optimization according to the provided symbol resolutions using the
11// resolution-based LTO interface, and outputs one or more object files.
12//
13// This program is intended to eventually replace llvm-lto which uses the legacy
14// LTO interface.
15//
16//===----------------------------------------------------------------------===//

134static void check(Error E, std::string Msg) {
135 if (!E)
136 return;
137 handleAllErrors(std::move(E), [&](ErrorInfoBase &EIB) {
138 errs() << "llvm-lto2: " << Msg << ": " << EIB.message().c_str() << '\n';
139 });
140 exit(1);
141}
142
143template <typename T> static T check(Expected<T> E, std::string Msg) {
144 if (E)
145 return std::move(*E);
146 check(E.takeError(), Msg);
147 return T();
148}
149
150static void check(std::error_code EC, std::string Msg) {
151 check(errorCodeToError(EC), Msg);
152}
153
154template <typename T> static T check(ErrorOr<T> E, std::string Msg) {
155 if (E)
156 return std::move(*E);
157 check(E.getError(), Msg);
158 return T();
159}

166static int run(int argc, char **argv) {
167 cl::ParseCommandLineOptions(argc, argv, "Resolution-based LTO test harness");
168
169 // FIXME: Workaround PR30396 which means that a symbol can appear
170 // more than once if it is defined in module-level assembly and
171 // has a GV declaration. We allow (file, symbol) pairs to have multiple
172 // resolutions and apply them in the order observed.
173 std::map<std::pair<std::string, std::string>, std::list<SymbolResolution>>
174 CommandLineResolutions;
175 for (std::string R : SymbolResolutions) {
176 StringRef Rest = R;
177 StringRef FileName, SymbolName;
178 std::tie(FileName, Rest) = Rest.split(',');
179 if (Rest.empty()) {
180 llvm::errs() << "invalid resolution: " << R << '\n';
181 return 1;
182 }
183 std::tie(SymbolName, Rest) = Rest.split(',');
184 SymbolResolution Res;
185 for (char C : Rest) {
186 if (C == 'p')
187 Res.Prevailing = true;
188 else if (C == 'l')
189 Res.FinalDefinitionInLinkageUnit = true;
190 else if (C == 'x')
191 Res.VisibleToRegularObj = true;
192 else if (C == 'r')
193 Res.LinkerRedefined = true;
194 else {
195 llvm::errs() << "invalid character " << C << " in resolution: " << R
196 << '\n';
197 return 1;
198 }
199 }
200 CommandLineResolutions[{FileName, SymbolName}].push_back(Res);
201 }
202
203 std::vector<std::unique_ptr<MemoryBuffer>> MBs;
204
205 Config Conf;
206 Conf.DiagHandler = [](const DiagnosticInfo &DI) {
207 DiagnosticPrinterRawOStream DP(errs());
208 DI.print(DP);
209 errs() << '\n';
210 if (DI.getSeverity() == DS_Error)
211 exit(1);
212 };
213
214 Conf.CPU = MCPU;
215 Conf.Options = InitTargetOptionsFromCodeGenFlags();
216 Conf.MAttrs = MAttrs;
217 if (auto RM = getRelocModel())
218 Conf.RelocModel = *RM;
219 Conf.CodeModel = getCodeModel();
220
221 Conf.DebugPassManager = DebugPassManager;
222
223 if (SaveTemps)
224 check(Conf.addSaveTemps(OutputFilename + "."),
225 "Config::addSaveTemps failed");
226
227 // Optimization remarks.
228 Conf.RemarksFilename = OptRemarksOutput;
229 Conf.RemarksPasses = OptRemarksPasses;
230 Conf.RemarksWithHotness = OptRemarksWithHotness;
231
232 Conf.SampleProfile = SamplePGOFile;
233 Conf.CSIRProfile = CSPGOFile;
234 Conf.RunCSIRInstr = RunCSIRInstr;
235
236 // Run a custom pipeline, if asked for.
237 Conf.OptPipeline = OptPipeline;
238 Conf.AAPipeline = AAPipeline;
239
240 Conf.OptLevel = OptLevel - '0';
241 Conf.UseNewPM = UseNewPM;
242 switch (CGOptLevel) {
243 case '0':
244 Conf.CGOptLevel = CodeGenOpt::None;
245 break;
246 case '1':
247 Conf.CGOptLevel = CodeGenOpt::Less;
248 break;
249 case '2':
250 Conf.CGOptLevel = CodeGenOpt::Default;
251 break;
252 case '3':
253 Conf.CGOptLevel = CodeGenOpt::Aggressive;
254 break;
255 default:
256 llvm::errs() << "invalid cg optimization level: " << CGOptLevel << '\n';
257 return 1;
258 }
259
260 if (FileType.getNumOccurrences())
261 Conf.CGFileType = FileType;
262
263 Conf.OverrideTriple = OverrideTriple;
264 Conf.DefaultTriple = DefaultTriple;
265 Conf.StatsFile = StatsFile;
266
267 ThinBackend Backend;
268 if (ThinLTODistributedIndexes)
269 Backend = createWriteIndexesThinBackend(/* OldPrefix */ "",
270 /* NewPrefix */ "",
271 /* ShouldEmitImportsFiles */ true,
272 /* LinkedObjectsFile */ nullptr,
273 /* OnWrite */ {});
274 else
275 Backend = createInProcessThinBackend(Threads);
276 LTO Lto(std::move(Conf), std::move(Backend));
277
278 bool HasErrors = false;
279 for (std::string F : InputFilenames) {
280 std::unique_ptr<MemoryBuffer> MB = check(MemoryBuffer::getFile(F), F);
281 std::unique_ptr<InputFile> Input =
282 check(InputFile::create(MB->getMemBufferRef()), F);
283
284 std::vector<SymbolResolution> Res;
285 for (const InputFile::Symbol &Sym : Input->symbols()) {
286 auto I = CommandLineResolutions.find({F, Sym.getName()});
287 if (I == CommandLineResolutions.end()) {
288 llvm::errs() << argv[0] << ": missing symbol resolution for " << F
289 << ',' << Sym.getName() << '\n';
290 HasErrors = true;
291 } else {
292 Res.push_back(I->second.front());
293 I->second.pop_front();
294 if (I->second.empty())
295 CommandLineResolutions.erase(I);
296 }
297 }
298
299 if (HasErrors)
300 continue;
301
302 MBs.push_back(std::move(MB));
303 check(Lto.add(std::move(Input), Res), F);
304 }
305
306 if (!CommandLineResolutions.empty()) {
307 HasErrors = true;
308 for (auto UnusedRes : CommandLineResolutions)
309 llvm::errs() << argv[0] << ": unused symbol resolution for "
310 << UnusedRes.first.first << ',' << UnusedRes.first.second
311 << '\n';
312 }
313 if (HasErrors)
314 return 1;
315
316 auto AddStream =
317 [&](size_t Task) -> std::unique_ptr<lto::NativeObjectStream> {
318 std::string Path = OutputFilename + "." + utostr(Task);
319
320 std::error_code EC;
321 auto S = llvm::make_unique<raw_fd_ostream>(Path, EC, sys::fs::F_None);
322 check(EC, Path);
323 return llvm::make_unique<lto::NativeObjectStream>(std::move(S));
324 };
325
326 auto AddBuffer = [&](size_t Task, std::unique_ptr<MemoryBuffer> MB) {
327 *AddStream(Task)->OS << MB->getBuffer();
328 };
329
330 NativeObjectCache Cache;
331 if (!CacheDir.empty())
332 Cache = check(localCache(CacheDir, AddBuffer), "failed to create cache");
333
334 check(Lto.run(AddStream, Cache), "LTO::run failed");
335 return 0;
336}

338static int dumpSymtab(int argc, char **argv) {
339 for (StringRef F : make_range(argv + 1, argv + argc)) {
340 std::unique_ptr<MemoryBuffer> MB = check(MemoryBuffer::getFile(F), F);
341 BitcodeFileContents BFC = check(getBitcodeFileContents(*MB), F);
342
343 if (BFC.Symtab.size() >= sizeof(irsymtab::storage::Header)) {
344 auto *Hdr = reinterpret_cast<const irsymtab::storage::Header *>(
345 BFC.Symtab.data());
346 outs() << "version: " << Hdr->Version << '\n';
347 if (Hdr->Version == irsymtab::storage::Header::kCurrentVersion)
348 outs() << "producer: " << Hdr->Producer.get(BFC.StrtabForSymtab)
349 << '\n';
350 }
351
352 std::unique_ptr<InputFile> Input =
353 check(InputFile::create(MB->getMemBufferRef()), F);
354
355 outs() << "target triple: " << Input->getTargetTriple() << '\n';
356 Triple TT(Input->getTargetTriple());
357
358 outs() << "source filename: " << Input->getSourceFileName() << '\n';
359
360 if (TT.isOSBinFormatCOFF())
361 outs() << "linker opts: " << Input->getCOFFLinkerOpts() << '\n';
362
363 if (TT.isOSBinFormatELF()) {
364 outs() << "dependent libraries:";
365 for (auto L : Input->getDependentLibraries())
366 outs() << " \"" << L << "\"";
367 outs() << '\n';
368 }
369
370 std::vector<StringRef> ComdatTable = Input->getComdatTable();
371 for (const InputFile::Symbol &Sym : Input->symbols()) {
372 switch (Sym.getVisibility()) {
373 case GlobalValue::HiddenVisibility:
374 outs() << 'H';
375 break;
376 case GlobalValue::ProtectedVisibility:
377 outs() << 'P';
378 break;
379 case GlobalValue::DefaultVisibility:
380 outs() << 'D';
381 break;
382 }
383
384 auto PrintBool = [&](char C, bool B) { outs() << (B ? C : '-'); };
385 PrintBool('U', Sym.isUndefined());
386 PrintBool('C', Sym.isCommon());
387 PrintBool('W', Sym.isWeak());
388 PrintBool('I', Sym.isIndirect());
389 PrintBool('O', Sym.canBeOmittedFromSymbolTable());
390 PrintBool('T', Sym.isTLS());
391 PrintBool('X', Sym.isExecutable());
392 outs() << ' ' << Sym.getName() << '\n';
393
394 if (Sym.isCommon())
395 outs() << " size " << Sym.getCommonSize() << " align "
396 << Sym.getCommonAlignment() << '\n';
397
398 int Comdat = Sym.getComdatIndex();
399 if (Comdat != -1)
400 outs() << " comdat " << ComdatTable[Comdat] << '\n';
401
402 if (TT.isOSBinFormatCOFF() && Sym.isWeak() && Sym.isIndirect())
403 outs() << " fallback " << Sym.getCOFFWeakExternalFallback() << '\n';
404
405 if (!Sym.getSectionName().empty())
406 outs() << " section " << Sym.getSectionName() << "\n";
407 }
408
409 outs() << '\n';
410 }
411
412 return 0;
413}
414
415int main(int argc, char **argv) {
416 InitLLVM X(argc, argv);
417 InitializeAllTargets();
418 InitializeAllTargetMCs();
419 InitializeAllAsmPrinters();
420 InitializeAllAsmParsers();
421
422 // FIXME: This should use llvm::cl subcommands, but it isn't currently
423 // possible to pass an argument not associated with a subcommand to a
424 // subcommand (e.g. -use-new-pm).
425 if (argc < 2)
426 return usage();
427
428 StringRef Subcommand = argv[1];
429 // Ensure that argv[0] is correct after adjusting argv/argc.
430 argv[1] = argv[0];
431 if (Subcommand == "dump-symtab")
432 return dumpSymtab(argc - 1, argv + 1);
433 if (Subcommand == "run")
434 return run(argc - 1, argv + 1);
435 return usage();
436}
