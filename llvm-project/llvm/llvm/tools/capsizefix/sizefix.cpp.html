1#include "llvm/Support/Endian.h"
2#include "llvm/Object/ObjectFile.h"
3#include "llvm/Object/ELFObjectFile.h"
4#include <unordered_map>
5#include <stdio.h>
6
7using namespace llvm;
8using namespace llvm::object;
9
10static const std::string SizePrefix = ".size.";
11
12int main(int argc, char *argv[]) {
13 const char* InFile = nullptr;
14 bool Verbose = false;
15 if (argc == 2) {
16 InFile = argv[1];
17 } else if (argc == 3 && StringRef("--verbose") == argv[1]) {
18 Verbose = true;
19 InFile = argv[2];
20 } else {
21 fprintf(stderr, "Usage: %s [--verbose] {statically linked object}\n", argv[0]);
22 return EXIT_FAILURE;
23 }
24 auto OF = ObjectFile::createObjectFile(InFile);
25 // OF is an Expected<T> which requires mandatory checking for errors. This is
26 // done by invoking the bool conversion operator.
27 if (!OF) {
28 fprintf(stderr, "Cannot open %s: %s\n", InFile, toString(OF.takeError()).c_str());
29 return EXIT_FAILURE;
30 }
31
32 std::unordered_map<uint64_t, std::pair<uint64_t,bool>> SymbolSizes;
33 std::vector<std::tuple<uint64_t, uint64_t, bool>> Sections;
34 // ObjectFile doesn't allow in-place modification, so we open the file again
35 // and write it out.
36 FILE *F = fopen(InFile, "r+");
37 if (!F) {
38 fprintf(stderr, "Cannot open %s\n", InFile);
39 return EXIT_FAILURE;
40 }
41 StringMap<uint64_t> SizeForName;
42 std::vector<SymbolRef> SizeSymbols;
43 SectionRef SizesSection;
44
45 StringRef Data;
46 for (const SectionRef &Sec : OF->getBinary()->sections()) {
47 StringRef Name;
48 if (!Sec.getName(Name)) {
49 if (Name == "__cap_relocs") {
50 Expected<StringRef> Contents = Sec.getContents();
51 if (!Contents)
52 report_fatal_error(Contents.takeError(), false);
53 Data = *Contents;
54 continue;
55 } else if (Name == ".global_sizes") {
56 SizesSection = Sec;
57 continue;
58 }
59 Sections.push_back(std::make_tuple(Sec.getAddress(), Sec.getSize(),
60 Sec.isText()));
61 }
62 }
63
64 for (const SymbolRef &sym : OF->getBinary()->symbols()) {
65 uint64_t Size = ELFSymbolRef(sym).getSize();
66 if (Size == 0)
67 continue;
68 Expected<StringRef> Name = sym.getName();
69 if (Name) {
70 if (Name->startswith(SizePrefix) && SizesSection.containsSymbol(sym)) {
71 SizeSymbols.push_back(sym);
72 continue;
73 } else
74 SizeForName[*Name] = Size;
75 }
76 Expected<uint64_t> Start = sym.getAddress();
77 if (!Start)
78 continue;
79 Expected<SymbolRef::Type> SymbolType = sym.getType();
80 if (SymbolType) {
81 SymbolRef::Type type = SymbolType.get();
82 SymbolSizes.insert({Start.get(), {Size, (type == SymbolRef::ST_Function)}});
83 }
84 }
85 const size_t entry_size = 40;
86 MemoryBufferRef MB = OF->getBinary()->getMemoryBufferRef();
87 for (int i = 0, e = Data.size() / entry_size; i < e; i++) {
88 const char *entry = Data.data() + (entry_size * i);
89 uint64_t base = support::endian::read<uint64_t, support::big, 1>(entry + 8);
90 if (!((entry >= MB.getBufferStart()) && (entry <= MB.getBufferEnd()))) {
91 fprintf(stderr,
92 "Unexpected location for capability relocations section!\n");
93 return EXIT_FAILURE;
94 }
95 auto SizeAndType = SymbolSizes.find(base);
96 uint64_t Size = 0;
97 bool isFunction = false;
98 if (base == 0)
99 Size = -1;
100 else if (SizeAndType == SymbolSizes.end()) {
101 for (auto &Sec : Sections) {
102 if (std::get<0>(Sec) < base &&
103 (std::get<0>(Sec) + std::get<1>(Sec)) > base) {
104 Size = std::get<1>(Sec);
105 isFunction = std::get<2>(Sec);
106#ifndef NDEBUG
107 if (!isFunction && Verbose)
108 fprintf(stderr, "Unable to find size of symbol at 0%llx for pointer at 0x%llx\n"
109 "Using section size (%llu bytes) instead\n",
110 (unsigned long long)base,
111 static_cast<unsigned long long>(
112 support::endian::read<uint64_t, support::big, 1>(entry)),
113 static_cast<unsigned long long>(Size));
114#endif
115 break; // First match wins
116 }
117 }
118 if (Size == 0) {
119 if (Verbose)
120 fprintf(stderr, "Unable to find size of symbol at 0%llx for pointer at 0x%llx\n",
121 (unsigned long long)base,
122 static_cast<unsigned long long>(
123 support::endian::read<uint64_t, support::big, 1>(entry)));
124 continue;
125 }
126 } else {
127 Size = SizeAndType->second.first;
128 isFunction = SizeAndType->second.second;
129 }
130 uint64_t Perms = 0;
131 if (isFunction)
132 Perms |= (1ULL<<63);
133 uint64_t BigSize =
134 support::endian::byte_swap<uint64_t, support::big>(Size);
135 uint64_t BigPerms =
136 support::endian::byte_swap<uint64_t, support::big>(Perms);
137 // This is an ugly hack. object ought to allow modification
138 fseek(F, entry - MB.getBufferStart() + 24, SEEK_SET);
139 fwrite(&BigSize, sizeof(BigSize), 1, F);
140 fwrite(&BigPerms, sizeof(BigPerms), 1, F);
141 }
142 if (SizesSection != SectionRef()) {
143 Expected<StringRef> Contents = SizesSection.getContents();
144 if (!Contents)
145 report_fatal_error(Contents.takeError(), false);
146 Data = *Contents;
147 uint64_t SectionOffset = Data.data() - MB.getBufferStart();
148 for (SymbolRef &sym : SizeSymbols) {
149 Expected<uint64_t> Start = sym.getAddress();
150 if (!Start)
151 continue;
152 uint64_t offset = *Start - SizesSection.getAddress();
153 Expected<StringRef> SymName = sym.getName();
154 if (!SymName)
155 continue;
156 std::string Name = SymName->str();
157 Name = Name.substr(SizePrefix.size(), Name.length() - SizePrefix.size());
158 auto SizeIt = SizeForName.find(Name);
159 uint64_t Size;
160 if (SizeIt == SizeForName.end()) {
161 fprintf(stderr, "Unable to find size for symbol %s\n", Name.c_str());
162 Size = 0;
163 } else
164 Size = SizeIt->second;
165#ifndef NDEBUG
166 if (Verbose) {
167 fprintf(stderr, "Writing size %llu for symbol %s\n", (unsigned long long)Size, Name.c_str());
168 }
169#endif
170 fseek(F, SectionOffset + offset, SEEK_SET);
171 uint64_t BigSize =
172 support::endian::byte_swap<uint64_t, support::big>(Size);
173 fwrite(&BigSize, sizeof(BigSize), 1, F);
174 }
175 }
176 fclose(F);
177}
