744static void yamlToPdb(StringRef Path) {
745 BumpPtrAllocator Allocator;
746 ErrorOr<std::unique_ptr<MemoryBuffer>> ErrorOrBuffer =
747 MemoryBuffer::getFileOrSTDIN(Path, /*FileSize=*/-1,
748 /*RequiresNullTerminator=*/false);
749
750 if (ErrorOrBuffer.getError()) {
751 ExitOnErr(createFileError(Path, errorCodeToError(ErrorOrBuffer.getError())));
752 }
753
754 std::unique_ptr<MemoryBuffer> &Buffer = ErrorOrBuffer.get();
755
756 llvm::yaml::Input In(Buffer->getBuffer());
757 pdb::yaml::PdbObject YamlObj(Allocator);
758 In >> YamlObj;
759
760 PDBFileBuilder Builder(Allocator);
761
762 uint32_t BlockSize = 4096;
763 if (YamlObj.Headers.hasValue())
764 BlockSize = YamlObj.Headers->SuperBlock.BlockSize;
765 ExitOnErr(Builder.initialize(BlockSize));
766 // Add each of the reserved streams. We ignore stream metadata in the
767 // yaml, because we will reconstruct our own view of the streams. For
768 // example, the YAML may say that there were 20 streams in the original
769 // PDB, but maybe we only dump a subset of those 20 streams, so we will
770 // have fewer, and the ones we do have may end up with different indices
771 // than the ones in the original PDB. So we just start with a clean slate.
772 for (uint32_t I = 0; I < kSpecialStreamCount; ++I)
773 ExitOnErr(Builder.getMsfBuilder().addStream(0));
774
775 StringsAndChecksums Strings;
776 Strings.setStrings(std::make_shared<DebugStringTableSubsection>());
777
778 if (YamlObj.StringTable.hasValue()) {
779 for (auto S : *YamlObj.StringTable)
780 Strings.strings()->insert(S);
781 }
782
783 pdb::yaml::PdbInfoStream DefaultInfoStream;
784 pdb::yaml::PdbDbiStream DefaultDbiStream;
785 pdb::yaml::PdbTpiStream DefaultTpiStream;
786 pdb::yaml::PdbTpiStream DefaultIpiStream;
787
788 const auto &Info = YamlObj.PdbStream.getValueOr(DefaultInfoStream);
789
790 auto &InfoBuilder = Builder.getInfoBuilder();
791 InfoBuilder.setAge(Info.Age);
792 InfoBuilder.setGuid(Info.Guid);
793 InfoBuilder.setSignature(Info.Signature);
794 InfoBuilder.setVersion(Info.Version);
795 for (auto F : Info.Features)
796 InfoBuilder.addFeature(F);
797
798 const auto &Dbi = YamlObj.DbiStream.getValueOr(DefaultDbiStream);
799 auto &DbiBuilder = Builder.getDbiBuilder();
800 DbiBuilder.setAge(Dbi.Age);
801 DbiBuilder.setBuildNumber(Dbi.BuildNumber);
802 DbiBuilder.setFlags(Dbi.Flags);
803 DbiBuilder.setMachineType(Dbi.MachineType);
804 DbiBuilder.setPdbDllRbld(Dbi.PdbDllRbld);
805 DbiBuilder.setPdbDllVersion(Dbi.PdbDllVersion);
806 DbiBuilder.setVersionHeader(Dbi.VerHeader);
807 for (const auto &MI : Dbi.ModInfos) {
808 auto &ModiBuilder = ExitOnErr(DbiBuilder.addModuleInfo(MI.Mod));
809 ModiBuilder.setObjFileName(MI.Obj);
810
811 for (auto S : MI.SourceFiles)
812 ExitOnErr(DbiBuilder.addModuleSourceFile(ModiBuilder, S));
813 if (MI.Modi.hasValue()) {
814 const auto &ModiStream = *MI.Modi;
815 for (auto Symbol : ModiStream.Symbols) {
816 ModiBuilder.addSymbol(
817 Symbol.toCodeViewSymbol(Allocator, CodeViewContainer::Pdb));
818 }
819 }
820
821 // Each module has its own checksum subsection, so scan for it every time.
822 Strings.setChecksums(nullptr);
823 CodeViewYAML::initializeStringsAndChecksums(MI.Subsections, Strings);
824
825 auto CodeViewSubsections = ExitOnErr(CodeViewYAML::toCodeViewSubsectionList(
826 Allocator, MI.Subsections, Strings));
827 for (auto &SS : CodeViewSubsections) {
828 ModiBuilder.addDebugSubsection(SS);
829 }
830 }
831
832 auto &TpiBuilder = Builder.getTpiBuilder();
833 const auto &Tpi = YamlObj.TpiStream.getValueOr(DefaultTpiStream);
834 TpiBuilder.setVersionHeader(Tpi.Version);
835 AppendingTypeTableBuilder TS(Allocator);
836 for (const auto &R : Tpi.Records) {
837 CVType Type = R.toCodeViewRecord(TS);
838 TpiBuilder.addTypeRecord(Type.RecordData, None);
839 }
840
841 const auto &Ipi = YamlObj.IpiStream.getValueOr(DefaultIpiStream);
842 auto &IpiBuilder = Builder.getIpiBuilder();
843 IpiBuilder.setVersionHeader(Ipi.Version);
844 for (const auto &R : Ipi.Records) {
845 CVType Type = R.toCodeViewRecord(TS);
846 IpiBuilder.addTypeRecord(Type.RecordData, None);
847 }
848
849 Builder.getStringTableBuilder().setStrings(*Strings.strings());
850
851 codeview::GUID IgnoredOutGuid;
852 ExitOnErr(Builder.commit(opts::yaml2pdb::YamlPdbOutputFile, &IgnoredOutGuid));
853}

935static void dumpInjectedSources(LinePrinter &Printer, IPDBSession &Session) {
936 auto Sources = Session.getInjectedSources();
937 if (0 == Sources->getChildCount()) {
938 Printer.printLine("There are no injected sources.");
939 return;
940 }
941
942 while (auto IS = Sources->getNext()) {
943 Printer.NewLine();
944 std::string File = stringOr(IS->getFileName(), "<null>");
945 uint64_t Size = IS->getCodeByteSize();
946 std::string Obj = stringOr(IS->getObjectFileName(), "<null>");
947 std::string VFName = stringOr(IS->getVirtualFileName(), "<null>");
948 uint32_t CRC = IS->getCrc32();
949
950 std::string CompressionStr;
951 llvm::raw_string_ostream Stream(CompressionStr);
952 Stream << IS->getCompression();
953 WithColor(Printer, PDB_ColorItem::Path).get() << File;
954 Printer << " (";
955 WithColor(Printer, PDB_ColorItem::LiteralValue).get() << Size;
956 Printer << " bytes): ";
957 WithColor(Printer, PDB_ColorItem::Keyword).get() << "obj";
958 Printer << "=";
959 WithColor(Printer, PDB_ColorItem::Path).get() << Obj;
960 Printer << ", ";
961 WithColor(Printer, PDB_ColorItem::Keyword).get() << "vname";
962 Printer << "=";
963 WithColor(Printer, PDB_ColorItem::Path).get() << VFName;
964 Printer << ", ";
965 WithColor(Printer, PDB_ColorItem::Keyword).get() << "crc";
966 Printer << "=";
967 WithColor(Printer, PDB_ColorItem::LiteralValue).get() << CRC;
968 Printer << ", ";
969 WithColor(Printer, PDB_ColorItem::Keyword).get() << "compression";
970 Printer << "=";
971 WithColor(Printer, PDB_ColorItem::LiteralValue).get() << Stream.str();
972
973 if (!opts::pretty::ShowInjectedSourceContent)
974 continue;
975
976 // Set the indent level to 0 when printing file content.
977 int Indent = Printer.getIndentLevel();
978 Printer.Unindent(Indent);
979
980 Printer.printLine(IS->getCode());
981
982 // Re-indent back to the original level.
983 Printer.Indent(Indent);
984 }

987template <typename OuterT, typename ChildT>
988void diaDumpChildren(PDBSymbol &Outer, PdbSymbolIdField Ids,
989 PdbSymbolIdField Recurse) {
990 OuterT *ConcreteOuter = dyn_cast<OuterT>(&Outer);
991 if (!ConcreteOuter)
992 return;
993
994 auto Children = ConcreteOuter->template findAllChildren<ChildT>();
995 while (auto Child = Children->getNext()) {
996 outs() << " {";
997 Child->defaultDump(outs(), 4, Ids, Recurse);
998 outs() << "\n }\n";
999 }

1002static void dumpDia(StringRef Path) {
1003 std::unique_ptr<IPDBSession> Session;
1004
1005 const auto ReaderType =
1006 opts::diadump::Native ? PDB_ReaderType::Native : PDB_ReaderType::DIA;
1007 ExitOnErr(loadDataForPDB(ReaderType, Path, Session));
1008
1009 auto GlobalScope = Session->getGlobalScope();
1010
1011 std::vector<PDB_SymType> SymTypes;
1012
1013 if (opts::diadump::Compilands)
1014 SymTypes.push_back(PDB_SymType::Compiland);
1015 if (opts::diadump::Enums)
1016 SymTypes.push_back(PDB_SymType::Enum);
1017 if (opts::diadump::Pointers)
1018 SymTypes.push_back(PDB_SymType::PointerType);
1019 if (opts::diadump::UDTs)
1020 SymTypes.push_back(PDB_SymType::UDT);
1021 if (opts::diadump::Funcsigs)
1022 SymTypes.push_back(PDB_SymType::FunctionSig);
1023 if (opts::diadump::Arrays)
1024 SymTypes.push_back(PDB_SymType::ArrayType);
1025 if (opts::diadump::VTShapes)
1026 SymTypes.push_back(PDB_SymType::VTableShape);
1027 if (opts::diadump::Typedefs)
1028 SymTypes.push_back(PDB_SymType::Typedef);
1029 PdbSymbolIdField Ids = opts::diadump::NoSymIndexIds ? PdbSymbolIdField::None
1030 : PdbSymbolIdField::All;
1031
1032 PdbSymbolIdField Recurse = PdbSymbolIdField::None;
1033 if (opts::diadump::Recurse)
1034 Recurse = PdbSymbolIdField::All;
1035 if (!opts::diadump::ShowClassHierarchy)
1036 Ids &= ~(PdbSymbolIdField::ClassParent | PdbSymbolIdField::LexicalParent);
1037
1038 for (PDB_SymType ST : SymTypes) {
1039 auto Children = GlobalScope->findAllChildren(ST);
1040 while (auto Child = Children->getNext()) {
1041 outs() << "{";
1042 Child->defaultDump(outs(), 2, Ids, Recurse);
1043
1044 diaDumpChildren<PDBSymbolTypeEnum, PDBSymbolData>(*Child, Ids, Recurse);
1045 outs() << "\n}\n";
1046 }
1047 }

1050static void dumpPretty(StringRef Path) {
1051 std::unique_ptr<IPDBSession> Session;
1052
1053 const auto ReaderType =
1054 opts::pretty::Native ? PDB_ReaderType::Native : PDB_ReaderType::DIA;
1055 ExitOnErr(loadDataForPDB(ReaderType, Path, Session));
1056
1057 if (opts::pretty::LoadAddress)
1058 Session->setLoadAddress(opts::pretty::LoadAddress);
1059
1060 auto &Stream = outs();
1061 const bool UseColor = opts::pretty::ColorOutput == cl::BOU_UNSET
1062 ? Stream.has_colors()
1063 : opts::pretty::ColorOutput == cl::BOU_TRUE;
1064 LinePrinter Printer(2, UseColor, Stream);
1065
1066 auto GlobalScope(Session->getGlobalScope());
1067 if (!GlobalScope)
1068 return;
1069 std::string FileName(GlobalScope->getSymbolsFileName());
1070
1071 WithColor(Printer, PDB_ColorItem::None).get() << "Summary for ";
1072 WithColor(Printer, PDB_ColorItem::Path).get() << FileName;
1073 Printer.Indent();
1074 uint64_t FileSize = 0;
1075
1076 Printer.NewLine();
1077 WithColor(Printer, PDB_ColorItem::Identifier).get() << "Size";
1078 if (!sys::fs::file_size(FileName, FileSize)) {
1079 Printer << ": " << FileSize << " bytes";
1080 } else {
1081 Printer << ": (Unable to obtain file size)";
1082 }
1083
1084 Printer.NewLine();
1085 WithColor(Printer, PDB_ColorItem::Identifier).get() << "Guid";
1086 Printer << ": " << GlobalScope->getGuid();
1087
1088 Printer.NewLine();
1089 WithColor(Printer, PDB_ColorItem::Identifier).get() << "Age";
1090 Printer << ": " << GlobalScope->getAge();
1091
1092 Printer.NewLine();
1093 WithColor(Printer, PDB_ColorItem::Identifier).get() << "Attributes";
1094 Printer << ": ";
1095 if (GlobalScope->hasCTypes())
1096 outs() << "HasCTypes ";
1097 if (GlobalScope->hasPrivateSymbols())
1098 outs() << "HasPrivateSymbols ";
1099 Printer.Unindent();
1100
1101 if (!opts::pretty::WithName.empty()) {
1102 Printer.NewLine();
1103 WithColor(Printer, PDB_ColorItem::SectionHeader).get()
1104 << "---SYMBOLS & TYPES BY NAME---";
1105
1106 for (StringRef Name : opts::pretty::WithName) {
1107 auto Symbols = GlobalScope->findChildren(
1108 PDB_SymType::None, Name, PDB_NameSearchFlags::NS_CaseSensitive);
1109 if (!Symbols || Symbols->getChildCount() == 0) {
1110 Printer.formatLine("[not found] - {0}", Name);
1111 continue;
1112 }
1113 Printer.formatLine("[{0} occurrences] - {1}", Symbols->getChildCount(),
1114 Name);
1115
1116 AutoIndent Indent(Printer);
1117 Printer.NewLine();
1118
1119 while (auto Symbol = Symbols->getNext()) {
1120 switch (Symbol->getSymTag()) {
1121 case PDB_SymType::Typedef: {
1122 TypedefDumper TD(Printer);
1123 std::unique_ptr<PDBSymbolTypeTypedef> T =
1124 llvm::unique_dyn_cast<PDBSymbolTypeTypedef>(std::move(Symbol));
1125 TD.start(*T);
1126 break;
1127 }
1128 case PDB_SymType::Enum: {
1129 EnumDumper ED(Printer);
1130 std::unique_ptr<PDBSymbolTypeEnum> E =
1131 llvm::unique_dyn_cast<PDBSymbolTypeEnum>(std::move(Symbol));
1132 ED.start(*E);
1133 break;
1134 }
1135 case PDB_SymType::UDT: {
1136 ClassDefinitionDumper CD(Printer);
1137 std::unique_ptr<PDBSymbolTypeUDT> C =
1138 llvm::unique_dyn_cast<PDBSymbolTypeUDT>(std::move(Symbol));
1139 CD.start(*C);
1140 break;
1141 }
1142 case PDB_SymType::BaseClass:
1143 case PDB_SymType::Friend: {
1144 TypeDumper TD(Printer);
1145 Symbol->dump(TD);
1146 break;
1147 }
1148 case PDB_SymType::Function: {
1149 FunctionDumper FD(Printer);
1150 std::unique_ptr<PDBSymbolFunc> F =
1151 llvm::unique_dyn_cast<PDBSymbolFunc>(std::move(Symbol));
1152 FD.start(*F, FunctionDumper::PointerType::None);
1153 break;
1154 }
1155 case PDB_SymType::Data: {
1156 VariableDumper VD(Printer);
1157 std::unique_ptr<PDBSymbolData> D =
1158 llvm::unique_dyn_cast<PDBSymbolData>(std::move(Symbol));
1159 VD.start(*D);
1160 break;
1161 }
1162 case PDB_SymType::PublicSymbol: {
1163 ExternalSymbolDumper ED(Printer);
1164 std::unique_ptr<PDBSymbolPublicSymbol> PS =
1165 llvm::unique_dyn_cast<PDBSymbolPublicSymbol>(std::move(Symbol));
1166 ED.dump(*PS);
1167 break;
1168 }
1169 default:
1170 llvm_unreachable("Unexpected symbol tag!");
1171 }
1172 }
1173 }
1174 llvm::outs().flush();
1175 }
1176
1177 if (opts::pretty::Compilands) {
1178 Printer.NewLine();
1179 WithColor(Printer, PDB_ColorItem::SectionHeader).get()
1180 << "---COMPILANDS---";
1181 auto Compilands = GlobalScope->findAllChildren<PDBSymbolCompiland>();
1182
1183 if (Compilands) {
1184 Printer.Indent();
1185 CompilandDumper Dumper(Printer);
1186 CompilandDumpFlags options = CompilandDumper::Flags::None;
1187 if (opts::pretty::Lines)
1188 options = options | CompilandDumper::Flags::Lines;
1189 while (auto Compiland = Compilands->getNext())
1190 Dumper.start(*Compiland, options);
1191 Printer.Unindent();
1192 }
1193 }
1194
1195 if (opts::pretty::Classes || opts::pretty::Enums || opts::pretty::Typedefs ||
1196 opts::pretty::Funcsigs || opts::pretty::Pointers ||
1197 opts::pretty::Arrays || opts::pretty::VTShapes) {
1198 Printer.NewLine();
1199 WithColor(Printer, PDB_ColorItem::SectionHeader).get() << "---TYPES---";
1200 Printer.Indent();
1201 TypeDumper Dumper(Printer);
1202 Dumper.start(*GlobalScope);
1203 Printer.Unindent();
1204 }
1205
1206 if (opts::pretty::Symbols) {
1207 Printer.NewLine();
1208 WithColor(Printer, PDB_ColorItem::SectionHeader).get() << "---SYMBOLS---";
1209 if (auto Compilands = GlobalScope->findAllChildren<PDBSymbolCompiland>()) {
1210 Printer.Indent();
1211 CompilandDumper Dumper(Printer);
1212 while (auto Compiland = Compilands->getNext())
1213 Dumper.start(*Compiland, true);
1214 Printer.Unindent();
1215 }
1216 }
1217
1218 if (opts::pretty::Globals) {
1219 Printer.NewLine();
1220 WithColor(Printer, PDB_ColorItem::SectionHeader).get() << "---GLOBALS---";
1221 Printer.Indent();
1222 if (shouldDumpSymLevel(opts::pretty::SymLevel::Functions)) {
1223 if (auto Functions = GlobalScope->findAllChildren<PDBSymbolFunc>()) {
1224 FunctionDumper Dumper(Printer);
1225 if (opts::pretty::SymbolOrder == opts::pretty::SymbolSortMode::None) {
1226 while (auto Function = Functions->getNext()) {
1227 Printer.NewLine();
1228 Dumper.start(*Function, FunctionDumper::PointerType::None);
1229 }
1230 } else {
1231 std::vector<std::unique_ptr<PDBSymbolFunc>> Funcs;
1232 while (auto Func = Functions->getNext())
1233 Funcs.push_back(std::move(Func));
1234 llvm::sort(Funcs, opts::pretty::compareFunctionSymbols);
1235 for (const auto &Func : Funcs) {
1236 Printer.NewLine();
1237 Dumper.start(*Func, FunctionDumper::PointerType::None);
1238 }
1239 }
1240 }
1241 }
1242 if (shouldDumpSymLevel(opts::pretty::SymLevel::Data)) {
1243 if (auto Vars = GlobalScope->findAllChildren<PDBSymbolData>()) {
1244 VariableDumper Dumper(Printer);
1245 if (opts::pretty::SymbolOrder == opts::pretty::SymbolSortMode::None) {
1246 while (auto Var = Vars->getNext())
1247 Dumper.start(*Var);
1248 } else {
1249 std::vector<std::unique_ptr<PDBSymbolData>> Datas;
1250 while (auto Var = Vars->getNext())
1251 Datas.push_back(std::move(Var));
1252 llvm::sort(Datas, opts::pretty::compareDataSymbols);
1253 for (const auto &Var : Datas)
1254 Dumper.start(*Var);
1255 }
1256 }
1257 }
1258 if (shouldDumpSymLevel(opts::pretty::SymLevel::Thunks)) {
1259 if (auto Thunks = GlobalScope->findAllChildren<PDBSymbolThunk>()) {
1260 CompilandDumper Dumper(Printer);
1261 while (auto Thunk = Thunks->getNext())
1262 Dumper.dump(*Thunk);
1263 }
1264 }
1265 Printer.Unindent();
1266 }
1267 if (opts::pretty::Externals) {
1268 Printer.NewLine();
1269 WithColor(Printer, PDB_ColorItem::SectionHeader).get() << "---EXTERNALS---";
1270 Printer.Indent();
1271 ExternalSymbolDumper Dumper(Printer);
1272 Dumper.start(*GlobalScope);
1273 }
1274 if (opts::pretty::Lines) {
1275 Printer.NewLine();
1276 }
1277 if (opts::pretty::InjectedSources) {
1278 Printer.NewLine();
1279 WithColor(Printer, PDB_ColorItem::SectionHeader).get()
1280 << "---INJECTED SOURCES---";
1281 AutoIndent Indent1(Printer);
1282
1283 if (ReaderType == PDB_ReaderType::Native)
1284 Printer.printLine(
1285 "Injected sources are not supported with the native reader.");
1286 else
1287 dumpInjectedSources(Printer, *Session);
1288 }
1289
1290 Printer.NewLine();
1291 outs().flush();
1292}

1294static void mergePdbs() {
1295 BumpPtrAllocator Allocator;
1296 MergingTypeTableBuilder MergedTpi(Allocator);
1297 MergingTypeTableBuilder MergedIpi(Allocator);
1298
1299 // Create a Tpi and Ipi type table with all types from all input files.
1300 for (const auto &Path : opts::merge::InputFilenames) {
1301 std::unique_ptr<IPDBSession> Session;
1302 auto &File = loadPDB(Path, Session);
1303 SmallVector<TypeIndex, 128> TypeMap;
1304 SmallVector<TypeIndex, 128> IdMap;
1305 if (File.hasPDBTpiStream()) {
1306 auto &Tpi = ExitOnErr(File.getPDBTpiStream());
1307 ExitOnErr(
1308 codeview::mergeTypeRecords(MergedTpi, TypeMap, Tpi.typeArray()));
1309 }
1310 if (File.hasPDBIpiStream()) {
1311 auto &Ipi = ExitOnErr(File.getPDBIpiStream());
1312 ExitOnErr(codeview::mergeIdRecords(MergedIpi, TypeMap, IdMap,
1313 Ipi.typeArray()));
1314 }
1315 }
1316
1317 // Then write the PDB.
1318 PDBFileBuilder Builder(Allocator);
1319 ExitOnErr(Builder.initialize(4096));
1320 // Add each of the reserved streams. We might not put any data in them,
1321 // but at least they have to be present.
1322 for (uint32_t I = 0; I < kSpecialStreamCount; ++I)
1323 ExitOnErr(Builder.getMsfBuilder().addStream(0));
1324
1325 auto &DestTpi = Builder.getTpiBuilder();
1326 auto &DestIpi = Builder.getIpiBuilder();
1327 MergedTpi.ForEachRecord([&DestTpi](TypeIndex TI, const CVType &Type) {
1328 DestTpi.addTypeRecord(Type.RecordData, None);
1329 });
1330 MergedIpi.ForEachRecord([&DestIpi](TypeIndex TI, const CVType &Type) {
1331 DestIpi.addTypeRecord(Type.RecordData, None);
1332 });
1333 Builder.getInfoBuilder().addFeature(PdbRaw_FeatureSig::VC140);
1334
1335 SmallString<64> OutFile(opts::merge::PdbOutputFile);
1336 if (OutFile.empty()) {
1337 OutFile = opts::merge::InputFilenames[0];
1338 llvm::sys::path::replace_extension(OutFile, "merged.pdb");
1339 }
1340
1341 codeview::GUID IgnoredOutGuid;
1342 ExitOnErr(Builder.commit(OutFile, &IgnoredOutGuid));
1343}

1357static void exportStream() {
1358 std::unique_ptr<IPDBSession> Session;
1359 PDBFile &File = loadPDB(opts::exportstream::InputFilename.front(), Session);
1360
1361 std::unique_ptr<MappedBlockStream> SourceStream;
1362 uint32_t Index = 0;
1363 bool Success = false;
1364 std::string OutFileName = opts::exportstream::OutputFile;
1365
1366 if (!opts::exportstream::ForceName) {
1367 // First try to parse it as an integer, if it fails fall back to treating it
1368 // as a named stream.
1369 if (to_integer(opts::exportstream::Stream, Index)) {
1370 if (Index >= File.getNumStreams()) {
1371 errs() << "Error: " << Index << " is not a valid stream index.\n";
1372 exit(1);
1373 }
1374 Success = true;
1375 outs() << "Dumping contents of stream index " << Index << " to file "
1376 << OutFileName << ".\n";
1377 }
1378 }
1379
1380 if (!Success) {
1381 InfoStream &IS = cantFail(File.getPDBInfoStream());
1382 Index = ExitOnErr(IS.getNamedStreamIndex(opts::exportstream::Stream));
1383 outs() << "Dumping contents of stream '" << opts::exportstream::Stream
1384 << "' (index " << Index << ") to file " << OutFileName << ".\n";
1385 }
1386
1387 SourceStream = MappedBlockStream::createIndexedStream(
1388 File.getMsfLayout(), File.getMsfBuffer(), Index, File.getAllocator());
1389 auto OutFile = ExitOnErr(
1390 FileOutputBuffer::create(OutFileName, SourceStream->getLength()));
1391 FileBufferByteStream DestStream(std::move(OutFile), llvm::support::little);
1392 BinaryStreamWriter Writer(DestStream);
1393 ExitOnErr(Writer.writeStreamRef(*SourceStream));
1394 ExitOnErr(DestStream.commit());
1395}

1397static bool parseRange(StringRef Str,
1398 Optional<opts::bytes::NumberRange> &Parsed) {
1399 if (Str.empty())
1400 return true;
1401
1402 llvm::Regex R("^([^-]+)(-([^-]+))?$");
1403 llvm::SmallVector<llvm::StringRef, 2> Matches;
1404 if (!R.match(Str, &Matches))
1405 return false;
1406
1407 Parsed.emplace();
1408 if (!to_integer(Matches[1], Parsed->Min))
1409 return false;
1410
1411 if (!Matches[3].empty()) {
1412 Parsed->Max.emplace();
1413 if (!to_integer(Matches[3], *Parsed->Max))
1414 return false;
1415 }
1416 return true;
1417}
