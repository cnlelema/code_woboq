332static std::string formatGaps(uint32_t IndentLevel,
333 ArrayRef<LocalVariableAddrGap> Gaps) {
334 std::vector<std::string> GapStrs;
335 for (const auto &G : Gaps) {
336 GapStrs.push_back(formatv("({0},{1})", G.GapStartOffset, G.Range).str());
337 }
338 return typesetItemList(GapStrs, 7, IndentLevel, ", ");

345Error MinimalSymbolDumper::visitSymbolBegin(codeview::CVSymbol &Record,
346 uint32_t Offset) {
347 // formatLine puts the newline at the beginning, so we use formatLine here
348 // to start a new line, and then individual visit methods use format to
349 // append to the existing line.
350 P.formatLine("{0} | {1} [size = {2}]",
351 fmt_align(Offset, AlignStyle::Right, 6),
352 formatSymbolKind(Record.kind()), Record.length());
353 P.Indent();
354 return Error::success();
355}
356
357Error MinimalSymbolDumper::visitSymbolEnd(CVSymbol &Record) {
358 if (RecordBytes) {
359 AutoIndent Indent(P, 7);
360 P.formatBinary("bytes", Record.content(), 0);
361 }
362 P.Unindent();
363 return Error::success();
364}

366std::string MinimalSymbolDumper::typeOrIdIndex(codeview::TypeIndex TI,
367 bool IsType) const {
368 if (TI.isSimple() || TI.isDecoratedItemId())
369 return formatv("{0}", TI).str();
370 auto &Container = IsType ? Types : Ids;
371 StringRef Name = Container.getTypeName(TI);
372 if (Name.size() > 32) {
373 Name = Name.take_front(32);
374 return formatv("{0} ({1}...)", TI, Name);
375 } else
376 return formatv("{0} ({1})", TI, Name);
377}
378
379std::string MinimalSymbolDumper::idIndex(codeview::TypeIndex TI) const {
380 return typeOrIdIndex(TI, false);
381}

383std::string MinimalSymbolDumper::typeIndex(TypeIndex TI) const {
384 return typeOrIdIndex(TI, true);
385}

387Error MinimalSymbolDumper::visitKnownRecord(CVSymbol &CVR, BlockSym &Block) {
388 P.format(" `{0}`", Block.Name);
389 AutoIndent Indent(P, 7);
390 P.formatLine("parent = {0}, end = {1}", Block.Parent, Block.End);
391 P.formatLine("code size = {0}, addr = {1}", Block.CodeSize,

393 return Error::success();

478Error MinimalSymbolDumper::visitKnownRecord(CVSymbol &CVR, FileStaticSym &FS) {
479 P.format(" `{0}`", FS.Name);
480 AutoIndent Indent(P, 7);
481 if (SymGroup) {
482 Expected<StringRef> FileName =
483 SymGroup->getNameFromStringTable(FS.ModFilenameOffset);
484 if (FileName) {
485 P.formatLine("type = {0}, file name = {1} ({2}), flags = {3}",
486 typeIndex(FS.Index), FS.ModFilenameOffset, *FileName,
487 formatLocalSymFlags(P.getIndentLevel() + 9, FS.Flags));
488 }
489 return Error::success();
490 }
491
492 P.formatLine("type = {0}, file name offset = {1}, flags = {2}",
493 typeIndex(FS.Index), FS.ModFilenameOffset,
494 formatLocalSymFlags(P.getIndentLevel() + 9, FS.Flags));
495 return Error::success();
496}

506Error MinimalSymbolDumper::visitKnownRecord(CVSymbol &CVR,
507 Compile2Sym &Compile2) {
508 AutoIndent Indent(P, 7);
509 SourceLanguage Lang = static_cast<SourceLanguage>(
510 Compile2.Flags & CompileSym2Flags::SourceLanguageMask);
511 CompilationCPU = Compile2.Machine;
512 P.formatLine("machine = {0}, ver = {1}, language = {2}",
513 formatMachineType(Compile2.Machine), Compile2.Version,
514 formatSourceLanguage(Lang));
515 P.formatLine("frontend = {0}.{1}.{2}, backend = {3}.{4}.{5}",
516 Compile2.VersionFrontendMajor, Compile2.VersionFrontendMinor,
517 Compile2.VersionFrontendBuild, Compile2.VersionBackendMajor,
518 Compile2.VersionBackendMinor, Compile2.VersionBackendBuild);
519 P.formatLine("flags = {0}",
520 formatCompileSym2Flags(P.getIndentLevel() + 9, Compile2.Flags));
521 P.formatLine(
522 "extra strings = {0}",
523 typesetStringList(P.getIndentLevel() + 9 + 2, Compile2.ExtraStrings));
524 return Error::success();
525}

670Error MinimalSymbolDumper::visitKnownRecord(CVSymbol &CVR, InlineSiteSym &IS) {
671 AutoIndent Indent(P, 7);
672 P.formatLine("inlinee = {0}, parent = {1}, end = {2}", idIndex(IS.Inlinee),
673 IS.Parent, IS.End);
674
675 // Break down the annotation byte code and calculate code and line offsets.
676 // FIXME: It would be helpful if we could look up the initial file and inlinee
677 // lines offset using the inlinee index above.
678 uint32_t CodeOffset = 0;
679 int32_t LineOffset = 0;
680 for (auto &Annot : IS.annotations()) {
681 P.formatLine(" {0}", fmt_align(toHex(Annot.Bytes), AlignStyle::Left, 9));
682
683 auto formatCodeOffset = [&](uint32_t Delta) {
684 CodeOffset += Delta;
685 P.format(" code 0x{0} (+0x{1})", utohexstr(CodeOffset), utohexstr(Delta));
686 };
687 auto formatCodeLength = [&](uint32_t Length) {
688 // Notably, changing the code length does not affect the code offset.
689 P.format(" code end 0x{0} (+0x{1})", utohexstr(CodeOffset + Length),
690 utohexstr(Length));
691 };
692 auto formatLineOffset = [&](int32_t Delta) {
693 LineOffset += Delta;
694 char Sign = Delta > 0 ? '+' : '-';
695 P.format(" line {0} ({1}{2})", LineOffset, Sign, std::abs(Delta));
696 };
697
698 // Use the opcode to interpret the integer values.
699 switch (Annot.OpCode) {
700 case BinaryAnnotationsOpCode::Invalid:
701 break;
702 case BinaryAnnotationsOpCode::CodeOffset:
703 case BinaryAnnotationsOpCode::ChangeCodeOffset:
704 formatCodeOffset(Annot.U1);
705 break;
706 case BinaryAnnotationsOpCode::ChangeLineOffset:
707 formatLineOffset(Annot.S1);
708 break;
709 case BinaryAnnotationsOpCode::ChangeCodeLength:
710 formatCodeLength(Annot.U1);
711 // Apparently this annotation updates the code offset. It's hard to make
712 // MSVC produce this opcode, but clang uses it, and debuggers seem to use
713 // this interpretation.
714 CodeOffset += Annot.U1;
715 break;
716 case BinaryAnnotationsOpCode::ChangeCodeOffsetAndLineOffset:
717 formatCodeOffset(Annot.U1);
718 formatLineOffset(Annot.S1);
719 break;
720 case BinaryAnnotationsOpCode::ChangeCodeLengthAndCodeOffset:
721 formatCodeOffset(Annot.U2);
722 formatCodeLength(Annot.U1);
723 break;
724
725 case BinaryAnnotationsOpCode::ChangeFile: {
726 uint32_t FileOffset = Annot.U1;
727 StringRef Filename = "<unknown>";
728 if (SymGroup) {
729 if (Expected<StringRef> MaybeFile =
730 SymGroup->getNameFromStringTable(FileOffset))
731 Filename = *MaybeFile;
732 else
733 return MaybeFile.takeError();
734 }
735 P.format(" setfile {0} 0x{1}", utohexstr(FileOffset));
736 break;
737 }
738
739 // The rest of these are hard to convince MSVC to emit, so they are not as
740 // well understood.
741 case BinaryAnnotationsOpCode::ChangeCodeOffsetBase:
742 formatCodeOffset(Annot.U1);
743 break;
744 case BinaryAnnotationsOpCode::ChangeLineEndDelta:
745 case BinaryAnnotationsOpCode::ChangeRangeKind:
746 case BinaryAnnotationsOpCode::ChangeColumnStart:
747 case BinaryAnnotationsOpCode::ChangeColumnEnd:
748 P.format(" {0} {1}", Annot.Name, Annot.U1);
749 break;
750 case BinaryAnnotationsOpCode::ChangeColumnEndDelta:
751 P.format(" {0} {1}", Annot.Name, Annot.S1);
752 break;
753 }
754 }
755 return Error::success();
756}
