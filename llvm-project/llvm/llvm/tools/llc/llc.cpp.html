161struct RunPassOption {
162 void operator=(const std::string &Val) const {
163 if (Val.empty())
164 return;
165 SmallVector<StringRef, 8> PassNames;
166 StringRef(Val).split(PassNames, ',', -1, false);
167 for (auto PassName : PassNames)
168 RunPassNames->push_back(PassName);
169 }
170};
171}

182static std::unique_ptr<ToolOutputFile> GetOutputStream(const char *TargetName,
183 Triple::OSType OS,
184 const char *ProgName) {
185 // If we don't yet have an output filename, make one.
186 if (OutputFilename.empty()) {
187 if (InputFilename == "-")
188 OutputFilename = "-";
189 else {
190 // If InputFilename ends in .bc or .ll, remove it.
191 StringRef IFN = InputFilename;
192 if (IFN.endswith(".bc") || IFN.endswith(".ll"))
193 OutputFilename = IFN.drop_back(3);
194 else if (IFN.endswith(".mir"))
195 OutputFilename = IFN.drop_back(4);
196 else
197 OutputFilename = IFN;
198
199 switch (FileType) {
200 case TargetMachine::CGFT_AssemblyFile:
201 if (TargetName[0] == 'c') {
202 if (TargetName[1] == 0)
203 OutputFilename += ".cbe.c";
204 else if (TargetName[1] == 'p' && TargetName[2] == 'p')
205 OutputFilename += ".cpp";
206 else
207 OutputFilename += ".s";
208 } else
209 OutputFilename += ".s";
210 break;
211 case TargetMachine::CGFT_ObjectFile:
212 if (OS == Triple::Win32)
213 OutputFilename += ".obj";
214 else
215 OutputFilename += ".o";
216 break;
217 case TargetMachine::CGFT_Null:
218 OutputFilename += ".null";
219 break;
220 }
221 }
222 }
223
224 // Decide if we need "binary" output.
225 bool Binary = false;
226 switch (FileType) {
227 case TargetMachine::CGFT_AssemblyFile:
228 break;
229 case TargetMachine::CGFT_ObjectFile:
230 case TargetMachine::CGFT_Null:
231 Binary = true;
232 break;
233 }
234
235 // Open the file.
236 std::error_code EC;
237 sys::fs::OpenFlags OpenFlags = sys::fs::F_None;
238 if (!Binary)
239 OpenFlags |= sys::fs::F_Text;
240 auto FDOut = llvm::make_unique<ToolOutputFile>(OutputFilename, EC, OpenFlags);
241 if (EC) {
242 WithColor::error() << EC.message() << '\n';
243 return nullptr;
244 }
245
246 return FDOut;
247}

249struct LLCDiagnosticHandler : public DiagnosticHandler {
250 bool *HasError;
251 LLCDiagnosticHandler(bool *HasErrorPtr) : HasError(HasErrorPtr) {}
252 bool handleDiagnostics(const DiagnosticInfo &DI) override {
253 if (DI.getSeverity() == DS_Error)
254 *HasError = true;
255
256 if (auto *Remark = dyn_cast<DiagnosticInfoOptimizationBase>(&DI))
257 if (!Remark->isEnabled())
258 return true;
259
260 DiagnosticPrinterRawOStream DP(errs());
261 errs() << LLVMContext::getDiagnosticMessagePrefix(DI.getSeverity()) << ": ";
262 DI.print(DP);
263 errs() << "\n";
264 return true;
265 }
266};

268static void InlineAsmDiagHandler(const SMDiagnostic &SMD, void *Context,
269 unsigned LocCookie) {
270 bool *HasError = static_cast<bool *>(Context);
271 if (SMD.getKind() == SourceMgr::DK_Error)
272 *HasError = true;
273
274 SMD.print(nullptr, errs());
275
276 // For testing purposes, we print the LocCookie here.
277 if (LocCookie)
278 WithColor::note() << "!srcloc = " << LocCookie << "\n";
279}

281// main - Entry point for the llc compiler.
282//
283int main(int argc, char **argv) {
284 InitLLVM X(argc, argv);
285
286 // Enable debug stream buffering.
287 EnableDebugBuffering = true;
288
289 LLVMContext Context;
290
291 // Initialize targets first, so that --version shows registered targets.
292 InitializeAllTargets();
293 InitializeAllTargetMCs();
294 InitializeAllAsmPrinters();
295 InitializeAllAsmParsers();
296
297 // Initialize codegen and IR passes used by llc so that the -print-after,
298 // -print-before, and -stop-after options work.
299 PassRegistry *Registry = PassRegistry::getPassRegistry();
300 initializeCore(*Registry);
301 initializeCodeGen(*Registry);
302 initializeLoopStrengthReducePass(*Registry);
303 initializeLowerIntrinsicsPass(*Registry);
304 initializeEntryExitInstrumenterPass(*Registry);
305 initializePostInlineEntryExitInstrumenterPass(*Registry);
306 initializeUnreachableBlockElimLegacyPassPass(*Registry);
307 initializeConstantHoistingLegacyPassPass(*Registry);
308 initializeScalarOpts(*Registry);
309 initializeVectorization(*Registry);
310 initializeScalarizeMaskedMemIntrinPass(*Registry);
311 initializeExpandReductionsPass(*Registry);
312 initializeHardwareLoopsPass(*Registry);
313
314 // Initialize debugging passes.
315 initializeScavengerTestPass(*Registry);
316
317 // Register the target printer for --version.
318 cl::AddExtraVersionPrinter(TargetRegistry::printRegisteredTargetsForVersion);
319
320 cl::ParseCommandLineOptions(argc, argv, "llvm system compiler\n");
321
322 Context.setDiscardValueNames(DiscardValueNames);
323
324 // Set a diagnostic handler that doesn't exit on the first error
325 bool HasError = false;
326 Context.setDiagnosticHandler(
327 llvm::make_unique<LLCDiagnosticHandler>(&HasError));
328 Context.setInlineAsmDiagnosticHandler(InlineAsmDiagHandler, &HasError);
329
330 if (PassRemarksWithHotness)
331 Context.setDiagnosticsHotnessRequested(true);
332
333 if (PassRemarksHotnessThreshold)
334 Context.setDiagnosticsHotnessThreshold(PassRemarksHotnessThreshold);
335
336 std::unique_ptr<ToolOutputFile> YamlFile;
337 if (RemarksFilename != "") {
338 std::error_code EC;
339 YamlFile =
340 llvm::make_unique<ToolOutputFile>(RemarksFilename, EC, sys::fs::F_None);
341 if (EC) {
342 WithColor::error(errs(), argv[0]) << EC.message() << '\n';
343 return 1;
344 }
345 Context.setRemarkStreamer(llvm::make_unique<RemarkStreamer>(
346 RemarksFilename,
347 llvm::make_unique<remarks::YAMLSerializer>(YamlFile->os())));
348
349 if (!RemarksPasses.empty())
350 if (Error E = Context.getRemarkStreamer()->setFilter(RemarksPasses)) {
351 WithColor::error(errs(), argv[0]) << E << '\n';
352 return 1;
353 }
354 }
355
356 if (InputLanguage != "" && InputLanguage != "ir" &&
357 InputLanguage != "mir") {
358 WithColor::error(errs(), argv[0])
359 << "input language must be '', 'IR' or 'MIR'\n";
360 return 1;
361 }
362
363 // Compile the module TimeCompilations times to give better compile time
364 // metrics.
365 for (unsigned I = TimeCompilations; I; --I)
366 if (int RetVal = compileModule(argv, Context))
367 return RetVal;
368
369 if (YamlFile)
370 YamlFile->keep();
371 return 0;
372}

374static bool addPass(PassManagerBase &PM, const char *argv0,
375 StringRef PassName, TargetPassConfig &TPC) {
376 if (PassName == "none")
377 return false;
378
379 const PassRegistry *PR = PassRegistry::getPassRegistry();
380 const PassInfo *PI = PR->getPassInfo(PassName);
381 if (!PI) {
382 WithColor::error(errs(), argv0)
383 << "run-pass " << PassName << " is not registered.\n";
384 return true;
385 }
386
387 Pass *P;
388 if (PI->getNormalCtor())
389 P = PI->getNormalCtor()();
390 else {
391 WithColor::error(errs(), argv0)
392 << "cannot create pass: " << PI->getPassName() << "\n";
393 return true;
394 }
395 std::string Banner = std::string("After ") + std::string(P->getPassName());
396 PM.add(P);
397 TPC.printAndVerify(Banner);
398
399 return false;
400}

402static int compileModule(char **argv, LLVMContext &Context) {
403 // Load the module to be compiled...
404 SMDiagnostic Err;
405 std::unique_ptr<Module> M;
406 std::unique_ptr<MIRParser> MIR;
407 Triple TheTriple;
408
409 bool SkipModule = MCPU == "help" ||
410 (!MAttrs.empty() && MAttrs.front() == "help");
411
412 // If user just wants to list available options, skip module loading
413 if (!SkipModule) {
414 if (InputLanguage == "mir" ||
415 (InputLanguage == "" && StringRef(InputFilename).endswith(".mir"))) {
416 MIR = createMIRParserFromFile(InputFilename, Err, Context);
417 if (MIR)
418 M = MIR->parseIRModule();
419 } else
420 M = parseIRFile(InputFilename, Err, Context, false);
421 if (!M) {
422 Err.print(argv[0], WithColor::error(errs(), argv[0]));
423 return 1;
424 }
425
426 // If we are supposed to override the target triple, do so now.
427 if (!TargetTriple.empty())
428 M->setTargetTriple(Triple::normalize(TargetTriple));
429 TheTriple = Triple(M->getTargetTriple());
430 } else {
431 TheTriple = Triple(Triple::normalize(TargetTriple));
432 }
433
434 if (TheTriple.getTriple().empty())
435 TheTriple.setTriple(sys::getDefaultTargetTriple());
436
437 // Get the target specific parser.
438 std::string Error;
439 const Target *TheTarget = TargetRegistry::lookupTarget(MArch, TheTriple,
440 Error);
441 if (!TheTarget) {
442 WithColor::error(errs(), argv[0]) << Error;
443 return 1;
444 }
445
446 std::string CPUStr = getCPUStr(), FeaturesStr = getFeaturesStr();
447
448 CodeGenOpt::Level OLvl = CodeGenOpt::Default;
449 switch (OptLevel) {
450 default:
451 WithColor::error(errs(), argv[0]) << "invalid optimization level.\n";
452 return 1;
453 case ' ': break;
454 case '0': OLvl = CodeGenOpt::None; break;
455 case '1': OLvl = CodeGenOpt::Less; break;
456 case '2': OLvl = CodeGenOpt::Default; break;
457 case '3': OLvl = CodeGenOpt::Aggressive; break;
458 }
459
460 TargetOptions Options = InitTargetOptionsFromCodeGenFlags();
461 Options.DisableIntegratedAS = NoIntegratedAssembler;
462 Options.MCOptions.ShowMCEncoding = ShowMCEncoding;
463 Options.MCOptions.MCUseDwarfDirectory = EnableDwarfDirectory;
464 Options.MCOptions.AsmVerbose = AsmVerbose;
465 Options.MCOptions.PreserveAsmComments = PreserveComments;
466 Options.MCOptions.IASSearchPaths = IncludeDirs;
467 Options.MCOptions.SplitDwarfFile = SplitDwarfFile;
468
469 std::unique_ptr<TargetMachine> Target(TheTarget->createTargetMachine(
470 TheTriple.getTriple(), CPUStr, FeaturesStr, Options, getRelocModel(),
471 getCodeModel(), OLvl));
472
473 assert(Target && "Could not allocate target machine!");
474
475 // If we don't have a module then just exit now. We do this down
476 // here since the CPU/Feature help is underneath the target machine
477 // creation.
478 if (SkipModule)
479 return 0;
480
481 assert(M && "Should have exited if we didn't have a module!");
482 if (FloatABIForCalls != FloatABI::Default)
483 Options.FloatABIType = FloatABIForCalls;
484
485 // Figure out where we are going to send the output.
486 std::unique_ptr<ToolOutputFile> Out =
487 GetOutputStream(TheTarget->getName(), TheTriple.getOS(), argv[0]);
488 if (!Out) return 1;
489
490 std::unique_ptr<ToolOutputFile> DwoOut;
491 if (!SplitDwarfOutputFile.empty()) {
492 std::error_code EC;
493 DwoOut = llvm::make_unique<ToolOutputFile>(SplitDwarfOutputFile, EC,
494 sys::fs::F_None);
495 if (EC) {
496 WithColor::error(errs(), argv[0]) << EC.message() << '\n';
497 return 1;
498 }
499 }
500
501 // Build up all of the passes that we want to do to the module.
502 legacy::PassManager PM;
503
504 // Add an appropriate TargetLibraryInfo pass for the module's triple.
505 TargetLibraryInfoImpl TLII(Triple(M->getTargetTriple()));
506
507 // The -disable-simplify-libcalls flag actually disables all builtin optzns.
508 if (DisableSimplifyLibCalls)
509 TLII.disableAllFunctions();
510 PM.add(new TargetLibraryInfoWrapperPass(TLII));
511
512 // Add the target data from the target machine, if it exists, or the module.
513 M->setDataLayout(Target->createDataLayout());
514
515 // This needs to be done after setting datalayout since it calls verifier
516 // to check debug info whereas verifier relies on correct datalayout.
517 UpgradeDebugInfo(*M);
518
519 // Verify module immediately to catch problems before doInitialization() is
520 // called on any passes.
521 if (!NoVerify && verifyModule(*M, &errs())) {
522 std::string Prefix =
523 (Twine(argv[0]) + Twine(": ") + Twine(InputFilename)).str();
524 WithColor::error(errs(), Prefix) << "input module is broken!\n";
525 return 1;
526 }
527
528 // Override function attributes based on CPUStr, FeaturesStr, and command line
529 // flags.
530 setFunctionAttributes(CPUStr, FeaturesStr, *M);
531
532 if (RelaxAll.getNumOccurrences() > 0 &&
533 FileType != TargetMachine::CGFT_ObjectFile)
534 WithColor::warning(errs(), argv[0])
535 << ": warning: ignoring -mc-relax-all because filetype != obj";
536
537 {
538 raw_pwrite_stream *OS = &Out->os();
539
540 // Manually do the buffering rather than using buffer_ostream,
541 // so we can memcmp the contents in CompileTwice mode
542 SmallVector<char, 0> Buffer;
543 std::unique_ptr<raw_svector_ostream> BOS;
544 if ((FileType != TargetMachine::CGFT_AssemblyFile &&
545 !Out->os().supportsSeeking()) ||
546 CompileTwice) {
547 BOS = make_unique<raw_svector_ostream>(Buffer);
548 OS = BOS.get();
549 }
550
551 const char *argv0 = argv[0];
552 LLVMTargetMachine &LLVMTM = static_cast<LLVMTargetMachine&>(*Target);
553 MachineModuleInfo *MMI = new MachineModuleInfo(&LLVMTM);
554
555 // Construct a custom pass pipeline that starts after instruction
556 // selection.
557 if (!RunPassNames->empty()) {
558 if (!MIR) {
559 WithColor::warning(errs(), argv[0])
560 << "run-pass is for .mir file only.\n";
561 return 1;
562 }
563 TargetPassConfig &TPC = *LLVMTM.createPassConfig(PM);
564 if (TPC.hasLimitedCodeGenPipeline()) {
565 WithColor::warning(errs(), argv[0])
566 << "run-pass cannot be used with "
567 << TPC.getLimitedCodeGenPipelineReason(" and ") << ".\n";
568 return 1;
569 }
570
571 TPC.setDisableVerify(NoVerify);
572 PM.add(&TPC);
573 PM.add(MMI);
574 TPC.printAndVerify("");
575 for (const std::string &RunPassName : *RunPassNames) {
576 if (addPass(PM, argv0, RunPassName, TPC))
577 return 1;
578 }
579 TPC.setInitialized();
580 PM.add(createPrintMIRPass(*OS));
581 if (cheri::ShouldCollectCSetBoundsStats) {
582 PM.add(createLogCheriSetBoundsPass());
583 }
584 PM.add(createFreeMachineFunctionPass());
585 } else if (Target->addPassesToEmitFile(PM, *OS,
586 DwoOut ? &DwoOut->os() : nullptr,
587 FileType, NoVerify, MMI)) {
588 WithColor::warning(errs(), argv[0])
589 << "target does not support generation of this"
590 << " file type!\n";
591 return 1;
592 } else if (cheri::ShouldCollectCSetBoundsStats) {
593 PM.add(createLogCheriSetBoundsPass());
594 }
595
596 if (MIR) {
597 assert(MMI && "Forgot to create MMI?");
598 if (MIR->parseMachineFunctions(*M, *MMI))
599 return 1;
600 }
601
602 // Before executing passes, print the final values of the LLVM options.
603 cl::PrintOptionValues();
604
605 // If requested, run the pass manager over the same module again,
606 // to catch any bugs due to persistent state in the passes. Note that
607 // opt has the same functionality, so it may be worth abstracting this out
608 // in the future.
609 SmallVector<char, 0> CompileTwiceBuffer;
610 if (CompileTwice) {
611 std::unique_ptr<Module> M2(llvm::CloneModule(*M));
612 PM.run(*M2);
613 CompileTwiceBuffer = Buffer;
614 Buffer.clear();
615 }
616
617 PM.run(*M);
618
619 auto HasError =
620 ((const LLCDiagnosticHandler *)(Context.getDiagHandlerPtr()))->HasError;
621 if (*HasError)
622 return 1;
623
624 // Compare the two outputs and make sure they're the same
625 if (CompileTwice) {
626 if (Buffer.size() != CompileTwiceBuffer.size() ||
627 (memcmp(Buffer.data(), CompileTwiceBuffer.data(), Buffer.size()) !=
628 0)) {
629 errs()
630 << "Running the pass manager twice changed the output.\n"
631 "Writing the result of the second run to the specified output\n"
632 "To generate the one-run comparison binary, just run without\n"
633 "the compile-twice option\n";
634 Out->os() << Buffer;
635 Out->keep();
636 return 1;
637 }
638 }
639
640 if (BOS) {
641 Out->os() << Buffer;
642 }
643 }
644
645 // Declare success.
646 Out->keep();
647 if (DwoOut)
648 DwoOut->keep();
649
650 if (cheri::ShouldCollectCSetBoundsStats) {
651 auto StatsFile = llvm::cheri::StatsOutputFile::open(
652 cheri::CSetBoundsStatistics::outputFile(),
653 [](StringRef StatsFile, const std::error_code &EC) {
654 errs() << "Couldn't open CSetBounds stats file " << StatsFile << ": " << EC.message();
655 },
656 [](StringRef StatsFile, const std::error_code &EC) {
657 errs() << "Couldn't lock CSetBounds stats file " << StatsFile << ": " << EC.message();
658 });
659 if (StatsFile)
660 llvm::cheri::CSetBoundsStats->print(*StatsFile, InputFilename);
661 }
662
663 return 0;
664}
