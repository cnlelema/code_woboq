125static void failIfError(std::error_code EC, Twine Context = "") {
126 if (!EC)
127 return;
128
129 std::string ContextStr = Context.str();
130 if (ContextStr.empty())
131 fail(EC.message());
132 fail(Context + ": " + EC.message());
133}

135static void failIfError(Error E, Twine Context = "") {
136 if (!E)
137 return;
138
139 handleAllErrors(std::move(E), [&](const llvm::ErrorInfoBase &EIB) {
140 std::string ContextStr = Context.str();
141 if (ContextStr.empty())
142 fail(EIB.message());
143 fail(Context + ": " + EIB.message());
144 });
145}

159// This enumeration delineates the kinds of operations on an archive
160// that are permitted.
161enum ArchiveOperation {
162 Print, ///< Print the contents of the archive
163 Delete, ///< Delete the specified members
164 Move, ///< Move members to end or as given by {a,b,i} modifiers
165 QuickAppend, ///< Quickly append to end of archive
166 ReplaceOrInsert, ///< Replace or Insert members
167 DisplayTable, ///< Display the table of contents
168 Extract, ///< Extract files back to file system
169 CreateSymTab ///< Create a symbol table in an existing archive
170};

185// Relative Positional Argument (for insert/move). This variable holds
186// the name of the archive member to which the 'a', 'b' or 'i' modifier
187// refers. Only one of 'a', 'b' or 'i' can be specified so we only need
188// one variable.

191// Count parameter for 'N' modifier. This variable specifies which file should

207// Extract the member filename from the command line for the [relpos] argument
208// associated with a, b, and i modifiers
209static void getRelPos() {
210 if (PositionalArgs.empty())
211 fail("Expected [relpos] for a, b, or i modifier");
212 RelPos = PositionalArgs[0];
213 PositionalArgs.erase(PositionalArgs.begin());
214}

216// Extract the parameter from the command line for the [count] argument
217// associated with the N modifier
218static void getCountParam() {
219 if (PositionalArgs.empty())
220 fail("Expected [count] for N modifier");
221 auto CountParamArg = StringRef(PositionalArgs[0]);
222 if (CountParamArg.getAsInteger(10, CountParam))
223 fail("Value for [count] must be numeric, got: " + CountParamArg);
224 if (CountParam < 1)
225 fail("Value for [count] must be positive, got: " + CountParamArg);
226 PositionalArgs.erase(PositionalArgs.begin());
227}

229// Get the archive file name from the command line
230static void getArchive() {
231 if (PositionalArgs.empty())
232 fail("An archive name must be specified");
233 ArchiveName = PositionalArgs[0];
234 PositionalArgs.erase(PositionalArgs.begin());
235}

237// Copy over remaining items in PositionalArgs to our Members vector
238static void getMembers() {
239 for (auto &Arg : PositionalArgs)
240 Members.push_back(Arg);

246static object::Archive &readLibrary(const Twine &Library) {
247 auto BufOrErr = MemoryBuffer::getFile(Library, -1, false);
248 failIfError(BufOrErr.getError(), "Could not open library " + Library);
249 ArchiveBuffers.push_back(std::move(*BufOrErr));
250 auto LibOrErr =
251 object::Archive::create(ArchiveBuffers.back()->getMemBufferRef());
252 failIfError(errorToErrorCode(LibOrErr.takeError()),
253 "Could not parse library");
254 Archives.push_back(std::move(*LibOrErr));
255 return *Archives.back();
256}

260// Parse the command line options as presented and return the operation
261// specified. Process all modifiers and check to make sure that constraints on
262// modifier/operation pairs have not been violated.
263static ArchiveOperation parseCommandLine() {
264 if (MRI) {
265 if (!PositionalArgs.empty() || !Options.empty())
266 fail("Cannot mix -M and other options");
267 runMRIScript();
268 }
269
270 // Keep track of number of operations. We can only specify one
271 // per execution.
272 unsigned NumOperations = 0;
273
274 // Keep track of the number of positional modifiers (a,b,i). Only
275 // one can be specified.
276 unsigned NumPositional = 0;
277
278 // Keep track of which operation was requested
279 ArchiveOperation Operation;
280
281 bool MaybeJustCreateSymTab = false;
282
283 for (unsigned i = 0; i < Options.size(); ++i) {
284 switch (Options[i]) {
285 case 'd':
286 ++NumOperations;
287 Operation = Delete;
288 break;
289 case 'm':
290 ++NumOperations;
291 Operation = Move;
292 break;
293 case 'p':
294 ++NumOperations;
295 Operation = Print;
296 break;
297 case 'q':
298 ++NumOperations;
299 Operation = QuickAppend;
300 break;
301 case 'r':
302 ++NumOperations;
303 Operation = ReplaceOrInsert;
304 break;
305 case 't':
306 ++NumOperations;
307 Operation = DisplayTable;
308 break;
309 case 'x':
310 ++NumOperations;
311 Operation = Extract;
312 break;
313 case 'c':
314 Create = true;
315 break;
316 case 'l': /* accepted but unused */
317 break;
318 case 'o':
319 OriginalDates = true;
320 break;
321 case 'P':
322 CompareFullPath = true;
323 break;
324 case 's':
325 Symtab = true;
326 MaybeJustCreateSymTab = true;
327 break;
328 case 'S':
329 Symtab = false;
330 break;
331 case 'u':
332 OnlyUpdate = true;
333 break;
334 case 'v':
335 Verbose = true;
336 break;
337 case 'a':
338 getRelPos();
339 AddAfter = true;
340 NumPositional++;
341 break;
342 case 'b':
343 getRelPos();
344 AddBefore = true;
345 NumPositional++;
346 break;
347 case 'i':
348 getRelPos();
349 AddBefore = true;
350 NumPositional++;
351 break;
352 case 'D':
353 Deterministic = true;
354 break;
355 case 'U':
356 Deterministic = false;
357 break;
358 case 'N':
359 getCountParam();
360 break;
361 case 'T':
362 Thin = true;
363 // Thin archives store path names, so P should be forced.
364 CompareFullPath = true;
365 break;
366 case 'L':
367 AddLibrary = true;
368 break;
369 default:
370 fail(std::string("unknown option ") + Options[i]);
371 }
372 }
373
374 // At this point, the next thing on the command line must be
375 // the archive name.
376 getArchive();
377
378 // Everything on the command line at this point is a member.
379 getMembers();
380
381 if (NumOperations == 0 && MaybeJustCreateSymTab) {
382 NumOperations = 1;
383 Operation = CreateSymTab;
384 if (!Members.empty())
385 fail("The s operation takes only an archive as argument");
386 }
387
388 // Perform various checks on the operation/modifier specification
389 // to make sure we are dealing with a legal request.
390 if (NumOperations == 0)
391 fail("You must specify at least one of the operations");
392 if (NumOperations > 1)
393 fail("Only one operation may be specified");
394 if (NumPositional > 1)
395 fail("You may only specify one of a, b, and i modifiers");
396 if (AddAfter || AddBefore)
397 if (Operation != Move && Operation != ReplaceOrInsert)
398 fail("The 'a', 'b' and 'i' modifiers can only be specified with "
399 "the 'm' or 'r' operations");
400 if (CountParam)
401 if (Operation != Extract && Operation != Delete)
402 fail("The 'N' modifier can only be specified with the 'x' or 'd' "
403 "operations");
404 if (OriginalDates && Operation != Extract)
405 fail("The 'o' modifier is only applicable to the 'x' operation");
406 if (OnlyUpdate && Operation != ReplaceOrInsert)
407 fail("The 'u' modifier is only applicable to the 'r' operation");
408 if (AddLibrary && Operation != QuickAppend)
409 fail("The 'L' modifier is only applicable to the 'q' operation");
410
411 // Return the parsed operation to the caller
412 return Operation;
413}

415// Implements the 'p' operation. This function traverses the archive
416// looking for members that match the path list.
417static void doPrint(StringRef Name, const object::Archive::Child &C) {
418 if (Verbose)
419 outs() << "Printing " << Name << "\n";
420
421 Expected<StringRef> DataOrErr = C.getBuffer();
422 failIfError(DataOrErr.takeError());
423 StringRef Data = *DataOrErr;
424 outs().write(Data.data(), Data.size());
425}

427// Utility function for printing out the file mode when the 't' operation is in
428// verbose mode.
429static void printMode(unsigned mode) {
430 outs() << ((mode & 004) ? "r" : "-");
431 outs() << ((mode & 002) ? "w" : "-");
432 outs() << ((mode & 001) ? "x" : "-");
433}
434
435// Implement the 't' operation. This function prints out just
436// the file names of each of the members. However, if verbose mode is requested
437// ('v' modifier) then the file type, permission mode, user, group, size, and
438// modification time are also printed.
439static void doDisplayTable(StringRef Name, const object::Archive::Child &C) {
440 if (Verbose) {
441 Expected<sys::fs::perms> ModeOrErr = C.getAccessMode();
442 failIfError(ModeOrErr.takeError());
443 sys::fs::perms Mode = ModeOrErr.get();
444 printMode((Mode >> 6) & 007);
445 printMode((Mode >> 3) & 007);
446 printMode(Mode & 007);
447 Expected<unsigned> UIDOrErr = C.getUID();
448 failIfError(UIDOrErr.takeError());
449 outs() << ' ' << UIDOrErr.get();
450 Expected<unsigned> GIDOrErr = C.getGID();
451 failIfError(GIDOrErr.takeError());
452 outs() << '/' << GIDOrErr.get();
453 Expected<uint64_t> Size = C.getSize();
454 failIfError(Size.takeError());
455 outs() << ' ' << format("%6llu", Size.get());
456 auto ModTimeOrErr = C.getLastModified();
457 failIfError(ModTimeOrErr.takeError());
458 // Note: formatv() only handles the default TimePoint<>, which is in
459 // nanoseconds.
460 // TODO: fix format_provider<TimePoint<>> to allow other units.
461 sys::TimePoint<> ModTimeInNs = ModTimeOrErr.get();
462 outs() << ' ' << formatv("{0:%b %e %H:%M %Y}", ModTimeInNs);
463 outs() << ' ';
464 }
465
466 if (C.getParent()->isThin()) {
467 if (!sys::path::is_absolute(Name)) {
468 StringRef ParentDir = sys::path::parent_path(ArchiveName);
469 if (!ParentDir.empty())
470 outs() << sys::path::convert_to_slash(ParentDir) << '/';
471 }
472 }
473 outs() << Name << "\n";
474}

476static StringRef normalizePath(StringRef Path) {
477 return CompareFullPath ? Path : sys::path::filename(Path);
478}

480// Implement the 'x' operation. This function extracts files back to the file
481// system.
482static void doExtract(StringRef Name, const object::Archive::Child &C) {
483 // Retain the original mode.
484 Expected<sys::fs::perms> ModeOrErr = C.getAccessMode();
485 failIfError(ModeOrErr.takeError());
486 sys::fs::perms Mode = ModeOrErr.get();
487
488 int FD;
489 failIfError(sys::fs::openFileForWrite(sys::path::filename(Name), FD,
490 sys::fs::CD_CreateAlways,
491 sys::fs::F_None, Mode),
492 Name);
493
494 {
495 raw_fd_ostream file(FD, false);
496
497 // Get the data and its length
498 Expected<StringRef> BufOrErr = C.getBuffer();
499 failIfError(BufOrErr.takeError());
500 StringRef Data = BufOrErr.get();
501
502 // Write the data.
503 file.write(Data.data(), Data.size());
504 }
505
506 // If we're supposed to retain the original modification times, etc. do so
507 // now.
508 if (OriginalDates) {
509 auto ModTimeOrErr = C.getLastModified();
510 failIfError(ModTimeOrErr.takeError());
511 failIfError(
512 sys::fs::setLastAccessAndModificationTime(FD, ModTimeOrErr.get()));
513 }
514
515 if (close(FD))
516 fail("Could not close the file");
517}

519static bool shouldCreateArchive(ArchiveOperation Op) {
520 switch (Op) {
521 case Print:
522 case Delete:
523 case Move:
524 case DisplayTable:
525 case Extract:
526 case CreateSymTab:
527 return false;
528
529 case QuickAppend:
530 case ReplaceOrInsert:
531 return true;
532 }
533
534 llvm_unreachable("Missing entry in covered switch.");
535}

537static void performReadOperation(ArchiveOperation Operation,
538 object::Archive *OldArchive) {
539 if (Operation == Extract && OldArchive->isThin())
540 fail("extracting from a thin archive is not supported");
541
542 bool Filter = !Members.empty();
543 StringMap<int> MemberCount;
544 {
545 Error Err = Error::success();
546 for (auto &C : OldArchive->children(Err)) {
547 Expected<StringRef> NameOrErr = C.getName();
548 failIfError(NameOrErr.takeError());
549 StringRef Name = NameOrErr.get();
550
551 if (Filter) {
552 auto I = find_if(Members, [Name](StringRef Path) {
553 return Name == normalizePath(Path);
554 });
555 if (I == Members.end())
556 continue;
557 if (CountParam && ++MemberCount[Name] != CountParam)
558 continue;
559 Members.erase(I);
560 }
561
562 switch (Operation) {
563 default:
564 llvm_unreachable("Not a read operation");
565 case Print:
566 doPrint(Name, C);
567 break;
568 case DisplayTable:
569 doDisplayTable(Name, C);
570 break;
571 case Extract:
572 doExtract(Name, C);
573 break;
574 }
575 }
576 failIfError(std::move(Err));
577 }
578
579 if (Members.empty())
580 return;
581 for (StringRef Name : Members)
582 WithColor::error(errs(), ToolName) << "'" << Name << "' was not found\n";
583 exit(1);
584}

586static void addChildMember(std::vector<NewArchiveMember> &Members,
587 const object::Archive::Child &M,
588 bool FlattenArchive = false) {
589 if (Thin && !M.getParent()->isThin())
590 fail("Cannot convert a regular archive to a thin one");
591 Expected<NewArchiveMember> NMOrErr =
592 NewArchiveMember::getOldMember(M, Deterministic);
593 failIfError(NMOrErr.takeError());
594 // If the child member we're trying to add is thin, use the path relative to
595 // the archive it's in, so the file resolves correctly.
596 if (Thin && FlattenArchive) {
597 StringSaver Saver(Alloc);
598 Expected<std::string> FileNameOrErr = M.getName();
599 failIfError(FileNameOrErr.takeError());
600 if (sys::path::is_absolute(*FileNameOrErr)) {
601 NMOrErr->MemberName = Saver.save(sys::path::convert_to_slash(*FileNameOrErr));
602 } else {
603 FileNameOrErr = M.getFullName();
604 failIfError(FileNameOrErr.takeError());
605 Expected<std::string> PathOrErr =
606 computeArchiveRelativePath(ArchiveName, *FileNameOrErr);
607 NMOrErr->MemberName = Saver.save(
608 PathOrErr ? *PathOrErr : sys::path::convert_to_slash(*FileNameOrErr));
609 }
610 }
611 if (FlattenArchive &&
612 identify_magic(NMOrErr->Buf->getBuffer()) == file_magic::archive) {
613 Expected<std::string> FileNameOrErr = M.getFullName();
614 failIfError(FileNameOrErr.takeError());
615 object::Archive &Lib = readLibrary(*FileNameOrErr);
616 // When creating thin archives, only flatten if the member is also thin.
617 if (!Thin || Lib.isThin()) {
618 Error Err = Error::success();
619 // Only Thin archives are recursively flattened.
620 for (auto &Child : Lib.children(Err))
621 addChildMember(Members, Child, /*FlattenArchive=*/Thin);
622 failIfError(std::move(Err));
623 return;
624 }
625 }
626 Members.push_back(std::move(*NMOrErr));
627}

629static void addMember(std::vector<NewArchiveMember> &Members,
630 StringRef FileName, bool FlattenArchive = false) {
631 Expected<NewArchiveMember> NMOrErr =
632 NewArchiveMember::getFile(FileName, Deterministic);
633 failIfError(NMOrErr.takeError(), FileName);
634 StringSaver Saver(Alloc);
635 // For regular archives, use the basename of the object path for the member
636 // name. For thin archives, use the full relative paths so the file resolves
637 // correctly.
638 if (!Thin) {
639 NMOrErr->MemberName = sys::path::filename(NMOrErr->MemberName);
640 } else {
641 if (sys::path::is_absolute(FileName))
642 NMOrErr->MemberName = Saver.save(sys::path::convert_to_slash(FileName));
643 else {
644 Expected<std::string> PathOrErr =
645 computeArchiveRelativePath(ArchiveName, FileName);
646 NMOrErr->MemberName = Saver.save(
647 PathOrErr ? *PathOrErr : sys::path::convert_to_slash(FileName));
648 }
649 }
650
651 if (FlattenArchive &&
652 identify_magic(NMOrErr->Buf->getBuffer()) == file_magic::archive) {
653 object::Archive &Lib = readLibrary(FileName);
654 // When creating thin archives, only flatten if the member is also thin.
655 if (!Thin || Lib.isThin()) {
656 Error Err = Error::success();
657 // Only Thin archives are recursively flattened.
658 for (auto &Child : Lib.children(Err))
659 addChildMember(Members, Child, /*FlattenArchive=*/Thin);
660 failIfError(std::move(Err));
661 return;
662 }
663 }
664 Members.push_back(std::move(*NMOrErr));
665}

667enum InsertAction {
668 IA_AddOldMember,
669 IA_AddNewMember,
670 IA_Delete,
671 IA_MoveOldMember,
672 IA_MoveNewMember
673};
674
675static InsertAction computeInsertAction(ArchiveOperation Operation,
676 const object::Archive::Child &Member,
677 StringRef Name,
678 std::vector<StringRef>::iterator &Pos,
679 StringMap<int> &MemberCount) {
680 if (Operation == QuickAppend || Members.empty())
681 return IA_AddOldMember;
682 auto MI = find_if(
683 Members, [Name](StringRef Path) { return Name == normalizePath(Path); });
684
685 if (MI == Members.end())
686 return IA_AddOldMember;
687
688 Pos = MI;
689
690 if (Operation == Delete) {
691 if (CountParam && ++MemberCount[Name] != CountParam)
692 return IA_AddOldMember;
693 return IA_Delete;
694 }
695
696 if (Operation == Move)
697 return IA_MoveOldMember;
698
699 if (Operation == ReplaceOrInsert) {
700 StringRef PosName = normalizePath(RelPos);
701 if (!OnlyUpdate) {
702 if (PosName.empty())
703 return IA_AddNewMember;
704 return IA_MoveNewMember;
705 }
706
707 // We could try to optimize this to a fstat, but it is not a common
708 // operation.
709 sys::fs::file_status Status;
710 failIfError(sys::fs::status(*MI, Status), *MI);
711 auto ModTimeOrErr = Member.getLastModified();
712 failIfError(ModTimeOrErr.takeError());
713 if (Status.getLastModificationTime() < ModTimeOrErr.get()) {
714 if (PosName.empty())
715 return IA_AddOldMember;
716 return IA_MoveOldMember;
717 }
718
719 if (PosName.empty())
720 return IA_AddNewMember;
721 return IA_MoveNewMember;
722 }
723 llvm_unreachable("No such operation");
724}

726// We have to walk this twice and computing it is not trivial, so creating an
727// explicit std::vector is actually fairly efficient.
728static std::vector<NewArchiveMember>
729computeNewArchiveMembers(ArchiveOperation Operation,
730 object::Archive *OldArchive) {
731 std::vector<NewArchiveMember> Ret;
732 std::vector<NewArchiveMember> Moved;
733 int InsertPos = -1;
734 StringRef PosName = normalizePath(RelPos);
735 if (OldArchive) {
736 Error Err = Error::success();
737 StringMap<int> MemberCount;
738 for (auto &Child : OldArchive->children(Err)) {
739 int Pos = Ret.size();
740 Expected<StringRef> NameOrErr = Child.getName();
741 failIfError(NameOrErr.takeError());
742 StringRef Name = NameOrErr.get();
743 if (Name == PosName) {
744 assert(AddAfter || AddBefore);
745 if (AddBefore)
746 InsertPos = Pos;
747 else
748 InsertPos = Pos + 1;
749 }
750
751 std::vector<StringRef>::iterator MemberI = Members.end();
752 InsertAction Action =
753 computeInsertAction(Operation, Child, Name, MemberI, MemberCount);
754 switch (Action) {
755 case IA_AddOldMember:
756 addChildMember(Ret, Child, /*FlattenArchive=*/Thin);
757 break;
758 case IA_AddNewMember:
759 addMember(Ret, *MemberI);
760 break;
761 case IA_Delete:
762 break;
763 case IA_MoveOldMember:
764 addChildMember(Moved, Child, /*FlattenArchive=*/Thin);
765 break;
766 case IA_MoveNewMember:
767 addMember(Moved, *MemberI);
768 break;
769 }
770 // When processing elements with the count param, we need to preserve the
771 // full members list when iterating over all archive members. For
772 // instance, "llvm-ar dN 2 archive.a member.o" should delete the second
773 // file named member.o it sees; we are not done with member.o the first
774 // time we see it in the archive.
775 if (MemberI != Members.end() && !CountParam)
776 Members.erase(MemberI);
777 }
778 failIfError(std::move(Err));
779 }
780
781 if (Operation == Delete)
782 return Ret;
783
784 if (!RelPos.empty() && InsertPos == -1)
785 fail("Insertion point not found");
786
787 if (RelPos.empty())
788 InsertPos = Ret.size();
789
790 assert(unsigned(InsertPos) <= Ret.size());
791 int Pos = InsertPos;
792 for (auto &M : Moved) {
793 Ret.insert(Ret.begin() + Pos, std::move(M));
794 ++Pos;
795 }
796
797 if (AddLibrary) {
798 assert(Operation == QuickAppend);
799 for (auto &Member : Members)
800 addMember(Ret, Member, /*FlattenArchive=*/true);
801 return Ret;
802 }
803
804 std::vector<NewArchiveMember> NewMembers;
805 for (auto &Member : Members)
806 addMember(NewMembers, Member, /*FlattenArchive=*/Thin);
807 Ret.reserve(Ret.size() + NewMembers.size());
808 std::move(NewMembers.begin(), NewMembers.end(),
809 std::inserter(Ret, std::next(Ret.begin(), InsertPos)));
810
811 return Ret;
812}

815 return Triple(sys::getProcessTriple()).isOSDarwin()
816 ? object::Archive::K_DARWIN
817 : object::Archive::K_GNU;

820static object::Archive::Kind getKindFromMember(const NewArchiveMember &Member) {
821 Expected<std::unique_ptr<object::ObjectFile>> OptionalObject =
822 object::ObjectFile::createObjectFile(Member.Buf->getMemBufferRef());
823
824 if (OptionalObject)
825 return isa<object::MachOObjectFile>(**OptionalObject)
826 ? object::Archive::K_DARWIN
827 : object::Archive::K_GNU;
828
829 // squelch the error in case we had a non-object file
830 consumeError(OptionalObject.takeError());
831 return getDefaultForHost();

834static void performWriteOperation(ArchiveOperation Operation,
835 object::Archive *OldArchive,
836 std::unique_ptr<MemoryBuffer> OldArchiveBuf,
837 std::vector<NewArchiveMember> *NewMembersP) {
838 std::vector<NewArchiveMember> NewMembers;
839 if (!NewMembersP)
840 NewMembers = computeNewArchiveMembers(Operation, OldArchive);
841
842 object::Archive::Kind Kind;
843 switch (FormatType) {
844 case Default:
845 if (Thin)
846 Kind = object::Archive::K_GNU;
847 else if (OldArchive)
848 Kind = OldArchive->kind();
849 else if (NewMembersP)
850 Kind = !NewMembersP->empty() ? getKindFromMember(NewMembersP->front())
851 : getDefaultForHost();
852 else
853 Kind = !NewMembers.empty() ? getKindFromMember(NewMembers.front())
854 : getDefaultForHost();
855 break;
856 case GNU:
857 Kind = object::Archive::K_GNU;
858 break;
859 case BSD:
860 if (Thin)
861 fail("Only the gnu format has a thin mode");
862 Kind = object::Archive::K_BSD;
863 break;
864 case DARWIN:
865 if (Thin)
866 fail("Only the gnu format has a thin mode");
867 Kind = object::Archive::K_DARWIN;
868 break;
869 case Unknown:
870 llvm_unreachable("");
871 }
872
873 Error E =
874 writeArchive(ArchiveName, NewMembersP ? *NewMembersP : NewMembers, Symtab,
875 Kind, Deterministic, Thin, std::move(OldArchiveBuf));
876 failIfError(std::move(E), ArchiveName);
877}

879static void createSymbolTable(object::Archive *OldArchive) {
880 // When an archive is created or modified, if the s option is given, the
881 // resulting archive will have a current symbol table. If the S option
882 // is given, it will have no symbol table.
883 // In summary, we only need to update the symbol table if we have none.
884 // This is actually very common because of broken build systems that think
885 // they have to run ranlib.
886 if (OldArchive->hasSymbolTable())
887 return;
888
889 performWriteOperation(CreateSymTab, OldArchive, nullptr, nullptr);
890}

892static void performOperation(ArchiveOperation Operation,
893 object::Archive *OldArchive,
894 std::unique_ptr<MemoryBuffer> OldArchiveBuf,
895 std::vector<NewArchiveMember> *NewMembers) {
896 switch (Operation) {
897 case Print:
898 case DisplayTable:
899 case Extract:
900 performReadOperation(Operation, OldArchive);
901 return;
902
903 case Delete:
904 case Move:
905 case QuickAppend:
906 case ReplaceOrInsert:
907 performWriteOperation(Operation, OldArchive, std::move(OldArchiveBuf),
908 NewMembers);
909 return;
910 case CreateSymTab:
911 createSymbolTable(OldArchive);
912 return;
913 }
914 llvm_unreachable("Unknown operation.");
915}

917static int performOperation(ArchiveOperation Operation,
918 std::vector<NewArchiveMember> *NewMembers) {
919 // Create or open the archive object.
920 ErrorOr<std::unique_ptr<MemoryBuffer>> Buf =
921 MemoryBuffer::getFile(ArchiveName, -1, false);
922 std::error_code EC = Buf.getError();
923 if (EC && EC != errc::no_such_file_or_directory)
924 fail("error opening '" + ArchiveName + "': " + EC.message() + "!");
925
926 if (!EC) {
927 Error Err = Error::success();
928 object::Archive Archive(Buf.get()->getMemBufferRef(), Err);
929 EC = errorToErrorCode(std::move(Err));
930 failIfError(EC,
931 "error loading '" + ArchiveName + "': " + EC.message() + "!");
932 if (Archive.isThin())
933 CompareFullPath = true;
934 performOperation(Operation, &Archive, std::move(Buf.get()), NewMembers);
935 return 0;
936 }
937
938 assert(EC == errc::no_such_file_or_directory);
939
940 if (!shouldCreateArchive(Operation)) {
941 failIfError(EC, Twine("error loading '") + ArchiveName + "'");
942 } else {
943 if (!Create) {
944 // Produce a warning if we should and we're creating the archive
945 WithColor::warning(errs(), ToolName)
946 << "creating " << ArchiveName << "\n";
947 }
948 }
949
950 performOperation(Operation, nullptr, nullptr, NewMembers);
951 return 0;
952}

954static void runMRIScript() {
955 enum class MRICommand { AddLib, AddMod, Create, CreateThin, Delete, Save, End, Invalid };
956
957 ErrorOr<std::unique_ptr<MemoryBuffer>> Buf = MemoryBuffer::getSTDIN();
958 failIfError(Buf.getError());
959 const MemoryBuffer &Ref = *Buf.get();
960 bool Saved = false;
961 std::vector<NewArchiveMember> NewMembers;
962
963 for (line_iterator I(Ref, /*SkipBlanks*/ false), E; I != E; ++I) {
964 StringRef Line = *I;
965 Line = Line.split(';').first;
966 Line = Line.split('*').first;
967 Line = Line.trim();
968 if (Line.empty())
969 continue;
970 StringRef CommandStr, Rest;
971 std::tie(CommandStr, Rest) = Line.split(' ');
972 Rest = Rest.trim();
973 if (!Rest.empty() && Rest.front() == '"' && Rest.back() == '"')
974 Rest = Rest.drop_front().drop_back();
975 auto Command = StringSwitch<MRICommand>(CommandStr.lower())
976 .Case("addlib", MRICommand::AddLib)
977 .Case("addmod", MRICommand::AddMod)
978 .Case("create", MRICommand::Create)
979 .Case("createthin", MRICommand::CreateThin)
980 .Case("delete", MRICommand::Delete)
981 .Case("save", MRICommand::Save)
982 .Case("end", MRICommand::End)
983 .Default(MRICommand::Invalid);
984
985 switch (Command) {
986 case MRICommand::AddLib: {
987 object::Archive &Lib = readLibrary(Rest);
988 {
989 Error Err = Error::success();
990 for (auto &Member : Lib.children(Err))
991 addChildMember(NewMembers, Member, /*FlattenArchive=*/Thin);
992 failIfError(std::move(Err));
993 }
994 break;
995 }
996 case MRICommand::AddMod:
997 addMember(NewMembers, Rest);
998 break;
999 case MRICommand::CreateThin:
1000 Thin = true;
1001 LLVM_FALLTHROUGH;
1002 case MRICommand::Create:
1003 Create = true;
1004 if (!ArchiveName.empty())
1005 fail("Editing multiple archives not supported");
1006 if (Saved)
1007 fail("File already saved");
1008 ArchiveName = Rest;
1009 break;
1010 case MRICommand::Delete: {
1011 StringRef Name = normalizePath(Rest);
1012 llvm::erase_if(NewMembers,
1013 [=](NewArchiveMember &M) { return M.MemberName == Name; });
1014 break;
1015 }
1016 case MRICommand::Save:
1017 Saved = true;
1018 break;
1019 case MRICommand::End:
1020 break;
1021 case MRICommand::Invalid:
1022 fail("Unknown command: " + CommandStr);
1023 }
1024 }
1025
1026 // Nothing to do if not saved.
1027 if (Saved)
1028 performOperation(ReplaceOrInsert, &NewMembers);
1029 exit(0);
1030}

1044static int ar_main(int argc, char **argv) {
1045 SmallVector<const char *, 0> Argv(argv, argv + argc);
1046 StringSaver Saver(Alloc);
1047 cl::ExpandResponseFiles(Saver, cl::TokenizeGNUCommandLine, Argv);
1048 for (size_t i = 1; i < Argv.size(); ++i) {
1049 StringRef Arg = Argv[i];
1050 const char *match;
1051 auto MatchFlagWithArg = [&](const char *expected) {
1052 size_t len = strlen(expected);
1053 if (Arg == expected) {
1054 if (++i >= Argv.size())
1055 fail(std::string(expected) + " requires an argument");
1056 match = Argv[i];
1057 return true;
1058 }
1059 if (Arg.startswith(expected) && Arg.size() > len && Arg[len] == '=') {
1060 match = Arg.data() + len + 1;
1061 return true;
1062 }
1063 return false;
1064 };
1065 if (handleGenericOption(Argv[i]))
1066 return 0;
1067 if (Arg == "--") {
1068 for (; i < Argv.size(); ++i)
1069 PositionalArgs.push_back(Argv[i]);
1070 break;
1071 }
1072 if (Arg[0] == '-') {
1073 if (Arg.startswith("--"))
1074 Arg = Argv[i] + 2;
1075 else
1076 Arg = Argv[i] + 1;
1077 if (Arg == "M") {
1078 MRI = true;
1079 } else if (MatchFlagWithArg("format")) {
1080 FormatType = StringSwitch<Format>(match)
1081 .Case("default", Default)
1082 .Case("gnu", GNU)
1083 .Case("darwin", DARWIN)
1084 .Case("bsd", BSD)
1085 .Default(Unknown);
1086 if (FormatType == Unknown)
1087 fail(std::string("Invalid format ") + match);
1088 } else if (MatchFlagWithArg("plugin")) {
1089 // Ignored.
1090 } else {
1091 Options += Argv[i] + 1;
1092 }
1093 } else if (Options.empty()) {
1094 Options += Argv[i];
1095 } else {
1096 PositionalArgs.push_back(Argv[i]);
1097 }
1098 }
1099 ArchiveOperation Operation = parseCommandLine();
1100 return performOperation(Operation, nullptr);
1101}

1103static int ranlib_main(int argc, char **argv) {
1104 bool ArchiveSpecified = false;
1105 for (int i = 1; i < argc; ++i) {

1118int main(int argc, char **argv) {
1119 InitLLVM X(argc, argv);
1120 ToolName = argv[0];

1122 llvm::InitializeAllTargetInfos();
1123 llvm::InitializeAllTargetMCs();
1124 llvm::InitializeAllAsmParsers();

1126 Stem = sys::path::stem(ToolName);
