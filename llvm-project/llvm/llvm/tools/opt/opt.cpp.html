304class OptCustomPassManager : public legacy::PassManager {
305 DebugifyStatsMap DIStatsMap;
306
307public:
308 using super = legacy::PassManager;
309
310 void add(Pass *P) override {
311 // Wrap each pass with (-check)-debugify passes if requested, making
312 // exceptions for passes which shouldn't see -debugify instrumentation.
313 bool WrapWithDebugify = DebugifyEach && !P->getAsImmutablePass() &&
314 !isIRPrintingPass(P) && !isBitcodeWriterPass(P);
315 if (!WrapWithDebugify) {
316 super::add(P);
317 return;
318 }
319
320 // Apply -debugify/-check-debugify before/after each pass and collect
321 // debug info loss statistics.
322 PassKind Kind = P->getPassKind();
323 StringRef Name = P->getPassName();
324
325 // TODO: Implement Debugify for BasicBlockPass, LoopPass.
326 switch (Kind) {
327 case PT_Function:
328 super::add(createDebugifyFunctionPass());
329 super::add(P);
330 super::add(createCheckDebugifyFunctionPass(true, Name, &DIStatsMap));
331 break;
332 case PT_Module:
333 super::add(createDebugifyModulePass());
334 super::add(P);
335 super::add(createCheckDebugifyModulePass(true, Name, &DIStatsMap));
336 break;
337 default:
338 super::add(P);
339 break;
340 }
341 }
342
343 const DebugifyStatsMap &getDebugifyStatsMap() const { return DIStatsMap; }
344};

346static inline void addPass(legacy::PassManagerBase &PM, Pass *P) {
347 // Add the pass to the pass manager...
348 PM.add(P);
349
350 // If we are verifying all of the intermediate steps, add the verifier...
351 if (VerifyEach)
352 PM.add(createVerifierPass());
353}

355/// This routine adds optimization passes based on selected optimization level,
356/// OptLevel.
357///
358/// OptLevel - Optimization Level
359static void AddOptimizationPasses(legacy::PassManagerBase &MPM,
360 legacy::FunctionPassManager &FPM,
361 TargetMachine *TM, unsigned OptLevel,
362 unsigned SizeLevel) {
363 if (!NoVerify || VerifyEach)
364 FPM.add(createVerifierPass()); // Verify that input is correct
365
366 PassManagerBuilder Builder;
367 Builder.OptLevel = OptLevel;
368 Builder.SizeLevel = SizeLevel;
369
370 if (DisableInline) {
371 // No inlining pass
372 } else if (OptLevel > 1) {
373 Builder.Inliner = createFunctionInliningPass(OptLevel, SizeLevel, false);
374 } else {
375 Builder.Inliner = createAlwaysInlinerLegacyPass();
376 }
377 Builder.DisableUnrollLoops = (DisableLoopUnrolling.getNumOccurrences() > 0) ?
378 DisableLoopUnrolling : OptLevel == 0;
379
380 // Check if vectorization is explicitly disabled via -vectorize-loops=false.
381 // The flag enables vectorization in the LoopVectorize pass, it is on by
382 // default, and if it was disabled, leave it disabled here.
383 // Another flag that exists: -loop-vectorize, controls adding the pass to the
384 // pass manager. If set, the pass is added, and there is no additional check
385 // here for it.
386 if (Builder.LoopVectorize)
387 Builder.LoopVectorize = OptLevel > 1 && SizeLevel < 2;
388
389 // When #pragma vectorize is on for SLP, do the same as above
390 Builder.SLPVectorize =
391 DisableSLPVectorization ? false : OptLevel > 1 && SizeLevel < 2;
392
393 if (TM)
394 TM->adjustPassManager(Builder);
395
396 if (Coroutines)
397 addCoroutinePassesToExtensionPoints(Builder);
398
399 switch (PGOKindFlag) {
400 case InstrGen:
401 Builder.EnablePGOInstrGen = true;
402 Builder.PGOInstrGen = ProfileFile;
403 break;
404 case InstrUse:
405 Builder.PGOInstrUse = ProfileFile;
406 break;
407 case SampleUse:
408 Builder.PGOSampleUse = ProfileFile;
409 break;
410 default:
411 break;
412 }
413
414 switch (CSPGOKindFlag) {
415 case CSInstrGen:
416 Builder.EnablePGOCSInstrGen = true;
417 break;
418 case CSInstrUse:
419 Builder.EnablePGOCSInstrUse = true;
420 break;
421 default:
422 break;
423 }
424
425 Builder.populateFunctionPassManager(FPM);
426 Builder.populateModulePassManager(MPM);
427}

429static void AddStandardLinkPasses(legacy::PassManagerBase &PM) {
430 PassManagerBuilder Builder;
431 Builder.VerifyInput = true;
432 if (DisableOptimizations)
433 Builder.OptLevel = 0;
434
435 if (!DisableInline)
436 Builder.Inliner = createFunctionInliningPass();
437 Builder.populateLTOPassManager(PM);
438}

444static CodeGenOpt::Level GetCodeGenOptLevel() {
445 if (CodeGenOptLevel.getNumOccurrences())
446 return static_cast<CodeGenOpt::Level>(unsigned(CodeGenOptLevel));
447 if (OptLevelO1)
448 return CodeGenOpt::Less;
449 if (OptLevelO2)
450 return CodeGenOpt::Default;
451 if (OptLevelO3)
452 return CodeGenOpt::Aggressive;
453 return CodeGenOpt::None;
454}

456// Returns the TargetMachine instance or zero if no triple is provided.
457static TargetMachine* GetTargetMachine(Triple TheTriple, StringRef CPUStr,
458 StringRef FeaturesStr,
459 const TargetOptions &Options) {
460 std::string Error;
461 const Target *TheTarget = TargetRegistry::lookupTarget(MArch, TheTriple,
462 Error);
463 // Some modules don't specify a triple, and this is okay.
464 if (!TheTarget) {
465 return nullptr;
466 }
467
468 return TheTarget->createTargetMachine(TheTriple.getTriple(), CPUStr,
469 FeaturesStr, Options, getRelocModel(),
470 getCodeModel(), GetCodeGenOptLevel());
471}

479//===----------------------------------------------------------------------===//
480// main for opt
481//
482int main(int argc, char **argv) {
483 InitLLVM X(argc, argv);
484
485 // Enable debug stream buffering.
486 EnableDebugBuffering = true;
487
488 LLVMContext Context;
489
490 InitializeAllTargets();
491 InitializeAllTargetMCs();
492 InitializeAllAsmPrinters();
493 InitializeAllAsmParsers();
494
495 // Initialize passes
496 PassRegistry &Registry = *PassRegistry::getPassRegistry();
497 initializeCore(Registry);
498 initializeCoroutines(Registry);
499 initializeScalarOpts(Registry);
500 initializeObjCARCOpts(Registry);
501 initializeVectorization(Registry);
502 initializeIPO(Registry);
503 initializeAnalysis(Registry);
504 initializeTransformUtils(Registry);
505 initializeInstCombine(Registry);
506 initializeAggressiveInstCombine(Registry);
507 initializeInstrumentation(Registry);
508 initializeTarget(Registry);
509 // For codegen passes, only passes that do IR to IR transformation are
510 // supported.
511 initializeExpandMemCmpPassPass(Registry);
512 initializeScalarizeMaskedMemIntrinPass(Registry);
513 initializeCodeGenPreparePass(Registry);
514 initializeAtomicExpandPass(Registry);
515 initializeRewriteSymbolsLegacyPassPass(Registry);
516 initializeWinEHPreparePass(Registry);
517 initializeDwarfEHPreparePass(Registry);
518 initializeSafeStackLegacyPassPass(Registry);
519 initializeSjLjEHPreparePass(Registry);
520 initializeStackProtectorPass(Registry);
521 initializePreISelIntrinsicLoweringLegacyPassPass(Registry);
522 initializeGlobalMergePass(Registry);
523 initializeIndirectBrExpandPassPass(Registry);
524 initializeInterleavedLoadCombinePass(Registry);
525 initializeInterleavedAccessPass(Registry);
526 initializeEntryExitInstrumenterPass(Registry);
527 initializePostInlineEntryExitInstrumenterPass(Registry);
528 initializeUnreachableBlockElimLegacyPassPass(Registry);
529 initializeExpandReductionsPass(Registry);
530 initializeWasmEHPreparePass(Registry);
531 initializeWriteBitcodePassPass(Registry);
532 initializeHardwareLoopsPass(Registry);
533
534 // Add the Cheri IR -> IR passes
535 initializeCHERICapFoldIntrinsicsPass(Registry);
536 initializeCHERICapDirectCallsPass(Registry);
537
538#ifdef LINK_POLLY_INTO_TOOLS
539 polly::initializePollyPasses(Registry);
540#endif
541
542 cl::ParseCommandLineOptions(argc, argv,
543 "llvm .bc -> .bc modular optimizer and analysis printer\n");
544
545 if (AnalyzeOnly && NoOutput) {
546 errs() << argv[0] << ": analyze mode conflicts with no-output mode.\n";
547 return 1;
548 }
549
550 SMDiagnostic Err;
551
552 Context.setDiscardValueNames(DiscardValueNames);
553 if (!DisableDITypeMap)
554 Context.enableDebugTypeODRUniquing();
555
556 if (PassRemarksWithHotness)
557 Context.setDiagnosticsHotnessRequested(true);
558
559 if (PassRemarksHotnessThreshold)
560 Context.setDiagnosticsHotnessThreshold(PassRemarksHotnessThreshold);
561
562 std::unique_ptr<ToolOutputFile> OptRemarkFile;
563 if (RemarksFilename != "") {
564 std::error_code EC;
565 OptRemarkFile =
566 llvm::make_unique<ToolOutputFile>(RemarksFilename, EC, sys::fs::F_None);
567 if (EC) {
568 errs() << EC.message() << '\n';
569 return 1;
570 }
571 Context.setRemarkStreamer(llvm::make_unique<RemarkStreamer>(
572 RemarksFilename,
573 llvm::make_unique<remarks::YAMLSerializer>(OptRemarkFile->os())));
574
575 if (!RemarksPasses.empty())
576 if (Error E = Context.getRemarkStreamer()->setFilter(RemarksPasses)) {
577 errs() << E << '\n';
578 return 1;
579 }
580 }
581
582 // Load the input module...
583 std::unique_ptr<Module> M =
584 parseIRFile(InputFilename, Err, Context, !NoVerify, ClDataLayout);
585
586 if (!M) {
587 Err.print(argv[0], errs());
588 return 1;
589 }
590
591 // Strip debug info before running the verifier.
592 if (StripDebug)
593 StripDebugInfo(*M);
594
595 // Erase module-level named metadata, if requested.
596 if (StripNamedMetadata) {
597 while (!M->named_metadata_empty()) {
598 NamedMDNode *NMD = &*M->named_metadata_begin();
599 M->eraseNamedMetadata(NMD);
600 }
601 }
602
603 // If we are supposed to override the target triple or data layout, do so now.
604 if (!TargetTriple.empty())
605 M->setTargetTriple(Triple::normalize(TargetTriple));
606
607 // Immediately run the verifier to catch any problems before starting up the
608 // pass pipelines. Otherwise we can crash on broken code during
609 // doInitialization().
610 if (!NoVerify && verifyModule(*M, &errs())) {
611 errs() << argv[0] << ": " << InputFilename
612 << ": error: input module is broken!\n";
613 return 1;
614 }
615
616 // Figure out what stream we are supposed to write to...
617 std::unique_ptr<ToolOutputFile> Out;
618 std::unique_ptr<ToolOutputFile> ThinLinkOut;
619 if (NoOutput) {
620 if (!OutputFilename.empty())
621 errs() << "WARNING: The -o (output filename) option is ignored when\n"
622 "the --disable-output option is used.\n";
623 } else {
624 // Default to standard output.
625 if (OutputFilename.empty())
626 OutputFilename = "-";
627
628 std::error_code EC;
629 Out.reset(new ToolOutputFile(OutputFilename, EC, sys::fs::F_None));
630 if (EC) {
631 errs() << EC.message() << '\n';
632 return 1;
633 }
634
635 if (!ThinLinkBitcodeFile.empty()) {
636 ThinLinkOut.reset(
637 new ToolOutputFile(ThinLinkBitcodeFile, EC, sys::fs::F_None));
638 if (EC) {
639 errs() << EC.message() << '\n';
640 return 1;
641 }
642 }
643 }
644
645 Triple ModuleTriple(M->getTargetTriple());
646 std::string CPUStr, FeaturesStr;
647 TargetMachine *Machine = nullptr;
648 const TargetOptions Options = InitTargetOptionsFromCodeGenFlags();
649
650 if (ModuleTriple.getArch()) {
651 CPUStr = getCPUStr();
652 FeaturesStr = getFeaturesStr();
653 Machine = GetTargetMachine(ModuleTriple, CPUStr, FeaturesStr, Options);
654 } else if (ModuleTriple.getArchName() != "unknown" &&
655 ModuleTriple.getArchName() != "") {
656 errs() << argv[0] << ": unrecognized architecture '"
657 << ModuleTriple.getArchName() << "' provided.\n";
658 return 1;
659 }
660
661 std::unique_ptr<TargetMachine> TM(Machine);
662
663 // Override function attributes based on CPUStr, FeaturesStr, and command line
664 // flags.
665 setFunctionAttributes(CPUStr, FeaturesStr, *M);
666
667 // If the output is set to be emitted to standard out, and standard out is a
668 // console, print out a warning message and refuse to do it. We don't
669 // impress anyone by spewing tons of binary goo to a terminal.
670 if (!Force && !NoOutput && !AnalyzeOnly && !OutputAssembly)
671 if (CheckBitcodeOutputToConsole(Out->os(), !Quiet))
672 NoOutput = true;
673
674 if (OutputThinLTOBC)
675 M->addModuleFlag(Module::Error, "EnableSplitLTOUnit", SplitLTOUnit);
676
677 if (PassPipeline.getNumOccurrences() > 0) {
678 OutputKind OK = OK_NoOutput;
679 if (!NoOutput)
680 OK = OutputAssembly
681 ? OK_OutputAssembly
682 : (OutputThinLTOBC ? OK_OutputThinLTOBitcode : OK_OutputBitcode);
683
684 VerifierKind VK = VK_VerifyInAndOut;
685 if (NoVerify)
686 VK = VK_NoVerifier;
687 else if (VerifyEach)
688 VK = VK_VerifyEachPass;
689
690 // The user has asked to use the new pass manager and provided a pipeline
691 // string. Hand off the rest of the functionality to the new code for that
692 // layer.
693 return runPassPipeline(argv[0], *M, TM.get(), Out.get(), ThinLinkOut.get(),
694 OptRemarkFile.get(), PassPipeline, OK, VK,
695 PreserveAssemblyUseListOrder,
696 PreserveBitcodeUseListOrder, EmitSummaryIndex,
697 EmitModuleHash, EnableDebugify)
698 ? 0
699 : 1;
700 }
701
702 // Create a PassManager to hold and optimize the collection of passes we are
703 // about to build.
704 OptCustomPassManager Passes;
705 bool AddOneTimeDebugifyPasses = EnableDebugify && !DebugifyEach;
706
707 // Add an appropriate TargetLibraryInfo pass for the module's triple.
708 TargetLibraryInfoImpl TLII(ModuleTriple);
709
710 // The -disable-simplify-libcalls flag actually disables all builtin optzns.
711 if (DisableSimplifyLibCalls)
712 TLII.disableAllFunctions();
713 Passes.add(new TargetLibraryInfoWrapperPass(TLII));
714
715 // Add internal analysis passes from the target machine.
716 Passes.add(createTargetTransformInfoWrapperPass(TM ? TM->getTargetIRAnalysis()
717 : TargetIRAnalysis()));
718
719 if (AddOneTimeDebugifyPasses)
720 Passes.add(createDebugifyModulePass());
721
722 std::unique_ptr<legacy::FunctionPassManager> FPasses;
723 if (OptLevelO0 || OptLevelO1 || OptLevelO2 || OptLevelOs || OptLevelOz ||
724 OptLevelO3) {
725 FPasses.reset(new legacy::FunctionPassManager(M.get()));
726 FPasses->add(createTargetTransformInfoWrapperPass(
727 TM ? TM->getTargetIRAnalysis() : TargetIRAnalysis()));
728 }
729
730 if (PrintBreakpoints) {
731 // Default to standard output.
732 if (!Out) {
733 if (OutputFilename.empty())
734 OutputFilename = "-";
735
736 std::error_code EC;
737 Out = llvm::make_unique<ToolOutputFile>(OutputFilename, EC,
738 sys::fs::F_None);
739 if (EC) {
740 errs() << EC.message() << '\n';
741 return 1;
742 }
743 }
744 Passes.add(createBreakpointPrinter(Out->os()));
745 NoOutput = true;
746 }
747
748 if (TM) {
749 // FIXME: We should dyn_cast this when supported.
750 auto &LTM = static_cast<LLVMTargetMachine &>(*TM);
751 Pass *TPC = LTM.createPassConfig(Passes);
752 Passes.add(TPC);
753 }
754
755 // Create a new optimization pass for each one specified on the command line
756 for (unsigned i = 0; i < PassList.size(); ++i) {
757 if (StandardLinkOpts &&
758 StandardLinkOpts.getPosition() < PassList.getPosition(i)) {
759 AddStandardLinkPasses(Passes);
760 StandardLinkOpts = false;
761 }
762
763 if (OptLevelO0 && OptLevelO0.getPosition() < PassList.getPosition(i)) {
764 AddOptimizationPasses(Passes, *FPasses, TM.get(), 0, 0);
765 OptLevelO0 = false;
766 }
767
768 if (OptLevelO1 && OptLevelO1.getPosition() < PassList.getPosition(i)) {
769 AddOptimizationPasses(Passes, *FPasses, TM.get(), 1, 0);
770 OptLevelO1 = false;
771 }
772
773 if (OptLevelO2 && OptLevelO2.getPosition() < PassList.getPosition(i)) {
774 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 0);
775 OptLevelO2 = false;
776 }
777
778 if (OptLevelOs && OptLevelOs.getPosition() < PassList.getPosition(i)) {
779 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 1);
780 OptLevelOs = false;
781 }
782
783 if (OptLevelOz && OptLevelOz.getPosition() < PassList.getPosition(i)) {
784 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 2);
785 OptLevelOz = false;
786 }
787
788 if (OptLevelO3 && OptLevelO3.getPosition() < PassList.getPosition(i)) {
789 AddOptimizationPasses(Passes, *FPasses, TM.get(), 3, 0);
790 OptLevelO3 = false;
791 }
792
793 const PassInfo *PassInf = PassList[i];
794 Pass *P = nullptr;
795 if (PassInf->getNormalCtor())
796 P = PassInf->getNormalCtor()();
797 else
798 errs() << argv[0] << ": cannot create pass: "
799 << PassInf->getPassName() << "\n";
800 if (P) {
801 PassKind Kind = P->getPassKind();
802 addPass(Passes, P);
803
804 if (AnalyzeOnly) {
805 switch (Kind) {
806 case PT_BasicBlock:
807 Passes.add(createBasicBlockPassPrinter(PassInf, Out->os(), Quiet));
808 break;
809 case PT_Region:
810 Passes.add(createRegionPassPrinter(PassInf, Out->os(), Quiet));
811 break;
812 case PT_Loop:
813 Passes.add(createLoopPassPrinter(PassInf, Out->os(), Quiet));
814 break;
815 case PT_Function:
816 Passes.add(createFunctionPassPrinter(PassInf, Out->os(), Quiet));
817 break;
818 case PT_CallGraphSCC:
819 Passes.add(createCallGraphPassPrinter(PassInf, Out->os(), Quiet));
820 break;
821 default:
822 Passes.add(createModulePassPrinter(PassInf, Out->os(), Quiet));
823 break;
824 }
825 }
826 }
827
828 if (PrintEachXForm)
829 Passes.add(
830 createPrintModulePass(errs(), "", PreserveAssemblyUseListOrder));
831 }
832
833 if (StandardLinkOpts) {
834 AddStandardLinkPasses(Passes);
835 StandardLinkOpts = false;
836 }
837
838 if (OptLevelO0)
839 AddOptimizationPasses(Passes, *FPasses, TM.get(), 0, 0);
840
841 if (OptLevelO1)
842 AddOptimizationPasses(Passes, *FPasses, TM.get(), 1, 0);
843
844 if (OptLevelO2)
845 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 0);
846
847 if (OptLevelOs)
848 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 1);
849
850 if (OptLevelOz)
851 AddOptimizationPasses(Passes, *FPasses, TM.get(), 2, 2);
852
853 if (OptLevelO3)
854 AddOptimizationPasses(Passes, *FPasses, TM.get(), 3, 0);
855
856 if (FPasses) {
857 FPasses->doInitialization();
858 for (Function &F : *M)
859 FPasses->run(F);
860 FPasses->doFinalization();
861 }
862
863 // Check that the module is well formed on completion of optimization
864 if (!NoVerify && !VerifyEach)
865 Passes.add(createVerifierPass());
866
867 if (AddOneTimeDebugifyPasses)
868 Passes.add(createCheckDebugifyModulePass(false));
869
870 // In run twice mode, we want to make sure the output is bit-by-bit
871 // equivalent if we run the pass manager again, so setup two buffers and
872 // a stream to write to them. Note that llc does something similar and it
873 // may be worth to abstract this out in the future.
874 SmallVector<char, 0> Buffer;
875 SmallVector<char, 0> FirstRunBuffer;
876 std::unique_ptr<raw_svector_ostream> BOS;
877 raw_ostream *OS = nullptr;
878
879 // Write bitcode or assembly to the output as the last step...
880 if (!NoOutput && !AnalyzeOnly) {
881 assert(Out);
882 OS = &Out->os();
883 if (RunTwice) {
884 BOS = make_unique<raw_svector_ostream>(Buffer);
885 OS = BOS.get();
886 }
887 if (OutputAssembly) {
888 if (EmitSummaryIndex)
889 report_fatal_error("Text output is incompatible with -module-summary");
890 if (EmitModuleHash)
891 report_fatal_error("Text output is incompatible with -module-hash");
892 Passes.add(createPrintModulePass(*OS, "", PreserveAssemblyUseListOrder));
893 } else if (OutputThinLTOBC)
894 Passes.add(createWriteThinLTOBitcodePass(
895 *OS, ThinLinkOut ? &ThinLinkOut->os() : nullptr));
896 else
897 Passes.add(createBitcodeWriterPass(*OS, PreserveBitcodeUseListOrder,
898 EmitSummaryIndex, EmitModuleHash));
899 }
900
901 // Before executing passes, print the final values of the LLVM options.
902 cl::PrintOptionValues();
903
904 if (!RunTwice) {
905 // Now that we have all of the passes ready, run them.
906 Passes.run(*M);
907 } else {
908 // If requested, run all passes twice with the same pass manager to catch
909 // bugs caused by persistent state in the passes.
910 std::unique_ptr<Module> M2(CloneModule(*M));
911 // Run all passes on the original module first, so the second run processes
912 // the clone to catch CloneModule bugs.
913 Passes.run(*M);
914 FirstRunBuffer = Buffer;
915 Buffer.clear();
916
917 Passes.run(*M2);
918
919 // Compare the two outputs and make sure they're the same
920 assert(Out);
921 if (Buffer.size() != FirstRunBuffer.size() ||
922 (memcmp(Buffer.data(), FirstRunBuffer.data(), Buffer.size()) != 0)) {
923 errs()
924 << "Running the pass manager twice changed the output.\n"
925 "Writing the result of the second run to the specified output.\n"
926 "To generate the one-run comparison binary, just run without\n"
927 "the compile-twice option\n";
928 Out->os() << BOS->str();
929 Out->keep();
930 if (OptRemarkFile)
931 OptRemarkFile->keep();
932 return 1;
933 }
934 Out->os() << BOS->str();
935 }
936
937 if (DebugifyEach && !DebugifyExport.empty())
938 exportDebugifyStats(DebugifyExport, Passes.getDebugifyStatsMap());
939
940 // Declare success.
941 if (!NoOutput || PrintBreakpoints)
942 Out->keep();
943
944 if (OptRemarkFile)
945 OptRemarkFile->keep();
946
947 if (ThinLinkOut)
948 ThinLinkOut->keep();
949
950 return 0;
951}
