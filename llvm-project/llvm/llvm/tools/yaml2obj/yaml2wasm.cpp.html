61static int writeUint64(raw_ostream &OS, uint64_t Value) {
62 char Data[sizeof(Value)];
63 support::endian::write64le(Data, Value);
64 OS.write(Data, sizeof(Data));
65 return 0;
66}
67
68static int writeUint32(raw_ostream &OS, uint32_t Value) {
69 char Data[sizeof(Value)];
70 support::endian::write32le(Data, Value);
71 OS.write(Data, sizeof(Data));
72 return 0;
73}
74
75static int writeUint8(raw_ostream &OS, uint8_t Value) {
76 char Data[sizeof(Value)];
77 memcpy(Data, &Value, sizeof(Data));
78 OS.write(Data, sizeof(Data));
79 return 0;
80}

82static int writeStringRef(const StringRef &Str, raw_ostream &OS) {
83 encodeULEB128(Str.size(), OS);
84 OS << Str;
85 return 0;
86}
87
88static int writeLimits(const WasmYAML::Limits &Lim, raw_ostream &OS) {
89 writeUint8(OS, Lim.Flags);
90 encodeULEB128(Lim.Initial, OS);
91 if (Lim.Flags & wasm::WASM_LIMITS_FLAG_HAS_MAX)
92 encodeULEB128(Lim.Maximum, OS);
93 return 0;
94}
95
96static int writeInitExpr(const wasm::WasmInitExpr &InitExpr, raw_ostream &OS) {
97 writeUint8(OS, InitExpr.Opcode);
98 switch (InitExpr.Opcode) {
99 case wasm::WASM_OPCODE_I32_CONST:
100 encodeSLEB128(InitExpr.Value.Int32, OS);
101 break;
102 case wasm::WASM_OPCODE_I64_CONST:
103 encodeSLEB128(InitExpr.Value.Int64, OS);
104 break;
105 case wasm::WASM_OPCODE_F32_CONST:
106 writeUint32(OS, InitExpr.Value.Float32);
107 break;
108 case wasm::WASM_OPCODE_F64_CONST:
109 writeUint64(OS, InitExpr.Value.Float64);
110 break;
111 case wasm::WASM_OPCODE_GLOBAL_GET:
112 encodeULEB128(InitExpr.Value.Global, OS);
113 break;
114 default:
115 errs() << "Unknown opcode in init_expr: " << InitExpr.Opcode << "\n";
116 return 1;
117 }
118 writeUint8(OS, wasm::WASM_OPCODE_END);
119 return 0;
120}

122class SubSectionWriter {
123 raw_ostream &OS;
124 std::string OutString;
125 raw_string_ostream StringStream;
126
127public:
128 SubSectionWriter(raw_ostream &OS) : OS(OS), StringStream(OutString) {}
129
130 void done() {
131 StringStream.flush();
132 encodeULEB128(OutString.size(), OS);
133 OS << OutString;
134 OutString.clear();
135 }
136
137 raw_ostream &getStream() { return StringStream; }
138};

140int WasmWriter::writeSectionContent(raw_ostream &OS,
141 WasmYAML::DylinkSection &Section) {
142 writeStringRef(Section.Name, OS);
143 encodeULEB128(Section.MemorySize, OS);
144 encodeULEB128(Section.MemoryAlignment, OS);
145 encodeULEB128(Section.TableSize, OS);
146 encodeULEB128(Section.TableAlignment, OS);
147 encodeULEB128(Section.Needed.size(), OS);
148 for (StringRef Needed : Section.Needed) {
149 writeStringRef(Needed, OS);
150 }
151 return 0;
152}

154int WasmWriter::writeSectionContent(raw_ostream &OS,
155 WasmYAML::LinkingSection &Section) {
156 writeStringRef(Section.Name, OS);
157 encodeULEB128(Section.Version, OS);
158
159 SubSectionWriter SubSection(OS);
160
161 // SYMBOL_TABLE subsection
162 if (Section.SymbolTable.size()) {
163 writeUint8(OS, wasm::WASM_SYMBOL_TABLE);
164
165 encodeULEB128(Section.SymbolTable.size(), SubSection.getStream());
166#ifndef NDEBUG
167 uint32_t SymbolIndex = 0;
168#endif
169 for (const WasmYAML::SymbolInfo &Info : Section.SymbolTable) {
170 assert(Info.Index == SymbolIndex++);
171 writeUint8(SubSection.getStream(), Info.Kind);
172 encodeULEB128(Info.Flags, SubSection.getStream());
173 switch (Info.Kind) {
174 case wasm::WASM_SYMBOL_TYPE_FUNCTION:
175 case wasm::WASM_SYMBOL_TYPE_GLOBAL:
176 case wasm::WASM_SYMBOL_TYPE_EVENT:
177 encodeULEB128(Info.ElementIndex, SubSection.getStream());
178 if ((Info.Flags & wasm::WASM_SYMBOL_UNDEFINED) == 0 ||
179 (Info.Flags & wasm::WASM_SYMBOL_EXPLICIT_NAME) != 0)
180 writeStringRef(Info.Name, SubSection.getStream());
181 break;
182 case wasm::WASM_SYMBOL_TYPE_DATA:
183 writeStringRef(Info.Name, SubSection.getStream());
184 if ((Info.Flags & wasm::WASM_SYMBOL_UNDEFINED) == 0) {
185 encodeULEB128(Info.DataRef.Segment, SubSection.getStream());
186 encodeULEB128(Info.DataRef.Offset, SubSection.getStream());
187 encodeULEB128(Info.DataRef.Size, SubSection.getStream());
188 }
189 break;
190 case wasm::WASM_SYMBOL_TYPE_SECTION:
191 encodeULEB128(Info.ElementIndex, SubSection.getStream());
192 break;
193 default:
194 llvm_unreachable("unexpected kind");
195 }
196 }
197
198 SubSection.done();
199 }
200
201 // SEGMENT_NAMES subsection
202 if (Section.SegmentInfos.size()) {
203 writeUint8(OS, wasm::WASM_SEGMENT_INFO);
204 encodeULEB128(Section.SegmentInfos.size(), SubSection.getStream());
205 for (const WasmYAML::SegmentInfo &SegmentInfo : Section.SegmentInfos) {
206 writeStringRef(SegmentInfo.Name, SubSection.getStream());
207 encodeULEB128(SegmentInfo.Alignment, SubSection.getStream());
208 encodeULEB128(SegmentInfo.Flags, SubSection.getStream());
209 }
210 SubSection.done();
211 }
212
213 // INIT_FUNCS subsection
214 if (Section.InitFunctions.size()) {
215 writeUint8(OS, wasm::WASM_INIT_FUNCS);
216 encodeULEB128(Section.InitFunctions.size(), SubSection.getStream());
217 for (const WasmYAML::InitFunction &Func : Section.InitFunctions) {
218 encodeULEB128(Func.Priority, SubSection.getStream());
219 encodeULEB128(Func.Symbol, SubSection.getStream());
220 }
221 SubSection.done();
222 }
223
224 // COMDAT_INFO subsection
225 if (Section.Comdats.size()) {
226 writeUint8(OS, wasm::WASM_COMDAT_INFO);
227 encodeULEB128(Section.Comdats.size(), SubSection.getStream());
228 for (const auto &C : Section.Comdats) {
229 writeStringRef(C.Name, SubSection.getStream());
230 encodeULEB128(0, SubSection.getStream()); // flags for future use
231 encodeULEB128(C.Entries.size(), SubSection.getStream());
232 for (const WasmYAML::ComdatEntry &Entry : C.Entries) {
233 writeUint8(SubSection.getStream(), Entry.Kind);
234 encodeULEB128(Entry.Index, SubSection.getStream());
235 }
236 }
237 SubSection.done();
238 }
239
240 return 0;
241}

243int WasmWriter::writeSectionContent(raw_ostream &OS,
244 WasmYAML::NameSection &Section) {
245 writeStringRef(Section.Name, OS);
246 if (Section.FunctionNames.size()) {
247 writeUint8(OS, wasm::WASM_NAMES_FUNCTION);
248
249 SubSectionWriter SubSection(OS);
250
251 encodeULEB128(Section.FunctionNames.size(), SubSection.getStream());
252 for (const WasmYAML::NameEntry &NameEntry : Section.FunctionNames) {
253 encodeULEB128(NameEntry.Index, SubSection.getStream());
254 writeStringRef(NameEntry.Name, SubSection.getStream());
255 }
256
257 SubSection.done();
258 }
259 return 0;
260}

262int WasmWriter::writeSectionContent(raw_ostream &OS,
263 WasmYAML::ProducersSection &Section) {
264 writeStringRef(Section.Name, OS);
265 int Fields = int(!Section.Languages.empty()) + int(!Section.Tools.empty()) +
266 int(!Section.SDKs.empty());
267 if (Fields == 0)
268 return 0;
269 encodeULEB128(Fields, OS);
270 for (auto &Field : {std::make_pair(StringRef("language"), &Section.Languages),
271 std::make_pair(StringRef("processed-by"), &Section.Tools),
272 std::make_pair(StringRef("sdk"), &Section.SDKs)}) {
273 if (Field.second->empty())
274 continue;
275 writeStringRef(Field.first, OS);
276 encodeULEB128(Field.second->size(), OS);
277 for (auto &Entry : *Field.second) {
278 writeStringRef(Entry.Name, OS);
279 writeStringRef(Entry.Version, OS);
280 }
281 }
282 return 0;
283}

296int WasmWriter::writeSectionContent(raw_ostream &OS,
297 WasmYAML::CustomSection &Section) {
298 if (auto S = dyn_cast<WasmYAML::DylinkSection>(&Section)) {
299 if (auto Err = writeSectionContent(OS, *S))
300 return Err;
301 } else if (auto S = dyn_cast<WasmYAML::NameSection>(&Section)) {
302 if (auto Err = writeSectionContent(OS, *S))
303 return Err;
304 } else if (auto S = dyn_cast<WasmYAML::LinkingSection>(&Section)) {
305 if (auto Err = writeSectionContent(OS, *S))
306 return Err;
307 } else if (auto S = dyn_cast<WasmYAML::ProducersSection>(&Section)) {
308 if (auto Err = writeSectionContent(OS, *S))
309 return Err;
310 } else if (auto S = dyn_cast<WasmYAML::TargetFeaturesSection>(&Section)) {
311 if (auto Err = writeSectionContent(OS, *S))
312 return Err;
313 } else {
314 writeStringRef(Section.Name, OS);
315 Section.Payload.writeAsBinary(OS);
316 }
317 return 0;
318}

320int WasmWriter::writeSectionContent(raw_ostream &OS,
321 WasmYAML::TypeSection &Section) {
322 encodeULEB128(Section.Signatures.size(), OS);
323 uint32_t ExpectedIndex = 0;
324 for (const WasmYAML::Signature &Sig : Section.Signatures) {
325 if (Sig.Index != ExpectedIndex) {
326 errs() << "Unexpected type index: " << Sig.Index << "\n";
327 return 1;
328 }
329 ++ExpectedIndex;
330 writeUint8(OS, Sig.Form);
331 encodeULEB128(Sig.ParamTypes.size(), OS);
332 for (auto ParamType : Sig.ParamTypes)
333 writeUint8(OS, ParamType);
334 if (Sig.ReturnType == wasm::WASM_TYPE_NORESULT) {
335 encodeULEB128(0, OS);
336 } else {
337 encodeULEB128(1, OS);
338 writeUint8(OS, Sig.ReturnType);
339 }
340 }
341 return 0;
342}

344int WasmWriter::writeSectionContent(raw_ostream &OS,
345 WasmYAML::ImportSection &Section) {
346 encodeULEB128(Section.Imports.size(), OS);
347 for (const WasmYAML::Import &Import : Section.Imports) {
348 writeStringRef(Import.Module, OS);
349 writeStringRef(Import.Field, OS);
350 writeUint8(OS, Import.Kind);
351 switch (Import.Kind) {
352 case wasm::WASM_EXTERNAL_FUNCTION:
353 encodeULEB128(Import.SigIndex, OS);
354 NumImportedFunctions++;
355 break;
356 case wasm::WASM_EXTERNAL_GLOBAL:
357 writeUint8(OS, Import.GlobalImport.Type);
358 writeUint8(OS, Import.GlobalImport.Mutable);
359 NumImportedGlobals++;
360 break;
361 case wasm::WASM_EXTERNAL_EVENT:
362 writeUint32(OS, Import.EventImport.Attribute);
363 writeUint32(OS, Import.EventImport.SigIndex);
364 NumImportedGlobals++;
365 break;
366 case wasm::WASM_EXTERNAL_MEMORY:
367 writeLimits(Import.Memory, OS);
368 break;
369 case wasm::WASM_EXTERNAL_TABLE:
370 writeUint8(OS, Import.TableImport.ElemType);
371 writeLimits(Import.TableImport.TableLimits, OS);
372 break;
373 default:
374 errs() << "Unknown import type: " << Import.Kind << "\n";
375 return 1;
376 }
377 }
378 return 0;
379}

474int WasmWriter::writeSectionContent(raw_ostream &OS,
475 WasmYAML::CodeSection &Section) {
476 encodeULEB128(Section.Functions.size(), OS);
477 uint32_t ExpectedIndex = NumImportedFunctions;
478 for (auto &Func : Section.Functions) {
479 std::string OutString;
480 raw_string_ostream StringStream(OutString);
481 if (Func.Index != ExpectedIndex) {
482 errs() << "Unexpected function index: " << Func.Index << "\n";
483 return 1;
484 }
485 ++ExpectedIndex;
486
487 encodeULEB128(Func.Locals.size(), StringStream);
488 for (auto &LocalDecl : Func.Locals) {
489 encodeULEB128(LocalDecl.Count, StringStream);
490 writeUint8(StringStream, LocalDecl.Type);
491 }
492
493 Func.Body.writeAsBinary(StringStream);
494
495 // Write the section size followed by the content
496 StringStream.flush();
497 encodeULEB128(OutString.size(), OS);
498 OS << OutString;
499 }
500 return 0;
501}

503int WasmWriter::writeSectionContent(raw_ostream &OS,
504 WasmYAML::DataSection &Section) {
505 encodeULEB128(Section.Segments.size(), OS);
506 for (auto &Segment : Section.Segments) {
507 encodeULEB128(Segment.InitFlags, OS);
508 if (Segment.InitFlags & wasm::WASM_SEGMENT_HAS_MEMINDEX)
509 encodeULEB128(Segment.MemoryIndex, OS);
510 if ((Segment.InitFlags & wasm::WASM_SEGMENT_IS_PASSIVE) == 0)
511 writeInitExpr(Segment.Offset, OS);
512 encodeULEB128(Segment.Content.binary_size(), OS);
513 Segment.Content.writeAsBinary(OS);
514 }
515 return 0;
516}

524int WasmWriter::writeRelocSection(raw_ostream &OS, WasmYAML::Section &Sec,
525 uint32_t SectionIndex) {
526 switch (Sec.Type) {
527 case wasm::WASM_SEC_CODE:
528 writeStringRef("reloc.CODE", OS);
529 break;
530 case wasm::WASM_SEC_DATA:
531 writeStringRef("reloc.DATA", OS);
532 break;
533 case wasm::WASM_SEC_CUSTOM: {
534 auto CustomSection = dyn_cast<WasmYAML::CustomSection>(&Sec);
535 writeStringRef(("reloc." + CustomSection->Name).str(), OS);
536 break;
537 }
538 default:
539 llvm_unreachable("not yet implemented");
540 return 1;
541 }
542
543 encodeULEB128(SectionIndex, OS);
544 encodeULEB128(Sec.Relocations.size(), OS);
545
546 for (auto Reloc : Sec.Relocations) {
547 writeUint8(OS, Reloc.Type);
548 encodeULEB128(Reloc.Offset, OS);
549 encodeULEB128(Reloc.Index, OS);
550 switch (Reloc.Type) {
551 case wasm::R_WASM_MEMORY_ADDR_LEB:
552 case wasm::R_WASM_MEMORY_ADDR_SLEB:
553 case wasm::R_WASM_MEMORY_ADDR_I32:
554 case wasm::R_WASM_FUNCTION_OFFSET_I32:
555 case wasm::R_WASM_SECTION_OFFSET_I32:
556 encodeULEB128(Reloc.Addend, OS);
557 }
558 }
559 return 0;
560}

562int WasmWriter::writeWasm(raw_ostream &OS) {
563 // Write headers
564 OS.write(wasm::WasmMagic, sizeof(wasm::WasmMagic));
565 writeUint32(OS, Obj.Header.Version);
566
567 // Write each section
568 llvm::object::WasmSectionOrderChecker Checker;
569 for (const std::unique_ptr<WasmYAML::Section> &Sec : Obj.Sections) {
570 StringRef SecName = "";
571 if (auto S = dyn_cast<WasmYAML::CustomSection>(Sec.get()))
572 SecName = S->Name;
573 if (!Checker.isValidSectionOrder(Sec->Type, SecName)) {
574 errs() << "Out of order section type: " << Sec->Type << "\n";
575 return 1;
576 }
577 encodeULEB128(Sec->Type, OS);
578 std::string OutString;
579 raw_string_ostream StringStream(OutString);
580 if (auto S = dyn_cast<WasmYAML::CustomSection>(Sec.get())) {
581 if (auto Err = writeSectionContent(StringStream, *S))
582 return Err;
583 } else if (auto S = dyn_cast<WasmYAML::TypeSection>(Sec.get())) {
584 if (auto Err = writeSectionContent(StringStream, *S))
585 return Err;
586 } else if (auto S = dyn_cast<WasmYAML::ImportSection>(Sec.get())) {
587 if (auto Err = writeSectionContent(StringStream, *S))
588 return Err;
589 } else if (auto S = dyn_cast<WasmYAML::FunctionSection>(Sec.get())) {
590 if (auto Err = writeSectionContent(StringStream, *S))
591 return Err;
592 } else if (auto S = dyn_cast<WasmYAML::TableSection>(Sec.get())) {
593 if (auto Err = writeSectionContent(StringStream, *S))
594 return Err;
595 } else if (auto S = dyn_cast<WasmYAML::MemorySection>(Sec.get())) {
596 if (auto Err = writeSectionContent(StringStream, *S))
597 return Err;
598 } else if (auto S = dyn_cast<WasmYAML::GlobalSection>(Sec.get())) {
599 if (auto Err = writeSectionContent(StringStream, *S))
600 return Err;
601 } else if (auto S = dyn_cast<WasmYAML::EventSection>(Sec.get())) {
602 if (auto Err = writeSectionContent(StringStream, *S))
603 return Err;
604 } else if (auto S = dyn_cast<WasmYAML::ExportSection>(Sec.get())) {
605 if (auto Err = writeSectionContent(StringStream, *S))
606 return Err;
607 } else if (auto S = dyn_cast<WasmYAML::StartSection>(Sec.get())) {
608 if (auto Err = writeSectionContent(StringStream, *S))
609 return Err;
610 } else if (auto S = dyn_cast<WasmYAML::ElemSection>(Sec.get())) {
611 if (auto Err = writeSectionContent(StringStream, *S))
612 return Err;
613 } else if (auto S = dyn_cast<WasmYAML::CodeSection>(Sec.get())) {
614 if (auto Err = writeSectionContent(StringStream, *S))
615 return Err;
616 } else if (auto S = dyn_cast<WasmYAML::DataSection>(Sec.get())) {
617 if (auto Err = writeSectionContent(StringStream, *S))
618 return Err;
619 } else if (auto S = dyn_cast<WasmYAML::DataCountSection>(Sec.get())) {
620 if (auto Err = writeSectionContent(StringStream, *S))
621 return Err;
622 } else {
623 errs() << "Unknown section type: " << Sec->Type << "\n";
624 return 1;
625 }
626 StringStream.flush();
627
628 // Write the section size followed by the content
629 encodeULEB128(OutString.size(), OS);
630 OS << OutString;
631 }
632
633 // write reloc sections for any section that have relocations
634 uint32_t SectionIndex = 0;
635 for (const std::unique_ptr<WasmYAML::Section> &Sec : Obj.Sections) {
636 if (Sec->Relocations.empty()) {
637 SectionIndex++;
638 continue;
639 }
640
641 writeUint8(OS, wasm::WASM_SEC_CUSTOM);
642 std::string OutString;
643 raw_string_ostream StringStream(OutString);
644 writeRelocSection(StringStream, *Sec, SectionIndex++);
645 StringStream.flush();
646
647 encodeULEB128(OutString.size(), OS);
648 OS << OutString;
649 }
650
651 return 0;
652}

654int yaml2wasm(llvm::WasmYAML::Object &Doc, raw_ostream &Out) {
655 WasmWriter Writer(Doc);
