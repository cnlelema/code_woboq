28// This class is used to build up a contiguous binary blob while keeping
29// track of an offset in the output (which notionally begins at
30// `InitialOffset`).
31namespace {
32class ContiguousBlobAccumulator {
33 const uint64_t InitialOffset;
34 SmallVector<char, 128> Buf;
35 raw_svector_ostream OS;
36
37 /// \returns The new offset.
38 uint64_t padToAlignment(unsigned Align) {
39 if (Align == 0)
40 Align = 1;
41 uint64_t CurrentOffset = InitialOffset + OS.tell();
42 uint64_t AlignedOffset = alignTo(CurrentOffset, Align);
43 OS.write_zeros(AlignedOffset - CurrentOffset);
44 return AlignedOffset; // == CurrentOffset;
45 }
46
47public:
48 ContiguousBlobAccumulator(uint64_t InitialOffset_)
49 : InitialOffset(InitialOffset_), Buf(), OS(Buf) {}
50 template <class Integer>
51 raw_ostream &getOSAndAlignedOffset(Integer &Offset, unsigned Align) {
52 Offset = padToAlignment(Align);
53 return OS;
54 }
55 void writeBlobToStream(raw_ostream &Out) { Out << OS.str(); }
56};
57} // end anonymous namespace

59// Used to keep track of section and symbol names, so that in the YAML file
60// sections and symbols can be referenced by name instead of by index.
61namespace {
62class NameToIdxMap {
63 StringMap<unsigned> Map;
64
65public:
66 /// \Returns false if name is already present in the map.
67 bool addName(StringRef Name, unsigned Ndx) {
68 return Map.insert({Name, Ndx}).second;
69 }
70 /// \Returns false if name is not present in the map.
71 bool lookup(StringRef Name, unsigned &Idx) const {
72 auto I = Map.find(Name);
73 if (I == Map.end())
74 return false;
75 Idx = I->getValue();
76 return true;
77 }
78 /// Asserts if name is not present in the map.
79 unsigned get(StringRef Name) const {
80 unsigned Idx;
81 if (lookup(Name, Idx))
82 return Idx;
83 assert(false && "Expected section not found in index");
84 return 0;
85 }
86 unsigned size() const { return Map.size(); }
87};
88} // end anonymous namespace

191template <class ELFT>
192void ELFState<ELFT>::initELFHeader(Elf_Ehdr &Header) {
193 using namespace llvm::ELF;
194 zero(Header);
195 Header.e_ident[EI_MAG0] = 0x7f;
196 Header.e_ident[EI_MAG1] = 'E';
197 Header.e_ident[EI_MAG2] = 'L';
198 Header.e_ident[EI_MAG3] = 'F';
199 Header.e_ident[EI_CLASS] = ELFT::Is64Bits ? ELFCLASS64 : ELFCLASS32;
200 Header.e_ident[EI_DATA] = Doc.Header.Data;
201 Header.e_ident[EI_VERSION] = EV_CURRENT;
202 Header.e_ident[EI_OSABI] = Doc.Header.OSABI;
203 Header.e_ident[EI_ABIVERSION] = Doc.Header.ABIVersion;
204 Header.e_type = Doc.Header.Type;
205 Header.e_machine = Doc.Header.Machine;
206 Header.e_version = EV_CURRENT;
207 Header.e_entry = Doc.Header.Entry;
208 Header.e_phoff = sizeof(Header);
209 Header.e_flags = Doc.Header.Flags;
210 Header.e_ehsize = sizeof(Elf_Ehdr);
211 Header.e_phentsize = sizeof(Elf_Phdr);
212 Header.e_phnum = Doc.ProgramHeaders.size();
213 Header.e_shentsize = sizeof(Elf_Shdr);
214 // Immediately following the ELF header and program headers.
215 Header.e_shoff =
216 sizeof(Header) + sizeof(Elf_Phdr) * Doc.ProgramHeaders.size();
217 Header.e_shnum = SN2I.size() + 1;
218 Header.e_shstrndx = SN2I.get(".shstrtab");
219}

221template <class ELFT>
222void ELFState<ELFT>::initProgramHeaders(std::vector<Elf_Phdr> &PHeaders) {
223 for (const auto &YamlPhdr : Doc.ProgramHeaders) {
224 Elf_Phdr Phdr;
225 Phdr.p_type = YamlPhdr.Type;
226 Phdr.p_flags = YamlPhdr.Flags;
227 Phdr.p_vaddr = YamlPhdr.VAddr;
228 Phdr.p_paddr = YamlPhdr.PAddr;
229 PHeaders.push_back(Phdr);
230 }
231}

233static bool convertSectionIndex(NameToIdxMap &SN2I, StringRef SecName,
234 StringRef IndexSrc, unsigned &IndexDest) {
235 if (!SN2I.lookup(IndexSrc, IndexDest) && !to_integer(IndexSrc, IndexDest)) {
236 WithColor::error() << "Unknown section referenced: '" << IndexSrc
237 << "' at YAML section '" << SecName << "'.\n";
238 return false;
239 }
240 return true;
241}

243template <class ELFT>
244bool ELFState<ELFT>::initImplicitHeader(ELFState<ELFT> &State,
245 ContiguousBlobAccumulator &CBA,
246 Elf_Shdr &Header, StringRef SecName,
247 ELFYAML::Section *YAMLSec) {
248 // Check if the header was already initialized.
249 if (Header.sh_offset)
250 return false;
251
252 if (SecName == ".symtab")
253 State.initSymtabSectionHeader(Header, SymtabType::Static, CBA, YAMLSec);
254 else if (SecName == ".strtab")
255 State.initStrtabSectionHeader(Header, SecName, State.DotStrtab, CBA,
256 YAMLSec);
257 else if (SecName == ".shstrtab")
258 State.initStrtabSectionHeader(Header, SecName, State.DotShStrtab, CBA,
259 YAMLSec);
260
261 else if (SecName == ".dynsym")
262 State.initSymtabSectionHeader(Header, SymtabType::Dynamic, CBA, YAMLSec);
263 else if (SecName == ".dynstr")
264 State.initStrtabSectionHeader(Header, SecName, State.DotDynstr, CBA,
265 YAMLSec);
266 else
267 return false;
268 return true;
269}

271template <class ELFT>
272bool ELFState<ELFT>::initSectionHeaders(ELFState<ELFT> &State,
273 std::vector<Elf_Shdr> &SHeaders,
274 ContiguousBlobAccumulator &CBA) {
275 // Build a list of sections we are going to add implicitly.
276 std::vector<StringRef> ImplicitSections;
277 for (StringRef Name : State.implicitSectionNames())
278 if (State.SN2I.get(Name) > Doc.Sections.size())
279 ImplicitSections.push_back(Name);
280
281 // Ensure SHN_UNDEF entry is present. An all-zero section header is a
282 // valid SHN_UNDEF entry since SHT_NULL == 0.
283 SHeaders.resize(Doc.Sections.size() + ImplicitSections.size() + 1);
284 zero(SHeaders[0]);
285
286 for (size_t I = 1; I < Doc.Sections.size() + ImplicitSections.size() + 1; ++I) {
287 Elf_Shdr &SHeader = SHeaders[I];
288 zero(SHeader);
289 ELFYAML::Section *Sec =
290 I > Doc.Sections.size() ? nullptr : Doc.Sections[I - 1].get();
291
292 // We have a few sections like string or symbol tables that are usually
293 // added implicitly to the end. However, if they are explicitly specified
294 // in the YAML, we need to write them here. This ensures the file offset
295 // remains correct.
296 StringRef SecName =
297 Sec ? Sec->Name : ImplicitSections[I - Doc.Sections.size() - 1];
298 if (initImplicitHeader(State, CBA, SHeader, SecName, Sec))
299 continue;
300
301 assert(Sec && "It can't be null unless it is an implicit section. But all "
302 "implicit sections should already have been handled above.");
303
304 SHeader.sh_name = DotShStrtab.getOffset(SecName);
305 SHeader.sh_type = Sec->Type;
306 SHeader.sh_flags = Sec->Flags;
307 SHeader.sh_addr = Sec->Address;
308 SHeader.sh_addralign = Sec->AddressAlign;
309
310 if (!Sec->Link.empty()) {
311 unsigned Index;
312 if (!convertSectionIndex(SN2I, Sec->Name, Sec->Link, Index))
313 return false;
314 SHeader.sh_link = Index;
315 }
316
317 if (auto S = dyn_cast<ELFYAML::RawContentSection>(Sec)) {
318 if (!writeSectionContent(SHeader, *S, CBA))
319 return false;
320 } else if (auto S = dyn_cast<ELFYAML::RelocationSection>(Sec)) {
321 if (!writeSectionContent(SHeader, *S, CBA))
322 return false;
323 } else if (auto S = dyn_cast<ELFYAML::Group>(Sec)) {
324 if (!writeSectionContent(SHeader, *S, CBA))
325 return false;
326 } else if (auto S = dyn_cast<ELFYAML::MipsABIFlags>(Sec)) {
327 if (!writeSectionContent(SHeader, *S, CBA))
328 return false;
329 } else if (auto S = dyn_cast<ELFYAML::NoBitsSection>(Sec)) {
330 SHeader.sh_entsize = 0;
331 SHeader.sh_size = S->Size;
332 // SHT_NOBITS section does not have content
333 // so just to setup the section offset.
334 CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
335 } else if (auto S = dyn_cast<ELFYAML::DynamicSection>(Sec)) {
336 if (!writeSectionContent(SHeader, *S, CBA))
337 return false;
338 } else if (auto S = dyn_cast<ELFYAML::SymverSection>(Sec)) {
339 if (!writeSectionContent(SHeader, *S, CBA))
340 return false;
341 } else if (auto S = dyn_cast<ELFYAML::VerneedSection>(Sec)) {
342 if (!writeSectionContent(SHeader, *S, CBA))
343 return false;
344 } else if (auto S = dyn_cast<ELFYAML::VerdefSection>(Sec)) {
345 if (!writeSectionContent(SHeader, *S, CBA))
346 return false;
347 } else
348 llvm_unreachable("Unknown section type");
349 }
350
351 return true;
352}

354static size_t findFirstNonGlobal(ArrayRef<ELFYAML::Symbol> Symbols) {
355 for (size_t I = 0; I < Symbols.size(); ++I)
356 if (Symbols[I].Binding.value != ELF::STB_LOCAL)
357 return I;
358 return Symbols.size();
359}
360
361static uint64_t writeRawSectionData(raw_ostream &OS,
362 const ELFYAML::RawContentSection &RawSec) {
363 size_t ContentSize = 0;
364 if (RawSec.Content) {
365 RawSec.Content->writeAsBinary(OS);
366 ContentSize = RawSec.Content->binary_size();
367 }
368
369 if (!RawSec.Size)
370 return ContentSize;
371
372 OS.write_zeros(*RawSec.Size - ContentSize);
373 return *RawSec.Size;
374}

376template <class ELFT>
377void ELFState<ELFT>::initSymtabSectionHeader(Elf_Shdr &SHeader,
378 SymtabType STType,
379 ContiguousBlobAccumulator &CBA,
380 ELFYAML::Section *YAMLSec) {
381
382 bool IsStatic = STType == SymtabType::Static;
383 const auto &Symbols = IsStatic ? Doc.Symbols : Doc.DynamicSymbols;
384
385 ELFYAML::RawContentSection *RawSec =
386 dyn_cast_or_null<ELFYAML::RawContentSection>(YAMLSec);
387 if (RawSec && !Symbols.empty() && (RawSec->Content || RawSec->Size)) {
388 if (RawSec->Content)
389 WithColor::error() << "Cannot specify both `Content` and " +
390 (IsStatic ? Twine("`Symbols`")
391 : Twine("`DynamicSymbols`")) +
392 " for symbol table section '"
393 << RawSec->Name << "'.\n";
394 if (RawSec->Size)
395 WithColor::error() << "Cannot specify both `Size` and " +
396 (IsStatic ? Twine("`Symbols`")
397 : Twine("`DynamicSymbols`")) +
398 " for symbol table section '"
399 << RawSec->Name << "'.\n";
400 exit(1);
401 }
402
403 zero(SHeader);
404 SHeader.sh_name = DotShStrtab.getOffset(IsStatic ? ".symtab" : ".dynsym");
405 SHeader.sh_type = IsStatic ? ELF::SHT_SYMTAB : ELF::SHT_DYNSYM;
406
407 if (RawSec && !RawSec->Link.empty()) {
408 // If the Link field is explicitly defined in the document,
409 // we should use it.
410 unsigned Index;
411 if (!convertSectionIndex(SN2I, RawSec->Name, RawSec->Link, Index))
412 return;
413 SHeader.sh_link = Index;
414 } else {
415 // When we describe the .dynsym section in the document explicitly, it is
416 // allowed to omit the "DynamicSymbols" tag. In this case .dynstr is not
417 // added implicitly and we should be able to leave the Link zeroed if
418 // .dynstr is not defined.
419 unsigned Link = 0;
420 if (IsStatic)
421 Link = SN2I.get(".strtab");
422 else
423 SN2I.lookup(".dynstr", Link);
424 SHeader.sh_link = Link;
425 }
426
427 if (!IsStatic)
428 SHeader.sh_flags |= ELF::SHF_ALLOC;
429
430 // If the symbol table section is explicitly described in the YAML
431 // then we should set the fields requested.
432 SHeader.sh_info =
433 RawSec ? (unsigned)RawSec->Info : findFirstNonGlobal(Symbols) + 1;
434 SHeader.sh_entsize = (YAMLSec && YAMLSec->EntSize)
435 ? (uint64_t)(*YAMLSec->EntSize)
436 : sizeof(Elf_Sym);
437 SHeader.sh_addralign = YAMLSec ? (uint64_t)YAMLSec->AddressAlign : 8;
438 SHeader.sh_addr = YAMLSec ? (uint64_t)YAMLSec->Address : 0;
439
440 auto &OS = CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
441 if (RawSec && (RawSec->Content || RawSec->Size)) {
442 assert(Symbols.empty());
443 SHeader.sh_size = writeRawSectionData(OS, *RawSec);
444 return;
445 }
446
447 std::vector<Elf_Sym> Syms;
448 {
449 // Ensure STN_UNDEF is present
450 Elf_Sym Sym;
451 zero(Sym);
452 Syms.push_back(Sym);
453 }
454
455 addSymbols(Symbols, Syms, IsStatic ? DotStrtab : DotDynstr);
456 writeArrayData(OS, makeArrayRef(Syms));
457 SHeader.sh_size = arrayDataSize(makeArrayRef(Syms));
458}

460template <class ELFT>
461void ELFState<ELFT>::initStrtabSectionHeader(Elf_Shdr &SHeader, StringRef Name,
462 StringTableBuilder &STB,
463 ContiguousBlobAccumulator &CBA,
464 ELFYAML::Section *YAMLSec) {
465 zero(SHeader);
466 SHeader.sh_name = DotShStrtab.getOffset(Name);
467 SHeader.sh_type = ELF::SHT_STRTAB;
468 SHeader.sh_addralign = YAMLSec ? (uint64_t)YAMLSec->AddressAlign : 1;
469
470 ELFYAML::RawContentSection *RawSec =
471 dyn_cast_or_null<ELFYAML::RawContentSection>(YAMLSec);
472
473 auto &OS = CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
474 if (RawSec && (RawSec->Content || RawSec->Size)) {
475 SHeader.sh_size = writeRawSectionData(OS, *RawSec);
476 } else {
477 STB.write(OS);
478 SHeader.sh_size = STB.getSize();
479 }
480
481 if (YAMLSec && YAMLSec->EntSize)
482 SHeader.sh_entsize = *YAMLSec->EntSize;
483
484 // If .dynstr section is explicitly described in the YAML
485 // then we want to use its section address.
486 if (Name == ".dynstr") {
487 if (YAMLSec)
488 SHeader.sh_addr = YAMLSec->Address;
489 // We assume that .dynstr is always allocatable.
490 SHeader.sh_flags |= ELF::SHF_ALLOC;
491 }
492}

494template <class ELFT>
495void ELFState<ELFT>::setProgramHeaderLayout(std::vector<Elf_Phdr> &PHeaders,
496 std::vector<Elf_Shdr> &SHeaders) {
497 uint32_t PhdrIdx = 0;
498 for (auto &YamlPhdr : Doc.ProgramHeaders) {
499 Elf_Phdr &PHeader = PHeaders[PhdrIdx++];
500
501 std::vector<Elf_Shdr *> Sections;
502 for (const ELFYAML::SectionName &SecName : YamlPhdr.Sections) {
503 unsigned Index;
504 if (!SN2I.lookup(SecName.Section, Index)) {
505 WithColor::error() << "Unknown section referenced: '" << SecName.Section
506 << "' by program header.\n";
507 exit(1);
508 }
509 Sections.push_back(&SHeaders[Index]);
510 }
511
512 if (YamlPhdr.Offset) {
513 PHeader.p_offset = *YamlPhdr.Offset;
514 } else {
515 if (YamlPhdr.Sections.size())
516 PHeader.p_offset = UINT32_MAX;
517 else
518 PHeader.p_offset = 0;
519
520 // Find the minimum offset for the program header.
521 for (Elf_Shdr *SHeader : Sections)
522 PHeader.p_offset = std::min(PHeader.p_offset, SHeader->sh_offset);
523 }
524
525 // Find the maximum offset of the end of a section in order to set p_filesz,
526 // if not set explicitly.
527 if (YamlPhdr.FileSize) {
528 PHeader.p_filesz = *YamlPhdr.FileSize;
529 } else {
530 PHeader.p_filesz = 0;
531 for (Elf_Shdr *SHeader : Sections) {
532 uint64_t EndOfSection;
533 if (SHeader->sh_type == llvm::ELF::SHT_NOBITS)
534 EndOfSection = SHeader->sh_offset;
535 else
536 EndOfSection = SHeader->sh_offset + SHeader->sh_size;
537 uint64_t EndOfSegment = PHeader.p_offset + PHeader.p_filesz;
538 EndOfSegment = std::max(EndOfSegment, EndOfSection);
539 PHeader.p_filesz = EndOfSegment - PHeader.p_offset;
540 }
541 }
542
543 // If not set explicitly, find the memory size by adding the size of
544 // sections at the end of the segment. These should be empty (size of zero)
545 // and NOBITS sections.
546 if (YamlPhdr.MemSize) {
547 PHeader.p_memsz = *YamlPhdr.MemSize;
548 } else {
549 PHeader.p_memsz = PHeader.p_filesz;
550 for (Elf_Shdr *SHeader : Sections)
551 if (SHeader->sh_offset == PHeader.p_offset + PHeader.p_filesz)
552 PHeader.p_memsz += SHeader->sh_size;
553 }
554
555 // Set the alignment of the segment to be the same as the maximum alignment
556 // of the sections with the same offset so that by default the segment
557 // has a valid and sensible alignment.
558 if (YamlPhdr.Align) {
559 PHeader.p_align = *YamlPhdr.Align;
560 } else {
561 PHeader.p_align = 1;
562 for (Elf_Shdr *SHeader : Sections)
563 if (SHeader->sh_offset == PHeader.p_offset)
564 PHeader.p_align = std::max(PHeader.p_align, SHeader->sh_addralign);
565 }
566 }
567}

569template <class ELFT>
570void ELFState<ELFT>::addSymbols(ArrayRef<ELFYAML::Symbol> Symbols,
571 std::vector<Elf_Sym> &Syms,
572 const StringTableBuilder &Strtab) {
573 for (const auto &Sym : Symbols) {
574 Elf_Sym Symbol;
575 zero(Symbol);
576
577 // If NameIndex, which contains the name offset, is explicitly specified, we
578 // use it. This is useful for preparing broken objects. Otherwise, we add
579 // the specified Name to the string table builder to get its offset.
580 if (Sym.NameIndex)
581 Symbol.st_name = *Sym.NameIndex;
582 else if (!Sym.Name.empty())
583 Symbol.st_name = Strtab.getOffset(Sym.Name);
584
585 Symbol.setBindingAndType(Sym.Binding, Sym.Type);
586 if (!Sym.Section.empty()) {
587 unsigned Index;
588 if (!SN2I.lookup(Sym.Section, Index)) {
589 WithColor::error() << "Unknown section referenced: '" << Sym.Section
590 << "' by YAML symbol " << Sym.Name << ".\n";
591 exit(1);
592 }
593 Symbol.st_shndx = Index;
594 } else if (Sym.Index) {
595 Symbol.st_shndx = *Sym.Index;
596 }
597 // else Symbol.st_shndex == SHN_UNDEF (== 0), since it was zero'd earlier.
598 Symbol.st_value = Sym.Value;
599 Symbol.st_other = Sym.Other;
600 Symbol.st_size = Sym.Size;
601 Syms.push_back(Symbol);
602 }
603}

605template <class ELFT>
606bool ELFState<ELFT>::writeSectionContent(
607 Elf_Shdr &SHeader, const ELFYAML::RawContentSection &Section,
608 ContiguousBlobAccumulator &CBA) {
609 raw_ostream &OS =
610 CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
611 SHeader.sh_size = writeRawSectionData(OS, Section);
612
613 if (Section.EntSize)
614 SHeader.sh_entsize = *Section.EntSize;
615 else if (Section.Type == llvm::ELF::SHT_RELR)
616 SHeader.sh_entsize = sizeof(Elf_Relr);
617 else
618 SHeader.sh_entsize = 0;
619 SHeader.sh_info = Section.Info;
620 return true;
621}

629template <class ELFT>
630bool
631ELFState<ELFT>::writeSectionContent(Elf_Shdr &SHeader,
632 const ELFYAML::RelocationSection &Section,
633 ContiguousBlobAccumulator &CBA) {
634 assert((Section.Type == llvm::ELF::SHT_REL ||
635 Section.Type == llvm::ELF::SHT_RELA) &&
636 "Section type is not SHT_REL nor SHT_RELA");
637
638 bool IsRela = Section.Type == llvm::ELF::SHT_RELA;
639 SHeader.sh_entsize = IsRela ? sizeof(Elf_Rela) : sizeof(Elf_Rel);
640 SHeader.sh_size = SHeader.sh_entsize * Section.Relocations.size();
641
642 // For relocation section set link to .symtab by default.
643 if (Section.Link.empty())
644 SHeader.sh_link = SN2I.get(".symtab");
645
646 unsigned Index = 0;
647 if (!Section.RelocatableSec.empty() &&
648 !convertSectionIndex(SN2I, Section.Name, Section.RelocatableSec, Index))
649 return false;
650 SHeader.sh_info = Index;
651
652 auto &OS = CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
653
654 for (const auto &Rel : Section.Relocations) {
655 unsigned SymIdx = 0;
656 // If a relocation references a symbol, try to look one up in the symbol
657 // table. If it is not there, treat the value as a symbol index.
658 if (Rel.Symbol && !SymN2I.lookup(*Rel.Symbol, SymIdx) &&
659 !to_integer(*Rel.Symbol, SymIdx)) {
660 WithColor::error() << "Unknown symbol referenced: '" << *Rel.Symbol
661 << "' at YAML section '" << Section.Name << "'.\n";
662 return false;
663 }
664
665 if (IsRela) {
666 Elf_Rela REntry;
667 zero(REntry);
668 REntry.r_offset = Rel.Offset;
669 REntry.r_addend = Rel.Addend;
670 REntry.setSymbolAndType(SymIdx, Rel.Type, isMips64EL(Doc));
671 OS.write((const char *)&REntry, sizeof(REntry));
672 } else {
673 Elf_Rel REntry;
674 zero(REntry);
675 REntry.r_offset = Rel.Offset;
676 REntry.setSymbolAndType(SymIdx, Rel.Type, isMips64EL(Doc));
677 OS.write((const char *)&REntry, sizeof(REntry));
678 }
679 }
680 return true;
681}

684bool ELFState<ELFT>::writeSectionContent(Elf_Shdr &SHeader,
685 const ELFYAML::Group &Section,
686 ContiguousBlobAccumulator &CBA) {
687 assert(Section.Type == llvm::ELF::SHT_GROUP &&
688 "Section type is not SHT_GROUP");
689
690 SHeader.sh_entsize = 4;
691 SHeader.sh_size = SHeader.sh_entsize * Section.Members.size();

693 unsigned SymIdx;
694 if (!SymN2I.lookup(Section.Signature, SymIdx) &&
695 !to_integer(Section.Signature, SymIdx)) {
696 WithColor::error() << "Unknown symbol referenced: '" << Section.Signature
697 << "' at YAML section '" << Section.Name << "'.\n";
698 return false;
699 }
700 SHeader.sh_info = SymIdx;
701
702 raw_ostream &OS =
703 CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
704
705 for (const ELFYAML::SectionOrType &Member : Section.Members) {
706 unsigned int SectionIndex = 0;
707 if (Member.sectionNameOrType == "GRP_COMDAT")
708 SectionIndex = llvm::ELF::GRP_COMDAT;
709 else if (!convertSectionIndex(SN2I, Section.Name, Member.sectionNameOrType,
710 SectionIndex))
711 return false;
712 support::endian::write<uint32_t>(OS, SectionIndex, ELFT::TargetEndianness);
713 }
714 return true;

731template <class ELFT>
732bool ELFState<ELFT>::writeSectionContent(Elf_Shdr &SHeader,
733 const ELFYAML::VerdefSection &Section,
734 ContiguousBlobAccumulator &CBA) {
735 typedef typename ELFT::Verdef Elf_Verdef;
736 typedef typename ELFT::Verdaux Elf_Verdaux;
737 raw_ostream &OS =
738 CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
739
740 uint64_t AuxCnt = 0;
741 for (size_t I = 0; I < Section.Entries.size(); ++I) {
742 const ELFYAML::VerdefEntry &E = Section.Entries[I];
743
744 Elf_Verdef VerDef;
745 VerDef.vd_version = E.Version;
746 VerDef.vd_flags = E.Flags;
747 VerDef.vd_ndx = E.VersionNdx;
748 VerDef.vd_hash = E.Hash;
749 VerDef.vd_aux = sizeof(Elf_Verdef);
750 VerDef.vd_cnt = E.VerNames.size();
751 if (I == Section.Entries.size() - 1)
752 VerDef.vd_next = 0;
753 else
754 VerDef.vd_next =
755 sizeof(Elf_Verdef) + E.VerNames.size() * sizeof(Elf_Verdaux);
756 OS.write((const char *)&VerDef, sizeof(Elf_Verdef));
757
758 for (size_t J = 0; J < E.VerNames.size(); ++J, ++AuxCnt) {
759 Elf_Verdaux VernAux;
760 VernAux.vda_name = DotDynstr.getOffset(E.VerNames[J]);
761 if (J == E.VerNames.size() - 1)
762 VernAux.vda_next = 0;
763 else
764 VernAux.vda_next = sizeof(Elf_Verdaux);
765 OS.write((const char *)&VernAux, sizeof(Elf_Verdaux));
766 }
767 }
768
769 SHeader.sh_size = Section.Entries.size() * sizeof(Elf_Verdef) +
770 AuxCnt * sizeof(Elf_Verdaux);
771 SHeader.sh_info = Section.Info;
772
773 return true;
774}

853template <class ELFT>
854bool ELFState<ELFT>::writeSectionContent(Elf_Shdr &SHeader,
855 const ELFYAML::DynamicSection &Section,
856 ContiguousBlobAccumulator &CBA) {
857 typedef typename ELFT::uint uintX_t;
858
859 assert(Section.Type == llvm::ELF::SHT_DYNAMIC &&
860 "Section type is not SHT_DYNAMIC");
861
862 if (!Section.Entries.empty() && Section.Content) {
863 WithColor::error()
864 << "Cannot specify both raw content and explicit entries "
865 "for dynamic section '"
866 << Section.Name << "'.\n";
867 return false;
868 }
869
870 if (Section.Content)
871 SHeader.sh_size = Section.Content->binary_size();
872 else
873 SHeader.sh_size = 2 * sizeof(uintX_t) * Section.Entries.size();
874 if (Section.EntSize)
875 SHeader.sh_entsize = *Section.EntSize;
876 else
877 SHeader.sh_entsize = sizeof(Elf_Dyn);
878
879 raw_ostream &OS = CBA.getOSAndAlignedOffset(SHeader.sh_offset, SHeader.sh_addralign);
880 for (const ELFYAML::DynamicEntry &DE : Section.Entries) {
881 support::endian::write<uintX_t>(OS, DE.Tag, ELFT::TargetEndianness);
882 support::endian::write<uintX_t>(OS, DE.Val, ELFT::TargetEndianness);
883 }
884 if (Section.Content)
885 Section.Content->writeAsBinary(OS);
886
887 return true;
888}

890template <class ELFT> bool ELFState<ELFT>::buildSectionIndex() {
891 for (unsigned i = 0, e = Doc.Sections.size(); i != e; ++i) {
892 StringRef Name = Doc.Sections[i]->Name;
893 DotShStrtab.add(Name);
894 // "+ 1" to take into account the SHT_NULL entry.
895 if (!SN2I.addName(Name, i + 1)) {
896 WithColor::error() << "Repeated section name: '" << Name
897 << "' at YAML section number " << i << ".\n";
898 return false;
899 }
900 }
901
902 auto SecNo = 1 + Doc.Sections.size();
903 // Add special sections after input sections, if necessary.
904 for (StringRef Name : implicitSectionNames())
905 if (SN2I.addName(Name, SecNo)) {
906 // Account for this section, since it wasn't in the Doc
907 ++SecNo;
908 DotShStrtab.add(Name);
909 }
910
911 DotShStrtab.finalize();
912 return true;
913}

915template <class ELFT>
916bool ELFState<ELFT>::buildSymbolIndex(ArrayRef<ELFYAML::Symbol> Symbols) {
917 bool GlobalSymbolSeen = false;
918 std::size_t I = 0;
919 for (const auto &Sym : Symbols) {
920 ++I;
921
922 StringRef Name = Sym.Name;
923 if (Sym.Binding.value == ELF::STB_LOCAL && GlobalSymbolSeen) {
924 WithColor::error() << "Local symbol '" + Name +
925 "' after global in Symbols list.\n";
926 return false;
927 }
928 if (Sym.Binding.value != ELF::STB_LOCAL)
929 GlobalSymbolSeen = true;
930
931 if (!Name.empty() && !SymN2I.addName(Name, I)) {
932 WithColor::error() << "Repeated symbol name: '" << Name << "'.\n";
933 return false;
934 }
935 }
936 return true;
937}

939template <class ELFT> void ELFState<ELFT>::finalizeStrings() {
940 // Add the regular symbol names to .strtab section.
941 for (const ELFYAML::Symbol &Sym : Doc.Symbols)
942 DotStrtab.add(Sym.Name);
943 DotStrtab.finalize();
944
945 // Add the dynamic symbol names to .dynstr section.
946 for (const ELFYAML::Symbol &Sym : Doc.DynamicSymbols)
947 DotDynstr.add(Sym.Name);
948
949 // SHT_GNU_verdef and SHT_GNU_verneed sections might also
950 // add strings to .dynstr section.
951 for (const std::unique_ptr<ELFYAML::Section> &Sec : Doc.Sections) {
952 if (auto VerNeed = dyn_cast<ELFYAML::VerneedSection>(Sec.get())) {
953 for (const ELFYAML::VerneedEntry &VE : VerNeed->VerneedV) {
954 DotDynstr.add(VE.File);
955 for (const ELFYAML::VernauxEntry &Aux : VE.AuxV)
956 DotDynstr.add(Aux.Name);
957 }
958 } else if (auto VerDef = dyn_cast<ELFYAML::VerdefSection>(Sec.get())) {
959 for (const ELFYAML::VerdefEntry &E : VerDef->Entries)
960 for (StringRef Name : E.VerNames)
961 DotDynstr.add(Name);
962 }
963 }
964
965 DotDynstr.finalize();
966}

968template <class ELFT>
969int ELFState<ELFT>::writeELF(raw_ostream &OS, const ELFYAML::Object &Doc) {
970 ELFState<ELFT> State(Doc);
971
972 // Finalize .strtab and .dynstr sections. We do that early because want to
973 // finalize the string table builders before writing the content of the
974 // sections that might want to use them.
975 State.finalizeStrings();
976
977 if (!State.buildSectionIndex())
978 return 1;
979
980 if (!State.buildSymbolIndex(Doc.Symbols))
981 return 1;
982
983 Elf_Ehdr Header;
984 State.initELFHeader(Header);
985
986 // TODO: Flesh out section header support.
987
988 std::vector<Elf_Phdr> PHeaders;
989 State.initProgramHeaders(PHeaders);
990
991 // XXX: This offset is tightly coupled with the order that we write
992 // things to `OS`.
993 const size_t SectionContentBeginOffset = Header.e_ehsize +
994 Header.e_phentsize * Header.e_phnum +
995 Header.e_shentsize * Header.e_shnum;
996 ContiguousBlobAccumulator CBA(SectionContentBeginOffset);
997
998 std::vector<Elf_Shdr> SHeaders;
999 if (!State.initSectionHeaders(State, SHeaders, CBA))
1000 return 1;
1001
1002 // Now we can decide segment offsets
1003 State.setProgramHeaderLayout(PHeaders, SHeaders);
1004
1005 OS.write((const char *)&Header, sizeof(Header));
1006 writeArrayData(OS, makeArrayRef(PHeaders));
1007 writeArrayData(OS, makeArrayRef(SHeaders));
1008 CBA.writeBlobToStream(OS);
1009 return 0;
1010}

1019int yaml2elf(llvm::ELFYAML::Object &Doc, raw_ostream &Out) {
1020 bool IsLE = Doc.Header.Data == ELFYAML::ELF_ELFDATA(ELF::ELFDATA2LSB);
1021 bool Is64Bit = Doc.Header.Class == ELFYAML::ELF_ELFCLASS(ELF::ELFCLASS64);
1022 if (Is64Bit) {
1023 if (IsLE)
1024 return ELFState<object::ELF64LE>::writeELF(Out, Doc);

1027 if (IsLE)
