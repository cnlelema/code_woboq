9#include "llvm/ADT/StringRef.h"
10#include "llvm/Support/Regex.h"
11#include "llvm/Support/YAMLTraits.h"
12#include <cassert>
13#include <string>
14
15llvm::Regex Infinity("^[-+]?(\\.inf|\\.Inf|\\.INF)$");
16llvm::Regex Base8("^0o[0-7]+$");
17llvm::Regex Base16("^0x[0-9a-fA-F]+$");
18llvm::Regex Float("^[-+]?(\\.[0-9]+|[0-9]+(\\.[0-9]*)?)([eE][-+]?[0-9]+)?$");
19
20inline bool isNumericRegex(llvm::StringRef S) {
21
22 if (S.equals(".nan") || S.equals(".NaN") || S.equals(".NAN"))
23 return true;
24
25 if (Infinity.match(S))
26 return true;
27
28 if (Base8.match(S))
29 return true;
30
31 if (Base16.match(S))
32 return true;
33
34 if (Float.match(S))
35 return true;
36
37 return false;
38}
39
40extern "C" int LLVMFuzzerTestOneInput(const uint8_t *Data, size_t Size) {
41 std::string Input(reinterpret_cast<const char *>(Data), Size);
42 Input.erase(std::remove(Input.begin(), Input.end(), 0), Input.end());
43 if (!Input.empty() && llvm::yaml::isNumeric(Input) != isNumericRegex(Input))
44 LLVM_BUILTIN_TRAP;
45 return 0;
46}
