9// This tool may be invoked in the following manner:
10// llvm-bcanalyzer [options] - Read LLVM bitcode from stdin
11// llvm-bcanalyzer [options] x.bc - Read LLVM bitcode from the x.bc file
12//
13// Options:
14// --help - Output information about command line switches
15// --dump - Dump low-level bitcode structure in readable format
16//
17// This tool provides analytical information about a bitcode file. It is
18// intended as an aid to developers of bitcode reading and writing software. It
19// produces on std::out a summary of the bitcode file that shows various
20// statistics about the contents of the file. By default this information is
21// detailed and contains information about individual bitcode blocks and the
22// functions in the module.
23// The tool is also able to print a bitcode file in a straight forward text
24// format that shows the containment and relationships of the information in
25// the bitcode file (-dump option).
26//
27//===----------------------------------------------------------------------===//

84/// GetBlockName - Return a symbolic block name if known, otherwise return
85/// null.
86static const char *GetBlockName(unsigned BlockID,
87 const BitstreamBlockInfo &BlockInfo,
88 CurStreamTypeType CurStreamType) {
89 // Standard blocks for all bitcode files.
90 if (BlockID < bitc::FIRST_APPLICATION_BLOCKID) {
91 if (BlockID == bitc::BLOCKINFO_BLOCK_ID)
92 return "BLOCKINFO_BLOCK";
93 return nullptr;
94 }
95
96 // Check to see if we have a blockinfo record for this block, with a name.
97 if (const BitstreamBlockInfo::BlockInfo *Info =
98 BlockInfo.getBlockInfo(BlockID)) {
99 if (!Info->Name.empty())
100 return Info->Name.c_str();
101 }
102
103
104 if (CurStreamType != LLVMIRBitstream) return nullptr;
105
106 switch (BlockID) {
107 default: return nullptr;
108 case bitc::OPERAND_BUNDLE_TAGS_BLOCK_ID: return "OPERAND_BUNDLE_TAGS_BLOCK";
109 case bitc::MODULE_BLOCK_ID: return "MODULE_BLOCK";
110 case bitc::PARAMATTR_BLOCK_ID: return "PARAMATTR_BLOCK";
111 case bitc::PARAMATTR_GROUP_BLOCK_ID: return "PARAMATTR_GROUP_BLOCK_ID";
112 case bitc::TYPE_BLOCK_ID_NEW: return "TYPE_BLOCK_ID";
113 case bitc::CONSTANTS_BLOCK_ID: return "CONSTANTS_BLOCK";
114 case bitc::FUNCTION_BLOCK_ID: return "FUNCTION_BLOCK";
115 case bitc::IDENTIFICATION_BLOCK_ID:
116 return "IDENTIFICATION_BLOCK_ID";
117 case bitc::VALUE_SYMTAB_BLOCK_ID: return "VALUE_SYMTAB";
118 case bitc::METADATA_BLOCK_ID: return "METADATA_BLOCK";
119 case bitc::METADATA_KIND_BLOCK_ID: return "METADATA_KIND_BLOCK";
120 case bitc::METADATA_ATTACHMENT_ID: return "METADATA_ATTACHMENT_BLOCK";
121 case bitc::USELIST_BLOCK_ID: return "USELIST_BLOCK_ID";
122 case bitc::GLOBALVAL_SUMMARY_BLOCK_ID:
123 return "GLOBALVAL_SUMMARY_BLOCK";
124 case bitc::FULL_LTO_GLOBALVAL_SUMMARY_BLOCK_ID:
125 return "FULL_LTO_GLOBALVAL_SUMMARY_BLOCK";
126 case bitc::MODULE_STRTAB_BLOCK_ID: return "MODULE_STRTAB_BLOCK";
127 case bitc::STRTAB_BLOCK_ID: return "STRTAB_BLOCK";
128 case bitc::SYMTAB_BLOCK_ID: return "SYMTAB_BLOCK";
129 }
130}

417struct PerRecordStats {
418 unsigned NumInstances;
419 unsigned NumAbbrev;
420 uint64_t TotalBits;
421
422 PerRecordStats() : NumInstances(0), NumAbbrev(0), TotalBits(0) {}
423};
424
425struct PerBlockIDStats {
426 /// NumInstances - This the number of times this block ID has been seen.
427 unsigned NumInstances;
428
429 /// NumBits - The total size in bits of all of these blocks.
430 uint64_t NumBits;
431
432 /// NumSubBlocks - The total number of blocks these blocks contain.
433 unsigned NumSubBlocks;
434
435 /// NumAbbrevs - The total number of abbreviations.
436 unsigned NumAbbrevs;
437
438 /// NumRecords - The total number of records these blocks contain, and the
439 /// number that are abbreviated.
440 unsigned NumRecords, NumAbbreviatedRecords;
441
442 /// CodeFreq - Keep track of the number of times we see each code.
443 std::vector<PerRecordStats> CodeFreq;
444
445 PerBlockIDStats()
446 : NumInstances(0), NumBits(0),
447 NumSubBlocks(0), NumAbbrevs(0), NumRecords(0), NumAbbreviatedRecords(0) {}
448};

454/// ReportError - All bitcode analysis errors go through this function, making this a
455/// good place to breakpoint if debugging.
456static bool ReportError(const Twine &Err) {
457 WithColor::error() << Err << "\n";
458 return true;
459}
460
461static bool decodeMetadataStringsBlob(StringRef Indent,
462 ArrayRef<uint64_t> Record,
463 StringRef Blob) {
464 if (Blob.empty())
465 return true;
466
467 if (Record.size() != 2)
468 return true;
469
470 unsigned NumStrings = Record[0];
471 unsigned StringsOffset = Record[1];
472 outs() << " num-strings = " << NumStrings << " {\n";
473
474 StringRef Lengths = Blob.slice(0, StringsOffset);
475 SimpleBitstreamCursor R(Lengths);
476 StringRef Strings = Blob.drop_front(StringsOffset);
477 do {
478 if (R.AtEndOfStream())
479 return ReportError("bad length");
480
481 unsigned Size = R.ReadVBR(6);
482 if (Strings.size() < Size)
483 return ReportError("truncated chars");
484
485 outs() << Indent << " '";
486 outs().write_escaped(Strings.slice(0, Size), /*hex=*/true);
487 outs() << "'\n";
488 Strings = Strings.drop_front(Size);
489 } while (--NumStrings);
490
491 outs() << Indent << " }";
492 return false;
493}
494
495static bool decodeBlob(unsigned Code, unsigned BlockID, StringRef Indent,
496 ArrayRef<uint64_t> Record, StringRef Blob) {
497 if (BlockID != bitc::METADATA_BLOCK_ID)
498 return true;
499 if (Code != bitc::METADATA_STRINGS)
500 return true;
501
502 return decodeMetadataStringsBlob(Indent, Record, Blob);
503}

505/// ParseBlock - Read a block, updating statistics, etc.
506static bool ParseBlock(BitstreamCursor &Stream, BitstreamBlockInfo &BlockInfo,
507 unsigned BlockID, unsigned IndentLevel,
508 CurStreamTypeType CurStreamType) {
509 std::string Indent(IndentLevel*2, ' ');
510 uint64_t BlockBitStart = Stream.GetCurrentBitNo();
511
512 // Get the statistics for this BlockID.
513 PerBlockIDStats &BlockStats = BlockIDStats[BlockID];
514
515 BlockStats.NumInstances++;
516
517 // BLOCKINFO is a special part of the stream.
518 bool DumpRecords = Dump;
519 if (BlockID == bitc::BLOCKINFO_BLOCK_ID) {
520 if (Dump) outs() << Indent << "<BLOCKINFO_BLOCK/>\n";
521 Optional<BitstreamBlockInfo> NewBlockInfo =
522 Stream.ReadBlockInfoBlock(/*ReadBlockInfoNames=*/true);
523 if (!NewBlockInfo)
524 return ReportError("Malformed BlockInfoBlock");
525 BlockInfo = std::move(*NewBlockInfo);
526 Stream.JumpToBit(BlockBitStart);
527 // It's not really interesting to dump the contents of the blockinfo block.
528 DumpRecords = false;
529 }
530
531 unsigned NumWords = 0;
532 if (Stream.EnterSubBlock(BlockID, &NumWords))
533 return ReportError("Malformed block record");
534
535 // Keep it for later, when we see a MODULE_HASH record
536 uint64_t BlockEntryPos = Stream.getCurrentByteNo();
537
538 const char *BlockName = nullptr;
539 if (DumpRecords) {
540 outs() << Indent << "<";
541 if ((BlockName = GetBlockName(BlockID, BlockInfo, CurStreamType)))
542 outs() << BlockName;
543 else
544 outs() << "UnknownBlock" << BlockID;
545
546 if (NonSymbolic && BlockName)
547 outs() << " BlockID=" << BlockID;
548
549 outs() << " NumWords=" << NumWords
550 << " BlockCodeSize=" << Stream.getAbbrevIDWidth() << ">\n";
551 }
552
553 SmallVector<uint64_t, 64> Record;
554
555 // Keep the offset to the metadata index if seen.
556 uint64_t MetadataIndexOffset = 0;
557
558 // Read all the records for this block.
559 while (1) {
560 if (Stream.AtEndOfStream())
561 return ReportError("Premature end of bitstream");
562
563 uint64_t RecordStartBit = Stream.GetCurrentBitNo();
564
565 BitstreamEntry Entry =
566 Stream.advance(BitstreamCursor::AF_DontAutoprocessAbbrevs);
567
568 switch (Entry.Kind) {
569 case BitstreamEntry::Error:
570 return ReportError("malformed bitcode file");
571 case BitstreamEntry::EndBlock: {
572 uint64_t BlockBitEnd = Stream.GetCurrentBitNo();
573 BlockStats.NumBits += BlockBitEnd-BlockBitStart;
574 if (DumpRecords) {
575 outs() << Indent << "</";
576 if (BlockName)
577 outs() << BlockName << ">\n";
578 else
579 outs() << "UnknownBlock" << BlockID << ">\n";
580 }
581 return false;
582 }
583
584 case BitstreamEntry::SubBlock: {
585 uint64_t SubBlockBitStart = Stream.GetCurrentBitNo();
586 if (ParseBlock(Stream, BlockInfo, Entry.ID, IndentLevel + 1,
587 CurStreamType))
588 return true;
589 ++BlockStats.NumSubBlocks;
590 uint64_t SubBlockBitEnd = Stream.GetCurrentBitNo();
591
592 // Don't include subblock sizes in the size of this block.
593 BlockBitStart += SubBlockBitEnd-SubBlockBitStart;
594 continue;
595 }
596 case BitstreamEntry::Record:
597 // The interesting case.
598 break;
599 }
600
601 if (Entry.ID == bitc::DEFINE_ABBREV) {
602 Stream.ReadAbbrevRecord();
603 ++BlockStats.NumAbbrevs;
604 continue;
605 }
606
607 Record.clear();
608
609 ++BlockStats.NumRecords;
610
611 StringRef Blob;
612 uint64_t CurrentRecordPos = Stream.GetCurrentBitNo();
613 unsigned Code = Stream.readRecord(Entry.ID, Record, &Blob);
614
615 // Increment the # occurrences of this code.
616 if (BlockStats.CodeFreq.size() <= Code)
617 BlockStats.CodeFreq.resize(Code+1);
618 BlockStats.CodeFreq[Code].NumInstances++;
619 BlockStats.CodeFreq[Code].TotalBits +=
620 Stream.GetCurrentBitNo()-RecordStartBit;
621 if (Entry.ID != bitc::UNABBREV_RECORD) {
622 BlockStats.CodeFreq[Code].NumAbbrev++;
623 ++BlockStats.NumAbbreviatedRecords;
624 }
625
626 if (DumpRecords) {
627 outs() << Indent << " <";
628 if (const char *CodeName =
629 GetCodeName(Code, BlockID, BlockInfo, CurStreamType))
630 outs() << CodeName;
631 else
632 outs() << "UnknownCode" << Code;
633 if (NonSymbolic && GetCodeName(Code, BlockID, BlockInfo, CurStreamType))
634 outs() << " codeid=" << Code;
635 const BitCodeAbbrev *Abbv = nullptr;
636 if (Entry.ID != bitc::UNABBREV_RECORD) {
637 Abbv = Stream.getAbbrev(Entry.ID);
638 outs() << " abbrevid=" << Entry.ID;
639 }
640
641 for (unsigned i = 0, e = Record.size(); i != e; ++i)
642 outs() << " op" << i << "=" << (int64_t)Record[i];
643
644 // If we found a metadata index, let's verify that we had an offset before
645 // and validate its forward reference offset was correct!
646 if (BlockID == bitc::METADATA_BLOCK_ID) {
647 if (Code == bitc::METADATA_INDEX_OFFSET) {
648 if (Record.size() != 2)
649 outs() << "(Invalid record)";
650 else {
651 auto Offset = Record[0] + (Record[1] << 32);
652 MetadataIndexOffset = Stream.GetCurrentBitNo() + Offset;
653 }
654 }
655 if (Code == bitc::METADATA_INDEX) {
656 outs() << " (offset ";
657 if (MetadataIndexOffset == RecordStartBit)
658 outs() << "match)";
659 else
660 outs() << "mismatch: " << MetadataIndexOffset << " vs "
661 << RecordStartBit << ")";
662 }
663 }
664
665 // If we found a module hash, let's verify that it matches!
666 if (BlockID == bitc::MODULE_BLOCK_ID && Code == bitc::MODULE_CODE_HASH &&
667 !CheckHash.empty()) {
668 if (Record.size() != 5)
669 outs() << " (invalid)";
670 else {
671 // Recompute the hash and compare it to the one in the bitcode
672 SHA1 Hasher;
673 StringRef Hash;
674 Hasher.update(CheckHash);
675 {
676 int BlockSize = (CurrentRecordPos / 8) - BlockEntryPos;
677 auto Ptr = Stream.getPointerToByte(BlockEntryPos, BlockSize);
678 Hasher.update(ArrayRef<uint8_t>(Ptr, BlockSize));
679 Hash = Hasher.result();
680 }
681 SmallString<20> RecordedHash;
682 RecordedHash.resize(20);
683 int Pos = 0;
684 for (auto &Val : Record) {
685 assert(!(Val >> 32) && "Unexpected high bits set");
686 RecordedHash[Pos++] = (Val >> 24) & 0xFF;
687 RecordedHash[Pos++] = (Val >> 16) & 0xFF;
688 RecordedHash[Pos++] = (Val >> 8) & 0xFF;
689 RecordedHash[Pos++] = (Val >> 0) & 0xFF;
690 }
691 if (Hash == RecordedHash)
692 outs() << " (match)";
693 else
694 outs() << " (!mismatch!)";
695 }
696 }
697
698 outs() << "/>";
699
700 if (Abbv) {
701 for (unsigned i = 1, e = Abbv->getNumOperandInfos(); i != e; ++i) {
702 const BitCodeAbbrevOp &Op = Abbv->getOperandInfo(i);
703 if (!Op.isEncoding() || Op.getEncoding() != BitCodeAbbrevOp::Array)
704 continue;
705 assert(i + 2 == e && "Array op not second to last");
706 std::string Str;
707 bool ArrayIsPrintable = true;
708 for (unsigned j = i - 1, je = Record.size(); j != je; ++j) {
709 if (!isPrint(static_cast<unsigned char>(Record[j]))) {
710 ArrayIsPrintable = false;
711 break;
712 }
713 Str += (char)Record[j];
714 }
715 if (ArrayIsPrintable)
716 outs() << " record string = '" << Str << "'";
717 break;
718 }
719 }
720
721 if (Blob.data() && decodeBlob(Code, BlockID, Indent, Record, Blob)) {
722 outs() << " blob data = ";
723 if (ShowBinaryBlobs) {
724 outs() << "'";
725 outs().write_escaped(Blob, /*hex=*/true) << "'";
726 } else {
727 bool BlobIsPrintable = true;
728 for (unsigned i = 0, e = Blob.size(); i != e; ++i)
729 if (!isPrint(static_cast<unsigned char>(Blob[i]))) {
730 BlobIsPrintable = false;
731 break;
732 }
733
734 if (BlobIsPrintable)
735 outs() << "'" << Blob << "'";
736 else
737 outs() << "unprintable, " << Blob.size() << " bytes.";
738 }
739 }
740
741 outs() << "\n";
742 }
743
744 // Make sure that we can skip the current record.
745 Stream.JumpToBit(CurrentRecordPos);
746 Stream.skipRecord(Entry.ID);
747 }
748}

758static CurStreamTypeType ReadSignature(BitstreamCursor &Stream) {
759 char Signature[6];
760 Signature[0] = Stream.Read(8);
761 Signature[1] = Stream.Read(8);
762
763 // Autodetect the file contents, if it is one we know.
764 if (Signature[0] == 'C' && Signature[1] == 'P') {
765 Signature[2] = Stream.Read(8);
766 Signature[3] = Stream.Read(8);
767 if (Signature[2] == 'C' && Signature[3] == 'H')
768 return ClangSerializedASTBitstream;
769 } else if (Signature[0] == 'D' && Signature[1] == 'I') {
770 Signature[2] = Stream.Read(8);
771 Signature[3] = Stream.Read(8);
772 if (Signature[2] == 'A' && Signature[3] == 'G')
773 return ClangSerializedDiagnosticsBitstream;
774 } else {
775 Signature[2] = Stream.Read(4);
776 Signature[3] = Stream.Read(4);
777 Signature[4] = Stream.Read(4);
778 Signature[5] = Stream.Read(4);
779 if (Signature[0] == 'B' && Signature[1] == 'C' &&
780 Signature[2] == 0x0 && Signature[3] == 0xC &&
781 Signature[4] == 0xE && Signature[5] == 0xD)
782 return LLVMIRBitstream;
783 }
784 return UnknownBitstream;
785}

787static bool openBitcodeFile(StringRef Path,
788 std::unique_ptr<MemoryBuffer> &MemBuf,
789 BitstreamCursor &Stream,
790 CurStreamTypeType &CurStreamType) {
791 // Read the input file.
792 ErrorOr<std::unique_ptr<MemoryBuffer>> MemBufOrErr =
793 MemoryBuffer::getFileOrSTDIN(Path);
794 if (std::error_code EC = MemBufOrErr.getError())
795 return ReportError(Twine("ReportError reading '") + Path + "': " + EC.message());
796 MemBuf = std::move(MemBufOrErr.get());
797
798 if (MemBuf->getBufferSize() & 3)
799 return ReportError("Bitcode stream should be a multiple of 4 bytes in length");
800
801 const unsigned char *BufPtr = (const unsigned char *)MemBuf->getBufferStart();
802 const unsigned char *EndBufPtr = BufPtr + MemBuf->getBufferSize();
803
804 // If we have a wrapper header, parse it and ignore the non-bc file contents.
805 // The magic number is 0x0B17C0DE stored in little endian.
806 if (isBitcodeWrapper(BufPtr, EndBufPtr)) {
807 if (MemBuf->getBufferSize() < BWH_HeaderSize)
808 return ReportError("Invalid bitcode wrapper header");
809
810 if (Dump) {
811 unsigned Magic = support::endian::read32le(&BufPtr[BWH_MagicField]);
812 unsigned Version = support::endian::read32le(&BufPtr[BWH_VersionField]);
813 unsigned Offset = support::endian::read32le(&BufPtr[BWH_OffsetField]);
814 unsigned Size = support::endian::read32le(&BufPtr[BWH_SizeField]);
815 unsigned CPUType = support::endian::read32le(&BufPtr[BWH_CPUTypeField]);
816
817 outs() << "<BITCODE_WRAPPER_HEADER"
818 << " Magic=" << format_hex(Magic, 10)
819 << " Version=" << format_hex(Version, 10)
820 << " Offset=" << format_hex(Offset, 10)
821 << " Size=" << format_hex(Size, 10)
822 << " CPUType=" << format_hex(CPUType, 10) << "/>\n";
823 }
824
825 if (SkipBitcodeWrapperHeader(BufPtr, EndBufPtr, true))
826 return ReportError("Invalid bitcode wrapper header");
827 }
828
829 Stream = BitstreamCursor(ArrayRef<uint8_t>(BufPtr, EndBufPtr));
830 CurStreamType = ReadSignature(Stream);
831
832 return false;
833}

835/// AnalyzeBitcode - Analyze the bitcode file specified by InputFilename.
836static int AnalyzeBitcode() {
837 std::unique_ptr<MemoryBuffer> StreamBuffer;
838 BitstreamCursor Stream;
839 BitstreamBlockInfo BlockInfo;
840 CurStreamTypeType CurStreamType;
841 if (openBitcodeFile(InputFilename, StreamBuffer, Stream, CurStreamType))
842 return true;
843 Stream.setBlockInfo(&BlockInfo);
844
845 // Read block info from BlockInfoFilename, if specified.
846 // The block info must be a top-level block.
847 if (!BlockInfoFilename.empty()) {
848 std::unique_ptr<MemoryBuffer> BlockInfoBuffer;
849 BitstreamCursor BlockInfoCursor;
850 CurStreamTypeType BlockInfoStreamType;
851 if (openBitcodeFile(BlockInfoFilename, BlockInfoBuffer, BlockInfoCursor,
852 BlockInfoStreamType))
853 return true;
854
855 while (!BlockInfoCursor.AtEndOfStream()) {
856 unsigned Code = BlockInfoCursor.ReadCode();
857 if (Code != bitc::ENTER_SUBBLOCK)
858 return ReportError("Invalid record at top-level in block info file");
859
860 unsigned BlockID = BlockInfoCursor.ReadSubBlockID();
861 if (BlockID == bitc::BLOCKINFO_BLOCK_ID) {
862 Optional<BitstreamBlockInfo> NewBlockInfo =
863 BlockInfoCursor.ReadBlockInfoBlock(/*ReadBlockInfoNames=*/true);
864 if (!NewBlockInfo)
865 return ReportError("Malformed BlockInfoBlock in block info file");
866 BlockInfo = std::move(*NewBlockInfo);
867 break;
868 }
869
870 BlockInfoCursor.SkipBlock();
871 }
872 }
873
874 unsigned NumTopBlocks = 0;
875
876 // Parse the top-level structure. We only allow blocks at the top-level.
877 while (!Stream.AtEndOfStream()) {
878 unsigned Code = Stream.ReadCode();
879 if (Code != bitc::ENTER_SUBBLOCK)
880 return ReportError("Invalid record at top-level");
881
882 unsigned BlockID = Stream.ReadSubBlockID();
883
884 if (ParseBlock(Stream, BlockInfo, BlockID, 0, CurStreamType))
885 return true;
886 ++NumTopBlocks;
887 }
888
889 if (Dump) outs() << "\n\n";
890
891 uint64_t BufferSizeBits = Stream.getBitcodeBytes().size() * CHAR_BIT;
892 // Print a summary of the read file.
893 outs() << "Summary of " << InputFilename << ":\n";
894 outs() << " Total size: ";
895 PrintSize(BufferSizeBits);
896 outs() << "\n";
897 outs() << " Stream type: ";
898 switch (CurStreamType) {
899 case UnknownBitstream:
900 outs() << "unknown\n";
901 break;
902 case LLVMIRBitstream:
903 outs() << "LLVM IR\n";
904 break;
905 case ClangSerializedASTBitstream:
906 outs() << "Clang Serialized AST\n";
907 break;
908 case ClangSerializedDiagnosticsBitstream:
909 outs() << "Clang Serialized Diagnostics\n";
910 break;
911 }
912 outs() << " # Toplevel Blocks: " << NumTopBlocks << "\n";
913 outs() << "\n";
914
915 // Emit per-block stats.
916 outs() << "Per-block Summary:\n";
917 for (std::map<unsigned, PerBlockIDStats>::iterator I = BlockIDStats.begin(),
918 E = BlockIDStats.end(); I != E; ++I) {
919 outs() << " Block ID #" << I->first;
920 if (const char *BlockName =
921 GetBlockName(I->first, BlockInfo, CurStreamType))
922 outs() << " (" << BlockName << ")";
923 outs() << ":\n";
924
925 const PerBlockIDStats &Stats = I->second;
926 outs() << " Num Instances: " << Stats.NumInstances << "\n";
927 outs() << " Total Size: ";
928 PrintSize(Stats.NumBits);
929 outs() << "\n";
930 double pct = (Stats.NumBits * 100.0) / BufferSizeBits;
931 outs() << " Percent of file: " << format("%2.4f%%", pct) << "\n";
932 if (Stats.NumInstances > 1) {
933 outs() << " Average Size: ";
934 PrintSize(Stats.NumBits/(double)Stats.NumInstances);
935 outs() << "\n";
936 outs() << " Tot/Avg SubBlocks: " << Stats.NumSubBlocks << "/"
937 << Stats.NumSubBlocks/(double)Stats.NumInstances << "\n";
938 outs() << " Tot/Avg Abbrevs: " << Stats.NumAbbrevs << "/"
939 << Stats.NumAbbrevs/(double)Stats.NumInstances << "\n";
940 outs() << " Tot/Avg Records: " << Stats.NumRecords << "/"
941 << Stats.NumRecords/(double)Stats.NumInstances << "\n";
942 } else {
943 outs() << " Num SubBlocks: " << Stats.NumSubBlocks << "\n";
944 outs() << " Num Abbrevs: " << Stats.NumAbbrevs << "\n";
945 outs() << " Num Records: " << Stats.NumRecords << "\n";
946 }
947 if (Stats.NumRecords) {
948 double pct = (Stats.NumAbbreviatedRecords * 100.0) / Stats.NumRecords;
949 outs() << " Percent Abbrevs: " << format("%2.4f%%", pct) << "\n";
950 }
951 outs() << "\n";
952
953 // Print a histogram of the codes we see.
954 if (!NoHistogram && !Stats.CodeFreq.empty()) {
955 std::vector<std::pair<unsigned, unsigned> > FreqPairs; // <freq,code>
956 for (unsigned i = 0, e = Stats.CodeFreq.size(); i != e; ++i)
957 if (unsigned Freq = Stats.CodeFreq[i].NumInstances)
958 FreqPairs.push_back(std::make_pair(Freq, i));
959 llvm::stable_sort(FreqPairs);
960 std::reverse(FreqPairs.begin(), FreqPairs.end());
961
962 outs() << "\tRecord Histogram:\n";
963 outs() << "\t\t Count # Bits b/Rec % Abv Record Kind\n";
964 for (unsigned i = 0, e = FreqPairs.size(); i != e; ++i) {
965 const PerRecordStats &RecStats = Stats.CodeFreq[FreqPairs[i].second];
966
967 outs() << format("\t\t%7d %9lu",
968 RecStats.NumInstances,
969 (unsigned long)RecStats.TotalBits);
970
971 if (RecStats.NumInstances > 1)
972 outs() << format(" %9.1f",
973 (double)RecStats.TotalBits/RecStats.NumInstances);
974 else
975 outs() << " ";
976
977 if (RecStats.NumAbbrev)
978 outs() <<
979 format(" %7.2f",
980 (double)RecStats.NumAbbrev/RecStats.NumInstances*100);
981 else
982 outs() << " ";
983
984 outs() << " ";
985 if (const char *CodeName = GetCodeName(FreqPairs[i].second, I->first,
986 BlockInfo, CurStreamType))
987 outs() << CodeName << "\n";
988 else
989 outs() << "UnknownCode" << FreqPairs[i].second << "\n";
990 }
991 outs() << "\n";
992
993 }
994 }
995 return 0;
996}
