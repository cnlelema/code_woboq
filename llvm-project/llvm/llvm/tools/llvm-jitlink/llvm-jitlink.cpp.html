140static uint64_t computeTotalAtomSizes(AtomGraph &G) {
141 uint64_t TotalSize = 0;
142 for (auto *DA : G.defined_atoms())
143 if (DA->isZeroFill())
144 TotalSize += DA->getZeroFillSize();
145 else
146 TotalSize += DA->getContent().size();
147 return TotalSize;
148}

150static void dumpSectionContents(raw_ostream &OS, AtomGraph &G) {
151 constexpr JITTargetAddress DumpWidth = 16;
152 static_assert(isPowerOf2_64(DumpWidth), "DumpWidth must be a power of two");
153
154 // Put sections in address order.
155 std::vector<Section *> Sections;
156 for (auto &S : G.sections())
157 Sections.push_back(&S);
158
159 std::sort(Sections.begin(), Sections.end(),
160 [](const Section *LHS, const Section *RHS) {
161 if (LHS->atoms_empty() && RHS->atoms_empty())
162 return false;
163 if (LHS->atoms_empty())
164 return false;
165 if (RHS->atoms_empty())
166 return true;
167 return (*LHS->atoms().begin())->getAddress() <
168 (*RHS->atoms().begin())->getAddress();
169 });
170
171 for (auto *S : Sections) {
172 OS << S->getName() << " content:";
173 if (S->atoms_empty()) {
174 OS << "\n section empty\n";
175 continue;
176 }
177
178 // Sort atoms into order, then render.
179 std::vector<DefinedAtom *> Atoms(S->atoms().begin(), S->atoms().end());
180 std::sort(Atoms.begin(), Atoms.end(),
181 [](const DefinedAtom *LHS, const DefinedAtom *RHS) {
182 return LHS->getAddress() < RHS->getAddress();
183 });
184
185 JITTargetAddress NextAddr = Atoms.front()->getAddress() & ~(DumpWidth - 1);
186 for (auto *DA : Atoms) {
187 bool IsZeroFill = DA->isZeroFill();
188 JITTargetAddress AtomStart = DA->getAddress();
189 JITTargetAddress AtomSize =
190 IsZeroFill ? DA->getZeroFillSize() : DA->getContent().size();
191 JITTargetAddress AtomEnd = AtomStart + AtomSize;
192 const uint8_t *AtomData =
193 IsZeroFill ? nullptr : DA->getContent().bytes_begin();
194
195 // Pad any space before the atom starts.
196 while (NextAddr != AtomStart) {
197 if (NextAddr % DumpWidth == 0)
198 OS << formatv("\n{0:x16}:", NextAddr);
199 OS << " ";
200 ++NextAddr;
201 }
202
203 // Render the atom content.
204 while (NextAddr != AtomEnd) {
205 if (NextAddr % DumpWidth == 0)
206 OS << formatv("\n{0:x16}:", NextAddr);
207 if (IsZeroFill)
208 OS << " 00";
209 else
210 OS << formatv(" {0:x-2}", AtomData[NextAddr - AtomStart]);
211 ++NextAddr;
212 }
213 }
214 OS << "\n";
215 }

244void Session::modifyPassConfig(const Triple &FTT,
245 PassConfiguration &PassConfig) {
246 if (!CheckFiles.empty())
247 PassConfig.PostFixupPasses.push_back([this](AtomGraph &G) {
248 if (TT.getObjectFormat() == Triple::MachO)
249 return registerMachOStubsAndGOT(*this, G);
250 return make_error<StringError>("Unsupported object format for GOT/stub "
251 "registration",
252 inconvertibleErrorCode());
253 });
254
255 if (ShowAtomGraph)
256 PassConfig.PostFixupPasses.push_back([](AtomGraph &G) -> Error {
257 outs() << "Atom graph post-fixup:\n";
258 G.dump(outs());
259 return Error::success();
260 });
261
262
263 if (ShowSizes) {
264 PassConfig.PrePrunePasses.push_back([this](AtomGraph &G) -> Error {
265 SizeBeforePruning += computeTotalAtomSizes(G);
266 return Error::success();
267 });
268 PassConfig.PostFixupPasses.push_back([this](AtomGraph &G) -> Error {
269 SizeAfterFixups += computeTotalAtomSizes(G);
270 return Error::success();
271 });
272 }
273
274 if (ShowRelocatedSectionContents)
275 PassConfig.PostFixupPasses.push_back([](AtomGraph &G) -> Error {
276 outs() << "Relocated section contents for " << G.getName() << ":\n";
277 dumpSectionContents(outs(), G);
278 return Error::success();
279 });
280}

371Error loadProcessSymbols(Session &S) {
372 std::string ErrMsg;
373 if (sys::DynamicLibrary::LoadLibraryPermanently(nullptr, &ErrMsg))
374 return make_error<StringError>(std::move(ErrMsg), inconvertibleErrorCode());
375
376 char GlobalPrefix = S.TT.getObjectFormat() == Triple::MachO ? '_' : '\0';
377 auto InternedEntryPointName = S.ES.intern(EntryPointName);
378 auto FilterMainEntryPoint = [InternedEntryPointName](SymbolStringPtr Name) {
379 return Name != InternedEntryPointName;
380 };
381 S.ES.getMainJITDylib().setGenerator(
382 ExitOnErr(orc::DynamicLibrarySearchGenerator::GetForCurrentProcess(
383 GlobalPrefix, FilterMainEntryPoint)));
384
385 return Error::success();
386}

402Error loadObjects(Session &S) {
403
404 std::map<unsigned, JITDylib *> IdxToJLD;
405
406 // First, set up JITDylibs.
407 LLVM_DEBUG(dbgs() << "Creating JITDylibs...\n");
408 {
409 // Create a "main" JITLinkDylib.
410 auto &MainJD = S.ES.getMainJITDylib();
411 IdxToJLD[0] = &MainJD;
412 S.JDSearchOrder.push_back(&MainJD);
413 LLVM_DEBUG(dbgs() << " 0: " << MainJD.getName() << "\n");
414
415 // Add any extra JITLinkDylibs from the command line.
416 std::string JDNamePrefix("lib");
417 for (auto JLDItr = JITLinkDylibs.begin(), JLDEnd = JITLinkDylibs.end();
418 JLDItr != JLDEnd; ++JLDItr) {
419 auto &JD = S.ES.createJITDylib(JDNamePrefix + *JLDItr);
420 unsigned JDIdx =
421 JITLinkDylibs.getPosition(JLDItr - JITLinkDylibs.begin());
422 IdxToJLD[JDIdx] = &JD;
423 S.JDSearchOrder.push_back(&JD);
424 LLVM_DEBUG(dbgs() << " " << JDIdx << ": " << JD.getName() << "\n");
425 }
426
427 // Set every dylib to link against every other, in command line order.
428 for (auto *JD : S.JDSearchOrder) {
429 JITDylibSearchList O;
430 for (auto *JD2 : S.JDSearchOrder) {
431 if (JD2 == JD)
432 continue;
433 O.push_back(std::make_pair(JD2, false));
434 }
435 JD->setSearchOrder(std::move(O));
436 }
437 }
438
439 // Load each object into the corresponding JITDylib..
440 LLVM_DEBUG(dbgs() << "Adding objects...\n");
441 for (auto InputFileItr = InputFiles.begin(), InputFileEnd = InputFiles.end();
442 InputFileItr != InputFileEnd; ++InputFileItr) {
443 unsigned InputFileArgIdx =
444 InputFiles.getPosition(InputFileItr - InputFiles.begin());
445 StringRef InputFile = *InputFileItr;
446 auto &JD = *std::prev(IdxToJLD.lower_bound(InputFileArgIdx))->second;
447 LLVM_DEBUG(dbgs() << " " << InputFileArgIdx << ": \"" << InputFile
448 << "\" to " << JD.getName() << "\n";);
449 auto ObjBuffer =
450 ExitOnErr(errorOrToExpected(MemoryBuffer::getFile(InputFile)));
451 ExitOnErr(S.ObjLayer.add(JD, std::move(ObjBuffer)));
452 }
453
454 // Define absolute symbols.
455 LLVM_DEBUG(dbgs() << "Defining absolute symbols...\n");
456 for (auto AbsDefItr = AbsoluteDefs.begin(), AbsDefEnd = AbsoluteDefs.end();
457 AbsDefItr != AbsDefEnd; ++AbsDefItr) {
458 unsigned AbsDefArgIdx =
459 AbsoluteDefs.getPosition(AbsDefItr - AbsoluteDefs.begin());
460 auto &JD = *std::prev(IdxToJLD.lower_bound(AbsDefArgIdx))->second;
461
462 StringRef AbsDefStmt = *AbsDefItr;
463 size_t EqIdx = AbsDefStmt.find_first_of('=');
464 if (EqIdx == StringRef::npos)
465 return make_error<StringError>("Invalid absolute define \"" + AbsDefStmt +
466 "\". Syntax: <name>=<addr>",
467 inconvertibleErrorCode());
468 StringRef Name = AbsDefStmt.substr(0, EqIdx).trim();
469 StringRef AddrStr = AbsDefStmt.substr(EqIdx + 1).trim();
470
471 uint64_t Addr;
472 if (AddrStr.getAsInteger(0, Addr))
473 return make_error<StringError>("Invalid address expression \"" + AddrStr +
474 "\" in absolute define \"" + AbsDefStmt +
475 "\"",
476 inconvertibleErrorCode());
477 JITEvaluatedSymbol AbsDef(Addr, JITSymbolFlags::Exported);
478 if (auto Err = JD.define(absoluteSymbols({{S.ES.intern(Name), AbsDef}})))
479 return Err;
480
481 // Register the absolute symbol with the session symbol infos.
482 S.SymbolInfos[Name] = { StringRef(), Addr };
483 }
484
485 LLVM_DEBUG({
486 dbgs() << "Dylib search order is [ ";
487 for (auto *JD : S.JDSearchOrder)
488 dbgs() << JD->getName() << " ";
489 dbgs() << "]\n";
490 });
491
492 return Error::success();
493}

495Error runChecks(Session &S) {
496
497 auto TripleName = S.TT.str();
498 std::string ErrorStr;
499 const Target *TheTarget = TargetRegistry::lookupTarget("", S.TT, ErrorStr);
500 if (!TheTarget)
501 ExitOnErr(make_error<StringError>("Error accessing target '" + TripleName +
502 "': " + ErrorStr,
503 inconvertibleErrorCode()));
504
505 std::unique_ptr<MCSubtargetInfo> STI(
506 TheTarget->createMCSubtargetInfo(TripleName, "", ""));
507 if (!STI)
508 ExitOnErr(
509 make_error<StringError>("Unable to create subtarget for " + TripleName,
510 inconvertibleErrorCode()));
511
512 std::unique_ptr<MCRegisterInfo> MRI(TheTarget->createMCRegInfo(TripleName));
513 if (!MRI)
514 ExitOnErr(make_error<StringError>("Unable to create target register info "
515 "for " +
516 TripleName,
517 inconvertibleErrorCode()));
518
519 std::unique_ptr<MCAsmInfo> MAI(TheTarget->createMCAsmInfo(*MRI, TripleName));
520 if (!MAI)
521 ExitOnErr(make_error<StringError>("Unable to create target asm info " +
522 TripleName,
523 inconvertibleErrorCode()));
524
525 MCContext Ctx(MAI.get(), MRI.get(), nullptr);
526
527 std::unique_ptr<MCDisassembler> Disassembler(
528 TheTarget->createMCDisassembler(*STI, Ctx));
529 if (!Disassembler)
530 ExitOnErr(make_error<StringError>("Unable to create disassembler for " +
531 TripleName,
532 inconvertibleErrorCode()));
533
534 std::unique_ptr<MCInstrInfo> MII(TheTarget->createMCInstrInfo());
535
536 std::unique_ptr<MCInstPrinter> InstPrinter(
537 TheTarget->createMCInstPrinter(Triple(TripleName), 0, *MAI, *MII, *MRI));
538
539 auto IsSymbolValid = [&S](StringRef Symbol) {
540 return S.isSymbolRegistered(Symbol);
541 };
542
543 auto GetSymbolInfo = [&S](StringRef Symbol) {
544 return S.findSymbolInfo(Symbol, "Can not get symbol info");
545 };
546
547 auto GetSectionInfo = [&S](StringRef FileName, StringRef SectionName) {
548 return S.findSectionInfo(FileName, SectionName);
549 };
550
551 auto GetStubInfo = [&S](StringRef FileName, StringRef SectionName) {
552 return S.findStubInfo(FileName, SectionName);
553 };
554
555 auto GetGOTInfo = [&S](StringRef FileName, StringRef SectionName) {
556 return S.findGOTEntryInfo(FileName, SectionName);
557 };
558
559 RuntimeDyldChecker Checker(
560 IsSymbolValid, GetSymbolInfo, GetSectionInfo, GetStubInfo, GetGOTInfo,
561 S.TT.isLittleEndian() ? support::little : support::big,
562 Disassembler.get(), InstPrinter.get(), dbgs());
563
564 for (auto &CheckFile : CheckFiles) {
565 auto CheckerFileBuf =
566 ExitOnErr(errorOrToExpected(MemoryBuffer::getFile(CheckFile)));
567 if (!Checker.checkAllRulesInBuffer("# jitlink-check:", &*CheckerFileBuf))
568 ExitOnErr(make_error<StringError>(
569 "Some checks in " + CheckFile + " failed", inconvertibleErrorCode()));
570 }
571
572 return Error::success();
573}

586Expected<int> runEntryPoint(Session &S, JITEvaluatedSymbol EntryPoint) {
587 assert(EntryPoint.getAddress() && "Entry point address should not be null");
588
589 constexpr const char *JITProgramName = "<llvm-jitlink jit'd code>";
590 auto PNStorage = llvm::make_unique<char[]>(strlen(JITProgramName) + 1);
591 strcpy(PNStorage.get(), JITProgramName);
592
593 std::vector<const char *> EntryPointArgs;
594 EntryPointArgs.push_back(PNStorage.get());
595 for (auto &InputArg : InputArgv)
596 EntryPointArgs.push_back(InputArg.data());
597 EntryPointArgs.push_back(nullptr);
598
599 using MainTy = int (*)(int, const char *[]);
600 MainTy EntryPointPtr = reinterpret_cast<MainTy>(EntryPoint.getAddress());
601
602 return EntryPointPtr(EntryPointArgs.size() - 1, EntryPointArgs.data());
603}
604
605int main(int argc, char *argv[]) {
606 InitLLVM X(argc, argv);
607
608 InitializeAllTargetInfos();
609 InitializeAllTargetMCs();
610 InitializeAllDisassemblers();
611
612 cl::ParseCommandLineOptions(argc, argv, "llvm jitlink tool");
613 ExitOnErr.setBanner(std::string(argv[0]) + ": ");
614
615 Session S(getFirstFileTriple());
616
617 ExitOnErr(sanitizeArguments(S));
618
619 if (!NoProcessSymbols)
620 ExitOnErr(loadProcessSymbols(S));
621 ExitOnErr(loadDylibs());
622
623 ExitOnErr(loadObjects(S));
624
625 auto EntryPoint = ExitOnErr(getMainEntryPoint(S));
626
627 if (ShowAddrs)
628 S.dumpSessionInfo(outs());
629
630 ExitOnErr(runChecks(S));
631
632 dumpSessionStats(S);
633
634 if (NoExec)
635 return 0;
636
637 return ExitOnErr(runEntryPoint(S, EntryPoint));
638}
