42// Mask for permissions (discardable, writable, readable, executable, etc).
43const uint32_t PermMask = 0xFE000000;
44
45// Mask for section types (code, data, bss).
46const uint32_t TypeMask = 0x000000E0;
47
48// The log base 2 of the largest section alignment, which is log2(8192), or 13.
49enum : unsigned { Log2MaxSectionAlignment = 13 };
50
51// A Chunk represents a chunk of data that will occupy space in the
52// output (if the resolver chose that). It may or may not be backed by
53// a section of an input file. It could be linker-created data, or
54// doesn't even have actual data (if common or bss).
55class Chunk {
56public:
57 enum Kind : uint8_t { SectionKind, OtherKind, ImportThunkKind };
58 Kind kind() const { return ChunkKind; }
59
60 // Returns the size of this chunk (even if this is a common or BSS.)
61 size_t getSize() const;
62
63 // Returns chunk alignment in power of two form. Value values are powers of
64 // two from 1 to 8192.
65 uint32_t getAlignment() const { return 1U << P2Align; }
66
67 // Update the chunk section alignment measured in bytes. Internally alignment
68 // is stored in log2.
69 void setAlignment(uint32_t Align) {
70 // Treat zero byte alignment as 1 byte alignment.
71 Align = Align ? Align : 1;
72 assert(llvm::isPowerOf2_32(Align) && "alignment is not a power of 2");
73 P2Align = llvm::Log2_32(Align);
74 assert(P2Align <= Log2MaxSectionAlignment &&
75 "impossible requested alignment");
76 }
77
78 // Write this chunk to a mmap'ed file, assuming Buf is pointing to
79 // beginning of the file. Because this function may use RVA values
80 // of other chunks for relocations, you need to set them properly
81 // before calling this function.
82 void writeTo(uint8_t *Buf) const;
83
84 // The writer sets and uses the addresses. In practice, PE images cannot be
85 // larger than 2GB. Chunks are always laid as part of the image, so Chunk RVAs
86 // can be stored with 32 bits.
87 uint32_t getRVA() const { return RVA; }
88 void setRVA(uint64_t V) {
89 RVA = (uint32_t)V;
90 assert(RVA == V && "RVA truncated");
91 }
92
93 // Returns true if this has non-zero data. BSS chunks return
94 // false. If false is returned, the space occupied by this chunk
95 // will be filled with zeros.
96 bool hasData() const { return HasData; }
97
98 // Returns readable/writable/executable bits.
99 uint32_t getOutputCharacteristics() const;
100
101 // Returns the section name if this is a section chunk.
102 // It is illegal to call this function on non-section chunks.
103 StringRef getSectionName() const;
104
105 // An output section has pointers to chunks in the section, and each
106 // chunk has a back pointer to an output section.
107 void setOutputSectionIdx(uint16_t O) { OSIdx = O; }
108 uint16_t getOutputSectionIdx() const { return OSIdx; }
109 OutputSection *getOutputSection() const;
110
111 // Windows-specific.
112 // Collect all locations that contain absolute addresses for base relocations.
113 void getBaserels(std::vector<Baserel> *Res);
114
115 // Returns a human-readable name of this chunk. Chunks are unnamed chunks of
116 // bytes, so this is used only for logging or debugging.
117 StringRef getDebugName() const;
118
119 // Return true if this file has the hotpatch flag set to true in the
120 // S_COMPILE3 record in codeview debug info. Also returns true for some thunks
121 // synthesized by the linker.
122 bool isHotPatchable() const;
123
124protected:
125 Chunk(Kind K = OtherKind) : ChunkKind(K), HasData(true), P2Align(0) {}
126
127 const Kind ChunkKind;
128
129 // True if the section has data. Corresponds to the
130 // IMAGE_SCN_CNT_UNINITIALIZED_DATA section characteristic bit.
131 uint8_t HasData : 1;
132
133 // The alignment of this chunk, stored in log2 form. The writer uses the
134 // value.
135 uint8_t P2Align : 7;
136
137 // The output section index for this chunk. The first valid section number is
138 // one.
139 uint16_t OSIdx = 0;
140
141 // The RVA of this chunk in the output. The writer sets a value.
142 uint32_t RVA = 0;
143};

145class NonSectionChunk : public Chunk {
146public:
147 virtual ~NonSectionChunk() = default;
148
149 // Returns the size of this chunk (even if this is a common or BSS.)
150 virtual size_t getSize() const = 0;
151
152 virtual uint32_t getOutputCharacteristics() const { return 0; }
153
154 // Write this chunk to a mmap'ed file, assuming Buf is pointing to
155 // beginning of the file. Because this function may use RVA values
156 // of other chunks for relocations, you need to set them properly
157 // before calling this function.
158 virtual void writeTo(uint8_t *Buf) const {}
159
160 // Returns the section name if this is a section chunk.
161 // It is illegal to call this function on non-section chunks.
162 virtual StringRef getSectionName() const {
163 llvm_unreachable("unimplemented getSectionName");
164 }
165
166 // Windows-specific.
167 // Collect all locations that contain absolute addresses for base relocations.
168 virtual void getBaserels(std::vector<Baserel> *Res) {}
169
170 // Returns a human-readable name of this chunk. Chunks are unnamed chunks of
171 // bytes, so this is used only for logging or debugging.
172 virtual StringRef getDebugName() const { return ""; }
173
174 static bool classof(const Chunk *C) { return C->kind() != SectionKind; }
175
176protected:
177 NonSectionChunk(Kind K = OtherKind) : Chunk(K) {}
178};

237 // True if this is a codeview debug info chunk. These will not be laid out in
238 // the image. Instead they will end up in the PDB, if one is requested.
239 bool isCodeView() const {
240 return getSectionName() == ".debug" || getSectionName().startswith(".debug$");

242
243 // True if this is a DWARF debug info or exception handling chunk.
244 bool isDWARF() const {
245 return getSectionName().startswith(".debug_") || getSectionName() == ".eh_frame";
246 }

248 // Allow iteration over the bodies of this chunk's relocated symbols.
249 llvm::iterator_range<symbol_iterator> symbols() const {
250 return llvm::make_range(symbol_iterator(File, RelocsData),
251 symbol_iterator(File, RelocsData + RelocsSize));
252 }
253
254 ArrayRef<coff_relocation> getRelocs() const {
255 return llvm::makeArrayRef(RelocsData, RelocsSize);
256 }

258 // Reloc setter used by ARM range extension thunk insertion.
259 void setRelocs(ArrayRef<coff_relocation> NewRelocs) {
260 RelocsData = NewRelocs.data();
261 RelocsSize = NewRelocs.size();
262 assert(RelocsSize == NewRelocs.size() && "reloc size truncation");

265 // Single linked list iterator for associated comdat children.
266 class AssociatedIterator
267 : public llvm::iterator_facade_base<
268 AssociatedIterator, std::forward_iterator_tag, SectionChunk> {
269 public:
270 AssociatedIterator() = default;
271 AssociatedIterator(SectionChunk *Head) : Cur(Head) {}
272 AssociatedIterator &operator=(const AssociatedIterator &R) {
273 Cur = R.Cur;
274 return *this;
275 }
276 bool operator==(const AssociatedIterator &R) const { return Cur == R.Cur; }
277 const SectionChunk &operator*() const { return *Cur; }
278 SectionChunk &operator*() { return *Cur; }
279 AssociatedIterator &operator++() {
280 Cur = Cur->AssocChildren;
281 return *this;
282 }
283
284 private:
285 SectionChunk *Cur = nullptr;
286 };
287
288 // Allow iteration over the associated child chunks for this section.
289 llvm::iterator_range<AssociatedIterator> children() const {
290 return llvm::make_range(AssociatedIterator(AssocChildren),
291 AssociatedIterator(nullptr));
292 }

334private:

396// This class is used to implement an lld-specific feature (not implemented in
397// MSVC) that minimizes the output size by finding string literals sharing tail
398// parts and merging them.
399//
400// If string tail merging is enabled and a section is identified as containing a
401// string literal, it is added to a MergeChunk with an appropriate alignment.
402// The MergeChunk then tail merges the strings using the StringTableBuilder
403// class and assigns RVAs and section offsets to each of the member chunks based
404// on the offsets assigned by the StringTableBuilder.
405class MergeChunk : public NonSectionChunk {
406public:
407 MergeChunk(uint32_t Alignment);
408 static void addSection(SectionChunk *C);
409 void finalizeContents();
410 void assignSubsectionRVAs();
411
412 uint32_t getOutputCharacteristics() const override;
413 StringRef getSectionName() const override { return ".rdata"; }
414 size_t getSize() const override;
415 void writeTo(uint8_t *Buf) const override;
416
417 static MergeChunk *Instances[Log2MaxSectionAlignment + 1];
418 std::vector<SectionChunk *> Sections;
419
420private:
421 llvm::StringTableBuilder Builder;
422 bool Finalized = false;
423};

540// Duplicate RVAs are not allowed in RVA tables, so unique symbols by chunk and
541// offset into the chunk. Order does not matter as the RVA table will be sorted
542// later.
543struct ChunkAndOffset {
544 Chunk *InputChunk;
545 uint32_t Offset;
546
547 struct DenseMapInfo {
548 static ChunkAndOffset getEmptyKey() {
549 return {llvm::DenseMapInfo<Chunk *>::getEmptyKey(), 0};
550 }
551 static ChunkAndOffset getTombstoneKey() {
552 return {llvm::DenseMapInfo<Chunk *>::getTombstoneKey(), 0};
553 }
554 static unsigned getHashValue(const ChunkAndOffset &CO) {
555 return llvm::DenseMapInfo<std::pair<Chunk *, uint32_t>>::getHashValue(
556 {CO.InputChunk, CO.Offset});
557 }
558 static bool isEqual(const ChunkAndOffset &LHS, const ChunkAndOffset &RHS) {
559 return LHS.InputChunk == RHS.InputChunk && LHS.Offset == RHS.Offset;
560 }
561 };
562};
