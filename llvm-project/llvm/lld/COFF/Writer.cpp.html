92public:
93 DebugDirectoryChunk(const std::vector<Chunk *> &R, bool WriteRepro)
94 : Records(R), WriteRepro(WriteRepro) {}
95
96 size_t getSize() const override {
97 return (Records.size() + int(WriteRepro)) * sizeof(debug_directory);
98 }
99
100 void writeTo(uint8_t *B) const override {
101 auto *D = reinterpret_cast<debug_directory *>(B);
102
103 for (const Chunk *Record : Records) {
104 OutputSection *OS = Record->getOutputSection();
105 uint64_t Offs = OS->getFileOff() + (Record->getRVA() - OS->getRVA());
106 fillEntry(D, COFF::IMAGE_DEBUG_TYPE_CODEVIEW, Record->getSize(),
107 Record->getRVA(), Offs);
108 ++D;
109 }
110
111 if (WriteRepro) {
112 // FIXME: The COFF spec allows either a 0-sized entry to just say
113 // "the timestamp field is really a hash", or a 4-byte size field
114 // followed by that many bytes containing a longer hash (with the
115 // lowest 4 bytes usually being the timestamp in little-endian order).
116 // Consider storing the full 8 bytes computed by xxHash64 here.
117 fillEntry(D, COFF::IMAGE_DEBUG_TYPE_REPRO, 0, 0, 0);
118 }
119 }

336// Check whether the target address S is in range from a relocation
337// of type RelType at address P.
338static bool isInRange(uint16_t RelType, uint64_t S, uint64_t P, int Margin) {
339 if (Config->Machine == ARMNT) {
340 int64_t Diff = AbsoluteDifference(S, P + 4) + Margin;
341 switch (RelType) {
342 case IMAGE_REL_ARM_BRANCH20T:
343 return isInt<21>(Diff);
344 case IMAGE_REL_ARM_BRANCH24T:
345 case IMAGE_REL_ARM_BLX23T:
346 return isInt<25>(Diff);
347 default:
348 return true;
349 }
350 } else if (Config->Machine == ARM64) {
351 int64_t Diff = AbsoluteDifference(S, P) + Margin;
352 switch (RelType) {
353 case IMAGE_REL_ARM64_BRANCH26:
354 return isInt<28>(Diff);
355 case IMAGE_REL_ARM64_BRANCH19:
356 return isInt<21>(Diff);
357 case IMAGE_REL_ARM64_BRANCH14:
358 return isInt<16>(Diff);
359 default:
360 return true;
361 }
362 } else {
363 llvm_unreachable("Unexpected architecture");
364 }
365}

367// Return the last thunk for the given target if it is in range,
368// or create a new one.
369static std::pair<Defined *, bool>
370getThunk(DenseMap<uint64_t, Defined *> &LastThunks, Defined *Target, uint64_t P,
371 uint16_t Type, int Margin) {
372 Defined *&LastThunk = LastThunks[Target->getRVA()];
373 if (LastThunk && isInRange(Type, LastThunk->getRVA(), P, Margin))
374 return {LastThunk, false};
375 Chunk *C;
376 switch (Config->Machine) {
377 case ARMNT:
378 C = make<RangeExtensionThunkARM>(Target);
379 break;
380 case ARM64:
381 C = make<RangeExtensionThunkARM64>(Target);
382 break;
383 default:
384 llvm_unreachable("Unexpected architecture");
385 }
386 Defined *D = make<DefinedSynthetic>("", C);
387 LastThunk = D;
388 return {D, true};
389}

391// This checks all relocations, and for any relocation which isn't in range
392// it adds a thunk after the section chunk that contains the relocation.
393// If the latest thunk for the specific target is in range, that is used
394// instead of creating a new thunk. All range checks are done with the
395// specified margin, to make sure that relocations that originally are in
396// range, but only barely, also get thunks - in case other added thunks makes
397// the target go out of range.
398//
399// After adding thunks, we verify that all relocations are in range (with
400// no extra margin requirements). If this failed, we restart (throwing away
401// the previously created thunks) and retry with a wider margin.
402static bool createThunks(OutputSection *OS, int Margin) {
403 bool AddressesChanged = false;
404 DenseMap<uint64_t, Defined *> LastThunks;
405 DenseMap<std::pair<ObjFile *, Defined *>, uint32_t> ThunkSymtabIndices;
406 size_t ThunksSize = 0;
407 // Recheck Chunks.size() each iteration, since we can insert more
408 // elements into it.
409 for (size_t I = 0; I != OS->Chunks.size(); ++I) {
410 SectionChunk *SC = dyn_cast_or_null<SectionChunk>(OS->Chunks[I]);
411 if (!SC)
412 continue;
413 size_t ThunkInsertionSpot = I + 1;
414
415 // Try to get a good enough estimate of where new thunks will be placed.
416 // Offset this by the size of the new thunks added so far, to make the
417 // estimate slightly better.
418 size_t ThunkInsertionRVA = SC->getRVA() + SC->getSize() + ThunksSize;
419 ObjFile *File = SC->File;
420 std::vector<std::pair<uint32_t, uint32_t>> RelocReplacements;
421 ArrayRef<coff_relocation> OriginalRelocs =
422 File->getCOFFObj()->getRelocations(SC->Header);
423 for (size_t J = 0, E = OriginalRelocs.size(); J < E; ++J) {
424 const coff_relocation &Rel = OriginalRelocs[J];
425 Symbol *RelocTarget = File->getSymbol(Rel.SymbolTableIndex);
426
427 // The estimate of the source address P should be pretty accurate,
428 // but we don't know whether the target Symbol address should be
429 // offset by ThunkSize or not (or by some of ThunksSize but not all of
430 // it), giving us some uncertainty once we have added one thunk.
431 uint64_t P = SC->getRVA() + Rel.VirtualAddress + ThunksSize;
432
433 Defined *Sym = dyn_cast_or_null<Defined>(RelocTarget);
434 if (!Sym)
435 continue;
436
437 uint64_t S = Sym->getRVA();
438
439 if (isInRange(Rel.Type, S, P, Margin))
440 continue;
441
442 // If the target isn't in range, hook it up to an existing or new
443 // thunk.
444 Defined *Thunk;
445 bool WasNew;
446 std::tie(Thunk, WasNew) = getThunk(LastThunks, Sym, P, Rel.Type, Margin);
447 if (WasNew) {
448 Chunk *ThunkChunk = Thunk->getChunk();
449 ThunkChunk->setRVA(
450 ThunkInsertionRVA); // Estimate of where it will be located.
451 OS->Chunks.insert(OS->Chunks.begin() + ThunkInsertionSpot, ThunkChunk);
452 ThunkInsertionSpot++;
453 ThunksSize += ThunkChunk->getSize();
454 ThunkInsertionRVA += ThunkChunk->getSize();
455 AddressesChanged = true;
456 }
457
458 // To redirect the relocation, add a symbol to the parent object file's
459 // symbol table, and replace the relocation symbol table index with the
460 // new index.
461 auto Insertion = ThunkSymtabIndices.insert({{File, Thunk}, ~0U});
462 uint32_t &ThunkSymbolIndex = Insertion.first->second;
463 if (Insertion.second)
464 ThunkSymbolIndex = File->addRangeThunkSymbol(Thunk);
465 RelocReplacements.push_back({J, ThunkSymbolIndex});
466 }
467
468 // Get a writable copy of this section's relocations so they can be
469 // modified. If the relocations point into the object file, allocate new
470 // memory. Otherwise, this must be previously allocated memory that can be
471 // modified in place.
472 ArrayRef<coff_relocation> CurRelocs = SC->getRelocs();
473 MutableArrayRef<coff_relocation> NewRelocs;
474 if (OriginalRelocs.data() == CurRelocs.data()) {
475 NewRelocs = makeMutableArrayRef(
476 BAlloc.Allocate<coff_relocation>(OriginalRelocs.size()),
477 OriginalRelocs.size());
478 } else {
479 NewRelocs = makeMutableArrayRef(
480 const_cast<coff_relocation *>(CurRelocs.data()), CurRelocs.size());
481 }
482
483 // Copy each relocation, but replace the symbol table indices which need
484 // thunks.
485 auto NextReplacement = RelocReplacements.begin();
486 auto EndReplacement = RelocReplacements.end();
487 for (size_t I = 0, E = OriginalRelocs.size(); I != E; ++I) {
488 NewRelocs[I] = OriginalRelocs[I];
489 if (NextReplacement != EndReplacement && NextReplacement->first == I) {
490 NewRelocs[I].SymbolTableIndex = NextReplacement->second;
491 ++NextReplacement;
492 }
493 }
494
495 SC->setRelocs(NewRelocs);
496 }
497 return AddressesChanged;
498}

500// Verify that all relocations are in range, with no extra margin requirements.
501static bool verifyRanges(const std::vector<Chunk *> Chunks) {
502 for (Chunk *C : Chunks) {
503 SectionChunk *SC = dyn_cast_or_null<SectionChunk>(C);
504 if (!SC)
505 continue;
506
507 ArrayRef<coff_relocation> Relocs = SC->getRelocs();
508 for (size_t J = 0, E = Relocs.size(); J < E; ++J) {
509 const coff_relocation &Rel = Relocs[J];
510 Symbol *RelocTarget = SC->File->getSymbol(Rel.SymbolTableIndex);
511
512 Defined *Sym = dyn_cast_or_null<Defined>(RelocTarget);
513 if (!Sym)
514 continue;
515
516 uint64_t P = SC->getRVA() + Rel.VirtualAddress;
517 uint64_t S = Sym->getRVA();
518
519 if (!isInRange(Rel.Type, S, P, 0))
520 return false;
521 }
522 }
523 return true;
524}

526// Assign addresses and add thunks if necessary.
527void Writer::finalizeAddresses() {
528 assignAddresses();
529 if (Config->Machine != ARMNT && Config->Machine != ARM64)
530 return;
531
532 size_t OrigNumChunks = 0;
533 for (OutputSection *Sec : OutputSections) {
534 Sec->OrigChunks = Sec->Chunks;
535 OrigNumChunks += Sec->Chunks.size();
536 }
537
538 int Pass = 0;
539 int Margin = 1024 * 100;
540 while (true) {
541 // First check whether we need thunks at all, or if the previous pass of
542 // adding them turned out ok.
543 bool RangesOk = true;
544 size_t NumChunks = 0;
545 for (OutputSection *Sec : OutputSections) {
546 if (!verifyRanges(Sec->Chunks)) {
547 RangesOk = false;
548 break;
549 }
550 NumChunks += Sec->Chunks.size();
551 }
552 if (RangesOk) {
553 if (Pass > 0)
554 log("Added " + Twine(NumChunks - OrigNumChunks) + " thunks with " +
555 "margin " + Twine(Margin) + " in " + Twine(Pass) + " passes");
556 return;
557 }
558
559 if (Pass >= 10)
560 fatal("adding thunks hasn't converged after " + Twine(Pass) + " passes");
561
562 if (Pass > 0) {
563 // If the previous pass didn't work out, reset everything back to the
564 // original conditions before retrying with a wider margin. This should
565 // ideally never happen under real circumstances.
566 for (OutputSection *Sec : OutputSections)
567 Sec->Chunks = Sec->OrigChunks;
568 Margin *= 2;
569 }
570
571 // Try adding thunks everywhere where it is needed, with a margin
572 // to avoid things going out of range due to the added thunks.
573 bool AddressesChanged = false;
574 for (OutputSection *Sec : OutputSections)
575 AddressesChanged |= createThunks(Sec, Margin);
576 // If the verification above thought we needed thunks, we should have
577 // added some.
578 assert(AddressesChanged);
579
580 // Recalculate the layout for the whole image (and verify the ranges at
581 // the start of the next round).
582 assignAddresses();
583
584 Pass++;
585 }

588// The main function of the writer.
589void Writer::run() {
590 ScopedTimer T1(CodeLayoutTimer);
591
592 createImportTables();
593 createSections();
594 createMiscChunks();
595 appendImportThunks();
596 createExportTable();
597 mergeSections();
598 removeUnusedSections();
599 finalizeAddresses();
600 removeEmptySections();
601 assignOutputSectionIndices();
602 setSectionPermissions();
603 createSymbolAndStringTable();
604
605 if (FileSize > UINT32_MAX)
606 fatal("image size (" + Twine(FileSize) + ") " +
607 "exceeds maximum allowable size (" + Twine(UINT32_MAX) + ")");
608
609 openFile(Config->OutputFile);
610 if (Config->is64()) {
611 writeHeader<pe32plus_header>();
612 } else {
613 writeHeader<pe32_header>();
614 }
615 writeSections();
616 sortExceptionTable();
617
618 T1.stop();
619
620 if (!Config->PDBPath.empty() && Config->Debug) {
621 assert(BuildId);
622 createPDB(Symtab, OutputSections, SectionTable, BuildId->BuildId);
623 }
624 writeBuildId();
625
626 writeMapFile(OutputSections);
627
628 ScopedTimer T2(DiskCommitTimer);
629 if (auto E = Buffer->commit())
630 fatal("failed to write the output file: " + toString(std::move(E)));
631}

655// Sort concrete section chunks from GNU import libraries.
656//
657// GNU binutils doesn't use short import files, but instead produces import
658// libraries that consist of object files, with section chunks for the .idata$*
659// sections. These are linked just as regular static libraries. Each import
660// library consists of one header object, one object file for every imported
661// symbol, and one trailer object. In order for the .idata tables/lists to
662// be formed correctly, the section chunks within each .idata$* section need
663// to be grouped by library, and sorted alphabetically within each library
664// (which makes sure the header comes first and the trailer last).
665bool Writer::fixGnuImportChunks() {
666 uint32_t RDATA = IMAGE_SCN_CNT_INITIALIZED_DATA | IMAGE_SCN_MEM_READ;
667
668 // Make sure all .idata$* section chunks are mapped as RDATA in order to
669 // be sorted into the same sections as our own synthesized .idata chunks.
670 for (auto It : PartialSections) {
671 PartialSection *PSec = It.second;
672 if (!PSec->Name.startswith(".idata"))
673 continue;
674 if (PSec->Characteristics == RDATA)
675 continue;
676 PartialSection *RDataSec = createPartialSection(PSec->Name, RDATA);
677 RDataSec->Chunks.insert(RDataSec->Chunks.end(), PSec->Chunks.begin(),
678 PSec->Chunks.end());
679 PSec->Chunks.clear();
680 }
681
682 bool HasIdata = false;
683 // Sort all .idata$* chunks, grouping chunks from the same library,
684 // with alphabetical ordering of the object fils within a library.
685 for (auto It : PartialSections) {
686 PartialSection *PSec = It.second;
687 if (!PSec->Name.startswith(".idata"))
688 continue;
689
690 if (!PSec->Chunks.empty())
691 HasIdata = true;
692 llvm::stable_sort(PSec->Chunks, [&](Chunk *S, Chunk *T) {
693 SectionChunk *SC1 = dyn_cast_or_null<SectionChunk>(S);
694 SectionChunk *SC2 = dyn_cast_or_null<SectionChunk>(T);
695 if (!SC1 || !SC2) {
696 // if SC1, order them ascending. If SC2 or both null,
697 // S is not less than T.
698 return SC1 != nullptr;
699 }
700 // Make a string with "libraryname/objectfile" for sorting, achieving
701 // both grouping by library and sorting of objects within a library,
702 // at once.
703 std::string Key1 =
704 (SC1->File->ParentName + "/" + SC1->File->getName()).str();
705 std::string Key2 =
706 (SC2->File->ParentName + "/" + SC2->File->getName()).str();
707 return Key1 < Key2;
708 });
709 }
710 return HasIdata;
711}

756// Create output section objects and add them to OutputSections.
757void Writer::createSections() {
758 // First, create the builtin sections.
759 const uint32_t DATA = IMAGE_SCN_CNT_INITIALIZED_DATA;
760 const uint32_t BSS = IMAGE_SCN_CNT_UNINITIALIZED_DATA;
761 const uint32_t CODE = IMAGE_SCN_CNT_CODE;
762 const uint32_t DISCARDABLE = IMAGE_SCN_MEM_DISCARDABLE;
763 const uint32_t R = IMAGE_SCN_MEM_READ;
764 const uint32_t W = IMAGE_SCN_MEM_WRITE;
765 const uint32_t X = IMAGE_SCN_MEM_EXECUTE;
766
767 SmallDenseMap<std::pair<StringRef, uint32_t>, OutputSection *> Sections;
768 auto CreateSection = [&](StringRef Name, uint32_t OutChars) {
769 OutputSection *&Sec = Sections[{Name, OutChars}];
770 if (!Sec) {
771 Sec = make<OutputSection>(Name, OutChars);
772 OutputSections.push_back(Sec);
773 }
774 return Sec;
775 };
776
777 // Try to match the section order used by link.exe.
778 TextSec = CreateSection(".text", CODE | R | X);
779 CreateSection(".bss", BSS | R | W);
780 RdataSec = CreateSection(".rdata", DATA | R);
781 BuildidSec = CreateSection(".buildid", DATA | R);
782 DataSec = CreateSection(".data", DATA | R | W);
783 PdataSec = CreateSection(".pdata", DATA | R);
784 IdataSec = CreateSection(".idata", DATA | R);
785 EdataSec = CreateSection(".edata", DATA | R);
786 DidatSec = CreateSection(".didat", DATA | R);
787 RsrcSec = CreateSection(".rsrc", DATA | R);
788 RelocSec = CreateSection(".reloc", DATA | DISCARDABLE | R);
789 CtorsSec = CreateSection(".ctors", DATA | R | W);
790 DtorsSec = CreateSection(".dtors", DATA | R | W);
791
792 // Then bin chunks by name and output characteristics.
793 for (Chunk *C : Symtab->getChunks()) {
794 auto *SC = dyn_cast<SectionChunk>(C);
795 if (SC && !SC->Live) {
796 if (Config->Verbose)
797 SC->printDiscardedMessage();
798 continue;
799 }
800 PartialSection *PSec = createPartialSection(C->getSectionName(),
801 C->getOutputCharacteristics());
802 PSec->Chunks.push_back(C);
803 }
804
805 // Even in non MinGW cases, we might need to link against GNU import
806 // libraries.
807 bool HasIdata = fixGnuImportChunks();
808 if (!Idata.empty())
809 HasIdata = true;
810
811 if (HasIdata)
812 addSyntheticIdata();
813
814 // Process an /order option.
815 if (!Config->Order.empty())
816 for (auto It : PartialSections)
817 sortBySectionOrder(It.second->Chunks);
818
819 if (HasIdata)
820 locateImportTables();
821
822 // Then create an OutputSection for each section.
823 // '$' and all following characters in input section names are
824 // discarded when determining output section. So, .text$foo
825 // contributes to .text, for example. See PE/COFF spec 3.2.
826 for (auto It : PartialSections) {
827 PartialSection *PSec = It.second;
828 StringRef Name = getOutputSectionName(PSec->Name);
829 uint32_t OutChars = PSec->Characteristics;
830
831 if (Name == ".CRT") {
832 // In link.exe, there is a special case for the I386 target where .CRT
833 // sections are treated as if they have output characteristics DATA | R if
834 // their characteristics are DATA | R | W. This implements the same
835 // special case for all architectures.
836 OutChars = DATA | R;
837
838 log("Processing section " + PSec->Name + " -> " + Name);
839
840 sortCRTSectionChunks(PSec->Chunks);
841 }
842
843 OutputSection *Sec = CreateSection(Name, OutChars);
844 for (Chunk *C : PSec->Chunks)
845 Sec->addChunk(C);
846
847 Sec->addContributingPartialSection(PSec);
848 }
849
850 // Finally, move some output sections to the end.
851 auto SectionOrder = [&](const OutputSection *S) {
852 // Move DISCARDABLE (or non-memory-mapped) sections to the end of file
853 // because the loader cannot handle holes. Stripping can remove other
854 // discardable ones than .reloc, which is first of them (created early).
855 if (S->Header.Characteristics & IMAGE_SCN_MEM_DISCARDABLE)
856 return 2;
857 // .rsrc should come at the end of the non-discardable sections because its
858 // size may change by the Win32 UpdateResources() function, causing
859 // subsequent sections to move (see https://crbug.com/827082).
860 if (S == RsrcSec)
861 return 1;
862 return 0;
863 };
864 llvm::stable_sort(OutputSections,
865 [&](const OutputSection *S, const OutputSection *T) {
866 return SectionOrder(S) < SectionOrder(T);
867 });
868}

1037Optional<coff_symbol16> Writer::createSymbol(Defined *Def) {
1038 coff_symbol16 Sym;
1039 switch (Def->kind()) {
1040 case Symbol::DefinedAbsoluteKind:
1041 Sym.Value = Def->getRVA();
1042 Sym.SectionNumber = IMAGE_SYM_ABSOLUTE;
1043 break;
1044 case Symbol::DefinedSyntheticKind:
1045 // Relative symbols are unrepresentable in a COFF symbol table.
1046 return None;
1047 default: {
1048 // Don't write symbols that won't be written to the output to the symbol
1049 // table.
1050 Chunk *C = Def->getChunk();
1051 if (!C)
1052 return None;
1053 OutputSection *OS = C->getOutputSection();
1054 if (!OS)
1055 return None;
1056
1057 Sym.Value = Def->getRVA() - OS->getRVA();
1058 Sym.SectionNumber = OS->SectionIndex;
1059 break;
1060 }
1061 }
1062
1063 StringRef Name = Def->getName();
1064 if (Name.size() > COFF::NameSize) {
1065 Sym.Name.Offset.Zeroes = 0;
1066 Sym.Name.Offset.Offset = addEntryToStringTable(Name);
1067 } else {
1068 memset(Sym.Name.ShortName, 0, COFF::NameSize);
1069 memcpy(Sym.Name.ShortName, Name.data(), Name.size());
1070 }
1071
1072 if (auto *D = dyn_cast<DefinedCOFF>(Def)) {
1073 COFFSymbolRef Ref = D->getCOFFSymbol();
1074 Sym.Type = Ref.getType();
1075 Sym.StorageClass = Ref.getStorageClass();
1076 } else {
1077 Sym.Type = IMAGE_SYM_TYPE_NULL;
1078 Sym.StorageClass = IMAGE_SYM_CLASS_EXTERNAL;
1079 }
1080 Sym.NumberOfAuxSymbols = 0;
1081 return Sym;
1082}

1084void Writer::createSymbolAndStringTable() {
1085 // PE/COFF images are limited to 8 byte section names. Longer names can be
1086 // supported by writing a non-standard string table, but this string table is
1087 // not mapped at runtime and the long names will therefore be inaccessible.
1088 // link.exe always truncates section names to 8 bytes, whereas binutils always
1089 // preserves long section names via the string table. LLD adopts a hybrid
1090 // solution where discardable sections have long names preserved and
1091 // non-discardable sections have their names truncated, to ensure that any
1092 // section which is mapped at runtime also has its name mapped at runtime.
1093 for (OutputSection *Sec : OutputSections) {
1094 if (Sec->Name.size() <= COFF::NameSize)
1095 continue;
1096 if ((Sec->Header.Characteristics & IMAGE_SCN_MEM_DISCARDABLE) == 0)
1097 continue;
1098 Sec->setStringTableOff(addEntryToStringTable(Sec->Name));
1099 }
1100
1101 if (Config->DebugDwarf || Config->DebugSymtab) {
1102 for (ObjFile *File : ObjFile::Instances) {
1103 for (Symbol *B : File->getSymbols()) {
1104 auto *D = dyn_cast_or_null<Defined>(B);
1105 if (!D || D->WrittenToSymtab)
1106 continue;
1107 D->WrittenToSymtab = true;
1108
1109 if (Optional<coff_symbol16> Sym = createSymbol(D))
1110 OutputSymtab.push_back(*Sym);
1111 }
1112 }
1113 }
1114
1115 if (OutputSymtab.empty() && Strtab.empty())
1116 return;
1117
1118 // We position the symbol table to be adjacent to the end of the last section.
1119 uint64_t FileOff = FileSize;
1120 PointerToSymbolTable = FileOff;
1121 FileOff += OutputSymtab.size() * sizeof(coff_symbol16);
1122 FileOff += 4 + Strtab.size();
1123 FileSize = alignTo(FileOff, Config->FileAlign);
1124}

1157// Visits all sections to assign incremental, non-overlapping RVAs and
1158// file offsets.
1159void Writer::assignAddresses() {
1160 SizeOfHeaders = DOSStubSize + sizeof(PEMagic) + sizeof(coff_file_header) +
1161 sizeof(data_directory) * NumberOfDataDirectory +
1162 sizeof(coff_section) * OutputSections.size();
1163 SizeOfHeaders +=
1164 Config->is64() ? sizeof(pe32plus_header) : sizeof(pe32_header);
1165 SizeOfHeaders = alignTo(SizeOfHeaders, Config->FileAlign);
1166 uint64_t RVA = PageSize; // The first page is kept unmapped.
1167 FileSize = SizeOfHeaders;
1168
1169 for (OutputSection *Sec : OutputSections) {
1170 if (Sec == RelocSec)
1171 addBaserels();
1172 uint64_t RawSize = 0, VirtualSize = 0;
1173 Sec->Header.VirtualAddress = RVA;
1174
1175 // If /FUNCTIONPADMIN is used, functions are padded in order to create a
1176 // hotpatchable image.
1177 const bool IsCodeSection =
1178 (Sec->Header.Characteristics & IMAGE_SCN_CNT_CODE) &&
1179 (Sec->Header.Characteristics & IMAGE_SCN_MEM_READ) &&
1180 (Sec->Header.Characteristics & IMAGE_SCN_MEM_EXECUTE);
1181 uint32_t Padding = IsCodeSection ? Config->FunctionPadMin : 0;
1182
1183 for (Chunk *C : Sec->Chunks) {
1184 if (Padding && C->isHotPatchable())
1185 VirtualSize += Padding;
1186 VirtualSize = alignTo(VirtualSize, C->getAlignment());
1187 C->setRVA(RVA + VirtualSize);
1188 VirtualSize += C->getSize();
1189 if (C->hasData())
1190 RawSize = alignTo(VirtualSize, Config->FileAlign);
1191 }
1192 if (VirtualSize > UINT32_MAX)
1193 error("section larger than 4 GiB: " + Sec->Name);
1194 Sec->Header.VirtualSize = VirtualSize;
1195 Sec->Header.SizeOfRawData = RawSize;
1196 if (RawSize != 0)
1197 Sec->Header.PointerToRawData = FileSize;
1198 RVA += alignTo(VirtualSize, PageSize);
1199 FileSize += alignTo(RawSize, Config->FileAlign);
1200 }
1201 SizeOfImage = alignTo(RVA, PageSize);
1202
1203 // Assign addresses to sections in MergeChunks.
1204 for (MergeChunk *MC : MergeChunk::Instances)
1205 if (MC)
1206 MC->assignSubsectionRVAs();
1207}

1209template <typename PEHeaderTy> void Writer::writeHeader() {
1210 // Write DOS header. For backwards compatibility, the first part of a PE/COFF
1211 // executable consists of an MS-DOS MZ executable. If the executable is run
1212 // under DOS, that program gets run (usually to just print an error message).
1213 // When run under Windows, the loader looks at AddressOfNewExeHeader and uses
1214 // the PE header instead.
1215 uint8_t *Buf = Buffer->getBufferStart();
1216 auto *DOS = reinterpret_cast<dos_header *>(Buf);
1217 Buf += sizeof(dos_header);
1218 DOS->Magic[0] = 'M';
1219 DOS->Magic[1] = 'Z';
1220 DOS->UsedBytesInTheLastPage = DOSStubSize % 512;
1221 DOS->FileSizeInPages = divideCeil(DOSStubSize, 512);
1222 DOS->HeaderSizeInParagraphs = sizeof(dos_header) / 16;
1223
1224 DOS->AddressOfRelocationTable = sizeof(dos_header);
1225 DOS->AddressOfNewExeHeader = DOSStubSize;
1226
1227 // Write DOS program.
1228 memcpy(Buf, DOSProgram, sizeof(DOSProgram));
1229 Buf += sizeof(DOSProgram);
1230
1231 // Write PE magic
1232 memcpy(Buf, PEMagic, sizeof(PEMagic));
1233 Buf += sizeof(PEMagic);
1234
1235 // Write COFF header
1236 auto *COFF = reinterpret_cast<coff_file_header *>(Buf);
1237 Buf += sizeof(*COFF);
1238 COFF->Machine = Config->Machine;
1239 COFF->NumberOfSections = OutputSections.size();
1240 COFF->Characteristics = IMAGE_FILE_EXECUTABLE_IMAGE;
1241 if (Config->LargeAddressAware)
1242 COFF->Characteristics |= IMAGE_FILE_LARGE_ADDRESS_AWARE;
1243 if (!Config->is64())
1244 COFF->Characteristics |= IMAGE_FILE_32BIT_MACHINE;
1245 if (Config->DLL)
1246 COFF->Characteristics |= IMAGE_FILE_DLL;
1247 if (!Config->Relocatable)
1248 COFF->Characteristics |= IMAGE_FILE_RELOCS_STRIPPED;
1249 if (Config->SwaprunCD)
1250 COFF->Characteristics |= IMAGE_FILE_REMOVABLE_RUN_FROM_SWAP;
1251 if (Config->SwaprunNet)
1252 COFF->Characteristics |= IMAGE_FILE_NET_RUN_FROM_SWAP;
1253 COFF->SizeOfOptionalHeader =
1254 sizeof(PEHeaderTy) + sizeof(data_directory) * NumberOfDataDirectory;
1255
1256 // Write PE header
1257 auto *PE = reinterpret_cast<PEHeaderTy *>(Buf);
1258 Buf += sizeof(*PE);
1259 PE->Magic = Config->is64() ? PE32Header::PE32_PLUS : PE32Header::PE32;
1260
1261 // If {Major,Minor}LinkerVersion is left at 0.0, then for some
1262 // reason signing the resulting PE file with Authenticode produces a
1263 // signature that fails to validate on Windows 7 (but is OK on 10).
1264 // Set it to 14.0, which is what VS2015 outputs, and which avoids
1265 // that problem.
1266 PE->MajorLinkerVersion = 14;
1267 PE->MinorLinkerVersion = 0;
1268
1269 PE->ImageBase = Config->ImageBase;
1270 PE->SectionAlignment = PageSize;
1271 PE->FileAlignment = Config->FileAlign;
1272 PE->MajorImageVersion = Config->MajorImageVersion;
1273 PE->MinorImageVersion = Config->MinorImageVersion;
1274 PE->MajorOperatingSystemVersion = Config->MajorOSVersion;
1275 PE->MinorOperatingSystemVersion = Config->MinorOSVersion;
1276 PE->MajorSubsystemVersion = Config->MajorOSVersion;
1277 PE->MinorSubsystemVersion = Config->MinorOSVersion;
1278 PE->Subsystem = Config->Subsystem;
1279 PE->SizeOfImage = SizeOfImage;
1280 PE->SizeOfHeaders = SizeOfHeaders;
1281 if (!Config->NoEntry) {
1282 Defined *Entry = cast<Defined>(Config->Entry);
1283 PE->AddressOfEntryPoint = Entry->getRVA();
1284 // Pointer to thumb code must have the LSB set, so adjust it.
1285 if (Config->Machine == ARMNT)
1286 PE->AddressOfEntryPoint |= 1;
1287 }
1288 PE->SizeOfStackReserve = Config->StackReserve;
1289 PE->SizeOfStackCommit = Config->StackCommit;
1290 PE->SizeOfHeapReserve = Config->HeapReserve;
1291 PE->SizeOfHeapCommit = Config->HeapCommit;
1292 if (Config->AppContainer)
1293 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_APPCONTAINER;
1294 if (Config->DynamicBase)
1295 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_DYNAMIC_BASE;
1296 if (Config->HighEntropyVA)
1297 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_HIGH_ENTROPY_VA;
1298 if (!Config->AllowBind)
1299 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_NO_BIND;
1300 if (Config->NxCompat)
1301 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_NX_COMPAT;
1302 if (!Config->AllowIsolation)
1303 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_NO_ISOLATION;
1304 if (Config->GuardCF != GuardCFLevel::Off)
1305 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_GUARD_CF;
1306 if (Config->IntegrityCheck)
1307 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_FORCE_INTEGRITY;
1308 if (SetNoSEHCharacteristic)
1309 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_NO_SEH;
1310 if (Config->TerminalServerAware)
1311 PE->DLLCharacteristics |= IMAGE_DLL_CHARACTERISTICS_TERMINAL_SERVER_AWARE;

1313 if (TextSec->getVirtualSize()) {
1314 PE->BaseOfCode = TextSec->getRVA();
1315 PE->SizeOfCode = TextSec->getRawSize();
1316 }
1317 PE->SizeOfInitializedData = getSizeOfInitializedData();

1320 auto *Dir = reinterpret_cast<data_directory *>(Buf);
1321 Buf += sizeof(*Dir) * NumberOfDataDirectory;
1322 if (!Config->Exports.empty()) {
1323 Dir[EXPORT_TABLE].RelativeVirtualAddress = Edata.getRVA();
1324 Dir[EXPORT_TABLE].Size = Edata.getSize();
1325 }
1326 if (ImportTableStart) {
1327 Dir[IMPORT_TABLE].RelativeVirtualAddress = ImportTableStart->getRVA();
1328 Dir[IMPORT_TABLE].Size = ImportTableSize;
1329 }
1330 if (IATStart) {
1331 Dir[IAT].RelativeVirtualAddress = IATStart->getRVA();
1332 Dir[IAT].Size = IATSize;
1333 }
1334 if (RsrcSec->getVirtualSize()) {
1335 Dir[RESOURCE_TABLE].RelativeVirtualAddress = RsrcSec->getRVA();
1336 Dir[RESOURCE_TABLE].Size = RsrcSec->getVirtualSize();
1337 }
1338 if (FirstPdata) {
1339 Dir[EXCEPTION_TABLE].RelativeVirtualAddress = FirstPdata->getRVA();
1340 Dir[EXCEPTION_TABLE].Size =
1341 LastPdata->getRVA() + LastPdata->getSize() - FirstPdata->getRVA();
1342 }
1343 if (RelocSec->getVirtualSize()) {
1344 Dir[BASE_RELOCATION_TABLE].RelativeVirtualAddress = RelocSec->getRVA();
1345 Dir[BASE_RELOCATION_TABLE].Size = RelocSec->getVirtualSize();
1346 }

1348 if (Defined *B = dyn_cast<Defined>(Sym)) {
1349 Dir[TLS_TABLE].RelativeVirtualAddress = B->getRVA();
1350 Dir[TLS_TABLE].Size = Config->is64()
1351 ? sizeof(object::coff_tls_directory64)
1352 : sizeof(object::coff_tls_directory32);
1353 }
1354 }
1355 if (DebugDirectory) {
1356 Dir[DEBUG_DIRECTORY].RelativeVirtualAddress = DebugDirectory->getRVA();
1357 Dir[DEBUG_DIRECTORY].Size = DebugDirectory->getSize();
1358 }
1359 if (Symbol *Sym = Symtab->findUnderscore("_load_config_used")) {
1360 if (auto *B = dyn_cast<DefinedRegular>(Sym)) {
1361 SectionChunk *SC = B->getChunk();
1362 assert(B->getRVA() >= SC->getRVA());
1363 uint64_t OffsetInChunk = B->getRVA() - SC->getRVA();
1364 if (!SC->hasData() || OffsetInChunk + 4 > SC->getSize())
1365 fatal("_load_config_used is malformed");
1366
1367 ArrayRef<uint8_t> SecContents = SC->getContents();
1368 uint32_t LoadConfigSize =
1369 *reinterpret_cast<const ulittle32_t *>(&SecContents[OffsetInChunk]);
1370 if (OffsetInChunk + LoadConfigSize > SC->getSize())
1371 fatal("_load_config_used is too large");
1372 Dir[LOAD_CONFIG_TABLE].RelativeVirtualAddress = B->getRVA();
1373 Dir[LOAD_CONFIG_TABLE].Size = LoadConfigSize;
1374 }
1375 }
1376 if (!DelayIdata.empty()) {
1377 Dir[DELAY_IMPORT_DESCRIPTOR].RelativeVirtualAddress =
1378 DelayIdata.getDirRVA();
1379 Dir[DELAY_IMPORT_DESCRIPTOR].Size = DelayIdata.getDirSize();
1380 }
1381
1382 // Write section table
1383 for (OutputSection *Sec : OutputSections) {
1384 Sec->writeHeaderTo(Buf);
1385 Buf += sizeof(coff_section);
1386 }
1387 SectionTable = ArrayRef<uint8_t>(
1388 Buf - OutputSections.size() * sizeof(coff_section), Buf);
1389
1390 if (OutputSymtab.empty() && Strtab.empty())
1391 return;
1392
1393 COFF->PointerToSymbolTable = PointerToSymbolTable;
1394 uint32_t NumberOfSymbols = OutputSymtab.size();
1395 COFF->NumberOfSymbols = NumberOfSymbols;
1396 auto *SymbolTable = reinterpret_cast<coff_symbol16 *>(
1397 Buffer->getBufferStart() + COFF->PointerToSymbolTable);
1398 for (size_t I = 0; I != NumberOfSymbols; ++I)
1399 SymbolTable[I] = OutputSymtab[I];
1400 // Create the string table, it follows immediately after the symbol table.
1401 // The first 4 bytes is length including itself.
1402 Buf = reinterpret_cast<uint8_t *>(&SymbolTable[NumberOfSymbols]);
1403 write32le(Buf, Strtab.size() + 4);
1404 if (!Strtab.empty())
1405 memcpy(Buf + 4, Strtab.data(), Strtab.size());
1406}

1439// Add a symbol to an RVA set. Two symbols may have the same RVA, but an RVA set
1440// cannot contain duplicates. Therefore, the set is uniqued by Chunk and the
1441// symbol's offset into that Chunk.
1442static void addSymbolToRVASet(SymbolRVASet &RVASet, Defined *S) {
1443 Chunk *C = S->getChunk();
1444 if (auto *SC = dyn_cast<SectionChunk>(C))
1445 C = SC->Repl; // Look through ICF replacement.
1446 uint32_t Off = S->getRVA() - (C ? C->getRVA() : 0);
1447 RVASet.insert({C, Off});
1448}

1450// Given a symbol, add it to the GFIDs table if it is a live, defined, function
1451// symbol in an executable section.
1452static void maybeAddAddressTakenFunction(SymbolRVASet &AddressTakenSyms,
1453 Symbol *S) {
1454 if (!S)
1455 return;
1456
1457 switch (S->kind()) {
1458 case Symbol::DefinedLocalImportKind:
1459 case Symbol::DefinedImportDataKind:
1460 // Defines an __imp_ pointer, so it is data, so it is ignored.
1461 break;
1462 case Symbol::DefinedCommonKind:
1463 // Common is always data, so it is ignored.
1464 break;
1465 case Symbol::DefinedAbsoluteKind:
1466 case Symbol::DefinedSyntheticKind:
1467 // Absolute is never code, synthetic generally isn't and usually isn't
1468 // determinable.
1469 break;
1470 case Symbol::LazyKind:
1471 case Symbol::UndefinedKind:
1472 // Undefined symbols resolve to zero, so they don't have an RVA. Lazy
1473 // symbols shouldn't have relocations.
1474 break;
1475
1476 case Symbol::DefinedImportThunkKind:
1477 // Thunks are always code, include them.
1478 addSymbolToRVASet(AddressTakenSyms, cast<Defined>(S));
1479 break;
1480
1481 case Symbol::DefinedRegularKind: {
1482 // This is a regular, defined, symbol from a COFF file. Mark the symbol as
1483 // address taken if the symbol type is function and it's in an executable
1484 // section.
1485 auto *D = cast<DefinedRegular>(S);
1486 if (D->getCOFFSymbol().getComplexType() == COFF::IMAGE_SYM_DTYPE_FUNCTION) {
1487 SectionChunk *SC = dyn_cast<SectionChunk>(D->getChunk());
1488 if (SC && SC->Live &&
1489 SC->getOutputCharacteristics() & IMAGE_SCN_MEM_EXECUTE)
1490 addSymbolToRVASet(AddressTakenSyms, D);
1491 }
1492 break;
1493 }
1494 }
1495}

1497// Visit all relocations from all section contributions of this object file and
1498// mark the relocation target as address-taken.
1499static void markSymbolsWithRelocations(ObjFile *File,
1500 SymbolRVASet &UsedSymbols) {
1501 for (Chunk *C : File->getChunks()) {
1502 // We only care about live section chunks. Common chunks and other chunks
1503 // don't generally contain relocations.
1504 SectionChunk *SC = dyn_cast<SectionChunk>(C);
1505 if (!SC || !SC->Live)
1506 continue;
1507
1508 for (const coff_relocation &Reloc : SC->getRelocs()) {
1509 if (Config->Machine == I386 && Reloc.Type == COFF::IMAGE_REL_I386_REL32)
1510 // Ignore relative relocations on x86. On x86_64 they can't be ignored
1511 // since they're also used to compute absolute addresses.
1512 continue;
1513
1514 Symbol *Ref = SC->File->getSymbol(Reloc.SymbolTableIndex);
1515 maybeAddAddressTakenFunction(UsedSymbols, Ref);
1516 }
1517 }
1518}

1520// Create the guard function id table. This is a table of RVAs of all
1521// address-taken functions. It is sorted and uniqued, just like the safe SEH
1522// table.
1523void Writer::createGuardCFTables() {
1524 SymbolRVASet AddressTakenSyms;
1525 SymbolRVASet LongJmpTargets;
1526 for (ObjFile *File : ObjFile::Instances) {
1527 // If the object was compiled with /guard:cf, the address taken symbols
1528 // are in .gfids$y sections, and the longjmp targets are in .gljmp$y
1529 // sections. If the object was not compiled with /guard:cf, we assume there
1530 // were no setjmp targets, and that all code symbols with relocations are
1531 // possibly address-taken.
1532 if (File->hasGuardCF()) {
1533 markSymbolsForRVATable(File, File->getGuardFidChunks(), AddressTakenSyms);
1534 markSymbolsForRVATable(File, File->getGuardLJmpChunks(), LongJmpTargets);
1535 } else {
1536 markSymbolsWithRelocations(File, AddressTakenSyms);
1537 }
1538 }
1539
1540 // Mark the image entry as address-taken.
1541 if (Config->Entry)
1542 maybeAddAddressTakenFunction(AddressTakenSyms, Config->Entry);
1543
1544 // Mark exported symbols in executable sections as address-taken.
1545 for (Export &E : Config->Exports)
1546 maybeAddAddressTakenFunction(AddressTakenSyms, E.Sym);
1547
1548 // Ensure sections referenced in the gfid table are 16-byte aligned.
1549 for (const ChunkAndOffset &C : AddressTakenSyms)
1550 if (C.InputChunk->getAlignment() < 16)
1551 C.InputChunk->setAlignment(16);
1552
1553 maybeAddRVATable(std::move(AddressTakenSyms), "__guard_fids_table",
1554 "__guard_fids_count");
1555
1556 // Add the longjmp target table unless the user told us not to.
1557 if (Config->GuardCF == GuardCFLevel::Full)
1558 maybeAddRVATable(std::move(LongJmpTargets), "__guard_longjmp_table",
1559 "__guard_longjmp_count");
1560
1561 // Set __guard_flags, which will be used in the load config to indicate that
1562 // /guard:cf was enabled.
1563 uint32_t GuardFlags = uint32_t(coff_guard_flags::CFInstrumented) |
1564 uint32_t(coff_guard_flags::HasFidTable);
1565 if (Config->GuardCF == GuardCFLevel::Full)
1566 GuardFlags |= uint32_t(coff_guard_flags::HasLongJmpTable);
1567 Symbol *FlagSym = Symtab->findUnderscore("__guard_flags");
1568 cast<DefinedAbsolute>(FlagSym)->setVA(GuardFlags);
1569}

1571// Take a list of input sections containing symbol table indices and add those
1572// symbols to an RVA table. The challenge is that symbol RVAs are not known and
1573// depend on the table size, so we can't directly build a set of integers.
1574void Writer::markSymbolsForRVATable(ObjFile *File,
1575 ArrayRef<SectionChunk *> SymIdxChunks,
1576 SymbolRVASet &TableSymbols) {
1577 for (SectionChunk *C : SymIdxChunks) {
1578 // Skip sections discarded by linker GC. This comes up when a .gfids section
1579 // is associated with something like a vtable and the vtable is discarded.
1580 // In this case, the associated gfids section is discarded, and we don't
1581 // mark the virtual member functions as address-taken by the vtable.
1582 if (!C->Live)
1583 continue;
1584
1585 // Validate that the contents look like symbol table indices.
1586 ArrayRef<uint8_t> Data = C->getContents();
1587 if (Data.size() % 4 != 0) {
1588 warn("ignoring " + C->getSectionName() +
1589 " symbol table index section in object " + toString(File));
1590 continue;
1591 }
1592
1593 // Read each symbol table index and check if that symbol was included in the
1594 // final link. If so, add it to the table symbol set.
1595 ArrayRef<ulittle32_t> SymIndices(
1596 reinterpret_cast<const ulittle32_t *>(Data.data()), Data.size() / 4);
1597 ArrayRef<Symbol *> ObjSymbols = File->getSymbols();
1598 for (uint32_t SymIndex : SymIndices) {
1599 if (SymIndex >= ObjSymbols.size()) {
1600 warn("ignoring invalid symbol table index in section " +
1601 C->getSectionName() + " in object " + toString(File));
1602 continue;
1603 }
1604 if (Symbol *S = ObjSymbols[SymIndex]) {
1605 if (S->isLive())
1606 addSymbolToRVASet(TableSymbols, cast<Defined>(S));
1607 }
1608 }
1609 }
1610}

1711void Writer::writeBuildId() {
1712 // There are two important parts to the build ID.
1713 // 1) If building with debug info, the COFF debug directory contains a
1714 // timestamp as well as a Guid and Age of the PDB.
1715 // 2) In all cases, the PE COFF file header also contains a timestamp.
1716 // For reproducibility, instead of a timestamp we want to use a hash of the
1717 // PE contents.
1718 if (Config->Debug) {
1719 assert(BuildId && "BuildId is not set!");
1720 // BuildId->BuildId was filled in when the PDB was written.
1721 }
1722
1723 // At this point the only fields in the COFF file which remain unset are the
1724 // "timestamp" in the COFF file header, and the ones in the coff debug
1725 // directory. Now we can hash the file and write that hash to the various
1726 // timestamp fields in the file.
1727 StringRef OutputFileData(
1728 reinterpret_cast<const char *>(Buffer->getBufferStart()),
1729 Buffer->getBufferSize());
1730
1731 uint32_t Timestamp = Config->Timestamp;
1732 uint64_t Hash = 0;
1733 bool GenerateSyntheticBuildId =
1734 Config->MinGW && Config->Debug && Config->PDBPath.empty();
1735
1736 if (Config->Repro || GenerateSyntheticBuildId)
1737 Hash = xxHash64(OutputFileData);
1738
1739 if (Config->Repro)
1740 Timestamp = static_cast<uint32_t>(Hash);
1741
1742 if (GenerateSyntheticBuildId) {
1743 // For MinGW builds without a PDB file, we still generate a build id
1744 // to allow associating a crash dump to the executable.
1745 BuildId->BuildId->PDB70.CVSignature = OMF::Signature::PDB70;
1746 BuildId->BuildId->PDB70.Age = 1;
1747 memcpy(BuildId->BuildId->PDB70.Signature, &Hash, 8);
1748 // xxhash only gives us 8 bytes, so put some fixed data in the other half.
1749 memcpy(&BuildId->BuildId->PDB70.Signature[8], "LLD PDB.", 8);
1750 }
1751
1752 if (DebugDirectory)
1753 DebugDirectory->setTimeDateStamp(Timestamp);
1754
1755 uint8_t *Buf = Buffer->getBufferStart();
1756 Buf += DOSStubSize + sizeof(PEMagic);
1757 object::coff_file_header *CoffHeader =
1758 reinterpret_cast<coff_file_header *>(Buf);
1759 CoffHeader->TimeDateStamp = Timestamp;
1760}

1762// Sort .pdata section contents according to PE/COFF spec 5.5.
1763void Writer::sortExceptionTable() {
1764 if (!FirstPdata)
1765 return;
1766 // We assume .pdata contains function table entries only.
1767 auto BufAddr = [&](Chunk *C) {
1768 OutputSection *OS = C->getOutputSection();
1769 return Buffer->getBufferStart() + OS->getFileOff() + C->getRVA() -
1770 OS->getRVA();
1771 };
1772 uint8_t *Begin = BufAddr(FirstPdata);
1773 uint8_t *End = BufAddr(LastPdata) + LastPdata->getSize();
1774 if (Config->Machine == AMD64) {
1775 struct Entry { ulittle32_t Begin, End, Unwind; };
1776 parallelSort(
1777 MutableArrayRef<Entry>((Entry *)Begin, (Entry *)End),
1778 [](const Entry &A, const Entry &B) { return A.Begin < B.Begin; });
1779 return;
1780 }
1781 if (Config->Machine == ARMNT || Config->Machine == ARM64) {
1782 struct Entry { ulittle32_t Begin, Unwind; };
1783 parallelSort(
1784 MutableArrayRef<Entry>((Entry *)Begin, (Entry *)End),
1785 [](const Entry &A, const Entry &B) { return A.Begin < B.Begin; });
1786 return;
1787 }
1788 errs() << "warning: don't know how to handle .pdata.\n";
1789}

1791// The CRT section contains, among other things, the array of function
1792// pointers that initialize every global variable that is not trivially
1793// constructed. The CRT calls them one after the other prior to invoking
1794// main().
1795//
1796// As per C++ spec, 3.6.2/2.3,
1797// "Variables with ordered initialization defined within a single
1798// translation unit shall be initialized in the order of their definitions
1799// in the translation unit"
1800//
1801// It is therefore critical to sort the chunks containing the function
1802// pointers in the order that they are listed in the object file (top to
1803// bottom), otherwise global objects might not be initialized in the
1804// correct order.
1805void Writer::sortCRTSectionChunks(std::vector<Chunk *> &Chunks) {
1806 auto SectionChunkOrder = [](const Chunk *A, const Chunk *B) {
1807 auto SA = dyn_cast<SectionChunk>(A);
1808 auto SB = dyn_cast<SectionChunk>(B);
1809 assert(SA && SB && "Non-section chunks in CRT section!");
1810
1811 StringRef SAObj = SA->File->MB.getBufferIdentifier();
1812 StringRef SBObj = SB->File->MB.getBufferIdentifier();
1813
1814 return SAObj == SBObj && SA->getSectionNumber() < SB->getSectionNumber();
1815 };
1816 llvm::stable_sort(Chunks, SectionChunkOrder);
1817
1818 if (Config->Verbose) {
1819 for (auto &C : Chunks) {
1820 auto SC = dyn_cast<SectionChunk>(C);
1821 log(" " + SC->File->MB.getBufferIdentifier().str() +
1822 ", SectionID: " + Twine(SC->getSectionNumber()));
1823 }
1824 }
1825}
