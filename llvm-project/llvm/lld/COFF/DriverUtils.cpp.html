48class Executor {
49public:
50 explicit Executor(StringRef S) : Prog(Saver.save(S)) {}
51 void add(StringRef S) { Args.push_back(Saver.save(S)); }
52 void add(std::string &S) { Args.push_back(Saver.save(S)); }
53 void add(Twine S) { Args.push_back(Saver.save(S)); }
54 void add(const char *S) { Args.push_back(Saver.save(S)); }
55
56 void run() {
57 ErrorOr<std::string> ExeOrErr = sys::findProgramByName(Prog);
58 if (auto EC = ExeOrErr.getError())
59 fatal("unable to find " + Prog + " in PATH: " + EC.message());
60 StringRef Exe = Saver.save(*ExeOrErr);
61 Args.insert(Args.begin(), Exe);
62
63 if (sys::ExecuteAndWait(Args[0], Args) != 0)
64 fatal("ExecuteAndWait failed: " +
65 llvm::join(Args.begin(), Args.end(), " "));
66 }
67
68private:
69 StringRef Prog;
70 std::vector<StringRef> Args;
71};

75// Parses a string in the form of "<integer>[,<integer>]".
76void parseNumbers(StringRef Arg, uint64_t *Addr, uint64_t *Size) {
77 StringRef S1, S2;
78 std::tie(S1, S2) = Arg.split(',');
79 if (S1.getAsInteger(0, *Addr))
80 fatal("invalid number: " + S1);
81 if (Size && !S2.empty() && S2.getAsInteger(0, *Size))
82 fatal("invalid number: " + S2);
83}

85// Parses a string in the form of "<integer>[.<integer>]".
86// If second number is not present, Minor is set to 0.
87void parseVersion(StringRef Arg, uint32_t *Major, uint32_t *Minor) {
88 StringRef S1, S2;
89 std::tie(S1, S2) = Arg.split('.');
90 if (S1.getAsInteger(0, *Major))
91 fatal("invalid number: " + S1);
92 *Minor = 0;
93 if (!S2.empty() && S2.getAsInteger(0, *Minor))
94 fatal("invalid number: " + S2);
95}

97void parseGuard(StringRef FullArg) {
98 SmallVector<StringRef, 1> SplitArgs;
99 FullArg.split(SplitArgs, ",");
100 for (StringRef Arg : SplitArgs) {
101 if (Arg.equals_lower("no"))
102 Config->GuardCF = GuardCFLevel::Off;
103 else if (Arg.equals_lower("nolongjmp"))
104 Config->GuardCF = GuardCFLevel::NoLongJmp;
105 else if (Arg.equals_lower("cf") || Arg.equals_lower("longjmp"))
106 Config->GuardCF = GuardCFLevel::Full;
107 else
108 fatal("invalid argument to /guard: " + Arg);
109 }
110}

112// Parses a string in the form of "<subsystem>[,<integer>[.<integer>]]".
113void parseSubsystem(StringRef Arg, WindowsSubsystem *Sys, uint32_t *Major,
114 uint32_t *Minor) {
115 StringRef SysStr, Ver;
116 std::tie(SysStr, Ver) = Arg.split(',');
117 *Sys = StringSwitch<WindowsSubsystem>(SysStr.lower())
118 .Case("boot_application", IMAGE_SUBSYSTEM_WINDOWS_BOOT_APPLICATION)
119 .Case("console", IMAGE_SUBSYSTEM_WINDOWS_CUI)
120 .Case("efi_application", IMAGE_SUBSYSTEM_EFI_APPLICATION)
121 .Case("efi_boot_service_driver", IMAGE_SUBSYSTEM_EFI_BOOT_SERVICE_DRIVER)
122 .Case("efi_rom", IMAGE_SUBSYSTEM_EFI_ROM)
123 .Case("efi_runtime_driver", IMAGE_SUBSYSTEM_EFI_RUNTIME_DRIVER)
124 .Case("native", IMAGE_SUBSYSTEM_NATIVE)
125 .Case("posix", IMAGE_SUBSYSTEM_POSIX_CUI)
126 .Case("windows", IMAGE_SUBSYSTEM_WINDOWS_GUI)
127 .Default(IMAGE_SUBSYSTEM_UNKNOWN);
128 if (*Sys == IMAGE_SUBSYSTEM_UNKNOWN)
129 fatal("unknown subsystem: " + SysStr);
130 if (!Ver.empty())
131 parseVersion(Ver, Major, Minor);
132}

134// Parse a string of the form of "<from>=<to>".
135// Results are directly written to Config.
136void parseAlternateName(StringRef S) {
137 StringRef From, To;
138 std::tie(From, To) = S.split('=');
139 if (From.empty() || To.empty())
140 fatal("/alternatename: invalid argument: " + S);
141 auto It = Config->AlternateNames.find(From);
142 if (It != Config->AlternateNames.end() && It->second != To)
143 fatal("/alternatename: conflicts: " + S);
144 Config->AlternateNames.insert(It, std::make_pair(From, To));
145}

147// Parse a string of the form of "<from>=<to>".
148// Results are directly written to Config.
149void parseMerge(StringRef S) {
150 StringRef From, To;
151 std::tie(From, To) = S.split('=');
152 if (From.empty() || To.empty())
153 fatal("/merge: invalid argument: " + S);
154 if (From == ".rsrc" || To == ".rsrc")
155 fatal("/merge: cannot merge '.rsrc' with any section");
156 if (From == ".reloc" || To == ".reloc")
157 fatal("/merge: cannot merge '.reloc' with any section");
158 auto Pair = Config->Merge.insert(std::make_pair(From, To));
159 bool Inserted = Pair.second;
160 if (!Inserted) {
161 StringRef Existing = Pair.first->second;
162 if (Existing != To)
163 warn(S + ": already merged into " + Existing);
164 }
165}

167static uint32_t parseSectionAttributes(StringRef S) {
168 uint32_t Ret = 0;
169 for (char C : S.lower()) {
170 switch (C) {
171 case 'd':
172 Ret |= IMAGE_SCN_MEM_DISCARDABLE;
173 break;
174 case 'e':
175 Ret |= IMAGE_SCN_MEM_EXECUTE;
176 break;
177 case 'k':
178 Ret |= IMAGE_SCN_MEM_NOT_CACHED;
179 break;
180 case 'p':
181 Ret |= IMAGE_SCN_MEM_NOT_PAGED;
182 break;
183 case 'r':
184 Ret |= IMAGE_SCN_MEM_READ;
185 break;
186 case 's':
187 Ret |= IMAGE_SCN_MEM_SHARED;
188 break;
189 case 'w':
190 Ret |= IMAGE_SCN_MEM_WRITE;
191 break;
192 default:
193 fatal("/section: invalid argument: " + S);
194 }
195 }
196 return Ret;
197}

208// Parses /aligncomm option argument.
209void parseAligncomm(StringRef S) {
210 StringRef Name, Align;
211 std::tie(Name, Align) = S.split(',');
212 if (Name.empty() || Align.empty()) {
213 error("/aligncomm: invalid argument: " + S);
214 return;
215 }
216 int V;
217 if (Align.getAsInteger(0, V)) {
218 error("/aligncomm: invalid argument: " + S);
219 return;
220 }
221 Config->AlignComm[Name] = std::max(Config->AlignComm[Name], 1 << V);
222}

224// Parses /functionpadmin option argument.
225void parseFunctionPadMin(llvm::opt::Arg *A, llvm::COFF::MachineTypes Machine) {
226 StringRef Arg = A->getNumValues() ? A->getValue() : "";
227 if (!Arg.empty()) {
228 // Optional padding in bytes is given.
229 if (Arg.getAsInteger(0, Config->FunctionPadMin))
230 error("/functionpadmin: invalid argument: " + Arg);
231 return;
232 }
233 // No optional argument given.
234 // Set default padding based on machine, similar to link.exe.
235 // There is no default padding for ARM platforms.
236 if (Machine == I386) {
237 Config->FunctionPadMin = 5;
238 } else if (Machine == AMD64) {
239 Config->FunctionPadMin = 6;
240 } else {
241 error("/functionpadmin: invalid argument for this machine: " + Arg);
242 }
243}

245// Parses a string in the form of "EMBED[,=<integer>]|NO".
246// Results are directly written to Config.
247void parseManifest(StringRef Arg) {
248 if (Arg.equals_lower("no")) {
249 Config->Manifest = Configuration::No;
250 return;
251 }
252 if (!Arg.startswith_lower("embed"))
253 fatal("invalid option " + Arg);
254 Config->Manifest = Configuration::Embed;
255 Arg = Arg.substr(strlen("embed"));
256 if (Arg.empty())
257 return;
258 if (!Arg.startswith_lower(",id="))
259 fatal("invalid option " + Arg);
260 Arg = Arg.substr(strlen(",id="));
261 if (Arg.getAsInteger(0, Config->ManifestID))
262 fatal("invalid option " + Arg);
263}

265// Parses a string in the form of "level=<string>|uiAccess=<string>|NO".
266// Results are directly written to Config.
267void parseManifestUAC(StringRef Arg) {
268 if (Arg.equals_lower("no")) {
269 Config->ManifestUAC = false;
270 return;
271 }
272 for (;;) {
273 Arg = Arg.ltrim();
274 if (Arg.empty())
275 return;
276 if (Arg.startswith_lower("level=")) {
277 Arg = Arg.substr(strlen("level="));
278 std::tie(Config->ManifestLevel, Arg) = Arg.split(" ");
279 continue;
280 }
281 if (Arg.startswith_lower("uiaccess=")) {
282 Arg = Arg.substr(strlen("uiaccess="));
283 std::tie(Config->ManifestUIAccess, Arg) = Arg.split(" ");
284 continue;
285 }
286 fatal("invalid option " + Arg);
287 }
288}

290// Parses a string in the form of "cd|net[,(cd|net)]*"
291// Results are directly written to Config.
292void parseSwaprun(StringRef Arg) {
293 do {
294 StringRef Swaprun, NewArg;
295 std::tie(Swaprun, NewArg) = Arg.split(',');
296 if (Swaprun.equals_lower("cd"))
297 Config->SwaprunCD = true;
298 else if (Swaprun.equals_lower("net"))
299 Config->SwaprunNet = true;
300 else if (Swaprun.empty())
301 error("/swaprun: missing argument");
302 else
303 error("/swaprun: invalid argument: " + Swaprun);
304 // To catch trailing commas, e.g. `/spawrun:cd,`
305 if (NewArg.empty() && Arg.endswith(","))
306 error("/swaprun: missing argument");
307 Arg = NewArg;
308 } while (!Arg.empty());
309}

311// An RAII temporary file class that automatically removes a temporary file.
312namespace {
313class TemporaryFile {
314public:
315 TemporaryFile(StringRef Prefix, StringRef Extn, StringRef Contents = "") {
316 SmallString<128> S;
317 if (auto EC = sys::fs::createTemporaryFile("lld-" + Prefix, Extn, S))
318 fatal("cannot create a temporary file: " + EC.message());
319 Path = S.str();
320
321 if (!Contents.empty()) {
322 std::error_code EC;
323 raw_fd_ostream OS(Path, EC, sys::fs::F_None);
324 if (EC)
325 fatal("failed to open " + Path + ": " + EC.message());
326 OS << Contents;
327 }
328 }
329
330 TemporaryFile(TemporaryFile &&Obj) {
331 std::swap(Path, Obj.Path);
332 }
333
334 ~TemporaryFile() {
335 if (Path.empty())
336 return;
337 if (sys::fs::remove(Path))
338 fatal("failed to remove " + Path);
339 }
340
341 // Returns a memory buffer of this temporary file.
342 // Note that this function does not leave the file open,
343 // so it is safe to remove the file immediately after this function
344 // is called (you cannot remove an opened file on Windows.)
345 std::unique_ptr<MemoryBuffer> getMemoryBuffer() {
346 // IsVolatileSize=true forces MemoryBuffer to not use mmap().
347 return CHECK(MemoryBuffer::getFile(Path, /*FileSize=*/-1,
348 /*RequiresNullTerminator=*/false,
349 /*IsVolatileSize=*/true),
350 "could not open " + Path);
351 }
352
353 std::string Path;
354};
355}

357static std::string createDefaultXml() {
358 std::string Ret;
359 raw_string_ostream OS(Ret);
360
361 // Emit the XML. Note that we do *not* verify that the XML attributes are
362 // syntactically correct. This is intentional for link.exe compatibility.
363 OS << "<?xml version=\"1.0\" standalone=\"yes\"?>\n"
364 << "<assembly xmlns=\"urn:schemas-microsoft-com:asm.v1\"\n"
365 << " manifestVersion=\"1.0\">\n";
366 if (Config->ManifestUAC) {
367 OS << " <trustInfo>\n"
368 << " <security>\n"
369 << " <requestedPrivileges>\n"
370 << " <requestedExecutionLevel level=" << Config->ManifestLevel
371 << " uiAccess=" << Config->ManifestUIAccess << "/>\n"
372 << " </requestedPrivileges>\n"
373 << " </security>\n"
374 << " </trustInfo>\n";
375 }
376 if (!Config->ManifestDependency.empty()) {
377 OS << " <dependency>\n"
378 << " <dependentAssembly>\n"
379 << " <assemblyIdentity " << Config->ManifestDependency << " />\n"
380 << " </dependentAssembly>\n"
381 << " </dependency>\n";
382 }
383 OS << "</assembly>\n";
384 return OS.str();
385}

387static std::string createManifestXmlWithInternalMt(StringRef DefaultXml) {
388 std::unique_ptr<MemoryBuffer> DefaultXmlCopy =
389 MemoryBuffer::getMemBufferCopy(DefaultXml);
390
391 windows_manifest::WindowsManifestMerger Merger;
392 if (auto E = Merger.merge(*DefaultXmlCopy.get()))
393 fatal("internal manifest tool failed on default xml: " +
394 toString(std::move(E)));
395
396 for (StringRef Filename : Config->ManifestInput) {
397 std::unique_ptr<MemoryBuffer> Manifest =
398 check(MemoryBuffer::getFile(Filename));
399 if (auto E = Merger.merge(*Manifest.get()))
400 fatal("internal manifest tool failed on file " + Filename + ": " +
401 toString(std::move(E)));
402 }
403
404 return Merger.getMergedManifest().get()->getBuffer();
405}

407static std::string createManifestXmlWithExternalMt(StringRef DefaultXml) {
408 // Create the default manifest file as a temporary file.
409 TemporaryFile Default("defaultxml", "manifest");
410 std::error_code EC;
411 raw_fd_ostream OS(Default.Path, EC, sys::fs::F_Text);
412 if (EC)
413 fatal("failed to open " + Default.Path + ": " + EC.message());
414 OS << DefaultXml;
415 OS.close();
416
417 // Merge user-supplied manifests if they are given. Since libxml2 is not
418 // enabled, we must shell out to Microsoft's mt.exe tool.
419 TemporaryFile User("user", "manifest");
420
421 Executor E("mt.exe");
422 E.add("/manifest");
423 E.add(Default.Path);
424 for (StringRef Filename : Config->ManifestInput) {
425 E.add("/manifest");
426 E.add(Filename);
427 }
428 E.add("/nologo");
429 E.add("/out:" + StringRef(User.Path));
430 E.run();
431
432 return CHECK(MemoryBuffer::getFile(User.Path), "could not open " + User.Path)
433 .get()
434 ->getBuffer();
435}

437static std::string createManifestXml() {
438 std::string DefaultXml = createDefaultXml();
439 if (Config->ManifestInput.empty())
440 return DefaultXml;
441
442 if (windows_manifest::isAvailable())
443 return createManifestXmlWithInternalMt(DefaultXml);
444
445 return createManifestXmlWithExternalMt(DefaultXml);

448static std::unique_ptr<WritableMemoryBuffer>
449createMemoryBufferForManifestRes(size_t ManifestSize) {
450 size_t ResSize = alignTo(
451 object::WIN_RES_MAGIC_SIZE + object::WIN_RES_NULL_ENTRY_SIZE +
452 sizeof(object::WinResHeaderPrefix) + sizeof(object::WinResIDs) +
453 sizeof(object::WinResHeaderSuffix) + ManifestSize,
454 object::WIN_RES_DATA_ALIGNMENT);
455 return WritableMemoryBuffer::getNewMemBuffer(ResSize, Config->OutputFile +
456 ".manifest.res");
457}
458
459static void writeResFileHeader(char *&Buf) {
460 memcpy(Buf, COFF::WinResMagic, sizeof(COFF::WinResMagic));
461 Buf += sizeof(COFF::WinResMagic);
462 memset(Buf, 0, object::WIN_RES_NULL_ENTRY_SIZE);
463 Buf += object::WIN_RES_NULL_ENTRY_SIZE;
464}
465
466static void writeResEntryHeader(char *&Buf, size_t ManifestSize) {
467 // Write the prefix.
468 auto *Prefix = reinterpret_cast<object::WinResHeaderPrefix *>(Buf);
469 Prefix->DataSize = ManifestSize;
470 Prefix->HeaderSize = sizeof(object::WinResHeaderPrefix) +
471 sizeof(object::WinResIDs) +
472 sizeof(object::WinResHeaderSuffix);
473 Buf += sizeof(object::WinResHeaderPrefix);
474
475 // Write the Type/Name IDs.
476 auto *IDs = reinterpret_cast<object::WinResIDs *>(Buf);
477 IDs->setType(RT_MANIFEST);
478 IDs->setName(Config->ManifestID);
479 Buf += sizeof(object::WinResIDs);
480
481 // Write the suffix.
482 auto *Suffix = reinterpret_cast<object::WinResHeaderSuffix *>(Buf);
483 Suffix->DataVersion = 0;
484 Suffix->MemoryFlags = object::WIN_RES_PURE_MOVEABLE;
485 Suffix->Language = SUBLANG_ENGLISH_US;
486 Suffix->Version = 0;
487 Suffix->Characteristics = 0;
488 Buf += sizeof(object::WinResHeaderSuffix);
489}

491// Create a resource file containing a manifest XML.
492std::unique_ptr<MemoryBuffer> createManifestRes() {
493 std::string Manifest = createManifestXml();
494
495 std::unique_ptr<WritableMemoryBuffer> Res =
496 createMemoryBufferForManifestRes(Manifest.size());
497
498 char *Buf = Res->getBufferStart();
499 writeResFileHeader(Buf);
500 writeResEntryHeader(Buf, Manifest.size());
501
502 // Copy the manifest data into the .res file.
503 std::copy(Manifest.begin(), Manifest.end(), Buf);
504 return std::move(Res);
505}

518// Parse a string in the form of
519// "<name>[=<internalname>][,@ordinal[,NONAME]][,DATA][,PRIVATE]"
520// or "<name>=<dllname>.<name>".
521// Used for parsing /export arguments.
522Export parseExport(StringRef Arg) {
523 Export E;
524 StringRef Rest;
525 std::tie(E.Name, Rest) = Arg.split(",");
526 if (E.Name.empty())
527 goto err;
528
529 if (E.Name.contains('=')) {
530 StringRef X, Y;
531 std::tie(X, Y) = E.Name.split("=");
532
533 // If "<name>=<dllname>.<name>".
534 if (Y.contains(".")) {
535 E.Name = X;
536 E.ForwardTo = Y;
537 return E;
538 }
539
540 E.ExtName = X;
541 E.Name = Y;
542 if (E.Name.empty())
543 goto err;
544 }
545
546 // If "<name>=<internalname>[,@ordinal[,NONAME]][,DATA][,PRIVATE]"
547 while (!Rest.empty()) {
548 StringRef Tok;
549 std::tie(Tok, Rest) = Rest.split(",");
550 if (Tok.equals_lower("noname")) {
551 if (E.Ordinal == 0)
552 goto err;
553 E.Noname = true;
554 continue;
555 }
556 if (Tok.equals_lower("data")) {
557 E.Data = true;
558 continue;
559 }
560 if (Tok.equals_lower("constant")) {
561 E.Constant = true;
562 continue;
563 }
564 if (Tok.equals_lower("private")) {
565 E.Private = true;
566 continue;
567 }
568 if (Tok.startswith("@")) {
569 int32_t Ord;
570 if (Tok.substr(1).getAsInteger(0, Ord))
571 goto err;
572 if (Ord <= 0 || 65535 < Ord)
573 goto err;
574 E.Ordinal = Ord;
575 continue;
576 }
577 goto err;
578 }
579 return E;
580
581err:
582 fatal("invalid /export: " + Arg);
583}

585static StringRef undecorate(StringRef Sym) {
586 if (Config->Machine != I386)
587 return Sym;
588 // In MSVC mode, a fully decorated stdcall function is exported
589 // as-is with the leading underscore (with type IMPORT_NAME).
590 // In MinGW mode, a decorated stdcall function gets the underscore
591 // removed, just like normal cdecl functions.
592 if (Sym.startswith("_") && Sym.contains('@') && !Config->MinGW)
593 return Sym;
594 return Sym.startswith("_") ? Sym.substr(1) : Sym;
595}

597// Convert stdcall/fastcall style symbols into unsuffixed symbols,
598// with or without a leading underscore. (MinGW specific.)
599static StringRef killAt(StringRef Sym, bool Prefix) {
600 if (Sym.empty())
601 return Sym;
602 // Strip any trailing stdcall suffix
603 Sym = Sym.substr(0, Sym.find('@', 1));
604 if (!Sym.startswith("@")) {
605 if (Prefix && !Sym.startswith("_"))
606 return Saver.save("_" + Sym);
607 return Sym;
608 }
609 // For fastcall, remove the leading @ and replace it with an
610 // underscore, if prefixes are used.
611 Sym = Sym.substr(1);
612 if (Prefix)
613 Sym = Saver.save("_" + Sym);
614 return Sym;
615}

617// Performs error checking on all /export arguments.
618// It also sets ordinals.
619void fixupExports() {
620 // Symbol ordinals must be unique.
621 std::set<uint16_t> Ords;
622 for (Export &E : Config->Exports) {
623 if (E.Ordinal == 0)
624 continue;
625 if (!Ords.insert(E.Ordinal).second)
626 fatal("duplicate export ordinal: " + E.Name);
627 }
628
629 for (Export &E : Config->Exports) {
630 if (!E.ForwardTo.empty()) {
631 E.ExportName = undecorate(E.Name);
632 } else {
633 E.ExportName = undecorate(E.ExtName.empty() ? E.Name : E.ExtName);
634 }
635 }
636
637 if (Config->KillAt && Config->Machine == I386) {
638 for (Export &E : Config->Exports) {
639 E.Name = killAt(E.Name, true);
640 E.ExportName = killAt(E.ExportName, false);
641 E.ExtName = killAt(E.ExtName, true);
642 E.SymbolName = killAt(E.SymbolName, true);
643 }
644 }
645
646 // Uniquefy by name.
647 DenseMap<StringRef, Export *> Map(Config->Exports.size());
648 std::vector<Export> V;
649 for (Export &E : Config->Exports) {
650 auto Pair = Map.insert(std::make_pair(E.ExportName, &E));
651 bool Inserted = Pair.second;
652 if (Inserted) {
653 V.push_back(E);
654 continue;
655 }
656 Export *Existing = Pair.first->second;
657 if (E == *Existing || E.Name != Existing->Name)
658 continue;
659 warn("duplicate /export option: " + E.Name);
660 }
661 Config->Exports = std::move(V);
662
663 // Sort by name.
664 std::sort(Config->Exports.begin(), Config->Exports.end(),
665 [](const Export &A, const Export &B) {
666 return A.ExportName < B.ExportName;
667 });
668}

670void assignExportOrdinals() {
671 // Assign unique ordinals if default (= 0).
672 uint16_t Max = 0;
673 for (Export &E : Config->Exports)
674 Max = std::max(Max, E.Ordinal);
675 for (Export &E : Config->Exports)
676 if (E.Ordinal == 0)
677 E.Ordinal = ++Max;
678}

680// Parses a string in the form of "key=value" and check
681// if value matches previous values for the same key.
682void checkFailIfMismatch(StringRef Arg, InputFile *Source) {
683 StringRef K, V;
684 std::tie(K, V) = Arg.split('=');
685 if (K.empty() || V.empty())
686 fatal("/failifmismatch: invalid argument: " + Arg);
687 std::pair<StringRef, InputFile *> Existing = Config->MustMatch[K];
688 if (!Existing.first.empty() && V != Existing.first) {
689 std::string SourceStr = Source ? toString(Source) : "cmd-line";
690 std::string ExistingStr =
691 Existing.second ? toString(Existing.second) : "cmd-line";
692 fatal("/failifmismatch: mismatch detected for '" + K + "':\n>>> " +
693 ExistingStr + " has value " + Existing.first + "\n>>> " + SourceStr +
694 " has value " + V);
695 }
696 Config->MustMatch[K] = {V, Source};
697}

699// Convert Windows resource files (.res files) to a .obj file.
700// Does what cvtres.exe does, but in-process and cross-platform.
701MemoryBufferRef convertResToCOFF(ArrayRef<MemoryBufferRef> MBs) {
702 object::WindowsResourceParser Parser;
703
704 for (MemoryBufferRef MB : MBs) {
705 std::unique_ptr<object::Binary> Bin = check(object::createBinary(MB));
706 object::WindowsResource *RF = dyn_cast<object::WindowsResource>(Bin.get());
707 if (!RF)
708 fatal("cannot compile non-resource file as resource");
709
710 std::vector<std::string> Duplicates;
711 if (auto EC = Parser.parse(RF, Duplicates))
712 fatal(toString(std::move(EC)));
713
714 for (const auto &DupeDiag : Duplicates)
715 if (Config->ForceMultipleRes)
716 warn(DupeDiag);
717 else
718 error(DupeDiag);
719 }
720
721 Expected<std::unique_ptr<MemoryBuffer>> E =
722 llvm::object::writeWindowsResourceCOFF(Config->Machine, Parser,
723 Config->Timestamp);
724 if (!E)
725 fatal("failed to write .res to COFF: " + toString(E.takeError()));
726
727 MemoryBufferRef MBRef = **E;
728 make<std::unique_ptr<MemoryBuffer>>(std::move(*E)); // take ownership
729 return MBRef;
730}

750// Set color diagnostics according to --color-diagnostics={auto,always,never}
751// or --no-color-diagnostics flags.
752static void handleColorDiagnostics(opt::InputArgList &Args) {
753 auto *Arg = Args.getLastArg(OPT_color_diagnostics, OPT_color_diagnostics_eq,
754 OPT_no_color_diagnostics);
755 if (!Arg)
756 return;
757 if (Arg->getOption().getID() == OPT_color_diagnostics) {
758 errorHandler().ColorDiagnostics = true;
759 } else if (Arg->getOption().getID() == OPT_no_color_diagnostics) {
760 errorHandler().ColorDiagnostics = false;
761 } else {
762 StringRef S = Arg->getValue();
763 if (S == "always")
764 errorHandler().ColorDiagnostics = true;
765 else if (S == "never")
766 errorHandler().ColorDiagnostics = false;
767 else if (S != "auto")
768 error("unknown option: --color-diagnostics=" + S);
769 }
770}

772static cl::TokenizerCallback getQuotingStyle(opt::InputArgList &Args) {
773 if (auto *Arg = Args.getLastArg(OPT_rsp_quoting)) {
774 StringRef S = Arg->getValue();
775 if (S != "windows" && S != "posix")
776 error("invalid response file quoting: " + S);
777 if (S == "windows")
778 return cl::TokenizeWindowsCommandLine;
779 return cl::TokenizeGNUCommandLine;
780 }
781 // The COFF linker always defaults to Windows quoting.
782 return cl::TokenizeWindowsCommandLine;
783}

785// Parses a given list of options.
786opt::InputArgList ArgParser::parse(ArrayRef<const char *> Argv) {
787 // Make InputArgList from string vectors.
788 unsigned MissingIndex;
789 unsigned MissingCount;
790
791 // We need to get the quoting style for response files before parsing all
792 // options so we parse here before and ignore all the options but
793 // --rsp-quoting.
794 opt::InputArgList Args = Table.ParseArgs(Argv, MissingIndex, MissingCount);
795
796 // Expand response files (arguments in the form of @<filename>)
797 // and then parse the argument again.
798 SmallVector<const char *, 256> ExpandedArgv(Argv.data(),
799 Argv.data() + Argv.size());
800 cl::ExpandResponseFiles(Saver, getQuotingStyle(Args), ExpandedArgv);
801 Args = Table.ParseArgs(makeArrayRef(ExpandedArgv).drop_front(), MissingIndex,
802 MissingCount);
803
804 // Print the real command line if response files are expanded.
805 if (Args.hasArg(OPT_verbose) && Argv.size() != ExpandedArgv.size()) {
806 std::string Msg = "Command line:";
807 for (const char *S : ExpandedArgv)
808 Msg += " " + std::string(S);
809 message(Msg);
810 }
811
812 // Save the command line after response file expansion so we can write it to
813 // the PDB if necessary.
814 Config->Argv = {ExpandedArgv.begin(), ExpandedArgv.end()};
815
816 // Handle /WX early since it converts missing argument warnings to errors.
817 errorHandler().FatalWarnings = Args.hasFlag(OPT_WX, OPT_WX_no, false);
818
819 if (MissingCount)
820 fatal(Twine(Args.getArgString(MissingIndex)) + ": missing argument");
821
822 handleColorDiagnostics(Args);
823
824 for (auto *Arg : Args.filtered(OPT_UNKNOWN)) {
825 std::string Nearest;
826 if (Table.findNearest(Arg->getAsString(Args), Nearest) > 1)
827 warn("ignoring unknown argument '" + Arg->getSpelling() + "'");
828 else
829 warn("ignoring unknown argument '" + Arg->getSpelling() +
830 "', did you mean '" + Nearest + "'");
831 }
832
833 if (Args.hasArg(OPT_lib))
834 warn("ignoring /lib since it's not the first argument");
835
836 return Args;
837}

839// Tokenizes and parses a given string as command line in .drective section.
840// /EXPORT options are processed in fastpath.
841std::pair<opt::InputArgList, std::vector<StringRef>>
842ArgParser::parseDirectives(StringRef S) {
843 std::vector<StringRef> Exports;
844 SmallVector<const char *, 16> Rest;
845
846 for (StringRef Tok : tokenize(S)) {
847 if (Tok.startswith_lower("/export:") || Tok.startswith_lower("-export:"))
848 Exports.push_back(Tok.substr(strlen("/export:")));
849 else
850 Rest.push_back(Tok.data());
851 }
852
853 // Make InputArgList from unparsed string vectors.
854 unsigned MissingIndex;
855 unsigned MissingCount;
856
857 opt::InputArgList Args = Table.ParseArgs(Rest, MissingIndex, MissingCount);
858
859 if (MissingCount)
860 fatal(Twine(Args.getArgString(MissingIndex)) + ": missing argument");
861 for (auto *Arg : Args.filtered(OPT_UNKNOWN))
862 warn("ignoring unknown argument: " + Arg->getSpelling());
863 return {std::move(Args), std::move(Exports)};
864}

866// link.exe has an interesting feature. If LINK or _LINK_ environment
867// variables exist, their contents are handled as command line strings.
868// So you can pass extra arguments using them.
869opt::InputArgList ArgParser::parseLINK(std::vector<const char *> Argv) {
870 // Concatenate LINK env and command line arguments, and then parse them.
871 if (Optional<std::string> S = Process::GetEnv("LINK")) {
872 std::vector<const char *> V = tokenize(*S);
873 Argv.insert(std::next(Argv.begin()), V.begin(), V.end());
874 }
875 if (Optional<std::string> S = Process::GetEnv("_LINK_")) {
876 std::vector<const char *> V = tokenize(*S);
877 Argv.insert(std::next(Argv.begin()), V.begin(), V.end());
878 }
879 return parse(Argv);
880}

882std::vector<const char *> ArgParser::tokenize(StringRef S) {
883 SmallVector<const char *, 16> Tokens;
884 cl::TokenizeWindowsCommandLine(S, Saver, Tokens);
885 return std::vector<const char *>(Tokens.begin(), Tokens.end());
886}
