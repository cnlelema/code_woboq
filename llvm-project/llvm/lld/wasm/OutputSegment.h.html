9#ifndef LLD_WASM_OUTPUT_SEGMENT_H
10#define LLD_WASM_OUTPUT_SEGMENT_H
11
12#include "InputChunks.h"
13#include "lld/Common/ErrorHandler.h"
14#include "llvm/Object/Wasm.h"
15
16namespace lld {
17namespace wasm {
18
19class InputSegment;
20
21class OutputSegment {
22public:
23 OutputSegment(StringRef N, uint32_t Index) : Name(N), Index(Index) {}
24
25 void addInputSegment(InputSegment *InSeg) {
26 Alignment = std::max(Alignment, InSeg->getAlignment());
27 InputSegments.push_back(InSeg);
28 Size = llvm::alignTo(Size, 1ULL << InSeg->getAlignment());
29 InSeg->OutputSeg = this;
30 InSeg->OutputSegmentOffset = Size;
31 Size += InSeg->getSize();
32 }
33
34 StringRef Name;
35 const uint32_t Index;
36 uint32_t SectionOffset = 0;
37 uint32_t Alignment = 0;
38 uint32_t StartVA = 0;
39 std::vector<InputSegment *> InputSegments;
40
41 // Sum of the size of the all the input segments
42 uint32_t Size = 0;
43
44 // Segment header
45 std::string Header;
46};
47
48} // namespace wasm
49} // namespace lld
50
51#endif // LLD_WASM_OUTPUT_SEGMENT_H
