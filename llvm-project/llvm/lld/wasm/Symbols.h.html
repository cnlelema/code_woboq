41// The base class for real symbol classes.
42class Symbol {
43public:
44 enum Kind {
45 DefinedFunctionKind,
46 DefinedDataKind,
47 DefinedGlobalKind,
48 DefinedEventKind,
49 SectionKind,
50 OutputSectionKind,
51 UndefinedFunctionKind,
52 UndefinedDataKind,
53 UndefinedGlobalKind,
54 LazyKind,
55 };
56
57 Kind kind() const { return SymbolKind; }
58
59 bool isDefined() const { return !isLazy() && !isUndefined(); }
60
61 bool isUndefined() const {
62 return SymbolKind == UndefinedFunctionKind ||
63 SymbolKind == UndefinedDataKind || SymbolKind == UndefinedGlobalKind;
64 }
65
66 bool isLazy() const { return SymbolKind == LazyKind; }
67
68 bool isLocal() const;
69 bool isWeak() const;
70 bool isHidden() const;
71
72 // Returns true if this symbol exists in a discarded (due to COMDAT) section
73 bool isDiscarded() const;
74
75 // True if this is an undefined weak symbol. This only works once
76 // all input files have been added.
77 bool isUndefWeak() const {
78 // See comment on lazy symbols for details.
79 return isWeak() && (isUndefined() || isLazy());
80 }
81
82 // Returns the symbol name.
83 StringRef getName() const { return Name; }
84
85 // Returns the file from which this symbol was created.
86 InputFile *getFile() const { return File; }
87
88 uint32_t getFlags() const { return Flags; }
89
90 InputChunk *getChunk() const;
91
92 // Indicates that the section or import for this symbol will be included in
93 // the final image.
94 bool isLive() const;
95
96 // Marks the symbol's InputChunk as Live, so that it will be included in the
97 // final image.
98 void markLive();
99
100 void setHidden(bool IsHidden);
101
102 // Get/set the index in the output symbol table. This is only used for
103 // relocatable output.
104 uint32_t getOutputSymbolIndex() const;
105 void setOutputSymbolIndex(uint32_t Index);
106
107 WasmSymbolType getWasmType() const;
108 bool isExported() const;
109
110 // True if the symbol was used for linking and thus need to be added to the
111 // output file's symbol table. This is true for all symbols except for
112 // unreferenced DSO symbols, lazy (archive) symbols, and bitcode symbols that
113 // are unreferenced except by other bitcode objects.
114 unsigned IsUsedInRegularObj : 1;
115
116 // True if ths symbol is explicity marked for export (i.e. via the -e/--export
117 // command line flag)
118 unsigned ForceExport : 1;
119
120 // False if LTO shouldn't inline whatever this symbol points to. If a symbol
121 // is overwritten after LTO, LTO shouldn't inline the symbol because it
122 // doesn't know the final contents of the symbol.
123 unsigned CanInline : 1;
124
125 // True if this symbol is specified by --trace-symbol option.
126 unsigned Traced : 1;
127
128 const WasmSignature* getSignature() const;
129
130 bool isInGOT() const { return GOTIndex != INVALID_INDEX; }
131
132 uint32_t getGOTIndex() const {
133 assert(GOTIndex != INVALID_INDEX);
134 return GOTIndex;
135 }
136
137 void setGOTIndex(uint32_t Index);
138 bool hasGOTIndex() const { return GOTIndex != INVALID_INDEX; }
139
140protected:
141 Symbol(StringRef Name, Kind K, uint32_t Flags, InputFile *F)
142 : IsUsedInRegularObj(false), ForceExport(false), CanInline(false),
143 Traced(false), Name(Name), SymbolKind(K), Flags(Flags), File(F),
144 Referenced(!Config->GcSections) {}
145
146 StringRef Name;
147 Kind SymbolKind;
148 uint32_t Flags;
149 InputFile *File;
150 uint32_t OutputSymbolIndex = INVALID_INDEX;
151 uint32_t GOTIndex = INVALID_INDEX;
152 bool Referenced;
153};

343// Wasm events are features that suspend the current execution and transfer the
344// control flow to a corresponding handler. Currently the only supported event
345// kind is exceptions.
346//
347// Event tags are values to distinguish different events. For exceptions, they
348// can be used to distinguish different language's exceptions, i.e., all C++
349// exceptions have the same tag. Wasm can generate code capable of doing
350// different handling actions based on the tag of caught exceptions.
351//
352// A single EventSymbol object represents a single tag. C++ exception event
353// symbol is a weak symbol generated in every object file in which exceptions
354// are used, and has name '__cpp_exception' for linking.
355class EventSymbol : public Symbol {
356public:
357 static bool classof(const Symbol *S) { return S->kind() == DefinedEventKind; }
358
359 const WasmEventType *getEventType() const { return EventType; }
360
361 // Get/set the event index
362 uint32_t getEventIndex() const;
363 void setEventIndex(uint32_t Index);
364 bool hasEventIndex() const;
365
366 const WasmSignature *Signature;
367
368protected:
369 EventSymbol(StringRef Name, Kind K, uint32_t Flags, InputFile *F,
370 const WasmEventType *EventType, const WasmSignature *Sig)
371 : Symbol(Name, K, Flags, F), Signature(Sig), EventType(EventType) {}
372
373 const WasmEventType *EventType;
374 uint32_t EventIndex = INVALID_INDEX;
375};

387// LazySymbol represents a symbol that is not yet in the link, but we know where
388// to find it if needed. If the resolver finds both Undefined and Lazy for the
389// same name, it will ask the Lazy to load a file.
390//
391// A special complication is the handling of weak undefined symbols. They should
392// not load a file, but we have to remember we have seen both the weak undefined
393// and the lazy. We represent that with a lazy symbol with a weak binding. This
394// means that code looking for undefined symbols normally also has to take lazy
395// symbols into consideration.
396class LazySymbol : public Symbol {
397public:
398 LazySymbol(StringRef Name, uint32_t Flags, InputFile *File,
399 const llvm::object::Archive::Symbol &Sym)
400 : Symbol(Name, LazyKind, Flags, File), ArchiveSymbol(Sym) {}
401
402 static bool classof(const Symbol *S) { return S->kind() == LazyKind; }
403 void fetch();
404
405 // Lazy symbols can have a signature because they can replace an
406 // UndefinedFunction which which case we need to be able to preserve the
407 // signture.
408 // TODO(sbc): This repetition of the signature field is inelegant. Revisit
409 // the use of class hierarchy to represent symbol taxonomy.
410 const WasmSignature *Signature = nullptr;
411
412private:
413 llvm::object::Archive::Symbol ArchiveSymbol;
414};

416// linker-generated symbols
417struct WasmSym {
418 // __stack_pointer
419 // Global that holds the address of the top of the explicit value stack in
420 // linear memory.
421 static GlobalSymbol *StackPointer;
422
423 // __data_end
424 // Symbol marking the end of the data and bss.
425 static DefinedData *DataEnd;
426
427 // __heap_base
428 // Symbol marking the end of the data, bss and explicit stack. Any linear
429 // memory following this address is not used by the linked code and can
430 // therefore be used as a backing store for brk()/malloc() implementations.
431 static DefinedData *HeapBase;
432
433 // __wasm_call_ctors
434 // Function that directly calls all ctors in priority order.
435 static DefinedFunction *CallCtors;
436
437 // __wasm_apply_relocs
438 // Function that applies relocations to data segment post-instantiation.
439 static DefinedFunction *ApplyRelocs;
440
441 // __dso_handle
442 // Symbol used in calls to __cxa_atexit to determine current DLL
443 static DefinedData *DsoHandle;
444
445 // __table_base
446 // Used in PIC code for offset of indirect function table
447 static UndefinedGlobal *TableBase;
448
449 // __memory_base
450 // Used in PIC code for offset of global data
451 static UndefinedGlobal *MemoryBase;
452};

454// A buffer class that is large enough to hold any Symbol-derived
455// object. We allocate memory using this class and instantiate a symbol
456// using the placement new.
457union SymbolUnion {
458 alignas(DefinedFunction) char A[sizeof(DefinedFunction)];
459 alignas(DefinedData) char B[sizeof(DefinedData)];
460 alignas(DefinedGlobal) char C[sizeof(DefinedGlobal)];
461 alignas(DefinedEvent) char D[sizeof(DefinedEvent)];
462 alignas(LazySymbol) char E[sizeof(LazySymbol)];
463 alignas(UndefinedFunction) char F[sizeof(UndefinedFunction)];
464 alignas(UndefinedData) char G[sizeof(UndefinedData)];
465 alignas(UndefinedGlobal) char H[sizeof(UndefinedGlobal)];
466 alignas(SectionSymbol) char I[sizeof(SectionSymbol)];
467};

472template <typename T, typename... ArgT>
473T *replaceSymbol(Symbol *S, ArgT &&... Arg) {
474 static_assert(std::is_trivially_destructible<T>(),
475 "Symbol types must be trivially destructible");
476 static_assert(sizeof(T) <= sizeof(SymbolUnion), "SymbolUnion too small");
477 static_assert(alignof(T) <= alignof(SymbolUnion),
478 "SymbolUnion not aligned enough");
479 assert(static_cast<Symbol *>(static_cast<T *>(nullptr)) == nullptr &&
480 "Not a Symbol");
481
482 Symbol SymCopy = *S;
483
484 T *S2 = new (S) T(std::forward<ArgT>(Arg)...);
485 S2->IsUsedInRegularObj = SymCopy.IsUsedInRegularObj;
486 S2->ForceExport = SymCopy.ForceExport;
487 S2->CanInline = SymCopy.CanInline;
488 S2->Traced = SymCopy.Traced;
489
490 // Print out a log message if --trace-symbol was specified.
491 // This is for debugging.
492 if (S2->Traced)
493 printTraceSymbol(S2);
494
495 return S2;
496}
