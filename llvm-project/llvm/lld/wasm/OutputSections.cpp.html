26static StringRef sectionTypeToString(uint32_t SectionType) {
27 switch (SectionType) {
28 case WASM_SEC_CUSTOM:
29 return "CUSTOM";
30 case WASM_SEC_TYPE:
31 return "TYPE";
32 case WASM_SEC_IMPORT:
33 return "IMPORT";
34 case WASM_SEC_FUNCTION:
35 return "FUNCTION";
36 case WASM_SEC_TABLE:
37 return "TABLE";
38 case WASM_SEC_MEMORY:
39 return "MEMORY";
40 case WASM_SEC_GLOBAL:
41 return "GLOBAL";
42 case WASM_SEC_EVENT:
43 return "EVENT";
44 case WASM_SEC_EXPORT:
45 return "EXPORT";
46 case WASM_SEC_START:
47 return "START";
48 case WASM_SEC_ELEM:
49 return "ELEM";
50 case WASM_SEC_CODE:
51 return "CODE";
52 case WASM_SEC_DATA:
53 return "DATA";
54 case WASM_SEC_DATACOUNT:
55 return "DATACOUNT";
56 default:
57 fatal("invalid section type");
58 }
59}

61// Returns a string, e.g. "FUNCTION(.text)".
62std::string lld::toString(const OutputSection &Sec) {
63 if (!Sec.Name.empty())
64 return (Sec.getSectionName() + "(" + Sec.Name + ")").str();
65 return Sec.getSectionName();

72void OutputSection::createHeader(size_t BodySize) {
73 raw_string_ostream OS(Header);
74 debugWrite(OS.tell(), "section type [" + getSectionName() + "]");
75 encodeULEB128(Type, OS);
76 writeUleb128(OS, BodySize, "section size");
77 OS.flush();
78 log("createHeader: " + toString(*this) + " body=" + Twine(BodySize) +
79 " total=" + Twine(getSize()));
80}

82void CodeSection::finalizeContents() {
83 raw_string_ostream OS(CodeSectionHeader);
84 writeUleb128(OS, Functions.size(), "function count");
85 OS.flush();
86 BodySize = CodeSectionHeader.size();
87
88 for (InputFunction *Func : Functions) {
89 Func->OutputOffset = BodySize;
90 Func->calculateSize();
91 BodySize += Func->getSize();
92 }
93
94 createHeader(BodySize);
95}

97void CodeSection::writeTo(uint8_t *Buf) {
98 log("writing " + toString(*this));
99 log(" size=" + Twine(getSize()));
100 log(" headersize=" + Twine(Header.size()));
101 log(" codeheadersize=" + Twine(CodeSectionHeader.size()));
102 Buf += Offset;
103
104 // Write section header
105 memcpy(Buf, Header.data(), Header.size());
106 Buf += Header.size();
107
108 // Write code section headers
109 memcpy(Buf, CodeSectionHeader.data(), CodeSectionHeader.size());
110
111 // Write code section bodies
112 for (const InputChunk *Chunk : Functions)
113 Chunk->writeTo(Buf);
114}

128void DataSection::finalizeContents() {
129 raw_string_ostream OS(DataSectionHeader);
130
131 writeUleb128(OS, Segments.size(), "data segment count");
132 OS.flush();
133 BodySize = DataSectionHeader.size();
134
135 for (OutputSegment *Segment : Segments) {
136 raw_string_ostream OS(Segment->Header);
137 writeUleb128(OS, 0, "memory index");
138 WasmInitExpr InitExpr;
139 if (Config->Pic) {
140 assert(Segments.size() <= 1 &&
141 "Currenly only a single data segment is supported in PIC mode");
142 InitExpr.Opcode = WASM_OPCODE_GLOBAL_GET;
143 InitExpr.Value.Global = WasmSym::MemoryBase->getGlobalIndex();
144 } else {
145 InitExpr.Opcode = WASM_OPCODE_I32_CONST;
146 InitExpr.Value.Int32 = Segment->StartVA;
147 }
148 writeInitExpr(OS, InitExpr);
149 writeUleb128(OS, Segment->Size, "segment size");
150 OS.flush();
151
152 Segment->SectionOffset = BodySize;
153 BodySize += Segment->Header.size() + Segment->Size;
154 log("Data segment: size=" + Twine(Segment->Size) + ", startVA=" +
155 Twine::utohexstr(Segment->StartVA) + ", name=" + Segment->Name);
156
157 for (InputSegment *InputSeg : Segment->InputSegments)
158 InputSeg->OutputOffset = Segment->SectionOffset + Segment->Header.size() +
159 InputSeg->OutputSegmentOffset;
160 }
161
162 createHeader(BodySize);
163}

165void DataSection::writeTo(uint8_t *Buf) {
166 log("writing " + toString(*this) + " size=" + Twine(getSize()) +
167 " body=" + Twine(BodySize));
168 Buf += Offset;
169
170 // Write section header
171 memcpy(Buf, Header.data(), Header.size());
172 Buf += Header.size();
173
174 // Write data section headers
175 memcpy(Buf, DataSectionHeader.data(), DataSectionHeader.size());
176
177 for (const OutputSegment *Segment : Segments) {
178 // Write data segment header
179 uint8_t *SegStart = Buf + Segment->SectionOffset;
180 memcpy(SegStart, Segment->Header.data(), Segment->Header.size());
181
182 // Write segment data payload
183 for (const InputChunk *Chunk : Segment->InputSegments)
184 Chunk->writeTo(Buf);
185 }
186}

188uint32_t DataSection::numRelocations() const {
189 uint32_t Count = 0;
190 for (const OutputSegment *Seg : Segments)
191 for (const InputChunk *InputSeg : Seg->InputSegments)

193 return Count;

202void CustomSection::finalizeContents() {
203 raw_string_ostream OS(NameData);
204 encodeULEB128(Name.size(), OS);
205 OS << Name;
206 OS.flush();
207
208 for (InputSection *Section : InputSections) {
209 Section->OutputOffset = PayloadSize;
210 Section->OutputSec = this;
211 PayloadSize += Section->getSize();
212 }
213
214 createHeader(PayloadSize + NameData.size());
215}
216
217void CustomSection::writeTo(uint8_t *Buf) {
218 log("writing " + toString(*this) + " size=" + Twine(getSize()) +
219 " chunks=" + Twine(InputSections.size()));
220
221 assert(Offset);
222 Buf += Offset;
223
224 // Write section header
225 memcpy(Buf, Header.data(), Header.size());
226 Buf += Header.size();
227 memcpy(Buf, NameData.data(), NameData.size());
228 Buf += NameData.size();
229
230 // Write custom sections payload
231 for (const InputSection *Section : InputSections)
232 Section->writeTo(Buf);
233}

238 Count += InputSect->NumRelocations();
239 return Count;
240}
