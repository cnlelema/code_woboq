36WasmSymbolType Symbol::getWasmType() const {
37 if (isa<FunctionSymbol>(this))
38 return WASM_SYMBOL_TYPE_FUNCTION;
39 if (isa<DataSymbol>(this))
40 return WASM_SYMBOL_TYPE_DATA;
41 if (isa<GlobalSymbol>(this))
42 return WASM_SYMBOL_TYPE_GLOBAL;
43 if (isa<EventSymbol>(this))
44 return WASM_SYMBOL_TYPE_EVENT;
45 if (isa<SectionSymbol>(this) || isa<OutputSectionSymbol>(this))
46 return WASM_SYMBOL_TYPE_SECTION;
47 llvm_unreachable("invalid symbol kind");
48}

50const WasmSignature *Symbol::getSignature() const {
51 if (auto* F = dyn_cast<FunctionSymbol>(this))
52 return F->Signature;
53 if (auto *L = dyn_cast<LazySymbol>(this))
54 return L->Signature;
55 return nullptr;
56}

58InputChunk *Symbol::getChunk() const {
59 if (auto *F = dyn_cast<DefinedFunction>(this))
60 return F->Function;
61 if (auto *D = dyn_cast<DefinedData>(this))
62 return D->Segment;
63 return nullptr;
64}

66bool Symbol::isDiscarded() const {
67 if (InputChunk *C = getChunk())
68 return C->Discarded;
69 return false;
70}

72bool Symbol::isLive() const {
73 if (auto *G = dyn_cast<DefinedGlobal>(this))
74 return G->Global->Live;
75 if (auto *E = dyn_cast<DefinedEvent>(this))
76 return E->Event->Live;
77 if (InputChunk *C = getChunk())
78 return C->Live;
79 return Referenced;
80}
81
82void Symbol::markLive() {
83 assert(!isDiscarded());
84 if (auto *G = dyn_cast<DefinedGlobal>(this))
85 G->Global->Live = true;
86 if (auto *E = dyn_cast<DefinedEvent>(this))
87 E->Event->Live = true;
88 if (InputChunk *C = getChunk())
89 C->Live = true;
90 Referenced = true;
91}

105void Symbol::setGOTIndex(uint32_t Index) {
106 LLVM_DEBUG(dbgs() << "setGOTIndex " << Name << " -> " << Index << "\n");
107 assert(GOTIndex == INVALID_INDEX);
108 // Any symbol that is assigned a GOT entry must be exported othewise the
109 // dynamic linker won't be able create the entry that contains it.
110 ForceExport = true;
111 GOTIndex = Index;
112}

123 return (Flags & WASM_SYMBOL_VISIBILITY_MASK) == WASM_SYMBOL_VISIBILITY_HIDDEN;

126void Symbol::setHidden(bool IsHidden) {
127 LLVM_DEBUG(dbgs() << "setHidden: " << Name << " -> " << IsHidden << "\n");
128 Flags &= ~WASM_SYMBOL_VISIBILITY_MASK;
129 if (IsHidden)
130 Flags |= WASM_SYMBOL_VISIBILITY_HIDDEN;
131 else
132 Flags |= WASM_SYMBOL_VISIBILITY_DEFAULT;
133}
134
135bool Symbol::isExported() const {
136 if (!isDefined() || isLocal())
137 return false;
138
139 if (ForceExport || Config->ExportAll)
140 return true;
141
142 if (Config->ExportDynamic && !isHidden())
143 return true;
144
145 return Flags & WASM_SYMBOL_EXPORTED;
146}

180void FunctionSymbol::setTableIndex(uint32_t Index) {
181 // For imports, we set the table index here on the Symbol; for defined
182 // functions we set the index on the InputFunction so that we don't export
183 // the same thing twice (keeps the table size down).
184 if (auto *F = dyn_cast<DefinedFunction>(this)) {
185 F->Function->setTableIndex(Index);
186 return;
187 }
188 LLVM_DEBUG(dbgs() << "setTableIndex " << Name << " -> " << Index << "\n");
189 assert(TableIndex == INVALID_INDEX);
190 TableIndex = Index;
191}

199uint32_t DefinedData::getVirtualAddress() const {
200 LLVM_DEBUG(dbgs() << "getVirtualAddress: " << getName() << "\n");

202 return Segment->OutputSeg->StartVA + Segment->OutputSegmentOffset + Offset;
203 return Offset;
204}

284std::string lld::maybeDemangleSymbol(StringRef Name) {
285 if (Config->Demangle)
286 if (Optional<std::string> S = demangleItanium(Name))
287 return *S;
288 return Name;
289}

291std::string lld::toString(wasm::Symbol::Kind Kind) {
292 switch (Kind) {
293 case wasm::Symbol::DefinedFunctionKind:
294 return "DefinedFunction";
295 case wasm::Symbol::DefinedDataKind:
296 return "DefinedData";
297 case wasm::Symbol::DefinedGlobalKind:
298 return "DefinedGlobal";
299 case wasm::Symbol::DefinedEventKind:
300 return "DefinedEvent";
301 case wasm::Symbol::UndefinedFunctionKind:
302 return "UndefinedFunction";
303 case wasm::Symbol::UndefinedDataKind:
304 return "UndefinedData";
305 case wasm::Symbol::UndefinedGlobalKind:
306 return "UndefinedGlobal";
307 case wasm::Symbol::LazyKind:
308 return "LazyKind";
309 case wasm::Symbol::SectionKind:
310 return "SectionKind";
311 case wasm::Symbol::OutputSectionKind:
312 return "OutputSectionKind";
313 }
314 llvm_unreachable("invalid symbol kind");
315}

322// Print out a log message for --trace-symbol.
323void lld::wasm::printTraceSymbol(Symbol *Sym) {
324 // Undefined symbols are traced via printTraceSymbolUndefined
325 if (Sym->isUndefined())
326 return;
327
328 std::string S;
329 if (Sym->isLazy())
330 S = ": lazy definition of ";
331 else
332 S = ": definition of ";
333
334 message(toString(Sym->getFile()) + S + Sym->getName());
335}
