34Defined *SymbolTable::ensureSymbolWillBeInDynsym(Symbol* Original) {
35 assert(!Original->includeInDynsym() && "Already included in dynsym?");
36 assert(Original->isFunc() && "This should only be used for functions");
37 // Hack: Add a new global symbol with a unique name so that we can use
38 // a dynamic relocation against it.
39 // TODO: should it be possible to add STB_LOCAL symbols to .dynsymtab?
40 // create a unique name:
41
42 auto it = LocalSymbolsForDynsym.find(Original);
43 if (it != LocalSymbolsForDynsym.end()) {
44 if (Config->VerboseCapRelocs)
45 message("Reusing existing 'fake' symbol " + toString(*it->second) +
46 " to allow relocation against " + verboseToString(Original));
47 return it->second;
48 }
49
50 std::string UniqueName = ("__cheri_fnptr_" + Original->getName()).str();
51 for (int i = 2; Symtab->find(UniqueName); i++) {
52 UniqueName = ("__cheri_fnptr" + Twine(i) + "_" + Original->getName()).str();
53 }
54 auto LocalDef = cast<Defined>(Original);
55 Defined* NewSym = cast<Defined>(Symtab->addSymbol(Defined{LocalDef->File,
56 Saver.save(UniqueName), llvm::ELF::STB_GLOBAL, llvm::ELF::STV_HIDDEN,
57 LocalDef->Type, LocalDef->Value, LocalDef->Size, LocalDef->Section}));
58
59 assert(NewSym->isFunc() && "This should only be used for functions");
60 // TODO: would be nice to just set this on Sym, but we can't have
61 // STB_LOCAL symbols in .dynsym
62 NewSym->UsedByDynReloc = true;
63 NewSym->IsPreemptible = false;
64 if (Config->VerboseCapRelocs)
65 message("Adding new symbol " + toString(*NewSym) +
66 " to allow relocation against " + verboseToString(Original));
67 LocalSymbolsForDynsym[Original] = NewSym;
68 return NewSym;
69}

72void SymbolTable::wrap(Symbol *Sym, Symbol *Real, Symbol *Wrap) {
73 // Swap symbols as instructed by -wrap.
74 int &Idx1 = SymMap[CachedHashStringRef(Sym->getName())];
75 int &Idx2 = SymMap[CachedHashStringRef(Real->getName())];
76 int &Idx3 = SymMap[CachedHashStringRef(Wrap->getName())];
77
78 Idx2 = Idx1;
79 Idx1 = Idx3;
80
81 // Now renaming is complete. No one refers Real symbol. We could leave
82 // Real as-is, but if Real is written to the symbol table, that may
83 // contain irrelevant values. So, we copy all values from Sym to Real.
84 StringRef S = Real->getName();
85 memcpy(Real, Sym, sizeof(SymbolUnion));
86 Real->setName(S);
87}

89// Find an existing symbol or create a new one.
90Symbol *SymbolTable::insert(StringRef Name) {
91 // <name>@@<version> means the symbol is the default version. In that
92 // case <name>@@<version> will be used to resolve references to <name>.
93 //
94 // Since this is a hot path, the following string search code is
95 // optimized for speed. StringRef::find(char) is much faster than
96 // StringRef::find(StringRef).
97 size_t Pos = Name.find('@');
98 if (Pos != StringRef::npos && Pos + 1 < Name.size() && Name[Pos + 1] == '@')
99 Name = Name.take_front(Pos);
100
101 auto P = SymMap.insert({CachedHashStringRef(Name), (int)SymVector.size()});
102 int &SymIndex = P.first->second;
103 bool IsNew = P.second;
104
105 if (!IsNew)
106 return SymVector[SymIndex];
107
108 Symbol *Sym = reinterpret_cast<Symbol *>(make<SymbolUnion>());
109 SymVector.push_back(Sym);
110
111 Sym->setName(Name);
112 Sym->SymbolKind = Symbol::PlaceholderKind;
113 Sym->VersionId = Config->DefaultSymbolVersion;
114 Sym->Visibility = STV_DEFAULT;
115 Sym->IsUsedInRegularObj = false;
116 Sym->ExportDynamic = false;
117 Sym->CanInline = true;
118 Sym->ScriptDefined = false;
119 Sym->Partition = 1;
120 return Sym;
121}

123Symbol *SymbolTable::addSymbol(const Symbol &New) {
124 Symbol *Sym = Symtab->insert(New.getName());
125 Sym->resolve(New);
126 return Sym;
127}

129Symbol *SymbolTable::find(StringRef Name) {
130 auto It = SymMap.find(CachedHashStringRef(Name));
131 if (It == SymMap.end())
132 return nullptr;
133 Symbol *Sym = SymVector[It->second];
134 if (Sym->isPlaceholder())
135 return nullptr;
136 return Sym;
137}

139// Initialize DemangledSyms with a map from demangled symbols to symbol
140// objects. Used to handle "extern C++" directive in version scripts.
141//
142// The map will contain all demangled symbols. That can be very large,
143// and in LLD we generally want to avoid do anything for each symbol.
144// Then, why are we doing this? Here's why.
145//
146// Users can use "extern C++ {}" directive to match against demangled
147// C++ symbols. For example, you can write a pattern such as
148// "llvm::*::foo(int, ?)". Obviously, there's no way to handle this
149// other than trying to match a pattern against all demangled symbols.
150// So, if "extern C++" feature is used, we need to demangle all known
151// symbols.
152StringMap<std::vector<Symbol *>> &SymbolTable::getDemangledSyms() {
153 if (!DemangledSyms) {
154 DemangledSyms.emplace();
155 for (Symbol *Sym : SymVector) {
156 if (!Sym->isDefined() && !Sym->isCommon())
157 continue;
158 if (Optional<std::string> S = demangleItanium(Sym->getName()))
159 (*DemangledSyms)[*S].push_back(Sym);
160 else
161 (*DemangledSyms)[Sym->getName()].push_back(Sym);
162 }
163 }
164 return *DemangledSyms;
165}

167std::vector<Symbol *> SymbolTable::findByVersion(SymbolVersion Ver) {
168 if (Ver.IsExternCpp)
169 return getDemangledSyms().lookup(Ver.Name);
170 if (Symbol *B = find(Ver.Name))
171 if (B->isDefined() || B->isCommon())
172 return {B};
173 return {};
174}
175
176std::vector<Symbol *> SymbolTable::findAllByVersion(SymbolVersion Ver) {
177 std::vector<Symbol *> Res;
178 StringMatcher M(Ver.Name);
179
180 if (Ver.IsExternCpp) {
181 for (auto &P : getDemangledSyms())
182 if (M.match(P.first()))
183 Res.insert(Res.end(), P.second.begin(), P.second.end());
184 return Res;
185 }
186
187 for (Symbol *Sym : SymVector)
188 if ((Sym->isDefined() || Sym->isCommon()) && M.match(Sym->getName()))
189 Res.push_back(Sym);
190 return Res;
191}

193// If there's only one anonymous version definition in a version
194// script file, the script does not actually define any symbol version,
195// but just specifies symbols visibilities.
196void SymbolTable::handleAnonymousVersion() {
197 for (SymbolVersion &Ver : Config->VersionScriptGlobals)
198 assignExactVersion(Ver, VER_NDX_GLOBAL, "global");
199 for (SymbolVersion &Ver : Config->VersionScriptGlobals)
200 assignWildcardVersion(Ver, VER_NDX_GLOBAL);
201 for (SymbolVersion &Ver : Config->VersionScriptLocals)
202 assignExactVersion(Ver, VER_NDX_LOCAL, "local");
203 for (SymbolVersion &Ver : Config->VersionScriptLocals)
204 assignWildcardVersion(Ver, VER_NDX_LOCAL);
205}

207// Handles -dynamic-list.
208void SymbolTable::handleDynamicList() {
209 for (SymbolVersion &Ver : Config->DynamicList) {
210 std::vector<Symbol *> Syms;
211 if (Ver.HasWildcard)
212 Syms = findAllByVersion(Ver);
213 else
214 Syms = findByVersion(Ver);
215
216 for (Symbol *B : Syms) {
217 if (!Config->Shared)
218 B->ExportDynamic = true;
219 else if (B->includeInDynsym())
220 B->IsPreemptible = true;
221 }
222 }
223}

225// Set symbol versions to symbols. This function handles patterns
226// containing no wildcard characters.
227void SymbolTable::assignExactVersion(SymbolVersion Ver, uint16_t VersionId,
228 StringRef VersionName) {
229 if (Ver.HasWildcard)
230 return;
231
232 // Get a list of symbols which we need to assign the version to.
233 std::vector<Symbol *> Syms = findByVersion(Ver);
234 if (Syms.empty()) {
235 if (!Config->UndefinedVersion)
236 error("version script assignment of '" + VersionName + "' to symbol '" +
237 Ver.Name + "' failed: symbol not defined");
238 return;
239 }
240
241 // Assign the version.
242 for (Symbol *Sym : Syms) {
243 // Skip symbols containing version info because symbol versions
244 // specified by symbol names take precedence over version scripts.
245 // See parseSymbolVersion().
246 if (Sym->getName().contains('@'))
247 continue;
248
249 if (Sym->VersionId != Config->DefaultSymbolVersion &&
250 Sym->VersionId != VersionId)
251 error("duplicate symbol '" + Ver.Name + "' in version script");
252 Sym->VersionId = VersionId;
253 }
254}
255
256void SymbolTable::assignWildcardVersion(SymbolVersion Ver, uint16_t VersionId) {
257 if (!Ver.HasWildcard)
258 return;
259
260 // Exact matching takes precendence over fuzzy matching,
261 // so we set a version to a symbol only if no version has been assigned
262 // to the symbol. This behavior is compatible with GNU.
263 for (Symbol *B : findAllByVersion(Ver))
264 if (B->VersionId == Config->DefaultSymbolVersion)
265 B->VersionId = VersionId;
266}

268// This function processes version scripts by updating VersionId
269// member of symbols.
270void SymbolTable::scanVersionScript() {
271 // Handle edge cases first.
272 handleAnonymousVersion();
273 handleDynamicList();
274
275 // Now we have version definitions, so we need to set version ids to symbols.
276 // Each version definition has a glob pattern, and all symbols that match
277 // with the pattern get that version.
278
279 // First, we assign versions to exact matching symbols,
280 // i.e. version definitions not containing any glob meta-characters.
281 for (VersionDefinition &V : Config->VersionDefinitions)
282 for (SymbolVersion &Ver : V.Globals)
283 assignExactVersion(Ver, V.Id, V.Name);
284
285 // Next, we assign versions to fuzzy matching symbols,
286 // i.e. version definitions containing glob meta-characters.
287 // Note that because the last match takes precedence over previous matches,
288 // we iterate over the definitions in the reverse order.
289 for (VersionDefinition &V : llvm::reverse(Config->VersionDefinitions))
290 for (SymbolVersion &Ver : V.Globals)
291 assignWildcardVersion(Ver, V.Id);
292
293 // Symbol themselves might know their versions because symbols
294 // can contain versions in the form of <name>@<version>.
295 // Let them parse and update their names to exclude version suffix.
296 for (Symbol *Sym : SymVector)
297 Sym->parseSymbolVersion();
298}
