60// The root class of input files.
61class InputFile {
62public:
63 enum Kind {
64 ObjKind,
65 SharedKind,
66 LazyObjKind,
67 ArchiveKind,
68 BitcodeKind,
69 BinaryKind,
70 };
71
72 Kind kind() const { return FileKind; }
73
74 bool isElf() const {
75 Kind K = kind();
76 return K == ObjKind || K == SharedKind;
77 }
78
79 StringRef getName() const { return MB.getBufferIdentifier(); }
80 MemoryBufferRef MB;
81
82 // Returns sections. It is a runtime error to call this function
83 // on files that don't have the notion of sections.
84 ArrayRef<InputSectionBase *> getSections() const {
85 assert(FileKind == ObjKind || FileKind == BinaryKind);
86 return Sections;
87 }
88
89 // Returns object file symbols. It is a runtime error to call this
90 // function on files of other types.
91 ArrayRef<Symbol *> getSymbols() { return getMutableSymbols(); }
92
93 MutableArrayRef<Symbol *> getMutableSymbols() {
94 assert(FileKind == BinaryKind || FileKind == ObjKind ||
95 FileKind == BitcodeKind);
96 return Symbols;
97 }
98
99 // Filename of .a which contained this file. If this file was
100 // not in an archive file, it is the empty string. We use this
101 // string for creating error messages.
102 std::string ArchiveName;
103
104 // If this is an architecture-specific file, the following members
105 // have ELF type (i.e. ELF{32,64}{LE,BE}) and target machine type.
106 ELFKind EKind = ELFNoneKind;
107 uint16_t EMachine = llvm::ELF::EM_NONE;
108 uint64_t EFlags = 0;
109 uint8_t OSABI = 0;
110 uint8_t ABIVersion = 0;
111
112 // Cache for toString(). Only toString() should use this member.
113 mutable std::string ToStringCache;
114
115 std::string getSrcMsg(const Symbol &Sym, const InputSectionBase &Sec,
116 uint64_t Offset);
117
118 // True if this is an argument for --just-symbols. Usually false.
119 bool JustSymbols = false;
120
121 // OutSecOff of .got2 in the current file. This is used by PPC32 -fPIC/-fPIE
122 // to compute offsets in PLT call stubs.
123 uint32_t PPC32Got2OutSecOff = 0;
124
125 // On PPC64 we need to keep track of which files contain small code model
126 // relocations that access the .toc section. To minimize the chance of a
127 // relocation overflow, files that do contain said relocations should have
128 // their .toc sections sorted closer to the .got section than files that do
129 // not contain any small code model relocations. Thats because the toc-pointer
130 // is defined to point at .got + 0x8000 and the instructions used with small
131 // code model relocations support immediates in the range [-0x8000, 0x7FFC],
132 // making the addressable range relative to the toc pointer
133 // [.got, .got + 0xFFFC].
134 bool PPC64SmallCodeModelTocRelocs = false;
135
136 // GroupId is used for --warn-backrefs which is an optional error
137 // checking feature. All files within the same --{start,end}-group or
138 // --{start,end}-lib get the same group ID. Otherwise, each file gets a new
139 // group ID. For more info, see checkDependency() in SymbolTable.cpp.
140 uint32_t GroupId;
141 static bool IsInGroup;
142 static uint32_t NextGroupId;
143
144 // Index of MIPS GOT built for this file.
145 llvm::Optional<size_t> MipsGotIndex;
146
147 std::vector<Symbol *> Symbols;
148
149protected:
150 InputFile(Kind K, MemoryBufferRef M);
151 std::vector<InputSectionBase *> Sections;
152
153private:
154 const Kind FileKind;
155};

157class ELFFileBase : public InputFile {
158public:
159 ELFFileBase(Kind K, MemoryBufferRef M);
160 static bool classof(const InputFile *F) { return F->isElf(); }
161
162 template <typename ELFT> llvm::object::ELFFile<ELFT> getObj() const {
163 return check(llvm::object::ELFFile<ELFT>::create(MB.getBuffer()));
164 }
165
166 StringRef getStringTable() const { return StringTable; }
167
168 template <typename ELFT> typename ELFT::SymRange getELFSyms() const {
169 return typename ELFT::SymRange(
170 reinterpret_cast<const typename ELFT::Sym *>(ELFSyms), NumELFSyms);
171 }
172 template <typename ELFT> typename ELFT::SymRange getGlobalELFSyms() const {
173 return getELFSyms<ELFT>().slice(FirstGlobal);
174 }
175
176protected:
177 // Initializes this class's member variables.
178 template <typename ELFT> void init();
179
180 const void *ELFSyms = nullptr;
181 size_t NumELFSyms = 0;
182 uint32_t FirstGlobal = 0;
183 StringRef StringTable;
184};

186// .o file.
187template <class ELFT> class ObjFile : public ELFFileBase {
188 using Elf_Rel = typename ELFT::Rel;
189 using Elf_Rela = typename ELFT::Rela;
190 using Elf_Sym = typename ELFT::Sym;
191 using Elf_Shdr = typename ELFT::Shdr;
192 using Elf_Word = typename ELFT::Word;
193 using Elf_CGProfile = typename ELFT::CGProfile;
194
195public:
196 static bool classof(const InputFile *F) { return F->kind() == ObjKind; }
197
198 llvm::object::ELFFile<ELFT> getObj() const {
199 return this->ELFFileBase::getObj<ELFT>();
200 }
201
202 ArrayRef<Symbol *> getLocalSymbols();
203 ArrayRef<Symbol *> getGlobalSymbols();
204
205 ObjFile(MemoryBufferRef M, StringRef ArchiveName) : ELFFileBase(ObjKind, M) {
206 this->ArchiveName = ArchiveName;
207 }
208
209 void parse(bool IgnoreComdats = false);
210
211 StringRef getShtGroupSignature(ArrayRef<Elf_Shdr> Sections,
212 const Elf_Shdr &Sec);
213
214 Symbol &getSymbol(uint32_t SymbolIndex) const {
215 if (SymbolIndex >= this->Symbols.size())
216 fatal(toString(this) + ": invalid symbol index");
217 return *this->Symbols[SymbolIndex];
218 }
219
220 uint32_t getSectionIndex(const Elf_Sym &Sym) const;
221
222 template <typename RelT> Symbol &getRelocTargetSym(const RelT &Rel) const {
223 uint32_t SymIndex = Rel.getSymbol(Config->IsMips64EL);
224 return getSymbol(SymIndex);
225 }
226
227 llvm::Optional<llvm::DILineInfo> getDILineInfo(const InputSectionBase *, uint64_t);
228 llvm::Optional<std::pair<std::string, unsigned>> getVariableLoc(StringRef Name);
229
230 // MIPS GP0 value defined by this file. This value represents the gp value
231 // used to create the relocatable object and required to support
232 // R_MIPS_GPREL16 / R_MIPS_GPREL32 relocations.
233 uint32_t MipsGp0 = 0;
234
235 uint32_t AndFeatures = 0;
236
237 // Name of source file obtained from STT_FILE symbol value,
238 // or empty string if there is no such symbol in object file
239 // symbol table.
240 StringRef SourceFile;
241
242 // True if the file defines functions compiled with
243 // -fsplit-stack. Usually false.
244 bool SplitStack = false;
245
246 // True if the file defines functions compiled with -fsplit-stack,
247 // but had one or more functions with the no_split_stack attribute.
248 bool SomeNoSplitStack = false;
249
250 // Pointer to this input file's .llvm_addrsig section, if it has one.
251 const Elf_Shdr *AddrsigSec = nullptr;
252
253 // SHT_LLVM_CALL_GRAPH_PROFILE table
254 ArrayRef<Elf_CGProfile> CGProfile;
255
256private:
257 void initializeSections(bool IgnoreComdats);
258 void initializeSymbols();
259 void initializeJustSymbols();
260 void initializeDwarf();
261 InputSectionBase *getRelocTarget(const Elf_Shdr &Sec);
262 InputSectionBase *createInputSection(const Elf_Shdr &Sec);
263 StringRef getSectionName(const Elf_Shdr &Sec);
264
265 bool shouldMerge(const Elf_Shdr &Sec);
266
267 // Each ELF symbol contains a section index which the symbol belongs to.
268 // However, because the number of bits dedicated for that is limited, a
269 // symbol can directly point to a section only when the section index is
270 // equal to or smaller than 65280.
271 //
272 // If an object file contains more than 65280 sections, the file must
273 // contain .symtab_shndx section. The section contains an array of
274 // 32-bit integers whose size is the same as the number of symbols.
275 // Nth symbol's section index is in the Nth entry of .symtab_shndx.
276 //
277 // The following variable contains the contents of .symtab_shndx.
278 // If the section does not exist (which is common), the array is empty.
279 ArrayRef<Elf_Word> ShndxTable;
280
281 // .shstrtab contents.
282 StringRef SectionStringTable;
283
284 // Debugging information to retrieve source file and line for error
285 // reporting. Linker may find reasonable number of errors in a
286 // single object file, so we cache debugging information in order to
287 // parse it only once for each object file we link.
288 std::unique_ptr<llvm::DWARFContext> Dwarf;
289 std::vector<const llvm::DWARFDebugLine::LineTable *> LineTables;
290 struct VarLoc {
291 const llvm::DWARFDebugLine::LineTable *LT;
292 unsigned File;
293 unsigned Line;
294 };
295 llvm::DenseMap<StringRef, VarLoc> VariableLoc;
296 llvm::once_flag InitDwarfLine;
297};

299// LazyObjFile is analogous to ArchiveFile in the sense that
300// the file contains lazy symbols. The difference is that
301// LazyObjFile wraps a single file instead of multiple files.
302//
303// This class is used for --start-lib and --end-lib options which
304// instruct the linker to link object files between them with the
305// archive file semantics.
306class LazyObjFile : public InputFile {
307public:
308 LazyObjFile(MemoryBufferRef M, StringRef ArchiveName,
309 uint64_t OffsetInArchive)
310 : InputFile(LazyObjKind, M), OffsetInArchive(OffsetInArchive) {
311 this->ArchiveName = ArchiveName;
312 }
313
314 static bool classof(const InputFile *F) { return F->kind() == LazyObjKind; }
315
316 template <class ELFT> void parse();
317 void fetch();
318
319private:
320 uint64_t OffsetInArchive;
321};

323// An ArchiveFile object represents a .a file.
324class ArchiveFile : public InputFile {
325public:
326 explicit ArchiveFile(std::unique_ptr<Archive> &&File);
327 static bool classof(const InputFile *F) { return F->kind() == ArchiveKind; }
328 void parse();
329
330 // Pulls out an object file that contains a definition for Sym and
331 // returns it. If the same file was instantiated before, this
332 // function does nothing (so we don't instantiate the same file
333 // more than once.)
334 void fetch(const Archive::Symbol &Sym);
335
336private:
337 std::unique_ptr<Archive> File;
338 llvm::DenseSet<uint64_t> Seen;
339};

365

370

374
375 // Used for --no-allow-shlib-undefined.
376 bool AllNeededIsKnown;
377
378 // Used for --as-needed
379 bool IsNeeded;
380};
