9// This file contains platform-independent functions to process relocations.
10// I'll describe the overview of this file here.
11//
12// Simple relocations are easy to handle for the linker. For example,
13// for R_X86_64_PC64 relocs, the linker just has to fix up locations
14// with the relative offsets to the target symbols. It would just be
15// reading records from relocation sections and applying them to output.
16//
17// But not all relocations are that easy to handle. For example, for
18// R_386_GOTOFF relocs, the linker has to create new GOT entries for
19// symbols if they don't exist, and fix up locations with GOT entry
20// offsets from the beginning of GOT section. So there is more than
21// fixing addresses in relocation processing.
22//
23// ELF defines a large number of complex relocations.
24//
25// The functions in this file analyze relocations and do whatever needs
26// to be done. It includes, but not limited to, the following.
27//
28// - create GOT/PLT entries
29// - create new relocations in .dynsym to let the dynamic linker resolve
30// them at runtime (since ELF supports dynamic linking, not all
31// relocations can be resolved at link-time)
32// - create COPY relocs and reserve space in .bss
33// - replace expensive relocs (in terms of runtime cost) with cheap ones
34// - error out infeasible combinations such as PIC and non-relative relocs
35//
36// Note that the functions in this file don't actually apply relocations
37// because it doesn't know about the output file nor the output file buffer.
38// It instead stores Relocation objects to InputSection's Relocations
39// vector to let it apply later in InputSection::writeTo.
40//
41//===----------------------------------------------------------------------===//

71static Optional<std::string> getLinkerScriptLocation(const Symbol &Sym) {
72 for (BaseCommand *Base : Script->SectionCommands)
73 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base))
74 if (Cmd->Sym == &Sym)
75 return Cmd->Location;
76 return None;

79// Construct a message in the following format.
80//
81// >>> defined in /home/alice/src/foo.o
82// >>> referenced by bar.c:12 (/home/alice/src/bar.c:12)
83// >>> /home/alice/src/bar.o:(.text+0x1)
84static std::string getLocation(const InputSectionBase &S, const Symbol &Sym,
85 uint64_t Off) {
86 std::string Msg = "\n>>> defined in ";
87 if (Sym.File)
88 Msg += toString(Sym.File);
89 else if (Optional<std::string> Loc = getLinkerScriptLocation(Sym))
90 Msg += *Loc;
91
92 Msg += "\n>>> referenced by ";
93 std::string Src = S.getSrcMsg(Sym, Off);
94 if (!Src.empty())
95 Msg += Src + "\n>>> ";
96 return Msg + S.getObjMsg(Off);
97}

101 return getLocation(S, Sym, Off);

105// Build a bitmask with one bit set for each RelExpr.
106//
107// Constexpr function arguments can't be used in static asserts, so we
108// use template arguments to build the mask.
109// But function template partial specializations don't exist (needed
110// for base case of the recursion), so we need a dummy struct.
111template <RelExpr... Exprs> struct RelExprMaskBuilder {
112 static inline std::pair<uint64_t, uint64_t> build() { return {0, 0}; }
113};
114
115// Specialization for recursive case.
116template <RelExpr Head, RelExpr... Tail>
117struct RelExprMaskBuilder<Head, Tail...> {
118 using Mask = std::pair<uint64_t, uint64_t>;
119 static inline Mask combine_masks(Mask a, Mask b) {
120 return {a.first | b.first, a.second | b.second};
121 }
122 static inline Mask build() {
123 Mask p{0, 0};
124 uint64_t &part = (Head > 63) ? p.first : p.second;
125 uint64_t shift = static_cast<uint64_t>(Head);
126 if (shift > 63)
127 shift -= 64;
128 assert(shift < 64);
129 part = (uint64_t(1) << shift);
130 return combine_masks(p, RelExprMaskBuilder<Tail...>::build());
131 }
132};

135// Return true if `Expr` is one of `Exprs`.
136// There are fewer than 64 RelExpr's, so we can represent any set of
137// RelExpr's as a constant bit mask and test for membership with a
138// couple cheap bitwise operations.
139template <RelExpr... Exprs> bool oneof(RelExpr Expr) {
140 auto mask = RelExprMaskBuilder<Exprs...>::build();
141 unsigned Bit = (unsigned)Expr;
142 assert(0 <= Bit && Bit < 128 &&
143 "RelExpr is too large for 128-bit mask!");
144 uint64_t &part = (Bit > 63) ? mask.first : mask.second;
145 if (Bit > 63)
146 Bit -= 64;
147 return (uint64_t(1) << Bit) & part;
148}

150// This function is similar to the `handleTlsRelocation`. MIPS does not
151// support any relaxations for TLS relocations so by factoring out MIPS
152// handling in to the separate function we can simplify the code and do not
153// pollute other `handleTlsRelocation` by MIPS `ifs` statements.
154// Mips has a custom MipsGotSection that handles the writing of GOT entries
155// without dynamic relocations.
156static unsigned handleMipsTlsRelocation(RelType Type, Symbol &Sym,
157 InputSectionBase &C, uint64_t Offset,
158 int64_t Addend, RelExpr Expr) {
159 if (Expr == R_MIPS_TLSLD) {
160 In.MipsGot->addTlsIndex(*C.File);
161 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
162 return 1;
163 }
164 if (Expr == R_MIPS_TLSGD) {
165 In.MipsGot->addDynTlsEntry(*C.File, Sym);
166 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
167 return 1;
168 }
169 if (Expr == R_MIPS_CHERI_CAPTAB_TLSLD) {
170 In.CheriCapTable->addTlsIndex();
171 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
172 return 1;
173 }
174 if (Expr == R_MIPS_CHERI_CAPTAB_TLSGD) {
175 In.CheriCapTable->addDynTlsEntry(Sym);
176 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
177 return 1;
178 }
179 if (Expr == R_MIPS_CHERI_CAPTAB_TPREL) {
180 In.CheriCapTable->addTlsEntry(Sym);
181 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
182 return 1;
183 }
184 return 0;
185}

187// This function is similar to the `handleMipsTlsRelocation`. ARM also does not
188// support any relaxations for TLS relocations. ARM is logically similar to Mips
189// in how it handles TLS, but Mips uses its own custom GOT which handles some
190// of the cases that ARM uses GOT relocations for.
191//
192// We look for TLS global dynamic and local dynamic relocations, these may
193// require the generation of a pair of GOT entries that have associated
194// dynamic relocations. When the results of the dynamic relocations can be
195// resolved at static link time we do so. This is necessary for static linking
196// as there will be no dynamic loader to resolve them at load-time.
197//
198// The pair of GOT entries created are of the form
199// GOT[e0] Module Index (Used to find pointer to TLS block at run-time)
200// GOT[e1] Offset of symbol in TLS block
201static unsigned handleARMTlsRelocation(RelType Type, Symbol &Sym,
202 InputSectionBase &C, uint64_t Offset,
203 int64_t Addend, RelExpr Expr) {
204 // The Dynamic TLS Module Index Relocation for a symbol defined in an
205 // executable is always 1. If the target Symbol is not preemptible then
206 // we know the offset into the TLS block at static link time.
207 bool NeedDynId = Sym.IsPreemptible || Config->Shared;
208 bool NeedDynOff = Sym.IsPreemptible;
209
210 auto AddTlsReloc = [&](uint64_t Off, RelType Type, Symbol *Dest, bool Dyn) {
211 if (Dyn)
212 Main->RelaDyn->addReloc(Type, In.Got, Off, Dest);
213 else
214 In.Got->Relocations.push_back({R_ABS, Type, Off, 0, Dest});
215 };
216
217 // Local Dynamic is for access to module local TLS variables, while still
218 // being suitable for being dynamically loaded via dlopen.
219 // GOT[e0] is the module index, with a special value of 0 for the current
220 // module. GOT[e1] is unused. There only needs to be one module index entry.
221 if (Expr == R_TLSLD_PC && In.Got->addTlsIndex()) {
222 AddTlsReloc(In.Got->getTlsIndexOff(), Target->TlsModuleIndexRel,
223 NeedDynId ? nullptr : &Sym, NeedDynId);
224 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
225 return 1;
226 }
227
228 // Global Dynamic is the most general purpose access model. When we know
229 // the module index and offset of symbol in TLS block we can fill these in
230 // using static GOT relocations.
231 if (Expr == R_TLSGD_PC) {
232 if (In.Got->addDynTlsEntry(Sym)) {
233 uint64_t Off = In.Got->getGlobalDynOffset(Sym);
234 AddTlsReloc(Off, Target->TlsModuleIndexRel, &Sym, NeedDynId);
235 AddTlsReloc(Off + Config->Wordsize, Target->TlsOffsetRel, &Sym,
236 NeedDynOff);
237 }
238 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
239 return 1;
240 }
241 return 0;
242}

244// Returns the number of relocations processed.
245template <class ELFT>
246static unsigned
247handleTlsRelocation(RelType Type, Symbol &Sym, InputSectionBase &C,
248 typename ELFT::uint Offset, int64_t Addend, RelExpr Expr) {
249 if (!Sym.isTls())
250 return 0;
251
252 if (Config->EMachine == EM_ARM)
253 return handleARMTlsRelocation(Type, Sym, C, Offset, Addend, Expr);
254 if (Config->EMachine == EM_MIPS)
255 return handleMipsTlsRelocation(Type, Sym, C, Offset, Addend, Expr);
256
257 if (oneof<R_AARCH64_TLSDESC_PAGE, R_TLSDESC, R_TLSDESC_CALL, R_TLSDESC_PC>(
258 Expr) &&
259 Config->Shared) {
260 if (In.Got->addDynTlsEntry(Sym)) {
261 uint64_t Off = In.Got->getGlobalDynOffset(Sym);
262 Main->RelaDyn->addReloc(
263 {Target->TlsDescRel, In.Got, Off, !Sym.IsPreemptible, &Sym, 0});
264 }
265 if (Expr != R_TLSDESC_CALL)
266 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
267 return 1;
268 }
269
270 if (oneof<R_TLSLD_GOT, R_TLSLD_GOTPLT, R_TLSLD_PC, R_TLSLD_HINT>(
271 Expr)) {
272 // Local-Dynamic relocs can be relaxed to Local-Exec.
273 if (!Config->Shared) {
274 C.Relocations.push_back(
275 {Target->adjustRelaxExpr(Type, nullptr, R_RELAX_TLS_LD_TO_LE), Type,
276 Offset, Addend, &Sym});
277 return Target->getTlsGdRelaxSkip(Type);
278 }
279 if (Expr == R_TLSLD_HINT)
280 return 1;
281 if (In.Got->addTlsIndex())
282 Main->RelaDyn->addReloc(Target->TlsModuleIndexRel, In.Got,
283 In.Got->getTlsIndexOff(), nullptr);
284 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
285 return 1;
286 }
287
288 // Local-Dynamic relocs can be relaxed to Local-Exec.
289 if (Expr == R_DTPREL && !Config->Shared) {
290 C.Relocations.push_back(
291 {Target->adjustRelaxExpr(Type, nullptr, R_RELAX_TLS_LD_TO_LE), Type,
292 Offset, Addend, &Sym});
293 return 1;
294 }
295
296 // Local-Dynamic sequence where offset of tls variable relative to dynamic
297 // thread pointer is stored in the got. This cannot be relaxed to Local-Exec.
298 if (Expr == R_TLSLD_GOT_OFF) {
299 if (!Sym.isInGot()) {
300 In.Got->addEntry(Sym);
301 uint64_t Off = Sym.getGotOffset();
302 In.Got->Relocations.push_back(
303 {R_ABS, Target->TlsOffsetRel, Off, 0, &Sym});
304 }
305 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
306 return 1;
307 }
308
309 if (oneof<R_AARCH64_TLSDESC_PAGE, R_TLSDESC, R_TLSDESC_CALL, R_TLSDESC_PC,
310 R_TLSGD_GOT, R_TLSGD_GOTPLT, R_TLSGD_PC>(Expr)) {
311 if (Config->Shared) {
312 if (In.Got->addDynTlsEntry(Sym)) {
313 uint64_t Off = In.Got->getGlobalDynOffset(Sym);
314 Main->RelaDyn->addReloc(Target->TlsModuleIndexRel, In.Got, Off, &Sym);
315
316 // If the symbol is preemptible we need the dynamic linker to write
317 // the offset too.
318 uint64_t OffsetOff = Off + Config->Wordsize;
319 if (Sym.IsPreemptible)
320 Main->RelaDyn->addReloc(Target->TlsOffsetRel, In.Got, OffsetOff,
321 &Sym);
322 else
323 In.Got->Relocations.push_back(
324 {R_ABS, Target->TlsOffsetRel, OffsetOff, 0, &Sym});
325 }
326 C.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
327 return 1;
328 }
329
330 // Global-Dynamic relocs can be relaxed to Initial-Exec or Local-Exec
331 // depending on the symbol being locally defined or not.
332 if (Sym.IsPreemptible) {
333 C.Relocations.push_back(
334 {Target->adjustRelaxExpr(Type, nullptr, R_RELAX_TLS_GD_TO_IE), Type,
335 Offset, Addend, &Sym});
336 if (!Sym.isInGot()) {
337 In.Got->addEntry(Sym);
338 Main->RelaDyn->addReloc(Target->TlsGotRel, In.Got, Sym.getGotOffset(),
339 &Sym);
340 }
341 } else {
342 C.Relocations.push_back(
343 {Target->adjustRelaxExpr(Type, nullptr, R_RELAX_TLS_GD_TO_LE), Type,
344 Offset, Addend, &Sym});
345 }
346 return Target->getTlsGdRelaxSkip(Type);
347 }
348
349 // Initial-Exec relocs can be relaxed to Local-Exec if the symbol is locally
350 // defined.
351 if (oneof<R_GOT, R_GOTPLT, R_GOT_PC, R_AARCH64_GOT_PAGE_PC, R_GOT_OFF,
352 R_TLSIE_HINT>(Expr) &&
353 !Config->Shared && !Sym.IsPreemptible) {
354 C.Relocations.push_back({R_RELAX_TLS_IE_TO_LE, Type, Offset, Addend, &Sym});
355 return 1;
356 }
357
358 if (Expr == R_TLSIE_HINT)
359 return 1;
360 return 0;
361}

363static RelType getMipsPairType(RelType Type, bool IsLocal) {
364 switch (Type) {
365 case R_MIPS_HI16:
366 return R_MIPS_LO16;
367 case R_MIPS_GOT16:
368 // In case of global symbol, the R_MIPS_GOT16 relocation does not
369 // have a pair. Each global symbol has a unique entry in the GOT
370 // and a corresponding instruction with help of the R_MIPS_GOT16
371 // relocation loads an address of the symbol. In case of local
372 // symbol, the R_MIPS_GOT16 relocation creates a GOT entry to hold
373 // the high 16 bits of the symbol's value. A paired R_MIPS_LO16
374 // relocations handle low 16 bits of the address. That allows
375 // to allocate only one GOT entry for every 64 KBytes of local data.
376 return IsLocal ? R_MIPS_LO16 : R_MIPS_NONE;
377 case R_MICROMIPS_GOT16:
378 return IsLocal ? R_MICROMIPS_LO16 : R_MIPS_NONE;
379 case R_MIPS_PCHI16:
380 return R_MIPS_PCLO16;
381 case R_MICROMIPS_HI16:
382 return R_MICROMIPS_LO16;
383 default:
384 return R_MIPS_NONE;
385 }
386}

424// Returns true if a given relocation can be computed at link-time.
425//
426// For instance, we know the offset from a relocation to its target at
427// link-time if the relocation is PC-relative and refers a
428// non-interposable function in the same executable. This function
429// will return true for such relocation.
430//
431// If this function returns false, that means we need to emit a
432// dynamic relocation so that the relocation will be fixed at load-time.
433static bool isStaticLinkTimeConstant(RelExpr E, RelType Type, const Symbol &Sym,
434 InputSectionBase &S, uint64_t RelOff) {
435 // These expressions always compute a constant
436 if (oneof<R_DTPREL, R_GOTPLT, R_GOT_OFF, R_HEXAGON_GOT, R_TLSLD_GOT_OFF,
437 R_CHERI_CAPABILITY_TABLE_INDEX,
438 R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE,
439 R_CHERI_CAPABILITY_TABLE_INDEX_CALL,
440 R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE,
441 R_CHERI_CAPABILITY_TABLE_REL,
442 R_MIPS_GOT_LOCAL_PAGE, R_MIPS_GOTREL, R_MIPS_GOT_OFF,
443 R_MIPS_GOT_OFF32, R_MIPS_GOT_GP_PC, R_MIPS_TLSGD,
444 R_AARCH64_GOT_PAGE_PC, R_GOT_PC, R_GOTONLY_PC, R_GOTPLTONLY_PC,
445 R_PLT_PC, R_TLSGD_GOT, R_TLSGD_GOTPLT, R_TLSGD_PC, R_PPC32_PLTREL,
446 R_PPC64_CALL_PLT, R_PPC64_RELAX_TOC, R_RISCV_ADD, R_TLSDESC_CALL,
447 R_TLSDESC_PC, R_AARCH64_TLSDESC_PAGE, R_HINT, R_TLSLD_HINT,
448 R_TLSIE_HINT>(E))
449 return true;
450
451 // Cheri capability relocations are never static link time constants since
452 // even if we know the exact value of the capability we can't write it since
453 // there is no way to store the tag bit
454 // TODO: for undef weak -> 0 (or other untagged values) it actually is okay
455 if (E == R_CHERI_CAPABILITY)
456 return false;
457
458 // These never do, except if the entire file is position dependent or if
459 // only the low bits are used.
460 if (E == R_GOT || E == R_PLT || E == R_TLSDESC)
461 return Target->usesOnlyLowPageBits(Type) || !Config->Pic;
462
463 if (Sym.IsPreemptible)
464 return false;
465 if (!Config->Pic)
466 return true;
467
468 // The size of a non preemptible symbol is a constant.
469 if (E == R_SIZE)
470 return true;
471
472 // For the target and the relocation, we want to know if they are
473 // absolute or relative.
474 bool AbsVal = isAbsoluteValue(Sym);
475 bool RelE = isRelExpr(E);
476 if (AbsVal && !RelE)
477 return true;
478 if (!AbsVal && RelE)
479 return true;
480 if (!AbsVal && !RelE)
481 return Target->usesOnlyLowPageBits(Type);
482
483 // Relative relocation to an absolute value. This is normally unrepresentable,
484 // but if the relocation refers to a weak undefined symbol, we allow it to
485 // resolve to the image base. This is a little strange, but it allows us to
486 // link function calls to such symbols. Normally such a call will be guarded
487 // with a comparison, which will load a zero from the GOT.
488 // Another special case is MIPS _gp_disp symbol which represents offset
489 // between start of a function and '_gp' value and defined as absolute just
490 // to simplify the code.
491 assert(AbsVal && RelE);
492 if (Sym.isUndefWeak())
493 return true;
494
495 // We set the final symbols values for linker script defined symbols later.
496 // They always can be computed as a link time constant.
497 if (Sym.ScriptDefined)
498 return true;
499
500 error("relocation " + toString(Type) + " cannot refer to absolute symbol: " +
501 toString(Sym) + getLocation(S, Sym, RelOff));
502 return true;
503}

534// Returns true if a given shared symbol is in a read-only segment in a DSO.
535template <class ELFT> static bool isReadOnly(SharedSymbol &SS) {
536 using Elf_Phdr = typename ELFT::Phdr;
537
538 // Determine if the symbol is read-only by scanning the DSO's program headers.
539 const SharedFile &File = SS.getFile();
540 for (const Elf_Phdr &Phdr :
541 check(File.template getObj<ELFT>().program_headers()))
542 if ((Phdr.p_type == ELF::PT_LOAD || Phdr.p_type == ELF::PT_GNU_RELRO) &&
543 !(Phdr.p_flags & ELF::PF_W) && SS.Value >= Phdr.p_vaddr &&
544 SS.Value < Phdr.p_vaddr + Phdr.p_memsz)
545 return true;
546 return false;
547}

549// Returns symbols at the same offset as a given symbol, including SS itself.
550//
551// If two or more symbols are at the same offset, and at least one of
552// them are copied by a copy relocation, all of them need to be copied.
553// Otherwise, they would refer to different places at runtime.
554template <class ELFT>
555static SmallSet<SharedSymbol *, 4> getSymbolsAt(SharedSymbol &SS) {
556 using Elf_Sym = typename ELFT::Sym;
557
558 SharedFile &File = SS.getFile();
559
560 SmallSet<SharedSymbol *, 4> Ret;
561 for (const Elf_Sym &S : File.template getGlobalELFSyms<ELFT>()) {
562 if (S.st_shndx == SHN_UNDEF || S.st_shndx == SHN_ABS ||
563 S.getType() == STT_TLS || S.st_value != SS.Value)
564 continue;
565 StringRef Name = check(S.getName(File.getStringTable()));
566 Symbol *Sym = Symtab->find(Name);
567 if (auto *Alias = dyn_cast_or_null<SharedSymbol>(Sym))
568 Ret.insert(Alias);
569 }
570 return Ret;
571}

573// When a symbol is copy relocated or we create a canonical plt entry, it is
574// effectively a defined symbol. In the case of copy relocation the symbol is
575// in .bss and in the case of a canonical plt entry it is in .plt. This function
576// replaces the existing symbol with a Defined pointing to the appropriate
577// location.
578static void replaceWithDefined(Symbol &Sym, SectionBase *Sec, uint64_t Value,
579 uint64_t Size) {
580 Symbol Old = Sym;
581
582 Sym.replace(Defined{Sym.File, Sym.getName(), Sym.Binding, Sym.StOther,
583 Sym.Type, Value, Size, Sec});
584

595// Reserve space in .bss or .bss.rel.ro for copy relocation.
596//
597// The copy relocation is pretty much a hack. If you use a copy relocation
598// in your program, not only the symbol name but the symbol's size, RW/RO
599// bit and alignment become part of the ABI. In addition to that, if the
600// symbol has aliases, the aliases become part of the ABI. That's subtle,
601// but if you violate that implicit ABI, that can cause very counter-
602// intuitive consequences.
603//
604// So, what is the copy relocation? It's for linking non-position
605// independent code to DSOs. In an ideal world, all references to data
606// exported by DSOs should go indirectly through GOT. But if object files
607// are compiled as non-PIC, all data references are direct. There is no
608// way for the linker to transform the code to use GOT, as machine
609// instructions are already set in stone in object files. This is where
610// the copy relocation takes a role.
611//
612// A copy relocation instructs the dynamic linker to copy data from a DSO
613// to a specified address (which is usually in .bss) at load-time. If the
614// static linker (that's us) finds a direct data reference to a DSO
615// symbol, it creates a copy relocation, so that the symbol can be
616// resolved as if it were in .bss rather than in a DSO.
617//
618// As you can see in this function, we create a copy relocation for the
619// dynamic linker, and the relocation contains not only symbol name but
620// various other informtion about the symbol. So, such attributes become a
621// part of the ABI.
622//
623// Note for application developers: I can give you a piece of advice if
624// you are writing a shared library. You probably should export only
625// functions from your library. You shouldn't export variables.
626//
627// As an example what can happen when you export variables without knowing
628// the semantics of copy relocations, assume that you have an exported
629// variable of type T. It is an ABI-breaking change to add new members at
630// end of T even though doing that doesn't change the layout of the
631// existing members. That's because the space for the new members are not
632// reserved in .bss unless you recompile the main program. That means they
633// are likely to overlap with other data that happens to be laid out next
634// to the variable in .bss. This kind of issue is sometimes very hard to
635// debug. What's a solution? Instead of exporting a varaible V from a DSO,
636// define an accessor getV().
637template <class ELFT> static void addCopyRelSymbol(SharedSymbol &SS) {
638 // Copy relocation against zero-sized symbol doesn't make sense.
639 uint64_t SymSize = SS.getSize();
640 if (SymSize == 0 || SS.Alignment == 0)
641 fatal("cannot create a copy relocation for symbol " + toString(SS));
642
643 // See if this symbol is in a read-only segment. If so, preserve the symbol's
644 // memory protection by reserving space in the .bss.rel.ro section.
645 bool IsReadOnly = isReadOnly<ELFT>(SS);
646 BssSection *Sec = make<BssSection>(IsReadOnly ? ".bss.rel.ro" : ".bss",
647 SymSize, SS.Alignment);
648 if (IsReadOnly)
649 In.BssRelRo->getParent()->addSection(Sec);
650 else
651 In.Bss->getParent()->addSection(Sec);
652
653 // Look through the DSO's dynamic symbol table for aliases and create a
654 // dynamic symbol for each one. This causes the copy relocation to correctly
655 // interpose any aliases.
656 for (SharedSymbol *Sym : getSymbolsAt<ELFT>(SS))
657 replaceWithDefined(*Sym, Sec, 0, Sym->Size);
658
659 Main->RelaDyn->addReloc(Target->CopyRel, Sec, 0, &SS);
660}

662// MIPS has an odd notion of "paired" relocations to calculate addends.
663// For example, if a relocation is of R_MIPS_HI16, there must be a
664// R_MIPS_LO16 relocation after that, and an addend is calculated using
665// the two relocations.
666template <class ELFT, class RelTy>
667static int64_t computeMipsAddend(const RelTy &Rel, const RelTy *End,
668 InputSectionBase &Sec, RelExpr Expr,
669 bool IsLocal) {
670 if (Expr == R_MIPS_GOTREL && IsLocal)
671 return Sec.getFile<ELFT>()->MipsGp0;
672
673 // The ABI says that the paired relocation is used only for REL.
674 // See p. 4-17 at ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
675 if (RelTy::IsRela)
676 return 0;
677
678 RelType Type = Rel.getType(Config->IsMips64EL);
679 uint32_t PairTy = getMipsPairType(Type, IsLocal);
680 if (PairTy == R_MIPS_NONE)
681 return 0;
682
683 const uint8_t *Buf = Sec.data().data();
684 uint32_t SymIndex = Rel.getSymbol(Config->IsMips64EL);
685
686 // To make things worse, paired relocations might not be contiguous in
687 // the relocation table, so we need to do linear search. *sigh*
688 for (const RelTy *RI = &Rel; RI != End; ++RI)
689 if (RI->getType(Config->IsMips64EL) == PairTy &&
690 RI->getSymbol(Config->IsMips64EL) == SymIndex)
691 return Target->getImplicitAddend(Buf + RI->r_offset, PairTy);
692
693 warn("can't find matching " + toString(PairTy) + " relocation for " +
694 toString(Type));
695 return 0;
696}

698// Returns an addend of a given relocation. If it is RELA, an addend
699// is in a relocation itself. If it is REL, we need to read it from an
700// input section.
701template <class ELFT, class RelTy>
702static int64_t computeAddend(const RelTy &Rel, const RelTy *End,
703 InputSectionBase &Sec, RelExpr Expr,
704 bool IsLocal) {
705 int64_t Addend;
706 RelType Type = Rel.getType(Config->IsMips64EL);
707
708 if (RelTy::IsRela) {
709 Addend = getAddend<ELFT>(Rel);
710 } else {
711 const uint8_t *Buf = Sec.data().data();
712 Addend = Target->getImplicitAddend(Buf + Rel.r_offset, Type);
713 }
714
715 if (Config->EMachine == EM_PPC64 && Config->Pic && Type == R_PPC64_TOC)
716 Addend += getPPC64TocBase();
717 if (Config->EMachine == EM_MIPS)
718 Addend += computeMipsAddend<ELFT>(Rel, End, Sec, Expr, IsLocal);
719
720 return Addend;
721}

818// MIPS N32 ABI treats series of successive relocations with the same offset
819// as a single relocation. The similar approach used by N64 ABI, but this ABI
820// packs all relocations into the single relocation record. Here we emulate
821// this for the N32 ABI. Iterate over relocation with the same offset and put
822// theirs types into the single bit-set.
823template <class RelTy> static RelType getMipsN32RelType(RelTy *&Rel, RelTy *End) {
824 RelType Type = 0;
825 uint64_t Offset = Rel->r_offset;
826
827 int N = 0;
828 while (Rel != End && Rel->r_offset == Offset)
829 Type |= (Rel++)->getType(Config->IsMips64EL) << (8 * N++);
830 return Type;
831}

833// .eh_frame sections are mergeable input sections, so their input
834// offsets are not linearly mapped to output section. For each input
835// offset, we need to find a section piece containing the offset and
836// add the piece's base address to the input offset to compute the
837// output offset. That isn't cheap.
838//
839// This class is to speed up the offset computation. When we process
840// relocations, we access offsets in the monotonically increasing
841// order. So we can optimize for that access pattern.
842//
843// For sections other than .eh_frame, this class doesn't do anything.
844namespace {
845class OffsetGetter {
846public:
847 explicit OffsetGetter(InputSectionBase &Sec) {
848 if (auto *Eh = dyn_cast<EhInputSection>(&Sec))
849 Pieces = Eh->Pieces;
850 }
851
852 // Translates offsets in input sections to offsets in output sections.
853 // Given offset must increase monotonically. We assume that Piece is
854 // sorted by InputOff.
855 uint64_t get(uint64_t Off) {
856 if (Pieces.empty())
857 return Off;
858
859 while (I != Pieces.size() && Pieces[I].InputOff + Pieces[I].Size <= Off)
860 ++I;
861 if (I == Pieces.size())
862 fatal(".eh_frame: relocation is not in any piece");
863
864 // Pieces must be contiguous, so there must be no holes in between.
865 assert(Pieces[I].InputOff <= Off && "Relocation not in any piece");
866
867 // Offset -1 means that the piece is dead (i.e. garbage collected).
868 if (Pieces[I].OutputOff == -1)
869 return -1;
870 return Pieces[I].OutputOff + Off - Pieces[I].InputOff;
871 }
872
873private:
874 ArrayRef<EhSectionPiece> Pieces;
875 size_t I = 0;
876};
877} // namespace

879static void addRelativeReloc(InputSectionBase *IS, uint64_t OffsetInSec,
880 Symbol *Sym, int64_t Addend, RelExpr Expr,
881 RelType Type) {
882 Partition &Part = IS->getPartition();
883
884 // Add a relative relocation. If RelrDyn section is enabled, and the
885 // relocation offset is guaranteed to be even, add the relocation to
886 // the RelrDyn section, otherwise add it to the RelaDyn section.
887 // RelrDyn sections don't support odd offsets. Also, RelrDyn sections
888 // don't store the addend values, so we must write it to the relocated

890 if (Part.RelrDyn && IS->Alignment >= 2 && OffsetInSec % 2 == 0) {
891 IS->Relocations.push_back({Expr, Type, OffsetInSec, Addend, Sym});
892 Part.RelrDyn->Relocs.push_back({IS, OffsetInSec});
893 return;
894 }
895 Part.RelaDyn->addReloc(Target->RelativeRel, IS, OffsetInSec, Sym, Addend,
896 Expr, Type);

908static void addGotEntry(Symbol &Sym) {
909 In.Got->addEntry(Sym);
910
911 RelExpr Expr = Sym.isTls() ? R_TLS : R_ABS;
912 uint64_t Off = Sym.getGotOffset();

914 // If a GOT slot value can be calculated at link-time, which is now,
915 // we can just fill that out.
916 //
917 // (We don't actually write a value to a GOT slot right now, but we
918 // add a static relocation to a Relocations vector so that
919 // InputSection::relocate will do the work for us. We may be able
920 // to just write a value now, but it is a TODO.)

929 // the GOT slot will be fixed at load-time.

931 addRelativeReloc(In.Got, Off, &Sym, 0, R_ABS, Target->SymbolicRel);
932 return;
933 }
934 Main->RelaDyn->addReloc(
935 Sym.isTls() ? Target->TlsGotRel : Target->GotRel, In.Got, Off, &Sym, 0,
936 Sym.IsPreemptible ? R_ADDEND : R_ABS, Target->SymbolicRel);
937}

939// Return true if we can define a symbol in the executable that
940// contains the value/function of a symbol defined in a shared
941// library.
942static bool canDefineSymbolInExecutable(Symbol &Sym) {
943 // If the symbol has default visibility the symbol defined in the
944 // executable will preempt it.
945 // Note that we want the visibility of the shared symbol itself, not
946 // the visibility of the symbol in the output file we are producing. That is
947 // why we use Sym.StOther.
948 if ((Sym.StOther & 0x3) == STV_DEFAULT)
949 return true;
950
951 // If we are allowed to break address equality of functions, defining
952 // a plt entry will allow the program to call the function in the
953 // .so, but the .so and the executable will no agree on the address
954 // of the function. Similar logic for objects.
955 return ((Sym.isFunc() && Config->IgnoreFunctionAddressEquality) ||
956 (Sym.isObject() && Config->IgnoreDataAddressEquality));
957}

959// The reason we have to do this early scan is as follows
960// * To mmap the output file, we need to know the size
961// * For that, we need to know how many dynamic relocs we will have.
962// It might be possible to avoid this by outputting the file with write:
963// * Write the allocated output sections, computing addresses.
964// * Apply relocations, recording which ones require a dynamic reloc.
965// * Write the dynamic relocations.
966// * Write the rest of the file.
967// This would have some drawbacks. For example, we would only know if .rela.dyn
968// is needed after applying relocations. If it is, it will go after rw and rx
969// sections. Given that it is ro, we will need an extra PT_LOAD. This
970// complicates things for the dynamic linker and means we would have to reserve
971// space for the extra PT_LOAD even if we end up not using it.
972template <class ELFT, class RelTy>
973static void processRelocAux(InputSectionBase &Sec, RelExpr Expr, RelType Type,
974 uint64_t Offset, Symbol &Sym, const RelTy &Rel,
975 int64_t Addend) {
976 if (isStaticLinkTimeConstant(Expr, Type, Sym, Sec, Offset)) {
977 Sec.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
978 return;
979 }
980 bool CanWrite = (Sec.Flags & SHF_WRITE) || !Config->ZText;
981 // HACK: clang emits a read-only __cap_relocs, but for PIC code we need to
982 // emit dynamic relocations for its contents (REL32/64/NONE).
983 if (Config->EMachine == EM_MIPS && Config->Pic && Sec.Name == "__cap_relocs") {
984 if (!Config->ProcessCapRelocs)
985 error("capsizefix will not work with dynamic libraries, please remove "
986 "-no-process-cap-relocs from the linker invocation!");
987 return;
988 }
989
990 if (Expr == R_CHERI_CAPABILITY) {
991 static auto getRelocTargetLocation = [&]() -> std::string {
992 auto RelocTarget = SymbolAndOffset::fromSectionWithOffset(&Sec, Offset);
993 return "\n>>> referenced by " + RelocTarget.verboseToString();
994 };
995 if (!CanWrite) {
996 readOnlyCapRelocsError(Sym, getRelocTargetLocation());
997 return;
998 }
999 addCapabilityRelocation<ELFT>(&Sym, Type, &Sec, Offset, Expr, Addend,
1000 /* IsCallExpr=*/false,
1001 getRelocTargetLocation);
1002 // TODO: check if it is a call and needs a plt stub
1003 return;
1004 }
1005
1006 if (CanWrite) {
1007 bool IsPreemptibleValue = Sym.IsPreemptible;
1008
1009 // XXXAR: HACK: the locations of the cap_relocs should not be preemptible
1010 // This is only needed for compatibility with capsizefix and RTLD without RELA support
1011 if (Sec.Name == "__cap_relocs" && (Offset % 40) == 0) {
1012 // The location field should never be marked as preemptible, we want to relocate the local symbol
1013 // XXXAR: this is probably wrong in some cases but this is only needed for external capsizefix
1014 IsPreemptibleValue = false;
1015 }
1016
1017 // FIXME Improve the way we handle absolute relocation types that will
1018 // change to relative relocations. ARM has a relocation type R_ARM_TARGET1
1019 // that is similar to SymbolicRel. PPC64 may have similar relocation types.
1020 if ((!IsPreemptibleValue &&
1021 (Config->EMachine == EM_ARM || Config->EMachine == EM_PPC64 ||
1022 Type == Target->SymbolicRel)) ||
1023 Expr == R_GOT) {
1024 // If this is a symbolic relocation to a non-preemptable symbol, or an
1025 // R_GOT, its address is its link-time value plus load address. Represent
1026 // it with a relative relocation.
1027 addRelativeReloc(&Sec, Offset, &Sym, Addend, Expr, Type);
1028 return;
1029 } else if (RelType Rel = Target->getDynRel(Type)) {
1030 Sec.getPartition().RelaDyn->addReloc(Rel, &Sec, Offset, &Sym, Addend,
1031 R_ADDEND, Type);
1032
1033 // MIPS ABI turns using of GOT and dynamic relocations inside out.
1034 // While regular ABI uses dynamic relocations to fill up GOT entries
1035 // MIPS ABI requires dynamic linker to fills up GOT entries using
1036 // specially sorted dynamic symbol table. This affects even dynamic
1037 // relocations against symbols which do not require GOT entries
1038 // creation explicitly, i.e. do not have any GOT-relocations. So if
1039 // a preemptible symbol has a dynamic relocation we anyway have
1040 // to create a GOT entry for it.
1041 // If a non-preemptible symbol has a dynamic relocation against it,
1042 // dynamic linker takes it st_value, adds offset and writes down
1043 // result of the dynamic relocation. In case of preemptible symbol
1044 // dynamic linker performs symbol resolution, writes the symbol value
1045 // to the GOT entry and reads the GOT entry when it needs to perform
1046 // a dynamic relocation.
1047 // ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf p.4-19
1048 if (Config->EMachine == EM_MIPS)
1049 In.MipsGot->addEntry(*Sec.File, Sym, Addend, Expr);
1050 return;
1051 }
1052 }
1053
1054 // If the relocation is to a weak undef, and we are producing
1055 // executable, give up on it and produce a non preemptible 0.
1056 if (!Config->Shared && Sym.isUndefWeak()) {
1057 if (Expr == R_CHERI_CAPABILITY)
1058 error("Not implemented yet");
1059 Sec.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
1060 return;
1061 }
1062
1063 if (!CanWrite && (Config->Pic && !isRelExpr(Expr))) {
1064 error(
1065 "can't create dynamic relocation " + toString(Type) + " against " +
1066 (Sym.getName().empty() ? "local symbol" : "symbol: " + toString(Sym)) +
1067 " in readonly segment; recompile object files with -fPIC "
1068 "or pass '-Wl,-z,notext' to allow text relocations in the output" +
1069 getLocation(Sec, Sym, Offset));
1070 return;
1071 }
1072
1073 // Copy relocations (for STT_OBJECT) and canonical PLT (for STT_FUNC) are only
1074 // possible in an executable.
1075 //
1076 // Among R_ABS relocatoin types, SymbolicRel has the same size as the word
1077 // size. Others have fewer bits and may cause runtime overflow in -pie/-shared
1078 // mode. Disallow them.
1079 if (Config->Shared ||
1080 (Config->Pie && Expr == R_ABS && Type != Target->SymbolicRel)) {
1081 errorOrWarn(
1082 "relocation " + toString(Type) + " cannot be used against " +
1083 (Sym.getName().empty() ? "local symbol" : "symbol " + toString(Sym)) +
1084 "; recompile with -fPIC" + getLocation(Sec, Sym, Offset));
1085 return;
1086 }
1087
1088 // If the symbol is undefined we already reported any relevant errors.
1089 if (Sym.isUndefined())
1090 return;
1091
1092 if (!canDefineSymbolInExecutable(Sym)) {
1093 error("cannot preempt symbol: " + toString(Sym) +
1094 getLocation(Sec, Sym, Offset));
1095 return;
1096 }
1097
1098 if (Sym.isObject()) {
1099 // Produce a copy relocation.
1100 if (auto *SS = dyn_cast<SharedSymbol>(&Sym)) {
1101 if (!Config->ZCopyreloc)
1102 error("unresolvable relocation " + toString(Type) +
1103 " against symbol '" + toString(*SS) +
1104 "'; recompile with -fPIC or remove '-z nocopyreloc'" +
1105 getLocation(Sec, Sym, Offset));
1106 addCopyRelSymbol<ELFT>(*SS);
1107 }
1108 Sec.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
1109 return;
1110 }
1111
1112 if (Sym.isFunc()) {
1113 // This handles a non PIC program call to function in a shared library. In
1114 // an ideal world, we could just report an error saying the relocation can
1115 // overflow at runtime. In the real world with glibc, crt1.o has a
1116 // R_X86_64_PC32 pointing to libc.so.
1117 //
1118 // The general idea on how to handle such cases is to create a PLT entry and
1119 // use that as the function value.
1120 //
1121 // For the static linking part, we just return a plt expr and everything
1122 // else will use the PLT entry as the address.
1123 //
1124 // The remaining problem is making sure pointer equality still works. We
1125 // need the help of the dynamic linker for that. We let it know that we have
1126 // a direct reference to a so symbol by creating an undefined symbol with a
1127 // non zero st_value. Seeing that, the dynamic linker resolves the symbol to
1128 // the value of the symbol we created. This is true even for got entries, so
1129 // pointer equality is maintained. To avoid an infinite loop, the only entry
1130 // that points to the real function is a dedicated got entry used by the
1131 // plt. That is identified by special relocation types (R_X86_64_JUMP_SLOT,
1132 // R_386_JMP_SLOT, etc).
1133
1134 // For position independent executable on i386, the plt entry requires ebx
1135 // to be set. This causes two problems:
1136 // * If some code has a direct reference to a function, it was probably
1137 // compiled without -fPIE/-fPIC and doesn't maintain ebx.
1138 // * If a library definition gets preempted to the executable, it will have
1139 // the wrong ebx value.
1140 if (Config->Pie && Config->EMachine == EM_386)
1141 errorOrWarn("symbol '" + toString(Sym) +
1142 "' cannot be preempted; recompile with -fPIE" +
1143 getLocation(Sec, Sym, Offset));
1144 if (!Sym.isInPlt())
1145 addPltEntry<ELFT>(In.Plt, In.GotPlt, In.RelaPlt, Target->PltRel, Sym);
1146 if (!Sym.isDefined())
1147 replaceWithDefined(
1148 Sym, In.Plt,
1149 Target->PltHeaderSize + Target->PltEntrySize * Sym.PltIndex, 0);
1150 Sym.NeedsPltAddr = true;
1151 Sec.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
1152 return;
1153 }
1154
1155 errorOrWarn("symbol '" + toString(Sym) + "' has no type" +
1156 getLocation(Sec, Sym, Offset));
1157}

1168template <class ELFT, class RelTy>
1169static void scanReloc(InputSectionBase &Sec, OffsetGetter &GetOffset, RelTy *&I,
1170 RelTy *End) {
1171 const RelTy &Rel = *I;
1172 uint32_t SymIndex = Rel.getSymbol(Config->IsMips64EL);
1173 Symbol &Sym = Sec.getFile<ELFT>()->getSymbol(SymIndex);
1174 RelType Type;
1175
1176 // Deal with MIPS oddity.
1177 if (Config->MipsN32Abi) {
1178 Type = getMipsN32RelType(I, End);
1179 } else {
1180 Type = Rel.getType(Config->IsMips64EL);
1181 ++I;
1182 }
1183
1184 // Get an offset in an output section this relocation is applied to.
1185 uint64_t Offset = GetOffset.get(Rel.r_offset);
1186 if (Offset == uint64_t(-1))
1187 return;
1188
1189 // Error if the target symbol is undefined. Symbol index 0 may be used by
1190 // marker relocations, e.g. R_*_NONE and R_ARM_V4BX. Don't error on them.
1191 if (SymIndex != 0 && maybeReportUndefined<ELFT>(Sym, Sec, Rel.r_offset))
1192 return;
1193
1194 const uint8_t *RelocatedAddr = Sec.data().begin() + Rel.r_offset;
1195 RelExpr Expr = Target->getRelExpr(Type, Sym, RelocatedAddr);
1196
1197 // Ignore "hint" relocations because they are only markers for relaxation.
1198 if (oneof<R_HINT, R_NONE>(Expr))
1199 return;
1200
1201 // We can separate the small code model relocations into 2 categories:
1202 // 1) Those that access the compiler generated .toc sections.
1203 // 2) Those that access the linker allocated got entries.
1204 // lld allocates got entries to symbols on demand. Since we don't try to sort
1205 // the got entries in any way, we don't have to track which objects have
1206 // got-based small code model relocs. The .toc sections get placed after the
1207 // end of the linker allocated .got section and we do sort those so sections
1208 // addressed with small code model relocations come first.
1209 if (Config->EMachine == EM_PPC64 && isPPC64SmallCodeModelTocReloc(Type))
1210 Sec.File->PPC64SmallCodeModelTocRelocs = true;
1211
1212 if (Sym.isGnuIFunc() && !Config->ZText && Config->WarnIfuncTextrel) {
1213 warn("using ifunc symbols when text relocations are allowed may produce "
1214 "a binary that will segfault, if the object file is linked with "
1215 "old version of glibc (glibc 2.28 and earlier). If this applies to "
1216 "you, consider recompiling the object files without -fPIC and "
1217 "without -Wl,-z,notext option. Use -no-warn-ifunc-textrel to "
1218 "turn off this warning." +
1219 getLocation(Sec, Sym, Offset));
1220 }
1221
1222 // Read an addend.
1223 int64_t Addend = computeAddend<ELFT>(Rel, End, Sec, Expr, Sym.isLocal());
1224
1225 // Relax relocations.
1226 //
1227 // If we know that a PLT entry will be resolved within the same ELF module, we
1228 // can skip PLT access and directly jump to the destination function. For
1229 // example, if we are linking a main exectuable, all dynamic symbols that can
1230 // be resolved within the executable will actually be resolved that way at
1231 // runtime, because the main exectuable is always at the beginning of a search
1232 // list. We can leverage that fact.
1233 if (!Sym.IsPreemptible && (!Sym.isGnuIFunc() || Config->ZIfuncNoplt)) {
1234 if (Expr == R_GOT_PC && !isAbsoluteValue(Sym)) {
1235 Expr = Target->adjustRelaxExpr(Type, RelocatedAddr, Expr);
1236 } else {
1237 // Addend of R_PPC_PLTREL24 is used to choose call stub type. It should be
1238 // ignored if optimized to R_PC.
1239 if (Config->EMachine == EM_PPC && Expr == R_PPC32_PLTREL)
1240 Addend = 0;
1241 Expr = fromPlt(Expr);
1242 }
1243 }
1244
1245 // If the relocation does not emit a GOT or GOTPLT entry but its computation
1246 // uses their addresses, we need GOT or GOTPLT to be created.
1247 //
1248 // The 4 types that relative GOTPLT are all x86 and x86-64 specific.
1249 if (oneof<R_GOTPLTONLY_PC, R_GOTPLTREL, R_GOTPLT, R_TLSGD_GOTPLT>(Expr)) {
1250 In.GotPlt->HasGotPltOffRel = true;
1251 } else if (oneof<R_GOTONLY_PC, R_GOTREL, R_PPC64_TOCBASE, R_PPC64_RELAX_TOC>(
1252 Expr)) {
1253 In.Got->HasGotOffRel = true;
1254 }
1255
1256 // Process some TLS relocations, including relaxing TLS relocations.
1257 // Note that this function does not handle all TLS relocations.
1258 if (unsigned Processed =
1259 handleTlsRelocation<ELFT>(Type, Sym, Sec, Offset, Addend, Expr)) {
1260 I += (Processed - 1);
1261 return;
1262 }
1263
1264 // We were asked not to generate PLT entries for ifuncs. Instead, pass the
1265 // direct relocation on through.
1266 if (Sym.isGnuIFunc() && Config->ZIfuncNoplt) {
1267 Sym.ExportDynamic = true;
1268 Main->RelaDyn->addReloc(Type, &Sec, Offset, &Sym, Addend, R_ADDEND, Type);
1269 return;
1270 }
1271
1272 if (oneof<R_CHERI_CAPABILITY_TABLE_INDEX,
1273 R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE,
1274 R_CHERI_CAPABILITY_TABLE_INDEX_CALL,
1275 R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE>(Expr)) {
1276 assert(Config->ProcessCapRelocs);
1277 In.CheriCapTable->addEntry(Sym, Expr, &Sec, Offset);
1278 // Write out the index into the instruction
1279 Sec.Relocations.push_back({Expr, Type, Offset, Addend, &Sym});
1280 return;
1281 }
1282
1283 // Non-preemptible ifuncs require special handling. First, handle the usual
1284 // case where the symbol isn't one of these.
1285 if (!Sym.isGnuIFunc() || Sym.IsPreemptible) {
1286 // If a relocation needs PLT, we create PLT and GOTPLT slots for the symbol.
1287 if (needsPlt(Expr) && !Sym.isInPlt())
1288 addPltEntry<ELFT>(In.Plt, In.GotPlt, In.RelaPlt, Target->PltRel, Sym);
1289
1290 // Create a GOT slot if a relocation needs GOT.
1291 if (needsGot(Expr)) {
1292 if (Config->EMachine == EM_MIPS) {
1293 // MIPS ABI has special rules to process GOT entries and doesn't
1294 // require relocation entries for them. A special case is TLS
1295 // relocations. In that case dynamic loader applies dynamic
1296 // relocations to initialize TLS GOT entries.
1297 // See "Global Offset Table" in Chapter 5 in the following document
1298 // for detailed description:
1299 // ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
1300 In.MipsGot->addEntry(*Sec.File, Sym, Addend, Expr);
1301 } else if (!Sym.isInGot()) {
1302 addGotEntry(Sym);
1303 }
1304 }
1305 } else {
1306 // Handle a reference to a non-preemptible ifunc. These are special in a
1307 // few ways:
1308 //
1309 // - Unlike most non-preemptible symbols, non-preemptible ifuncs do not have
1310 // a fixed value. But assuming that all references to the ifunc are
1311 // GOT-generating or PLT-generating, the handling of an ifunc is
1312 // relatively straightforward. We create a PLT entry in Iplt, which is
1313 // usually at the end of .plt, which makes an indirect call using a
1314 // matching GOT entry in IgotPlt, which is usually at the end of .got.plt.
1315 // The GOT entry is relocated using an IRELATIVE relocation in RelaIplt,
1316 // which is usually at the end of .rela.plt. Unlike most relocations in
1317 // .rela.plt, which may be evaluated lazily without -z now, dynamic
1318 // loaders evaluate IRELATIVE relocs eagerly, which means that for
1319 // IRELATIVE relocs only, GOT-generating relocations can point directly to
1320 // .got.plt without requiring a separate GOT entry.
1321 //
1322 // - Despite the fact that an ifunc does not have a fixed value, compilers
1323 // that are not passed -fPIC will assume that they do, and will emit
1324 // direct (non-GOT-generating, non-PLT-generating) relocations to the
1325 // symbol. This means that if a direct relocation to the symbol is
1326 // seen, the linker must set a value for the symbol, and this value must
1327 // be consistent no matter what type of reference is made to the symbol.
1328 // This can be done by creating a PLT entry for the symbol in the way
1329 // described above and making it canonical, that is, making all references
1330 // point to the PLT entry instead of the resolver. In lld we also store
1331 // the address of the PLT entry in the dynamic symbol table, which means
1332 // that the symbol will also have the same value in other modules.
1333 // Because the value loaded from the GOT needs to be consistent with
1334 // the value computed using a direct relocation, a non-preemptible ifunc
1335 // may end up with two GOT entries, one in .got.plt that points to the
1336 // address returned by the resolver and is used only by the PLT entry,
1337 // and another in .got that points to the PLT entry and is used by
1338 // GOT-generating relocations.
1339 //
1340 // - The fact that these symbols do not have a fixed value makes them an
1341 // exception to the general rule that a statically linked executable does
1342 // not require any form of dynamic relocation. To handle these relocations
1343 // correctly, the IRELATIVE relocations are stored in an array which a
1344 // statically linked executable's startup code must enumerate using the
1345 // linker-defined symbols __rela?_iplt_{start,end}.
1346 //
1347 // - An absolute relocation to a non-preemptible ifunc (such as a global
1348 // variable containing a pointer to the ifunc) needs to be relocated in
1349 // the exact same way as a GOT entry, so we can avoid needing to make the
1350 // PLT entry canonical by translating such relocations into IRELATIVE
1351 // relocations in the RelaIplt.
1352 if (!Sym.isInPlt()) {
1353 // Create PLT and GOTPLT slots for the symbol.
1354 Sym.IsInIplt = true;
1355
1356 // Create a copy of the symbol to use as the target of the IRELATIVE
1357 // relocation in the IgotPlt. This is in case we make the PLT canonical
1358 // later, which would overwrite the original symbol.
1359 //
1360 // FIXME: Creating a copy of the symbol here is a bit of a hack. All
1361 // that's really needed to create the IRELATIVE is the section and value,
1362 // so ideally we should just need to copy those.
1363 auto *DirectSym = make<Defined>(cast<Defined>(Sym));
1364 addPltEntry<ELFT>(In.Iplt, In.IgotPlt, In.RelaIplt, Target->IRelativeRel,
1365 *DirectSym);
1366 Sym.PltIndex = DirectSym->PltIndex;
1367 }
1368 if (Expr == R_ABS && Addend == 0 && (Sec.Flags & SHF_WRITE)) {
1369 // We might be able to represent this as an IRELATIVE. But we don't know
1370 // yet whether some later relocation will make the symbol point to a
1371 // canonical PLT, which would make this either a dynamic RELATIVE (PIC) or
1372 // static (non-PIC) relocation. So we keep a record of the information
1373 // required to process the relocation, and after scanRelocs() has been
1374 // called on all relocations, the relocation is resolved by
1375 // addIRelativeRelocs().
1376 IRelativeRelocs.push_back({Type, &Sec, Offset, &Sym});
1377 return;
1378 }
1379 if (needsGot(Expr)) {
1380 // Redirect GOT accesses to point to the Igot.
1381 //
1382 // This field is also used to keep track of whether we ever needed a GOT
1383 // entry. If we did and we make the PLT canonical later, we'll need to
1384 // create a GOT entry pointing to the PLT entry for Sym.
1385 Sym.GotInIgot = true;
1386 } else if (!needsPlt(Expr)) {
1387 // Make the ifunc's PLT entry canonical by changing the value of its
1388 // symbol to redirect all references to point to it.
1389 unsigned EntryOffset = Sym.PltIndex * Target->PltEntrySize;
1390 if (Config->ZRetpolineplt)
1391 EntryOffset += Target->PltHeaderSize;
1392
1393 auto &D = cast<Defined>(Sym);
1394 D.Section = In.Iplt;
1395 D.Value = EntryOffset;
1396 D.Size = 0;
1397 // It's important to set the symbol type here so that dynamic loaders
1398 // don't try to call the PLT as if it were an ifunc resolver.
1399 D.Type = STT_FUNC;
1400
1401 if (Sym.GotInIgot) {
1402 // We previously encountered a GOT generating reference that we
1403 // redirected to the Igot. Now that the PLT entry is canonical we must
1404 // clear the redirection to the Igot and add a GOT entry. As we've
1405 // changed the symbol type to STT_FUNC future GOT generating references
1406 // will naturally use this GOT entry.
1407 //
1408 // We don't need to worry about creating a MIPS GOT here because ifuncs
1409 // aren't a thing on MIPS.
1410 Sym.GotInIgot = false;
1411 addGotEntry(Sym);
1412 }
1413 }
1414 }
1415
1416 processRelocAux<ELFT>(Sec, Expr, Type, Offset, Sym, Rel, Addend);
1417}

1461static bool mergeCmp(const InputSection *A, const InputSection *B) {
1462 // std::merge requires a strict weak ordering.
1463 if (A->OutSecOff < B->OutSecOff)
1464 return true;
1465
1466 if (A->OutSecOff == B->OutSecOff) {
1467 auto *TA = dyn_cast<ThunkSection>(A);
1468 auto *TB = dyn_cast<ThunkSection>(B);
1469
1470 // Check if Thunk is immediately before any specific Target
1471 // InputSection for example Mips LA25 Thunks.
1472 if (TA && TA->getTargetInputSection() == B)
1473 return true;
1474
1475 // Place Thunk Sections without specific targets before
1476 // non-Thunk Sections.
1477 if (TA && !TB && !TA->getTargetInputSection())
1478 return true;
1479 }
1480
1481 return false;
1482}

1498// Thunk Implementation
1499//
1500// Thunks (sometimes called stubs, veneers or branch islands) are small pieces
1501// of code that the linker inserts inbetween a caller and a callee. The thunks
1502// are added at link time rather than compile time as the decision on whether
1503// a thunk is needed, such as the caller and callee being out of range, can only
1504// be made at link time.
1505//
1506// It is straightforward to tell given the current state of the program when a
1507// thunk is needed for a particular call. The more difficult part is that
1508// the thunk needs to be placed in the program such that the caller can reach
1509// the thunk and the thunk can reach the callee; furthermore, adding thunks to
1510// the program alters addresses, which can mean more thunks etc.
1511//
1512// In lld we have a synthetic ThunkSection that can hold many Thunks.
1513// The decision to have a ThunkSection act as a container means that we can
1514// more easily handle the most common case of a single block of contiguous
1515// Thunks by inserting just a single ThunkSection.
1516//
1517// The implementation of Thunks in lld is split across these areas
1518// Relocations.cpp : Framework for creating and placing thunks
1519// Thunks.cpp : The code generated for each supported thunk
1520// Target.cpp : Target specific hooks that the framework uses to decide when
1521// a thunk is used
1522// Synthetic.cpp : Implementation of ThunkSection
1523// Writer.cpp : Iteratively call framework until no more Thunks added
1524//
1525// Thunk placement requirements:
1526// Mips LA25 thunks. These must be placed immediately before the callee section
1527// We can assume that the caller is in range of the Thunk. These are modelled
1528// by Thunks that return the section they must precede with
1529// getTargetInputSection().
1530//
1531// ARM interworking and range extension thunks. These thunks must be placed
1532// within range of the caller. All implemented ARM thunks can always reach the
1533// callee as they use an indirect jump via a register that has no range
1534// restrictions.
1535//
1536// Thunk placement algorithm:
1537// For Mips LA25 ThunkSections; the placement is explicit, it has to be before
1538// getTargetInputSection().
1539//
1540// For thunks that must be placed within range of the caller there are many
1541// possible choices given that the maximum range from the caller is usually
1542// much larger than the average InputSection size. Desirable properties include:
1543// - Maximize reuse of thunks by multiple callers
1544// - Minimize number of ThunkSections to simplify insertion
1545// - Handle impact of already added Thunks on addresses
1546// - Simple to understand and implement
1547//
1548// In lld for the first pass, we pre-create one or more ThunkSections per
1549// InputSectionDescription at Target specific intervals. A ThunkSection is
1550// placed so that the estimated end of the ThunkSection is within range of the
1551// start of the InputSectionDescription or the previous ThunkSection. For
1552// example:
1553// InputSectionDescription
1554// Section 0
1555// ...
1556// Section N
1557// ThunkSection 0
1558// Section N + 1
1559// ...
1560// Section N + K
1561// Thunk Section 1
1562//
1563// The intention is that we can add a Thunk to a ThunkSection that is well
1564// spaced enough to service a number of callers without having to do a lot
1565// of work. An important principle is that it is not an error if a Thunk cannot
1566// be placed in a pre-created ThunkSection; when this happens we create a new
1567// ThunkSection placed next to the caller. This allows us to handle the vast
1568// majority of thunks simply, but also handle rare cases where the branch range
1569// is smaller than the target specific spacing.
1570//
1571// The algorithm is expected to create all the thunks that are needed in a
1572// single pass, with a small number of programs needing a second pass due to
1573// the insertion of thunks in the first pass increasing the offset between
1574// callers and callees that were only just in range.
1575//
1576// A consequence of allowing new ThunkSections to be created outside of the
1577// pre-created ThunkSections is that in rare cases calls to Thunks that were in
1578// range in pass K, are out of range in some pass > K due to the insertion of
1579// more Thunks in between the caller and callee. When this happens we retarget
1580// the relocation back to the original target and create another Thunk.
1581
1582// Remove ThunkSections that are empty, this should only be the initial set
1583// precreated on pass 0.
1584
1585// Insert the Thunks for OutputSection OS into their designated place
1586// in the Sections vector, and recalculate the InputSection output section
1587// offsets.
1588// This may invalidate any output section offsets stored outside of InputSection
1589void ThunkCreator::mergeThunks(ArrayRef<OutputSection *> OutputSections) {
1590 forEachInputSectionDescription(
1591 OutputSections, [&](OutputSection *OS, InputSectionDescription *ISD) {
1592 if (ISD->ThunkSections.empty())
1593 return;
1594
1595 // Remove any zero sized precreated Thunks.
1596 llvm::erase_if(ISD->ThunkSections,
1597 [](const std::pair<ThunkSection *, uint32_t> &TS) {
1598 return TS.first->getSize() == 0;
1599 });
1600
1601 // ISD->ThunkSections contains all created ThunkSections, including
1602 // those inserted in previous passes. Extract the Thunks created this
1603 // pass and order them in ascending OutSecOff.
1604 std::vector<ThunkSection *> NewThunks;
1605 for (const std::pair<ThunkSection *, uint32_t> TS : ISD->ThunkSections)
1606 if (TS.second == Pass)
1607 NewThunks.push_back(TS.first);
1608 llvm::stable_sort(NewThunks,
1609 [](const ThunkSection *A, const ThunkSection *B) {
1610 return A->OutSecOff < B->OutSecOff;
1611 });
1612
1613 // Merge sorted vectors of Thunks and InputSections by OutSecOff
1614 std::vector<InputSection *> Tmp;
1615 Tmp.reserve(ISD->Sections.size() + NewThunks.size());
1616
1617 std::merge(ISD->Sections.begin(), ISD->Sections.end(),
1618 NewThunks.begin(), NewThunks.end(), std::back_inserter(Tmp),
1619 mergeCmp);
1620
1621 ISD->Sections = std::move(Tmp);
1622 });
1623}

1625// Find or create a ThunkSection within the InputSectionDescription (ISD) that
1626// is in range of Src. An ISD maps to a range of InputSections described by a
1627// linker script section pattern such as { .text .text.* }.
1628ThunkSection *ThunkCreator::getISDThunkSec(OutputSection *OS, InputSection *IS,
1629 InputSectionDescription *ISD,
1630 uint32_t Type, uint64_t Src) {
1631 for (std::pair<ThunkSection *, uint32_t> TP : ISD->ThunkSections) {
1632 ThunkSection *TS = TP.first;
1633 uint64_t TSBase = OS->Addr + TS->OutSecOff;
1634 uint64_t TSLimit = TSBase + TS->getSize();
1635 if (Target->inBranchRange(Type, Src, (Src > TSLimit) ? TSBase : TSLimit))
1636 return TS;
1637 }
1638
1639 // No suitable ThunkSection exists. This can happen when there is a branch
1640 // with lower range than the ThunkSection spacing or when there are too
1641 // many Thunks. Create a new ThunkSection as close to the InputSection as
1642 // possible. Error if InputSection is so large we cannot place ThunkSection
1643 // anywhere in Range.
1644 uint64_t ThunkSecOff = IS->OutSecOff;
1645 if (!Target->inBranchRange(Type, Src, OS->Addr + ThunkSecOff)) {
1646 ThunkSecOff = IS->OutSecOff + IS->getSize();
1647 if (!Target->inBranchRange(Type, Src, OS->Addr + ThunkSecOff))
1648 fatal("InputSection too large for range extension thunk " +
1649 IS->getObjMsg(Src - (OS->Addr + IS->OutSecOff)));
1650 }
1651 return addThunkSection(OS, ISD, ThunkSecOff);
1652}

1654// Add a Thunk that needs to be placed in a ThunkSection that immediately
1655// precedes its Target.
1656ThunkSection *ThunkCreator::getISThunkSec(InputSection *IS) {
1657 ThunkSection *TS = ThunkedSections.lookup(IS);
1658 if (TS)
1659 return TS;
1660
1661 // Find InputSectionRange within Target Output Section (TOS) that the
1662 // InputSection (IS) that we need to precede is in.
1663 OutputSection *TOS = IS->getParent();
1664 for (BaseCommand *BC : TOS->SectionCommands) {
1665 auto *ISD = dyn_cast<InputSectionDescription>(BC);
1666 if (!ISD || ISD->Sections.empty())
1667 continue;
1668
1669 InputSection *First = ISD->Sections.front();
1670 InputSection *Last = ISD->Sections.back();
1671
1672 if (IS->OutSecOff < First->OutSecOff || Last->OutSecOff < IS->OutSecOff)
1673 continue;
1674
1675 TS = addThunkSection(TOS, ISD, IS->OutSecOff);
1676 ThunkedSections[IS] = TS;
1677 return TS;
1678 }
1679
1680 return nullptr;
1681}

1683// Create one or more ThunkSections per OS that can be used to place Thunks.
1684// We attempt to place the ThunkSections using the following desirable
1685// properties:
1686// - Within range of the maximum number of callers
1687// - Minimise the number of ThunkSections
1688//
1689// We follow a simple but conservative heuristic to place ThunkSections at
1690// offsets that are multiples of a Target specific branch range.
1691// For an InputSectionDescription that is smaller than the range, a single
1692// ThunkSection at the end of the range will do.
1693//
1694// For an InputSectionDescription that is more than twice the size of the range,
1695// we place the last ThunkSection at range bytes from the end of the
1696// InputSectionDescription in order to increase the likelihood that the
1697// distance from a thunk to its target will be sufficiently small to
1698// allow for the creation of a short thunk.
1699void ThunkCreator::createInitialThunkSections(
1700 ArrayRef<OutputSection *> OutputSections) {
1701 uint32_t ThunkSectionSpacing = Target->getThunkSectionSpacing();
1702
1703 forEachInputSectionDescription(
1704 OutputSections, [&](OutputSection *OS, InputSectionDescription *ISD) {
1705 if (ISD->Sections.empty())
1706 return;
1707
1708 uint32_t ISDBegin = ISD->Sections.front()->OutSecOff;
1709 uint32_t ISDEnd =
1710 ISD->Sections.back()->OutSecOff + ISD->Sections.back()->getSize();
1711 uint32_t LastThunkLowerBound = -1;
1712 if (ISDEnd - ISDBegin > ThunkSectionSpacing * 2)
1713 LastThunkLowerBound = ISDEnd - ThunkSectionSpacing;
1714
1715 uint32_t ISLimit;
1716 uint32_t PrevISLimit = ISDBegin;
1717 uint32_t ThunkUpperBound = ISDBegin + ThunkSectionSpacing;
1718
1719 for (const InputSection *IS : ISD->Sections) {
1720 ISLimit = IS->OutSecOff + IS->getSize();
1721 if (ISLimit > ThunkUpperBound) {
1722 addThunkSection(OS, ISD, PrevISLimit);
1723 ThunkUpperBound = PrevISLimit + ThunkSectionSpacing;
1724 }
1725 if (ISLimit > LastThunkLowerBound)
1726 break;
1727 PrevISLimit = ISLimit;
1728 }
1729 addThunkSection(OS, ISD, ISLimit);
1730 });
1731}

1734 InputSectionDescription *ISD,
1735 uint64_t Off) {
1736 auto *TS = make<ThunkSection>(OS, Off);
1737 TS->Partition = OS->Partition;
1738 ISD->ThunkSections.push_back({TS, Pass});
1739 return TS;
1740}

1742static bool isThunkSectionCompatible(InputSection *Source,
1743 SectionBase *Target) {
1744 // We can't reuse thunks in different loadable partitions because they might
1745 // not be loaded. But partition 1 (the main partition) will always be loaded.
1746 if (Source->Partition != Target->Partition)
1747 return Target->Partition == 1;
1748 return true;

1751std::pair<Thunk *, bool> ThunkCreator::getThunk(InputSection *IS,
1752 Relocation &Rel, uint64_t Src) {
1753 std::vector<Thunk *> *ThunkVec = nullptr;
1754
1755 // We use (section, offset) pair to find the thunk position if possible so
1756 // that we create only one thunk for aliased symbols or ICFed sections.
1757 if (auto *D = dyn_cast<Defined>(Rel.Sym))
1758 if (!D->isInPlt() && D->Section)
1759 ThunkVec = &ThunkedSymbolsBySection[{D->Section->Repl, D->Value}];
1760 if (!ThunkVec)
1761 ThunkVec = &ThunkedSymbols[Rel.Sym];
1762
1763 // Check existing Thunks for Sym to see if they can be reused
1764 for (Thunk *T : *ThunkVec)
1765 if (isThunkSectionCompatible(IS, T->getThunkTargetSym()->Section) &&
1766 T->isCompatibleWith(*IS, Rel) &&
1767 Target->inBranchRange(Rel.Type, Src, T->getThunkTargetSym()->getVA()))
1768 return std::make_pair(T, false);
1769
1770 // No existing compatible Thunk in range, create a new one
1771 Thunk *T = addThunk(*IS, Rel);
1772 ThunkVec->push_back(T);
1773 return std::make_pair(T, true);
1774}
1775
1776// Return true if the relocation target is an in range Thunk.
1777// Return false if the relocation is not to a Thunk. If the relocation target
1778// was originally to a Thunk, but is no longer in range we revert the
1779// relocation back to its original non-Thunk target.
1780bool ThunkCreator::normalizeExistingThunk(Relocation &Rel, uint64_t Src) {
1781 if (Thunk *T = Thunks.lookup(Rel.Sym)) {
1782 if (Target->inBranchRange(Rel.Type, Src, Rel.Sym->getVA()))
1783 return true;
1784 Rel.Sym = &T->Destination;
1785 if (Rel.Sym->isInPlt())
1786 Rel.Expr = toPlt(Rel.Expr);
1787 }
1788 return false;
1789}

1791// Process all relocations from the InputSections that have been assigned
1792// to InputSectionDescriptions and redirect through Thunks if needed. The
1793// function should be called iteratively until it returns false.
1794//
1795// PreConditions:
1796// All InputSections that may need a Thunk are reachable from
1797// OutputSectionCommands.
1798//
1799// All OutputSections have an address and all InputSections have an offset
1800// within the OutputSection.
1801//
1802// The offsets between caller (relocation place) and callee
1803// (relocation target) will not be modified outside of createThunks().
1804//
1805// PostConditions:
1806// If return value is true then ThunkSections have been inserted into
1807// OutputSections. All relocations that needed a Thunk based on the information
1808// available to createThunks() on entry have been redirected to a Thunk. Note
1809// that adding Thunks changes offsets between caller and callee so more Thunks
1810// may be required.
1811//
1812// If return value is false then no more Thunks are needed, and createThunks has
1813// made no changes. If the target requires range extension thunks, currently
1814// ARM, then any future change in offset between caller and callee risks a
1815// relocation out of range error.
1816bool ThunkCreator::createThunks(ArrayRef<OutputSection *> OutputSections) {
1817 bool AddressesChanged = false;
1818
1819 if (Pass == 0 && Target->getThunkSectionSpacing())
1820 createInitialThunkSections(OutputSections);
1821
1822 // With Thunk Size much smaller than branch range we expect to
1823 // converge quickly; if we get to 10 something has gone wrong.
1824 if (Pass == 10)
1825 fatal("thunk creation not converged");
1826
1827 // Create all the Thunks and insert them into synthetic ThunkSections. The
1828 // ThunkSections are later inserted back into InputSectionDescriptions.
1829 // We separate the creation of ThunkSections from the insertion of the
1830 // ThunkSections as ThunkSections are not always inserted into the same
1831 // InputSectionDescription as the caller.
1832 forEachInputSectionDescription(
1833 OutputSections, [&](OutputSection *OS, InputSectionDescription *ISD) {
1834 for (InputSection *IS : ISD->Sections)
1835 for (Relocation &Rel : IS->Relocations) {
1836 uint64_t Src = IS->getVA(Rel.Offset);
1837
1838 // If we are a relocation to an existing Thunk, check if it is
1839 // still in range. If not then Rel will be altered to point to its
1840 // original target so another Thunk can be generated.
1841 if (Pass > 0 && normalizeExistingThunk(Rel, Src))
1842 continue;
1843
1844 if (!Target->needsThunk(Rel.Expr, Rel.Type, IS->File, Src,
1845 *Rel.Sym))
1846 continue;
1847
1848 Thunk *T;
1849 bool IsNew;
1850 std::tie(T, IsNew) = getThunk(IS, Rel, Src);
1851
1852 if (IsNew) {
1853 // Find or create a ThunkSection for the new Thunk
1854 ThunkSection *TS;
1855 if (auto *TIS = T->getTargetInputSection())
1856 TS = getISThunkSec(TIS);
1857 else
1858 TS = getISDThunkSec(OS, IS, ISD, Rel.Type, Src);
1859 TS->addThunk(T);
1860 Thunks[T->getThunkTargetSym()] = T;
1861 }
1862
1863 // Redirect relocation to Thunk, we never go via the PLT to a Thunk
1864 Rel.Sym = T->getThunkTargetSym();
1865 Rel.Expr = fromPlt(Rel.Expr);
1866
1867 // Addend of R_PPC_PLTREL24 should be ignored after changing to R_PC.
1868 if (Config->EMachine == EM_PPC && Rel.Type == R_PPC_PLTREL24)
1869 Rel.Addend = 0;
1870 }
1871
1872 for (auto &P : ISD->ThunkSections)
1873 AddressesChanged |= P.first->assignOffsets();
1874 });
1875
1876 for (auto &P : ThunkedSections)
1877 AddressesChanged |= P.second->assignOffsets();
1878
1879 // Merge all created synthetic ThunkSections back into OutputSection
1880 mergeThunks(OutputSections);
1881 ++Pass;
1882 return AddressesChanged;
1883}
