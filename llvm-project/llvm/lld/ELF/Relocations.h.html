25// Represents a relocation type, such as R_X86_64_PC32 or R_ARM_THM_CALL.
26using RelType = uint32_t;

28// List of target-independent relocation types. Relocations read
29// from files are converted to these types so that the main code
30// doesn't have to know about architecture-specific details.

32 R_ABS,
33 R_ADDEND,
34 R_DTPREL,
35 R_GOT,
36 R_GOT_OFF,
37 R_GOT_PC,
38 R_GOTONLY_PC,
39 R_GOTPLTONLY_PC,
40 R_GOTPLT,
41 R_GOTPLTREL,
42 R_GOTREL,
43 R_HINT,
44 R_NEG_TLS,
45 R_NONE,
46 R_PC,
47 R_PLT,
48 R_PLT_PC,
49 R_RELAX_GOT_PC,
50 R_RELAX_GOT_PC_NOPIC,
51 R_RELAX_TLS_GD_TO_IE,
52 R_RELAX_TLS_GD_TO_IE_ABS,
53 R_RELAX_TLS_GD_TO_IE_GOT_OFF,
54 R_RELAX_TLS_GD_TO_IE_GOTPLT,
55 R_RELAX_TLS_GD_TO_LE,
56 R_RELAX_TLS_GD_TO_LE_NEG,
57 R_RELAX_TLS_IE_TO_LE,
58 R_RELAX_TLS_LD_TO_LE,
59 R_RELAX_TLS_LD_TO_LE_ABS,
60 R_SIZE,
61 R_TLS,
62 R_TLSDESC,
63 R_TLSDESC_CALL,
64 R_TLSDESC_PC,
65 R_TLSGD_GOT,
66 R_TLSGD_GOTPLT,
67 R_TLSGD_PC,
68 R_TLSIE_HINT,
69 R_TLSLD_GOT,
70 R_TLSLD_GOTPLT,
71 R_TLSLD_GOT_OFF,

75 // The following is abstract relocation types used for only one target.

77 // Even though RelExpr is intended to be a target-neutral representation

80 R_AARCH64_GOT_PAGE_PC,
81 R_AARCH64_PAGE_PC,
82 R_AARCH64_RELAX_TLS_GD_TO_IE_PAGE_PC,
83 R_AARCH64_TLSDESC_PAGE,
84 R_ARM_SBREL,
85 R_HEXAGON_GOT,
86 R_MIPS_GOTREL,
87 R_MIPS_GOT_GP,

90 R_MIPS_GOT_OFF,
91 R_MIPS_GOT_OFF32,
92 R_MIPS_TLSGD,
93 R_MIPS_TLSLD,
94 R_PPC32_PLTREL,
95 R_PPC64_CALL,
96 R_PPC64_CALL_PLT,
97 R_PPC64_RELAX_TOC,
98 R_PPC64_TOCBASE,
99 R_RISCV_ADD,
100 R_RISCV_PC_INDIRECT,
101 R_CHERI_CAPABILITY_TABLE_INDEX,
102 R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE,
103 R_CHERI_CAPABILITY_TABLE_INDEX_CALL,
104 R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE,
105 R_CHERI_CAPABILITY_TABLE_REL, // relative offset to _CHERI_CAPABILITY_TABLE_
106 R_MIPS_CHERI_CAPTAB_TLSGD,
107 R_MIPS_CHERI_CAPTAB_TLSLD,
108 R_MIPS_CHERI_CAPTAB_TPREL,
109 R_CHERI_CAPABILITY
110};

112// Architecture-neutral representation of relocation.
113struct Relocation {
114 RelExpr Expr;
115 RelType Type;
116 uint64_t Offset;
117 int64_t Addend;
118 Symbol *Sym;
119};

125class ThunkSection;
126class Thunk;
127struct InputSectionDescription;
128
129class ThunkCreator {
130public:
131 // Return true if Thunks have been added to OutputSections
132 bool createThunks(ArrayRef<OutputSection *> OutputSections);
133
134 // The number of completed passes of createThunks this permits us
135 // to do one time initialization on Pass 0 and put a limit on the
136 // number of times it can be called to prevent infinite loops.
137 uint32_t Pass = 0;
138
139private:
140 void mergeThunks(ArrayRef<OutputSection *> OutputSections);
141
142 ThunkSection *getISDThunkSec(OutputSection *OS, InputSection *IS,
143 InputSectionDescription *ISD, uint32_t Type,
144 uint64_t Src);
145
146 ThunkSection *getISThunkSec(InputSection *IS);
147
148 void createInitialThunkSections(ArrayRef<OutputSection *> OutputSections);
149
150 std::pair<Thunk *, bool> getThunk(InputSection *IS, Relocation &Rel,
151 uint64_t Src);
152
153 ThunkSection *addThunkSection(OutputSection *OS, InputSectionDescription *,
154 uint64_t Off);
155
156 bool normalizeExistingThunk(Relocation &Rel, uint64_t Src);
157
158 // Record all the available Thunks for a Symbol
159 llvm::DenseMap<std::pair<SectionBase *, uint64_t>, std::vector<Thunk *>>
160 ThunkedSymbolsBySection;
161 llvm::DenseMap<Symbol *, std::vector<Thunk *>> ThunkedSymbols;
162
163 // Find a Thunk from the Thunks symbol definition, we can use this to find
164 // the Thunk from a relocation to the Thunks symbol definition.
165 llvm::DenseMap<Symbol *, Thunk *> Thunks;
166
167 // Track InputSections that have an inline ThunkSection placed in front
168 // an inline ThunkSection may have control fall through to the section below
169 // so we need to make sure that there is only one of them.
170 // The Mips LA25 Thunk is an example of an inline ThunkSection.
171 llvm::DenseMap<InputSection *, ThunkSection *> ThunkedSections;
172};

174// Return a int64_t to make sure we get the sign extension out of the way as
175// early as possible.
176template <class ELFT>
177static inline int64_t getAddend(const typename ELFT::Rel &Rel) {
178 return 0;
179}
180template <class ELFT>
181static inline int64_t getAddend(const typename ELFT::Rela &Rel) {
182 return Rel.r_addend;
183}
