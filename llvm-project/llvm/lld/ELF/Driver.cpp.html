9// The driver drives the entire linking process. It is responsible for
10// parsing command line options and doing whatever it is instructed to do.
11//
12// One notable thing in the LLD's driver when compared to other linkers is
13// that the LLD's driver is agnostic on the host operating system.
14// Other linkers usually have implicit default values (such as a dynamic
15// linker path or library paths) for each host OS.
16//
17// I don't think implicit default values are useful because they are
18// usually explicitly specified by the compiler driver. They can even
19// be harmful when you are doing cross-linking. Therefore, in LLD, we
20// simply trust the compiler driver to pass all required options and
21// don't try to make effort on our side.
22//
23//===----------------------------------------------------------------------===//

76bool elf::link(ArrayRef<const char *> Args, bool CanExitEarly,
77 raw_ostream &Error) {
78 errorHandler().LogName = args::getFilenameWithoutExe(Args[0]);
79 errorHandler().ErrorLimitExceededMsg =
80 "too many errors emitted, stopping now (use "
81 "-error-limit=0 to see all errors)";
82 errorHandler().WarningLimitExceededMsg =
83 "too many warnings emitted, stopping now (use "
84 "-warning-limit=0 to see all warnings)\n";
85 errorHandler().ErrorOS = &Error;
86 errorHandler().ExitEarly = CanExitEarly;
87 errorHandler().ColorDiagnostics = Error.has_colors();
88
89 InputSections.clear();
90 OutputSections.clear();
91 BinaryFiles.clear();
92 BitcodeFiles.clear();
93 ObjectFiles.clear();
94 SharedFiles.clear();
95
96 Config = make<Configuration>();
97 Driver = make<LinkerDriver>();
98 Script = make<LinkerScript>();
99 Symtab = make<SymbolTable>();
100
101 Tar = nullptr;
102 memset(&In, 0, sizeof(In));
103
104 Partitions = {Partition()};
105
106 SharedFile::VernauxNum = 0;
107
108 Config->ProgName = Args[0];
109
110 Driver->main(Args);
111
112 // Exit immediately if we don't need to return to the caller.
113 // This saves time because the overhead of calling destructors
114 // for all globally-allocated objects is not negligible.
115 if (CanExitEarly)
116 exitLld(errorCount() ? 1 : 0);
117
118 freeArena();
119 return !errorCount();
120}

122// Parses a linker -m option.
123static std::tuple<ELFKind, uint16_t, uint8_t, bool> parseEmulation(
124 StringRef Emul) {
125 uint8_t OSABI = 0;
126 bool IsCheriABI = false;
127 StringRef S = Emul;
128 if (S.endswith("_fbsd")) {
129 S = S.drop_back(5);
130 OSABI = ELFOSABI_FREEBSD;
131 if (S.endswith("_cheri")) {
132 S = S.drop_back(6);
133 IsCheriABI = true;
134 }
135 }
136
137 std::pair<ELFKind, uint16_t> Ret =
138 StringSwitch<std::pair<ELFKind, uint16_t>>(S)
139 .Cases("aarch64elf", "aarch64linux", "aarch64_elf64_le_vec",
140 {ELF64LEKind, EM_AARCH64})
141 .Cases("armelf", "armelf_linux_eabi", {ELF32LEKind, EM_ARM})
142 .Case("elf32_x86_64", {ELF32LEKind, EM_X86_64})
143 .Cases("elf32btsmip", "elf32btsmipn32", {ELF32BEKind, EM_MIPS})
144 .Cases("elf32ltsmip", "elf32ltsmipn32", {ELF32LEKind, EM_MIPS})
145 .Case("elf32lriscv", {ELF32LEKind, EM_RISCV})
146 .Cases("elf32ppc", "elf32ppclinux", {ELF32BEKind, EM_PPC})
147 .Case("elf64btsmip", {ELF64BEKind, EM_MIPS})
148 .Case("elf64ltsmip", {ELF64LEKind, EM_MIPS})
149 .Case("elf64lriscv", {ELF64LEKind, EM_RISCV})
150 .Case("elf64ppc", {ELF64BEKind, EM_PPC64})
151 .Case("elf64lppc", {ELF64LEKind, EM_PPC64})
152 .Cases("elf_amd64", "elf_x86_64", {ELF64LEKind, EM_X86_64})
153 .Case("elf_i386", {ELF32LEKind, EM_386})
154 .Case("elf_iamcu", {ELF32LEKind, EM_IAMCU})
155 .Default({ELFNoneKind, EM_NONE});
156
157 if (Ret.first == ELFNoneKind)
158 error("unknown emulation: " + Emul);
159 return std::make_tuple(Ret.first, Ret.second, OSABI, IsCheriABI);
160}

162// Returns slices of MB by parsing MB as an archive file.
163// Each slice consists of a member file in the archive.
164std::vector<std::pair<MemoryBufferRef, uint64_t>> static getArchiveMembers(
165 MemoryBufferRef MB) {
166 std::unique_ptr<Archive> File =
167 CHECK(Archive::create(MB),
168 MB.getBufferIdentifier() + ": failed to parse archive");
169
170 std::vector<std::pair<MemoryBufferRef, uint64_t>> V;
171 Error Err = Error::success();
172 bool AddToTar = File->isThin() && Tar;
173 for (const ErrorOr<Archive::Child> &COrErr : File->children(Err)) {
174 Archive::Child C =
175 CHECK(COrErr, MB.getBufferIdentifier() +
176 ": could not get the child of the archive");
177 MemoryBufferRef MBRef =
178 CHECK(C.getMemoryBufferRef(),
179 MB.getBufferIdentifier() +
180 ": could not get the buffer for a child of the archive");
181 if (AddToTar)
182 Tar->append(relativeToRoot(check(C.getFullName())), MBRef.getBuffer());
183 V.push_back(std::make_pair(MBRef, C.getChildOffset()));
184 }
185 if (Err)
186 fatal(MB.getBufferIdentifier() + ": Archive::children failed: " +
187 toString(std::move(Err)));
188
189 // Take ownership of memory buffers created for members of thin archives.
190 for (std::unique_ptr<MemoryBuffer> &MB : File->takeThinBuffers())
191 make<std::unique_ptr<MemoryBuffer>>(std::move(MB));
192
193 return V;
194}

196// Opens a file and create a file object. Path has to be resolved already.
197void LinkerDriver::addFile(StringRef Path, bool WithLOption) {
198 using namespace sys::fs;
199
200 Optional<MemoryBufferRef> Buffer = readFile(Path);
201 if (!Buffer.hasValue())
202 return;
203 MemoryBufferRef MBRef = *Buffer;
204
205 if (Config->FormatBinary) {
206 Files.push_back(make<BinaryFile>(MBRef));
207 return;
208 }
209
210 switch (identify_magic(MBRef.getBuffer())) {
211 case file_magic::unknown:
212 readLinkerScript(MBRef);
213 return;
214 case file_magic::archive: {
215 // Handle -whole-archive.
216 if (InWholeArchive) {
217 for (const auto &P : getArchiveMembers(MBRef))
218 Files.push_back(createObjectFile(P.first, Path, P.second));
219 return;
220 }
221
222 std::unique_ptr<Archive> File =
223 CHECK(Archive::create(MBRef), Path + ": failed to parse archive");
224
225 // If an archive file has no symbol table, it is likely that a user
226 // is attempting LTO and using a default ar command that doesn't
227 // understand the LLVM bitcode file. It is a pretty common error, so
228 // we'll handle it as if it had a symbol table.
229 if (!File->isEmpty() && !File->hasSymbolTable()) {
230 // Check if all members are bitcode files. If not, ignore, which is the
231 // default action without the LTO hack described above.
232 for (const std::pair<MemoryBufferRef, uint64_t> &P :
233 getArchiveMembers(MBRef))
234 if (identify_magic(P.first.getBuffer()) != file_magic::bitcode)
235 return;
236
237 for (const std::pair<MemoryBufferRef, uint64_t> &P :
238 getArchiveMembers(MBRef))
239 Files.push_back(make<LazyObjFile>(P.first, Path, P.second));
240 return;
241 }
242
243 // Handle the regular case.
244 Files.push_back(make<ArchiveFile>(std::move(File)));
245 return;
246 }
247 case file_magic::elf_shared_object:
248 if (Config->Static || Config->Relocatable) {
249 error("attempted static link of dynamic object " + Path);
250 return;
251 }
252
253 // DSOs usually have DT_SONAME tags in their ELF headers, and the
254 // sonames are used to identify DSOs. But if they are missing,
255 // they are identified by filenames. We don't know whether the new
256 // file has a DT_SONAME or not because we haven't parsed it yet.
257 // Here, we set the default soname for the file because we might
258 // need it later.
259 //
260 // If a file was specified by -lfoo, the directory part is not
261 // significant, as a user did not specify it. This behavior is
262 // compatible with GNU.
263 Files.push_back(
264 make<SharedFile>(MBRef, WithLOption ? path::filename(Path) : Path));
265 return;
266 case file_magic::bitcode:
267 case file_magic::elf_relocatable:
268 if (InLib)
269 Files.push_back(make<LazyObjFile>(MBRef, "", 0));
270 else
271 Files.push_back(createObjectFile(MBRef));
272 break;
273 default:
274 error(Path + ": unknown file type");
275 }
276}

278// Add a given library by searching it from input search paths.
279void LinkerDriver::addLibrary(StringRef Name) {
280 if (Optional<std::string> Path = searchLibrary(Name))
281 addFile(*Path, /*WithLOption=*/true);
282 else
283 error("unable to find library -l" + Name);
284}

297// Some command line options or some combinations of them are not allowed.
298// This function checks for such errors.
299static void checkOptions() {
300 // The MIPS ABI as of 2016 does not support the GNU-style symbol lookup
301 // table which is a relatively new feature.
302 if (Config->EMachine == EM_MIPS && Config->GnuHash)
303 error("the .gnu.hash section is not compatible with the MIPS target");
304
305 if (Config->FixCortexA53Errata843419 && Config->EMachine != EM_AARCH64)
306 error("--fix-cortex-a53-843419 is only supported on AArch64 targets");
307
308 if (Config->TocOptimize && Config->EMachine != EM_PPC64)
309 error("--toc-optimize is only supported on the PowerPC64 target");
310
311 if (Config->Pie && Config->Shared)
312 error("-shared and -pie may not be used together");
313
314 if (!Config->Shared && !Config->FilterList.empty())
315 error("-F may not be used without -shared");
316
317 if (!Config->Shared && !Config->AuxiliaryList.empty())
318 error("-f may not be used without -shared");
319
320 if (!Config->Relocatable && !Config->DefineCommon)
321 error("-no-define-common not supported in non relocatable output");
322
323 if (Config->ZText && Config->ZIfuncNoplt)
324 error("-z text and -z ifunc-noplt may not be used together");
325
326 if (Config->Relocatable) {
327 if (Config->Shared)
328 error("-r and -shared may not be used together");
329 if (Config->GcSections)
330 error("-r and --gc-sections may not be used together");
331 if (Config->GdbIndex)
332 error("-r and --gdb-index may not be used together");
333 if (Config->ICF != ICFLevel::None)
334 error("-r and --icf may not be used together");
335 if (Config->Pie)
336 error("-r and -pie may not be used together");
337 }
338 if (Config->LocalCapRelocsMode == CapRelocsMode::ElfReloc)
339 error("local-cap-relocs=elf is not implemented yet");
340 if (Config->LocalCapRelocsMode == CapRelocsMode::CBuildCap)
341 error("local-cap-relocs=cbuildcap is not implemented yet");
342 assert(Config->PreemptibleCapRelocsMode != CapRelocsMode::CBuildCap);
343
344 if (Config->PreemptibleCapRelocsMode == CapRelocsMode::Legacy &&
345 Config->RelativeCapRelocsOnly)
346 error("--preemptible-caprelocs=legacy is not compatible with "
347 "--relative-cap-relocs");
348
349 if (Config->ExecuteOnly) {
350 if (Config->EMachine != EM_AARCH64)
351 error("-execute-only is only supported on AArch64 targets");
352
353 if (Config->SingleRoRx && !Script->HasSectionsCommand)
354 error("-execute-only and -no-rosegment cannot be used together");
355 }
356
357 if (Config->ZRetpolineplt && Config->RequireCET)
358 error("--require-cet may not be used with -z retpolineplt");
359
360 if (Config->EMachine != EM_AARCH64) {
361 if (Config->PacPlt)
362 error("--pac-plt only supported on AArch64");
363 if (Config->ForceBTI)
364 error("--force-bti only supported on AArch64");
365 }
366}

413void LinkerDriver::main(ArrayRef<const char *> ArgsArr) {
414 ELFOptTable Parser;
415 opt::InputArgList Args = Parser.parse(ArgsArr.slice(1));
416
417 // Interpret this flag early because error() depends on them.
418 errorHandler().ErrorLimit = args::getInteger(Args, OPT_error_limit, 20);
419 errorHandler().WarningLimit = args::getInteger(Args, OPT_warning_limit, 20);
420 checkZOptions(Args);
421
422 // Handle -help
423 if (Args.hasArg(OPT_help)) {
424 printHelp();
425 return;
426 }
427
428 // Handle -v or -version.
429 //
430 // A note about "compatible with GNU linkers" message: this is a hack for
431 // scripts generated by GNU Libtool 2.4.6 (released in February 2014 and
432 // still the newest version in March 2017) or earlier to recognize LLD as
433 // a GNU compatible linker. As long as an output for the -v option
434 // contains "GNU" or "with BFD", they recognize us as GNU-compatible.
435 //
436 // This is somewhat ugly hack, but in reality, we had no choice other
437 // than doing this. Considering the very long release cycle of Libtool,
438 // it is not easy to improve it to recognize LLD as a GNU compatible
439 // linker in a timely manner. Even if we can make it, there are still a
440 // lot of "configure" scripts out there that are generated by old version
441 // of Libtool. We cannot convince every software developer to migrate to
442 // the latest version and re-generate scripts. So we have this hack.
443 if (Args.hasArg(OPT_v) || Args.hasArg(OPT_version))
444 message(getLLDVersion() + " (compatible with GNU linkers)");
445
446 if (const char *Path = getReproduceOption(Args)) {
447 // Note that --reproduce is a debug option so you can ignore it
448 // if you are trying to understand the whole picture of the code.
449 Expected<std::unique_ptr<TarWriter>> ErrOrWriter =
450 TarWriter::create(Path, path::stem(Path));
451 if (ErrOrWriter) {
452 Tar = std::move(*ErrOrWriter);
453 Tar->append("response.txt", createResponseFile(Args));
454 Tar->append("version.txt", getLLDVersion() + "\n");
455 } else {
456 error("--reproduce: " + toString(ErrOrWriter.takeError()));
457 }
458 }
459
460 readConfigs(Args);
461
462 // The behavior of -v or --version is a bit strange, but this is
463 // needed for compatibility with GNU linkers.
464 if (Args.hasArg(OPT_v) && !Args.hasArg(OPT_INPUT))
465 return;
466 if (Args.hasArg(OPT_version))
467 return;
468
469 initLLVM();
470 createFiles(Args);
471 if (errorCount())
472 return;
473
474 inferMachineType();
475 setConfigs(Args);
476 checkOptions();
477 if (errorCount())
478 return;
479
480 // The Target instance handles target-specific stuff, such as applying
481 // relocations or writing a PLT section. It also contains target-dependent
482 // values such as a default image base address.
483 Target = getTarget();
484
485 switch (Config->EKind) {
486 case ELF32LEKind:
487 link<ELF32LE>(Args);
488 return;
489 case ELF32BEKind:
490 link<ELF32BE>(Args);
491 return;
492 case ELF64LEKind:
493 link<ELF64LE>(Args);
494 return;
495 case ELF64BEKind:
496 link<ELF64BE>(Args);
497 return;
498 default:
499 llvm_unreachable("unknown Config->EKind");
500 }
501}

508// Determines what we should do if there are remaining unresolved
509// symbols after the name resolution.
510static UnresolvedPolicy getUnresolvedSymbolPolicy(opt::InputArgList &Args) {
511 UnresolvedPolicy ErrorOrWarn = Args.hasFlag(OPT_error_unresolved_symbols,
512 OPT_warn_unresolved_symbols, true)
513 ? UnresolvedPolicy::ReportError
514 : UnresolvedPolicy::Warn;
515
516 // Process the last of -unresolved-symbols, -no-undefined or -z defs.
517 for (auto *Arg : llvm::reverse(Args)) {
518 switch (Arg->getOption().getID()) {
519 case OPT_unresolved_symbols: {
520 StringRef S = Arg->getValue();
521 if (S == "ignore-all" || S == "ignore-in-object-files")
522 return UnresolvedPolicy::Ignore;
523 if (S == "ignore-in-shared-libs" || S == "report-all")
524 return ErrorOrWarn;
525 error("unknown --unresolved-symbols value: " + S);
526 continue;
527 }
528 case OPT_no_undefined:
529 return ErrorOrWarn;
530 case OPT_z:
531 if (StringRef(Arg->getValue()) == "defs")
532 return ErrorOrWarn;
533 continue;
534 }
535 }
536
537 // -shared implies -unresolved-symbols=ignore-all because missing
538 // symbols are likely to be resolved at runtime using other DSOs.
539 if (Config->Shared)
540 return UnresolvedPolicy::Ignore;
541 return ErrorOrWarn;
542}

746static void readCallGraph(MemoryBufferRef MB) {
747 // Build a map from symbol name to section
748 DenseMap<StringRef, Symbol *> Map;
749 for (InputFile *File : ObjectFiles)
750 for (Symbol *Sym : File->getSymbols())
751 Map[Sym->getName()] = Sym;
752
753 auto FindSection = [&](StringRef Name) -> InputSectionBase * {
754 Symbol *Sym = Map.lookup(Name);
755 if (!Sym) {
756 if (Config->WarnSymbolOrdering)
757 warn(MB.getBufferIdentifier() + ": no such symbol: " + Name);
758 return nullptr;
759 }
760 maybeWarnUnorderableSymbol(Sym);
761
762 if (Defined *DR = dyn_cast_or_null<Defined>(Sym))
763 return dyn_cast_or_null<InputSectionBase>(DR->Section);
764 return nullptr;
765 };
766
767 for (StringRef Line : args::getLines(MB)) {
768 SmallVector<StringRef, 3> Fields;
769 Line.split(Fields, ' ');
770 uint64_t Count;
771
772 if (Fields.size() != 3 || !to_integer(Fields[2], Count)) {
773 error(MB.getBufferIdentifier() + ": parse error");
774 return;
775 }
776
777 if (InputSectionBase *From = FindSection(Fields[0]))
778 if (InputSectionBase *To = FindSection(Fields[1]))
779 Config->CallGraphProfile[std::make_pair(From, To)] += Count;
780 }
781}

783template <class ELFT> static void readCallGraphsFromObjectFiles() {
784 for (auto File : ObjectFiles) {
785 auto *Obj = cast<ObjFile<ELFT>>(File);
786
787 for (const Elf_CGProfile_Impl<ELFT> &CGPE : Obj->CGProfile) {
788 auto *FromSym = dyn_cast<Defined>(&Obj->getSymbol(CGPE.cgp_from));
789 auto *ToSym = dyn_cast<Defined>(&Obj->getSymbol(CGPE.cgp_to));
790 if (!FromSym || !ToSym)
791 continue;
792
793 auto *From = dyn_cast_or_null<InputSectionBase>(FromSym->Section);
794 auto *To = dyn_cast_or_null<InputSectionBase>(ToSym->Section);
795 if (From && To)
796 Config->CallGraphProfile[{From, To}] += CGPE.cgp_weight;
797 }
798 }
799}

1134// This function initialize such members. See Config.h for the details
1135// of these values.
1136static void setConfigs(opt::InputArgList &Args) {
1137 ELFKind K = Config->EKind;
1138 uint16_t M = Config->EMachine;
1139
1140 Config->CopyRelocs = (Config->Relocatable || Config->EmitRelocs);
1141 Config->Is64 = (K == ELF64LEKind || K == ELF64BEKind);
1142 Config->IsLE = (K == ELF32LEKind || K == ELF64LEKind);
1143 Config->Endianness = Config->IsLE ? endianness::little : endianness::big;
1144 Config->IsMips64EL = (K == ELF64LEKind && M == EM_MIPS);
1145 Config->Pic = Config->Pie || Config->Shared;
1146 Config->PicThunk = Args.hasArg(OPT_pic_veneer, Config->Pic);
1147 Config->Wordsize = Config->Is64 ? 8 : 4;
1148
1149 // ELF defines two different ways to store relocation addends as shown below:
1150 //
1151 // Rel: Addends are stored to the location where relocations are applied.
1152 // Rela: Addends are stored as part of relocation entry.
1153 //
1154 // In other words, Rela makes it easy to read addends at the price of extra
1155 // 4 or 8 byte for each relocation entry. We don't know why ELF defined two
1156 // different mechanisms in the first place, but this is how the spec is
1157 // defined.
1158 //
1159 // You cannot choose which one, Rel or Rela, you want to use. Instead each
1160 // ABI defines which one you need to use. The following expression expresses
1161 // that.
1162 Config->IsRela = M == EM_AARCH64 || M == EM_AMDGPU || M == EM_HEXAGON ||
1163 M == EM_PPC || M == EM_PPC64 || M == EM_RISCV ||
1164 M == EM_X86_64;
1165
1166 // If the output uses REL relocations we must store the dynamic relocation
1167 // addends to the output sections. We also store addends for RELA relocations
1168 // if --apply-dynamic-relocs is used.
1169 // We default to not writing the addends when using RELA relocations since
1170 // any standard conforming tool can find it in r_addend.
1171 Config->WriteAddends = Args.hasFlag(OPT_apply_dynamic_relocs,
1172 OPT_no_apply_dynamic_relocs, false) ||
1173 !Config->IsRela;
1174
1175 // Avoid dynamic relocations for __cap_relocs unless we are building legacy
1176 // TODO: remove once benchmarking is done.
1177 bool RelativeCapRelocsDefault =
1178 !Config->Pic || Config->PreemptibleCapRelocsMode != CapRelocsMode::Legacy;
1179 Config->RelativeCapRelocsOnly =
1180 Args.hasFlag(OPT_relative_cap_relocs, OPT_no_relative_cap_relocs,
1181 RelativeCapRelocsDefault);
1182
1183 Config->TocOptimize =
1184 Args.hasFlag(OPT_toc_optimize, OPT_no_toc_optimize, M == EM_PPC64);
1185}

1198void LinkerDriver::createFiles(opt::InputArgList &Args) {
1199 // For --{push,pop}-state.
1200 std::vector<std::tuple<bool, bool, bool>> Stack;
1201
1202 // Iterate over argv to process input files and positional arguments.
1203 for (auto *Arg : Args) {
1204 switch (Arg->getOption().getUnaliasedOption().getID()) {
1205 case OPT_library:
1206 addLibrary(Arg->getValue());
1207 break;
1208 case OPT_INPUT:
1209 addFile(Arg->getValue(), /*WithLOption=*/false);
1210 break;
1211 case OPT_defsym: {
1212 StringRef From;
1213 StringRef To;
1214 std::tie(From, To) = StringRef(Arg->getValue()).split('=');
1215 if (From.empty() || To.empty())
1216 error("-defsym: syntax error: " + StringRef(Arg->getValue()));
1217 else
1218 readDefsym(From, MemoryBufferRef(To, "-defsym"));
1219 break;
1220 }
1221 case OPT_script:
1222 if (Optional<std::string> Path = searchScript(Arg->getValue())) {
1223 if (Optional<MemoryBufferRef> MB = readFile(*Path))
1224 readLinkerScript(*MB);
1225 break;
1226 }
1227 error(Twine("cannot find linker script ") + Arg->getValue());
1228 break;
1229 case OPT_as_needed:
1230 Config->AsNeeded = true;
1231 break;
1232 case OPT_format:
1233 Config->FormatBinary = isFormatBinary(Arg->getValue());
1234 break;
1235 case OPT_no_as_needed:
1236 Config->AsNeeded = false;
1237 break;
1238 case OPT_Bstatic:
1239 case OPT_omagic:
1240 case OPT_nmagic:
1241 Config->Static = true;
1242 break;
1243 case OPT_Bdynamic:
1244 Config->Static = false;
1245 break;
1246 case OPT_whole_archive:
1247 InWholeArchive = true;
1248 break;
1249 case OPT_no_whole_archive:
1250 InWholeArchive = false;
1251 break;
1252 case OPT_just_symbols:
1253 if (Optional<MemoryBufferRef> MB = readFile(Arg->getValue())) {
1254 Files.push_back(createObjectFile(*MB));
1255 Files.back()->JustSymbols = true;
1256 }
1257 break;
1258 case OPT_start_group:
1259 if (InputFile::IsInGroup)
1260 error("nested --start-group");
1261 InputFile::IsInGroup = true;
1262 break;
1263 case OPT_end_group:
1264 if (!InputFile::IsInGroup)
1265 error("stray --end-group");
1266 InputFile::IsInGroup = false;
1267 ++InputFile::NextGroupId;
1268 break;
1269 case OPT_start_lib:
1270 if (InLib)
1271 error("nested --start-lib");
1272 if (InputFile::IsInGroup)
1273 error("may not nest --start-lib in --start-group");
1274 InLib = true;
1275 InputFile::IsInGroup = true;
1276 break;
1277 case OPT_end_lib:
1278 if (!InLib)
1279 error("stray --end-lib");
1280 InLib = false;
1281 InputFile::IsInGroup = false;
1282 ++InputFile::NextGroupId;
1283 break;
1284 case OPT_push_state:
1285 Stack.emplace_back(Config->AsNeeded, Config->Static, InWholeArchive);
1286 break;
1287 case OPT_pop_state:
1288 if (Stack.empty()) {
1289 error("unbalanced --push-state/--pop-state");
1290 break;
1291 }
1292 std::tie(Config->AsNeeded, Config->Static, InWholeArchive) = Stack.back();
1293 Stack.pop_back();
1294 break;
1295 }
1296 }
1297
1298 if (Files.empty() && errorCount() == 0)
1299 error("no input files");
1300}

1492// Record sections that define symbols mentioned in --keep-unique <symbol>
1493// and symbols referred to by address-significance tables. These sections are
1494// ineligible for ICF.
1495template <class ELFT>
1496static void findKeepUniqueSections(opt::InputArgList &Args) {
1497 for (auto *Arg : Args.filtered(OPT_keep_unique)) {
1498 StringRef Name = Arg->getValue();
1499 auto *D = dyn_cast_or_null<Defined>(Symtab->find(Name));
1500 if (!D || !D->Section) {
1501 warn("could not find symbol " + Name + " to keep unique");
1502 continue;
1503 }
1504 D->Section->KeepUnique = true;
1505 }
1506
1507 // --icf=all --ignore-data-address-equality means that we can ignore
1508 // the dynsym and address-significance tables entirely.
1509 if (Config->ICF == ICFLevel::All && Config->IgnoreDataAddressEquality)
1510 return;
1511
1512 // Symbols in the dynsym could be address-significant in other executables
1513 // or DSOs, so we conservatively mark them as address-significant.
1514 Symtab->forEachSymbol([&](Symbol *Sym) {
1515 if (Sym->includeInDynsym())
1516 markAddrsig(Sym);
1517 });
1518
1519 // Visit the address-significance table in each object file and mark each
1520 // referenced symbol as address-significant.
1521 for (InputFile *F : ObjectFiles) {
1522 auto *Obj = cast<ObjFile<ELFT>>(F);
1523 ArrayRef<Symbol *> Syms = Obj->getSymbols();
1524 if (Obj->AddrsigSec) {
1525 ArrayRef<uint8_t> Contents =
1526 check(Obj->getObj().getSectionContents(Obj->AddrsigSec));
1527 const uint8_t *Cur = Contents.begin();
1528 while (Cur != Contents.end()) {
1529 unsigned Size;
1530 const char *Err;
1531 uint64_t SymIndex = decodeULEB128(Cur, &Size, Contents.end(), &Err);
1532 if (Err)
1533 fatal(toString(F) + ": could not decode addrsig section: " + Err);
1534 markAddrsig(Syms[SymIndex]);
1535 Cur += Size;
1536 }
1537 } else {
1538 // If an object file does not have an address-significance table,
1539 // conservatively mark all of its symbols as address-significant.
1540 for (Symbol *S : Syms)
1541 markAddrsig(S);
1542 }
1543 }
1544}

1546// This function reads a symbol partition specification section. These sections
1547// are used to control which partition a symbol is allocated to. See
1548// https://lld.llvm.org/Partitions.html for more details on partitions.
1549template <typename ELFT>
1550static void readSymbolPartitionSection(InputSectionBase *S) {
1551 // Read the relocation that refers to the partition's entry point symbol.
1552 Symbol *Sym;
1553 if (S->AreRelocsRela)
1554 Sym = &S->getFile<ELFT>()->getRelocTargetSym(S->template relas<ELFT>()[0]);
1555 else
1556 Sym = &S->getFile<ELFT>()->getRelocTargetSym(S->template rels<ELFT>()[0]);
1557 if (!isa<Defined>(Sym) || !Sym->includeInDynsym())
1558 return;
1559
1560 StringRef PartName = reinterpret_cast<const char *>(S->data().data());
1561 for (Partition &Part : Partitions) {
1562 if (Part.Name == PartName) {
1563 Sym->Partition = Part.getNumber();
1564 return;
1565 }
1566 }
1567
1568 // Forbid partitions from being used on incompatible targets, and forbid them
1569 // from being used together with various linker features that assume a single
1570 // set of output sections.
1571 if (Script->HasSectionsCommand)
1572 error(toString(S->File) +
1573 ": partitions cannot be used with the SECTIONS command");
1574 if (Script->hasPhdrsCommands())
1575 error(toString(S->File) +
1576 ": partitions cannot be used with the PHDRS command");
1577 if (!Config->SectionStartMap.empty())
1578 error(toString(S->File) + ": partitions cannot be used with "
1579 "--section-start, -Ttext, -Tdata or -Tbss");
1580 if (Config->EMachine == EM_MIPS)
1581 error(toString(S->File) + ": partitions cannot be used on this target");
1582
1583 // Impose a limit of no more than 254 partitions. This limit comes from the
1584 // sizes of the Partition fields in InputSectionBase and Symbol, as well as
1585 // the amount of space devoted to the partition number in RankFlags.
1586 if (Partitions.size() == 254)
1587 fatal("may not have more than 254 partitions");
1588
1589 Partitions.emplace_back();
1590 Partition &NewPart = Partitions.back();
1591 NewPart.Name = PartName;
1592 Sym->Partition = NewPart.getNumber();
1593}

1622// The --wrap option is a feature to rename symbols so that you can write
1623// wrappers for existing functions. If you pass `-wrap=foo`, all
1624// occurrences of symbol `foo` are resolved to `wrap_foo` (so, you are
1625// expected to write `wrap_foo` function as a wrapper). The original
1626// symbol becomes accessible as `real_foo`, so you can call that from your
1627// wrapper.
1628//
1629// This data structure is instantiated for each -wrap option.
1630struct WrappedSymbol {
1631 Symbol *Sym;
1632 Symbol *Real;
1633 Symbol *Wrap;
1634};
1635
1636// Handles -wrap option.
1637//
1638// This function instantiates wrapper symbols. At this point, they seem
1639// like they are not being used at all, so we explicitly set some flags so
1640// that LTO won't eliminate them.
1641static std::vector<WrappedSymbol> addWrappedSymbols(opt::InputArgList &Args) {
1642 std::vector<WrappedSymbol> V;
1643 DenseSet<StringRef> Seen;
1644
1645 for (auto *Arg : Args.filtered(OPT_wrap)) {
1646 StringRef Name = Arg->getValue();
1647 if (!Seen.insert(Name).second)
1648 continue;
1649
1650 Symbol *Sym = Symtab->find(Name);
1651 if (!Sym)
1652 continue;
1653
1654 Symbol *Real = addUndefined(Saver.save("__real_" + Name));
1655 Symbol *Wrap = addUndefined(Saver.save("__wrap_" + Name));
1656 V.push_back({Sym, Real, Wrap});
1657
1658 // We want to tell LTO not to inline symbols to be overwritten
1659 // because LTO doesn't know the final symbol contents after renaming.
1660 Real->CanInline = false;
1661 Sym->CanInline = false;
1662
1663 // Tell LTO not to eliminate these symbols.
1664 Sym->IsUsedInRegularObj = true;
1665 Wrap->IsUsedInRegularObj = true;
1666 }
1667 return V;
1668}

1670// Do renaming for -wrap by updating pointers to symbols.
1671//
1672// When this function is executed, only InputFiles and symbol table
1673// contain pointers to symbol objects. We visit them to replace pointers,
1674// so that wrapped symbols are swapped as instructed by the command line.
1675static void wrapSymbols(ArrayRef<WrappedSymbol> Wrapped) {
1676 DenseMap<Symbol *, Symbol *> Map;
1677 for (const WrappedSymbol &W : Wrapped) {
1678 Map[W.Sym] = W.Wrap;
1679 Map[W.Real] = W.Sym;
1680 }
1681
1682 // Update pointers in input files.
1683 parallelForEach(ObjectFiles, [&](InputFile *File) {
1684 MutableArrayRef<Symbol *> Syms = File->getMutableSymbols();
1685 for (size_t I = 0, E = Syms.size(); I != E; ++I)
1686 if (Symbol *S = Map.lookup(Syms[I]))
1687 Syms[I] = S;
1688 });
1689
1690 // Update pointers in the symbol table.
1691 for (const WrappedSymbol &W : Wrapped)
1692 Symtab->wrap(W.Sym, W.Real, W.Wrap);
1693}

1695// To enable CET (x86's hardware-assited control flow enforcement), each
1696// source file must be compiled with -fcf-protection. Object files compiled
1697// with the flag contain feature flags indicating that they are compatible
1698// with CET. We enable the feature only when all object files are compatible
1699// with CET.
1700//
1701// This function returns the merged feature flags. If 0, we cannot enable CET.
1702// This is also the case with AARCH64's BTI and PAC which use the similar
1703// GNU_PROPERTY_AARCH64_FEATURE_1_AND mechanism.
1704//
1705// Note that the CET-aware PLT is not implemented yet. We do error
1706// check only.
1707template <class ELFT> static uint32_t getAndFeatures() {
1708 if (Config->EMachine != EM_386 && Config->EMachine != EM_X86_64 &&
1709 Config->EMachine != EM_AARCH64)
1710 return 0;
1711
1712 uint32_t Ret = -1;
1713 for (InputFile *F : ObjectFiles) {
1714 uint32_t Features = cast<ObjFile<ELFT>>(F)->AndFeatures;
1715 if (Config->ForceBTI && !(Features & GNU_PROPERTY_AARCH64_FEATURE_1_BTI)) {
1716 warn(toString(F) + ": --force-bti: file does not have BTI property");
1717 Features |= GNU_PROPERTY_AARCH64_FEATURE_1_BTI;
1718 } else if (!Features && Config->RequireCET)
1719 error(toString(F) + ": --require-cet: file is not compatible with CET");
1720 Ret &= Features;
1721 }
1722
1723 // Force enable pointer authentication Plt, we don't warn in this case as
1724 // this does not require support in the object for correctness.
1725 if (Config->PacPlt)
1726 Ret |= GNU_PROPERTY_AARCH64_FEATURE_1_PAC;
1727
1728 return Ret;
1729}

1743 // If a -hash-style option was not given, set to a default value,
1744 // which varies depending on the target.
1745 if (!Args.hasArg(OPT_hash_style)) {
1746 if (Config->EMachine == EM_MIPS)
1747 Config->SysvHash = true;
1748 else
1749 Config->SysvHash = Config->GnuHash = true;
1750 }
1751
1752 // Default output filename is "a.out" by the Unix tradition.
1753 if (Config->OutputFile.empty())
1754 Config->OutputFile = "a.out";
1755
1756 // Fail early if the output file or map file is not writable. If a user has a
1757 // long link, e.g. due to a large LTO link, they do not wish to run it and
1758 // find that it failed because there was a mistake in their command-line.
1759 if (auto E = tryCreateFile(Config->OutputFile))
1760 error("cannot open output file " + Config->OutputFile + ": " + E.message());
1761 if (auto E = tryCreateFile(Config->MapFile))
1762 error("cannot open map file " + Config->MapFile + ": " + E.message());
1763 if (errorCount())
1764 return;
1765
1766 // Use default entry point name if no name was given via the command
1767 // line nor linker scripts. For some reason, MIPS entry point name is
1768 // different from others.
1769 Config->WarnMissingEntry =
1770 (!Config->Entry.empty() || (!Config->Shared && !Config->Relocatable));
1771 if (Config->Entry.empty() && !Config->Relocatable)
1772 Config->Entry = (Config->EMachine == EM_MIPS) ? "__start" : "_start";
1773
1774 // Handle --trace-symbol.
1775 for (auto *Arg : Args.filtered(OPT_trace_symbol))
1776 Symtab->insert(Arg->getValue())->Traced = true;
1777
1778 // Add all files to the symbol table. This will add almost all
1779 // symbols that we need to the symbol table. This process might
1780 // add files to the link, via autolinking, these files are always
1781 // appended to the Files vector.
1782 for (size_t I = 0; I < Files.size(); ++I)
1783 parseFile(Files[I]);
1784
1785 // Now that we have every file, we can decide if we will need a
1786 // dynamic symbol table.
1787 // We need one if we were asked to export dynamic symbols or if we are
1788 // producing a shared library.
1789 // We also need one if any shared libraries are used and for pie executables
1790 // (probably because the dynamic linker needs it).
1791 Config->HasDynSymTab =
1792 !SharedFiles.empty() || Config->Pic || (Config->ExportDynamic && !Config->Static);
1793
1794 // Some symbols (such as __ehdr_start) are defined lazily only when there
1795 // are undefined symbols for them, so we add these to trigger that logic.
1796 for (StringRef Name : Script->ReferencedSymbols)
1797 addUndefined(Name);
1798
1799 // Handle the `--undefined <sym>` options.
1800 for (StringRef S : Config->Undefined)
1801 handleUndefined(S);
1802
1803 // If an entry symbol is in a static archive, pull out that file now.
1804 handleUndefined(Config->Entry);
1805
1806 // If any of our inputs are bitcode files, the LTO code generator may create
1807 // references to certain library functions that might not be explicit in the
1808 // bitcode file's symbol table. If any of those library functions are defined
1809 // in a bitcode file in an archive member, we need to arrange to use LTO to
1810 // compile those archive members by adding them to the link beforehand.
1811 //
1812 // However, adding all libcall symbols to the link can have undesired
1813 // consequences. For example, the libgcc implementation of
1814 // __sync_val_compare_and_swap_8 on 32-bit ARM pulls in an .init_array entry
1815 // that aborts the program if the Linux kernel does not support 64-bit
1816 // atomics, which would prevent the program from running even if it does not
1817 // use 64-bit atomics.
1818 //
1819 // Therefore, we only add libcall symbols to the link before LTO if we have
1820 // to, i.e. if the symbol's definition is in bitcode. Any other required
1821 // libcall symbols will be added to the link after LTO when we add the LTO
1822 // object file to the link.
1823 if (!BitcodeFiles.empty())
1824 for (const char *S : LibcallRoutineNames)
1825 handleLibcall(S);
1826
1827 // Return if there were name resolution errors.
1828 if (errorCount())
1829 return;
1830
1831 // Now when we read all script files, we want to finalize order of linker
1832 // script commands, which can be not yet final because of INSERT commands.
1833 Script->processInsertCommands();
1834
1835 // We want to declare linker script's symbols early,
1836 // so that we can version them.
1837 // They also might be exported if referenced by DSOs.
1838 Script->declareSymbols();
1839
1840 // Handle the -exclude-libs option.
1841 if (Args.hasArg(OPT_exclude_libs))
1842 excludeLibs(Args);
1843
1844 // Create ElfHeader early. We need a dummy section in
1845 // addReservedSymbols to mark the created symbols as not absolute.
1846 Out::ElfHeader = make<OutputSection>("", 0, SHF_ALLOC);
1847 Out::ElfHeader->Size = sizeof(typename ELFT::Ehdr);
1848
1849 // Create wrapped symbols for -wrap option.
1850 std::vector<WrappedSymbol> Wrapped = addWrappedSymbols(Args);
1851
1852 // We need to create some reserved symbols such as _end. Create them.
1853 if (!Config->Relocatable)
1854 addReservedSymbols();
1855
1856 // Apply version scripts.
1857 //
1858 // For a relocatable output, version scripts don't make sense, and
1859 // parsing a symbol version string (e.g. dropping "@ver1" from a symbol
1860 // name "foo@ver1") rather do harm, so we don't call this if -r is given.
1861 if (!Config->Relocatable)
1862 Symtab->scanVersionScript();
1863
1864 // Do link-time optimization if given files are LLVM bitcode files.
1865 // This compiles bitcode files into real object files.
1866 //
1867 // With this the symbol table should be complete. After this, no new names
1868 // except a few linker-synthesized ones will be added to the symbol table.
1869 compileBitcodeFiles<ELFT>();
1870 if (errorCount())
1871 return;
1872
1873 // If -thinlto-index-only is given, we should create only "index
1874 // files" and not object files. Index file creation is already done
1875 // in addCombinedLTOObject, so we are done if that's the case.
1876 if (Config->ThinLTOIndexOnly)
1877 return;
1878
1879 // Likewise, --plugin-opt=emit-llvm is an option to make LTO create
1880 // an output file in bitcode and exit, so that you can just get a
1881 // combined bitcode file.
1882 if (Config->EmitLLVM)
1883 return;
1884
1885 // Apply symbol renames for -wrap.
1886 if (!Wrapped.empty())
1887 wrapSymbols(Wrapped);
1888
1889 // Now that we have a complete list of input files.
1890 // Beyond this point, no new files are added.
1891 // Aggregate all input sections into one place.
1892 for (InputFile *F : ObjectFiles)
1893 for (InputSectionBase *S : F->getSections())
1894 if (S && S != &InputSection::Discarded)
1895 InputSections.push_back(S);
1896 for (BinaryFile *F : BinaryFiles)
1897 for (InputSectionBase *S : F->getSections())
1898 InputSections.push_back(cast<InputSection>(S));
1899
1900 llvm::erase_if(InputSections, [](InputSectionBase *S) {
1901 if (S->Type == SHT_LLVM_SYMPART) {
1902 readSymbolPartitionSection<ELFT>(S);
1903 return true;
1904 }
1905
1906 // We do not want to emit debug sections if --strip-all
1907 // or -strip-debug are given.
1908 return Config->Strip != StripPolicy::None &&
1909 (S->Name.startswith(".debug") || S->Name.startswith(".zdebug"));
1910 });
1911
1912 // Now that the number of partitions is fixed, save a pointer to the main
1913 // partition.
1914 Main = &Partitions[0];
1915
1916 // Read .note.gnu.property sections from input object files which
1917 // contain a hint to tweak linker's and loader's behaviors.
1918 Config->AndFeatures = getAndFeatures<ELFT>();
1919
1920 // The Target instance handles target-specific stuff, such as applying
1921 // relocations or writing a PLT section. It also contains target-dependent
1922 // values such as a default image base address.
1923 Target = getTarget();
1924
1925 Config->EFlags = Target->calcEFlags();
1926 // MaxPageSize (sometimes called abi page size) is the maximum page size that
1927 // the output can be run on. For example if the OS can use 4k or 64k page
1928 // sizes then MaxPageSize must be 64 for the output to be useable on both.
1929 // All important alignment decisions must use this value.
1930 Config->MaxPageSize = getMaxPageSize(Args);
1931 // CommonPageSize is the most common page size that the output will be run on.
1932 // For example if an OS can use 4k or 64k page sizes and 4k is more common
1933 // than 64k then CommonPageSize is set to 4k. CommonPageSize can be used for
1934 // optimizations such as DATA_SEGMENT_ALIGN in linker scripts. LLD's use of it
1935 // is limited to writing trap instructions on the last executable segment.
1936 Config->CommonPageSize = getCommonPageSize(Args);
1937
1938 Config->ImageBase = getImageBase(Args);
1939 Config->CapabilitySize = Target->getCapabilitySize();
1940
1941 // CapabilitySize must be set if we are targeting the purecap ABI
1942 if (Config->isCheriABI()) {
1943 if (errorCount())
1944 return;
1945 assert(Config->CapabilitySize > 0);
1946 }
1947
1948 if (Config->EMachine == EM_ARM) {
1949 // FIXME: These warnings can be removed when lld only uses these features
1950 // when the input objects have been compiled with an architecture that
1951 // supports them.
1952 if (Config->ARMHasBlx == false)
1953 warn("lld uses blx instruction, no object with architecture supporting "
1954 "feature detected");
1955 }
1956
1957 // This adds a .comment section containing a version string. We have to add it
1958 // before mergeSections because the .comment section is a mergeable section.
1959 if (!Config->Relocatable)
1960 InputSections.push_back(createCommentSection());
1961
1962 // Replace common symbols with regular symbols.
1963 replaceCommonSymbols();
1964
1965 // Do size optimizations: garbage collection, merging of SHF_MERGE sections
1966 // and identical code folding.
1967 splitSections<ELFT>();
1968 markLive<ELFT>();
1969 demoteSharedSymbols();
1970 mergeSections();
1971 if (Config->ICF != ICFLevel::None) {
1972 findKeepUniqueSections<ELFT>(Args);
1973 doIcf<ELFT>();
1974 }
1975
1976 // Read the callgraph now that we know what was gced or icfed
1977 if (Config->CallGraphProfileSort) {
1978 if (auto *Arg = Args.getLastArg(OPT_call_graph_ordering_file))
1979 if (Optional<MemoryBufferRef> Buffer = readFile(Arg->getValue()))
1980 readCallGraph(*Buffer);
1981 readCallGraphsFromObjectFiles<ELFT>();
1982 }
1983
1984 // Write the result to the file.
1985 writeResult<ELFT>();
1986}
