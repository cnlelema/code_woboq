59InputSectionBase::InputSectionBase(InputFile *File, uint64_t Flags,
60 uint32_t Type, uint64_t Entsize,
61 uint32_t Link, uint32_t Info,
62 uint32_t Alignment, ArrayRef<uint8_t> Data,
63 StringRef Name, Kind SectionKind)
64 : SectionBase(SectionKind, Name, Flags, Entsize, Alignment, Type, Info,
65 Link),
66 File(File), RawData(Data) {
67 // In order to reduce memory allocation, we assume that mergeable
68 // sections are smaller than 4 GiB, which is not an unreasonable
69 // assumption as of 2017.
70 if (SectionKind == SectionBase::Merge && RawData.size() > UINT32_MAX)
71 error(toString(this) + ": section too large");
72
73 NumRelocations = 0;
74 AreRelocsRela = false;
75
76 // The ELF spec states that a value of 0 means the section has
77 // no alignment constraits.
78 uint32_t V = std::max<uint32_t>(Alignment, 1);
79 if (!isPowerOf2_64(V))
80 fatal(toString(this) + ": sh_addralign is not a power of 2");
81 this->Alignment = V;
82
83 // In ELF, each section can be compressed by zlib, and if compressed,
84 // section name may be mangled by appending "z" (e.g. ".zdebug_info").
85 // If that's the case, demangle section name so that we can handle a
86 // section as if it weren't compressed.
87 if ((Flags & SHF_COMPRESSED) || Name.startswith(".zdebug")) {
88 if (!zlib::isAvailable())
89 error(toString(File) + ": contains a compressed section, " +
90 "but zlib is not available");
91 parseCompressedHeader();
92 }
93}

95// Drop SHF_GROUP bit unless we are producing a re-linkable object file.
96// SHF_GROUP is a marker that a section belongs to some comdat group.
97// That flag doesn't make sense in an executable.
98static uint64_t getFlags(uint64_t Flags) {
99 Flags &= ~(uint64_t)SHF_INFO_LINK;
100 if (!Config->Relocatable)
101 Flags &= ~(uint64_t)SHF_GROUP;
102 return Flags;
103}

105// GNU assembler 2.24 and LLVM 4.0.0's MC (the newest release as of
106// March 2017) fail to infer section types for sections starting with
107// ".init_array." or ".fini_array.". They set SHT_PROGBITS instead of
108// SHF_INIT_ARRAY. As a result, the following assembler directive
109// creates ".init_array.100" with SHT_PROGBITS, for example.
110//
111// .section .init_array.100, "aw"
112//
113// This function forces SHT_{INIT,FINI}_ARRAY so that we can handle
114// incorrect inputs as if they were correct from the beginning.
115static uint64_t getType(uint64_t Type, StringRef Name) {
116 if (Type == SHT_PROGBITS && Name.startswith(".init_array."))
117 return SHT_INIT_ARRAY;
118 if (Type == SHT_PROGBITS && Name.startswith(".fini_array."))
119 return SHT_FINI_ARRAY;
120 return Type;
121}

146void InputSectionBase::uncompress() const {
147 size_t Size = UncompressedSize;
148 char *UncompressedBuf;
149 {
150 static std::mutex Mu;
151 std::lock_guard<std::mutex> Lock(Mu);
152 UncompressedBuf = BAlloc.Allocate<char>(Size);
153 }
154
155 if (Error E = zlib::uncompress(toStringRef(RawData), UncompressedBuf, Size))
156 fatal(toString(this) +
157 ": uncompress failed: " + llvm::toString(std::move(E)));
158 RawData = makeArrayRef((uint8_t *)UncompressedBuf, Size);
159 UncompressedSize = -1;
160}

168uint64_t SectionBase::getOffset(uint64_t Offset) const {
169 switch (kind()) {
170 case Output: {
171 auto *OS = cast<OutputSection>(this);
172 // For output sections we treat offset -1 as the end of the section.
173 return Offset == uint64_t(-1) ? OS->Size : Offset;
174 }
175 case Regular:
176 case Synthetic:
177 return cast<InputSection>(this)->getOffset(Offset);
178 case EHFrame:
179 // The file crtbeginT.o has relocations pointing to the start of an empty
180 // .eh_frame that is known to be the first in the link. It does that to
181 // identify the start of the output .eh_frame.
182 return Offset;
183 case Merge:
184 const MergeInputSection *MS = cast<MergeInputSection>(this);
185 if (InputSection *IS = MS->getParent())
186 return IS->getOffset(MS->getParentOffset(Offset));
187 return MS->getParentOffset(Offset);
188 }
189 llvm_unreachable("invalid section kind");
190}

192uint64_t SectionBase::getVA(uint64_t Offset) const {
193 const OutputSection *Out = getOutputSection();
194 return (Out ? Out->Addr : 0) + getOffset(Offset);

197OutputSection *SectionBase::getOutputSection() {
198 InputSection *Sec;
199 if (auto *IS = dyn_cast<InputSection>(this))
200 Sec = IS;
201 else if (auto *MS = dyn_cast<MergeInputSection>(this))
202 Sec = MS->getParent();
203 else if (auto *EH = dyn_cast<EhInputSection>(this))
204 Sec = EH->getParent();
205 else
206 return cast<OutputSection>(this);
207 return Sec ? Sec->getParent() : nullptr;
208}

210// When a section is compressed, `RawData` consists with a header followed
211// by zlib-compressed data. This function parses a header to initialize
212// `UncompressedSize` member and remove the header from `RawData`.
213void InputSectionBase::parseCompressedHeader() {
214 using Chdr64 = typename ELF64LE::Chdr;
215 using Chdr32 = typename ELF32LE::Chdr;
216
217 // Old-style header
218 if (Name.startswith(".zdebug")) {
219 if (!toStringRef(RawData).startswith("ZLIB")) {
220 error(toString(this) + ": corrupted compressed section header");
221 return;
222 }
223 RawData = RawData.slice(4);
224
225 if (RawData.size() < 8) {
226 error(toString(this) + ": corrupted compressed section header");
227 return;
228 }
229
230 UncompressedSize = read64be(RawData.data());
231 RawData = RawData.slice(8);
232
233 // Restore the original section name.
234 // (e.g. ".zdebug_info" -> ".debug_info")
235 Name = Saver.save("." + Name.substr(2));
236 return;
237 }
238
239 assert(Flags & SHF_COMPRESSED);
240 Flags &= ~(uint64_t)SHF_COMPRESSED;
241
242 // New-style 64-bit header
243 if (Config->Is64) {
244 if (RawData.size() < sizeof(Chdr64)) {
245 error(toString(this) + ": corrupted compressed section");
246 return;
247 }
248
249 auto *Hdr = reinterpret_cast<const Chdr64 *>(RawData.data());
250 if (Hdr->ch_type != ELFCOMPRESS_ZLIB) {
251 error(toString(this) + ": unsupported compression type");
252 return;
253 }
254
255 UncompressedSize = Hdr->ch_size;
256 Alignment = std::max<uint32_t>(Hdr->ch_addralign, 1);
257 RawData = RawData.slice(sizeof(*Hdr));
258 return;
259 }
260
261 // New-style 32-bit header
262 if (RawData.size() < sizeof(Chdr32)) {
263 error(toString(this) + ": corrupted compressed section");
264 return;
265 }
266
267 auto *Hdr = reinterpret_cast<const Chdr32 *>(RawData.data());
268 if (Hdr->ch_type != ELFCOMPRESS_ZLIB) {
269 error(toString(this) + ": unsupported compression type");
270 return;
271 }
272
273 UncompressedSize = Hdr->ch_size;
274 Alignment = std::max<uint32_t>(Hdr->ch_addralign, 1);
275 RawData = RawData.slice(sizeof(*Hdr));
276}

305// Returns a source location string. Used to construct an error message.
306template <class ELFT>
307std::string InputSectionBase::getLocation(uint64_t Offset) const {
308 std::string SecAndOffset = (Name + "+0x" + utohexstr(Offset)).str();
309
310 // We don't have file for synthetic sections.
311 if (getFile<ELFT>() == nullptr)
312 return (Config->OutputFile + ":(" + SecAndOffset + ")")
313 .str();
314
315 // First check if we can get desired values from debugging information.
316 if (Optional<DILineInfo> Info = getFile<ELFT>()->getDILineInfo(this, Offset))
317 return Info->FileName + ":" + std::to_string(Info->Line) + ":(" +
318 SecAndOffset + ")";
319
320 // File->SourceFile contains STT_FILE symbol that contains a
321 // source file name. If it's missing, we use an object file name.
322 std::string SrcFile = getFile<ELFT>()->SourceFile;
323 if (SrcFile.empty())
324 SrcFile = toString(File);
325
326 if (Defined *D = getEnclosingFunction<ELFT>(Offset))
327 return SrcFile + ":(function " + toString(*D) + ": " + SecAndOffset + ")";
328 else if (Defined *D = getEnclosingObject<ELFT>(Offset))
329 return SrcFile + ":(object " + toString(*D) + ": " + SecAndOffset + ")";
330
331 // If there's no symbol, print out the offset in the section.
332 return (SrcFile + ":(" + SecAndOffset + ")");
333}

348// Returns a filename string along with an optional section name. This
349// function is intended to be used for constructing an error
350// message. The returned message looks like this:
351//
352// path/to/foo.o:(function bar)
353//
354// or
355//
356// path/to/foo.o:(function bar) in archive path/to/bar.a
357std::string InputSectionBase::getObjMsg(uint64_t Off) const {
358 // Synthetic sections don't have input files.
359 if (!File)
360 return ("<internal>:(" + Name + "+0x" + utohexstr(Off) + ")").str();
361 std::string Filename = File->getName();
362
363 std::string Archive;
364 if (!File->ArchiveName.empty())
365 Archive = " in archive " + File->ArchiveName;
366
367 // Find a symbol that encloses a given location.
368 for (Symbol *B : File->getSymbols())
369 if (auto *D = dyn_cast<Defined>(B))
370 if (D->Section == this && D->Value <= Off && Off < D->Value + D->Size)
371 return Filename + ":(" + toString(*D) + ")" + Archive;
372
373 // If there's no symbol, print out the offset in the section.
374 return (Filename + ":(" + Name + "+0x" + utohexstr(Off) + ")" + Archive)
375 .str();
376}

401// Copy SHT_GROUP section contents. Used only for the -r option.
402template <class ELFT> void InputSection::copyShtGroup(uint8_t *Buf) {
403 // ELFT::Word is the 32-bit integral type in the target endianness.
404 using u32 = typename ELFT::Word;
405 ArrayRef<u32> From = getDataAs<u32>();
406 auto *To = reinterpret_cast<u32 *>(Buf);
407
408 // The first entry is not a section number but a flag.
409 *To++ = From[0];
410
411 // Adjust section numbers because section numbers in an input object
412 // files are different in the output.
413 ArrayRef<InputSectionBase *> Sections = File->getSections();
414 for (uint32_t Idx : From.slice(1))
415 *To++ = Sections[Idx]->getOutputSection()->SectionIndex;
416}

425// This is used for -r and --emit-relocs. We can't use memcpy to copy
426// relocations because we need to update symbol table offset and section index
427// for each relocation. So we copy relocations one by one.
428template <class ELFT, class RelTy>
429void InputSection::copyRelocations(uint8_t *Buf, ArrayRef<RelTy> Rels) {
430 InputSectionBase *Sec = getRelocatedSection();
431
432 for (const RelTy &Rel : Rels) {
433 RelType Type = Rel.getType(Config->IsMips64EL);
434 const ObjFile<ELFT> *File = getFile<ELFT>();
435 Symbol &Sym = File->getRelocTargetSym(Rel);
436
437 auto *P = reinterpret_cast<typename ELFT::Rela *>(Buf);
438 Buf += sizeof(RelTy);
439
440 if (RelTy::IsRela)
441 P->r_addend = getAddend<ELFT>(Rel);
442
443 // Output section VA is zero for -r, so r_offset is an offset within the
444 // section, but for --emit-relocs it is an virtual address.
445 P->r_offset = Sec->getVA(Rel.r_offset);
446 P->setSymbolAndType(In.SymTab->getSymbolIndex(&Sym), Type,
447 Config->IsMips64EL);
448
449 if (Sym.Type == STT_SECTION) {
450 // We combine multiple section symbols into only one per
451 // section. This means we have to update the addend. That is
452 // trivial for Elf_Rela, but for Elf_Rel we have to write to the
453 // section data. We do that by adding to the Relocation vector.
454
455 // .eh_frame is horribly special and can reference discarded sections. To
456 // avoid having to parse and recreate .eh_frame, we just replace any
457 // relocation in it pointing to discarded sections with R_*_NONE, which
458 // hopefully creates a frame that is ignored at runtime. Also, don't warn
459 // on .gcc_except_table and debug sections.
460 //
461 // See the comment in maybeReportUndefined for PPC64 .toc .
462 auto *D = dyn_cast<Defined>(&Sym);
463 if (!D) {
464 if (!Sec->Name.startswith(".debug") &&
465 !Sec->Name.startswith(".zdebug") && Sec->Name != ".eh_frame" &&
466 Sec->Name != ".gcc_except_table" && Sec->Name != ".toc") {
467 uint32_t SecIdx = cast<Undefined>(Sym).DiscardedSecIdx;
468 Elf_Shdr_Impl<ELFT> Sec =
469 CHECK(File->getObj().sections(), File)[SecIdx];
470 warn("relocation refers to a discarded section: " +
471 CHECK(File->getObj().getSectionName(&Sec), File) +
472 "\n>>> referenced by " + getObjMsg(P->r_offset));
473 }
474 P->setSymbolAndType(0, 0, false);
475 continue;
476 }
477 SectionBase *Section = D->Section->Repl;
478 if (!Section->isLive()) {
479 P->setSymbolAndType(0, 0, false);
480 continue;
481 }
482
483 int64_t Addend = getAddend<ELFT>(Rel);
484 const uint8_t *BufLoc = Sec->data().begin() + Rel.r_offset;
485 if (!RelTy::IsRela)
486 Addend = Target->getImplicitAddend(BufLoc, Type);
487
488 if (Config->EMachine == EM_MIPS && Config->Relocatable &&
489 Target->getRelExpr(Type, Sym, BufLoc) == R_MIPS_GOTREL) {
490 // Some MIPS relocations depend on "gp" value. By default,
491 // this value has 0x7ff0 offset from a .got section. But
492 // relocatable files produced by a complier or a linker
493 // might redefine this default value and we must use it
494 // for a calculation of the relocation result. When we
495 // generate EXE or DSO it's trivial. Generating a relocatable
496 // output is more difficult case because the linker does
497 // not calculate relocations in this mode and loses
498 // individual "gp" values used by each input object file.
499 // As a workaround we add the "gp" value to the relocation
500 // addend and save it back to the file.
501 Addend += Sec->getFile<ELFT>()->MipsGp0;
502 }
503
504 if (RelTy::IsRela)
505 P->r_addend = Sym.getVA(Addend) - Section->getOutputSection()->Addr;
506 else if (Config->Relocatable && Type != Target->NoneRel)
507 Sec->Relocations.push_back({R_ABS, Type, Rel.r_offset, Addend, &Sym});
508 }
509 }

512// The ARM and AArch64 ABI handle pc-relative relocations to undefined weak
513// references specially. The general rule is that the value of the symbol in
514// this context is the address of the place P. A further special case is that
515// branch relocations to an undefined weak reference resolve to the next
516// instruction.
517static uint32_t getARMUndefinedRelativeWeakVA(RelType Type, uint32_t A,
518 uint32_t P) {
519 switch (Type) {
520 // Unresolved branch relocations to weak references resolve to next
521 // instruction, this will be either 2 or 4 bytes on from P.
522 case R_ARM_THM_JUMP11:
523 return P + 2 + A;
524 case R_ARM_CALL:
525 case R_ARM_JUMP24:
526 case R_ARM_PC24:
527 case R_ARM_PLT32:
528 case R_ARM_PREL31:
529 case R_ARM_THM_JUMP19:
530 case R_ARM_THM_JUMP24:
531 return P + 4 + A;
532 case R_ARM_THM_CALL:
533 // We don't want an interworking BLX to ARM
534 return P + 5 + A;
535 // Unresolved non branch pc-relative relocations
536 // R_ARM_TARGET2 which can be resolved relatively is not present as it never
537 // targets a weak-reference.
538 case R_ARM_MOVW_PREL_NC:
539 case R_ARM_MOVT_PREL:
540 case R_ARM_REL32:
541 case R_ARM_THM_MOVW_PREL_NC:
542 case R_ARM_THM_MOVT_PREL:
543 return P + A;
544 }
545 llvm_unreachable("ARM pc-relative relocation expected\n");
546}

548// The comment above getARMUndefinedRelativeWeakVA applies to this function.
549static uint64_t getAArch64UndefinedRelativeWeakVA(uint64_t Type, uint64_t A,
550 uint64_t P) {
551 switch (Type) {
552 // Unresolved branch relocations to weak references resolve to next
553 // instruction, this is 4 bytes on from P.
554 case R_AARCH64_CALL26:
555 case R_AARCH64_CONDBR19:
556 case R_AARCH64_JUMP26:
557 case R_AARCH64_TSTBR14:
558 return P + 4 + A;
559 // Unresolved non branch pc-relative relocations
560 case R_AARCH64_PREL16:
561 case R_AARCH64_PREL32:
562 case R_AARCH64_PREL64:
563 case R_AARCH64_ADR_PREL_LO21:
564 case R_AARCH64_LD_PREL_LO19:
565 return P + A;
566 }
567 llvm_unreachable("AArch64 pc-relative relocation expected\n");
568}

570// ARM SBREL relocations are of the form S + A - B where B is the static base
571// The ARM ABI defines base to be "addressing origin of the output segment
572// defining the symbol S". We defined the "addressing origin"/static base to be
573// the base of the PT_LOAD segment containing the Sym.
574// The procedure call standard only defines a Read Write Position Independent
575// RWPI variant so in practice we should expect the static base to be the base
576// of the RW segment.
577static uint64_t getARMStaticBase(const Symbol &Sym) {
578 OutputSection *OS = Sym.getOutputSection();
579 if (!OS || !OS->PtLoad || !OS->PtLoad->FirstSec)
580 fatal("SBREL relocation to " + Sym.getName() + " without static base");
581 return OS->PtLoad->FirstSec->Addr;
582}

584// For R_RISCV_PC_INDIRECT (R_RISCV_PCREL_LO12_{I,S}), the symbol actually
585// points the corresponding R_RISCV_PCREL_HI20 relocation, and the target VA
586// is calculated using PCREL_HI20's symbol.
587//
588// This function returns the R_RISCV_PCREL_HI20 relocation from
589// R_RISCV_PCREL_LO12's symbol and addend.
590static Relocation *getRISCVPCRelHi20(const Symbol *Sym, uint64_t Addend) {
591 const Defined *D = cast<Defined>(Sym);
592 InputSection *IS = cast<InputSection>(D->Section);
593
594 if (Addend != 0)
595 warn("Non-zero addend in R_RISCV_PCREL_LO12 relocation to " +
596 IS->getObjMsg(D->Value) + " is ignored");
597
598 // Relocations are sorted by offset, so we can use std::equal_range to do
599 // binary search.
600 Relocation R;
601 R.Offset = D->Value;
602 auto Range =
603 std::equal_range(IS->Relocations.begin(), IS->Relocations.end(), R,
604 [](const Relocation &LHS, const Relocation &RHS) {
605 return LHS.Offset < RHS.Offset;
606 });
607
608 for (auto It = Range.first; It != Range.second; ++It)
609 if (It->Expr == R_PC)
610 return &*It;
611
612 error("R_RISCV_PCREL_LO12 relocation points to " + IS->getObjMsg(D->Value) +
613 " without an associated R_RISCV_PCREL_HI20 relocation");
614 return nullptr;
615}

617// A TLS symbol's virtual address is relative to the TLS segment. Add a
618// target-specific adjustment to produce a thread-pointer-relative offset.
619static int64_t getTlsTpOffset(const Symbol &S) {
620 // On targets that support TLSDESC, _TLS_MODULE_BASE_@tpoff = 0.
621 if (&S == ElfSym::TlsModuleBase)
622 return 0;
623
624 switch (Config->EMachine) {
625 case EM_ARM:
626 case EM_AARCH64:
627 // Variant 1. The thread pointer points to a TCB with a fixed 2-word size,
628 // followed by a variable amount of alignment padding, followed by the TLS
629 // segment.
630 return S.getVA(0) + alignTo(Config->Wordsize * 2, Out::TlsPhdr->p_align);
631 case EM_386:
632 case EM_X86_64:
633 // Variant 2. The TLS segment is located just before the thread pointer.
634 return S.getVA(0) - alignTo(Out::TlsPhdr->p_memsz, Out::TlsPhdr->p_align);
635 case EM_PPC:
636 case EM_PPC64:
637 // The thread pointer points to a fixed offset from the start of the
638 // executable's TLS segment. An offset of 0x7000 allows a signed 16-bit
639 // offset to reach 0x1000 of TCB/thread-library data and 0xf000 of the
640 // program's TLS segment.
641 return S.getVA(0) - 0x7000;
642 default:
643 llvm_unreachable("unhandled Config->EMachine");
644 }
645}

647static uint64_t getRelocTargetVA(const InputFile *File, RelType Type, int64_t A,
648 uint64_t P, const Symbol &Sym, RelExpr Expr,
649 InputSectionBase *IS, uint64_t Offset) {
650 switch (Expr) {
651 case R_ABS:
652 case R_DTPREL:
653 case R_RELAX_TLS_LD_TO_LE_ABS:
654 case R_RELAX_GOT_PC_NOPIC:
655 case R_RISCV_ADD:
656 return Sym.getVA(A);
657 case R_ADDEND:
658 return A;
659 case R_ARM_SBREL:
660 return Sym.getVA(A) - getARMStaticBase(Sym);
661 case R_GOT:
662 case R_RELAX_TLS_GD_TO_IE_ABS:
663 return Sym.getGotVA() + A;
664 case R_GOTONLY_PC:
665 return In.Got->getVA() + A - P;
666 case R_GOTPLTONLY_PC:
667 return In.GotPlt->getVA() + A - P;
668 case R_GOTREL:
669 case R_PPC64_RELAX_TOC:
670 return Sym.getVA(A) - In.Got->getVA();
671 case R_GOTPLTREL:
672 return Sym.getVA(A) - In.GotPlt->getVA();
673 case R_GOTPLT:
674 case R_RELAX_TLS_GD_TO_IE_GOTPLT:
675 return Sym.getGotVA() + A - In.GotPlt->getVA();
676 case R_TLSLD_GOT_OFF:
677 case R_GOT_OFF:
678 case R_RELAX_TLS_GD_TO_IE_GOT_OFF:
679 return Sym.getGotOffset() + A;
680 case R_AARCH64_GOT_PAGE_PC:
681 case R_AARCH64_RELAX_TLS_GD_TO_IE_PAGE_PC:
682 return getAArch64Page(Sym.getGotVA() + A) - getAArch64Page(P);
683 case R_GOT_PC:
684 case R_RELAX_TLS_GD_TO_IE:
685 return Sym.getGotVA() + A - P;
686 case R_HEXAGON_GOT:
687 return Sym.getGotVA() - In.GotPlt->getVA();
688 case R_MIPS_GOTREL:
689 return Sym.getVA(A) - In.MipsGot->getGp(File);
690 case R_MIPS_GOT_GP:
691 return In.MipsGot->getGp(File) + A;
692 case R_MIPS_GOT_GP_PC: {
693 // R_MIPS_LO16 expression has R_MIPS_GOT_GP_PC type iif the target
694 // is _gp_disp symbol. In that case we should use the following
695 // formula for calculation "AHL + GP - P + 4". For details see p. 4-19 at
696 // ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
697 // microMIPS variants of these relocations use slightly different
698 // expressions: AHL + GP - P + 3 for %lo() and AHL + GP - P - 1 for %hi()
699 // to correctly handle less-sugnificant bit of the microMIPS symbol.
700 uint64_t V = In.MipsGot->getGp(File) + A - P;
701 if (Type == R_MIPS_LO16 || Type == R_MICROMIPS_LO16)
702 V += 4;
703 if (Type == R_MICROMIPS_LO16 || Type == R_MICROMIPS_HI16)
704 V -= 1;
705 return V;
706 }
707 case R_MIPS_GOT_LOCAL_PAGE:
708 // If relocation against MIPS local symbol requires GOT entry, this entry
709 // should be initialized by 'page address'. This address is high 16-bits
710 // of sum the symbol's value and the addend.
711 return In.MipsGot->getVA() + In.MipsGot->getPageEntryOffset(File, Sym, A) -
712 In.MipsGot->getGp(File);
713 case R_MIPS_GOT_OFF:
714 case R_MIPS_GOT_OFF32:
715 // In case of MIPS if a GOT relocation has non-zero addend this addend
716 // should be applied to the GOT entry content not to the GOT entry offset.
717 // That is why we use separate expression type.
718 return In.MipsGot->getVA() + In.MipsGot->getSymEntryOffset(File, Sym, A) -
719 In.MipsGot->getGp(File);
720 case R_MIPS_TLSGD:
721 return In.MipsGot->getVA() + In.MipsGot->getGlobalDynOffset(File, Sym) -
722 In.MipsGot->getGp(File);
723 case R_MIPS_TLSLD:
724 return In.MipsGot->getVA() + In.MipsGot->getTlsIndexOffset(File) -
725 In.MipsGot->getGp(File);
726 case R_AARCH64_PAGE_PC: {
727 uint64_t Val = Sym.isUndefWeak() ? P + A : Sym.getVA(A);
728 return getAArch64Page(Val) - getAArch64Page(P);
729 }
730 case R_RISCV_PC_INDIRECT: {
731 if (const Relocation *HiRel = getRISCVPCRelHi20(&Sym, A))
732 return getRelocTargetVA(File, HiRel->Type, HiRel->Addend, Sym.getVA(),
733 *HiRel->Sym, HiRel->Expr, IS, Offset);
734 return 0;
735 }
736 case R_PC: {
737 uint64_t Dest;
738 if (Sym.isUndefWeak()) {
739 // On ARM and AArch64 a branch to an undefined weak resolves to the
740 // next instruction, otherwise the place.
741 if (Config->EMachine == EM_ARM)
742 Dest = getARMUndefinedRelativeWeakVA(Type, A, P);
743 else if (Config->EMachine == EM_AARCH64)
744 Dest = getAArch64UndefinedRelativeWeakVA(Type, A, P);
745 else if (Config->EMachine == EM_PPC)
746 Dest = P;
747 else
748 Dest = Sym.getVA(A);
749 } else {
750 Dest = Sym.getVA(A);
751 }
752 return Dest - P;
753 }
754 case R_PLT:
755 return Sym.getPltVA() + A;
756 case R_PLT_PC:
757 case R_PPC64_CALL_PLT:
758 return Sym.getPltVA() + A - P;
759 case R_PPC32_PLTREL:
760 // R_PPC_PLTREL24 uses the addend (usually 0 or 0x8000) to indicate r30
761 // stores _GLOBAL_OFFSET_TABLE_ or .got2+0x8000. The addend is ignored for
762 // target VA compuation.
763 return Sym.getPltVA() - P;
764 case R_PPC64_CALL: {
765 uint64_t SymVA = Sym.getVA(A);
766 // If we have an undefined weak symbol, we might get here with a symbol
767 // address of zero. That could overflow, but the code must be unreachable,
768 // so don't bother doing anything at all.
769 if (!SymVA)
770 return 0;
771
772 // PPC64 V2 ABI describes two entry points to a function. The global entry
773 // point is used for calls where the caller and callee (may) have different
774 // TOC base pointers and r2 needs to be modified to hold the TOC base for
775 // the callee. For local calls the caller and callee share the same
776 // TOC base and so the TOC pointer initialization code should be skipped by
777 // branching to the local entry point.
778 return SymVA - P + getPPC64GlobalEntryToLocalEntryOffset(Sym.StOther);
779 }
780 case R_PPC64_TOCBASE:
781 return getPPC64TocBase() + A;
782 case R_RELAX_GOT_PC:
783 return Sym.getVA(A) - P;
784 case R_RELAX_TLS_GD_TO_LE:
785 case R_RELAX_TLS_IE_TO_LE:
786 case R_RELAX_TLS_LD_TO_LE:
787 case R_TLS:
788 // It is not very clear what to return if the symbol is undefined. With
789 // --noinhibit-exec, even a non-weak undefined reference may reach here.
790 // Just return A, which matches R_ABS, and the behavior of some dynamic
791 // loaders.
792 if (Sym.isUndefined())
793 return A;
794 return getTlsTpOffset(Sym) + A;
795 case R_RELAX_TLS_GD_TO_LE_NEG:
796 case R_NEG_TLS:
797 if (Sym.isUndefined())
798 return A;
799 return -getTlsTpOffset(Sym) + A;
800 case R_SIZE:
801 return Sym.getSize() + A;
802 case R_TLSDESC:
803 return In.Got->getGlobalDynAddr(Sym) + A;
804 case R_TLSDESC_PC:
805 return In.Got->getGlobalDynAddr(Sym) + A - P;
806 case R_AARCH64_TLSDESC_PAGE:
807 return getAArch64Page(In.Got->getGlobalDynAddr(Sym) + A) -
808 getAArch64Page(P);
809 case R_TLSGD_GOT:
810 return In.Got->getGlobalDynOffset(Sym) + A;
811 case R_TLSGD_GOTPLT:
812 return In.Got->getVA() + In.Got->getGlobalDynOffset(Sym) + A - In.GotPlt->getVA();
813 case R_TLSGD_PC:
814 return In.Got->getGlobalDynAddr(Sym) + A - P;
815 case R_TLSLD_GOTPLT:
816 return In.Got->getVA() + In.Got->getTlsIndexOff() + A - In.GotPlt->getVA();
817 case R_TLSLD_GOT:
818 return In.Got->getTlsIndexOff() + A;
819 case R_TLSLD_PC:
820 return In.Got->getTlsIndexVA() + A - P;
821 case R_CHERI_CAPABILITY:
822 llvm_unreachable("R_CHERI_CAPABILITY should not be handled here!");
823 case R_CHERI_CAPABILITY_TABLE_INDEX:
824 case R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE:
825 case R_CHERI_CAPABILITY_TABLE_INDEX_CALL:
826 case R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE:
827 assert(A == 0 && "capability table index relocs should not have addends");
828 return Config->CapabilitySize * In.CheriCapTable->getIndex(Sym, IS, Offset);
829 case R_CHERI_CAPABILITY_TABLE_REL:
830 if (!ElfSym::CheriCapabilityTable) {
831 error("cannot compute difference between non-existent "
832 "CheriCapabilityTable and symbol " + toString(Sym));
833 return Sym.getVA(A);
834 }
835 return Sym.getVA(A) - ElfSym::CheriCapabilityTable->getVA();
836 case R_MIPS_CHERI_CAPTAB_TLSGD:
837 assert(A == 0 && "capability table index relocs should not have addends");
838 return In.CheriCapTable->getDynTlsOffset(Sym);
839 case R_MIPS_CHERI_CAPTAB_TLSLD:
840 assert(A == 0 && "capability table index relocs should not have addends");
841 return In.CheriCapTable->getTlsIndexOffset();
842 case R_MIPS_CHERI_CAPTAB_TPREL:
843 assert(A == 0 && "capability table index relocs should not have addends");
844 return In.CheriCapTable->getTlsOffset(Sym);
845 default:
846 llvm_unreachable("invalid expression");
847 }
848}

850// This function applies relocations to sections without SHF_ALLOC bit.
851// Such sections are never mapped to memory at runtime. Debug sections are
852// an example. Relocations in non-alloc sections are much easier to
853// handle than in allocated sections because it will never need complex
854// treatement such as GOT or PLT (because at runtime no one refers them).
855// So, we handle relocations for non-alloc sections directly in this
856// function as a performance optimization.
857template <class ELFT, class RelTy>
858void InputSection::relocateNonAlloc(uint8_t *Buf, ArrayRef<RelTy> Rels) {
859 const unsigned Bits = sizeof(typename ELFT::uint) * 8;
860
861 for (const RelTy &Rel : Rels) {
862 RelType Type = Rel.getType(Config->IsMips64EL);
863
864 // GCC 8.0 or earlier have a bug that they emit R_386_GOTPC relocations
865 // against _GLOBAL_OFFSET_TABLE_ for .debug_info. The bug has been fixed
866 // in 2017 (https://gcc.gnu.org/bugzilla/show_bug.cgi?id=82630), but we
867 // need to keep this bug-compatible code for a while.
868 if (Config->EMachine == EM_386 && Type == R_386_GOTPC)
869 continue;
870
871 uint64_t Offset = getOffset(Rel.r_offset);
872 uint8_t *BufLoc = Buf + Offset;
873 int64_t Addend = getAddend<ELFT>(Rel);
874 if (!RelTy::IsRela)
875 Addend += Target->getImplicitAddend(BufLoc, Type);
876
877 Symbol &Sym = getFile<ELFT>()->getRelocTargetSym(Rel);
878 RelExpr Expr = Target->getRelExpr(Type, Sym, BufLoc);
879 if (Expr == R_NONE)
880 continue;
881
882 if (Expr != R_ABS && Expr != R_DTPREL) {
883 std::string Msg = getLocation<ELFT>(Offset) +
884 ": has non-ABS relocation " + toString(Type) +
885 " against symbol '" + toString(Sym) + "'";
886 if (Expr != R_PC) {
887 error(Msg);
888 return;
889 }
890
891 // If the control reaches here, we found a PC-relative relocation in a
892 // non-ALLOC section. Since non-ALLOC section is not loaded into memory
893 // at runtime, the notion of PC-relative doesn't make sense here. So,
894 // this is a usage error. However, GNU linkers historically accept such
895 // relocations without any errors and relocate them as if they were at
896 // address 0. For bug-compatibilty, we accept them with warnings. We
897 // know Steel Bank Common Lisp as of 2018 have this bug.
898 warn(Msg);
899 Target->relocateOne(BufLoc, Type,
900 SignExtend64<Bits>(Sym.getVA(Addend - Offset)));
901 continue;
902 }
903
904 if (Sym.isTls() && !Out::TlsPhdr)
905 Target->relocateOne(BufLoc, Type, 0);
906 else
907 Target->relocateOne(BufLoc, Type, SignExtend64<Bits>(Sym.getVA(Addend)));
908 }
909}

913// relocations aimed to update addends. They are handled in relocateAlloc()
914// for allocatable sections, and this function does the same for
915// non-allocatable sections, such as sections with debug information.
916static void relocateNonAllocForRelocatable(InputSection *Sec, uint8_t *Buf) {

923 uint64_t TargetVA = SignExtend64(Rel.Sym->getVA(Rel.Addend), Bits);
924 Target->relocateOne(BufLoc, Rel.Type, TargetVA);
925 }

947void InputSectionBase::relocateAlloc(uint8_t *Buf, uint8_t *BufEnd) {
948 assert(Flags & SHF_ALLOC);
949 const unsigned Bits = Config->Wordsize * 8;
950
951 for (const Relocation &Rel : Relocations) {
952 uint64_t Offset = Rel.Offset;
953 if (auto *Sec = dyn_cast<InputSection>(this))
954 Offset += Sec->OutSecOff;
955 uint8_t *BufLoc = Buf + Offset;
956 RelType Type = Rel.Type;
957
958 uint64_t AddrLoc = getOutputSection()->Addr + Offset;
959 RelExpr Expr = Rel.Expr;
960 // FIXME: this should not be happening
961 if (!Rel.Sym) {
962 error(getErrorLocation(BufLoc) + "cannot create relocation against NULL symbol");
963 continue;
964 }
965 uint64_t TargetVA =
966 SignExtend64(getRelocTargetVA(File, Type, Rel.Addend, AddrLoc, *Rel.Sym,
967 Expr, this, Rel.Offset),
968 Bits);
969
970 switch (Expr) {
971 case R_RELAX_GOT_PC:
972 case R_RELAX_GOT_PC_NOPIC:
973 Target->relaxGot(BufLoc, Type, TargetVA);
974 break;
975 case R_PPC64_RELAX_TOC:
976 if (!tryRelaxPPC64TocIndirection(Type, Rel, BufLoc))
977 Target->relocateOne(BufLoc, Type, TargetVA);
978 break;
979 case R_RELAX_TLS_IE_TO_LE:
980 Target->relaxTlsIeToLe(BufLoc, Type, TargetVA);
981 break;
982 case R_RELAX_TLS_LD_TO_LE:
983 case R_RELAX_TLS_LD_TO_LE_ABS:
984 Target->relaxTlsLdToLe(BufLoc, Type, TargetVA);
985 break;
986 case R_RELAX_TLS_GD_TO_LE:
987 case R_RELAX_TLS_GD_TO_LE_NEG:
988 Target->relaxTlsGdToLe(BufLoc, Type, TargetVA);
989 break;
990 case R_AARCH64_RELAX_TLS_GD_TO_IE_PAGE_PC:
991 case R_RELAX_TLS_GD_TO_IE:
992 case R_RELAX_TLS_GD_TO_IE_ABS:
993 case R_RELAX_TLS_GD_TO_IE_GOT_OFF:
994 case R_RELAX_TLS_GD_TO_IE_GOTPLT:
995 Target->relaxTlsGdToIe(BufLoc, Type, TargetVA);
996 break;
997 case R_PPC64_CALL:
998 // If this is a call to __tls_get_addr, it may be part of a TLS
999 // sequence that has been relaxed and turned into a nop. In this
1000 // case, we don't want to handle it as a call.
1001 if (read32(BufLoc) == 0x60000000) // nop
1002 break;
1003
1004 // Patch a nop (0x60000000) to a ld.
1005 if (Rel.Sym->NeedsTocRestore) {
1006 if (BufLoc + 8 > BufEnd || read32(BufLoc + 4) != 0x60000000) {
1007 error(getErrorLocation(BufLoc) + "call lacks nop, can't restore toc");
1008 break;
1009 }
1010 write32(BufLoc + 4, 0xe8410018); // ld %r2, 24(%r1)
1011 }
1012 Target->relocateOne(BufLoc, Type, TargetVA);
1013 break;
1014 default:
1015 Target->relocateOne(BufLoc, Type, TargetVA);
1016 break;
1017 }
1018 }
1019 for (const DynamicReloc &Reloc : FreeBSDMipsRelocationsHack) {
1020 int64_t Addend = Reloc.computeAddend();
1021 // getOffset adds the output section base address here
1022 uint64_t Offset = Reloc.getOffset() - getOutputSection()->Addr;
1023 if (errorHandler().Verbose) {
1024 message("Adding hack: addend=0x" + utohexstr(Addend) +
1025 " offset=0x" + utohexstr(Reloc.getOffset()) + " Type: 0x" + utohexstr(Reloc.Type));
1026 }
1027 uint8_t *BufLoc = Buf + Offset;
1028 Target->relocateOne(BufLoc, Reloc.Type, /*TargetVA=*/Addend);
1029 }

1033template <class ELFT>
1034static void fillGlobalSizesSection(InputSection* IS, uint8_t* Buf, uint8_t* BufEnd) {
1035 static std::mutex Mu;
1036 std::lock_guard<std::mutex> Lock(Mu);
1037 const endianness E = ELFT::TargetEndianness;
1038
1039#ifdef DEBUG_CAP_RELOCS
1040 if (Config->VerboseCapRelocs)
1041 message("Write .global_sizes: IS = " + toString(IS));
1042#endif
1043
1044 foreachGlobalSizesSymbol<ELFT>(IS, [&](StringRef RealSymName, Symbol *Target,
1045 uint64_t Offset) {
1046 uint64_t ResolvedSize = Target->getSize();
1047 uint8_t *Location = Buf + Offset;
1048 assert(Location + 8 <= BufEnd); // Should use a span type instead
1049 bool SizeIsUnknown = ResolvedSize == 0;
1050 if (SizeIsUnknown || Target->isUndefined()) {
1051 // HACK for environ and __progname (both are capabilities):
1052 if (Target->IsSectionStartSymbol) {
1053 assert(!Target->getOutputSection() ||
1054 Target->getOutputSection()->Size == 0);
1055 SizeIsUnknown = false; // output section is empty -> size 0 is fine
1056 } else if (isSectionEndSymbol(RealSymName) ||
1057 isSectionStartSymbol(RealSymName)) {
1058 SizeIsUnknown = false; // zero size is fine for section start/end
1059 } else if (Config->Shared &&
1060 (RealSymName == "__progname" || RealSymName == "environ")) {
1061 message("Using .global_size for symbol " + RealSymName +
1062 " in shared lib (assuming size==sizeof(void* __capability)");
1063 ResolvedSize = Config->CapabilitySize;
1064 SizeIsUnknown = false;
1065 } else {
1066 warn("Could not find .global_size for " + verboseToString(Target));
1067 }
1068 }
1069 if (SizeIsUnknown && (IS->getOutputSection()->Flags & SHF_WRITE) == 0) {
1070 error("Unknown .global_sizes value for " + RealSymName +
1071 " but section was not marked as writable");
1072 }
1073 uint64_t Existing = read64<E>(Location);
1074 if (Existing != 0 && Existing != ResolvedSize) {
1075 // The value might not be zero if we are linking against a file built
1076 // with -r (e.g. openpam_static_modules.o) In that case we need to check
1077 // whether the existing value and the value we want to write matches
1078 error("Conflicting values for .size." + RealSymName +
1079 "\n>>> was already initialized to 0x" + utohexstr(Existing) +
1080 " in " + toString(IS) + "\n>>> expected value is 0x" +
1081 utohexstr(ResolvedSize));
1082 }
1083
1084 write64<E>(Location, ResolvedSize);
1085 if (Config->VerboseCapRelocs)
1086 message("Writing size 0x" + utohexstr(ResolvedSize) + " for " +
1087 verboseToString(Target));
1088 });
1089}

1091// For each function-defining prologue, find any calls to __morestack,
1092// and replace them with calls to __morestack_non_split.
1093static void switchMorestackCallsToMorestackNonSplit(
1094 DenseSet<Defined *> &Prologues, std::vector<Relocation *> &MorestackCalls) {
1095
1096 // If the target adjusted a function's prologue, all calls to
1097 // __morestack inside that function should be switched to
1098 // __morestack_non_split.
1099 Symbol *MoreStackNonSplit = Symtab->find("__morestack_non_split");
1100 if (!MoreStackNonSplit) {
1101 error("Mixing split-stack objects requires a definition of "
1102 "__morestack_non_split");
1103 return;
1104 }
1105
1106 // Sort both collections to compare addresses efficiently.
1107 llvm::sort(MorestackCalls, [](const Relocation *L, const Relocation *R) {
1108 return L->Offset < R->Offset;
1109 });
1110 std::vector<Defined *> Functions(Prologues.begin(), Prologues.end());
1111 llvm::sort(Functions, [](const Defined *L, const Defined *R) {
1112 return L->Value < R->Value;
1113 });
1114
1115 auto It = MorestackCalls.begin();
1116 for (Defined *F : Functions) {
1117 // Find the first call to __morestack within the function.
1118 while (It != MorestackCalls.end() && (*It)->Offset < F->Value)
1119 ++It;
1120 // Adjust all calls inside the function.
1121 while (It != MorestackCalls.end() && (*It)->Offset < F->Value + F->Size) {
1122 (*It)->Sym = MoreStackNonSplit;
1123 ++It;
1124 }
1125 }
1126}

1128static bool enclosingPrologueAttempted(uint64_t Offset,
1129 const DenseSet<Defined *> &Prologues) {
1130 for (Defined *F : Prologues)
1131 if (F->Value <= Offset && Offset < F->Value + F->Size)
1132 return true;
1133 return false;
1134}

1136// If a function compiled for split stack calls a function not
1137// compiled for split stack, then the caller needs its prologue
1138// adjusted to ensure that the called function will have enough stack
1139// available. Find those functions, and adjust their prologues.
1140template <class ELFT>
1141void InputSectionBase::adjustSplitStackFunctionPrologues(uint8_t *Buf,
1142 uint8_t *End) {
1143 if (!getFile<ELFT>()->SplitStack)
1144 return;
1145 DenseSet<Defined *> Prologues;
1146 std::vector<Relocation *> MorestackCalls;
1147
1148 for (Relocation &Rel : Relocations) {
1149 // Local symbols can't possibly be cross-calls, and should have been
1150 // resolved long before this line.
1151 if (Rel.Sym->isLocal())
1152 continue;
1153
1154 // Ignore calls into the split-stack api.
1155 if (Rel.Sym->getName().startswith("__morestack")) {
1156 if (Rel.Sym->getName().equals("__morestack"))
1157 MorestackCalls.push_back(&Rel);
1158 continue;
1159 }
1160
1161 // A relocation to non-function isn't relevant. Sometimes
1162 // __morestack is not marked as a function, so this check comes
1163 // after the name check.
1164 if (Rel.Sym->Type != STT_FUNC)
1165 continue;
1166
1167 // If the callee's-file was compiled with split stack, nothing to do. In
1168 // this context, a "Defined" symbol is one "defined by the binary currently
1169 // being produced". So an "undefined" symbol might be provided by a shared
1170 // library. It is not possible to tell how such symbols were compiled, so be
1171 // conservative.
1172 if (Defined *D = dyn_cast<Defined>(Rel.Sym))
1173 if (InputSection *IS = cast_or_null<InputSection>(D->Section))
1174 if (!IS || !IS->getFile<ELFT>() || IS->getFile<ELFT>()->SplitStack)
1175 continue;
1176
1177 if (enclosingPrologueAttempted(Rel.Offset, Prologues))
1178 continue;
1179
1180 if (Defined *F = getEnclosingFunction<ELFT>(Rel.Offset)) {
1181 Prologues.insert(F);
1182 if (Target->adjustPrologueForCrossSplitStack(Buf + getOffset(F->Value),
1183 End, F->StOther))
1184 continue;
1185 if (!getFile<ELFT>()->SomeNoSplitStack)
1186 error(lld::toString(this) + ": " + F->getName() +
1187 " (with -fsplit-stack) calls " + Rel.Sym->getName() +
1188 " (without -fsplit-stack), but couldn't adjust its prologue");
1189 }
1190 }
1191
1192 if (Target->NeedsMoreStackNonSplit)
1193 switchMorestackCallsToMorestackNonSplit(Prologues, MorestackCalls);
1194}

1196template <class ELFT> void InputSection::writeTo(uint8_t *Buf) {
1197 if (Type == SHT_NOBITS)
1198 return;
1199
1200 if (auto *S = dyn_cast<SyntheticSection>(this)) {
1201 S->writeTo(Buf + OutSecOff);
1202 return;
1203 }
1204
1205 // If -r or --emit-relocs is given, then an InputSection
1206 // may be a relocation section.
1207 if (Type == SHT_RELA) {
1208 copyRelocations<ELFT>(Buf + OutSecOff, getDataAs<typename ELFT::Rela>());
1209 return;
1210 }
1211 if (Type == SHT_REL) {
1212 copyRelocations<ELFT>(Buf + OutSecOff, getDataAs<typename ELFT::Rel>());
1213 return;
1214 }
1215
1216 // If -r is given, we may have a SHT_GROUP section.
1217 if (Type == SHT_GROUP) {
1218 copyShtGroup<ELFT>(Buf + OutSecOff);
1219 return;
1220 }
1221
1222 // If this is a compressed section, uncompress section contents directly
1223 // to the buffer.
1224 if (UncompressedSize >= 0) {
1225 size_t Size = UncompressedSize;
1226 if (Error E = zlib::uncompress(toStringRef(RawData),
1227 (char *)(Buf + OutSecOff), Size))
1228 fatal(toString(this) +
1229 ": uncompress failed: " + llvm::toString(std::move(E)));
1230 uint8_t *BufEnd = Buf + OutSecOff + Size;
1231 relocate<ELFT>(Buf, BufEnd);
1232 return;
1233 }
1234
1235 // Copy section contents from source object file to output file
1236 // and then apply relocations.
1237 memcpy(Buf + OutSecOff, data().data(), data().size());
1238 uint8_t *BufEnd = Buf + OutSecOff + data().size();
1239 if (Config->ProcessCapRelocs && !Config->Relocatable &&
1240 Name == ".global_sizes") {
1241 fillGlobalSizesSection<ELFT>(this, Buf + OutSecOff, BufEnd);
1242 }
1243
1244 relocate<ELFT>(Buf, BufEnd);
1245}

1274// Returns the index of the first relocation that points to a region between
1275// Begin and Begin+Size.
1276template <class IntTy, class RelTy>
1277static unsigned getReloc(IntTy Begin, IntTy Size, const ArrayRef<RelTy> &Rels,
1278 unsigned &RelocI) {
1279 // Start search from RelocI for fast access. That works because the
1280 // relocations are sorted in .eh_frame.
1281 for (unsigned N = Rels.size(); RelocI < N; ++RelocI) {
1282 const RelTy &Rel = Rels[RelocI];
1283 if (Rel.r_offset < Begin)
1284 continue;
1285
1286 if (Rel.r_offset < Begin + Size)
1287 return RelocI;
1288 return -1;
1289 }
1290 return -1;
1291}

1315static size_t findNull(StringRef S, size_t EntSize) {
1316 // Optimize the common case.
1317 if (EntSize == 1)
1318 return S.find(0);
1319
1320 for (unsigned I = 0, N = S.size(); I != N; I += EntSize) {
1321 const char *B = S.begin() + I;
1322 if (std::all_of(B, B + EntSize, [](char C) { return C == 0; }))
1323 return I;
1324 }
1325 return StringRef::npos;
1326}

1332// Split SHF_STRINGS section. Such section is a sequence of
1333// null-terminated strings.
1334void MergeInputSection::splitStrings(ArrayRef<uint8_t> Data, size_t EntSize) {
1335 size_t Off = 0;
1336 bool IsAlloc = Flags & SHF_ALLOC;
1337 StringRef S = toStringRef(Data);
1338
1339 while (!S.empty()) {
1340 size_t End = findNull(S, EntSize);
1341 if (End == StringRef::npos)
1342 fatal(toString(this) + ": string is not null terminated");
1343 size_t Size = End + EntSize;
1344
1345 Pieces.emplace_back(Off, xxHash64(S.substr(0, Size)), !IsAlloc);
1346 S = S.substr(Size);
1347 Off += Size;
1348 }
1349}

1352// fixed size records.
1353void MergeInputSection::splitNonStrings(ArrayRef<uint8_t> Data,
