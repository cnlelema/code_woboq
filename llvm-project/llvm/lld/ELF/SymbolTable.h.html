22// SymbolTable is a bucket of all known symbols, including defined,
23// undefined, or lazy symbols (the last one is symbols in archive
24// files whose archive members are not yet loaded).
25//
26// We put all symbols of all files to a SymbolTable, and the
27// SymbolTable selects the "best" symbols if there are name
28// conflicts. For example, obviously, a defined symbol is better than
29// an undefined symbol. Or, if there's a conflict between a lazy and a
30// undefined, it'll read an archive member to read a real definition
31// to replace the lazy symbol. The logic is implemented in the
32// add*() functions, which are called by input files as they are parsed. There
33// is one add* function per symbol type.
34class SymbolTable {
35public:
36 void wrap(Symbol *Sym, Symbol *Real, Symbol *Wrap);
37
38 void forEachSymbol(llvm::function_ref<void(Symbol *)> Fn) {
39 for (Symbol *Sym : SymVector)
40 if (!Sym->isPlaceholder())
41 Fn(Sym);
42 }
43
44 Symbol *insert(StringRef Name);
45
46 Symbol *addSymbol(const Symbol &New);
47
48 void scanVersionScript();
49
50 Symbol *find(StringRef Name);
51
52 void handleDynamicList();
53
54 // Set of .so files to not link the same shared object file more than once.
55 llvm::DenseMap<StringRef, SharedFile *> SoNames;
56
57 // This is a workaround for CHERI to add local symbols to .dynsym for relocations
58 Defined *ensureSymbolWillBeInDynsym(Symbol* Original);
59
60 // Comdat groups define "link once" sections. If two comdat groups have the
61 // same name, only one of them is linked, and the other is ignored. This map
62 // is used to uniquify them.
63 llvm::DenseMap<llvm::CachedHashStringRef, const InputFile *> ComdatGroups;
64
65 // Removed upstream, needed for CHERI
66 std::vector<Symbol*> &getSymbols() { return SymVector; }
67
68private:
69 std::vector<Symbol *> findByVersion(SymbolVersion Ver);
70 std::vector<Symbol *> findAllByVersion(SymbolVersion Ver);
71
72 llvm::StringMap<std::vector<Symbol *>> &getDemangledSyms();
73 void handleAnonymousVersion();
74 void assignExactVersion(SymbolVersion Ver, uint16_t VersionId,
75 StringRef VersionName);
76 void assignWildcardVersion(SymbolVersion Ver, uint16_t VersionId);
77
78 // The order the global symbols are in is not defined. We can use an arbitrary
79 // order, but it has to be reproducible. That is true even when cross linking.
80 // The default hashing of StringRef produces different results on 32 and 64
81 // bit systems so we use a map to a vector. That is arbitrary, deterministic
82 // but a bit inefficient.
83 // FIXME: Experiment with passing in a custom hashing or sorting the symbols
84 // once symbol resolution is finished.
85 llvm::DenseMap<llvm::CachedHashStringRef, int> SymMap;
86 std::vector<Symbol *> SymVector;
87 llvm::DenseMap<Symbol *, Defined *> LocalSymbolsForDynsym;
88
89
90 // A map from demangled symbol names to their symbol objects.
91 // This mapping is 1:N because two symbols with different versions
92 // can have the same name. We use this map to handle "extern C++ {}"
93 // directive in version scripts.
94 llvm::Optional<llvm::StringMap<std::vector<Symbol *>>> DemangledSyms;
95};
96
97extern SymbolTable *Symtab;
