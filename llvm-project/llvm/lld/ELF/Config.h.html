276 // True if the target is the little-endian MIPS64.
277 //
278 // The reason why we have this variable only for the MIPS is because
279 // we use this often. Some ELF headers for MIPS64EL are in a
280 // mixed-endian (which is horrible and I'd say that's a serious spec
281 // bug), and we need to know whether we are reading MIPS ELF files or
282 // not in various places.
283 //
284 // (Note that MIPS64EL is not a typo for MIPS64LE. This is the official
285 // name whatever that means. A fun hypothesis is that "EL" is short for
286 // little-endian written in the little-endian order, but I don't know
287 // if that's true.)
288 bool IsMips64EL;
289
290 // True if we need to set the DF_STATIC_TLS flag to an output file,
291 // which works as a hint to the dynamic loader that the file contains
292 // code compiled with the static TLS model. The thread-local variable
293 // compiled with the static TLS model is faster but less flexible, and
294 // it may not be loaded using dlopen().
295 //
296 // We set this flag to true when we see a relocation for the static TLS
297 // model. Once this becomes true, it will never become false.
298 //
299 // Since the flag is updated by multi-threaded code, we use std::atomic.
300 // (Writing to a variable is not considered thread-safe even if the
301 // variable is boolean and we always set the same value from all threads.)
302 std::atomic<bool> HasStaticTlsModel{false};
303
304 // Holds set of ELF header flags for the target.
305 uint32_t EFlags = 0;
306
307 // The ELF spec defines two types of relocation table entries, RELA and
308 // REL. RELA is a triplet of (offset, info, addend) while REL is a
309 // tuple of (offset, info). Addends for REL are implicit and read from
310 // the location where the relocations are applied. So, REL is more
311 // compact than RELA but requires a bit of more work to process.
312 //
313 // (From the linker writer's view, this distinction is not necessary.
314 // If the ELF had chosen whichever and sticked with it, it would have
315 // been easier to write code to process relocations, but it's too late
316 // to change the spec.)
317 //
318 // Each ABI defines its relocation type. IsRela is true if target
319 // uses RELA. As far as we know, all 64-bit ABIs are using RELA. A
320 // few 32-bit ABIs are using RELA too.
321 bool IsRela;
322

332 inline bool isCheriABI() const { return CheriABI; }
333 // We need to set the searchPaths before createFiles() is called since linker
334 // scripts might contain INPUT() commands. Add a getter and setter for
335 // CheriABI to ensure this is always the case
336 inline void setIsCheriABI(bool Set) {
337 if (!Set)
338 return;
339 CheriABI = true;
340 if (DynamicLinker.empty())
341 DynamicLinker = "/libexec/ld-cheri-elf.so.1";
342 // add the default search paths for CheriABI
343 SearchPaths.emplace_back("=/libcheri");
344 SearchPaths.emplace_back("=/usr/libcheri");
345 SearchPaths.emplace_back("=/usr/local/libcheri");
346 }
