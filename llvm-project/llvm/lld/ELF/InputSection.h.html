37// This is the base class of all sections that lld handles. Some are sections in
38// input files, some are sections in the produced output file and some exist
39// just as a convenience for implementing special ways of combining some
40// sections.
41class SectionBase {
42public:
43 enum Kind { Regular, EHFrame, Merge, Synthetic, Output };
44
45 Kind kind() const { return (Kind)SectionKind; }
46
47 StringRef Name;
48
49 // This pointer points to the "real" instance of this instance.
50 // Usually Repl == this. However, if ICF merges two sections,
51 // Repl pointer of one section points to another section. So,
52 // if you need to get a pointer to this instance, do not use
53 // this but instead this->Repl.
54 SectionBase *Repl;
55
56 unsigned SectionKind : 3;
57
58 // The next three bit fields are only used by InputSectionBase, but we
59 // put them here so the struct packs better.
60
61 // True if this section has already been placed to a linker script
62 // output section. This is needed because, in a linker script, you
63 // can refer to the same section more than once. For example, in
64 // the following linker script,
65 //
66 // .foo : { *(.text) }
67 // .bar : { *(.text) }
68 //
69 // .foo takes all .text sections, and .bar becomes empty. To achieve
70 // this, we need to memorize whether a section has been placed or
71 // not for each input section.
72 unsigned Assigned : 1;
73
74 unsigned Bss : 1;
75
76 // Set for sections that should not be folded by ICF.
77 unsigned KeepUnique : 1;
78
79 // The 1-indexed partition that this section is assigned to by the garbage
80 // collector, or 0 if this section is dead. Normally there is only one
81 // partition, so this will either be 0 or 1.
82 uint8_t Partition;
83 elf::Partition &getPartition() const;
84
85 // These corresponds to the fields in Elf_Shdr.
86 uint32_t Alignment;
87 uint64_t Flags;
88 uint64_t Entsize;
89 uint32_t Type;
90 uint32_t Link;
91 uint32_t Info;
92
93 OutputSection *getOutputSection();
94 const OutputSection *getOutputSection() const {
95 return const_cast<SectionBase *>(this)->getOutputSection();
96 }
97
98 // Translate an offset in the input section to an offset in the output
99 // section.
100 uint64_t getOffset(uint64_t Offset) const;
101
102 uint64_t getVA(uint64_t Offset = 0) const;
103
104 bool isLive() const { return Partition != 0; }
105 void markLive() { Partition = 1; }
106 void markDead() { Partition = 0; }
107
108protected:
109 SectionBase(Kind SectionKind, StringRef Name, uint64_t Flags,
110 uint64_t Entsize, uint64_t Alignment, uint32_t Type,
111 uint32_t Info, uint32_t Link)
112 : Name(Name), Repl(this), SectionKind(SectionKind), Assigned(false),
113 Bss(false), KeepUnique(false), Partition(0), Alignment(Alignment),
114 Flags(Flags), Entsize(Entsize), Type(Type), Link(Link), Info(Info) {}
115};

117// This corresponds to a section of an input file.
118class InputSectionBase : public SectionBase {
119public:
120 template <class ELFT>
121 InputSectionBase(ObjFile<ELFT> &File, const typename ELFT::Shdr &Header,
122 StringRef Name, Kind SectionKind);
123
124 InputSectionBase(InputFile *File, uint64_t Flags, uint32_t Type,
125 uint64_t Entsize, uint32_t Link, uint32_t Info,
126 uint32_t Alignment, ArrayRef<uint8_t> Data, StringRef Name,
127 Kind SectionKind);
128
129 static bool classof(const SectionBase *S) { return S->kind() != Output; }
130
131 // Relocations that refer to this section.
132 unsigned NumRelocations : 31;
133 unsigned AreRelocsRela : 1;
134 const void *FirstRelocation = nullptr;
135
136 // The file which contains this section. Its dynamic type is always
137 // ObjFile<ELFT>, but in order to avoid ELFT, we use InputFile as
138 // its static type.
139 InputFile *File;
140
141 template <class ELFT> ObjFile<ELFT> *getFile() const {
142 return cast_or_null<ObjFile<ELFT>>(File);
143 }
144
145 ArrayRef<uint8_t> data() const {
146 if (UncompressedSize >= 0)
147 uncompress();
148 return RawData;
149 }
150
151 uint64_t getOffsetInFile() const;
152
153 // Input sections are part of an output section. Special sections
154 // like .eh_frame and merge sections are first combined into a
155 // synthetic section that is then added to an output section. In all
156 // cases this points one level up.
157 SectionBase *Parent = nullptr;
158
159 template <class ELFT> ArrayRef<typename ELFT::Rel> rels() const {
160 assert(!AreRelocsRela);
161 return llvm::makeArrayRef(
162 static_cast<const typename ELFT::Rel *>(FirstRelocation),
163 NumRelocations);
164 }
165
166 template <class ELFT> ArrayRef<typename ELFT::Rela> relas() const {
167 assert(AreRelocsRela);
168 return llvm::makeArrayRef(
169 static_cast<const typename ELFT::Rela *>(FirstRelocation),
170 NumRelocations);
171 }
172
173 // InputSections that are dependent on us (reverse dependency for GC)
174 llvm::TinyPtrVector<InputSection *> DependentSections;
175
176 // Returns the size of this section (even if this is a common or BSS.)
177 size_t getSize() const;
178
179 InputSection *getLinkOrderDep() const;
180
181 // Get the function symbol that encloses this offset from within the
182 // section.
183 template <class ELFT>
184 Defined *getEnclosingFunction(uint64_t Offset) const;
185 template <class ELFT>
186 Defined *getEnclosingObject(uint64_t Offset) const;
187 template <class ELFT, unsigned SymbolType>
188 Defined *getEnclosingSymbol(uint64_t Offset) const;
189
190 // Returns a source location string. Used to construct an error message.
191 template <class ELFT> std::string getLocation(uint64_t Offset) const;
192 std::string getSrcMsg(const Symbol &Sym, uint64_t Offset) const;
193 std::string getObjMsg(uint64_t Offset) const;
194
195 // Each section knows how to relocate itself. These functions apply
196 // relocations, assuming that Buf points to this section's copy in
197 // the mmap'ed output buffer.
198 template <class ELFT> void relocate(uint8_t *Buf, uint8_t *BufEnd);
199 void relocateAlloc(uint8_t *Buf, uint8_t *BufEnd);
200
201 // The native ELF reloc data type is not very convenient to handle.
202 // So we convert ELF reloc records to our own records in Relocations.cpp.
203 // This vector contains such "cooked" relocations.
204 std::vector<Relocation> Relocations;
205 std::vector<DynamicReloc> FreeBSDMipsRelocationsHack;
206
207 // A function compiled with -fsplit-stack calling a function
208 // compiled without -fsplit-stack needs its prologue adjusted. Find
209 // such functions and adjust their prologues. This is very similar
210 // to relocation. See https://gcc.gnu.org/wiki/SplitStacks for more
211 // information.
212 template <typename ELFT>
213 void adjustSplitStackFunctionPrologues(uint8_t *Buf, uint8_t *End);
214
215
216 template <typename T> llvm::ArrayRef<T> getDataAs() const {
217 size_t S = data().size();
218 assert(S % sizeof(T) == 0);
219 return llvm::makeArrayRef<T>((const T *)data().data(), S / sizeof(T));
220 }
221
222protected:
223 void parseCompressedHeader();
224 void uncompress() const;
225
226 mutable ArrayRef<uint8_t> RawData;
227
228 // This field stores the uncompressed size of the compressed data in RawData,
229 // or -1 if RawData is not compressed (either because the section wasn't
230 // compressed in the first place, or because we ended up uncompressing it).
231 // Since the feature is not used often, this is usually -1.
232 mutable int64_t UncompressedSize = -1;
233};

235// SectionPiece represents a piece of splittable section contents.
236// We allocate a lot of these and binary search on them. This means that they
237// have to be as compact as possible, which is why we don't store the size (can
238// be found by looking at the next one).
239struct SectionPiece {
240 SectionPiece(size_t Off, uint32_t Hash, bool Live)
241 : InputOff(Off), Live(Live || !Config->GcSections), Hash(Hash >> 1) {}
242
243 uint32_t InputOff;
244 uint32_t Live : 1;
245 uint32_t Hash : 31;
246 uint64_t OutputOff = 0;
247};
248
249static_assert(sizeof(SectionPiece) == 16, "SectionPiece is too big");

251// This corresponds to a SHF_MERGE section of an input file.
252class MergeInputSection : public InputSectionBase {
253public:
254 template <class ELFT>
255 MergeInputSection(ObjFile<ELFT> &F, const typename ELFT::Shdr &Header,
256 StringRef Name);
257 MergeInputSection(uint64_t Flags, uint32_t Type, uint64_t Entsize,
258 ArrayRef<uint8_t> Data, StringRef Name);
259
260 static bool classof(const SectionBase *S) { return S->kind() == Merge; }
261 void splitIntoPieces();
262
263 // Translate an offset in the input section to an offset in the parent
264 // MergeSyntheticSection.
265 uint64_t getParentOffset(uint64_t Offset) const;
266
267 // Splittable sections are handled as a sequence of data
268 // rather than a single large blob of data.
269 std::vector<SectionPiece> Pieces;
270
271 // Returns I'th piece's data. This function is very hot when
272 // string merging is enabled, so we want to inline.
273 LLVM_ATTRIBUTE_ALWAYS_INLINE
274 llvm::CachedHashStringRef getData(size_t I) const {
275 size_t Begin = Pieces[I].InputOff;
276 size_t End =
277 (Pieces.size() - 1 == I) ? data().size() : Pieces[I + 1].InputOff;
278 return {toStringRef(data().slice(Begin, End - Begin)), Pieces[I].Hash};
279 }
280
281 // Returns the SectionPiece at a given input section offset.
282 SectionPiece *getSectionPiece(uint64_t Offset);
283 const SectionPiece *getSectionPiece(uint64_t Offset) const {
284 return const_cast<MergeInputSection *>(this)->getSectionPiece(Offset);
285 }
286
287 SyntheticSection *getParent() const;
288
289private:
290 void splitStrings(ArrayRef<uint8_t> A, size_t Size);
291 void splitNonStrings(ArrayRef<uint8_t> A, size_t Size);
292};

294struct EhSectionPiece {
295 EhSectionPiece(size_t Off, InputSectionBase *Sec, uint32_t Size,
296 unsigned FirstRelocation)
297 : InputOff(Off), Sec(Sec), Size(Size), FirstRelocation(FirstRelocation) {}
298
299 ArrayRef<uint8_t> data() {
300 return {Sec->data().data() + this->InputOff, Size};
301 }
302
303 size_t InputOff;
304 ssize_t OutputOff = -1;
305 InputSectionBase *Sec;
306 uint32_t Size;
307 unsigned FirstRelocation;
308};

327// This is a section that is added directly to an output section
328// instead of needing special combination via a synthetic section. This
329// includes all input sections with the exceptions of SHF_MERGE and
330// .eh_frame. It also includes the synthetic sections themselves.
331class InputSection : public InputSectionBase {
332public:
333 InputSection(InputFile *F, uint64_t Flags, uint32_t Type, uint32_t Alignment,
334 ArrayRef<uint8_t> Data, StringRef Name, Kind K = Regular);
335 template <class ELFT>
336 InputSection(ObjFile<ELFT> &F, const typename ELFT::Shdr &Header,
337 StringRef Name);
338
339 // Write this section to a mmap'ed file, assuming Buf is pointing to
340 // beginning of the output section.
341 template <class ELFT> void writeTo(uint8_t *Buf);
342
343 uint64_t getOffset(uint64_t Offset) const { return OutSecOff + Offset; }
344
345 OutputSection *getParent() const;
346
347 // This variable has two usages. Initially, it represents an index in the
348 // OutputSection's InputSection list, and is used when ordering SHF_LINK_ORDER
349 // sections. After assignAddresses is called, it represents the offset from
350 // the beginning of the output section this section was assigned to.
351 uint64_t OutSecOff = 0;
352
353 static bool classof(const SectionBase *S);
354
355 InputSectionBase *getRelocatedSection() const;
356
357 template <class ELFT, class RelTy>
358 void relocateNonAlloc(uint8_t *Buf, llvm::ArrayRef<RelTy> Rels);
359
360 // Used by ICF.
361 uint32_t Class[2] = {0, 0};
362
363 // Called by ICF to merge two input sections.
364 void replace(InputSection *Other);
365
366 static InputSection Discarded;
367
368private:
369 template <class ELFT, class RelTy>
370 void copyRelocations(uint8_t *Buf, llvm::ArrayRef<RelTy> Rels);
371
372 template <class ELFT> void copyShtGroup(uint8_t *Buf);
373};
