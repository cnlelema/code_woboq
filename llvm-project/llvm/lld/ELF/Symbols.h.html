41// This is a StringRef-like container that doesn't run strlen().
42//
43// ELF string tables contain a lot of null-terminated strings. Most of them
44// are not necessary for the linker because they are names of local symbols,
45// and the linker doesn't use local symbol names for name resolution. So, we
46// use this class to represents strings read from string tables.
47struct StringRefZ {
48 StringRefZ(const char *S) : Data(S), Size(-1) {}
49 StringRefZ(StringRef S) : Data(S.data()), Size(S.size()) {}
50
51 const char *Data;
52 const uint32_t Size;
53};

55// The base class for real symbol classes.
56class Symbol {
57public:
58 enum Kind {
59 PlaceholderKind,
60 DefinedKind,
61 CommonKind,
62 SharedKind,
63 UndefinedKind,
64 LazyArchiveKind,
65 LazyObjectKind,
66 };
67
68 Kind kind() const { return static_cast<Kind>(SymbolKind); }

70 // The file from which this symbol was created.
71 InputFile *File;
72
73protected:
74 const char *NameData;
75 mutable uint32_t NameSize;
76
77public:
78 uint32_t DynsymIndex = 0;
79 uint32_t GotIndex = -1;
80 uint32_t PltIndex = -1;
81
82 uint32_t GlobalDynIndex = -1;
83
84 // This field is a index to the symbol's version definition.
85 uint32_t VerdefIndex = -1;
86
87 // Version definition index.
88 uint16_t VersionId;
89
90 // An index into the .branch_lt section on PPC64.
91 uint16_t PPC64BranchltIndex = -1;
92
93 // Symbol binding. This is not overwritten by replace() to track
94 // changes during resolution. In particular:
95 // - An undefined weak is still weak when it resolves to a shared library.
96 // - An undefined weak will not fetch archive members, but we have to
97 // remember it is weak.
98 uint8_t Binding;
99
100 // The following fields have the same meaning as the ELF symbol attributes.
101 uint8_t Type; // symbol type
102 uint8_t StOther; // st_other field value
103
104 uint8_t SymbolKind;
105
106 // Symbol visibility. This is the computed minimum visibility of all
107 // observed non-DSO symbols.
108 unsigned Visibility : 2;
109
110 // True if the symbol was used for linking and thus need to be added to the
111 // output file's symbol table. This is true for all symbols except for
112 // unreferenced DSO symbols, lazy (archive) symbols, and bitcode symbols that
113 // are unreferenced except by other bitcode objects.
114 unsigned IsUsedInRegularObj : 1;
115
116 // If this flag is true and the symbol has protected or default visibility, it
117 // will appear in .dynsym. This flag is set by interposable DSO symbols in
118 // executables, by most symbols in DSOs and executables built with
119 // --export-dynamic, and by dynamic lists.
120 unsigned ExportDynamic : 1;
121
122 // False if LTO shouldn't inline whatever this symbol points to. If a symbol
123 // is overwritten after LTO, LTO shouldn't inline the symbol because it
124 // doesn't know the final contents of the symbol.
125 unsigned CanInline : 1;
126
127 // True if this symbol is specified by --trace-symbol option.
128 unsigned Traced : 1;
129
130 inline void replace(const Symbol &New);
131
132 bool includeInDynsym() const;
133 uint8_t computeBinding() const;
134 bool isWeak() const { return Binding == llvm::ELF::STB_WEAK; }
135
136 bool isUndefined() const { return SymbolKind == UndefinedKind; }
137 bool isCommon() const { return SymbolKind == CommonKind; }
138 bool isDefined() const { return SymbolKind == DefinedKind; }
139 bool isShared() const { return SymbolKind == SharedKind; }
140 bool isPlaceholder() const { return SymbolKind == PlaceholderKind; }
141
142 bool isLocal() const { return Binding == llvm::ELF::STB_LOCAL; }
143
144 bool isLazy() const {
145 return SymbolKind == LazyArchiveKind || SymbolKind == LazyObjectKind;
146 }
147
148 // True if this is an undefined weak symbol. This only works once
149 // all input files have been added.
150 bool isUndefWeak() const {
151 // See comment on lazy symbols for details.
152 return isWeak() && (isUndefined() || isLazy());
153 }

184 // The following two functions are used for symbol resolution.
185 //
186 // You are expected to call mergeProperties for all symbols in input
187 // files so that attributes that are attached to names rather than
188 // indivisual symbol (such as visibility) are merged together.
189 //
190 // Every time you read a new symbol from an input, you are supposed
191 // to call resolve() with the new symbol. That function replaces
192 // "this" object as a result of name resolution if the new symbol is
193 // more appropriate to be included in the output.
194 //
195 // For example, if "this" is an undefined symbol and a new symbol is
196 // a defined symbol, "this" is replaced with the new symbol.
197 void mergeProperties(const Symbol &Other);
198 void resolve(const Symbol &Other);
199
200 // If this is a lazy symbol, fetch an input file and add the symbol
201 // in the file to the symbol table. Calling this function on
202 // non-lazy object causes a runtime error.
203 void fetch() const;
204
205private:
206 static bool isExportDynamic(Kind K, uint8_t Visibility) {
207 if (K == SharedKind)
208 return Visibility == llvm::ELF::STV_DEFAULT;
209 return Config->Shared || Config->ExportDynamic;
210 }
211
212 void resolveUndefined(const Undefined &Other);
213 void resolveCommon(const CommonSymbol &Other);
214 void resolveDefined(const Defined &Other);
215 template <class LazyT> void resolveLazy(const LazyT &Other);
216 void resolveShared(const SharedSymbol &Other);
217
218 int compare(const Symbol *Other) const;
219
220 inline size_t getSymbolSize() const;
221
222protected:
223 Symbol(Kind K, InputFile *File, StringRefZ Name, uint8_t Binding,
224 uint8_t StOther, uint8_t Type)
225 : File(File), NameData(Name.Data), NameSize(Name.Size), Binding(Binding),
226 Type(Type), StOther(StOther), SymbolKind(K), Visibility(StOther & 3),
227 IsUsedInRegularObj(!File || File->kind() == InputFile::ObjKind),
228 ExportDynamic(isExportDynamic(K, Visibility)), CanInline(false),
229 Traced(false), NeedsPltAddr(false), IsInIplt(false), GotInIgot(false),
230 IsPreemptible(false), Used(!Config->GcSections), UsedByDynReloc(false),
231 IsSectionStartSymbol(false), NeedsTocRestore(false),
232 ScriptDefined(false) {}
233
234public:
235 // True the symbol should point to its PLT entry.
236 // For SharedSymbol only.
237 unsigned NeedsPltAddr : 1;
238
239 // True if this symbol is in the Iplt sub-section of the Plt and the Igot
240 // sub-section of the .got.plt or .got.
241 unsigned IsInIplt : 1;
242
243 // True if this symbol needs a GOT entry and its GOT entry is actually in
244 // Igot. This will be true only for certain non-preemptible ifuncs.
245 unsigned GotInIgot : 1;
246
247 // True if this symbol is preemptible at load time.
248 unsigned IsPreemptible : 1;
249
250 // True if an undefined or shared symbol is used from a live section.
251 unsigned Used : 1;
252
253 // True if a symbol is referenced by a dynamic relocation and therefore needs
254 // to be included in the dynamic symbol table.
255 unsigned UsedByDynReloc : 1;
256
257 // True if the linker should set the size of this symbol to be the size of the
258 // section it references. For compatibility reason this is only used when
259 // building for CHERI
260 unsigned IsSectionStartSymbol : 1;
261
262 // True if a call to this symbol needs to be followed by a restore of the
263 // PPC64 toc pointer.
264 unsigned NeedsTocRestore : 1;
265
266 // True if this symbol is defined by a linker script.
267 unsigned ScriptDefined : 1;
268
269 // The partition whose dynamic symbol table contains this symbol's definition.
270 uint8_t Partition = 1;
271
272 bool isSection() const { return Type == llvm::ELF::STT_SECTION; }
273 bool isTls() const { return Type == llvm::ELF::STT_TLS; }
274 bool isFunc() const { return Type == llvm::ELF::STT_FUNC; }
275 bool isGnuIFunc() const { return Type == llvm::ELF::STT_GNU_IFUNC; }
276 bool isObject() const { return Type == llvm::ELF::STT_OBJECT; }
277 bool isFile() const { return Type == llvm::ELF::STT_FILE; }
278};

295// Represents a common symbol.
296//
297// On Unix, it is traditionally allowed to write variable definitions
298// without initialization expressions (such as "int foo;") to header
299// files. Such definition is called "tentative definition".
300//
301// Using tentative definition is usually considered a bad practice
302// because you should write only declarations (such as "extern int
303// foo;") to header files. Nevertheless, the linker and the compiler
304// have to do something to support bad code by allowing duplicate
305// definitions for this particular case.
306//
307// Common symbols represent variable definitions without initializations.
308// The compiler creates common symbols when it sees varaible definitions
309// without initialization (you can suppress this behavior and let the
310// compiler create a regular defined symbol by -fno-common).
311//
312// The linker allows common symbols to be replaced by regular defined
313// symbols. If there are remaining common symbols after name resolution is
314// complete, they are converted to regular defined symbols in a .bss
315// section. (Therefore, the later passes don't see any CommonSymbols.)
316class CommonSymbol : public Symbol {
317public:
318 CommonSymbol(InputFile *File, StringRefZ Name, uint8_t Binding,
319 uint8_t StOther, uint8_t Type, uint64_t Alignment, uint64_t Size)
320 : Symbol(CommonKind, File, Name, Binding, StOther, Type),
321 Alignment(Alignment), Size(Size) {}
322
323 static bool classof(const Symbol *S) { return S->isCommon(); }
324
325 uint32_t Alignment;
326 uint64_t Size;
327};

336 static bool classof(const Symbol *S) { return S->kind() == UndefinedKind; }

338 // The section index if in a discarded section, 0 otherwise.
339 uint32_t DiscardedSecIdx;

342class SharedSymbol : public Symbol {
343public:
344 static bool classof(const Symbol *S) { return S->kind() == SharedKind; }
345
346 SharedSymbol(InputFile &File, StringRef Name, uint8_t Binding,
347 uint8_t StOther, uint8_t Type, uint64_t Value, uint64_t Size,
348 uint32_t Alignment, uint32_t VerdefIndex)
349 : Symbol(SharedKind, &File, Name, Binding, StOther, Type),
350 Alignment(Alignment), Value(Value), Size(Size) {
351 this->VerdefIndex = VerdefIndex;
352 // GNU ifunc is a mechanism to allow user-supplied functions to
353 // resolve PLT slot values at load-time. This is contrary to the
354 // regular symbol resolution scheme in which symbols are resolved just
355 // by name. Using this hook, you can program how symbols are solved
356 // for you program. For example, you can make "memcpy" to be resolved
357 // to a SSE-enabled version of memcpy only when a machine running the
358 // program supports the SSE instruction set.
359 //
360 // Naturally, such symbols should always be called through their PLT
361 // slots. What GNU ifunc symbols point to are resolver functions, and
362 // calling them directly doesn't make sense (unless you are writing a
363 // loader).
364 //
365 // For DSO symbols, we always call them through PLT slots anyway.
366 // So there's no difference between GNU ifunc and regular function
367 // symbols if they are in DSOs. So we can handle GNU_IFUNC as FUNC.
368 if (this->Type == llvm::ELF::STT_GNU_IFUNC)
369 this->Type = llvm::ELF::STT_FUNC;
370 }
371
372 SharedFile &getFile() const { return *cast<SharedFile>(File); }
373
374 uint32_t Alignment;
375
376 uint64_t Value; // st_value
377 uint64_t Size; // st_size
378};

380// LazyArchive and LazyObject represent a symbols that is not yet in the link,
381// but we know where to find it if needed. If the resolver finds both Undefined
382// and Lazy for the same name, it will ask the Lazy to load a file.
383//
384// A special complication is the handling of weak undefined symbols. They should
385// not load a file, but we have to remember we have seen both the weak undefined
386// and the lazy. We represent that with a lazy symbol with a weak binding. This
387// means that code looking for undefined symbols normally also has to take lazy
388// symbols into consideration.
389
390// This class represents a symbol defined in an archive file. It is
391// created from an archive file header, and it knows how to load an
392// object file from an archive to replace itself with a defined
393// symbol.
394class LazyArchive : public Symbol {
395public:
396 LazyArchive(InputFile &File, const llvm::object::Archive::Symbol S)
397 : Symbol(LazyArchiveKind, &File, S.getName(), llvm::ELF::STB_GLOBAL,
398 llvm::ELF::STV_DEFAULT, llvm::ELF::STT_NOTYPE),
399 Sym(S) {}
400
401 static bool classof(const Symbol *S) { return S->kind() == LazyArchiveKind; }
402
403 MemoryBufferRef getMemberBuffer();
404
405 const llvm::object::Archive::Symbol Sym;
406};

459// A buffer class that is large enough to hold any Symbol-derived
460// object. We allocate memory using this class and instantiate a symbol
461// using the placement new.
462union SymbolUnion {
463 alignas(Defined) char A[sizeof(Defined)];
464 alignas(CommonSymbol) char B[sizeof(CommonSymbol)];
465 alignas(Undefined) char C[sizeof(Undefined)];
466 alignas(SharedSymbol) char D[sizeof(SharedSymbol)];
467 alignas(LazyArchive) char E[sizeof(LazyArchive)];
468 alignas(LazyObject) char F[sizeof(LazyObject)];
469};
470
471template <typename T> struct AssertSymbol {
472 static_assert(std::is_trivially_destructible<T>(),
473 "Symbol types must be trivially destructible");
474 static_assert(sizeof(T) <= sizeof(SymbolUnion), "SymbolUnion too small");
475 static_assert(alignof(T) <= alignof(SymbolUnion),
476 "SymbolUnion not aligned enough");
477};

486}

490size_t Symbol::getSymbolSize() const {
491 switch (kind()) {
492 case CommonKind:
493 return sizeof(CommonSymbol);
494 case DefinedKind:
495 return sizeof(Defined);
496 case LazyArchiveKind:
497 return sizeof(LazyArchive);
498 case LazyObjectKind:
499 return sizeof(LazyObject);
500 case SharedKind:
501 return sizeof(SharedSymbol);
502 case UndefinedKind:
503 return sizeof(Undefined);
504 case PlaceholderKind:
505 return sizeof(Symbol);
506 }
507 llvm_unreachable("unknown symbol kind");
508}

510// replace() replaces "this" object with a given symbol by memcpy'ing
511// it over to "this". This function is called as a result of name
512// resolution, e.g. to replace an undefind symbol with a defined symbol.
513void Symbol::replace(const Symbol &New) {
514 using llvm::ELF::STT_TLS;
515
516 // Symbols representing thread-local variables must be referenced by
517 // TLS-aware relocations, and non-TLS symbols must be reference by
518 // non-TLS relocations, so there's a clear distinction between TLS
519 // and non-TLS symbols. It is an error if the same symbol is defined
520 // as a TLS symbol in one file and as a non-TLS symbol in other file.
521 if (SymbolKind != PlaceholderKind && !isLazy() && !New.isLazy()) {
522 bool TlsMismatch = (Type == STT_TLS && New.Type != STT_TLS) ||
523 (Type != STT_TLS && New.Type == STT_TLS);
524 if (TlsMismatch)
525 error("TLS attribute mismatch: " + toString(*this) + "\n>>> defined in " +
526 toString(New.File) + "\n>>> defined in " + toString(File));
527 }
528
529 Symbol Old = *this;
530 memcpy(this, &New, New.getSymbolSize());
531
532 VersionId = Old.VersionId;
533 Visibility = Old.Visibility;
534 IsUsedInRegularObj = Old.IsUsedInRegularObj;
535 ExportDynamic = Old.ExportDynamic;
536 CanInline = Old.CanInline;
537 Traced = Old.Traced;
538 IsPreemptible = Old.IsPreemptible;
539 ScriptDefined = Old.ScriptDefined;
540 Partition = Old.Partition;
541
542 // Symbol length is computed lazily. If we already know a symbol length,
543 // propagate it.
544 if (NameData == Old.NameData && NameSize == 0 && Old.NameSize != 0)
545 NameSize = Old.NameSize;
546
547 // Print out a log message if --trace-symbol was specified.
548 // This is for debugging.
549 if (Traced)
550 printTraceSymbol(this);
551}
