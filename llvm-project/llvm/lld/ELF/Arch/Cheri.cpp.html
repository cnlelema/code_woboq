23template <class ELFT> struct InMemoryCapRelocEntry {
24 using NativeUint = typename ELFT::uint;
25 using CapRelocUint = llvm::support::detail::packed_endian_specific_integral<
26 NativeUint, ELFT::TargetEndianness, llvm::support::aligned>;
27 InMemoryCapRelocEntry(NativeUint Loc, NativeUint Obj, NativeUint Off,
28 NativeUint S, NativeUint Perms)
29 : capability_location(Loc), object(Obj), offset(Off), size(S),
30 permissions(Perms) {}
31 CapRelocUint capability_location;
32 CapRelocUint object;
33 CapRelocUint offset;
34 CapRelocUint size;
35 CapRelocUint permissions;
36};

48template <class ELFT>
49void CheriCapRelocsSection<ELFT>::addSection(InputSectionBase *S) {
50 // FIXME: can this happen with ld -r ?
51 // error("Compiler should not have generated __cap_relocs section for " + toString(S));
52 assert(S->Name == "__cap_relocs");
53 assert(S->AreRelocsRela && "__cap_relocs should be RELA");
54 // make sure the section is no longer processed
55 S->markDead();
56
57 if ((S->getSize() % Entsize) != 0) {
58 error("__cap_relocs section size is not a multiple of " + Twine(Entsize) +
59 ": " + toString(S));
60 return;
61 }
62 size_t NumCapRelocs = S->getSize() / RelocSize;
63 if (NumCapRelocs * 2 != S->NumRelocations) {
64 error("expected " + Twine(NumCapRelocs * 2) + " relocations for " +
65 toString(S) + " but got " + Twine(S->NumRelocations));
66 return;
67 }
68 if (Config->VerboseCapRelocs)
69 message("Adding cap relocs from " + toString(S->File) + "\n");
70
71 LegacyInputs.push_back(S);
72}

89SymbolAndOffset
90SymbolAndOffset::fromSectionWithOffset(InputSectionBase *IS, int64_t Offset,
91 const SymbolAndOffset *Default) {
92 Symbol *FallbackResult = nullptr;
93 assert((int64_t)Offset >= 0);
94 int64_t FallbackOffset = Offset;
95 // For internal symbols we don't have a matching InputFile, just return
96 auto* File = IS->File;
97 if (!File) {
98 if (Default)
99 return *Default;
100 return {IS, Offset};
101 }
102 for (Symbol *B : IS->File->getSymbols()) {
103 if (auto *D = dyn_cast<Defined>(B)) {
104 if (D->Section != IS)
105 continue;
106 if ((int64_t)D->Value <= Offset &&
107 Offset <= (int64_t)D->Value + (int64_t)D->Size) {
108 // XXXAR: should we accept any symbol that encloses or only exact
109 // matches?
110 if ((int64_t)D->Value == Offset && (D->isFunc() || D->isObject()))
111 return {D, 0}; // perfect match
112 FallbackResult = D;
113 FallbackOffset = Offset - D->Value;
114 }
115 }
116 }
117 // When using the legacy __cap_relocs style (where clang emits __cap_relocs
118 // instead of R_CHERI_CAPABILITY) the local symbols might not exist so we
119 // may have fall back to the section.
120 if (!FallbackResult) {
121 // worst case we fall back to the section + offset
122 // Don't warn if the relocation is against an anonymous string constant
123 // since clang won't emit a symbol (and no size) for those
124 if (!IS->Name.startswith(".rodata.str"))
125 nonFatalWarning("Could not find a real symbol for __cap_reloc against " + IS->Name +
126 "+0x" + utohexstr(Offset) + " in " + toString(IS->File));
127 // Could not find a symbol -> return section+offset
128 assert(Offset == FallbackOffset);
129 if (Default)
130 return *Default;
131 return {IS, Offset};
132 }
133 return {FallbackResult, FallbackOffset};
134}

136SymbolAndOffset SymbolAndOffset::findRealSymbol() const {
137 // If we only have a section, see if we can resolve section+offset to a
138 // symbol (that may have been added later).
139 if (SymOrSec.is<InputSectionBase *>()) {
140 return SymbolAndOffset::fromSectionWithOffset(
141 SymOrSec.get<InputSectionBase *>(), Offset, this);
142 }
143 auto S = SymOrSec.get<Symbol *>();
144 if (!S->isSection())
145 return *this;
146 if (Defined *DefinedSym = dyn_cast<Defined>(S)) {
147 if (auto *IS = dyn_cast<InputSectionBase>(DefinedSym->Section)) {
148 return SymbolAndOffset::fromSectionWithOffset(IS, Offset, this);
149 }
150 }
151 return *this;
152}

160template <class ELFT>
161void CheriCapRelocsSection<ELFT>::processSection(InputSectionBase *S) {
162 // TODO: sort by offset (or is that always true?
163 const auto Rels = S->relas<ELFT>();
164 for (auto I = Rels.begin(), End = Rels.end(); I != End; ++I) {
165 const auto &LocationRel = *I;
166 ++I;
167 const auto &TargetRel = *I;
168 if ((LocationRel.r_offset % Entsize) != 0) {
169 error("corrupted __cap_relocs: expected Relocation offset to be a "
170 "multiple of " +
171 Twine(Entsize) + " but got " + Twine(LocationRel.r_offset));
172 return;
173 }
174 if (TargetRel.r_offset != LocationRel.r_offset + FieldSize) {
175 error("corrupted __cap_relocs: expected target relocation (" +
176 Twine(TargetRel.r_offset) +
177 " to directly follow location relocation (" +
178 Twine(LocationRel.r_offset) + ")");
179 return;
180 }
181 if (LocationRel.r_addend < 0) {
182 error("corrupted __cap_relocs: addend is less than zero in" +
183 toString(S) + ": " + Twine(LocationRel.r_addend));
184 return;
185 }
186 uint64_t CapRelocsOffset = LocationRel.r_offset;
187 assert(CapRelocsOffset + Entsize <= S->getSize());
188 if (Config->EMachine == EM_MIPS) {
189 if (LocationRel.getType(Config->IsMips64EL) != R_MIPS_64) {
190 error("Exptected a R_MIPS_64 relocation in __cap_relocs but got " +
191 toString(LocationRel.getType(Config->IsMips64EL)));
192 continue;
193 }
194 if (TargetRel.getType(Config->IsMips64EL) != R_MIPS_64) {
195 error("Exptected a R_MIPS_64 relocation in __cap_relocs but got " +
196 toString(TargetRel.getType(Config->IsMips64EL)));
197 continue;
198 }
199 } else {
200 if (LocationRel.getType(Config->IsMips64EL) != *Target->AbsPointerRel) {
201 error("Exptected an absolute pointer relocation in __cap_relocs "
202 "but got " + toString(LocationRel.getType(Config->IsMips64EL)));
203 continue;
204 }
205 if (TargetRel.getType(Config->IsMips64EL) != *Target->AbsPointerRel) {
206 error("Exptected an absolute pointer relocation in __cap_relocs "
207 "but got " + toString(TargetRel.getType(Config->IsMips64EL)));
208 continue;
209 }
210 }
211 Symbol *LocationSym =
212 &S->getFile<ELFT>()->getRelocTargetSym(LocationRel);
213 Symbol &TargetSym = S->getFile<ELFT>()->getRelocTargetSym(TargetRel);
214
215 if (LocationSym->File != S->File) {
216 error("Expected capability relocation to point to " + toString(S->File) +
217 " but got " + toString(LocationSym->File));
218 continue;
219 }
220 // errs() << "Adding cap reloc at " << toString(LocationSym) << " type "
221 // << Twine((int)LocationSym.Type) << " against "
222 // << toString(TargetSym) << "\n";
223 auto *RawInput = reinterpret_cast<const InMemoryCapRelocEntry<ELFT> *>(
224 S->data().begin() + CapRelocsOffset);
225 int64_t TargetCapabilityOffset = (int64_t)RawInput->offset;
226 assert(RawInput->size == 0 && "Clang should not have set size in __cap_relocs");
227 if (!isa<Defined>(LocationSym)) {
228 error("Unhandled symbol kind for cap_reloc: " +
229 Twine(LocationSym->kind()));
230 continue;
231 }
232
233 const SymbolAndOffset RelocLocation{LocationSym, LocationRel.r_addend};
234 const SymbolAndOffset RelocTarget{&TargetSym, TargetRel.r_addend};
235 SymbolAndOffset RealLocation = RelocLocation.findRealSymbol();
236 SymbolAndOffset RealTarget = RelocTarget.findRealSymbol();
237 if (Config->VerboseCapRelocs) {
238 message("Adding capability relocation at " +
239 RealLocation.verboseToString() + "\nagainst " +
240 RealTarget.verboseToString());
241 }
242
243 bool TargetNeedsDynReloc = false;
244 if (TargetSym.IsPreemptible) {
245 // Do we need this?
246 // TargetNeedsDynReloc = true;
247 }
248 switch (TargetSym.kind()) {
249 case Symbol::DefinedKind:
250 break;
251 case Symbol::SharedKind:
252 if (!hasDynamicLinker()) {
253 error("cannot create a capability relocation against a shared symbol"
254 " when linking statically");
255 continue;
256 }
257 TargetNeedsDynReloc = true;
258 break;
259 case Symbol::UndefinedKind:
260 // addCapReloc() will add an error if we are building an executable
261 // instead of a shlib
262 // TODO: we really should add a dynamic SIZE relocation as well
263 TargetNeedsDynReloc = true;
264 break;
265 default:
266 error("Unhandled symbol kind for cap_reloc target: " +
267 Twine(TargetSym.kind()));
268 continue;
269 }
270 assert(LocationSym->isSection());
271 auto *LocationDef = cast<Defined>(LocationSym);
272 auto *LocationSec = cast<InputSectionBase>(LocationDef->Section);
273 addCapReloc({LocationSec, (uint64_t)LocationRel.r_addend, false},
274 RealTarget, TargetNeedsDynReloc, TargetCapabilityOffset,
275 RealLocation.sym());
276 }

279template <class ELFT>
280void CheriCapRelocsSection<ELFT>::addCapReloc(CheriCapRelocLocation Loc,
281 const SymbolAndOffset &Target,
282 bool TargetNeedsDynReloc,
283 int64_t CapabilityOffset,
284 Symbol *SourceSymbol) {
285 if (Config->RelativeCapRelocsOnly) {
286 assert(!Loc.NeedsDynReloc);
287 if (TargetNeedsDynReloc) {
288 error("Cannot add __cap_reloc against target that needs a dynamic "
289 "relocation when --relative-cap-relocs is enabled: " +
290 Target.verboseToString());
291 return;
292 }
293 } else {
294 // Allow relocations in __cap_relocs section for legacy mode
295 Loc.NeedsDynReloc = Loc.NeedsDynReloc || Config->Pic || Config->Pie;
296 TargetNeedsDynReloc = TargetNeedsDynReloc || Config->Pic || Config->Pie;
297 }
298 uint64_t CurrentEntryOffset = RelocsMap.size() * RelocSize;
299
300 std::string SourceMsg =
301 SourceSymbol ? verboseToString(SourceSymbol) : Loc.toString();
302 if (Target.sym()->isUndefined() && !Target.sym()->isUndefWeak()) {
303 std::string Msg =
304 "cap_reloc against undefined symbol: " + toString(*Target.sym()) +
305 "\n>>> referenced by " + SourceMsg;
306 if (Config->UnresolvedSymbols == UnresolvedPolicy::ReportError)
307 error(Msg);
308 else
309 nonFatalWarning(Msg);
310 }
311
312 // assert(CapabilityOffset >= 0 && "Negative offsets not supported");
313 if (errorHandler().Verbose && CapabilityOffset < 0)
314 message("global capability offset " + Twine(CapabilityOffset) +
315 " is less than 0:\n>>> Location: " + Loc.toString() +
316 "\n>>> Target: " + Target.verboseToString());
317
318 bool CanWriteLoc = (Loc.Section->Flags & SHF_WRITE) || !Config->ZText;
319 if (!CanWriteLoc) {
320 readOnlyCapRelocsError(*Target.sym(), "\n>>> referenced by " + SourceMsg);
321 return;
322 }
323
324 if (!addEntry(Loc, {Target, CapabilityOffset, TargetNeedsDynReloc})) {
325 return; // Maybe happens with vtables?
326 }
327 if (Loc.NeedsDynReloc) {
328 // XXXAR: We don't need to create a symbol here since if we pass nullptr
329 // to the dynamic reloc it will add a relocation against the load address
330#ifdef DEBUG_CAP_RELOCS
331 llvm::sys::path::filename(Loc.Section->File->getName());
332 StringRef Filename = llvm::sys::path::filename(Loc.Section->File->getName());
333 std::string SymbolHackName = ("__caprelocs_hack_" + Loc.Section->Name + "_" +
334 Filename).str();
335 auto LocationSym = Symtab->find(SymbolHackName);
336 if (!LocationSym) {
337 Symtab->addDefined(Saver.save(SymbolHackName), STV_DEFAULT, STT_OBJECT,
338 Loc.Offset, Config->CapabilitySize, STB_GLOBAL,
339 Loc.Section, Loc.Section->File);
340 LocationSym = Symtab->find(SymbolHackName);
341 assert(LocationSym);
342 }
343
344 // Needed because local symbols cannot be used in dynamic relocations
345 // TODO: do this better
346 // message("Adding dyn reloc at " + toString(this) + "+0x" +
347 // utohexstr(CurrentEntryOffset))
348#endif
349 assert(CurrentEntryOffset < getSize());
350 // Add a dynamic relocation so that RTLD fills in the right base address
351 // We only have the offset relative to the load address...
352 // Ideally RTLD/crt_init_globals would just add the load address to all
353 // cap_relocs entries that have a RELATIVE flag set instead of requiring a
354 // full Elf_Rel/Elf_Rela
355 // The addend is zero here since it will be written in writeTo()
356 assert(!Config->IsRela);
357 Main->RelaDyn->addReloc({elf::Target->RelativeRel, this, CurrentEntryOffset,
358 true, nullptr, 0});
359 ContainsDynamicRelocations = true;
360 }
361 if (TargetNeedsDynReloc) {
362#ifdef DEBUG_CAP_RELOCS
363 message("Adding dyn reloc at " + toString(this) + "+0x" +
364 utohexstr(CurrentEntryOffset) + " against " +
365 Target.verboseToString());
366 message("Symbol preemptible:" + Twine(Target.Sym->IsPreemptible));
367#endif
368
369 bool RelativeToLoadAddress = false;
370 RelType RelocKind;
371 if (Target.sym()->IsPreemptible) {
372 RelocKind = *elf::Target->AbsPointerRel;
373 } else {
374 // If the target is not preemptible we can optimize this to a relative
375 // relocation agaist the image base
376 RelativeToLoadAddress = true;
377 RelocKind = elf::Target->RelativeRel;
378 }
379 // The addend is not used as the offset into the capability here, as we
380 // have the offset field in the __cap_relocs for that. The Addend
381 // will be zero unless we are targetting a string constant as these
382 // don't have a symbol and will be like .rodata.str+0x1234
383 int64_t Addend = Target.Offset;
384 // Capability target is the second field
385 assert((CurrentEntryOffset + FieldSize) < getSize());
386 Main->RelaDyn->addReloc({RelocKind, this, CurrentEntryOffset + FieldSize,
387 RelativeToLoadAddress, Target.sym(), Addend});
388 ContainsDynamicRelocations = true;
389 if (!RelativeToLoadAddress) {
390 // We also add a size relocation for the size field here
391 RelType SizeRel = *elf::Target->SizeRel;
392 // Capability size is the fourth field
393 assert((CurrentEntryOffset + 3*FieldSize) < getSize());
394 Main->RelaDyn->addReloc(SizeRel, this, CurrentEntryOffset + 3*FieldSize,
395 Target.sym());
396 }
397 }

400template<typename ELFT>
401static uint64_t getTargetSize(const CheriCapRelocLocation &Location,
402 const CheriCapReloc &Reloc, bool Strict) {
403 uint64_t TargetSize = Reloc.Target.sym()->getSize();
404 if (TargetSize > INT_MAX) {
405 error("Insanely large symbol size for " + Reloc.Target.verboseToString() +
406 "for cap_reloc at" + Location.toString());
407 return 0;
408 }
409 auto TargetSym = Reloc.Target.sym();
410 if (TargetSize == 0 && !TargetSym->IsPreemptible) {
411 StringRef Name = TargetSym->getName();
412 // Section end symbols like __preinit_array_end, etc. should actually be
413 // zero size symbol since they are just markers for the end of a section
414 // and not usable as a valid pointer
415 if (isSectionEndSymbol(Name) || isSectionStartSymbol(Name))
416 return TargetSize;
417
418 bool IsAbsoluteSym = TargetSym->getOutputSection() == nullptr;
419 // Symbols previously declared as weak can have size 0 (if they then resolve
420 // to NULL). For example __preinit_array_start, etc. are generated by the
421 // linker as ABS symbols with value 0.
422 // A symbol is linker-synthesized/linker script generated if File == nullptr
423 if (IsAbsoluteSym && TargetSym->File == nullptr)
424 return TargetSize;
425
426 if (TargetSym->isUndefWeak() && TargetSym->getVA(0) == 0)
427 // Weak symbol resolved to NULL -> zero size is fine
428 return 0;
429
430 // Absolute value provided by -defsym or assignment in .o file is fine
431 if (IsAbsoluteSym)
432 return TargetSize;
433
434 // Otherwise warn about missing sizes for symbols
435 bool WarnAboutUnknownSize = true;
436 // currently clang doesn't emit the necessary symbol information for local
437 // string constants such as: struct config_opt opts[] = { { ..., "foo" },
438 // { ..., "bar" } }; As this pattern is quite common don't warn if the
439 // target section is .rodata.str
440 if (Defined *DefinedSym =
441 dyn_cast<Defined>(TargetSym)) {
442 if (DefinedSym->isSection() &&
443 DefinedSym->Section->Name.startswith(".rodata.str")) {
444 WarnAboutUnknownSize = false;
445 }
446 }
447 // TODO: are there any other cases that can be ignored?
448
449 if (WarnAboutUnknownSize || errorHandler().Verbose) {
450 std::string Msg = "could not determine size of cap reloc against " +
451 Reloc.Target.verboseToString() +
452 "\n>>> referenced by " + Location.toString();
453 if (Strict)
454 warn(Msg);
455 else
456 nonFatalWarning(Msg);
457 }
458 if (OutputSection *OS = TargetSym->getOutputSection()) {
459 // For negative offsets use 0 instead (we wan the range of the full symbol in that case)
460 int64_t Offset = std::max((int64_t)0, Reloc.Target.Offset);
461 uint64_t TargetVA = TargetSym->getVA(Offset);
462 assert(TargetVA >= OS->Addr);
463 uint64_t OffsetInOS = TargetVA - OS->Addr;
464 // Use less-or-equal here to account for __end_foo symbols which point 1 past the section
465 assert(OffsetInOS <= OS->Size);
466 TargetSize = OS->Size - OffsetInOS;
467#ifdef DEBUG_CAP_RELOCS
468 if (Config->VerboseCapRelocs)
469 errs() << " OS OFFSET 0x" << utohexstr(OS->Addr) << "SYM OFFSET 0x"
470 << utohexstr(OffsetInOS) << " SECLEN 0x" << utohexstr(OS->Size)
471 << " -> target size 0x" << utohexstr(TargetSize) << "\n";
472#endif
473 } else {
474 warn("Could not find size for symbol " + Reloc.Target.verboseToString() +
475 " and could not determine section size. Using 0.");
476 // TargetSize = std::numeric_limits<uint64_t>::max();
477 return 0;
478 }
479 }
480 return TargetSize;
481}

483template <class ELFT>
484struct CaptablePermissions {
485 static const uint64_t Function =
486 UINT64_C(1) << ((sizeof(typename ELFT::uint) * 8) - 1);
487 static const uint64_t ReadOnly =
488 UINT64_C(1) << ((sizeof(typename ELFT::uint) * 8) - 2);
489};

491template <class ELFT> void CheriCapRelocsSection<ELFT>::writeTo(uint8_t *Buf) {
492 static_assert(RelocSize == sizeof(InMemoryCapRelocEntry<ELFT>),
493 "cap relocs size mismatch");
494 uint64_t Offset = 0;
495 for (const auto &I : RelocsMap) {
496 const CheriCapRelocLocation &Location = I.first;
497 const CheriCapReloc &Reloc = I.second;
498 assert(Location.Offset <= Location.Section->getSize());
499 // We write the virtual address of the location in in both static and the
500 // shared library case:
501 // In the static case we can compute the final virtual address and write it
502 // In the dynamic case we write the virtual address relative to the load
503 // address and the runtime linker will add the load address to that
504 uint64_t OutSecOffset = Location.Section->getOffset(Location.Offset);
505 uint64_t LocationVA =
506 Location.Section->getOutputSection()->Addr + OutSecOffset;
507
508 // For the target the virtual address the addend is always zero so
509 // if we need a dynamic reloc we write zero
510 // TODO: would it be more efficient for local symbols to write the DSO VA
511 // and add a relocation against the load address?
512 // Also this would make llvm-objdump --cap-relocs more useful because it
513 // would actually display the symbol that the relocation is against
514 uint64_t TargetVA = Reloc.Target.sym()->getVA(Reloc.Target.Offset);
515 bool PreemptibleDynReloc =
516 Reloc.NeedsDynReloc && Reloc.Target.sym()->IsPreemptible;
517 uint64_t TargetSize = 0;
518 if (PreemptibleDynReloc) {
519 // If we have a relocation against a preemptible symbol (even in the
520 // current DSO) we can't compute the virtual address here so we only write
521 // the addend
522 if (Reloc.Target.Offset != 0)
523 error("Dyn Reloc Target offset was nonzero: " +
524 Twine(Reloc.Target.Offset) + " - " +
525 Reloc.Target.verboseToString());
526 TargetVA = Reloc.Target.Offset;
527 } else {
528 // For non-preemptible symbols we can write the target size:
529 TargetSize = getTargetSize<ELFT>(Location, Reloc,
530 /*Strict=*/!containsLegacyCapRelocs());
531 }
532 uint64_t TargetOffset = Reloc.CapabilityOffset;
533 uint64_t Permissions = 0;
534 // Fow now Function implies ReadOnly so don't add the flag
535 if (Reloc.Target.sym()->isFunc()) {
536 Permissions |= CaptablePermissions<ELFT>::Function;
537 } else if (auto OS = Reloc.Target.sym()->getOutputSection()) {
538 assert(!Reloc.Target.sym()->isTls());
539 assert((OS->Flags & SHF_TLS) == 0);
540 // if ((OS->getPhdrFlags() & PF_W) == 0) {
541 if (((OS->Flags & SHF_WRITE) == 0) || isRelroSection(OS)) {
542 Permissions |= CaptablePermissions<ELFT>::ReadOnly;
543 } else if (OS->Flags & SHF_EXECINSTR) {
544 warn("Non-function __cap_reloc against symbol in section with "
545 "SHF_EXECINSTR (" + toString(OS->Name) + ") for symbol " +
546 Reloc.Target.verboseToString());
547 }
548 }
549
550 // TODO: should we warn about symbols that are out-of-bounds?
551 // mandoc seems to do it so I guess we need it
552 // if (TargetOffset < 0 || TargetOffset > TargetSize) warn(...);
553
554 InMemoryCapRelocEntry<ELFT> Entry(LocationVA, TargetVA, TargetOffset,
555 TargetSize, Permissions);
556 memcpy(Buf + Offset, &Entry, sizeof(Entry));
557 // if (errorHandler().Verbose) {
558 // errs() << "Added capability reloc: loc=" << utohexstr(LocationVA)
559 // << ", object=" << utohexstr(TargetVA)
560 // << ", offset=" << utohexstr(TargetOffset)
561 // << ", size=" << utohexstr(TargetSize)
562 // << ", permissions=" << utohexstr(Permissions) << "\n";
563 // }
564 Offset += RelocSize;
565 }
566
567 // FIXME: this totally breaks dynamic relocs!!! need to do in finalize()
568
569 // Sort the cap_relocs by target address for better cache and TLB locality
570 // It also makes it much easier to read the llvm-objdump -C output since it
571 // is sorted in a sensible order
572 // However, we can't do this if we added any dynamic relocations since it
573 // will mean the dynamic relocation offset refers to a different location
574 // FIXME: do the sorting in finalizeSection instead
575 if (Config->SortCapRelocs && !ContainsDynamicRelocations)
576 std::stable_sort(reinterpret_cast<InMemoryCapRelocEntry<ELFT> *>(Buf),
577 reinterpret_cast<InMemoryCapRelocEntry<ELFT> *>(Buf + Offset),
578 [](const InMemoryCapRelocEntry<ELFT> &a,
579 const InMemoryCapRelocEntry<ELFT> &b) {
580 return a.capability_location < b.capability_location;
581 });
582 assert(Offset == getSize() && "Not all data written?");
583}

593void CheriCapTableSection::writeTo(uint8_t* Buf) {
594 // Should be filled with all zeros and crt_init_globals fills it in
595 // TODO: fill in the raw bits and use csettag
596
597 auto Write = [&](size_t I, const Symbol *S, int64_t A) {
598 uint64_t VA = A;
599 if (S)
600 VA = S->getVA(A);
601
602 if (Config->Is64)
603 write64(Buf + I * 8, VA);
604 else
605 write32(Buf + I * 4, VA);
606 };
607 // If TLS entry has a corresponding dynamic relocations, leave it
608 // initialized by zero. Write down adjusted TLS symbol's values otherwise.
609 // To calculate the adjustments use offsets for thread-local storage.
610 for (auto &it : DynTlsEntries.Map) {
611 if (it.first == nullptr && !Config->Pic)
612 Write(it.second.Index.getValue(), nullptr, 1);
613 else if (it.first && !it.first->IsPreemptible) {
614 // If we are emitting PIC code with relocations we mustn't write
615 // anything to the GOT here. When using Elf_Rel relocations the value
616 // one will be treated as an addend and will cause crashes at runtime
617 if (!Config->Pic)
618 Write(it.second.Index.getValue(), nullptr, 1);
619 Write(it.second.Index.getValue() + 1, it.first, 0);
620 }
621 }
622
623 for (auto &it : TlsEntries.Map)
624 Write(it.second.Index.getValue(), it.first, 0);
625}

629 default:
630 llvm_unreachable("Invalid kind");
631 case ELF32LEKind:
632 return IS->getEnclosingFunction<ELF32LE>(SymOffset);
633 case ELF32BEKind:
634 return IS->getEnclosingFunction<ELF32BE>(SymOffset);
635 case ELF64LEKind:

637 case ELF64BEKind:

639 }

644 uint64_t Offset) {

665void CheriCapTableSection::addEntry(Symbol &Sym, RelExpr Expr,
666 InputSectionBase *IS, uint64_t Offset) {
667 // FIXME: can this be called from multiple threads?
668 CapTableIndex Idx;
669 Idx.NeedsSmallImm = false;
670 Idx.UsedInCallExpr = false;
671 Idx.FirstUse = SymbolAndOffset::fromSectionWithOffset(IS, Offset);
672 assert(!Idx.FirstUse->SymOrSec.isNull());
673 switch (Expr) {
674 case R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE:
675 case R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE:
676 Idx.NeedsSmallImm = true;
677 break;
678 default:
679 break;
680 }
681 // If the symbol is only ever referenced by the captable call relocations we
682 // can emit a capability call relocation instead of a normal capability
683 // relocation. This indicates to the runtime linker that the capability is
684 // not used as a function pointer and therefore does not need a unique
685 // address (plt stub) across all DSOs.
686 switch (Expr) {
687 case R_CHERI_CAPABILITY_TABLE_INDEX_CALL:
688 case R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE:
689 if (!Sym.isFunc() && !Sym.isUndefWeak()) {
690 CheriCapRelocLocation Loc{IS, Offset, false};
691 std::string Msg = "call relocation against non-function symbol " + verboseToString(&Sym, 0) +
692 "\n>>> referenced by " + Loc.toString();
693 if (Sym.isUndefined() && Config->AllowShlibUndefined) {
694 // Don't fail the build for shared libraries unless
695 nonFatalWarning(Msg);
696 } else {
697 warn(Msg);
698 }
699 }
700 Idx.UsedInCallExpr = true;
701 break;
702 default:
703 break;
704 }
705 CaptableMap &Entries = getCaptableMapForFileAndOffset(IS, Offset);
706 if (Config->ZCapTableDebug) {
707 // Add a local helper symbol to improve disassembly:
708 StringRef HelperSymName = Saver.save("$captable_load_" + (Sym.getName().empty() ? "$anonymous_symbol" : Sym.getName()));
709 addSyntheticLocal(HelperSymName, STT_NOTYPE, Offset, 0, *IS);
710 }
711
712 auto it = Entries.Map.insert(std::make_pair(&Sym, Idx));
713 if (!it.second) {
714 // If it is references by a small immediate relocation we need to update
715 // the small immediate flag
716 if (Idx.NeedsSmallImm)
717 it.first->second.NeedsSmallImm = true;
718 // If one of the uses is not a call expression, we need to emit a unique
719 // function pointer and reuse that for the call expression, too.
720 // TODO: should we emit two relocations instead?
721 if (!Idx.UsedInCallExpr)
722 it.first->second.UsedInCallExpr = false;
723 }
724#if defined(DEBUG_CAP_TABLE)
725 std::string DbgContext;
726 if (Config->CapTableScope == CapTableScopePolicy::File) {
727 DbgContext = " for file '" + toString(IS->File) + "'";
728 } else if (Config->CapTableScope == CapTableScopePolicy::Function) {
729 DbgContext = " for function '" + toString(*findMatchingFunction(IS, Offset)) + "'";
730 }
731 llvm::errs() << "Added symbol " << toString(Sym) << " to .captable"
732 << DbgContext << ". Total count " << Entries.size() << "\n";
733#endif
734}

750 assert(ValuesAssigned && "getIndex called before index assignment");

790template <class ELFT>
791uint64_t CheriCapTableSection::assignIndices(uint64_t StartIndex,
792 CaptableMap &Entries,
793 const Twine &SymContext) {
794 // Usually StartIndex will be zero (one global captable) but if we are
795 // compiling with per-file/per-function
796 uint64_t SmallEntryCount = 0;
797 assert(Entries.FirstIndex == std::numeric_limits<uint64_t>::max() &&
798 "Should not be initialized yet!");
799 Entries.FirstIndex = StartIndex;
800 for (auto &it : Entries.Map) {
801 // TODO: looping twice is inefficient, we could keep track of the number of
802 // small entries during insertion
803 if (it.second.NeedsSmallImm) {
804 SmallEntryCount++;
805 }
806 }
807
808 if (Config->EMachine == EM_MIPS) {
809 unsigned MaxSmallEntries = (1 << 19) / Config->CapabilitySize;
810 if (SmallEntryCount > MaxSmallEntries) {
811 // Use warn here since the calculation may be wrong if the 11 bit clc is
812 // used. We will error when writing the relocation values later anyway
813 // so this will help find the error
814 warn("added " + Twine(SmallEntryCount) + " entries to .captable but "
815 "current maximum is " + Twine(MaxSmallEntries) + "; try recompiling "
816 "non-performance critical source files with -mxcaptable");
817 }
818 if (errorHandler().Verbose) {
819 message("Total " + Twine(Entries.size()) + " .captable entries: " +
820 Twine(SmallEntryCount) + " use a small immediate and " +
821 Twine(Entries.size() - SmallEntryCount) + " use -mxcaptable. ");
822 }
823 }
824
825 // Only add the @CAPTABLE symbols when running the LLD unit tests
826 // errorHandler().ExitEarly is set to false if LLD_IN_TEST=1 so just reuse
827 // that instead of calling getenv on every iteration
828 const bool ShouldAddAtCaptableSymbols = !errorHandler().ExitEarly;
829 uint32_t AssignedSmallIndexes = 0;
830 uint32_t AssignedLargeIndexes = 0;
831 for (auto &it : Entries.Map) {
832 CapTableIndex &CTI = it.second;
833 if (CTI.NeedsSmallImm) {
834 assert(AssignedSmallIndexes < SmallEntryCount);
835 CTI.Index = StartIndex + AssignedSmallIndexes;
836 AssignedSmallIndexes++;
837 } else {
838 CTI.Index = StartIndex + SmallEntryCount + AssignedLargeIndexes;
839 AssignedLargeIndexes++;
840 }
841
842 uint32_t Index = *CTI.Index;
843 assert(Index >= StartIndex && Index < StartIndex + Entries.size());
844 Symbol *TargetSym = it.first;
845
846 StringRef Name = TargetSym->getName();
847 // Avoid duplicate symbol name errors for unnamed string constants:
848 StringRef RefName;
849 // For now always append .INDEX to local symbols @CAPTABLE names since they
850 // might not be unique. If there is a global with the same name we always
851 // want the global to have the plain @CAPTABLE name
852 if (Name.empty() /* || Name.startswith(".L") */ || TargetSym->isLocal())
853 RefName = Saver.save(Name + "@CAPTABLE" + SymContext + "." + Twine(Index));
854 else
855 RefName = Saver.save(Name + "@CAPTABLE" + SymContext);
856 // XXXAR: This should no longer be necessary now that I am using addSyntheticLocal?
857#if 0
858 if (Symtab->find(RefName)) {
859 std::string NewRefName =
860 (Name + "@CAPTABLE" + SymContext + "." + Twine(Index)).str();
861 // XXXAR: for some reason we sometimes create more than one cap table entry
862 // for a given global name, for now just rename the symbol
863 // assert(TargetSym->isLocal());
864 if (!TargetSym->isLocal()) {
865 error("Found duplicate global captable ref name " + RefName +
866 " but referenced symbol was not local\n>>> " +
867 verboseToString(TargetSym));
868 } else {
869 // TODO: make this a warning
870 message("Found duplicate captable name " + RefName +
871 "\n>>> Replacing with " + NewRefName);
872 }
873 RefName = std::move(NewRefName);
874 assert(!Symtab->find(RefName) && "RefName should be unique");
875 }
876#endif
877 uint64_t Off = Index * Config->CapabilitySize;
878 if (ShouldAddAtCaptableSymbols) {
879 addSyntheticLocal(RefName, STT_OBJECT, Off, Config->CapabilitySize, *this);
880 }
881 // If the symbol is used as a function pointer the runtime linker has to
882 // ensure that all pointers to that function compare equal. This is done
883 // by ensuring that they all point to the same PLT stub.
884 // If it is not used as a function pointer we can use a capability call
885 // relocation instead which allows the runtime linker to create non-unique
886 // plt stubs.
887 RelType ElfCapabilityReloc = it.second.UsedInCallExpr
888 ? *Target->CheriCapCallRel
889 : *Target->CheriCapRel;
890 // All capability call relocations should end up in the pltrel section
891 // rather than the normal relocation section to make processing of PLT
892 // relocations in RTLD more efficient.
893 RelocationBaseSection *DynRelSec =
894 it.second.UsedInCallExpr ? In.RelaPlt : Main->RelaDyn;
895 addCapabilityRelocation<ELFT>(
896 TargetSym, ElfCapabilityReloc, In.CheriCapTable, Off,
897 R_CHERI_CAPABILITY, 0, it.second.UsedInCallExpr,
898 [&]() {
899 return ("\n>>> referenced by " + RefName + "\n>>> first used in " +
900 it.second.FirstUse->verboseToString())
901 .str();
902 },
903 DynRelSec);
904 }
905 assert(AssignedSmallIndexes + AssignedLargeIndexes == Entries.size());
906 return AssignedSmallIndexes + AssignedLargeIndexes;
907}

909template <class ELFT>
910void CheriCapTableSection::assignValuesAndAddCapTableSymbols() {
911 // First assign the global indices (which will usually be the only ones)
912 uint64_t AssignedEntries = assignIndices<ELFT>(0, GlobalEntries, "");
913 if (LLVM_UNLIKELY(Config->CapTableScope != CapTableScopePolicy::All)) {
914 assert(AssignedEntries == 0 && "Should not have any global entries in"
915 " per-file/per-function captable mode");
916 for (auto &it : PerFileEntries) {
917 std::string FullContext = toString(it.first);
918 auto LastSlash = StringRef(FullContext).find_last_of("/\\") + 1;
919 StringRef Context = StringRef(FullContext).substr(LastSlash);
920 AssignedEntries += assignIndices<ELFT>(AssignedEntries, it.second,
921 "@" + Context);
922 }
923 for (auto &it : PerFunctionEntries)
924 AssignedEntries += assignIndices<ELFT>(AssignedEntries, it.second,
925 "@" + toString(*it.first));
926 }
927 assert(AssignedEntries == nonTlsEntryCount());
928
929 uint32_t AssignedTlsIndexes = 0;
930 uint32_t TlsBaseIndex =
931 AssignedEntries * (Config->CapabilitySize / Config->Wordsize);
932
933 // TODO: support TLS for per-function captable
934 if (Config->CapTableScope != CapTableScopePolicy::All &&
935 (!DynTlsEntries.empty() || !TlsEntries.empty())) {
936 error("TLS is not supported yet with per-file or per-function captable");
937 return;
938 }
939
940 for (auto &it : DynTlsEntries.Map) {
941 CapTableIndex &CTI = it.second;
942 assert(!CTI.NeedsSmallImm);
943 CTI.Index = TlsBaseIndex + AssignedTlsIndexes;
944 AssignedTlsIndexes += 2;
945 Symbol *S = it.first;
946 uint64_t Offset = CTI.Index.getValue() * Config->Wordsize;
947 if (S == nullptr) {
948 if (!Config->Pic)
949 continue;
950 Main->RelaDyn->addReloc(Target->TlsModuleIndexRel, this, Offset, S);
951 } else {
952 // When building a shared library we still need a dynamic relocation
953 // for the module index. Therefore only checking for
954 // S->IsPreemptible is not sufficient (this happens e.g. for
955 // thread-locals that have been marked as local through a linker script)
956 if (!S->IsPreemptible && !Config->Pic)
957 continue;
958 Main->RelaDyn->addReloc(Target->TlsModuleIndexRel, this, Offset, S);
959 // However, we can skip writing the TLS offset reloc for non-preemptible
960 // symbols since it is known even in shared libraries
961 if (!S->IsPreemptible)
962 continue;
963 Offset += Config->Wordsize;
964 Main->RelaDyn->addReloc(Target->TlsOffsetRel, this, Offset, S);
965 }
966 }
967
968 for (auto &it : TlsEntries.Map) {
969 CapTableIndex &CTI = it.second;
970 assert(!CTI.NeedsSmallImm);
971 CTI.Index = TlsBaseIndex + AssignedTlsIndexes++;
972 Symbol *S = it.first;
973 uint64_t Offset = CTI.Index.getValue() * Config->Wordsize;
974 if (S->IsPreemptible)
975 Main->RelaDyn->addReloc(Target->TlsGotRel, this, Offset, S);
976 }
977
978 ValuesAssigned = true;
979}

1018
1019void CheriCapTableMappingSection::writeTo(uint8_t *Buf) {
1020 assert(Config->CapTableScope != CapTableScopePolicy::All);
1021 if (!In.CheriCapTable)
1022 return;
1023 if (!In.SymTab) {
1024 error("Cannot write " + this->Name + " without .symtab section!");
1025 return;
1026 }
1027
1028 // Write the mapping from function vaddr -> captable subset for RTLD
1029 std::vector<CaptableMappingEntry> Entries;
1030 // Note: Symtab->getSymbols() only returns the symbols in .dynsym. We need
1031 // to use In.sym()tab instead since we also want to add all local functions!
1032 for (const SymbolTableEntry &STE : In.SymTab->getSymbols()) {
1033 Symbol* Sym = STE.Sym;
1034 if (!Sym->isDefined() || !Sym->isFunc())
1035 continue;
1036 const CheriCapTableSection::CaptableMap *CapTableMap = nullptr;
1037 if (Config->CapTableScope == CapTableScopePolicy::Function) {
1038 auto it = In.CheriCapTable->PerFunctionEntries.find(Sym);
1039 if (it != In.CheriCapTable->PerFunctionEntries.end())
1040 CapTableMap = &it->second;
1041 } else if (Config->CapTableScope == CapTableScopePolicy::File) {
1042 auto it = In.CheriCapTable->PerFileEntries.find(Sym->File);
1043 if (it != In.CheriCapTable->PerFileEntries.end())
1044 CapTableMap = &it->second;
1045 } else {
1046 llvm_unreachable("Invalid mode!");
1047 }
1048 CaptableMappingEntry Entry;
1049 Entry.FuncStart = Sym->getVA(0);
1050 Entry.FuncEnd = Entry.FuncStart + Sym->getSize();
1051 if (CapTableMap) {
1052 assert(CapTableMap->FirstIndex != std::numeric_limits<uint64_t>::max());
1053 Entry.CapTableOffset = CapTableMap->FirstIndex * Config->CapabilitySize;
1054 Entry.SubTableSize = CapTableMap->size() * Config->CapabilitySize;
1055 } else {
1056 // TODO: don't write an entry for functions that don't use the captable
1057 Entry.CapTableOffset = 0;
1058 Entry.SubTableSize = 0;
1059 }
1060 Entries.push_back(Entry);
1061 }
1062 // Sort all the entries so that RTLD can do a binary search to find the
1063 // correct entry instead of having to scan all of them.
1064 // Do this before swapping to target endianess to simplify the comparisons.
1065 llvm::sort(Entries, [](const CaptableMappingEntry &E1,
1066 const CaptableMappingEntry &E2) {
1067 if (E1.FuncStart == E2.FuncStart)
1068 return E1.FuncEnd < E2.FuncEnd;
1069 return E1.FuncStart < E2.FuncStart;
1070 });
1071 // Byte-swap all the values so that we can memcpy the sorted buffer
1072 for (CaptableMappingEntry &E : Entries) {
1073 E.FuncStart = support::endian::byte_swap(E.FuncStart, Config->Endianness);
1074 E.FuncEnd = support::endian::byte_swap(E.FuncEnd, Config->Endianness);
1075 E.CapTableOffset =
1076 support::endian::byte_swap(E.CapTableOffset, Config->Endianness);
1077 E.SubTableSize =
1078 support::endian::byte_swap(E.SubTableSize, Config->Endianness);
1079 }
1080 assert(Entries.size() * sizeof(CaptableMappingEntry) == getSize());
1081 memcpy(Buf, Entries.data(), Entries.size() * sizeof(CaptableMappingEntry));
1082}

1084template <typename ELFT>
1085void addCapabilityRelocation(Symbol *Sym, RelType Type, InputSectionBase *Sec,
1086 uint64_t Offset, RelExpr Expr, int64_t Addend,
1087 bool IsCallExpr,
1088 llvm::function_ref<std::string()> ReferencedBy,
1089 RelocationBaseSection *DynRelSec) {
1090 // Emit either the legacy __cap_relocs section or a R_CHERI_CAPABILITY reloc
1091 // For local symbols we can also emit the untagged capability bits and
1092 // instruct csu/rtld to run CBuildCap
1093 assert(Config->ProcessCapRelocs);
1094 CapRelocsMode CapRelocMode = Sym->IsPreemptible
1095 ? Config->PreemptibleCapRelocsMode
1096 : Config->LocalCapRelocsMode;
1097 bool NeedTrampoline = false;
1098 // In the PLT ABI (and fndesc?) we have to use an elf relocation for function
1099 // pointers to ensure that the runtime linker adds the required trampolines
1100 // that sets $cgp:
1101 if (!IsCallExpr && Config->EMachine == llvm::ELF::EM_MIPS && Sym->isFunc()) {
1102 if (!lld::elf::hasDynamicLinker()) {
1103 // In static binaries we do not need PLT stubs for function pointers since
1104 // all functions share the same $cgp
1105 // TODO: this is no longer true if we were to support dlopen() in static
1106 // binaries
1107 if (Config->VerboseCapRelocs)
1108 message("Do not need function pointer trampoline for " +
1109 toString(*Sym) + " in static binary");
1110 NeedTrampoline = false;
1111 } else if (auto ABI = InX<ELFT>::MipsAbiFlags->getCheriAbiVariant()) {
1112 if (*ABI == llvm::ELF::DF_MIPS_CHERI_ABI_PLT ||
1113 *ABI == llvm::ELF::DF_MIPS_CHERI_ABI_FNDESC)
1114 NeedTrampoline = true;
1115 }
1116 }
1117 if (NeedTrampoline) {
1118 CapRelocMode = CapRelocsMode::ElfReloc;
1119 assert(CapRelocMode == Config->PreemptibleCapRelocsMode);
1120 if (Config->VerboseCapRelocs)
1121 message("Using trampoline for function pointer against " +
1122 verboseToString(Sym));
1123 }
1124
1125 // local cap relocs don't need a Elf relocation with a full symbol lookup:
1126 if (CapRelocMode == CapRelocsMode::ElfReloc) {
1127 assert((Sym->IsPreemptible || NeedTrampoline) &&
1128 "ELF relocs should not be used for non-preemptible symbols");
1129 assert((!Sym->isLocal() || NeedTrampoline) &&
1130 "ELF relocs should not be used for local symbols");
1131 if (Config->EMachine == llvm::ELF::EM_MIPS && Config->BuildingFreeBSDRtld) {
1132 error("relocation " + toString(Type) + " against " +
1133 verboseToString(Sym) +
1134 " cannot be using when building FreeBSD RTLD" + ReferencedBy());
1135 return;
1136 }
1137 if (!lld::elf::hasDynamicLinker()) {
1138 error("attempting to emit a R_CAPABILITY relocation against " +
1139 (Sym->getName().empty() ? "local symbol"
1140 : "symbol " + toString(*Sym)) +
1141 " in binary without a dynamic linker; try removing -Wl,-" +
1142 (Sym->IsPreemptible ? "preemptible" : "local") + "-caprelocs=elf" +
1143 ReferencedBy());
1144 return;
1145 }
1146 assert(Config->HasDynSymTab && "Should have been checked in Driver.cpp");
1147 // We don't use a R_MIPS_CHERI_CAPABILITY relocation for the input but
1148 // instead need to use an absolute pointer size relocation to write
1149 // the offset addend
1150 if (!DynRelSec)
1151 DynRelSec = Main->RelaDyn;
1152 // in the case that -local-caprelocs=elf is passed we need to ensure that
1153 // the target symbol is included in the dynamic symbol table
1154 if (!Main->DynSymTab) {
1155 error("R_CHERI_CAPABILITY relocations need a dynamic symbol table");
1156 return;
1157 }
1158 if (!Sym->includeInDynsym()) {
1159 if (!NeedTrampoline) {
1160 error("added a R_CHERI_CAPABILITY relocation but symbol not included "
1161 "in dynamic symbol: " +
1162 verboseToString(Sym));
1163 return;
1164 }
1165 // Hack: Add a new global symbol with a unique name so that we can use
1166 // a dynamic relocation against it.
1167 // TODO: should it be possible to add STB_LOCAL symbols to .dynsymtab?
1168
1169
1170
1171 std::string UniqueName = ("__cheri_fnptr_" + Sym->getName()).str();
