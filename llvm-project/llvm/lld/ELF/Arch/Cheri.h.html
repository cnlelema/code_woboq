61struct CheriCapReloc {
62 // We can't use a plain Symbol* here as capabilities to string constants
63 // will be e.g. `.rodata.str + 0x90` -> need to store offset as well
64 SymbolAndOffset Target;
65 int64_t CapabilityOffset;
66 bool NeedsDynReloc;
67 bool operator==(const CheriCapReloc &Other) const {
68 return Target.SymOrSec == Other.Target.SymOrSec &&
69 Target.Offset == Other.Target.Offset &&
70 CapabilityOffset == Other.CapabilityOffset &&
71 NeedsDynReloc == Other.NeedsDynReloc;
72 }
73};

90private:
91 void processSection(InputSectionBase *S);
92 bool addEntry(CheriCapRelocLocation Loc, CheriCapReloc Relocation) {
93 auto it = RelocsMap.insert(std::make_pair(Loc, Relocation));
94 // assert(it.first->second == Relocation);
95 if (!(it.first->second == Relocation)) {
96 error("Newly inserted relocation at " + Loc.toString() +
97 " does not match existing one:\n> Existing: " +
98 it.first->second.Target.verboseToString() +
99 ", cap offset=" + Twine(it.first->second.CapabilityOffset) +
100 ", dyn=" + Twine(it.first->second.NeedsDynReloc) +
101 "\n> New: " + Relocation.Target.verboseToString() +
102 ", cap offset=" + Twine(Relocation.CapabilityOffset) +
103 ", dyn=" + Twine(Relocation.NeedsDynReloc));
104 }
105 return it.second;
106 }

118// General cap table structure (for CapSize = 2*WordSize):
119//
120// +-------------------------------+
121// | Small Index Cap 1 |
122// +-------------------------------+
123// | : |
124// +-------------------------------+
125// | Small Index Cap m |
126// +-------------------------------+
127// | Large Index Cap 1 |
128// +-------------------------------+
129// | : |
130// +-------------------------------+
131// | Large Index Cap n |
132// +---------------+---------------+
133// | Dyn TLS Mod 1 | Dyn TLS Off 2 |
134// +---------------+---------------+
135// | : |
136// +---------------+---------------+
137// | Dyn TLS Mod p | Dyn TLS Off p |
138// +---------------+---------------+
139// | IE TPREL 1 | IE TPREL 2 |
140// +---------------+---------------+
141// | : |
142// +---------------+---------------+
143// | IE TPREL q-1 | IE TPREL q |
144// +-------------------------------+
145//
146// If we are compiling with per function/per file tables it starts as follows:
147// +---------------------------------------+
148// | Small Index Cap 1 for File/Function 1 |
149// +---------------------------------------+
150// | : |
151// +---------------------------------------+
152// | Small Index Cap m for File/Function 1 |
153// +---------------------------------------+
154// | Large Index Cap 1 for File/Function 1 |
155// +---------------------------------------+
156// | : |
157// +---------------------------------------+
158// | Large Index Cap n for File/Function 1 |
159// +---------------------------------------+
160// | Small Index Cap 1 for File/Function 2 |
161// +---------------------------------------+
162// | : |
163// +---------------------------------------+
164// | Small Index Cap m for File/Function 2 |
165// +---------------------------------------+
166// | Large Index Cap 1 for File/Function 2 |
167// +---------------------------------------+
168// | : |
169// +---------------------------------------+
170// | Large Index Cap n for File/Function 2 |
171// +---------------------------------------+
172// | : |
173// | : |
174// +---------------------------------------+
175// | Small Index Cap 1 for File/Function N |
176// +---------------------------------------+
177// | : |
178// +---------------------------------------+
179// | Small Index Cap m for File/Function N |
180// +---------------------------------------+
181// | Large Index Cap 1 for File/Function N |
182// +---------------------------------------+
183// | : |
184// +---------------------------------------+
185// | Large Index Cap n for File/Function N |
186// +---------------------------------------+
187// TODO: TLS caps also need to be per file/function

210 size_t getSize() const override {
211 size_t NonTlsEntries = nonTlsEntryCount();
212 if (NonTlsEntries > 0 || !TlsEntries.empty() || !DynTlsEntries.empty()) {

253 size_t nonTlsEntryCount() const {
254 size_t TotalCount = GlobalEntries.size();
255 if (LLVM_LIKELY(Config->CapTableScope == CapTableScopePolicy::All)) {

300template <typename ELFT, typename CallBack>
301static void foreachGlobalSizesSymbol(InputSection *IS, CallBack &&CB) {
302 assert(IS->Name == ".global_sizes");
303 for (Symbol *B : IS->File->getSymbols()) {
304 if (auto *D = dyn_cast<Defined>(B)) {
305 if (D->Section != IS)
306 continue;
307 // skip the initial .global_sizes symbol (exists e.g. in
308 // openpam_static_modules.o)
309 if (D->isSection() && D->isLocal() && D->getName().empty())
310 continue;
311 StringRef Name = D->getName();
312 if (!Name.startswith(".size.")) {
313 error(".global_sizes symbol name is invalid: " + verboseToString(D));
314 continue;
315 }
316 StringRef RealSymName = Name.drop_front(strlen(".size."));
317 Symbol *Target = Symtab->find(RealSymName);
318 CB(RealSymName, Target, D->Value);
319 }
320 }

323inline bool isSectionEndSymbol(StringRef Name) {
324 // Section end symbols like __preinit_array_end, etc. should actually be
325 // zero size symbol since they are just markers for the end of a section
326 // and not usable as a valid pointer
327 return Name.startswith("__stop_") ||
328 (Name.startswith("__") && Name.endswith("_end")) || Name == "end" ||
329 Name == "_end" || Name == "etext" || Name == "_etext" ||
330 Name == "edata" || Name == "_edata";
331}
332
333inline bool isSectionStartSymbol(StringRef Name) {
334 // Section end symbols like __preinit_array_start, might end up pointing to
335 // .text (see commments in Writer.cpp) if they are emtpy to avoid relocation
336 // overflow. In that case returning a size of 0 is fine too.
337 return Name.startswith("__start_") ||
338 (Name.startswith("__") && Name.endswith("_start")) ||
339 Name == "_DYNAMIC";
340}
