47template <class ELFT> MIPS<ELFT>::MIPS() {
48 GotPltHeaderEntriesNum = 2;
49 DefaultMaxPageSize = 65536;
50 GotBaseSymInGotPlt = false;
51 PltEntrySize = 16;
52 PltHeaderSize = 32;
53 CopyRel = R_MIPS_COPY;
54 NoneRel = R_MIPS_NONE;
55 PltRel = R_MIPS_JUMP_SLOT;
56 CheriCapRel = R_MIPS_CHERI_CAPABILITY;
57 CheriCapCallRel = R_MIPS_CHERI_CAPABILITY_CALL;
58 NeedsThunks = true;
59
60 // Set `sigrie 1` as a trap instruction.
61 write32(TrapInstr.data(), 0x04170001);
62
63 if (ELFT::Is64Bits) {
64 RelativeRel = (R_MIPS_64 << 8) | R_MIPS_REL32;
65 SymbolicRel = R_MIPS_64;
66 TlsGotRel = R_MIPS_TLS_TPREL64;
67 TlsModuleIndexRel = R_MIPS_TLS_DTPMOD64;
68 TlsOffsetRel = R_MIPS_TLS_DTPREL64;
69 AbsPointerRel = (R_MIPS_64 << 8) | R_MIPS_CHERI_ABSPTR;
70 SizeRel = (R_MIPS_64 << 8) | R_MIPS_CHERI_SIZE;
71 } else {
72 RelativeRel = R_MIPS_REL32;
73 SymbolicRel = R_MIPS_32;
74 TlsGotRel = R_MIPS_TLS_TPREL32;
75 TlsModuleIndexRel = R_MIPS_TLS_DTPMOD32;
76 TlsOffsetRel = R_MIPS_TLS_DTPREL32;
77 AbsPointerRel = R_MIPS_CHERI_ABSPTR;
78 SizeRel = R_MIPS_CHERI_SIZE;
79 }
80 // Make the CheriABI start address more similar to the BFD output
81 if (Config->OSABI == ELFOSABI_FREEBSD && Config->isCheriABI()) {
82 // All the FreeBSD MIPS linker scripts use 0x120000000 (18*256M)
83 // XXXAR: TODO: actually it is 0x120000000 + SIZEOF_HEADERS (0x120000ae0)
84 DefaultImageBase = 0x0000000120000000;
85 }

92template <class ELFT> int MIPS<ELFT>::getCapabilitySize() const {
93 // Compute the size of a CHERI capability based on the MIPS ABI flags:
94 if ((Config->EFlags & EF_MIPS_MACH) == EF_MIPS_MACH_CHERI128)
95 return 16;
96 if ((Config->EFlags & EF_MIPS_MACH) == EF_MIPS_MACH_CHERI256)
97 return 32;
98 return 0;

101template <class ELFT>
102RelExpr MIPS<ELFT>::getRelExpr(RelType Type, const Symbol &S,
103 const uint8_t *Loc) const {
104 // See comment in the calculateMipsRelChain.
105 if (ELFT::Is64Bits || Config->MipsN32Abi)
106 Type &= 0xff;
107
108 switch (Type) {
109 case R_MIPS_JALR:
110 case R_MICROMIPS_JALR:
111 return R_HINT;
112 case R_MIPS_GPREL16:
113 case R_MIPS_GPREL32:
114 case R_MICROMIPS_GPREL16:
115 case R_MICROMIPS_GPREL7_S2:
116 return R_MIPS_GOTREL;
117 case R_MIPS_CHERI_CAPTABLEREL16:
118 return R_CHERI_CAPABILITY_TABLE_REL;
119 case R_MIPS_26:
120 case R_MICROMIPS_26_S1:
121 return R_PLT;
122 case R_MICROMIPS_PC26_S1:
123 return R_PLT_PC;
124 case R_MIPS_HI16:
125 case R_MIPS_LO16:
126 case R_MIPS_HIGHER:
127 case R_MIPS_HIGHEST:
128 case R_MICROMIPS_HI16:
129 case R_MICROMIPS_LO16:
130 // R_MIPS_HI16/R_MIPS_LO16 relocations against _gp_disp calculate
131 // offset between start of function and 'gp' value which by default
132 // equal to the start of .got section. In that case we consider these
133 // relocations as relative.
134 if (&S == ElfSym::MipsGpDisp)
135 return R_MIPS_GOT_GP_PC;
136 if (&S == ElfSym::MipsLocalGp)
137 return R_MIPS_GOT_GP;
138 LLVM_FALLTHROUGH;
139 case R_MIPS_32:
140 case R_MIPS_64:
141 case R_MIPS_GOT_OFST:
142 case R_MIPS_SUB:
143 case R_MIPS_TLS_DTPREL_HI16:
144 case R_MIPS_TLS_DTPREL_LO16:
145 case R_MIPS_TLS_DTPREL32:
146 case R_MIPS_TLS_DTPREL64:
147 case R_MIPS_TLS_TPREL_HI16:
148 case R_MIPS_TLS_TPREL_LO16:
149 case R_MIPS_TLS_TPREL32:
150 case R_MIPS_TLS_TPREL64:
151 case R_MICROMIPS_TLS_DTPREL_HI16:
152 case R_MICROMIPS_TLS_DTPREL_LO16:
153 case R_MICROMIPS_TLS_TPREL_HI16:
154 case R_MICROMIPS_TLS_TPREL_LO16:
155 return R_ABS;
156 case R_MIPS_PC32:
157 case R_MIPS_PC16:
158 case R_MIPS_PC19_S2:
159 case R_MIPS_PC21_S2:
160 case R_MIPS_PC26_S2:
161 case R_MIPS_PCHI16:
162 case R_MIPS_PCLO16:
163 case R_MICROMIPS_PC7_S1:
164 case R_MICROMIPS_PC10_S1:
165 case R_MICROMIPS_PC16_S1:
166 case R_MICROMIPS_PC18_S3:
167 case R_MICROMIPS_PC19_S2:
168 case R_MICROMIPS_PC23_S2:
169 case R_MICROMIPS_PC21_S1:
170 return R_PC;
171 case R_MIPS_GOT16:
172 case R_MICROMIPS_GOT16:
173 if (S.isLocal())
174 return R_MIPS_GOT_LOCAL_PAGE;
175 LLVM_FALLTHROUGH;
176 case R_MIPS_CALL16:
177 case R_MIPS_GOT_DISP:
178 case R_MIPS_TLS_GOTTPREL:
179 case R_MICROMIPS_CALL16:
180 case R_MICROMIPS_TLS_GOTTPREL:
181 return R_MIPS_GOT_OFF;
182 case R_MIPS_CALL_HI16:
183 case R_MIPS_CALL_LO16:
184 case R_MIPS_GOT_HI16:
185 case R_MIPS_GOT_LO16:
186 case R_MICROMIPS_CALL_HI16:
187 case R_MICROMIPS_CALL_LO16:
188 case R_MICROMIPS_GOT_HI16:
189 case R_MICROMIPS_GOT_LO16:
190 return R_MIPS_GOT_OFF32;
191 case R_MIPS_GOT_PAGE:
192 return R_MIPS_GOT_LOCAL_PAGE;
193 case R_MIPS_TLS_GD:
194 case R_MICROMIPS_TLS_GD:
195 return R_MIPS_TLSGD;
196 case R_MIPS_TLS_LDM:
197 case R_MICROMIPS_TLS_LDM:
198 return R_MIPS_TLSLD;
199 case R_MIPS_NONE:
200 return R_NONE;
201 case R_MIPS_CHERI_CAPABILITY:
202 return R_CHERI_CAPABILITY;
203 case R_MIPS_CHERI_CAPTAB_LO16:
204 case R_MIPS_CHERI_CAPTAB_HI16:
205 return R_CHERI_CAPABILITY_TABLE_INDEX;
206 case R_MIPS_CHERI_CAPCALL_LO16:
207 case R_MIPS_CHERI_CAPCALL_HI16:
208 return R_CHERI_CAPABILITY_TABLE_INDEX_CALL;
209 case R_MIPS_CHERI_CAPTAB_CLC11:
210 case R_MIPS_CHERI_CAPTAB20:
211 return R_CHERI_CAPABILITY_TABLE_INDEX_SMALL_IMMEDIATE;
212 case R_MIPS_CHERI_CAPCALL_CLC11:
213 case R_MIPS_CHERI_CAPCALL20:
214 return R_CHERI_CAPABILITY_TABLE_INDEX_CALL_SMALL_IMMEDIATE;
215 case R_MIPS_CHERI_CAPTAB_TLS_GD_LO16:
216 case R_MIPS_CHERI_CAPTAB_TLS_GD_HI16:
217 return R_MIPS_CHERI_CAPTAB_TLSGD;
218 case R_MIPS_CHERI_CAPTAB_TLS_LDM_LO16:
219 case R_MIPS_CHERI_CAPTAB_TLS_LDM_HI16:
220 return R_MIPS_CHERI_CAPTAB_TLSLD;
221 case R_MIPS_CHERI_CAPTAB_TLS_TPREL_LO16:
222 case R_MIPS_CHERI_CAPTAB_TLS_TPREL_HI16:
223 return R_MIPS_CHERI_CAPTAB_TPREL;
224 default:
225 error(getErrorLocation(Loc) + "unknown relocation (" + Twine(Type) +
226 ") against symbol " + toString(S));
227 return R_NONE;
228 }
229}

245template <endianness E> static uint32_t readShuffle(const uint8_t *Loc) {
246 // The major opcode of a microMIPS instruction needs to appear
247 // in the first 16-bit word (lowest address) for efficient hardware
248 // decode so that it knows if the instruction is 16-bit or 32-bit
249 // as early as possible. To do so, little-endian binaries keep 16-bit
250 // words in a big-endian order. That is why we have to swap these
251 // words to get a correct value.
252 uint32_t V = read32<E>(Loc);
253 if (E == support::little)
254 return (V << 16) | (V >> 16);
255 return V;
256}
257
258template <endianness E>
259static void writeValue(uint8_t *Loc, uint64_t V, uint8_t BitsSize,
260 uint8_t Shift) {
261 uint32_t Instr = read32<E>(Loc);
262 uint32_t Mask = 0xffffffff >> (32 - BitsSize);
263 uint32_t Data = (Instr & ~Mask) | ((V >> Shift) & Mask);
264 write32<E>(Loc, Data);
265}
266
267template <endianness E>
268static void writeShuffleValue(uint8_t *Loc, uint64_t V, uint8_t BitsSize,
269 uint8_t Shift) {
270 // See comments in readShuffle for purpose of this code.
271 uint16_t *Words = (uint16_t *)Loc;
272 if (E == support::little)
273 std::swap(Words[0], Words[1]);
274
275 writeValue<E>(Loc, V, BitsSize, Shift);
276
277 if (E == support::little)
278 std::swap(Words[0], Words[1]);
279}

281template <endianness E>
282static void writeMicroRelocation16(uint8_t *Loc, uint64_t V, uint8_t BitsSize,
283 uint8_t Shift) {
284 uint16_t Instr = read16<E>(Loc);
285 uint16_t Mask = 0xffff >> (16 - BitsSize);
286 uint16_t Data = (Instr & ~Mask) | ((V >> Shift) & Mask);
287 write16<E>(Loc, Data);
288}

290template <class ELFT> void MIPS<ELFT>::writePltHeader(uint8_t *Buf) const {
291 const endianness E = ELFT::TargetEndianness;
292 if (isMicroMips()) {
293 uint64_t GotPlt = In.GotPlt->getVA();
294 uint64_t Plt = In.Plt->getVA();
295 // Overwrite trap instructions written by Writer::writeTrapInstr.
296 memset(Buf, 0, PltHeaderSize);
297
298 write16<E>(Buf, isMipsR6() ? 0x7860 : 0x7980); // addiupc v1, (GOTPLT) - .
299 write16<E>(Buf + 4, 0xff23); // lw $25, 0($3)
300 write16<E>(Buf + 8, 0x0535); // subu16 $2, $2, $3
301 write16<E>(Buf + 10, 0x2525); // srl16 $2, $2, 2
302 write16<E>(Buf + 12, 0x3302); // addiu $24, $2, -2
303 write16<E>(Buf + 14, 0xfffe);
304 write16<E>(Buf + 16, 0x0dff); // move $15, $31
305 if (isMipsR6()) {
306 write16<E>(Buf + 18, 0x0f83); // move $28, $3
307 write16<E>(Buf + 20, 0x472b); // jalrc $25
308 write16<E>(Buf + 22, 0x0c00); // nop
309 relocateOne(Buf, R_MICROMIPS_PC19_S2, GotPlt - Plt);
310 } else {
311 write16<E>(Buf + 18, 0x45f9); // jalrc $25
312 write16<E>(Buf + 20, 0x0f83); // move $28, $3
313 write16<E>(Buf + 22, 0x0c00); // nop
314 relocateOne(Buf, R_MICROMIPS_PC23_S2, GotPlt - Plt);
315 }
316 return;
317 }
318
319 if (Config->MipsN32Abi) {
320 write32<E>(Buf, 0x3c0e0000); // lui $14, %hi(&GOTPLT[0])
321 write32<E>(Buf + 4, 0x8dd90000); // lw $25, %lo(&GOTPLT[0])($14)
322 write32<E>(Buf + 8, 0x25ce0000); // addiu $14, $14, %lo(&GOTPLT[0])
323 write32<E>(Buf + 12, 0x030ec023); // subu $24, $24, $14
324 write32<E>(Buf + 16, 0x03e07825); // move $15, $31
325 write32<E>(Buf + 20, 0x0018c082); // srl $24, $24, 2
326 } else if (ELFT::Is64Bits) {
327 write32<E>(Buf, 0x3c0e0000); // lui $14, %hi(&GOTPLT[0])
328 write32<E>(Buf + 4, 0xddd90000); // ld $25, %lo(&GOTPLT[0])($14)
329 write32<E>(Buf + 8, 0x25ce0000); // addiu $14, $14, %lo(&GOTPLT[0])
330 write32<E>(Buf + 12, 0x030ec023); // subu $24, $24, $14
331 write32<E>(Buf + 16, 0x03e07825); // move $15, $31
332 write32<E>(Buf + 20, 0x0018c0c2); // srl $24, $24, 3
333 } else {
334 write32<E>(Buf, 0x3c1c0000); // lui $28, %hi(&GOTPLT[0])
335 write32<E>(Buf + 4, 0x8f990000); // lw $25, %lo(&GOTPLT[0])($28)
336 write32<E>(Buf + 8, 0x279c0000); // addiu $28, $28, %lo(&GOTPLT[0])
337 write32<E>(Buf + 12, 0x031cc023); // subu $24, $24, $28
338 write32<E>(Buf + 16, 0x03e07825); // move $15, $31
339 write32<E>(Buf + 20, 0x0018c082); // srl $24, $24, 2
340 }
341
342 uint32_t JalrInst = Config->ZHazardplt ? 0x0320fc09 : 0x0320f809;
343 write32<E>(Buf + 24, JalrInst); // jalr.hb $25 or jalr $25
344 write32<E>(Buf + 28, 0x2718fffe); // subu $24, $24, 2
345
346 uint64_t GotPlt = In.GotPlt->getVA();
347 writeValue<E>(Buf, GotPlt + 0x8000, 16, 16);
348 writeValue<E>(Buf + 4, GotPlt, 16, 0);
349 writeValue<E>(Buf + 8, GotPlt, 16, 0);
350}

352template <class ELFT>
353void MIPS<ELFT>::writePlt(uint8_t *Buf, uint64_t GotPltEntryAddr,
354 uint64_t PltEntryAddr, int32_t Index,
355 unsigned RelOff) const {
356 const endianness E = ELFT::TargetEndianness;
357 if (isMicroMips()) {
358 // Overwrite trap instructions written by Writer::writeTrapInstr.
359 memset(Buf, 0, PltEntrySize);
360
361 if (isMipsR6()) {
362 write16<E>(Buf, 0x7840); // addiupc $2, (GOTPLT) - .
363 write16<E>(Buf + 4, 0xff22); // lw $25, 0($2)
364 write16<E>(Buf + 8, 0x0f02); // move $24, $2
365 write16<E>(Buf + 10, 0x4723); // jrc $25 / jr16 $25
366 relocateOne(Buf, R_MICROMIPS_PC19_S2, GotPltEntryAddr - PltEntryAddr);
367 } else {
368 write16<E>(Buf, 0x7900); // addiupc $2, (GOTPLT) - .
369 write16<E>(Buf + 4, 0xff22); // lw $25, 0($2)
370 write16<E>(Buf + 8, 0x4599); // jrc $25 / jr16 $25
371 write16<E>(Buf + 10, 0x0f02); // move $24, $2
372 relocateOne(Buf, R_MICROMIPS_PC23_S2, GotPltEntryAddr - PltEntryAddr);
373 }
374 return;
375 }
376
377 uint32_t LoadInst = ELFT::Is64Bits ? 0xddf90000 : 0x8df90000;
378 uint32_t JrInst = isMipsR6() ? (Config->ZHazardplt ? 0x03200409 : 0x03200009)
379 : (Config->ZHazardplt ? 0x03200408 : 0x03200008);
380 uint32_t AddInst = ELFT::Is64Bits ? 0x65f80000 : 0x25f80000;
381
382 write32<E>(Buf, 0x3c0f0000); // lui $15, %hi(.got.plt entry)
383 write32<E>(Buf + 4, LoadInst); // l[wd] $25, %lo(.got.plt entry)($15)
384 write32<E>(Buf + 8, JrInst); // jr $25 / jr.hb $25
385 write32<E>(Buf + 12, AddInst); // [d]addiu $24, $15, %lo(.got.plt entry)
386 writeValue<E>(Buf, GotPltEntryAddr + 0x8000, 16, 16);
387 writeValue<E>(Buf + 4, GotPltEntryAddr, 16, 0);
388 writeValue<E>(Buf + 12, GotPltEntryAddr, 16, 0);
389}

391template <class ELFT>
392bool MIPS<ELFT>::needsThunk(RelExpr Expr, RelType Type, const InputFile *File,
393 uint64_t BranchAddr, const Symbol &S) const {
394 // Any MIPS PIC code function is invoked with its address in register $t9.
395 // So if we have a branch instruction from non-PIC code to the PIC one
396 // we cannot make the jump directly and need to create a small stubs
397 // to save the target function address.
398 // See page 3-38 ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
399 if (Type != R_MIPS_26 && Type != R_MIPS_PC26_S2 &&
400 Type != R_MICROMIPS_26_S1 && Type != R_MICROMIPS_PC26_S1)
401 return false;
402 auto *F = dyn_cast_or_null<ObjFile<ELFT>>(File);
403 if (!F)
404 return false;
405 // If current file has PIC code, LA25 stub is not required.
406 if (F->getObj().getHeader()->e_flags & EF_MIPS_PIC)
407 return false;
408 auto *D = dyn_cast<Defined>(&S);
409 // LA25 is required if target file has PIC code
410 // or target symbol is a PIC symbol.
411 return D && isMipsPIC<ELFT>(D);
412}

414template <class ELFT>
415int64_t MIPS<ELFT>::getImplicitAddend(const uint8_t *Buf, RelType Type) const {
416 const endianness E = ELFT::TargetEndianness;
417 switch (Type) {
418 case R_MIPS_32:
419 case R_MIPS_GPREL32:
420 case R_MIPS_TLS_DTPREL32:
421 case R_MIPS_TLS_TPREL32:
422 return SignExtend64<32>(read32<E>(Buf));
423 case R_MIPS_26:
424 // FIXME (simon): If the relocation target symbol is not a PLT entry
425 // we should use another expression for calculation:
426 // ((A << 2) | (P & 0xf0000000)) >> 2
427 return SignExtend64<28>(read32<E>(Buf) << 2);
428 case R_MIPS_GOT16:
429 case R_MIPS_HI16:
430 case R_MIPS_PCHI16:
431 return SignExtend64<16>(read32<E>(Buf)) << 16;
432 case R_MIPS_GPREL16:
433 case R_MIPS_LO16:
434 case R_MIPS_PCLO16:
435 case R_MIPS_TLS_DTPREL_HI16:
436 case R_MIPS_TLS_DTPREL_LO16:
437 case R_MIPS_TLS_TPREL_HI16:
438 case R_MIPS_TLS_TPREL_LO16:
439 return SignExtend64<16>(read32<E>(Buf));
440 case R_MICROMIPS_GOT16:
441 case R_MICROMIPS_HI16:
442 return SignExtend64<16>(readShuffle<E>(Buf)) << 16;
443 case R_MICROMIPS_GPREL16:
444 case R_MICROMIPS_LO16:
445 case R_MICROMIPS_TLS_DTPREL_HI16:
446 case R_MICROMIPS_TLS_DTPREL_LO16:
447 case R_MICROMIPS_TLS_TPREL_HI16:
448 case R_MICROMIPS_TLS_TPREL_LO16:
449 return SignExtend64<16>(readShuffle<E>(Buf));
450 case R_MICROMIPS_GPREL7_S2:
451 return SignExtend64<9>(readShuffle<E>(Buf) << 2);
452 case R_MIPS_PC16:
453 return SignExtend64<18>(read32<E>(Buf) << 2);
454 case R_MIPS_PC19_S2:
455 return SignExtend64<21>(read32<E>(Buf) << 2);
456 case R_MIPS_PC21_S2:
457 return SignExtend64<23>(read32<E>(Buf) << 2);
458 case R_MIPS_PC26_S2:
459 return SignExtend64<28>(read32<E>(Buf) << 2);
460 case R_MIPS_PC32:
461 return SignExtend64<32>(read32<E>(Buf));
462 case R_MICROMIPS_26_S1:
463 return SignExtend64<27>(readShuffle<E>(Buf) << 1);
464 case R_MICROMIPS_PC7_S1:
465 return SignExtend64<8>(read16<E>(Buf) << 1);
466 case R_MICROMIPS_PC10_S1:
467 return SignExtend64<11>(read16<E>(Buf) << 1);
468 case R_MICROMIPS_PC16_S1:
469 return SignExtend64<17>(readShuffle<E>(Buf) << 1);
470 case R_MICROMIPS_PC18_S3:
471 return SignExtend64<21>(readShuffle<E>(Buf) << 3);
472 case R_MICROMIPS_PC19_S2:
473 return SignExtend64<21>(readShuffle<E>(Buf) << 2);
474 case R_MICROMIPS_PC21_S1:
475 return SignExtend64<22>(readShuffle<E>(Buf) << 1);
476 case R_MICROMIPS_PC23_S2:
477 return SignExtend64<25>(readShuffle<E>(Buf) << 2);
478 case R_MICROMIPS_PC26_S1:
479 return SignExtend64<27>(readShuffle<E>(Buf) << 1);
480 default:
481 return 0;
482 }
483}

485static std::pair<uint32_t, uint64_t>
486calculateMipsRelChain(uint8_t *Loc, RelType Type, uint64_t Val) {
487 // MIPS N64 ABI packs multiple relocations into the single relocation
488 // record. In general, all up to three relocations can have arbitrary
489 // types. In fact, Clang and GCC uses only a few combinations. For now,
490 // we support two of them. That is allow to pass at least all LLVM
491 // test suite cases.
492 // <any relocation> / R_MIPS_SUB / R_MIPS_HI16 | R_MIPS_LO16
493 // <any relocation> / R_MIPS_64 / R_MIPS_NONE
494 // The first relocation is a 'real' relocation which is calculated
495 // using the corresponding symbol's value. The second and the third
496 // relocations used to modify result of the first one: extend it to
497 // 64-bit, extract high or low part etc. For details, see part 2.9 Relocation
498 // at the https://dmz-portal.mips.com/mw/images/8/82/007-4658-001.pdf
499 RelType Type2 = (Type >> 8) & 0xff;
500 RelType Type3 = (Type >> 16) & 0xff;
501 if (Type2 == R_MIPS_NONE && Type3 == R_MIPS_NONE)
502 return std::make_pair(Type, Val);
503 if (Type2 == R_MIPS_64 && Type3 == R_MIPS_NONE)
504 return std::make_pair(Type2, Val);
505 if (Type2 == R_MIPS_SUB && (Type3 == R_MIPS_HI16 || Type3 == R_MIPS_LO16))
506 return std::make_pair(Type3, -Val);
507 error(getErrorLocation(Loc) + "unsupported relocations combination " +
508 Twine(Type));
509 return std::make_pair(Type & 0xff, Val);
510}

512static bool isBranchReloc(RelType Type) {
513 return Type == R_MIPS_26 || Type == R_MIPS_PC26_S2 ||
514 Type == R_MIPS_PC21_S2 || Type == R_MIPS_PC16;
515}

517static bool isMicroBranchReloc(RelType Type) {
518 return Type == R_MICROMIPS_26_S1 || Type == R_MICROMIPS_PC16_S1 ||
519 Type == R_MICROMIPS_PC10_S1 || Type == R_MICROMIPS_PC7_S1;
520}

522template <class ELFT>
523static uint64_t fixupCrossModeJump(uint8_t *Loc, RelType Type, uint64_t Val) {
524 // Here we need to detect jump/branch from regular MIPS code
525 // to a microMIPS target and vice versa. In that cases jump
526 // instructions need to be replaced by their "cross-mode"
527 // equivalents.
528 const endianness E = ELFT::TargetEndianness;
529 bool IsMicroTgt = Val & 0x1;
530 bool IsCrossJump = (IsMicroTgt && isBranchReloc(Type)) ||
531 (!IsMicroTgt && isMicroBranchReloc(Type));
532 if (!IsCrossJump)
533 return Val;
534
535 switch (Type) {
536 case R_MIPS_26: {
537 uint32_t Inst = read32<E>(Loc) >> 26;
538 if (Inst == 0x3 || Inst == 0x1d) { // JAL or JALX
539 writeValue<E>(Loc, 0x1d << 26, 32, 0);
540 return Val;
541 }
542 break;
543 }
544 case R_MICROMIPS_26_S1: {
545 uint32_t Inst = readShuffle<E>(Loc) >> 26;
546 if (Inst == 0x3d || Inst == 0x3c) { // JAL32 or JALX32
547 Val >>= 1;
548 writeShuffleValue<E>(Loc, 0x3c << 26, 32, 0);
549 return Val;
550 }
551 break;
552 }
553 case R_MIPS_PC26_S2:
554 case R_MIPS_PC21_S2:
555 case R_MIPS_PC16:
556 case R_MICROMIPS_PC16_S1:
557 case R_MICROMIPS_PC10_S1:
558 case R_MICROMIPS_PC7_S1:
559 // FIXME (simon): Support valid branch relocations.
560 break;
561 default:
562 llvm_unreachable("unexpected jump/branch relocation");
563 }
564
565 error(getErrorLocation(Loc) +
566 "unsupported jump/branch instruction between ISA modes referenced by " +
567 toString(Type) + " relocation");
568 return Val;
569}

571template <class ELFT>
572void MIPS<ELFT>::relocateOne(uint8_t *Loc, RelType Type, uint64_t Val) const {
573 const endianness E = ELFT::TargetEndianness;
574
575 if (ELFT::Is64Bits || Config->MipsN32Abi)
576 std::tie(Type, Val) = calculateMipsRelChain(Loc, Type, Val);
577
578 // Detect cross-mode jump/branch and fix instruction.
579 Val = fixupCrossModeJump<ELFT>(Loc, Type, Val);
580
581 if (!Config->isCheriABI()) {
582 // Thread pointer and DRP offsets from the start of TLS data area.
583 // https://www.linux-mips.org/wiki/NPTL
584 if (Type == R_MIPS_TLS_DTPREL_HI16 || Type == R_MIPS_TLS_DTPREL_LO16 ||
585 Type == R_MIPS_TLS_DTPREL32 || Type == R_MIPS_TLS_DTPREL64 ||
586 Type == R_MICROMIPS_TLS_DTPREL_HI16 ||
587 Type == R_MICROMIPS_TLS_DTPREL_LO16) {
588 Val -= 0x8000;
589 } else if (Type == R_MIPS_TLS_TPREL_HI16 || Type == R_MIPS_TLS_TPREL_LO16 ||
590 Type == R_MIPS_TLS_TPREL32 || Type == R_MIPS_TLS_TPREL64 ||
591 Type == R_MICROMIPS_TLS_TPREL_HI16 ||
592 Type == R_MICROMIPS_TLS_TPREL_LO16) {
593 Val -= 0x7000;
594 }
595 }
596
597 switch (Type) {
598 case R_MIPS_32:
599 case R_MIPS_GPREL32:
600 case R_MIPS_TLS_DTPREL32:
601 case R_MIPS_TLS_TPREL32:
602 write32<E>(Loc, Val);
603 break;
604 case R_MIPS_64:
605 case R_MIPS_TLS_DTPREL64:
606 case R_MIPS_TLS_TPREL64:
607 write64<E>(Loc, Val);
608 break;
609 case R_MIPS_26:
610 writeValue<E>(Loc, Val, 26, 2);
611 break;
612 case R_MIPS_GOT16:
613 // The R_MIPS_GOT16 relocation's value in "relocatable" linking mode
614 // is updated addend (not a GOT index). In that case write high 16 bits
615 // to store a correct addend value.
616 if (Config->Relocatable) {
617 writeValue<E>(Loc, Val + 0x8000, 16, 16);
618 } else {
619 checkInt(Loc, Val, 16, Type);
620 writeValue<E>(Loc, Val, 16, 0);
621 }
622 break;
623 case R_MICROMIPS_GOT16:
624 if (Config->Relocatable) {
625 writeShuffleValue<E>(Loc, Val + 0x8000, 16, 16);
626 } else {
627 checkInt(Loc, Val, 16, Type);
628 writeShuffleValue<E>(Loc, Val, 16, 0);
629 }
630 break;
631 case R_MIPS_CALL16:
632 case R_MIPS_GOT_DISP:
633 case R_MIPS_GOT_PAGE:
634 case R_MIPS_GPREL16:
635 case R_MIPS_TLS_GD:
636 case R_MIPS_TLS_GOTTPREL:
637 case R_MIPS_TLS_LDM:
638 checkInt(Loc, Val, 16, Type);
639 LLVM_FALLTHROUGH;
640 case R_MIPS_CALL_LO16:
641 case R_MIPS_GOT_LO16:
642 case R_MIPS_GOT_OFST:
643 case R_MIPS_LO16:
644 case R_MIPS_PCLO16:
645 case R_MIPS_TLS_DTPREL_LO16:
646 case R_MIPS_TLS_TPREL_LO16:
647 case R_MIPS_CHERI_CAPTAB_LO16:
648 case R_MIPS_CHERI_CAPCALL_LO16:
649 case R_MIPS_CHERI_CAPTAB_TLS_GD_LO16:
650 case R_MIPS_CHERI_CAPTAB_TLS_LDM_LO16:
651 case R_MIPS_CHERI_CAPTAB_TLS_TPREL_LO16:
652 writeValue<E>(Loc, Val, 16, 0);
653 break;
654 case R_MICROMIPS_GPREL16:
655 case R_MICROMIPS_TLS_GD:
656 case R_MICROMIPS_TLS_LDM:
657 checkInt(Loc, Val, 16, Type);
658 writeShuffleValue<E>(Loc, Val, 16, 0);
659 break;
660 case R_MICROMIPS_CALL16:
661 case R_MICROMIPS_CALL_LO16:
662 case R_MICROMIPS_LO16:
663 case R_MICROMIPS_TLS_DTPREL_LO16:
664 case R_MICROMIPS_TLS_GOTTPREL:
665 case R_MICROMIPS_TLS_TPREL_LO16:
666 writeShuffleValue<E>(Loc, Val, 16, 0);
667 break;
668 case R_MICROMIPS_GPREL7_S2:
669 checkInt(Loc, Val, 7, Type);
670 writeShuffleValue<E>(Loc, Val, 7, 2);
671 break;
672 case R_MIPS_CALL_HI16:
673 case R_MIPS_GOT_HI16:
674 case R_MIPS_HI16:
675 case R_MIPS_PCHI16:
676 case R_MIPS_TLS_DTPREL_HI16:
677 case R_MIPS_TLS_TPREL_HI16:
678 case R_MIPS_CHERI_CAPTAB_HI16:
679 case R_MIPS_CHERI_CAPCALL_HI16:
680 case R_MIPS_CHERI_CAPTAB_TLS_GD_HI16:
681 case R_MIPS_CHERI_CAPTAB_TLS_LDM_HI16:
682 case R_MIPS_CHERI_CAPTAB_TLS_TPREL_HI16:
683 writeValue<E>(Loc, Val + 0x8000, 16, 16);
684 break;
685 case R_MIPS_CHERI_CAPTAB_CLC11:
686 case R_MIPS_CHERI_CAPCALL_CLC11:
687 // The clc relocations have a signed 15 bit range (11 bit immediate shifted
688 // by 4). This is the same for 128 and 256 even though they have different
689 // capability sizes
690 assert((Val & 0xf) == 0 && "Bottom 4 bits should always be zero!");
691 checkInt(Loc, Val >> 4, 11, Type);
692 writeValue<E>(Loc, Val, 11, 4);
693 break;
694 case R_MIPS_CHERI_CAPTAB20:
695 case R_MIPS_CHERI_CAPCALL20:
696 // The new clc instruction has a 20 bit signed range (16 bit immediate
697 // shifted by 4). This is the same for 128 and 256 even though they have
698 // different capability sizes
699 assert((Val & 0xf) == 0 && "Bottom 4 bits should always be zero!");
700 checkInt(Loc, Val >> 4, 16, Type);
701 writeValue<E>(Loc, Val, 16, 4);
702 break;
703 case R_MICROMIPS_CALL_HI16:
704 case R_MICROMIPS_GOT_HI16:
705 case R_MICROMIPS_HI16:
706 case R_MICROMIPS_TLS_DTPREL_HI16:
707 case R_MICROMIPS_TLS_TPREL_HI16:
708 writeShuffleValue<E>(Loc, Val + 0x8000, 16, 16);
709 break;
710 case R_MIPS_HIGHER:
711 writeValue<E>(Loc, Val + 0x80008000, 16, 32);
712 break;
713 case R_MIPS_HIGHEST:
714 writeValue<E>(Loc, Val + 0x800080008000, 16, 48);
715 break;
716 case R_MIPS_JALR:
717 case R_MICROMIPS_JALR:
718 // Ignore this optimization relocation for now
719 break;
720 case R_MIPS_PC16:
721 checkAlignment(Loc, Val, 4, Type);
722 checkInt(Loc, Val, 18, Type);
723 writeValue<E>(Loc, Val, 16, 2);
724 break;
725 case R_MIPS_PC19_S2:
726 checkAlignment(Loc, Val, 4, Type);
727 checkInt(Loc, Val, 21, Type);
728 writeValue<E>(Loc, Val, 19, 2);
729 break;
730 case R_MIPS_PC21_S2:
731 checkAlignment(Loc, Val, 4, Type);
732 checkInt(Loc, Val, 23, Type);
733 writeValue<E>(Loc, Val, 21, 2);
734 break;
735 case R_MIPS_PC26_S2:
736 checkAlignment(Loc, Val, 4, Type);
737 checkInt(Loc, Val, 28, Type);
738 writeValue<E>(Loc, Val, 26, 2);
739 break;
740 case R_MIPS_PC32:
741 writeValue<E>(Loc, Val, 32, 0);
742 break;
743 case R_MICROMIPS_26_S1:
744 case R_MICROMIPS_PC26_S1:
745 checkInt(Loc, Val, 27, Type);
746 writeShuffleValue<E>(Loc, Val, 26, 1);
747 break;
748 case R_MICROMIPS_PC7_S1:
749 checkInt(Loc, Val, 8, Type);
750 writeMicroRelocation16<E>(Loc, Val, 7, 1);
751 break;
752 case R_MICROMIPS_PC10_S1:
753 checkInt(Loc, Val, 11, Type);
754 writeMicroRelocation16<E>(Loc, Val, 10, 1);
755 break;
756 case R_MICROMIPS_PC16_S1:
757 checkInt(Loc, Val, 17, Type);
758 writeShuffleValue<E>(Loc, Val, 16, 1);
759 break;
760 case R_MICROMIPS_PC18_S3:
761 checkInt(Loc, Val, 21, Type);
762 writeShuffleValue<E>(Loc, Val, 18, 3);
763 break;
764 case R_MICROMIPS_PC19_S2:
765 checkInt(Loc, Val, 21, Type);
766 writeShuffleValue<E>(Loc, Val, 19, 2);
767 break;
768 case R_MICROMIPS_PC21_S1:
769 checkInt(Loc, Val, 22, Type);
770 writeShuffleValue<E>(Loc, Val, 21, 1);
771 break;
772 case R_MICROMIPS_PC23_S2:
773 checkInt(Loc, Val, 25, Type);
774 writeShuffleValue<E>(Loc, Val, 23, 2);
775 break;
776 case R_MIPS_CHERI_CAPABILITY:
777 llvm_unreachable("R_MIPS_CHERI_CAPABILITY should never be handled here!");
778 default:
779 llvm_unreachable("unknown relocation");
780 }
781}

788// Return true if the symbol is a PIC function.
789template <class ELFT> bool elf::isMipsPIC(const Defined *Sym) {
790 if (!Sym->isFunc())
791 return false;
792
793 if (Sym->StOther & STO_MIPS_PIC)
794 return true;
795
796 if (!Sym->Section)
797 return false;
798
799 ObjFile<ELFT> *File =
800 cast<InputSectionBase>(Sym->Section)->template getFile<ELFT>();
801 if (!File)
802 return false;
803
804 return File->getObj().getHeader()->e_flags & EF_MIPS_PIC;
805}
