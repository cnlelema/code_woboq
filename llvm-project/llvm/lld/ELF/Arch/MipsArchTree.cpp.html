45 return "n64";
46 case EF_MIPS_ABI2:

69static void checkFlags(ArrayRef<FileFlags> Files) {
70 assert(!Files.empty() && "expected non-empty file list");
71
72 uint32_t ABI = Files[0].Flags & (EF_MIPS_ABI | EF_MIPS_ABI2);
73 bool Nan = Files[0].Flags & EF_MIPS_NAN2008;
74 bool Fp = Files[0].Flags & EF_MIPS_FP64;
75
76 // Handle explicit ABI selection via -melf64btsmip_cheri_fbsd
77 if ((Config->isCheriABI() && ABI != EF_MIPS_ABI_CHERIABI) ||
78 (!Config->isCheriABI() && ABI == EF_MIPS_ABI_CHERIABI)) {
79 error(toString(Files[0].File) + ": ABI '" + getAbiName(ABI) +
80 "' is incompatible with explicitly selected linker emulation '" +
81 Config->Emulation + "'");
82 ABI = Config->isCheriABI() ? EF_MIPS_ABI_CHERIABI : 0;
83 }
84
85 for (const FileFlags &F : Files) {
86 if (Config->Is64 && F.Flags & EF_MIPS_MICROMIPS)
87 error(toString(F.File) + ": microMIPS 64-bit is not supported");
88
89 uint32_t ABI2 = F.Flags & (EF_MIPS_ABI | EF_MIPS_ABI2);
90 if (ABI != ABI2)
91 error(toString(F.File) + ": ABI '" + getAbiName(ABI2) +
92 "' is incompatible with target ABI '" + getAbiName(ABI) + "'");
93
94 bool Nan2 = F.Flags & EF_MIPS_NAN2008;
95 if (Nan != Nan2)
96 error(toString(F.File) + ": -mnan=" + getNanName(Nan2) +
97 " is incompatible with target -mnan=" + getNanName(Nan));
98
99 bool Fp2 = F.Flags & EF_MIPS_FP64;
100 if (Fp != Fp2)
101 error(toString(F.File) + ": -mfp" + getFpName(Fp2) +
102 " is incompatible with target -mfp" + getFpName(Fp));
103 }
104}

106static uint32_t getMiscFlags(ArrayRef<FileFlags> Files) {
107 uint32_t Ret = 0;
108 for (const FileFlags &F : Files)
109 Ret |= F.Flags &
110 (EF_MIPS_ABI | EF_MIPS_ABI2 | EF_MIPS_ARCH_ASE | EF_MIPS_NOREORDER |
111 EF_MIPS_MICROMIPS | EF_MIPS_NAN2008 | EF_MIPS_32BITMODE);
112 return Ret;
113}

115static uint32_t getPicFlags(ArrayRef<FileFlags> Files) {
116 // Check PIC/non-PIC compatibility.
117 bool IsPic = Files[0].Flags & (EF_MIPS_PIC | EF_MIPS_CPIC);
118 for (const FileFlags &F : Files.slice(1)) {
119 bool IsPic2 = F.Flags & (EF_MIPS_PIC | EF_MIPS_CPIC);
120 if (IsPic && !IsPic2)
121 warn(toString(F.File) +
122 ": linking non-abicalls code with abicalls code " +
123 toString(Files[0].File));
124 if (!IsPic && IsPic2)
125 warn(toString(F.File) +
126 ": linking abicalls code with non-abicalls code " +
127 toString(Files[0].File));
128 }
129
130 // Compute the result PIC/non-PIC flag.
131 uint32_t Ret = Files[0].Flags & (EF_MIPS_PIC | EF_MIPS_CPIC);
132 for (const FileFlags &F : Files.slice(1))
133 Ret &= F.Flags & (EF_MIPS_PIC | EF_MIPS_CPIC);
134
135 // PIC code is inherently CPIC and may not set CPIC flag explicitly.
136 if (Ret & EF_MIPS_PIC)
137 Ret |= EF_MIPS_CPIC;
138 return Ret;
139}

186static bool isArchMatched(uint32_t New, uint32_t Res) {
187 // llvm::errs() << __func__ << ": new=" << utohexstr(New) << " res=" << utohexstr(Res) << "\n";
188 if (New == Res)
189 return true;
190 if (New == EF_MIPS_ARCH_32 && isArchMatched(EF_MIPS_ARCH_64, Res))
191 return true;
192 if (New == EF_MIPS_ARCH_32R2 && isArchMatched(EF_MIPS_ARCH_64R2, Res))
193 return true;
194
195 // check for cheri128 vs cheri256 and upgrade non-cheri to cheri
196 uint32_t NewMach = (New & EF_MIPS_MACH);
197 uint32_t ResMach = (Res & EF_MIPS_MACH);
198 if (ResMach == EF_MIPS_MACH_CHERI128) {
199 if (NewMach != 0 && NewMach != EF_MIPS_MACH_CHERI128 && NewMach != EF_MIPS_MACH_BERI)
200 return false;
201 return isArchMatched(New & ~EF_MIPS_MACH, Res & ~EF_MIPS_MACH);
202 }
203 if (ResMach == EF_MIPS_MACH_CHERI256) {
204 if (NewMach != 0 && NewMach != EF_MIPS_MACH_CHERI256 && NewMach != EF_MIPS_MACH_BERI)
205 return false;
206 return isArchMatched(New & ~EF_MIPS_MACH, Res & ~EF_MIPS_MACH);
207 }
208
209 for (const auto &Edge : ArchTree) {
210 if (Res == Edge.Child) {
211 Res = Edge.Parent;
212 if (Res == New)
213 return true;
214 }
215 }
216 return false;
217}

307static inline bool isBeriOrCheri(uint32_t Flags) {
308 uint32_t Mach = Flags & EF_MIPS_MACH;

310 Mach == EF_MIPS_MACH_CHERI256;

313// There are (arguably too) many MIPS ISAs out there. Their relationships
314// can be represented as a forest. If all input files have ISAs which
315// reachable by repeated proceeding from the single child to the parent,
316// these input files are compatible. In that case we need to return "highest"
317// ISA. If there are incompatible input files, we show an error.
318// For example, mips1 is a "parent" of mips2 and such files are compatible.
319// Output file gets EF_MIPS_ARCH_2 flag. From the other side mips3 and mips32
320// are incompatible because nor mips3 is a parent for misp32, nor mips32
321// is a parent for mips3.
322static uint32_t getArchFlags(ArrayRef<FileFlags> Files) {
323 uint32_t Ret = Files[0].Flags & (EF_MIPS_ARCH | EF_MIPS_MACH);
324
325 for (const FileFlags &F : Files.slice(1)) {
326 uint32_t New = F.Flags & (EF_MIPS_ARCH | EF_MIPS_MACH);
327
328 // Warn about linking BERI/CHERI and non BERI/CHERI
329 // This is required because the default -mcpu=mips4 instruction scheduling
330 // results in lots of pipeline bubbles that prevent MIPS performance from
331 // being comparable to CHERI performance.
332 if (isBeriOrCheri(Ret) != isBeriOrCheri(New)) {
333 warn("linking files compiled for BERI/CHERI and non-BERI/CHERI can "
334 "result in surprising performance:\n>>> " +
335 toString(Files[0].File) + ": " + getFullArchName(Ret) + "\n>>> " +
336 toString(F.File) + ": " + getFullArchName(New));
337 }
338
339 // Check ISA compatibility.
340 if (isArchMatched(New, Ret))
341 continue;
342 if (!isArchMatched(Ret, New)) {
343 error("incompatible target ISA:\n>>> " + toString(Files[0].File) + ": " +
344 getFullArchName(Ret) + "\n>>> " + toString(F.File) + ": " +
345 getFullArchName(New));
346 return 0;
347 }
348 Ret = New;
349 }
350 return Ret;
351}

353template <class ELFT> uint32_t elf::calcMipsEFlags() {
354 std::vector<FileFlags> V;
355 for (InputFile *F : ObjectFiles)
356 V.push_back({F, cast<ObjFile<ELFT>>(F)->getObj().getHeader()->e_flags});
357 if (V.empty())
358 return 0;
359 checkFlags(V);
360 return getMiscFlags(V) | getPicFlags(V) | getArchFlags(V);
361}

363static int compareMipsFpAbi(uint8_t FpA, uint8_t FpB) {
364 if (FpA == FpB)
365 return 0;
366 if (FpB == Mips::Val_GNU_MIPS_ABI_FP_ANY)
367 return 1;
368 if (FpB == Mips::Val_GNU_MIPS_ABI_FP_64A &&
369 FpA == Mips::Val_GNU_MIPS_ABI_FP_64)
370 return 1;
371 if (FpB != Mips::Val_GNU_MIPS_ABI_FP_XX)
372 return -1;
373 if (FpA == Mips::Val_GNU_MIPS_ABI_FP_DOUBLE ||
374 FpA == Mips::Val_GNU_MIPS_ABI_FP_64 ||
375 FpA == Mips::Val_GNU_MIPS_ABI_FP_64A)
376 return 1;
377 return -1;
378}

403uint8_t elf::getMipsFpAbiFlag(uint8_t OldFlag, StringRef OldFile,
404 uint8_t NewFlag, StringRef NewFile) {
405 if (compareMipsFpAbi(NewFlag, OldFlag) >= 0)
406 return NewFlag;
407 if (compareMipsFpAbi(OldFlag, NewFlag) < 0)
408 error(NewFile + ": floating point ABI '" + getMipsFpAbiName(NewFlag) +
409 "' is incompatible with target floating point ABI '" +
410 getMipsFpAbiName(OldFlag) + "'");
411 return OldFlag;
412}

451uint8_t elf::getMipsIsaExt(uint64_t OldExt, StringRef OldFile, uint64_t NewExt,
452 StringRef NewFile) {
453 if (OldExt == NewExt)
454 return NewExt;
455
456 // ISA_EXT is different, now check if we want to allow this
457 // No ext -> any ext is always fine (XXXAR: well at least for now it is)
458 // TODO: require isa_ext to be set for CHERI purecap programs in the future
459 if (OldExt == Mips::AFL_EXT_NONE)
460 return NewExt;
461 if (NewExt == Mips::AFL_EXT_NONE)
462 return OldExt;
463 Mips::AFL_EXT CheriABIs[] = {
464 Mips::AFL_EXT_CHERI_ABI_LEGACY,
465 Mips::AFL_EXT_CHERI_ABI_PLT,
466 Mips::AFL_EXT_CHERI_ABI_PCREL,
467 Mips::AFL_EXT_CHERI_ABI_FNDESC,
468 };
469 if (llvm::is_contained(CheriABIs, OldExt) ||
470 llvm::is_contained(CheriABIs, NewExt)) {
471 // incompatible cheri purecap ABIs:
472 error("incompatible pure-capability ABIs:\n>>> " + OldFile + " uses " +
473 getMipsIsaExtName((Mips::AFL_EXT)OldExt) + "\n>>> " + NewFile +
474 " uses " + getMipsIsaExtName((Mips::AFL_EXT)NewExt));
475 // return NewExt to get sensible error messages with multiple mismatches
476 return NewExt;
477 }
478 // non-cheri isa_ext -> just return the maximum
479 return std::max(OldExt, NewExt);
480}

482static Mips::AFL_EXT cheriFlagsToAFL_EXT(uint64_t CheriFlags) {
483 assert(CheriFlags < DF_MIPS_CHERI_ABI_MASK);
484 switch (CheriFlags) {
485 case DF_MIPS_CHERI_ABI_LEGACY:
486 return Mips::AFL_EXT::AFL_EXT_CHERI_ABI_LEGACY;
487 case DF_MIPS_CHERI_ABI_PCREL:
488 return Mips::AFL_EXT::AFL_EXT_CHERI_ABI_PCREL;
489 case DF_MIPS_CHERI_ABI_PLT:
490 return Mips::AFL_EXT::AFL_EXT_CHERI_ABI_PLT;
491 case DF_MIPS_CHERI_ABI_FNDESC:
492 return Mips::AFL_EXT::AFL_EXT_CHERI_ABI_FNDESC;
493 default:
494 llvm_unreachable("Invalid ABI");
495 }
496 return Mips::AFL_EXT::AFL_EXT_NONE;
497}

499void elf::checkMipsShlibCompatible(InputFile *F, uint64_t InputCheriFlags,
500 uint64_t TargetCheriFlags) {
501 const uint32_t TargetABI = Config->EFlags & (EF_MIPS_ABI | EF_MIPS_ABI2);
502 assert(F->EMachine == Config->EMachine);
503 uint32_t ABI = F->EFlags & (EF_MIPS_ABI | EF_MIPS_ABI2);
504 // Mips can't link against cheriabi and the other way around
505 if ((Config->isCheriABI() && ABI != EF_MIPS_ABI_CHERIABI) ||
506 (!Config->isCheriABI() && ABI == EF_MIPS_ABI_CHERIABI)) {
507 // assert(errorCount() && "Should have already caused an errors");
508 // llvm_unreachable("Should have been checked earlier!");
509 if (!errorCount())
510 error(toString(F) + ": ABI '" + getAbiName(ABI) +
511 "' is incompatible with target ABI: " + getAbiName(TargetABI));
512 } else {
513 uint64_t InputCheriAbi = InputCheriFlags & DF_MIPS_CHERI_ABI_MASK;
514 uint64_t TargetCheriAbi = TargetCheriFlags & DF_MIPS_CHERI_ABI_MASK;
515 if (InputCheriAbi != TargetCheriAbi) {
516 std::string Msg = "target pure-capability ABI " +
517 getMipsIsaExtName(cheriFlagsToAFL_EXT(TargetCheriAbi)) +
518 " is incompatible with linked shared library\n>>> " +
519 toString(F) + " uses " +
520 getMipsIsaExtName(cheriFlagsToAFL_EXT(InputCheriAbi));
521 // mixing legacy/non-legacy is an error, anything a warning
522 if (InputCheriAbi == DF_MIPS_CHERI_ABI_LEGACY ||
523 TargetCheriAbi == DF_MIPS_CHERI_ABI_LEGACY)
524 error(Msg);
525 else
526 warn(Msg);
527 }
528 }
