51static uint64_t getOutputSectionVA(SectionBase *InputSec, StringRef Loc) {
52 if (OutputSection *OS = InputSec->getOutputSection())
53 return OS->Addr;
54 error(Loc + ": unable to evaluate expression: input section " +
55 InputSec->Name + " has no output section assigned");
56 return 0;
57}
58
59uint64_t ExprValue::getValue() const {
60 if (Sec)
61 return alignTo(Sec->getOffset(Val) + getOutputSectionVA(Sec, Loc),
62 Alignment);
63 return alignTo(Val, Alignment);
64}
65
66uint64_t ExprValue::getSecAddr() const {
67 if (Sec)
68 return Sec->getOffset(0) + getOutputSectionVA(Sec, Loc);
69 return 0;
70}
71
72uint64_t ExprValue::getSectionOffset() const {
73 // If the alignment is trivial, we don't have to compute the full
74 // value to know the offset. This allows this function to succeed in
75 // cases where the output section is not yet known.
76 if (Alignment == 1 && (!Sec || !Sec->getOutputSection()))
77 return Val;
78 return getValue() - getSecAddr();
79}

81OutputSection *LinkerScript::createOutputSection(StringRef Name,
82 StringRef Location) {
83 OutputSection *&SecRef = NameToOutputSection[Name];
84 OutputSection *Sec;
85 if (SecRef && SecRef->Location.empty()) {
86 // There was a forward reference.
87 Sec = SecRef;
88 } else {
89 Sec = make<OutputSection>(Name, SHT_PROGBITS, 0);
90 if (!SecRef)
91 SecRef = Sec;
92 }
93 Sec->Location = Location;
94 return Sec;
95}

97OutputSection *LinkerScript::getOrCreateOutputSection(StringRef Name) {
98 OutputSection *&CmdRef = NameToOutputSection[Name];
99 if (!CmdRef)
100 CmdRef = make<OutputSection>(Name, SHT_PROGBITS, 0);
101 return CmdRef;
102}

104// Expands the memory region by the specified size.
105static void expandMemoryRegion(MemoryRegion *MemRegion, uint64_t Size,
106 StringRef RegionName, StringRef SecName) {
107 MemRegion->CurPos += Size;
108 uint64_t NewSize = MemRegion->CurPos - MemRegion->Origin;
109 if (NewSize > MemRegion->Length)
110 error("section '" + SecName + "' will not fit in region '" + RegionName +
111 "': overflowed by " + Twine(NewSize - MemRegion->Length) + " bytes");
112}
113
114void LinkerScript::expandMemoryRegions(uint64_t Size) {
115 if (Ctx->MemRegion)
116 expandMemoryRegion(Ctx->MemRegion, Size, Ctx->MemRegion->Name,
117 Ctx->OutSec->Name);
118 // Only expand the LMARegion if it is different from MemRegion.
119 if (Ctx->LMARegion && Ctx->MemRegion != Ctx->LMARegion)
120 expandMemoryRegion(Ctx->LMARegion, Size, Ctx->LMARegion->Name,
121 Ctx->OutSec->Name);
122}
123
124void LinkerScript::expandOutputSection(uint64_t Size) {
125 Ctx->OutSec->Size += Size;
126 expandMemoryRegions(Size);
127}

129void LinkerScript::setDot(Expr E, const Twine &Loc, bool InSec) {
130 uint64_t Val = E().getValue();
131 if (Val < Dot && InSec)
132 error(Loc + ": unable to move location counter backward for: " +
133 Ctx->OutSec->Name);
134
135 // Update to location counter means update to section size.
136 if (InSec)
137 expandOutputSection(Val - Dot);
138
139 Dot = Val;
140}

142// Used for handling linker symbol assignments, for both finalizing
143// their values and doing early declarations. Returns true if symbol
144// should be defined from linker script.
145static bool shouldDefineSym(SymbolAssignment *Cmd) {
146 if (Cmd->Name == ".")
147 return false;
148
149 if (!Cmd->Provide)
150 return true;
151
152 // If a symbol was in PROVIDE(), we need to define it only
153 // when it is a referenced undefined symbol.
154 Symbol *B = Symtab->find(Cmd->Name);
155 if (B && !B->isDefined())
156 return true;
157 return false;
158}
159
160// This function is called from processSectionCommands,
161// while we are fixing the output section layout.
162void LinkerScript::addSymbol(SymbolAssignment *Cmd) {
163 if (!shouldDefineSym(Cmd))
164 return;
165
166 // Define a symbol.
167 ExprValue Value = Cmd->Expression();
168 SectionBase *Sec = Value.isAbsolute() ? nullptr : Value.Sec;
169 uint8_t Visibility = Cmd->Hidden ? STV_HIDDEN : STV_DEFAULT;
170
171 // When this function is called, section addresses have not been
172 // fixed yet. So, we may or may not know the value of the RHS
173 // expression.
174 //
175 // For example, if an expression is `x = 42`, we know x is always 42.
176 // However, if an expression is `x = .`, there's no way to know its
177 // value at the moment.
178 //
179 // We want to set symbol values early if we can. This allows us to
180 // use symbols as variables in linker scripts. Doing so allows us to
181 // write expressions like this: `alignment = 16; . = ALIGN(., alignment)`.
182 uint64_t SymValue = Value.Sec ? 0 : Value.getValue();
183
184 Defined New(nullptr, Cmd->Name, STB_GLOBAL, Visibility, STT_NOTYPE, SymValue,
185 0, Sec);
186
187 Symbol *Sym = Symtab->insert(Cmd->Name);
188 Sym->mergeProperties(New);
189 Sym->replace(New);
190 Cmd->Sym = cast<Defined>(Sym);
191}
192
193// This function is called from LinkerScript::declareSymbols.
194// It creates a placeholder symbol if needed.
195static void declareSymbol(SymbolAssignment *Cmd) {
196 if (!shouldDefineSym(Cmd))
197 return;
198
199 uint8_t Visibility = Cmd->Hidden ? STV_HIDDEN : STV_DEFAULT;
200 Defined New(nullptr, Cmd->Name, STB_GLOBAL, Visibility, STT_NOTYPE, 0, 0,
201 nullptr);
202
203 // We can't calculate final value right now.
204 Symbol *Sym = Symtab->insert(Cmd->Name);
205 Sym->mergeProperties(New);
206 Sym->replace(New);
207
208 Cmd->Sym = cast<Defined>(Sym);
209 Cmd->Provide = false;
210 Sym->ScriptDefined = true;
211}

213// This method is used to handle INSERT AFTER statement. Here we rebuild
214// the list of script commands to mix sections inserted into.
215void LinkerScript::processInsertCommands() {
216 std::vector<BaseCommand *> V;
217 auto Insert = [&](std::vector<BaseCommand *> &From) {
218 V.insert(V.end(), From.begin(), From.end());
219 From.clear();
220 };
221
222 for (BaseCommand *Base : SectionCommands) {
223 if (auto *OS = dyn_cast<OutputSection>(Base)) {
224 Insert(InsertBeforeCommands[OS->Name]);
225 V.push_back(Base);
226 Insert(InsertAfterCommands[OS->Name]);
227 continue;
228 }
229 V.push_back(Base);
230 }
231
232 for (auto &Cmds : {InsertBeforeCommands, InsertAfterCommands})
233 for (const std::pair<StringRef, std::vector<BaseCommand *>> &P : Cmds)
234 if (!P.second.empty())
235 error("unable to INSERT AFTER/BEFORE " + P.first +
236 ": section not defined");
237
238 SectionCommands = std::move(V);
239}

241// Symbols defined in script should not be inlined by LTO. At the same time
242// we don't know their final values until late stages of link. Here we scan
243// over symbol assignment commands and create placeholder symbols if needed.
244void LinkerScript::declareSymbols() {
245 assert(!Ctx);
246 for (BaseCommand *Base : SectionCommands) {
247 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base)) {
248 declareSymbol(Cmd);
249 continue;
250 }
251
252 // If the output section directive has constraints,
253 // we can't say for sure if it is going to be included or not.
254 // Skip such sections for now. Improve the checks if we ever
255 // need symbols from that sections to be declared early.
256 auto *Sec = cast<OutputSection>(Base);
257 if (Sec->Constraint != ConstraintKind::NoConstraint)
258 continue;
259 for (BaseCommand *Base2 : Sec->SectionCommands)
260 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base2))
261 declareSymbol(Cmd);
262 }
263}

265// This function is called from assignAddresses, while we are
266// fixing the output section addresses. This function is supposed
267// to set the final value for a given symbol assignment.
268void LinkerScript::assignSymbol(SymbolAssignment *Cmd, bool InSec) {
269 if (Cmd->Name == ".") {
270 setDot(Cmd->Expression, Cmd->Location, InSec);
271 return;
272 }
273
274 if (!Cmd->Sym)
275 return;
276
277 ExprValue V = Cmd->Expression();
278 if (V.isAbsolute()) {
279 Cmd->Sym->Section = nullptr;
280 Cmd->Sym->Value = V.getValue();
281 } else {
282 Cmd->Sym->Section = V.Sec;
283 Cmd->Sym->Value = V.getSectionOffset();
284 }
285}

295bool LinkerScript::shouldKeep(InputSectionBase *S) {
296 if (KeptSections.empty())
297 return false;
298 std::string Filename = getFilename(S->File);
299 for (InputSectionDescription *ID : KeptSections)
300 if (ID->FilePat.match(Filename))
301 for (SectionPattern &P : ID->SectionPatterns)
302 if (P.SectionPat.match(S->Name))
303 return true;
304 return false;
305}

307// A helper function for the SORT() command.
308static std::function<bool(InputSectionBase *, InputSectionBase *)>
309getComparator(SortSectionPolicy K) {
310 switch (K) {
311 case SortSectionPolicy::Alignment:
312 return [](InputSectionBase *A, InputSectionBase *B) {
313 // ">" is not a mistake. Sections with larger alignments are placed
314 // before sections with smaller alignments in order to reduce the
315 // amount of padding necessary. This is compatible with GNU.
316 return A->Alignment > B->Alignment;
317 };
318 case SortSectionPolicy::Name:
319 return [](InputSectionBase *A, InputSectionBase *B) {
320 return A->Name < B->Name;
321 };
322 case SortSectionPolicy::Priority:
323 return [](InputSectionBase *A, InputSectionBase *B) {
324 return getPriority(A->Name) < getPriority(B->Name);
325 };
326 default:
327 llvm_unreachable("unknown sort policy");
328 }
329}

331// A helper function for the SORT() command.
332static bool matchConstraints(ArrayRef<InputSection *> Sections,
333 ConstraintKind Kind) {
334 if (Kind == ConstraintKind::NoConstraint)
335 return true;
336
337 bool IsRW = llvm::any_of(
338 Sections, [](InputSection *Sec) { return Sec->Flags & SHF_WRITE; });
339
340 return (IsRW && Kind == ConstraintKind::ReadWrite) ||
341 (!IsRW && Kind == ConstraintKind::ReadOnly);
342}

344static void sortSections(MutableArrayRef<InputSection *> Vec,
345 SortSectionPolicy K) {
346 if (K != SortSectionPolicy::Default && K != SortSectionPolicy::None)
347 llvm::stable_sort(Vec, getComparator(K));
348}

350// Sort sections as instructed by SORT-family commands and --sort-section
351// option. Because SORT-family commands can be nested at most two depth
352// (e.g. SORT_BY_NAME(SORT_BY_ALIGNMENT(.text.*))) and because the command
353// line option is respected even if a SORT command is given, the exact
354// behavior we have here is a bit complicated. Here are the rules.
355//
356// 1. If two SORT commands are given, --sort-section is ignored.
357// 2. If one SORT command is given, and if it is not SORT_NONE,
358// --sort-section is handled as an inner SORT command.
359// 3. If one SORT command is given, and if it is SORT_NONE, don't sort.
360// 4. If no SORT command is given, sort according to --sort-section.
361static void sortInputSections(MutableArrayRef<InputSection *> Vec,
362 const SectionPattern &Pat) {
363 if (Pat.SortOuter == SortSectionPolicy::None)
364 return;
365
366 if (Pat.SortInner == SortSectionPolicy::Default)
367 sortSections(Vec, Config->SortSection);
368 else
369 sortSections(Vec, Pat.SortInner);
370 sortSections(Vec, Pat.SortOuter);
371}

373// Compute and remember which sections the InputSectionDescription matches.
374std::vector<InputSection *>
375LinkerScript::computeInputSections(const InputSectionDescription *Cmd) {
376 std::vector<InputSection *> Ret;
377
378 // Collects all sections that satisfy constraints of Cmd.
379 for (const SectionPattern &Pat : Cmd->SectionPatterns) {
380 size_t SizeBefore = Ret.size();
381
382 for (InputSectionBase *Sec : InputSections) {
383 if (!Sec->isLive() || Sec->Assigned)
384 continue;
385
386 // For -emit-relocs we have to ignore entries like
387 // .rela.dyn : { *(.rela.data) }
388 // which are common because they are in the default bfd script.
389 // We do not ignore SHT_REL[A] linker-synthesized sections here because
390 // want to support scripts that do custom layout for them.
391 if (auto *IS = dyn_cast<InputSection>(Sec))
392 if (IS->getRelocatedSection())
393 continue;
394
395 std::string Filename = getFilename(Sec->File);
396 if (!Cmd->FilePat.match(Filename) ||
397 Pat.ExcludedFilePat.match(Filename) ||
398 !Pat.SectionPat.match(Sec->Name))
399 continue;
400
401 // It is safe to assume that Sec is an InputSection
402 // because mergeable or EH input sections have already been
403 // handled and eliminated.
404 Ret.push_back(cast<InputSection>(Sec));
405 Sec->Assigned = true;
406 }
407
408 sortInputSections(MutableArrayRef<InputSection *>(Ret).slice(SizeBefore),
409 Pat);
410 }
411 return Ret;
412}

414void LinkerScript::discard(ArrayRef<InputSection *> V) {
415 for (InputSection *S : V) {
416 if (S == In.ShStrTab || S == Main->RelaDyn || S == Main->RelrDyn)
417 error("discarding " + S->Name + " section is not allowed");
418
419 // You can discard .hash and .gnu.hash sections by linker scripts. Since
420 // they are synthesized sections, we need to handle them differently than
421 // other regular sections.
422 if (S == Main->GnuHashTab)
423 Main->GnuHashTab = nullptr;
424 if (S == Main->HashTab)
425 Main->HashTab = nullptr;
426
427 S->Assigned = false;
428 S->markDead();
429 discard(S->DependentSections);
430 }
431}

433std::vector<InputSection *>
434LinkerScript::createInputSectionList(OutputSection &OutCmd) {
435 std::vector<InputSection *> Ret;
436
437 for (BaseCommand *Base : OutCmd.SectionCommands) {
438 if (auto *Cmd = dyn_cast<InputSectionDescription>(Base)) {
439 Cmd->Sections = computeInputSections(Cmd);
440 Ret.insert(Ret.end(), Cmd->Sections.begin(), Cmd->Sections.end());
441 }
442 }
443 return Ret;
444}

446void LinkerScript::processSectionCommands() {
447 // A symbol can be assigned before any section is mentioned in the linker
448 // script. In an DSO, the symbol values are addresses, so the only important
449 // section values are:
450 // * SHN_UNDEF
451 // * SHN_ABS
452 // * Any value meaning a regular section.
453 // To handle that, create a dummy aether section that fills the void before
454 // the linker scripts switches to another section. It has an index of one
455 // which will map to whatever the first actual section is.
456 Aether = make<OutputSection>("", 0, SHF_ALLOC);
457 Aether->SectionIndex = 1;
458
459 // Ctx captures the local AddressState and makes it accessible deliberately.
460 // This is needed as there are some cases where we cannot just
461 // thread the current state through to a lambda function created by the
462 // script parser.
463 auto Deleter = make_unique<AddressState>();
464 Ctx = Deleter.get();
465 Ctx->OutSec = Aether;
466
467 size_t I = 0;
468 // Add input sections to output sections.
469 for (BaseCommand *Base : SectionCommands) {
470 // Handle symbol assignments outside of any output section.
471 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base)) {
472 addSymbol(Cmd);
473 continue;
474 }
475
476 if (auto *Sec = dyn_cast<OutputSection>(Base)) {
477 std::vector<InputSection *> V = createInputSectionList(*Sec);
478
479 // The output section name `/DISCARD/' is special.
480 // Any input section assigned to it is discarded.
481 if (Sec->Name == "/DISCARD/") {
482 discard(V);
483 Sec->SectionCommands.clear();
484 continue;
485 }
486
487 // This is for ONLY_IF_RO and ONLY_IF_RW. An output section directive
488 // ".foo : ONLY_IF_R[OW] { ... }" is handled only if all member input
489 // sections satisfy a given constraint. If not, a directive is handled
490 // as if it wasn't present from the beginning.
491 //
492 // Because we'll iterate over SectionCommands many more times, the easy
493 // way to "make it as if it wasn't present" is to make it empty.
494 if (!matchConstraints(V, Sec->Constraint)) {
495 for (InputSectionBase *S : V)
496 S->Assigned = false;
497 Sec->SectionCommands.clear();
498 continue;
499 }
500
501 // A directive may contain symbol definitions like this:
502 // ".foo : { ...; bar = .; }". Handle them.
503 for (BaseCommand *Base : Sec->SectionCommands)
504 if (auto *OutCmd = dyn_cast<SymbolAssignment>(Base))
505 addSymbol(OutCmd);
506
507 // Handle subalign (e.g. ".foo : SUBALIGN(32) { ... }"). If subalign
508 // is given, input sections are aligned to that value, whether the
509 // given value is larger or smaller than the original section alignment.
510 if (Sec->SubalignExpr) {
511 uint32_t Subalign = Sec->SubalignExpr().getValue();
512 for (InputSectionBase *S : V)
513 S->Alignment = Subalign;
514 }
515
516 // Add input sections to an output section.
517 for (InputSection *S : V)
518 Sec->addSection(S);
519
520 Sec->SectionIndex = I++;
521 if (Sec->Noload)
522 Sec->Type = SHT_NOBITS;
523 if (Sec->NonAlloc)
524 Sec->Flags &= ~(uint64_t)SHF_ALLOC;
525 }
526 }
527 Ctx = nullptr;
528}

530static OutputSection *findByName(ArrayRef<BaseCommand *> Vec,
531 StringRef Name) {
532 for (BaseCommand *Base : Vec)
533 if (auto *Sec = dyn_cast<OutputSection>(Base))
534 if (Sec->Name == Name)
535 return Sec;
536 return nullptr;
537}

539static OutputSection *createSection(InputSectionBase *IS,
540 StringRef OutsecName) {
541 OutputSection *Sec = Script->createOutputSection(OutsecName, "<internal>");
542 if (OutsecName == "__cap_relocs" && !Config->RelativeCapRelocsOnly) {
543 Sec->Flags |= SHF_WRITE;
544 }
545 Sec->addSection(cast<InputSection>(IS));
546 return Sec;
547}

549static OutputSection *
550addInputSec(StringMap<TinyPtrVector<OutputSection *>> &Map,
551 InputSectionBase *IS, StringRef OutsecName) {
552 // Sections with SHT_GROUP or SHF_GROUP attributes reach here only when the -r
553 // option is given. A section with SHT_GROUP defines a "section group", and
554 // its members have SHF_GROUP attribute. Usually these flags have already been
555 // stripped by InputFiles.cpp as section groups are processed and uniquified.
556 // However, for the -r option, we want to pass through all section groups
557 // as-is because adding/removing members or merging them with other groups
558 // change their semantics.
559 if (IS->Type == SHT_GROUP || (IS->Flags & SHF_GROUP))
560 return createSection(IS, OutsecName);
561
562 // Imagine .zed : { *(.foo) *(.bar) } script. Both foo and bar may have
563 // relocation sections .rela.foo and .rela.bar for example. Most tools do
564 // not allow multiple REL[A] sections for output section. Hence we
565 // should combine these relocation sections into single output.
566 // We skip synthetic sections because it can be .rela.dyn/.rela.plt or any
567 // other REL[A] sections created by linker itself.
568 if (!isa<SyntheticSection>(IS) &&
569 (IS->Type == SHT_REL || IS->Type == SHT_RELA)) {
570 auto *Sec = cast<InputSection>(IS);
571 OutputSection *Out = Sec->getRelocatedSection()->getOutputSection();
572
573 if (Out->RelocationSection) {
574 Out->RelocationSection->addSection(Sec);
575 return nullptr;
576 }
577
578 Out->RelocationSection = createSection(IS, OutsecName);
579 return Out->RelocationSection;
580 }
581
582 // When control reaches here, mergeable sections have already been merged into
583 // synthetic sections. For relocatable case we want to create one output
584 // section per syntetic section so that they have a valid sh_entsize.
585 if (Config->Relocatable && (IS->Flags & SHF_MERGE))
586 return createSection(IS, OutsecName);
587
588 // The ELF spec just says
589 // ----------------------------------------------------------------
590 // In the first phase, input sections that match in name, type and
591 // attribute flags should be concatenated into single sections.
592 // ----------------------------------------------------------------
593 //
594 // However, it is clear that at least some flags have to be ignored for
595 // section merging. At the very least SHF_GROUP and SHF_COMPRESSED have to be
596 // ignored. We should not have two output .text sections just because one was
597 // in a group and another was not for example.
598 //
599 // It also seems that wording was a late addition and didn't get the
600 // necessary scrutiny.
601 //
602 // Merging sections with different flags is expected by some users. One
603 // reason is that if one file has
604 //
605 // int *const bar __attribute__((section(".foo"))) = (int *)0;
606 //
607 // gcc with -fPIC will produce a read only .foo section. But if another
608 // file has
609 //
610 // int zed;
611 // int *const bar __attribute__((section(".foo"))) = (int *)&zed;
612 //
613 // gcc with -fPIC will produce a read write section.
614 //
615 // Last but not least, when using linker script the merge rules are forced by
616 // the script. Unfortunately, linker scripts are name based. This means that
617 // expressions like *(.foo*) can refer to multiple input sections with
618 // different flags. We cannot put them in different output sections or we
619 // would produce wrong results for
620 //
621 // start = .; *(.foo.*) end = .; *(.bar)
622 //
623 // and a mapping of .foo1 and .bar1 to one section and .foo2 and .bar2 to
624 // another. The problem is that there is no way to layout those output
625 // sections such that the .foo sections are the only thing between the start
626 // and end symbols.
627 //
628 // Given the above issues, we instead merge sections by name and error on
629 // incompatible types and flags.
630 TinyPtrVector<OutputSection *> &V = Map[OutsecName];
631 for (OutputSection *Sec : V) {
632 if (Sec->Partition != IS->Partition)
633 continue;
634 Sec->addSection(cast<InputSection>(IS));
635 return nullptr;
636 }
637
638 OutputSection *Sec = createSection(IS, OutsecName);
639 V.push_back(Sec);
640 return Sec;
641}

643// Add sections that didn't match any sections command.
644void LinkerScript::addOrphanSections() {
645 StringMap<TinyPtrVector<OutputSection *>> Map;
646 std::vector<OutputSection *> V;
647
648 auto Add = [&](InputSectionBase *S) {
649 if (!S->isLive() || S->Parent)
650 return;
651
652 StringRef Name = getOutputSectionName(S);
653
654 if (Config->OrphanHandling == OrphanHandlingPolicy::Error)
655 error(toString(S) + " is being placed in '" + Name + "'");
656 else if (Config->OrphanHandling == OrphanHandlingPolicy::Warn)
657 warn(toString(S) + " is being placed in '" + Name + "'");
658
659 if (OutputSection *Sec = findByName(SectionCommands, Name)) {
660 Sec->addSection(cast<InputSection>(S));
661 return;
662 }
663
664 if (OutputSection *OS = addInputSec(Map, S, Name))
665 V.push_back(OS);
666 assert(S->getOutputSection()->SectionIndex == UINT32_MAX);
667 };
668
669 // For futher --emit-reloc handling code we need target output section
670 // to be created before we create relocation output section, so we want
671 // to create target sections first. We do not want priority handling
672 // for synthetic sections because them are special.
673 for (InputSectionBase *IS : InputSections) {
674 if (auto *Sec = dyn_cast<InputSection>(IS))
675 if (InputSectionBase *Rel = Sec->getRelocatedSection())
676 if (auto *RelIS = dyn_cast_or_null<InputSectionBase>(Rel->Parent))
677 Add(RelIS);
678 Add(IS);
679 }
680
681 // If no SECTIONS command was given, we should insert sections commands
682 // before others, so that we can handle scripts which refers them,
683 // for example: "foo = ABSOLUTE(ADDR(.text)));".
684 // When SECTIONS command is present we just add all orphans to the end.
685 if (HasSectionsCommand)
686 SectionCommands.insert(SectionCommands.end(), V.begin(), V.end());
687 else
688 SectionCommands.insert(SectionCommands.begin(), V.begin(), V.end());
689}

691uint64_t LinkerScript::advance(uint64_t Size, unsigned Alignment) {
692 bool IsTbss =
693 (Ctx->OutSec->Flags & SHF_TLS) && Ctx->OutSec->Type == SHT_NOBITS;
694 uint64_t Start = IsTbss ? Dot + Ctx->ThreadBssOffset : Dot;
695 Start = alignTo(Start, Alignment);
696 uint64_t End = Start + Size;
697
698 if (IsTbss)
699 Ctx->ThreadBssOffset = End - Dot;
700 else
701 Dot = End;
702 return End;
703}

705void LinkerScript::output(InputSection *S) {
706 assert(Ctx->OutSec == S->getParent());
707 uint64_t Before = advance(0, 1);
708 uint64_t Pos = advance(S->getSize(), S->Alignment);
709 S->OutSecOff = Pos - S->getSize() - Ctx->OutSec->Addr;
710
711 // Update output section size after adding each section. This is so that
712 // SIZEOF works correctly in the case below:
713 // .foo { *(.aaa) a = SIZEOF(.foo); *(.bbb) }
714 expandOutputSection(Pos - Before);
715}

717void LinkerScript::switchTo(OutputSection *Sec) {
718 Ctx->OutSec = Sec;
719
720 uint64_t Before = advance(0, 1);
721 Ctx->OutSec->Addr = advance(0, Ctx->OutSec->Alignment);
722 expandMemoryRegions(Ctx->OutSec->Addr - Before);
723}

725// This function searches for a memory region to place the given output
726// section in. If found, a pointer to the appropriate memory region is
727// returned. Otherwise, a nullptr is returned.
728MemoryRegion *LinkerScript::findMemoryRegion(OutputSection *Sec) {
729 // If a memory region name was specified in the output section command,
730 // then try to find that region first.
731 if (!Sec->MemoryRegionName.empty()) {
732 if (MemoryRegion *M = MemoryRegions.lookup(Sec->MemoryRegionName))
733 return M;
734 error("memory region '" + Sec->MemoryRegionName + "' not declared");
735 return nullptr;
736 }
737
738 // If at least one memory region is defined, all sections must
739 // belong to some memory region. Otherwise, we don't need to do
740 // anything for memory regions.
741 if (MemoryRegions.empty())
742 return nullptr;
743
744 // See if a region can be found by matching section flags.
745 for (auto &Pair : MemoryRegions) {
746 MemoryRegion *M = Pair.second;
747 if ((M->Flags & Sec->Flags) && (M->NegFlags & Sec->Flags) == 0)
748 return M;
749 }
750
751 // Otherwise, no suitable region was found.
752 if (Sec->Flags & SHF_ALLOC)
753 error("no memory region specified for section '" + Sec->Name + "'");
754 return nullptr;
755}

757static OutputSection *findFirstSection(PhdrEntry *Load) {
758 for (OutputSection *Sec : OutputSections)
759 if (Sec->PtLoad == Load)
760 return Sec;
761 return nullptr;
762}

764// This function assigns offsets to input sections and an output section
765// for a single sections command (e.g. ".text { *(.text); }").
766void LinkerScript::assignOffsets(OutputSection *Sec) {
767 if (!(Sec->Flags & SHF_ALLOC))
768 Dot = 0;
769
770 Ctx->MemRegion = Sec->MemRegion;
771 Ctx->LMARegion = Sec->LMARegion;
772 if (Ctx->MemRegion)
773 Dot = Ctx->MemRegion->CurPos;
774
775 if ((Sec->Flags & SHF_ALLOC) && Sec->AddrExpr)
776 setDot(Sec->AddrExpr, Sec->Location, false);
777
778 switchTo(Sec);
779
780 if (Sec->LMAExpr)
781 Ctx->LMAOffset = Sec->LMAExpr().getValue() - Dot;
782
783 if (MemoryRegion *MR = Sec->LMARegion)
784 Ctx->LMAOffset = MR->CurPos - Dot;
785
786 // If neither AT nor AT> is specified for an allocatable section, the linker
787 // will set the LMA such that the difference between VMA and LMA for the
788 // section is the same as the preceding output section in the same region
789 // https://sourceware.org/binutils/docs-2.20/ld/Output-Section-LMA.html
790 // This, however, should only be done by the first "non-header" section
791 // in the segment.
792 if (PhdrEntry *L = Ctx->OutSec->PtLoad)
793 if (Sec == findFirstSection(L))
794 L->LMAOffset = Ctx->LMAOffset;
795
796 // We can call this method multiple times during the creation of
797 // thunks and want to start over calculation each time.
798 Sec->Size = 0;
799
800 // We visited SectionsCommands from processSectionCommands to
801 // layout sections. Now, we visit SectionsCommands again to fix
802 // section offsets.
803 for (BaseCommand *Base : Sec->SectionCommands) {
804 // This handles the assignments to symbol or to the dot.
805 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base)) {
806 Cmd->Addr = Dot;
807 assignSymbol(Cmd, true);
808 Cmd->Size = Dot - Cmd->Addr;
809 continue;
810 }
811
812 // Handle BYTE(), SHORT(), LONG(), or QUAD().
813 if (auto *Cmd = dyn_cast<ByteCommand>(Base)) {
814 Cmd->Offset = Dot - Ctx->OutSec->Addr;
815 Dot += Cmd->Size;
816 expandOutputSection(Cmd->Size);
817 continue;
818 }
819
820 // Handle a single input section description command.
821 // It calculates and assigns the offsets for each section and also
822 // updates the output section size.
823 for (InputSection *Sec : cast<InputSectionDescription>(Base)->Sections)
824 output(Sec);
825 }
826}

828static bool isDiscardable(OutputSection &Sec) {
829 if (Sec.Name == "/DISCARD/")
830 return true;
831
832 // We do not remove empty sections that are explicitly
833 // assigned to any segment.
834 if (!Sec.Phdrs.empty())
835 return false;
836
837 // We do not want to remove OutputSections with expressions that reference
838 // symbols even if the OutputSection is empty. We want to ensure that the
839 // expressions can be evaluated and report an error if they cannot.
840 if (Sec.ExpressionsUseSymbols)
841 return false;
842
843 // OutputSections may be referenced by name in ADDR and LOADADDR expressions,
844 // as an empty Section can has a valid VMA and LMA we keep the OutputSection
845 // to maintain the integrity of the other Expression.
846 if (Sec.UsedInExpression)
847 return false;
848
849 for (BaseCommand *Base : Sec.SectionCommands) {
850 if (auto Cmd = dyn_cast<SymbolAssignment>(Base))
851 // Don't create empty output sections just for unreferenced PROVIDE
852 // symbols.
853 if (Cmd->Name != "." && !Cmd->Sym)
854 continue;
855
856 if (!isa<InputSectionDescription>(*Base))
857 return false;
858 }
859 return true;
860}

862void LinkerScript::adjustSectionsBeforeSorting() {
863 // If the output section contains only symbol assignments, create a
864 // corresponding output section. The issue is what to do with linker script
865 // like ".foo : { symbol = 42; }". One option would be to convert it to
866 // "symbol = 42;". That is, move the symbol out of the empty section
867 // description. That seems to be what bfd does for this simple case. The
868 // problem is that this is not completely general. bfd will give up and
869 // create a dummy section too if there is a ". = . + 1" inside the section
870 // for example.
871 // Given that we want to create the section, we have to worry what impact
872 // it will have on the link. For example, if we just create a section with
873 // 0 for flags, it would change which PT_LOADs are created.
874 // We could remember that particular section is dummy and ignore it in
875 // other parts of the linker, but unfortunately there are quite a few places
876 // that would need to change:
877 // * The program header creation.
878 // * The orphan section placement.
879 // * The address assignment.
880 // The other option is to pick flags that minimize the impact the section
881 // will have on the rest of the linker. That is why we copy the flags from
882 // the previous sections. Only a few flags are needed to keep the impact low.
883 uint64_t Flags = SHF_ALLOC;
884
885 for (BaseCommand *&Cmd : SectionCommands) {
886 auto *Sec = dyn_cast<OutputSection>(Cmd);
887 if (!Sec)
888 continue;
889
890 // Handle align (e.g. ".foo : ALIGN(16) { ... }").
891 if (Sec->AlignExpr)
892 Sec->Alignment =
893 std::max<uint32_t>(Sec->Alignment, Sec->AlignExpr().getValue());
894
895 // The input section might have been removed (if it was an empty synthetic
896 // section), but we at least know the flags.
897 if (Sec->HasInputSections)
898 Flags = Sec->Flags;
899
900 // We do not want to keep any special flags for output section
901 // in case it is empty.
902 bool IsEmpty = getInputSections(Sec).empty();
903 if (IsEmpty)
904 Sec->Flags = Flags & ((Sec->NonAlloc ? 0 : (uint64_t)SHF_ALLOC) |
905 SHF_WRITE | SHF_EXECINSTR);
906
907 if (IsEmpty && isDiscardable(*Sec)) {
908 Sec->markDead();
909 Cmd = nullptr;
910 } else if (!Sec->isLive()) {
911 Sec->markLive();
912 }
913 }
914
915 // It is common practice to use very generic linker scripts. So for any
916 // given run some of the output sections in the script will be empty.
917 // We could create corresponding empty output sections, but that would
918 // clutter the output.
919 // We instead remove trivially empty sections. The bfd linker seems even
920 // more aggressive at removing them.
921 llvm::erase_if(SectionCommands, [&](BaseCommand *Base) { return !Base; });
922}

924void LinkerScript::adjustSectionsAfterSorting() {
925 // Try and find an appropriate memory region to assign offsets in.
926 for (BaseCommand *Base : SectionCommands) {
927 if (auto *Sec = dyn_cast<OutputSection>(Base)) {
928 if (!Sec->LMARegionName.empty()) {
929 if (MemoryRegion *M = MemoryRegions.lookup(Sec->LMARegionName))
930 Sec->LMARegion = M;
931 else
932 error("memory region '" + Sec->LMARegionName + "' not declared");
933 }
934 Sec->MemRegion = findMemoryRegion(Sec);
935 }
936 }
937
938 // If output section command doesn't specify any segments,
939 // and we haven't previously assigned any section to segment,
940 // then we simply assign section to the very first load segment.
941 // Below is an example of such linker script:
942 // PHDRS { seg PT_LOAD; }
943 // SECTIONS { .aaa : { *(.aaa) } }
944 std::vector<StringRef> DefPhdrs;
945 auto FirstPtLoad = llvm::find_if(PhdrsCommands, [](const PhdrsCommand &Cmd) {
946 return Cmd.Type == PT_LOAD;
947 });
948 if (FirstPtLoad != PhdrsCommands.end())
949 DefPhdrs.push_back(FirstPtLoad->Name);
950
951 // Walk the commands and propagate the program headers to commands that don't
952 // explicitly specify them.
953 for (BaseCommand *Base : SectionCommands) {
954 auto *Sec = dyn_cast<OutputSection>(Base);
955 if (!Sec)
956 continue;
957
958 if (Sec->Phdrs.empty()) {
959 // To match the bfd linker script behaviour, only propagate program
960 // headers to sections that are allocated.
961 if (Sec->Flags & SHF_ALLOC)
962 Sec->Phdrs = DefPhdrs;
963 } else {
964 DefPhdrs = Sec->Phdrs;
965 }
966 }
967}

969static uint64_t computeBase(uint64_t Min, bool AllocateHeaders) {
970 // If there is no SECTIONS or if the linkerscript is explicit about program
971 // headers, do our best to allocate them.
972 if (!Script->HasSectionsCommand || AllocateHeaders)
973 return 0;
974 // Otherwise only allocate program headers if that would not add a page.
975 return alignDown(Min, Config->MaxPageSize);
976}
977
978// Try to find an address for the file and program headers output sections,
979// which were unconditionally added to the first PT_LOAD segment earlier.
980//
981// When using the default layout, we check if the headers fit below the first
982// allocated section. When using a linker script, we also check if the headers
983// are covered by the output section. This allows omitting the headers by not
984// leaving enough space for them in the linker script; this pattern is common
985// in embedded systems.
986//
987// If there isn't enough space for these sections, we'll remove them from the
988// PT_LOAD segment, and we'll also remove the PT_PHDR segment.
989void LinkerScript::allocateHeaders(std::vector<PhdrEntry *> &Phdrs) {
990 uint64_t Min = std::numeric_limits<uint64_t>::max();
991 for (OutputSection *Sec : OutputSections)
992 if (Sec->Flags & SHF_ALLOC)
993 Min = std::min<uint64_t>(Min, Sec->Addr);
994
995 auto It = llvm::find_if(
996 Phdrs, [](const PhdrEntry *E) { return E->p_type == PT_LOAD; });
997 if (It == Phdrs.end())
998 return;
999 PhdrEntry *FirstPTLoad = *It;
1000
1001 bool HasExplicitHeaders =
1002 llvm::any_of(PhdrsCommands, [](const PhdrsCommand &Cmd) {
1003 return Cmd.HasPhdrs || Cmd.HasFilehdr;
1004 });
1005 bool Paged = !Config->Omagic && !Config->Nmagic;
1006 uint64_t HeaderSize = getHeaderSize();
1007 if ((Paged || HasExplicitHeaders) &&
1008 HeaderSize <= Min - computeBase(Min, HasExplicitHeaders)) {
1009 Min = alignDown(Min - HeaderSize, Config->MaxPageSize);
1010 Out::ElfHeader->Addr = Min;
1011 Out::ProgramHeaders->Addr = Min + Out::ElfHeader->Size;
1012 return;
1013 }
1014
1015 // Error if we were explicitly asked to allocate headers.
1016 if (HasExplicitHeaders)
1017 error("could not allocate headers");
1018
1019 Out::ElfHeader->PtLoad = nullptr;
1020 Out::ProgramHeaders->PtLoad = nullptr;
1021 FirstPTLoad->FirstSec = findFirstSection(FirstPTLoad);
1022
1023 llvm::erase_if(Phdrs,
1024 [](const PhdrEntry *E) { return E->p_type == PT_PHDR; });
1025}

1034static uint64_t getInitialDot() {
1035 // By default linker scripts use an initial value of 0 for '.',
1036 // but prefer -image-base if set.
1037 if (Script->HasSectionsCommand)
1038 return Config->ImageBase ? *Config->ImageBase : 0;
1039
1040 uint64_t StartAddr = UINT64_MAX;
1041 // The Sections with -T<section> have been sorted in order of ascending
1042 // address. We must lower StartAddr if the lowest -T<section address> as
1043 // calls to setDot() must be monotonically increasing.
1044 for (auto &KV : Config->SectionStartMap)
1045 StartAddr = std::min(StartAddr, KV.second);
1046 return std::min(StartAddr, Target->getImageBase() + elf::getHeaderSize());
1047}

1049// Here we assign addresses as instructed by linker script SECTIONS
1050// sub-commands. Doing that allows us to use final VA values, so here
1051// we also handle rest commands like symbol assignments and ASSERTs.
1052void LinkerScript::assignAddresses() {
1053 Dot = getInitialDot();
1054
1055 auto Deleter = make_unique<AddressState>();
1056 Ctx = Deleter.get();
1057 ErrorOnMissingSection = true;
1058 switchTo(Aether);
1059
1060 for (BaseCommand *Base : SectionCommands) {
1061 if (auto *Cmd = dyn_cast<SymbolAssignment>(Base)) {
1062 Cmd->Addr = Dot;
1063 assignSymbol(Cmd, false);
1064 Cmd->Size = Dot - Cmd->Addr;
1065 continue;
1066 }
1067 assignOffsets(cast<OutputSection>(Base));
1068 }
1069 Ctx = nullptr;
1070}

1072// Creates program headers as instructed by PHDRS linker script command.
1073std::vector<PhdrEntry *> LinkerScript::createPhdrs() {
1074 std::vector<PhdrEntry *> Ret;
1075
1076 // Process PHDRS and FILEHDR keywords because they are not
1077 // real output sections and cannot be added in the following loop.
1078 for (const PhdrsCommand &Cmd : PhdrsCommands) {
1079 PhdrEntry *Phdr = make<PhdrEntry>(Cmd.Type, Cmd.Flags ? *Cmd.Flags : PF_R);
1080
1081 if (Cmd.HasFilehdr)
1082 Phdr->add(Out::ElfHeader);
1083 if (Cmd.HasPhdrs)
1084 Phdr->add(Out::ProgramHeaders);
1085
1086 if (Cmd.LMAExpr) {
1087 Phdr->p_paddr = Cmd.LMAExpr().getValue();
1088 Phdr->HasLMA = true;
1089 }
1090 Ret.push_back(Phdr);
1091 }
1092
1093 // Add output sections to program headers.
1094 for (OutputSection *Sec : OutputSections) {
1095 // Assign headers specified by linker script
1096 for (size_t Id : getPhdrIndices(Sec)) {
1097 Ret[Id]->add(Sec);
1098 if (!PhdrsCommands[Id].Flags.hasValue())
1099 Ret[Id]->p_flags |= Sec->getPhdrFlags();
1100 }
1101 }
1102 return Ret;
1103}

1119ExprValue LinkerScript::getSymbolValue(StringRef Name, const Twine &Loc) {
1120 if (Name == ".") {
1121 if (Ctx)
1122 return {Ctx->OutSec, false, Dot - Ctx->OutSec->Addr, Loc};
1123 error(Loc + ": unable to get location counter value");
1124 return 0;
1125 }
1126
1127 if (Symbol *Sym = Symtab->find(Name)) {
1128 if (auto *DS = dyn_cast<Defined>(Sym))
1129 return {DS->Section, false, DS->Value, Loc};
1130 if (isa<SharedSymbol>(Sym))
1131 if (!ErrorOnMissingSection)
1132 return {nullptr, false, 0, Loc};
1133 }
1134
1135 error(Loc + ": symbol not found: " + Name);
1136 return 0;
1137}

1148// Returns indices of ELF headers containing specific section. Each index is a
1149// zero based number of ELF header listed within PHDRS {} script block.

1152
