98StringRef elf::getOutputSectionName(const InputSectionBase *S) {
99 if (Config->Relocatable)
100 return S->Name;
101
102 // This is for --emit-relocs. If .text.foo is emitted as .text.bar, we want
103 // to emit .rela.text.foo as .rela.text.bar for consistency (this is not
104 // technically required, but not doing it is odd). This code guarantees that.
105 if (auto *IS = dyn_cast<InputSection>(S)) {
106 if (InputSectionBase *Rel = IS->getRelocatedSection()) {
107 OutputSection *Out = Rel->getOutputSection();
108 if (S->Type == SHT_RELA)
109 return Saver.save(".rela" + Out->Name);
110 return Saver.save(".rel" + Out->Name);
111 }
112 }
113
114 // This check is for -z keep-text-section-prefix. This option separates text
115 // sections with prefix ".text.hot", ".text.unlikely", ".text.startup" or
116 // ".text.exit".
117 // When enabled, this allows identifying the hot code region (.text.hot) in
118 // the final binary which can be selectively mapped to huge pages or mlocked,
119 // for instance.
120 if (Config->ZKeepTextSectionPrefix)
121 for (StringRef V :
122 {".text.hot.", ".text.unlikely.", ".text.startup.", ".text.exit."})
123 if (isSectionPrefix(V, S->Name))
124 return V.drop_back();
125
126 for (StringRef V :
127 {".text.", ".rodata.", ".data.rel.ro.", ".data.", ".bss.rel.ro.",
128 ".bss.", ".init_array.", ".fini_array.", ".ctors.", ".dtors.", ".tbss.",
129 ".gcc_except_table.", ".tdata.", ".ARM.exidx.", ".ARM.extab."})
130 if (isSectionPrefix(V, S->Name))
131 return V.drop_back();
132
133 // CommonSection is identified as "COMMON" in linker scripts.
134 // By default, it should go to .bss section.
135 if (S->Name == "COMMON")
136 return ".bss";
137
138 return S->Name;
139}

153void Writer<ELFT>::removeEmptyPTLoad(std::vector<PhdrEntry *> &Phdrs) {
154 llvm::erase_if(Phdrs, [&](const PhdrEntry *P) {
155 if (P->p_type != PT_LOAD)
156 return false;
157 if (!P->FirstSec)
158 return true;
159 uint64_t Size = P->LastSec->Addr + P->LastSec->Size - P->FirstSec->Addr;
160 return Size == 0;
161 });
162}

164template <class ELFT> static void copySectionsIntoPartitions() {
165 std::vector<InputSectionBase *> NewSections;
166 for (unsigned Part = 2; Part != Partitions.size() + 1; ++Part) {
167 for (InputSectionBase *S : InputSections) {
168 if (!(S->Flags & SHF_ALLOC) || !S->isLive())
169 continue;
170 InputSectionBase *Copy;
171 if (S->Type == SHT_NOTE)
172 Copy = make<InputSection>(cast<InputSection>(*S));
173 else if (auto *ES = dyn_cast<EhInputSection>(S))
174 Copy = make<EhInputSection>(*ES);
175 else
176 continue;
177 Copy->Partition = Part;
178 NewSections.push_back(Copy);
179 }
180 }
181
182 InputSections.insert(InputSections.end(), NewSections.begin(),
183 NewSections.end());
184}

186template <class ELFT> static void combineEhSections() {
187 for (InputSectionBase *&S : InputSections) {
188 // Ignore dead sections and the partition end marker (.part.end),
189 // whose partition number is out of bounds.
190 if (!S->isLive() || S->Partition == 255)
191 continue;
192
193 Partition &Part = S->getPartition();
194 if (auto *ES = dyn_cast<EhInputSection>(S)) {
195 Part.EhFrame->addSection<ELFT>(ES);
196 S = nullptr;
197 } else if (S->kind() == SectionBase::Regular && Part.ARMExidx &&
198 Part.ARMExidx->addSection(cast<InputSection>(S))) {
199 S = nullptr;
200 }
201 }
202
203 std::vector<InputSectionBase *> &V = InputSections;
204 V.erase(std::remove(V.begin(), V.end(), nullptr), V.end());
205}

207template <class ELFT> void Writer<ELFT>::combineCapRelocsSections() {
208 for (InputSectionBase *&S : InputSections) {
209 if (S->Name != "__cap_relocs")
210 continue;
211 // Factory.addInputSec(S, getOutputSectionName(S->Name), InX<ELFT>::CapRelocs->OutSec);
212
213 // We only gather the sections here and add the cap_relocs during
214 // finalizeContents() The reason for this is that we don't know if symbols
215 // are preemptible when this function is called.
216 InX<ELFT>::CapRelocs->addSection(S);
217 S = nullptr;
218 }
219 std::vector<InputSectionBase *> &V = InputSections;
220 V.erase(std::remove(V.begin(), V.end(), nullptr), V.end());
221}

223static Defined *addOptionalRegular(StringRef Name, SectionBase *Sec,
224 uint64_t Val, uint8_t StOther = STV_HIDDEN,
225 uint8_t Binding = STB_GLOBAL,
226 bool CanBeSectionStart = true) {
227 Symbol *S = Symtab->find(Name);
228 if (!S || S->isDefined())
229 return nullptr;
230
231 S->resolve(Defined{/*File=*/nullptr, Name, Binding, StOther, STT_NOTYPE, Val,
232 /*Size=*/0, Sec});
233 // If Val == 0 assume this symbol references the start of a section.
234 // When targetting CHERI we set the size of that symbol since otherwise
235 // an expression like foo = &_DYNAMIC will create a zero-length capability
236 // for foo and most likely crash the program.
237 // TODO: I would like to do this for all targets but that might cause
238 // compatibility issues
239 if (Val == 0 && CanBeSectionStart) {
240 log("Treating " + Name + " as a section start symbol");
241 S->IsSectionStartSymbol = true;
242 }
243
244 return cast<Defined>(S);
245}

247static Defined *addAbsolute(StringRef Name) {
248 Symbol *Sym = Symtab->addSymbol(Defined{nullptr, Name, STB_GLOBAL, STV_HIDDEN,
249 STT_NOTYPE, 0, 0, nullptr});
250 return cast<Defined>(Sym);
251}

253// The linker is expected to define some symbols depending on
254// the linking result. This function defines such symbols.
255void elf::addReservedSymbols() {
256 if (Config->EMachine == EM_MIPS) {
257 // Define _gp for MIPS. st_value of _gp symbol will be updated by Writer
258 // so that it points to an absolute address which by default is relative
259 // to GOT. Default offset is 0x7ff0.
260 // See "Global Data Symbols" in Chapter 6 in the following document:
261 // ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
262 ElfSym::MipsGp = addAbsolute("_gp");
263
264 // On MIPS O32 ABI, _gp_disp is a magic symbol designates offset between
265 // start of function and 'gp' pointer into GOT.
266 if (Symtab->find("_gp_disp"))
267 ElfSym::MipsGpDisp = addAbsolute("_gp_disp");
268
269 // The __gnu_local_gp is a magic symbol equal to the current value of 'gp'
270 // pointer. This symbol is used in the code generated by .cpload pseudo-op
271 // in case of using -mno-shared option.
272 // https://sourceware.org/ml/binutils/2004-12/msg00094.html
273 if (Symtab->find("__gnu_local_gp"))
274 ElfSym::MipsLocalGp = addAbsolute("__gnu_local_gp");
275 } else if (Config->EMachine == EM_PPC) {
276 // glibc *crt1.o has a undefined reference to _SDA_BASE_. Since we don't
277 // support Small Data Area, define it arbitrarily as 0.
278 addOptionalRegular("_SDA_BASE_", nullptr, 0, STV_HIDDEN);
279 }
280
281 // The Power Architecture 64-bit v2 ABI defines a TableOfContents (TOC) which
282 // combines the typical ELF GOT with the small data sections. It commonly
283 // includes .got .toc .sdata .sbss. The .TOC. symbol replaces both
284 // _GLOBAL_OFFSET_TABLE_ and _SDA_BASE_ from the 32-bit ABI. It is used to
285 // represent the TOC base which is offset by 0x8000 bytes from the start of
286 // the .got section.
287 // We do not allow _GLOBAL_OFFSET_TABLE_ to be defined by input objects as the
288 // correctness of some relocations depends on its value.
289 StringRef GotSymName =
290 (Config->EMachine == EM_PPC64) ? ".TOC." : "_GLOBAL_OFFSET_TABLE_";
291
292 if (Symbol *S = Symtab->find(GotSymName)) {
293 if (S->isDefined()) {
294 error(toString(S->File) + " cannot redefine linker defined symbol '" +
295 GotSymName + "'");
296 return;
297 }
298
299 uint64_t GotOff = 0;
300 if (Config->EMachine == EM_PPC64)
301 GotOff = 0x8000;
302
303 S->resolve(Defined{/*File=*/nullptr, GotSymName, STB_GLOBAL, STV_HIDDEN,
304 STT_NOTYPE, GotOff, /*Size=*/0, Out::ElfHeader});
305 ElfSym::GlobalOffsetTable = cast<Defined>(S);
306 }
307
308 // __ehdr_start is the location of ELF file headers. Note that we define
309 // this symbol unconditionally even when using a linker script, which
310 // differs from the behavior implemented by GNU linker which only define
311 // this symbol if ELF headers are in the memory mapped segment.
312 addOptionalRegular("__ehdr_start", Out::ElfHeader, 0, STV_HIDDEN);
313
314 // __executable_start is not documented, but the expectation of at
315 // least the Android libc is that it points to the ELF header.
316 addOptionalRegular("__executable_start", Out::ElfHeader, 0, STV_HIDDEN);
317
318 // __dso_handle symbol is passed to cxa_finalize as a marker to identify
319 // each DSO. The address of the symbol doesn't matter as long as they are
320 // different in different DSOs, so we chose the start address of the DSO.
321 addOptionalRegular("__dso_handle", Out::ElfHeader, 0, STV_HIDDEN);
322
323 // If linker script do layout we do not need to create any standart symbols.
324 if (Script->HasSectionsCommand)
325 return;
326
327 auto Add = [](StringRef S, int64_t Pos) {
328 return addOptionalRegular(S, Out::ElfHeader, Pos, STV_DEFAULT);
329 };
330
331 ElfSym::Bss = Add("__bss_start", 0);
332 ElfSym::End1 = Add("end", -1);
333 ElfSym::End2 = Add("_end", -1);
334 ElfSym::Etext1 = Add("etext", -1);
335 ElfSym::Etext2 = Add("_etext", -1);
336 ElfSym::Edata1 = Add("edata", -1);
337 ElfSym::Edata2 = Add("_edata", -1);
338}

340static OutputSection *findSection(StringRef Name, unsigned Partition = 1) {
341 for (BaseCommand *Base : Script->SectionCommands)
342 if (auto *Sec = dyn_cast<OutputSection>(Base))
343 if (Sec->Name == Name && Sec->Partition == Partition)
344 return Sec;
345 return nullptr;
346}

348// Initialize Out members.
349template <class ELFT> static void createSyntheticSections() {
350 // Initialize all pointers with NULL. This is needed because
351 // you can call lld::elf::main more than once as a library.
352 memset(&Out::First, 0, sizeof(Out));
353
354 auto Add = [](InputSectionBase *Sec) { InputSections.push_back(Sec); };
355
356 In.ShStrTab = make<StringTableSection>(".shstrtab", false);
357
358 Out::ProgramHeaders = make<OutputSection>("", 0, SHF_ALLOC);
359 Out::ProgramHeaders->Alignment = Config->Wordsize;
360
361 if (Config->Strip != StripPolicy::All) {
362 In.StrTab = make<StringTableSection>(".strtab", false);
363 In.SymTab = make<SymbolTableSection<ELFT>>(*In.StrTab);
364 In.SymTabShndx = make<SymtabShndxSection>();
365 }
366
367 In.Bss = make<BssSection>(".bss", 0, 1);
368 Add(In.Bss);
369
370 // If there is a SECTIONS command and a .data.rel.ro section name use name
371 // .data.rel.ro.bss so that we match in the .data.rel.ro output section.
372 // This makes sure our relro is contiguous.
373 bool HasDataRelRo =
374 Script->HasSectionsCommand && findSection(".data.rel.ro", 0);
375 In.BssRelRo =
376 make<BssSection>(HasDataRelRo ? ".data.rel.ro.bss" : ".bss.rel.ro", 0, 1);
377 Add(In.BssRelRo);
378
379 if (Config->ProcessCapRelocs) {
380 InX<ELFT>::CapRelocs = make<CheriCapRelocsSection<ELFT>>();
381 }
382
383 if (Config->CapabilitySize > 0) {
384 In.CheriCapTable = make<CheriCapTableSection>();
385 Add(In.CheriCapTable);
386 if (Config->CapTableScope != CapTableScopePolicy::All) {
387 In.CheriCapTableMapping = make<CheriCapTableMappingSection>();
388 Add(In.CheriCapTableMapping);
389 }
390 }
391
392 // Add MIPS-specific sections.
393 if (Config->EMachine == EM_MIPS) {
394 // XXXAR: also add the RLD_MAP dynamic tags to rtld so that we can use
395 // gdb with rtld direct exec mode.
396 // TODO: should probably try to build rtld as PIE instead?
397 if ((!Config->Shared || Config->BuildingFreeBSDRtld) && Config->HasDynSymTab) {
398 In.MipsRldMap = make<MipsRldMapSection>();
399 Add(In.MipsRldMap);
400 }
401 InX<ELFT>::MipsAbiFlags = MipsAbiFlagsSection<ELFT>::create();
402 if (InX<ELFT>::MipsAbiFlags)
403 Add(InX<ELFT>::MipsAbiFlags);
404 if (auto *Sec = MipsOptionsSection<ELFT>::create())
405 Add(Sec);
406 if (auto *Sec = MipsReginfoSection<ELFT>::create())
407 Add(Sec);
408 }
409
410 for (Partition &Part : Partitions) {
411 auto Add = [&](InputSectionBase *Sec) {
412 Sec->Partition = Part.getNumber();
413 InputSections.push_back(Sec);
414 };
415
416 if (!Part.Name.empty()) {
417 Part.ElfHeader = make<PartitionElfHeaderSection<ELFT>>();
418 Part.ElfHeader->Name = Part.Name;
419 Add(Part.ElfHeader);
420
421 Part.ProgramHeaders = make<PartitionProgramHeadersSection<ELFT>>();
422 Add(Part.ProgramHeaders);
423 }
424
425 if (Config->BuildId != BuildIdKind::None) {
426 Part.BuildId = make<BuildIdSection>();
427 Add(Part.BuildId);
428 }
429
430 Part.DynStrTab = make<StringTableSection>(".dynstr", true);
431 Part.DynSymTab = make<SymbolTableSection<ELFT>>(*Part.DynStrTab);
432 Part.Dynamic = make<DynamicSection<ELFT>>();
433 if (Config->AndroidPackDynRelocs) {
434 Part.RelaDyn = make<AndroidPackedRelocationSection<ELFT>>(
435 Config->IsRela ? ".rela.dyn" : ".rel.dyn");
436 } else {
437 Part.RelaDyn = make<RelocationSection<ELFT>>(
438 Config->IsRela ? ".rela.dyn" : ".rel.dyn", Config->ZCombreloc);
439 }
440
441 if (needsInterpSection())
442 Add(createInterpSection());
443
444 if (Config->HasDynSymTab) {
445 Part.DynSymTab = make<SymbolTableSection<ELFT>>(*Part.DynStrTab);
446 Add(Part.DynSymTab);
447
448 Part.VerSym = make<VersionTableSection>();
449 Add(Part.VerSym);
450
451 if (!Config->VersionDefinitions.empty()) {
452 Part.VerDef = make<VersionDefinitionSection>();
453 Add(Part.VerDef);
454 }
455
456 Part.VerNeed = make<VersionNeedSection<ELFT>>();
457 Add(Part.VerNeed);
458
459 if (Config->GnuHash) {
460 Part.GnuHashTab = make<GnuHashTableSection>();
461 Add(Part.GnuHashTab);
462 }
463
464 if (Config->SysvHash) {
465 Part.HashTab = make<HashTableSection>();
466 Add(Part.HashTab);
467 }
468
469 Add(Part.Dynamic);
470 Add(Part.DynStrTab);
471 Add(Part.RelaDyn);
472 }
473
474 if (Config->RelrPackDynRelocs) {
475 Part.RelrDyn = make<RelrSection<ELFT>>();
476 Add(Part.RelrDyn);
477 }
478
479 if (!Config->Relocatable) {
480 if (Config->EhFrameHdr) {
481 Part.EhFrameHdr = make<EhFrameHeader>();
482 Add(Part.EhFrameHdr);
483 }
484 Part.EhFrame = make<EhFrameSection>();
485 Add(Part.EhFrame);
486 }
487
488 if (Config->EMachine == EM_ARM && !Config->Relocatable) {
489 // The ARMExidxsyntheticsection replaces all the individual .ARM.exidx
490 // InputSections.
491 Part.ARMExidx = make<ARMExidxSyntheticSection>();
492 Add(Part.ARMExidx);
493 }
494 }
495
496 if (Partitions.size() != 1) {
497 // Create the partition end marker. This needs to be in partition number 255
498 // so that it is sorted after all other partitions. It also has other
499 // special handling (see createPhdrs() and combineEhSections()).
500 In.PartEnd = make<BssSection>(".part.end", Config->MaxPageSize, 1);
501 In.PartEnd->Partition = 255;
502 Add(In.PartEnd);
503
504 In.PartIndex = make<PartitionIndexSection>();
505 addOptionalRegular("__part_index_begin", In.PartIndex, 0);
506 addOptionalRegular("__part_index_end", In.PartIndex,
507 In.PartIndex->getSize());
508 Add(In.PartIndex);
509 }
510
511 // Add .got. MIPS' .got is so different from the other archs,
512 // it has its own class.
513 if (Config->EMachine == EM_MIPS) {
514 In.MipsGot = make<MipsGotSection>();
515 Add(In.MipsGot);
516 } else {
517 In.Got = make<GotSection>();
518 Add(In.Got);
519 }
520
521 if (Config->EMachine == EM_PPC) {
522 In.PPC32Got2 = make<PPC32Got2Section>();
523 Add(In.PPC32Got2);
524 }
525
526 if (Config->EMachine == EM_PPC64) {
527 In.PPC64LongBranchTarget = make<PPC64LongBranchTargetSection>();
528 Add(In.PPC64LongBranchTarget);
529 }
530
531 In.GotPlt = make<GotPltSection>();
532 Add(In.GotPlt);
533 In.IgotPlt = make<IgotPltSection>();
534 Add(In.IgotPlt);
535
536 // _GLOBAL_OFFSET_TABLE_ is defined relative to either .got.plt or .got. Treat
537 // it as a relocation and ensure the referenced section is created.
538 if (ElfSym::GlobalOffsetTable && Config->EMachine != EM_MIPS) {
539 if (Target->GotBaseSymInGotPlt)
540 In.GotPlt->HasGotPltOffRel = true;
541 else
542 In.Got->HasGotOffRel = true;
543 }
544
545 if (Config->GdbIndex)
546 Add(GdbIndexSection::create<ELFT>());
547
548 // We always need to add rel[a].plt to output if it has entries.
549 // Even for static linking it can contain R_[*]_IRELATIVE relocations.
550 In.RelaPlt = make<RelocationSection<ELFT>>(
551 Config->IsRela ? ".rela.plt" : ".rel.plt", false /*Sort*/);
552 Add(In.RelaPlt);
553
554 // The RelaIplt immediately follows .rel.plt (.rel.dyn for ARM) to ensure
555 // that the IRelative relocations are processed last by the dynamic loader.
556 // We cannot place the iplt section in .rel.dyn when Android relocation
557 // packing is enabled because that would cause a section type mismatch.
558 // However, because the Android dynamic loader reads .rel.plt after .rel.dyn,
559 // we can get the desired behaviour by placing the iplt section in .rel.plt.
560 In.RelaIplt = make<RelocationSection<ELFT>>(
561 (Config->EMachine == EM_ARM && !Config->AndroidPackDynRelocs)
562 ? ".rel.dyn"
563 : In.RelaPlt->Name,
564 false /*Sort*/);
565 Add(In.RelaIplt);
566
567 In.Plt = make<PltSection>(false);
568 Add(In.Plt);
569 In.Iplt = make<PltSection>(true);
570 Add(In.Iplt);
571
572 if (Config->AndFeatures)
573 Add(make<GnuPropertySection>());
574
575 // .note.GNU-stack is always added when we are creating a re-linkable
576 // object file. Other linkers are using the presence of this marker
577 // section to control the executable-ness of the stack area, but that
578 // is irrelevant these days. Stack area should always be non-executable
579 // by default. So we emit this section unconditionally.
580 if (Config->Relocatable)
581 Add(make<GnuStackSection>());
582
583 if (In.SymTab)
584 Add(In.SymTab);
585 if (In.SymTabShndx)
586 Add(In.SymTabShndx);
587 Add(In.ShStrTab);
588 if (In.StrTab)
589 Add(In.StrTab);
590}

592// The main function of the writer.
593template <class ELFT> void Writer<ELFT>::run() {
594 // Make copies of any input sections that need to be copied into each
595 // partition.
596 copySectionsIntoPartitions<ELFT>();
597
598 // Create linker-synthesized sections such as .got or .plt.
599 // Such sections are of type input section.
600 createSyntheticSections<ELFT>();
601
602 // Some input sections that are used for exception handling need to be moved
603 // into synthetic sections. Do that now so that they aren't assigned to
604 // output sections in the usual way.
605 if (!Config->Relocatable) {
606 combineEhSections<ELFT>();
607 if (Config->ProcessCapRelocs && InX<ELFT>::CapRelocs) {
608 combineCapRelocsSections();
609 InputSections.push_back(InX<ELFT>::CapRelocs);
610 }
611 }
612
613 // We want to process linker script commands. When SECTIONS command
614 // is given we let it create sections.
615 Script->processSectionCommands();
616
617 // Linker scripts controls how input sections are assigned to output sections.
618 // Input sections that were not handled by scripts are called "orphans", and
619 // they are assigned to output sections by the default rule. Process that.
620 Script->addOrphanSections();
621
622 if (Config->Discard != DiscardPolicy::All)
623 copyLocalSymbols();
624
625 if (Config->CopyRelocs)
626 addSectionSymbols();
627
628 // Now that we have a complete set of output sections. This function
629 // completes section contents. For example, we need to add strings
630 // to the string table, and add entries to .got and .plt.
631 // finalizeSections does that.
632 finalizeSections();
633 checkExecuteOnly();
634 if (errorCount())
635 return;
636
637 Script->assignAddresses();
638
639 // If -compressed-debug-sections is specified, we need to compress
640 // .debug_* sections. Do it right now because it changes the size of
641 // output sections.
642 for (OutputSection *Sec : OutputSections)
643 Sec->maybeCompress<ELFT>();
644
645 Script->allocateHeaders(Main->Phdrs);
646
647 // Remove empty PT_LOAD to avoid causing the dynamic linker to try to mmap a
648 // 0 sized region. This has to be done late since only after assignAddresses
649 // we know the size of the sections.
650 for (Partition &Part : Partitions)
651 removeEmptyPTLoad(Part.Phdrs);
652
653 if (!Config->OFormatBinary)
654 assignFileOffsets();
655 else
656 assignFileOffsetsBinary();
657
658 for (Partition &Part : Partitions)
659 setPhdrs(Part);
660
661 if (Config->Relocatable)
662 for (OutputSection *Sec : OutputSections)
663 Sec->Addr = 0;
664
665 if (Config->CheckSections)
666 checkSections();
667
668 // It does not make sense try to open the file if we have error already.
669 if (errorCount())
670 return;
671 // Write the result down to a file.
672 openFile();
673 if (errorCount())
674 return;
675
676 if (!Config->OFormatBinary) {
677 writeTrapInstr();
678 writeHeader();
679 writeSections();
680 } else {
681 writeSectionsBinary();
682 }
683
684 // Backfill .note.gnu.build-id section content. This is done at last
685 // because the content is usually a hash value of the entire output file.
686 writeBuildId();
687 if (errorCount())
688 return;
689
690 // Handle -Map and -cref options.
691 writeMapFile();
692 writeCrossReferenceTable();
693 if (errorCount())
694 return;
695
696 if (auto E = Buffer->commit())
697 error("failed to write to the output file: " + toString(std::move(E)));
698}

700static bool shouldKeepInSymtab(const Defined &Sym) {
701 if (Sym.isSection())
702 return false;
703
704 if (Config->Discard == DiscardPolicy::None)
705 return true;
706
707 // If -emit-reloc is given, all symbols including local ones need to be
708 // copied because they may be referenced by relocations.
709 if (Config->EmitRelocs)
710 return true;
711
712 // In ELF assembly .L symbols are normally discarded by the assembler.
713 // If the assembler fails to do so, the linker discards them if
714 // * --discard-locals is used.
715 // * The symbol is in a SHF_MERGE section, which is normally the reason for
716 // the assembler keeping the .L symbol.
717 StringRef Name = Sym.getName();
718 bool IsLocal = Name.startswith(".L") || Name.empty();
719 if (!IsLocal)
720 return true;
721
722 if (Config->Discard == DiscardPolicy::Locals)
723 return false;
724
725 SectionBase *Sec = Sym.Section;
726 return !Sec || !(Sec->Flags & SHF_MERGE);

813// Today's loaders have a feature to make segments read-only after
814// processing dynamic relocations to enhance security. PT_GNU_RELRO
815// is defined for that.
816//
817// This function returns true if a section needs to be put into a
818// PT_GNU_RELRO segment.
819bool lld::elf::isRelroSection(const OutputSection *Sec) {
820 if (!Config->ZRelro)
821 return false;
822
823 uint64_t Flags = Sec->Flags;
824
825 // Non-allocatable or non-writable sections don't need RELRO because
826 // they are not writable or not even mapped to memory in the first place.
827 // RELRO is for sections that are essentially read-only but need to
828 // be writable only at process startup to allow dynamic linker to
829 // apply relocations.
830 if (!(Flags & SHF_ALLOC) || !(Flags & SHF_WRITE))
831 return false;
832
833 // Once initialized, TLS data segments are used as data templates
834 // for a thread-local storage. For each new thread, runtime
835 // allocates memory for a TLS and copy templates there. No thread
836 // are supposed to use templates directly. Thus, it can be in RELRO.
837 if (Flags & SHF_TLS)
838 return true;
839
840 // .init_array, .preinit_array and .fini_array contain pointers to
841 // functions that are executed on process startup or exit. These
842 // pointers are set by the static linker, and they are not expected
843 // to change at runtime. But if you are an attacker, you could do
844 // interesting things by manipulating pointers in .fini_array, for
845 // example. So they are put into RELRO.
846 uint32_t Type = Sec->Type;
847 if (Type == SHT_INIT_ARRAY || Type == SHT_FINI_ARRAY ||
848 Type == SHT_PREINIT_ARRAY)
849 return true;
850
851 // .got contains pointers to external symbols. They are resolved by
852 // the dynamic linker when a module is loaded into memory, and after
853 // that they are not expected to change. So, it can be in RELRO.
854 if (In.Got && Sec == In.Got->getParent())
855 return true;
856
857 // .toc is a GOT-ish section for PowerPC64. Their contents are accessed
858 // through r2 register, which is reserved for that purpose. Since r2 is used
859 // for accessing .got as well, .got and .toc need to be close enough in the
860 // virtual address space. Usually, .toc comes just after .got. Since we place
861 // .got into RELRO, .toc needs to be placed into RELRO too.
862 if (Sec->Name.equals(".toc"))
863 return true;
864
865 // .got.plt contains pointers to external function symbols. They are
866 // by default resolved lazily, so we usually cannot put it into RELRO.
867 // However, if "-z now" is given, the lazy symbol resolution is
868 // disabled, which enables us to put it into RELRO.
869 if (Sec == In.GotPlt->getParent())
870 return Config->ZNow;
871
872 // Similarly the CHERI capability table is also relro since the capabilities
873 // in the table need to be initialized at runtime to set the tag bits
874 if (In.CheriCapTable && Sec == In.CheriCapTable->getParent()) {
875 // Without -z now, the PLT stubs can update the captable entries so we
876 // can't mark it as relro. It can also be relro for static binaries:
877 return Config->ZNow || !Config->Pic;
878 }
879
880 // .dynamic section contains data for the dynamic linker, and
881 // there's no need to write to it at runtime, so it's better to put
882 // it into RELRO.
883 if (Sec->Name == ".dynamic")
884 return true;
885
886 // Sections with some special names are put into RELRO. This is a
887 // bit unfortunate because section names shouldn't be significant in
888 // ELF in spirit. But in reality many linker features depend on
889 // magic section names.
890 StringRef S = Sec->Name;
891 return S == ".data.rel.ro" || S == ".bss.rel.ro" || S == ".ctors" ||
892 S == ".dtors" || S == ".jcr" || S == ".eh_frame" ||
893 S == ".openbsd.randomdata" || S == "__cap_relocs";
894}

896// We compute a rank for each section. The rank indicates where the
897// section should be placed in the file. Instead of using simple
898// numbers (0,1,2...), we use a series of flags. One for each decision
899// point when placing the section.
900// Using flags has two key properties:
901// * It is easy to check if a give branch was taken.
902// * It is easy two see how similar two ranks are (see getRankProximity).
903enum RankFlags {
904 RF_NOT_ADDR_SET = 1 << 27,
905 RF_NOT_ALLOC = 1 << 26,
906 RF_PARTITION = 1 << 18, // Partition number (8 bits)
907 RF_NOT_PART_EHDR = 1 << 17,
908 RF_NOT_PART_PHDR = 1 << 16,
909 RF_NOT_INTERP = 1 << 15,
910 RF_NOT_NOTE = 1 << 14,
911 RF_WRITE = 1 << 13,
912 RF_EXEC_WRITE = 1 << 12,
913 RF_EXEC = 1 << 11,
914 RF_RODATA = 1 << 10,
915 RF_NOT_RELRO = 1 << 9,
916 RF_NOT_TLS = 1 << 8,
917 RF_BSS = 1 << 7,
918 RF_PPC_NOT_TOCBSS = 1 << 6,
919 RF_PPC_TOCL = 1 << 5,
920 RF_PPC_TOC = 1 << 4,
921 RF_PPC_GOT = 1 << 3,
922 RF_PPC_BRANCH_LT = 1 << 2,
923 RF_MIPS_GPREL = 1 << 1,
924 RF_MIPS_NOT_GOT = 1 << 0
925};
926
927static unsigned getSectionRank(const OutputSection *Sec) {
928 unsigned Rank = Sec->Partition * RF_PARTITION;
929
930 // We want to put section specified by -T option first, so we
931 // can start assigning VA starting from them later.
932 if (Config->SectionStartMap.count(Sec->Name))
933 return Rank;
934 Rank |= RF_NOT_ADDR_SET;
935
936 // Allocatable sections go first to reduce the total PT_LOAD size and
937 // so debug info doesn't change addresses in actual code.
938 if (!(Sec->Flags & SHF_ALLOC))
939 return Rank | RF_NOT_ALLOC;
940
941 if (Sec->Type == SHT_LLVM_PART_EHDR)
942 return Rank;
943 Rank |= RF_NOT_PART_EHDR;
944
945 if (Sec->Type == SHT_LLVM_PART_PHDR)
946 return Rank;
947 Rank |= RF_NOT_PART_PHDR;
948
949 // Put .interp first because some loaders want to see that section
950 // on the first page of the executable file when loaded into memory.
951 if (Sec->Name == ".interp")
952 return Rank;
953 Rank |= RF_NOT_INTERP;
954
955 // Put .note sections (which make up one PT_NOTE) at the beginning so that
956 // they are likely to be included in a core file even if core file size is
957 // limited. In particular, we want a .note.gnu.build-id and a .note.tag to be
958 // included in a core to match core files with executables.
959 if (Sec->Type == SHT_NOTE)
960 return Rank;
961 Rank |= RF_NOT_NOTE;
962
963 // Sort sections based on their access permission in the following
964 // order: R, RX, RWX, RW. This order is based on the following
965 // considerations:
966 // * Read-only sections come first such that they go in the
967 // PT_LOAD covering the program headers at the start of the file.
968 // * Read-only, executable sections come next.
969 // * Writable, executable sections follow such that .plt on
970 // architectures where it needs to be writable will be placed
971 // between .text and .data.
972 // * Writable sections come last, such that .bss lands at the very
973 // end of the last PT_LOAD.
974 bool IsExec = Sec->Flags & SHF_EXECINSTR;
975 bool IsWrite = Sec->Flags & SHF_WRITE;
976
977 if (IsExec) {
978 if (IsWrite)
979 Rank |= RF_EXEC_WRITE;
980 else
981 Rank |= RF_EXEC;
982 } else if (IsWrite) {
983 Rank |= RF_WRITE;
984 } else if (Sec->Type == SHT_PROGBITS) {
985 // Make non-executable and non-writable PROGBITS sections (e.g .rodata
986 // .eh_frame) closer to .text. They likely contain PC or GOT relative
987 // relocations and there could be relocation overflow if other huge sections
988 // (.dynstr .dynsym) were placed in between.
989 Rank |= RF_RODATA;
990 }
991
992 // Place RelRo sections first. After considering SHT_NOBITS below, the
993 // ordering is PT_LOAD(PT_GNU_RELRO(.data.rel.ro .bss.rel.ro) | .data .bss),
994 // where | marks where page alignment happens. An alternative ordering is
995 // PT_LOAD(.data | PT_GNU_RELRO( .data.rel.ro .bss.rel.ro) | .bss), but it may
996 // waste more bytes due to 2 alignment places.
997 if (!isRelroSection(Sec))
998 Rank |= RF_NOT_RELRO;
999
1000 // If we got here we know that both A and B are in the same PT_LOAD.
1001
1002 // The TLS initialization block needs to be a single contiguous block in a R/W
1003 // PT_LOAD, so stick TLS sections directly before the other RelRo R/W
1004 // sections. Since p_filesz can be less than p_memsz, place NOBITS sections
1005 // after PROGBITS.
1006 if (!(Sec->Flags & SHF_TLS))
1007 Rank |= RF_NOT_TLS;
1008
1009 // Within TLS sections, or within other RelRo sections, or within non-RelRo
1010 // sections, place non-NOBITS sections first.
1011 if (Sec->Type == SHT_NOBITS)
1012 Rank |= RF_BSS;
1013
1014 // Some architectures have additional ordering restrictions for sections
1015 // within the same PT_LOAD.
1016 if (Config->EMachine == EM_PPC64) {
1017 // PPC64 has a number of special SHT_PROGBITS+SHF_ALLOC+SHF_WRITE sections
1018 // that we would like to make sure appear is a specific order to maximize
1019 // their coverage by a single signed 16-bit offset from the TOC base
1020 // pointer. Conversely, the special .tocbss section should be first among
1021 // all SHT_NOBITS sections. This will put it next to the loaded special
1022 // PPC64 sections (and, thus, within reach of the TOC base pointer).
1023 StringRef Name = Sec->Name;
1024 if (Name != ".tocbss")
1025 Rank |= RF_PPC_NOT_TOCBSS;
1026
1027 if (Name == ".toc1")
1028 Rank |= RF_PPC_TOCL;
1029
1030 if (Name == ".toc")
1031 Rank |= RF_PPC_TOC;
1032
1033 if (Name == ".got")
1034 Rank |= RF_PPC_GOT;
1035
1036 if (Name == ".branch_lt")
1037 Rank |= RF_PPC_BRANCH_LT;
1038 }
1039
1040 if (Config->EMachine == EM_MIPS) {
1041 // All sections with SHF_MIPS_GPREL flag should be grouped together
1042 // because data in these sections is addressable with a gp relative address.
1043 if (Sec->Flags & SHF_MIPS_GPREL)
1044 Rank |= RF_MIPS_GPREL;
1045
1046 if (Sec->Name != ".got")
1047 Rank |= RF_MIPS_NOT_GOT;
1048 }
1049
1050 return Rank;
1051}

1202// We want to find how similar two ranks are.
1203// The more branches in getSectionRank that match, the more similar they are.
1204// Since each branch corresponds to a bit flag, we can just use
1205// countLeadingZeros.
1206static int getRankProximityAux(OutputSection *A, OutputSection *B) {
1207 return countLeadingZeros(A->SortRank ^ B->SortRank);
1208}

1210static int getRankProximity(OutputSection *A, BaseCommand *B) {
1211 auto *Sec = dyn_cast<OutputSection>(B);
1212 return (Sec && Sec->HasInputSections) ? getRankProximityAux(A, Sec) : -1;

1215// When placing orphan sections, we want to place them after symbol assignments
1216// so that an orphan after
1217// begin_foo = .;
1218// foo : { *(foo) }
1219// end_foo = .;
1220// doesn't break the intended meaning of the begin/end symbols.
1221// We don't want to go over sections since findOrphanPos is the
1222// one in charge of deciding the order of the sections.
1223// We don't want to go over changes to '.', since doing so in
1224// rx_sec : { *(rx_sec) }
1225// . = ALIGN(0x1000);
1226// /* The RW PT_LOAD starts here*/
1227// rw_sec : { *(rw_sec) }
1228// would mean that the RW PT_LOAD would become unaligned.
1229static bool shouldSkip(BaseCommand *Cmd) {
1230 if (auto *Assign = dyn_cast<SymbolAssignment>(Cmd))
1231 return Assign->Name != ".";
1232 return false;
1233}

1235// We want to place orphan sections so that they share as much
1236// characteristics with their neighbors as possible. For example, if
1237// both are rw, or both are tls.
1238static std::vector<BaseCommand *>::iterator
1239findOrphanPos(std::vector<BaseCommand *>::iterator B,
1240 std::vector<BaseCommand *>::iterator E) {
1241 OutputSection *Sec = cast<OutputSection>(*E);
1242
1243 // Find the first element that has as close a rank as possible.
1244 auto I = std::max_element(B, E, [=](BaseCommand *A, BaseCommand *B) {
1245 return getRankProximity(Sec, A) < getRankProximity(Sec, B);
1246 });
1247 if (I == E)
1248 return E;
1249
1250 // Consider all existing sections with the same proximity.
1251 int Proximity = getRankProximity(Sec, *I);
1252 for (; I != E; ++I) {
1253 auto *CurSec = dyn_cast<OutputSection>(*I);
1254 if (!CurSec || !CurSec->HasInputSections)
1255 continue;
1256 if (getRankProximity(Sec, CurSec) != Proximity ||
1257 Sec->SortRank < CurSec->SortRank)
1258 break;
1259 }

1263 return OS && OS->HasInputSections;
1264 };
1265 auto J = std::find_if(llvm::make_reverse_iterator(I),
1266 llvm::make_reverse_iterator(B),
1267 IsOutputSecWithInputSections);
1268 I = J.base();
1269
1270 // As a special case, if the orphan section is the last section, put
1271 // it at the very end, past any other commands.
1272 // This matches bfd's behavior and is convenient when the linker script fully
1273 // specifies the start of the file, but doesn't care about the end (the non
1274 // alloc sections for example).
1275 auto NextSec = std::find_if(I, E, IsOutputSecWithInputSections);
1276 if (NextSec == E)
1277 return E;
1278
1279 while (I != E && shouldSkip(*I))
1280 ++I;
1281 return I;
1282}

1284// Builds section order for handling --symbol-ordering-file.
1285static DenseMap<const InputSectionBase *, int> buildSectionOrder() {
1286 DenseMap<const InputSectionBase *, int> SectionOrder;
1287 // Use the rarely used option -call-graph-ordering-file to sort sections.
1288 if (!Config->CallGraphProfile.empty())
1289 return computeCallGraphProfileOrder();
1290
1291 if (Config->SymbolOrderingFile.empty())
1292 return SectionOrder;
1293
1294 struct SymbolOrderEntry {
1295 int Priority;
1296 bool Present;
1297 };
1298
1299 // Build a map from symbols to their priorities. Symbols that didn't
1300 // appear in the symbol ordering file have the lowest priority 0.
1301 // All explicitly mentioned symbols have negative (higher) priorities.
1302 DenseMap<StringRef, SymbolOrderEntry> SymbolOrder;
1303 int Priority = -Config->SymbolOrderingFile.size();
1304 for (StringRef S : Config->SymbolOrderingFile)
1305 SymbolOrder.insert({S, {Priority++, false}});
1306
1307 // Build a map from sections to their priorities.
1308 auto AddSym = [&](Symbol &Sym) {
1309 auto It = SymbolOrder.find(Sym.getName());
1310 if (It == SymbolOrder.end())
1311 return;
1312 SymbolOrderEntry &Ent = It->second;
1313 Ent.Present = true;
1314
1315 maybeWarnUnorderableSymbol(&Sym);
1316
1317 if (auto *D = dyn_cast<Defined>(&Sym)) {
1318 if (auto *Sec = dyn_cast_or_null<InputSectionBase>(D->Section)) {
1319 int &Priority = SectionOrder[cast<InputSectionBase>(Sec->Repl)];
1320 Priority = std::min(Priority, Ent.Priority);
1321 }
1322 }
1323 };
1324
1325 // We want both global and local symbols. We get the global ones from the
1326 // symbol table and iterate the object files for the local ones.
1327 Symtab->forEachSymbol([&](Symbol *Sym) {
1328 if (!Sym->isLazy())
1329 AddSym(*Sym);
1330 });
1331
1332 for (InputFile *File : ObjectFiles)
1333 for (Symbol *Sym : File->getSymbols())
1334 if (Sym->isLocal())
1335 AddSym(*Sym);
1336
1337 if (Config->WarnSymbolOrdering)
1338 for (auto OrderEntry : SymbolOrder)
1339 if (!OrderEntry.second.Present)
1340 warn("symbol ordering file: no such symbol: " + OrderEntry.first);
1341
1342 return SectionOrder;
1343}

1345// Sorts the sections in ISD according to the provided section order.
1346static void
1347sortISDBySectionOrder(InputSectionDescription *ISD,
1348 const DenseMap<const InputSectionBase *, int> &Order) {
1349 std::vector<InputSection *> UnorderedSections;
1350 std::vector<std::pair<InputSection *, int>> OrderedSections;
1351 uint64_t UnorderedSize = 0;
1352
1353 for (InputSection *IS : ISD->Sections) {
1354 auto I = Order.find(IS);
1355 if (I == Order.end()) {
1356 UnorderedSections.push_back(IS);
1357 UnorderedSize += IS->getSize();
1358 continue;
1359 }
1360 OrderedSections.push_back({IS, I->second});
1361 }
1362 llvm::sort(OrderedSections, [&](std::pair<InputSection *, int> A,
1363 std::pair<InputSection *, int> B) {
1364 return A.second < B.second;
1365 });
1366
1367 // Find an insertion point for the ordered section list in the unordered
1368 // section list. On targets with limited-range branches, this is the mid-point
1369 // of the unordered section list. This decreases the likelihood that a range
1370 // extension thunk will be needed to enter or exit the ordered region. If the
1371 // ordered section list is a list of hot functions, we can generally expect
1372 // the ordered functions to be called more often than the unordered functions,
1373 // making it more likely that any particular call will be within range, and
1374 // therefore reducing the number of thunks required.
1375 //
1376 // For example, imagine that you have 8MB of hot code and 32MB of cold code.
1377 // If the layout is:
1378 //
1379 // 8MB hot
1380 // 32MB cold
1381 //
1382 // only the first 8-16MB of the cold code (depending on which hot function it
1383 // is actually calling) can call the hot code without a range extension thunk.
1384 // However, if we use this layout:
1385 //
1386 // 16MB cold
1387 // 8MB hot
1388 // 16MB cold
1389 //
1390 // both the last 8-16MB of the first block of cold code and the first 8-16MB
1391 // of the second block of cold code can call the hot code without a thunk. So
1392 // we effectively double the amount of code that could potentially call into
1393 // the hot code without a thunk.
1394 size_t InsPt = 0;
1395 if (Target->getThunkSectionSpacing() && !OrderedSections.empty()) {
1396 uint64_t UnorderedPos = 0;
1397 for (; InsPt != UnorderedSections.size(); ++InsPt) {
1398 UnorderedPos += UnorderedSections[InsPt]->getSize();
1399 if (UnorderedPos > UnorderedSize / 2)
1400 break;
1401 }
1402 }
1403
1404 ISD->Sections.clear();
1405 for (InputSection *IS : makeArrayRef(UnorderedSections).slice(0, InsPt))
1406 ISD->Sections.push_back(IS);
1407 for (std::pair<InputSection *, int> P : OrderedSections)
1408 ISD->Sections.push_back(P.first);
1409 for (InputSection *IS : makeArrayRef(UnorderedSections).slice(InsPt))
1410 ISD->Sections.push_back(IS);
1411}

1413static void sortSection(OutputSection *Sec,
1414 const DenseMap<const InputSectionBase *, int> &Order) {
1415 StringRef Name = Sec->Name;
1416
1417 // Sort input sections by section name suffixes for
1418 // __attribute__((init_priority(N))).
1419 if (Name == ".init_array" || Name == ".fini_array") {
1420 if (!Script->HasSectionsCommand)
1421 Sec->sortInitFini();
1422 return;
1423 }
1424
1425 // Sort input sections by the special rule for .ctors and .dtors.
1426 if (Name == ".ctors" || Name == ".dtors") {
1427 if (!Script->HasSectionsCommand)
1428 Sec->sortCtorsDtors();
1429 return;
1430 }
1431
1432 // Never sort these.
1433 if (Name == ".init" || Name == ".fini")
1434 return;
1435
1436 // .toc is allocated just after .got and is accessed using GOT-relative
1437 // relocations. Object files compiled with small code model have an
1438 // addressable range of [.got, .got + 0xFFFC] for GOT-relative relocations.
1439 // To reduce the risk of relocation overflow, .toc contents are sorted so that
1440 // sections having smaller relocation offsets are at beginning of .toc
1441 if (Config->EMachine == EM_PPC64 && Name == ".toc") {
1442 if (Script->HasSectionsCommand)
1443 return;
1444 assert(Sec->SectionCommands.size() == 1);
1445 auto *ISD = cast<InputSectionDescription>(Sec->SectionCommands[0]);
1446 llvm::stable_sort(ISD->Sections,
1447 [](const InputSection *A, const InputSection *B) -> bool {
1448 return A->File->PPC64SmallCodeModelTocRelocs &&
1449 !B->File->PPC64SmallCodeModelTocRelocs;
1450 });
1451 return;
1452 }
1453
1454 // Sort input sections by priority using the list provided
1455 // by --symbol-ordering-file.
1456 if (!Order.empty())
1457 for (BaseCommand *B : Sec->SectionCommands)
1458 if (auto *ISD = dyn_cast<InputSectionDescription>(B))
1459 sortISDBySectionOrder(ISD, Order);
1460}

1462// If no layout was provided by linker script, we want to apply default
1463// sorting for special input sections. This also handles --symbol-ordering-file.
1464template <class ELFT> void Writer<ELFT>::sortInputSections() {
1465 // Build the order once since it is expensive.
1466 DenseMap<const InputSectionBase *, int> Order = buildSectionOrder();
1467 for (BaseCommand *Base : Script->SectionCommands)
1468 if (auto *Sec = dyn_cast<OutputSection>(Base))
1469 sortSection(Sec, Order);
1470}

1472template <class ELFT> void Writer<ELFT>::sortSections() {
1473 Script->adjustSectionsBeforeSorting();
1474
1475 // Don't sort if using -r. It is not necessary and we want to preserve the
1476 // relative order for SHF_LINK_ORDER sections.
1477 if (Config->Relocatable)
1478 return;
1479
1480 sortInputSections();
1481
1482 for (BaseCommand *Base : Script->SectionCommands) {
1483 auto *OS = dyn_cast<OutputSection>(Base);
1484 if (!OS)
1485 continue;
1486 OS->SortRank = getSectionRank(OS);
1487
1488 // We want to assign rude approximation values to OutSecOff fields
1489 // to know the relative order of the input sections. We use it for
1490 // sorting SHF_LINK_ORDER sections. See resolveShfLinkOrder().
1491 uint64_t I = 0;
1492 for (InputSection *Sec : getInputSections(OS))
1493 Sec->OutSecOff = I++;
1494 }
1495
1496 if (!Script->HasSectionsCommand) {
1497 // We know that all the OutputSections are contiguous in this case.
1498 auto IsSection = [](BaseCommand *Base) { return isa<OutputSection>(Base); };
1499 std::stable_sort(
1500 llvm::find_if(Script->SectionCommands, IsSection),
1501 llvm::find_if(llvm::reverse(Script->SectionCommands), IsSection).base(),
1502 compareSections);
1503 return;
1504 }
1505
1506 // Orphan sections are sections present in the input files which are
1507 // not explicitly placed into the output file by the linker script.
1508 //
1509 // The sections in the linker script are already in the correct
1510 // order. We have to figuere out where to insert the orphan
1511 // sections.
1512 //
1513 // The order of the sections in the script is arbitrary and may not agree with
1514 // compareSections. This means that we cannot easily define a strict weak
1515 // ordering. To see why, consider a comparison of a section in the script and
1516 // one not in the script. We have a two simple options:
1517 // * Make them equivalent (a is not less than b, and b is not less than a).
1518 // The problem is then that equivalence has to be transitive and we can
1519 // have sections a, b and c with only b in a script and a less than c
1520 // which breaks this property.
1521 // * Use compareSectionsNonScript. Given that the script order doesn't have
1522 // to match, we can end up with sections a, b, c, d where b and c are in the
1523 // script and c is compareSectionsNonScript less than b. In which case d
1524 // can be equivalent to c, a to b and d < a. As a concrete example:
1525 // .a (rx) # not in script
1526 // .b (rx) # in script
1527 // .c (ro) # in script
1528 // .d (ro) # not in script
1529 //
1530 // The way we define an order then is:
1531 // * Sort only the orphan sections. They are in the end right now.
1532 // * Move each orphan section to its preferred position. We try
1533 // to put each section in the last position where it can share
1534 // a PT_LOAD.
1535 //
1536 // There is some ambiguity as to where exactly a new entry should be
1537 // inserted, because Commands contains not only output section
1538 // commands but also other types of commands such as symbol assignment
1539 // expressions. There's no correct answer here due to the lack of the
1540 // formal specification of the linker script. We use heuristics to
1541 // determine whether a new output command should be added before or
1542 // after another commands. For the details, look at shouldSkip
1543 // function.
1544
1545 auto I = Script->SectionCommands.begin();
1546 auto E = Script->SectionCommands.end();
1547 auto NonScriptI = std::find_if(I, E, [](BaseCommand *Base) {
1548 if (auto *Sec = dyn_cast<OutputSection>(Base))
1549 return Sec->SectionIndex == UINT32_MAX;
1550 return false;
1551 });
1552
1553 // Sort the orphan sections.
1554 std::stable_sort(NonScriptI, E, compareSections);
1555
1556 // As a horrible special case, skip the first . assignment if it is before any
1557 // section. We do this because it is common to set a load address by starting
1558 // the script with ". = 0xabcd" and the expectation is that every section is
1559 // after that.
1560 auto FirstSectionOrDotAssignment =
1561 std::find_if(I, E, [](BaseCommand *Cmd) { return !shouldSkip(Cmd); });
1562 if (FirstSectionOrDotAssignment != E &&
1563 isa<SymbolAssignment>(**FirstSectionOrDotAssignment))
1564 ++FirstSectionOrDotAssignment;
1565 I = FirstSectionOrDotAssignment;
1566
1567 while (NonScriptI != E) {
1568 auto Pos = findOrphanPos(I, NonScriptI);
1569 OutputSection *Orphan = cast<OutputSection>(*NonScriptI);
1570
1571 // As an optimization, find all sections with the same sort rank
1572 // and insert them with one rotate.
1573 unsigned Rank = Orphan->SortRank;
1574 auto End = std::find_if(NonScriptI + 1, E, [=](BaseCommand *Cmd) {
1575 return cast<OutputSection>(Cmd)->SortRank != Rank;
1576 });
1577 std::rotate(Pos, NonScriptI, End);
1578 NonScriptI = End;
1579 }
1580
1581 Script->adjustSectionsAfterSorting();
1582}

1584static bool compareByFilePosition(InputSection *A, InputSection *B) {
1585 InputSection *LA = A->getLinkOrderDep();
1586 InputSection *LB = B->getLinkOrderDep();
1587 OutputSection *AOut = LA->getParent();
1588 OutputSection *BOut = LB->getParent();
1589
1590 if (AOut != BOut)
1591 return AOut->SectionIndex < BOut->SectionIndex;
1592 return LA->OutSecOff < LB->OutSecOff;
1593}

1595template <class ELFT> void Writer<ELFT>::resolveShfLinkOrder() {
1596 for (OutputSection *Sec : OutputSections) {
1597 if (!(Sec->Flags & SHF_LINK_ORDER))
1598 continue;
1599
1600 // Link order may be distributed across several InputSectionDescriptions
1601 // but sort must consider them all at once.
1602 std::vector<InputSection **> ScriptSections;
1603 std::vector<InputSection *> Sections;
1604 for (BaseCommand *Base : Sec->SectionCommands) {
1605 if (auto *ISD = dyn_cast<InputSectionDescription>(Base)) {
1606 for (InputSection *&IS : ISD->Sections) {
1607 ScriptSections.push_back(&IS);
1608 Sections.push_back(IS);
1609 }
1610 }
1611 }
1612
1613 // The ARM.exidx section use SHF_LINK_ORDER, but we have consolidated
1614 // this processing inside the ARMExidxsyntheticsection::finalizeContents().
1615 if (!Config->Relocatable && Config->EMachine == EM_ARM &&
1616 Sec->Type == SHT_ARM_EXIDX)
1617 continue;
1618
1619 llvm::stable_sort(Sections, compareByFilePosition);
1620
1621 for (int I = 0, N = Sections.size(); I < N; ++I)
1622 *ScriptSections[I] = Sections[I];
1623 }
1624}

1626// We need to generate and finalize the content that depends on the address of
1627// InputSections. As the generation of the content may also alter InputSection
1628// addresses we must converge to a fixed point. We do that here. See the comment
1629// in Writer<ELFT>::finalizeSections().
1630template <class ELFT> void Writer<ELFT>::finalizeAddressDependentContent() {
1631 ThunkCreator TC;
1632 AArch64Err843419Patcher A64P;
1633
1634 // For some targets, like x86, this loop iterates only once.
1635 for (;;) {
1636 bool Changed = false;
1637
1638 Script->assignAddresses();
1639
1640 if (Target->NeedsThunks)
1641 Changed |= TC.createThunks(OutputSections);
1642
1643 if (Config->FixCortexA53Errata843419) {
1644 if (Changed)
1645 Script->assignAddresses();
1646 Changed |= A64P.createFixes();
1647 }
1648
1649 if (In.MipsGot)
1650 In.MipsGot->updateAllocSize();
1651
1652 for (Partition &Part : Partitions) {
1653 Changed |= Part.RelaDyn->updateAllocSize();
1654 if (Part.RelrDyn)
1655 Changed |= Part.RelrDyn->updateAllocSize();
1656 }
1657
1658 if (!Changed)
1659 return;
1660 }

1668// In order to allow users to manipulate linker-synthesized sections,
1669// we had to add synthetic sections to the input section list early,
1670// even before we make decisions whether they are needed. This allows
1671// users to write scripts like this: ".mygot : { .got }".
1672//
1673// Doing it has an unintended side effects. If it turns out that we
1674// don't need a .got (for example) at all because there's no
1675// relocation that needs a .got, we don't want to emit .got.
1676//
1677// To deal with the above problem, this function is called after
1678// scanRelocations is called to remove synthetic sections that turn
1679// out to be empty.
1680static void removeUnusedSyntheticSections() {
1681 // All input synthetic sections that can be empty are placed after
1682 // all regular ones. We iterate over them all and exit at first
1683 // non-synthetic.
1684 for (InputSectionBase *S : llvm::reverse(InputSections)) {
1685 SyntheticSection *SS = dyn_cast<SyntheticSection>(S);
1686 if (!SS)
1687 return;
1688 OutputSection *OS = SS->getParent();
1689 if (!OS || SS->isNeeded())
1690 continue;
1691
1692 // If we reach here, then SS is an unused synthetic section and we want to
1693 // remove it from corresponding input section description of output section.
1694 for (BaseCommand *B : OS->SectionCommands)
1695 if (auto *ISD = dyn_cast<InputSectionDescription>(B))
1696 llvm::erase_if(ISD->Sections,
1697 [=](InputSection *IS) { return IS == SS; });
1698 }
1699}

1701// Returns true if a symbol can be replaced at load-time by a symbol
1702// with the same name defined in other ELF executable or DSO.
1703static bool computeIsPreemptible(const Symbol &B) {
1704 assert(!B.isLocal());
1705
1706 // Only symbols that appear in dynsym can be preempted.
1707 if (!B.includeInDynsym())
1708 return false;
1709
1710 // Only default visibility symbols can be preempted.
1711 if (B.Visibility != STV_DEFAULT)
1712 return false;
1713
1714 // At this point copy relocations have not been created yet, so any
1715 // symbol that is not defined locally is preemptible.
1716 if (!B.isDefined())
1717 return true;
1718
1719 // If we have a dynamic list it specifies which local symbols are preemptible.
1720 if (Config->HasDynamicList)
1721 return false;
1722
1723 if (!Config->Shared)
1724 return false;
1725
1726 // -Bsymbolic means that definitions are not preempted.
1727 if (Config->Bsymbolic || (Config->BsymbolicFunctions && B.isFunc()))
1728 return false;
1729 return true;
1730}

1732// Create output section objects and add them to OutputSections.
1733template <class ELFT> void Writer<ELFT>::finalizeSections() {
1734 Out::PreinitArray = findSection(".preinit_array");
1735 Out::InitArray = findSection(".init_array");
1736 Out::FiniArray = findSection(".fini_array");
1737
1738 // The linker needs to define SECNAME_start, SECNAME_end and SECNAME_stop
1739 // symbols for sections, so that the runtime can get the start and end
1740 // addresses of each section by section name. Add such symbols.
1741 if (!Config->Relocatable) {
1742 addStartEndSymbols();
1743 for (BaseCommand *Base : Script->SectionCommands)
1744 if (auto *Sec = dyn_cast<OutputSection>(Base))
1745 addStartStopSymbols(Sec);
1746 }
1747
1748 // Add _DYNAMIC symbol. Unlike GNU gold, our _DYNAMIC symbol has no type.
1749 // It should be okay as no one seems to care about the type.
1750 // Even the author of gold doesn't remember why gold behaves that way.
1751 // https://sourceware.org/ml/binutils/2002-03/msg00360.html
1752 bool NeedsDYNAMIC = (Config->Pic || !SharedFiles.empty()); // TODO: --as-needed?
1753 if (Main->Dynamic->Parent && NeedsDYNAMIC) {
1754 // Set _DYNAMIC to null for static binaries without shared libraries
1755 // Note: This is needed in order to not have a valid _DYNAMIC if
1756 // --export-dynamic is passed to a static executable. Some programs check
1757 // if they are dynamically linked using `if (&_DYNAMIC != 0)` so we should
1758 // keep this check working.
1759
1760 auto *S = Symtab->addSymbol(Defined{/*File=*/nullptr, "_DYNAMIC", STB_WEAK,
1761 STV_HIDDEN, STT_NOTYPE,
1762 /*Value=*/0, /*Size=*/0, Main->Dynamic});
1763 // In CheriABI we want sensible bounds if we do &_DYNAMIC in C code
1764 S->IsSectionStartSymbol = true;
1765 }
1766
1767 // The common check if a program is dynamically linked (&_DYNAMIC != 0)
1768 // will not work in the early CHERI startup. In PIC code we also can't
1769 // use dla _DYNAMIC since that needs either $gp or text relocations
1770 // Instead we just let the linker generate a new symbol _HAS__DYNAMIC
1771 if (auto *Reference = Symtab->find("_HAS__DYNAMIC"))
1772 if (!Reference->isDefined()) {
1773 Defined *HasDynamicSym = addAbsolute("_HAS__DYNAMIC");
1774 HasDynamicSym->Value = NeedsDYNAMIC ? 1 : 0;
1775 }
1776
1777 // Define __rel[a]_iplt_{start,end} symbols if needed.
1778 addRelIpltSymbols();
1779
1780 // RISC-V's gp can address +/- 2 KiB, set it to .sdata + 0x800 if not defined.
1781 if (Config->EMachine == EM_RISCV)
1782 if (!dyn_cast_or_null<Defined>(Symtab->find("__global_pointer$")))
1783 addOptionalRegular("__global_pointer$", findSection(".sdata"), 0x800);
1784
1785 if (Config->EMachine == EM_X86_64) {
1786 // On targets that support TLSDESC, _TLS_MODULE_BASE_ is defined in such a
1787 // way that:
1788 //
1789 // 1) Without relaxation: it produces a dynamic TLSDESC relocation that
1790 // computes 0.
1791 // 2) With LD->LE relaxation: _TLS_MODULE_BASE_@tpoff = 0 (lowest address in
1792 // the TLS block).
1793 //
1794 // 2) is special cased in @tpoff computation. To satisfy 1), we define it as
1795 // an absolute symbol of zero. This is different from GNU linkers which
1796 // define _TLS_MODULE_BASE_ relative to the first TLS section.
1797 Symbol *S = Symtab->find("_TLS_MODULE_BASE_");
1798 if (S && S->isUndefined()) {
1799 S->resolve(Defined{/*File=*/nullptr, S->getName(), STB_GLOBAL, STV_HIDDEN,
1800 STT_TLS, /*Value=*/0, 0,
1801 /*Section=*/nullptr});
1802 ElfSym::TlsModuleBase = cast<Defined>(S);
1803 }
1804 }
1805
1806 StringRef CaptableSym = "_CHERI_CAPABILITY_TABLE_";
1807 if (In.CheriCapTable) {
1808 // When creating relocatable output we should not define the
1809 // _CHERI_CAPABILITY_TABLE_ symbol because otherwise we get duplicate
1810 // symbol errors when linking that into a final executable
1811 if (!Config->Relocatable)
1812 ElfSym::CheriCapabilityTable =
1813 addOptionalRegular(CaptableSym, In.CheriCapTable, 0);
1814 }
1815
1816 // This responsible for splitting up .eh_frame section into
1817 // pieces. The relocation scan uses those pieces, so this has to be
1818 // earlier.
1819 for (Partition &Part : Partitions)
1820 finalizeSynthetic(Part.EhFrame);
1821
1822 Symtab->forEachSymbol([](Symbol *S) {
1823 if (!S->IsPreemptible)
1824 S->IsPreemptible = computeIsPreemptible(*S);
1825 });
1826
1827 // Scan relocations. This must be done after every symbol is declared so that
1828 // we can correctly decide if a dynamic relocation is needed.
1829 if (!Config->Relocatable)
1830 forEachRelSec(scanRelocations<ELFT>);
1831
1832 addIRelativeRelocs();
1833
1834 // Do the cap table index assignment
1835 // Must come before CapRelocs->finalizeContents() because it can add
1836 // __cap_relocs
1837 if (In.CheriCapTable) {
1838 // Ensure that we always have a _CHERI_CAPABILITY_TABLE_ symbol if the
1839 // cap table exists. This makes llvm-objdump more useful since it can now
1840 // print the target of a cap table load
1841 if (!ElfSym::CheriCapabilityTable && In.CheriCapTable->isNeeded()) {
1842 ElfSym::CheriCapabilityTable = cast<Defined>(
1843 Symtab->addSymbol(Defined{nullptr, CaptableSym, STB_LOCAL,
1844 STV_HIDDEN, STT_NOTYPE, 0, 0, In.CheriCapTable}));
1845 ElfSym::CheriCapabilityTable->IsSectionStartSymbol = true;
1846 assert(!ElfSym::CheriCapabilityTable->IsPreemptible);
1847 }
1848 In.CheriCapTable->assignValuesAndAddCapTableSymbols<ELFT>();
1849 }
1850
1851 // Now handle __cap_relocs (must be before RelaDyn because it might
1852 // result in new dynamic relocations being added)
1853 if (Config->ProcessCapRelocs) {
1854 finalizeSynthetic(InX<ELFT>::CapRelocs);
1855
1856 if (OutputSection *GS = findSection(".global_sizes"))
1857 // Also check if the .global_sizes section needs to be writable:
1858 for (BaseCommand *Cmd : GS->SectionCommands)
1859 if (auto *ISD = dyn_cast<InputSectionDescription>(Cmd))

2113// The linker is expected to define SECNAME_start and SECNAME_end
2114// symbols for a few sections. This function defines them.
2115template <class ELFT> void Writer<ELFT>::addStartEndSymbols() {
2116 // If a section does not exist, there's ambiguity as to how we
2117 // define _start and _end symbols for an init/fini section. Since
2118 // the loader assume that the symbols are always defined, we need to
2119 // always define them. But what value? The loader iterates over all
2120 // pointers between _start and _end to run global ctors/dtors, so if
2121 // the section is empty, their symbol values don't actually matter
2122 // as long as _start and _end point to the same location.
2123 //
2124 // That said, we don't want to set the symbols to 0 (which is
2125 // probably the simplest value) because that could cause some
2126 // program to fail to link due to relocation overflow, if their
2127 // program text is above 2 GiB. We use the address of the .text
2128 // section instead to prevent that failure.
2129 //
2130 // In a rare sitaution, .text section may not exist. If that's the
2131 // case, use the image base address as a last resort.
2132 OutputSection *Default = findSection(".text");
2133 if (!Default)
2134 Default = Out::ElfHeader;
2135
2136 auto Define = [=](StringRef Start, StringRef End, OutputSection *OS) {
2137 if (OS) {
2138 addOptionalRegular(Start, OS, 0);
2139 addOptionalRegular(End, OS, -1);
2140 } else {
2141 // Since this is an empty section we don't want to set CanBeSectionStart
2142 // Iterating over this should terminate immediately so setting the size
2143 // to zero is fine
2144 addOptionalRegular(Start, Default, 0, STV_HIDDEN, STB_GLOBAL,
2145 /*CanBeSectionStart=*/false);
2146 // End is not a section start symbol even though it has value 0:
2147 addOptionalRegular(End, Default, 0, STV_HIDDEN, STB_GLOBAL,
2148 /*CanBeSectionStart=*/false);
2149 }
2150 };

2152 Define("__preinit_array_start", "__preinit_array_end", Out::PreinitArray);

2155 Define("__ctors_start", "__ctors_end", findSection(".ctors"));

2208std::vector<PhdrEntry *> Writer<ELFT>::createPhdrs(Partition &Part) {

2210 auto AddHdr = [&](unsigned Type, unsigned Flags) -> PhdrEntry * {
2211 Ret.push_back(make<PhdrEntry>(Type, Flags));
2212 return Ret.back();
2213 };
2214
2215 unsigned PartNo = Part.getNumber();
2216 bool IsMain = PartNo == 1;
2217
2218 // The first phdr entry is PT_PHDR which describes the program header itself.
2219 if (IsMain)
2220 AddHdr(PT_PHDR, PF_R)->add(Out::ProgramHeaders);
2221 else
2222 AddHdr(PT_PHDR, PF_R)->add(Part.ProgramHeaders->getParent());
2223
2224 // PT_INTERP must be the second entry if exists.
2225 if (OutputSection *Cmd = findSection(".interp", PartNo))
2226 AddHdr(PT_INTERP, Cmd->getPhdrFlags())->add(Cmd);
2227

2229 uint64_t Flags = computeFlags(PF_R);
2230 PhdrEntry *Load = nullptr;
2231
2232 // Add the headers. We will remove them if they don't fit.
2233 // In the other partitions the headers are ordinary sections, so they don't
2234 // need to be added here.
2235 if (IsMain) {
2236 Load = AddHdr(PT_LOAD, Flags);
2237 Load->add(Out::ElfHeader);
2238 Load->add(Out::ProgramHeaders);
2239 }

2241 // PT_GNU_RELRO includes all sections that should be marked as
2242 // read-only by dynamic linker after proccessing relocations.
2243 // Current dynamic loaders only support one PT_GNU_RELRO PHDR, give
2244 // an error message if more than one PT_GNU_RELRO PHDR is required.
2245 PhdrEntry *RelRo = make<PhdrEntry>(PT_GNU_RELRO, PF_R);
2246 bool InRelroPhdr = false;
2247 OutputSection *RelroEnd = nullptr;
2248 for (OutputSection *Sec : OutputSections) {
2249 if (Sec->Partition != PartNo || !needsPtLoad(Sec))
2250 continue;

2255 else
2256 error("section: " + Sec->Name + " is not contiguous with other relro" +

2261 }

2273 // partition so that a segment is created for it in the main partition,
2274 // which will cause the dynamic loader to reserve space for the other

2414// Compute an in-file position for a given section. The file offset must be the
2415// same with its virtual address modulo the page size, so that the loader can
2416// load executables without any address adjustment.
2417static uint64_t computeFileOffset(OutputSection *OS, uint64_t Off) {
2418 // If the section is not in a PT_LOAD, we just have to align it.
2419 if (!OS->PtLoad) {
2420 // File offsets are not significant for .bss sections when not the first
2421 // section in a PT_LOAD. By convention, we keep section offsets
2422 // monotonically increasing rather than setting to zero.
2423 if (OS->Type == SHT_NOBITS)
2424 return Off;
2425
2426 return alignTo(Off, OS->Alignment);
2427 }
2428
2429 // The first section in a PT_LOAD has to have congruent offset and address
2430 // module the page size.
2431 OutputSection *First = OS->PtLoad->FirstSec;
2432 if (OS == First) {
2433 uint64_t Alignment = std::max<uint64_t>(OS->Alignment, Config->MaxPageSize);
2434 return alignTo(Off, Alignment, OS->Addr);
2435 }

2437 // This is a .bss section that isn't first in its PT_LOAD; see above.
2438 if (OS->Type == SHT_NOBITS)
2439 return Off;
2440
2441 // If two sections share the same PT_LOAD the file offset is calculated
2442 // using this formula: Off2 = Off1 + (VA2 - VA1).
2443 return First->Offset + OS->Addr - First->Addr;
2444}

2469// Assign file offsets to output sections.
2470template <class ELFT> void Writer<ELFT>::assignFileOffsets() {
2471 uint64_t Off = 0;
2472 Off = setFileOffset(Out::ElfHeader, Off);
2473 Off = setFileOffset(Out::ProgramHeaders, Off);
2474
2475 PhdrEntry *LastRX = nullptr;
2476 for (Partition &Part : Partitions)
2477 for (PhdrEntry *P : Part.Phdrs)
2478 if (P->p_type == PT_LOAD && (P->p_flags & PF_X))
2479 LastRX = P;
2480
2481 for (OutputSection *Sec : OutputSections) {
2482 Off = setFileOffset(Sec, Off);
2483 if (Script->HasSectionsCommand)
2484 continue;
2485
2486 // If this is a last section of the last executable segment and that
2487 // segment is the last loadable segment, align the offset of the
2488 // following section to avoid loading non-segments parts of the file.
2489 if (LastRX && LastRX->LastSec == Sec)
2490 Off = alignTo(Off, Config->CommonPageSize);
2491 }
2492
2493 SectionHeaderOff = alignTo(Off, Config->Wordsize);
2494 FileSize = SectionHeaderOff + (OutputSections.size() + 1) * sizeof(Elf_Shdr);
2495
2496 // Our logic assumes that sections have rising VA within the same segment.
2497 // With use of linker scripts it is possible to violate this rule and get file
2498 // offset overlaps or overflows. That should never happen with a valid script
2499 // which does not move the location counter backwards and usually scripts do
2500 // not do that. Unfortunately, there are apps in the wild, for example, Linux
2501 // kernel, which control segment distribution explicitly and move the counter
2502 // backwards, so we have to allow doing that to support linking them. We
2503 // perform non-critical checks for overlaps in checkSectionOverlap(), but here
2504 // we want to prevent file size overflows because it would crash the linker.
2505 for (OutputSection *Sec : OutputSections) {
2506 if (Sec->Type == SHT_NOBITS)
2507 continue;
2508 if ((Sec->Offset > FileSize) || (Sec->Offset + Sec->Size > FileSize))
2509 error("unable to place section " + Sec->Name + " at file offset " +
2510 rangeToString(Sec->Offset, Sec->Size) +
2511 "; check your linker script for overflows");
2512 }
2513}

2515// Finalize the program headers. We call this function after we assign
2516// file offsets and VAs to all sections.
2517template <class ELFT> void Writer<ELFT>::setPhdrs(Partition &Part) {
2518 for (PhdrEntry *P : Part.Phdrs) {
2519 OutputSection *First = P->FirstSec;
2520 OutputSection *Last = P->LastSec;
2521
2522 if (First) {
2523 P->p_filesz = Last->Offset - First->Offset;
2524 if (Last->Type != SHT_NOBITS)
2525 P->p_filesz += Last->Size;
2526
2527 P->p_memsz = Last->Addr + Last->Size - First->Addr;
2528 P->p_offset = First->Offset;
2529 P->p_vaddr = First->Addr;
2530
2531 // File offsets in partitions other than the main partition are relative
2532 // to the offset of the ELF headers. Perform that adjustment now.
2533 if (Part.ElfHeader)
2534 P->p_offset -= Part.ElfHeader->getParent()->Offset;
2535
2536 if (!P->HasLMA)
2537 P->p_paddr = First->getLMA();
2538 }
2539
2540 if (P->p_type == PT_LOAD) {
2541 P->p_align = std::max<uint64_t>(P->p_align, Config->MaxPageSize);
2542 } else if (P->p_type == PT_GNU_RELRO) {
2543 P->p_align = 1;
2544 // The glibc dynamic loader rounds the size down, so we need to round up
2545 // to protect the last page. This is a no-op on FreeBSD which always
2546 // rounds up.
2547 P->p_memsz = alignTo(P->p_memsz, Config->CommonPageSize);
2548 }
2549 }

2560// Check whether sections overlap for a specific address range (file offsets,
2561// load and virtual adresses).
2562static void checkOverlap(StringRef Name, std::vector<SectionOffset> &Sections,
2563 bool IsVirtualAddr) {
2564 llvm::sort(Sections, [=](const SectionOffset &A, const SectionOffset &B) {
2565 return A.Offset < B.Offset;
2566 });
2567
2568 // Finding overlap is easy given a vector is sorted by start position.
2569 // If an element starts before the end of the previous element, they overlap.
2570 for (size_t I = 1, End = Sections.size(); I < End; ++I) {
2571 SectionOffset A = Sections[I - 1];
2572 SectionOffset B = Sections[I];
2573 if (B.Offset >= A.Offset + A.Sec->Size)
2574 continue;
2575
2576 // If both sections are in OVERLAY we allow the overlapping of virtual
2577 // addresses, because it is what OVERLAY was designed for.
2578 if (IsVirtualAddr && A.Sec->InOverlay && B.Sec->InOverlay)
2579 continue;
2580
2581 errorOrWarn("section " + A.Sec->Name + " " + Name +
2582 " range overlaps with " + B.Sec->Name + "\n>>> " + A.Sec->Name +
2583 " range is " + rangeToString(A.Offset, A.Sec->Size) + "\n>>> " +
2584 B.Sec->Name + " range is " +
2585 rangeToString(B.Offset, B.Sec->Size));
2586 }
2587}

2589// Check for overlapping sections and address overflows.
2590//
2591// In this function we check that none of the output sections have overlapping
2592// file offsets. For SHF_ALLOC sections we also check that the load address
2593// ranges and the virtual address ranges don't overlap
2594template <class ELFT> void Writer<ELFT>::checkSections() {
2595 // First, check that section's VAs fit in available address space for target.
2596 for (OutputSection *OS : OutputSections)
2597 if ((OS->Addr + OS->Size < OS->Addr) ||
2598 (!ELFT::Is64Bits && OS->Addr + OS->Size > UINT32_MAX))
2599 errorOrWarn("section " + OS->Name + " at 0x" + utohexstr(OS->Addr) +
2600 " of size 0x" + utohexstr(OS->Size) +
2601 " exceeds available address space");
2602
2603 // Check for overlapping file offsets. In this case we need to skip any
2604 // section marked as SHT_NOBITS. These sections don't actually occupy space in
2605 // the file so Sec->Offset + Sec->Size can overlap with others. If --oformat
2606 // binary is specified only add SHF_ALLOC sections are added to the output
2607 // file so we skip any non-allocated sections in that case.
2608 std::vector<SectionOffset> FileOffs;
2609 for (OutputSection *Sec : OutputSections)
2610 if (Sec->Size > 0 && Sec->Type != SHT_NOBITS &&
2611 (!Config->OFormatBinary || (Sec->Flags & SHF_ALLOC)))
2612 FileOffs.push_back({Sec, Sec->Offset});
2613 checkOverlap("file", FileOffs, false);
2614
2615 // When linking with -r there is no need to check for overlapping virtual/load
2616 // addresses since those addresses will only be assigned when the final
2617 // executable/shared object is created.
2618 if (Config->Relocatable)
2619 return;
2620
2621 // Checking for overlapping virtual and load addresses only needs to take
2622 // into account SHF_ALLOC sections since others will not be loaded.
2623 // Furthermore, we also need to skip SHF_TLS sections since these will be
2624 // mapped to other addresses at runtime and can therefore have overlapping
2625 // ranges in the file.
2626 std::vector<SectionOffset> VMAs;
2627 for (OutputSection *Sec : OutputSections)
2628 if (Sec->Size > 0 && (Sec->Flags & SHF_ALLOC) && !(Sec->Flags & SHF_TLS))
2629 VMAs.push_back({Sec, Sec->Addr});
2630 checkOverlap("virtual address", VMAs, true);
2631
2632 // Finally, check that the load addresses don't overlap. This will usually be
2633 // the same as the virtual addresses but can be different when using a linker
2634 // script with AT().
2635 std::vector<SectionOffset> LMAs;
2636 for (OutputSection *Sec : OutputSections)
2637 if (Sec->Size > 0 && (Sec->Flags & SHF_ALLOC) && !(Sec->Flags & SHF_TLS))
2638 LMAs.push_back({Sec, Sec->getLMA()});
2639 checkOverlap("load address", LMAs, false);
2640}

2642// The entry point address is chosen in the following ways.
2643//
2644// 1. the '-e' entry command-line option;
2645// 2. the ENTRY(symbol) command in a linker control script;
2646// 3. the value of the symbol _start, if present;
2647// 4. the number represented by the entry symbol, if it is a number;
2648// 5. the address of the first byte of the .text section, if present;
2649// 6. the address 0.
2650static uint64_t getEntryAddr() {
2651 // Case 1, 2 or 3
2652 if (Symbol *B = Symtab->find(Config->Entry))
2653 return B->getVA();
2654
2655 // Case 4
2656 uint64_t Addr;
2657 if (to_integer(Config->Entry, Addr))
2658 return Addr;
2659
2660 // Case 5
2661 if (OutputSection *Sec = findSection(".text")) {
2662 if (Config->WarnMissingEntry)
2663 warn("cannot find entry symbol " + Config->Entry + "; defaulting to 0x" +
2664 utohexstr(Sec->Addr));
2665 return Sec->Addr;
2666 }
2667
2668 // Case 6
2669 if (Config->WarnMissingEntry)
2670 warn("cannot find entry symbol " + Config->Entry +
2671 "; not setting start address");
2672 return 0;
2673}

2683template <class ELFT> void Writer<ELFT>::writeHeader() {
2684 writeEhdr<ELFT>(Out::BufferStart, *Main);
2685 writePhdrs<ELFT>(Out::BufferStart + sizeof(Elf_Ehdr), *Main);
2686
2687 auto *EHdr = reinterpret_cast<Elf_Ehdr *>(Out::BufferStart);
2688 EHdr->e_type = getELFType();
2689 EHdr->e_entry = getEntryAddr();
2690 EHdr->e_shoff = SectionHeaderOff;
2691
2692 // Write the section header table.
2693 //
2694 // The ELF header can only store numbers up to SHN_LORESERVE in the e_shnum
2695 // and e_shstrndx fields. When the value of one of these fields exceeds
2696 // SHN_LORESERVE ELF requires us to put sentinel values in the ELF header and
2697 // use fields in the section header at index 0 to store
2698 // the value. The sentinel values and fields are:
2699 // e_shnum = 0, SHdrs[0].sh_size = number of sections.
2700 // e_shstrndx = SHN_XINDEX, SHdrs[0].sh_link = .shstrtab section index.
2701 auto *SHdrs = reinterpret_cast<Elf_Shdr *>(Out::BufferStart + EHdr->e_shoff);
2702 size_t Num = OutputSections.size() + 1;
2703 if (Num >= SHN_LORESERVE)
2704 SHdrs->sh_size = Num;
2705 else
2706 EHdr->e_shnum = Num;
2707
2708 uint32_t StrTabIndex = In.ShStrTab->getParent()->SectionIndex;
2709 if (StrTabIndex >= SHN_LORESERVE) {
2710 SHdrs->sh_link = StrTabIndex;
2711 EHdr->e_shstrndx = SHN_XINDEX;
2712 } else {
2713 EHdr->e_shstrndx = StrTabIndex;
2714 }
2715
2716 for (OutputSection *Sec : OutputSections)
2717 Sec->writeHeaderTo<ELFT>(++SHdrs);
2718}

2750static void fillTrap(uint8_t *I, uint8_t *End) {
2751 for (; I + 4 <= End; I += 4)
2752 memcpy(I, &Target->TrapInstr, 4);

2755// Fill the last page of executable segments with trap instructions
2756// instead of leaving them as zero. Even though it is not required by any
2757// standard, it is in general a good thing to do for security reasons.
2758//
2759// We'll leave other pages in segments as-is because the rest will be
2760// overwritten by output sections.
2761template <class ELFT> void Writer<ELFT>::writeTrapInstr() {
2762 if (Script->HasSectionsCommand)
2763 return;
2764
2765 for (Partition &Part : Partitions) {
2766 // Fill the last page.
2767 for (PhdrEntry *P : Part.Phdrs)
2768 if (P->p_type == PT_LOAD && (P->p_flags & PF_X))
2769 fillTrap(Out::BufferStart + alignDown(P->FirstSec->Offset + P->p_filesz,
2770 Config->CommonPageSize),
2771 Out::BufferStart + alignTo(P->FirstSec->Offset + P->p_filesz,
2772 Config->CommonPageSize));
2773
2774 // Round up the file size of the last segment to the page boundary iff it is
2775 // an executable segment to ensure that other tools don't accidentally
2776 // trim the instruction padding (e.g. when stripping the file).
2777 PhdrEntry *Last = nullptr;
2778 for (PhdrEntry *P : Part.Phdrs)
2779 if (P->p_type == PT_LOAD)
2780 Last = P;
2781
2782 if (Last && (Last->p_flags & PF_X))
2783 Last->p_memsz = Last->p_filesz =
2784 alignTo(Last->p_filesz, Config->CommonPageSize);
2785 }
2786}

2802// Split one uint8 array into small pieces of uint8 arrays.
2803static std::vector<ArrayRef<uint8_t>> split(ArrayRef<uint8_t> Arr,
2804 size_t ChunkSize) {
2805 std::vector<ArrayRef<uint8_t>> Ret;
2806 while (Arr.size() > ChunkSize) {
2807 Ret.push_back(Arr.take_front(ChunkSize));
2808 Arr = Arr.drop_front(ChunkSize);
2809 }
2810 if (!Arr.empty())
2811 Ret.push_back(Arr);
2812 return Ret;
2813}

2815// Computes a hash value of Data using a given hash function.
2816// In order to utilize multiple cores, we first split data into 1MB
2817// chunks, compute a hash for each chunk, and then compute a hash value
2818// of the hash values.
2819static void
2820computeHash(llvm::MutableArrayRef<uint8_t> HashBuf,
2821 llvm::ArrayRef<uint8_t> Data,
2822 std::function<void(uint8_t *Dest, ArrayRef<uint8_t> Arr)> HashFn) {
2823 std::vector<ArrayRef<uint8_t>> Chunks = split(Data, 1024 * 1024);
2824 std::vector<uint8_t> Hashes(Chunks.size() * HashBuf.size());
2825
2826 // Compute hash values.
2827 parallelForEachN(0, Chunks.size(), [&](size_t I) {
2828 HashFn(Hashes.data() + I * HashBuf.size(), Chunks[I]);
2829 });
2830
2831 // Write to the final output buffer.
2832 HashFn(HashBuf.data(), Hashes);
2833}

2835template <class ELFT> void Writer<ELFT>::writeBuildId() {
2836 if (!Main->BuildId || !Main->BuildId->getParent())
2837 return;
2838
2839 if (Config->BuildId == BuildIdKind::Hexstring) {
2840 for (Partition &Part : Partitions)
2841 Part.BuildId->writeBuildId(Config->BuildIdVector);
2842 return;
2843 }
2844
2845 // Compute a hash of all sections of the output file.
2846 size_t HashSize = Main->BuildId->HashSize;
2847 std::vector<uint8_t> BuildId(HashSize);
2848 llvm::ArrayRef<uint8_t> Buf{Out::BufferStart, size_t(FileSize)};
2849
2850 switch (Config->BuildId) {
2851 case BuildIdKind::Fast:
2852 computeHash(BuildId, Buf, [](uint8_t *Dest, ArrayRef<uint8_t> Arr) {
2853 write64le(Dest, xxHash64(Arr));
2854 });
2855 break;
2856 case BuildIdKind::Md5:
2857 computeHash(BuildId, Buf, [&](uint8_t *Dest, ArrayRef<uint8_t> Arr) {
2858 memcpy(Dest, MD5::hash(Arr).data(), HashSize);
2859 });
2860 break;
2861 case BuildIdKind::Sha1:
2862 computeHash(BuildId, Buf, [&](uint8_t *Dest, ArrayRef<uint8_t> Arr) {
2863 memcpy(Dest, SHA1::hash(Arr).data(), HashSize);
2864 });
2865 break;
2866 case BuildIdKind::Uuid:
2867 if (auto EC = llvm::getRandomBytes(BuildId.data(), HashSize))
2868 error("entropy source failure: " + EC.message());
2869 break;
2870 default:
2871 llvm_unreachable("unknown BuildIdKind");
2872 }
2873 for (Partition &Part : Partitions)
2874 Part.BuildId->writeBuildId(BuildId);
2875}
