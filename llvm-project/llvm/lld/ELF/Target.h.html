26class TargetInfo {
27public:
28 virtual uint32_t calcEFlags() const { return 0; }
29 virtual int getCapabilitySize() const { return 0; }
30 virtual RelType getDynRel(RelType Type) const { return Type; }
31 virtual void writeGotPltHeader(uint8_t *Buf) const {}
32 virtual void writeGotHeader(uint8_t *Buf) const {}
33 virtual void writeGotPlt(uint8_t *Buf, const Symbol &S) const {};
34 virtual void writeIgotPlt(uint8_t *Buf, const Symbol &S) const;
35 virtual int64_t getImplicitAddend(const uint8_t *Buf, RelType Type) const;
36 virtual int getTlsGdRelaxSkip(RelType Type) const { return 1; }
37
38 // If lazy binding is supported, the first entry of the PLT has code
39 // to call the dynamic linker to resolve PLT entries the first time
40 // they are called. This function writes that code.
41 virtual void writePltHeader(uint8_t *Buf) const {}
42
43 virtual void writePlt(uint8_t *Buf, uint64_t GotEntryAddr,
44 uint64_t PltEntryAddr, int32_t Index,
45 unsigned RelOff) const {}
46 virtual void addPltHeaderSymbols(InputSection &IS) const {}
47 virtual void addPltSymbols(InputSection &IS, uint64_t Off) const {}
48
49 // Returns true if a relocation only uses the low bits of a value such that
50 // all those bits are in the same page. For example, if the relocation
51 // only uses the low 12 bits in a system with 4k pages. If this is true, the
52 // bits will always have the same value at runtime and we don't have to emit
53 // a dynamic relocation.
54 virtual bool usesOnlyLowPageBits(RelType Type) const;
55
56 // Decide whether a Thunk is needed for the relocation from File
57 // targeting S.
58 virtual bool needsThunk(RelExpr Expr, RelType RelocType,
59 const InputFile *File, uint64_t BranchAddr,
60 const Symbol &S) const;
61
62 // On systems with range extensions we place collections of Thunks at
63 // regular spacings that enable the majority of branches reach the Thunks.
64 // a value of 0 means range extension thunks are not supported.
65 virtual uint32_t getThunkSectionSpacing() const { return 0; }
66
67 // The function with a prologue starting at Loc was compiled with
68 // -fsplit-stack and it calls a function compiled without. Adjust the prologue
69 // to do the right thing. See https://gcc.gnu.org/wiki/SplitStacks.
70 // The symbols st_other flags are needed on PowerPC64 for determining the
71 // offset to the split-stack prologue.
72 virtual bool adjustPrologueForCrossSplitStack(uint8_t *Loc, uint8_t *End,
73 uint8_t StOther) const;
74
75 // Return true if we can reach Dst from Src with Relocation RelocType
76 virtual bool inBranchRange(RelType Type, uint64_t Src,
77 uint64_t Dst) const;
78 virtual RelExpr getRelExpr(RelType Type, const Symbol &S,
79 const uint8_t *Loc) const = 0;
80
81 virtual void relocateOne(uint8_t *Loc, RelType Type, uint64_t Val) const = 0;
82
83 virtual ~TargetInfo();
84
85 unsigned DefaultCommonPageSize = 4096;
86 unsigned DefaultMaxPageSize = 4096;
87
88 uint64_t getImageBase() const;
89
90 // True if _GLOBAL_OFFSET_TABLE_ is relative to .got.plt, false if .got.
91 bool GotBaseSymInGotPlt = true;
92
93 RelType CopyRel;
94 RelType GotRel;
95 RelType NoneRel;
96 RelType PltRel;
97 RelType RelativeRel;
98 RelType IRelativeRel;
99 RelType SymbolicRel;
100 RelType TlsDescRel;
101 RelType TlsGotRel;
102 RelType TlsModuleIndexRel;
103 RelType TlsOffsetRel;
104 llvm::Optional<RelType> AbsPointerRel; // TODO: remove the optional
105 llvm::Optional<RelType> SizeRel;
106 llvm::Optional<RelType> CheriCapRel;
107 llvm::Optional<RelType> CheriCapCallRel;
108 unsigned PltEntrySize;
109 unsigned PltHeaderSize;
110
111 // At least on x86_64 positions 1 and 2 are used by the first plt entry
112 // to support lazy loading.
113 unsigned GotPltHeaderEntriesNum = 3;
114
115 // On PPC ELF V2 abi, the first entry in the .got is the .TOC.
116 unsigned GotHeaderEntriesNum = 0;
117
118 bool NeedsThunks = false;
119
120 // A 4-byte field corresponding to one or more trap instructions, used to pad
121 // executable OutputSections.
122 std::array<uint8_t, 4> TrapInstr;
123
124 // If a target needs to rewrite calls to __morestack to instead call
125 // __morestack_non_split when a split-stack enabled caller calls a
126 // non-split-stack callee this will return true. Otherwise returns false.
127 bool NeedsMoreStackNonSplit = true;
128
129 virtual RelExpr adjustRelaxExpr(RelType Type, const uint8_t *Data,
130 RelExpr Expr) const;
131 virtual void relaxGot(uint8_t *Loc, RelType Type, uint64_t Val) const;
132 virtual void relaxTlsGdToIe(uint8_t *Loc, RelType Type, uint64_t Val) const;
133 virtual void relaxTlsGdToLe(uint8_t *Loc, RelType Type, uint64_t Val) const;
134 virtual void relaxTlsIeToLe(uint8_t *Loc, RelType Type, uint64_t Val) const;
135 virtual void relaxTlsLdToLe(uint8_t *Loc, RelType Type, uint64_t Val) const;
136
137protected:
138 // On FreeBSD x86_64 the first page cannot be mmaped.
139 // On Linux that is controled by vm.mmap_min_addr. At least on some x86_64
140 // installs that is 65536, so the first 15 pages cannot be used.
141 // Given that, the smallest value that can be used in here is 0x10000.
142 uint64_t DefaultImageBase = 0x10000;
143};

177// In the PowerPC64 Elf V2 abi a function can have 2 entry points. The first
178// is a global entry point (GEP) which typically is used to initialize the TOC
179// pointer in general purpose register 2. The second is a local entry
180// point (LEP) which bypasses the TOC pointer initialization code. The
181// offset between GEP and LEP is encoded in a function's st_other flags.
182// This function will return the offset (in bytes) from the global entry-point
183// to the local entry-point.
184unsigned getPPC64GlobalEntryToLocalEntryOffset(uint8_t StOther);

198static inline void reportRangeError(uint8_t *Loc, RelType Type, const Twine &V,
199 int64_t Min, uint64_t Max) {
200 ErrorPlace ErrPlace = getErrorPlace(Loc);
201 StringRef Hint;
202 if (ErrPlace.IS && ErrPlace.IS->Name.startswith(".debug"))
203 Hint = "; consider recompiling with -fdebug-types-section to reduce size "
204 "of debug sections";
205
206 errorOrWarn(ErrPlace.Loc + "relocation " + lld::toString(Type) +
207 " out of range: " + V.str() + " is not in [" + Twine(Min).str() +
208 ", " + Twine(Max).str() + "]" + Hint);
209}

211// Make sure that V can be represented as an N bit signed integer.
212inline void checkInt(uint8_t *Loc, int64_t V, int N, RelType Type) {
213 if (V != llvm::SignExtend64(V, N))
214 reportRangeError(Loc, Type, Twine(V), llvm::minIntN(N), llvm::maxIntN(N));
215}
216
217// Make sure that V can be represented as an N bit unsigned integer.
218inline void checkUInt(uint8_t *Loc, uint64_t V, int N, RelType Type) {
219 if ((V >> N) != 0)
220 reportRangeError(Loc, Type, Twine(V), 0, llvm::maxUIntN(N));
221}
222
223// Make sure that V can be represented as an N bit signed or unsigned integer.
224inline void checkIntUInt(uint8_t *Loc, uint64_t V, int N, RelType Type) {
225 // For the error message we should cast V to a signed integer so that error
226 // messages show a small negative value rather than an extremely large one
227 if (V != (uint64_t)llvm::SignExtend64(V, N) && (V >> N) != 0)
228 reportRangeError(Loc, Type, Twine((int64_t)V), llvm::minIntN(N),
229 llvm::maxIntN(N));
230}
231
232inline void checkAlignment(uint8_t *Loc, uint64_t V, int N, RelType Type) {
233 if ((V & (N - 1)) != 0)
234 error(getErrorLocation(Loc) + "improper alignment for relocation " +
235 lld::toString(Type) + ": 0x" + llvm::utohexstr(V) +
236 " is not aligned to " + Twine(N) + " bytes");
237}
