45static uint64_t getSymVA(const Symbol &Sym, int64_t &Addend) {
46 switch (Sym.kind()) {
47 case Symbol::DefinedKind: {
48 auto &D = cast<Defined>(Sym);
49 SectionBase *IS = D.Section;
50
51 // This is an absolute symbol.
52 if (!IS)
53 return D.Value;
54
55 assert(IS != &InputSection::Discarded);
56 IS = IS->Repl;
57
58 uint64_t Offset = D.Value;
59
60 // An object in an SHF_MERGE section might be referenced via a
61 // section symbol (as a hack for reducing the number of local
62 // symbols).
63 // Depending on the addend, the reference via a section symbol
64 // refers to a different object in the merge section.
65 // Since the objects in the merge section are not necessarily
66 // contiguous in the output, the addend can thus affect the final
67 // VA in a non-linear way.
68 // To make this work, we incorporate the addend into the section
69 // offset (and zero out the addend for later processing) so that
70 // we find the right object in the section.
71 if (D.isSection()) {
72 Offset += Addend;
73 Addend = 0;
74 }
75
76 // In the typical case, this is actually very simple and boils
77 // down to adding together 3 numbers:
78 // 1. The address of the output section.
79 // 2. The offset of the input section within the output section.
80 // 3. The offset within the input section (this addition happens
81 // inside InputSection::getOffset).
82 //
83 // If you understand the data structures involved with this next
84 // line (and how they get built), then you have a pretty good
85 // understanding of the linker.
86 uint64_t VA = IS->getVA(Offset);
87
88 // MIPS relocatable files can mix regular and microMIPS code.
89 // Linker needs to distinguish such code. To do so microMIPS
90 // symbols has the `STO_MIPS_MICROMIPS` flag in the `st_other`
91 // field. Unfortunately, the `MIPS::relocateOne()` method has
92 // a symbol value only. To pass type of the symbol (regular/microMIPS)
93 // to that routine as well as other places where we write
94 // a symbol value as-is (.dynamic section, `Elf_Ehdr::e_entry`
95 // field etc) do the same trick as compiler uses to mark microMIPS
96 // for CPU - set the less-significant bit.
97 if (Config->EMachine == EM_MIPS && isMicroMips() &&
98 ((Sym.StOther & STO_MIPS_MICROMIPS) || Sym.NeedsPltAddr))
99 VA |= 1;
100
101 if (D.isTls() && !Config->Relocatable) {
102 // Use the address of the TLS segment's first section rather than the
103 // segment's address, because segment addresses aren't initialized until
104 // after sections are finalized. (e.g. Measuring the size of .rela.dyn
105 // for Android relocation packing requires knowing TLS symbol addresses
106 // during section finalization.)
107 if (!Out::TlsPhdr || !Out::TlsPhdr->FirstSec)
108 fatal(toString(D.File) +
109 " has an STT_TLS symbol but doesn't have an SHF_TLS section");
110 return VA - Out::TlsPhdr->FirstSec->Addr;
111 }
112 return VA;
113 }
114 case Symbol::SharedKind:
115 case Symbol::UndefinedKind:
116 return 0;
117 case Symbol::LazyArchiveKind:
118 case Symbol::LazyObjectKind:
119 assert(Sym.IsUsedInRegularObj && "lazy symbol reached writer");
120 return 0;
121 case Symbol::CommonKind:
122 llvm_unreachable("common symbol reached writer");
123 case Symbol::PlaceholderKind:
124 llvm_unreachable("placeholder symbol reached writer");
125 }
126 llvm_unreachable("invalid symbol kind");
127}

159uint64_t Symbol::getPltVA() const {
160 PltSection *Plt = IsInIplt ? In.Iplt : In.Plt;
161 uint64_t OutVA =
162 Plt->getVA() + Plt->HeaderSize + PltIndex * Target->PltEntrySize;
163 // While linking microMIPS code PLT code are always microMIPS
164 // code. Set the less-significant bit to track that fact.
165 // See detailed comment in the `getSymVA` function.
166 if (Config->EMachine == EM_MIPS && isMicroMips())
167 OutVA |= 1;
168 return OutVA;
169}

177uint64_t Symbol::getSize() const {
178 if (const auto *DR = dyn_cast<Defined>(this)) {
179 if (Config->isCheriABI() && DR->IsSectionStartSymbol) {
180 assert(DR->Value == 0 && "Bad section start symbol?");
181 if (!DR->Section)
182 return 0; // Section is not included in the output
183 return DR->Section->getOutputSection()->Size;
184 }
185 return DR->Size;
186 }
187 // FIXME: assuming it is always shared broke this
188 if (isa<Undefined>(this))
189 return 0;
190 if (isUndefWeak())
191 return 0;
192 // errs() << "Should be a Shared symbol " << toString(*this) << ":" << this->kind() << "\n";
193 return cast<SharedSymbol>(this)->Size;
194}

198 if (auto *Sec = S->Section)
199 return Sec->Repl->getOutputSection();
200 return nullptr;
201 }
202 return nullptr;

205// If a symbol name contains '@', the characters after that is
206// a symbol version name. This function parses that.
207void Symbol::parseSymbolVersion() {
208 StringRef S = getName();
209 size_t Pos = S.find('@');
210 if (Pos == 0 || Pos == StringRef::npos)
211 return;
212 StringRef Verstr = S.substr(Pos + 1);
213 if (Verstr.empty())
214 return;
215
216 // Truncate the symbol name so that it doesn't include the version string.
217 NameSize = Pos;
218
219 // If this is not in this DSO, it is not a definition.
220 if (!isDefined())
221 return;
222
223 // '@@' in a symbol name means the default version.
224 // It is usually the most recent one.
225 bool IsDefault = (Verstr[0] == '@');
226 if (IsDefault)
227 Verstr = Verstr.substr(1);
228
229 for (VersionDefinition &Ver : Config->VersionDefinitions) {
230 if (Ver.Name != Verstr)
231 continue;
232
233 if (IsDefault)
234 VersionId = Ver.Id;
235 else
236 VersionId = Ver.Id | VERSYM_HIDDEN;
237 return;
238 }
239
240 // It is an error if the specified version is not defined.
241 // Usually version script is not provided when linking executable,
242 // but we may still want to override a versioned symbol from DSO,
243 // so we do not report error in this case. We also do not error
244 // if the symbol has a local version as it won't be in the dynamic
245 // symbol table.
246 if (Config->Shared && VersionId != VER_NDX_LOCAL)
247 error(toString(File) + ": symbol " + S + " has undefined version " +
248 Verstr);
249}

252 if (auto *Sym = dyn_cast<LazyArchive>(this)) {
253 cast<ArchiveFile>(Sym->File)->fetch(Sym->Sym);

255 }
256

258 dyn_cast<LazyObjFile>(Sym->File)->fetch();
259 return;
260 }
261
262 llvm_unreachable("Symbol::fetch() is called on a non-lazy symbol");
263}

274uint8_t Symbol::computeBinding() const {
275 if (Config->Relocatable)
276 return Binding;
277 if (Visibility != STV_DEFAULT && Visibility != STV_PROTECTED)
278 return STB_LOCAL;
279 if (VersionId == VER_NDX_LOCAL && isDefined() && !IsPreemptible)
280 return STB_LOCAL;
281 if (!Config->GnuUnique && Binding == STB_GNU_UNIQUE)
282 return STB_GLOBAL;
283 return Binding;
284}

286bool Symbol::includeInDynsym() const {
287 if (!Config->HasDynSymTab)
288 return false;
289 if (UsedByDynReloc)
290 return true;
291 if (computeBinding() == STB_LOCAL)
292 return false;
293 // If a PIE binary was not linked against any shared libraries, then we can
294 // safely drop weak undef symbols from .dynsym.
295 if (isUndefWeak() && Config->Pie && SharedFiles.empty())
296 return false;
297 if (!isDefined())
298 return true;
299 return ExportDynamic;
300}

304 std::string S;

306 S = ": reference to ";
307 else if (Sym->isLazy())
308 S = ": lazy definition of ";
309 else if (Sym->isShared())

312 S = ": common definition of ";

314 S = ": definition of ";

316 message(toString(Sym->File) + S + Sym->getName());
317}

319void elf::maybeWarnUnorderableSymbol(const Symbol *Sym) {
320 if (!Config->WarnSymbolOrdering)
321 return;
322
323 // If UnresolvedPolicy::Ignore is used, no "undefined symbol" error/warning
324 // is emitted. It makes sense to not warn on undefined symbols.
325 //
326 // Note, ld.bfd --symbol-ordering-file= does not warn on undefined symbols,
327 // but we don't have to be compatible here.
328 if (Sym->isUndefined() &&
329 Config->UnresolvedSymbols == UnresolvedPolicy::Ignore)
330 return;
331
332 const InputFile *File = Sym->File;
333 auto *D = dyn_cast<Defined>(Sym);
334
335 auto Warn = [&](StringRef S) { warn(toString(File) + S + Sym->getName()); };
336
337 if (Sym->isUndefined())
338 Warn(": unable to order undefined symbol: ");
339 else if (Sym->isShared())
340 Warn(": unable to order shared symbol: ");
341 else if (D && !D->Section)
342 Warn(": unable to order absolute symbol: ");
343 else if (D && isa<OutputSection>(D->Section))
344 Warn(": unable to order synthetic symbol: ");
345 else if (D && !D->Section->Repl->isLive())
346 Warn(": unable to order discarded symbol: ");
347}

373std::string lld::verboseToString(Symbol *B, uint64_t SymOffset) {
374 std::string Msg;
375
376 if (B->isLocal())
377 Msg += "local ";
378 if (B->isWeak())
379 Msg += "weak ";
380 if (B->isShared())
381 Msg += "shared ";
382 // else if (B->isDefined())
383 // Msg += "defined ";
384 if (B->Type == STT_COMMON)
385 Msg += "common ";
386 else if (B->Type == STT_TLS)
387 Msg += "TLS ";
388 if (B->isSection())
389 Msg += "section ";
390 else if (B->isTls())
391 Msg += "tls ";
392 else if (B->isFunc())
393 Msg += "function ";
394 else if (B->isGnuIFunc())
395 Msg += "gnu ifunc ";
396 else if (B->isObject())
397 Msg += "object ";
398 else if (B->isFile())
399 Msg += "file ";
400 else if (B->isUndefined())
401 Msg += "<undefined> ";
402 else
403 Msg += "<unknown kind> ";
404
405 if (B->isInGot())
406 Msg += "(in GOT) ";
407 if (B->isInPlt())
408 Msg += "(in PLT) ";
409
410 Defined* DR = dyn_cast<Defined>(B);
411 InputSectionBase* IS = nullptr;
412 if (DR && DR->Section) {
413 SymOffset = DR->isSection() ? SymOffset : DR->Section->getOffset(DR->Value);
414 IS = dyn_cast<InputSectionBase>(DR->Section);
415 }
416 std::string Name = toString(*B);
417 if (Name.empty()) {
418 if (DR && DR->Section) {
419 if (IS) {
420 Name = getLocationNonTemplate(IS, SymOffset);
421 } else {
422 Name = (DR->Section->Name + "+0x" + utohexstr(SymOffset)).str();
423 }
424 } else if (OutputSection* OS = B->getOutputSection()) {
425 Name = (OS->Name + "+(unknown offset)").str();
426 }
427 }
428 if (Name.empty()) {
429 Name = "<unknown symbol>";
430 }
431 Msg += Name;
432 if (!B->isUndefined()) {
433 std::string Src = IS ? IS->getSrcMsg(*B, SymOffset) : toString(B->File);
434 if (IS)
435 Src += " (" + IS->getObjMsg(SymOffset) + ")";
436 Msg += "\n>>> defined in " + Src;
437 }
438 return Msg;
439}

449// Merge symbol properties.
450//
451// When we have many symbols of the same name, we choose one of them,
452// and that's the result of symbol resolution. However, symbols that
453// were not chosen still affect some symbol properties.
454void Symbol::mergeProperties(const Symbol &Other) {
455 if (Other.ExportDynamic)
456 ExportDynamic = true;
457 if (Other.IsUsedInRegularObj)
458 IsUsedInRegularObj = true;
459
460 // DSO symbols do not affect visibility in the output.
461 if (!Other.isShared())
462 Visibility = getMinVisibility(Visibility, Other.Visibility);
463}
464
465void Symbol::resolve(const Symbol &Other) {
466 mergeProperties(Other);
467
468 if (isPlaceholder()) {
469 replace(Other);
470 return;
471 }
472
473 switch (Other.kind()) {
474 case Symbol::UndefinedKind:
475 resolveUndefined(cast<Undefined>(Other));
476 break;
477 case Symbol::CommonKind:
478 resolveCommon(cast<CommonSymbol>(Other));
479 break;
480 case Symbol::DefinedKind:
481 resolveDefined(cast<Defined>(Other));
482 break;
483 case Symbol::LazyArchiveKind:
484 resolveLazy(cast<LazyArchive>(Other));
485 break;
486 case Symbol::LazyObjectKind:
487 resolveLazy(cast<LazyObject>(Other));
488 break;
489 case Symbol::SharedKind:
490 resolveShared(cast<SharedSymbol>(Other));
491 break;
492 case Symbol::PlaceholderKind:
493 llvm_unreachable("bad symbol kind");
494 }
495}

497void Symbol::resolveUndefined(const Undefined &Other) {
498 // An undefined symbol with non default visibility must be satisfied
499 // in the same DSO.
500 //
501 // If this is a non-weak defined symbol in a discarded section, override the
502 // existing undefined symbol for better error message later.
503 if ((isShared() && Other.Visibility != STV_DEFAULT) ||
504 (isUndefined() && Other.Binding != STB_WEAK && Other.DiscardedSecIdx)) {
505 replace(Other);
506 return;
507 }
508
509 if (Traced)
510 printTraceSymbol(&Other);
511
512 if (isShared() || isLazy() || (isUndefined() && Other.Binding != STB_WEAK))
513 Binding = Other.Binding;
514
515 if (isLazy()) {
516 // An undefined weak will not fetch archive members. See comment on Lazy in
517 // Symbols.h for the details.
518 if (Other.Binding == STB_WEAK) {
519 Type = Other.Type;
520 return;
521 }
522
523 // Do extra check for --warn-backrefs.
524 //
525 // --warn-backrefs is an option to prevent an undefined reference from
526 // fetching an archive member written earlier in the command line. It can be
527 // used to keep compatibility with GNU linkers to some degree.
528 // I'll explain the feature and why you may find it useful in this comment.
529 //
530 // lld's symbol resolution semantics is more relaxed than traditional Unix
531 // linkers. For example,
532 //
533 // ld.lld foo.a bar.o
534 //
535 // succeeds even if bar.o contains an undefined symbol that has to be
536 // resolved by some object file in foo.a. Traditional Unix linkers don't
537 // allow this kind of backward reference, as they visit each file only once
538 // from left to right in the command line while resolving all undefined
539 // symbols at the moment of visiting.
540 //
541 // In the above case, since there's no undefined symbol when a linker visits
542 // foo.a, no files are pulled out from foo.a, and because the linker forgets
543 // about foo.a after visiting, it can't resolve undefined symbols in bar.o
544 // that could have been resolved otherwise.
545 //
546 // That lld accepts more relaxed form means that (besides it'd make more
547 // sense) you can accidentally write a command line or a build file that
548 // works only with lld, even if you have a plan to distribute it to wider
549 // users who may be using GNU linkers. With --warn-backrefs, you can detect
550 // a library order that doesn't work with other Unix linkers.
551 //
552 // The option is also useful to detect cyclic dependencies between static
553 // archives. Again, lld accepts
554 //
555 // ld.lld foo.a bar.a
556 //
557 // even if foo.a and bar.a depend on each other. With --warn-backrefs, it is
558 // handled as an error.
559 //
560 // Here is how the option works. We assign a group ID to each file. A file
561 // with a smaller group ID can pull out object files from an archive file
562 // with an equal or greater group ID. Otherwise, it is a reverse dependency
563 // and an error.
564 //
565 // A file outside --{start,end}-group gets a fresh ID when instantiated. All
566 // files within the same --{start,end}-group get the same group ID. E.g.
567 //
568 // ld.lld A B --start-group C D --end-group E
569 //
570 // A forms group 0. B form group 1. C and D (including their member object
571 // files) form group 2. E forms group 3. I think that you can see how this
572 // group assignment rule simulates the traditional linker's semantics.
573 bool Backref = Config->WarnBackrefs && Other.File &&
574 File->GroupId < Other.File->GroupId;
575 fetch();
576
577 // We don't report backward references to weak symbols as they can be
578 // overridden later.
579 if (Backref && !isWeak())
580 warn("backward reference detected: " + Other.getName() + " in " +
581 toString(Other.File) + " refers to " + toString(File));
582 }
583}

585// Using .symver foo,foo@@VER unfortunately creates two symbols: foo and
586// foo@@VER. We want to effectively ignore foo, so give precedence to
587// foo@@VER.
588// FIXME: If users can transition to using
589// .symver foo,foo@@@VER
590// we can delete this hack.
591static int compareVersion(StringRef A, StringRef B) {
592 bool X = A.contains("@@");
593 bool Y = B.contains("@@");
594 if (!X && Y)
595 return 1;
596 if (X && !Y)
597 return -1;
598 return 0;
599}

601// Compare two symbols. Return 1 if the new symbol should win, -1 if
602// the new symbol should lose, or 0 if there is a conflict.
603int Symbol::compare(const Symbol *Other) const {
604 assert(Other->isDefined() || Other->isCommon());
605
606 if (!isDefined() && !isCommon())
607 return 1;
608
609 if (int Cmp = compareVersion(getName(), Other->getName()))
610 return Cmp;
611
612 if (Other->isWeak())
613 return -1;
614
615 if (isWeak())
616 return 1;
617
618 if (isCommon() && Other->isCommon()) {
619 if (Config->WarnCommon)
620 warn("multiple common of " + getName());
621 return 0;
622 }
623
624 if (isCommon()) {
625 if (Config->WarnCommon)
626 warn("common " + getName() + " is overridden");
627 return 1;
628 }
629
630 if (Other->isCommon()) {
631 if (Config->WarnCommon)
632 warn("common " + getName() + " is overridden");
633 return -1;
634 }
635
636 auto *OldSym = cast<Defined>(this);
637 auto *NewSym = cast<Defined>(Other);
638
639 if (Other->File && isa<BitcodeFile>(Other->File))
640 return 0;
641
642 if (!OldSym->Section && !NewSym->Section && OldSym->Value == NewSym->Value &&
643 NewSym->Binding == STB_GLOBAL)
644 return -1;
645
646 return 0;
647}

649static void reportDuplicate(Symbol *Sym, InputFile *NewFile,
650 InputSectionBase *ErrSec, uint64_t ErrOffset) {
651 if (Config->AllowMultipleDefinition)
652 return;
653
654 Defined *D = cast<Defined>(Sym);
655 if (!D->Section || !ErrSec) {
656 error("duplicate symbol: " + toString(*Sym) + "\n>>> defined in " +
657 toString(Sym->File) + "\n>>> defined in " + toString(NewFile));
658 return;
659 }
660
661 // Construct and print an error message in the form of:
662 //
663 // ld.lld: error: duplicate symbol: foo
664 // >>> defined at bar.c:30
665 // >>> bar.o (/home/alice/src/bar.o)
666 // >>> defined at baz.c:563
667 // >>> baz.o in archive libbaz.a
668 auto *Sec1 = cast<InputSectionBase>(D->Section);
669 std::string Src1 = Sec1->getSrcMsg(*Sym, D->Value);
670 std::string Obj1 = Sec1->getObjMsg(D->Value);
671 std::string Src2 = ErrSec->getSrcMsg(*Sym, ErrOffset);
672 std::string Obj2 = ErrSec->getObjMsg(ErrOffset);
673
674 std::string Msg = "duplicate symbol: " + toString(*Sym) + "\n>>> defined at ";
675 if (!Src1.empty())
676 Msg += Src1 + "\n>>> ";
677 Msg += Obj1 + "\n>>> defined at ";
678 if (!Src2.empty())
679 Msg += Src2 + "\n>>> ";
680 Msg += Obj2;
681 error(Msg);
682}

684void Symbol::resolveCommon(const CommonSymbol &Other) {
685 int Cmp = compare(&Other);
686 if (Cmp < 0)
687 return;
688
689 if (Cmp > 0) {
690 replace(Other);
691 return;
692 }
693
694 CommonSymbol *OldSym = cast<CommonSymbol>(this);
695
696 OldSym->Alignment = std::max(OldSym->Alignment, Other.Alignment);
697 if (OldSym->Size < Other.Size) {
698 OldSym->File = Other.File;
699 OldSym->Size = Other.Size;
700 }
701}

703void Symbol::resolveDefined(const Defined &Other) {
704 int Cmp = compare(&Other);
705 if (Cmp > 0)
706 replace(Other);
707 else if (Cmp == 0)
708 reportDuplicate(this, Other.File,
709 dyn_cast_or_null<InputSectionBase>(Other.Section),
710 Other.Value);
711}

713template <class LazyT> void Symbol::resolveLazy(const LazyT &Other) {
714 if (!isUndefined())
715 return;
716
717 // An undefined weak will not fetch archive members. See comment on Lazy in
718 // Symbols.h for the details.
719 if (isWeak()) {
720 uint8_t Ty = Type;
721 replace(Other);
722 Type = Ty;
723 Binding = STB_WEAK;
724 return;
725 }
726
727 Other.fetch();
728}
