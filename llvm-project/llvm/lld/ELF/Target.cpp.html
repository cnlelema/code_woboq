9// Machine-specific things, such as applying relocations, creation of
10// GOT or PLT entries, etc., are handled in this file.
11//
12// Refer the ELF spec for the single letter variables, S, A or P, used
13// in this file.
14//
15// Some functions defined in this file has "relaxTls" as part of their names.
16// They do peephole optimization for TLS variables by rewriting instructions.
17// They are not part of the ABI but optional optimization, so you can skip
18// them if you are not interested in how TLS variables are optimized.
19// See the following paper for the details.
20//
21// Ulrich Drepper, ELF Handling For Thread-Local Storage

42std::string lld::toString(RelType Type) {
43 auto Machine = elf::Config->EMachine;
44 if (Machine == EM_MIPS && Type > 0xff) {
45 uint32_t Type1 = Type & 0xff;
46 llvm::Twine Result = getELFRelocationTypeName(Machine, Type1);
47 uint32_t Type2 = (Type >> 8) & 0xff;
48 uint32_t Type3 = (Type >> 16) & 0xff;
49 if (Type2 || Type3) {
50 return (Result + "/" + getELFRelocationTypeName(Machine, Type2) + "/" +
51 getELFRelocationTypeName(Machine, Type3)).str();
52 }
53 return Result.str();
54 }
55 StringRef S = getELFRelocationTypeName(Machine, Type);
56 if (S == "Unknown")
57 return ("Unknown (" + Twine(Type) + ")").str();
58 return S;
59}

61TargetInfo *elf::getTarget() {
62 switch (Config->EMachine) {
63 case EM_386:
64 case EM_IAMCU:
65 return getX86TargetInfo();
66 case EM_AARCH64:
67 return getAArch64TargetInfo();
68 case EM_AMDGPU:
69 return getAMDGPUTargetInfo();
70 case EM_ARM:
71 return getARMTargetInfo();
72 case EM_AVR:
73 return getAVRTargetInfo();
74 case EM_HEXAGON:
75 return getHexagonTargetInfo();
76 case EM_MIPS:
77 if (Config->isCheriABI())
78 return getMipsTargetInfo<ELF64BE>(); // TODO: custom target info?
79 switch (Config->EKind) {
80 case ELF32LEKind:
81 return getMipsTargetInfo<ELF32LE>();
82 case ELF32BEKind:
83 return getMipsTargetInfo<ELF32BE>();
84 case ELF64LEKind:
85 return getMipsTargetInfo<ELF64LE>();
86 case ELF64BEKind:
87 return getMipsTargetInfo<ELF64BE>();
88 default:
89 llvm_unreachable("unsupported MIPS target");
90 }
91 case EM_MSP430:
92 return getMSP430TargetInfo();
93 case EM_PPC:
94 return getPPCTargetInfo();
95 case EM_PPC64:
96 return getPPC64TargetInfo();
97 case EM_RISCV:
98 return getRISCVTargetInfo();
99 case EM_SPARCV9:
100 return getSPARCV9TargetInfo();
101 case EM_X86_64:
102 return getX86_64TargetInfo();
103 }
104 llvm_unreachable("unknown target machine");
105}

107template <class ELFT> static ErrorPlace getErrPlace(const uint8_t *Loc) {
108 for (InputSectionBase *D : InputSections) {
109 auto *IS = cast<InputSection>(D);
110 if (!IS->getParent())
111 continue;
112
113 uint8_t *ISLoc = Out::BufferStart + IS->getParent()->Offset + IS->OutSecOff;
114 if (ISLoc <= Loc && Loc < ISLoc + IS->getSize())
115 return {IS, IS->template getLocation<ELFT>(Loc - ISLoc) + ": "};
116 }
117 return {};
118}
119
120ErrorPlace elf::getErrorPlace(const uint8_t *Loc) {
121 switch (Config->EKind) {
122 case ELF32LEKind:
123 return getErrPlace<ELF32LE>(Loc);
124 case ELF32BEKind:
125 return getErrPlace<ELF32BE>(Loc);
126 case ELF64LEKind:
127 return getErrPlace<ELF64LE>(Loc);
128 case ELF64BEKind:
129 return getErrPlace<ELF64BE>(Loc);
130 default:
131 llvm_unreachable("unknown ELF type");
132 }
133}
