107bool Minimizer::reportError(const char *CurPtr, unsigned Err) {
108 if (!Diags)
109 return true;
110 assert(CurPtr >= Input.data() && "invalid buffer ptr");
111 Diags->Report(InputSourceLoc.getLocWithOffset(CurPtr - Input.data()), Err);

115static void skipOverSpaces(const char *&First, const char *const End) {
116 while (First != End && isHorizontalWhitespace(*First))
117 ++First;
118}

120LLVM_NODISCARD static bool isRawStringLiteral(const char *First,
121 const char *Current) {
122 assert(First <= Current);
123
124 // Check if we can even back up.
125 if (*Current != '"' || First == Current)
126 return false;
127
128 // Check for an "R".
129 --Current;
130 if (*Current != 'R')
131 return false;
132 if (First == Current || !isIdentifierBody(*--Current))
133 return true;
134
135 // Check for a prefix of "u", "U", or "L".
136 if (*Current == 'u' || *Current == 'U' || *Current == 'L')
137 return First == Current || !isIdentifierBody(*--Current);
138
139 // Check for a prefix of "u8".
140 if (*Current != '8' || First == Current || *Current-- != 'u')
141 return false;
142 return First == Current || !isIdentifierBody(*--Current);
143}
144
145static void skipRawString(const char *&First, const char *const End) {
146 assert(First[0] == '"');
147 assert(First[-1] == 'R');
148
149 const char *Last = ++First;
150 while (Last != End && *Last != '(')
151 ++Last;
152 if (Last == End) {
153 First = Last; // Hit the end... just give up.
154 return;
155 }
156
157 StringRef Terminator(First, Last - First);
158 for (;;) {
159 // Move First to just past the next ")".
160 First = Last;
161 while (First != End && *First != ')')
162 ++First;
163 if (First == End)
164 return;
165 ++First;
166
167 // Look ahead for the terminator sequence.
168 Last = First;
169 while (Last != End && size_t(Last - First) < Terminator.size() &&
170 Terminator[Last - First] == *Last)
171 ++Last;
172
173 // Check if we hit it (or the end of the file).
174 if (Last == End) {
175 First = Last;
176 return;
177 }
178 if (size_t(Last - First) < Terminator.size())
179 continue;
180 if (*Last != '"')
181 continue;
182 First = Last + 1;
183 return;
184 }
185}

187static void skipString(const char *&First, const char *const End) {
188 assert(*First == '\'' || *First == '"');
189 const char Terminator = *First;
190 for (++First; First != End && *First != Terminator; ++First)
191 if (*First == '\\')
192 if (++First == End)
193 return;
194 if (First != End)
195 ++First; // Finish off the string.
196}
197
198static void skipNewline(const char *&First, const char *End) {
199 assert(isVerticalWhitespace(*First));
200 ++First;
201 if (First == End)
202 return;
203
204 // Check for "\n\r" and "\r\n".
205 if (LLVM_UNLIKELY(isVerticalWhitespace(*First) && First[-1] != First[0]))
206 ++First;
207}
208
209static void skipToNewlineRaw(const char *&First, const char *const End) {
210 for (;;) {
211 if (First == End)
212 return;
213
214 if (isVerticalWhitespace(*First))
215 return;
216
217 while (!isVerticalWhitespace(*First))
218 if (++First == End)
219 return;
220
221 if (First[-1] != '\\')
222 return;
223
224 ++First; // Keep going...
225 }
226}
227
228static const char *reverseOverSpaces(const char *First, const char *Last) {
229 assert(First <= Last);
230 while (First != Last && isHorizontalWhitespace(Last[-1]))
231 --Last;
232 return Last;
233}
234
235static void skipLineComment(const char *&First, const char *const End) {
236 assert(First[0] == '/' && First[1] == '/');
237 First += 2;
238 skipToNewlineRaw(First, End);
239}
240
241static void skipBlockComment(const char *&First, const char *const End) {
242 assert(First[0] == '/' && First[1] == '*');
243 if (End - First < 4) {
244 First = End;
245 return;
246 }
247 for (First += 3; First != End; ++First)
248 if (First[-1] == '*' && First[0] == '/') {
249 ++First;
250 return;
251 }
252}

254/// \returns True if the current single quotation mark character is a C++ 14
255/// digit separator.
256static bool isQuoteCppDigitSeparator(const char *const Start,
257 const char *const Cur,
258 const char *const End) {
259 assert(*Cur == '\'' && "expected quotation character");
260 // skipLine called in places where we don't expect a valid number
261 // body before `start` on the same line, so always return false at the start.
262 if (Start == Cur)
263 return false;
264 // The previous character must be a valid PP number character.
265 if (!isPreprocessingNumberBody(*(Cur - 1)))
266 return false;
267 // The next character should be a valid identifier body character.
268 return (Cur + 1) < End && isIdentifierBody(*(Cur + 1));
269}
270
271static void skipLine(const char *&First, const char *const End) {
272 do {
273 assert(First <= End);
274 if (First == End)
275 return;
276
277 if (isVerticalWhitespace(*First)) {
278 skipNewline(First, End);
279 return;
280 }
281 const char *Start = First;
282 while (First != End && !isVerticalWhitespace(*First)) {
283 // Iterate over strings correctly to avoid comments and newlines.
284 if (*First == '"' ||
285 (*First == '\'' && !isQuoteCppDigitSeparator(Start, First, End))) {
286 if (isRawStringLiteral(Start, First))
287 skipRawString(First, End);
288 else
289 skipString(First, End);
290 continue;
291 }
292
293 // Iterate over comments correctly.
294 if (*First != '/' || End - First < 2) {
295 ++First;
296 continue;
297 }
298
299 if (First[1] == '/') {
300 // "//...".
301 skipLineComment(First, End);
302 continue;
303 }
304
305 if (First[1] != '*') {
306 ++First;
307 continue;
308 }
309
310 // "/*...*/".
311 skipBlockComment(First, End);
312 }
313 if (First == End)
314 return;
315
316 // Skip over the newline.
317 assert(isVerticalWhitespace(*First));
318 skipNewline(First, End);
319 } while (First[-2] == '\\'); // Continue past line-continuations.
320}

322static void skipDirective(StringRef Name, const char *&First,
323 const char *const End) {
324 if (llvm::StringSwitch<bool>(Name)
325 .Case("warning", true)
326 .Case("error", true)
327 .Default(false))
328 // Do not process quotes or comments.
329 skipToNewlineRaw(First, End);
330 else
331 skipLine(First, End);
332}

334void Minimizer::printToNewline(const char *&First, const char *const End) {
335 while (First != End && !isVerticalWhitespace(*First)) {
336 const char *Last = First;
337 do {
338 // Iterate over strings correctly to avoid comments and newlines.
339 if (*Last == '"' || *Last == '\'') {
340 if (LLVM_UNLIKELY(isRawStringLiteral(First, Last)))
341 skipRawString(Last, End);
342 else
343 skipString(Last, End);
344 continue;
345 }
346 if (*Last != '/' || End - Last < 2) {
347 ++Last;
348 continue; // Gather the rest up to print verbatim.
349 }
350
351 if (Last[1] != '/' && Last[1] != '*') {
352 ++Last;
353 continue;
354 }
355
356 // Deal with "//..." and "/*...*/".
357 append(First, reverseOverSpaces(First, Last));
358 First = Last;
359
360 if (Last[1] == '/') {
361 skipLineComment(First, End);
362 return;
363 }
364
365 put(' ');
366 skipBlockComment(First, End);
367 skipOverSpaces(First, End);
368 Last = First;
369 } while (Last != End && !isVerticalWhitespace(*Last));
370
371 // Print out the string.
372 if (Last == End || Last == First || Last[-1] != '\\') {
373 append(First, reverseOverSpaces(First, Last));
374 return;
375 }
376
377 // Print up to the backslash, backing up over spaces.
378 append(First, reverseOverSpaces(First, Last - 1));
379
380 First = Last;
381 skipNewline(First, End);
382 skipOverSpaces(First, End);
383 }
384}

386static void skipWhitespace(const char *&First, const char *const End) {
387 for (;;) {
388 assert(First <= End);
389 skipOverSpaces(First, End);
390
391 if (End - First < 2)
392 return;
393
394 if (First[0] == '\\' && isVerticalWhitespace(First[1])) {
395 skipNewline(++First, End);
396 continue;
397 }
398
399 // Check for a non-comment character.
400 if (First[0] != '/')
401 return;
402
403 // "// ...".
404 if (First[1] == '/') {
405 skipLineComment(First, End);
406 return;
407 }
408
409 // Cannot be a comment.
410 if (First[1] != '*')
411 return;
412
413 // "/*...*/".
414 skipBlockComment(First, End);
415 }
416}

418void Minimizer::printAdjacentModuleNameParts(const char *&First,
419 const char *const End) {
420 // Skip over parts of the body.
421 const char *Last = First;
422 do
423 ++Last;
424 while (Last != End && (isIdentifierBody(*Last) || *Last == '.'));
425 append(First, Last);
426 First = Last;

429bool Minimizer::printAtImportBody(const char *&First, const char *const End) {
430 for (;;) {
431 skipWhitespace(First, End);
432 if (First == End)
433 return true;
434
435 if (isVerticalWhitespace(*First)) {
436 skipNewline(First, End);
437 continue;
438 }
439
440 // Found a semicolon.
441 if (*First == ';') {
442 put(*First++).put('\n');
443 return false;
444 }
445
446 // Don't handle macro expansions inside @import for now.
447 if (!isIdentifierBody(*First) && *First != '.')
448 return true;
449
450 printAdjacentModuleNameParts(First, End);
451 }
452}

454void Minimizer::printDirectiveBody(const char *&First, const char *const End) {
455 skipWhitespace(First, End); // Skip initial whitespace.
456 printToNewline(First, End);
457 while (Out.back() == ' ')
458 Out.pop_back();
459 put('\n');
460}

462LLVM_NODISCARD static const char *lexRawIdentifier(const char *First,
463 const char *const End) {
464 assert(isIdentifierBody(*First) && "invalid identifer");
465 const char *Last = First + 1;
466 while (Last != End && isIdentifierBody(*Last))
467 ++Last;
468 return Last;
469}

471LLVM_NODISCARD static const char *
472getIdentifierContinuation(const char *First, const char *const End) {
473 if (End - First < 3 || First[0] != '\\' || !isVerticalWhitespace(First[1]))
474 return nullptr;
475
476 ++First;
477 skipNewline(First, End);
478 if (First == End)
479 return nullptr;
480 return isIdentifierBody(First[0]) ? First : nullptr;
481}

483Minimizer::IdInfo Minimizer::lexIdentifier(const char *First,
484 const char *const End) {
485 const char *Last = lexRawIdentifier(First, End);
486 const char *Next = getIdentifierContinuation(Last, End);
487 if (LLVM_LIKELY(!Next))
488 return IdInfo{Last, StringRef(First, Last - First)};
489
490 // Slow path, where identifiers are split over lines.
491 SmallVector<char, 64> Id(First, Last);
492 while (Next) {
493 Last = lexRawIdentifier(Next, End);
494 Id.append(Next, Last);
495 Next = getIdentifierContinuation(Last, End);
496 }
497 return IdInfo{
498 Last,
499 SplitIds.try_emplace(StringRef(Id.begin(), Id.size()), 0).first->first()};
500}

502void Minimizer::printAdjacentMacroArgs(const char *&First,
503 const char *const End) {
504 // Skip over parts of the body.
505 const char *Last = First;
506 do
507 ++Last;
508 while (Last != End &&
509 (isIdentifierBody(*Last) || *Last == '.' || *Last == ','));
510 append(First, Last);
511 First = Last;
512}

514bool Minimizer::printMacroArgs(const char *&First, const char *const End) {
515 assert(*First == '(');
516 put(*First++);
517 for (;;) {
518 skipWhitespace(First, End);
519 if (First == End)
520 return true;
521
522 if (*First == ')') {
523 put(*First++);
524 return false;
525 }
526
527 // This is intentionally fairly liberal.
528 if (!(isIdentifierBody(*First) || *First == '.' || *First == ','))
529 return true;
530
531 printAdjacentMacroArgs(First, End);
532 }

535/// Looks for an identifier starting from Last.
536///
537/// Updates "First" to just past the next identifier, if any. Returns true iff
538/// the identifier matches "Id".
539bool Minimizer::isNextIdentifier(StringRef Id, const char *&First,
540 const char *const End) {
541 skipWhitespace(First, End);
542 if (First == End || !isIdentifierHead(*First))
543 return false;
544
545 IdInfo FoundId = lexIdentifier(First, End);
546 First = FoundId.Last;
547 return FoundId.Name == Id;
548}

550bool Minimizer::lexAt(const char *&First, const char *const End) {
551 // Handle "@import".
552 const char *ImportLoc = First++;
553 if (!isNextIdentifier("import", First, End)) {
554 skipLine(First, End);
555 return false;
556 }
557 makeToken(decl_at_import);
558 append("@import ");
559 if (printAtImportBody(First, End))
560 return reportError(
561 ImportLoc, diag::err_dep_source_minimizer_missing_sema_after_at_import);
562 skipWhitespace(First, End);
563 if (First == End)
564 return false;
565 if (!isVerticalWhitespace(*First))
566 return reportError(
567 ImportLoc, diag::err_dep_source_minimizer_unexpected_tokens_at_import);
568 skipNewline(First, End);
569 return false;
570}

572bool Minimizer::lexDefine(const char *&First, const char *const End) {
573 makeToken(pp_define);
574 append("#define ");
575 skipWhitespace(First, End);
576
577 if (!isIdentifierHead(*First))
578 return reportError(First, diag::err_pp_macro_not_identifier);
579
580 IdInfo Id = lexIdentifier(First, End);
581 const char *Last = Id.Last;
582 append(Id.Name);
583 if (Last == End)
584 return false;
585 if (*Last == '(') {
586 size_t Size = Out.size();
587 if (printMacroArgs(Last, End)) {
588 // Be robust to bad macro arguments, since they can show up in disabled
589 // code.
590 Out.resize(Size);
591 append("(/* invalid */\n");
592 skipLine(Last, End);
593 return false;
594 }
595 }
596 skipWhitespace(Last, End);
597 if (Last == End)
598 return false;
599 if (!isVerticalWhitespace(*Last))
600 put(' ');
601 printDirectiveBody(Last, End);
602 First = Last;
603 return false;
604}

606bool Minimizer::lexPragma(const char *&First, const char *const End) {
607 // #pragma.
608 if (!isNextIdentifier("clang", First, End)) {
609 skipLine(First, End);
610 return false;
611 }
612
613 // #pragma clang.
614 if (!isNextIdentifier("module", First, End)) {
615 skipLine(First, End);
616 return false;
617 }
618
619 // #pragma clang module.
620 if (!isNextIdentifier("import", First, End)) {
621 skipLine(First, End);
622 return false;
623 }
624
625 // #pragma clang module import.
626 makeToken(pp_pragma_import);
627 append("#pragma clang module import ");
628 printDirectiveBody(First, End);
629 return false;
630}

632bool Minimizer::lexEndif(const char *&First, const char *const End) {
633 // Strip out "#else" if it's empty.
634 if (top() == pp_else)
635 popToken();
636
637 // Strip out "#elif" if they're empty.
638 while (top() == pp_elif)
639 popToken();
640
641 // If "#if" is empty, strip it and skip the "#endif".
642 if (top() == pp_if || top() == pp_ifdef || top() == pp_ifndef) {
643 popToken();
644 skipLine(First, End);
645 return false;
646 }
647
648 return lexDefault(pp_endif, "endif", First, End);
649}

651bool Minimizer::lexDefault(TokenKind Kind, StringRef Directive,
652 const char *&First, const char *const End) {
653 makeToken(Kind);
654 put('#').append(Directive).put(' ');
655 printDirectiveBody(First, End);
656 return false;
657}

659bool Minimizer::lexPPLine(const char *&First, const char *const End) {
660 assert(First != End);
661
662 skipWhitespace(First, End);
663 assert(First <= End);
664 if (First == End)
665 return false;
666
667 if (*First != '#' && *First != '@') {
668 skipLine(First, End);
669 assert(First <= End);
670 return false;
671 }
672
673 // Handle "@import".
674 if (*First == '@')
675 return lexAt(First, End);
676
677 // Handle preprocessing directives.
678 ++First; // Skip over '#'.
679 skipWhitespace(First, End);
680
681 if (First == End)
682 return reportError(First, diag::err_pp_expected_eol);
683
684 if (!isIdentifierHead(*First)) {
685 skipLine(First, End);
686 return false;
687 }
688
689 // Figure out the token.
690 IdInfo Id = lexIdentifier(First, End);
691 First = Id.Last;
692 auto Kind = llvm::StringSwitch<TokenKind>(Id.Name)
693 .Case("include", pp_include)
694 .Case("__include_macros", pp___include_macros)
695 .Case("define", pp_define)
696 .Case("undef", pp_undef)
697 .Case("import", pp_import)
698 .Case("include_next", pp_include_next)
699 .Case("if", pp_if)
700 .Case("ifdef", pp_ifdef)
701 .Case("ifndef", pp_ifndef)
702 .Case("elif", pp_elif)
703 .Case("else", pp_else)
704 .Case("endif", pp_endif)
705 .Case("pragma", pp_pragma_import)
706 .Default(pp_none);
707 if (Kind == pp_none) {
708 skipDirective(Id.Name, First, End);
709 return false;
710 }
711
712 if (Kind == pp_endif)
713 return lexEndif(First, End);
714
715 if (Kind == pp_define)
716 return lexDefine(First, End);
717
718 if (Kind == pp_pragma_import)
719 return lexPragma(First, End);
720
721 // Everything else.
722 return lexDefault(Kind, Id.Name, First, End);
723}

725bool Minimizer::minimizeImpl(const char *First, const char *const End) {
726 while (First != End)
727 if (lexPPLine(First, End))
728 return true;
729 return false;
730}

732bool Minimizer::minimize() {
733 bool Error = minimizeImpl(Input.begin(), Input.end());
734
735 if (!Error) {
736 // Add a trailing newline and an EOF on success.
737 if (!Out.empty() && Out.back() != '\n')
738 Out.push_back('\n');
739 makeToken(pp_eof);
740 }
741
742 // Null-terminate the output. This way the memory buffer that's passed to
743 // Clang will not have to worry about the terminating '\0'.
744 Out.push_back(0);
745 Out.pop_back();
746 return Error;
747}

750 StringRef Input, SmallVectorImpl<char> &Output,
751 SmallVectorImpl<Token> &Tokens, DiagnosticsEngine *Diags,

754 Tokens.clear();
755 return Minimizer(Output, Tokens, Input, Diags, InputSourceLoc).minimize();
