62void RegistryMaps::registerMatcher(
63 StringRef MatcherName, std::unique_ptr<MatcherDescriptor> Callback) {
64 assert(Constructors.find(MatcherName) == Constructors.end());
65 Constructors[MatcherName] = std::move(Callback);
66}

68#define REGISTER_MATCHER(name) \
69 registerMatcher(#name, internal::makeMatcherAutoMarshall( \
70 ::clang::ast_matchers::name, #name));
71
72#define REGISTER_MATCHER_OVERLOAD(name) \
73 registerMatcher(#name, \
74 llvm::make_unique<internal::OverloadedMatcherDescriptor>(name##Callbacks))
75
76#define SPECIFIC_MATCHER_OVERLOAD(name, Id) \
77 static_cast<::clang::ast_matchers::name##_Type##Id>( \
78 ::clang::ast_matchers::name)
79
80#define MATCHER_OVERLOAD_ENTRY(name, Id) \
81 internal::makeMatcherAutoMarshall(SPECIFIC_MATCHER_OVERLOAD(name, Id), \
82 #name)
83
84#define REGISTER_OVERLOADED_2(name) \
85 do { \
86 std::unique_ptr<MatcherDescriptor> name##Callbacks[] = { \
87 MATCHER_OVERLOAD_ENTRY(name, 0), \
88 MATCHER_OVERLOAD_ENTRY(name, 1)}; \
89 REGISTER_MATCHER_OVERLOAD(name); \
90 } while (false)

529llvm::Optional<MatcherCtor> Registry::lookupMatcherCtor(StringRef MatcherName) {
530 auto it = RegistryData->constructors().find(MatcherName);
531 return it == RegistryData->constructors().end()
532 ? llvm::Optional<MatcherCtor>()
533 : it->second.get();
534}

536static llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
537 const std::set<ASTNodeKind> &KS) {
538 unsigned Count = 0;
539 for (std::set<ASTNodeKind>::const_iterator I = KS.begin(), E = KS.end();
540 I != E; ++I) {
541 if (I != KS.begin())
542 OS << "|";
543 if (Count++ == 3) {
544 OS << "...";
545 break;
546 }
547 OS << *I;
548 }
549 return OS;
550}

552std::vector<ArgKind> Registry::getAcceptedCompletionTypes(
553 ArrayRef<std::pair<MatcherCtor, unsigned>> Context) {
554 ASTNodeKind InitialTypes[] = {
555 ASTNodeKind::getFromNodeKind<Decl>(),
556 ASTNodeKind::getFromNodeKind<QualType>(),
557 ASTNodeKind::getFromNodeKind<Type>(),
558 ASTNodeKind::getFromNodeKind<Stmt>(),
559 ASTNodeKind::getFromNodeKind<NestedNameSpecifier>(),
560 ASTNodeKind::getFromNodeKind<NestedNameSpecifierLoc>(),
561 ASTNodeKind::getFromNodeKind<TypeLoc>()};
562
563 // Starting with the above seed of acceptable top-level matcher types, compute
564 // the acceptable type set for the argument indicated by each context element.
565 std::set<ArgKind> TypeSet(std::begin(InitialTypes), std::end(InitialTypes));
566 for (const auto &CtxEntry : Context) {
567 MatcherCtor Ctor = CtxEntry.first;
568 unsigned ArgNumber = CtxEntry.second;
569 std::vector<ArgKind> NextTypeSet;
570 for (const ArgKind &Kind : TypeSet) {
571 if (Kind.getArgKind() == Kind.AK_Matcher &&
572 Ctor->isConvertibleTo(Kind.getMatcherKind()) &&
573 (Ctor->isVariadic() || ArgNumber < Ctor->getNumArgs()))
574 Ctor->getArgKinds(Kind.getMatcherKind(), ArgNumber, NextTypeSet);
575 }
576 TypeSet.clear();
577 TypeSet.insert(NextTypeSet.begin(), NextTypeSet.end());
578 }
579 return std::vector<ArgKind>(TypeSet.begin(), TypeSet.end());
580}

582std::vector<MatcherCompletion>
583Registry::getMatcherCompletions(ArrayRef<ArgKind> AcceptedTypes) {
584 std::vector<MatcherCompletion> Completions;
585
586 // Search the registry for acceptable matchers.
587 for (const auto &M : RegistryData->constructors()) {
588 const MatcherDescriptor& Matcher = *M.getValue();
589 StringRef Name = M.getKey();
590
591 std::set<ASTNodeKind> RetKinds;
592 unsigned NumArgs = Matcher.isVariadic() ? 1 : Matcher.getNumArgs();
593 bool IsPolymorphic = Matcher.isPolymorphic();
594 std::vector<std::vector<ArgKind>> ArgsKinds(NumArgs);
595 unsigned MaxSpecificity = 0;
596 for (const ArgKind& Kind : AcceptedTypes) {
597 if (Kind.getArgKind() != Kind.AK_Matcher)
598 continue;
599 unsigned Specificity;
600 ASTNodeKind LeastDerivedKind;
601 if (Matcher.isConvertibleTo(Kind.getMatcherKind(), &Specificity,
602 &LeastDerivedKind)) {
603 if (MaxSpecificity < Specificity)
604 MaxSpecificity = Specificity;
605 RetKinds.insert(LeastDerivedKind);
606 for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
607 Matcher.getArgKinds(Kind.getMatcherKind(), Arg, ArgsKinds[Arg]);
608 if (IsPolymorphic)
609 break;
610 }
611 }
612
613 if (!RetKinds.empty() && MaxSpecificity > 0) {
614 std::string Decl;
615 llvm::raw_string_ostream OS(Decl);
616
617 if (IsPolymorphic) {
618 OS << "Matcher<T> " << Name << "(Matcher<T>";
619 } else {
620 OS << "Matcher<" << RetKinds << "> " << Name << "(";
621 for (const std::vector<ArgKind> &Arg : ArgsKinds) {
622 if (&Arg != &ArgsKinds[0])
623 OS << ", ";
624
625 bool FirstArgKind = true;
626 std::set<ASTNodeKind> MatcherKinds;
627 // Two steps. First all non-matchers, then matchers only.
628 for (const ArgKind &AK : Arg) {
629 if (AK.getArgKind() == ArgKind::AK_Matcher) {
630 MatcherKinds.insert(AK.getMatcherKind());
631 } else {
632 if (!FirstArgKind) OS << "|";
633 FirstArgKind = false;
634 OS << AK.asString();
635 }
636 }
637 if (!MatcherKinds.empty()) {
638 if (!FirstArgKind) OS << "|";
639 OS << "Matcher<" << MatcherKinds << ">";
640 }
641 }
642 }
643 if (Matcher.isVariadic())
644 OS << "...";
645 OS << ")";
646
647 std::string TypedText = Name;
648 TypedText += "(";
649 if (ArgsKinds.empty())
650 TypedText += ")";
651 else if (ArgsKinds[0][0].getArgKind() == ArgKind::AK_String)
652 TypedText += "\"";
653
654 Completions.emplace_back(TypedText, OS.str(), MaxSpecificity);
655 }
656 }
657
658 return Completions;
659}

662 SourceRange NameRange,
663 ArrayRef<ParserValue> Args,
664 Diagnostics *Error) {

668VariantMatcher Registry::constructBoundMatcher(MatcherCtor Ctor,
669 SourceRange NameRange,
670 StringRef BindID,
671 ArrayRef<ParserValue> Args,
672 Diagnostics *Error) {
673 VariantMatcher Out = constructMatcher(Ctor, NameRange, Args, Error);
674 if (Out.isNull()) return Out;
675
676 llvm::Optional<DynTypedMatcher> Result = Out.getSingleMatcher();
677 if (Result.hasValue()) {
678 llvm::Optional<DynTypedMatcher> Bound = Result->tryBind(BindID);
679 if (Bound.hasValue()) {
680 return VariantMatcher::SingleMatcher(*Bound);
681 }
682 }
683 Error->addError(NameRange, Error->ET_RegistryNotBindable);
684 return VariantMatcher();
685}
