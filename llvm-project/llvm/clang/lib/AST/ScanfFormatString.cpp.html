31static bool ParseScanList(FormatStringHandler &H,
32 ScanfConversionSpecifier &CS,
33 const char *&Beg, const char *E) {
34 const char *I = Beg;
35 const char *start = I - 1;
36 UpdateOnReturn <const char*> UpdateBeg(Beg, I);
37
38 // No more characters?
39 if (I == E) {
40 H.HandleIncompleteScanList(start, I);
41 return true;
42 }
43
44 // Special case: ']' is the first character.
45 if (*I == ']') {
46 if (++I == E) {
47 H.HandleIncompleteScanList(start, I - 1);
48 return true;
49 }
50 }
51
52 // Special case: "^]" are the first characters.
53 if (I + 1 != E && I[0] == '^' && I[1] == ']') {
54 I += 2;
55 if (I == E) {
56 H.HandleIncompleteScanList(start, I - 1);
57 return true;
58 }
59 }
60
61 // Look for a ']' character which denotes the end of the scan list.
62 while (*I != ']') {
63 if (++I == E) {
64 H.HandleIncompleteScanList(start, I - 1);
65 return true;
66 }
67 }
68
69 CS.setEndScanList(I);
70 return false;
71}

73// FIXME: Much of this is copy-paste from ParsePrintfSpecifier.
74// We can possibly refactor.
75static ScanfSpecifierResult ParseScanfSpecifier(FormatStringHandler &H,
76 const char *&Beg,
77 const char *E,
78 unsigned &argIndex,
79 const LangOptions &LO,
80 const TargetInfo &Target) {
81 using namespace clang::analyze_format_string;
82 using namespace clang::analyze_scanf;
83 const char *I = Beg;
84 const char *Start = nullptr;
85 UpdateOnReturn <const char*> UpdateBeg(Beg, I);
86
87 // Look for a '%' character that indicates the start of a format specifier.
88 for ( ; I != E ; ++I) {
89 char c = *I;
90 if (c == '\0') {
91 // Detect spurious null characters, which are likely errors.
92 H.HandleNullChar(I);
93 return true;
94 }
95 if (c == '%') {
96 Start = I++; // Record the start of the format specifier.
97 break;
98 }
99 }
100
101 // No format specifier found?
102 if (!Start)
103 return false;
104
105 if (I == E) {
106 // No more characters left?
107 H.HandleIncompleteSpecifier(Start, E - Start);
108 return true;
109 }
110
111 ScanfSpecifier FS;
112 if (ParseArgPosition(H, FS, Start, I, E))
113 return true;
114
115 if (I == E) {
116 // No more characters left?
117 H.HandleIncompleteSpecifier(Start, E - Start);
118 return true;
119 }
120
121 // Look for '*' flag if it is present.
122 if (*I == '*') {
123 FS.setSuppressAssignment(I);
124 if (++I == E) {
125 H.HandleIncompleteSpecifier(Start, E - Start);
126 return true;
127 }
128 }
129
130 // Look for the field width (if any). Unlike printf, this is either
131 // a fixed integer or isn't present.
132 const OptionalAmount &Amt = clang::analyze_format_string::ParseAmount(I, E);
133 if (Amt.getHowSpecified() != OptionalAmount::NotSpecified) {
134 assert(Amt.getHowSpecified() == OptionalAmount::Constant);
135 FS.setFieldWidth(Amt);
136
137 if (I == E) {
138 // No more characters left?
139 H.HandleIncompleteSpecifier(Start, E - Start);
140 return true;
141 }
142 }
143
144 // Look for the length modifier.
145 if (ParseLengthModifier(FS, I, E, LO, /*scanf=*/true) && I == E) {
146 // No more characters left?
147 H.HandleIncompleteSpecifier(Start, E - Start);
148 return true;
149 }
150
151 // Detect spurious null characters, which are likely errors.
152 if (*I == '\0') {
153 H.HandleNullChar(I);
154 return true;
155 }
156
157 // Finally, look for the conversion specifier.
158 const char *conversionPosition = I++;
159 ScanfConversionSpecifier::Kind k = ScanfConversionSpecifier::InvalidSpecifier;
160 switch (*conversionPosition) {
161 default:
162 break;
163 case '%': k = ConversionSpecifier::PercentArg; break;
164 case 'A': k = ConversionSpecifier::AArg; break;
165 case 'E': k = ConversionSpecifier::EArg; break;
166 case 'F': k = ConversionSpecifier::FArg; break;
167 case 'G': k = ConversionSpecifier::GArg; break;
168 case 'X': k = ConversionSpecifier::XArg; break;
169 case 'a': k = ConversionSpecifier::aArg; break;
170 case 'd': k = ConversionSpecifier::dArg; break;
171 case 'e': k = ConversionSpecifier::eArg; break;
172 case 'f': k = ConversionSpecifier::fArg; break;
173 case 'g': k = ConversionSpecifier::gArg; break;
174 case 'i': k = ConversionSpecifier::iArg; break;
175 case 'n': k = ConversionSpecifier::nArg; break;
176 case 'c': k = ConversionSpecifier::cArg; break;
177 case 'C': k = ConversionSpecifier::CArg; break;
178 case 'S': k = ConversionSpecifier::SArg; break;
179 case '[': k = ConversionSpecifier::ScanListArg; break;
180 case 'u': k = ConversionSpecifier::uArg; break;
181 case 'x': k = ConversionSpecifier::xArg; break;
182 case 'o': k = ConversionSpecifier::oArg; break;
183 case 's': k = ConversionSpecifier::sArg; break;
184 case 'p': k = ConversionSpecifier::pArg; break;
185 // Apple extensions
186 // Apple-specific
187 case 'D':
188 if (Target.getTriple().isOSDarwin())
189 k = ConversionSpecifier::DArg;
190 break;
191 case 'O':
192 if (Target.getTriple().isOSDarwin())
193 k = ConversionSpecifier::OArg;
194 break;
195 case 'U':
196 if (Target.getTriple().isOSDarwin())
197 k = ConversionSpecifier::UArg;
198 break;
199 }
200 ScanfConversionSpecifier CS(conversionPosition, k);
201 if (k == ScanfConversionSpecifier::ScanListArg) {
202 if (ParseScanList(H, CS, I, E))
203 return true;
204 }
205 FS.setConversionSpecifier(CS);
206 if (CS.consumesDataArgument() && !FS.getSuppressAssignment()
207 && !FS.usesPositionalArg())
208 FS.setArgIndex(argIndex++);
209
210 // FIXME: '%' and '*' doesn't make sense. Issue a warning.
211 // FIXME: 'ConsumedSoFar' and '*' doesn't make sense.
212
213 if (k == ScanfConversionSpecifier::InvalidSpecifier) {
214 unsigned Len = I - Beg;
215 if (ParseUTF8InvalidSpecifier(Beg, E, Len)) {
216 CS.setEndScanList(Beg + Len);
217 FS.setConversionSpecifier(CS);
218 }
219 // Assume the conversion takes one argument.
220 return !H.HandleInvalidScanfConversionSpecifier(FS, Beg, Len);
221 }
222 return ScanfSpecifierResult(Start, FS);
223}

225ArgType ScanfSpecifier::getArgType(ASTContext &Ctx) const {
226 const ScanfConversionSpecifier &CS = getConversionSpecifier();
227
228 if (!CS.consumesDataArgument())
229 return ArgType::Invalid();
230
231 switch(CS.getKind()) {
232 // Signed int.
233 case ConversionSpecifier::dArg:
234 case ConversionSpecifier::DArg:
235 case ConversionSpecifier::iArg:
236 switch (LM.getKind()) {
237 case LengthModifier::None:
238 return ArgType::PtrTo(Ctx.IntTy);
239 case LengthModifier::AsChar:
240 return ArgType::PtrTo(ArgType::AnyCharTy);
241 case LengthModifier::AsShort:
242 return ArgType::PtrTo(Ctx.ShortTy);
243 case LengthModifier::AsLong:
244 return ArgType::PtrTo(Ctx.LongTy);
245 case LengthModifier::AsLongLong:
246 case LengthModifier::AsQuad:
247 return ArgType::PtrTo(Ctx.LongLongTy);
248 case LengthModifier::AsInt64:
249 return ArgType::PtrTo(ArgType(Ctx.LongLongTy, "__int64"));
250 case LengthModifier::AsIntMax:
251 return ArgType::PtrTo(ArgType(Ctx.getIntMaxType(), "intmax_t"));
252 case LengthModifier::AsIntPtr:
253 return ArgType::PtrTo(ArgType(Ctx.getIntPtrType(), "intptr_t"));
254 case LengthModifier::AsSizeT:
255 return ArgType::PtrTo(ArgType(Ctx.getSignedSizeType(), "ssize_t"));
256 case LengthModifier::AsPtrDiff:
257 return ArgType::PtrTo(ArgType(Ctx.getPointerDiffType(), "ptrdiff_t"));
258 case LengthModifier::AsLongDouble:
259 // GNU extension.
260 return ArgType::PtrTo(Ctx.LongLongTy);
261 case LengthModifier::AsAllocate:
262 case LengthModifier::AsMAllocate:
263 case LengthModifier::AsInt32:
264 case LengthModifier::AsInt3264:
265 case LengthModifier::AsWide:
266 case LengthModifier::AsShortLong:
267 return ArgType::Invalid();
268 }
269 llvm_unreachable("Unsupported LengthModifier Type");
270
271 // Unsigned int.
272 case ConversionSpecifier::oArg:
273 case ConversionSpecifier::OArg:
274 case ConversionSpecifier::uArg:
275 case ConversionSpecifier::UArg:
276 case ConversionSpecifier::xArg:
277 case ConversionSpecifier::XArg:
278 switch (LM.getKind()) {
279 case LengthModifier::None:
280 return ArgType::PtrTo(Ctx.UnsignedIntTy);
281 case LengthModifier::AsChar:
282 return ArgType::PtrTo(Ctx.UnsignedCharTy);
283 case LengthModifier::AsShort:
284 return ArgType::PtrTo(Ctx.UnsignedShortTy);
285 case LengthModifier::AsLong:
286 return ArgType::PtrTo(Ctx.UnsignedLongTy);
287 case LengthModifier::AsLongLong:
288 case LengthModifier::AsQuad:
289 return ArgType::PtrTo(Ctx.UnsignedLongLongTy);
290 case LengthModifier::AsInt64:
291 return ArgType::PtrTo(ArgType(Ctx.UnsignedLongLongTy, "unsigned __int64"));
292 case LengthModifier::AsIntMax:
293 return ArgType::PtrTo(ArgType(Ctx.getUIntMaxType(), "uintmax_t"));
294 case LengthModifier::AsIntPtr:
295 return ArgType::PtrTo(ArgType(Ctx.getUIntPtrType(), "uintptr_t"));
296 case LengthModifier::AsSizeT:
297 return ArgType::PtrTo(ArgType(Ctx.getSizeType(), "size_t"));
298 case LengthModifier::AsPtrDiff:
299 return ArgType::PtrTo(
300 ArgType(Ctx.getUnsignedPointerDiffType(), "unsigned ptrdiff_t"));
301 case LengthModifier::AsLongDouble:
302 // GNU extension.
303 return ArgType::PtrTo(Ctx.UnsignedLongLongTy);
304 case LengthModifier::AsAllocate:
305 case LengthModifier::AsMAllocate:
306 case LengthModifier::AsInt32:
307 case LengthModifier::AsInt3264:
308 case LengthModifier::AsWide:
309 case LengthModifier::AsShortLong:
310 return ArgType::Invalid();
311 }
312 llvm_unreachable("Unsupported LengthModifier Type");
313
314 // Float.
315 case ConversionSpecifier::aArg:
316 case ConversionSpecifier::AArg:
317 case ConversionSpecifier::eArg:
318 case ConversionSpecifier::EArg:
319 case ConversionSpecifier::fArg:
320 case ConversionSpecifier::FArg:
321 case ConversionSpecifier::gArg:
322 case ConversionSpecifier::GArg:
323 switch (LM.getKind()) {
324 case LengthModifier::None:
325 return ArgType::PtrTo(Ctx.FloatTy);
326 case LengthModifier::AsLong:
327 return ArgType::PtrTo(Ctx.DoubleTy);
328 case LengthModifier::AsLongDouble:
329 return ArgType::PtrTo(Ctx.LongDoubleTy);
330 default:
331 return ArgType::Invalid();
332 }
333
334 // Char, string and scanlist.
335 case ConversionSpecifier::cArg:
336 case ConversionSpecifier::sArg:
337 case ConversionSpecifier::ScanListArg:
338 switch (LM.getKind()) {
339 case LengthModifier::None:
340 return ArgType::PtrTo(ArgType::AnyCharTy);
341 case LengthModifier::AsLong:
342 case LengthModifier::AsWide:
343 return ArgType::PtrTo(ArgType(Ctx.getWideCharType(), "wchar_t"));
344 case LengthModifier::AsAllocate:
345 case LengthModifier::AsMAllocate:
346 return ArgType::PtrTo(ArgType::CStrTy);
347 case LengthModifier::AsShort:
348 if (Ctx.getTargetInfo().getTriple().isOSMSVCRT())
349 return ArgType::PtrTo(ArgType::AnyCharTy);
350 LLVM_FALLTHROUGH;
351 default:
352 return ArgType::Invalid();
353 }
354 case ConversionSpecifier::CArg:
355 case ConversionSpecifier::SArg:
356 // FIXME: Mac OS X specific?
357 switch (LM.getKind()) {
358 case LengthModifier::None:
359 case LengthModifier::AsWide:
360 return ArgType::PtrTo(ArgType(Ctx.getWideCharType(), "wchar_t"));
361 case LengthModifier::AsAllocate:
362 case LengthModifier::AsMAllocate:
363 return ArgType::PtrTo(ArgType(ArgType::WCStrTy, "wchar_t *"));
364 case LengthModifier::AsShort:
365 if (Ctx.getTargetInfo().getTriple().isOSMSVCRT())
366 return ArgType::PtrTo(ArgType::AnyCharTy);
367 LLVM_FALLTHROUGH;
368 default:
369 return ArgType::Invalid();
370 }
371
372 // Pointer.
373 case ConversionSpecifier::pArg:
374 return ArgType::PtrTo(ArgType::CPointerTy);
375
376 // Write-back.
377 case ConversionSpecifier::nArg:
378 switch (LM.getKind()) {
379 case LengthModifier::None:
380 return ArgType::PtrTo(Ctx.IntTy);
381 case LengthModifier::AsChar:
382 return ArgType::PtrTo(Ctx.SignedCharTy);
383 case LengthModifier::AsShort:
384 return ArgType::PtrTo(Ctx.ShortTy);
385 case LengthModifier::AsLong:
386 return ArgType::PtrTo(Ctx.LongTy);
387 case LengthModifier::AsLongLong:
388 case LengthModifier::AsQuad:
389 return ArgType::PtrTo(Ctx.LongLongTy);
390 case LengthModifier::AsInt64:
391 return ArgType::PtrTo(ArgType(Ctx.LongLongTy, "__int64"));
392 case LengthModifier::AsIntMax:
393 return ArgType::PtrTo(ArgType(Ctx.getIntMaxType(), "intmax_t"));
394 case LengthModifier::AsIntPtr:
395 return ArgType::PtrTo(ArgType(Ctx.getIntPtrType(), "intptr_t"));
396 case LengthModifier::AsSizeT:
397 return ArgType::PtrTo(ArgType(Ctx.getSignedSizeType(), "ssize_t"));
398 case LengthModifier::AsPtrDiff:
399 return ArgType::PtrTo(ArgType(Ctx.getPointerDiffType(), "ptrdiff_t"));
400 case LengthModifier::AsLongDouble:
401 return ArgType(); // FIXME: Is this a known extension?
402 case LengthModifier::AsAllocate:
403 case LengthModifier::AsMAllocate:
404 case LengthModifier::AsInt32:
405 case LengthModifier::AsInt3264:
406 case LengthModifier::AsWide:
407 case LengthModifier::AsShortLong:
408 return ArgType::Invalid();
409 }
410
411 default:
412 break;
413 }
414
415 return ArgType();
416}

418bool ScanfSpecifier::fixType(QualType QT, QualType RawQT,
419 const LangOptions &LangOpt,
420 ASTContext &Ctx) {
421
422 // %n is different from other conversion specifiers; don't try to fix it.
423 if (CS.getKind() == ConversionSpecifier::nArg)
424 return false;
425
426 if (!QT->isPointerType())
427 return false;
428
429 QualType PT = QT->getPointeeType();
430
431 // If it's an enum, get its underlying type.
432 if (const EnumType *ETy = PT->getAs<EnumType>()) {
433 // Don't try to fix incomplete enums.
434 if (!ETy->getDecl()->isComplete())
435 return false;
436 PT = ETy->getDecl()->getIntegerType();
437 }
438
439 const BuiltinType *BT = PT->getAs<BuiltinType>();
440 if (!BT)
441 return false;
442
443 // Pointer to a character.
444 if (PT->isAnyCharacterType()) {
445 CS.setKind(ConversionSpecifier::sArg);
446 if (PT->isWideCharType())
447 LM.setKind(LengthModifier::AsWideChar);
448 else
449 LM.setKind(LengthModifier::None);
450
451 // If we know the target array length, we can use it as a field width.
452 if (const ConstantArrayType *CAT = Ctx.getAsConstantArrayType(RawQT)) {
453 if (CAT->getSizeModifier() == ArrayType::Normal)
454 FieldWidth = OptionalAmount(OptionalAmount::Constant,
455 CAT->getSize().getZExtValue() - 1,
456 "", 0, false);
457
458 }
459 return true;
460 }
461
462 // Figure out the length modifier.
463 switch (BT->getKind()) {
464 // no modifier
465 case BuiltinType::UInt:
466 case BuiltinType::Int:
467 case BuiltinType::Float:
468 LM.setKind(LengthModifier::None);
469 break;
470
471 // hh
472 case BuiltinType::Char_U:
473 case BuiltinType::UChar:
474 case BuiltinType::Char_S:
475 case BuiltinType::SChar:
476 LM.setKind(LengthModifier::AsChar);
477 break;
478
479 // h
480 case BuiltinType::Short:
481 case BuiltinType::UShort:
482 LM.setKind(LengthModifier::AsShort);
483 break;
484
485 // l
486 case BuiltinType::Long:
487 case BuiltinType::ULong:
488 case BuiltinType::Double:
489 LM.setKind(LengthModifier::AsLong);
490 break;
491
492 // ll
493 case BuiltinType::LongLong:
494 case BuiltinType::ULongLong:
495 LM.setKind(LengthModifier::AsLongLong);
496 break;
497
498 // L
499 case BuiltinType::LongDouble:
500 LM.setKind(LengthModifier::AsLongDouble);
501 break;
502
503 // Don't know.
504 default:
505 return false;
506 }
507
508 // Handle size_t, ptrdiff_t, etc. that have dedicated length modifiers in C99.
509 if (isa<TypedefType>(PT) && (LangOpt.C99 || LangOpt.CPlusPlus11))
510 namedTypeToLengthModifier(PT, LM);
511
512 // If fixing the length modifier was enough, we are done.
513 if (hasValidLengthModifier(Ctx.getTargetInfo(), LangOpt)) {
514 const analyze_scanf::ArgType &AT = getArgType(Ctx);
515 if (AT.isValid() && AT.matchesType(Ctx, QT))
516 return true;
517 }
518
519 // Figure out the conversion specifier.
520 if (PT->isRealFloatingType())
521 CS.setKind(ConversionSpecifier::fArg);
522 else if (PT->isSignedIntegerType())
523 CS.setKind(ConversionSpecifier::dArg);
524 else if (PT->isUnsignedIntegerType())
525 CS.setKind(ConversionSpecifier::uArg);
526 else
527 llvm_unreachable("Unexpected type");
528
529 return true;
530}

532void ScanfSpecifier::toString(raw_ostream &os) const {
533 os << "%";
534
535 if (usesPositionalArg())
536 os << getPositionalArgIndex() << "$";
537 if (SuppressAssignment)
538 os << "*";
539
540 FieldWidth.toString(os);
541 os << LM.toString();
542 os << CS.toString();
543}

545bool clang::analyze_format_string::ParseScanfString(FormatStringHandler &H,
546 const char *I,
547 const char *E,
548 const LangOptions &LO,
549 const TargetInfo &Target) {
550
551 unsigned argIndex = 0;
552
553 // Keep looking for a format specifier until we have exhausted the string.
554 while (I != E) {
555 const ScanfSpecifierResult &FSR = ParseScanfSpecifier(H, I, E, argIndex,
556 LO, Target);
557 // Did a fail-stop error of any kind occur when parsing the specifier?
558 // If so, don't do any more processing.
559 if (FSR.shouldStop())
560 return true;
561 // Did we exhaust the string or encounter an error that
562 // we can recover from?
563 if (!FSR.hasValue())
564 continue;
565 // We have a format specifier. Pass it to the callback.
566 if (!H.HandleScanfSpecifier(FSR.getValue(), FSR.getStart(),
567 I - FSR.getStart())) {
568 return true;
569 }
570 }
571 assert(I == E && "Format string not exhausted");
572 return false;
573}
