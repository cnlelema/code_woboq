90/// Returns the size of the type source info data block.
91unsigned TypeLoc::getFullDataSizeForType(QualType Ty) {
92 unsigned Total = 0;
93 TypeLoc TyLoc(Ty, nullptr);
94 unsigned MaxAlign = 1;
95 while (!TyLoc.isNull()) {
96 unsigned Align = getLocalAlignmentForType(TyLoc.getType());
97 MaxAlign = std::max(Align, MaxAlign);
98 Total = llvm::alignTo(Total, Align);
99 Total += TypeSizer().Visit(TyLoc);
100 TyLoc = TyLoc.getNextTypeLoc();
101 }
102 Total = llvm::alignTo(Total, MaxAlign);
103 return Total;
104}

126/// Initializes a type location, and all of its children
127/// recursively, as if the entire tree had been written in the
128/// given location.
129void TypeLoc::initializeImpl(ASTContext &Context, TypeLoc TL,
130 SourceLocation Loc) {
131 while (true) {
132 switch (TL.getTypeLocClass()) {
133#define ABSTRACT_TYPELOC(CLASS, PARENT)
134#define TYPELOC(CLASS, PARENT) \
135 case CLASS: { \
136 CLASS##TypeLoc TLCasted = TL.castAs<CLASS##TypeLoc>(); \
137 TLCasted.initializeLocal(Context, Loc); \
138 TL = TLCasted.getNextTypeLoc(); \
139 if (!TL) return; \
140 continue; \
141 }
142#include "clang/AST/TypeLocNodes.def"
143 }
144 }
145}

165void TypeLoc::copy(TypeLoc other) {
166 assert(getFullDataSize() == other.getFullDataSize());
167
168 // If both data pointers are aligned to the maximum alignment, we
169 // can memcpy because getFullDataSize() accurately reflects the
170 // layout of the data.
171 if (reinterpret_cast<uintptr_t>(Data) ==
172 llvm::alignTo(reinterpret_cast<uintptr_t>(Data),
173 TypeLocMaxDataAlign) &&
174 reinterpret_cast<uintptr_t>(other.Data) ==
175 llvm::alignTo(reinterpret_cast<uintptr_t>(other.Data),
176 TypeLocMaxDataAlign)) {
177 memcpy(Data, other.Data, getFullDataSize());
178 return;
179 }
180
181 // Copy each of the pieces.
182 TypeLoc TL(getType(), Data);
183 do {
184 TypeLocCopier(other).Visit(TL);
185 other = other.getNextTypeLoc();
186 } while ((TL = TL.getNextTypeLoc()));
187}

189SourceLocation TypeLoc::getBeginLoc() const {
190 TypeLoc Cur = *this;
191 TypeLoc LeftMost = Cur;
192 while (true) {
193 switch (Cur.getTypeLocClass()) {
194 case Elaborated:
195 LeftMost = Cur;
196 break;
197 case FunctionProto:
198 if (Cur.castAs<FunctionProtoTypeLoc>().getTypePtr()
199 ->hasTrailingReturn()) {
200 LeftMost = Cur;
201 break;
202 }
203 LLVM_FALLTHROUGH;
204 case FunctionNoProto:
205 case ConstantArray:
206 case DependentSizedArray:
207 case IncompleteArray:
208 case VariableArray:
209 // FIXME: Currently QualifiedTypeLoc does not have a source range
210 case Qualified:
211 Cur = Cur.getNextTypeLoc();
212 continue;
213 default:
214 if (Cur.getLocalSourceRange().getBegin().isValid())
215 LeftMost = Cur;
216 Cur = Cur.getNextTypeLoc();
217 if (Cur.isNull())
218 break;
219 continue;
220 } // switch
221 break;
222 } // while
223 return LeftMost.getLocalSourceRange().getBegin();
224}
225
226SourceLocation TypeLoc::getEndLoc() const {
227 TypeLoc Cur = *this;
228 TypeLoc Last;
229 while (true) {
230 switch (Cur.getTypeLocClass()) {
231 default:
232 if (!Last)
233 Last = Cur;
234 return Last.getLocalSourceRange().getEnd();
235 case Paren:
236 case ConstantArray:
237 case DependentSizedArray:
238 case IncompleteArray:
239 case VariableArray:
240 case FunctionNoProto:
241 Last = Cur;
242 break;
243 case FunctionProto:
244 if (Cur.castAs<FunctionProtoTypeLoc>().getTypePtr()->hasTrailingReturn())
245 Last = TypeLoc();
246 else
247 Last = Cur;
248 break;
249 case Pointer:
250 case BlockPointer:
251 case MemberPointer:
252 case LValueReference:
253 case RValueReference:
254 case PackExpansion:
255 if (!Last)
256 Last = Cur;
257 break;
258 case Qualified:
259 case Elaborated:
260 break;
261 }
262 Cur = Cur.getNextTypeLoc();
263 }

275 bool Visit##CLASS##TypeLoc(CLASS##TypeLoc TyLoc) { \
276 return isTypeSpec(TyLoc); \

283/// Determines if the given type loc corresponds to a
284/// TypeSpecTypeLoc. Since there is not actually a TypeSpecType in
285/// the type hierarchy, this is made somewhat complicated.
286///
287/// There are a lot of types that currently use TypeSpecTypeLoc
288/// because it's a convenient base class. Ideally we would not accept
289/// those here, but ideally we would have better implementations for
290/// them.
291bool TypeSpecTypeLoc::isKind(const TypeLoc &TL) {
292 if (TL.getType().hasLocalQualifiers()) return false;
293 return TSTChecker().Visit(TL);
294}

300 if (getRParenLoc().isValid())
301 return SourceRange(getTypeofLoc(), getRParenLoc());
302 else
303 return SourceRange(getTypeofLoc(),
304 getUnderlyingExpr()->getSourceRange().getEnd());

308TypeSpecifierType BuiltinTypeLoc::getWrittenTypeSpec() const {
309 if (needsExtraLocalData())
310 return static_cast<TypeSpecifierType>(getWrittenBuiltinSpecs().Type);
311 switch (getTypePtr()->getKind()) {

404TypeLoc TypeLoc::IgnoreParensImpl(TypeLoc TL) {
405 while (ParenTypeLoc PTL = TL.getAs<ParenTypeLoc>())
406 TL = PTL.getInnerLoc();
407 return TL;

410SourceLocation TypeLoc::findNullabilityLoc() const {
411 if (auto ATL = getAs<AttributedTypeLoc>()) {
412 const Attr *A = ATL.getAttr();
413 if (A && (isa<TypeNullableAttr>(A) || isa<TypeNonNullAttr>(A) ||
414 isa<TypeNullUnspecifiedAttr>(A)))
415 return A->getLocation();
416 }
417
418 return {};
419}

421TypeLoc TypeLoc::findExplicitQualifierLoc() const {
422 // Qualified types.
423 if (auto qual = getAs<QualifiedTypeLoc>())
424 return qual;
425
426 TypeLoc loc = IgnoreParens();
427
428 // Attributed types.
429 if (auto attr = loc.getAs<AttributedTypeLoc>()) {
430 if (attr.isQualifier()) return attr;
431 return attr.getModifiedLoc().findExplicitQualifierLoc();
432 }
433
434 // C11 _Atomic types.
435 if (auto atomic = loc.getAs<AtomicTypeLoc>()) {
436 return atomic;
437 }
438
439 return {};
440}

442void ObjCTypeParamTypeLoc::initializeLocal(ASTContext &Context,
443 SourceLocation Loc) {
444 setNameLoc(Loc);
445 if (!getNumProtocols()) return;
446
447 setProtocolLAngleLoc(Loc);
448 setProtocolRAngleLoc(Loc);
449 for (unsigned i = 0, e = getNumProtocols(); i != e; ++i)
450 setProtocolLoc(i, Loc);
451}

453void ObjCObjectTypeLoc::initializeLocal(ASTContext &Context,
454 SourceLocation Loc) {
455 setHasBaseTypeAsWritten(true);
456 setTypeArgsLAngleLoc(Loc);
457 setTypeArgsRAngleLoc(Loc);
458 for (unsigned i = 0, e = getNumTypeArgs(); i != e; ++i) {
459 setTypeArgTInfo(i,
460 Context.getTrivialTypeSourceInfo(
461 getTypePtr()->getTypeArgsAsWritten()[i], Loc));
462 }
463 setProtocolLAngleLoc(Loc);
464 setProtocolRAngleLoc(Loc);
465 for (unsigned i = 0, e = getNumProtocols(); i != e; ++i)
466 setProtocolLoc(i, Loc);
467}

504DependentTemplateSpecializationTypeLoc::initializeLocal(ASTContext &Context,
505 SourceLocation Loc) {
506 setElaboratedKeywordLoc(Loc);
507 if (getTypePtr()->getQualifier()) {
508 NestedNameSpecifierLocBuilder Builder;
509 Builder.MakeTrivial(Context, getTypePtr()->getQualifier(), Loc);
510 setQualifierLoc(Builder.getWithLocInContext(Context));
511 } else {
512 setQualifierLoc(NestedNameSpecifierLoc());
513 }
514 setTemplateKeywordLoc(Loc);
515 setTemplateNameLoc(Loc);
516 setLAngleLoc(Loc);
517 setRAngleLoc(Loc);
518 TemplateSpecializationTypeLoc::initializeArgLocs(Context, getNumArgs(),
519 getTypePtr()->getArgs(),
520 getArgInfos(), Loc);
521}

523void TemplateSpecializationTypeLoc::initializeArgLocs(ASTContext &Context,
524 unsigned NumArgs,
525 const TemplateArgument *Args,
526 TemplateArgumentLocInfo *ArgInfos,
527 SourceLocation Loc) {
528 for (unsigned i = 0, e = NumArgs; i != e; ++i) {
529 switch (Args[i].getKind()) {
530 case TemplateArgument::Null:
531 llvm_unreachable("Impossible TemplateArgument");
532
533 case TemplateArgument::Integral:
534 case TemplateArgument::Declaration:
535 case TemplateArgument::NullPtr:
536 ArgInfos[i] = TemplateArgumentLocInfo();
537 break;
538
539 case TemplateArgument::Expression:
540 ArgInfos[i] = TemplateArgumentLocInfo(Args[i].getAsExpr());
541 break;
542
543 case TemplateArgument::Type:
544 ArgInfos[i] = TemplateArgumentLocInfo(
545 Context.getTrivialTypeSourceInfo(Args[i].getAsType(),
546 Loc));
547 break;
548
549 case TemplateArgument::Template:
550 case TemplateArgument::TemplateExpansion: {
551 NestedNameSpecifierLocBuilder Builder;
552 TemplateName Template = Args[i].getAsTemplateOrTemplatePattern();
553 if (DependentTemplateName *DTN = Template.getAsDependentTemplateName())
554 Builder.MakeTrivial(Context, DTN->getQualifier(), Loc);
555 else if (QualifiedTemplateName *QTN = Template.getAsQualifiedTemplateName())
556 Builder.MakeTrivial(Context, QTN->getQualifier(), Loc);
557
558 ArgInfos[i] = TemplateArgumentLocInfo(
559 Builder.getWithLocInContext(Context), Loc,
560 Args[i].getKind() == TemplateArgument::Template ? SourceLocation()
561 : Loc);
562 break;
563 }
564
565 case TemplateArgument::Pack:
566 ArgInfos[i] = TemplateArgumentLocInfo();
567 break;
568 }
569 }
