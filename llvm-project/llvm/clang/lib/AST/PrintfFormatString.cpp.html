37static bool ParsePrecision(FormatStringHandler &H, PrintfSpecifier &FS,
38 const char *Start, const char *&Beg, const char *E,
39 unsigned *argIndex) {
40 if (argIndex) {
41 FS.setPrecision(ParseNonPositionAmount(Beg, E, *argIndex));
42 } else {
43 const OptionalAmount Amt = ParsePositionAmount(H, Start, Beg, E,
44 analyze_format_string::PrecisionPos);
45 if (Amt.isInvalid())
46 return true;
47 FS.setPrecision(Amt);
48 }
49 return false;
50}

52static bool ParseObjCFlags(FormatStringHandler &H, PrintfSpecifier &FS,
53 const char *FlagBeg, const char *E, bool Warn) {
54 StringRef Flag(FlagBeg, E - FlagBeg);
55 // Currently there is only one flag.
56 if (Flag == "tt") {
57 FS.setHasObjCTechnicalTerm(FlagBeg);
58 return false;
59 }
60 // Handle either the case of no flag or an invalid flag.
61 if (Warn) {
62 if (Flag == "")
63 H.HandleEmptyObjCModifierFlag(FlagBeg, E - FlagBeg);
64 else
65 H.HandleInvalidObjCModifierFlag(FlagBeg, E - FlagBeg);
66 }
67 return true;
68}

70static PrintfSpecifierResult ParsePrintfSpecifier(FormatStringHandler &H,
71 const char *&Beg,
72 const char *E,
73 unsigned &argIndex,
74 const LangOptions &LO,
75 const TargetInfo &Target,
76 bool Warn,
77 bool isFreeBSDKPrintf) {
78
79 using namespace clang::analyze_format_string;
80 using namespace clang::analyze_printf;
81
82 const char *I = Beg;
83 const char *Start = nullptr;
84 UpdateOnReturn <const char*> UpdateBeg(Beg, I);
85
86 // Look for a '%' character that indicates the start of a format specifier.
87 for ( ; I != E ; ++I) {
88 char c = *I;
89 if (c == '\0') {
90 // Detect spurious null characters, which are likely errors.
91 H.HandleNullChar(I);
92 return true;
93 }
94 if (c == '%') {
95 Start = I++; // Record the start of the format specifier.
96 break;
97 }
98 }
99
100 // No format specifier found?
101 if (!Start)
102 return false;
103
104 if (I == E) {
105 // No more characters left?
106 if (Warn)
107 H.HandleIncompleteSpecifier(Start, E - Start);
108 return true;
109 }
110
111 PrintfSpecifier FS;
112 if (ParseArgPosition(H, FS, Start, I, E))
113 return true;
114
115 if (I == E) {
116 // No more characters left?
117 if (Warn)
118 H.HandleIncompleteSpecifier(Start, E - Start);
119 return true;
120 }
121
122 if (*I == '{') {
123 ++I;
124 unsigned char PrivacyFlags = 0;
125 StringRef MatchedStr;
126
127 do {
128 StringRef Str(I, E - I);
129 std::string Match = "^[[:space:]]*"
130 "(private|public|sensitive|mask\\.[^[:space:],}]*)"
131 "[[:space:]]*(,|})";
132 llvm::Regex R(Match);
133 SmallVector<StringRef, 2> Matches;
134
135 if (R.match(Str, &Matches)) {
136 MatchedStr = Matches[1];
137 I += Matches[0].size();
138
139 // Set the privacy flag if the privacy annotation in the
140 // comma-delimited segment is at least as strict as the privacy
141 // annotations in previous comma-delimited segments.
142 if (MatchedStr.startswith("mask")) {
143 StringRef MaskType = MatchedStr.substr(sizeof("mask.") - 1);
144 unsigned Size = MaskType.size();
145 if (Warn && (Size == 0 || Size > 8))
146 H.handleInvalidMaskType(MaskType);
147 FS.setMaskType(MaskType);
148 } else if (MatchedStr.equals("sensitive"))
149 PrivacyFlags = clang::analyze_os_log::OSLogBufferItem::IsSensitive;
150 else if (PrivacyFlags !=
151 clang::analyze_os_log::OSLogBufferItem::IsSensitive &&
152 MatchedStr.equals("private"))
153 PrivacyFlags = clang::analyze_os_log::OSLogBufferItem::IsPrivate;
154 else if (PrivacyFlags == 0 && MatchedStr.equals("public"))
155 PrivacyFlags = clang::analyze_os_log::OSLogBufferItem::IsPublic;
156 } else {
157 size_t CommaOrBracePos =
158 Str.find_if([](char c) { return c == ',' || c == '}'; });
159
160 if (CommaOrBracePos == StringRef::npos) {
161 // Neither a comma nor the closing brace was found.
162 if (Warn)
163 H.HandleIncompleteSpecifier(Start, E - Start);
164 return true;
165 }
166
167 I += CommaOrBracePos + 1;
168 }
169 // Continue until the closing brace is found.
170 } while (*(I - 1) == ',');
171
172 // Set the privacy flag.
173 switch (PrivacyFlags) {
174 case 0:
175 break;
176 case clang::analyze_os_log::OSLogBufferItem::IsPrivate:
177 FS.setIsPrivate(MatchedStr.data());
178 break;
179 case clang::analyze_os_log::OSLogBufferItem::IsPublic:
180 FS.setIsPublic(MatchedStr.data());
181 break;
182 case clang::analyze_os_log::OSLogBufferItem::IsSensitive:
183 FS.setIsSensitive(MatchedStr.data());
184 break;
185 default:
186 llvm_unreachable("Unexpected privacy flag value");
187 }
188 }
189
190 // Look for flags (if any).
191 bool hasMore = true;
192 for ( ; I != E; ++I) {
193 switch (*I) {
194 default: hasMore = false; break;
195 case '\'':
196 // FIXME: POSIX specific. Always accept?
197 FS.setHasThousandsGrouping(I);
198 break;
199 case '-': FS.setIsLeftJustified(I); break;
200 case '+': FS.setHasPlusPrefix(I); break;
201 case ' ': FS.setHasSpacePrefix(I); break;
202 case '#': FS.setHasAlternativeForm(I); break;
203 case '0': FS.setHasLeadingZeros(I); break;
204 }
205 if (!hasMore)
206 break;
207 }
208
209 if (I == E) {
210 // No more characters left?
211 if (Warn)
212 H.HandleIncompleteSpecifier(Start, E - Start);
213 return true;
214 }
215
216 // Look for the field width (if any).
217 if (ParseFieldWidth(H, FS, Start, I, E,
218 FS.usesPositionalArg() ? nullptr : &argIndex))
219 return true;
220
221 if (I == E) {
222 // No more characters left?
223 if (Warn)
224 H.HandleIncompleteSpecifier(Start, E - Start);
225 return true;
226 }
227
228 // Look for the precision (if any).
229 if (*I == '.') {
230 ++I;
231 if (I == E) {
232 if (Warn)
233 H.HandleIncompleteSpecifier(Start, E - Start);
234 return true;
235 }
236
237 if (ParsePrecision(H, FS, Start, I, E,
238 FS.usesPositionalArg() ? nullptr : &argIndex))
239 return true;
240
241 if (I == E) {
242 // No more characters left?
243 if (Warn)
244 H.HandleIncompleteSpecifier(Start, E - Start);
245 return true;
246 }
247 }
248
249 if (ParseVectorModifier(H, FS, I, E, LO))
250 return true;
251
252 // Look for the length modifier.
253 if (ParseLengthModifier(FS, I, E, LO) && I == E) {
254 // No more characters left?
255 if (Warn)
256 H.HandleIncompleteSpecifier(Start, E - Start);
257 return true;
258 }
259
260 // Look for the Objective-C modifier flags, if any.
261 // We parse these here, even if they don't apply to
262 // the conversion specifier, and then emit an error
263 // later if the conversion specifier isn't '@'. This
264 // enables better recovery, and we don't know if
265 // these flags are applicable until later.
266 const char *ObjCModifierFlagsStart = nullptr,
267 *ObjCModifierFlagsEnd = nullptr;
268 if (*I == '[') {
269 ObjCModifierFlagsStart = I;
270 ++I;
271 auto flagStart = I;
272 for (;; ++I) {
273 ObjCModifierFlagsEnd = I;
274 if (I == E) {
275 if (Warn)
276 H.HandleIncompleteSpecifier(Start, E - Start);
277 return true;
278 }
279 // Did we find the closing ']'?
280 if (*I == ']') {
281 if (ParseObjCFlags(H, FS, flagStart, I, Warn))
282 return true;
283 ++I;
284 break;
285 }
286 // There are no separators defined yet for multiple
287 // Objective-C modifier flags. When those are
288 // defined, this is the place to check.
289 }
290 }
291
292 if (*I == '\0') {
293 // Detect spurious null characters, which are likely errors.
294 H.HandleNullChar(I);
295 return true;
296 }
297
298 // Finally, look for the conversion specifier.
299 const char *conversionPosition = I++;
300 ConversionSpecifier::Kind k = ConversionSpecifier::InvalidSpecifier;
301 switch (*conversionPosition) {
302 default:
303 break;
304 // C99: 7.19.6.1 (section 8).
305 case '%': k = ConversionSpecifier::PercentArg; break;
306 case 'A': k = ConversionSpecifier::AArg; break;
307 case 'E': k = ConversionSpecifier::EArg; break;
308 case 'F': k = ConversionSpecifier::FArg; break;
309 case 'G': k = ConversionSpecifier::GArg; break;
310 case 'X': k = ConversionSpecifier::XArg; break;
311 case 'a': k = ConversionSpecifier::aArg; break;
312 case 'c': k = ConversionSpecifier::cArg; break;
313 case 'd': k = ConversionSpecifier::dArg; break;
314 case 'e': k = ConversionSpecifier::eArg; break;
315 case 'f': k = ConversionSpecifier::fArg; break;
316 case 'g': k = ConversionSpecifier::gArg; break;
317 case 'i': k = ConversionSpecifier::iArg; break;
318 case 'n':
319 // Not handled, but reserved in OpenCL.
320 if (!LO.OpenCL)
321 k = ConversionSpecifier::nArg;
322 break;
323 case 'o': k = ConversionSpecifier::oArg; break;
324 case 'p': k = ConversionSpecifier::pArg; break;
325 case 's': k = ConversionSpecifier::sArg; break;
326 case 'u': k = ConversionSpecifier::uArg; break;
327 case 'x': k = ConversionSpecifier::xArg; break;
328 // POSIX specific.
329 case 'C': k = ConversionSpecifier::CArg; break;
330 case 'S': k = ConversionSpecifier::SArg; break;
331 // Apple extension for os_log
332 case 'P':
333 k = ConversionSpecifier::PArg;
334 break;
335 // Objective-C.
336 case '@': k = ConversionSpecifier::ObjCObjArg; break;
337 // Glibc specific.
338 case 'm': k = ConversionSpecifier::PrintErrno; break;
339 // FreeBSD kernel specific.
340 case 'b':
341 if (isFreeBSDKPrintf)
342 k = ConversionSpecifier::FreeBSDbArg; // int followed by char *
343 break;
344 case 'r':
345 if (isFreeBSDKPrintf)
346 k = ConversionSpecifier::FreeBSDrArg; // int
347 break;
348 case 'y':
349 if (isFreeBSDKPrintf)
350 k = ConversionSpecifier::FreeBSDyArg; // int
351 break;
352 // Apple-specific.
353 case 'D':
354 if (isFreeBSDKPrintf)
355 k = ConversionSpecifier::FreeBSDDArg; // void * followed by char *
356 else if (Target.getTriple().isOSDarwin())
357 k = ConversionSpecifier::DArg;
358 break;
359 case 'O':
360 if (Target.getTriple().isOSDarwin())
361 k = ConversionSpecifier::OArg;
362 break;
363 case 'U':
364 if (Target.getTriple().isOSDarwin())
365 k = ConversionSpecifier::UArg;
366 break;
367 // MS specific.
368 case 'Z':
369 if (Target.getTriple().isOSMSVCRT())
370 k = ConversionSpecifier::ZArg;
371 break;
372 }
373
374 // Check to see if we used the Objective-C modifier flags with
375 // a conversion specifier other than '@'.
376 if (k != ConversionSpecifier::ObjCObjArg &&
377 k != ConversionSpecifier::InvalidSpecifier &&
378 ObjCModifierFlagsStart) {
379 H.HandleObjCFlagsWithNonObjCConversion(ObjCModifierFlagsStart,
380 ObjCModifierFlagsEnd + 1,
381 conversionPosition);
382 return true;
383 }
384
385 PrintfConversionSpecifier CS(conversionPosition, k);
386 FS.setConversionSpecifier(CS);
387 if (CS.consumesDataArgument() && !FS.usesPositionalArg())
388 FS.setArgIndex(argIndex++);
389 // FreeBSD kernel specific.
390 if (k == ConversionSpecifier::FreeBSDbArg ||
391 k == ConversionSpecifier::FreeBSDDArg)
392 argIndex++;
393
394 if (k == ConversionSpecifier::InvalidSpecifier) {
395 unsigned Len = I - Start;
396 if (ParseUTF8InvalidSpecifier(Start, E, Len)) {
397 CS.setEndScanList(Start + Len);
398 FS.setConversionSpecifier(CS);
399 }
400 // Assume the conversion takes one argument.
401 return !H.HandleInvalidPrintfConversionSpecifier(FS, Start, Len);
402 }
403 return PrintfSpecifierResult(Start, FS);
404}

406bool clang::analyze_format_string::ParsePrintfString(FormatStringHandler &H,
407 const char *I,
408 const char *E,
409 const LangOptions &LO,
410 const TargetInfo &Target,
411 bool isFreeBSDKPrintf) {
412
413 unsigned argIndex = 0;
414
415 // Keep looking for a format specifier until we have exhausted the string.
416 while (I != E) {
417 const PrintfSpecifierResult &FSR = ParsePrintfSpecifier(H, I, E, argIndex,
418 LO, Target, true,
419 isFreeBSDKPrintf);
420 // Did a fail-stop error of any kind occur when parsing the specifier?
421 // If so, don't do any more processing.
422 if (FSR.shouldStop())
423 return true;
424 // Did we exhaust the string or encounter an error that
425 // we can recover from?
426 if (!FSR.hasValue())
427 continue;
428 // We have a format specifier. Pass it to the callback.
429 if (!H.HandlePrintfSpecifier(FSR.getValue(), FSR.getStart(),
430 I - FSR.getStart()))
431 return true;
432 }
433 assert(I == E && "Format string not exhausted");
434 return false;
435}

437bool clang::analyze_format_string::ParseFormatStringHasSArg(const char *I,
438 const char *E,
439 const LangOptions &LO,
440 const TargetInfo &Target) {
441
442 unsigned argIndex = 0;
443
444 // Keep looking for a %s format specifier until we have exhausted the string.
445 FormatStringHandler H;
446 while (I != E) {
447 const PrintfSpecifierResult &FSR = ParsePrintfSpecifier(H, I, E, argIndex,
448 LO, Target, false,
449 false);
450 // Did a fail-stop error of any kind occur when parsing the specifier?
451 // If so, don't do any more processing.
452 if (FSR.shouldStop())
453 return false;
454 // Did we exhaust the string or encounter an error that
455 // we can recover from?
456 if (!FSR.hasValue())
457 continue;
458 const analyze_printf::PrintfSpecifier &FS = FSR.getValue();
459 // Return true if this a %s format specifier.
460 if (FS.getConversionSpecifier().getKind() == ConversionSpecifier::Kind::sArg)
461 return true;
462 }
463 return false;
464}

470ArgType PrintfSpecifier::getScalarArgType(ASTContext &Ctx,
471 bool IsObjCLiteral) const {
472 if (CS.getKind() == ConversionSpecifier::cArg)
473 switch (LM.getKind()) {
474 case LengthModifier::None:
475 return Ctx.IntTy;
476 case LengthModifier::AsLong:
477 case LengthModifier::AsWide:
478 return ArgType(ArgType::WIntTy, "wint_t");
479 case LengthModifier::AsShort:
480 if (Ctx.getTargetInfo().getTriple().isOSMSVCRT())
481 return Ctx.IntTy;
482 LLVM_FALLTHROUGH;
483 default:
484 return ArgType::Invalid();
485 }
486
487 if (CS.isIntArg())
488 switch (LM.getKind()) {
489 case LengthModifier::AsLongDouble:
490 // GNU extension.
491 return Ctx.LongLongTy;
492 case LengthModifier::None:
493 case LengthModifier::AsShortLong:
494 return Ctx.IntTy;
495 case LengthModifier::AsInt32:
496 return ArgType(Ctx.IntTy, "__int32");
497 case LengthModifier::AsChar:
498 return ArgType::AnyCharTy;
499 case LengthModifier::AsShort: return Ctx.ShortTy;
500 case LengthModifier::AsLong: return Ctx.LongTy;
501 case LengthModifier::AsLongLong:
502 case LengthModifier::AsQuad:
503 return Ctx.LongLongTy;
504 case LengthModifier::AsInt64:
505 return ArgType(Ctx.LongLongTy, "__int64");
506 case LengthModifier::AsIntMax:
507 return ArgType(Ctx.getIntMaxType(), "intmax_t");
508 case LengthModifier::AsIntPtr:
509 return ArgType(Ctx.getIntPtrType(), "intptr_t");
510 case LengthModifier::AsSizeT:
511 return ArgType::makeSizeT(ArgType(Ctx.getSignedSizeType(), "ssize_t"));
512 case LengthModifier::AsInt3264:
513 return Ctx.getTargetInfo().getTriple().isArch64Bit()
514 ? ArgType(Ctx.LongLongTy, "__int64")
515 : ArgType(Ctx.IntTy, "__int32");
516 case LengthModifier::AsPtrDiff:
517 return ArgType::makePtrdiffT(
518 ArgType(Ctx.getPointerDiffType(), "ptrdiff_t"));
519 case LengthModifier::AsAllocate:
520 case LengthModifier::AsMAllocate:
521 case LengthModifier::AsWide:
522 return ArgType::Invalid();
523 }
524
525 if (CS.isUIntArg())
526 switch (LM.getKind()) {
527 case LengthModifier::AsLongDouble:
528 // GNU extension.
529 return Ctx.UnsignedLongLongTy;
530 case LengthModifier::None:
531 case LengthModifier::AsShortLong:
532 return Ctx.UnsignedIntTy;
533 case LengthModifier::AsInt32:
534 return ArgType(Ctx.UnsignedIntTy, "unsigned __int32");
535 case LengthModifier::AsChar: return Ctx.UnsignedCharTy;
536 case LengthModifier::AsShort: return Ctx.UnsignedShortTy;
537 case LengthModifier::AsLong: return Ctx.UnsignedLongTy;
538 case LengthModifier::AsLongLong:
539 case LengthModifier::AsQuad:
540 return Ctx.UnsignedLongLongTy;
541 case LengthModifier::AsInt64:
542 return ArgType(Ctx.UnsignedLongLongTy, "unsigned __int64");
543 case LengthModifier::AsIntMax:
544 return ArgType(Ctx.getUIntMaxType(), "uintmax_t");
545 case LengthModifier::AsIntPtr:
546 return ArgType(Ctx.getUIntPtrType(), "uintptr_t");
547 case LengthModifier::AsSizeT:
548 return ArgType::makeSizeT(ArgType(Ctx.getSizeType(), "size_t"));
549 case LengthModifier::AsInt3264:
550 return Ctx.getTargetInfo().getTriple().isArch64Bit()
551 ? ArgType(Ctx.UnsignedLongLongTy, "unsigned __int64")
552 : ArgType(Ctx.UnsignedIntTy, "unsigned __int32");
553 case LengthModifier::AsPtrDiff:
554 return ArgType::makePtrdiffT(
555 ArgType(Ctx.getUnsignedPointerDiffType(), "unsigned ptrdiff_t"));
556 case LengthModifier::AsAllocate:
557 case LengthModifier::AsMAllocate:
558 case LengthModifier::AsWide:
559 return ArgType::Invalid();
560 }
561
562 if (CS.isDoubleArg()) {
563 if (!VectorNumElts.isInvalid()) {
564 switch (LM.getKind()) {
565 case LengthModifier::AsShort:
566 return Ctx.HalfTy;
567 case LengthModifier::AsShortLong:
568 return Ctx.FloatTy;
569 case LengthModifier::AsLong:
570 default:
571 return Ctx.DoubleTy;
572 }
573 }
574
575 if (LM.getKind() == LengthModifier::AsLongDouble)
576 return Ctx.LongDoubleTy;
577 return Ctx.DoubleTy;
578 }
579
580 if (CS.getKind() == ConversionSpecifier::nArg) {
581 switch (LM.getKind()) {
582 case LengthModifier::None:
583 return ArgType::PtrTo(Ctx.IntTy);
584 case LengthModifier::AsChar:
585 return ArgType::PtrTo(Ctx.SignedCharTy);
586 case LengthModifier::AsShort:
587 return ArgType::PtrTo(Ctx.ShortTy);
588 case LengthModifier::AsLong:
589 return ArgType::PtrTo(Ctx.LongTy);
590 case LengthModifier::AsLongLong:
591 case LengthModifier::AsQuad:
592 return ArgType::PtrTo(Ctx.LongLongTy);
593 case LengthModifier::AsIntMax:
594 return ArgType::PtrTo(ArgType(Ctx.getIntMaxType(), "intmax_t"));
595 case LengthModifier::AsIntPtr:
596 return ArgType::PtrTo(ArgType(Ctx.getIntPtrType(), "intptr_t"));
597 case LengthModifier::AsSizeT:
598 return ArgType::PtrTo(ArgType(Ctx.getSignedSizeType(), "ssize_t"));
599 case LengthModifier::AsPtrDiff:
600 return ArgType::PtrTo(ArgType(Ctx.getPointerDiffType(), "ptrdiff_t"));
601 case LengthModifier::AsLongDouble:
602 return ArgType(); // FIXME: Is this a known extension?
603 case LengthModifier::AsAllocate:
604 case LengthModifier::AsMAllocate:
605 case LengthModifier::AsInt32:
606 case LengthModifier::AsInt3264:
607 case LengthModifier::AsInt64:
608 case LengthModifier::AsWide:
609 return ArgType::Invalid();
610 case LengthModifier::AsShortLong:
611 llvm_unreachable("only used for OpenCL which doesn not handle nArg");
612 }
613 }
614
615 switch (CS.getKind()) {
616 case ConversionSpecifier::sArg:
617 if (LM.getKind() == LengthModifier::AsWideChar) {
618 if (IsObjCLiteral)
619 return ArgType(Ctx.getPointerType(Ctx.UnsignedShortTy.withConst()),
620 "const unichar *");
621 return ArgType(ArgType::WCStrTy, "wchar_t *");
622 }
623 if (LM.getKind() == LengthModifier::AsWide)
624 return ArgType(ArgType::WCStrTy, "wchar_t *");
625 return ArgType::CStrTy;
626 case ConversionSpecifier::SArg:
627 if (IsObjCLiteral)
628 return ArgType(Ctx.getPointerType(Ctx.UnsignedShortTy.withConst()),
629 "const unichar *");
630 if (Ctx.getTargetInfo().getTriple().isOSMSVCRT() &&
631 LM.getKind() == LengthModifier::AsShort)
632 return ArgType::CStrTy;
633 return ArgType(ArgType::WCStrTy, "wchar_t *");
634 case ConversionSpecifier::CArg:
635 if (IsObjCLiteral)
636 return ArgType(Ctx.UnsignedShortTy, "unichar");
637 if (Ctx.getTargetInfo().getTriple().isOSMSVCRT() &&
638 LM.getKind() == LengthModifier::AsShort)
639 return Ctx.IntTy;
640 return ArgType(Ctx.WideCharTy, "wchar_t");
641 case ConversionSpecifier::pArg:
642 case ConversionSpecifier::PArg:
643 return ArgType::CPointerTy;
644 case ConversionSpecifier::ObjCObjArg:
645 return ArgType::ObjCPointerTy;
646 default:
647 break;
648 }
649
650 // FIXME: Handle other cases.
651 return ArgType();
652}

654
655ArgType PrintfSpecifier::getArgType(ASTContext &Ctx,
656 bool IsObjCLiteral) const {
657 const PrintfConversionSpecifier &CS = getConversionSpecifier();
658
659 if (!CS.consumesDataArgument())
660 return ArgType::Invalid();
661
662 ArgType ScalarTy = getScalarArgType(Ctx, IsObjCLiteral);
663 if (!ScalarTy.isValid() || VectorNumElts.isInvalid())
664 return ScalarTy;
665
666 return ScalarTy.makeVectorType(Ctx, VectorNumElts.getConstantAmount());
667}

669bool PrintfSpecifier::fixType(QualType QT, const LangOptions &LangOpt,
670 ASTContext &Ctx, bool IsObjCLiteral) {
671 // %n is different from other conversion specifiers; don't try to fix it.
672 if (CS.getKind() == ConversionSpecifier::nArg)
673 return false;
674
675 // Handle Objective-C objects first. Note that while the '%@' specifier will
676 // not warn for structure pointer or void pointer arguments (because that's
677 // how CoreFoundation objects are implemented), we only show a fixit for '%@'
678 // if we know it's an object (block, id, class, or __attribute__((NSObject))).
679 if (QT->isObjCRetainableType()) {
680 if (!IsObjCLiteral)
681 return false;
682
683 CS.setKind(ConversionSpecifier::ObjCObjArg);
684
685 // Disable irrelevant flags
686 HasThousandsGrouping = false;
687 HasPlusPrefix = false;
688 HasSpacePrefix = false;
689 HasAlternativeForm = false;
690 HasLeadingZeroes = false;
691 Precision.setHowSpecified(OptionalAmount::NotSpecified);
692 LM.setKind(LengthModifier::None);
693
694 return true;
695 }
696
697 // Handle strings next (char *, wchar_t *)
698 if (QT->isPointerType() && (QT->getPointeeType()->isAnyCharacterType())) {
699 CS.setKind(ConversionSpecifier::sArg);
700
701 // Disable irrelevant flags
702 HasAlternativeForm = 0;
703 HasLeadingZeroes = 0;
704
705 // Set the long length modifier for wide characters
706 if (QT->getPointeeType()->isWideCharType())
707 LM.setKind(LengthModifier::AsWideChar);
708 else
709 LM.setKind(LengthModifier::None);
710
711 return true;
712 }
713
714 // If it's an enum, get its underlying type.
715 if (const EnumType *ETy = QT->getAs<EnumType>())
716 QT = ETy->getDecl()->getIntegerType();
717
718 const BuiltinType *BT = QT->getAs<BuiltinType>();
719 if (!BT) {
720 const VectorType *VT = QT->getAs<VectorType>();
721 if (VT) {
722 QT = VT->getElementType();
723 BT = QT->getAs<BuiltinType>();
724 VectorNumElts = OptionalAmount(VT->getNumElements());
725 }
726 }
727
728 // We can only work with builtin types.
729 if (!BT)
730 return false;
731
732 // Set length modifier
733 switch (BT->getKind()) {
734 case BuiltinType::Bool:
735 case BuiltinType::WChar_U:
736 case BuiltinType::WChar_S:
737 case BuiltinType::Char8: // FIXME: Treat like 'char'?
738 case BuiltinType::Char16:
739 case BuiltinType::Char32:
740 case BuiltinType::UIntCap:
741 case BuiltinType::IntCap:
742 case BuiltinType::UInt128:
743 case BuiltinType::Int128:
744 case BuiltinType::Half:
745 case BuiltinType::Float16:
746 case BuiltinType::Float128:
747 case BuiltinType::ShortAccum:
748 case BuiltinType::Accum:
749 case BuiltinType::LongAccum:
750 case BuiltinType::UShortAccum:
751 case BuiltinType::UAccum:
752 case BuiltinType::ULongAccum:
753 case BuiltinType::ShortFract:
754 case BuiltinType::Fract:
755 case BuiltinType::LongFract:
756 case BuiltinType::UShortFract:
757 case BuiltinType::UFract:
758 case BuiltinType::ULongFract:
759 case BuiltinType::SatShortAccum:
760 case BuiltinType::SatAccum:
761 case BuiltinType::SatLongAccum:
762 case BuiltinType::SatUShortAccum:
763 case BuiltinType::SatUAccum:
764 case BuiltinType::SatULongAccum:
765 case BuiltinType::SatShortFract:
766 case BuiltinType::SatFract:
767 case BuiltinType::SatLongFract:
768 case BuiltinType::SatUShortFract:
769 case BuiltinType::SatUFract:
770 case BuiltinType::SatULongFract:
771 // Various types which are non-trivial to correct.
772 return false;
773
774#define IMAGE_TYPE(ImgType, Id, SingletonId, Access, Suffix) \
775 case BuiltinType::Id:
776#include "clang/Basic/OpenCLImageTypes.def"
777#define EXT_OPAQUE_TYPE(ExtType, Id, Ext) \
778 case BuiltinType::Id:
779#include "clang/Basic/OpenCLExtensionTypes.def"
780#define SIGNED_TYPE(Id, SingletonId)
781#define UNSIGNED_TYPE(Id, SingletonId)
782#define FLOATING_TYPE(Id, SingletonId)
783#define BUILTIN_TYPE(Id, SingletonId) \
784 case BuiltinType::Id:
785#include "clang/AST/BuiltinTypes.def"
786 // Misc other stuff which doesn't make sense here.
787 return false;
788
789 case BuiltinType::UInt:
790 case BuiltinType::Int:
791 case BuiltinType::Float:
792 LM.setKind(VectorNumElts.isInvalid() ?
793 LengthModifier::None : LengthModifier::AsShortLong);
794 break;
795 case BuiltinType::Double:
796 LM.setKind(VectorNumElts.isInvalid() ?
797 LengthModifier::None : LengthModifier::AsLong);
798 break;
799 case BuiltinType::Char_U:
800 case BuiltinType::UChar:
801 case BuiltinType::Char_S:
802 case BuiltinType::SChar:
803 LM.setKind(LengthModifier::AsChar);
804 break;
805
806 case BuiltinType::Short:
807 case BuiltinType::UShort:
808 LM.setKind(LengthModifier::AsShort);
809 break;
810
811 case BuiltinType::Long:
812 case BuiltinType::ULong:
813 LM.setKind(LengthModifier::AsLong);
814 break;
815
816 case BuiltinType::LongLong:
817 case BuiltinType::ULongLong:
818 LM.setKind(LengthModifier::AsLongLong);
819 break;
820
821 case BuiltinType::LongDouble:
822 LM.setKind(LengthModifier::AsLongDouble);
823 break;
824 }
825
826 // Handle size_t, ptrdiff_t, etc. that have dedicated length modifiers in C99.
827 if (isa<TypedefType>(QT) && (LangOpt.C99 || LangOpt.CPlusPlus11))
828 namedTypeToLengthModifier(QT, LM);
829
830 // If fixing the length modifier was enough, we might be done.
831 if (hasValidLengthModifier(Ctx.getTargetInfo(), LangOpt)) {
832 // If we're going to offer a fix anyway, make sure the sign matches.
833 switch (CS.getKind()) {
834 case ConversionSpecifier::uArg:
835 case ConversionSpecifier::UArg:
836 if (QT->isSignedIntegerType())
837 CS.setKind(clang::analyze_format_string::ConversionSpecifier::dArg);
838 break;
839 case ConversionSpecifier::dArg:
840 case ConversionSpecifier::DArg:
841 case ConversionSpecifier::iArg:
842 if (QT->isUnsignedIntegerType() && !HasPlusPrefix)
843 CS.setKind(clang::analyze_format_string::ConversionSpecifier::uArg);
844 break;
845 default:
846 // Other specifiers do not have signed/unsigned variants.
847 break;
848 }
849
850 const analyze_printf::ArgType &ATR = getArgType(Ctx, IsObjCLiteral);
851 if (ATR.isValid() && ATR.matchesType(Ctx, QT))
852 return true;
853 }
854
855 // Set conversion specifier and disable any flags which do not apply to it.
856 // Let typedefs to char fall through to int, as %c is silly for uint8_t.
857 if (!isa<TypedefType>(QT) && QT->isCharType()) {
858 CS.setKind(ConversionSpecifier::cArg);
859 LM.setKind(LengthModifier::None);
860 Precision.setHowSpecified(OptionalAmount::NotSpecified);
861 HasAlternativeForm = 0;
862 HasLeadingZeroes = 0;
863 HasPlusPrefix = 0;
864 }
865 // Test for Floating type first as LongDouble can pass isUnsignedIntegerType
866 else if (QT->isRealFloatingType()) {
867 CS.setKind(ConversionSpecifier::fArg);
868 }
869 else if (QT->isSignedIntegerType()) {
870 CS.setKind(ConversionSpecifier::dArg);
871 HasAlternativeForm = 0;
872 }
873 else if (QT->isUnsignedIntegerType()) {
874 CS.setKind(ConversionSpecifier::uArg);
875 HasAlternativeForm = 0;
876 HasPlusPrefix = 0;
877 } else {
878 llvm_unreachable("Unexpected type");
879 }
880
881 return true;
882}

884void PrintfSpecifier::toString(raw_ostream &os) const {
885 // Whilst some features have no defined order, we are using the order
886 // appearing in the C99 standard (ISO/IEC 9899:1999 (E) 7.19.6.1)
887 os << "%";
888
889 // Positional args
890 if (usesPositionalArg()) {
891 os << getPositionalArgIndex() << "$";
892 }
893
894 // Conversion flags
895 if (IsLeftJustified) os << "-";
896 if (HasPlusPrefix) os << "+";
897 if (HasSpacePrefix) os << " ";
898 if (HasAlternativeForm) os << "#";
899 if (HasLeadingZeroes) os << "0";
900
901 // Minimum field width
902 FieldWidth.toString(os);
903 // Precision
904 Precision.toString(os);
905
906 // Vector modifier
907 if (!VectorNumElts.isInvalid())
908 os << 'v' << VectorNumElts.getConstantAmount();
909
910 // Length modifier
911 os << LM.toString();
912 // Conversion specifier
913 os << CS.toString();
914}
