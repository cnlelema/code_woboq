26// Returns a desugared version of the QualType, and marks ShouldAKA as true
27// whenever we remove significant sugar from the type.
28static QualType Desugar(ASTContext &Context, QualType QT, bool &ShouldAKA) {
29 QualifierCollector QC;
30
31 while (true) {
32 const Type *Ty = QC.strip(QT);
33
34 // Don't aka just because we saw an elaborated type...
35 if (const ElaboratedType *ET = dyn_cast<ElaboratedType>(Ty)) {
36 QT = ET->desugar();
37 continue;
38 }
39 // ... or a paren type ...
40 if (const ParenType *PT = dyn_cast<ParenType>(Ty)) {
41 QT = PT->desugar();
42 continue;
43 }
44 // ... or a macro defined type ...
45 if (const MacroQualifiedType *MDT = dyn_cast<MacroQualifiedType>(Ty)) {
46 QT = MDT->desugar();
47 continue;
48 }
49 // ...or a substituted template type parameter ...
50 if (const SubstTemplateTypeParmType *ST =
51 dyn_cast<SubstTemplateTypeParmType>(Ty)) {
52 QT = ST->desugar();
53 continue;
54 }
55 // ...or an attributed type...
56 if (const AttributedType *AT = dyn_cast<AttributedType>(Ty)) {
57 QT = AT->desugar();
58 continue;
59 }
60 // ...or an adjusted type...
61 if (const AdjustedType *AT = dyn_cast<AdjustedType>(Ty)) {
62 QT = AT->desugar();
63 continue;
64 }
65 // ... or an auto type.
66 if (const AutoType *AT = dyn_cast<AutoType>(Ty)) {
67 if (!AT->isSugared())
68 break;
69 QT = AT->desugar();
70 continue;
71 }
72
73 // Desugar FunctionType if return type or any parameter type should be
74 // desugared. Preserve nullability attribute on desugared types.
75 if (const FunctionType *FT = dyn_cast<FunctionType>(Ty)) {
76 bool DesugarReturn = false;
77 QualType SugarRT = FT->getReturnType();
78 QualType RT = Desugar(Context, SugarRT, DesugarReturn);
79 if (auto nullability = AttributedType::stripOuterNullability(SugarRT)) {
80 RT = Context.getAttributedType(
81 AttributedType::getNullabilityAttrKind(*nullability), RT, RT);
82 }
83
84 bool DesugarArgument = false;
85 SmallVector<QualType, 4> Args;
86 const FunctionProtoType *FPT = dyn_cast<FunctionProtoType>(FT);
87 if (FPT) {
88 for (QualType SugarPT : FPT->param_types()) {
89 QualType PT = Desugar(Context, SugarPT, DesugarArgument);
90 if (auto nullability =
91 AttributedType::stripOuterNullability(SugarPT)) {
92 PT = Context.getAttributedType(
93 AttributedType::getNullabilityAttrKind(*nullability), PT, PT);
94 }
95 Args.push_back(PT);
96 }
97 }
98
99 if (DesugarReturn || DesugarArgument) {
100 ShouldAKA = true;
101 QT = FPT ? Context.getFunctionType(RT, Args, FPT->getExtProtoInfo())
102 : Context.getFunctionNoProtoType(RT, FT->getExtInfo());
103 break;
104 }
105 }
106
107 // Desugar template specializations if any template argument should be
108 // desugared.
109 if (const TemplateSpecializationType *TST =
110 dyn_cast<TemplateSpecializationType>(Ty)) {
111 if (!TST->isTypeAlias()) {
112 bool DesugarArgument = false;
113 SmallVector<TemplateArgument, 4> Args;
114 for (unsigned I = 0, N = TST->getNumArgs(); I != N; ++I) {
115 const TemplateArgument &Arg = TST->getArg(I);
116 if (Arg.getKind() == TemplateArgument::Type)
117 Args.push_back(Desugar(Context, Arg.getAsType(), DesugarArgument));
118 else
119 Args.push_back(Arg);
120 }
121
122 if (DesugarArgument) {
123 ShouldAKA = true;
124 QT = Context.getTemplateSpecializationType(
125 TST->getTemplateName(), Args, QT);
126 }
127 break;
128 }
129 }
130
131 // Don't desugar magic Objective-C types.
132 if (QualType(Ty,0) == Context.getObjCIdType() ||
133 QualType(Ty,0) == Context.getObjCClassType() ||
134 QualType(Ty,0) == Context.getObjCSelType() ||
135 QualType(Ty,0) == Context.getObjCProtoType())
136 break;
137
138 // Don't desugar va_list.
139 if (QualType(Ty, 0) == Context.getBuiltinVaListType() ||
140 QualType(Ty, 0) == Context.getBuiltinMSVaListType())
141 break;
142
143 // Otherwise, do a single-step desugar.
144 QualType Underlying;
145 bool IsSugar = false;
146 switch (Ty->getTypeClass()) {
147#define ABSTRACT_TYPE(Class, Base)
148#define TYPE(Class, Base) \
149case Type::Class: { \
150const Class##Type *CTy = cast<Class##Type>(Ty); \
151if (CTy->isSugared()) { \
152IsSugar = true; \
153Underlying = CTy->desugar(); \
154} \
155break; \
156}
157#include "clang/AST/TypeNodes.def"
158 }
159
160 // If it wasn't sugared, we're done.
161 if (!IsSugar)
162 break;
163
164 // If the desugared type is a vector type, we don't want to expand
165 // it, it will turn into an attribute mess. People want their "vec4".
166 if (isa<VectorType>(Underlying))
167 break;
168
169 // Don't desugar through the primary typedef of an anonymous type.
170 if (const TagType *UTT = Underlying->getAs<TagType>())
171 if (const TypedefType *QTT = dyn_cast<TypedefType>(QT))
172 if (UTT->getDecl()->getTypedefNameForAnonDecl() == QTT->getDecl())
173 break;
174
175 // Record that we actually looked through an opaque type here.
176 ShouldAKA = true;
177 QT = Underlying;
178 }
179
180 // If we have a pointer-like type, desugar the pointee as well.
181 // FIXME: Handle other pointer-like types.
182 if (const PointerType *Ty = QT->getAs<PointerType>()) {
183 ASTContext::PointerInterpretationKind PIK = Ty->isCHERICapability() ?
184 ASTContext::PIK_Capability : ASTContext::PIK_Integer;
185 QT = Context.getPointerType(Desugar(Context, Ty->getPointeeType(),
186 ShouldAKA), PIK);
187 } else if (const auto *Ty = QT->getAs<ObjCObjectPointerType>()) {
188 QT = Context.getObjCObjectPointerType(Desugar(Context, Ty->getPointeeType(),
189 ShouldAKA));
190 } else if (const LValueReferenceType *Ty = QT->getAs<LValueReferenceType>()) {
191 ASTContext::PointerInterpretationKind PIK =
192 Ty->isCHERICapability() ? ASTContext::PIK_Capability : ASTContext::PIK_Integer;
193 QT = Context.getLValueReferenceType(Desugar(Context, Ty->getPointeeType(),
194 ShouldAKA), true, PIK);
195 } else if (const RValueReferenceType *Ty = QT->getAs<RValueReferenceType>()) {
196 ASTContext::PointerInterpretationKind PIK =
197 Ty->isCHERICapability() ? ASTContext::PIK_Capability : ASTContext::PIK_Integer;
198 QT = Context.getRValueReferenceType(Desugar(Context, Ty->getPointeeType(),
199 ShouldAKA), PIK);
200 } else if (const auto *Ty = QT->getAs<ObjCObjectType>()) {
201 if (Ty->getBaseType().getTypePtr() != Ty && !ShouldAKA) {
202 QualType BaseType = Desugar(Context, Ty->getBaseType(), ShouldAKA);
203 QT = Context.getObjCObjectType(BaseType, Ty->getTypeArgsAsWritten(),
204 llvm::makeArrayRef(Ty->qual_begin(),
205 Ty->getNumProtocols()),
206 Ty->isKindOfTypeAsWritten());
207 }
208 }
209
210 return QC.apply(Context, QT);
211}

213/// Convert the given type to a string suitable for printing as part of
214/// a diagnostic.
215///
216/// There are four main criteria when determining whether we should have an
217/// a.k.a. clause when pretty-printing a type:
218///
219/// 1) Some types provide very minimal sugar that doesn't impede the
220/// user's understanding --- for example, elaborated type
221/// specifiers. If this is all the sugar we see, we don't want an
222/// a.k.a. clause.
223/// 2) Some types are technically sugared but are much more familiar
224/// when seen in their sugared form --- for example, va_list,
225/// vector types, and the magic Objective C types. We don't
226/// want to desugar these, even if we do produce an a.k.a. clause.
227/// 3) Some types may have already been desugared previously in this diagnostic.
228/// if this is the case, doing another "aka" would just be clutter.
229/// 4) Two different types within the same diagnostic have the same output
230/// string. In this case, force an a.k.a with the desugared type when
231/// doing so will provide additional information.
232///
233/// \param Context the context in which the type was allocated
234/// \param Ty the type to print
235/// \param QualTypeVals pointer values to QualTypes which are used in the
236/// diagnostic message
237static std::string
238ConvertTypeToDiagnosticString(ASTContext &Context, QualType Ty,
239 ArrayRef<DiagnosticsEngine::ArgumentValue> PrevArgs,
240 ArrayRef<intptr_t> QualTypeVals) {
241 // FIXME: Playing with std::string is really slow.
242 bool ForceAKA = false;
243 QualType CanTy = Ty.getCanonicalType();
244 std::string S = Ty.getAsString(Context.getPrintingPolicy());
245 std::string CanS = CanTy.getAsString(Context.getPrintingPolicy());
246
247 for (unsigned I = 0, E = QualTypeVals.size(); I != E; ++I) {
248 QualType CompareTy =
249 QualType::getFromOpaquePtr(reinterpret_cast<void*>(QualTypeVals[I]));
250 if (CompareTy.isNull())
251 continue;
252 if (CompareTy == Ty)
253 continue; // Same types
254 QualType CompareCanTy = CompareTy.getCanonicalType();
255 if (CompareCanTy == CanTy)
256 continue; // Same canonical types
257 std::string CompareS = CompareTy.getAsString(Context.getPrintingPolicy());
258 bool ShouldAKA = false;
259 QualType CompareDesugar = Desugar(Context, CompareTy, ShouldAKA);
260 std::string CompareDesugarStr =
261 CompareDesugar.getAsString(Context.getPrintingPolicy());
262 if (CompareS != S && CompareDesugarStr != S)
263 continue; // The type string is different than the comparison string
264 // and the desugared comparison string.
265 std::string CompareCanS =
266 CompareCanTy.getAsString(Context.getPrintingPolicy());
267
268 if (CompareCanS == CanS)
269 continue; // No new info from canonical type
270
271 ForceAKA = true;
272 break;
273 }
274
275 // Check to see if we already desugared this type in this
276 // diagnostic. If so, don't do it again.
277 bool Repeated = false;
278 for (unsigned i = 0, e = PrevArgs.size(); i != e; ++i) {
279 // TODO: Handle ak_declcontext case.
280 if (PrevArgs[i].first == DiagnosticsEngine::ak_qualtype) {
281 void *Ptr = (void*)PrevArgs[i].second;
282 QualType PrevTy(QualType::getFromOpaquePtr(Ptr));
283 if (PrevTy == Ty) {
284 Repeated = true;
285 break;
286 }
287 }
288 }
289
290 // Consider producing an a.k.a. clause if removing all the direct
291 // sugar gives us something "significantly different".
292 if (!Repeated) {
293 bool ShouldAKA = false;
294 QualType DesugaredTy = Desugar(Context, Ty, ShouldAKA);
295 if (ShouldAKA || ForceAKA) {
296 if (DesugaredTy == Ty) {
297 DesugaredTy = Ty.getCanonicalType();
298 }
299 std::string akaStr = DesugaredTy.getAsString(Context.getPrintingPolicy());
300 if (akaStr != S) {
301 S = "'" + S + "' (aka '" + akaStr + "')";
302 return S;
303 }
304 }
305
306 // Give some additional info on vector types. These are either not desugared
307 // or displaying complex __attribute__ expressions so add details of the
308 // type and element count.
309 if (Ty->isVectorType()) {
310 const VectorType *VTy = Ty->getAs<VectorType>();
311 std::string DecoratedString;
312 llvm::raw_string_ostream OS(DecoratedString);
313 const char *Values = VTy->getNumElements() > 1 ? "values" : "value";
314 OS << "'" << S << "' (vector of " << VTy->getNumElements() << " '"
315 << VTy->getElementType().getAsString(Context.getPrintingPolicy())
316 << "' " << Values << ")";
317 return OS.str();
318 }
319 }
320
321 S = "'" + S + "'";
322 return S;
323}

330void clang::FormatASTNodeDiagnosticArgument(
331 DiagnosticsEngine::ArgumentKind Kind,
332 intptr_t Val,
333 StringRef Modifier,
334 StringRef Argument,
335 ArrayRef<DiagnosticsEngine::ArgumentValue> PrevArgs,
336 SmallVectorImpl<char> &Output,
337 void *Cookie,
338 ArrayRef<intptr_t> QualTypeVals) {
339 ASTContext &Context = *static_cast<ASTContext*>(Cookie);
340
341 size_t OldEnd = Output.size();
342 llvm::raw_svector_ostream OS(Output);
343 bool NeedQuotes = true;
344
345 switch (Kind) {
346 default: llvm_unreachable("unknown ArgumentKind");
347 case DiagnosticsEngine::ak_qual: {
348 assert(Modifier.empty() && Argument.empty() &&
349 "Invalid modifier for Qualfiers argument");
350
351 Qualifiers Q(Qualifiers::fromOpaqueValue(Val));
352 auto S = Q.getAsString();
353 if (S.empty()) {
354 OS << "unqualified";
355 NeedQuotes = false;
356 } else {
357 OS << Q.getAsString();
358 }
359 break;
360 }
361 case DiagnosticsEngine::ak_qualtype_pair: {
362 TemplateDiffTypes &TDT = *reinterpret_cast<TemplateDiffTypes*>(Val);
363 QualType FromType =
364 QualType::getFromOpaquePtr(reinterpret_cast<void*>(TDT.FromType));
365 QualType ToType =
366 QualType::getFromOpaquePtr(reinterpret_cast<void*>(TDT.ToType));
367
368 if (FormatTemplateTypeDiff(Context, FromType, ToType, TDT.PrintTree,
369 TDT.PrintFromType, TDT.ElideType,
370 TDT.ShowColors, OS)) {
371 NeedQuotes = !TDT.PrintTree;
372 TDT.TemplateDiffUsed = true;
373 break;
374 }
375
376 // Don't fall-back during tree printing. The caller will handle
377 // this case.
378 if (TDT.PrintTree)
379 return;
380
381 // Attempting to do a template diff on non-templates. Set the variables
382 // and continue with regular type printing of the appropriate type.
383 Val = TDT.PrintFromType ? TDT.FromType : TDT.ToType;
384 Modifier = StringRef();
385 Argument = StringRef();
386 // Fall through
387 LLVM_FALLTHROUGH;
388 }
389 case DiagnosticsEngine::ak_qualtype: {
390 assert(Modifier.empty() && Argument.empty() &&
391 "Invalid modifier for QualType argument");
392
393 QualType Ty(QualType::getFromOpaquePtr(reinterpret_cast<void*>(Val)));
394 OS << ConvertTypeToDiagnosticString(Context, Ty, PrevArgs, QualTypeVals);
395 NeedQuotes = false;
396 break;
397 }
398 case DiagnosticsEngine::ak_declarationname: {
399 if (Modifier == "objcclass" && Argument.empty())
400 OS << '+';
401 else if (Modifier == "objcinstance" && Argument.empty())
402 OS << '-';
403 else
404 assert(Modifier.empty() && Argument.empty() &&
405 "Invalid modifier for DeclarationName argument");
406
407 OS << DeclarationName::getFromOpaqueInteger(Val);
408 break;
409 }
410 case DiagnosticsEngine::ak_nameddecl: {
411 bool Qualified;
412 if (Modifier == "q" && Argument.empty())
413 Qualified = true;
414 else {
415 assert(Modifier.empty() && Argument.empty() &&
416 "Invalid modifier for NamedDecl* argument");
417 Qualified = false;
418 }
419 const NamedDecl *ND = reinterpret_cast<const NamedDecl*>(Val);
420 ND->getNameForDiagnostic(OS, Context.getPrintingPolicy(), Qualified);
421 break;
422 }
423 case DiagnosticsEngine::ak_nestednamespec: {
424 NestedNameSpecifier *NNS = reinterpret_cast<NestedNameSpecifier*>(Val);
425 NNS->print(OS, Context.getPrintingPolicy());
426 NeedQuotes = false;
427 break;
428 }
429 case DiagnosticsEngine::ak_declcontext: {
430 DeclContext *DC = reinterpret_cast<DeclContext *> (Val);
431 assert(DC && "Should never have a null declaration context");
432 NeedQuotes = false;
433
434 // FIXME: Get the strings for DeclContext from some localized place
435 if (DC->isTranslationUnit()) {
436 if (Context.getLangOpts().CPlusPlus)
437 OS << "the global namespace";
438 else
439 OS << "the global scope";
440 } else if (DC->isClosure()) {
441 OS << "block literal";
442 } else if (isLambdaCallOperator(DC)) {
443 OS << "lambda expression";
444 } else if (TypeDecl *Type = dyn_cast<TypeDecl>(DC)) {
445 OS << ConvertTypeToDiagnosticString(Context,
446 Context.getTypeDeclType(Type),
447 PrevArgs, QualTypeVals);
448 } else {
449 assert(isa<NamedDecl>(DC) && "Expected a NamedDecl");
450 NamedDecl *ND = cast<NamedDecl>(DC);
451 if (isa<NamespaceDecl>(ND))
452 OS << "namespace ";
453 else if (isa<ObjCMethodDecl>(ND))
454 OS << "method ";
455 else if (isa<FunctionDecl>(ND))
456 OS << "function ";
457
458 OS << '\'';
459 ND->getNameForDiagnostic(OS, Context.getPrintingPolicy(), true);
460 OS << '\'';
461 }
462 break;
463 }
464 case DiagnosticsEngine::ak_attr: {
465 const Attr *At = reinterpret_cast<Attr *>(Val);
466 assert(At && "Received null Attr object!");
467 OS << '\'' << At->getSpelling() << '\'';
468 NeedQuotes = false;
469 break;
470 }
471 }
472
473 if (NeedQuotes) {
474 Output.insert(Output.begin()+OldEnd, '\'');
475 Output.push_back('\'');
476 }
477}

479/// TemplateDiff - A class that constructs a pretty string for a pair of
480/// QualTypes. For the pair of types, a diff tree will be created containing
481/// all the information about the templates and template arguments. Afterwards,
482/// the tree is transformed to a string according to the options passed in.
483namespace {
484class TemplateDiff {
485 /// Context - The ASTContext which is used for comparing template arguments.
486 ASTContext &Context;
487
488 /// Policy - Used during expression printing.
489 PrintingPolicy Policy;
490
491 /// ElideType - Option to elide identical types.
492 bool ElideType;
493
494 /// PrintTree - Format output string as a tree.
495 bool PrintTree;
496
497 /// ShowColor - Diagnostics support color, so bolding will be used.
498 bool ShowColor;
499
500 /// FromTemplateType - When single type printing is selected, this is the
501 /// type to be be printed. When tree printing is selected, this type will
502 /// show up first in the tree.
503 QualType FromTemplateType;
504
505 /// ToTemplateType - The type that FromType is compared to. Only in tree
506 /// printing will this type be outputed.
507 QualType ToTemplateType;
508
509 /// OS - The stream used to construct the output strings.
510 raw_ostream &OS;
511
512 /// IsBold - Keeps track of the bold formatting for the output string.
513 bool IsBold;
514
515 /// DiffTree - A tree representation the differences between two types.
516 class DiffTree {
517 public:
518 /// DiffKind - The difference in a DiffNode. Fields of
519 /// TemplateArgumentInfo needed by each difference can be found in the
520 /// Set* and Get* functions.
521 enum DiffKind {
522 /// Incomplete or invalid node.
523 Invalid,
524 /// Another level of templates
525 Template,
526 /// Type difference, all type differences except those falling under
527 /// the Template difference.
528 Type,
529 /// Expression difference, this is only when both arguments are
530 /// expressions. If one argument is an expression and the other is
531 /// Integer or Declaration, then use that diff type instead.
532 Expression,
533 /// Template argument difference
534 TemplateTemplate,
535 /// Integer difference
536 Integer,
537 /// Declaration difference, nullptr arguments are included here
538 Declaration,
539 /// One argument being integer and the other being declaration
540 FromIntegerAndToDeclaration,
541 FromDeclarationAndToInteger
542 };
543
544 private:
545 /// TemplateArgumentInfo - All the information needed to pretty print
546 /// a template argument. See the Set* and Get* functions to see which
547 /// fields are used for each DiffKind.
548 struct TemplateArgumentInfo {
549 QualType ArgType;
550 Qualifiers Qual;
551 llvm::APSInt Val;
552 bool IsValidInt = false;
553 Expr *ArgExpr = nullptr;
554 TemplateDecl *TD = nullptr;
555 ValueDecl *VD = nullptr;
556 bool NeedAddressOf = false;
557 bool IsNullPtr = false;
558 bool IsDefault = false;
559 };
560
561 /// DiffNode - The root node stores the original type. Each child node
562 /// stores template arguments of their parents. For templated types, the
563 /// template decl is also stored.
564 struct DiffNode {
565 DiffKind Kind = Invalid;
566
567 /// NextNode - The index of the next sibling node or 0.
568 unsigned NextNode = 0;
569
570 /// ChildNode - The index of the first child node or 0.
571 unsigned ChildNode = 0;
572
573 /// ParentNode - The index of the parent node.
574 unsigned ParentNode = 0;
575
576 TemplateArgumentInfo FromArgInfo, ToArgInfo;
577
578 /// Same - Whether the two arguments evaluate to the same value.
579 bool Same = false;
580
581 DiffNode(unsigned ParentNode = 0) : ParentNode(ParentNode) {}
582 };
583
584 /// FlatTree - A flattened tree used to store the DiffNodes.
585 SmallVector<DiffNode, 16> FlatTree;
586
587 /// CurrentNode - The index of the current node being used.
588 unsigned CurrentNode;
589
590 /// NextFreeNode - The index of the next unused node. Used when creating
591 /// child nodes.
592 unsigned NextFreeNode;
593
594 /// ReadNode - The index of the current node being read.
595 unsigned ReadNode;
596
597 public:
598 DiffTree() :
599 CurrentNode(0), NextFreeNode(1) {
600 FlatTree.push_back(DiffNode());
601 }

603 // Node writing functions, one for each valid DiffKind element.
604 void SetTemplateDiff(TemplateDecl *FromTD, TemplateDecl *ToTD,
605 Qualifiers FromQual, Qualifiers ToQual,
606 bool FromDefault, bool ToDefault) {
607 assert(FlatTree[CurrentNode].Kind == Invalid && "Node is not empty.");
608 FlatTree[CurrentNode].Kind = Template;
609 FlatTree[CurrentNode].FromArgInfo.TD = FromTD;
610 FlatTree[CurrentNode].ToArgInfo.TD = ToTD;
611 FlatTree[CurrentNode].FromArgInfo.Qual = FromQual;
612 FlatTree[CurrentNode].ToArgInfo.Qual = ToQual;
613 SetDefault(FromDefault, ToDefault);
614 }

619 FlatTree[CurrentNode].Kind = Type;
620 FlatTree[CurrentNode].FromArgInfo.ArgType = FromType;
621 FlatTree[CurrentNode].ToArgInfo.ArgType = ToType;
622 SetDefault(FromDefault, ToDefault);

914 /// TSTiterator - a pair of iterators that walks the
915 /// TemplateSpecializationType and the desugared TemplateSpecializationType.
916 /// The deseguared TemplateArgument should provide the canonical argument
917 /// for comparisons.
918 class TSTiterator {
919 typedef const TemplateArgument& reference;
920 typedef const TemplateArgument* pointer;
921
922 /// InternalIterator - an iterator that is used to enter a
923 /// TemplateSpecializationType and read TemplateArguments inside template
924 /// parameter packs in order with the rest of the TemplateArguments.
925 struct InternalIterator {
926 /// TST - the template specialization whose arguments this iterator
927 /// traverse over.
928 const TemplateSpecializationType *TST;
929
930 /// Index - the index of the template argument in TST.
931 unsigned Index;
932
933 /// CurrentTA - if CurrentTA is not the same as EndTA, then CurrentTA
934 /// points to a TemplateArgument within a parameter pack.
935 TemplateArgument::pack_iterator CurrentTA;
936
937 /// EndTA - the end iterator of a parameter pack
938 TemplateArgument::pack_iterator EndTA;
939
940 /// InternalIterator - Constructs an iterator and sets it to the first
941 /// template argument.
942 InternalIterator(const TemplateSpecializationType *TST)
943 : TST(TST), Index(0), CurrentTA(nullptr), EndTA(nullptr) {
944 if (!TST) return;
945
946 if (isEnd()) return;
947
948 // Set to first template argument. If not a parameter pack, done.
949 TemplateArgument TA = TST->getArg(0);
950 if (TA.getKind() != TemplateArgument::Pack) return;
951
952 // Start looking into the parameter pack.
953 CurrentTA = TA.pack_begin();
954 EndTA = TA.pack_end();
955
956 // Found a valid template argument.
957 if (CurrentTA != EndTA) return;
958
959 // Parameter pack is empty, use the increment to get to a valid
960 // template argument.
961 ++(*this);
962 }
963
964 /// Return true if the iterator is non-singular.
965 bool isValid() const { return TST; }
966
967 /// isEnd - Returns true if the iterator is one past the end.
968 bool isEnd() const {
969 assert(TST && "InternalIterator is invalid with a null TST.");
970 return Index >= TST->getNumArgs();
971 }
972
973 /// &operator++ - Increment the iterator to the next template argument.
974 InternalIterator &operator++() {
975 assert(TST && "InternalIterator is invalid with a null TST.");
976 if (isEnd()) {
977 return *this;
978 }
979
980 // If in a parameter pack, advance in the parameter pack.
981 if (CurrentTA != EndTA) {
982 ++CurrentTA;
983 if (CurrentTA != EndTA)
984 return *this;
985 }
986
987 // Loop until a template argument is found, or the end is reached.
988 while (true) {
989 // Advance to the next template argument. Break if reached the end.
990 if (++Index == TST->getNumArgs())
991 break;
992
993 // If the TemplateArgument is not a parameter pack, done.
994 TemplateArgument TA = TST->getArg(Index);
995 if (TA.getKind() != TemplateArgument::Pack)
996 break;
997
998 // Handle parameter packs.
999 CurrentTA = TA.pack_begin();
1000 EndTA = TA.pack_end();
1001
1002 // If the parameter pack is empty, try to advance again.
1003 if (CurrentTA != EndTA)
1004 break;
1005 }
1006 return *this;
1007 }
1008
1009 /// operator* - Returns the appropriate TemplateArgument.
1010 reference operator*() const {
1011 assert(TST && "InternalIterator is invalid with a null TST.");
1012 assert(!isEnd() && "Index exceeds number of arguments.");
1013 if (CurrentTA == EndTA)
1014 return TST->getArg(Index);
1015 else
1016 return *CurrentTA;
1017 }
1018
1019 /// operator-> - Allow access to the underlying TemplateArgument.
1020 pointer operator->() const {
1021 assert(TST && "InternalIterator is invalid with a null TST.");
1022 return &operator*();
1023 }
1024 };
1025
1026 InternalIterator SugaredIterator;
1027 InternalIterator DesugaredIterator;
1028
1029 public:
1030 TSTiterator(ASTContext &Context, const TemplateSpecializationType *TST)
1031 : SugaredIterator(TST),
1032 DesugaredIterator(
1033 (TST->isSugared() && !TST->isTypeAlias())
1034 ? GetTemplateSpecializationType(Context, TST->desugar())
1035 : nullptr) {}
1036
1037 /// &operator++ - Increment the iterator to the next template argument.
1038 TSTiterator &operator++() {
1039 ++SugaredIterator;
1040 if (DesugaredIterator.isValid())
1041 ++DesugaredIterator;
1042 return *this;
1043 }
1044
1045 /// operator* - Returns the appropriate TemplateArgument.
1046 reference operator*() const {
1047 return *SugaredIterator;
1048 }
1049
1050 /// operator-> - Allow access to the underlying TemplateArgument.
1051 pointer operator->() const {
1052 return &operator*();
1053 }
1054
1055 /// isEnd - Returns true if no more TemplateArguments are available.
1056 bool isEnd() const {
1057 return SugaredIterator.isEnd();
1058 }
1059
1060 /// hasDesugaredTA - Returns true if there is another TemplateArgument
1061 /// available.
1062 bool hasDesugaredTA() const {
1063 return DesugaredIterator.isValid() && !DesugaredIterator.isEnd();
1064 }
1065
1066 /// getDesugaredTA - Returns the desugared TemplateArgument.
1067 reference getDesugaredTA() const {
1068 assert(DesugaredIterator.isValid() &&
1069 "Desugared TemplateArgument should not be used.");
1070 return *DesugaredIterator;
1071 }
1072 };

1078 ASTContext &Context, QualType Ty) {

1080 Ty->getAs<TemplateSpecializationType>())

1090
1091 if (!CTSD)
1092 return nullptr;

1094 Ty = Context.getTemplateSpecializationType(
1095 TemplateName(CTSD->getSpecializedTemplate()),

1097 Ty.getLocalUnqualifiedType().getCanonicalType());

1243 bool HasFromInt = false, HasToInt = false, FromNullPtr = false,

1247 FromIntType, FromNullPtr, FromExpr, FromValueDecl, NeedFromAddressOf);

1305 /// DiffTemplate - recursively visits template arguments and stores the
1306 /// argument info into a tree.
1307 void DiffTemplate(const TemplateSpecializationType *FromTST,
1308 const TemplateSpecializationType *ToTST) {
1309 // Begin descent into diffing template tree.
1310 TemplateParameterList *ParamsFrom =
1311 FromTST->getTemplateName().getAsTemplateDecl()->getTemplateParameters();
1312 TemplateParameterList *ParamsTo =
1313 ToTST->getTemplateName().getAsTemplateDecl()->getTemplateParameters();
1314 unsigned TotalArgs = 0;
1315 for (TSTiterator FromIter(Context, FromTST), ToIter(Context, ToTST);
1316 !FromIter.isEnd() || !ToIter.isEnd(); ++TotalArgs) {
1317 Tree.AddNode();
1318
1319 // Get the parameter at index TotalArgs. If index is larger
1320 // than the total number of parameters, then there is an
1321 // argument pack, so re-use the last parameter.
1322 unsigned FromParamIndex = std::min(TotalArgs, ParamsFrom->size() - 1);
1323 unsigned ToParamIndex = std::min(TotalArgs, ParamsTo->size() - 1);
1324 NamedDecl *FromParamND = ParamsFrom->getParam(FromParamIndex);
1325 NamedDecl *ToParamND = ParamsTo->getParam(ToParamIndex);
1326
1327 assert(FromParamND->getKind() == ToParamND->getKind() &&
1328 "Parameter Decl are not the same kind.");
1329
1330 if (isa<TemplateTypeParmDecl>(FromParamND)) {
1331 DiffTypes(FromIter, ToIter);
1332 } else if (isa<TemplateTemplateParmDecl>(FromParamND)) {
1333 DiffTemplateTemplates(FromIter, ToIter);
1334 } else if (isa<NonTypeTemplateParmDecl>(FromParamND)) {
1335 NonTypeTemplateParmDecl *FromDefaultNonTypeDecl =
1336 cast<NonTypeTemplateParmDecl>(FromParamND);
1337 NonTypeTemplateParmDecl *ToDefaultNonTypeDecl =
1338 cast<NonTypeTemplateParmDecl>(ToParamND);
1339 DiffNonTypes(FromIter, ToIter, FromDefaultNonTypeDecl,
1340 ToDefaultNonTypeDecl);
1341 } else {
1342 llvm_unreachable("Unexpected Decl type.");
1343 }
1344
1345 ++FromIter;
1346 ++ToIter;
1347 Tree.Up();
1348 }
1349 }

1363 /// hasSameBaseTemplate - Returns true when the base templates are the same,
1364 /// even if the template arguments are not.
1365 static bool hasSameBaseTemplate(const TemplateSpecializationType *FromTST,
1366 const TemplateSpecializationType *ToTST) {

1368 ToTST->getTemplateName().getAsTemplateDecl()->getCanonicalDecl();

1371 /// hasSameTemplate - Returns true if both types are specialized from the
1372 /// same template declaration. If they come from different template aliases,
1373 /// do a parallel ascension search to determine the highest template alias in
1374 /// common and set the arguments to them.
1375 static bool hasSameTemplate(const TemplateSpecializationType *&FromTST,
1376 const TemplateSpecializationType *&ToTST) {
1377 // Check the top templates if they are the same.
1378 if (hasSameBaseTemplate(FromTST, ToTST))
1379 return true;
1380
1381 // Create vectors of template aliases.
1382 SmallVector<const TemplateSpecializationType*, 1> FromTemplateList,
1383 ToTemplateList;
1384
1385 makeTemplateList(FromTemplateList, FromTST);
1386 makeTemplateList(ToTemplateList, ToTST);
1387
1388 SmallVectorImpl<const TemplateSpecializationType *>::reverse_iterator
1389 FromIter = FromTemplateList.rbegin(), FromEnd = FromTemplateList.rend(),
1390 ToIter = ToTemplateList.rbegin(), ToEnd = ToTemplateList.rend();
1391
1392 // Check if the lowest template types are the same. If not, return.
1393 if (!hasSameBaseTemplate(*FromIter, *ToIter))
1394 return false;
1395
1396 // Begin searching up the template aliases. The bottom most template
1397 // matches so move up until one pair does not match. Use the template
1398 // right before that one.
1399 for (; FromIter != FromEnd && ToIter != ToEnd; ++FromIter, ++ToIter) {
1400 if (!hasSameBaseTemplate(*FromIter, *ToIter))
1401 break;
1402 }
1403
1404 FromTST = FromIter[-1];
1405 ToTST = ToIter[-1];
1406
1407 return true;
1408 }

1449 /// TreeToString - Converts the Tree object into a character stream which
1450 /// will later be turned into the output string.
1451 void TreeToString(int Indent = 1) {
1452 if (PrintTree) {
1453 OS << '\n';
1454 OS.indent(2 * Indent);
1455 ++Indent;
1456 }
1457
1458 // Handle cases where the difference is not templates with different
1459 // arguments.
1460 switch (Tree.GetKind()) {
1461 case DiffTree::Invalid:
1462 llvm_unreachable("Template diffing failed with bad DiffNode");
1463 case DiffTree::Type: {
1464 QualType FromType, ToType;
1465 Tree.GetTypeDiff(FromType, ToType);
1466 PrintTypeNames(FromType, ToType, Tree.FromDefault(), Tree.ToDefault(),
1467 Tree.NodeIsSame());
1468 return;
1469 }
1470 case DiffTree::Expression: {
1471 Expr *FromExpr, *ToExpr;
1472 Tree.GetExpressionDiff(FromExpr, ToExpr);
1473 PrintExpr(FromExpr, ToExpr, Tree.FromDefault(), Tree.ToDefault(),
1474 Tree.NodeIsSame());
1475 return;
1476 }
1477 case DiffTree::TemplateTemplate: {
1478 TemplateDecl *FromTD, *ToTD;
1479 Tree.GetTemplateTemplateDiff(FromTD, ToTD);
1480 PrintTemplateTemplate(FromTD, ToTD, Tree.FromDefault(),
1481 Tree.ToDefault(), Tree.NodeIsSame());
1482 return;
1483 }
1484 case DiffTree::Integer: {
1485 llvm::APSInt FromInt, ToInt;
1486 Expr *FromExpr, *ToExpr;
1487 bool IsValidFromInt, IsValidToInt;
1488 QualType FromIntType, ToIntType;
1489 Tree.GetIntegerDiff(FromInt, ToInt, IsValidFromInt, IsValidToInt,
1490 FromIntType, ToIntType, FromExpr, ToExpr);
1491 PrintAPSInt(FromInt, ToInt, IsValidFromInt, IsValidToInt, FromIntType,
1492 ToIntType, FromExpr, ToExpr, Tree.FromDefault(),
1493 Tree.ToDefault(), Tree.NodeIsSame());
1494 return;
1495 }
1496 case DiffTree::Declaration: {
1497 ValueDecl *FromValueDecl, *ToValueDecl;
1498 bool FromAddressOf, ToAddressOf;
1499 bool FromNullPtr, ToNullPtr;
1500 Expr *FromExpr, *ToExpr;
1501 Tree.GetDeclarationDiff(FromValueDecl, ToValueDecl, FromAddressOf,
1502 ToAddressOf, FromNullPtr, ToNullPtr, FromExpr,
1503 ToExpr);
1504 PrintValueDecl(FromValueDecl, ToValueDecl, FromAddressOf, ToAddressOf,
1505 FromNullPtr, ToNullPtr, FromExpr, ToExpr,
1506 Tree.FromDefault(), Tree.ToDefault(), Tree.NodeIsSame());
1507 return;
1508 }
1509 case DiffTree::FromDeclarationAndToInteger: {
1510 ValueDecl *FromValueDecl;
1511 bool FromAddressOf;
1512 bool FromNullPtr;
1513 Expr *FromExpr;
1514 llvm::APSInt ToInt;
1515 bool IsValidToInt;
1516 QualType ToIntType;
1517 Expr *ToExpr;
1518 Tree.GetFromDeclarationAndToIntegerDiff(
1519 FromValueDecl, FromAddressOf, FromNullPtr, FromExpr, ToInt,
1520 IsValidToInt, ToIntType, ToExpr);
1521 assert((FromValueDecl || FromNullPtr) && IsValidToInt);
1522 PrintValueDeclAndInteger(FromValueDecl, FromAddressOf, FromNullPtr,
1523 FromExpr, Tree.FromDefault(), ToInt, ToIntType,
1524 ToExpr, Tree.ToDefault());
1525 return;
1526 }
1527 case DiffTree::FromIntegerAndToDeclaration: {
1528 llvm::APSInt FromInt;
1529 bool IsValidFromInt;
1530 QualType FromIntType;
1531 Expr *FromExpr;
1532 ValueDecl *ToValueDecl;
1533 bool ToAddressOf;
1534 bool ToNullPtr;
1535 Expr *ToExpr;
1536 Tree.GetFromIntegerAndToDeclarationDiff(
1537 FromInt, IsValidFromInt, FromIntType, FromExpr, ToValueDecl,
1538 ToAddressOf, ToNullPtr, ToExpr);
1539 assert(IsValidFromInt && (ToValueDecl || ToNullPtr));
1540 PrintIntegerAndValueDecl(FromInt, FromIntType, FromExpr,
1541 Tree.FromDefault(), ToValueDecl, ToAddressOf,
1542 ToNullPtr, ToExpr, Tree.ToDefault());
1543 return;
1544 }
1545 case DiffTree::Template: {
1546 // Node is root of template. Recurse on children.
1547 TemplateDecl *FromTD, *ToTD;
1548 Qualifiers FromQual, ToQual;
1549 Tree.GetTemplateDiff(FromTD, ToTD, FromQual, ToQual);
1550
1551 PrintQualifiers(FromQual, ToQual);
1552
1553 if (!Tree.HasChildren()) {
1554 // If we're dealing with a template specialization with zero
1555 // arguments, there are no children; special-case this.
1556 OS << FromTD->getNameAsString() << "<>";
1557 return;
1558 }
1559
1560 OS << FromTD->getNameAsString() << '<';
1561 Tree.MoveToChild();
1562 unsigned NumElideArgs = 0;
1563 bool AllArgsElided = true;
1564 do {
1565 if (ElideType) {
1566 if (Tree.NodeIsSame()) {
1567 ++NumElideArgs;
1568 continue;
1569 }
1570 AllArgsElided = false;
1571 if (NumElideArgs > 0) {
1572 PrintElideArgs(NumElideArgs, Indent);
1573 NumElideArgs = 0;
1574 OS << ", ";
1575 }
1576 }
1577 TreeToString(Indent);
1578 if (Tree.HasNextSibling())
1579 OS << ", ";
1580 } while (Tree.AdvanceSibling());
1581 if (NumElideArgs > 0) {
1582 if (AllArgsElided)
1583 OS << "...";
1584 else
1585 PrintElideArgs(NumElideArgs, Indent);
1586 }
1587
1588 Tree.Parent();
1589 OS << ">";
1590 return;
1591 }
1592 }
1593 }

1937 // Prints and highlights differences in Qualifiers.
1938 void PrintQualifiers(Qualifiers FromQual, Qualifiers ToQual) {
1939 // Both types have no qualifiers
1940 if (FromQual.empty() && ToQual.empty())
1941 return;
1942
1943 // Both types have same qualifiers
1944 if (FromQual == ToQual) {
1945 PrintQualifier(FromQual, /*ApplyBold*/false);
1946 return;
1947 }
1948
1949 // Find common qualifiers and strip them from FromQual and ToQual.
1950 Qualifiers CommonQual = Qualifiers::removeCommonQualifiers(FromQual,
1951 ToQual);
1952
1953 // The qualifiers are printed before the template name.
1954 // Inline printing:
1955 // The common qualifiers are printed. Then, qualifiers only in this type
1956 // are printed and highlighted. Finally, qualifiers only in the other
1957 // type are printed and highlighted inside parentheses after "missing".
1958 // Tree printing:
1959 // Qualifiers are printed next to each other, inside brackets, and
1960 // separated by "!=". The printing order is:
1961 // common qualifiers, highlighted from qualifiers, "!=",
1962 // common qualifiers, highlighted to qualifiers
1963 if (PrintTree) {
1964 OS << "[";
1965 if (CommonQual.empty() && FromQual.empty()) {
1966 Bold();
1967 OS << "(no qualifiers) ";
1968 Unbold();
1969 } else {
1970 PrintQualifier(CommonQual, /*ApplyBold*/false);
1971 PrintQualifier(FromQual, /*ApplyBold*/true);
1972 }
1973 OS << "!= ";
1974 if (CommonQual.empty() && ToQual.empty()) {
1975 Bold();
1976 OS << "(no qualifiers)";
1977 Unbold();
1978 } else {
1979 PrintQualifier(CommonQual, /*ApplyBold*/false,
1980 /*appendSpaceIfNonEmpty*/!ToQual.empty());
1981 PrintQualifier(ToQual, /*ApplyBold*/true,
1982 /*appendSpaceIfNonEmpty*/false);
1983 }
1984 OS << "] ";
1985 } else {
1986 PrintQualifier(CommonQual, /*ApplyBold*/false);
1987 PrintQualifier(FromQual, /*ApplyBold*/true);
1988 }
1989 }

1993 if (Q.empty()) return;
1994 if (ApplyBold) Bold();
1995 Q.print(OS, Policy, AppendSpaceIfNonEmpty);
1996 if (ApplyBold) Unbold();

2016 /// DiffTemplate - Start the template type diffing.
2017 void DiffTemplate() {
2018 Qualifiers FromQual = FromTemplateType.getQualifiers(),
2019 ToQual = ToTemplateType.getQualifiers();
2020
2021 const TemplateSpecializationType *FromOrigTST =
2022 GetTemplateSpecializationType(Context, FromTemplateType);
2023 const TemplateSpecializationType *ToOrigTST =
2024 GetTemplateSpecializationType(Context, ToTemplateType);
2025
2026 // Only checking templates.
2027 if (!FromOrigTST || !ToOrigTST)
2028 return;
2029
2030 // Different base templates.
2031 if (!hasSameTemplate(FromOrigTST, ToOrigTST)) {
2032 return;
2033 }
2034
2035 FromQual -= QualType(FromOrigTST, 0).getQualifiers();
2036 ToQual -= QualType(ToOrigTST, 0).getQualifiers();
2037
2038 // Same base template, but different arguments.
2039 Tree.SetTemplateDiff(FromOrigTST->getTemplateName().getAsTemplateDecl(),
2040 ToOrigTST->getTemplateName().getAsTemplateDecl(),
2041 FromQual, ToQual, false /*FromDefault*/,
2042 false /*ToDefault*/);
2043
2044 DiffTemplate(FromOrigTST, ToOrigTST);
2045 }

2048 /// base template, a string representation of the type difference will be
2049 /// emitted to the stream and return true. Otherwise, return false.

2051 Tree.StartTraverse();
2052 if (Tree.Empty())
2053 return false;
2054
2055 TreeToString();
2056 assert(!IsBold && "Bold is applied to end of string.");
2057 return true;
2058 }
