50static struct StmtClassNameTable {
51 const char *Name;
52 unsigned Counter;
53 unsigned Size;
54} StmtClassInfo[Stmt::lastStmtConstant+1];

56static StmtClassNameTable &getStmtInfoTableEntry(Stmt::StmtClass E) {
57 static bool Initialized = false;
58 if (Initialized)
59 return StmtClassInfo[E];
60
61 // Initialize the table on the first use.
62 Initialized = true;
63#define ABSTRACT_STMT(STMT)
64#define STMT(CLASS, PARENT) \
65 StmtClassInfo[(unsigned)Stmt::CLASS##Class].Name = #CLASS; \
66 StmtClassInfo[(unsigned)Stmt::CLASS##Class].Size = sizeof(CLASS);
67#include "clang/AST/StmtNodes.inc"
68
69 return StmtClassInfo[E];
70}

89void Stmt::PrintStats() {
90 // Ensure the table is primed.
91 getStmtInfoTableEntry(Stmt::NullStmtClass);
92
93 unsigned sum = 0;
94 llvm::errs() << "\n*** Stmt/Expr Stats:\n";
95 for (int i = 0; i != Stmt::lastStmtConstant+1; i++) {
96 if (StmtClassInfo[i].Name == nullptr) continue;
97 sum += StmtClassInfo[i].Counter;
98 }
99 llvm::errs() << " " << sum << " stmts/exprs total.\n";
100 sum = 0;
101 for (int i = 0; i != Stmt::lastStmtConstant+1; i++) {
102 if (StmtClassInfo[i].Name == nullptr) continue;
103 if (StmtClassInfo[i].Counter == 0) continue;
104 llvm::errs() << " " << StmtClassInfo[i].Counter << " "
105 << StmtClassInfo[i].Name << ", " << StmtClassInfo[i].Size
106 << " each (" << StmtClassInfo[i].Counter*StmtClassInfo[i].Size
107 << " bytes)\n";
108 sum += StmtClassInfo[i].Counter*StmtClassInfo[i].Size;
109 }
110
111 llvm::errs() << "Total bytes = " << sum << "\n";
112}

123/// Skip no-op (attributed, compound) container stmts and skip captured
124/// stmt at the top, if \a IgnoreCaptured is true.
125Stmt *Stmt::IgnoreContainers(bool IgnoreCaptured) {
126 Stmt *S = this;
127 if (IgnoreCaptured)
128 if (auto CapS = dyn_cast_or_null<CapturedStmt>(S))
129 S = CapS->getCapturedStmt();
130 while (true) {
131 if (auto AS = dyn_cast_or_null<AttributedStmt>(S))
132 S = AS->getSubStmt();
133 else if (auto CS = dyn_cast_or_null<CompoundStmt>(S)) {
134 if (CS->size() != 1)
135 break;
136 S = CS->body_back();
137 } else
138 break;
139 }
140 return S;
141}

143/// Strip off all label-like statements.
144///
145/// This will strip off label statements, case statements, attributed
146/// statements and default statements recursively.
147const Stmt *Stmt::stripLabelLikeStatements() const {
148 const Stmt *S = this;
149 while (true) {
150 if (const auto *LS = dyn_cast<LabelStmt>(S))
151 S = LS->getSubStmt();
152 else if (const auto *SC = dyn_cast<SwitchCase>(S))
153 S = SC->getSubStmt();
154 else if (const auto *AS = dyn_cast<AttributedStmt>(S))
155 S = AS->getSubStmt();
156 else
157 return S;
158 }
159}

228// a more specific implementation of getSourceRange.
229//
230// See also Expr.cpp:getExprLoc().

233 /// This implementation is used when a class provides a custom
234 /// implementation of getSourceRange.
235 template <class S, class T>
236 SourceRange getSourceRangeImpl(const Stmt *stmt,
237 SourceRange (T::*v)() const) {
238 return static_cast<const S*>(stmt)->getSourceRange();
239 }
240
241 /// This implementation is used when a class doesn't provide a custom
242 /// implementation of getSourceRange. Overload resolution should pick it over
243 /// the implementation above because it's more specialized according to
244 /// function template partial ordering.
245 template <class S>
246 SourceRange getSourceRangeImpl(const Stmt *stmt,
247 SourceRange (Stmt::*v)() const) {
248 return SourceRange(static_cast<const S *>(stmt)->getBeginLoc(),
249 static_cast<const S *>(stmt)->getEndLoc());
250 }

252} // namespace

326const Expr *ValueStmt::getExprStmt() const {
327 const Stmt *S = this;
328 do {
329 if (const auto *E = dyn_cast<Expr>(S))
330 return E;
331
332 if (const auto *LS = dyn_cast<LabelStmt>(S))
333 S = LS->getSubStmt();
334 else if (const auto *AS = dyn_cast<AttributedStmt>(S))
335 S = AS->getSubStmt();
336 else
337 llvm_unreachable("unknown kind of ValueStmt");
338 } while (isa<ValueStmt>(S));
339
340 return nullptr;
341}

464void GCCAsmStmt::setOutputsAndInputsAndClobbers(const ASTContext &C,
465 IdentifierInfo **Names,
466 StringLiteral **Constraints,
467 Stmt **Exprs,
468 unsigned NumOutputs,
469 unsigned NumInputs,
470 unsigned NumLabels,
471 StringLiteral **Clobbers,
472 unsigned NumClobbers) {
473 this->NumOutputs = NumOutputs;
474 this->NumInputs = NumInputs;
475 this->NumClobbers = NumClobbers;
476 this->NumLabels = NumLabels;
477 assert(!(NumOutputs && NumLabels) && "asm goto cannot have outputs");
478
479 unsigned NumExprs = NumOutputs + NumInputs + NumLabels;
480
481 C.Deallocate(this->Names);
482 this->Names = new (C) IdentifierInfo*[NumExprs];
483 std::copy(Names, Names + NumExprs, this->Names);
484
485 C.Deallocate(this->Exprs);
486 this->Exprs = new (C) Stmt*[NumExprs];
487 std::copy(Exprs, Exprs + NumExprs, this->Exprs);
488
489 unsigned NumConstraints = NumOutputs + NumInputs;
490 C.Deallocate(this->Constraints);
491 this->Constraints = new (C) StringLiteral*[NumConstraints];
492 std::copy(Constraints, Constraints + NumConstraints, this->Constraints);
493
494 C.Deallocate(this->Clobbers);
495 this->Clobbers = new (C) StringLiteral*[NumClobbers];
496 std::copy(Clobbers, Clobbers + NumClobbers, this->Clobbers);
497}

499/// getNamedOperand - Given a symbolic operand reference like %[foo],
500/// translate this into a numeric value needed to reference the same operand.
501/// This returns -1 if the operand name is invalid.
502int GCCAsmStmt::getNamedOperand(StringRef SymbolicName) const {
503 unsigned NumPlusOperands = 0;
504
505 // Check if this is an output operand.
506 for (unsigned i = 0, e = getNumOutputs(); i != e; ++i) {
507 if (getOutputName(i) == SymbolicName)
508 return i;
509 }
510
511 for (unsigned i = 0, e = getNumInputs(); i != e; ++i)
512 if (getInputName(i) == SymbolicName)
513 return getNumOutputs() + NumPlusOperands + i;
514
515 for (unsigned i = 0, e = getNumLabels(); i != e; ++i)
516 if (getLabelName(i) == SymbolicName)
517 return i + getNumInputs();
518
519 // Not found.
520 return -1;
521}

523/// AnalyzeAsmString - Analyze the asm string of the current asm, decomposing
524/// it into pieces. If the asm string is erroneous, emit errors and return
525/// true, otherwise return false.
526unsigned GCCAsmStmt::AnalyzeAsmString(SmallVectorImpl<AsmStringPiece>&Pieces,
527 const ASTContext &C, unsigned &DiagOffs) const {
528 StringRef Str = getAsmString()->getString();
529 const char *StrStart = Str.begin();
530 const char *StrEnd = Str.end();
531 const char *CurPtr = StrStart;
532
533 // "Simple" inline asms have no constraints or operands, just convert the asm
534 // string to escape $'s.
535 if (isSimple()) {
536 std::string Result;
537 for (; CurPtr != StrEnd; ++CurPtr) {
538 switch (*CurPtr) {
539 case '$':
540 Result += "$$";
541 break;
542 default:
543 Result += *CurPtr;
544 break;
545 }
546 }
547 Pieces.push_back(AsmStringPiece(Result));
548 return 0;
549 }
550
551 // CurStringPiece - The current string that we are building up as we scan the
552 // asm string.
553 std::string CurStringPiece;
554
555 bool HasVariants = !C.getTargetInfo().hasNoAsmVariants();
556
557 unsigned LastAsmStringToken = 0;
558 unsigned LastAsmStringOffset = 0;
559
560 while (true) {
561 // Done with the string?
562 if (CurPtr == StrEnd) {
563 if (!CurStringPiece.empty())
564 Pieces.push_back(AsmStringPiece(CurStringPiece));
565 return 0;
566 }
567
568 char CurChar = *CurPtr++;
569 switch (CurChar) {
570 case '$': CurStringPiece += "$$"; continue;
571 case '{': CurStringPiece += (HasVariants ? "$(" : "{"); continue;
572 case '|': CurStringPiece += (HasVariants ? "$|" : "|"); continue;
573 case '}': CurStringPiece += (HasVariants ? "$)" : "}"); continue;
574 case '%':
575 break;
576 default:
577 CurStringPiece += CurChar;
578 continue;
579 }
580
581 // Escaped "%" character in asm string.
582 if (CurPtr == StrEnd) {
583 // % at end of string is invalid (no escape).
584 DiagOffs = CurPtr-StrStart-1;
585 return diag::err_asm_invalid_escape;
586 }
587 // Handle escaped char and continue looping over the asm string.
588 char EscapedChar = *CurPtr++;
589 switch (EscapedChar) {
590 default:
591 break;
592 case '%': // %% -> %
593 case '{': // %{ -> {
594 case '}': // %} -> }
595 CurStringPiece += EscapedChar;
596 continue;
597 case '=': // %= -> Generate a unique ID.
598 CurStringPiece += "${:uid}";
599 continue;
600 }
601
602 // Otherwise, we have an operand. If we have accumulated a string so far,
603 // add it to the Pieces list.
604 if (!CurStringPiece.empty()) {
605 Pieces.push_back(AsmStringPiece(CurStringPiece));
606 CurStringPiece.clear();
607 }
608
609 // Handle operands that have asmSymbolicName (e.g., %x[foo]) and those that
610 // don't (e.g., %x4). 'x' following the '%' is the constraint modifier.
611
612 const char *Begin = CurPtr - 1; // Points to the character following '%'.
613 const char *Percent = Begin - 1; // Points to '%'.
614
615 if (isLetter(EscapedChar)) {
616 if (CurPtr == StrEnd) { // Premature end.
617 DiagOffs = CurPtr-StrStart-1;
618 return diag::err_asm_invalid_escape;
619 }
620 EscapedChar = *CurPtr++;
621 }
622
623 const TargetInfo &TI = C.getTargetInfo();
624 const SourceManager &SM = C.getSourceManager();
625 const LangOptions &LO = C.getLangOpts();
626
627 // Handle operands that don't have asmSymbolicName (e.g., %x4).
628 if (isDigit(EscapedChar)) {
629 // %n - Assembler operand n
630 unsigned N = 0;
631
632 --CurPtr;
633 while (CurPtr != StrEnd && isDigit(*CurPtr))
634 N = N*10 + ((*CurPtr++)-'0');
635
636 unsigned NumOperands = getNumOutputs() + getNumPlusOperands() +
637 getNumInputs() + getNumLabels();
638 if (N >= NumOperands) {
639 DiagOffs = CurPtr-StrStart-1;
640 return diag::err_asm_invalid_operand_number;
641 }
642
643 // Str contains "x4" (Operand without the leading %).
644 std::string Str(Begin, CurPtr - Begin);
645
646 // (BeginLoc, EndLoc) represents the range of the operand we are currently
647 // processing. Unlike Str, the range includes the leading '%'.
648 SourceLocation BeginLoc = getAsmString()->getLocationOfByte(
649 Percent - StrStart, SM, LO, TI, &LastAsmStringToken,
650 &LastAsmStringOffset);
651 SourceLocation EndLoc = getAsmString()->getLocationOfByte(
652 CurPtr - StrStart, SM, LO, TI, &LastAsmStringToken,
653 &LastAsmStringOffset);
654
655 Pieces.emplace_back(N, std::move(Str), BeginLoc, EndLoc);
656 continue;
657 }
658
659 // Handle operands that have asmSymbolicName (e.g., %x[foo]).
660 if (EscapedChar == '[') {
661 DiagOffs = CurPtr-StrStart-1;
662
663 // Find the ']'.
664 const char *NameEnd = (const char*)memchr(CurPtr, ']', StrEnd-CurPtr);
665 if (NameEnd == nullptr)
666 return diag::err_asm_unterminated_symbolic_operand_name;
667 if (NameEnd == CurPtr)
668 return diag::err_asm_empty_symbolic_operand_name;
669
670 StringRef SymbolicName(CurPtr, NameEnd - CurPtr);
671
672 int N = getNamedOperand(SymbolicName);
673 if (N == -1) {
674 // Verify that an operand with that name exists.
675 DiagOffs = CurPtr-StrStart;
676 return diag::err_asm_unknown_symbolic_operand_name;
677 }
678
679 // Str contains "x[foo]" (Operand without the leading %).
680 std::string Str(Begin, NameEnd + 1 - Begin);
681
682 // (BeginLoc, EndLoc) represents the range of the operand we are currently
683 // processing. Unlike Str, the range includes the leading '%'.
684 SourceLocation BeginLoc = getAsmString()->getLocationOfByte(
685 Percent - StrStart, SM, LO, TI, &LastAsmStringToken,
686 &LastAsmStringOffset);
687 SourceLocation EndLoc = getAsmString()->getLocationOfByte(
688 NameEnd + 1 - StrStart, SM, LO, TI, &LastAsmStringToken,
689 &LastAsmStringOffset);
690
691 Pieces.emplace_back(N, std::move(Str), BeginLoc, EndLoc);
692
693 CurPtr = NameEnd+1;
694 continue;
695 }
696
697 DiagOffs = CurPtr-StrStart-1;
698 return diag::err_asm_invalid_escape;
699 }
700}

702/// Assemble final IR asm string (GCC-style).
703std::string GCCAsmStmt::generateAsmString(const ASTContext &C) const {
704 // Analyze the asm string to decompose it into its pieces. We know that Sema
705 // has already done this, so it is guaranteed to be successful.
706 SmallVector<GCCAsmStmt::AsmStringPiece, 4> Pieces;
707 unsigned DiagOffs;
708 AnalyzeAsmString(Pieces, C, DiagOffs);
709
710 std::string AsmString;
711 for (const auto &Piece : Pieces) {
712 if (Piece.isString())
713 AsmString += Piece.getString();
714 else if (Piece.getModifier() == '\0')
715 AsmString += '$' + llvm::utostr(Piece.getOperandNo());
716 else
717 AsmString += "${" + llvm::utostr(Piece.getOperandNo()) + ':' +
718 Piece.getModifier() + '}';
719 }
720 return AsmString;
721}

1154CapturedStmt::Capture::Capture(SourceLocation Loc, VariableCaptureKind Kind,
1155 VarDecl *Var)
1156 : VarAndKind(Var, Kind), Loc(Loc) {
1157 switch (Kind) {
1158 case VCK_This:
1159 assert(!Var && "'this' capture cannot have a variable!");
1160 break;
1161 case VCK_ByRef:
1162 assert(Var && "capturing by reference must have a variable!");
1163 break;
1164 case VCK_ByCopy:
1165 assert(Var && "capturing by copy must have a variable!");
1166 assert(
1167 (Var->getType()->isScalarType() || (Var->getType()->isReferenceType() &&
1168 Var->getType()
1169 ->castAs<ReferenceType>()
1170 ->getPointeeType()
1171 ->isScalarType())) &&
1172 "captures by copy are expected to have a scalar type!");
1173 break;
1174 case VCK_VLAType:
1175 assert(!Var &&
1176 "Variable-length array type capture cannot have a variable!");
1177 break;
1178 }
1179}

1192CapturedStmt::Capture *CapturedStmt::getStoredCaptures() const {
1193 unsigned Size = sizeof(CapturedStmt) + sizeof(Stmt *) * (NumCaptures + 1);
1194
1195 // Offset of the first Capture object.
1196 unsigned FirstCaptureOffset = llvm::alignTo(Size, alignof(Capture));

1199 reinterpret_cast<char *>(const_cast<CapturedStmt *>(this))

1203CapturedStmt::CapturedStmt(Stmt *S, CapturedRegionKind Kind,
1204 ArrayRef<Capture> Captures,
1205 ArrayRef<Expr *> CaptureInits,
1206 CapturedDecl *CD,
1207 RecordDecl *RD)
1208 : Stmt(CapturedStmtClass), NumCaptures(Captures.size()),
1209 CapDeclAndKind(CD, Kind), TheRecordDecl(RD) {
1210 assert( S && "null captured statement");
1211 assert(CD && "null captured declaration for captured statement");
1212 assert(RD && "null record declaration for captured statement");
1213
1214 // Copy initialization expressions.
1215 Stmt **Stored = getStoredStmts();
1216 for (unsigned I = 0, N = NumCaptures; I != N; ++I)
1217 *Stored++ = CaptureInits[I];
1218
1219 // Copy the statement being captured.
1220 *Stored = S;
1221
1222 // Copy all Capture objects.
1223 Capture *Buffer = getStoredCaptures();
1224 std::copy(Captures.begin(), Captures.end(), Buffer);
1225}

1233CapturedStmt *CapturedStmt::Create(const ASTContext &Context, Stmt *S,
1234 CapturedRegionKind Kind,
1235 ArrayRef<Capture> Captures,
1236 ArrayRef<Expr *> CaptureInits,
1237 CapturedDecl *CD,
1238 RecordDecl *RD) {
1239 // The layout is
1240 //
1241 // -----------------------------------------------------------
1242 // | CapturedStmt, Init, ..., Init, S, Capture, ..., Capture |
1243 // ----------------^-------------------^----------------------
1244 // getStoredStmts() getStoredCaptures()
1245 //
1246 // where S is the statement being captured.
1247 //
1248 assert(CaptureInits.size() == Captures.size() && "wrong number of arguments");
1249
1250 unsigned Size = sizeof(CapturedStmt) + sizeof(Stmt *) * (Captures.size() + 1);
1251 if (!Captures.empty()) {
1252 // Realign for the following Capture array.
1253 Size = llvm::alignTo(Size, alignof(Capture));
1254 Size += sizeof(Capture) * Captures.size();
1255 }
1256
1257 void *Mem = Context.Allocate(Size);
1258 return new (Mem) CapturedStmt(S, Kind, Captures, CaptureInits, CD, RD);
1259}
