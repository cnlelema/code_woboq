9// This file implement StructuralEquivalenceContext class and helper functions
10// for layout matching.
11//
12// The structural equivalence check could have been implemented as a parallel
13// BFS on a pair of graphs. That must have been the original approach at the
14// beginning.
15// Let's consider this simple BFS algorithm from the `s` source:
16// ```
17// void bfs(Graph G, int s)
18// {
19// Queue<Integer> queue = new Queue<Integer>();
20// marked[s] = true; // Mark the source
21// queue.enqueue(s); // and put it on the queue.
22// while (!q.isEmpty()) {
23// int v = queue.dequeue(); // Remove next vertex from the queue.
24// for (int w : G.adj(v))
25// if (!marked[w]) // For every unmarked adjacent vertex,
26// {
27// marked[w] = true;
28// queue.enqueue(w);
29// }
30// }
31// }
32// ```
33// Indeed, it has it's queue, which holds pairs of nodes, one from each graph,
34// this is the `DeclsToCheck` and it's pair is in `TentativeEquivalences`.
35// `TentativeEquivalences` also plays the role of the marking (`marked`)
36// functionality above, we use it to check whether we've already seen a pair of
37// nodes.
38//
39// We put in the elements into the queue only in the toplevel decl check
40// function:
41// ```
42// static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
43// Decl *D1, Decl *D2);
44// ```
45// The `while` loop where we iterate over the children is implemented in
46// `Finish()`. And `Finish` is called only from the two **member** functions
47// which check the equivalency of two Decls or two Types. ASTImporter (and
48// other clients) call only these functions.
49//
50// The `static` implementation functions are called from `Finish`, these push
51// the children nodes to the queue via `static bool
52// IsStructurallyEquivalent(StructuralEquivalenceContext &Context, Decl *D1,
53// Decl *D2)`. So far so good, this is almost like the BFS. However, if we
54// let a static implementation function to call `Finish` via another **member**
55// function that means we end up with two nested while loops each of them
56// working on the same queue. This is wrong and nobody can reason about it's
57// doing. Thus, static implementation functions must not call the **member**
58// functions.
59//
60// So, now `TentativeEquivalences` plays two roles. It is used to store the
61// second half of the decls which we want to compare, plus it plays a role in
62// closing the recursion. On a long term, we could refactor structural
63// equivalency to be more alike to the traditional BFS.
64//

123/// Determine whether two nested-name-specifiers are equivalent.
124static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
125 NestedNameSpecifier *NNS1,
126 NestedNameSpecifier *NNS2) {
127 if (NNS1->getKind() != NNS2->getKind())
128 return false;
129
130 NestedNameSpecifier *Prefix1 = NNS1->getPrefix(),
131 *Prefix2 = NNS2->getPrefix();
132 if ((bool)Prefix1 != (bool)Prefix2)
133 return false;
134
135 if (Prefix1)
136 if (!IsStructurallyEquivalent(Context, Prefix1, Prefix2))
137 return false;
138
139 switch (NNS1->getKind()) {
140 case NestedNameSpecifier::Identifier:
141 return IsStructurallyEquivalent(NNS1->getAsIdentifier(),
142 NNS2->getAsIdentifier());
143 case NestedNameSpecifier::Namespace:
144 return IsStructurallyEquivalent(Context, NNS1->getAsNamespace(),
145 NNS2->getAsNamespace());
146 case NestedNameSpecifier::NamespaceAlias:
147 return IsStructurallyEquivalent(Context, NNS1->getAsNamespaceAlias(),
148 NNS2->getAsNamespaceAlias());
149 case NestedNameSpecifier::TypeSpec:
150 case NestedNameSpecifier::TypeSpecWithTemplate:
151 return IsStructurallyEquivalent(Context, QualType(NNS1->getAsType(), 0),
152 QualType(NNS2->getAsType(), 0));
153 case NestedNameSpecifier::Global:
154 return true;
155 case NestedNameSpecifier::Super:
156 return IsStructurallyEquivalent(Context, NNS1->getAsRecordDecl(),
157 NNS2->getAsRecordDecl());
158 }
159 return false;
160}

162static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
163 const TemplateName &N1,
164 const TemplateName &N2) {
165 if (N1.getKind() != N2.getKind())
166 return false;
167 switch (N1.getKind()) {
168 case TemplateName::Template:
169 return IsStructurallyEquivalent(Context, N1.getAsTemplateDecl(),
170 N2.getAsTemplateDecl());
171
172 case TemplateName::OverloadedTemplate: {
173 OverloadedTemplateStorage *OS1 = N1.getAsOverloadedTemplate(),
174 *OS2 = N2.getAsOverloadedTemplate();
175 OverloadedTemplateStorage::iterator I1 = OS1->begin(), I2 = OS2->begin(),
176 E1 = OS1->end(), E2 = OS2->end();
177 for (; I1 != E1 && I2 != E2; ++I1, ++I2)
178 if (!IsStructurallyEquivalent(Context, *I1, *I2))
179 return false;
180 return I1 == E1 && I2 == E2;
181 }
182
183 case TemplateName::AssumedTemplate: {
184 AssumedTemplateStorage *TN1 = N1.getAsAssumedTemplateName(),
185 *TN2 = N1.getAsAssumedTemplateName();
186 return TN1->getDeclName() == TN2->getDeclName();
187 }
188
189 case TemplateName::QualifiedTemplate: {
190 QualifiedTemplateName *QN1 = N1.getAsQualifiedTemplateName(),
191 *QN2 = N2.getAsQualifiedTemplateName();
192 return IsStructurallyEquivalent(Context, QN1->getDecl(), QN2->getDecl()) &&
193 IsStructurallyEquivalent(Context, QN1->getQualifier(),
194 QN2->getQualifier());
195 }
196
197 case TemplateName::DependentTemplate: {
198 DependentTemplateName *DN1 = N1.getAsDependentTemplateName(),
199 *DN2 = N2.getAsDependentTemplateName();
200 if (!IsStructurallyEquivalent(Context, DN1->getQualifier(),
201 DN2->getQualifier()))
202 return false;
203 if (DN1->isIdentifier() && DN2->isIdentifier())
204 return IsStructurallyEquivalent(DN1->getIdentifier(),
205 DN2->getIdentifier());
206 else if (DN1->isOverloadedOperator() && DN2->isOverloadedOperator())
207 return DN1->getOperator() == DN2->getOperator();
208 return false;
209 }
210
211 case TemplateName::SubstTemplateTemplateParm: {
212 SubstTemplateTemplateParmStorage *TS1 = N1.getAsSubstTemplateTemplateParm(),
213 *TS2 = N2.getAsSubstTemplateTemplateParm();
214 return IsStructurallyEquivalent(Context, TS1->getParameter(),
215 TS2->getParameter()) &&
216 IsStructurallyEquivalent(Context, TS1->getReplacement(),
217 TS2->getReplacement());
218 }
219
220 case TemplateName::SubstTemplateTemplateParmPack: {
221 SubstTemplateTemplateParmPackStorage
222 *P1 = N1.getAsSubstTemplateTemplateParmPack(),
223 *P2 = N2.getAsSubstTemplateTemplateParmPack();
224 return IsStructurallyEquivalent(Context, P1->getArgumentPack(),
225 P2->getArgumentPack()) &&
226 IsStructurallyEquivalent(Context, P1->getParameterPack(),
227 P2->getParameterPack());
228 }
229 }
230 return false;
231}

233/// Determine whether two template arguments are equivalent.
234static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
235 const TemplateArgument &Arg1,
236 const TemplateArgument &Arg2) {
237 if (Arg1.getKind() != Arg2.getKind())
238 return false;
239
240 switch (Arg1.getKind()) {
241 case TemplateArgument::Null:
242 return true;
243
244 case TemplateArgument::Type:
245 return IsStructurallyEquivalent(Context, Arg1.getAsType(), Arg2.getAsType());
246
247 case TemplateArgument::Integral:
248 if (!IsStructurallyEquivalent(Context, Arg1.getIntegralType(),
249 Arg2.getIntegralType()))
250 return false;
251
252 return llvm::APSInt::isSameValue(Arg1.getAsIntegral(),
253 Arg2.getAsIntegral());
254
255 case TemplateArgument::Declaration:
256 return IsStructurallyEquivalent(Context, Arg1.getAsDecl(), Arg2.getAsDecl());
257
258 case TemplateArgument::NullPtr:
259 return true; // FIXME: Is this correct?
260
261 case TemplateArgument::Template:
262 return IsStructurallyEquivalent(Context, Arg1.getAsTemplate(),
263 Arg2.getAsTemplate());
264
265 case TemplateArgument::TemplateExpansion:
266 return IsStructurallyEquivalent(Context,
267 Arg1.getAsTemplateOrTemplatePattern(),
268 Arg2.getAsTemplateOrTemplatePattern());
269
270 case TemplateArgument::Expression:
271 return IsStructurallyEquivalent(Context, Arg1.getAsExpr(),
272 Arg2.getAsExpr());
273
274 case TemplateArgument::Pack:
275 if (Arg1.pack_size() != Arg2.pack_size())
276 return false;
277
278 for (unsigned I = 0, N = Arg1.pack_size(); I != N; ++I)
279 if (!IsStructurallyEquivalent(Context, Arg1.pack_begin()[I],
280 Arg2.pack_begin()[I]))
281 return false;
282
283 return true;
284 }
285
286 llvm_unreachable("Invalid template argument kind");
287}

289/// Determine structural equivalence for the common part of array
290/// types.
291static bool IsArrayStructurallyEquivalent(StructuralEquivalenceContext &Context,
292 const ArrayType *Array1,
293 const ArrayType *Array2) {
294 if (!IsStructurallyEquivalent(Context, Array1->getElementType(),
295 Array2->getElementType()))
296 return false;
297 if (Array1->getSizeModifier() != Array2->getSizeModifier())
298 return false;
299 if (Array1->getIndexTypeQualifiers() != Array2->getIndexTypeQualifiers())
300 return false;
301
302 return true;
303}

305/// Determine structural equivalence based on the ExtInfo of functions. This
306/// is inspired by ASTContext::mergeFunctionTypes(), we compare calling
307/// conventions bits but must not compare some other bits.
308static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
309 FunctionType::ExtInfo EI1,
310 FunctionType::ExtInfo EI2) {
311 // Compatible functions must have compatible calling conventions.
312 if (EI1.getCC() != EI2.getCC())
313 return false;
314
315 // Regparm is part of the calling convention.
316 if (EI1.getHasRegParm() != EI2.getHasRegParm())
317 return false;
318 if (EI1.getRegParm() != EI2.getRegParm())
319 return false;
320
321 if (EI1.getProducesResult() != EI2.getProducesResult())
322 return false;
323 if (EI1.getNoCallerSavedRegs() != EI2.getNoCallerSavedRegs())
324 return false;
325 if (EI1.getNoCfCheck() != EI2.getNoCfCheck())
326 return false;
327
328 return true;
329}

331/// Check the equivalence of exception specifications.
332static bool IsEquivalentExceptionSpec(StructuralEquivalenceContext &Context,
333 const FunctionProtoType *Proto1,
334 const FunctionProtoType *Proto2) {
335
336 auto Spec1 = Proto1->getExceptionSpecType();
337 auto Spec2 = Proto2->getExceptionSpecType();
338
339 if (isUnresolvedExceptionSpec(Spec1) || isUnresolvedExceptionSpec(Spec2))
340 return true;
341
342 if (Spec1 != Spec2)
343 return false;
344 if (Spec1 == EST_Dynamic) {
345 if (Proto1->getNumExceptions() != Proto2->getNumExceptions())
346 return false;
347 for (unsigned I = 0, N = Proto1->getNumExceptions(); I != N; ++I) {
348 if (!IsStructurallyEquivalent(Context, Proto1->getExceptionType(I),
349 Proto2->getExceptionType(I)))
350 return false;
351 }
352 } else if (isComputedNoexcept(Spec1)) {
353 if (!IsStructurallyEquivalent(Context, Proto1->getNoexceptExpr(),
354 Proto2->getNoexceptExpr()))
355 return false;
356 }
357
358 return true;
359}

361/// Determine structural equivalence of two types.
362static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
363 QualType T1, QualType T2) {
364 if (T1.isNull() || T2.isNull())
365 return T1.isNull() && T2.isNull();
366
367 QualType OrigT1 = T1;
368 QualType OrigT2 = T2;
369
370 if (!Context.StrictTypeSpelling) {
371 // We aren't being strict about token-to-token equivalence of types,
372 // so map down to the canonical type.
373 T1 = Context.FromCtx.getCanonicalType(T1);
374 T2 = Context.ToCtx.getCanonicalType(T2);
375 }
376
377 if (T1.getQualifiers() != T2.getQualifiers())
378 return false;
379
380 Type::TypeClass TC = T1->getTypeClass();
381
382 if (T1->getTypeClass() != T2->getTypeClass()) {
383 // Compare function types with prototypes vs. without prototypes as if
384 // both did not have prototypes.
385 if (T1->getTypeClass() == Type::FunctionProto &&
386 T2->getTypeClass() == Type::FunctionNoProto)
387 TC = Type::FunctionNoProto;
388 else if (T1->getTypeClass() == Type::FunctionNoProto &&
389 T2->getTypeClass() == Type::FunctionProto)
390 TC = Type::FunctionNoProto;
391 else
392 return false;
393 }
394
395 switch (TC) {
396 case Type::Builtin:
397 // FIXME: Deal with Char_S/Char_U.
398 if (cast<BuiltinType>(T1)->getKind() != cast<BuiltinType>(T2)->getKind())
399 return false;
400 break;
401
402 case Type::Complex:
403 if (!IsStructurallyEquivalent(Context,
404 cast<ComplexType>(T1)->getElementType(),
405 cast<ComplexType>(T2)->getElementType()))
406 return false;
407 break;
408
409 case Type::Adjusted:
410 case Type::Decayed:
411 if (!IsStructurallyEquivalent(Context,
412 cast<AdjustedType>(T1)->getOriginalType(),
413 cast<AdjustedType>(T2)->getOriginalType()))
414 return false;
415 break;
416
417 case Type::Pointer: {
418 const auto *Ptr1 = cast<PointerType>(T1);
419 const auto *Ptr2 = cast<PointerType>(T2);
420 if (Ptr1->isCHERICapability() != Ptr2->isCHERICapability())
421 return false;
422 if (!IsStructurallyEquivalent(Context, Ptr1->getPointeeType(), Ptr2->getPointeeType()))
423 return false;
424 break;
425 }
426
427 case Type::BlockPointer:
428 if (!IsStructurallyEquivalent(Context,
429 cast<BlockPointerType>(T1)->getPointeeType(),
430 cast<BlockPointerType>(T2)->getPointeeType()))
431 return false;
432 break;
433
434 case Type::LValueReference:
435 case Type::RValueReference: {
436 const auto *Ref1 = cast<ReferenceType>(T1);
437 const auto *Ref2 = cast<ReferenceType>(T2);
438 if (Ref1->isSpelledAsLValue() != Ref2->isSpelledAsLValue())
439 return false;
440 if (Ref1->isCHERICapability() != Ref2->isCHERICapability())
441 return false;
442 if (Ref1->isInnerRef() != Ref2->isInnerRef())
443 return false;
444 if (!IsStructurallyEquivalent(Context, Ref1->getPointeeTypeAsWritten(),
445 Ref2->getPointeeTypeAsWritten()))
446 return false;
447 break;
448 }
449
450 case Type::MemberPointer: {
451 const auto *MemPtr1 = cast<MemberPointerType>(T1);
452 const auto *MemPtr2 = cast<MemberPointerType>(T2);
453 if (!IsStructurallyEquivalent(Context, MemPtr1->getPointeeType(),
454 MemPtr2->getPointeeType()))
455 return false;
456 if (!IsStructurallyEquivalent(Context, QualType(MemPtr1->getClass(), 0),
457 QualType(MemPtr2->getClass(), 0)))
458 return false;
459 break;
460 }
461
462 case Type::ConstantArray: {
463 const auto *Array1 = cast<ConstantArrayType>(T1);
464 const auto *Array2 = cast<ConstantArrayType>(T2);
465 if (!llvm::APInt::isSameValue(Array1->getSize(), Array2->getSize()))
466 return false;
467
468 if (!IsArrayStructurallyEquivalent(Context, Array1, Array2))
469 return false;
470 break;
471 }
472
473 case Type::IncompleteArray:
474 if (!IsArrayStructurallyEquivalent(Context, cast<ArrayType>(T1),
475 cast<ArrayType>(T2)))
476 return false;
477 break;
478
479 case Type::VariableArray: {
480 const auto *Array1 = cast<VariableArrayType>(T1);
481 const auto *Array2 = cast<VariableArrayType>(T2);
482 if (!IsStructurallyEquivalent(Context, Array1->getSizeExpr(),
483 Array2->getSizeExpr()))
484 return false;
485
486 if (!IsArrayStructurallyEquivalent(Context, Array1, Array2))
487 return false;
488
489 break;
490 }
491
492 case Type::DependentSizedArray: {
493 const auto *Array1 = cast<DependentSizedArrayType>(T1);
494 const auto *Array2 = cast<DependentSizedArrayType>(T2);
495 if (!IsStructurallyEquivalent(Context, Array1->getSizeExpr(),
496 Array2->getSizeExpr()))
497 return false;
498
499 if (!IsArrayStructurallyEquivalent(Context, Array1, Array2))
500 return false;
501
502 break;
503 }
504
505 case Type::DependentAddressSpace: {
506 const auto *DepAddressSpace1 = cast<DependentAddressSpaceType>(T1);
507 const auto *DepAddressSpace2 = cast<DependentAddressSpaceType>(T2);
508 if (!IsStructurallyEquivalent(Context, DepAddressSpace1->getAddrSpaceExpr(),
509 DepAddressSpace2->getAddrSpaceExpr()))
510 return false;
511 if (!IsStructurallyEquivalent(Context, DepAddressSpace1->getPointeeType(),
512 DepAddressSpace2->getPointeeType()))
513 return false;
514
515 break;
516 }
517
518 case Type::DependentSizedExtVector: {
519 const auto *Vec1 = cast<DependentSizedExtVectorType>(T1);
520 const auto *Vec2 = cast<DependentSizedExtVectorType>(T2);
521 if (!IsStructurallyEquivalent(Context, Vec1->getSizeExpr(),
522 Vec2->getSizeExpr()))
523 return false;
524 if (!IsStructurallyEquivalent(Context, Vec1->getElementType(),
525 Vec2->getElementType()))
526 return false;
527 break;
528 }
529
530 case Type::DependentVector: {
531 const auto *Vec1 = cast<DependentVectorType>(T1);
532 const auto *Vec2 = cast<DependentVectorType>(T2);
533 if (Vec1->getVectorKind() != Vec2->getVectorKind())
534 return false;
535 if (!IsStructurallyEquivalent(Context, Vec1->getSizeExpr(),
536 Vec2->getSizeExpr()))
537 return false;
538 if (!IsStructurallyEquivalent(Context, Vec1->getElementType(),
539 Vec2->getElementType()))
540 return false;
541 break;
542 }
543
544 case Type::Vector:
545 case Type::ExtVector: {
546 const auto *Vec1 = cast<VectorType>(T1);
547 const auto *Vec2 = cast<VectorType>(T2);
548 if (!IsStructurallyEquivalent(Context, Vec1->getElementType(),
549 Vec2->getElementType()))
550 return false;
551 if (Vec1->getNumElements() != Vec2->getNumElements())
552 return false;
553 if (Vec1->getVectorKind() != Vec2->getVectorKind())
554 return false;
555 break;
556 }
557
558 case Type::FunctionProto: {
559 const auto *Proto1 = cast<FunctionProtoType>(T1);
560 const auto *Proto2 = cast<FunctionProtoType>(T2);
561
562 if (Proto1->getNumParams() != Proto2->getNumParams())
563 return false;
564 for (unsigned I = 0, N = Proto1->getNumParams(); I != N; ++I) {
565 if (!IsStructurallyEquivalent(Context, Proto1->getParamType(I),
566 Proto2->getParamType(I)))
567 return false;
568 }
569 if (Proto1->isVariadic() != Proto2->isVariadic())
570 return false;
571
572 if (Proto1->getMethodQuals() != Proto2->getMethodQuals())
573 return false;
574
575 // Check exceptions, this information is lost in canonical type.
576 const auto *OrigProto1 =
577 cast<FunctionProtoType>(OrigT1.getDesugaredType(Context.FromCtx));
578 const auto *OrigProto2 =
579 cast<FunctionProtoType>(OrigT2.getDesugaredType(Context.ToCtx));
580 if (!IsEquivalentExceptionSpec(Context, OrigProto1, OrigProto2))
581 return false;
582
583 // Fall through to check the bits common with FunctionNoProtoType.
584 LLVM_FALLTHROUGH;
585 }
586
587 case Type::FunctionNoProto: {
588 const auto *Function1 = cast<FunctionType>(T1);
589 const auto *Function2 = cast<FunctionType>(T2);
590 if (!IsStructurallyEquivalent(Context, Function1->getReturnType(),
591 Function2->getReturnType()))
592 return false;
593 if (!IsStructurallyEquivalent(Context, Function1->getExtInfo(),
594 Function2->getExtInfo()))
595 return false;
596 break;
597 }
598
599 case Type::UnresolvedUsing:
600 if (!IsStructurallyEquivalent(Context,
601 cast<UnresolvedUsingType>(T1)->getDecl(),
602 cast<UnresolvedUsingType>(T2)->getDecl()))
603 return false;
604 break;
605
606 case Type::Attributed:
607 if (!IsStructurallyEquivalent(Context,
608 cast<AttributedType>(T1)->getModifiedType(),
609 cast<AttributedType>(T2)->getModifiedType()))
610 return false;
611 if (!IsStructurallyEquivalent(
612 Context, cast<AttributedType>(T1)->getEquivalentType(),
613 cast<AttributedType>(T2)->getEquivalentType()))
614 return false;
615 break;
616
617 case Type::Paren:
618 if (!IsStructurallyEquivalent(Context, cast<ParenType>(T1)->getInnerType(),
619 cast<ParenType>(T2)->getInnerType()))
620 return false;
621 break;
622
623 case Type::MacroQualified:
624 if (!IsStructurallyEquivalent(
625 Context, cast<MacroQualifiedType>(T1)->getUnderlyingType(),
626 cast<MacroQualifiedType>(T2)->getUnderlyingType()))
627 return false;
628 break;
629
630 case Type::Typedef:
631 if (!IsStructurallyEquivalent(Context, cast<TypedefType>(T1)->getDecl(),
632 cast<TypedefType>(T2)->getDecl()))
633 return false;
634 break;
635
636 case Type::TypeOfExpr:
637 if (!IsStructurallyEquivalent(
638 Context, cast<TypeOfExprType>(T1)->getUnderlyingExpr(),
639 cast<TypeOfExprType>(T2)->getUnderlyingExpr()))
640 return false;
641 break;
642
643 case Type::TypeOf:
644 if (!IsStructurallyEquivalent(Context,
645 cast<TypeOfType>(T1)->getUnderlyingType(),
646 cast<TypeOfType>(T2)->getUnderlyingType()))
647 return false;
648 break;
649
650 case Type::UnaryTransform:
651 if (!IsStructurallyEquivalent(
652 Context, cast<UnaryTransformType>(T1)->getUnderlyingType(),
653 cast<UnaryTransformType>(T2)->getUnderlyingType()))
654 return false;
655 break;
656
657 case Type::Decltype:
658 if (!IsStructurallyEquivalent(Context,
659 cast<DecltypeType>(T1)->getUnderlyingExpr(),
660 cast<DecltypeType>(T2)->getUnderlyingExpr()))
661 return false;
662 break;
663
664 case Type::Auto:
665 if (!IsStructurallyEquivalent(Context, cast<AutoType>(T1)->getDeducedType(),
666 cast<AutoType>(T2)->getDeducedType()))
667 return false;
668 break;
669
670 case Type::DeducedTemplateSpecialization: {
671 const auto *DT1 = cast<DeducedTemplateSpecializationType>(T1);
672 const auto *DT2 = cast<DeducedTemplateSpecializationType>(T2);
673 if (!IsStructurallyEquivalent(Context, DT1->getTemplateName(),
674 DT2->getTemplateName()))
675 return false;
676 if (!IsStructurallyEquivalent(Context, DT1->getDeducedType(),
677 DT2->getDeducedType()))
678 return false;
679 break;
680 }
681
682 case Type::Record:
683 case Type::Enum:
684 if (!IsStructurallyEquivalent(Context, cast<TagType>(T1)->getDecl(),
685 cast<TagType>(T2)->getDecl()))
686 return false;
687 break;
688
689 case Type::TemplateTypeParm: {
690 const auto *Parm1 = cast<TemplateTypeParmType>(T1);
691 const auto *Parm2 = cast<TemplateTypeParmType>(T2);
692 if (Parm1->getDepth() != Parm2->getDepth())
693 return false;
694 if (Parm1->getIndex() != Parm2->getIndex())
695 return false;
696 if (Parm1->isParameterPack() != Parm2->isParameterPack())
697 return false;
698
699 // Names of template type parameters are never significant.
700 break;
701 }
702
703 case Type::SubstTemplateTypeParm: {
704 const auto *Subst1 = cast<SubstTemplateTypeParmType>(T1);
705 const auto *Subst2 = cast<SubstTemplateTypeParmType>(T2);
706 if (!IsStructurallyEquivalent(Context,
707 QualType(Subst1->getReplacedParameter(), 0),
708 QualType(Subst2->getReplacedParameter(), 0)))
709 return false;
710 if (!IsStructurallyEquivalent(Context, Subst1->getReplacementType(),
711 Subst2->getReplacementType()))
712 return false;
713 break;
714 }
715
716 case Type::SubstTemplateTypeParmPack: {
717 const auto *Subst1 = cast<SubstTemplateTypeParmPackType>(T1);
718 const auto *Subst2 = cast<SubstTemplateTypeParmPackType>(T2);
719 if (!IsStructurallyEquivalent(Context,
720 QualType(Subst1->getReplacedParameter(), 0),
721 QualType(Subst2->getReplacedParameter(), 0)))
722 return false;
723 if (!IsStructurallyEquivalent(Context, Subst1->getArgumentPack(),
724 Subst2->getArgumentPack()))
725 return false;
726 break;
727 }
728
729 case Type::TemplateSpecialization: {
730 const auto *Spec1 = cast<TemplateSpecializationType>(T1);
731 const auto *Spec2 = cast<TemplateSpecializationType>(T2);
732 if (!IsStructurallyEquivalent(Context, Spec1->getTemplateName(),
733 Spec2->getTemplateName()))
734 return false;
735 if (Spec1->getNumArgs() != Spec2->getNumArgs())
736 return false;
737 for (unsigned I = 0, N = Spec1->getNumArgs(); I != N; ++I) {
738 if (!IsStructurallyEquivalent(Context, Spec1->getArg(I),
739 Spec2->getArg(I)))
740 return false;
741 }
742 break;
743 }
744
745 case Type::Elaborated: {
746 const auto *Elab1 = cast<ElaboratedType>(T1);
747 const auto *Elab2 = cast<ElaboratedType>(T2);
748 // CHECKME: what if a keyword is ETK_None or ETK_typename ?
749 if (Elab1->getKeyword() != Elab2->getKeyword())
750 return false;
751 if (!IsStructurallyEquivalent(Context, Elab1->getQualifier(),
752 Elab2->getQualifier()))
753 return false;
754 if (!IsStructurallyEquivalent(Context, Elab1->getNamedType(),
755 Elab2->getNamedType()))
756 return false;
757 break;
758 }

760 case Type::InjectedClassName: {
761 const auto *Inj1 = cast<InjectedClassNameType>(T1);
762 const auto *Inj2 = cast<InjectedClassNameType>(T2);
763 if (!IsStructurallyEquivalent(Context,
764 Inj1->getInjectedSpecializationType(),
765 Inj2->getInjectedSpecializationType()))
766 return false;
767 break;
768 }

770 case Type::DependentName: {
771 const auto *Typename1 = cast<DependentNameType>(T1);
772 const auto *Typename2 = cast<DependentNameType>(T2);
773 if (!IsStructurallyEquivalent(Context, Typename1->getQualifier(),
774 Typename2->getQualifier()))
775 return false;
776 if (!IsStructurallyEquivalent(Typename1->getIdentifier(),
777 Typename2->getIdentifier()))
778 return false;

780 break;
781 }

784 const auto *Spec1 = cast<DependentTemplateSpecializationType>(T1);
785 const auto *Spec2 = cast<DependentTemplateSpecializationType>(T2);
786 if (!IsStructurallyEquivalent(Context, Spec1->getQualifier(),
787 Spec2->getQualifier()))
788 return false;
789 if (!IsStructurallyEquivalent(Spec1->getIdentifier(),
790 Spec2->getIdentifier()))
791 return false;

802 case Type::PackExpansion:
803 if (!IsStructurallyEquivalent(Context,
804 cast<PackExpansionType>(T1)->getPattern(),
805 cast<PackExpansionType>(T2)->getPattern()))
806 return false;
807 break;
808
809 case Type::ObjCInterface: {
810 const auto *Iface1 = cast<ObjCInterfaceType>(T1);
811 const auto *Iface2 = cast<ObjCInterfaceType>(T2);
812 if (!IsStructurallyEquivalent(Context, Iface1->getDecl(),
813 Iface2->getDecl()))
814 return false;
815 break;
816 }
817
818 case Type::ObjCTypeParam: {
819 const auto *Obj1 = cast<ObjCTypeParamType>(T1);
820 const auto *Obj2 = cast<ObjCTypeParamType>(T2);

822 return false;
823

825 return false;

828 Obj2->getProtocol(I)))
829 return false;
830 }
831 break;
832 }

834 case Type::ObjCObject: {

838 Obj2->getBaseType()))
839 return false;
840 if (Obj1->getNumProtocols() != Obj2->getNumProtocols())

867 cast<PipeType>(T2)->getElementType()))

869 break;

875/// Determine structural equivalence of two fields.
876static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
877 FieldDecl *Field1, FieldDecl *Field2) {
878 const auto *Owner2 = cast<RecordDecl>(Field2->getDeclContext());
879
880 // For anonymous structs/unions, match up the anonymous struct/union type
881 // declarations directly, so that we don't go off searching for anonymous
882 // types
883 if (Field1->isAnonymousStructOrUnion() &&
884 Field2->isAnonymousStructOrUnion()) {
885 RecordDecl *D1 = Field1->getType()->castAs<RecordType>()->getDecl();
886 RecordDecl *D2 = Field2->getType()->castAs<RecordType>()->getDecl();
887 return IsStructurallyEquivalent(Context, D1, D2);
888 }
889
890 // Check for equivalent field names.
891 IdentifierInfo *Name1 = Field1->getIdentifier();
892 IdentifierInfo *Name2 = Field2->getIdentifier();
893 if (!::IsStructurallyEquivalent(Name1, Name2)) {
894 if (Context.Complain) {
895 Context.Diag2(
896 Owner2->getLocation(),
897 Context.getApplicableDiagnostic(diag::err_odr_tag_type_inconsistent))
898 << Context.ToCtx.getTypeDeclType(Owner2);
899 Context.Diag2(Field2->getLocation(), diag::note_odr_field_name)
900 << Field2->getDeclName();
901 Context.Diag1(Field1->getLocation(), diag::note_odr_field_name)
902 << Field1->getDeclName();
903 }
904 return false;
905 }
906
907 if (!IsStructurallyEquivalent(Context, Field1->getType(),
908 Field2->getType())) {
909 if (Context.Complain) {
910 Context.Diag2(
911 Owner2->getLocation(),
912 Context.getApplicableDiagnostic(diag::err_odr_tag_type_inconsistent))
913 << Context.ToCtx.getTypeDeclType(Owner2);
914 Context.Diag2(Field2->getLocation(), diag::note_odr_field)
915 << Field2->getDeclName() << Field2->getType();
916 Context.Diag1(Field1->getLocation(), diag::note_odr_field)
917 << Field1->getDeclName() << Field1->getType();
918 }
919 return false;
920 }
921
922 if (Field1->isBitField() != Field2->isBitField()) {
923 if (Context.Complain) {
924 Context.Diag2(
925 Owner2->getLocation(),
926 Context.getApplicableDiagnostic(diag::err_odr_tag_type_inconsistent))
927 << Context.ToCtx.getTypeDeclType(Owner2);
928 if (Field1->isBitField()) {
929 Context.Diag1(Field1->getLocation(), diag::note_odr_bit_field)
930 << Field1->getDeclName() << Field1->getType()
931 << Field1->getBitWidthValue(Context.FromCtx);
932 Context.Diag2(Field2->getLocation(), diag::note_odr_not_bit_field)
933 << Field2->getDeclName();
934 } else {
935 Context.Diag2(Field2->getLocation(), diag::note_odr_bit_field)
936 << Field2->getDeclName() << Field2->getType()
937 << Field2->getBitWidthValue(Context.ToCtx);
938 Context.Diag1(Field1->getLocation(), diag::note_odr_not_bit_field)
939 << Field1->getDeclName();
940 }
941 }
942 return false;
943 }
944
945 if (Field1->isBitField()) {
946 // Make sure that the bit-fields are the same length.
947 unsigned Bits1 = Field1->getBitWidthValue(Context.FromCtx);
948 unsigned Bits2 = Field2->getBitWidthValue(Context.ToCtx);
949
950 if (Bits1 != Bits2) {
951 if (Context.Complain) {
952 Context.Diag2(Owner2->getLocation(),
953 Context.getApplicableDiagnostic(
954 diag::err_odr_tag_type_inconsistent))
955 << Context.ToCtx.getTypeDeclType(Owner2);
956 Context.Diag2(Field2->getLocation(), diag::note_odr_bit_field)
957 << Field2->getDeclName() << Field2->getType() << Bits2;
958 Context.Diag1(Field1->getLocation(), diag::note_odr_bit_field)
959 << Field1->getDeclName() << Field1->getType() << Bits1;
960 }
961 return false;
962 }
963 }
964
965 return true;
966}

968/// Determine structural equivalence of two methods.
969static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
970 CXXMethodDecl *Method1,
971 CXXMethodDecl *Method2) {
972 bool PropertiesEqual =
973 Method1->getDeclKind() == Method2->getDeclKind() &&
974 Method1->getRefQualifier() == Method2->getRefQualifier() &&
975 Method1->getAccess() == Method2->getAccess() &&
976 Method1->getOverloadedOperator() == Method2->getOverloadedOperator() &&
977 Method1->isStatic() == Method2->isStatic() &&
978 Method1->isConst() == Method2->isConst() &&
979 Method1->isVolatile() == Method2->isVolatile() &&
980 Method1->isVirtual() == Method2->isVirtual() &&
981 Method1->isPure() == Method2->isPure() &&
982 Method1->isDefaulted() == Method2->isDefaulted() &&
983 Method1->isDeleted() == Method2->isDeleted();
984 if (!PropertiesEqual)
985 return false;
986 // FIXME: Check for 'final'.
987
988 if (auto *Constructor1 = dyn_cast<CXXConstructorDecl>(Method1)) {
989 auto *Constructor2 = cast<CXXConstructorDecl>(Method2);
990 if (!Constructor1->getExplicitSpecifier().isEquivalent(
991 Constructor2->getExplicitSpecifier()))
992 return false;
993 }
994
995 if (auto *Conversion1 = dyn_cast<CXXConversionDecl>(Method1)) {
996 auto *Conversion2 = cast<CXXConversionDecl>(Method2);
997 if (!Conversion1->getExplicitSpecifier().isEquivalent(
998 Conversion2->getExplicitSpecifier()))
999 return false;
1000 if (!IsStructurallyEquivalent(Context, Conversion1->getConversionType(),
1001 Conversion2->getConversionType()))
1002 return false;
1003 }
1004
1005 const IdentifierInfo *Name1 = Method1->getIdentifier();
1006 const IdentifierInfo *Name2 = Method2->getIdentifier();
1007 if (!::IsStructurallyEquivalent(Name1, Name2)) {
1008 return false;
1009 // TODO: Names do not match, add warning like at check for FieldDecl.
1010 }
1011
1012 // Check the prototypes.
1013 if (!::IsStructurallyEquivalent(Context,
1014 Method1->getType(), Method2->getType()))
1015 return false;
1016
1017 return true;
1018}

1020/// Determine structural equivalence of two records.
1021static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
1022 RecordDecl *D1, RecordDecl *D2) {
1023 if (D1->isUnion() != D2->isUnion()) {
1024 if (Context.Complain) {
1025 Context.Diag2(D2->getLocation(), Context.getApplicableDiagnostic(
1026 diag::err_odr_tag_type_inconsistent))
1027 << Context.ToCtx.getTypeDeclType(D2);
1028 Context.Diag1(D1->getLocation(), diag::note_odr_tag_kind_here)
1029 << D1->getDeclName() << (unsigned)D1->getTagKind();
1030 }
1031 return false;
1032 }
1033
1034 if (!D1->getDeclName() && !D2->getDeclName()) {
1035 // If both anonymous structs/unions are in a record context, make sure
1036 // they occur in the same location in the context records.
1037 if (Optional<unsigned> Index1 =
1038 StructuralEquivalenceContext::findUntaggedStructOrUnionIndex(D1)) {
1039 if (Optional<unsigned> Index2 =
1040 StructuralEquivalenceContext::findUntaggedStructOrUnionIndex(
1041 D2)) {
1042 if (*Index1 != *Index2)
1043 return false;
1044 }
1045 }
1046 }
1047
1048 // If both declarations are class template specializations, we know
1049 // the ODR applies, so check the template and template arguments.
1050 const auto *Spec1 = dyn_cast<ClassTemplateSpecializationDecl>(D1);
1051 const auto *Spec2 = dyn_cast<ClassTemplateSpecializationDecl>(D2);
1052 if (Spec1 && Spec2) {
1053 // Check that the specialized templates are the same.
1054 if (!IsStructurallyEquivalent(Context, Spec1->getSpecializedTemplate(),
1055 Spec2->getSpecializedTemplate()))
1056 return false;
1057
1058 // Check that the template arguments are the same.
1059 if (Spec1->getTemplateArgs().size() != Spec2->getTemplateArgs().size())
1060 return false;
1061
1062 for (unsigned I = 0, N = Spec1->getTemplateArgs().size(); I != N; ++I)
1063 if (!IsStructurallyEquivalent(Context, Spec1->getTemplateArgs().get(I),
1064 Spec2->getTemplateArgs().get(I)))
1065 return false;
1066 }
1067 // If one is a class template specialization and the other is not, these
1068 // structures are different.
1069 else if (Spec1 || Spec2)
1070 return false;
1071
1072 // Compare the definitions of these two records. If either or both are
1073 // incomplete (i.e. it is a forward decl), we assume that they are
1074 // equivalent.
1075 D1 = D1->getDefinition();
1076 D2 = D2->getDefinition();
1077 if (!D1 || !D2)
1078 return true;
1079
1080 // If any of the records has external storage and we do a minimal check (or
1081 // AST import) we assume they are equivalent. (If we didn't have this
1082 // assumption then `RecordDecl::LoadFieldsFromExternalStorage` could trigger
1083 // another AST import which in turn would call the structural equivalency
1084 // check again and finally we'd have an improper result.)
1085 if (Context.EqKind == StructuralEquivalenceKind::Minimal)
1086 if (D1->hasExternalLexicalStorage() || D2->hasExternalLexicalStorage())
1087 return true;
1088
1089 // If one definition is currently being defined, we do not compare for
1090 // equality and we assume that the decls are equal.
1091 if (D1->isBeingDefined() || D2->isBeingDefined())
1092 return true;
1093
1094 if (auto *D1CXX = dyn_cast<CXXRecordDecl>(D1)) {
1095 if (auto *D2CXX = dyn_cast<CXXRecordDecl>(D2)) {
1096 if (D1CXX->hasExternalLexicalStorage() &&
1097 !D1CXX->isCompleteDefinition()) {
1098 D1CXX->getASTContext().getExternalSource()->CompleteType(D1CXX);
1099 }
1100
1101 if (D1CXX->getNumBases() != D2CXX->getNumBases()) {
1102 if (Context.Complain) {
1103 Context.Diag2(D2->getLocation(),
1104 Context.getApplicableDiagnostic(
1105 diag::err_odr_tag_type_inconsistent))
1106 << Context.ToCtx.getTypeDeclType(D2);
1107 Context.Diag2(D2->getLocation(), diag::note_odr_number_of_bases)
1108 << D2CXX->getNumBases();
1109 Context.Diag1(D1->getLocation(), diag::note_odr_number_of_bases)
1110 << D1CXX->getNumBases();
1111 }
1112 return false;
1113 }
1114
1115 // Check the base classes.
1116 for (CXXRecordDecl::base_class_iterator Base1 = D1CXX->bases_begin(),
1117 BaseEnd1 = D1CXX->bases_end(),
1118 Base2 = D2CXX->bases_begin();
1119 Base1 != BaseEnd1; ++Base1, ++Base2) {
1120 if (!IsStructurallyEquivalent(Context, Base1->getType(),
1121 Base2->getType())) {
1122 if (Context.Complain) {
1123 Context.Diag2(D2->getLocation(),
1124 Context.getApplicableDiagnostic(
1125 diag::err_odr_tag_type_inconsistent))
1126 << Context.ToCtx.getTypeDeclType(D2);
1127 Context.Diag2(Base2->getBeginLoc(), diag::note_odr_base)
1128 << Base2->getType() << Base2->getSourceRange();
1129 Context.Diag1(Base1->getBeginLoc(), diag::note_odr_base)
1130 << Base1->getType() << Base1->getSourceRange();
1131 }
1132 return false;
1133 }
1134
1135 // Check virtual vs. non-virtual inheritance mismatch.
1136 if (Base1->isVirtual() != Base2->isVirtual()) {
1137 if (Context.Complain) {
1138 Context.Diag2(D2->getLocation(),
1139 Context.getApplicableDiagnostic(
1140 diag::err_odr_tag_type_inconsistent))
1141 << Context.ToCtx.getTypeDeclType(D2);
1142 Context.Diag2(Base2->getBeginLoc(), diag::note_odr_virtual_base)
1143 << Base2->isVirtual() << Base2->getSourceRange();
1144 Context.Diag1(Base1->getBeginLoc(), diag::note_odr_base)
1145 << Base1->isVirtual() << Base1->getSourceRange();
1146 }
1147 return false;
1148 }
1149 }
1150
1151 // Check the friends for consistency.
1152 CXXRecordDecl::friend_iterator Friend2 = D2CXX->friend_begin(),
1153 Friend2End = D2CXX->friend_end();
1154 for (CXXRecordDecl::friend_iterator Friend1 = D1CXX->friend_begin(),
1155 Friend1End = D1CXX->friend_end();
1156 Friend1 != Friend1End; ++Friend1, ++Friend2) {
1157 if (Friend2 == Friend2End) {
1158 if (Context.Complain) {
1159 Context.Diag2(D2->getLocation(),
1160 Context.getApplicableDiagnostic(
1161 diag::err_odr_tag_type_inconsistent))
1162 << Context.ToCtx.getTypeDeclType(D2CXX);
1163 Context.Diag1((*Friend1)->getFriendLoc(), diag::note_odr_friend);
1164 Context.Diag2(D2->getLocation(), diag::note_odr_missing_friend);
1165 }
1166 return false;
1167 }
1168
1169 if (!IsStructurallyEquivalent(Context, *Friend1, *Friend2)) {
1170 if (Context.Complain) {
1171 Context.Diag2(D2->getLocation(),
1172 Context.getApplicableDiagnostic(
1173 diag::err_odr_tag_type_inconsistent))
1174 << Context.ToCtx.getTypeDeclType(D2CXX);
1175 Context.Diag1((*Friend1)->getFriendLoc(), diag::note_odr_friend);
1176 Context.Diag2((*Friend2)->getFriendLoc(), diag::note_odr_friend);
1177 }
1178 return false;
1179 }
1180 }
1181
1182 if (Friend2 != Friend2End) {
1183 if (Context.Complain) {
1184 Context.Diag2(D2->getLocation(),
1185 Context.getApplicableDiagnostic(
1186 diag::err_odr_tag_type_inconsistent))
1187 << Context.ToCtx.getTypeDeclType(D2);
1188 Context.Diag2((*Friend2)->getFriendLoc(), diag::note_odr_friend);
1189 Context.Diag1(D1->getLocation(), diag::note_odr_missing_friend);
1190 }
1191 return false;
1192 }
1193 } else if (D1CXX->getNumBases() > 0) {
1194 if (Context.Complain) {
1195 Context.Diag2(D2->getLocation(),
1196 Context.getApplicableDiagnostic(
1197 diag::err_odr_tag_type_inconsistent))
1198 << Context.ToCtx.getTypeDeclType(D2);
1199 const CXXBaseSpecifier *Base1 = D1CXX->bases_begin();
1200 Context.Diag1(Base1->getBeginLoc(), diag::note_odr_base)
1201 << Base1->getType() << Base1->getSourceRange();
1202 Context.Diag2(D2->getLocation(), diag::note_odr_missing_base);
1203 }
1204 return false;
1205 }
1206 }
1207
1208 // Check the fields for consistency.
1209 RecordDecl::field_iterator Field2 = D2->field_begin(),
1210 Field2End = D2->field_end();
1211 for (RecordDecl::field_iterator Field1 = D1->field_begin(),
1212 Field1End = D1->field_end();
1213 Field1 != Field1End; ++Field1, ++Field2) {
1214 if (Field2 == Field2End) {
1215 if (Context.Complain) {
1216 Context.Diag2(D2->getLocation(),
1217 Context.getApplicableDiagnostic(
1218 diag::err_odr_tag_type_inconsistent))
1219 << Context.ToCtx.getTypeDeclType(D2);
1220 Context.Diag1(Field1->getLocation(), diag::note_odr_field)
1221 << Field1->getDeclName() << Field1->getType();
1222 Context.Diag2(D2->getLocation(), diag::note_odr_missing_field);
1223 }
1224 return false;
1225 }
1226
1227 if (!IsStructurallyEquivalent(Context, *Field1, *Field2))
1228 return false;
1229 }
1230
1231 if (Field2 != Field2End) {
1232 if (Context.Complain) {
1233 Context.Diag2(D2->getLocation(), Context.getApplicableDiagnostic(
1234 diag::err_odr_tag_type_inconsistent))
1235 << Context.ToCtx.getTypeDeclType(D2);
1236 Context.Diag2(Field2->getLocation(), diag::note_odr_field)
1237 << Field2->getDeclName() << Field2->getType();
1238 Context.Diag1(D1->getLocation(), diag::note_odr_missing_field);
1239 }
1240 return false;
1241 }
1242
1243 return true;
1244}

1246/// Determine structural equivalence of two enums.
1247static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
1248 EnumDecl *D1, EnumDecl *D2) {
1249
1250 // Compare the definitions of these two enums. If either or both are
1251 // incomplete (i.e. forward declared), we assume that they are equivalent.
1252 D1 = D1->getDefinition();
1253 D2 = D2->getDefinition();
1254 if (!D1 || !D2)
1255 return true;
1256
1257 EnumDecl::enumerator_iterator EC2 = D2->enumerator_begin(),
1258 EC2End = D2->enumerator_end();
1259 for (EnumDecl::enumerator_iterator EC1 = D1->enumerator_begin(),
1260 EC1End = D1->enumerator_end();
1261 EC1 != EC1End; ++EC1, ++EC2) {
1262 if (EC2 == EC2End) {
1263 if (Context.Complain) {
1264 Context.Diag2(D2->getLocation(),
1265 Context.getApplicableDiagnostic(
1266 diag::err_odr_tag_type_inconsistent))
1267 << Context.ToCtx.getTypeDeclType(D2);
1268 Context.Diag1(EC1->getLocation(), diag::note_odr_enumerator)
1269 << EC1->getDeclName() << EC1->getInitVal().toString(10);
1270 Context.Diag2(D2->getLocation(), diag::note_odr_missing_enumerator);
1271 }
1272 return false;
1273 }
1274
1275 llvm::APSInt Val1 = EC1->getInitVal();
1276 llvm::APSInt Val2 = EC2->getInitVal();
1277 if (!llvm::APSInt::isSameValue(Val1, Val2) ||
1278 !IsStructurallyEquivalent(EC1->getIdentifier(), EC2->getIdentifier())) {

1280 Context.Diag2(D2->getLocation(),
1281 Context.getApplicableDiagnostic(
1282 diag::err_odr_tag_type_inconsistent))
1283 << Context.ToCtx.getTypeDeclType(D2);
1284 Context.Diag2(EC2->getLocation(), diag::note_odr_enumerator)
1285 << EC2->getDeclName() << EC2->getInitVal().toString(10);
1286 Context.Diag1(EC1->getLocation(), diag::note_odr_enumerator)
1287 << EC1->getDeclName() << EC1->getInitVal().toString(10);
1288 }
1289 return false;
1290 }

1472/// Determine structural equivalence of two declarations.
1473static bool IsStructurallyEquivalent(StructuralEquivalenceContext &Context,
1474 Decl *D1, Decl *D2) {
1475 // FIXME: Check for known structural equivalences via a callback of some sort.
1476
1477 // Check whether we already know that these two declarations are not
1478 // structurally equivalent.
1479 if (Context.NonEquivalentDecls.count(
1480 std::make_pair(D1->getCanonicalDecl(), D2->getCanonicalDecl())))
1481 return false;
1482
1483 // Determine whether we've already produced a tentative equivalence for D1.
1484 Decl *&EquivToD1 = Context.TentativeEquivalences[D1->getCanonicalDecl()];
1485 if (EquivToD1)
1486 return EquivToD1 == D2->getCanonicalDecl();
1487
1488 // Produce a tentative equivalence D1 <-> D2, which will be checked later.
1489 EquivToD1 = D2->getCanonicalDecl();
1490 Context.DeclsToCheck.push_back(D1->getCanonicalDecl());
1491 return true;
1492}

1512Optional<unsigned>
1513StructuralEquivalenceContext::findUntaggedStructOrUnionIndex(RecordDecl *Anon) {
1514 ASTContext &Context = Anon->getASTContext();
1515 QualType AnonTy = Context.getRecordType(Anon);
1516
1517 const auto *Owner = dyn_cast<RecordDecl>(Anon->getDeclContext());
1518 if (!Owner)
1519 return None;
1520
1521 unsigned Index = 0;
1522 for (const auto *D : Owner->noload_decls()) {
1523 const auto *F = dyn_cast<FieldDecl>(D);
1524 if (!F)
1525 continue;
1526
1527 if (F->isAnonymousStructOrUnion()) {
1528 if (Context.hasSameType(F->getType(), AnonTy))
1529 break;
1530 ++Index;
1531 continue;
1532 }
1533
1534 // If the field looks like this:
1535 // struct { ... } A;
1536 QualType FieldType = F->getType();
1537 // In case of nested structs.
1538 while (const auto *ElabType = dyn_cast<ElaboratedType>(FieldType))
1539 FieldType = ElabType->getNamedType();
1540
1541 if (const auto *RecType = dyn_cast<RecordType>(FieldType)) {
1542 const RecordDecl *RecDecl = RecType->getDecl();
1543 if (RecDecl->getDeclContext() == Owner && !RecDecl->getIdentifier()) {
1544 if (Context.hasSameType(FieldType, AnonTy))
1545 break;
1546 ++Index;
1547 continue;
1548 }
1549 }
1550 }
1551
1552 return Index;
1553}

1604 // never call the public ASTStructuralEquivalence::IsEquivalent() functions,

1643 // FIXME: Switch on all declaration kinds. For now, we're just going to
1644 // check the obvious ones.
1645 if (auto *Record1 = dyn_cast<RecordDecl>(D1)) {
1646 if (auto *Record2 = dyn_cast<RecordDecl>(D2)) {
1647 // Check for equivalent structure names.
1648 IdentifierInfo *Name1 = Record1->getIdentifier();
1649 if (!Name1 && Record1->getTypedefNameForAnonDecl())
1650 Name1 = Record1->getTypedefNameForAnonDecl()->getIdentifier();
1651 IdentifierInfo *Name2 = Record2->getIdentifier();
1652 if (!Name2 && Record2->getTypedefNameForAnonDecl())
1653 Name2 = Record2->getTypedefNameForAnonDecl()->getIdentifier();
1654 if (!::IsStructurallyEquivalent(Name1, Name2) ||
1655 !::IsStructurallyEquivalent(*this, Record1, Record2))
1656 return false;
1657 } else {
1658 // Record/non-record mismatch.
1659 return false;
1660 }
1661 } else if (auto *Enum1 = dyn_cast<EnumDecl>(D1)) {
1662 if (auto *Enum2 = dyn_cast<EnumDecl>(D2)) {
1663 // Check for equivalent enum names.
1664 IdentifierInfo *Name1 = Enum1->getIdentifier();
1665 if (!Name1 && Enum1->getTypedefNameForAnonDecl())
1666 Name1 = Enum1->getTypedefNameForAnonDecl()->getIdentifier();
1667 IdentifierInfo *Name2 = Enum2->getIdentifier();
1668 if (!Name2 && Enum2->getTypedefNameForAnonDecl())
1669 Name2 = Enum2->getTypedefNameForAnonDecl()->getIdentifier();
1670 if (!::IsStructurallyEquivalent(Name1, Name2) ||
1671 !::IsStructurallyEquivalent(*this, Enum1, Enum2))
1672 return false;
1673 } else {
1674 // Enum/non-enum mismatch
1675 return false;
1676 }
1677 } else if (const auto *Typedef1 = dyn_cast<TypedefNameDecl>(D1)) {
1678 if (const auto *Typedef2 = dyn_cast<TypedefNameDecl>(D2)) {
1679 if (!::IsStructurallyEquivalent(Typedef1->getIdentifier(),
1680 Typedef2->getIdentifier()) ||
1681 !::IsStructurallyEquivalent(*this, Typedef1->getUnderlyingType(),
1682 Typedef2->getUnderlyingType()))

1685 // Typedef/non-typedef mismatch.

1689 if (auto *ClassTemplate2 = dyn_cast<ClassTemplateDecl>(D2)) {
1690 if (!::IsStructurallyEquivalent(*this, ClassTemplate1,
1691 ClassTemplate2))
1692 return false;
1693 } else {
1694 // Class template/non-class-template mismatch.
1695 return false;

1698 if (auto *FunctionTemplate2 = dyn_cast<FunctionTemplateDecl>(D2)) {
1699 if (!::IsStructurallyEquivalent(*this, FunctionTemplate1,
1700 FunctionTemplate2))
1701 return false;
1702 } else {
1703 // Class template/non-class-template mismatch.
1704 return false;
1705 }

1707 if (auto *TTP2 = dyn_cast<TemplateTypeParmDecl>(D2)) {

1768bool StructuralEquivalenceContext::Finish() {
1769 while (!DeclsToCheck.empty()) {
1770 // Check the next declaration.
1771 Decl *D1 = DeclsToCheck.front();
1772 DeclsToCheck.pop_front();
1773
1774 Decl *D2 = TentativeEquivalences[D1];
1775 assert(D2 && "Unrecorded tentative equivalence?");
1776
1777 bool Equivalent =
1778 CheckCommonEquivalence(D1, D2) && CheckKindSpecificEquivalence(D1, D2);
1779
1780 if (!Equivalent) {
1781 // Note that these two declarations are not equivalent (and we already
1782 // know about it).
1783 NonEquivalentDecls.insert(
1784 std::make_pair(D1->getCanonicalDecl(), D2->getCanonicalDecl()));
1785 return true;
1786 }
1787 }
1788
1789 return false;
