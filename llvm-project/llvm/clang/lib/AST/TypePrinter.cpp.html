137static void AppendTypeQualList(raw_ostream &OS, unsigned TypeQuals,
138 bool HasRestrictKeyword) {
139 bool appendSpace = false;
140 if (TypeQuals & Qualifiers::Const) {
141 OS << "const";
142 appendSpace = true;
143 }
144 if (TypeQuals & Qualifiers::Volatile) {
145 if (appendSpace) OS << ' ';
146 OS << "volatile";
147 appendSpace = true;
148 }
149 if (TypeQuals & Qualifiers::Restrict) {
150 if (appendSpace) OS << ' ';
151 if (HasRestrictKeyword) {
152 OS << "restrict";
153 } else {
154 OS << "__restrict";
155 }
156 }

176void TypePrinter::print(const Type *T, Qualifiers Quals, raw_ostream &OS,
177 StringRef PlaceHolder) {
178 if (!T) {
179 OS << "NULL TYPE";
180 return;
181 }
182
183 SaveAndRestore<bool> PHVal(HasEmptyPlaceHolder, PlaceHolder.empty());

185 printBefore(T, Quals, OS);
186 OS << PlaceHolder;
187 printAfter(T, Quals, OS);

190bool TypePrinter::canPrefixQualifiers(const Type *T,
191 bool &NeedARCStrongQualifier) {
192 // CanPrefixQualifiers - We prefer to print type qualifiers before the type,
193 // so that we get "const int" instead of "int const", but we can't do this if
194 // the type is complex. For example if the type is "int*", we *must* print
195 // "int * const", printing "const int *" is different. Only do this when the
196 // type expands to a simple string.
197 bool CanPrefixQualifiers = false;
198 NeedARCStrongQualifier = false;
199 Type::TypeClass TC = T->getTypeClass();
200 if (const auto *AT = dyn_cast<AutoType>(T))
201 TC = AT->desugar()->getTypeClass();
202 if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(T))
203 TC = Subst->getReplacementType()->getTypeClass();
204
205 switch (TC) {
206 case Type::Auto:
207 case Type::Builtin:
208 case Type::Complex:
209 case Type::UnresolvedUsing:
210 case Type::Typedef:
211 case Type::TypeOfExpr:
212 case Type::TypeOf:
213 case Type::Decltype:
214 case Type::UnaryTransform:
215 case Type::Record:
216 case Type::Enum:
217 case Type::Elaborated:
218 case Type::TemplateTypeParm:
219 case Type::SubstTemplateTypeParmPack:
220 case Type::DeducedTemplateSpecialization:
221 case Type::TemplateSpecialization:
222 case Type::InjectedClassName:
223 case Type::DependentName:
224 case Type::DependentTemplateSpecialization:
225 case Type::ObjCObject:
226 case Type::ObjCTypeParam:
227 case Type::ObjCInterface:
228 case Type::Atomic:
229 case Type::Pipe:
230 CanPrefixQualifiers = true;
231 break;
232
233 case Type::ObjCObjectPointer:
234 CanPrefixQualifiers = T->isObjCIdType() || T->isObjCClassType() ||
235 T->isObjCQualifiedIdType() || T->isObjCQualifiedClassType();
236 break;
237
238 case Type::ConstantArray:
239 case Type::IncompleteArray:
240 case Type::VariableArray:
241 case Type::DependentSizedArray:
242 NeedARCStrongQualifier = true;
243 LLVM_FALLTHROUGH;
244
245 case Type::Adjusted:
246 case Type::Decayed:
247 case Type::Pointer:
248 case Type::BlockPointer:
249 case Type::LValueReference:
250 case Type::RValueReference:
251 case Type::MemberPointer:
252 case Type::DependentAddressSpace:
253 case Type::DependentVector:
254 case Type::DependentSizedExtVector:
255 case Type::Vector:
256 case Type::ExtVector:
257 case Type::FunctionProto:
258 case Type::FunctionNoProto:
259 case Type::Paren:
260 case Type::PackExpansion:
261 case Type::SubstTemplateTypeParm:
262 case Type::MacroQualified:
263 CanPrefixQualifiers = false;
264 break;
265
266 case Type::Attributed: {
267 // We still want to print the address_space before the type if it is an
268 // address_space attribute.
269 const auto *AttrTy = cast<AttributedType>(T);
270 CanPrefixQualifiers = AttrTy->getAttrKind() == attr::AddressSpace;
271 }
272 }
273
274 return CanPrefixQualifiers;
275}

277void TypePrinter::printBefore(QualType T, raw_ostream &OS) {
278 SplitQualType Split = splitAccordingToPolicy(T, Policy);
279
280 // If we have cv1 T, where T is substituted for cv2 U, only print cv1 - cv2
281 // at this level.
282 Qualifiers Quals = Split.Quals;
283 if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(Split.Ty))
284 Quals -= QualType(Subst, 0).getQualifiers();
285
286 printBefore(Split.Ty, Quals, OS);
287}

289/// Prints the part of the type string before an identifier, e.g. for
290/// "int foo[10]" it prints "int ".
291void TypePrinter::printBefore(const Type *T,Qualifiers Quals, raw_ostream &OS) {
292 if (Policy.SuppressSpecifiers && T->isSpecifierType())
293 return;
294
295 SaveAndRestore<bool> PrevPHIsEmpty(HasEmptyPlaceHolder);
296
297 // Print qualifiers as appropriate.
298
299 bool CanPrefixQualifiers = false;
300 bool NeedARCStrongQualifier = false;
301 CanPrefixQualifiers = canPrefixQualifiers(T, NeedARCStrongQualifier);
302
303 if (CanPrefixQualifiers && !Quals.empty()) {
304 if (NeedARCStrongQualifier) {
305 IncludeStrongLifetimeRAII Strong(Policy);
306 Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
307 } else {
308 Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
309 }
310 }
311
312 bool hasAfterQuals = false;
313 if (!CanPrefixQualifiers && !Quals.empty()) {
314 hasAfterQuals = !Quals.isEmptyWhenPrinted(Policy);
315 if (hasAfterQuals)
316 HasEmptyPlaceHolder = false;
317 }
318
319 switch (T->getTypeClass()) {
320#define ABSTRACT_TYPE(CLASS, PARENT)
321#define TYPE(CLASS, PARENT) case Type::CLASS: \
322 print##CLASS##Before(cast<CLASS##Type>(T), OS); \
323 break;
324#include "clang/AST/TypeNodes.def"
325 }
326
327 // Print __capability
328 if (const PointerType *PTy = dyn_cast<PointerType>(T)) {
329 if (PTy->isCHERICapability()) {
330 OS << " __capability";
331 if (hasAfterQuals || !PrevPHIsEmpty.get())
332 OS << " ";
333 }
334 }
335 else if (const ReferenceType *RTy = dyn_cast<ReferenceType>(T)) {
336 if (RTy->isCHERICapability()) {
337 OS << " __capability";
338 if (hasAfterQuals || !PrevPHIsEmpty.get())
339 OS << " ";
340 }
341 }
342
343 if (hasAfterQuals) {
344 if (NeedARCStrongQualifier) {
345 IncludeStrongLifetimeRAII Strong(Policy);
346 Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
347 } else {
348 Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
349 }
350 }
351}

637void TypePrinter::printVectorBefore(const VectorType *T, raw_ostream &OS) {
638 switch (T->getVectorKind()) {
639 case VectorType::AltiVecPixel:
640 OS << "__vector __pixel ";
641 break;
642 case VectorType::AltiVecBool:
643 OS << "__vector __bool ";
644 printBefore(T->getElementType(), OS);
645 break;
646 case VectorType::AltiVecVector:
647 OS << "__vector ";
648 printBefore(T->getElementType(), OS);
649 break;
650 case VectorType::NeonVector:
651 OS << "__attribute__((neon_vector_type("
652 << T->getNumElements() << "))) ";
653 printBefore(T->getElementType(), OS);
654 break;
655 case VectorType::NeonPolyVector:
656 OS << "__attribute__((neon_polyvector_type(" <<
657 T->getNumElements() << "))) ";
658 printBefore(T->getElementType(), OS);
659 break;
660 case VectorType::GenericVector: {
661 // FIXME: We prefer to print the size directly here, but have no way
662 // to get the size of the type.
663 OS << "__attribute__((__vector_size__("
664 << T->getNumElements()
665 << " * sizeof(";
666 print(T->getElementType(), OS, StringRef());
667 OS << ")))) ";
668 printBefore(T->getElementType(), OS);
669 break;
670 }
671 }

800 // If needed for precedence reasons, wrap the inner part in grouping parens.
801 if (!HasEmptyPlaceHolder)

1134/// Appends the given scope to the end of a string.
1135void TypePrinter::AppendScope(DeclContext *DC, raw_ostream &OS) {
1136 if (DC->isTranslationUnit()) return;
1137 if (DC->isFunctionOrMethod()) return;
1138 AppendScope(DC->getParent(), OS);
1139
1140 if (const auto *NS = dyn_cast<NamespaceDecl>(DC)) {
1141 if (Policy.SuppressUnwrittenScope &&
1142 (NS->isAnonymousNamespace() || NS->isInline()))
1143 return;
1144 if (NS->getIdentifier())
1145 OS << NS->getName() << "::";
1146 else
1147 OS << "(anonymous namespace)::";
1148 } else if (const auto *Spec = dyn_cast<ClassTemplateSpecializationDecl>(DC)) {
1149 IncludeStrongLifetimeRAII Strong(Policy);
1150 OS << Spec->getIdentifier()->getName();
1151 const TemplateArgumentList &TemplateArgs = Spec->getTemplateArgs();
1152 printTemplateArgumentList(OS, TemplateArgs.asArray(), Policy);
1153 OS << "::";
1154 } else if (const auto *Tag = dyn_cast<TagDecl>(DC)) {
1155 if (TypedefNameDecl *Typedef = Tag->getTypedefNameForAnonDecl())
1156 OS << Typedef->getIdentifier()->getName() << "::";
1157 else if (Tag->getIdentifier())
1158 OS << Tag->getIdentifier()->getName() << "::";
1159 else
1160 return;
1161 }
1162}

1164void TypePrinter::printTag(TagDecl *D, raw_ostream &OS) {
1165 if (Policy.IncludeTagDefinition) {
1166 PrintingPolicy SubPolicy = Policy;
1167 SubPolicy.IncludeTagDefinition = false;
1168 D->print(OS, SubPolicy, Indentation);
1169 spaceBeforePlaceHolder(OS);
1170 return;
1171 }
1172
1173 bool HasKindDecoration = false;
1174
1175 // We don't print tags unless this is an elaborated type.
1176 // In C, we just assume every RecordType is an elaborated type.
1177 if (!Policy.SuppressTagKeyword && !D->getTypedefNameForAnonDecl()) {
1178 HasKindDecoration = true;
1179 OS << D->getKindName();
1180 OS << ' ';
1181 }
1182
1183 // Compute the full nested-name-specifier for this type.
1184 // In C, this will always be empty except when the type
1185 // being printed is anonymous within other Record.
1186 if (!Policy.SuppressScope)
1187 AppendScope(D->getDeclContext(), OS);
1188
1189 if (const IdentifierInfo *II = D->getIdentifier())
1190 OS << II->getName();
1191 else if (TypedefNameDecl *Typedef = D->getTypedefNameForAnonDecl()) {
1192 assert(Typedef->getIdentifier() && "Typedef without identifier?");
1193 OS << Typedef->getIdentifier()->getName();
1194 } else {
1195 // Make an unambiguous representation for anonymous types, e.g.
1196 // (anonymous enum at /usr/include/string.h:120:9)
1197 OS << (Policy.MSVCFormatting ? '`' : '(');
1198
1199 if (isa<CXXRecordDecl>(D) && cast<CXXRecordDecl>(D)->isLambda()) {
1200 OS << "lambda";
1201 HasKindDecoration = true;
1202 } else {
1203 OS << "anonymous";
1204 }
1205
1206 if (Policy.AnonymousTagLocations) {
1207 // Suppress the redundant tag keyword if we just printed one.
1208 // We don't have to worry about ElaboratedTypes here because you can't
1209 // refer to an anonymous type with one.
1210 if (!HasKindDecoration)
1211 OS << " " << D->getKindName();
1212
1213 PresumedLoc PLoc = D->getASTContext().getSourceManager().getPresumedLoc(
1214 D->getLocation());
1215 if (PLoc.isValid()) {
1216 OS << " at ";
1217 StringRef File = PLoc.getFilename();
1218 if (Policy.RemapFilePaths)
1219 OS << Policy.remapPath(File);
1220 else
1221 OS << File;
1222 OS << ':' << PLoc.getLine() << ':' << PLoc.getColumn();
1223 }
1224 }
1225
1226 OS << (Policy.MSVCFormatting ? '\'' : ')');
1227 }
1228
1229 // If this is a class template specialization, print the template
1230 // arguments.
1231 if (const auto *Spec = dyn_cast<ClassTemplateSpecializationDecl>(D)) {
1232 ArrayRef<TemplateArgument> Args;
1233 if (TypeSourceInfo *TAW = Spec->getTypeAsWritten()) {
1234 const TemplateSpecializationType *TST =
1235 cast<TemplateSpecializationType>(TAW->getType());
1236 Args = TST->template_arguments();
1237 } else {
1238 const TemplateArgumentList &TemplateArgs = Spec->getTemplateArgs();
1239 Args = TemplateArgs.asArray();
1240 }
1241 IncludeStrongLifetimeRAII Strong(Policy);
1242 printTemplateArgumentList(OS, Args, Policy);
1243 }
1244
1245 spaceBeforePlaceHolder(OS);
1246}

1722template<typename TA>
1723static void printTo(raw_ostream &OS, ArrayRef<TA> Args,
1724 const PrintingPolicy &Policy, bool SkipBrackets) {
1725 const char *Comma = Policy.MSVCFormatting ? "," : ", ";
1726 if (!SkipBrackets)
1727 OS << '<';
1728
1729 bool NeedSpace = false;
1730 bool FirstArg = true;
1731 for (const auto &Arg : Args) {
1732 // Print the argument into a string.
1733 SmallString<128> Buf;
1734 llvm::raw_svector_ostream ArgOS(Buf);
1735 const TemplateArgument &Argument = getArgument(Arg);
1736 if (Argument.getKind() == TemplateArgument::Pack) {
1737 if (Argument.pack_size() && !FirstArg)
1738 OS << Comma;
1739 printTo(ArgOS, Argument.getPackAsArray(), Policy, true);
1740 } else {
1741 if (!FirstArg)
1742 OS << Comma;
1743 // Tries to print the argument with location info if exists.
1744 printArgument(Arg, Policy, ArgOS);
1745 }
1746 StringRef ArgString = ArgOS.str();
1747
1748 // If this is the first argument and its string representation
1749 // begins with the global scope specifier ('::foo'), add a space
1750 // to avoid printing the diagraph '<:'.
1751 if (FirstArg && !ArgString.empty() && ArgString[0] == ':')
1752 OS << ' ';
1753
1754 OS << ArgString;
1755
1756 NeedSpace = (!ArgString.empty() && ArgString.back() == '>');
1757 FirstArg = false;
1758 }
1759
1760 // If the last character of our string is '>', add another space to
1761 // keep the two '>''s separate tokens. We don't *have* to do this in
1762 // C++0x, but it's still good hygiene.
1763 if (NeedSpace)
1764 OS << ' ';
1765
1766 if (!SkipBrackets)
1767 OS << '>';
1768}

1820// Appends qualifiers to the given string, separated by spaces. Will
1821// prefix a space if the string is non-empty. Will not append a final
1822// space.
1823void Qualifiers::print(raw_ostream &OS, const PrintingPolicy& Policy,
1824 bool appendSpaceIfNonEmpty) const {
1825 bool addSpace = false;
1826
1827 unsigned quals = getCVRQualifiers();
1828 if (quals) {
1829 AppendTypeQualList(OS, quals, Policy.Restrict);
1830 addSpace = true;
1831 }
1832 if (hasUnaligned()) {
1833 if (addSpace)
1834 OS << ' ';
1835 OS << "__unaligned";
1836 addSpace = true;
1837 }
1838 LangAS addrspace = getAddressSpace();
1839 if (addrspace != LangAS::Default) {
1840 if (addrspace != LangAS::opencl_private) {
1841 if (addSpace)
1842 OS << ' ';
1843 addSpace = true;
1844 switch (addrspace) {
1845 case LangAS::opencl_global:
1846 OS << "__global";
1847 break;
1848 case LangAS::opencl_local:
1849 OS << "__local";
1850 break;
1851 case LangAS::opencl_private:
1852 break;
1853 case LangAS::opencl_constant:
1854 OS << "__constant";
1855 break;
1856 case LangAS::opencl_generic:
1857 OS << "__generic";
1858 break;
1859 case LangAS::cuda_device:
1860 OS << "__device__";
1861 break;
1862 case LangAS::cuda_constant:
1863 OS << "__constant__";
1864 break;
1865 case LangAS::cuda_shared:
1866 OS << "__shared__";
1867 break;
1868 default:
1869 OS << "__attribute__((address_space(";
1870 OS << toTargetAddressSpace(addrspace);
1871 OS << ")))";
1872 }
1873 }
1874 }
1875 if (Qualifiers::GC gc = getObjCGCAttr()) {
1876 if (addSpace)
1877 OS << ' ';
1878 addSpace = true;
1879 if (gc == Qualifiers::Weak)
1880 OS << "__weak";
1881 else
1882 OS << "__strong";
1883 }
1884 if (Qualifiers::ObjCLifetime lifetime = getObjCLifetime()) {
1885 if (!(lifetime == Qualifiers::OCL_Strong && Policy.SuppressStrongLifetime)){
1886 if (addSpace)
1887 OS << ' ';
1888 addSpace = true;
1889 }
1890
1891 switch (lifetime) {
1892 case Qualifiers::OCL_None: llvm_unreachable("none but true");
1893 case Qualifiers::OCL_ExplicitNone: OS << "__unsafe_unretained"; break;
1894 case Qualifiers::OCL_Strong:
1895 if (!Policy.SuppressStrongLifetime)
1896 OS << "__strong";
1897 break;
1898
1899 case Qualifiers::OCL_Weak: OS << "__weak"; break;
1900 case Qualifiers::OCL_Autoreleasing: OS << "__autoreleasing"; break;
1901 }
1902 }
1903
1904 if (appendSpaceIfNonEmpty && addSpace)
1905 OS << ' ';
1906}
