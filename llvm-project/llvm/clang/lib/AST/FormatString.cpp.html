36OptionalAmount
37clang::analyze_format_string::ParseAmount(const char *&Beg, const char *E) {
38 const char *I = Beg;
39 UpdateOnReturn <const char*> UpdateBeg(Beg, I);
40
41 unsigned accumulator = 0;
42 bool hasDigits = false;
43
44 for ( ; I != E; ++I) {
45 char c = *I;
46 if (c >= '0' && c <= '9') {
47 hasDigits = true;
48 accumulator = (accumulator * 10) + (c - '0');
49 continue;
50 }
51
52 if (hasDigits)
53 return OptionalAmount(OptionalAmount::Constant, accumulator, Beg, I - Beg,
54 false);
55
56 break;
57 }
58
59 return OptionalAmount();
60}

62OptionalAmount
63clang::analyze_format_string::ParseNonPositionAmount(const char *&Beg,
64 const char *E,
65 unsigned &argIndex) {
66 if (*Beg == '*') {
67 ++Beg;
68 return OptionalAmount(OptionalAmount::Arg, argIndex++, Beg, 0, false);
69 }
70
71 return ParseAmount(Beg, E);
72}

74OptionalAmount
75clang::analyze_format_string::ParsePositionAmount(FormatStringHandler &H,
76 const char *Start,
77 const char *&Beg,
78 const char *E,
79 PositionContext p) {
80 if (*Beg == '*') {
81 const char *I = Beg + 1;
82 const OptionalAmount &Amt = ParseAmount(I, E);
83
84 if (Amt.getHowSpecified() == OptionalAmount::NotSpecified) {
85 H.HandleInvalidPosition(Beg, I - Beg, p);
86 return OptionalAmount(false);
87 }
88
89 if (I == E) {
90 // No more characters left?
91 H.HandleIncompleteSpecifier(Start, E - Start);
92 return OptionalAmount(false);
93 }
94
95 assert(Amt.getHowSpecified() == OptionalAmount::Constant);
96
97 if (*I == '$') {
98 // Handle positional arguments
99
100 // Special case: '*0$', since this is an easy mistake.
101 if (Amt.getConstantAmount() == 0) {
102 H.HandleZeroPosition(Beg, I - Beg + 1);
103 return OptionalAmount(false);
104 }
105
106 const char *Tmp = Beg;
107 Beg = ++I;
108
109 return OptionalAmount(OptionalAmount::Arg, Amt.getConstantAmount() - 1,
110 Tmp, 0, true);
111 }
112
113 H.HandleInvalidPosition(Beg, I - Beg, p);
114 return OptionalAmount(false);
115 }
116
117 return ParseAmount(Beg, E);
118}

121bool
122clang::analyze_format_string::ParseFieldWidth(FormatStringHandler &H,
123 FormatSpecifier &CS,
124 const char *Start,
125 const char *&Beg, const char *E,
126 unsigned *argIndex) {
127 // FIXME: Support negative field widths.
128 if (argIndex) {
129 CS.setFieldWidth(ParseNonPositionAmount(Beg, E, *argIndex));
130 }
131 else {
132 const OptionalAmount Amt =
133 ParsePositionAmount(H, Start, Beg, E,
134 analyze_format_string::FieldWidthPos);
135
136 if (Amt.isInvalid())
137 return true;
138 CS.setFieldWidth(Amt);
139 }
140 return false;
141}

143bool
144clang::analyze_format_string::ParseArgPosition(FormatStringHandler &H,
145 FormatSpecifier &FS,
146 const char *Start,
147 const char *&Beg,
148 const char *E) {
149 const char *I = Beg;
150
151 const OptionalAmount &Amt = ParseAmount(I, E);
152
153 if (I == E) {
154 // No more characters left?
155 H.HandleIncompleteSpecifier(Start, E - Start);
156 return true;
157 }
158
159 if (Amt.getHowSpecified() == OptionalAmount::Constant && *(I++) == '$') {
160 // Warn that positional arguments are non-standard.
161 H.HandlePosition(Start, I - Start);
162
163 // Special case: '%0$', since this is an easy mistake.
164 if (Amt.getConstantAmount() == 0) {
165 H.HandleZeroPosition(Start, I - Start);
166 return true;
167 }
168
169 FS.setArgIndex(Amt.getConstantAmount() - 1);
170 FS.setUsesPositionalArg();
171 // Update the caller's pointer if we decided to consume
172 // these characters.
173 Beg = I;
174 return false;
175 }
176
177 return false;
178}

180bool
181clang::analyze_format_string::ParseVectorModifier(FormatStringHandler &H,
182 FormatSpecifier &FS,
183 const char *&I,
184 const char *E,
185 const LangOptions &LO) {
186 if (!LO.OpenCL)
187 return false;
188
189 const char *Start = I;
190 if (*I == 'v') {
191 ++I;
192
193 if (I == E) {
194 H.HandleIncompleteSpecifier(Start, E - Start);
195 return true;
196 }
197
198 OptionalAmount NumElts = ParseAmount(I, E);
199 if (NumElts.getHowSpecified() != OptionalAmount::Constant) {
200 H.HandleIncompleteSpecifier(Start, E - Start);
201 return true;
202 }
203
204 FS.setVectorNumElts(NumElts);
205 }
206
207 return false;
208}

210bool
211clang::analyze_format_string::ParseLengthModifier(FormatSpecifier &FS,
212 const char *&I,
213 const char *E,
214 const LangOptions &LO,
215 bool IsScanf) {
216 LengthModifier::Kind lmKind = LengthModifier::None;
217 const char *lmPosition = I;
218 switch (*I) {
219 default:
220 return false;
221 case 'h':
222 ++I;
223 if (I != E && *I == 'h') {
224 ++I;
225 lmKind = LengthModifier::AsChar;
226 } else if (I != E && *I == 'l' && LO.OpenCL) {
227 ++I;
228 lmKind = LengthModifier::AsShortLong;
229 } else {
230 lmKind = LengthModifier::AsShort;
231 }
232 break;
233 case 'l':
234 ++I;
235 if (I != E && *I == 'l') {
236 ++I;
237 lmKind = LengthModifier::AsLongLong;
238 } else {
239 lmKind = LengthModifier::AsLong;
240 }
241 break;
242 case 'j': lmKind = LengthModifier::AsIntMax; ++I; break;
243 case 'P':
244 if ((I+1) == E)
245 return false;
246 else {
247 ++I;
248 lmKind = LengthModifier::AsIntPtr;
249 }
250 break;
251 case 'z': lmKind = LengthModifier::AsSizeT; ++I; break;
252 case 't': lmKind = LengthModifier::AsPtrDiff; ++I; break;
253 case 'L': lmKind = LengthModifier::AsLongDouble; ++I; break;
254 case 'q': lmKind = LengthModifier::AsQuad; ++I; break;
255 case 'a':
256 if (IsScanf && !LO.C99 && !LO.CPlusPlus11) {
257 // For scanf in C90, look at the next character to see if this should
258 // be parsed as the GNU extension 'a' length modifier. If not, this
259 // will be parsed as a conversion specifier.
260 ++I;
261 if (I != E && (*I == 's' || *I == 'S' || *I == '[')) {
262 lmKind = LengthModifier::AsAllocate;
263 break;
264 }
265 --I;
266 }
267 return false;
268 case 'm':
269 if (IsScanf) {
270 lmKind = LengthModifier::AsMAllocate;
271 ++I;
272 break;
273 }
274 return false;
275 // printf: AsInt64, AsInt32, AsInt3264
276 // scanf: AsInt64
277 case 'I':
278 if (I + 1 != E && I + 2 != E) {
279 if (I[1] == '6' && I[2] == '4') {
280 I += 3;
281 lmKind = LengthModifier::AsInt64;
282 break;
283 }
284 if (IsScanf)
285 return false;
286
287 if (I[1] == '3' && I[2] == '2') {
288 I += 3;
289 lmKind = LengthModifier::AsInt32;
290 break;
291 }
292 }
293 ++I;
294 lmKind = LengthModifier::AsInt3264;
295 break;
296 case 'w':
297 lmKind = LengthModifier::AsWide; ++I; break;
298 }
299 LengthModifier lm(lmPosition, lmKind);
300 FS.setLengthModifier(lm);
301 return true;
302}

304bool clang::analyze_format_string::ParseUTF8InvalidSpecifier(
305 const char *SpecifierBegin, const char *FmtStrEnd, unsigned &Len) {
306 if (SpecifierBegin + 1 >= FmtStrEnd)
307 return false;
308
309 const llvm::UTF8 *SB =
310 reinterpret_cast<const llvm::UTF8 *>(SpecifierBegin + 1);
311 const llvm::UTF8 *SE = reinterpret_cast<const llvm::UTF8 *>(FmtStrEnd);
312 const char FirstByte = *SB;
313
314 // If the invalid specifier is a multibyte UTF-8 string, return the
315 // total length accordingly so that the conversion specifier can be
316 // properly updated to reflect a complete UTF-8 specifier.
317 unsigned NumBytes = llvm::getNumBytesForUTF8(FirstByte);
318 if (NumBytes == 1)
319 return false;
320 if (SB + NumBytes > SE)
321 return false;
322
323 Len = NumBytes + 1;
324 return true;
325}

331clang::analyze_format_string::ArgType::MatchKind
332ArgType::matchesType(ASTContext &C, QualType argTy) const {
333 if (Ptr) {
334 // It has to be a pointer.
335 const PointerType *PT = argTy->getAs<PointerType>();
336 if (!PT)
337 return NoMatch;
338
339 // We cannot write through a const qualified pointer.
340 if (PT->getPointeeType().isConstQualified())
341 return NoMatch;
342
343 argTy = PT->getPointeeType();
344 }
345
346 switch (K) {
347 case InvalidTy:
348 llvm_unreachable("ArgType must be valid");
349
350 case UnknownTy:
351 return Match;
352
353 case AnyCharTy: {
354 if (const EnumType *ETy = argTy->getAs<EnumType>()) {
355 // If the enum is incomplete we know nothing about the underlying type.
356 // Assume that it's 'int'.
357 if (!ETy->getDecl()->isComplete())
358 return NoMatch;
359 argTy = ETy->getDecl()->getIntegerType();
360 }
361
362 if (const BuiltinType *BT = argTy->getAs<BuiltinType>())
363 switch (BT->getKind()) {
364 default:
365 break;
366 case BuiltinType::Char_S:
367 case BuiltinType::SChar:
368 case BuiltinType::UChar:
369 case BuiltinType::Char_U:
370 return Match;
371 }
372 return NoMatch;
373 }
374
375 case SpecificTy: {
376 if (const EnumType *ETy = argTy->getAs<EnumType>()) {
377 // If the enum is incomplete we know nothing about the underlying type.
378 // Assume that it's 'int'.
379 if (!ETy->getDecl()->isComplete())
380 argTy = C.IntTy;
381 else
382 argTy = ETy->getDecl()->getIntegerType();
383 }
384 argTy = C.getCanonicalType(argTy).getUnqualifiedType();
385
386 if (T == argTy)
387 return Match;
388 // Check for "compatible types".
389 if (const BuiltinType *BT = argTy->getAs<BuiltinType>())
390 switch (BT->getKind()) {
391 default:
392 break;
393 case BuiltinType::Char_S:
394 case BuiltinType::SChar:
395 case BuiltinType::Char_U:
396 case BuiltinType::UChar:
397 return T == C.UnsignedCharTy || T == C.SignedCharTy ? Match
398 : NoMatch;
399 case BuiltinType::Short:
400 return T == C.UnsignedShortTy ? Match : NoMatch;
401 case BuiltinType::UShort:
402 return T == C.ShortTy ? Match : NoMatch;
403 case BuiltinType::Int:
404 return T == C.UnsignedIntTy ? Match : NoMatch;
405 case BuiltinType::UInt:
406 return T == C.IntTy ? Match : NoMatch;
407 case BuiltinType::Long:
408 return T == C.UnsignedLongTy ? Match : NoMatch;
409 case BuiltinType::ULong:
410 return T == C.LongTy ? Match : NoMatch;
411 case BuiltinType::LongLong:
412 return T == C.UnsignedLongLongTy ? Match : NoMatch;
413 case BuiltinType::ULongLong:
414 return T == C.LongLongTy ? Match : NoMatch;
415 }
416 return NoMatch;
417 }
418
419 case CStrTy: {
420 const PointerType *PT = argTy->getAs<PointerType>();
421 if (!PT)
422 return NoMatch;
423 if (C.getTargetInfo().areAllPointersCapabilities() != PT->isCHERICapability())
424 return NoMatch;
425 QualType pointeeTy = PT->getPointeeType();
426 if (const BuiltinType *BT = pointeeTy->getAs<BuiltinType>())
427 switch (BT->getKind()) {
428 case BuiltinType::Void:
429 case BuiltinType::Char_U:
430 case BuiltinType::UChar:
431 case BuiltinType::Char_S:
432 case BuiltinType::SChar:
433 return Match;
434 default:
435 break;
436 }
437
438 return NoMatch;
439 }
440
441 case WCStrTy: {
442 const PointerType *PT = argTy->getAs<PointerType>();
443 if (!PT)
444 return NoMatch;
445 QualType pointeeTy =
446 C.getCanonicalType(PT->getPointeeType()).getUnqualifiedType();
447 return pointeeTy == C.getWideCharType() ? Match : NoMatch;
448 }
449
450 case WIntTy: {
451 QualType WInt = C.getCanonicalType(C.getWIntType()).getUnqualifiedType();
452
453 if (C.getCanonicalType(argTy).getUnqualifiedType() == WInt)
454 return Match;
455
456 QualType PromoArg = argTy->isPromotableIntegerType()
457 ? C.getPromotedIntegerType(argTy)
458 : argTy;
459 PromoArg = C.getCanonicalType(PromoArg).getUnqualifiedType();
460
461 // If the promoted argument is the corresponding signed type of the
462 // wint_t type, then it should match.
463 if (PromoArg->hasSignedIntegerRepresentation() &&
464 C.getCorrespondingUnsignedType(PromoArg) == WInt)
465 return Match;
466
467 return WInt == PromoArg ? Match : NoMatch;
468 }
469
470 case CPointerTy:
471 if (const PointerType *PT = argTy->getAs<PointerType>()) {
472 if (C.getTargetInfo().areAllPointersCapabilities() != PT->isCHERICapability())
473 return NoMatch;
474 }
475 if (argTy->isVoidPointerType()) {
476 return Match;
477 } if (argTy->isPointerType() || argTy->isObjCObjectPointerType() ||
478 argTy->isBlockPointerType() || argTy->isNullPtrType()) {
479 return NoMatchPedantic;
480 } else {
481 return NoMatch;
482 }
483
484 case ObjCPointerTy: {
485 if (argTy->getAs<ObjCObjectPointerType>() ||
486 argTy->getAs<BlockPointerType>())
487 return Match;
488
489 // Handle implicit toll-free bridging.
490 if (const PointerType *PT = argTy->getAs<PointerType>()) {
491 // Things such as CFTypeRef are really just opaque pointers
492 // to C structs representing CF types that can often be bridged
493 // to Objective-C objects. Since the compiler doesn't know which
494 // structs can be toll-free bridged, we just accept them all.
495 QualType pointee = PT->getPointeeType();
496 if (pointee->getAsStructureType() || pointee->isVoidType())
497 return Match;
498 }
499 return NoMatch;
500 }
501 }
502
503 llvm_unreachable("Invalid ArgType Kind!");
504}

506ArgType ArgType::makeVectorType(ASTContext &C, unsigned NumElts) const {
507 // Check for valid vector element types.
508 if (T.isNull())
509 return ArgType::Invalid();
510
511 QualType Vec = C.getExtVectorType(T, NumElts);
512 return ArgType(Vec, Name);
513}

515QualType ArgType::getRepresentativeType(ASTContext &C) const {
516 QualType Res;
517 switch (K) {
518 case InvalidTy:
519 llvm_unreachable("No representative type for Invalid ArgType");
520 case UnknownTy:
521 llvm_unreachable("No representative type for Unknown ArgType");
522 case AnyCharTy:
523 Res = C.CharTy;
524 break;
525 case SpecificTy:
526 Res = T;
527 break;
528 case CStrTy:
529 Res = C.getPointerType(C.CharTy);
530 break;
531 case WCStrTy:
532 Res = C.getPointerType(C.getWideCharType());
533 break;
534 case ObjCPointerTy:
535 Res = C.ObjCBuiltinIdTy;
536 break;
537 case CPointerTy:
538 Res = C.VoidPtrTy;
539 break;
540 case WIntTy: {
541 Res = C.getWIntType();
542 break;
543 }
544 }
545
546 if (Ptr)
547 Res = C.getPointerType(Res);
548 return Res;
549}
550
551std::string ArgType::getRepresentativeTypeName(ASTContext &C) const {
552 std::string S = getRepresentativeType(C).getAsString();
553
554 std::string Alias;
555 if (Name) {
556 // Use a specific name for this type, e.g. "size_t".
557 Alias = Name;
558 if (Ptr) {
559 // If ArgType is actually a pointer to T, append an asterisk.
560 Alias += (Alias[Alias.size()-1] == '*') ? "*" : " *";
561 }
562 // If Alias is the same as the underlying type, e.g. wchar_t, then drop it.
563 if (S == Alias)
564 Alias.clear();
565 }
566
567 if (!Alias.empty())
568 return std::string("'") + Alias + "' (aka '" + S + "')";
569 return std::string("'") + S + "'";
570}

711void OptionalAmount::toString(raw_ostream &os) const {
712 switch (hs) {
713 case Invalid:
714 case NotSpecified:
715 return;
716 case Arg:
717 if (UsesDotPrefix)
718 os << ".";
719 if (usesPositionalArg())
720 os << "*" << getPositionalArgIndex() << "$";
721 else
722 os << "*";
723 break;
724 case Constant:
725 if (UsesDotPrefix)
726 os << ".";
727 os << amt;
728 break;
729 }
730}

732bool FormatSpecifier::hasValidLengthModifier(const TargetInfo &Target,
733 const LangOptions &LO) const {
734 switch (LM.getKind()) {
735 case LengthModifier::None:
736 return true;
737
738 // Handle most integer flags
739 case LengthModifier::AsShort:
740 // Length modifier only applies to FP vectors.
741 if (LO.OpenCL && CS.isDoubleArg())
742 return !VectorNumElts.isInvalid();
743
744 if (Target.getTriple().isOSMSVCRT()) {
745 switch (CS.getKind()) {
746 case ConversionSpecifier::cArg:
747 case ConversionSpecifier::CArg:
748 case ConversionSpecifier::sArg:
749 case ConversionSpecifier::SArg:
750 case ConversionSpecifier::ZArg:
751 return true;
752 default:
753 break;
754 }
755 }
756 LLVM_FALLTHROUGH;
757 case LengthModifier::AsChar:
758 case LengthModifier::AsLongLong:
759 case LengthModifier::AsQuad:
760 case LengthModifier::AsIntMax:
761 case LengthModifier::AsIntPtr:
762 case LengthModifier::AsSizeT:
763 case LengthModifier::AsPtrDiff:
764 switch (CS.getKind()) {
765 case ConversionSpecifier::dArg:
766 case ConversionSpecifier::DArg:
767 case ConversionSpecifier::iArg:
768 case ConversionSpecifier::oArg:
769 case ConversionSpecifier::OArg:
770 case ConversionSpecifier::uArg:
771 case ConversionSpecifier::UArg:
772 case ConversionSpecifier::xArg:
773 case ConversionSpecifier::XArg:
774 case ConversionSpecifier::nArg:
775 return true;
776 case ConversionSpecifier::FreeBSDrArg:
777 case ConversionSpecifier::FreeBSDyArg:
778 return Target.getTriple().isOSFreeBSD() || Target.getTriple().isPS4();
779 default:
780 return false;
781 }
782
783 case LengthModifier::AsShortLong:
784 return LO.OpenCL && !VectorNumElts.isInvalid();
785
786 // Handle 'l' flag
787 case LengthModifier::AsLong: // or AsWideChar
788 if (CS.isDoubleArg()) {
789 // Invalid for OpenCL FP scalars.
790 if (LO.OpenCL && VectorNumElts.isInvalid())
791 return false;
792 return true;
793 }
794
795 switch (CS.getKind()) {
796 case ConversionSpecifier::dArg:
797 case ConversionSpecifier::DArg:
798 case ConversionSpecifier::iArg:
799 case ConversionSpecifier::oArg:
800 case ConversionSpecifier::OArg:
801 case ConversionSpecifier::uArg:
802 case ConversionSpecifier::UArg:
803 case ConversionSpecifier::xArg:
804 case ConversionSpecifier::XArg:
805 case ConversionSpecifier::nArg:
806 case ConversionSpecifier::cArg:
807 case ConversionSpecifier::sArg:
808 case ConversionSpecifier::ScanListArg:
809 case ConversionSpecifier::ZArg:
810 return true;
811 case ConversionSpecifier::FreeBSDrArg:
812 case ConversionSpecifier::FreeBSDyArg:
813 return Target.getTriple().isOSFreeBSD() || Target.getTriple().isPS4();
814 default:
815 return false;
816 }
817
818 case LengthModifier::AsLongDouble:
819 switch (CS.getKind()) {
820 case ConversionSpecifier::aArg:
821 case ConversionSpecifier::AArg:
822 case ConversionSpecifier::fArg:
823 case ConversionSpecifier::FArg:
824 case ConversionSpecifier::eArg:
825 case ConversionSpecifier::EArg:
826 case ConversionSpecifier::gArg:
827 case ConversionSpecifier::GArg:
828 return true;
829 // GNU libc extension.
830 case ConversionSpecifier::dArg:
831 case ConversionSpecifier::iArg:
832 case ConversionSpecifier::oArg:
833 case ConversionSpecifier::uArg:
834 case ConversionSpecifier::xArg:
835 case ConversionSpecifier::XArg:
836 return !Target.getTriple().isOSDarwin() &&
837 !Target.getTriple().isOSWindows();
838 default:
839 return false;
840 }
841
842 case LengthModifier::AsAllocate:
843 switch (CS.getKind()) {
844 case ConversionSpecifier::sArg:
845 case ConversionSpecifier::SArg:
846 case ConversionSpecifier::ScanListArg:
847 return true;
848 default:
849 return false;
850 }
851
852 case LengthModifier::AsMAllocate:
853 switch (CS.getKind()) {
854 case ConversionSpecifier::cArg:
855 case ConversionSpecifier::CArg:
856 case ConversionSpecifier::sArg:
857 case ConversionSpecifier::SArg:
858 case ConversionSpecifier::ScanListArg:
859 return true;
860 default:
861 return false;
862 }
863 case LengthModifier::AsInt32:
864 case LengthModifier::AsInt3264:
865 case LengthModifier::AsInt64:
866 switch (CS.getKind()) {
867 case ConversionSpecifier::dArg:
868 case ConversionSpecifier::iArg:
869 case ConversionSpecifier::oArg:
870 case ConversionSpecifier::uArg:
871 case ConversionSpecifier::xArg:
872 case ConversionSpecifier::XArg:
873 return Target.getTriple().isOSMSVCRT();
874 default:
875 return false;
876 }
877 case LengthModifier::AsWide:
878 switch (CS.getKind()) {
879 case ConversionSpecifier::cArg:
880 case ConversionSpecifier::CArg:
881 case ConversionSpecifier::sArg:
882 case ConversionSpecifier::SArg:
883 case ConversionSpecifier::ZArg:
884 return Target.getTriple().isOSMSVCRT();
885 default:
886 return false;
887 }
888 }
889 llvm_unreachable("Invalid LengthModifier Kind!");
890}

992bool FormatSpecifier::namedTypeToLengthModifier(QualType QT,
993 LengthModifier &LM) {
994 assert(isa<TypedefType>(QT) && "Expected a TypedefType");
995 const TypedefNameDecl *Typedef = cast<TypedefType>(QT)->getDecl();
996
997 for (;;) {
998 const IdentifierInfo *Identifier = Typedef->getIdentifier();
999 if (Identifier->getName() == "size_t") {
1000 LM.setKind(LengthModifier::AsSizeT);
1001 return true;
1002 } else if (Identifier->getName() == "ssize_t") {
1003 // Not C99, but common in Unix.
1004 LM.setKind(LengthModifier::AsSizeT);
1005 return true;
1006 } else if (Identifier->getName() == "intmax_t") {
1007 LM.setKind(LengthModifier::AsIntMax);
1008 return true;
1009 } else if (Identifier->getName() == "uintmax_t") {
1010 LM.setKind(LengthModifier::AsIntMax);
1011 return true;
1012 } else if (Identifier->getName() == "intptr_t") {
1013 LM.setKind(LengthModifier::AsIntPtr);
1014 return true;
1015 } else if (Identifier->getName() == "uintptr_t") {
1016 LM.setKind(LengthModifier::AsIntPtr);
1017 return true;
1018 } else if (Identifier->getName() == "ptrdiff_t") {
1019 LM.setKind(LengthModifier::AsPtrDiff);
1020 return true;
1021 }
1022
1023 QualType T = Typedef->getUnderlyingType();
1024 if (!isa<TypedefType>(T))
1025 break;
1026
1027 Typedef = cast<TypedefType>(T)->getDecl();
1028 }
1029 return false;
1030}
