39// Did the text at this location originate in a macro definition (aka. body)?
40// For example,
41//
42// #define NESTED(x) x
43// #define MACRO(y) { int y = NESTED(3); }
44// if (true) MACRO(foo)
45//
46// The if statement expands to
47//
48// if (true) { int foo = 3; }
49// ^ ^
50// Loc1 Loc2
51//
52// For SourceManager SM, SM.isMacroArgExpansion(Loc1) and
53// SM.isMacroArgExpansion(Loc2) are both true, but isOriginMacroBody(sm, Loc1)
54// is false, because "foo" originated in the source file (as an argument to a
55// macro), whereas isOriginMacroBody(SM, Loc2) is true, because "3" originated
56// in the definition of MACRO.
57static bool isOriginMacroBody(const clang::SourceManager &SM,
58 clang::SourceLocation Loc) {
59 while (Loc.isMacroID()) {
60 if (SM.isMacroBodyExpansion(Loc))
61 return true;
62 // Otherwise, it must be in an argument, so we continue searching up the
63 // invocation stack. getImmediateMacroCallerLoc() gives the location of the
64 // argument text, inside the call text.
65 Loc = SM.getImmediateMacroCallerLoc(Loc);
66 }
67 return false;
68}

70Expected<SmallVector<tooling::detail::Transformation, 1>>
71tooling::detail::translateEdits(const MatchResult &Result,
72 llvm::ArrayRef<ASTEdit> Edits) {
73 SmallVector<tooling::detail::Transformation, 1> Transformations;
74 for (const auto &Edit : Edits) {
75 Expected<CharSourceRange> Range = Edit.TargetRange(Result);
76 if (!Range)
77 return Range.takeError();
78 if (Range->isInvalid() ||
79 isOriginMacroBody(*Result.SourceManager, Range->getBegin()))
80 return SmallVector<Transformation, 0>();
81 auto Replacement = Edit.Replacement(Result);
82 if (!Replacement)
83 return Replacement.takeError();
84 tooling::detail::Transformation T;
85 T.Range = *Range;
86 T.Replacement = std::move(*Replacement);
87 Transformations.push_back(std::move(T));
88 }
89 return Transformations;
90}

92ASTEdit tooling::change(RangeSelector S, TextGenerator Replacement) {

105// Determines whether A is a base type of B in the class hierarchy, including
106// the implicit relationship of Type and QualType.
107static bool isBaseOf(ASTNodeKind A, ASTNodeKind B) {
108 static auto TypeKind = ASTNodeKind::getFromNodeKind<Type>();
109 static auto QualKind = ASTNodeKind::getFromNodeKind<QualType>();
110 /// Mimic the implicit conversions of Matcher<>.
111 /// - From Matcher<Type> to Matcher<QualType>
112 /// - From Matcher<Base> to Matcher<Derived>
113 return (A.isSame(TypeKind) && B.isSame(QualKind)) || A.isBaseOf(B);
114}

116// Try to find a common kind to which all of the rule's matchers can be
117// converted.
118static ASTNodeKind
119findCommonKind(const SmallVectorImpl<RewriteRule::Case> &Cases) {
120 assert(!Cases.empty() && "Rule must have at least one case.");
121 ASTNodeKind JoinKind = Cases[0].Matcher.getSupportedKind();
122 // Find a (least) Kind K, for which M.canConvertTo(K) holds, for all matchers
123 // M in Rules.
124 for (const auto &Case : Cases) {
125 auto K = Case.Matcher.getSupportedKind();
126 if (isBaseOf(JoinKind, K)) {
127 JoinKind = K;
128 continue;
129 }
130 if (K.isSame(JoinKind) || isBaseOf(K, JoinKind))
131 // JoinKind is already the lowest.
132 continue;
133 // K and JoinKind are unrelated -- there is no least common kind.
134 return ASTNodeKind();
135 }
136 return JoinKind;
137}

139// Binds each rule's matcher to a unique (and deterministic) tag based on
140// `TagBase`.
141static std::vector<DynTypedMatcher>
142taggedMatchers(StringRef TagBase,
143 const SmallVectorImpl<RewriteRule::Case> &Cases) {
144 std::vector<DynTypedMatcher> Matchers;
145 Matchers.reserve(Cases.size());
146 size_t count = 0;
147 for (const auto &Case : Cases) {
148 std::string Tag = (TagBase + Twine(count)).str();
149 ++count;
150 auto M = Case.Matcher.tryBind(Tag);
151 assert(M && "RewriteRule matchers should be bindable.");
152 Matchers.push_back(*std::move(M));
153 }
154 return Matchers;
155}

157// Simply gathers the contents of the various rules into a single rule. The
158// actual work to combine these into an ordered choice is deferred to matcher
159// registration.
160RewriteRule tooling::applyFirst(ArrayRef<RewriteRule> Rules) {
161 RewriteRule R;
162 for (auto &Rule : Rules)
163 R.Cases.append(Rule.Cases.begin(), Rule.Cases.end());
164 return R;
165}

167static DynTypedMatcher joinCaseMatchers(const RewriteRule &Rule) {
168 assert(!Rule.Cases.empty() && "Rule must have at least one case.");
169 if (Rule.Cases.size() == 1)
170 return Rule.Cases[0].Matcher;
171
172 auto CommonKind = findCommonKind(Rule.Cases);
173 assert(!CommonKind.isNone() && "Cases must have compatible matchers.");
174 return DynTypedMatcher::constructVariadic(
175 DynTypedMatcher::VO_AnyOf, CommonKind, taggedMatchers("Tag", Rule.Cases));
176}
177
178DynTypedMatcher tooling::detail::buildMatcher(const RewriteRule &Rule) {
179 DynTypedMatcher M = joinCaseMatchers(Rule);
180 M.setAllowBind(true);
181 // `tryBind` is guaranteed to succeed, because `AllowBind` was set to true.
182 return *M.tryBind(RewriteRule::RootID);
183}

185// Finds the case that was "selected" -- that is, whose matcher triggered the
186// `MatchResult`.
187const RewriteRule::Case &
188tooling::detail::findSelectedCase(const MatchResult &Result,
189 const RewriteRule &Rule) {
190 if (Rule.Cases.size() == 1)
191 return Rule.Cases[0];
192
193 auto &NodesMap = Result.Nodes.getMap();
194 for (size_t i = 0, N = Rule.Cases.size(); i < N; ++i) {
195 std::string Tag = ("Tag" + Twine(i)).str();
196 if (NodesMap.find(Tag) != NodesMap.end())
197 return Rule.Cases[i];
198 }
199 llvm_unreachable("No tag found for this rule.");
200}

208void Transformer::run(const MatchResult &Result) {
209 if (Result.Context->getDiagnostics().hasErrorOccurred())
210 return;
211
212 // Verify the existence and validity of the AST node that roots this rule.
213 auto &NodesMap = Result.Nodes.getMap();
214 auto Root = NodesMap.find(RewriteRule::RootID);
215 assert(Root != NodesMap.end() && "Transformation failed: missing root node.");
216 SourceLocation RootLoc = Result.SourceManager->getExpansionLoc(
217 Root->second.getSourceRange().getBegin());
218 assert(RootLoc.isValid() && "Invalid location for Root node of match.");
219
220 auto Transformations = tooling::detail::translateEdits(
221 Result, tooling::detail::findSelectedCase(Result, Rule).Edits);
222 if (!Transformations) {
223 Consumer(Transformations.takeError());
224 return;
225 }
226
227 if (Transformations->empty()) {
228 // No rewrite applied (but no error encountered either).
229 RootLoc.print(llvm::errs() << "note: skipping match at loc ",
230 *Result.SourceManager);
231 llvm::errs() << "\n";
232 return;
233 }
234
235 // Record the results in the AtomicChange.
236 AtomicChange AC(*Result.SourceManager, RootLoc);
237 for (const auto &T : *Transformations) {
238 if (auto Err = AC.replace(*Result.SourceManager, T.Range, T.Replacement)) {
239 Consumer(std::move(Err));
240 return;
241 }
242 }
243
244 Consumer(std::move(AC));
245}
