58static ToolChain::RTTIMode CalculateRTTIMode(const ArgList &Args,
59 const llvm::Triple &Triple,
60 const Arg *CachedRTTIArg) {
61 // Explicit rtti/no-rtti args
62 if (CachedRTTIArg) {
63 if (CachedRTTIArg->getOption().matches(options::OPT_frtti))
64 return ToolChain::RM_Enabled;
65 else
66 return ToolChain::RM_Disabled;
67 }
68
69 // -frtti is default, except for the PS4 CPU.
70 return (Triple.isPS4CPU()) ? ToolChain::RM_Disabled : ToolChain::RM_Enabled;
71}

73ToolChain::ToolChain(const Driver &D, const llvm::Triple &T,
74 const ArgList &Args)
75 : D(D), Triple(T), Args(Args), CachedRTTIArg(GetRTTIArgument(Args)),
76 CachedRTTIMode(CalculateRTTIMode(Args, Triple, CachedRTTIArg)) {
77 if (D.CCCIsCXX()) {
78 if (auto CXXStdlibPath = getCXXStdlibPath())
79 getFilePaths().push_back(*CXXStdlibPath);
80 }
81
82 if (auto RuntimePath = getRuntimePath())
83 getLibraryPaths().push_back(*RuntimePath);
84
85 std::string CandidateLibPath = getArchSpecificLibPath();
86 if (getVFS().exists(CandidateLibPath))
87 getFilePaths().push_back(CandidateLibPath);
88}

132 const char *ModeFlag;

137static const DriverSuffix *FindDriverSuffix(StringRef ProgName, size_t &Pos) {
138 // A list of known driver suffixes. Suffixes are compared against the
139 // program name in order. If there is a match, the frontend type is updated as
140 // necessary by applying the ModeFlag.
141 static const DriverSuffix DriverSuffixes[] = {
142 {"clang", nullptr},
143 {"clang++", "--driver-mode=g++"},
144 {"clang-c++", "--driver-mode=g++"},
145 {"clang-cc", nullptr},
146 {"clang-cpp", "--driver-mode=cpp"},
147 {"clang-g++", "--driver-mode=g++"},
148 {"clang-gcc", nullptr},
149 {"clang-cl", "--driver-mode=cl"},
150 {"cc", nullptr},
151 {"cpp", "--driver-mode=cpp"},
152 {"cl", "--driver-mode=cl"},
153 {"++", "--driver-mode=g++"},
154 };
155
156 for (size_t i = 0; i < llvm::array_lengthof(DriverSuffixes); ++i) {
157 StringRef Suffix(DriverSuffixes[i].Suffix);
158 if (ProgName.endswith(Suffix)) {
159 Pos = ProgName.size() - Suffix.size();
160 return &DriverSuffixes[i];
161 }
162 }
163 return nullptr;
164}
165
166/// Normalize the program name from argv[0] by stripping the file extension if
167/// present and lower-casing the string on Windows.
168static std::string normalizeProgramName(llvm::StringRef Argv0) {
169 std::string ProgName = llvm::sys::path::stem(Argv0);
170#ifdef _WIN32
171 // Transform to lowercase for case insensitive file systems.
172 std::transform(ProgName.begin(), ProgName.end(), ProgName.begin(), ::tolower);
173#endif
174 return ProgName;
175}
176
177static const DriverSuffix *parseDriverSuffix(StringRef ProgName, size_t &Pos) {
178 // Try to infer frontend type and default target from the program name by
179 // comparing it against DriverSuffixes in order.
180
181 // If there is a match, the function tries to identify a target as prefix.
182 // E.g. "x86_64-linux-clang" as interpreted as suffix "clang" with target
183 // prefix "x86_64-linux". If such a target prefix is found, it may be
184 // added via -target as implicit first argument.
185 const DriverSuffix *DS = FindDriverSuffix(ProgName, Pos);
186
187 if (!DS) {
188 // Try again after stripping any trailing version number:
189 // clang++3.5 -> clang++
190 ProgName = ProgName.rtrim("0123456789.");
191 DS = FindDriverSuffix(ProgName, Pos);
192 }
193
194 if (!DS) {
195 // Try again after stripping trailing -component.
196 // clang++-tot -> clang++
197 ProgName = ProgName.slice(0, ProgName.rfind('-'));
198 DS = FindDriverSuffix(ProgName, Pos);
199 }
200 return DS;
201}
202
203ParsedClangName
204ToolChain::getTargetAndModeFromProgramName(StringRef PN) {
205 std::string ProgName = normalizeProgramName(PN);
206 size_t SuffixPos;
207 const DriverSuffix *DS = parseDriverSuffix(ProgName, SuffixPos);
208 if (!DS)
209 return {};
210 size_t SuffixEnd = SuffixPos + strlen(DS->Suffix);
211
212 size_t LastComponent = ProgName.rfind('-', SuffixPos);
213 if (LastComponent == std::string::npos)
214 return ParsedClangName(ProgName.substr(0, SuffixEnd), DS->ModeFlag);
215 std::string ModeSuffix = ProgName.substr(LastComponent + 1,
216 SuffixEnd - LastComponent - 1);
217
218 // Infer target from the prefix.
219 StringRef Prefix(ProgName);
220 Prefix = Prefix.slice(0, LastComponent);
221 std::string IgnoredError;
222 bool IsRegistered = llvm::TargetRegistry::lookupTarget(Prefix, IgnoredError);
223 return ParsedClangName{Prefix, ModeSuffix, DS->ModeFlag, IsRegistered};
224}

226StringRef ToolChain::getDefaultUniversalArchName() const {
227 // In universal driver terms, the arch name accepted by -arch isn't exactly
228 // the same as the ones that appear in the triple. Roughly speaking, this is
229 // an inverse of the darwin::getArchTypeForDarwinArchName() function, but the
230 // only interesting special case is powerpc.
231 switch (Triple.getArch()) {
232 case llvm::Triple::ppc:
233 return "ppc";
234 case llvm::Triple::ppc64:
235 return "ppc64";
236 case llvm::Triple::ppc64le:
237 return "ppc64le";
238 default:
239 return Triple.getArchName();
240 }
241}

289Tool *ToolChain::getTool(Action::ActionClass AC) const {
290 switch (AC) {
291 case Action::AssembleJobClass:
292 return getAssemble();
293
294 case Action::LinkJobClass:
295 return getLink();
296
297 case Action::InputClass:
298 case Action::BindArchClass:
299 case Action::OffloadClass:
300 case Action::LipoJobClass:
301 case Action::DsymutilJobClass:
302 case Action::VerifyDebugInfoJobClass:
303 llvm_unreachable("Invalid tool kind.");
304
305 case Action::CompileJobClass:
306 case Action::PrecompileJobClass:
307 case Action::HeaderModulePrecompileJobClass:
308 case Action::PreprocessJobClass:
309 case Action::AnalyzeJobClass:
310 case Action::MigrateJobClass:
311 case Action::VerifyPCHJobClass:
312 case Action::BackendJobClass:
313 return getClang();
314
315 case Action::OffloadBundlingJobClass:
316 case Action::OffloadUnbundlingJobClass:
317 return getOffloadBundler();
318 }
319
320 llvm_unreachable("Invalid tool kind.");
321}

323static StringRef getArchNameForCompilerRTLib(const ToolChain &TC,
324 const ArgList &Args) {
325 const llvm::Triple &Triple = TC.getTriple();
326 bool IsWindows = Triple.isOSWindows();
327
328 if (TC.getArch() == llvm::Triple::arm || TC.getArch() == llvm::Triple::armeb)
329 return (arm::getARMFloatABI(TC, Args) == arm::FloatABI::Hard && !IsWindows)
330 ? "armhf"
331 : "arm";
332
333 // For historic reasons, Android library is using i686 instead of i386.
334 if (TC.getArch() == llvm::Triple::x86 && Triple.isAndroid())
335 return "i686";
336
337 if (Triple.getEnvironment() == llvm::Triple::CheriPurecap) {
338 assert(Triple.getSubArch() != llvm::Triple::NoSubArch && "purecap triple should have subarch");
339 return Triple.getArchName();
340 }
341
342 return llvm::Triple::getArchTypeName(TC.getArch());
343}

370std::string ToolChain::getCompilerRT(const ArgList &Args, StringRef Component,
371 FileType Type) const {
372 const llvm::Triple &TT = getTriple();
373 bool IsITANMSVCWindows =
374 TT.isWindowsMSVCEnvironment() || TT.isWindowsItaniumEnvironment();
375
376 const char *Prefix =
377 IsITANMSVCWindows || Type == ToolChain::FT_Object ? "" : "lib";
378 const char *Suffix;
379 switch (Type) {
380 case ToolChain::FT_Object:
381 Suffix = IsITANMSVCWindows ? ".obj" : ".o";
382 break;
383 case ToolChain::FT_Static:
384 Suffix = IsITANMSVCWindows ? ".lib" : ".a";
385 break;
386 case ToolChain::FT_Shared:
387 Suffix = Triple.isOSWindows()
388 ? (Triple.isWindowsGNUEnvironment() ? ".dll.a" : ".lib")
389 : ".so";
390 break;
391 }
392
393 for (const auto &LibPath : getLibraryPaths()) {
394 SmallString<128> P(LibPath);
395 llvm::sys::path::append(P, Prefix + Twine("clang_rt.") + Component + Suffix);
396 if (getVFS().exists(P))
397 return P.str();
398 }
399
400 StringRef Arch = getArchNameForCompilerRTLib(*this, Args);
401 const char *Env = TT.isAndroid() ? "-android" : "";
402 SmallString<128> Path(getCompilerRTPath());
403 llvm::sys::path::append(Path, Prefix + Twine("clang_rt.") + Component + "-" +
404 Arch + Env + Suffix);
405 return Path.str();
406}

415Optional<std::string> ToolChain::getRuntimePath() const {
416 SmallString<128> P;
417
418 // First try the triple passed to driver as --target=<triple>.
419 P.assign(D.ResourceDir);
420 llvm::sys::path::append(P, "lib", D.getTargetTriple());
421 if (getVFS().exists(P))
422 return llvm::Optional<std::string>(P.str());
423
424 // Second try the normalized triple.
425 P.assign(D.ResourceDir);
426 llvm::sys::path::append(P, "lib", Triple.str());
427 if (getVFS().exists(P))
428 return llvm::Optional<std::string>(P.str());
429
430 return None;
431}

433Optional<std::string> ToolChain::getCXXStdlibPath() const {
434 SmallString<128> P;
435
436 // First try the triple passed to driver as --target=<triple>.
437 P.assign(D.Dir);
438 llvm::sys::path::append(P, "..", "lib", D.getTargetTriple(), "c++");
439 if (getVFS().exists(P))
440 return llvm::Optional<std::string>(P.str());
441
442 // Second try the normalized triple.
443 P.assign(D.Dir);
444 llvm::sys::path::append(P, "..", "lib", Triple.str(), "c++");
445 if (getVFS().exists(P))
446 return llvm::Optional<std::string>(P.str());
447
448 return None;
449}

495std::string ToolChain::GetLinkerPath() const {
496 const Arg* A = Args.getLastArg(options::OPT_fuse_ld_EQ);
497 StringRef UseLinker = A ? A->getValue() : CLANG_DEFAULT_LINKER;
498
499 if (llvm::sys::path::is_absolute(UseLinker)) {
500 // If we're passed what looks like an absolute path, don't attempt to
501 // second-guess that.
502 if (llvm::sys::fs::can_execute(UseLinker))
503 return UseLinker;
504 } else if (UseLinker.empty() || UseLinker == "ld") {
505 // If we're passed -fuse-ld= with no argument, or with the argument ld,
506 // then use whatever the default system linker is.
507 return GetProgramPath(getDefaultLinker());
508 } else {
509 llvm::SmallString<8> LinkerName;
510 if (Triple.isOSDarwin())
511 LinkerName.append("ld64.");
512 else
513 LinkerName.append("ld.");
514 LinkerName.append(UseLinker);
515
516 std::string LinkerPath(GetProgramPath(LinkerName.c_str()));
517 if (llvm::sys::fs::can_execute(LinkerPath))
518 return LinkerPath;
519 }
520
521 if (A)
522 getDriver().Diag(diag::err_drv_invalid_linker_name) << A->getAsString(Args);
523
524 return GetProgramPath(getDefaultLinker());
525}

535bool ToolChain::isCrossCompiling() const {
536 llvm::Triple HostTriple(LLVM_HOST_TRIPLE);
537 switch (HostTriple.getArch()) {
538 // The A32/T32/T16 instruction sets are not separate architectures in this
539 // context.
540 case llvm::Triple::arm:
541 case llvm::Triple::armeb:
542 case llvm::Triple::thumb:
543 case llvm::Triple::thumbeb:
544 return getArch() != llvm::Triple::arm && getArch() != llvm::Triple::thumb &&
545 getArch() != llvm::Triple::armeb && getArch() != llvm::Triple::thumbeb;
546 default:
547 return HostTriple.getArch() != getArch();
548 }
549}

562 if (Model == "single") {
563 // FIXME: 'single' is only supported on ARM and WebAssembly so far.
564 return Triple.getArch() == llvm::Triple::arm ||
565 Triple.getArch() == llvm::Triple::armeb ||
566 Triple.getArch() == llvm::Triple::thumb ||
567 Triple.getArch() == llvm::Triple::thumbeb ||
568 Triple.getArch() == llvm::Triple::wasm32 ||
569 Triple.getArch() == llvm::Triple::wasm64;
570 } else if (Model == "posix")
571 return true;
572
573 return false;
574}

576std::string ToolChain::ComputeLLVMTriple(const ArgList &Args,
577 types::ID InputType) const {
578 switch (getTriple().getArch()) {
579 default:
580 return getTripleString();
581
582 case llvm::Triple::x86_64: {
583 llvm::Triple Triple = getTriple();
584 if (!Triple.isOSBinFormatMachO())
585 return getTripleString();
586
587 if (Arg *A = Args.getLastArg(options::OPT_march_EQ)) {
588 // x86_64h goes in the triple. Other -march options just use the
589 // vanilla triple we already have.
590 StringRef MArch = A->getValue();
591 if (MArch == "x86_64h")
592 Triple.setArchName(MArch);
593 }
594 return Triple.getTriple();
595 }
596 case llvm::Triple::aarch64: {
597 llvm::Triple Triple = getTriple();
598 if (!Triple.isOSBinFormatMachO())
599 return getTripleString();
600
601 // FIXME: older versions of ld64 expect the "arm64" component in the actual
602 // triple string and query it to determine whether an LTO file can be
603 // handled. Remove this when we don't care any more.
604 Triple.setArchName("arm64");
605 return Triple.getTriple();
606 }
607 case llvm::Triple::arm:
608 case llvm::Triple::armeb:
609 case llvm::Triple::thumb:
610 case llvm::Triple::thumbeb: {
611 // FIXME: Factor into subclasses.
612 llvm::Triple Triple = getTriple();
613 bool IsBigEndian = getTriple().getArch() == llvm::Triple::armeb ||
614 getTriple().getArch() == llvm::Triple::thumbeb;
615
616 // Handle pseudo-target flags '-mlittle-endian'/'-EL' and
617 // '-mbig-endian'/'-EB'.
618 if (Arg *A = Args.getLastArg(options::OPT_mlittle_endian,
619 options::OPT_mbig_endian)) {
620 IsBigEndian = !A->getOption().matches(options::OPT_mlittle_endian);
621 }
622
623 // Thumb2 is the default for V7 on Darwin.
624 //
625 // FIXME: Thumb should just be another -target-feaure, not in the triple.
626 StringRef MCPU, MArch;
627 if (const Arg *A = Args.getLastArg(options::OPT_mcpu_EQ))
628 MCPU = A->getValue();
629 if (const Arg *A = Args.getLastArg(options::OPT_march_EQ))
630 MArch = A->getValue();
631 std::string CPU =
632 Triple.isOSBinFormatMachO()
633 ? tools::arm::getARMCPUForMArch(MArch, Triple).str()
634 : tools::arm::getARMTargetCPU(MCPU, MArch, Triple);
635 StringRef Suffix =
636 tools::arm::getLLVMArchSuffixForARM(CPU, MArch, Triple);
637 bool IsMProfile = ARM::parseArchProfile(Suffix) == ARM::ProfileKind::M;
638 bool ThumbDefault = IsMProfile || (ARM::parseArchVersion(Suffix) == 7 &&
639 getTriple().isOSBinFormatMachO());
640 // FIXME: this is invalid for WindowsCE
641 if (getTriple().isOSWindows())
642 ThumbDefault = true;
643 std::string ArchName;
644 if (IsBigEndian)
645 ArchName = "armeb";
646 else
647 ArchName = "arm";
648
649 // Check if ARM ISA was explicitly selected (using -mno-thumb or -marm) for
650 // M-Class CPUs/architecture variants, which is not supported.
651 bool ARMModeRequested = !Args.hasFlag(options::OPT_mthumb,
652 options::OPT_mno_thumb, ThumbDefault);
653 if (IsMProfile && ARMModeRequested) {
654 if (!MCPU.empty())
655 getDriver().Diag(diag::err_cpu_unsupported_isa) << CPU << "ARM";
656 else
657 getDriver().Diag(diag::err_arch_unsupported_isa)
658 << tools::arm::getARMArch(MArch, getTriple()) << "ARM";
659 }
660
661 // Check to see if an explicit choice to use thumb has been made via
662 // -mthumb. For assembler files we must check for -mthumb in the options
663 // passed to the assembler via -Wa or -Xassembler.
664 bool IsThumb = false;
665 if (InputType != types::TY_PP_Asm)
666 IsThumb = Args.hasFlag(options::OPT_mthumb, options::OPT_mno_thumb,
667 ThumbDefault);
668 else {
669 // Ideally we would check for these flags in
670 // CollectArgsForIntegratedAssembler but we can't change the ArchName at
671 // that point. There is no assembler equivalent of -mno-thumb, -marm, or
672 // -mno-arm.
673 for (const auto *A :
674 Args.filtered(options::OPT_Wa_COMMA, options::OPT_Xassembler)) {
675 for (StringRef Value : A->getValues()) {
676 if (Value == "-mthumb")
677 IsThumb = true;
678 }
679 }
680 }
681 // Assembly files should start in ARM mode, unless arch is M-profile, or
682 // -mthumb has been passed explicitly to the assembler. Windows is always
683 // thumb.
684 if (IsThumb || IsMProfile || getTriple().isOSWindows()) {
685 if (IsBigEndian)
686 ArchName = "thumbeb";
687 else
688 ArchName = "thumb";
689 }
690 Triple.setArchName(ArchName + Suffix.str());
691
692 return Triple.getTriple();
693 }
694 }
695}

744 if (LibName == "none")

753 if (GetRuntimeLibType(Args) == RLT_Libgcc)
754 getDriver().Diag(diag::err_drv_incompatible_unwindlib);
755 return ToolChain::UNW_CompilerRT;
756 } else if (LibName == "libgcc")
757 return ToolChain::UNW_Libgcc;
758
759 if (A)
760 getDriver().Diag(diag::err_drv_invalid_unwindlib_name)
761 << A->getAsString(Args);

763 return GetDefaultUnwindLibType();

873bool ToolChain::AddFastMathRuntimeIfAvailable(const ArgList &Args,
874 ArgStringList &CmdArgs) const {
875 // Do not check for -fno-fast-math or -fno-unsafe-math when -Ofast passed
876 // (to keep the linker options consistent with gcc and clang itself).
877 if (!isOptimizationLevelFast(Args)) {
878 // Check if -ffast-math or -funsafe-math.
879 Arg *A =
880 Args.getLastArg(options::OPT_ffast_math, options::OPT_fno_fast_math,
881 options::OPT_funsafe_math_optimizations,
882 options::OPT_fno_unsafe_math_optimizations);
883
884 if (!A || A->getOption().getID() == options::OPT_fno_fast_math ||
885 A->getOption().getID() == options::OPT_fno_unsafe_math_optimizations)
886 return false;
887 }
888 // If crtfastmath.o exists add it to the arguments.
889 std::string Path = GetFilePath("crtfastmath.o");
890 if (Path == "crtfastmath.o") // Not found.
891 return false;
892
893 CmdArgs.push_back(Args.MakeArgString(Path));
894 return true;
895}

897SanitizerMask ToolChain::getSupportedSanitizers() const {
898 // Return sanitizers which don't require runtime support and are not
899 // platform dependent.
900
901 SanitizerMask Res = (SanitizerKind::Undefined & ~SanitizerKind::Vptr &
902 ~SanitizerKind::Function) |
903 (SanitizerKind::CFI & ~SanitizerKind::CFIICall) |
904 SanitizerKind::CFICastStrict |
905 SanitizerKind::UnsignedIntegerOverflow |
906 SanitizerKind::ImplicitConversion |
907 SanitizerKind::Nullability | SanitizerKind::LocalBounds;
908 if (getTriple().getArch() == llvm::Triple::x86 ||
909 getTriple().getArch() == llvm::Triple::x86_64 ||
910 getTriple().getArch() == llvm::Triple::arm ||
911 getTriple().getArch() == llvm::Triple::aarch64 ||
912 getTriple().getArch() == llvm::Triple::wasm32 ||
913 getTriple().getArch() == llvm::Triple::wasm64)
914 Res |= SanitizerKind::CFIICall;
915 if (getTriple().getArch() == llvm::Triple::x86_64 ||
916 getTriple().getArch() == llvm::Triple::aarch64)
917 Res |= SanitizerKind::ShadowCallStack;
918 return Res;
919}

927static VersionTuple separateMSVCFullVersion(unsigned Version) {
928 if (Version < 100)
929 return VersionTuple(Version);
930
931 if (Version < 10000)
932 return VersionTuple(Version / 100, Version % 100);
933
934 unsigned Build = 0, Factor = 1;
935 for (; Version > 10000; Version = Version / 10, Factor = Factor * 10)
936 Build = Build + (Version % 10) * Factor;
937 return VersionTuple(Version / 100, Version % 100, Build);
938}

940VersionTuple
941ToolChain::computeMSVCVersion(const Driver *D,
942 const llvm::opt::ArgList &Args) const {
943 const Arg *MSCVersion = Args.getLastArg(options::OPT_fmsc_version);
944 const Arg *MSCompatibilityVersion =
945 Args.getLastArg(options::OPT_fms_compatibility_version);
946
947 if (MSCVersion && MSCompatibilityVersion) {
948 if (D)
949 D->Diag(diag::err_drv_argument_not_allowed_with)
950 << MSCVersion->getAsString(Args)
951 << MSCompatibilityVersion->getAsString(Args);
952 return VersionTuple();
953 }
954
955 if (MSCompatibilityVersion) {
956 VersionTuple MSVT;
957 if (MSVT.tryParse(MSCompatibilityVersion->getValue())) {
958 if (D)
959 D->Diag(diag::err_drv_invalid_value)
960 << MSCompatibilityVersion->getAsString(Args)
961 << MSCompatibilityVersion->getValue();
962 } else {
963 return MSVT;
964 }
965 }
966
967 if (MSCVersion) {
968 unsigned Version = 0;
969 if (StringRef(MSCVersion->getValue()).getAsInteger(10, Version)) {
970 if (D)
971 D->Diag(diag::err_drv_invalid_value)
972 << MSCVersion->getAsString(Args) << MSCVersion->getValue();
973 } else {
974 return separateMSVCFullVersion(Version);
975 }
976 }
977
978 return VersionTuple();
979}

981llvm::opt::DerivedArgList *ToolChain::TranslateOpenMPTargetArgs(
982 const llvm::opt::DerivedArgList &Args, bool SameTripleAsHost,
983 SmallVectorImpl<llvm::opt::Arg *> &AllocatedArgs) const {
984 DerivedArgList *DAL = new DerivedArgList(Args.getBaseArgs());
985 const OptTable &Opts = getDriver().getOpts();
986 bool Modified = false;
987
988 // Handle -Xopenmp-target flags
989 for (auto *A : Args) {
990 // Exclude flags which may only apply to the host toolchain.
991 // Do not exclude flags when the host triple (AuxTriple)
992 // matches the current toolchain triple. If it is not present
993 // at all, target and host share a toolchain.
994 if (A->getOption().matches(options::OPT_m_Group)) {
995 if (SameTripleAsHost)
996 DAL->append(A);
997 else
998 Modified = true;
999 continue;
1000 }
1001
1002 unsigned Index;
1003 unsigned Prev;
1004 bool XOpenMPTargetNoTriple =
1005 A->getOption().matches(options::OPT_Xopenmp_target);
1006
1007 if (A->getOption().matches(options::OPT_Xopenmp_target_EQ)) {
1008 // Passing device args: -Xopenmp-target=<triple> -opt=val.
1009 if (A->getValue(0) == getTripleString())
1010 Index = Args.getBaseArgs().MakeIndex(A->getValue(1));
1011 else
1012 continue;
1013 } else if (XOpenMPTargetNoTriple) {
1014 // Passing device args: -Xopenmp-target -opt=val.
1015 Index = Args.getBaseArgs().MakeIndex(A->getValue(0));
1016 } else {
1017 DAL->append(A);
1018 continue;
1019 }
1020
1021 // Parse the argument to -Xopenmp-target.
1022 Prev = Index;
1023 std::unique_ptr<Arg> XOpenMPTargetArg(Opts.ParseOneArg(Args, Index));
1024 if (!XOpenMPTargetArg || Index > Prev + 1) {
1025 getDriver().Diag(diag::err_drv_invalid_Xopenmp_target_with_args)
1026 << A->getAsString(Args);
1027 continue;
1028 }
1029 if (XOpenMPTargetNoTriple && XOpenMPTargetArg &&
1030 Args.getAllArgValues(options::OPT_fopenmp_targets_EQ).size() != 1) {
1031 getDriver().Diag(diag::err_drv_Xopenmp_target_missing_triple);
1032 continue;
1033 }
1034 XOpenMPTargetArg->setBaseArg(A);
1035 A = XOpenMPTargetArg.release();
1036 AllocatedArgs.push_back(A);
1037 DAL->append(A);
1038 Modified = true;
1039 }
1040
1041 if (Modified)
1042 return DAL;
1043
1044 delete DAL;
1045 return nullptr;
1046}
