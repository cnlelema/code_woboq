67void tools::handleTargetFeaturesGroup(const ArgList &Args,
68 std::vector<StringRef> &Features,
69 OptSpecifier Group) {
70 for (const Arg *A : Args.filtered(Group)) {
71 StringRef Name = A->getOption().getName();
72 A->claim();
73
74 // Skip over "-m".
75 assert(Name.startswith("m") && "Invalid feature name.");
76 Name = Name.substr(1);
77
78 bool IsNegative = Name.startswith("no-");
79 if (IsNegative)
80 Name = Name.substr(3);
81 Features.push_back(Args.MakeArgString((IsNegative ? "-" : "+") + Name));
82 }
83}

85void tools::addDirectoryList(const ArgList &Args, ArgStringList &CmdArgs,
86 const char *ArgName, const char *EnvVar) {
87 const char *DirList = ::getenv(EnvVar);
88 bool CombinedArg = false;
89
90 if (!DirList)
91 return; // Nothing to do.
92
93 StringRef Name(ArgName);
94 if (Name.equals("-I") || Name.equals("-L"))
95 CombinedArg = true;
96
97 StringRef Dirs(DirList);
98 if (Dirs.empty()) // Empty string should not add '.'.
99 return;
100
101 StringRef::size_type Delim;
102 while ((Delim = Dirs.find(llvm::sys::EnvPathSeparator)) != StringRef::npos) {
103 if (Delim == 0) { // Leading colon.
104 if (CombinedArg) {
105 CmdArgs.push_back(Args.MakeArgString(std::string(ArgName) + "."));
106 } else {
107 CmdArgs.push_back(ArgName);
108 CmdArgs.push_back(".");
109 }
110 } else {
111 if (CombinedArg) {
112 CmdArgs.push_back(
113 Args.MakeArgString(std::string(ArgName) + Dirs.substr(0, Delim)));
114 } else {
115 CmdArgs.push_back(ArgName);
116 CmdArgs.push_back(Args.MakeArgString(Dirs.substr(0, Delim)));
117 }
118 }
119 Dirs = Dirs.substr(Delim + 1);
120 }
121
122 if (Dirs.empty()) { // Trailing colon.
123 if (CombinedArg) {
124 CmdArgs.push_back(Args.MakeArgString(std::string(ArgName) + "."));
125 } else {
126 CmdArgs.push_back(ArgName);
127 CmdArgs.push_back(".");
128 }
129 } else { // Add the last path.
130 if (CombinedArg) {
131 CmdArgs.push_back(Args.MakeArgString(std::string(ArgName) + Dirs));
132 } else {
133 CmdArgs.push_back(ArgName);
134 CmdArgs.push_back(Args.MakeArgString(Dirs));
135 }
136 }

139void tools::AddLinkerInputs(const ToolChain &TC, const InputInfoList &Inputs,
140 const ArgList &Args, ArgStringList &CmdArgs,
141 const JobAction &JA) {
142 const Driver &D = TC.getDriver();
143
144 // Add extra linker input arguments which are not treated as inputs
145 // (constructed via -Xarch_).
146 Args.AddAllArgValues(CmdArgs, options::OPT_Zlinker_input);
147
148 for (const auto &II : Inputs) {
149 // If the current tool chain refers to an OpenMP or HIP offloading host, we
150 // should ignore inputs that refer to OpenMP or HIP offloading devices -
151 // they will be embedded according to a proper linker script.
152 if (auto *IA = II.getAction())
153 if ((JA.isHostOffloading(Action::OFK_OpenMP) &&
154 IA->isDeviceOffloading(Action::OFK_OpenMP)) ||
155 (JA.isHostOffloading(Action::OFK_HIP) &&
156 IA->isDeviceOffloading(Action::OFK_HIP)))
157 continue;
158
159 if (!TC.HasNativeLLVMSupport() && types::isLLVMIR(II.getType()))
160 // Don't try to pass LLVM inputs unless we have native support.
161 D.Diag(diag::err_drv_no_linker_llvm_support) << TC.getTripleString();
162
163 // Add filenames immediately.
164 if (II.isFilename()) {
165 CmdArgs.push_back(II.getFilename());
166 continue;
167 }
168
169 // Otherwise, this is a linker input argument.
170 const Arg &A = II.getInputArg();
171
172 // Handle reserved library options.
173 if (A.getOption().matches(options::OPT_Z_reserved_lib_stdcxx))
174 TC.AddCXXStdlibLibArgs(Args, CmdArgs);
175 else if (A.getOption().matches(options::OPT_Z_reserved_lib_cckext))
176 TC.AddCCKextLibArgs(Args, CmdArgs);
177 else if (A.getOption().matches(options::OPT_z)) {
178 // Pass -z prefix for gcc linker compatibility.
179 A.claim();
180 A.render(Args, CmdArgs);
181 } else {
182 A.renderAsInput(Args, CmdArgs);
183 }
184 }
185
186 // LIBRARY_PATH - included following the user specified library paths.
187 // and only supported on native toolchains.
188 if (!TC.isCrossCompiling()) {
189 addDirectoryList(Args, CmdArgs, "-L", "LIBRARY_PATH");
190 }
191}

205/// Get the (LLVM) name of the R600 gpu we are targeting.
206static std::string getR600TargetGPU(const ArgList &Args) {
207 if (Arg *A = Args.getLastArg(options::OPT_mcpu_EQ)) {
208 const char *GPUName = A->getValue();
209 return llvm::StringSwitch<const char *>(GPUName)
210 .Cases("rv630", "rv635", "r600")
211 .Cases("rv610", "rv620", "rs780", "rs880")
212 .Case("rv740", "rv770")
213 .Case("palm", "cedar")
214 .Cases("sumo", "sumo2", "sumo")
215 .Case("hemlock", "cypress")
216 .Case("aruba", "cayman")
217 .Default(GPUName);
218 }
219 return "";
220}

229/// Get the (LLVM) name of the WebAssembly cpu we are targeting.
230static StringRef getWebAssemblyTargetCPU(const ArgList &Args) {
231 // If we have -mcpu=, use that.
232 if (Arg *A = Args.getLastArg(options::OPT_mcpu_EQ)) {
233 StringRef CPU = A->getValue();
234
235#ifdef __wasm__
236 // Handle "native" by examining the host. "native" isn't meaningful when
237 // cross compiling, so only support this when the host is also WebAssembly.
238 if (CPU == "native")
239 return llvm::sys::getHostCPUName();
240#endif
241
242 return CPU;
243 }
244
245 return "generic";
246}

248std::string tools::getCPUName(const ArgList &Args, const llvm::Triple &T,
249 bool FromAs) {
250 Arg *A;
251
252 switch (T.getArch()) {
253 default:
254 return "";
255
256 case llvm::Triple::aarch64:
257 case llvm::Triple::aarch64_be:
258 return aarch64::getAArch64TargetCPU(Args, T, A);
259
260 case llvm::Triple::arm:
261 case llvm::Triple::armeb:
262 case llvm::Triple::thumb:
263 case llvm::Triple::thumbeb: {
264 StringRef MArch, MCPU;
265 arm::getARMArchCPUFromArgs(Args, MArch, MCPU, FromAs);
266 return arm::getARMTargetCPU(MCPU, MArch, T);
267 }
268
269 case llvm::Triple::avr:
270 if (const Arg *A = Args.getLastArg(options::OPT_mmcu_EQ))
271 return A->getValue();
272 return "";
273
274 case llvm::Triple::mips:
275 case llvm::Triple::mipsel:
276 case llvm::Triple::mips64:
277 case llvm::Triple::mips64el:
278 case llvm::Triple::cheri: {
279 StringRef CPUName;
280 StringRef ABIName;
281 mips::getMipsCPUAndABI(Args, T, CPUName, ABIName);
282 return CPUName;
283 }
284
285 case llvm::Triple::nvptx:
286 case llvm::Triple::nvptx64:
287 if (const Arg *A = Args.getLastArg(options::OPT_march_EQ))
288 return A->getValue();
289 return "";
290
291 case llvm::Triple::ppc:
292 case llvm::Triple::ppc64:
293 case llvm::Triple::ppc64le: {
294 std::string TargetCPUName = ppc::getPPCTargetCPU(Args);
295 // LLVM may default to generating code for the native CPU,
296 // but, like gcc, we default to a more generic option for
297 // each architecture. (except on Darwin)
298 if (TargetCPUName.empty() && !T.isOSDarwin()) {
299 if (T.getArch() == llvm::Triple::ppc64)
300 TargetCPUName = "ppc64";
301 else if (T.getArch() == llvm::Triple::ppc64le)
302 TargetCPUName = "ppc64le";
303 else
304 TargetCPUName = "ppc";
305 }
306 return TargetCPUName;
307 }
308
309 case llvm::Triple::bpfel:
310 case llvm::Triple::bpfeb:
311 case llvm::Triple::sparc:
312 case llvm::Triple::sparcel:
313 case llvm::Triple::sparcv9:
314 if (const Arg *A = Args.getLastArg(options::OPT_mcpu_EQ))
315 return A->getValue();
316 return "";
317
318 case llvm::Triple::x86:
319 case llvm::Triple::x86_64:
320 return x86::getX86TargetCPU(Args, T);
321
322 case llvm::Triple::hexagon:
323 return "hexagon" +
324 toolchains::HexagonToolChain::GetTargetCPUVersion(Args).str();
325
326 case llvm::Triple::lanai:
327 return getLanaiTargetCPU(Args);
328
329 case llvm::Triple::systemz:
330 return systemz::getSystemZTargetCPU(Args);
331
332 case llvm::Triple::r600:
333 case llvm::Triple::amdgcn:
334 return getR600TargetGPU(Args);
335
336 case llvm::Triple::wasm32:
337 case llvm::Triple::wasm64:
338 return getWebAssemblyTargetCPU(Args);
339 }
340}

342unsigned tools::getLTOParallelism(const ArgList &Args, const Driver &D) {
343 unsigned Parallelism = 0;
344 Arg *LtoJobsArg = Args.getLastArg(options::OPT_flto_jobs_EQ);

349 return Parallelism;

357void tools::AddGoldPlugin(const ToolChain &ToolChain, const ArgList &Args,
358 ArgStringList &CmdArgs, const InputInfo &Output,
359 const InputInfo &Input, bool IsThinLTO) {
360 // Tell the linker to load the plugin. This has to come before AddLinkerInputs
361 // as gold requires -plugin to come before any -plugin-opt that -Wl might
362 // forward.
363 CmdArgs.push_back("-plugin");
364
365#if defined(_WIN32)
366 const char *Suffix = ".dll";
367#elif defined(__APPLE__)
368 const char *Suffix = ".dylib";
369#else
370 const char *Suffix = ".so";
371#endif
372
373 SmallString<1024> Plugin;
374 llvm::sys::path::native(Twine(ToolChain.getDriver().Dir) +
375 "/../lib" CLANG_LIBDIR_SUFFIX "/LLVMgold" +
376 Suffix,
377 Plugin);
378 CmdArgs.push_back(Args.MakeArgString(Plugin));
379
380 // Try to pass driver level flags relevant to LTO code generation down to
381 // the plugin.
382
383 // Handle flags for selecting CPU variants.
384 std::string CPU = getCPUName(Args, ToolChain.getTriple());
385 if (!CPU.empty())
386 CmdArgs.push_back(Args.MakeArgString(Twine("-plugin-opt=mcpu=") + CPU));
387
388 if (Arg *A = Args.getLastArg(options::OPT_O_Group)) {
389 StringRef OOpt;
390 if (A->getOption().matches(options::OPT_O4) ||
391 A->getOption().matches(options::OPT_Ofast))
392 OOpt = "3";
393 else if (A->getOption().matches(options::OPT_O))
394 OOpt = A->getValue();
395 else if (A->getOption().matches(options::OPT_O0))
396 OOpt = "0";
397 if (!OOpt.empty())
398 CmdArgs.push_back(Args.MakeArgString(Twine("-plugin-opt=O") + OOpt));
399 }
400
401 if (Args.hasArg(options::OPT_gsplit_dwarf)) {
402 CmdArgs.push_back(
403 Args.MakeArgString(Twine("-plugin-opt=dwo_dir=") +
404 Output.getFilename() + "_dwo"));
405 }
406
407 if (IsThinLTO)
408 CmdArgs.push_back("-plugin-opt=thinlto");
409
410 if (unsigned Parallelism = getLTOParallelism(Args, ToolChain.getDriver()))
411 CmdArgs.push_back(
412 Args.MakeArgString("-plugin-opt=jobs=" + Twine(Parallelism)));
413
414 // If an explicit debugger tuning argument appeared, pass it along.
415 if (Arg *A = Args.getLastArg(options::OPT_gTune_Group,
416 options::OPT_ggdbN_Group)) {
417 if (A->getOption().matches(options::OPT_glldb))
418 CmdArgs.push_back("-plugin-opt=-debugger-tune=lldb");
419 else if (A->getOption().matches(options::OPT_gsce))
420 CmdArgs.push_back("-plugin-opt=-debugger-tune=sce");
421 else
422 CmdArgs.push_back("-plugin-opt=-debugger-tune=gdb");
423 }
424
425 bool UseSeparateSections =
426 isUseSeparateSections(ToolChain.getEffectiveTriple());
427
428 if (Args.hasFlag(options::OPT_ffunction_sections,
429 options::OPT_fno_function_sections, UseSeparateSections)) {
430 CmdArgs.push_back("-plugin-opt=-function-sections");
431 }
432
433 if (Args.hasFlag(options::OPT_fdata_sections, options::OPT_fno_data_sections,
434 UseSeparateSections)) {
435 CmdArgs.push_back("-plugin-opt=-data-sections");
436 }
437
438 if (Arg *A = getLastProfileSampleUseArg(Args)) {
439 StringRef FName = A->getValue();
440 if (!llvm::sys::fs::exists(FName))
441 ToolChain.getDriver().Diag(diag::err_drv_no_such_file) << FName;
442 else
443 CmdArgs.push_back(
444 Args.MakeArgString(Twine("-plugin-opt=sample-profile=") + FName));
445 }
446
447 auto *CSPGOGenerateArg = Args.getLastArg(options::OPT_fcs_profile_generate,
448 options::OPT_fcs_profile_generate_EQ,
449 options::OPT_fno_profile_generate);
450 if (CSPGOGenerateArg &&
451 CSPGOGenerateArg->getOption().matches(options::OPT_fno_profile_generate))
452 CSPGOGenerateArg = nullptr;
453
454 auto *ProfileUseArg = getLastProfileUseArg(Args);
455
456 if (CSPGOGenerateArg) {
457 CmdArgs.push_back(Args.MakeArgString("-plugin-opt=cs-profile-generate"));
458 if (CSPGOGenerateArg->getOption().matches(
459 options::OPT_fcs_profile_generate_EQ)) {
460 SmallString<128> Path(CSPGOGenerateArg->getValue());
461 llvm::sys::path::append(Path, "default_%m.profraw");
462 CmdArgs.push_back(
463 Args.MakeArgString(Twine("-plugin-opt=cs-profile-path=") + Path));
464 } else
465 CmdArgs.push_back(
466 Args.MakeArgString("-plugin-opt=cs-profile-path=default_%m.profraw"));
467 } else if (ProfileUseArg) {
468 SmallString<128> Path(
469 ProfileUseArg->getNumValues() == 0 ? "" : ProfileUseArg->getValue());
470 if (Path.empty() || llvm::sys::fs::is_directory(Path))
471 llvm::sys::path::append(Path, "default.profdata");
472 CmdArgs.push_back(Args.MakeArgString(Twine("-plugin-opt=cs-profile-path=") +
473 Path));
474 }
475
476 // Need this flag to turn on new pass manager via Gold plugin.
477 if (Args.hasFlag(options::OPT_fexperimental_new_pass_manager,
478 options::OPT_fno_experimental_new_pass_manager,
479 /* Default */ ENABLE_EXPERIMENTAL_NEW_PASS_MANAGER)) {
480 CmdArgs.push_back("-plugin-opt=new-pass-manager");
481 }
482
483 // Setup statistics file output.
484 SmallString<128> StatsFile =
485 getStatsFileName(Args, Output, Input, ToolChain.getDriver());
486 if (!StatsFile.empty())
487 CmdArgs.push_back(
488 Args.MakeArgString(Twine("-plugin-opt=stats-file=") + StatsFile));
489}

504bool tools::addOpenMPRuntime(ArgStringList &CmdArgs, const ToolChain &TC,
505 const ArgList &Args, bool IsOffloadingHost,
506 bool GompNeedsRT) {
507 if (!Args.hasFlag(options::OPT_fopenmp, options::OPT_fopenmp_EQ,
508 options::OPT_fno_openmp, false))
509 return false;
510
511 switch (TC.getDriver().getOpenMPRuntime(Args)) {
512 case Driver::OMPRT_OMP:
513 CmdArgs.push_back("-lomp");
514 break;
515 case Driver::OMPRT_GOMP:
516 CmdArgs.push_back("-lgomp");
517
518 if (GompNeedsRT)
519 CmdArgs.push_back("-lrt");
520 break;
521 case Driver::OMPRT_IOMP5:
522 CmdArgs.push_back("-liomp5");
523 break;
524 case Driver::OMPRT_Unknown:
525 // Already diagnosed.
526 return false;
527 }
528
529 if (IsOffloadingHost)
530 CmdArgs.push_back("-lomptarget");
531
532 addArchSpecificRPath(TC, Args, CmdArgs);
533
534 return true;
535}

547 if (IsShared) {
548 addArchSpecificRPath(TC, Args, CmdArgs);

552// Tries to use a file with the list of dynamic symbols that need to be exported
553// from the runtime library. Returns true if the file was found.
554static bool addSanitizerDynamicList(const ToolChain &TC, const ArgList &Args,
555 ArgStringList &CmdArgs,
556 StringRef Sanitizer) {
557 // Solaris ld defaults to --export-dynamic behaviour but doesn't support
558 // the option, so don't try to pass it.
559 if (TC.getTriple().getOS() == llvm::Triple::Solaris)
560 return true;
561 // Myriad is static linking only. Furthermore, some versions of its
562 // linker have the bug where --export-dynamic overrides -static, so
563 // don't use --export-dynamic on that platform.
564 if (TC.getTriple().getVendor() == llvm::Triple::Myriad)
565 return true;
566 SmallString<128> SanRT(TC.getCompilerRT(Args, Sanitizer));
567 if (llvm::sys::fs::exists(SanRT + ".syms")) {
568 CmdArgs.push_back(Args.MakeArgString("--dynamic-list=" + SanRT + ".syms"));
569 return true;
570 }
571 return false;
572}

574void tools::linkSanitizerRuntimeDeps(const ToolChain &TC,
575 ArgStringList &CmdArgs) {
576 // Force linking against the system libraries sanitizers depends on
577 // (see PR15823 why this is necessary).
578 CmdArgs.push_back("--no-as-needed");
579 // There's no libpthread or librt on RTEMS & Android.
580 if (TC.getTriple().getOS() != llvm::Triple::RTEMS &&
581 !TC.getTriple().isAndroid()) {
582 CmdArgs.push_back("-lpthread");
583 if (!TC.getTriple().isOSOpenBSD())
584 CmdArgs.push_back("-lrt");
585 }
586 CmdArgs.push_back("-lm");
587 // There's no libdl on all OSes.
588 if (!TC.getTriple().isOSFreeBSD() &&
589 !TC.getTriple().isOSNetBSD() &&
590 !TC.getTriple().isOSOpenBSD() &&
591 TC.getTriple().getOS() != llvm::Triple::RTEMS)
592 CmdArgs.push_back("-ldl");
593 // Required for backtrace on some OSes
594 if (TC.getTriple().isOSFreeBSD() ||
595 TC.getTriple().isOSNetBSD())
596 CmdArgs.push_back("-lexecinfo");
597}

599static void
600collectSanitizerRuntimes(const ToolChain &TC, const ArgList &Args,
601 SmallVectorImpl<StringRef> &SharedRuntimes,
602 SmallVectorImpl<StringRef> &StaticRuntimes,
603 SmallVectorImpl<StringRef> &NonWholeStaticRuntimes,
604 SmallVectorImpl<StringRef> &HelperStaticRuntimes,
605 SmallVectorImpl<StringRef> &RequiredSymbols) {
606 const SanitizerArgs &SanArgs = TC.getSanitizerArgs();
607 // Collect shared runtimes.
608 if (SanArgs.needsSharedRt()) {
609 if (SanArgs.needsAsanRt()) {
610 SharedRuntimes.push_back("asan");
611 if (!Args.hasArg(options::OPT_shared) && !TC.getTriple().isAndroid())
612 HelperStaticRuntimes.push_back("asan-preinit");
613 }
614 if (SanArgs.needsUbsanRt()) {
615 if (SanArgs.requiresMinimalRuntime())
616 SharedRuntimes.push_back("ubsan_minimal");
617 else
618 SharedRuntimes.push_back("ubsan_standalone");
619 }
620 if (SanArgs.needsScudoRt()) {
621 if (SanArgs.requiresMinimalRuntime())
622 SharedRuntimes.push_back("scudo_minimal");
623 else
624 SharedRuntimes.push_back("scudo");
625 }
626 if (SanArgs.needsHwasanRt())
627 SharedRuntimes.push_back("hwasan");
628 }

630 // The stats_client library is also statically linked into DSOs.

873/// Parses the various -fpic/-fPIC/-fpie/-fPIE arguments. Then,
874/// smooshes them together with platform defaults, to decide whether
875/// this compile should be using PIC mode or not. Returns a tuple of
876/// (RelocationModel, PICLevel, IsPIE).
877std::tuple<llvm::Reloc::Model, unsigned, bool>
878tools::ParsePICArgs(const ToolChain &ToolChain, const ArgList &Args) {
879 const llvm::Triple &EffectiveTriple = ToolChain.getEffectiveTriple();
880 const llvm::Triple &Triple = ToolChain.getTriple();
881
882 bool PIE = ToolChain.isPIEDefault();
883 bool PIC = PIE || ToolChain.isPICDefault();
884 // The Darwin/MachO default to use PIC does not apply when using -static.
885 if (Triple.isOSBinFormatMachO() && Args.hasArg(options::OPT_static))
886 PIE = PIC = false;
887 bool IsPICLevelTwo = PIC;
888
889 bool KernelOrKext =
890 Args.hasArg(options::OPT_mkernel, options::OPT_fapple_kext);
891
892 // Android-specific defaults for PIC/PIE
893 if (Triple.isAndroid()) {
894 switch (Triple.getArch()) {
895 case llvm::Triple::arm:
896 case llvm::Triple::armeb:
897 case llvm::Triple::thumb:
898 case llvm::Triple::thumbeb:
899 case llvm::Triple::aarch64:
900 case llvm::Triple::mips:
901 case llvm::Triple::mipsel:
902 case llvm::Triple::mips64:
903 case llvm::Triple::mips64el:
904 case llvm::Triple::cheri:
905 PIC = true; // "-fpic"
906 break;
907
908 case llvm::Triple::x86:
909 case llvm::Triple::x86_64:
910 PIC = true; // "-fPIC"
911 IsPICLevelTwo = true;
912 break;
913
914 default:
915 break;
916 }
917 }
918
919 // OpenBSD-specific defaults for PIE
920 if (Triple.isOSOpenBSD()) {
921 switch (ToolChain.getArch()) {
922 case llvm::Triple::arm:
923 case llvm::Triple::aarch64:
924 case llvm::Triple::mips64:
925 case llvm::Triple::mips64el:
926 case llvm::Triple::x86:
927 case llvm::Triple::x86_64:
928 IsPICLevelTwo = false; // "-fpie"
929 break;
930
931 case llvm::Triple::ppc:
932 case llvm::Triple::sparc:
933 case llvm::Triple::sparcel:
934 case llvm::Triple::sparcv9:
935 IsPICLevelTwo = true; // "-fPIE"
936 break;
937
938 default:
939 break;
940 }
941 }
942
943 // AMDGPU-specific defaults for PIC.
944 if (Triple.getArch() == llvm::Triple::amdgcn)
945 PIC = true;
946
947 // The last argument relating to either PIC or PIE wins, and no
948 // other argument is used. If the last argument is any flavor of the
949 // '-fno-...' arguments, both PIC and PIE are disabled. Any PIE
950 // option implicitly enables PIC at the same level.
951 Arg *LastPICArg = Args.getLastArg(options::OPT_fPIC, options::OPT_fno_PIC,
952 options::OPT_fpic, options::OPT_fno_pic,
953 options::OPT_fPIE, options::OPT_fno_PIE,
954 options::OPT_fpie, options::OPT_fno_pie);
955 if (Triple.isOSWindows() && LastPICArg &&
956 LastPICArg ==
957 Args.getLastArg(options::OPT_fPIC, options::OPT_fpic,
958 options::OPT_fPIE, options::OPT_fpie)) {
959 ToolChain.getDriver().Diag(diag::err_drv_unsupported_opt_for_target)
960 << LastPICArg->getSpelling() << Triple.str();
961 if (Triple.getArch() == llvm::Triple::x86_64)
962 return std::make_tuple(llvm::Reloc::PIC_, 2U, false);
963 return std::make_tuple(llvm::Reloc::Static, 0U, false);
964 }
965
966 // Check whether the tool chain trumps the PIC-ness decision. If the PIC-ness
967 // is forced, then neither PIC nor PIE flags will have no effect.
968 if (!ToolChain.isPICDefaultForced()) {
969 if (LastPICArg) {
970 Option O = LastPICArg->getOption();
971 if (O.matches(options::OPT_fPIC) || O.matches(options::OPT_fpic) ||
972 O.matches(options::OPT_fPIE) || O.matches(options::OPT_fpie)) {
973 PIE = O.matches(options::OPT_fPIE) || O.matches(options::OPT_fpie);
974 PIC =
975 PIE || O.matches(options::OPT_fPIC) || O.matches(options::OPT_fpic);
976 IsPICLevelTwo =
977 O.matches(options::OPT_fPIE) || O.matches(options::OPT_fPIC);
978 } else {
979 PIE = PIC = false;
980 if (EffectiveTriple.isPS4CPU()) {
981 Arg *ModelArg = Args.getLastArg(options::OPT_mcmodel_EQ);
982 StringRef Model = ModelArg ? ModelArg->getValue() : "";
983 if (Model != "kernel") {
984 PIC = true;
985 ToolChain.getDriver().Diag(diag::warn_drv_ps4_force_pic)
986 << LastPICArg->getSpelling();
987 }
988 }
989 }
990 }
991 }
992
993 // Introduce a Darwin and PS4-specific hack. If the default is PIC, but the
994 // PIC level would've been set to level 1, force it back to level 2 PIC
995 // instead.
996 if (PIC && (Triple.isOSDarwin() || EffectiveTriple.isPS4CPU()))
997 IsPICLevelTwo |= ToolChain.isPICDefault();
998
999 // This kernel flags are a trump-card: they will disable PIC/PIE
1000 // generation, independent of the argument order.
1001 if (KernelOrKext &&
1002 ((!EffectiveTriple.isiOS() || EffectiveTriple.isOSVersionLT(6)) &&
1003 !EffectiveTriple.isWatchOS()))
1004 PIC = PIE = false;
1005
1006 if (Arg *A = Args.getLastArg(options::OPT_mdynamic_no_pic)) {
1007 // This is a very special mode. It trumps the other modes, almost no one
1008 // uses it, and it isn't even valid on any OS but Darwin.
1009 if (!Triple.isOSDarwin())
1010 ToolChain.getDriver().Diag(diag::err_drv_unsupported_opt_for_target)
1011 << A->getSpelling() << Triple.str();
1012
1013 // FIXME: Warn when this flag trumps some other PIC or PIE flag.
1014
1015 // Only a forced PIC mode can cause the actual compile to have PIC defines
1016 // etc., no flags are sufficient. This behavior was selected to closely
1017 // match that of llvm-gcc and Apple GCC before that.
1018 PIC = ToolChain.isPICDefault() && ToolChain.isPICDefaultForced();
1019
1020 return std::make_tuple(llvm::Reloc::DynamicNoPIC, PIC ? 2U : 0U, false);
1021 }
1022
1023 bool EmbeddedPISupported;
1024 switch (Triple.getArch()) {
1025 case llvm::Triple::arm:
1026 case llvm::Triple::armeb:
1027 case llvm::Triple::thumb:
1028 case llvm::Triple::thumbeb:
1029 EmbeddedPISupported = true;
1030 break;
1031 default:
1032 EmbeddedPISupported = false;
1033 break;
1034 }
1035
1036 bool ROPI = false, RWPI = false;
1037 Arg* LastROPIArg = Args.getLastArg(options::OPT_fropi, options::OPT_fno_ropi);
1038 if (LastROPIArg && LastROPIArg->getOption().matches(options::OPT_fropi)) {
1039 if (!EmbeddedPISupported)
1040 ToolChain.getDriver().Diag(diag::err_drv_unsupported_opt_for_target)
1041 << LastROPIArg->getSpelling() << Triple.str();
1042 ROPI = true;
1043 }
1044 Arg *LastRWPIArg = Args.getLastArg(options::OPT_frwpi, options::OPT_fno_rwpi);
1045 if (LastRWPIArg && LastRWPIArg->getOption().matches(options::OPT_frwpi)) {
1046 if (!EmbeddedPISupported)
1047 ToolChain.getDriver().Diag(diag::err_drv_unsupported_opt_for_target)
1048 << LastRWPIArg->getSpelling() << Triple.str();
1049 RWPI = true;
1050 }
1051
1052 // ROPI and RWPI are not compatible with PIC or PIE.
1053 if ((ROPI || RWPI) && (PIC || PIE))
1054 ToolChain.getDriver().Diag(diag::err_drv_ropi_rwpi_incompatible_with_pic);
1055
1056 if (Triple.isMIPS()) {
1057 StringRef CPUName;
1058 StringRef ABIName;
1059 mips::getMipsCPUAndABI(Args, Triple, CPUName, ABIName);
1060 // When targeting the N64 ABI, PIC is the default, except in the case
1061 // when the -mno-abicalls option is used. In that case we exit
1062 // at next check regardless of PIC being set below.
1063 if (ABIName == "n64")
1064 PIC = true;
1065 // When targettng MIPS with -mno-abicalls, it's always static.
1066 if(Args.hasArg(options::OPT_mno_abicalls))
1067 return std::make_tuple(llvm::Reloc::Static, 0U, false);
1068 // Unlike other architectures, MIPS, even with -fPIC/-mxgot/multigot,
1069 // does not use PIC level 2 for historical reasons.
1070 IsPICLevelTwo = false;
1071 }
1072
1073 if (PIC)
1074 return std::make_tuple(llvm::Reloc::PIC_, IsPICLevelTwo ? 2U : 1U, PIE);
1075
1076 llvm::Reloc::Model RelocM = llvm::Reloc::Static;
1077 if (ROPI && RWPI)
1078 RelocM = llvm::Reloc::ROPI_RWPI;
1079 else if (ROPI)
1080 RelocM = llvm::Reloc::ROPI;
1081 else if (RWPI)
1082 RelocM = llvm::Reloc::RWPI;
1083
1084 return std::make_tuple(RelocM, 0U, false);
1085}

1087// `-falign-functions` indicates that the functions should be aligned to a
1088// 16-byte boundary.
1089//
1090// `-falign-functions=1` is the same as `-fno-align-functions`.
1091//
1092// The scalar `n` in `-falign-functions=n` must be an integral value between
1093// [0, 65536]. If the value is not a power-of-two, it will be rounded up to
1094// the nearest power-of-two.
1095//
1096// If we return `0`, the frontend will default to the backend's preferred
1097// alignment.
1098//
1099// NOTE: icc only allows values between [0, 4096]. icc uses `-falign-functions`
1100// to mean `-falign-functions=16`. GCC defaults to the backend's preferred
1101// alignment. For unaligned functions, we default to the backend's preferred
1102// alignment.
1103unsigned tools::ParseFunctionAlignment(const ToolChain &TC,
1104 const ArgList &Args) {
1105 const Arg *A = Args.getLastArg(options::OPT_falign_functions,
1106 options::OPT_falign_functions_EQ,
1107 options::OPT_fno_align_functions);
1108 if (!A || A->getOption().matches(options::OPT_fno_align_functions))
1109 return 0;
1110
1111 if (A->getOption().matches(options::OPT_falign_functions))
1112 return 0;
1113
1114 unsigned Value = 0;
1115 if (StringRef(A->getValue()).getAsInteger(10, Value) || Value > 65536)
1116 TC.getDriver().Diag(diag::err_drv_invalid_int_value)
1117 << A->getAsString(Args) << A->getValue();
1118 return Value ? llvm::Log2_32_Ceil(std::min(Value, 65536u)) : Value;
1119}

1153// Gcc adds libgcc arguments in various ways:
1154//
1155// gcc <none>: -lgcc --as-needed -lgcc_s --no-as-needed
1156// g++ <none>: -lgcc_s -lgcc
1157// gcc shared: -lgcc_s -lgcc
1158// g++ shared: -lgcc_s -lgcc
1159// gcc static: -lgcc -lgcc_eh
1160// g++ static: -lgcc -lgcc_eh
1161// gcc static-pie: -lgcc -lgcc_eh
1162// g++ static-pie: -lgcc -lgcc_eh
1163//
1164// Also, certain targets need additional adjustments.
1165
1166static void AddUnwindLibrary(const ToolChain &TC, const Driver &D,
1167 ArgStringList &CmdArgs, const ArgList &Args) {
1168 ToolChain::UnwindLibType UNW = TC.GetUnwindLibType(Args);
1169 // Targets that don't use unwind libraries.
1170 if (TC.getTriple().isAndroid() || TC.getTriple().isOSIAMCU() ||
1171 TC.getTriple().isOSBinFormatWasm() ||
1172 UNW == ToolChain::UNW_None)
1173 return;
1174
1175 LibGccType LGT = getLibGccType(Args);
1176 bool AsNeeded = D.CCCIsCC() && LGT == LibGccType::UnspecifiedLibGcc &&
1177 !TC.getTriple().isAndroid() && !TC.getTriple().isOSCygMing();
1178 if (AsNeeded)
1179 CmdArgs.push_back("--as-needed");
1180
1181 switch (UNW) {
1182 case ToolChain::UNW_None:
1183 return;
1184 case ToolChain::UNW_Libgcc: {
1185 LibGccType LGT = getLibGccType(Args);
1186 if (LGT == LibGccType::UnspecifiedLibGcc || LGT == LibGccType::SharedLibGcc)
1187 CmdArgs.push_back("-lgcc_s");
1188 else if (LGT == LibGccType::StaticLibGcc)
1189 CmdArgs.push_back("-lgcc_eh");
1190 break;
1191 }
1192 case ToolChain::UNW_CompilerRT:
1193 CmdArgs.push_back("-lunwind");
1194 break;
1195 }
1196
1197 if (AsNeeded)
1198 CmdArgs.push_back("--no-as-needed");
1199}

1201static void AddLibgcc(const ToolChain &TC, const Driver &D,
1202 ArgStringList &CmdArgs, const ArgList &Args) {
1203 bool isAndroid = TC.getTriple().isAndroid();
1204
1205 LibGccType LGT = getLibGccType(Args);
1206 bool LibGccFirst = (D.CCCIsCC() && LGT == LibGccType::UnspecifiedLibGcc) ||
1207 LGT == LibGccType::StaticLibGcc;
1208 if (LibGccFirst)
1209 CmdArgs.push_back("-lgcc");
1210
1211 AddUnwindLibrary(TC, D, CmdArgs, Args);
1212
1213 if (!LibGccFirst)
1214 CmdArgs.push_back("-lgcc");
1215
1216 // According to Android ABI, we have to link with libdl if we are
1217 // linking with non-static libgcc.
1218 //
1219 // NOTE: This fixes a link error on Android MIPS as well. The non-static
1220 // libgcc for MIPS relies on _Unwind_Find_FDE and dl_iterate_phdr from libdl.
1221 if (isAndroid && getLibGccType(Args) != LibGccType::StaticLibGcc)
1222 CmdArgs.push_back("-ldl");
1223}

1231 case ToolChain::RLT_CompilerRT:
1232 CmdArgs.push_back(TC.getCompilerRTArgString(Args, "builtins"));

1235 case ToolChain::RLT_Libgcc:

1244 } else
1245 AddLibgcc(TC, D, CmdArgs, Args);
1246 break;
1247 }

1250/// Add OpenMP linker script arguments at the end of the argument list so that
1251/// the fat binary is built by embedding each of the device images into the
1252/// host. The linker script also defines a few symbols required by the code
1253/// generation so that the images can be easily retrieved at runtime by the
1254/// offloading library. This should be used only in tool chains that support
1255/// linker scripts.
1256void tools::AddOpenMPLinkerScript(const ToolChain &TC, Compilation &C,
1257 const InputInfo &Output,
1258 const InputInfoList &Inputs,
1259 const ArgList &Args, ArgStringList &CmdArgs,
1260 const JobAction &JA) {
1261
1262 // If this is not an OpenMP host toolchain, we don't need to do anything.
1263 if (!JA.isHostOffloading(Action::OFK_OpenMP))
1264 return;
1265
1266 // Create temporary linker script. Keep it if save-temps is enabled.
1267 const char *LKS;
1268 SmallString<256> Name = llvm::sys::path::filename(Output.getFilename());
1269 if (C.getDriver().isSaveTempsEnabled()) {
1270 llvm::sys::path::replace_extension(Name, "lk");
1271 LKS = C.getArgs().MakeArgString(Name.c_str());
1272 } else {
1273 llvm::sys::path::replace_extension(Name, "");
1274 Name = C.getDriver().GetTemporaryPath(Name, "lk");
1275 LKS = C.addTempFile(C.getArgs().MakeArgString(Name.c_str()));
1276 }
1277
1278 // Add linker script option to the command.
1279 CmdArgs.push_back("-T");
1280 CmdArgs.push_back(LKS);
1281
1282 // Create a buffer to write the contents of the linker script.
1283 std::string LksBuffer;
1284 llvm::raw_string_ostream LksStream(LksBuffer);
1285
1286 // Get the OpenMP offload tool chains so that we can extract the triple
1287 // associated with each device input.
1288 auto OpenMPToolChains = C.getOffloadToolChains<Action::OFK_OpenMP>();
1289 assert(OpenMPToolChains.first != OpenMPToolChains.second &&
1290 "No OpenMP toolchains??");
1291
1292 // Track the input file name and device triple in order to build the script,
1293 // inserting binaries in the designated sections.
1294 SmallVector<std::pair<std::string, const char *>, 8> InputBinaryInfo;
1295
1296 // Add commands to embed target binaries. We ensure that each section and
1297 // image is 16-byte aligned. This is not mandatory, but increases the
1298 // likelihood of data to be aligned with a cache block in several main host
1299 // machines.
1300 LksStream << "/*\n";
1301 LksStream << " OpenMP Offload Linker Script\n";
1302 LksStream << " *** Automatically generated by Clang ***\n";
1303 LksStream << "*/\n";
1304 LksStream << "TARGET(binary)\n";
1305 auto DTC = OpenMPToolChains.first;
1306 for (auto &II : Inputs) {
1307 const Action *A = II.getAction();
1308 // Is this a device linking action?
1309 if (A && isa<LinkJobAction>(A) &&
1310 A->isDeviceOffloading(Action::OFK_OpenMP)) {
1311 assert(DTC != OpenMPToolChains.second &&
1312 "More device inputs than device toolchains??");
1313 InputBinaryInfo.push_back(std::make_pair(
1314 DTC->second->getTriple().normalize(), II.getFilename()));
1315 ++DTC;
1316 LksStream << "INPUT(" << II.getFilename() << ")\n";
1317 }
1318 }
1319
1320 assert(DTC == OpenMPToolChains.second &&
1321 "Less device inputs than device toolchains??");
1322
1323 LksStream << "SECTIONS\n";
1324 LksStream << "{\n";
1325
1326 // Put each target binary into a separate section.
1327 for (const auto &BI : InputBinaryInfo) {
1328 LksStream << " .omp_offloading." << BI.first << " :\n";
1329 LksStream << " ALIGN(0x10)\n";
1330 LksStream << " {\n";
1331 LksStream << " PROVIDE_HIDDEN(.omp_offloading.img_start." << BI.first
1332 << " = .);\n";
1333 LksStream << " " << BI.second << "\n";
1334 LksStream << " PROVIDE_HIDDEN(.omp_offloading.img_end." << BI.first
1335 << " = .);\n";
1336 LksStream << " }\n";
1337 }
1338
1339 // Add commands to define host entries begin and end. We use 1-byte subalign
1340 // so that the linker does not add any padding and the elements in this
1341 // section form an array.
1342 LksStream << " .omp_offloading.entries :\n";
1343 LksStream << " ALIGN(0x10)\n";
1344 LksStream << " SUBALIGN(0x01)\n";
1345 LksStream << " {\n";
1346 LksStream << " PROVIDE_HIDDEN(.omp_offloading.entries_begin = .);\n";
1347 LksStream << " *(.omp_offloading.entries)\n";
1348 LksStream << " PROVIDE_HIDDEN(.omp_offloading.entries_end = .);\n";
1349 LksStream << " }\n";
1350 LksStream << "}\n";
1351 LksStream << "INSERT BEFORE .data\n";
1352 LksStream.flush();
1353
1354 // Dump the contents of the linker script if the user requested that. We
1355 // support this option to enable testing of behavior with -###.
1356 if (C.getArgs().hasArg(options::OPT_fopenmp_dump_offload_linker_script))
1357 llvm::errs() << LksBuffer;
1358
1359 // If this is a dry run, do not create the linker script file.
1360 if (C.getArgs().hasArg(options::OPT__HASH_HASH_HASH))
1361 return;
1362
1363 // Open script file and write the contents.
1364 std::error_code EC;
1365 llvm::raw_fd_ostream Lksf(LKS, EC, llvm::sys::fs::F_None);
1366
1367 if (EC) {
1368 C.getDriver().Diag(clang::diag::err_unable_to_make_temp) << EC.message();
1369 return;
1370 }
1371
1372 Lksf << LksBuffer;
1373}

1375/// Add HIP linker script arguments at the end of the argument list so that
1376/// the fat binary is built by embedding the device images into the host. The
1377/// linker script also defines a symbol required by the code generation so that
1378/// the image can be retrieved at runtime. This should be used only in tool
1379/// chains that support linker scripts.
1380void tools::AddHIPLinkerScript(const ToolChain &TC, Compilation &C,
1381 const InputInfo &Output,
1382 const InputInfoList &Inputs, const ArgList &Args,
1383 ArgStringList &CmdArgs, const JobAction &JA,
1384 const Tool &T) {

1387 if (!JA.isHostOffloading(Action::OFK_HIP))
1388 return;

1390 InputInfoList DeviceInputs;
1391 for (const auto &II : Inputs) {
1392 const Action *A = II.getAction();
1393 // Is this a device linking action?
1394 if (A && isa<LinkJobAction>(A) && A->isDeviceOffloading(Action::OFK_HIP)) {
1395 DeviceInputs.push_back(II);
1396 }
1397 }
1398
1399 if (DeviceInputs.empty())
1400 return;
1401
1402 // Create temporary linker script. Keep it if save-temps is enabled.
1403 const char *LKS;
1404 SmallString<256> Name = llvm::sys::path::filename(Output.getFilename());
1405 if (C.getDriver().isSaveTempsEnabled()) {
1406 llvm::sys::path::replace_extension(Name, "lk");
1407 LKS = C.getArgs().MakeArgString(Name.c_str());
1408 } else {
1409 llvm::sys::path::replace_extension(Name, "");
1410 Name = C.getDriver().GetTemporaryPath(Name, "lk");
1411 LKS = C.addTempFile(C.getArgs().MakeArgString(Name.c_str()));
1412 }
1413
1414 // Add linker script option to the command.
1415 CmdArgs.push_back("-T");
1416 CmdArgs.push_back(LKS);
1417

1419 std::string LksBuffer;
1420 llvm::raw_string_ostream LksStream(LksBuffer);
1421
1422 // Get the HIP offload tool chain.
1423 auto *HIPTC = static_cast<const toolchains::CudaToolChain *>(
1424 C.getSingleOffloadToolChain<Action::OFK_HIP>());
1425 assert(HIPTC->getTriple().getArch() == llvm::Triple::amdgcn &&
1426 "Wrong platform");
1427 (void)HIPTC;
1428
1429 // The output file name needs to persist through the compilation, therefore
1430 // it needs to be created through MakeArgString.
1431 std::string BundleFileName = C.getDriver().GetTemporaryPath("BUNDLE", "hipfb");
1432 const char *BundleFile =
1433 C.addTempFile(C.getArgs().MakeArgString(BundleFileName.c_str()));
1434 AMDGCN::constructHIPFatbinCommand(C, JA, BundleFile, DeviceInputs, Args, T);
1435
1436 // Add commands to embed target binaries. We ensure that each section and
1437 // image is 16-byte aligned. This is not mandatory, but increases the
1438 // likelihood of data to be aligned with a cache block in several main host
1439 // machines.
1440 LksStream << "/*\n";
1441 LksStream << " HIP Offload Linker Script\n";
1442 LksStream << " *** Automatically generated by Clang ***\n";
1443 LksStream << "*/\n";
1444 LksStream << "TARGET(binary)\n";
1445 LksStream << "INPUT(" << BundleFileName << ")\n";
1446 LksStream << "SECTIONS\n";
1447 LksStream << "{\n";
1448 LksStream << " .hip_fatbin :\n";
1449 LksStream << " ALIGN(0x10)\n";
1450 LksStream << " {\n";
1451 LksStream << " PROVIDE_HIDDEN(__hip_fatbin = .);\n";
1452 LksStream << " " << BundleFileName << "\n";
1453 LksStream << " }\n";
1454 LksStream << " /DISCARD/ :\n";
1455 LksStream << " {\n";
1456 LksStream << " * ( __CLANG_OFFLOAD_BUNDLE__* )\n";
1457 LksStream << " }\n";
1458 LksStream << "}\n";
1459 LksStream << "INSERT BEFORE .data\n";
1460 LksStream.flush();
1461
1462 // Dump the contents of the linker script if the user requested that. We
1463 // support this option to enable testing of behavior with -###.
1464 if (C.getArgs().hasArg(options::OPT_fhip_dump_offload_linker_script))
1465 llvm::errs() << LksBuffer;
1466
1467 // If this is a dry run, do not create the linker script file.
1468 if (C.getArgs().hasArg(options::OPT__HASH_HASH_HASH))
1469 return;
1470
1471 // Open script file and write the contents.
1472 std::error_code EC;
1473 llvm::raw_fd_ostream Lksf(LKS, EC, llvm::sys::fs::F_None);
1474
1475 if (EC) {
1476 C.getDriver().Diag(clang::diag::err_unable_to_make_temp) << EC.message();
1477 return;
1478 }
1479
1480 Lksf << LksBuffer;
1481}

1484 const InputInfo &Output,
1485 const InputInfo &Input,
1486 const Driver &D) {
1487 const Arg *A = Args.getLastArg(options::OPT_save_stats_EQ);

1491 StringRef SaveStats = A->getValue();
1492 SmallString<128> StatsFile;
1493 if (SaveStats == "obj" && Output.isFilename()) {

1503 llvm::sys::path::replace_extension(StatsFile, "stats");
