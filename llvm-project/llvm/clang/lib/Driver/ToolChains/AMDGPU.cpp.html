22void amdgpu::Linker::ConstructJob(Compilation &C, const JobAction &JA,
23 const InputInfo &Output,
24 const InputInfoList &Inputs,
25 const ArgList &Args,
26 const char *LinkingOutput) const {
27
28 std::string Linker = getToolChain().GetProgramPath(getShortName());
29 ArgStringList CmdArgs;
30 AddLinkerInputs(getToolChain(), Inputs, Args, CmdArgs, JA);
31 CmdArgs.push_back("-shared");
32 CmdArgs.push_back("-o");
33 CmdArgs.push_back(Output.getFilename());
34 C.addCommand(llvm::make_unique<Command>(JA, *this, Args.MakeArgString(Linker),
35 CmdArgs, Inputs));
36}

38void amdgpu::getAMDGPUTargetFeatures(const Driver &D,
39 const llvm::opt::ArgList &Args,
40 std::vector<StringRef> &Features) {
41 if (const Arg *dAbi = Args.getLastArg(options::OPT_mamdgpu_debugger_abi))
42 D.Diag(diag::err_drv_clang_unsupported) << dAbi->getAsString(Args);
43
44 handleTargetFeaturesGroup(
45 Args, Features, options::OPT_m_amdgpu_Features_Group);
46}

59DerivedArgList *
60AMDGPUToolChain::TranslateArgs(const DerivedArgList &Args, StringRef BoundArch,
61 Action::OffloadKind DeviceOffloadKind) const {
62
63 DerivedArgList *DAL =
64 Generic_ELF::TranslateArgs(Args, BoundArch, DeviceOffloadKind);
65
66 // Do nothing if not OpenCL (-x cl)
67 if (!Args.getLastArgValue(options::OPT_x).equals("cl"))
68 return DAL;
69
70 if (!DAL)
71 DAL = new DerivedArgList(Args.getBaseArgs());
72 for (auto *A : Args)
73 DAL->append(A);
74
75 const OptTable &Opts = getDriver().getOpts();
76
77 // Phase 1 (.cl -> .bc)
78 if (Args.hasArg(options::OPT_c) && Args.hasArg(options::OPT_emit_llvm)) {
79 DAL->AddFlagArg(nullptr, Opts.getOption(getTriple().isArch64Bit()
80 ? options::OPT_m64
81 : options::OPT_m32));
82
83 // Have to check OPT_O4, OPT_O0 & OPT_Ofast separately
84 // as they defined that way in Options.td
85 if (!Args.hasArg(options::OPT_O, options::OPT_O0, options::OPT_O4,
86 options::OPT_Ofast))
87 DAL->AddJoinedArg(nullptr, Opts.getOption(options::OPT_O),
88 getOptionDefault(options::OPT_O));
89 }
90
91 return DAL;
92}

94void AMDGPUToolChain::addClangTargetOptions(
95 const llvm::opt::ArgList &DriverArgs,
96 llvm::opt::ArgStringList &CC1Args,
97 Action::OffloadKind DeviceOffloadingKind) const {
98 // Default to "hidden" visibility, as object level linking will not be
99 // supported for the foreseeable future.
100 if (!DriverArgs.hasArg(options::OPT_fvisibility_EQ,
101 options::OPT_fvisibility_ms_compat)) {
102 CC1Args.push_back("-fvisibility");
103 CC1Args.push_back("hidden");
104 CC1Args.push_back("-fapply-global-visibility-to-externs");
105 }
