28void rtems::Linker::ConstructJob(Compilation &C, const JobAction &JA,
29 const InputInfo &Output,
30 const InputInfoList &Inputs,
31 const ArgList &Args,
32 const char *LinkingOutput) const {
33 const toolchains::RTEMS &ToolChain =
34 static_cast<const toolchains::RTEMS &>(getToolChain());
35 const Driver &D = ToolChain.getDriver();
36
37 ArgStringList CmdArgs;
38
39 // Silence warning for "clang -g foo.o -o foo"
40 Args.ClaimAllArgs(options::OPT_g_Group);
41 // and "clang -emit-llvm foo.o -o foo"
42 Args.ClaimAllArgs(options::OPT_emit_llvm);
43 // and for "clang -w foo.o -o foo". Other warning options are already
44 // handled somewhere else.
45 Args.ClaimAllArgs(options::OPT_w);
46 // TODO: warn if rtlib/c++lib is missing: err_drv_missing_argument
47
48 const char *Exec = Args.MakeArgString(ToolChain.GetLinkerPath());
49 if (llvm::sys::path::filename(Exec).equals_lower("ld.lld") ||
50 llvm::sys::path::stem(Exec).equals_lower("ld.lld")) {
51 CmdArgs.push_back("-z");
52 CmdArgs.push_back("rodynamic");
53 }
54
55 if (!D.SysRoot.empty())
56 CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
57
58 if (Args.hasArg(options::OPT_rdynamic))
59 CmdArgs.push_back("-export-dynamic");
60
61 if (Args.hasArg(options::OPT_s))
62 CmdArgs.push_back("-s");
63
64 if (Args.hasArg(options::OPT_r)) {
65 CmdArgs.push_back("-r");
66 } else {
67 CmdArgs.push_back("--build-id");
68 // LLD rejects gnu hash style for MIPS
69 //CmdArgs.push_back("--hash-style=gnu");
70 }
71
72 CmdArgs.push_back("--eh-frame-hdr");
73
74 // No support for -shared or -pie linking
75 if (Arg* A = Args.getLastArg(options::OPT_shared, options::OPT_pie)) {
76 D.Diag(diag::err_drv_unsupported_opt) << A->getAsString(Args);
77 }
78 CmdArgs.push_back("-Bstatic");
79 CmdArgs.push_back("-static"); // only find .a files
80
81 CmdArgs.push_back("-o");
82 CmdArgs.push_back(Output.getFilename());
83
84 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles)) {
85 if (!Args.hasArg(options::OPT_q_rtems)) {
86
87 // RTEMS must provide a --sysroot path to its installed C library
88 const char* arg = nullptr;
89 if (D.SysRoot.empty()) {
90 ToolChain.getDriver().Diag(diag::err_drv_argument_not_allowed_with)
91 << "missing --sysroot= option" << "rtems driver";
92 arg = Args.MakeArgString(ToolChain.GetFilePath("crt0.o"));
93 } else {
94 arg = Args.MakeArgString(D.SysRoot + "/lib/crt0.o");
95 }
96 CmdArgs.push_back(arg);
97 }
98 // Otherwise, RTEMS build system should provide its own start file
99 }
100
101 Args.AddAllArgs(CmdArgs, options::OPT_L);
102 Args.AddAllArgs(CmdArgs, options::OPT_u);
103
104 ToolChain.AddFilePathLibArgs(Args, CmdArgs);
105
106 addSanitizerRuntimes(ToolChain, Args, CmdArgs);
107
108 AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
109
110
111 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs)) {
112 // Add the rtems libns:
113 if (Args.hasArg(options::OPT_q_rtems)) {
114 CmdArgs.push_back("-Tlinkcmds");
115 CmdArgs.push_back("-lrtemsbsp");
116 CmdArgs.push_back("-lrtemscpu");
117 }
118 if (D.CCCIsCXX()) {
119 if (ToolChain.ShouldLinkCXXStdlib(Args))
120 ToolChain.AddCXXStdlibLibArgs(Args, CmdArgs);
121 CmdArgs.push_back("-lm");
122 }
123
124 AddRunTimeLibs(ToolChain, D, CmdArgs, Args);
125 // TODO: is there pthread lib?
126 if (Args.hasArg(options::OPT_pthread) ||
127 Args.hasArg(options::OPT_pthreads))
128 CmdArgs.push_back("-lpthread");
129 // TODO: is this needed?
130 if (Args.hasArg(options::OPT_fsplit_stack))
131 CmdArgs.push_back("--wrap=pthread_create");
132
133 CmdArgs.push_back("-lc");
134 }
135 CmdArgs.push_back("-e");
136 CmdArgs.push_back("_start");
137
138 C.addCommand(llvm::make_unique<Command>(JA, *this, Exec, CmdArgs, Inputs));
139}

141/// RTEMS - RTEMS tool chain which can call as(1) and ld(1) directly.
142
143RTEMS::RTEMS(const Driver &D, const llvm::Triple &Triple,
144 const ArgList &Args)
145 : ToolChain(D, Triple, Args) {
146 getProgramPaths().push_back(getDriver().getInstalledDir());
147 if (getDriver().getInstalledDir() != D.Dir)
148 getProgramPaths().push_back(D.Dir);
149#if 0
150 SmallString<128> P(getTargetDir(D, getTriple()));
151 llvm::sys::path::append(P, "lib");
152 getFilePaths().push_back(P.str());
153#endif
154
155 if (!D.SysRoot.empty()) {
156 SmallString<128> P(D.SysRoot);
157 llvm::sys::path::append(P, "lib");
158 getFilePaths().push_back(P.str());
159 }
160}

166ToolChain::RuntimeLibType RTEMS::GetRuntimeLibType(
167 const ArgList &Args) const {
168 if (Arg *A = Args.getLastArg(clang::driver::options::OPT_rtlib_EQ)) {
169 StringRef Value = A->getValue();
170 if (Value != "compiler-rt")
171 getDriver().Diag(clang::diag::err_drv_invalid_rtlib_name)
172 << A->getAsString(Args);
173 }
174 return ToolChain::RLT_CompilerRT;
175}
176
177std::string RTEMS::getCompilerRTPath() const {
178 // Pick compiler rt from the sysroot first:
179 const Driver &D = getDriver();
180 if (!D.SysRoot.empty()) {
181 llvm::SmallString<128> Path(D.SysRoot);
182 llvm::sys::path::append(Path, "lib");
183 return Path.str();
184 }
185 return ToolChain::getCompilerRTPath();
186}

188ToolChain::CXXStdlibType
189RTEMS::GetCXXStdlibType(const ArgList &Args) const {
190 if (Arg *A = Args.getLastArg(options::OPT_stdlib_EQ)) {
191 StringRef Value = A->getValue();
192 if (Value != "libc++")
193 getDriver().Diag(diag::err_drv_invalid_stdlib_name)
194 << A->getAsString(Args);
195 }
196 return ToolChain::CST_Libcxx;
197}

207void RTEMS::AddClangSystemIncludeArgs(const ArgList &DriverArgs,
208 ArgStringList &CC1Args) const {
209
210 const Driver &D = getDriver();
211
212 if (DriverArgs.hasArg(options::OPT_nostdinc))
213 return;
214
215 if (!DriverArgs.hasArg(options::OPT_nobuiltininc)) {
216 SmallString<128> P(D.ResourceDir);
217 llvm::sys::path::append(P, "include");
218 addSystemInclude(DriverArgs, CC1Args, P);
219 }
220
221 if (DriverArgs.hasArg(options::OPT_nostdlibinc))
222 return;
223
224 if (!D.SysRoot.empty()) {
225 SmallString<128> P(D.SysRoot);
226 llvm::sys::path::append(P, "include");
227 addExternCSystemInclude(DriverArgs, CC1Args, P.str());
228 }
229}

231void RTEMS::AddClangCXXStdlibIncludeArgs(const ArgList &DriverArgs,
232 ArgStringList &CC1Args) const {
233 if (DriverArgs.hasArg(options::OPT_nostdlibinc) ||
234 DriverArgs.hasArg(options::OPT_nostdincxx))
235 return;
236 const Driver &D = getDriver();
237 switch (GetCXXStdlibType(DriverArgs)) {
238 case ToolChain::CST_Libcxx: {
239 SmallString<128> P(!D.SysRoot.empty() ? D.SysRoot : "/");
240 llvm::sys::path::append(P, "include", "c++", "v1");
241 addSystemInclude(DriverArgs, CC1Args, P.str());
242 break;
243 }
244
245 default:
246 llvm_unreachable("invalid stdlib name");
247 }
248}
249
250void RTEMS::AddCXXStdlibLibArgs(const ArgList &Args,
251 ArgStringList &CmdArgs) const {
252 switch (GetCXXStdlibType(Args)) {
253 case ToolChain::CST_Libcxx:
254 // TODO: what is correct here
255 CmdArgs.push_back("-lc++");
256 // CmdArgs.push_back("-lc++abi");
257 CmdArgs.push_back("-lunwind");
258 break;
259
260 case ToolChain::CST_Libstdcxx:
261 llvm_unreachable("invalid stdlib name");
262 }
