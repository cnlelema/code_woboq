17namespace clang {
18namespace driver {
19
20namespace toolchains {
21
22class LLVM_LIBRARY_VISIBILITY BareMetal : public ToolChain {
23public:
24 BareMetal(const Driver &D, const llvm::Triple &Triple,
25 const llvm::opt::ArgList &Args);
26 ~BareMetal() override;
27
28 static bool handlesTarget(const llvm::Triple &Triple);
29protected:
30 Tool *buildLinker() const override;
31
32public:
33 bool useIntegratedAs() const override { return true; }
34 bool isCrossCompiling() const override { return true; }
35 bool isPICDefault() const override { return false; }
36 bool isPIEDefault() const override { return false; }
37 bool isPICDefaultForced() const override { return false; }
38 bool SupportsProfiling() const override { return false; }
39
40 RuntimeLibType GetDefaultRuntimeLibType() const override {
41 return ToolChain::RLT_CompilerRT;
42 }
43 CXXStdlibType GetDefaultCXXStdlibType() const override {
44 return ToolChain::CST_Libcxx;
45 }
46
47 const char *getDefaultLinker() const override { return "ld.lld"; }
48
49 std::string getRuntimesDir() const;
50 void AddClangSystemIncludeArgs(const llvm::opt::ArgList &DriverArgs,
51 llvm::opt::ArgStringList &CC1Args) const override;
52 void addClangTargetOptions(const llvm::opt::ArgList &DriverArgs,
53 llvm::opt::ArgStringList &CC1Args,
54 Action::OffloadKind DeviceOffloadKind) const override;
55 void AddClangCXXStdlibIncludeArgs(
56 const llvm::opt::ArgList &DriverArgs,
57 llvm::opt::ArgStringList &CC1Args) const override;
58 void AddCXXStdlibLibArgs(const llvm::opt::ArgList &Args,
59 llvm::opt::ArgStringList &CmdArgs) const override;
60 void AddLinkRuntimeLib(const llvm::opt::ArgList &Args,
61 llvm::opt::ArgStringList &CmdArgs) const;
62
63private:
64 enum class BaremetalTarget { Unknown, ARM, MIPS };
65 BaremetalTarget Target = BaremetalTarget::Unknown;
66 bool IsCheriPurecap = false;
67};
68
69} // namespace toolchains
70
71namespace tools {
72namespace baremetal {
73
74class LLVM_LIBRARY_VISIBILITY Linker : public Tool {
75public:
76 Linker(const ToolChain &TC) : Tool("baremetal::Linker", "ld.lld", TC) {}
77 bool isLinkJob() const override { return true; }
78 bool hasIntegratedCPP() const override { return false; }
79 void ConstructJob(Compilation &C, const JobAction &JA,
80 const InputInfo &Output, const InputInfoList &Inputs,
81 const llvm::opt::ArgList &TCArgs,
82 const char *LinkingOutput) const override;
83};
