31/// Is the triple {arm,thumb}-none-none-{eabi,eabihf} ?
32static bool isARMBareMetal(const llvm::Triple &Triple) {
33 if (Triple.getArch() != llvm::Triple::arm &&
34 Triple.getArch() != llvm::Triple::thumb)
35 return false;
36
37 if (Triple.getVendor() != llvm::Triple::UnknownVendor)
38 return false;
39
40 if (Triple.getOS() != llvm::Triple::UnknownOS)
41 return false;
42
43 if (Triple.getEnvironment() != llvm::Triple::EABI &&
44 Triple.getEnvironment() != llvm::Triple::EABIHF)
45 return false;
46
47 return true;
48}
49
50/// Allow mips*-none-elf
51static bool isMIPSBareMetal(const llvm::Triple& Triple) {
52 switch(Triple.getArch()) {
53 case llvm::Triple::mips:
54 case llvm::Triple::mips64el:
55 case llvm::Triple::mips64:
56 case llvm::Triple::mipsel:
57 case llvm::Triple::cheri:
58 if (Triple.getVendor() != llvm::Triple::UnknownVendor)
59 return false;
60 if (Triple.getOS() != llvm::Triple::UnknownOS)
61 return false;
62 return Triple.isOSBinFormatELF();
63 default:
64 return false;
65 }
66}

68BareMetal::BareMetal(const Driver &D, const llvm::Triple &Triple,
69 const ArgList &Args)
70 : ToolChain(D, Triple, Args) {
71 getProgramPaths().push_back(getDriver().getInstalledDir());
72 if (getDriver().getInstalledDir() != getDriver().Dir)
73 getProgramPaths().push_back(getDriver().Dir);
74 if (isARMBareMetal(Triple))
75 Target = BaremetalTarget::ARM;
76 else if (isMIPSBareMetal(Triple)) {
77 Target = BaremetalTarget::MIPS;
78 if (tools::mips::hasMipsAbiArg(Args, "purecap"))
79 IsCheriPurecap = true;
80 }
81}

85bool BareMetal::handlesTarget(const llvm::Triple &Triple) {
86 return isARMBareMetal(Triple) || isMIPSBareMetal(Triple);

93std::string BareMetal::getRuntimesDir() const {
94 if (Target == BaremetalTarget::MIPS) {
95 SmallString<128> Dir(getDriver().SysRoot);
96 if (Dir.empty())
97 Dir = getDriver().ResourceDir;
98 if (IsCheriPurecap)
99 llvm::sys::path::append(Dir, "libcheri");
100 else
101 llvm::sys::path::append(Dir, "lib");
102 return Dir.str();
103 }
104 assert(Target == BaremetalTarget::ARM);
105 SmallString<128> Dir(getDriver().ResourceDir);
106 llvm::sys::path::append(Dir, "lib", "baremetal");
107 return Dir.str();
108}

110void BareMetal::AddClangSystemIncludeArgs(const ArgList &DriverArgs,
111 ArgStringList &CC1Args) const {
112 if (DriverArgs.hasArg(options::OPT_nostdinc))
113 return;
114
115 if (!DriverArgs.hasArg(options::OPT_nobuiltininc)) {
116 SmallString<128> Dir(getDriver().ResourceDir);
117 llvm::sys::path::append(Dir, "include");
118 addSystemInclude(DriverArgs, CC1Args, Dir.str());
119 }
120
121 if (!DriverArgs.hasArg(options::OPT_nostdlibinc)) {
122 SmallString<128> Dir(getDriver().SysRoot);
123 llvm::sys::path::append(Dir, "include");
124 addSystemInclude(DriverArgs, CC1Args, Dir.str());
125 }
126}

134void BareMetal::AddClangCXXStdlibIncludeArgs(
135 const ArgList &DriverArgs, ArgStringList &CC1Args) const {
136 if (DriverArgs.hasArg(options::OPT_nostdinc) ||
137 DriverArgs.hasArg(options::OPT_nostdlibinc) ||
138 DriverArgs.hasArg(options::OPT_nostdincxx))
139 return;
140
141 StringRef SysRoot = getDriver().SysRoot;
142 if (SysRoot.empty())
143 return;
144
145 switch (GetCXXStdlibType(DriverArgs)) {
146 case ToolChain::CST_Libcxx: {
147 SmallString<128> Dir(SysRoot);
148 llvm::sys::path::append(Dir, "include", "c++", "v1");
149 addSystemInclude(DriverArgs, CC1Args, Dir.str());
150 break;
151 }
152 case ToolChain::CST_Libstdcxx: {
153 SmallString<128> Dir(SysRoot);
154 llvm::sys::path::append(Dir, "include", "c++");
155 std::error_code EC;
156 Generic_GCC::GCCVersion Version = {"", -1, -1, -1, "", "", ""};
157 // Walk the subdirs, and find the one with the newest gcc version:
158 for (llvm::vfs::directory_iterator
159 LI = getDriver().getVFS().dir_begin(Dir.str(), EC),
160 LE;
161 !EC && LI != LE; LI = LI.increment(EC)) {
162 StringRef VersionText = llvm::sys::path::filename(LI->path());
163 auto CandidateVersion = Generic_GCC::GCCVersion::Parse(VersionText);
164 if (CandidateVersion.Major == -1)
165 continue;
166 if (CandidateVersion <= Version)
167 continue;
168 Version = CandidateVersion;
169 }
170 if (Version.Major == -1)
171 return;
172 llvm::sys::path::append(Dir, Version.Text);
173 addSystemInclude(DriverArgs, CC1Args, Dir.str());
174 break;
175 }
176 }
177}

179void BareMetal::AddCXXStdlibLibArgs(const ArgList &Args,
180 ArgStringList &CmdArgs) const {
181 switch (GetCXXStdlibType(Args)) {
182 case ToolChain::CST_Libcxx:
183 CmdArgs.push_back("-lc++");
184 CmdArgs.push_back(Target == BaremetalTarget::ARM ? "-lc++abi" : "-lcxxrt");
185 break;
186 case ToolChain::CST_Libstdcxx:
187 CmdArgs.push_back("-lstdc++");
188 CmdArgs.push_back("-lsupc++");
189 break;
190 }
191 CmdArgs.push_back("-lunwind");
192}

194void BareMetal::AddLinkRuntimeLib(const ArgList &Args,
195 ArgStringList &CmdArgs) const {
196 SmallString<32> LibName("-lclang_rt.builtins-");
197 if (Target == BaremetalTarget::ARM) {
198 LibName += getTriple().getArchName();
199 LibName += ".a";
200 } else {
201 assert(Target == BaremetalTarget::MIPS);
202 if (getTriple().getArch() == llvm::Triple::cheri && !IsCheriPurecap) {
203 LibName += "mips64";
204 } else if (getTriple().isMIPS() && IsCheriPurecap) {
205 LibName += "cheri"; // TODO: would be nice to have CHERI size here
206 } else {
207 LibName += getTriple().getArchName();
208 }
209 }
210 CmdArgs.push_back(
211 Args.MakeArgString(LibName));
212}

214void baremetal::Linker::ConstructJob(Compilation &C, const JobAction &JA,
215 const InputInfo &Output,
216 const InputInfoList &Inputs,
217 const ArgList &Args,
218 const char *LinkingOutput) const {
219 ArgStringList CmdArgs;
220
221 auto &TC = static_cast<const toolchains::BareMetal&>(getToolChain());
222
223 AddLinkerInputs(TC, Inputs, Args, CmdArgs, JA);
224
225 CmdArgs.push_back("-Bstatic");
226
227 CmdArgs.push_back(Args.MakeArgString("-L" + TC.getRuntimesDir()));
228
229 Args.AddAllArgs(CmdArgs, {options::OPT_L, options::OPT_T_Group,
230 options::OPT_e, options::OPT_s, options::OPT_t,
231 options::OPT_Z_Flag, options::OPT_r});
232
233 if (TC.ShouldLinkCXXStdlib(Args))
234 TC.AddCXXStdlibLibArgs(Args, CmdArgs);
235 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs)) {
236 CmdArgs.push_back("-lc");
237 CmdArgs.push_back("-lm");
238
239 TC.AddLinkRuntimeLib(Args, CmdArgs);
240 }
241
242 CmdArgs.push_back("-o");
243 CmdArgs.push_back(Output.getFilename());
244
245 C.addCommand(llvm::make_unique<Command>(JA, *this,
246 Args.MakeArgString(TC.GetLinkerPath()),
247 CmdArgs, Inputs));
248}
