19class LLVM_LIBRARY_VISIBILITY RISCVToolChain : public Generic_ELF {

32 addLibStdCxxIncludePaths(const llvm::opt::ArgList &DriverArgs,
33 llvm::opt::ArgStringList &CC1Args) const override;

35protected:

38private:
