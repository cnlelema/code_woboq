23// Get CPU and ABI names. They are not independent
24// so we have to calculate them together.
25void mips::getMipsCPUAndABI(const ArgList &Args, const llvm::Triple &Triple,
26 StringRef &CPUName, StringRef &ABIName) {
27 const char *DefMips32CPU = "mips32r2";
28 const char *DefMips64CPU = "mips64r2";
29 const char *CHERICPU = "cheri128";
30
31 // MIPS32r6 is the default for mips(el)?-img-linux-gnu and MIPS64r6 is the
32 // default for mips64(el)?-img-linux-gnu.
33 if (Triple.getVendor() == llvm::Triple::ImaginationTechnologies &&
34 Triple.isGNUEnvironment()) {
35 DefMips32CPU = "mips32r6";
36 DefMips64CPU = "mips64r6";
37 }
38 if (Arg *A = Args.getLastArg(options::OPT_cheri, options::OPT_cheri_EQ)) {
39 if (A->getOption().matches(options::OPT_cheri))
40 CHERICPU = "cheri128";
41 else
42 CHERICPU = llvm::StringSwitch<const char *>(A->getValue())
43 .Case("64", "cheri64")
44 .Case("128", "cheri128")
45 .Case("256", "cheri256")
46 .Default("cheri128");
47 }
48
49 if (Triple.getSubArch() == llvm::Triple::MipsSubArch_r6) {
50 DefMips32CPU = "mips32r6";
51 DefMips64CPU = "mips64r6";
52 }
53
54 // MIPS64r6 is the default for Android MIPS64 (mips64el-linux-android).
55 if (Triple.isAndroid()) {
56 DefMips32CPU = "mips32";
57 DefMips64CPU = "mips64r6";
58 }
59
60 // MIPS3 is the default for mips64*-unknown-openbsd.
61 if (Triple.isOSOpenBSD())
62 DefMips64CPU = "mips3";
63
64 // MIPS2 is the default for mips(el)?-unknown-freebsd.
65 // MIPS3 is the default for mips64(el)?-unknown-freebsd.
66 if (Triple.isOSFreeBSD()) {
67 DefMips32CPU = "mips2";
68 DefMips64CPU = "mips3";
69 }
70
71 if (Arg *A = Args.getLastArg(clang::driver::options::OPT_march_EQ,
72 options::OPT_mcpu_EQ))
73 CPUName = A->getValue();
74
75 if (Arg *A = Args.getLastArg(options::OPT_mabi_EQ)) {
76 ABIName = A->getValue();
77 // Convert a GNU style Mips ABI name to the name
78 // accepted by LLVM Mips backend.
79 ABIName = llvm::StringSwitch<llvm::StringRef>(ABIName)
80 .Case("32", "o32")
81 .Case("64", "n64")
82 .Default(ABIName);
83 }
84
85 if (ABIName.empty() && (Triple.getEnvironment() == llvm::Triple::CheriPurecap))
86 ABIName = "purecap";
87 else if (ABIName.empty() && (Triple.getEnvironment() == llvm::Triple::GNUABIN32))
88 ABIName = "n32";
89 else if (ABIName.empty() && (Triple.getEnvironment() == llvm::Triple::GNUABI64))
90 ABIName = "n64";
91
92 if (Triple.getArch() == llvm::Triple::cheri || ABIName == "purecap" ||
93 Triple.getEnvironment() == llvm::Triple::CheriPurecap ||
94 Triple.getSubArch() == llvm::Triple::MipsSubArch_cheri64 ||
95 Triple.getSubArch() == llvm::Triple::MipsSubArch_cheri128 ||
96 Triple.getSubArch() == llvm::Triple::MipsSubArch_cheri256) {
97 DefMips32CPU = CHERICPU;
98 DefMips64CPU = CHERICPU;
99 }
100
101 // Setup default CPU and ABI names.
102 if (CPUName.empty() && ABIName.empty()) {
103 switch (Triple.getArch()) {
104 default:
105 llvm_unreachable("Unexpected triple arch name");
106 case llvm::Triple::mips:
107 case llvm::Triple::mipsel:
108 CPUName = DefMips32CPU;
109 break;
110 case llvm::Triple::mips64:
111 case llvm::Triple::mips64el:
112 CPUName = DefMips64CPU;
113 break;
114 case llvm::Triple::cheri:
115 CPUName = CHERICPU;
116 break;
117 }
118 }
119
120 if (ABIName.empty() &&
121 (Triple.getVendor() == llvm::Triple::MipsTechnologies ||
122 Triple.getVendor() == llvm::Triple::ImaginationTechnologies)) {
123 ABIName = llvm::StringSwitch<const char *>(CPUName)
124 .Case("mips1", "o32")
125 .Case("mips2", "o32")
126 .Case("mips3", "n64")
127 .Case("mips4", "n64")
128 .Case("mips5", "n64")
129 .Case("mips32", "o32")
130 .Case("mips32r2", "o32")
131 .Case("mips32r3", "o32")
132 .Case("mips32r5", "o32")
133 .Case("mips32r6", "o32")
134 .Case("mips64", "n64")
135 .Case("mips64r2", "n64")
136 .Case("mips64r3", "n64")
137 .Case("mips64r5", "n64")
138 .Case("mips64r6", "n64")
139 .Case("octeon", "n64")
140 .Case("p5600", "o32")
141 .Default("");
142 }
143
144 if (ABIName.empty()) {
145 // Deduce ABI name from the target triple.
146 ABIName = Triple.isMIPS32() ? "o32" : "n64";
147 }
148
149 if (CPUName.empty()) {
150 // Deduce CPU name from ABI name.
151 CPUName = llvm::StringSwitch<const char *>(ABIName)
152 .Case("o32", DefMips32CPU)
153 .Cases("n32", "n64", DefMips64CPU)
154 .Case("purecap", CHERICPU)
155 .Default("");
156 }
157
158 // change CPU from cheri to cheri128 if -mllvm -cheri128 was passed
159 if (CPUName == CHERICPU)
160 for (const Arg *A : Args.filtered(options::OPT_mllvm))
161 if (StringRef(A->getValue(0)) == "-cheri128")
162 CPUName = "cheri128";
163
164 // FIXME: Warn on inconsistent use of -march and -mabi.
165}

167std::string mips::getMipsABILibSuffix(const ArgList &Args,
168 const llvm::Triple &Triple) {
169 StringRef CPUName, ABIName;
170 tools::mips::getMipsCPUAndABI(Args, Triple, CPUName, ABIName);
171 return llvm::StringSwitch<std::string>(ABIName)
172 .Case("o32", "")
173 .Case("n32", "32")
174 .Case("n64", "64");
175}

177// Convert ABI name to the GNU tools acceptable variant.
178StringRef mips::getGnuCompatibleMipsABIName(StringRef ABI) {
179 return llvm::StringSwitch<llvm::StringRef>(ABI)
180 .Case("o32", "32")
181 .Case("n64", "64")
182 .Default(ABI);
183}

185// Select the MIPS float ABI as determined by -msoft-float, -mhard-float,
186// and -mfloat-abi=.
187mips::FloatABI mips::getMipsFloatABI(const Driver &D, const ArgList &Args) {
188 mips::FloatABI ABI = mips::FloatABI::Invalid;
189 if (Arg *A =
190 Args.getLastArg(options::OPT_msoft_float, options::OPT_mhard_float,
191 options::OPT_mfloat_abi_EQ)) {
192 if (A->getOption().matches(options::OPT_msoft_float))
193 ABI = mips::FloatABI::Soft;
194 else if (A->getOption().matches(options::OPT_mhard_float))
195 ABI = mips::FloatABI::Hard;
196 else {
197 ABI = llvm::StringSwitch<mips::FloatABI>(A->getValue())
198 .Case("soft", mips::FloatABI::Soft)
199 .Case("hard", mips::FloatABI::Hard)
200 .Default(mips::FloatABI::Invalid);
201 if (ABI == mips::FloatABI::Invalid && !StringRef(A->getValue()).empty()) {
202 D.Diag(clang::diag::err_drv_invalid_mfloat_abi) << A->getAsString(Args);
203 ABI = mips::FloatABI::Hard;
204 }
205 }
206 }
207
208 // If unspecified, choose the default based on the platform.
209 if (ABI == mips::FloatABI::Invalid) {
210 // Assume "hard", because it's a default value used by gcc.
211 // When we start to recognize specific target MIPS processors,
212 // we will be able to select the default more correctly.
213 ABI = mips::FloatABI::Hard;
214 }
215
216 assert(ABI != mips::FloatABI::Invalid && "must select an ABI");
217 return ABI;
218}

220void mips::getMIPSTargetFeatures(const Driver &D, const llvm::Triple &Triple,
221 const ArgList &Args,
222 std::vector<StringRef> &Features) {
223 StringRef CPUName;
224 StringRef ABIName;
225 getMipsCPUAndABI(Args, Triple, CPUName, ABIName);
226 ABIName = getGnuCompatibleMipsABIName(ABIName);
227 if (ABIName == "purecap" && !Triple.isMIPS64())
228 D.Diag(diag::err_drv_argument_not_allowed_with) << "-mabi=purecap"
229 << Triple.str();
230
231 // Historically, PIC code for MIPS was associated with -mabicalls, a.k.a
232 // SVR4 abicalls. Static code does not use SVR4 calling sequences. An ABI
233 // extension was developed by Richard Sandiford & Code Sourcery to support
234 // static code calling PIC code (CPIC). For O32 and N32 this means we have
235 // several combinations of PIC/static and abicalls. Pure static, static
236 // with the CPIC extension, and pure PIC code.
237
238 // At final link time, O32 and N32 with CPIC will have another section
239 // added to the binary which contains the stub functions to perform
240 // any fixups required for PIC code.
241
242 // For N64, the situation is more regular: code can either be static
243 // (non-abicalls) or PIC (abicalls). GCC has traditionally picked PIC code
244 // code for N64. Since Clang has already built the relocation model portion
245 // of the commandline, we pick add +noabicalls feature in the N64 static
246 // case.
247
248 // The is another case to be accounted for: -msym32, which enforces that all
249 // symbols have 32 bits in size. In this case, N64 can in theory use CPIC
250 // but it is unsupported.
251
252 // The combinations for N64 are:
253 // a) Static without abicalls and 64bit symbols.
254 // b) Static with abicalls and 32bit symbols.
255 // c) PIC with abicalls and 64bit symbols.
256
257 // For case (a) we need to add +noabicalls for N64.
258
259 bool IsN64 = ABIName == "64";
260 bool IsPIC = false;
261 bool NonPIC = false;
262
263 Arg *LastPICArg = Args.getLastArg(options::OPT_fPIC, options::OPT_fno_PIC,
264 options::OPT_fpic, options::OPT_fno_pic,
265 options::OPT_fPIE, options::OPT_fno_PIE,
266 options::OPT_fpie, options::OPT_fno_pie);
267 if (LastPICArg) {
268 Option O = LastPICArg->getOption();
269 NonPIC =
270 (O.matches(options::OPT_fno_PIC) || O.matches(options::OPT_fno_pic) ||
271 O.matches(options::OPT_fno_PIE) || O.matches(options::OPT_fno_pie));
272 IsPIC =
273 (O.matches(options::OPT_fPIC) || O.matches(options::OPT_fpic) ||
274 O.matches(options::OPT_fPIE) || O.matches(options::OPT_fpie));
275 }
276
277 bool UseAbiCalls = false;
278
279 Arg *ABICallsArg =
280 Args.getLastArg(options::OPT_mabicalls, options::OPT_mno_abicalls);
281 UseAbiCalls =
282 !ABICallsArg || ABICallsArg->getOption().matches(options::OPT_mabicalls);
283
284 if (IsN64 && NonPIC && (!ABICallsArg || UseAbiCalls)) {
285 D.Diag(diag::warn_drv_unsupported_pic_with_mabicalls)
286 << LastPICArg->getAsString(Args) << (!ABICallsArg ? 0 : 1);
287 NonPIC = false;
288 }
289
290 if (ABICallsArg && !UseAbiCalls && IsPIC) {
291 D.Diag(diag::err_drv_unsupported_noabicalls_pic);
292 }
293
294 if (!UseAbiCalls)
295 Features.push_back("+noabicalls");
296 else
297 Features.push_back("-noabicalls");
298
299 if (Arg *A = Args.getLastArg(options::OPT_mlong_calls,
300 options::OPT_mno_long_calls)) {
301 if (A->getOption().matches(options::OPT_mno_long_calls))
302 Features.push_back("-long-calls");
303 else if (!UseAbiCalls)
304 Features.push_back("+long-calls");
305 else
306 D.Diag(diag::warn_drv_unsupported_longcalls) << (ABICallsArg ? 0 : 1);
307 }
308
309 mips::FloatABI FloatABI = mips::getMipsFloatABI(D, Args);
310 if (FloatABI == mips::FloatABI::Soft) {
311 // FIXME: Note, this is a hack. We need to pass the selected float
312 // mode to the MipsTargetInfoBase to define appropriate macros there.
313 // Now it is the only method.
314 Features.push_back("+soft-float");
315 }
316
317 if (Arg *A = Args.getLastArg(options::OPT_mnan_EQ)) {
318 StringRef Val = StringRef(A->getValue());
319 if (Val == "2008") {
320 if (mips::getIEEE754Standard(CPUName) & mips::Std2008)
321 Features.push_back("+nan2008");
322 else {
323 Features.push_back("-nan2008");
324 D.Diag(diag::warn_target_unsupported_nan2008) << CPUName;
325 }
326 } else if (Val == "legacy") {
327 if (mips::getIEEE754Standard(CPUName) & mips::Legacy)
328 Features.push_back("-nan2008");
329 else {
330 Features.push_back("+nan2008");
331 D.Diag(diag::warn_target_unsupported_nanlegacy) << CPUName;
332 }
333 } else
334 D.Diag(diag::err_drv_unsupported_option_argument)
335 << A->getOption().getName() << Val;
336 }
337
338 if (Arg *A = Args.getLastArg(options::OPT_mabs_EQ)) {
339 StringRef Val = StringRef(A->getValue());
340 if (Val == "2008") {
341 if (mips::getIEEE754Standard(CPUName) & mips::Std2008) {
342 Features.push_back("+abs2008");
343 } else {
344 Features.push_back("-abs2008");
345 D.Diag(diag::warn_target_unsupported_abs2008) << CPUName;
346 }
347 } else if (Val == "legacy") {
348 if (mips::getIEEE754Standard(CPUName) & mips::Legacy) {
349 Features.push_back("-abs2008");
350 } else {
351 Features.push_back("+abs2008");
352 D.Diag(diag::warn_target_unsupported_abslegacy) << CPUName;
353 }
354 } else {
355 D.Diag(diag::err_drv_unsupported_option_argument)
356 << A->getOption().getName() << Val;
357 }
358 }
359
360 AddTargetFeature(Args, Features, options::OPT_msingle_float,
361 options::OPT_mdouble_float, "single-float");
362 AddTargetFeature(Args, Features, options::OPT_mips16, options::OPT_mno_mips16,
363 "mips16");
364 AddTargetFeature(Args, Features, options::OPT_mmicromips,
365 options::OPT_mno_micromips, "micromips");
366 AddTargetFeature(Args, Features, options::OPT_mdsp, options::OPT_mno_dsp,
367 "dsp");
368 AddTargetFeature(Args, Features, options::OPT_mdspr2, options::OPT_mno_dspr2,
369 "dspr2");
370 AddTargetFeature(Args, Features, options::OPT_mmsa, options::OPT_mno_msa,
371 "msa");
372
373 // Add the last -mfp32/-mfpxx/-mfp64, if none are given and the ABI is O32
374 // pass -mfpxx, or if none are given and fp64a is default, pass fp64 and
375 // nooddspreg.
376 if (Arg *A = Args.getLastArg(options::OPT_mfp32, options::OPT_mfpxx,
377 options::OPT_mfp64)) {
378 if (A->getOption().matches(options::OPT_mfp32))
379 Features.push_back("-fp64");
380 else if (A->getOption().matches(options::OPT_mfpxx)) {
381 Features.push_back("+fpxx");
382 Features.push_back("+nooddspreg");
383 } else
384 Features.push_back("+fp64");
385 } else if (mips::shouldUseFPXX(Args, Triple, CPUName, ABIName, FloatABI)) {
386 Features.push_back("+fpxx");
387 Features.push_back("+nooddspreg");
388 } else if (mips::isFP64ADefault(Triple, CPUName)) {
389 Features.push_back("+fp64");
390 Features.push_back("+nooddspreg");
391 }
392
393 AddTargetFeature(Args, Features, options::OPT_mno_odd_spreg,
394 options::OPT_modd_spreg, "nooddspreg");
395 AddTargetFeature(Args, Features, options::OPT_mno_madd4, options::OPT_mmadd4,
396 "nomadd4");
397 AddTargetFeature(Args, Features, options::OPT_mmt, options::OPT_mno_mt, "mt");
398 AddTargetFeature(Args, Features, options::OPT_mcrc, options::OPT_mno_crc,
399 "crc");
400 AddTargetFeature(Args, Features, options::OPT_mvirt, options::OPT_mno_virt,
401 "virt");
402 AddTargetFeature(Args, Features, options::OPT_mginv, options::OPT_mno_ginv,
403 "ginv");
404
405 if (Arg *A = Args.getLastArg(options::OPT_cheri, options::OPT_cheri_EQ)) {
406 Features.push_back("+chericap");
407 if (A->getOption().matches(options::OPT_cheri))
408 Features.push_back("+cheri128");
409 else {
410 auto SizeFeature = llvm::StringSwitch<const char *>(A->getValue())
411 .Case("64", "+cheri64")
412 .Case("128", "+cheri128")
413 .Case("256", "+cheri256")
414 .Default(nullptr);
415 if (SizeFeature)
416 Features.push_back(SizeFeature);
417 else
418 D.Diag(diag::err_drv_unsupported_option_argument)
419 << A->getOption().getName() << A->getValue();
420 }
421 }
422 if (Arg *A = Args.getLastArg(options::OPT_mindirect_jump_EQ)) {
423 StringRef Val = StringRef(A->getValue());
424 if (Val == "hazard") {
425 Arg *B =
426 Args.getLastArg(options::OPT_mmicromips, options::OPT_mno_micromips);
427 Arg *C = Args.getLastArg(options::OPT_mips16, options::OPT_mno_mips16);
428
429 if (B && B->getOption().matches(options::OPT_mmicromips))
430 D.Diag(diag::err_drv_unsupported_indirect_jump_opt)
431 << "hazard" << "micromips";
432 else if (C && C->getOption().matches(options::OPT_mips16))
433 D.Diag(diag::err_drv_unsupported_indirect_jump_opt)
434 << "hazard" << "mips16";
435 else if (mips::supportsIndirectJumpHazardBarrier(CPUName))
436 Features.push_back("+use-indirect-jump-hazard");
437 else
438 D.Diag(diag::err_drv_unsupported_indirect_jump_opt)
439 << "hazard" << CPUName;
440 } else
441 D.Diag(diag::err_drv_unknown_indirect_jump_opt) << Val;
442 }
443}

445mips::IEEE754Standard mips::getIEEE754Standard(StringRef &CPU) {
446 // Strictly speaking, mips32r2 and mips64r2 do not conform to the
447 // IEEE754-2008 standard. Support for this standard was first introduced
448 // in Release 3. However, other compilers have traditionally allowed it
449 // for Release 2 so we should do the same.
450 return (IEEE754Standard)llvm::StringSwitch<int>(CPU)
451 .Case("mips1", Legacy)
452 .Case("mips2", Legacy)
453 .Case("mips3", Legacy)
454 .Case("mips4", Legacy)
455 .Case("mips5", Legacy)
456 .Case("mips32", Legacy)
457 .Case("mips32r2", Legacy | Std2008)
458 .Case("mips32r3", Legacy | Std2008)
459 .Case("mips32r5", Legacy | Std2008)
460 .Case("mips32r6", Std2008)
461 .Case("mips64", Legacy)
462 .Case("mips64r2", Legacy | Std2008)
463 .Case("mips64r3", Legacy | Std2008)
464 .Case("mips64r5", Legacy | Std2008)
465 .Case("mips64r6", Std2008)
466 .Default(Std2008);
467}

487bool mips::isNaN2008(const ArgList &Args, const llvm::Triple &Triple) {
488 if (Arg *NaNArg = Args.getLastArg(options::OPT_mnan_EQ))
489 return llvm::StringSwitch<bool>(NaNArg->getValue())
490 .Case("2008", true)
491 .Case("legacy", false)

493

497 .Default(false);
498
499 return false;

512bool mips::isFPXXDefault(const llvm::Triple &Triple, StringRef CPUName,
513 StringRef ABIName, mips::FloatABI FloatABI) {
514 if (Triple.getVendor() != llvm::Triple::ImaginationTechnologies &&
515 Triple.getVendor() != llvm::Triple::MipsTechnologies &&
516 !Triple.isAndroid())
517 return false;
518
519 if (ABIName != "32")
520 return false;
521
522 // FPXX shouldn't be used if either -msoft-float or -mfloat-abi=soft is
523 // present.
524 if (FloatABI == mips::FloatABI::Soft)
525 return false;
526
527 return llvm::StringSwitch<bool>(CPUName)
528 .Cases("mips2", "mips3", "mips4", "mips5", true)
529 .Cases("mips32", "mips32r2", "mips32r3", "mips32r5", true)
530 .Cases("mips64", "mips64r2", "mips64r3", "mips64r5", true)
531 .Default(false);
532}
533
534bool mips::shouldUseFPXX(const ArgList &Args, const llvm::Triple &Triple,
535 StringRef CPUName, StringRef ABIName,
536 mips::FloatABI FloatABI) {
537 bool UseFPXX = isFPXXDefault(Triple, CPUName, ABIName, FloatABI);
538
539 // FPXX shouldn't be used if -msingle-float is present.
540 if (Arg *A = Args.getLastArg(options::OPT_msingle_float,
541 options::OPT_mdouble_float))
542 if (A->getOption().matches(options::OPT_msingle_float))
543 UseFPXX = false;
544
545 return UseFPXX;
546}

548bool mips::supportsIndirectJumpHazardBarrier(StringRef &CPU) {
549 // Supporting the hazard barrier method of dealing with indirect
550 // jumps requires MIPSR2 support.
551 return llvm::StringSwitch<bool>(CPU)
552 .Case("mips32r2", true)
553 .Case("mips32r3", true)
554 .Case("mips32r5", true)
555 .Case("mips32r6", true)
556 .Case("mips64r2", true)
557 .Case("mips64r3", true)
558 .Case("mips64r5", true)
559 .Case("mips64r6", true)
560 .Case("octeon", true)
561 .Case("p5600", true)
562 .Default(false);
563}
