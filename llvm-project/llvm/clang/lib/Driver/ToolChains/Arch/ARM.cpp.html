22// Get SubArch (vN).
23int arm::getARMSubArchVersionNumber(const llvm::Triple &Triple) {
24 llvm::StringRef Arch = Triple.getArchName();
25 return llvm::ARM::parseArchVersion(Arch);
26}

28// True if M-profile.
29bool arm::isARMMProfile(const llvm::Triple &Triple) {
30 llvm::StringRef Arch = Triple.getArchName();
31 return llvm::ARM::parseArchProfile(Arch) == llvm::ARM::ProfileKind::M;
32}

34// Get Arch/CPU from args.
35void arm::getARMArchCPUFromArgs(const ArgList &Args, llvm::StringRef &Arch,
36 llvm::StringRef &CPU, bool FromAs) {
37 if (const Arg *A = Args.getLastArg(clang::driver::options::OPT_mcpu_EQ))
38 CPU = A->getValue();
39 if (const Arg *A = Args.getLastArg(options::OPT_march_EQ))
40 Arch = A->getValue();
41 if (!FromAs)
42 return;
43
44 for (const Arg *A :
45 Args.filtered(options::OPT_Wa_COMMA, options::OPT_Xassembler)) {
46 StringRef Value = A->getValue();
47 if (Value.startswith("-mcpu="))
48 CPU = Value.substr(6);
49 if (Value.startswith("-march="))
50 Arch = Value.substr(7);
51 }
52}

73// Decode ARM features from string like +[no]featureA+[no]featureB+...
74static bool DecodeARMFeatures(const Driver &D, StringRef text,
75 StringRef CPU, llvm::ARM::ArchKind ArchKind,
76 std::vector<StringRef> &Features) {
77 SmallVector<StringRef, 8> Split;
78 text.split(Split, StringRef("+"), -1, false);
79
80 for (StringRef Feature : Split) {
81 if (!appendArchExtFeatures(CPU, ArchKind, Feature, Features))
82 return false;
83 }
84 return true;
85}
86
87static void DecodeARMFeaturesFromCPU(const Driver &D, StringRef CPU,
88 std::vector<StringRef> &Features) {
89 CPU = CPU.split("+").first;
90 if (CPU != "generic") {
91 llvm::ARM::ArchKind ArchKind = llvm::ARM::parseCPUArch(CPU);
92 unsigned Extension = llvm::ARM::getDefaultExtensions(CPU, ArchKind);
93 llvm::ARM::getExtensionFeatures(Extension, Features);
94 }
95}

97// Check if -march is valid by checking if it can be canonicalised and parsed.
98// getARMArch is used here instead of just checking the -march value in order
99// to handle -march=native correctly.
100static void checkARMArchName(const Driver &D, const Arg *A, const ArgList &Args,
101 llvm::StringRef ArchName, llvm::StringRef CPUName,
102 std::vector<StringRef> &Features,
103 std::vector<StringRef> &ExtensionFeatures,
104 const llvm::Triple &Triple) {
105 std::pair<StringRef, StringRef> Split = ArchName.split("+");
106
107 std::string MArch = arm::getARMArch(ArchName, Triple);
108 llvm::ARM::ArchKind ArchKind = llvm::ARM::parseArch(MArch);
109 if (ArchKind == llvm::ARM::ArchKind::INVALID ||
110 (Split.second.size() && !DecodeARMFeatures(
111 D, Split.second, CPUName, ArchKind, ExtensionFeatures)))
112 D.Diag(clang::diag::err_drv_clang_unsupported) << A->getAsString(Args);
113}

115// Check -mcpu=. Needs ArchName to handle -mcpu=generic.
116static void checkARMCPUName(const Driver &D, const Arg *A, const ArgList &Args,
117 llvm::StringRef CPUName, llvm::StringRef ArchName,
118 std::vector<StringRef> &Features,
119 std::vector<StringRef> &ExtensionFeatures,
120 const llvm::Triple &Triple) {
121 std::pair<StringRef, StringRef> Split = CPUName.split("+");
122
123 std::string CPU = arm::getARMTargetCPU(CPUName, ArchName, Triple);
124 llvm::ARM::ArchKind ArchKind =
125 arm::getLLVMArchKindForARM(CPU, ArchName, Triple);
126 if (ArchKind == llvm::ARM::ArchKind::INVALID ||
127 (Split.second.size() && !DecodeARMFeatures(
128 D, Split.second, CPU, ArchKind, ExtensionFeatures)))
129 D.Diag(clang::diag::err_drv_clang_unsupported) << A->getAsString(Args);
130}

132bool arm::useAAPCSForMachO(const llvm::Triple &T) {
133 // The backend is hardwired to assume AAPCS for M-class processors, ensure
134 // the frontend matches that.
135 return T.getEnvironment() == llvm::Triple::EABI ||
136 T.getOS() == llvm::Triple::UnknownOS || isARMMProfile(T);
137}

139// Select mode for reading thread pointer (-mtp=soft/cp15).
140arm::ReadTPMode arm::getReadTPMode(const ToolChain &TC, const ArgList &Args) {
141 if (Arg *A = Args.getLastArg(options::OPT_mtp_mode_EQ)) {
142 const Driver &D = TC.getDriver();
143 arm::ReadTPMode ThreadPointer =
144 llvm::StringSwitch<arm::ReadTPMode>(A->getValue())
145 .Case("cp15", ReadTPMode::Cp15)
146 .Case("soft", ReadTPMode::Soft)
147 .Default(ReadTPMode::Invalid);
148 if (ThreadPointer != ReadTPMode::Invalid)
149 return ThreadPointer;
150 if (StringRef(A->getValue()).empty())
151 D.Diag(diag::err_drv_missing_arg_mtp) << A->getAsString(Args);
152 else
153 D.Diag(diag::err_drv_invalid_mtp) << A->getAsString(Args);
154 return ReadTPMode::Invalid;
155 }
156 return ReadTPMode::Soft;
157}

159// Select the float ABI as determined by -msoft-float, -mhard-float, and
160// -mfloat-abi=.
161arm::FloatABI arm::getARMFloatABI(const ToolChain &TC, const ArgList &Args) {
162 const Driver &D = TC.getDriver();
163 const llvm::Triple &Triple = TC.getEffectiveTriple();
164 auto SubArch = getARMSubArchVersionNumber(Triple);
165 arm::FloatABI ABI = FloatABI::Invalid;
166 if (Arg *A =
167 Args.getLastArg(options::OPT_msoft_float, options::OPT_mhard_float,
168 options::OPT_mfloat_abi_EQ)) {
169 if (A->getOption().matches(options::OPT_msoft_float)) {
170 ABI = FloatABI::Soft;
171 } else if (A->getOption().matches(options::OPT_mhard_float)) {
172 ABI = FloatABI::Hard;
173 } else {
174 ABI = llvm::StringSwitch<arm::FloatABI>(A->getValue())
175 .Case("soft", FloatABI::Soft)
176 .Case("softfp", FloatABI::SoftFP)
177 .Case("hard", FloatABI::Hard)
178 .Default(FloatABI::Invalid);
179 if (ABI == FloatABI::Invalid && !StringRef(A->getValue()).empty()) {
180 D.Diag(diag::err_drv_invalid_mfloat_abi) << A->getAsString(Args);
181 ABI = FloatABI::Soft;
182 }
183 }
184
185 // It is incorrect to select hard float ABI on MachO platforms if the ABI is
186 // "apcs-gnu".
187 if (Triple.isOSBinFormatMachO() && !useAAPCSForMachO(Triple) &&
188 ABI == FloatABI::Hard) {
189 D.Diag(diag::err_drv_unsupported_opt_for_target) << A->getAsString(Args)
190 << Triple.getArchName();
191 }
192 }
193
194 // If unspecified, choose the default based on the platform.
195 if (ABI == FloatABI::Invalid) {
196 switch (Triple.getOS()) {
197 case llvm::Triple::Darwin:
198 case llvm::Triple::MacOSX:
199 case llvm::Triple::IOS:
200 case llvm::Triple::TvOS: {
201 // Darwin defaults to "softfp" for v6 and v7.
202 ABI = (SubArch == 6 || SubArch == 7) ? FloatABI::SoftFP : FloatABI::Soft;
203 ABI = Triple.isWatchABI() ? FloatABI::Hard : ABI;
204 break;
205 }
206 case llvm::Triple::WatchOS:
207 ABI = FloatABI::Hard;
208 break;
209
210 // FIXME: this is invalid for WindowsCE
211 case llvm::Triple::Win32:
212 ABI = FloatABI::Hard;
213 break;
214
215 case llvm::Triple::NetBSD:
216 switch (Triple.getEnvironment()) {
217 case llvm::Triple::EABIHF:
218 case llvm::Triple::GNUEABIHF:
219 ABI = FloatABI::Hard;
220 break;
221 default:
222 ABI = FloatABI::Soft;
223 break;
224 }
225 break;
226
227 case llvm::Triple::FreeBSD:
228 switch (Triple.getEnvironment()) {
229 case llvm::Triple::GNUEABIHF:
230 ABI = FloatABI::Hard;
231 break;
232 default:
233 // FreeBSD defaults to soft float
234 ABI = FloatABI::Soft;
235 break;
236 }
237 break;
238
239 case llvm::Triple::OpenBSD:
240 ABI = FloatABI::SoftFP;
241 break;
242
243 default:
244 switch (Triple.getEnvironment()) {
245 case llvm::Triple::GNUEABIHF:
246 case llvm::Triple::MuslEABIHF:
247 case llvm::Triple::EABIHF:
248 ABI = FloatABI::Hard;
249 break;
250 case llvm::Triple::GNUEABI:
251 case llvm::Triple::MuslEABI:
252 case llvm::Triple::EABI:
253 // EABI is always AAPCS, and if it was not marked 'hard', it's softfp
254 ABI = FloatABI::SoftFP;
255 break;
256 case llvm::Triple::Android:
257 ABI = (SubArch >= 7) ? FloatABI::SoftFP : FloatABI::Soft;
258 break;
259 default:
260 // Assume "soft", but warn the user we are guessing.
261 if (Triple.isOSBinFormatMachO() &&
262 Triple.getSubArch() == llvm::Triple::ARMSubArch_v7em)
263 ABI = FloatABI::Hard;
264 else
265 ABI = FloatABI::Soft;
266
267 if (Triple.getOS() != llvm::Triple::UnknownOS ||
268 !Triple.isOSBinFormatMachO())
269 D.Diag(diag::warn_drv_assuming_mfloat_abi_is) << "soft";
270 break;
271 }
272 }
273 }
274
275 assert(ABI != FloatABI::Invalid && "must select an ABI");
276 return ABI;
277}

279void arm::getARMTargetFeatures(const ToolChain &TC,
280 const llvm::Triple &Triple,
281 const ArgList &Args,
282 ArgStringList &CmdArgs,
283 std::vector<StringRef> &Features,
284 bool ForAS) {
285 const Driver &D = TC.getDriver();
286
287 bool KernelOrKext =
288 Args.hasArg(options::OPT_mkernel, options::OPT_fapple_kext);
289 arm::FloatABI ABI = arm::getARMFloatABI(TC, Args);
290 arm::ReadTPMode ThreadPointer = arm::getReadTPMode(TC, Args);
291 const Arg *WaCPU = nullptr, *WaFPU = nullptr;
292 const Arg *WaHDiv = nullptr, *WaArch = nullptr;
293
294 // This vector will accumulate features from the architecture
295 // extension suffixes on -mcpu and -march (e.g. the 'bar' in
296 // -mcpu=foo+bar). We want to apply those after the features derived
297 // from the FPU, in case -mfpu generates a negative feature which
298 // the +bar is supposed to override.
299 std::vector<StringRef> ExtensionFeatures;
300
301 if (!ForAS) {
302 // FIXME: Note, this is a hack, the LLVM backend doesn't actually use these
303 // yet (it uses the -mfloat-abi and -msoft-float options), and it is
304 // stripped out by the ARM target. We should probably pass this a new
305 // -target-option, which is handled by the -cc1/-cc1as invocation.
306 //
307 // FIXME2: For consistency, it would be ideal if we set up the target
308 // machine state the same when using the frontend or the assembler. We don't
309 // currently do that for the assembler, we pass the options directly to the
310 // backend and never even instantiate the frontend TargetInfo. If we did,
311 // and used its handleTargetFeatures hook, then we could ensure the
312 // assembler and the frontend behave the same.
313
314 // Use software floating point operations?
315 if (ABI == arm::FloatABI::Soft)
316 Features.push_back("+soft-float");
317
318 // Use software floating point argument passing?
319 if (ABI != arm::FloatABI::Hard)
320 Features.push_back("+soft-float-abi");
321 } else {
322 // Here, we make sure that -Wa,-mfpu/cpu/arch/hwdiv will be passed down
323 // to the assembler correctly.
324 for (const Arg *A :
325 Args.filtered(options::OPT_Wa_COMMA, options::OPT_Xassembler)) {
326 StringRef Value = A->getValue();
327 if (Value.startswith("-mfpu=")) {
328 WaFPU = A;
329 } else if (Value.startswith("-mcpu=")) {
330 WaCPU = A;
331 } else if (Value.startswith("-mhwdiv=")) {
332 WaHDiv = A;
333 } else if (Value.startswith("-march=")) {
334 WaArch = A;
335 }
336 }
337 }
338
339 if (ThreadPointer == arm::ReadTPMode::Cp15)
340 Features.push_back("+read-tp-hard");
341
342 const Arg *ArchArg = Args.getLastArg(options::OPT_march_EQ);
343 const Arg *CPUArg = Args.getLastArg(options::OPT_mcpu_EQ);
344 StringRef ArchName;
345 StringRef CPUName;
346
347 // Check -mcpu. ClangAs gives preference to -Wa,-mcpu=.
348 if (WaCPU) {
349 if (CPUArg)
350 D.Diag(clang::diag::warn_drv_unused_argument)
351 << CPUArg->getAsString(Args);
352 CPUName = StringRef(WaCPU->getValue()).substr(6);
353 CPUArg = WaCPU;
354 } else if (CPUArg)
355 CPUName = CPUArg->getValue();
356
357 // Check -march. ClangAs gives preference to -Wa,-march=.
358 if (WaArch) {
359 if (ArchArg)
360 D.Diag(clang::diag::warn_drv_unused_argument)
361 << ArchArg->getAsString(Args);
362 ArchName = StringRef(WaArch->getValue()).substr(7);
363 checkARMArchName(D, WaArch, Args, ArchName, CPUName,
364 Features, ExtensionFeatures, Triple);
365 // FIXME: Set Arch.
366 D.Diag(clang::diag::warn_drv_unused_argument) << WaArch->getAsString(Args);
367 } else if (ArchArg) {
368 ArchName = ArchArg->getValue();
369 checkARMArchName(D, ArchArg, Args, ArchName, CPUName,
370 Features, ExtensionFeatures, Triple);
371 }
372
373 // Add CPU features for generic CPUs
374 if (CPUName == "native") {
375 llvm::StringMap<bool> HostFeatures;
376 if (llvm::sys::getHostCPUFeatures(HostFeatures))
377 for (auto &F : HostFeatures)
378 Features.push_back(
379 Args.MakeArgString((F.second ? "+" : "-") + F.first()));
380 } else if (!CPUName.empty()) {
381 DecodeARMFeaturesFromCPU(D, CPUName, ExtensionFeatures);
382 }
383
384 if (CPUArg)
385 checkARMCPUName(D, CPUArg, Args, CPUName, ArchName,
386 Features, ExtensionFeatures, Triple);
387 // Honor -mfpu=. ClangAs gives preference to -Wa,-mfpu=.
388 const Arg *FPUArg = Args.getLastArg(options::OPT_mfpu_EQ);
389 if (WaFPU) {
390 if (FPUArg)
391 D.Diag(clang::diag::warn_drv_unused_argument)
392 << FPUArg->getAsString(Args);
393 getARMFPUFeatures(D, WaFPU, Args, StringRef(WaFPU->getValue()).substr(6),
394 Features);
395 } else if (FPUArg) {
396 getARMFPUFeatures(D, FPUArg, Args, FPUArg->getValue(), Features);
397 } else if (Triple.isAndroid() && getARMSubArchVersionNumber(Triple) >= 7) {
398 const char *AndroidFPU = "neon";
399 if (!llvm::ARM::getFPUFeatures(llvm::ARM::parseFPU(AndroidFPU), Features))
400 D.Diag(clang::diag::err_drv_clang_unsupported)
401 << std::string("-mfpu=") + AndroidFPU;
402 }
403
404 // Now we've finished accumulating features from arch, cpu and fpu,
405 // we can append the ones for architecture extensions that we
406 // collected separately.
407 Features.insert(std::end(Features),
408 std::begin(ExtensionFeatures), std::end(ExtensionFeatures));
409
410 // Honor -mhwdiv=. ClangAs gives preference to -Wa,-mhwdiv=.
411 const Arg *HDivArg = Args.getLastArg(options::OPT_mhwdiv_EQ);
412 if (WaHDiv) {
413 if (HDivArg)
414 D.Diag(clang::diag::warn_drv_unused_argument)
415 << HDivArg->getAsString(Args);
416 getARMHWDivFeatures(D, WaHDiv, Args,
417 StringRef(WaHDiv->getValue()).substr(8), Features);
418 } else if (HDivArg)
419 getARMHWDivFeatures(D, HDivArg, Args, HDivArg->getValue(), Features);
420
421 // Handle (arch-dependent) fp16fml/fullfp16 relationship.
422 // Must happen before any features are disabled due to soft-float.
423 // FIXME: this fp16fml option handling will be reimplemented after the
424 // TargetParser rewrite.
425 const auto ItRNoFullFP16 = std::find(Features.rbegin(), Features.rend(), "-fullfp16");
426 const auto ItRFP16FML = std::find(Features.rbegin(), Features.rend(), "+fp16fml");
427 if (Triple.getSubArch() == llvm::Triple::SubArchType::ARMSubArch_v8_4a) {
428 const auto ItRFullFP16 = std::find(Features.rbegin(), Features.rend(), "+fullfp16");
429 if (ItRFullFP16 < ItRNoFullFP16 && ItRFullFP16 < ItRFP16FML) {
430 // Only entangled feature that can be to the right of this +fullfp16 is -fp16fml.
431 // Only append the +fp16fml if there is no -fp16fml after the +fullfp16.
432 if (std::find(Features.rbegin(), ItRFullFP16, "-fp16fml") == ItRFullFP16)
433 Features.push_back("+fp16fml");
434 }
435 else
436 goto fp16_fml_fallthrough;
437 }
438 else {
439fp16_fml_fallthrough:
440 // In both of these cases, putting the 'other' feature on the end of the vector will
441 // result in the same effect as placing it immediately after the current feature.
442 if (ItRNoFullFP16 < ItRFP16FML)
443 Features.push_back("-fp16fml");
444 else if (ItRNoFullFP16 > ItRFP16FML)
445 Features.push_back("+fullfp16");
446 }
447
448 // Setting -msoft-float/-mfloat-abi=soft effectively disables the FPU (GCC
449 // ignores the -mfpu options in this case).
450 // Note that the ABI can also be set implicitly by the target selected.
451 if (ABI == arm::FloatABI::Soft) {
452 llvm::ARM::getFPUFeatures(llvm::ARM::FK_NONE, Features);
453
454 // Disable all features relating to hardware FP.
455 // FIXME: Disabling fpregs should be enough all by itself, since all
456 // the other FP features are dependent on it. However
457 // there is currently no easy way to test this in clang, so for
458 // now just be explicit and disable all known dependent features
459 // as well.
460 for (std::string Feature : {
461 "vfp2", "vfp2sp", "vfp2d16", "vfp2d16sp",
462 "vfp3", "vfp3sp", "vfp3d16", "vfp3d16sp",
463 "vfp4", "vfp4sp", "vfp4d16", "vfp4d16sp",
464 "fp-armv8", "fp-armv8sp", "fp-armv8d16", "fp-armv8d16sp",
465 "fullfp16", "neon", "crypto", "dotprod", "fp16fml",
466 "fp64", "d32", "fpregs"})
467 Features.push_back(Args.MakeArgString("-" + Feature));
468 }
469
470 // En/disable crc code generation.
471 if (Arg *A = Args.getLastArg(options::OPT_mcrc, options::OPT_mnocrc)) {
472 if (A->getOption().matches(options::OPT_mcrc))
473 Features.push_back("+crc");
474 else
475 Features.push_back("-crc");
476 }
477
478 // For Arch >= ARMv8.0: crypto = sha2 + aes
479 // FIXME: this needs reimplementation after the TargetParser rewrite
480 if (ArchName.find_lower("armv8a") != StringRef::npos ||
481 ArchName.find_lower("armv8.1a") != StringRef::npos ||
482 ArchName.find_lower("armv8.2a") != StringRef::npos ||
483 ArchName.find_lower("armv8.3a") != StringRef::npos ||
484 ArchName.find_lower("armv8.4a") != StringRef::npos) {
485 if (ArchName.find_lower("+crypto") != StringRef::npos) {
486 if (ArchName.find_lower("+nosha2") == StringRef::npos)
487 Features.push_back("+sha2");
488 if (ArchName.find_lower("+noaes") == StringRef::npos)
489 Features.push_back("+aes");
490 } else if (ArchName.find_lower("-crypto") != StringRef::npos) {
491 if (ArchName.find_lower("+sha2") == StringRef::npos)
492 Features.push_back("-sha2");
493 if (ArchName.find_lower("+aes") == StringRef::npos)
494 Features.push_back("-aes");
495 }
496 }
497
498 // CMSE: Check for target 8M (for -mcmse to be applicable) is performed later.
499 if (Args.getLastArg(options::OPT_mcmse))
500 Features.push_back("+8msecext");
501
502 // Look for the last occurrence of -mlong-calls or -mno-long-calls. If
503 // neither options are specified, see if we are compiling for kernel/kext and
504 // decide whether to pass "+long-calls" based on the OS and its version.
505 if (Arg *A = Args.getLastArg(options::OPT_mlong_calls,
506 options::OPT_mno_long_calls)) {
507 if (A->getOption().matches(options::OPT_mlong_calls))
508 Features.push_back("+long-calls");
509 } else if (KernelOrKext && (!Triple.isiOS() || Triple.isOSVersionLT(6)) &&
510 !Triple.isWatchOS()) {
511 Features.push_back("+long-calls");
512 }
513
514 // Generate execute-only output (no data access to code sections).
515 // This only makes sense for the compiler, not for the assembler.
516 if (!ForAS) {
517 // Supported only on ARMv6T2 and ARMv7 and above.
518 // Cannot be combined with -mno-movt or -mlong-calls
519 if (Arg *A = Args.getLastArg(options::OPT_mexecute_only, options::OPT_mno_execute_only)) {
520 if (A->getOption().matches(options::OPT_mexecute_only)) {
521 if (getARMSubArchVersionNumber(Triple) < 7 &&
522 llvm::ARM::parseArch(Triple.getArchName()) != llvm::ARM::ArchKind::ARMV6T2)
523 D.Diag(diag::err_target_unsupported_execute_only) << Triple.getArchName();
524 else if (Arg *B = Args.getLastArg(options::OPT_mno_movt))
525 D.Diag(diag::err_opt_not_valid_with_opt) << A->getAsString(Args) << B->getAsString(Args);
526 // Long calls create constant pool entries and have not yet been fixed up
527 // to play nicely with execute-only. Hence, they cannot be used in
528 // execute-only code for now
529 else if (Arg *B = Args.getLastArg(options::OPT_mlong_calls, options::OPT_mno_long_calls)) {
530 if (B->getOption().matches(options::OPT_mlong_calls))
531 D.Diag(diag::err_opt_not_valid_with_opt) << A->getAsString(Args) << B->getAsString(Args);
532 }
533 Features.push_back("+execute-only");
534 }
535 }
536 }
537
538 // Kernel code has more strict alignment requirements.
539 if (KernelOrKext)
540 Features.push_back("+strict-align");
541 else if (Arg *A = Args.getLastArg(options::OPT_mno_unaligned_access,
542 options::OPT_munaligned_access)) {
543 if (A->getOption().matches(options::OPT_munaligned_access)) {
544 // No v6M core supports unaligned memory access (v6M ARM ARM A3.2).
545 if (Triple.getSubArch() == llvm::Triple::SubArchType::ARMSubArch_v6m)
546 D.Diag(diag::err_target_unsupported_unaligned) << "v6m";
547 // v8M Baseline follows on from v6M, so doesn't support unaligned memory
548 // access either.
549 else if (Triple.getSubArch() == llvm::Triple::SubArchType::ARMSubArch_v8m_baseline)
550 D.Diag(diag::err_target_unsupported_unaligned) << "v8m.base";
551 } else
552 Features.push_back("+strict-align");
553 } else {
554 // Assume pre-ARMv6 doesn't support unaligned accesses.
555 //
556 // ARMv6 may or may not support unaligned accesses depending on the
557 // SCTLR.U bit, which is architecture-specific. We assume ARMv6
558 // Darwin and NetBSD targets support unaligned accesses, and others don't.
559 //
560 // ARMv7 always has SCTLR.U set to 1, but it has a new SCTLR.A bit
561 // which raises an alignment fault on unaligned accesses. Linux
562 // defaults this bit to 0 and handles it as a system-wide (not
563 // per-process) setting. It is therefore safe to assume that ARMv7+
564 // Linux targets support unaligned accesses. The same goes for NaCl.
565 //
566 // The above behavior is consistent with GCC.
567 int VersionNum = getARMSubArchVersionNumber(Triple);
568 if (Triple.isOSDarwin() || Triple.isOSNetBSD()) {
569 if (VersionNum < 6 ||
570 Triple.getSubArch() == llvm::Triple::SubArchType::ARMSubArch_v6m)
571 Features.push_back("+strict-align");
572 } else if (Triple.isOSLinux() || Triple.isOSNaCl()) {
573 if (VersionNum < 7)
574 Features.push_back("+strict-align");
575 } else
576 Features.push_back("+strict-align");
577 }
578
579 // llvm does not support reserving registers in general. There is support
580 // for reserving r9 on ARM though (defined as a platform-specific register
581 // in ARM EABI).
582 if (Args.hasArg(options::OPT_ffixed_r9))
583 Features.push_back("+reserve-r9");
584
585 // The kext linker doesn't know how to deal with movw/movt.
586 if (KernelOrKext || Args.hasArg(options::OPT_mno_movt))
587 Features.push_back("+no-movt");
588
589 if (Args.hasArg(options::OPT_mno_neg_immediates))
590 Features.push_back("+no-neg-immediates");
591}

593const std::string arm::getARMArch(StringRef Arch, const llvm::Triple &Triple) {
594 std::string MArch;
595 if (!Arch.empty())
596 MArch = Arch;
597 else
598 MArch = Triple.getArchName();
599 MArch = StringRef(MArch).split("+").first.lower();
600
601 // Handle -march=native.
602 if (MArch == "native") {
603 std::string CPU = llvm::sys::getHostCPUName();
604 if (CPU != "generic") {
605 // Translate the native cpu into the architecture suffix for that CPU.
606 StringRef Suffix = arm::getLLVMArchSuffixForARM(CPU, MArch, Triple);
607 // If there is no valid architecture suffix for this CPU we don't know how
608 // to handle it, so return no architecture.
609 if (Suffix.empty())
610 MArch = "";
611 else
612 MArch = std::string("arm") + Suffix.str();
613 }
614 }
615
616 return MArch;
617}

619/// Get the (LLVM) name of the minimum ARM CPU for the arch we are targeting.
620StringRef arm::getARMCPUForMArch(StringRef Arch, const llvm::Triple &Triple) {
621 std::string MArch = getARMArch(Arch, Triple);
622 // getARMCPUForArch defaults to the triple if MArch is empty, but empty MArch
623 // here means an -march=native that we can't handle, so instead return no CPU.
624 if (MArch.empty())
625 return StringRef();
626
627 // We need to return an empty string here on invalid MArch values as the
628 // various places that call this function can't cope with a null result.
629 return Triple.getARMCPUForArch(MArch);
630}

632/// getARMTargetCPU - Get the (LLVM) name of the ARM cpu we are targeting.
633std::string arm::getARMTargetCPU(StringRef CPU, StringRef Arch,
634 const llvm::Triple &Triple) {
635 // FIXME: Warn on inconsistent use of -mcpu and -march.
636 // If we have -mcpu=, use that.
637 if (!CPU.empty()) {
638 std::string MCPU = StringRef(CPU).split("+").first.lower();
639 // Handle -mcpu=native.
640 if (MCPU == "native")
641 return llvm::sys::getHostCPUName();
642 else
643 return MCPU;
644 }
645
646 return getARMCPUForMArch(Arch, Triple);
647}

649/// getLLVMArchSuffixForARM - Get the LLVM ArchKind value to use for a
650/// particular CPU (or Arch, if CPU is generic). This is needed to
651/// pass to functions like llvm::ARM::getDefaultFPU which need an
652/// ArchKind as well as a CPU name.
653llvm::ARM::ArchKind arm::getLLVMArchKindForARM(StringRef CPU, StringRef Arch,
654 const llvm::Triple &Triple) {
655 llvm::ARM::ArchKind ArchKind;
656 if (CPU == "generic") {
657 std::string ARMArch = tools::arm::getARMArch(Arch, Triple);
658 ArchKind = llvm::ARM::parseArch(ARMArch);
659 if (ArchKind == llvm::ARM::ArchKind::INVALID)
660 // In case of generic Arch, i.e. "arm",
661 // extract arch from default cpu of the Triple
662 ArchKind = llvm::ARM::parseCPUArch(Triple.getARMCPUForArch(ARMArch));
663 } else {
664 // FIXME: horrible hack to get around the fact that Cortex-A7 is only an
665 // armv7k triple if it's actually been specified via "-arch armv7k".
666 ArchKind = (Arch == "armv7k" || Arch == "thumbv7k")
667 ? llvm::ARM::ArchKind::ARMV7K
668 : llvm::ARM::parseCPUArch(CPU);
669 }
670 return ArchKind;
671}

673/// getLLVMArchSuffixForARM - Get the LLVM arch name to use for a particular
674/// CPU (or Arch, if CPU is generic).
675// FIXME: This is redundant with -mcpu, why does LLVM use this.
676StringRef arm::getLLVMArchSuffixForARM(StringRef CPU, StringRef Arch,
677 const llvm::Triple &Triple) {
678 llvm::ARM::ArchKind ArchKind = getLLVMArchKindForARM(CPU, Arch, Triple);
679 if (ArchKind == llvm::ARM::ArchKind::INVALID)
680 return "";
681 return llvm::ARM::getSubArch(ArchKind);
682}

684void arm::appendBE8LinkFlag(const ArgList &Args, ArgStringList &CmdArgs,
685 const llvm::Triple &Triple) {
686 if (Args.hasArg(options::OPT_r))
687 return;
688
689 // ARMv7 (and later) and ARMv6-M do not support BE-32, so instruct the linker
690 // to generate BE-8 executables.
691 if (arm::getARMSubArchVersionNumber(Triple) >= 7 || arm::isARMMProfile(Triple))
692 CmdArgs.push_back("--be8");
693}
