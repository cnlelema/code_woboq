13#include "Gnu.h"

17namespace clang {
18namespace driver {
19namespace tools {
20namespace rtems {
21class LLVM_LIBRARY_VISIBILITY Linker : public GnuTool {
22public:
23 Linker(const ToolChain &TC) : GnuTool("rtems::Linker", "ld.lld", TC) {}
24
25 bool hasIntegratedCPP() const override { return false; }
26 bool isLinkJob() const override { return true; }
27
28 void ConstructJob(Compilation &C, const JobAction &JA,
29 const InputInfo &Output, const InputInfoList &Inputs,
30 const llvm::opt::ArgList &TCArgs,
31 const char *LinkingOutput) const override;
32};
33} // end namespace rtems
34} // end namespace tools
35
36namespace toolchains {
37// TODO: make this actually work for upstream RTEMS
38// This is based on the Fuchsia toolchain (ajdusted to work for CHERI RTEMS)
39class LLVM_LIBRARY_VISIBILITY RTEMS : public ToolChain {
40public:
41 RTEMS(const Driver &D, const llvm::Triple &Triple,
42 const llvm::opt::ArgList &Args);
43
44 bool HasNativeLLVMSupport() const override { return true; }
45 bool IsIntegratedAssemblerDefault() const override { return true; }
46 bool IsMathErrnoDefault() const override { return false; }
47 RuntimeLibType GetDefaultRuntimeLibType() const override {
48 // FIXME: this is probably not correct for normal RTEMS, only cheri RTEMS
49 return ToolChain::RLT_CompilerRT;
50 }
51 std::string getCompilerRTPath() const override;
52
53 CXXStdlibType GetDefaultCXXStdlibType() const override {
54 return ToolChain::CST_Libcxx;
55 }
56 bool IsUnwindTablesDefault(const llvm::opt::ArgList &Args) const override {
57 return true;
58 }
59 bool isPICDefault() const override { return false; }
60 bool isPIEDefault() const override { return false; }
61 bool isPICDefaultForced() const override { return false; }
62 llvm::DebuggerKind getDefaultDebuggerTuning() const override {
63 return llvm::DebuggerKind::GDB;
64 }
65
66 RuntimeLibType
67 GetRuntimeLibType(const llvm::opt::ArgList &Args) const override;
68 CXXStdlibType
69 GetCXXStdlibType(const llvm::opt::ArgList &Args) const override;
70
71 void addClangTargetOptions(const llvm::opt::ArgList &DriverArgs,
72 llvm::opt::ArgStringList &CC1Args,
73 Action::OffloadKind DeviceOffloadKind) const override;
74 void
75 AddClangSystemIncludeArgs(const llvm::opt::ArgList &DriverArgs,
76 llvm::opt::ArgStringList &CC1Args) const override;
77 void
78 AddClangCXXStdlibIncludeArgs(const llvm::opt::ArgList &DriverArgs,
79 llvm::opt::ArgStringList &CC1Args) const override;
80 void AddCXXStdlibLibArgs(const llvm::opt::ArgList &Args,
81 llvm::opt::ArgStringList &CmdArgs) const override;
82
83 const char *getDefaultLinker() const override {
84 return "ld.lld";
85 }
86
87protected:
88 Tool *buildLinker() const override;
89};
90
91} // end namespace toolchains
