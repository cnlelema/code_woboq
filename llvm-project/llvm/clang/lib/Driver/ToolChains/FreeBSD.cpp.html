27void freebsd::Assembler::ConstructJob(Compilation &C, const JobAction &JA,
28 const InputInfo &Output,
29 const InputInfoList &Inputs,
30 const ArgList &Args,
31 const char *LinkingOutput) const {
32 claimNoWarnArgs(Args);
33 ArgStringList CmdArgs;
34
35 // When building 32-bit code on FreeBSD/amd64, we have to explicitly
36 // instruct as in the base system to assemble 32-bit code.
37 switch (getToolChain().getArch()) {
38 default:
39 break;
40 case llvm::Triple::x86:
41 CmdArgs.push_back("--32");
42 break;
43 case llvm::Triple::ppc:
44 CmdArgs.push_back("-a32");
45 break;
46 case llvm::Triple::mips:
47 case llvm::Triple::mipsel:
48 case llvm::Triple::mips64:
49 case llvm::Triple::mips64el:
50 case llvm::Triple::cheri: {
51 StringRef CPUName;
52 StringRef ABIName;
53 mips::getMipsCPUAndABI(Args, getToolChain().getTriple(), CPUName, ABIName);
54
55 CmdArgs.push_back("-march");
56 CmdArgs.push_back(CPUName.data());
57
58 CmdArgs.push_back("-mabi");
59 CmdArgs.push_back(mips::getGnuCompatibleMipsABIName(ABIName).data());
60
61 if (getToolChain().getTriple().isLittleEndian())
62 CmdArgs.push_back("-EL");
63 else
64 CmdArgs.push_back("-EB");
65
66 if (Arg *A = Args.getLastArg(options::OPT_G)) {
67 StringRef v = A->getValue();
68 CmdArgs.push_back(Args.MakeArgString("-G" + v));
69 A->claim();
70 }
71
72 AddAssemblerKPIC(getToolChain(), Args, CmdArgs);
73 break;
74 }
75 case llvm::Triple::arm:
76 case llvm::Triple::armeb:
77 case llvm::Triple::thumb:
78 case llvm::Triple::thumbeb: {
79 arm::FloatABI ABI = arm::getARMFloatABI(getToolChain(), Args);
80
81 if (ABI == arm::FloatABI::Hard)
82 CmdArgs.push_back("-mfpu=vfp");
83 else
84 CmdArgs.push_back("-mfpu=softvfp");
85
86 switch (getToolChain().getTriple().getEnvironment()) {
87 case llvm::Triple::GNUEABIHF:
88 case llvm::Triple::GNUEABI:
89 case llvm::Triple::EABI:
90 CmdArgs.push_back("-meabi=5");
91 break;
92
93 default:
94 CmdArgs.push_back("-matpcs");
95 }
96 break;
97 }
98 case llvm::Triple::sparc:
99 case llvm::Triple::sparcel:
100 case llvm::Triple::sparcv9: {
101 std::string CPU = getCPUName(Args, getToolChain().getTriple());
102 CmdArgs.push_back(sparc::getSparcAsmModeForCPU(CPU, getToolChain().getTriple()));
103 AddAssemblerKPIC(getToolChain(), Args, CmdArgs);
104 break;
105 }
106 }
107
108 Args.AddAllArgValues(CmdArgs, options::OPT_Wa_COMMA, options::OPT_Xassembler);
109
110 CmdArgs.push_back("-o");
111 CmdArgs.push_back(Output.getFilename());
112
113 for (const auto &II : Inputs)
114 CmdArgs.push_back(II.getFilename());
115
116 const char *Exec = Args.MakeArgString(getToolChain().GetProgramPath("as"));
117 C.addCommand(llvm::make_unique<Command>(JA, *this, Exec, CmdArgs, Inputs));
118}

120void freebsd::Linker::ConstructJob(Compilation &C, const JobAction &JA,
121 const InputInfo &Output,
122 const InputInfoList &Inputs,
123 const ArgList &Args,
124 const char *LinkingOutput) const {
125 const toolchains::FreeBSD &ToolChain =
126 static_cast<const toolchains::FreeBSD &>(getToolChain());
127 const Driver &D = ToolChain.getDriver();
128 const llvm::Triple::ArchType Arch = ToolChain.getArch();
129 bool IsCHERIPureCapABI = false;
130 if (ToolChain.getTriple().isMIPS())
131 IsCHERIPureCapABI = tools::mips::hasMipsAbiArg(Args, "purecap");
132 // For CheriABI default to -pie unless -static is also passed
133 // TODO: enable static PIE?
134 const bool CheriAbiPIEDefault =
135 IsCHERIPureCapABI && !Args.hasArg(options::OPT_static);
136 const bool IsPIEDefault = ToolChain.isPIEDefault() || CheriAbiPIEDefault;
137 // We can't pass -pie to the linker if any of -shared,-r,-no-pie,-no-pie are
138 // set
139 Arg *ConflictsWithPie = Args.getLastArg(options::OPT_r, options::OPT_shared);
140 // Have to negate here to handle the no-pie and nopie aliases
141 Arg *LastPIEArg = Args.getLastArg(options::OPT_pie, options::OPT_no_pie,
142 options::OPT_nopie);
143 const bool ExplicitPIE =
144 LastPIEArg && LastPIEArg->getOption().matches(options::OPT_pie);
145 if (ExplicitPIE && ConflictsWithPie) {
146 getToolChain().getDriver().Diag(diag::err_drv_argument_not_allowed_with)
147 << LastPIEArg->getAsString(Args) << ConflictsWithPie->getAsString(Args);
148 }
149 const bool IsPIE =
150 (LastPIEArg ? ExplicitPIE : IsPIEDefault) && !ConflictsWithPie;
151
152 ArgStringList CmdArgs;
153
154 // Silence warning for -cheri=NNN
155 Args.ClaimAllArgs(options::OPT_cheri_EQ);
156 Args.ClaimAllArgs(options::OPT_cheri);
157 // And -cheri-uintcap=
158 Args.ClaimAllArgs(options::OPT_cheri_uintcap_offset);
159 Args.ClaimAllArgs(options::OPT_cheri_uintcap_addr);
160 // Also pretend that all the captable flags were used
161 Args.ClaimAllArgs(options::OPT_cheri_cap_table_abi);
162 Args.ClaimAllArgs(options::OPT_cheri_cap_table);
163 Args.ClaimAllArgs(options::OPT_no_cheri_cap_table);
164 Args.ClaimAllArgs(options::OPT_cheri_large_cap_table);
165 Args.ClaimAllArgs(options::OPT_no_cheri_large_cap_table);
166 Args.ClaimAllArgs(options::OPT_cheri_cap_tls_abi);
167 // Various other CHERI flags can also be passed to the linker without warning:
168 Args.ClaimAllArgs(options::OPT_cheri_data_dependent_provenance);
169 Args.ClaimAllArgs(options::OPT_cheri_bounds_EQ);
170
171 // Silence warning for "clang -g foo.o -o foo"
172 Args.ClaimAllArgs(options::OPT_g_Group);
173 // and "clang -emit-llvm foo.o -o foo"
174 Args.ClaimAllArgs(options::OPT_emit_llvm);
175 // and for "clang -w foo.o -o foo". Other warning options are already
176 // handled somewhere else.
177 Args.ClaimAllArgs(options::OPT_w);
178
179 if (!D.SysRoot.empty())
180 CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
181
182 if (IsPIE)
183 CmdArgs.push_back("-pie");
184
185 CmdArgs.push_back("--eh-frame-hdr");
186 if (Args.hasArg(options::OPT_static)) {
187 CmdArgs.push_back("-Bstatic");
188 } else {
189 if (Args.hasArg(options::OPT_rdynamic))
190 CmdArgs.push_back("-export-dynamic");
191 if (Args.hasArg(options::OPT_shared)) {
192 CmdArgs.push_back("-Bshareable");
193 } else {
194 CmdArgs.push_back("-dynamic-linker");
195 if (IsCHERIPureCapABI)
196 CmdArgs.push_back("/libexec/ld-cheri-elf.so.1");
197 else
198 CmdArgs.push_back("/libexec/ld-elf.so.1");
199 }
200 if (ToolChain.getTriple().getOSMajorVersion() >= 9) {
201 if (Arch == llvm::Triple::arm || Arch == llvm::Triple::sparc ||
202 Arch == llvm::Triple::x86 || Arch == llvm::Triple::x86_64) {
203 CmdArgs.push_back("--hash-style=both");
204 }
205 }
206 CmdArgs.push_back("--enable-new-dtags");
207 }
208
209 // Explicitly set the linker emulation for platforms that might not
210 // be the default emulation for the linker.
211 switch (Arch) {
212 case llvm::Triple::x86:
213 CmdArgs.push_back("-m");
214 CmdArgs.push_back("elf_i386_fbsd");
215 break;
216 case llvm::Triple::ppc:
217 CmdArgs.push_back("-m");
218 CmdArgs.push_back("elf32ppc_fbsd");
219 break;
220 case llvm::Triple::mips:
221 CmdArgs.push_back("-m");
222 CmdArgs.push_back("elf32btsmip_fbsd");
223 break;
224 case llvm::Triple::mipsel:
225 CmdArgs.push_back("-m");
226 CmdArgs.push_back("elf32ltsmip_fbsd");
227 break;
228 case llvm::Triple::mips64:
229 case llvm::Triple::cheri:
230 CmdArgs.push_back("-m");
231 if (IsCHERIPureCapABI)
232 CmdArgs.push_back("elf64btsmip_cheri_fbsd");
233 else if (tools::mips::hasMipsAbiArg(Args, "n32"))
234 CmdArgs.push_back("elf32btsmipn32_fbsd");
235 else
236 CmdArgs.push_back("elf64btsmip_fbsd");
237 break;
238 case llvm::Triple::mips64el:
239 CmdArgs.push_back("-m");
240 if (tools::mips::hasMipsAbiArg(Args, "n32"))
241 CmdArgs.push_back("elf32ltsmipn32_fbsd");
242 else
243 CmdArgs.push_back("elf64ltsmip_fbsd");
244 break;
245 default:
246 break;
247 }
248
249 if (Arg *A = Args.getLastArg(options::OPT_G)) {
250 if (ToolChain.getTriple().isMIPS()) {
251 StringRef v = A->getValue();
252 CmdArgs.push_back(Args.MakeArgString("-G" + v));
253 A->claim();
254 }
255 }
256
257 // The FreeBSD/MIPS version of GNU ld is horribly buggy and errors out
258 // complaining about linking 32-bit and 64-bit code when linking CHERI code.
259 if (IsCHERIPureCapABI)
260 CmdArgs.push_back(Args.MakeArgString("--no-warn-mismatch"));
261
262 if (Output.isFilename()) {
263 CmdArgs.push_back("-o");
264 CmdArgs.push_back(Output.getFilename());
265 } else {
266 assert(Output.isNothing() && "Invalid output.");
267 }
268
269 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles)) {
270 const char *crt1 = nullptr;
271 if (!Args.hasArg(options::OPT_shared)) {
272 if (Args.hasArg(options::OPT_pg))
273 crt1 = "gcrt1.o";
274 else if (IsPIE)
275 crt1 = "Scrt1.o";
276 else
277 crt1 = "crt1.o";
278 }
279 if (crt1)
280 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crt1)));
281
282 // Don't support .init and .fini sections for CheriABI.
283 if (!IsCHERIPureCapABI)
284 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crti.o")));
285
286 const char *crtbegin = nullptr;
287 if (Args.hasArg(options::OPT_shared) || IsPIE)
288 crtbegin = "crtbeginS.o";
289 else if (Args.hasArg(options::OPT_static))
290 crtbegin = "crtbeginT.o";
291 else
292 crtbegin = "crtbegin.o";
293
294 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crtbegin)));
295 }
296
297 Args.AddAllArgs(CmdArgs, options::OPT_L);
298 ToolChain.AddFilePathLibArgs(Args, CmdArgs);
299 Args.AddAllArgs(CmdArgs, options::OPT_T_Group);
300 Args.AddAllArgs(CmdArgs, options::OPT_e);
301 Args.AddAllArgs(CmdArgs, options::OPT_s);
302 Args.AddAllArgs(CmdArgs, options::OPT_t);
303 Args.AddAllArgs(CmdArgs, options::OPT_Z_Flag);
304 Args.AddAllArgs(CmdArgs, options::OPT_r);
305
306 if (D.isUsingLTO()) {
307 assert(!Inputs.empty() && "Must have at least one input.");
308 AddGoldPlugin(ToolChain, Args, CmdArgs, Output, Inputs[0],
309 D.getLTOMode() == LTOK_Thin);
310 }
311
312 bool NeedsSanitizerDeps = addSanitizerRuntimes(ToolChain, Args, CmdArgs);
313 bool NeedsXRayDeps = addXRayRuntime(ToolChain, Args, CmdArgs);
314 AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
315
316 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs)) {
317 addOpenMPRuntime(CmdArgs, ToolChain, Args);
318 CmdArgs.push_back("--start-group");
319 if (D.CCCIsCXX()) {
320 if (ToolChain.ShouldLinkCXXStdlib(Args))
321 ToolChain.AddCXXStdlibLibArgs(Args, CmdArgs);
322 if (Args.hasArg(options::OPT_pg))
323 CmdArgs.push_back("-lm_p");
324 else
325 CmdArgs.push_back("-lm");
326 }
327 if (NeedsSanitizerDeps)
328 linkSanitizerRuntimeDeps(ToolChain, CmdArgs);
329 if (NeedsXRayDeps)
330 linkXRayRuntimeDeps(ToolChain, CmdArgs);
331 // FIXME: For some reason GCC passes -lgcc and -lgcc_s before adding
332 // the default system libraries. Just mimic this for now.
333 if (Args.hasArg(options::OPT_pg))
334 CmdArgs.push_back("-lgcc_p");
335 else
336 CmdArgs.push_back("-lgcc");
337 if (Args.hasArg(options::OPT_static)) {
338 CmdArgs.push_back("-lgcc_eh");
339 } else if (Args.hasArg(options::OPT_pg)) {
340 CmdArgs.push_back("-lgcc_eh_p");
341 } else {
342 CmdArgs.push_back("--as-needed");
343 CmdArgs.push_back("-lgcc_s");
344 CmdArgs.push_back("--no-as-needed");
345 }
346
347 if (Args.hasArg(options::OPT_pthread)) {
348 if (Args.hasArg(options::OPT_pg))
349 CmdArgs.push_back("-lpthread_p");
350 else
351 CmdArgs.push_back("-lpthread");
352 }
353
354 if (Args.hasArg(options::OPT_pg)) {
355 if (Args.hasArg(options::OPT_shared))
356 CmdArgs.push_back("-lc");
357 else
358 CmdArgs.push_back("-lc_p");
359 CmdArgs.push_back("-lgcc_p");
360 } else {
361 CmdArgs.push_back("-lc");
362 CmdArgs.push_back("-lgcc");
363 }
364
365 if (Args.hasArg(options::OPT_static)) {
366 CmdArgs.push_back("-lgcc_eh");
367 } else if (Args.hasArg(options::OPT_pg)) {
368 CmdArgs.push_back("-lgcc_eh_p");
369 } else {
370 CmdArgs.push_back("--as-needed");
371 CmdArgs.push_back("-lgcc_s");
372 CmdArgs.push_back("--no-as-needed");
373 }
374 CmdArgs.push_back("--end-group");
375 }
376
377 if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles)) {
378 if (Args.hasArg(options::OPT_shared) || IsPIE)
379 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crtendS.o")));
380 else
381 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crtend.o")));
382
383 // Don't support .init and .fini sections for CheriABI.
384 if (!IsCHERIPureCapABI)
385 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crtn.o")));
386 }
387
388 ToolChain.addProfileRTLibs(Args, CmdArgs);
389
390 const char *Exec = Args.MakeArgString(getToolChain().GetLinkerPath());
391 C.addCommand(llvm::make_unique<Command>(JA, *this, Exec, CmdArgs, Inputs));
392 if (Args.hasFlag(options::OPT_external_capsizefix, options::OPT_no_capsizefix,
393 false) &&
394 Args.hasFlag(options::OPT_cheri_linker, options::OPT_no_cheri_linker,
395 true)) {
396 Exec = Args.MakeArgString(getToolChain().GetProgramPath("capsizefix"));
397 ArgStringList SizeFixArgs;
398 if (Args.hasArg(options::OPT_verbose_capsizefix))
399 SizeFixArgs.push_back("--verbose");
400 SizeFixArgs.push_back(Output.getFilename());
401 InputInfoList In= {InputInfo(types::TY_Object, Output.getFilename(), Output.getBaseInput())};
402 C.addCommand(llvm::make_unique<Command>(JA, *this, Exec, SizeFixArgs, In));
403 }
404}

406/// FreeBSD - FreeBSD tool chain which can call as(1) and ld(1) directly.
407
408FreeBSD::FreeBSD(const Driver &D, const llvm::Triple &Triple,
409 const ArgList &Args)
410 : Generic_ELF(D, Triple, Args) {
411
412 IsCheriPurecap =
413 Triple.getEnvironment() == llvm::Triple::CheriPurecap ||
414 (Triple.isMIPS() && tools::mips::hasMipsAbiArg(Args, "purecap"));
415
416 // When targeting 32-bit platforms, look for '/usr/lib32/crt1.o' and fall
417 // back to '/usr/lib' if it doesn't exist.
418 if ((Triple.getArch() == llvm::Triple::x86 || Triple.isMIPS32() ||
419 Triple.getArch() == llvm::Triple::ppc) &&
420 D.getVFS().exists(getDriver().SysRoot + "/usr/lib32/crt1.o"))
421 getFilePaths().push_back(getDriver().SysRoot + "/usr/lib32");
422 else if (IsCheriPurecap &&
423 D.getVFS().exists(getDriver().SysRoot + "/usr/libcheri/crt1.o"))
424 getFilePaths().push_back(getDriver().SysRoot + "/usr/libcheri");
425 else
426 getFilePaths().push_back(getDriver().SysRoot + "/usr/lib");
427}

429ToolChain::CXXStdlibType FreeBSD::GetDefaultCXXStdlibType() const {
430 // Always use libc++ for CHERI purecap
431 // TODO: always use libc++ for MIPS64?
432 if (getTriple().getOSMajorVersion() >= 10 || IsCheriPurecap)
433 return ToolChain::CST_Libcxx;
434 // Always use libc++ for CHERI triples and CHERI subarch:
435 if (getTriple().getArch() == llvm::Triple::cheri)
436 return ToolChain::CST_Libcxx;
437 if (getTriple().isMIPS()) {
438 switch (getTriple().getSubArch()) {
439 case llvm::Triple::MipsSubArch_cheri64:
440 case llvm::Triple::MipsSubArch_cheri128:
441 case llvm::Triple::MipsSubArch_cheri256:
442 return ToolChain::CST_Libcxx;
443 default:
444 break;
445 }
446 }
447 return ToolChain::CST_Libstdcxx;
448}

457void FreeBSD::AddCXXStdlibLibArgs(const ArgList &Args,
458 ArgStringList &CmdArgs) const {
459 CXXStdlibType Type = GetCXXStdlibType(Args);
460 bool Profiling = Args.hasArg(options::OPT_pg);
461
462 switch (Type) {
463 case ToolChain::CST_Libcxx:
464 CmdArgs.push_back(Profiling ? "-lc++_p" : "-lc++");
465 break;
466
467 case ToolChain::CST_Libstdcxx:
468 CmdArgs.push_back(Profiling ? "-lstdc++_p" : "-lstdc++");
469 break;
470 }
471}

479llvm::ExceptionHandling FreeBSD::GetExceptionModel(const ArgList &Args) const {
480 // FreeBSD uses SjLj exceptions on ARM oabi.
481 switch (getTriple().getEnvironment()) {
482 case llvm::Triple::GNUEABIHF:
483 case llvm::Triple::GNUEABI:
484 case llvm::Triple::EABI:
485 return llvm::ExceptionHandling::None;
486 default:
487 if (getTriple().getArch() == llvm::Triple::arm ||
488 getTriple().getArch() == llvm::Triple::thumb)
489 return llvm::ExceptionHandling::SjLj;
490 return llvm::ExceptionHandling::None;
491 }
492}

498SanitizerMask FreeBSD::getSupportedSanitizers() const {
499 const bool IsX86 = getTriple().getArch() == llvm::Triple::x86;
500 const bool IsX86_64 = getTriple().getArch() == llvm::Triple::x86_64;
501 const bool IsMIPS64 = getTriple().isMIPS64();
502 SanitizerMask Res = ToolChain::getSupportedSanitizers();
503 Res |= SanitizerKind::Address;
504 Res |= SanitizerKind::PointerCompare;
505 Res |= SanitizerKind::PointerSubtract;
506 Res |= SanitizerKind::Vptr;
507 if (IsX86_64 || IsMIPS64) {
508 Res |= SanitizerKind::Leak;
509 Res |= SanitizerKind::Thread;
510 }
511 if (IsX86 || IsX86_64) {
512 Res |= SanitizerKind::Function;
513 Res |= SanitizerKind::SafeStack;
514 Res |= SanitizerKind::Fuzzer;
515 Res |= SanitizerKind::FuzzerNoLink;
516 } else if (IsMIPS64) {
517 Res |= SanitizerKind::Fuzzer;
518 Res |= SanitizerKind::FuzzerNoLink;
519 }
520 if (IsX86_64)
521 Res |= SanitizerKind::Memory;
522 return Res;
523}
