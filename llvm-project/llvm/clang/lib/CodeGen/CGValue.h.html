34/// RValue - This trivial value class is used to represent the result of an
35/// expression that is evaluated. It can be one of three things: either a
36/// simple LLVM SSA value, a pair of SSA values for complex numbers, or the
37/// address of an aggregate value in memory.
38class RValue {
39 enum Flavor { Scalar, Complex, Aggregate };
40
41 // The shift to make to an aggregate's alignment to make it look
42 // like a pointer.
43 enum { AggAlignShift = 4 };
44
45 // Stores first value and flavor.
46 llvm::PointerIntPair<llvm::Value *, 2, Flavor> V1;
47 // Stores second value and volatility.
48 llvm::PointerIntPair<llvm::Value *, 1, bool> V2;
49
50public:
51 bool isScalar() const { return V1.getInt() == Scalar; }
52 bool isComplex() const { return V1.getInt() == Complex; }
53 bool isAggregate() const { return V1.getInt() == Aggregate; }
54
55 bool isVolatileQualified() const { return V2.getInt(); }
56
57 /// getScalarVal() - Return the Value* of this scalar value.
58 llvm::Value *getScalarVal() const {
59 assert(isScalar() && "Not a scalar!");
60 return V1.getPointer();
61 }
62
63 unsigned getAlignment() {
64 assert(isScalar() && "Not a scalar!");
65 auto align = reinterpret_cast<uintptr_t>(V2.getPointer());
66 return align >> AggAlignShift;
67 }
68
69 /// getComplexVal - Return the real/imag components of this complex value.
70 ///
71 std::pair<llvm::Value *, llvm::Value *> getComplexVal() const {
72 return std::make_pair(V1.getPointer(), V2.getPointer());
73 }
74
75 /// getAggregateAddr() - Return the Value* of the address of the aggregate.
76 Address getAggregateAddress() const {
77 assert(isAggregate() && "Not an aggregate!");
78 auto align = reinterpret_cast<uintptr_t>(V2.getPointer()) >> AggAlignShift;
79 return Address(V1.getPointer(), CharUnits::fromQuantity(align));
80 }
81 llvm::Value *getAggregatePointer() const {
82 assert(isAggregate() && "Not an aggregate!");
83 return V1.getPointer();
84 }
85
86 static RValue getIgnored() {
87 // FIXME: should we make this a more explicit state?
88 return get(nullptr);
89 }
90
91 static RValue get(llvm::Value *V, unsigned Alignment = 0) {
92 RValue ER;
93 ER.V1.setPointer(V);
94 ER.V1.setInt(Scalar);
95 ER.V2.setPointer(
96 reinterpret_cast<llvm::Value *>(Alignment << AggAlignShift));
97 ER.V2.setInt(false);
98 return ER;
99 }
100 static RValue getComplex(llvm::Value *V1, llvm::Value *V2) {
101 RValue ER;
102 ER.V1.setPointer(V1);
103 ER.V2.setPointer(V2);
104 ER.V1.setInt(Complex);
105 ER.V2.setInt(false);
106 return ER;
107 }
108 static RValue getComplex(const std::pair<llvm::Value *, llvm::Value *> &C) {
109 return getComplex(C.first, C.second);
110 }
111 // FIXME: Aggregate rvalues need to retain information about whether they are
112 // volatile or not. Remove default to find all places that probably get this
113 // wrong.
114 static RValue getAggregate(Address addr, bool isVolatile = false) {
115 RValue ER;
116 ER.V1.setPointer(addr.getPointer());
117 ER.V1.setInt(Aggregate);
118
119 auto align = static_cast<uintptr_t>(addr.getAlignment().getQuantity());
120 ER.V2.setPointer(reinterpret_cast<llvm::Value*>(align << AggAlignShift));
121 ER.V2.setInt(isVolatile);
122 return ER;
123 }
124};

128 ARCImpreciseLifetime, ARCPreciseLifetime

131/// The source of the alignment of an l-value; an expression of
132/// confidence in the alignment actually matching the estimate.
133enum class AlignmentSource {
134 /// The l-value was an access to a declared entity or something
135 /// equivalently strong, like the address of an array allocated by a
136 /// language runtime.
137 Decl,
138
139 /// The l-value was considered opaque, so the alignment was
140 /// determined from a type, but that type was an explicitly-aligned
141 /// typedef.
142 AttributedType,
143
144 /// The l-value was considered opaque, so the alignment was
145 /// determined from a type.
146 Type
147};
148
149/// Given that the base address has the given alignment source, what's
150/// our confidence in the alignment of the field?
151static inline AlignmentSource getFieldAlignmentSource(AlignmentSource Source) {
152 // For now, we don't distinguish fields of opaque pointers from
153 // top-level declarations, but maybe we should.
154 return AlignmentSource::Decl;
155}
156
157class LValueBaseInfo {
158 AlignmentSource AlignSource;
159
160public:
161 explicit LValueBaseInfo(AlignmentSource Source = AlignmentSource::Type)
162 : AlignSource(Source) {}
163 AlignmentSource getAlignmentSource() const { return AlignSource; }
164 void setAlignmentSource(AlignmentSource Source) { AlignSource = Source; }
165
166 void mergeForCast(const LValueBaseInfo &Info) {
167 setAlignmentSource(Info.getAlignmentSource());
168 }

171/// LValue - This represents an lvalue references. Because C/C++ allow
172/// bitfields, this is not a simple LLVM pointer, it may be a pointer plus a
173/// bitrange.
174class LValue {
175 enum {
176 Simple, // This is a normal l-value, use getAddress().
177 VectorElt, // This is a vector element l-value (V[i]), use getVector*
178 BitField, // This is a bitfield l-value, use getBitfield*.
179 ExtVectorElt, // This is an extended vector subset, use getExtVectorComp
180 GlobalReg // This is a register l-value, use getGlobalReg()
181 } LVType;
182
183 llvm::Value *V;
184
185 union {
186 // Index into a vector subscript: V[i]
187 llvm::Value *VectorIdx;
188
189 // ExtVector element subset: V.xyx
190 llvm::Constant *VectorElts;
191
192 // BitField start bit and size
193 const CGBitFieldInfo *BitFieldInfo;
194 };
195
196 QualType Type;
197
198 // 'const' is unused here
199 Qualifiers Quals;
200
201 // The alignment to use when accessing this lvalue. (For vector elements,
202 // this is the alignment of the whole vector.)
203 unsigned Alignment;
204
205 // objective-c's ivar
206 bool Ivar:1;
207
208 // objective-c's ivar is an array
209 bool ObjIsArray:1;
210
211 // LValue is non-gc'able for any reason, including being a parameter or local
212 // variable.
213 bool NonGC: 1;
214
215 // Lvalue is a global reference of an objective-c object
216 bool GlobalObjCRef : 1;
217
218 // Lvalue is a thread local reference
219 bool ThreadLocalRef : 1;
220
221 // Lvalue has ARC imprecise lifetime. We store this inverted to try
222 // to make the default bitfield pattern all-zeroes.
223 bool ImpreciseLifetime : 1;
224
225 // This flag shows if a nontemporal load/stores should be used when accessing
226 // this lvalue.
227 bool Nontemporal : 1;
228
229 LValueBaseInfo BaseInfo;
230 TBAAAccessInfo TBAAInfo;
231
232 Expr *BaseIvarExp;
233
234private:
235 void Initialize(QualType Type, Qualifiers Quals, CharUnits Alignment,
236 LValueBaseInfo BaseInfo, TBAAAccessInfo TBAAInfo) {
237 assert((!Alignment.isZero() || Type->isIncompleteType()) &&
238 "initializing l-value with zero alignment!");
239 this->Type = Type;
240 this->Quals = Quals;
241 const unsigned MaxAlign = 1U << 31;
242 this->Alignment = Alignment.getQuantity() <= MaxAlign
243 ? Alignment.getQuantity()
244 : MaxAlign;
245 assert(this->Alignment == Alignment.getQuantity() &&
246 "Alignment exceeds allowed max!");
247 this->BaseInfo = BaseInfo;
248 this->TBAAInfo = TBAAInfo;
249
250 // Initialize Objective-C flags.
251 this->Ivar = this->ObjIsArray = this->NonGC = this->GlobalObjCRef = false;
252 this->ImpreciseLifetime = false;
253 this->Nontemporal = false;
254 this->ThreadLocalRef = false;
255 this->BaseIvarExp = nullptr;
256 }
257
258public:
259 bool isSimple() const { return LVType == Simple; }
260 bool isVectorElt() const { return LVType == VectorElt; }
261 bool isBitField() const { return LVType == BitField; }
262 bool isExtVectorElt() const { return LVType == ExtVectorElt; }
263 bool isGlobalReg() const { return LVType == GlobalReg; }
264
265 bool isVolatileQualified() const { return Quals.hasVolatile(); }
266 bool isRestrictQualified() const { return Quals.hasRestrict(); }
267 unsigned getVRQualifiers() const {
268 return Quals.getCVRQualifiers() & ~Qualifiers::Const;
269 }
270
271 QualType getType() const { return Type; }
272
273 Qualifiers::ObjCLifetime getObjCLifetime() const {
274 return Quals.getObjCLifetime();
275 }
276
277 bool isObjCIvar() const { return Ivar; }
278 void setObjCIvar(bool Value) { Ivar = Value; }
279
280 bool isObjCArray() const { return ObjIsArray; }
281 void setObjCArray(bool Value) { ObjIsArray = Value; }
282
283 bool isNonGC () const { return NonGC; }
284 void setNonGC(bool Value) { NonGC = Value; }
285
286 bool isGlobalObjCRef() const { return GlobalObjCRef; }
287 void setGlobalObjCRef(bool Value) { GlobalObjCRef = Value; }

374 static LValue MakeAddr(Address address, QualType type, ASTContext &Context,
375 LValueBaseInfo BaseInfo, TBAAAccessInfo TBAAInfo) {
376 Qualifiers qs = type.getQualifiers();
377 qs.setObjCGCAttr(Context.getObjCGCAttrKind(type));
378
379 LValue R;
380 R.LVType = Simple;
381 assert(address.getPointer()->getType()->isPointerTy());
382 R.V = address.getPointer();
383 R.Initialize(type, qs, address.getAlignment(), BaseInfo, TBAAInfo);
384 return R;
385 }
386
387 static LValue MakeVectorElt(Address vecAddress, llvm::Value *Idx,
388 QualType type, LValueBaseInfo BaseInfo,
389 TBAAAccessInfo TBAAInfo) {
390 LValue R;
391 R.LVType = VectorElt;
392 R.V = vecAddress.getPointer();
393 R.VectorIdx = Idx;
394 R.Initialize(type, type.getQualifiers(), vecAddress.getAlignment(),
395 BaseInfo, TBAAInfo);
396 return R;
397 }
398
399 static LValue MakeExtVectorElt(Address vecAddress, llvm::Constant *Elts,
400 QualType type, LValueBaseInfo BaseInfo,
401 TBAAAccessInfo TBAAInfo) {
402 LValue R;
403 R.LVType = ExtVectorElt;
404 R.V = vecAddress.getPointer();
405 R.VectorElts = Elts;
406 R.Initialize(type, type.getQualifiers(), vecAddress.getAlignment(),
407 BaseInfo, TBAAInfo);
408 return R;
409 }

411 /// Create a new object to represent a bit-field access.
412 ///
413 /// \param Addr - The base address of the bit-field sequence this
414 /// bit-field refers to.
415 /// \param Info - The information describing how to perform the bit-field
416 /// access.
417 static LValue MakeBitfield(Address Addr, const CGBitFieldInfo &Info,
418 QualType type, LValueBaseInfo BaseInfo,
419 TBAAAccessInfo TBAAInfo) {
420 LValue R;
421 R.LVType = BitField;
422 R.V = Addr.getPointer();
423 R.BitFieldInfo = &Info;
424 R.Initialize(type, type.getQualifiers(), Addr.getAlignment(), BaseInfo,
425 TBAAInfo);
426 return R;
427 }

429 static LValue MakeGlobalReg(Address Reg, QualType type) {
430 LValue R;
431 R.LVType = GlobalReg;
432 R.V = Reg.getPointer();
433 R.Initialize(type, type.getQualifiers(), Reg.getAlignment(),
434 LValueBaseInfo(AlignmentSource::Decl), TBAAAccessInfo());
435 return R;
436 }

443/// An aggregate value slot.
444class AggValueSlot {
445 /// The address.
446 llvm::Value *Addr;
447
448 // Qualifiers
449 Qualifiers Quals;
450
451 unsigned Alignment;
452
453 /// DestructedFlag - This is set to true if some external code is
454 /// responsible for setting up a destructor for the slot. Otherwise
455 /// the code which constructs it should push the appropriate cleanup.
456 bool DestructedFlag : 1;
457
458 /// ObjCGCFlag - This is set to true if writing to the memory in the
459 /// slot might require calling an appropriate Objective-C GC
460 /// barrier. The exact interaction here is unnecessarily mysterious.
461 bool ObjCGCFlag : 1;
462
463 /// ZeroedFlag - This is set to true if the memory in the slot is
464 /// known to be zero before the assignment into it. This means that
465 /// zero fields don't need to be set.
466 bool ZeroedFlag : 1;
467
468 /// AliasedFlag - This is set to true if the slot might be aliased
469 /// and it's not undefined behavior to access it through such an
470 /// alias. Note that it's always undefined behavior to access a C++
471 /// object that's under construction through an alias derived from
472 /// outside the construction process.
473 ///
474 /// This flag controls whether calls that produce the aggregate
475 /// value may be evaluated directly into the slot, or whether they
476 /// must be evaluated into an unaliased temporary and then memcpy'ed
477 /// over. Since it's invalid in general to memcpy a non-POD C++
478 /// object, it's important that this flag never be set when
479 /// evaluating an expression which constructs such an object.
480 bool AliasedFlag : 1;
481
482 /// This is set to true if the tail padding of this slot might overlap
483 /// another object that may have already been initialized (and whose
484 /// value must be preserved by this initialization). If so, we may only
485 /// store up to the dsize of the type. Otherwise we can widen stores to
486 /// the size of the type.
487 bool OverlapFlag : 1;
488
489 /// If is set to true, sanitizer checks are already generated for this address
490 /// or not required. For instance, if this address represents an object
491 /// created in 'new' expression, sanitizer checks for memory is made as a part
492 /// of 'operator new' emission and object constructor should not generate
493 /// them.
494 bool SanitizerCheckedFlag : 1;
495
496public:
497 enum IsAliased_t { IsNotAliased, IsAliased };
498 enum IsDestructed_t { IsNotDestructed, IsDestructed };
499 enum IsZeroed_t { IsNotZeroed, IsZeroed };
500 enum Overlap_t { DoesNotOverlap, MayOverlap };
501 enum NeedsGCBarriers_t { DoesNotNeedGCBarriers, NeedsGCBarriers };
502 enum IsSanitizerChecked_t { IsNotSanitizerChecked, IsSanitizerChecked };
503
504 /// ignored - Returns an aggregate value slot indicating that the
505 /// aggregate value is being ignored.
506 static AggValueSlot ignored() {
507 return forAddr(Address::invalid(), Qualifiers(), IsNotDestructed,
508 DoesNotNeedGCBarriers, IsNotAliased, DoesNotOverlap);
509 }
510
511 /// forAddr - Make a slot for an aggregate value.
512 ///
513 /// \param quals - The qualifiers that dictate how the slot should
514 /// be initialied. Only 'volatile' and the Objective-C lifetime
515 /// qualifiers matter.
516 ///
517 /// \param isDestructed - true if something else is responsible
518 /// for calling destructors on this object
519 /// \param needsGC - true if the slot is potentially located
520 /// somewhere that ObjC GC calls should be emitted for
521 static AggValueSlot forAddr(Address addr,
522 Qualifiers quals,
523 IsDestructed_t isDestructed,
524 NeedsGCBarriers_t needsGC,
525 IsAliased_t isAliased,
526 Overlap_t mayOverlap,
527 IsZeroed_t isZeroed = IsNotZeroed,
528 IsSanitizerChecked_t isChecked = IsNotSanitizerChecked) {
529 AggValueSlot AV;
530 if (addr.isValid()) {
531 AV.Addr = addr.getPointer();
532 AV.Alignment = addr.getAlignment().getQuantity();
533 } else {
534 AV.Addr = nullptr;
535 AV.Alignment = 0;
536 }
537 AV.Quals = quals;
538 AV.DestructedFlag = isDestructed;
539 AV.ObjCGCFlag = needsGC;
540 AV.ZeroedFlag = isZeroed;
541 AV.AliasedFlag = isAliased;
542 AV.OverlapFlag = mayOverlap;
543 AV.SanitizerCheckedFlag = isChecked;
544 return AV;
545 }
546
547 static AggValueSlot forLValue(const LValue &LV,
548 IsDestructed_t isDestructed,
549 NeedsGCBarriers_t needsGC,
550 IsAliased_t isAliased,
551 Overlap_t mayOverlap,
552 IsZeroed_t isZeroed = IsNotZeroed,
553 IsSanitizerChecked_t isChecked = IsNotSanitizerChecked) {
554 return forAddr(LV.getAddress(), LV.getQuals(), isDestructed, needsGC,
555 isAliased, mayOverlap, isZeroed, isChecked);
556 }
557
558 IsDestructed_t isExternallyDestructed() const {
559 return IsDestructed_t(DestructedFlag);
560 }
561 void setExternallyDestructed(bool destructed = true) {
562 DestructedFlag = destructed;
563 }
564
565 Qualifiers getQualifiers() const { return Quals; }
566
567 bool isVolatile() const {
568 return Quals.hasVolatile();
569 }
570
571 void setVolatile(bool flag) {
572 if (flag)
573 Quals.addVolatile();
574 else
575 Quals.removeVolatile();
576 }
577
578 Qualifiers::ObjCLifetime getObjCLifetime() const {
579 return Quals.getObjCLifetime();
580 }
581
582 NeedsGCBarriers_t requiresGCollection() const {
583 return NeedsGCBarriers_t(ObjCGCFlag);
584 }
585
586 llvm::Value *getPointer() const {
587 return Addr;
588 }
589
590 Address getAddress() const {
591 return Address(Addr, getAlignment());
592 }
593
594 bool isIgnored() const {
595 return Addr == nullptr;
596 }
597
598 CharUnits getAlignment() const {
599 return CharUnits::fromQuantity(Alignment);
600 }
601
602 IsAliased_t isPotentiallyAliased() const {
603 return IsAliased_t(AliasedFlag);
604 }
605
606 Overlap_t mayOverlap() const {
607 return Overlap_t(OverlapFlag);
608 }
609
610 bool isSanitizerChecked() const {
611 return SanitizerCheckedFlag;
612 }
613
614 RValue asRValue() const {
615 if (isIgnored()) {
616 return RValue::getIgnored();
617 } else {
618 return RValue::getAggregate(getAddress(), isVolatile());
619 }
620 }
621
622 void setZeroed(bool V = true) { ZeroedFlag = V; }
623 IsZeroed_t isZeroed() const {
624 return IsZeroed_t(ZeroedFlag);
625 }
626
627 /// Get the preferred size to use when storing a value to this slot. This
628 /// is the type size unless that might overlap another object, in which
629 /// case it's the dsize.
630 CharUnits getPreferredSize(ASTContext &Ctx, QualType Type) const {
631 return mayOverlap() ? Ctx.getTypeInfoDataSizeInChars(Type).first
632 : Ctx.getTypeSizeInChars(Type);
633 }
