49llvm::FunctionCallee CodeGenModule::getTerminateFn() {
50 // void __terminate();
51
52 llvm::FunctionType *FTy =
53 llvm::FunctionType::get(VoidTy, /*IsVarArgs=*/false);
54
55 StringRef name;
56
57 // In C++, use std::terminate().
58 if (getLangOpts().CPlusPlus &&
59 getTarget().getCXXABI().isItaniumFamily()) {
60 name = "_ZSt9terminatev";
61 } else if (getLangOpts().CPlusPlus &&
62 getTarget().getCXXABI().isMicrosoft()) {
63 if (getLangOpts().isCompatibleWithMSVC(LangOptions::MSVC2015))
64 name = "__std_terminate";
65 else
66 name = "?terminate@@YAXXZ";
67 } else if (getLangOpts().ObjC &&
68 getLangOpts().ObjCRuntime.hasTerminate())
69 name = "objc_terminate";
70 else
71 name = "abort";
72 return CreateRuntimeFunction(FTy, name);
73}

129static const EHPersonality &getObjCPersonality(const TargetInfo &Target,
130 const LangOptions &L) {
131 const llvm::Triple &T = Target.getTriple();
132 if (T.isWindowsMSVCEnvironment())
133 return EHPersonality::MSVC_CxxFrameHandler3;
134
135 switch (L.ObjCRuntime.getKind()) {
136 case ObjCRuntime::FragileMacOSX:
137 return getCPersonality(Target, L);
138 case ObjCRuntime::MacOSX:
139 case ObjCRuntime::iOS:
140 case ObjCRuntime::WatchOS:
141 return EHPersonality::NeXT_ObjC;
142 case ObjCRuntime::GNUstep:
143 if (L.ObjCRuntime.getVersion() >= VersionTuple(1, 7))
144 return EHPersonality::GNUstep_ObjC;
145 LLVM_FALLTHROUGH;
146 case ObjCRuntime::GCC:
147 case ObjCRuntime::ObjFW:
148 if (L.SjLjExceptions)
149 return EHPersonality::GNU_ObjC_SJLJ;
150 if (L.SEHExceptions)
151 return EHPersonality::GNU_ObjC_SEH;
152 return EHPersonality::GNU_ObjC;
153 }
154 llvm_unreachable("bad runtime kind");
155}

158 const LangOptions &L) {
159 const llvm::Triple &T = Target.getTriple();
160 if (T.isWindowsMSVCEnvironment())
161 return EHPersonality::MSVC_CxxFrameHandler3;

176/// Determines the personality function to use when both C++
177/// and Objective-C exceptions are being caught.
178static const EHPersonality &getObjCXXPersonality(const TargetInfo &Target,
179 const LangOptions &L) {
180 if (Target.getTriple().isWindowsMSVCEnvironment())
181 return EHPersonality::MSVC_CxxFrameHandler3;
182
183 switch (L.ObjCRuntime.getKind()) {
184 // In the fragile ABI, just use C++ exception handling and hope
185 // they're not doing crazy exception mixing.
186 case ObjCRuntime::FragileMacOSX:
187 return getCXXPersonality(Target, L);
188
189 // The ObjC personality defers to the C++ personality for non-ObjC
190 // handlers. Unlike the C++ case, we use the same personality
191 // function on targets using (backend-driven) SJLJ EH.
192 case ObjCRuntime::MacOSX:
193 case ObjCRuntime::iOS:
194 case ObjCRuntime::WatchOS:
195 return getObjCPersonality(Target, L);
196
197 case ObjCRuntime::GNUstep:
198 return EHPersonality::GNU_ObjCXX;
199
200 // The GCC runtime's personality function inherently doesn't support
201 // mixed EH. Use the ObjC personality just to avoid returning null.
202 case ObjCRuntime::GCC:
203 case ObjCRuntime::ObjFW:
204 return getObjCPersonality(Target, L);
205 }
206 llvm_unreachable("bad runtime kind");
207}

215const EHPersonality &EHPersonality::get(CodeGenModule &CGM,
216 const FunctionDecl *FD) {
217 const llvm::Triple &T = CGM.getTarget().getTriple();
218 const LangOptions &L = CGM.getLangOpts();
219 const TargetInfo &Target = CGM.getTarget();
220
221 // Functions using SEH get an SEH personality.
222 if (FD && FD->usesSEHTry())
223 return getSEHPersonalityMSVC(T);
224
225 if (L.ObjC)
226 return L.CPlusPlus ? getObjCXXPersonality(Target, L)
227 : getObjCPersonality(Target, L);
228 return L.CPlusPlus ? getCXXPersonality(Target, L)
229 : getCPersonality(Target, L);
230}

232const EHPersonality &EHPersonality::get(CodeGenFunction &CGF) {
233 const auto *FD = CGF.CurCodeDecl;
234 // For outlined finallys and filters, use the SEH personality in case they
235 // contain more SEH. This mostly only affects finallys. Filters could
236 // hypothetically use gnu statement expressions to sneak in nested SEH.
237 FD = FD ? FD : CGF.CurSEHParent;
238 return get(CGF.CGM, dyn_cast_or_null<FunctionDecl>(FD));
239}

259/// Check whether a landingpad instruction only uses C++ features.
260static bool LandingPadHasOnlyCXXUses(llvm::LandingPadInst *LPI) {
261 for (unsigned I = 0, E = LPI->getNumClauses(); I != E; ++I) {
262 // Look for something that would've been returned by the ObjC
263 // runtime's GetEHType() method.
264 llvm::Value *Val = LPI->getClause(I)->stripPointerCasts();
265 if (LPI->isCatch(I)) {
266 // Check if the catch value has the ObjC prefix.
267 if (llvm::GlobalVariable *GV = dyn_cast<llvm::GlobalVariable>(Val))
268 // ObjC EH selector entries are always global variables with
269 // names starting like this.
270 if (GV->getName().startswith("OBJC_EHTYPE"))
271 return false;
272 } else {
273 // Check if any of the filter values have the ObjC prefix.
274 llvm::Constant *CVal = cast<llvm::Constant>(Val);
275 for (llvm::User::op_iterator
276 II = CVal->op_begin(), IE = CVal->op_end(); II != IE; ++II) {
277 if (llvm::GlobalVariable *GV =
278 cast<llvm::GlobalVariable>((*II)->stripPointerCasts()))
279 // ObjC EH selector entries are always global variables with
280 // names starting like this.
281 if (GV->getName().startswith("OBJC_EHTYPE"))
282 return false;
283 }
284 }
285 }
286 return true;
287}

289/// Check whether a personality function could reasonably be swapped
290/// for a C++ personality function.
291static bool PersonalityHasOnlyCXXUses(llvm::Constant *Fn) {
292 for (llvm::User *U : Fn->users()) {
293 // Conditionally white-list bitcasts.
294 if (llvm::ConstantExpr *CE = dyn_cast<llvm::ConstantExpr>(U)) {
295 if (CE->getOpcode() != llvm::Instruction::BitCast) return false;
296 if (!PersonalityHasOnlyCXXUses(CE))
297 return false;
298 continue;
299 }
300
301 // Otherwise it must be a function.
302 llvm::Function *F = dyn_cast<llvm::Function>(U);
303 if (!F) return false;
304
305 for (auto BB = F->begin(), E = F->end(); BB != E; ++BB) {
306 if (BB->isLandingPad())
307 if (!LandingPadHasOnlyCXXUses(BB->getLandingPadInst()))
308 return false;
309 }
310 }
311
312 return true;
313}

315/// Try to use the C++ personality function in ObjC++. Not doing this
316/// can cause some incompatibilities with gcc, which is more
317/// aggressive about only using the ObjC++ personality in a function
318/// when it really needs it.
319void CodeGenModule::SimplifyPersonality() {
320 // If we're not in ObjC++ -fexceptions, there's nothing to do.
321 if (!LangOpts.CPlusPlus || !LangOpts.ObjC || !LangOpts.Exceptions)
322 return;
323
324 // Both the problem this endeavors to fix and the way the logic
325 // above works is specific to the NeXT runtime.
326 if (!LangOpts.ObjCRuntime.isNeXTFamily())
327 return;
328
329 const EHPersonality &ObjCXX = EHPersonality::get(*this, /*FD=*/nullptr);
330 const EHPersonality &CXX = getCXXPersonality(getTarget(), LangOpts);
331 if (&ObjCXX == &CXX)
332 return;
333
334 assert(std::strcmp(ObjCXX.PersonalityFn, CXX.PersonalityFn) != 0 &&
335 "Different EHPersonalities using the same personality function.");
336
337 llvm::Function *Fn = getModule().getFunction(ObjCXX.PersonalityFn);
338
339 // Nothing to do if it's unused.
340 if (!Fn || Fn->use_empty()) return;
341
342 // Can't do the optimization if it has non-C++ uses.
343 if (!PersonalityHasOnlyCXXUses(Fn)) return;
344
345 // Create the C++ personality function and kill off the old
346 // function.
347 llvm::FunctionCallee CXXFn = getPersonalityFn(*this, CXX);
348
349 // This can happen if the user is screwing with us.
350 if (Fn->getType() != CXXFn.getCallee()->getType())
351 return;
352
353 Fn->replaceAllUsesWith(CXXFn.getCallee());
354 Fn->eraseFromParent();
355}

376// Emits an exception expression into the given location. This
377// differs from EmitAnyExprToMem only in that, if a final copy-ctor
378// call is required, an exception within that copy ctor causes
379// std::terminate to be invoked.
380void CodeGenFunction::EmitAnyExprToExn(const Expr *e, Address addr) {
381 // Make sure the exception object is cleaned up if there's an
382 // exception during initialization.
383 pushFullExprCleanup<FreeException>(EHCleanup, addr.getPointer());
384 EHScopeStack::stable_iterator cleanup = EHStack.stable_begin();
385
386 // __cxa_allocate_exception returns a void*; we need to cast this
387 // to the appropriate type for the object.
388 unsigned DefaultAS = CGM.getTargetCodeGenInfo().getDefaultAS();
389 llvm::Type *ty = ConvertTypeForMem(e->getType())->getPointerTo(DefaultAS);
390 Address typedAddr = Builder.CreateBitCast(addr, ty);
391
392 // FIXME: this isn't quite right! If there's a final unelided call
393 // to a copy constructor, then according to [except.terminate]p1 we
394 // must call std::terminate() if that constructor throws, because
395 // technically that copy occurs after the exception expression is
396 // evaluated but before the exception is caught. But the best way
397 // to handle that is to teach EmitAggExpr to do the final copy
398 // differently if it can't be elided.
399 EmitAnyExprToMem(e, typedAddr, e->getType().getQualifiers(),
400 /*IsInit*/ true);
401
402 // Deactivate the cleanup block.
403 DeactivateCleanupBlock(cleanup,
404 cast<llvm::Instruction>(typedAddr.getPointer()));
405}

427void CodeGenFunction::EmitCXXThrowExpr(const CXXThrowExpr *E,
428 bool KeepInsertionPoint) {
429 if (const Expr *SubExpr = E->getSubExpr()) {
430 QualType ThrowType = SubExpr->getType();
431 if (ThrowType->isObjCObjectPointerType()) {
432 const Stmt *ThrowStmt = E->getSubExpr();
433 const ObjCAtThrowStmt S(E->getExprLoc(), const_cast<Stmt *>(ThrowStmt));
434 CGM.getObjCRuntime().EmitThrowStmt(*this, S, false);
435 } else {
436 CGM.getCXXABI().emitThrow(*this, E);
437 }
438 } else {
439 CGM.getCXXABI().emitRethrow(*this, /*isNoReturn=*/true);
440 }
441
442 // throw is an expression, and the expression emitters expect us
443 // to leave ourselves at a valid insertion point.
444 if (KeepInsertionPoint)
445 EmitBlock(createBasicBlock("throw.cont"));
446}

448void CodeGenFunction::EmitStartEHSpec(const Decl *D) {
449 if (!CGM.getLangOpts().CXXExceptions)
450 return;
451
452 const FunctionDecl* FD = dyn_cast_or_null<FunctionDecl>(D);
453 if (!FD) {
454 // Check if CapturedDecl is nothrow and create terminate scope for it.
455 if (const CapturedDecl* CD = dyn_cast_or_null<CapturedDecl>(D)) {
456 if (CD->isNothrow())
457 EHStack.pushTerminate();
458 }
459 return;
460 }
461 const FunctionProtoType *Proto = FD->getType()->getAs<FunctionProtoType>();
462 if (!Proto)
463 return;
464
465 ExceptionSpecificationType EST = Proto->getExceptionSpecType();
466 if (isNoexceptExceptionSpec(EST) && Proto->canThrow() == CT_Cannot) {
467 // noexcept functions are simple terminate scopes.
468 EHStack.pushTerminate();
469 } else if (EST == EST_Dynamic || EST == EST_DynamicNone) {
470 // TODO: Revisit exception specifications for the MS ABI. There is a way to
471 // encode these in an object file but MSVC doesn't do anything with it.
472 if (getTarget().getCXXABI().isMicrosoft())
473 return;
474 unsigned NumExceptions = Proto->getNumExceptions();
475 EHFilterScope *Filter = EHStack.pushFilter(NumExceptions);
476
477 for (unsigned I = 0; I != NumExceptions; ++I) {
478 QualType Ty = Proto->getExceptionType(I);
479 QualType ExceptType = Ty.getNonReferenceType().getUnqualifiedType();
480 llvm::Value *EHType = CGM.GetAddrOfRTTIDescriptor(ExceptType,
481 /*ForEH=*/true);
482 Filter->setFilter(I, EHType);
483 }
484 }
485}

487/// Emit the dispatch block for a filter scope if necessary.
488static void emitFilterDispatchBlock(CodeGenFunction &CGF,
489 EHFilterScope &filterScope) {
490 llvm::BasicBlock *dispatchBlock = filterScope.getCachedEHDispatchBlock();
491 if (!dispatchBlock) return;
492 if (dispatchBlock->use_empty()) {
493 delete dispatchBlock;
494 return;
495 }
496
497 CGF.EmitBlockAfterUses(dispatchBlock);
498
499 // If this isn't a catch-all filter, we need to check whether we got
500 // here because the filter triggered.
501 if (filterScope.getNumFilters()) {
502 // Load the selector value.
503 llvm::Value *selector = CGF.getSelectorFromSlot();
504 llvm::BasicBlock *unexpectedBB = CGF.createBasicBlock("ehspec.unexpected");
505
506 llvm::Value *zero = CGF.Builder.getInt32(0);
507 llvm::Value *failsFilter =
508 CGF.Builder.CreateICmpSLT(selector, zero, "ehspec.fails");
509 CGF.Builder.CreateCondBr(failsFilter, unexpectedBB,
510 CGF.getEHResumeBlock(false));
511
512 CGF.EmitBlock(unexpectedBB);
513 }
514
515 // Call __cxa_call_unexpected. This doesn't need to be an invoke
516 // because __cxa_call_unexpected magically filters exceptions
517 // according to the last landing pad the exception was thrown
518 // into. Seriously.
519 llvm::Value *exn = CGF.getExceptionFromSlot();
520 CGF.EmitRuntimeCall(getUnexpectedFn(CGF.CGM), exn)
521 ->setDoesNotReturn();
522 CGF.Builder.CreateUnreachable();
523}

525void CodeGenFunction::EmitEndEHSpec(const Decl *D) {
526 if (!CGM.getLangOpts().CXXExceptions)
527 return;
528
529 const FunctionDecl* FD = dyn_cast_or_null<FunctionDecl>(D);
530 if (!FD) {
531 // Check if CapturedDecl is nothrow and pop terminate scope for it.
532 if (const CapturedDecl* CD = dyn_cast_or_null<CapturedDecl>(D)) {
533 if (CD->isNothrow())
534 EHStack.popTerminate();
535 }
536 return;
537 }
538 const FunctionProtoType *Proto = FD->getType()->getAs<FunctionProtoType>();
539 if (!Proto)
540 return;
541
542 ExceptionSpecificationType EST = Proto->getExceptionSpecType();
543 if (isNoexceptExceptionSpec(EST) && Proto->canThrow() == CT_Cannot) {
544 EHStack.popTerminate();
545 } else if (EST == EST_Dynamic || EST == EST_DynamicNone) {
546 // TODO: Revisit exception specifications for the MS ABI. There is a way to
547 // encode these in an object file but MSVC doesn't do anything with it.
548 if (getTarget().getCXXABI().isMicrosoft())
549 return;
550 EHFilterScope &filterScope = cast<EHFilterScope>(*EHStack.begin());
551 emitFilterDispatchBlock(*this, filterScope);
552 EHStack.popFilter();
553 }

562void CodeGenFunction::EnterCXXTryStmt(const CXXTryStmt &S, bool IsFnTryBlock) {
563 unsigned NumHandlers = S.getNumHandlers();
564 EHCatchScope *CatchScope = EHStack.pushCatch(NumHandlers);
565
566 for (unsigned I = 0; I != NumHandlers; ++I) {
567 const CXXCatchStmt *C = S.getHandler(I);
568
569 llvm::BasicBlock *Handler = createBasicBlock("catch");
570 if (C->getExceptionDecl()) {
571 // FIXME: Dropping the reference type on the type into makes it
572 // impossible to correctly implement catch-by-reference
573 // semantics for pointers. Unfortunately, this is what all
574 // existing compilers do, and it's not clear that the standard
575 // personality routine is capable of doing this right. See C++ DR 388:
576 // http://www.open-std.org/jtc1/sc22/wg21/docs/cwg_active.html#388
577 Qualifiers CaughtTypeQuals;
578 QualType CaughtType = CGM.getContext().getUnqualifiedArrayType(
579 C->getCaughtType().getNonReferenceType(), CaughtTypeQuals);
580
581 CatchTypeInfo TypeInfo{nullptr, 0};
582 if (CaughtType->isObjCObjectPointerType()) {
583 TypeInfo.RTTI = CGM.getObjCRuntime().GetEHType(CaughtType);
584 TypeInfo.RTTI =
585 llvm::ConstantExpr::getPointerBitCastOrAddrSpaceCast(TypeInfo.RTTI,
586 cast<llvm::PointerType>(TypeInfo.RTTI->getType())
587 ->getElementType()->getPointerTo(0));
588 } else
589 TypeInfo = CGM.getCXXABI().getAddrOfCXXCatchHandlerType(
590 CaughtType, C->getCaughtType());
591 CatchScope->setHandler(I, TypeInfo, Handler);
592 } else {
593 // No exception decl indicates '...', a catch-all.
594 CatchScope->setHandler(I, CGM.getCXXABI().getCatchAllTypeInfo(), Handler);
595 }
596 }

599llvm::BasicBlock *
600CodeGenFunction::getEHDispatchBlock(EHScopeStack::stable_iterator si) {
601 if (EHPersonality::get(*this).usesFuncletPads())
602 return getFuncletEHDispatchBlock(si);
603
604 // The dispatch block for the end of the scope chain is a block that
605 // just resumes unwinding.
606 if (si == EHStack.stable_end())
607 return getEHResumeBlock(true);
608
609 // Otherwise, we should look at the actual scope.
610 EHScope &scope = *EHStack.find(si);
611
612 llvm::BasicBlock *dispatchBlock = scope.getCachedEHDispatchBlock();
613 if (!dispatchBlock) {
614 switch (scope.getKind()) {
615 case EHScope::Catch: {
616 // Apply a special case to a single catch-all.
617 EHCatchScope &catchScope = cast<EHCatchScope>(scope);
618 if (catchScope.getNumHandlers() == 1 &&
619 catchScope.getHandler(0).isCatchAll()) {
620 dispatchBlock = catchScope.getHandler(0).Block;
621
622 // Otherwise, make a dispatch block.
623 } else {
624 dispatchBlock = createBasicBlock("catch.dispatch");
625 }
626 break;
627 }
628
629 case EHScope::Cleanup:
630 dispatchBlock = createBasicBlock("ehcleanup");
631 break;
632
633 case EHScope::Filter:
634 dispatchBlock = createBasicBlock("filter.dispatch");
635 break;
636
637 case EHScope::Terminate:
638 dispatchBlock = getTerminateHandler();
639 break;
640
641 case EHScope::PadEnd:
642 llvm_unreachable("PadEnd unnecessary for Itanium!");
643 }
644 scope.setCachedEHDispatchBlock(dispatchBlock);
645 }
646 return dispatchBlock;
647}

649llvm::BasicBlock *
650CodeGenFunction::getFuncletEHDispatchBlock(EHScopeStack::stable_iterator SI) {
651 // Returning nullptr indicates that the previous dispatch block should unwind
652 // to caller.
653 if (SI == EHStack.stable_end())
654 return nullptr;
655
656 // Otherwise, we should look at the actual scope.
657 EHScope &EHS = *EHStack.find(SI);
658
659 llvm::BasicBlock *DispatchBlock = EHS.getCachedEHDispatchBlock();
660 if (DispatchBlock)
661 return DispatchBlock;
662
663 if (EHS.getKind() == EHScope::Terminate)
664 DispatchBlock = getTerminateFunclet();
665 else
666 DispatchBlock = createBasicBlock();
667 CGBuilderTy Builder(*this, DispatchBlock);
668
669 switch (EHS.getKind()) {
670 case EHScope::Catch:
671 DispatchBlock->setName("catch.dispatch");
672 break;
673
674 case EHScope::Cleanup:
675 DispatchBlock->setName("ehcleanup");
676 break;
677
678 case EHScope::Filter:
679 llvm_unreachable("exception specifications not handled yet!");
680
681 case EHScope::Terminate:
682 DispatchBlock->setName("terminate");
683 break;
684
685 case EHScope::PadEnd:
686 llvm_unreachable("PadEnd dispatch block missing!");
687 }
688 EHS.setCachedEHDispatchBlock(DispatchBlock);
689 return DispatchBlock;
690}

698 return !cast<EHCleanupScope>(S).isEHCleanup();
699 case EHScope::Filter:

709llvm::BasicBlock *CodeGenFunction::getInvokeDestImpl() {
710 assert(EHStack.requiresLandingPad());
711 assert(!EHStack.empty());
712
713 // If exceptions are disabled and SEH is not in use, then there is no invoke
714 // destination. SEH "works" even if exceptions are off. In practice, this
715 // means that C++ destructors and other EH cleanups don't run, which is
716 // consistent with MSVC's behavior.
717 const LangOptions &LO = CGM.getLangOpts();
718 if (!LO.Exceptions) {
719 if (!LO.Borland && !LO.MicrosoftExt)
720 return nullptr;
721 if (!currentFunctionUsesSEHTry())
722 return nullptr;
723 }
724
725 // CUDA device code doesn't have exceptions.
726 if (LO.CUDA && LO.CUDAIsDevice)
727 return nullptr;
728
729 // Check the innermost scope for a cached landing pad. If this is
730 // a non-EH cleanup, we'll check enclosing scopes in EmitLandingPad.
731 llvm::BasicBlock *LP = EHStack.begin()->getCachedLandingPad();
732 if (LP) return LP;
733
734 const EHPersonality &Personality = EHPersonality::get(*this);
735
736 if (!CurFn->hasPersonalityFn())
737 CurFn->setPersonalityFn(getOpaquePersonalityFn(CGM, Personality));
738
739 if (Personality.usesFuncletPads()) {
740 // We don't need separate landing pads in the funclet model.
741 LP = getEHDispatchBlock(EHStack.getInnermostEHScope());
742 } else {
743 // Build the landing pad for this scope.
744 LP = EmitLandingPad();
745 }
746
747 assert(LP);
748
749 // Cache the landing pad on the innermost scope. If this is a
750 // non-EH scope, cache the landing pad on the enclosing scope, too.
751 for (EHScopeStack::iterator ir = EHStack.begin(); true; ++ir) {
752 ir->setCachedLandingPad(LP);
753 if (!isNonEHScope(*ir)) break;
754 }
755
756 return LP;
757}

759llvm::BasicBlock *CodeGenFunction::EmitLandingPad() {
760 assert(EHStack.requiresLandingPad());
761
762 EHScope &innermostEHScope = *EHStack.find(EHStack.getInnermostEHScope());
763 switch (innermostEHScope.getKind()) {
764 case EHScope::Terminate:
765 return getTerminateLandingPad();
766
767 case EHScope::PadEnd:
768 llvm_unreachable("PadEnd unnecessary for Itanium!");
769
770 case EHScope::Catch:
771 case EHScope::Cleanup:
772 case EHScope::Filter:
773 if (llvm::BasicBlock *lpad = innermostEHScope.getCachedLandingPad())
774 return lpad;
775 }
776
777 // Save the current IR generation state.
778 CGBuilderTy::InsertPoint savedIP = Builder.saveAndClearIP();
779 auto DL = ApplyDebugLocation::CreateDefaultArtificial(*this, CurEHLocation);
780
781 // Create and configure the landing pad.
782 llvm::BasicBlock *lpad = createBasicBlock("lpad");
783 EmitBlock(lpad);
784
785 llvm::LandingPadInst *LPadInst =
786 Builder.CreateLandingPad(llvm::StructType::get(Int8PtrTy, Int32Ty), 0);
787
788 llvm::Value *LPadExn = Builder.CreateExtractValue(LPadInst, 0);
789 Builder.CreateStore(LPadExn, getExceptionSlot());
790 llvm::Value *LPadSel = Builder.CreateExtractValue(LPadInst, 1);
791 Builder.CreateStore(LPadSel, getEHSelectorSlot());
792
793 // Save the exception pointer. It's safe to use a single exception
794 // pointer per function because EH cleanups can never have nested
795 // try/catches.
796 // Build the landingpad instruction.
797
798 // Accumulate all the handlers in scope.
799 bool hasCatchAll = false;
800 bool hasCleanup = false;
801 bool hasFilter = false;
802 SmallVector<llvm::Value*, 4> filterTypes;
803 llvm::SmallPtrSet<llvm::Value*, 4> catchTypes;
804 for (EHScopeStack::iterator I = EHStack.begin(), E = EHStack.end(); I != E;
805 ++I) {
806
807 switch (I->getKind()) {
808 case EHScope::Cleanup:
809 // If we have a cleanup, remember that.
810 hasCleanup = (hasCleanup || cast<EHCleanupScope>(*I).isEHCleanup());
811 continue;
812
813 case EHScope::Filter: {
814 assert(I.next() == EHStack.end() && "EH filter is not end of EH stack");
815 assert(!hasCatchAll && "EH filter reached after catch-all");
816
817 // Filter scopes get added to the landingpad in weird ways.
818 EHFilterScope &filter = cast<EHFilterScope>(*I);
819 hasFilter = true;
820
821 // Add all the filter values.
822 for (unsigned i = 0, e = filter.getNumFilters(); i != e; ++i)
823 filterTypes.push_back(filter.getFilter(i));
824 goto done;
825 }
826
827 case EHScope::Terminate:
828 // Terminate scopes are basically catch-alls.
829 assert(!hasCatchAll);
830 hasCatchAll = true;
831 goto done;
832
833 case EHScope::Catch:
834 break;
835
836 case EHScope::PadEnd:
837 llvm_unreachable("PadEnd unnecessary for Itanium!");
838 }
839
840 EHCatchScope &catchScope = cast<EHCatchScope>(*I);
841 for (unsigned hi = 0, he = catchScope.getNumHandlers(); hi != he; ++hi) {
842 EHCatchScope::Handler handler = catchScope.getHandler(hi);
843 assert(handler.Type.Flags == 0 &&
844 "landingpads do not support catch handler flags");
845
846 // If this is a catch-all, register that and abort.
847 if (!handler.Type.RTTI) {
848 assert(!hasCatchAll);
849 hasCatchAll = true;
850 goto done;
851 }
852
853 // Check whether we already have a handler for this type.
854 if (catchTypes.insert(handler.Type.RTTI).second)
855 // If not, add it directly to the landingpad.
856 LPadInst->addClause(handler.Type.RTTI);
857 }
858 }
859
860 done:
861 // If we have a catch-all, add null to the landingpad.
862 assert(!(hasCatchAll && hasFilter));
863 if (hasCatchAll) {
864 LPadInst->addClause(getCatchAllValue(*this));
865
866 // If we have an EH filter, we need to add those handlers in the
867 // right place in the landingpad, which is to say, at the end.
868 } else if (hasFilter) {
869 // Create a filter expression: a constant array indicating which filter
870 // types there are. The personality routine only lands here if the filter
871 // doesn't match.
872 SmallVector<llvm::Constant*, 8> Filters;
873 llvm::ArrayType *AType =
874 llvm::ArrayType::get(!filterTypes.empty() ?
875 filterTypes[0]->getType() : Int8PtrTy,
876 filterTypes.size());
877
878 for (unsigned i = 0, e = filterTypes.size(); i != e; ++i)
879 Filters.push_back(cast<llvm::Constant>(filterTypes[i]));
880 llvm::Constant *FilterArray = llvm::ConstantArray::get(AType, Filters);
881 LPadInst->addClause(FilterArray);
882
883 // Also check whether we need a cleanup.
884 if (hasCleanup)
885 LPadInst->setCleanup(true);
886
887 // Otherwise, signal that we at least have cleanups.
888 } else if (hasCleanup) {
889 LPadInst->setCleanup(true);
890 }
891
892 assert((LPadInst->getNumClauses() > 0 || LPadInst->isCleanup()) &&
893 "landingpad instruction has no clauses!");
894
895 // Tell the backend how to generate the landing pad.
896 Builder.CreateBr(getEHDispatchBlock(EHStack.getInnermostEHScope()));
897
898 // Restore the old IR generation state.
899 Builder.restoreIP(savedIP);
900
901 return lpad;
902}

904static void emitCatchPadBlock(CodeGenFunction &CGF, EHCatchScope &CatchScope) {
905 llvm::BasicBlock *DispatchBlock = CatchScope.getCachedEHDispatchBlock();
906 assert(DispatchBlock);
907
908 CGBuilderTy::InsertPoint SavedIP = CGF.Builder.saveIP();
909 CGF.EmitBlockAfterUses(DispatchBlock);
910
911 llvm::Value *ParentPad = CGF.CurrentFuncletPad;
912 if (!ParentPad)
913 ParentPad = llvm::ConstantTokenNone::get(CGF.getLLVMContext());
914 llvm::BasicBlock *UnwindBB =
915 CGF.getEHDispatchBlock(CatchScope.getEnclosingEHScope());
916
917 unsigned NumHandlers = CatchScope.getNumHandlers();
918 llvm::CatchSwitchInst *CatchSwitch =
919 CGF.Builder.CreateCatchSwitch(ParentPad, UnwindBB, NumHandlers);
920
921 // Test against each of the exception types we claim to catch.
922 for (unsigned I = 0; I < NumHandlers; ++I) {
923 const EHCatchScope::Handler &Handler = CatchScope.getHandler(I);
924
925 CatchTypeInfo TypeInfo = Handler.Type;
926 if (!TypeInfo.RTTI)
927 TypeInfo.RTTI = llvm::Constant::getNullValue(CGF.VoidPtrTy);
928
929 CGF.Builder.SetInsertPoint(Handler.Block);
930
931 if (EHPersonality::get(CGF).isMSVCXXPersonality()) {
932 CGF.Builder.CreateCatchPad(
933 CatchSwitch, {TypeInfo.RTTI, CGF.Builder.getInt32(TypeInfo.Flags),
934 llvm::Constant::getNullValue(CGF.VoidPtrTy)});
935 } else {
936 CGF.Builder.CreateCatchPad(CatchSwitch, {TypeInfo.RTTI});
937 }
938
939 CatchSwitch->addHandler(Handler.Block);
940 }
941 CGF.Builder.restoreIP(SavedIP);
942}

944// Wasm uses Windows-style EH instructions, but it merges all catch clauses into
945// one big catchpad, within which we use Itanium's landingpad-style selector
946// comparison instructions.
947static void emitWasmCatchPadBlock(CodeGenFunction &CGF,
948 EHCatchScope &CatchScope) {
949 llvm::BasicBlock *DispatchBlock = CatchScope.getCachedEHDispatchBlock();
950 assert(DispatchBlock);
951
952 CGBuilderTy::InsertPoint SavedIP = CGF.Builder.saveIP();
953 CGF.EmitBlockAfterUses(DispatchBlock);
954
955 llvm::Value *ParentPad = CGF.CurrentFuncletPad;
956 if (!ParentPad)
957 ParentPad = llvm::ConstantTokenNone::get(CGF.getLLVMContext());
958 llvm::BasicBlock *UnwindBB =
959 CGF.getEHDispatchBlock(CatchScope.getEnclosingEHScope());
960
961 unsigned NumHandlers = CatchScope.getNumHandlers();
962 llvm::CatchSwitchInst *CatchSwitch =
963 CGF.Builder.CreateCatchSwitch(ParentPad, UnwindBB, NumHandlers);
964
965 // We don't use a landingpad instruction, so generate intrinsic calls to
966 // provide exception and selector values.
967 llvm::BasicBlock *WasmCatchStartBlock = CGF.createBasicBlock("catch.start");
968 CatchSwitch->addHandler(WasmCatchStartBlock);
969 CGF.EmitBlockAfterUses(WasmCatchStartBlock);
970
971 // Create a catchpad instruction.
972 SmallVector<llvm::Value *, 4> CatchTypes;
973 for (unsigned I = 0, E = NumHandlers; I < E; ++I) {
974 const EHCatchScope::Handler &Handler = CatchScope.getHandler(I);
975 CatchTypeInfo TypeInfo = Handler.Type;
976 if (!TypeInfo.RTTI)
977 TypeInfo.RTTI = llvm::Constant::getNullValue(CGF.VoidPtrTy);
978 CatchTypes.push_back(TypeInfo.RTTI);
979 }
980 auto *CPI = CGF.Builder.CreateCatchPad(CatchSwitch, CatchTypes);
981
982 // Create calls to wasm.get.exception and wasm.get.ehselector intrinsics.
983 // Before they are lowered appropriately later, they provide values for the
984 // exception and selector.
985 llvm::Function *GetExnFn =
986 CGF.CGM.getIntrinsic(llvm::Intrinsic::wasm_get_exception);
987 llvm::Function *GetSelectorFn =
988 CGF.CGM.getIntrinsic(llvm::Intrinsic::wasm_get_ehselector);
989 llvm::CallInst *Exn = CGF.Builder.CreateCall(GetExnFn, CPI);
990 CGF.Builder.CreateStore(Exn, CGF.getExceptionSlot());
991 llvm::CallInst *Selector = CGF.Builder.CreateCall(GetSelectorFn, CPI);
992
993 llvm::Function *TypeIDFn = CGF.CGM.getIntrinsic(llvm::Intrinsic::eh_typeid_for);
994
995 // If there's only a single catch-all, branch directly to its handler.
996 if (CatchScope.getNumHandlers() == 1 &&
997 CatchScope.getHandler(0).isCatchAll()) {
998 CGF.Builder.CreateBr(CatchScope.getHandler(0).Block);
999 CGF.Builder.restoreIP(SavedIP);
1000 return;
1001 }
1002
1003 // Test against each of the exception types we claim to catch.
1004 for (unsigned I = 0, E = NumHandlers;; ++I) {
1005 assert(I < E && "ran off end of handlers!");
1006 const EHCatchScope::Handler &Handler = CatchScope.getHandler(I);
1007 CatchTypeInfo TypeInfo = Handler.Type;
1008 if (!TypeInfo.RTTI)
1009 TypeInfo.RTTI = llvm::Constant::getNullValue(CGF.VoidPtrTy);
1010
1011 // Figure out the next block.
1012 llvm::BasicBlock *NextBlock;
1013
1014 bool EmitNextBlock = false, NextIsEnd = false;
1015
1016 // If this is the last handler, we're at the end, and the next block is a
1017 // block that contains a call to the rethrow function, so we can unwind to
1018 // the enclosing EH scope. The call itself will be generated later.
1019 if (I + 1 == E) {
1020 NextBlock = CGF.createBasicBlock("rethrow");
1021 EmitNextBlock = true;
1022 NextIsEnd = true;
1023
1024 // If the next handler is a catch-all, we're at the end, and the
1025 // next block is that handler.
1026 } else if (CatchScope.getHandler(I + 1).isCatchAll()) {
1027 NextBlock = CatchScope.getHandler(I + 1).Block;
1028 NextIsEnd = true;
1029
1030 // Otherwise, we're not at the end and we need a new block.
1031 } else {
1032 NextBlock = CGF.createBasicBlock("catch.fallthrough");
1033 EmitNextBlock = true;
1034 }
1035
1036 // Figure out the catch type's index in the LSDA's type table.
1037 llvm::CallInst *TypeIndex = CGF.Builder.CreateCall(TypeIDFn, TypeInfo.RTTI);
1038 TypeIndex->setDoesNotThrow();
1039
1040 llvm::Value *MatchesTypeIndex =
1041 CGF.Builder.CreateICmpEQ(Selector, TypeIndex, "matches");
1042 CGF.Builder.CreateCondBr(MatchesTypeIndex, Handler.Block, NextBlock);
1043
1044 if (EmitNextBlock)
1045 CGF.EmitBlock(NextBlock);
1046 if (NextIsEnd)
1047 break;
1048 }
1049
1050 CGF.Builder.restoreIP(SavedIP);
1051}

1053/// Emit the structure of the dispatch block for the given catch scope.
1054/// It is an invariant that the dispatch block already exists.
1055static void emitCatchDispatchBlock(CodeGenFunction &CGF,
1056 EHCatchScope &catchScope) {
1057 if (EHPersonality::get(CGF).isWasmPersonality())
1058 return emitWasmCatchPadBlock(CGF, catchScope);
1059 if (EHPersonality::get(CGF).usesFuncletPads())
1060 return emitCatchPadBlock(CGF, catchScope);
1061
1062 llvm::BasicBlock *dispatchBlock = catchScope.getCachedEHDispatchBlock();
1063 assert(dispatchBlock);
1064
1065 // If there's only a single catch-all, getEHDispatchBlock returned
1066 // that catch-all as the dispatch block.
1067 if (catchScope.getNumHandlers() == 1 &&
1068 catchScope.getHandler(0).isCatchAll()) {
1069 assert(dispatchBlock == catchScope.getHandler(0).Block);
1070 return;
1071 }
1072
1073 CGBuilderTy::InsertPoint savedIP = CGF.Builder.saveIP();
1074 CGF.EmitBlockAfterUses(dispatchBlock);
1075
1076 // Select the right handler.
1077 llvm::Function *llvm_eh_typeid_for =
1078 CGF.CGM.getIntrinsic(llvm::Intrinsic::eh_typeid_for);
1079
1080 // Load the selector value.
1081 llvm::Value *selector = CGF.getSelectorFromSlot();
1082
1083 // Test against each of the exception types we claim to catch.
1084 for (unsigned i = 0, e = catchScope.getNumHandlers(); ; ++i) {
1085 assert(i < e && "ran off end of handlers!");
1086 const EHCatchScope::Handler &handler = catchScope.getHandler(i);
1087
1088 llvm::Value *typeValue = handler.Type.RTTI;
1089 assert(handler.Type.Flags == 0 &&
1090 "landingpads do not support catch handler flags");
1091 assert(typeValue && "fell into catch-all case!");
1092 typeValue = CGF.Builder.CreatePointerBitCastOrAddrSpaceCast(typeValue,
1093 CGF.Int8Ty->getPointerTo(0));
1094
1095 // Figure out the next block.
1096 bool nextIsEnd;
1097 llvm::BasicBlock *nextBlock;
1098
1099 // If this is the last handler, we're at the end, and the next
1100 // block is the block for the enclosing EH scope.
1101 if (i + 1 == e) {
1102 nextBlock = CGF.getEHDispatchBlock(catchScope.getEnclosingEHScope());
1103 nextIsEnd = true;
1104
1105 // If the next handler is a catch-all, we're at the end, and the
1106 // next block is that handler.
1107 } else if (catchScope.getHandler(i+1).isCatchAll()) {
1108 nextBlock = catchScope.getHandler(i+1).Block;
1109 nextIsEnd = true;
1110
1111 // Otherwise, we're not at the end and we need a new block.
1112 } else {
1113 nextBlock = CGF.createBasicBlock("catch.fallthrough");
1114 nextIsEnd = false;
1115 }
1116
1117 // Figure out the catch type's index in the LSDA's type table.
1118 llvm::CallInst *typeIndex =
1119 CGF.Builder.CreateCall(llvm_eh_typeid_for, typeValue);
1120 typeIndex->setDoesNotThrow();
1121
1122 llvm::Value *matchesTypeIndex =
1123 CGF.Builder.CreateICmpEQ(selector, typeIndex, "matches");
1124 CGF.Builder.CreateCondBr(matchesTypeIndex, handler.Block, nextBlock);
1125
1126 // If the next handler is a catch-all, we're completely done.
1127 if (nextIsEnd) {
1128 CGF.Builder.restoreIP(savedIP);
1129 return;
1130 }
1131 // Otherwise we need to emit and continue at that block.
1132 CGF.EmitBlock(nextBlock);
1133 }

1143void CodeGenFunction::ExitCXXTryStmt(const CXXTryStmt &S, bool IsFnTryBlock) {
1144 unsigned NumHandlers = S.getNumHandlers();
1145 EHCatchScope &CatchScope = cast<EHCatchScope>(*EHStack.begin());
1146 assert(CatchScope.getNumHandlers() == NumHandlers);
1147 llvm::BasicBlock *DispatchBlock = CatchScope.getCachedEHDispatchBlock();
1148
1149 // If the catch was not required, bail out now.
1150 if (!CatchScope.hasEHBranches()) {
1151 CatchScope.clearHandlerBlocks();
1152 EHStack.popCatch();
1153 return;
1154 }
1155
1156 // Emit the structure of the EH dispatch for this catch.
1157 emitCatchDispatchBlock(*this, CatchScope);
1158
1159 // Copy the handler blocks off before we pop the EH stack. Emitting
1160 // the handlers might scribble on this memory.
1161 SmallVector<EHCatchScope::Handler, 8> Handlers(
1162 CatchScope.begin(), CatchScope.begin() + NumHandlers);
1163
1164 EHStack.popCatch();
1165
1166 // The fall-through block.
1167 llvm::BasicBlock *ContBB = createBasicBlock("try.cont");
1168
1169 // We just emitted the body of the try; jump to the continue block.
1170 if (HaveInsertPoint())
1171 Builder.CreateBr(ContBB);
1172
1173 // Determine if we need an implicit rethrow for all these catch handlers;
1174 // see the comment below.
1175 bool doImplicitRethrow = false;
1176 if (IsFnTryBlock)
1177 doImplicitRethrow = isa<CXXDestructorDecl>(CurCodeDecl) ||
1178 isa<CXXConstructorDecl>(CurCodeDecl);
1179
1180 // Wasm uses Windows-style EH instructions, but merges all catch clauses into
1181 // one big catchpad. So we save the old funclet pad here before we traverse
1182 // each catch handler.
1183 SaveAndRestore<llvm::Instruction *> RestoreCurrentFuncletPad(
1184 CurrentFuncletPad);
1185 llvm::BasicBlock *WasmCatchStartBlock = nullptr;
1186 if (EHPersonality::get(*this).isWasmPersonality()) {
1187 auto *CatchSwitch =
1188 cast<llvm::CatchSwitchInst>(DispatchBlock->getFirstNonPHI());
1189 WasmCatchStartBlock = CatchSwitch->hasUnwindDest()
1190 ? CatchSwitch->getSuccessor(1)
1191 : CatchSwitch->getSuccessor(0);
1192 auto *CPI = cast<llvm::CatchPadInst>(WasmCatchStartBlock->getFirstNonPHI());
1193 CurrentFuncletPad = CPI;
1194 }
1195
1196 // Perversely, we emit the handlers backwards precisely because we
1197 // want them to appear in source order. In all of these cases, the
1198 // catch block will have exactly one predecessor, which will be a
1199 // particular block in the catch dispatch. However, in the case of
1200 // a catch-all, one of the dispatch blocks will branch to two
1201 // different handlers, and EmitBlockAfterUses will cause the second
1202 // handler to be moved before the first.
1203 bool HasCatchAll = false;
1204 for (unsigned I = NumHandlers; I != 0; --I) {
1205 HasCatchAll |= Handlers[I - 1].isCatchAll();
1206 llvm::BasicBlock *CatchBlock = Handlers[I-1].Block;
1207 EmitBlockAfterUses(CatchBlock);
1208
1209 // Catch the exception if this isn't a catch-all.
1210 const CXXCatchStmt *C = S.getHandler(I-1);
1211
1212 // Enter a cleanup scope, including the catch variable and the
1213 // end-catch.
1214 RunCleanupsScope CatchScope(*this);
1215
1216 // Initialize the catch variable and set up the cleanups.
1217 SaveAndRestore<llvm::Instruction *> RestoreCurrentFuncletPad(
1218 CurrentFuncletPad);
1219 CGM.getCXXABI().emitBeginCatch(*this, C);
1220
1221 // Emit the PGO counter increment.
1222 incrementProfileCounter(C);
1223
1224 // Perform the body of the catch.
1225 EmitStmt(C->getHandlerBlock());
1226
1227 // [except.handle]p11:
1228 // The currently handled exception is rethrown if control
1229 // reaches the end of a handler of the function-try-block of a
1230 // constructor or destructor.
1231
1232 // It is important that we only do this on fallthrough and not on
1233 // return. Note that it's illegal to put a return in a
1234 // constructor function-try-block's catch handler (p14), so this
1235 // really only applies to destructors.
1236 if (doImplicitRethrow && HaveInsertPoint()) {
1237 CGM.getCXXABI().emitRethrow(*this, /*isNoReturn*/false);
1238 Builder.CreateUnreachable();
1239 Builder.ClearInsertionPoint();
1240 }
1241
1242 // Fall out through the catch cleanups.
1243 CatchScope.ForceCleanup();
1244
1245 // Branch out of the try.
1246 if (HaveInsertPoint())
1247 Builder.CreateBr(ContBB);
1248 }
1249
1250 // Because in wasm we merge all catch clauses into one big catchpad, in case
1251 // none of the types in catch handlers matches after we test against each of
1252 // them, we should unwind to the next EH enclosing scope. We generate a call
1253 // to rethrow function here to do that.
1254 if (EHPersonality::get(*this).isWasmPersonality() && !HasCatchAll) {
1255 assert(WasmCatchStartBlock);
1256 // Navigate for the "rethrow" block we created in emitWasmCatchPadBlock().
1257 // Wasm uses landingpad-style conditional branches to compare selectors, so
1258 // we follow the false destination for each of the cond branches to reach
1259 // the rethrow block.
1260 llvm::BasicBlock *RethrowBlock = WasmCatchStartBlock;
1261 while (llvm::Instruction *TI = RethrowBlock->getTerminator()) {
1262 auto *BI = cast<llvm::BranchInst>(TI);
1263 assert(BI->isConditional());
1264 RethrowBlock = BI->getSuccessor(1);
1265 }
1266 assert(RethrowBlock != WasmCatchStartBlock && RethrowBlock->empty());
1267 Builder.SetInsertPoint(RethrowBlock);
1268 llvm::Function *RethrowInCatchFn =
1269 CGM.getIntrinsic(llvm::Intrinsic::wasm_rethrow_in_catch);
1270 EmitNoreturnRuntimeCallOrInvoke(RethrowInCatchFn, {});
1271 }
1272
1273 EmitBlock(ContBB);
1274 incrementProfileCounter(&S);
1275}

1299 struct PerformFinally final : EHScopeStack::Cleanup {
1300 const Stmt *Body;
1301 llvm::Value *ForEHVar;
1302 llvm::FunctionCallee EndCatchFn;
1303 llvm::FunctionCallee RethrowFn;
1304 llvm::Value *SavedExnVar;
1305
1306 PerformFinally(const Stmt *Body, llvm::Value *ForEHVar,
1307 llvm::FunctionCallee EndCatchFn,
1308 llvm::FunctionCallee RethrowFn, llvm::Value *SavedExnVar)
1309 : Body(Body), ForEHVar(ForEHVar), EndCatchFn(EndCatchFn),
1310 RethrowFn(RethrowFn), SavedExnVar(SavedExnVar) {}
1311
1312 void Emit(CodeGenFunction &CGF, Flags flags) override {
1313 // Enter a cleanup to call the end-catch function if one was provided.
1314 if (EndCatchFn)
1315 CGF.EHStack.pushCleanup<CallEndCatchForFinally>(NormalAndEHCleanup,
1316 ForEHVar, EndCatchFn);
1317
1318 // Save the current cleanup destination in case there are
1319 // cleanups in the finally block.
1320 llvm::Value *SavedCleanupDest =
1321 CGF.Builder.CreateLoad(CGF.getNormalCleanupDestSlot(),
1322 "cleanup.dest.saved");
1323
1324 // Emit the finally block.
1325 CGF.EmitStmt(Body);
1326
1327 // If the end of the finally is reachable, check whether this was
1328 // for EH. If so, rethrow.
1329 if (CGF.HaveInsertPoint()) {
1330 llvm::BasicBlock *RethrowBB = CGF.createBasicBlock("finally.rethrow");
1331 llvm::BasicBlock *ContBB = CGF.createBasicBlock("finally.cont");
1332
1333 llvm::Value *ShouldRethrow =
1334 CGF.Builder.CreateFlagLoad(ForEHVar, "finally.shouldthrow");
1335 CGF.Builder.CreateCondBr(ShouldRethrow, RethrowBB, ContBB);
1336
1337 CGF.EmitBlock(RethrowBB);
1338 if (SavedExnVar) {
1339 CGF.EmitRuntimeCallOrInvoke(RethrowFn,
1340 CGF.Builder.CreateAlignedLoad(SavedExnVar, CGF.getPointerAlign()));
1341 } else {
1342 CGF.EmitRuntimeCallOrInvoke(RethrowFn);
1343 }
1344 CGF.Builder.CreateUnreachable();
1345
1346 CGF.EmitBlock(ContBB);
1347
1348 // Restore the cleanup destination.
1349 CGF.Builder.CreateStore(SavedCleanupDest,
1350 CGF.getNormalCleanupDestSlot());
1351 }
1352
1353 // Leave the end-catch cleanup. As an optimization, pretend that
1354 // the fallthrough path was inaccessible; we've dynamically proven
1355 // that we're not in the EH case along that path.
1356 if (EndCatchFn) {
1357 CGBuilderTy::InsertPoint SavedIP = CGF.Builder.saveAndClearIP();
1358 CGF.PopCleanupBlock();
1359 CGF.Builder.restoreIP(SavedIP);
1360 }
1361
1362 // Now make sure we actually have an insertion point or the
1363 // cleanup gods will hate us.
1364 CGF.EnsureInsertPoint();
1365 }
1366 };

1369/// Enters a finally block for an implementation using zero-cost
1370/// exceptions. This is mostly general, but hard-codes some
1371/// language/ABI-specific behavior in the catch-all sections.
1372void CodeGenFunction::FinallyInfo::enter(CodeGenFunction &CGF, const Stmt *body,
1373 llvm::FunctionCallee beginCatchFn,
1374 llvm::FunctionCallee endCatchFn,
1375 llvm::FunctionCallee rethrowFn) {
1376 assert((!!beginCatchFn) == (!!endCatchFn) &&
1377 "begin/end catch functions not paired");
1378 assert(rethrowFn && "rethrow function is required");
1379
1380 BeginCatchFn = beginCatchFn;
1381
1382 // The rethrow function has one of the following two types:
1383 // void (*)()
1384 // void (*)(void*)
1385 // In the latter case we need to pass it the exception object.
1386 // But we can't use the exception slot because the @finally might
1387 // have a landing pad (which would overwrite the exception slot).
1388 llvm::FunctionType *rethrowFnTy = rethrowFn.getFunctionType();
1389 SavedExnVar = nullptr;
1390 if (rethrowFnTy->getNumParams())
1391 SavedExnVar = CGF.CreateTempAlloca(CGF.Int8PtrTy, "finally.exn");
1392
1393 // A finally block is a statement which must be executed on any edge
1394 // out of a given scope. Unlike a cleanup, the finally block may
1395 // contain arbitrary control flow leading out of itself. In
1396 // addition, finally blocks should always be executed, even if there
1397 // are no catch handlers higher on the stack. Therefore, we
1398 // surround the protected scope with a combination of a normal
1399 // cleanup (to catch attempts to break out of the block via normal
1400 // control flow) and an EH catch-all (semantically "outside" any try
1401 // statement to which the finally block might have been attached).
1402 // The finally block itself is generated in the context of a cleanup
1403 // which conditionally leaves the catch-all.
1404
1405 // Jump destination for performing the finally block on an exception
1406 // edge. We'll never actually reach this block, so unreachable is
1407 // fine.
1408 RethrowDest = CGF.getJumpDestInCurrentScope(CGF.getUnreachableBlock());
1409
1410 // Whether the finally block is being executed for EH purposes.
1411 ForEHVar = CGF.CreateTempAlloca(CGF.Builder.getInt1Ty(), "finally.for-eh");
1412 CGF.Builder.CreateFlagStore(false, ForEHVar);
1413
1414 // Enter a normal cleanup which will perform the @finally block.
1415 CGF.EHStack.pushCleanup<PerformFinally>(NormalCleanup, body,
1416 ForEHVar, endCatchFn,
1417 rethrowFn, SavedExnVar);
1418
1419 // Enter a catch-all scope.
1420 llvm::BasicBlock *catchBB = CGF.createBasicBlock("finally.catchall");
1421 EHCatchScope *catchScope = CGF.EHStack.pushCatch(1);
1422 catchScope->setCatchAllHandler(0, catchBB);
1423}

1425void CodeGenFunction::FinallyInfo::exit(CodeGenFunction &CGF) {
1426 // Leave the finally catch-all.
1427 EHCatchScope &catchScope = cast<EHCatchScope>(*CGF.EHStack.begin());
1428 llvm::BasicBlock *catchBB = catchScope.getHandler(0).Block;
1429
1430 CGF.popCatchScope();
1431
1432 // If there are any references to the catch-all block, emit it.
1433 if (catchBB->use_empty()) {
1434 delete catchBB;
1435 } else {
1436 CGBuilderTy::InsertPoint savedIP = CGF.Builder.saveAndClearIP();
1437 CGF.EmitBlock(catchBB);
1438
1439 llvm::Value *exn = nullptr;
1440
1441 // If there's a begin-catch function, call it.
1442 if (BeginCatchFn) {
1443 exn = CGF.getExceptionFromSlot();
1444 CGF.EmitNounwindRuntimeCall(BeginCatchFn, exn);
1445 }
1446
1447 // If we need to remember the exception pointer to rethrow later, do so.
1448 if (SavedExnVar) {
1449 if (!exn) exn = CGF.getExceptionFromSlot();
1450 CGF.Builder.CreateAlignedStore(exn, SavedExnVar, CGF.getPointerAlign());
1451 }
1452
1453 // Tell the cleanups in the finally block that we're do this for EH.
1454 CGF.Builder.CreateFlagStore(true, ForEHVar);
1455
1456 // Thread a jump through the finally cleanup.
1457 CGF.EmitBranchThroughCleanup(RethrowDest);
1458
1459 CGF.Builder.restoreIP(savedIP);
1460 }
1461
1462 // Finally, leave the @finally cleanup.
1463 CGF.PopCleanupBlock();
1464}

1599

1662/// Find all local variable captures in the statement.
1663struct CaptureFinder : ConstStmtVisitor<CaptureFinder> {
1664 CodeGenFunction &ParentCGF;
1665 const VarDecl *ParentThis;
1666 llvm::SmallSetVector<const VarDecl *, 4> Captures;
1667 Address SEHCodeSlot = Address::invalid();
1668 CaptureFinder(CodeGenFunction &ParentCGF, const VarDecl *ParentThis)
1669 : ParentCGF(ParentCGF), ParentThis(ParentThis) {}
1670
1671 // Return true if we need to do any capturing work.
1672 bool foundCaptures() {
1673 return !Captures.empty() || SEHCodeSlot.isValid();
1674 }
1675
1676 void Visit(const Stmt *S) {
1677 // See if this is a capture, then recurse.
1678 ConstStmtVisitor<CaptureFinder>::Visit(S);
1679 for (const Stmt *Child : S->children())
1680 if (Child)
1681 Visit(Child);
1682 }
1683
1684 void VisitDeclRefExpr(const DeclRefExpr *E) {
1685 // If this is already a capture, just make sure we capture 'this'.
1686 if (E->refersToEnclosingVariableOrCapture()) {
1687 Captures.insert(ParentThis);
1688 return;
1689 }
1690
1691 const auto *D = dyn_cast<VarDecl>(E->getDecl());
1692 if (D && D->isLocalVarDeclOrParm() && D->hasLocalStorage())
1693 Captures.insert(D);
1694 }
1695
1696 void VisitCXXThisExpr(const CXXThisExpr *E) {
1697 Captures.insert(ParentThis);
1698 }
1699
1700 void VisitCallExpr(const CallExpr *E) {
1701 // We only need to add parent frame allocations for these builtins in x86.
1702 if (ParentCGF.getTarget().getTriple().getArch() != llvm::Triple::x86)
1703 return;
1704
1705 unsigned ID = E->getBuiltinCallee();
1706 switch (ID) {
1707 case Builtin::BI__exception_code:
1708 case Builtin::BI_exception_code:
1709 // This is the simple case where we are the outermost finally. All we
1710 // have to do here is make sure we escape this and recover it in the
1711 // outlined handler.
1712 if (!SEHCodeSlot.isValid())
1713 SEHCodeSlot = ParentCGF.SEHCodeSlotStack.back();
1714 break;
1715 }
1716 }
1717};

1720Address CodeGenFunction::recoverAddrOfEscapedLocal(CodeGenFunction &ParentCGF,
1721 Address ParentVar,
1722 llvm::Value *ParentFP) {
1723 llvm::CallInst *RecoverCall = nullptr;
1724 CGBuilderTy Builder(*this, AllocaInsertPt);
1725 if (auto *ParentAlloca = dyn_cast<llvm::AllocaInst>(ParentVar.getPointer())) {
1726 // Mark the variable escaped if nobody else referenced it and compute the
1727 // localescape index.
1728 auto InsertPair = ParentCGF.EscapedLocals.insert(
1729 std::make_pair(ParentAlloca, ParentCGF.EscapedLocals.size()));
1730 int FrameEscapeIdx = InsertPair.first->second;
1731 // call i8* @llvm.localrecover(i8* bitcast(@parentFn), i8* %fp, i32 N)
1732 llvm::Function *FrameRecoverFn = llvm::Intrinsic::getDeclaration(
1733 &CGM.getModule(), llvm::Intrinsic::localrecover);
1734 llvm::Constant *ParentI8Fn =
1735 llvm::ConstantExpr::getBitCast(ParentCGF.CurFn, Int8PtrTy);
1736 RecoverCall = Builder.CreateCall(
1737 FrameRecoverFn, {ParentI8Fn, ParentFP,
1738 llvm::ConstantInt::get(Int32Ty, FrameEscapeIdx)});
1739
1740 } else {
1741 // If the parent didn't have an alloca, we're doing some nested outlining.
1742 // Just clone the existing localrecover call, but tweak the FP argument to
1743 // use our FP value. All other arguments are constants.
1744 auto *ParentRecover =
1745 cast<llvm::IntrinsicInst>(ParentVar.getPointer()->stripPointerCasts());
1746 assert(ParentRecover->getIntrinsicID() == llvm::Intrinsic::localrecover &&
1747 "expected alloca or localrecover in parent LocalDeclMap");
1748 RecoverCall = cast<llvm::CallInst>(ParentRecover->clone());
1749 RecoverCall->setArgOperand(1, ParentFP);
1750 RecoverCall->insertBefore(AllocaInsertPt);
1751 }
1752
1753 // Bitcast the variable, rename it, and insert it in the local decl map.
1754 llvm::Value *ChildVar =
1755 Builder.CreateBitCast(RecoverCall, ParentVar.getType());
1756 ChildVar->setName(ParentVar.getName());
1757 return Address(ChildVar, ParentVar.getAlignment());
1758}

1760void CodeGenFunction::EmitCapturedLocals(CodeGenFunction &ParentCGF,
1761 const Stmt *OutlinedStmt,
1762 bool IsFilter) {
1763 // Find all captures in the Stmt.
1764 CaptureFinder Finder(ParentCGF, ParentCGF.CXXABIThisDecl);
1765 Finder.Visit(OutlinedStmt);
1766
1767 // We can exit early on x86_64 when there are no captures. We just have to
1768 // save the exception code in filters so that __exception_code() works.
1769 if (!Finder.foundCaptures() &&
1770 CGM.getTarget().getTriple().getArch() != llvm::Triple::x86) {
1771 if (IsFilter)
1772 EmitSEHExceptionCodeSave(ParentCGF, nullptr, nullptr);
1773 return;
1774 }
1775
1776 llvm::Value *EntryFP = nullptr;
1777 CGBuilderTy Builder(CGM, AllocaInsertPt);
1778 if (IsFilter && CGM.getTarget().getTriple().getArch() == llvm::Triple::x86) {
1779 // 32-bit SEH filters need to be careful about FP recovery. The end of the
1780 // EH registration is passed in as the EBP physical register. We can
1781 // recover that with llvm.frameaddress(1).
1782 EntryFP = Builder.CreateCall(
1783 CGM.getIntrinsic(llvm::Intrinsic::frameaddress, {CGM.AllocaInt8PtrTy}),
1784 {Builder.getInt32(1)});
1785 } else {
1786 // Otherwise, for x64 and 32-bit finally functions, the parent FP is the
1787 // second parameter.
1788 auto AI = CurFn->arg_begin();
1789 ++AI;
1790 EntryFP = &*AI;
1791 }
1792
1793 llvm::Value *ParentFP = EntryFP;
1794 if (IsFilter) {
1795 // Given whatever FP the runtime provided us in EntryFP, recover the true
1796 // frame pointer of the parent function. We only need to do this in filters,
1797 // since finally funclets recover the parent FP for us.
1798 llvm::Function *RecoverFPIntrin =
1799 CGM.getIntrinsic(llvm::Intrinsic::eh_recoverfp);
1800 llvm::Constant *ParentI8Fn =
1801 llvm::ConstantExpr::getBitCast(ParentCGF.CurFn, Int8PtrTy);
1802 ParentFP = Builder.CreateCall(RecoverFPIntrin, {ParentI8Fn, EntryFP});
1803 }
1804
1805 // Create llvm.localrecover calls for all captures.
1806 for (const VarDecl *VD : Finder.Captures) {
1807 if (isa<ImplicitParamDecl>(VD)) {
1808 CGM.ErrorUnsupported(VD, "'this' captured by SEH");
1809 CXXThisValue = llvm::UndefValue::get(ConvertTypeForMem(VD->getType()));
1810 continue;
1811 }
1812 if (VD->getType()->isVariablyModifiedType()) {
1813 CGM.ErrorUnsupported(VD, "VLA captured by SEH");
1814 continue;
1815 }
1816 assert((isa<ImplicitParamDecl>(VD) || VD->isLocalVarDeclOrParm()) &&
1817 "captured non-local variable");
1818
1819 // If this decl hasn't been declared yet, it will be declared in the
1820 // OutlinedStmt.
1821 auto I = ParentCGF.LocalDeclMap.find(VD);
1822 if (I == ParentCGF.LocalDeclMap.end())
1823 continue;
1824
1825 Address ParentVar = I->second;
1826 setAddrOfLocalVar(
1827 VD, recoverAddrOfEscapedLocal(ParentCGF, ParentVar, ParentFP));
1828 }
1829
1830 if (Finder.SEHCodeSlot.isValid()) {
1831 SEHCodeSlotStack.push_back(
1832 recoverAddrOfEscapedLocal(ParentCGF, Finder.SEHCodeSlot, ParentFP));
1833 }
1834
1835 if (IsFilter)
1836 EmitSEHExceptionCodeSave(ParentCGF, ParentFP, EntryFP);
1837}

1839/// Arrange a function prototype that can be called by Windows exception
1840/// handling personalities. On Win64, the prototype looks like:
1841/// RetTy func(void *EHPtrs, void *ParentFP);
1842void CodeGenFunction::startOutlinedSEHHelper(CodeGenFunction &ParentCGF,
1843 bool IsFilter,
1844 const Stmt *OutlinedStmt) {
1845 SourceLocation StartLoc = OutlinedStmt->getBeginLoc();
1846
1847 // Get the mangled function name.
1848 SmallString<128> Name;
1849 {
1850 llvm::raw_svector_ostream OS(Name);
1851 const NamedDecl *ParentSEHFn = ParentCGF.CurSEHParent;
1852 assert(ParentSEHFn && "No CurSEHParent!");
1853 MangleContext &Mangler = CGM.getCXXABI().getMangleContext();
1854 if (IsFilter)
1855 Mangler.mangleSEHFilterExpression(ParentSEHFn, OS);
1856 else
1857 Mangler.mangleSEHFinallyBlock(ParentSEHFn, OS);
1858 }
1859
1860 FunctionArgList Args;
1861 if (CGM.getTarget().getTriple().getArch() != llvm::Triple::x86 || !IsFilter) {
1862 // All SEH finally functions take two parameters. Win64 filters take two
1863 // parameters. Win32 filters take no parameters.
1864 if (IsFilter) {
1865 Args.push_back(ImplicitParamDecl::Create(
1866 getContext(), /*DC=*/nullptr, StartLoc,
1867 &getContext().Idents.get("exception_pointers"),
1868 getContext().VoidPtrTy, ImplicitParamDecl::Other));
1869 } else {
1870 Args.push_back(ImplicitParamDecl::Create(
1871 getContext(), /*DC=*/nullptr, StartLoc,
1872 &getContext().Idents.get("abnormal_termination"),
1873 getContext().UnsignedCharTy, ImplicitParamDecl::Other));
1874 }
1875 Args.push_back(ImplicitParamDecl::Create(
1876 getContext(), /*DC=*/nullptr, StartLoc,
1877 &getContext().Idents.get("frame_pointer"), getContext().VoidPtrTy,
1878 ImplicitParamDecl::Other));
1879 }
1880
1881 QualType RetTy = IsFilter ? getContext().LongTy : getContext().VoidTy;
1882
1883 const CGFunctionInfo &FnInfo =
1884 CGM.getTypes().arrangeBuiltinFunctionDeclaration(RetTy, Args);
1885
1886 llvm::FunctionType *FnTy = CGM.getTypes().GetFunctionType(FnInfo);
1887 llvm::Function *Fn = llvm::Function::Create(
1888 FnTy, llvm::GlobalValue::InternalLinkage, Name.str(), &CGM.getModule());
1889
1890 IsOutlinedSEHHelper = true;
1891
1892 StartFunction(GlobalDecl(), RetTy, Fn, FnInfo, Args,
1893 OutlinedStmt->getBeginLoc(), OutlinedStmt->getBeginLoc());
1894 CurSEHParent = ParentCGF.CurSEHParent;
1895
1896 CGM.SetLLVMFunctionAttributes(GlobalDecl(), FnInfo, CurFn);
1897 EmitCapturedLocals(ParentCGF, OutlinedStmt, IsFilter);
1898}

1900/// Create a stub filter function that will ultimately hold the code of the
1901/// filter expression. The EH preparation passes in LLVM will outline the code
1902/// from the main function body into this stub.
1903llvm::Function *
1904CodeGenFunction::GenerateSEHFilterFunction(CodeGenFunction &ParentCGF,
1905 const SEHExceptStmt &Except) {
1906 const Expr *FilterExpr = Except.getFilterExpr();
1907 startOutlinedSEHHelper(ParentCGF, true, FilterExpr);
1908
1909 // Emit the original filter expression, convert to i32, and return.
1910 llvm::Value *R = EmitScalarExpr(FilterExpr);
1911 R = Builder.CreateIntCast(R, ConvertType(getContext().LongTy),
1912 FilterExpr->getType()->isSignedIntegerType());
1913 Builder.CreateStore(R, ReturnValue);
1914
1915 FinishFunction(FilterExpr->getEndLoc());
1916
1917 return CurFn;
1918}

1920llvm::Function *
1921CodeGenFunction::GenerateSEHFinallyFunction(CodeGenFunction &ParentCGF,
1922 const SEHFinallyStmt &Finally) {
1923 const Stmt *FinallyBlock = Finally.getBlock();
1924 startOutlinedSEHHelper(ParentCGF, false, FinallyBlock);
1925

1927 EmitStmt(FinallyBlock);
1928
1929 FinishFunction(FinallyBlock->getEndLoc());
1930
1931 return CurFn;

1934void CodeGenFunction::EmitSEHExceptionCodeSave(CodeGenFunction &ParentCGF,
1935 llvm::Value *ParentFP,
1936 llvm::Value *EntryFP) {
1937 // Get the pointer to the EXCEPTION_POINTERS struct. This is returned by the
1938 // __exception_info intrinsic.
1939 if (CGM.getTarget().getTriple().getArch() != llvm::Triple::x86) {
1940 // On Win64, the info is passed as the first parameter to the filter.
1941 SEHInfo = &*CurFn->arg_begin();
1942 SEHCodeSlotStack.push_back(
1943 CreateMemTemp(getContext().IntTy, "__exception_code"));
1944 } else {
1945 // On Win32, the EBP on entry to the filter points to the end of an
1946 // exception registration object. It contains 6 32-bit fields, and the info
1947 // pointer is stored in the second field. So, GEP 20 bytes backwards and
1948 // load the pointer.
1949 SEHInfo = Builder.CreateConstInBoundsGEP1_32(Int8Ty, EntryFP, -20);
1950 unsigned DefaultAS = CGM.getTargetCodeGenInfo().getDefaultAS();
1951 SEHInfo = Builder.CreateBitCast(SEHInfo, Int8PtrTy->getPointerTo(DefaultAS));
1952 SEHInfo = Builder.CreateAlignedLoad(Int8PtrTy, SEHInfo, getPointerAlign());
1953 SEHCodeSlotStack.push_back(recoverAddrOfEscapedLocal(
1954 ParentCGF, ParentCGF.SEHCodeSlotStack.back(), ParentFP));
1955 }
1956
1957 // Save the exception code in the exception slot to unify exception access in
1958 // the filter function and the landing pad.
1959 // struct EXCEPTION_POINTERS {
1960 // EXCEPTION_RECORD *ExceptionRecord;
1961 // CONTEXT *ContextRecord;
1962 // };
1963 // int exceptioncode = exception_pointers->ExceptionRecord->ExceptionCode;
1964 unsigned DefaultAS = CGM.getTargetCodeGenInfo().getDefaultAS();
1965 llvm::Type *RecordTy = CGM.Int32Ty->getPointerTo(DefaultAS);
1966 llvm::Type *PtrsTy = llvm::StructType::get(RecordTy, CGM.VoidPtrTy);
1967 llvm::Value *Ptrs = Builder.CreateBitCast(SEHInfo,
1968 PtrsTy->getPointerTo(DefaultAS));
1969 llvm::Value *Rec = Builder.CreateStructGEP(PtrsTy, Ptrs, 0);
1970 Rec = Builder.CreateAlignedLoad(Rec, getPointerAlign());
1971 llvm::Value *Code = Builder.CreateAlignedLoad(Rec, getIntAlign());
1972 assert(!SEHCodeSlotStack.empty() && "emitting EH code outside of __except");
1973 Builder.CreateStore(Code, SEHCodeSlotStack.back());
1974}

2002void CodeGenFunction::EnterSEHTryStmt(const SEHTryStmt &S) {
2003 CodeGenFunction HelperCGF(CGM, /*suppressNewContext=*/true);
2004 if (const SEHFinallyStmt *Finally = S.getFinallyHandler()) {
2005 // Outline the finally block.
2006 llvm::Function *FinallyFunc =
2007 HelperCGF.GenerateSEHFinallyFunction(*this, *Finally);
2008
2009 // Push a cleanup for __finally blocks.
2010 EHStack.pushCleanup<PerformSEHFinally>(NormalAndEHCleanup, FinallyFunc);
2011 return;
2012 }
2013
2014 // Otherwise, we must have an __except block.
2015 const SEHExceptStmt *Except = S.getExceptHandler();
2016 assert(Except);
2017 EHCatchScope *CatchScope = EHStack.pushCatch(1);
2018 SEHCodeSlotStack.push_back(
2019 CreateMemTemp(getContext().IntTy, "__exception_code"));
2020
2021 // If the filter is known to evaluate to 1, then we can use the clause
2022 // "catch i8* null". We can't do this on x86 because the filter has to save
2023 // the exception code.
2024 llvm::Constant *C =
2025 ConstantEmitter(*this).tryEmitAbstract(Except->getFilterExpr(),
2026 getContext().IntTy);
2027 if (CGM.getTarget().getTriple().getArch() != llvm::Triple::x86 && C &&
2028 C->isOneValue()) {
2029 CatchScope->setCatchAllHandler(0, createBasicBlock("__except"));
2030 return;
2031 }
2032
2033 // In general, we have to emit an outlined filter function. Use the function
2034 // in place of the RTTI typeinfo global that C++ EH uses.
2035 llvm::Function *FilterFunc =
2036 HelperCGF.GenerateSEHFilterFunction(*this, *Except);
2037 llvm::Constant *OpaqueFunc =
2038 llvm::ConstantExpr::getBitCast(FilterFunc, Int8PtrTy);
2039 CatchScope->setHandler(0, OpaqueFunc, createBasicBlock("__except.ret"));
2040}

2042void CodeGenFunction::ExitSEHTryStmt(const SEHTryStmt &S) {
2043 // Just pop the cleanup if it's a __finally block.
2044 if (S.getFinallyHandler()) {
2045 PopCleanupBlock();
2046 return;
2047 }
2048
2049 // Otherwise, we must have an __except block.
2050 const SEHExceptStmt *Except = S.getExceptHandler();
2051 assert(Except && "__try must have __finally xor __except");
2052 EHCatchScope &CatchScope = cast<EHCatchScope>(*EHStack.begin());
2053
2054 // Don't emit the __except block if the __try block lacked invokes.
2055 // TODO: Model unwind edges from instructions, either with iload / istore or
2056 // a try body function.
2057 if (!CatchScope.hasEHBranches()) {
2058 CatchScope.clearHandlerBlocks();
2059 EHStack.popCatch();
2060 SEHCodeSlotStack.pop_back();
2061 return;
2062 }
2063
2064 // The fall-through block.
2065 llvm::BasicBlock *ContBB = createBasicBlock("__try.cont");
2066
2067 // We just emitted the body of the __try; jump to the continue block.
2068 if (HaveInsertPoint())
2069 Builder.CreateBr(ContBB);
2070
2071 // Check if our filter function returned true.
2072 emitCatchDispatchBlock(*this, CatchScope);
2073
2074 // Grab the block before we pop the handler.
2075 llvm::BasicBlock *CatchPadBB = CatchScope.getHandler(0).Block;
2076 EHStack.popCatch();
2077
2078 EmitBlockAfterUses(CatchPadBB);
2079
2080 // __except blocks don't get outlined into funclets, so immediately do a
2081 // catchret.
2082 llvm::CatchPadInst *CPI =
2083 cast<llvm::CatchPadInst>(CatchPadBB->getFirstNonPHI());
2084 llvm::BasicBlock *ExceptBB = createBasicBlock("__except");
2085 Builder.CreateCatchRet(CPI, ExceptBB);
2086 EmitBlock(ExceptBB);
2087
2088 // On Win64, the exception code is returned in EAX. Copy it into the slot.
2089 if (CGM.getTarget().getTriple().getArch() != llvm::Triple::x86) {
2090 llvm::Function *SEHCodeIntrin =
2091 CGM.getIntrinsic(llvm::Intrinsic::eh_exceptioncode);
2092 llvm::Value *Code = Builder.CreateCall(SEHCodeIntrin, {CPI});
2093 Builder.CreateStore(Code, SEHCodeSlotStack.back());
2094 }
2095
2096 // Emit the __except body.
2097 EmitStmt(Except->getBlock());
2098
2099 // End the lifetime of the exception code.
2100 SEHCodeSlotStack.pop_back();
2101
2102 if (HaveInsertPoint())
2103 Builder.CreateBr(ContBB);
2104
2105 EmitBlock(ContBB);
2106}

2108void CodeGenFunction::EmitSEHLeaveStmt(const SEHLeaveStmt &S) {
2109 // If this code is reachable then emit a stop point (if generating
2110 // debug info). We have to do this ourselves because we are on the
2111 // "simple" statement path.
2112 if (HaveInsertPoint())
2113 EmitStopPoint(&S);
2114
2115 // This must be a __leave from a __finally block, which we warn on and is UB.
2116 // Just emit unreachable.
2117 if (!isSEHTryScope()) {
2118 Builder.CreateUnreachable();
2119 Builder.ClearInsertionPoint();
2120 return;
2121 }
2122
2123 EmitBranchThroughCleanup(*SEHTryEpilogueStack.back());
2124}
