22llvm::Type *ConstantInitFuture::getType() const {
23 assert(Data && "dereferencing null future");
24 if (Data.is<llvm::Constant*>()) {
25 return Data.get<llvm::Constant*>()->getType();
26 } else {
27 return Data.get<ConstantInitBuilderBase*>()->Buffer[0]->getType();
28 }
29}
30
31void ConstantInitFuture::abandon() {
32 assert(Data && "abandoning null future");
33 if (auto builder = Data.dyn_cast<ConstantInitBuilderBase*>()) {
34 builder->abandon(0);
35 }
36 Data = nullptr;
37}
38
39void ConstantInitFuture::installInGlobal(llvm::GlobalVariable *GV) {
40 assert(Data && "installing null future");
41 if (Data.is<llvm::Constant*>()) {
42 GV->setInitializer(Data.get<llvm::Constant*>());
43 } else {
44 auto &builder = *Data.get<ConstantInitBuilderBase*>();
45 assert(builder.Buffer.size() == 1);
46 builder.setGlobalInitializer(GV, builder.Buffer[0]);
47 builder.Buffer.clear();
48 Data = nullptr;
49 }
50}

52ConstantInitFuture
53ConstantInitBuilderBase::createFuture(llvm::Constant *initializer) {
54 assert(Buffer.empty() && "buffer not current empty");
55 Buffer.push_back(initializer);
56 return ConstantInitFuture(this);
57}

67llvm::GlobalVariable *
68ConstantInitBuilderBase::createGlobal(llvm::Constant *initializer,
69 const llvm::Twine &name,
70 CharUnits alignment,
71 bool constant,
72 llvm::GlobalValue::LinkageTypes linkage,
73 unsigned addressSpace) {
74 if (addressSpace == static_cast<unsigned>(-1))
75 addressSpace = CGM.getTargetCodeGenInfo().getDefaultAS();
76 auto GV = new llvm::GlobalVariable(CGM.getModule(),
77 initializer->getType(),
78 constant,
79 linkage,
80 initializer,
81 name,
82 /*insert before*/ nullptr,
83 llvm::GlobalValue::NotThreadLocal,
84 addressSpace);
85 GV->setAlignment(alignment.getQuantity());
86 resolveSelfReferences(GV);
87 return GV;
88}

90void ConstantInitBuilderBase::setGlobalInitializer(llvm::GlobalVariable *GV,
91 llvm::Constant *initializer){
92 GV->setInitializer(initializer);
93
94 if (!SelfReferences.empty())
95 resolveSelfReferences(GV);
96}

98void ConstantInitBuilderBase::resolveSelfReferences(llvm::GlobalVariable *GV) {
99 for (auto &entry : SelfReferences) {
100 llvm::Constant *resolvedReference =
101 llvm::ConstantExpr::getInBoundsGetElementPtr(
102 GV->getValueType(), GV, entry.Indices);
103 auto dummy = entry.Dummy;
104 dummy->replaceAllUsesWith(resolvedReference);
105 dummy->eraseFromParent();
106 }
107 SelfReferences.clear();
108}

110void ConstantInitBuilderBase::abandon(size_t newEnd) {
111 // Remove all the entries we've added.
112 Buffer.erase(Buffer.begin() + newEnd, Buffer.end());
113
114 // If we're abandoning all the way to the beginning, destroy
115 // all the self-references, because we might not get another
116 // opportunity.
117 if (newEnd == 0) {
118 for (auto &entry : SelfReferences) {
119 auto dummy = entry.Dummy;
120 dummy->replaceAllUsesWith(llvm::UndefValue::get(dummy->getType()));
121 dummy->eraseFromParent();
122 }
123 SelfReferences.clear();
124 }
125}

131llvm::Constant *
132ConstantAggregateBuilderBase::getRelativeOffset(llvm::IntegerType *offsetType,
133 llvm::Constant *target) {
134 // Compute the address of the relative-address slot.
135 auto base = getAddrOfCurrentPosition(offsetType);
136
137 // Subtract.
138 base = llvm::ConstantExpr::getPtrToInt(base, Builder.CGM.IntPtrTy);
139 target = llvm::ConstantExpr::getPtrToInt(target, Builder.CGM.IntPtrTy);
140 llvm::Constant *offset = llvm::ConstantExpr::getSub(target, base);
141
142 // Truncate to the relative-address type if necessary.
143 if (Builder.CGM.IntPtrTy != offsetType) {
144 offset = llvm::ConstantExpr::getTrunc(offset, offsetType);
145 }
146
147 return offset;
148}

150llvm::Constant *
151ConstantAggregateBuilderBase::getAddrOfCurrentPosition(llvm::Type *type) {
152 // Make a global variable. We will replace this with a GEP to this
153 // position after installing the initializer.
154 auto dummy =
155 new llvm::GlobalVariable(Builder.CGM.getModule(), type, true,
156 llvm::GlobalVariable::PrivateLinkage,
157 nullptr, "");
158 Builder.SelfReferences.emplace_back(dummy);
159 auto &entry = Builder.SelfReferences.back();
160 (void) getGEPIndicesToCurrentPosition(entry.Indices);
161 return dummy;
162}

164void ConstantAggregateBuilderBase::getGEPIndicesTo(
165 llvm::SmallVectorImpl<llvm::Constant*> &indices,
166 size_t position) const {
167 // Recurse on the parent builder if present.
168 if (Parent) {
169 Parent->getGEPIndicesTo(indices, Begin);
170
171 // Otherwise, add an index to drill into the first level of pointer.
172 } else {
173 assert(indices.empty());
174 indices.push_back(llvm::ConstantInt::get(Builder.CGM.Int32Ty, 0));
175 }
176
177 assert(position >= Begin);
178 // We have to use i32 here because struct GEPs demand i32 indices.
179 // It's rather unlikely to matter in practice.
180 indices.push_back(llvm::ConstantInt::get(Builder.CGM.Int32Ty,
181 position - Begin));
182}

184ConstantAggregateBuilderBase::PlaceholderPosition
185ConstantAggregateBuilderBase::addPlaceholderWithSize(llvm::Type *type) {
186 // Bring the offset up to the last field.
187 CharUnits offset = getNextOffsetFromGlobal();
188
189 // Create the placeholder.
190 auto position = addPlaceholder();
191
192 // Advance the offset past that field.
193 auto &layout = Builder.CGM.getDataLayout();
194 if (!Packed)
195 offset = offset.alignTo(CharUnits::fromQuantity(
196 layout.getABITypeAlignment(type)));
197 offset += CharUnits::fromQuantity(layout.getTypeStoreSize(type));
198
199 CachedOffsetEnd = Builder.Buffer.size();
200 CachedOffsetFromGlobal = offset;
201
202 return position;
203}
204
205CharUnits ConstantAggregateBuilderBase::getOffsetFromGlobalTo(size_t end) const{
206 size_t cacheEnd = CachedOffsetEnd;
207 assert(cacheEnd <= end);
208
209 // Fast path: if the cache is valid, just use it.
210 if (cacheEnd == end) {
211 return CachedOffsetFromGlobal;
212 }
213
214 // If the cached range ends before the index at which the current
215 // aggregate starts, recurse for the parent.
216 CharUnits offset;
217 if (cacheEnd < Begin) {
218 assert(cacheEnd == 0);
219 assert(Parent && "Begin != 0 for root builder");
220 cacheEnd = Begin;
221 offset = Parent->getOffsetFromGlobalTo(Begin);
222 } else {
223 offset = CachedOffsetFromGlobal;
224 }
225
226 // Perform simple layout on the elements in cacheEnd..<end.
227 if (cacheEnd != end) {
228 auto &layout = Builder.CGM.getDataLayout();
229 do {
230 llvm::Constant *element = Builder.Buffer[cacheEnd];
231 assert(element != nullptr &&
232 "cannot compute offset when a placeholder is present");
233 llvm::Type *elementType = element->getType();
234 if (!Packed)
235 offset = offset.alignTo(CharUnits::fromQuantity(
236 layout.getABITypeAlignment(elementType)));
237 offset += CharUnits::fromQuantity(layout.getTypeStoreSize(elementType));
238 } while (++cacheEnd != end);
239 }
240
241 // Cache and return.
242 CachedOffsetEnd = cacheEnd;
243 CachedOffsetFromGlobal = offset;
244 return offset;
245}

247llvm::Constant *ConstantAggregateBuilderBase::finishArray(llvm::Type *eltTy) {
248 markFinished();
249
250 auto &buffer = getBuffer();
251 assert((Begin < buffer.size() ||
252 (Begin == buffer.size() && eltTy))
253 && "didn't add any array elements without element type");
254 auto elts = llvm::makeArrayRef(buffer).slice(Begin);
255 if (!eltTy) eltTy = elts[0]->getType();
256 auto type = llvm::ArrayType::get(eltTy, elts.size());
257 auto constant = llvm::ConstantArray::get(type, elts);
258 buffer.erase(buffer.begin() + Begin, buffer.end());
259 return constant;
260}

262llvm::Constant *
263ConstantAggregateBuilderBase::finishStruct(llvm::StructType *ty) {
264 markFinished();
265
266 auto &buffer = getBuffer();
267 auto elts = llvm::makeArrayRef(buffer).slice(Begin);
268
269 if (ty == nullptr && elts.empty())
270 ty = llvm::StructType::get(Builder.CGM.getLLVMContext(), {}, Packed);
271
272 llvm::Constant *constant;
273 if (ty) {
274 assert(ty->isPacked() == Packed);
275 constant = llvm::ConstantStruct::get(ty, elts);
276 } else {
277 constant = llvm::ConstantStruct::getAnon(elts, Packed);
278 }
279
280 buffer.erase(buffer.begin() + Begin, buffer.end());
281 return constant;
282}
