26static void EmitDeclInit(CodeGenFunction &CGF, const VarDecl &D,
27 ConstantAddress DeclPtr) {
28 // TODO: Enable assert for non-CHERI like OpenCL
29 //assert(
30 // (D.hasGlobalStorage() ||
31 // (D.hasLocalStorage() && CGF.getContext().getLangOpts().OpenCLCPlusPlus)) &&
32 // "VarDecl must have global or local (in the case of OpenCL) storage!");
33 assert(!D.getType()->isReferenceType() &&
34 "Should not call EmitDeclInit on a reference!");
35
36 QualType type = D.getType();
37 LValue lv = CGF.MakeAddrLValue(DeclPtr, type);
38
39 const Expr *Init = D.getInit();
40 switch (CGF.getEvaluationKind(type)) {
41 case TEK_Scalar: {
42 CodeGenModule &CGM = CGF.CGM;
43 if (lv.isObjCStrong())
44 CGM.getObjCRuntime().EmitObjCGlobalAssign(CGF, CGF.EmitScalarExpr(Init),
45 DeclPtr, D.getTLSKind());
46 else if (lv.isObjCWeak())
47 CGM.getObjCRuntime().EmitObjCWeakAssign(CGF, CGF.EmitScalarExpr(Init),
48 DeclPtr);
49 else
50 CGF.EmitScalarInit(Init, &D, lv, false);
51 return;
52 }
53 case TEK_Complex:
54 CGF.EmitComplexExprIntoLValue(Init, lv, /*isInit*/ true);
55 return;
56 case TEK_Aggregate:
57 CGF.EmitAggExpr(Init, AggValueSlot::forLValue(lv,AggValueSlot::IsDestructed,
58 AggValueSlot::DoesNotNeedGCBarriers,
59 AggValueSlot::IsNotAliased,
60 AggValueSlot::DoesNotOverlap));
61 return;
62 }
63 llvm_unreachable("bad evaluation kind");
64}

66/// Emit code to cause the destruction of the given variable with
67/// static storage duration.
68static void EmitDeclDestroy(CodeGenFunction &CGF, const VarDecl &D,
69 ConstantAddress Addr) {
70 // Honor __attribute__((no_destroy)) and bail instead of attempting
71 // to emit a reference to a possibly nonexistent destructor, which
72 // in turn can cause a crash. This will result in a global constructor
73 // that isn't balanced out by a destructor call as intended by the
74 // attribute. This also checks for -fno-c++-static-destructors and
75 // bails even if the attribute is not present.
76 if (D.isNoDestroy(CGF.getContext()))
77 return;
78
79 CodeGenModule &CGM = CGF.CGM;
80
81 // FIXME: __attribute__((cleanup)) ?
82
83 QualType Type = D.getType();
84 QualType::DestructionKind DtorKind = Type.isDestructedType();
85
86 switch (DtorKind) {
87 case QualType::DK_none:
88 return;
89
90 case QualType::DK_cxx_destructor:
91 break;
92
93 case QualType::DK_objc_strong_lifetime:
94 case QualType::DK_objc_weak_lifetime:
95 case QualType::DK_nontrivial_c_struct:
96 // We don't care about releasing objects during process teardown.
97 assert(!D.getTLSKind() && "should have rejected this");
98 return;
99 }
100
101 llvm::FunctionCallee Func;
102 llvm::Constant *Argument;
103
104 // Special-case non-array C++ destructors, if they have the right signature.
105 // Under some ABIs, destructors return this instead of void, and cannot be
106 // passed directly to __cxa_atexit if the target does not allow this
107 // mismatch.
108 const CXXRecordDecl *Record = Type->getAsCXXRecordDecl();
109 bool CanRegisterDestructor =
110 Record && (!CGM.getCXXABI().HasThisReturn(
111 GlobalDecl(Record->getDestructor(), Dtor_Complete)) ||
112 CGM.getCXXABI().canCallMismatchedFunctionType());
113 // If __cxa_atexit is disabled via a flag, a different helper function is
114 // generated elsewhere which uses atexit instead, and it takes the destructor
115 // directly.
116 bool UsingExternalHelper = !CGM.getCodeGenOpts().CXAAtExit;
117 if (Record && (CanRegisterDestructor || UsingExternalHelper)) {
118 assert(!Record->hasTrivialDestructor());
119 CXXDestructorDecl *Dtor = Record->getDestructor();
120
121 Func = CGM.getAddrAndTypeOfCXXStructor(GlobalDecl(Dtor, Dtor_Complete));
122 Argument = llvm::ConstantExpr::getBitCast(
123 Addr.getPointer(), CGF.getTypes().ConvertType(Type)->getPointerTo(
124 CGM.getTargetCodeGenInfo().getDefaultAS()));
125
126 // Otherwise, the standard logic requires a helper function.
127 } else {
128 Func = CodeGenFunction(CGM)
129 .generateDestroyHelper(Addr, Type, CGF.getDestroyer(DtorKind),
130 CGF.needsEHCleanup(DtorKind), &D);
131 Argument = llvm::Constant::getNullValue(CGF.Int8PtrTy);
132 }
133
134 CGM.getCXXABI().registerGlobalDtor(CGF, D, Func, Argument);
135}

137/// Emit code to cause the variable at the given address to be considered as
138/// constant from this point onwards.
139static void EmitDeclInvariant(CodeGenFunction &CGF, const VarDecl &D,
140 llvm::Constant *Addr) {
141 return CGF.EmitInvariantStart(
142 Addr, CGF.getContext().getTypeSizeInChars(D.getType()));
143}

145void CodeGenFunction::EmitInvariantStart(llvm::Constant *Addr, CharUnits Size) {
146 // If the address space isn't 0, do we can't mark it as invariant
147 if (Addr->getType()->getPointerAddressSpace() != 0)
148 return;
149 // Do not emit the intrinsic if we're not optimizing.
150 if (!CGM.getCodeGenOpts().OptimizationLevel)
151 return;
152
153 // Grab the llvm.invariant.start intrinsic.
154 llvm::Intrinsic::ID InvStartID = llvm::Intrinsic::invariant_start;
155 // Overloaded address space type.
156 llvm::Type *ObjectPtr[1] = {Int8PtrTy};
157 llvm::Function *InvariantStart = CGM.getIntrinsic(InvStartID, ObjectPtr);
158
159 // Emit a call with the size in bytes of the object.
160 uint64_t Width = Size.getQuantity();
161 llvm::Value *Args[2] = { llvm::ConstantInt::getSigned(Int64Ty, Width),
162 llvm::ConstantExpr::getBitCast(Addr, Int8PtrTy)};
163 Builder.CreateCall(InvariantStart, Args);
164}

166void CodeGenFunction::EmitCXXGlobalVarDeclInit(const VarDecl &D,
167 llvm::Constant *DeclPtr,
168 bool PerformInit) {
169
170 const Expr *Init = D.getInit();
171 QualType T = D.getType();
172
173 // The address space of a static local variable (DeclPtr) may be different
174 // from the address space of the "this" argument of the constructor. In that
175 // case, we need an addrspacecast before calling the constructor.
176 //
177 // struct StructWithCtor {
178 // __device__ StructWithCtor() {...}
179 // };
180 // __device__ void foo() {
181 // __shared__ StructWithCtor s;
182 // ...
183 // }
184 //
185 // For example, in the above CUDA code, the static local variable s has a
186 // "shared" address space qualifier, but the constructor of StructWithCtor
187 // expects "this" in the "generic" address space.
188 ASTContext &Context = getContext();
189 // This is not quite right, we don't want an __intcap_t alloca to be AS200
190 // unsigned ExpectedAddrSpace = CGM.getAddressSpaceForType(T);
191 unsigned ExpectedAddrSpace =
192 Context.getTargetInfo().areAllPointersCapabilities()
193 ? CGM.getTargetCodeGenInfo().getCHERICapabilityAS()
194 : CGM.getTargetAddressSpace(T.getAddressSpace());
195 unsigned ActualAddrSpace = DeclPtr->getType()->getPointerAddressSpace();
196 if (ActualAddrSpace != ExpectedAddrSpace) {
197 llvm::Type *LTy = CGM.getTypes().ConvertTypeForMem(T);
198 llvm::PointerType *PTy = llvm::PointerType::get(LTy, ExpectedAddrSpace);
199 DeclPtr = llvm::ConstantExpr::getAddrSpaceCast(DeclPtr, PTy);
200 }
201
202 ConstantAddress DeclAddr(DeclPtr, Context.getDeclAlign(&D));
203
204 if (!T->isReferenceType()) {
205 if (getLangOpts().OpenMP && !getLangOpts().OpenMPSimd &&
206 D.hasAttr<OMPThreadPrivateDeclAttr>()) {
207 (void)CGM.getOpenMPRuntime().emitThreadPrivateVarDefinition(
208 &D, DeclAddr, D.getAttr<OMPThreadPrivateDeclAttr>()->getLocation(),
209 PerformInit, this);
210 }
211 if (PerformInit)
212 EmitDeclInit(*this, D, DeclAddr);
213 if (CGM.isTypeConstant(D.getType(), true))
214 EmitDeclInvariant(*this, D, DeclPtr);
215 else
216 EmitDeclDestroy(*this, D, DeclAddr);
217 return;
218 }
219
220 assert(PerformInit && "cannot have constant initializer which needs "
221 "destruction for reference");
222 RValue RV = EmitReferenceBindingToExpr(Init);
223 EmitStoreOfScalar(RV.getScalarVal(), DeclAddr, false, T);
224}

226/// Create a stub function, suitable for being passed to atexit,
227/// which passes the given address to the given destructor function.
228llvm::Function *CodeGenFunction::createAtExitStub(const VarDecl &VD,
229 llvm::FunctionCallee dtor,
230 llvm::Constant *addr) {
231 // Get the destructor function type, void(*)(void).
232 llvm::FunctionType *ty = llvm::FunctionType::get(CGM.VoidTy, false);
233 SmallString<256> FnName;
234 {
235 llvm::raw_svector_ostream Out(FnName);
236 CGM.getCXXABI().getMangleContext().mangleDynamicAtExitDestructor(&VD, Out);
237 }
238
239 const CGFunctionInfo &FI = CGM.getTypes().arrangeNullaryFunction();
240 llvm::Function *fn = CGM.CreateGlobalInitOrDestructFunction(
241 ty, FnName.str(), FI, VD.getLocation());
242
243 CodeGenFunction CGF(CGM);
244
245 CGF.StartFunction(GlobalDecl(&VD, DynamicInitKind::AtExit),
246 CGM.getContext().VoidTy, fn, FI, FunctionArgList());
247
248 llvm::CallInst *call = CGF.Builder.CreateCall(dtor, addr);
249
250 // Make sure the call and the callee agree on calling convention.
251 if (llvm::Function *dtorFn =
252 dyn_cast<llvm::Function>(dtor.getCallee()->stripPointerCasts()))
253 call->setCallingConv(dtorFn->getCallingConv());
254
255 CGF.FinishFunction();
256
257 return fn;
258}

269void CodeGenFunction::registerGlobalDtorWithAtExit(llvm::Constant *dtorStub) {
270 // extern "C" int atexit(void (*f)(void));
271 llvm::FunctionType *atexitTy =
272 llvm::FunctionType::get(IntTy, dtorStub->getType(), false);
273
274 llvm::FunctionCallee atexit =
275 CGM.CreateRuntimeFunction(atexitTy, "atexit", llvm::AttributeList(),
276 /*Local=*/true);
277 if (llvm::Function *atexitFn = dyn_cast<llvm::Function>(atexit.getCallee()))
278 atexitFn->setDoesNotThrow();
279
280 EmitNounwindRuntimeCall(atexit, dtorStub);
281}

297void CodeGenFunction::EmitCXXGuardedInitBranch(llvm::Value *NeedsInit,
298 llvm::BasicBlock *InitBlock,
299 llvm::BasicBlock *NoInitBlock,
300 GuardKind Kind,
301 const VarDecl *D) {
302 assert((Kind == GuardKind::TlsGuard || D) && "no guarded variable");
303
304 // A guess at how many times we will enter the initialization of a
305 // variable, depending on the kind of variable.
306 static const uint64_t InitsPerTLSVar = 1024;
307 static const uint64_t InitsPerLocalVar = 1024 * 1024;
308
309 llvm::MDNode *Weights;
310 if (Kind == GuardKind::VariableGuard && !D->isLocalVarDecl()) {
311 // For non-local variables, don't apply any weighting for now. Due to our
312 // use of COMDATs, we expect there to be at most one initialization of the
313 // variable per DSO, but we have no way to know how many DSOs will try to
314 // initialize the variable.
315 Weights = nullptr;
316 } else {
317 uint64_t NumInits;
318 // FIXME: For the TLS case, collect and use profiling information to
319 // determine a more accurate brach weight.
320 if (Kind == GuardKind::TlsGuard || D->getTLSKind())
321 NumInits = InitsPerTLSVar;
322 else
323 NumInits = InitsPerLocalVar;
324
325 // The probability of us entering the initializer is
326 // 1 / (total number of times we attempt to initialize the variable).
327 llvm::MDBuilder MDHelper(CGM.getLLVMContext());
328 Weights = MDHelper.createBranchWeights(1, NumInits - 1);
329 }
330
331 Builder.CreateCondBr(NeedsInit, InitBlock, NoInitBlock, Weights);
332}

334llvm::Function *CodeGenModule::CreateGlobalInitOrDestructFunction(
335 llvm::FunctionType *FTy, const Twine &Name, const CGFunctionInfo &FI,
336 SourceLocation Loc, bool TLS) {
337 llvm::Function *Fn =
338 llvm::Function::Create(FTy, llvm::GlobalValue::InternalLinkage,
339 Name, &getModule());
340 if (!getLangOpts().AppleKext && !TLS) {
341 // Set the section if needed.
342 if (const char *Section = getTarget().getStaticInitSectionSpecifier())
343 Fn->setSection(Section);
344 }
345
346 SetInternalFunctionAttributes(GlobalDecl(), Fn, FI);
347
348 Fn->setCallingConv(getRuntimeCC());
349
350 if (!getLangOpts().Exceptions)
351 Fn->setDoesNotThrow();
352
353 if (getLangOpts().Sanitize.has(SanitizerKind::Address) &&
354 !isInSanitizerBlacklist(SanitizerKind::Address, Fn, Loc))
355 Fn->addFnAttr(llvm::Attribute::SanitizeAddress);
356
357 if (getLangOpts().Sanitize.has(SanitizerKind::KernelAddress) &&
358 !isInSanitizerBlacklist(SanitizerKind::KernelAddress, Fn, Loc))
359 Fn->addFnAttr(llvm::Attribute::SanitizeAddress);
360
361 if (getLangOpts().Sanitize.has(SanitizerKind::HWAddress) &&
362 !isInSanitizerBlacklist(SanitizerKind::HWAddress, Fn, Loc))
363 Fn->addFnAttr(llvm::Attribute::SanitizeHWAddress);
364
365 if (getLangOpts().Sanitize.has(SanitizerKind::KernelHWAddress) &&
366 !isInSanitizerBlacklist(SanitizerKind::KernelHWAddress, Fn, Loc))
367 Fn->addFnAttr(llvm::Attribute::SanitizeHWAddress);
368
369 if (getLangOpts().Sanitize.has(SanitizerKind::Thread) &&
370 !isInSanitizerBlacklist(SanitizerKind::Thread, Fn, Loc))
371 Fn->addFnAttr(llvm::Attribute::SanitizeThread);
372
373 if (getLangOpts().Sanitize.has(SanitizerKind::Memory) &&
374 !isInSanitizerBlacklist(SanitizerKind::Memory, Fn, Loc))
375 Fn->addFnAttr(llvm::Attribute::SanitizeMemory);
376
377 if (getLangOpts().Sanitize.has(SanitizerKind::KernelMemory) &&
378 !isInSanitizerBlacklist(SanitizerKind::KernelMemory, Fn, Loc))
379 Fn->addFnAttr(llvm::Attribute::SanitizeMemory);
380
381 if (getLangOpts().Sanitize.has(SanitizerKind::SafeStack) &&
382 !isInSanitizerBlacklist(SanitizerKind::SafeStack, Fn, Loc))
383 Fn->addFnAttr(llvm::Attribute::SafeStack);
384
385 if (getLangOpts().Sanitize.has(SanitizerKind::ShadowCallStack) &&
386 !isInSanitizerBlacklist(SanitizerKind::ShadowCallStack, Fn, Loc))
387 Fn->addFnAttr(llvm::Attribute::ShadowCallStack);
388
389 auto RASignKind = getCodeGenOpts().getSignReturnAddress();
390 if (RASignKind != CodeGenOptions::SignReturnAddressScope::None) {
391 Fn->addFnAttr("sign-return-address",
392 RASignKind == CodeGenOptions::SignReturnAddressScope::All
393 ? "all"
394 : "non-leaf");
395 auto RASignKey = getCodeGenOpts().getSignReturnAddressKey();
396 Fn->addFnAttr("sign-return-address-key",
397 RASignKey == CodeGenOptions::SignReturnAddressKeyValue::AKey
398 ? "a_key"
399 : "b_key");
400 }
401
402 if (getCodeGenOpts().BranchTargetEnforcement)
403 Fn->addFnAttr("branch-target-enforcement");
404
405 return Fn;
406}

426void
427CodeGenModule::EmitCXXGlobalVarDeclInitFunc(const VarDecl *D,
428 llvm::GlobalVariable *Addr,
429 bool PerformInit) {
430
431 // According to E.2.3.1 in CUDA-7.5 Programming guide: __device__,
432 // __constant__ and __shared__ variables defined in namespace scope,
433 // that are of class type, cannot have a non-empty constructor. All
434 // the checks have been done in Sema by now. Whatever initializers
435 // are allowed are empty and we just need to ignore them here.
436 if (getLangOpts().CUDA && getLangOpts().CUDAIsDevice &&
437 (D->hasAttr<CUDADeviceAttr>() || D->hasAttr<CUDAConstantAttr>() ||
438 D->hasAttr<CUDASharedAttr>()))
439 return;
440
441 if (getLangOpts().OpenMP &&
442 getOpenMPRuntime().emitDeclareTargetVarDefinition(D, Addr, PerformInit))
443 return;
444
445 // Check if we've already initialized this decl.
446 auto I = DelayedCXXInitPosition.find(D);
447 if (I != DelayedCXXInitPosition.end() && I->second == ~0U)
448 return;
449
450 llvm::FunctionType *FTy = llvm::FunctionType::get(VoidTy, false);
451 SmallString<256> FnName;
452 {
453 llvm::raw_svector_ostream Out(FnName);
454 getCXXABI().getMangleContext().mangleDynamicInitializer(D, Out);
455 }
456
457 // Create a variable initialization function.
458 llvm::Function *Fn =
459 CreateGlobalInitOrDestructFunction(FTy, FnName.str(),
460 getTypes().arrangeNullaryFunction(),
461 D->getLocation());
462
463 auto *ISA = D->getAttr<InitSegAttr>();
464 CodeGenFunction(*this).GenerateCXXGlobalVarDeclInitFunc(Fn, D, Addr,
465 PerformInit);
466
467 llvm::GlobalVariable *COMDATKey =
468 supportsCOMDAT() && D->isExternallyVisible() ? Addr : nullptr;
469
470 if (D->getTLSKind()) {
471 // FIXME: Should we support init_priority for thread_local?
472 // FIXME: We only need to register one __cxa_thread_atexit function for the
473 // entire TU.
474 CXXThreadLocalInits.push_back(Fn);
475 CXXThreadLocalInitVars.push_back(D);
476 } else if (PerformInit && ISA) {
477 EmitPointerToInitFunc(D, Addr, Fn, ISA);
478 } else if (auto *IPA = D->getAttr<InitPriorityAttr>()) {
479 OrderGlobalInits Key(IPA->getPriority(), PrioritizedCXXGlobalInits.size());
480 PrioritizedCXXGlobalInits.push_back(std::make_pair(Key, Fn));
481 } else if (isTemplateInstantiation(D->getTemplateSpecializationKind()) ||
482 getContext().GetGVALinkageForVariable(D) == GVA_DiscardableODR) {
483 // C++ [basic.start.init]p2:
484 // Definitions of explicitly specialized class template static data
485 // members have ordered initialization. Other class template static data
486 // members (i.e., implicitly or explicitly instantiated specializations)
487 // have unordered initialization.
488 //
489 // As a consequence, we can put them into their own llvm.global_ctors entry.
490 //
491 // If the global is externally visible, put the initializer into a COMDAT
492 // group with the global being initialized. On most platforms, this is a
493 // minor startup time optimization. In the MS C++ ABI, there are no guard
494 // variables, so this COMDAT key is required for correctness.
495 AddGlobalCtor(Fn, 65535, COMDATKey);
496 if (getTarget().getCXXABI().isMicrosoft() && COMDATKey) {
497 // In The MS C++, MS add template static data member in the linker
498 // drective.
499 addUsedGlobal(COMDATKey);
500 }
501 } else if (D->hasAttr<SelectAnyAttr>()) {
502 // SelectAny globals will be comdat-folded. Put the initializer into a
503 // COMDAT group associated with the global, so the initializers get folded
504 // too.
505 AddGlobalCtor(Fn, 65535, COMDATKey);
506 } else {
507 I = DelayedCXXInitPosition.find(D); // Re-do lookup in case of re-hash.
508 if (I == DelayedCXXInitPosition.end()) {
509 CXXGlobalInits.push_back(Fn);
510 } else if (I->second != ~0U) {
511 assert(I->second < CXXGlobalInits.size() &&
512 CXXGlobalInits[I->second] == nullptr);
513 CXXGlobalInits[I->second] = Fn;
514 }
515 }
516
517 // Remember that we already emitted the initializer for this global.
518 DelayedCXXInitPosition[D] = ~0U;
519}

530void
531CodeGenModule::EmitCXXGlobalInitFunc() {
532 while (!CXXGlobalInits.empty() && !CXXGlobalInits.back())
533 CXXGlobalInits.pop_back();
534
535 if (CXXGlobalInits.empty() && PrioritizedCXXGlobalInits.empty())
536 return;
537
538 llvm::FunctionType *FTy = llvm::FunctionType::get(VoidTy, false);
539 const CGFunctionInfo &FI = getTypes().arrangeNullaryFunction();
540
541 // Create our global initialization function.
542 if (!PrioritizedCXXGlobalInits.empty()) {
543 SmallVector<llvm::Function *, 8> LocalCXXGlobalInits;
544 llvm::array_pod_sort(PrioritizedCXXGlobalInits.begin(),
545 PrioritizedCXXGlobalInits.end());
546 // Iterate over "chunks" of ctors with same priority and emit each chunk
547 // into separate function. Note - everything is sorted first by priority,
548 // second - by lex order, so we emit ctor functions in proper order.
549 for (SmallVectorImpl<GlobalInitData >::iterator
550 I = PrioritizedCXXGlobalInits.begin(),
551 E = PrioritizedCXXGlobalInits.end(); I != E; ) {
552 SmallVectorImpl<GlobalInitData >::iterator
553 PrioE = std::upper_bound(I + 1, E, *I, GlobalInitPriorityCmp());
554
555 LocalCXXGlobalInits.clear();
556 unsigned Priority = I->first.priority;
557 // Compute the function suffix from priority. Prepend with zeroes to make
558 // sure the function names are also ordered as priorities.
559 std::string PrioritySuffix = llvm::utostr(Priority);
560 // Priority is always <= 65535 (enforced by sema).
561 PrioritySuffix = std::string(6-PrioritySuffix.size(), '0')+PrioritySuffix;
562 llvm::Function *Fn = CreateGlobalInitOrDestructFunction(
563 FTy, "_GLOBAL__I_" + PrioritySuffix, FI);
564
565 for (; I < PrioE; ++I)
566 LocalCXXGlobalInits.push_back(I->second);
567
568 CodeGenFunction(*this).GenerateCXXGlobalInitFunc(Fn, LocalCXXGlobalInits);
569 AddGlobalCtor(Fn, Priority);
570 }
571 PrioritizedCXXGlobalInits.clear();
572 }
573
574 // Include the filename in the symbol name. Including "sub_" matches gcc and
575 // makes sure these symbols appear lexicographically behind the symbols with
576 // priority emitted above.
577 SmallString<128> FileName = llvm::sys::path::filename(getModule().getName());
578 if (FileName.empty())
579 FileName = "<null>";
580
581 for (size_t i = 0; i < FileName.size(); ++i) {
582 // Replace everything that's not [a-zA-Z0-9._] with a _. This set happens
583 // to be the set of C preprocessing numbers.
584 if (!isPreprocessingNumberBody(FileName[i]))
585 FileName[i] = '_';
586 }
587
588 llvm::Function *Fn = CreateGlobalInitOrDestructFunction(
589 FTy, llvm::Twine("_GLOBAL__sub_I_", FileName), FI);
590
591 CodeGenFunction(*this).GenerateCXXGlobalInitFunc(Fn, CXXGlobalInits);
592 AddGlobalCtor(Fn);
593
594 // In OpenCL global init functions must be converted to kernels in order to
595 // be able to launch them from the host.
596 // FIXME: Some more work might be needed to handle destructors correctly.
597 // Current initialization function makes use of function pointers callbacks.
598 // We can't support function pointers especially between host and device.
599 // However it seems global destruction has little meaning without any
600 // dynamic resource allocation on the device and program scope variables are
601 // destroyed by the runtime when program is released.
602 if (getLangOpts().OpenCL) {
603 GenOpenCLArgMetadata(Fn);
604 Fn->setCallingConv(llvm::CallingConv::SPIR_KERNEL);
605 }
606
607 CXXGlobalInits.clear();
608}

612 return;

617 const CGFunctionInfo &FI = getTypes().arrangeNullaryFunction();

625/// Emit the code necessary to initialize the given global variable.
626void CodeGenFunction::GenerateCXXGlobalVarDeclInitFunc(llvm::Function *Fn,
627 const VarDecl *D,
628 llvm::GlobalVariable *Addr,
629 bool PerformInit) {
630 // Check if we need to emit debug info for variable initializer.
631 if (D->hasAttr<NoDebugAttr>())
632 DebugInfo = nullptr; // disable debug info indefinitely for this function
633
634 CurEHLocation = D->getBeginLoc();
635
636 StartFunction(GlobalDecl(D, DynamicInitKind::Initializer),
637 getContext().VoidTy, Fn, getTypes().arrangeNullaryFunction(),
638 FunctionArgList(), D->getLocation(),
639 D->getInit()->getExprLoc());
640
641 // Use guarded initialization if the global variable is weak. This
642 // occurs for, e.g., instantiated static data members and
643 // definitions explicitly marked weak.
644 //
645 // Also use guarded initialization for a variable with dynamic TLS and
646 // unordered initialization. (If the initialization is ordered, the ABI
647 // layer will guard the whole-TU initialization for us.)
648 if (Addr->hasWeakLinkage() || Addr->hasLinkOnceLinkage() ||
649 (D->getTLSKind() == VarDecl::TLS_Dynamic &&
650 isTemplateInstantiation(D->getTemplateSpecializationKind()))) {
651 EmitCXXGuardedInit(*D, Addr, PerformInit);
652 } else {
653 EmitCXXGlobalVarDeclInit(*D, Addr, PerformInit);
654 }
655
656 FinishFunction();
657}

659void
660CodeGenFunction::GenerateCXXGlobalInitFunc(llvm::Function *Fn,
661 ArrayRef<llvm::Function *> Decls,
662 ConstantAddress Guard) {
663 {
664 auto NL = ApplyDebugLocation::CreateEmpty(*this);
665 StartFunction(GlobalDecl(), getContext().VoidTy, Fn,
666 getTypes().arrangeNullaryFunction(), FunctionArgList());
667 // Emit an artificial location for this function.
668 auto AL = ApplyDebugLocation::CreateArtificial(*this);
669
670 llvm::BasicBlock *ExitBlock = nullptr;
671 if (Guard.isValid()) {
672 // If we have a guard variable, check whether we've already performed
673 // these initializations. This happens for TLS initialization functions.
674 llvm::Value *GuardVal = Builder.CreateLoad(Guard);
675 llvm::Value *Uninit = Builder.CreateIsNull(GuardVal,
676 "guard.uninitialized");
677 llvm::BasicBlock *InitBlock = createBasicBlock("init");
678 ExitBlock = createBasicBlock("exit");
679 EmitCXXGuardedInitBranch(Uninit, InitBlock, ExitBlock,
680 GuardKind::TlsGuard, nullptr);
681 EmitBlock(InitBlock);
682 // Mark as initialized before initializing anything else. If the
683 // initializers use previously-initialized thread_local vars, that's
684 // probably supposed to be OK, but the standard doesn't say.
685 Builder.CreateStore(llvm::ConstantInt::get(GuardVal->getType(),1), Guard);
686
687 // The guard variable can't ever change again.
688 EmitInvariantStart(
689 Guard.getPointer(),
690 CharUnits::fromQuantity(
691 CGM.getDataLayout().getTypeAllocSize(GuardVal->getType())));
692 }
693
694 RunCleanupsScope Scope(*this);
695
696 // When building in Objective-C++ ARC mode, create an autorelease pool
697 // around the global initializers.
698 if (getLangOpts().ObjCAutoRefCount && getLangOpts().CPlusPlus) {
699 llvm::Value *token = EmitObjCAutoreleasePoolPush();
700 EmitObjCAutoreleasePoolCleanup(token);
701 }
702
703 for (unsigned i = 0, e = Decls.size(); i != e; ++i)
704 if (Decls[i])
705 EmitRuntimeCall(Decls[i]);
706
707 Scope.ForceCleanup();
708
709 if (ExitBlock) {
710 Builder.CreateBr(ExitBlock);
711 EmitBlock(ExitBlock);
712 }
713 }
714
715 FinishFunction();
716}

718void CodeGenFunction::GenerateCXXGlobalDtorsFunc(
719 llvm::Function *Fn,
720 const std::vector<std::tuple<llvm::FunctionType *, llvm::WeakTrackingVH,
721 llvm::Constant *>> &DtorsAndObjects) {
722 {
723 auto NL = ApplyDebugLocation::CreateEmpty(*this);
724 StartFunction(GlobalDecl(), getContext().VoidTy, Fn,
725 getTypes().arrangeNullaryFunction(), FunctionArgList());
726 // Emit an artificial location for this function.
727 auto AL = ApplyDebugLocation::CreateArtificial(*this);
728
729 // Emit the dtors, in reverse order from construction.
730 for (unsigned i = 0, e = DtorsAndObjects.size(); i != e; ++i) {
731 llvm::FunctionType *CalleeTy;
732 llvm::Value *Callee;
733 llvm::Constant *Arg;
734 std::tie(CalleeTy, Callee, Arg) = DtorsAndObjects[e - i - 1];
735 llvm::CallInst *CI = Builder.CreateCall(CalleeTy, Callee, Arg);
736 // Make sure the call and the callee agree on calling convention.
737 if (llvm::Function *F = dyn_cast<llvm::Function>(Callee))
738 CI->setCallingConv(F->getCallingConv());
739 }
740 }
741
742 FinishFunction();
743}

745/// generateDestroyHelper - Generates a helper function which, when
746/// invoked, destroys the given object. The address of the object
747/// should be in global memory.
748llvm::Function *CodeGenFunction::generateDestroyHelper(
749 Address addr, QualType type, Destroyer *destroyer,
750 bool useEHCleanupForArray, const VarDecl *VD) {
751 FunctionArgList args;
752 ImplicitParamDecl Dst(getContext(), getContext().VoidPtrTy,
753 ImplicitParamDecl::Other);
754 args.push_back(&Dst);
755
756 const CGFunctionInfo &FI =
757 CGM.getTypes().arrangeBuiltinFunctionDeclaration(getContext().VoidTy, args);
758 llvm::FunctionType *FTy = CGM.getTypes().GetFunctionType(FI);
759 llvm::Function *fn = CGM.CreateGlobalInitOrDestructFunction(
760 FTy, "__cxx_global_array_dtor", FI, VD->getLocation());
761
762 CurEHLocation = VD->getBeginLoc();
763
764 StartFunction(VD, getContext().VoidTy, fn, FI, args);
765
766 emitDestroy(addr, type, destroyer, useEHCleanupForArray);
767
768 FinishFunction();
769
770 return fn;
771}
