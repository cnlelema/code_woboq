47/// shouldEmitLifetimeMarkers - Decide whether we need emit the life-time
48/// markers.
49static bool shouldEmitLifetimeMarkers(const CodeGenOptions &CGOpts,
50 const LangOptions &LangOpts) {
51 if (CGOpts.DisableLifetimeMarkers)
52 return false;
53
54 // Disable lifetime markers in msan builds.
55 // FIXME: Remove this when msan works with lifetime markers.
56 if (LangOpts.Sanitize.has(SanitizerKind::Memory))
57 return false;
58
59 // Asan uses markers for use-after-scope checks.
60 if (CGOpts.SanitizeAddressUseAfterScope)
61 return true;
62
63 // For now, only in optimized builds.
64 return CGOpts.OptimizationLevel != 0;
65}

135CharUnits CodeGenFunction::getNaturalTypeAlignment(QualType T,
136 LValueBaseInfo *BaseInfo,
137 TBAAAccessInfo *TBAAInfo,
138 bool forPointeeType) {
139 if (TBAAInfo)
140 *TBAAInfo = CGM.getTBAAAccessInfo(T);
141
142 // Honor alignment typedef attributes even on incomplete types.
143 // We also honor them straight for C++ class types, even as pointees;
144 // there's an expressivity gap here.
145 if (auto TT = T->getAs<TypedefType>()) {
146 if (auto Align = TT->getDecl()->getMaxAlignment()) {
147 if (BaseInfo)
148 *BaseInfo = LValueBaseInfo(AlignmentSource::AttributedType);
149 return getContext().toCharUnitsFromBits(Align);
150 }
151 }
152
153 if (BaseInfo)
154 *BaseInfo = LValueBaseInfo(AlignmentSource::Type);
155
156 CharUnits Alignment;
157 if (T->isIncompleteType()) {
158 Alignment = CharUnits::One(); // Shouldn't be used, but pessimistic is best.
159 } else {
160 // For C++ class pointees, we don't know whether we're pointing at a
161 // base or a complete object, so we generally need to use the
162 // non-virtual alignment.
163 const CXXRecordDecl *RD;
164 if (forPointeeType && (RD = T->getAsCXXRecordDecl())) {
165 Alignment = CGM.getClassPointerAlignment(RD);
166 } else {
167 Alignment = getContext().getTypeAlignInChars(T);
168 if (T.getQualifiers().hasUnaligned())
169 Alignment = CharUnits::One();
170 }
171
172 // Cap to the global maximum type alignment unless the alignment
173 // was somehow explicit on the type.
174 if (unsigned MaxAlign = getLangOpts().MaxTypeAlign) {
175 if (Alignment.getQuantity() > MaxAlign &&
176 !getContext().isAlignmentRequired(T))
177 Alignment = CharUnits::fromQuantity(MaxAlign);
178 }
179 }
180 return Alignment;
181}

215TypeEvaluationKind CodeGenFunction::getEvaluationKind(QualType type) {
216 type = type.getCanonicalType();
217 while (true) {
218 switch (type->getTypeClass()) {
219#define TYPE(name, parent)
220#define ABSTRACT_TYPE(name, parent)
221#define NON_CANONICAL_TYPE(name, parent) case Type::name:
222#define DEPENDENT_TYPE(name, parent) case Type::name:
223#define NON_CANONICAL_UNLESS_DEPENDENT_TYPE(name, parent) case Type::name:
224#include "clang/AST/TypeNodes.def"
225 llvm_unreachable("non-canonical or dependent type in IR-generation");
226
227 case Type::Auto:
228 case Type::DeducedTemplateSpecialization:
229 llvm_unreachable("undeduced type in IR-generation");
230
231 // Various scalar types.
232 case Type::Builtin:
233 case Type::Pointer:
234 case Type::BlockPointer:
235 case Type::LValueReference:
236 case Type::RValueReference:
237 case Type::MemberPointer:
238 case Type::Vector:
239 case Type::ExtVector:
240 case Type::FunctionProto:
241 case Type::FunctionNoProto:
242 case Type::Enum:
243 case Type::ObjCObjectPointer:
244 case Type::Pipe:
245 return TEK_Scalar;
246
247 // Complexes.
248 case Type::Complex:
249 return TEK_Complex;
250
251 // Arrays, records, and Objective-C objects.
252 case Type::ConstantArray:
253 case Type::IncompleteArray:
254 case Type::VariableArray:
255 case Type::Record:
256 case Type::ObjCObject:
257 case Type::ObjCInterface:
258 return TEK_Aggregate;
259
260 // We operate on atomic values according to their underlying type.
261 case Type::Atomic:
262 type = cast<AtomicType>(type)->getValueType();
263 continue;
264 }
265 llvm_unreachable("unknown type kind!");
266 }
267}

269llvm::DebugLoc CodeGenFunction::EmitReturnBlock() {
270 // For cleanliness, we try to avoid emitting the return block for
271 // simple cases.
272 llvm::BasicBlock *CurBB = Builder.GetInsertBlock();
273
274 if (CurBB) {
275 assert(!CurBB->getTerminator() && "Unexpected terminated block.");
276
277 // We have a valid insert point, reuse it if it is empty or there are no
278 // explicit jumps to the return block.
279 if (CurBB->empty() || ReturnBlock.getBlock()->use_empty()) {
280 ReturnBlock.getBlock()->replaceAllUsesWith(CurBB);
281 delete ReturnBlock.getBlock();
282 ReturnBlock = JumpDest();
283 } else
284 EmitBlock(ReturnBlock.getBlock());
285 return llvm::DebugLoc();
286 }
287
288 // Otherwise, if the return block is the target of a single direct
289 // branch then we can just put the code in that block instead. This
290 // cleans up functions which started with a unified return block.
291 if (ReturnBlock.getBlock()->hasOneUse()) {
292 llvm::BranchInst *BI =
293 dyn_cast<llvm::BranchInst>(*ReturnBlock.getBlock()->user_begin());
294 if (BI && BI->isUnconditional() &&
295 BI->getSuccessor(0) == ReturnBlock.getBlock()) {
296 // Record/return the DebugLoc of the simple 'return' expression to be used
297 // later by the actual 'ret' instruction.
298 llvm::DebugLoc Loc = BI->getDebugLoc();
299 Builder.SetInsertPoint(BI->getParent());
300 BI->eraseFromParent();
301 delete ReturnBlock.getBlock();
302 ReturnBlock = JumpDest();
303 return Loc;
304 }
305 }
306
307 // FIXME: We are at an unreachable point, there is no reason to emit the block
308 // unless it has uses. However, we still need a place to put the debug
309 // region.end for now.
310
311 EmitBlock(ReturnBlock.getBlock());
312 return llvm::DebugLoc();
313}

315static void EmitIfUsed(CodeGenFunction &CGF, llvm::BasicBlock *BB) {
316 if (!BB) return;
317 if (!BB->use_empty())
318 return CGF.CurFn->getBasicBlockList().push_back(BB);
319 delete BB;
320}

322void CodeGenFunction::FinishFunction(SourceLocation EndLoc) {
323 assert(BreakContinueStack.empty() &&
324 "mismatched push/pop in break/continue stack!");
325
326 bool OnlySimpleReturnStmts = NumSimpleReturnExprs > 0
327 && NumSimpleReturnExprs == NumReturnExprs
328 && ReturnBlock.getBlock()->use_empty();
329 // Usually the return expression is evaluated before the cleanup
330 // code. If the function contains only a simple return statement,
331 // such as a constant, the location before the cleanup code becomes
332 // the last useful breakpoint in the function, because the simple
333 // return expression will be evaluated after the cleanup code. To be
334 // safe, set the debug location for cleanup code to the location of
335 // the return statement. Otherwise the cleanup code should be at the
336 // end of the function's lexical scope.
337 //
338 // If there are multiple branches to the return block, the branch
339 // instructions will get the location of the return statements and
340 // all will be fine.
341 if (CGDebugInfo *DI = getDebugInfo()) {
342 if (OnlySimpleReturnStmts)
343 DI->EmitLocation(Builder, LastStopPoint);
344 else
345 DI->EmitLocation(Builder, EndLoc);
346 }
347
348 // Pop any cleanups that might have been associated with the
349 // parameters. Do this in whatever block we're currently in; it's
350 // important to do this before we enter the return block or return
351 // edges will be *really* confused.
352 bool HasCleanups = EHStack.stable_begin() != PrologueCleanupDepth;
353 bool HasOnlyLifetimeMarkers =
354 HasCleanups && EHStack.containsOnlyLifetimeMarkers(PrologueCleanupDepth);
355 bool EmitRetDbgLoc = !HasCleanups || HasOnlyLifetimeMarkers;
356 if (HasCleanups) {
357 // Make sure the line table doesn't jump back into the body for
358 // the ret after it's been at EndLoc.
359 if (CGDebugInfo *DI = getDebugInfo())
360 if (OnlySimpleReturnStmts)
361 DI->EmitLocation(Builder, EndLoc);
362
363 PopCleanupBlocks(PrologueCleanupDepth);
364 }
365
366 // Emit function epilog (to return).
367 llvm::DebugLoc Loc = EmitReturnBlock();
368
369 if (ShouldInstrumentFunction()) {
370 if (CGM.getCodeGenOpts().InstrumentFunctions)
371 CurFn->addFnAttr("instrument-function-exit", "__cyg_profile_func_exit");
372 if (CGM.getCodeGenOpts().InstrumentFunctionsAfterInlining)
373 CurFn->addFnAttr("instrument-function-exit-inlined",
374 "__cyg_profile_func_exit");
375 }
376
377 // Emit debug descriptor for function end.
378 if (CGDebugInfo *DI = getDebugInfo())
379 DI->EmitFunctionEnd(Builder, CurFn);
380
381 // Reset the debug location to that of the simple 'return' expression, if any
382 // rather than that of the end of the function's scope '}'.
383 ApplyDebugLocation AL(*this, Loc);
384 EmitFunctionEpilog(*CurFnInfo, EmitRetDbgLoc, EndLoc);
385 EmitEndEHSpec(CurCodeDecl);
386
387 assert(EHStack.empty() &&
388 "did not remove all scopes from cleanup stack!");
389
390 // If someone did an indirect goto, emit the indirect goto block at the end of
391 // the function.
392 if (IndirectBranch) {
393 EmitBlock(IndirectBranch->getParent());
394 Builder.ClearInsertionPoint();
395 }
396
397 // If some of our locals escaped, insert a call to llvm.localescape in the
398 // entry block.
399 if (!EscapedLocals.empty()) {
400 // Invert the map from local to index into a simple vector. There should be
401 // no holes.
402 SmallVector<llvm::Value *, 4> EscapeArgs;
403 EscapeArgs.resize(EscapedLocals.size());
404 for (auto &Pair : EscapedLocals)
405 EscapeArgs[Pair.second] = Pair.first;
406 llvm::Function *FrameEscapeFn = llvm::Intrinsic::getDeclaration(
407 &CGM.getModule(), llvm::Intrinsic::localescape);
408 CGBuilderTy(*this, AllocaInsertPt).CreateCall(FrameEscapeFn, EscapeArgs);
409 }
410
411 // Remove the AllocaInsertPt instruction, which is just a convenience for us.
412 llvm::Instruction *Ptr = AllocaInsertPt;
413 AllocaInsertPt = nullptr;
414 Ptr->eraseFromParent();
415
416 // If someone took the address of a label but never did an indirect goto, we
417 // made a zero entry PHI node, which is illegal, zap it now.
418 if (IndirectBranch) {
419 llvm::PHINode *PN = cast<llvm::PHINode>(IndirectBranch->getAddress());
420 if (PN->getNumIncomingValues() == 0) {
421 PN->replaceAllUsesWith(llvm::UndefValue::get(PN->getType()));
422 PN->eraseFromParent();
423 }
424 }
425
426 EmitIfUsed(*this, EHResumeBlock);
427 EmitIfUsed(*this, TerminateLandingPad);
428 EmitIfUsed(*this, TerminateHandler);
429 EmitIfUsed(*this, UnreachableBlock);
430
431 for (const auto &FuncletAndParent : TerminateFunclets)
432 EmitIfUsed(*this, FuncletAndParent.second);
433
434 if (CGM.getCodeGenOpts().EmitDeclMetadata)
435 EmitDeclMetadata();
436
437 for (SmallVectorImpl<std::pair<llvm::Instruction *, llvm::Value *> >::iterator
438 I = DeferredReplacements.begin(),
439 E = DeferredReplacements.end();
440 I != E; ++I) {
441 I->first->replaceAllUsesWith(I->second);
442 I->first->eraseFromParent();
443 }
444
445 // Eliminate CleanupDestSlot alloca by replacing it with SSA values and
446 // PHIs if the current function is a coroutine. We don't do it for all
447 // functions as it may result in slight increase in numbers of instructions
448 // if compiled with no optimizations. We do it for coroutine as the lifetime
449 // of CleanupDestSlot alloca make correct coroutine frame building very
450 // difficult.
451 if (NormalCleanupDest.isValid() && isCoroutine()) {
452 llvm::DominatorTree DT(*CurFn);
453 llvm::PromoteMemToReg(
454 cast<llvm::AllocaInst>(NormalCleanupDest.getPointer()), DT);
455 NormalCleanupDest = Address::invalid();
456 }
457
458 // Scan function arguments for vector width.
459 for (llvm::Argument &A : CurFn->args())
460 if (auto *VT = dyn_cast<llvm::VectorType>(A.getType()))
461 LargestVectorWidth = std::max(LargestVectorWidth,
462 VT->getPrimitiveSizeInBits());
463
464 // Update vector width based on return type.
465 if (auto *VT = dyn_cast<llvm::VectorType>(CurFn->getReturnType()))
466 LargestVectorWidth = std::max(LargestVectorWidth,
467 VT->getPrimitiveSizeInBits());
468
469 // Add the required-vector-width attribute. This contains the max width from:
470 // 1. min-vector-width attribute used in the source program.
471 // 2. Any builtins used that have a vector width specified.
472 // 3. Values passed in and out of inline assembly.
473 // 4. Width of vector arguments and return types for this function.
474 // 5. Width of vector aguments and return types for functions called by this
475 // function.
476 CurFn->addFnAttr("min-legal-vector-width", llvm::utostr(LargestVectorWidth));
477
478 // If we generated an unreachable return block, delete it now.
479 if (ReturnBlock.isValid() && ReturnBlock.getBlock()->use_empty()) {
480 Builder.ClearInsertionPoint();
481 ReturnBlock.getBlock()->eraseFromParent();
482 }
483 if (ReturnValue.isValid()) {
484 auto *RetAlloca = dyn_cast<llvm::AllocaInst>(ReturnValue.getPointer());
485 if (RetAlloca && RetAlloca->use_empty()) {
486 RetAlloca->eraseFromParent();
487 ReturnValue = Address::invalid();
488 }
489 }
490}

526llvm::Constant *
527CodeGenFunction::EncodeAddrForUseInPrologue(llvm::Function *F,
528 llvm::Constant *Addr) {
529 // Addresses stored in prologue data can't require run-time fixups and must
530 // be PC-relative. Run-time fixups are undesirable because they necessitate
531 // writable text segments, which are unsafe. And absolute addresses are
532 // undesirable because they break PIE mode.
533
534 // Add a layer of indirection through a private global. Taking its address
535 // won't result in a run-time fixup, even if Addr has linkonce_odr linkage.
536 auto *GV = new llvm::GlobalVariable(CGM.getModule(), Addr->getType(),
537 /*isConstant=*/true,
538 llvm::GlobalValue::PrivateLinkage, Addr);
539
540 // Create a PC-relative address.
541 auto *GOTAsInt = llvm::ConstantExpr::getPtrToInt(GV, IntPtrTy);
542 auto *FuncAsInt = llvm::ConstantExpr::getPtrToInt(F, IntPtrTy);
543 auto *PCRelAsInt = llvm::ConstantExpr::getSub(GOTAsInt, FuncAsInt);
544 return (IntPtrTy == Int32Ty)
545 ? PCRelAsInt
546 : llvm::ConstantExpr::getTrunc(PCRelAsInt, Int32Ty);
547}

549llvm::Value *
550CodeGenFunction::DecodeAddrUsedInPrologue(llvm::Value *F,
551 llvm::Value *EncodedAddr) {
552 // Reconstruct the address of the global.
553 auto *PCRelAsInt = Builder.CreateSExt(EncodedAddr, IntPtrTy);
554 auto *FuncAsInt = Builder.CreatePtrToInt(F, IntPtrTy, "func_addr.int");
555 auto *GOTAsInt = Builder.CreateAdd(PCRelAsInt, FuncAsInt, "global_addr.int");
556 auto *GOTAddr = Builder.CreateIntToPtr(GOTAsInt, Int8PtrPtrTy, "global_addr");
557
558 // Load the original pointer through the global.
559 return Builder.CreateLoad(Address(GOTAddr, getPointerAlign()),
560 "decoded_addr");
561}

563void CodeGenFunction::EmitOpenCLKernelMetadata(const FunctionDecl *FD,
564 llvm::Function *Fn)
565{
566 if (!FD->hasAttr<OpenCLKernelAttr>())
567 return;
568
569 llvm::LLVMContext &Context = getLLVMContext();
570
571 CGM.GenOpenCLArgMetadata(Fn, FD, this);
572
573 if (const VecTypeHintAttr *A = FD->getAttr<VecTypeHintAttr>()) {
574 QualType HintQTy = A->getTypeHint();
575 const ExtVectorType *HintEltQTy = HintQTy->getAs<ExtVectorType>();
576 bool IsSignedInteger =
577 HintQTy->isSignedIntegerType() ||
578 (HintEltQTy && HintEltQTy->getElementType()->isSignedIntegerType());
579 llvm::Metadata *AttrMDArgs[] = {
580 llvm::ConstantAsMetadata::get(llvm::UndefValue::get(
581 CGM.getTypes().ConvertType(A->getTypeHint()))),
582 llvm::ConstantAsMetadata::get(llvm::ConstantInt::get(
583 llvm::IntegerType::get(Context, 32),
584 llvm::APInt(32, (uint64_t)(IsSignedInteger ? 1 : 0))))};
585 Fn->setMetadata("vec_type_hint", llvm::MDNode::get(Context, AttrMDArgs));
586 }
587
588 if (const WorkGroupSizeHintAttr *A = FD->getAttr<WorkGroupSizeHintAttr>()) {
589 llvm::Metadata *AttrMDArgs[] = {
590 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getXDim())),
591 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getYDim())),
592 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getZDim()))};
593 Fn->setMetadata("work_group_size_hint", llvm::MDNode::get(Context, AttrMDArgs));
594 }
595
596 if (const ReqdWorkGroupSizeAttr *A = FD->getAttr<ReqdWorkGroupSizeAttr>()) {
597 llvm::Metadata *AttrMDArgs[] = {
598 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getXDim())),
599 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getYDim())),
600 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getZDim()))};
601 Fn->setMetadata("reqd_work_group_size", llvm::MDNode::get(Context, AttrMDArgs));
602 }
603
604 if (const OpenCLIntelReqdSubGroupSizeAttr *A =
605 FD->getAttr<OpenCLIntelReqdSubGroupSizeAttr>()) {
606 llvm::Metadata *AttrMDArgs[] = {
607 llvm::ConstantAsMetadata::get(Builder.getInt32(A->getSubGroupSize()))};
608 Fn->setMetadata("intel_reqd_sub_group_size",
609 llvm::MDNode::get(Context, AttrMDArgs));
610 }
611}

613/// Determine whether the function F ends with a return stmt.
614static bool endsWithReturn(const Decl* F) {
615 const Stmt *Body = nullptr;
616 if (auto *FD = dyn_cast_or_null<FunctionDecl>(F))
617 Body = FD->getBody();
618 else if (auto *OMD = dyn_cast_or_null<ObjCMethodDecl>(F))
619 Body = OMD->getBody();
620
621 if (auto *CS = dyn_cast_or_null<CompoundStmt>(Body)) {
622 auto LastStmt = CS->body_rbegin();
623 if (LastStmt != CS->body_rend())
624 return isa<ReturnStmt>(*LastStmt);
625 }
626 return false;
627}

636static bool matchesStlAllocatorFn(const Decl *D, const ASTContext &Ctx) {
637 auto *MD = dyn_cast_or_null<CXXMethodDecl>(D);
638 if (!MD || !MD->getDeclName().getAsIdentifierInfo() ||
639 !MD->getDeclName().getAsIdentifierInfo()->isStr("allocate") ||
640 (MD->getNumParams() != 1 && MD->getNumParams() != 2))
641 return false;
642
643 if (MD->parameters()[0]->getType().getCanonicalType() != Ctx.getSizeType())
644 return false;
645
646 if (MD->getNumParams() == 2) {
647 auto *PT = MD->parameters()[1]->getType()->getAs<PointerType>();
648 if (!PT || !PT->isVoidPointerType() ||
649 !PT->getPointeeType().isConstQualified())
650 return false;
651 }
652
653 return true;
654}

665void CodeGenFunction::StartFunction(GlobalDecl GD,
666 QualType RetTy,
667 llvm::Function *Fn,
668 const CGFunctionInfo &FnInfo,
669 const FunctionArgList &Args,
670 SourceLocation Loc,
671 SourceLocation StartLoc) {
672 assert(!CurFn &&
673 "Do not use a CodeGenFunction object for more than one function");
674
675 const Decl *D = GD.getDecl();
676
677 DidCallStackSave = false;
678 CurCodeDecl = D;
679 if (const auto *FD = dyn_cast_or_null<FunctionDecl>(D))
680 if (FD->usesSEHTry())
681 CurSEHParent = FD;
682 CurFuncDecl = (D ? D->getNonClosureContext() : nullptr);
683 FnRetTy = RetTy;
684 CurFn = Fn;
685 CurFnInfo = &FnInfo;
686 assert(CurFn->isDeclaration() && "Function already has body?");
687
688 // If this function has been blacklisted for any of the enabled sanitizers,
689 // disable the sanitizer for the function.
690 do {
691#define SANITIZER(NAME, ID) \
692 if (SanOpts.empty()) \
693 break; \
694 if (SanOpts.has(SanitizerKind::ID)) \
695 if (CGM.isInSanitizerBlacklist(SanitizerKind::ID, Fn, Loc)) \
696 SanOpts.set(SanitizerKind::ID, false);
697
698#include "clang/Basic/Sanitizers.def"
699#undef SANITIZER
700 } while (0);
701
702 if (D) {
703 // Apply the no_sanitize* attributes to SanOpts.
704 for (auto Attr : D->specific_attrs<NoSanitizeAttr>()) {
705 SanitizerMask mask = Attr->getMask();
706 SanOpts.Mask &= ~mask;
707 if (mask & SanitizerKind::Address)
708 SanOpts.set(SanitizerKind::KernelAddress, false);
709 if (mask & SanitizerKind::KernelAddress)
710 SanOpts.set(SanitizerKind::Address, false);
711 if (mask & SanitizerKind::HWAddress)
712 SanOpts.set(SanitizerKind::KernelHWAddress, false);
713 if (mask & SanitizerKind::KernelHWAddress)
714 SanOpts.set(SanitizerKind::HWAddress, false);
715 }
716 }
717
718 // Apply sanitizer attributes to the function.
719 if (SanOpts.hasOneOf(SanitizerKind::Address | SanitizerKind::KernelAddress))
720 Fn->addFnAttr(llvm::Attribute::SanitizeAddress);
721 if (SanOpts.hasOneOf(SanitizerKind::HWAddress | SanitizerKind::KernelHWAddress))
722 Fn->addFnAttr(llvm::Attribute::SanitizeHWAddress);
723 if (SanOpts.has(SanitizerKind::Thread))
724 Fn->addFnAttr(llvm::Attribute::SanitizeThread);
725 if (SanOpts.hasOneOf(SanitizerKind::Memory | SanitizerKind::KernelMemory))
726 Fn->addFnAttr(llvm::Attribute::SanitizeMemory);
727 if (SanOpts.has(SanitizerKind::SafeStack))
728 Fn->addFnAttr(llvm::Attribute::SafeStack);
729 if (SanOpts.has(SanitizerKind::ShadowCallStack))
730 Fn->addFnAttr(llvm::Attribute::ShadowCallStack);
731
732 // Apply fuzzing attribute to the function.
733 if (SanOpts.hasOneOf(SanitizerKind::Fuzzer | SanitizerKind::FuzzerNoLink))
734 Fn->addFnAttr(llvm::Attribute::OptForFuzzing);
735
736 // Ignore TSan memory acesses from within ObjC/ObjC++ dealloc, initialize,
737 // .cxx_destruct, __destroy_helper_block_ and all of their calees at run time.
738 if (SanOpts.has(SanitizerKind::Thread)) {
739 if (const auto *OMD = dyn_cast_or_null<ObjCMethodDecl>(D)) {
740 IdentifierInfo *II = OMD->getSelector().getIdentifierInfoForSlot(0);
741 if (OMD->getMethodFamily() == OMF_dealloc ||
742 OMD->getMethodFamily() == OMF_initialize ||
743 (OMD->getSelector().isUnarySelector() && II->isStr(".cxx_destruct"))) {
744 markAsIgnoreThreadCheckingAtRuntime(Fn);
745 }
746 }
747 }
748
749 // Ignore unrelated casts in STL allocate() since the allocator must cast
750 // from void* to T* before object initialization completes. Don't match on the
751 // namespace because not all allocators are in std::
752 if (D && SanOpts.has(SanitizerKind::CFIUnrelatedCast)) {
753 if (matchesStlAllocatorFn(D, getContext()))
754 SanOpts.Mask &= ~SanitizerKind::CFIUnrelatedCast;
755 }
756
757 // Apply xray attributes to the function (as a string, for now)
758 if (D) {
759 if (const auto *XRayAttr = D->getAttr<XRayInstrumentAttr>()) {
760 if (CGM.getCodeGenOpts().XRayInstrumentationBundle.has(
761 XRayInstrKind::Function)) {
762 if (XRayAttr->alwaysXRayInstrument() && ShouldXRayInstrumentFunction())
763 Fn->addFnAttr("function-instrument", "xray-always");
764 if (XRayAttr->neverXRayInstrument())
765 Fn->addFnAttr("function-instrument", "xray-never");
766 if (const auto *LogArgs = D->getAttr<XRayLogArgsAttr>())
767 if (ShouldXRayInstrumentFunction())
768 Fn->addFnAttr("xray-log-args",
769 llvm::utostr(LogArgs->getArgumentCount()));
770 }
771 } else {
772 if (ShouldXRayInstrumentFunction() && !CGM.imbueXRayAttrs(Fn, Loc))
773 Fn->addFnAttr(
774 "xray-instruction-threshold",
775 llvm::itostr(CGM.getCodeGenOpts().XRayInstructionThreshold));
776 }
777 }
778
779 if (const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D)) {
780 auto *FT =
781 dyn_cast<FunctionType>(FD->getType().getDesugaredType(getContext()));
782 if (FT && (FT->getCallConv() == CC_CHERICCallee))
783 if (auto *ClsAttr = FD->getAttr<CHERIMethodClassAttr>())
784 CGM.EmitSandboxDefinedMethod(ClsAttr->getDefaultClass()->getName(),
785 FD->getName(), Fn);
786 }
787 // Add no-jump-tables value.
788 Fn->addFnAttr("no-jump-tables",
789 llvm::toStringRef(CGM.getCodeGenOpts().NoUseJumpTables));
790
791 // Add profile-sample-accurate value.
792 if (CGM.getCodeGenOpts().ProfileSampleAccurate)
793 Fn->addFnAttr("profile-sample-accurate");
794
795 if (getLangOpts().OpenCL) {
796 // Add metadata for a kernel function.
797 if (const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D))
798 EmitOpenCLKernelMetadata(FD, Fn);
799 }
800
801 // If we are checking function types, emit a function type signature as
802 // prologue data.
803 if (getLangOpts().CPlusPlus && SanOpts.has(SanitizerKind::Function)) {
804 if (const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D)) {
805 if (llvm::Constant *PrologueSig = getPrologueSignature(CGM, FD)) {
806 // Remove any (C++17) exception specifications, to allow calling e.g. a
807 // noexcept function through a non-noexcept pointer.
808 auto ProtoTy =
809 getContext().getFunctionTypeWithExceptionSpec(FD->getType(),
810 EST_None);
811 llvm::Constant *FTRTTIConst =
812 CGM.GetAddrOfRTTIDescriptor(ProtoTy, /*ForEH=*/true);
813 llvm::Constant *FTRTTIConstEncoded =
814 EncodeAddrForUseInPrologue(Fn, FTRTTIConst);
815 llvm::Constant *PrologueStructElems[] = {PrologueSig,
816 FTRTTIConstEncoded};
817 llvm::Constant *PrologueStructConst =
818 llvm::ConstantStruct::getAnon(PrologueStructElems, /*Packed=*/true);
819 Fn->setPrologueData(PrologueStructConst);
820 }
821 }
822 }
823
824 if (D && D->hasAttr<SensitiveAttr>()) {
825 // Sensitive needs to be a function attribute, not metadata
826#if 0
827 llvm::LLVMContext &Context = getLLVMContext();
828 llvm::Value *attrMDArgs[] = { Fn };
829 llvm::MDNode *FNNode= llvm::MDNode::get(Context, attrMDArgs);
830 llvm::NamedMDNode *OpenCLKernelMetadata =
831 CGM.getModule().getOrInsertNamedMetadata("cheri.sensitive.functions");
832 OpenCLKernelMetadata->addOperand(FNNode);
833#endif
834 }
835
836
837 // If we're checking nullability, we need to know whether we can check the
838 // return value. Initialize the flag to 'true' and refine it in EmitParmDecl.
839 if (SanOpts.has(SanitizerKind::NullabilityReturn)) {
840 auto Nullability = FnRetTy->getNullability(getContext());
841 if (Nullability && *Nullability == NullabilityKind::NonNull) {
842 if (!(SanOpts.has(SanitizerKind::ReturnsNonnullAttribute) &&
843 CurCodeDecl && CurCodeDecl->getAttr<ReturnsNonNullAttr>()))
844 RetValNullabilityPrecondition =
845 llvm::ConstantInt::getTrue(getLLVMContext());
846 }
847 }
848
849 // If we're in C++ mode and the function name is "main", it is guaranteed
850 // to be norecurse by the standard (3.6.1.3 "The function main shall not be
851 // used within a program").
852 if (getLangOpts().CPlusPlus)
853 if (const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D))
854 if (FD->isMain())
855 Fn->addFnAttr(llvm::Attribute::NoRecurse);
856
857 // If a custom alignment is used, force realigning to this alignment on
858 // any main function which certainly will need it.
859 if (const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D))
860 if ((FD->isMain() || FD->isMSVCRTEntryPoint()) &&
861 CGM.getCodeGenOpts().StackAlignment)
862 Fn->addFnAttr("stackrealign");
863
864 llvm::BasicBlock *EntryBB = createBasicBlock("entry", CurFn);
865
866 // Create a marker to make it easy to insert allocas into the entryblock
867 // later. Don't create this with the builder, because we don't want it
868 // folded.
869 llvm::Value *Undef = llvm::UndefValue::get(Int32Ty);
870 AllocaInsertPt = new llvm::BitCastInst(Undef, Int32Ty, "allocapt", EntryBB);
871
872 ReturnBlock = getJumpDestInCurrentScope("return");
873
874 Builder.SetInsertPoint(EntryBB);
875
876 // If we're checking the return value, allocate space for a pointer to a
877 // precise source location of the checked return statement.
878 if (requiresReturnValueCheck()) {
879 ReturnLocation = CreateDefaultAlignTempAlloca(Int8PtrTy, "return.sloc.ptr");
880 InitTempAlloca(ReturnLocation, llvm::ConstantPointerNull::get(Int8PtrTy));
881 }
882
883 // Emit subprogram debug descriptor.
884 if (CGDebugInfo *DI = getDebugInfo()) {
885 // Reconstruct the type from the argument list so that implicit parameters,
886 // such as 'this' and 'vtt', show up in the debug info. Preserve the calling
887 // convention.
888 CallingConv CC = CallingConv::CC_C;
889 if (auto *FD = dyn_cast_or_null<FunctionDecl>(D))
890 if (const auto *SrcFnTy = FD->getType()->getAs<FunctionType>())
891 CC = SrcFnTy->getCallConv();
892 SmallVector<QualType, 16> ArgTypes;
893 for (const VarDecl *VD : Args)
894 ArgTypes.push_back(VD->getType());
895 QualType FnType = getContext().getFunctionType(
896 RetTy, ArgTypes, FunctionProtoType::ExtProtoInfo(CC));
897 DI->EmitFunctionStart(GD, Loc, StartLoc, FnType, CurFn, CurFuncIsThunk,
898 Builder);
899 }
900
901 if (ShouldInstrumentFunction()) {
902 if (CGM.getCodeGenOpts().InstrumentFunctions)
903 CurFn->addFnAttr("instrument-function-entry", "__cyg_profile_func_enter");
904 if (CGM.getCodeGenOpts().InstrumentFunctionsAfterInlining)
905 CurFn->addFnAttr("instrument-function-entry-inlined",
906 "__cyg_profile_func_enter");
907 if (CGM.getCodeGenOpts().InstrumentFunctionEntryBare)
908 CurFn->addFnAttr("instrument-function-entry-inlined",
909 "__cyg_profile_func_enter_bare");
910 }
911
912 // Since emitting the mcount call here impacts optimizations such as function
913 // inlining, we just add an attribute to insert a mcount call in backend.
914 // The attribute "counting-function" is set to mcount function name which is
915 // architecture dependent.
916 if (CGM.getCodeGenOpts().InstrumentForProfiling) {
917 // Calls to fentry/mcount should not be generated if function has
918 // the no_instrument_function attribute.
919 if (!CurFuncDecl || !CurFuncDecl->hasAttr<NoInstrumentFunctionAttr>()) {
920 if (CGM.getCodeGenOpts().CallFEntry)
921 Fn->addFnAttr("fentry-call", "true");
922 else {
923 Fn->addFnAttr("instrument-function-entry-inlined",
924 getTarget().getMCountName());
925 }
926 }
927 }
928
929 if (RetTy->isVoidType()) {
930 // Void type; nothing to return.
931 ReturnValue = Address::invalid();
932
933 // Count the implicit return.
934 if (!endsWithReturn(D))
935 ++NumReturnExprs;
936 } else if (CurFnInfo->getReturnInfo().getKind() == ABIArgInfo::Indirect) {
937 // Indirect return; emit returned value directly into sret slot.
938 // This reduces code size, and affects correctness in C++.
939 auto AI = CurFn->arg_begin();
940 if (CurFnInfo->getReturnInfo().isSRetAfterThis())
941 ++AI;
942 ReturnValue = Address(&*AI, CurFnInfo->getReturnInfo().getIndirectAlign());
943 } else if (CurFnInfo->getReturnInfo().getKind() == ABIArgInfo::InAlloca &&
944 !hasScalarEvaluationKind(CurFnInfo->getReturnType())) {
945 // Load the sret pointer from the argument struct and return into that.
946 unsigned Idx = CurFnInfo->getReturnInfo().getInAllocaFieldIndex();
947 llvm::Function::arg_iterator EI = CurFn->arg_end();
948 --EI;
949 llvm::Value *Addr = Builder.CreateStructGEP(nullptr, &*EI, Idx);
950 Addr = Builder.CreateAlignedLoad(Addr, getPointerAlign(), "agg.result");
951 ReturnValue = Address(Addr, getNaturalTypeAlignment(RetTy));
952 } else {
953 ReturnValue = CreateIRTemp(RetTy, "retval");
954
955 // Tell the epilog emitter to autorelease the result. We do this
956 // now so that various specialized functions can suppress it
957 // during their IR-generation.
958 if (getLangOpts().ObjCAutoRefCount &&
959 !CurFnInfo->isReturnsRetained() &&
960 RetTy->isObjCRetainableType())
961 AutoreleaseResult = true;
962 }
963
964 EmitStartEHSpec(CurCodeDecl);
965
966 PrologueCleanupDepth = EHStack.stable_begin();
967
968 // Emit OpenMP specific initialization of the device functions.
969 if (getLangOpts().OpenMP && CurCodeDecl)
970 CGM.getOpenMPRuntime().emitFunctionProlog(*this, CurCodeDecl);
971
972 EmitFunctionProlog(*CurFnInfo, CurFn, Args);
973
974 if (D && isa<CXXMethodDecl>(D) && cast<CXXMethodDecl>(D)->isInstance()) {
975 CGM.getCXXABI().EmitInstanceFunctionProlog(*this);
976 const CXXMethodDecl *MD = cast<CXXMethodDecl>(D);
977 if (MD->getParent()->isLambda() &&
978 MD->getOverloadedOperator() == OO_Call) {
979 // We're in a lambda; figure out the captures.
980 MD->getParent()->getCaptureFields(LambdaCaptureFields,
981 LambdaThisCaptureField);
982 if (LambdaThisCaptureField) {
983 // If the lambda captures the object referred to by '*this' - either by
984 // value or by reference, make sure CXXThisValue points to the correct
985 // object.
986
987 // Get the lvalue for the field (which is a copy of the enclosing object
988 // or contains the address of the enclosing object).
989 LValue ThisFieldLValue = EmitLValueForLambdaField(LambdaThisCaptureField);
990 if (!LambdaThisCaptureField->getType()->isPointerType()) {
991 // If the enclosing object was captured by value, just use its address.
992 CXXThisValue = ThisFieldLValue.getAddress().getPointer();
993 } else {
994 // Load the lvalue pointed to by the field, since '*this' was captured
995 // by reference.
996 CXXThisValue =
997 EmitLoadOfLValue(ThisFieldLValue, SourceLocation()).getScalarVal();
998 }
999 }
1000 for (auto *FD : MD->getParent()->fields()) {
1001 if (FD->hasCapturedVLAType()) {
1002 auto *ExprArg = EmitLoadOfLValue(EmitLValueForLambdaField(FD),
1003 SourceLocation()).getScalarVal();
1004 auto VAT = FD->getCapturedVLAType();
1005 VLASizeMap[VAT->getSizeExpr()] = ExprArg;
1006 }
1007 }
1008 } else {
1009 // Not in a lambda; just use 'this' from the method.
1010 // FIXME: Should we generate a new load for each use of 'this'? The
1011 // fast register allocator would be happier...
1012 CXXThisValue = CXXABIThisValue;
1013 }
1014
1015 // Check the 'this' pointer once per function, if it's available.
1016 if (CXXABIThisValue) {
1017 SanitizerSet SkippedChecks;
1018 SkippedChecks.set(SanitizerKind::ObjectSize, true);
1019 QualType ThisTy = MD->getThisType();
1020
1021 // If this is the call operator of a lambda with no capture-default, it
1022 // may have a static invoker function, which may call this operator with
1023 // a null 'this' pointer.
1024 if (isLambdaCallOperator(MD) &&
1025 MD->getParent()->getLambdaCaptureDefault() == LCD_None)
1026 SkippedChecks.set(SanitizerKind::Null, true);
1027
1028 EmitTypeCheck(isa<CXXConstructorDecl>(MD) ? TCK_ConstructorCall
1029 : TCK_MemberCall,
1030 Loc, CXXABIThisValue, ThisTy,
1031 getContext().getTypeAlignInChars(ThisTy->getPointeeType()),
1032 SkippedChecks);
1033 }
1034 }
1035
1036 // If any of the arguments have a variably modified type, make sure to
1037 // emit the type size.
1038 for (FunctionArgList::const_iterator i = Args.begin(), e = Args.end();
1039 i != e; ++i) {
1040 const VarDecl *VD = *i;
1041
1042 // Dig out the type as written from ParmVarDecls; it's unclear whether
1043 // the standard (C99 6.9.1p10) requires this, but we're following the
1044 // precedent set by gcc.
1045 QualType Ty;
1046 if (const ParmVarDecl *PVD = dyn_cast<ParmVarDecl>(VD))
1047 Ty = PVD->getOriginalType();
1048 else
1049 Ty = VD->getType();
1050
1051 if (Ty->isVariablyModifiedType())
1052 EmitVariablyModifiedType(Ty);
1053 }
1054 // Emit a location at the end of the prologue.
1055 if (CGDebugInfo *DI = getDebugInfo())
1056 DI->EmitLocation(Builder, StartLoc);
1057
1058 // TODO: Do we need to handle this in two places like we do with
1059 // target-features/target-cpu?
1060 if (CurFuncDecl)
1061 if (const auto *VecWidth = CurFuncDecl->getAttr<MinVectorWidthAttr>())
1062 LargestVectorWidth = VecWidth->getVectorWidth();
1063}

1095/// Tries to mark the given function nounwind based on the
1096/// non-existence of any throwing calls within it. We believe this is
1097/// lightweight enough to do at -O0.
1098static void TryMarkNoThrow(llvm::Function *F) {
1099 // LLVM treats 'nounwind' on a function as part of the type, so we
1100 // can't do this on functions that can be overwritten.
1101 if (F->isInterposable()) return;
1102
1103 for (llvm::BasicBlock &BB : *F)
1104 for (llvm::Instruction &I : BB)
1105 if (I.mayThrow())
1106 return;
1107
1108 F->setDoesNotThrow();
1109}

1111QualType CodeGenFunction::BuildFunctionArgList(GlobalDecl GD,
1112 FunctionArgList &Args) {
1113 const FunctionDecl *FD = cast<FunctionDecl>(GD.getDecl());
1114 QualType ResTy = FD->getReturnType();
1115
1116 const CXXMethodDecl *MD = dyn_cast<CXXMethodDecl>(FD);
1117 if (MD && MD->isInstance()) {
1118 if (CGM.getCXXABI().HasThisReturn(GD))
1119 ResTy = MD->getThisType();
1120 else if (CGM.getCXXABI().hasMostDerivedReturn(GD))
1121 ResTy = CGM.getContext().VoidPtrTy;
1122 CGM.getCXXABI().buildThisParam(*this, Args);
1123 }
1124
1125 // The base version of an inheriting constructor whose constructed base is a
1126 // virtual base is not passed any arguments (because it doesn't actually call
1127 // the inherited constructor).
1128 bool PassedParams = true;
1129 if (const CXXConstructorDecl *CD = dyn_cast<CXXConstructorDecl>(FD))
1130 if (auto Inherited = CD->getInheritedConstructor())
1131 PassedParams =
1132 getTypes().inheritingCtorHasParams(Inherited, GD.getCtorType());
1133
1134 if (PassedParams) {
1135 for (auto *Param : FD->parameters()) {
1136 Args.push_back(Param);
1137 if (!Param->hasAttr<PassObjectSizeAttr>())
1138 continue;
1139
1140 auto *Implicit = ImplicitParamDecl::Create(
1141 getContext(), Param->getDeclContext(), Param->getLocation(),
1142 /*Id=*/nullptr, getContext().getSizeType(), ImplicitParamDecl::Other);
1143 SizeArguments[Param] = Implicit;
1144 Args.push_back(Implicit);
1145 }
1146 }
1147
1148 if (MD && (isa<CXXConstructorDecl>(MD) || isa<CXXDestructorDecl>(MD)))
1149 CGM.getCXXABI().addImplicitStructorParams(*this, ResTy, Args);
1150
1151 return ResTy;
1152}

1154static bool
1155shouldUseUndefinedBehaviorReturnOptimization(const FunctionDecl *FD,
1156 const ASTContext &Context) {
1157 QualType T = FD->getReturnType();
1158 // Avoid the optimization for functions that return a record type with a
1159 // trivial destructor or another trivially copyable type.
1160 if (const RecordType *RT = T.getCanonicalType()->getAs<RecordType>()) {
1161 if (const auto *ClassDecl = dyn_cast<CXXRecordDecl>(RT->getDecl()))
1162 return !ClassDecl->hasTrivialDestructor();
1163 }
1164 return !T.isTriviallyCopyableType(Context);
1165}

1167void CodeGenFunction::GenerateCode(GlobalDecl GD, llvm::Function *Fn,
1168 const CGFunctionInfo &FnInfo) {
1169 const FunctionDecl *FD = cast<FunctionDecl>(GD.getDecl());
1170 CurGD = GD;
1171
1172 FunctionArgList Args;
1173 QualType ResTy = BuildFunctionArgList(GD, Args);
1174
1175 // Check if we should generate debug info for this function.
1176 if (FD->hasAttr<NoDebugAttr>())
1177 DebugInfo = nullptr; // disable debug info indefinitely for this function
1178
1179 // The function might not have a body if we're generating thunks for a
1180 // function declaration.
1181 SourceRange BodyRange;
1182 if (Stmt *Body = FD->getBody())
1183 BodyRange = Body->getSourceRange();
1184 else
1185 BodyRange = FD->getLocation();
1186 CurEHLocation = BodyRange.getEnd();
1187
1188 // Use the location of the start of the function to determine where
1189 // the function definition is located. By default use the location
1190 // of the declaration as the location for the subprogram. A function
1191 // may lack a declaration in the source code if it is created by code
1192 // gen. (examples: _GLOBAL__I_a, __cxx_global_array_dtor, thunk).
1193 SourceLocation Loc = FD->getLocation();
1194
1195 // If this is a function specialization then use the pattern body
1196 // as the location for the function.
1197 if (const FunctionDecl *SpecDecl = FD->getTemplateInstantiationPattern())
1198 if (SpecDecl->hasBody(SpecDecl))
1199 Loc = SpecDecl->getLocation();
1200
1201 Stmt *Body = FD->getBody();
1202
1203 // Initialize helper which will detect jumps which can cause invalid lifetime
1204 // markers.
1205 if (Body && ShouldEmitLifetimeMarkers)
1206 Bypasses.Init(Body);
1207
1208 // Emit the standard function prologue.
1209 StartFunction(GD, ResTy, Fn, FnInfo, Args, Loc, BodyRange.getBegin());
1210
1211 // Generate the body of the function.
1212 PGO.assignRegionCounters(GD, CurFn);
1213 if (isa<CXXDestructorDecl>(FD))
1214 EmitDestructorBody(Args);
1215 else if (isa<CXXConstructorDecl>(FD))
1216 EmitConstructorBody(Args);
1217 else if (getLangOpts().CUDA &&
1218 !getLangOpts().CUDAIsDevice &&
1219 FD->hasAttr<CUDAGlobalAttr>())
1220 CGM.getCUDARuntime().emitDeviceStub(*this, Args);
1221 else if (isa<CXXMethodDecl>(FD) &&
1222 cast<CXXMethodDecl>(FD)->isLambdaStaticInvoker()) {
1223 // The lambda static invoker function is special, because it forwards or
1224 // clones the body of the function call operator (but is actually static).
1225 EmitLambdaStaticInvokeBody(cast<CXXMethodDecl>(FD));
1226 } else if (FD->isDefaulted() && isa<CXXMethodDecl>(FD) &&
1227 (cast<CXXMethodDecl>(FD)->isCopyAssignmentOperator() ||
1228 cast<CXXMethodDecl>(FD)->isMoveAssignmentOperator())) {
1229 // Implicit copy-assignment gets the same special treatment as implicit
1230 // copy-constructors.
1231 emitImplicitAssignmentOperatorBody(Args);
1232 } else if (Body) {
1233 EmitFunctionBody(Body);
1234 } else
1235 llvm_unreachable("no definition for emitted function");
1236
1237 // C++11 [stmt.return]p2:
1238 // Flowing off the end of a function [...] results in undefined behavior in
1239 // a value-returning function.
1240 // C11 6.9.1p12:
1241 // If the '}' that terminates a function is reached, and the value of the
1242 // function call is used by the caller, the behavior is undefined.
1243 if (getLangOpts().CPlusPlus && !FD->hasImplicitReturnZero() && !SawAsmBlock &&
1244 !FD->getReturnType()->isVoidType() && Builder.GetInsertBlock()) {
1245 bool ShouldEmitUnreachable =
1246 CGM.getCodeGenOpts().StrictReturn ||
1247 shouldUseUndefinedBehaviorReturnOptimization(FD, getContext());
1248 if (SanOpts.has(SanitizerKind::Return)) {
1249 SanitizerScope SanScope(this);
1250 llvm::Value *IsFalse = Builder.getFalse();
1251 EmitCheck(std::make_pair(IsFalse, SanitizerKind::Return),
1252 SanitizerHandler::MissingReturn,
1253 EmitCheckSourceLocation(FD->getLocation()), None);
1254 } else if (ShouldEmitUnreachable) {
1255 if (CGM.getCodeGenOpts().OptimizationLevel == 0)
1256 EmitTrapCall(llvm::Intrinsic::trap);
1257 }
1258 if (SanOpts.has(SanitizerKind::Return) || ShouldEmitUnreachable) {
1259 Builder.CreateUnreachable();
1260 Builder.ClearInsertionPoint();
1261 }
1262 }
1263
1264 // Emit the standard function epilogue.
1265 FinishFunction(BodyRange.getEnd());
1266
1267 // If we haven't marked the function nothrow through other means, do
1268 // a quick pass now to see if we can.
1269 if (!CurFn->doesNotThrow())
1270 TryMarkNoThrow(CurFn);
1271}

1273/// ContainsLabel - Return true if the statement contains a label in it. If
1274/// this statement is not executed normally, it not containing a label means
1275/// that we can just remove the code.
1276bool CodeGenFunction::ContainsLabel(const Stmt *S, bool IgnoreCaseStmts) {
1277 // Null statement, not a label!
1278 if (!S) return false;
1279
1280 // If this is a label, we have to emit the code, consider something like:
1281 // if (0) { ... foo: bar(); } goto foo;
1282 //
1283 // TODO: If anyone cared, we could track __label__'s, since we know that you
1284 // can't jump to one from outside their declared region.
1285 if (isa<LabelStmt>(S))
1286 return true;
1287
1288 // If this is a case/default statement, and we haven't seen a switch, we have
1289 // to emit the code.
1290 if (isa<SwitchCase>(S) && !IgnoreCaseStmts)
1291 return true;
1292
1293 // If this is a switch statement, we want to ignore cases below it.
1294 if (isa<SwitchStmt>(S))
1295 IgnoreCaseStmts = true;
1296
1297 // Scan subexpressions for verboten labels.
1298 for (const Stmt *SubStmt : S->children())
1299 if (ContainsLabel(SubStmt, IgnoreCaseStmts))
1300 return true;
1301
1302 return false;
1303}

1305/// containsBreak - Return true if the statement contains a break out of it.
1306/// If the statement (recursively) contains a switch or loop with a break
1307/// inside of it, this is fine.
1308bool CodeGenFunction::containsBreak(const Stmt *S) {
1309 // Null statement, not a label!
1310 if (!S) return false;
1311
1312 // If this is a switch or loop that defines its own break scope, then we can
1313 // include it and anything inside of it.
1314 if (isa<SwitchStmt>(S) || isa<WhileStmt>(S) || isa<DoStmt>(S) ||
1315 isa<ForStmt>(S))
1316 return false;
1317
1318 if (isa<BreakStmt>(S))
1319 return true;
1320
1321 // Scan subexpressions for verboten breaks.
1322 for (const Stmt *SubStmt : S->children())
1323 if (containsBreak(SubStmt))
1324 return true;
1325
1326 return false;
1327}

1330 if (!S) return false;

1333 // scope. Note, this list is longer than the list of statements that might
1334 // have an unscoped decl nested within them, but this way is conservatively
1335 // correct even if more statement kinds are added.
1336 if (isa<IfStmt>(S) || isa<SwitchStmt>(S) || isa<WhileStmt>(S) ||

1346 if (mightAddDeclToScope(SubStmt))
1347 return true;

1349 return false;

1388/// EmitBranchOnBoolExpr - Emit a branch on a boolean condition (e.g. for an if
1389/// statement) to the specified blocks. Based on the condition, this might try
1390/// to simplify the codegen of the conditional based on the branch.
1391///
1392void CodeGenFunction::EmitBranchOnBoolExpr(const Expr *Cond,
1393 llvm::BasicBlock *TrueBlock,
1394 llvm::BasicBlock *FalseBlock,
1395 uint64_t TrueCount) {
1396 Cond = Cond->IgnoreParens();
1397
1398 if (const BinaryOperator *CondBOp = dyn_cast<BinaryOperator>(Cond)) {
1399
1400 // Handle X && Y in a condition.
1401 if (CondBOp->getOpcode() == BO_LAnd) {
1402 // If we have "1 && X", simplify the code. "0 && X" would have constant
1403 // folded if the case was simple enough.
1404 bool ConstantBool = false;
1405 if (ConstantFoldsToSimpleInteger(CondBOp->getLHS(), ConstantBool) &&
1406 ConstantBool) {
1407 // br(1 && X) -> br(X).
1408 incrementProfileCounter(CondBOp);
1409 return EmitBranchOnBoolExpr(CondBOp->getRHS(), TrueBlock, FalseBlock,
1410 TrueCount);
1411 }
1412
1413 // If we have "X && 1", simplify the code to use an uncond branch.
1414 // "X && 0" would have been constant folded to 0.
1415 if (ConstantFoldsToSimpleInteger(CondBOp->getRHS(), ConstantBool) &&
1416 ConstantBool) {
1417 // br(X && 1) -> br(X).
1418 return EmitBranchOnBoolExpr(CondBOp->getLHS(), TrueBlock, FalseBlock,
1419 TrueCount);
1420 }
1421
1422 // Emit the LHS as a conditional. If the LHS conditional is false, we
1423 // want to jump to the FalseBlock.
1424 llvm::BasicBlock *LHSTrue = createBasicBlock("land.lhs.true");
1425 // The counter tells us how often we evaluate RHS, and all of TrueCount
1426 // can be propagated to that branch.
1427 uint64_t RHSCount = getProfileCount(CondBOp->getRHS());
1428
1429 ConditionalEvaluation eval(*this);
1430 {
1431 ApplyDebugLocation DL(*this, Cond);
1432 EmitBranchOnBoolExpr(CondBOp->getLHS(), LHSTrue, FalseBlock, RHSCount);
1433 EmitBlock(LHSTrue);
1434 }
1435
1436 incrementProfileCounter(CondBOp);
1437 setCurrentProfileCount(getProfileCount(CondBOp->getRHS()));
1438
1439 // Any temporaries created here are conditional.
1440 eval.begin(*this);
1441 EmitBranchOnBoolExpr(CondBOp->getRHS(), TrueBlock, FalseBlock, TrueCount);
1442 eval.end(*this);
1443
1444 return;
1445 }
1446
1447 if (CondBOp->getOpcode() == BO_LOr) {
1448 // If we have "0 || X", simplify the code. "1 || X" would have constant
1449 // folded if the case was simple enough.
1450 bool ConstantBool = false;
1451 if (ConstantFoldsToSimpleInteger(CondBOp->getLHS(), ConstantBool) &&
1452 !ConstantBool) {
1453 // br(0 || X) -> br(X).
1454 incrementProfileCounter(CondBOp);
1455 return EmitBranchOnBoolExpr(CondBOp->getRHS(), TrueBlock, FalseBlock,
1456 TrueCount);
1457 }
1458
1459 // If we have "X || 0", simplify the code to use an uncond branch.
1460 // "X || 1" would have been constant folded to 1.
1461 if (ConstantFoldsToSimpleInteger(CondBOp->getRHS(), ConstantBool) &&
1462 !ConstantBool) {
1463 // br(X || 0) -> br(X).
1464 return EmitBranchOnBoolExpr(CondBOp->getLHS(), TrueBlock, FalseBlock,
1465 TrueCount);
1466 }
1467
1468 // Emit the LHS as a conditional. If the LHS conditional is true, we
1469 // want to jump to the TrueBlock.
1470 llvm::BasicBlock *LHSFalse = createBasicBlock("lor.lhs.false");
1471 // We have the count for entry to the RHS and for the whole expression
1472 // being true, so we can divy up True count between the short circuit and
1473 // the RHS.
1474 uint64_t LHSCount =
1475 getCurrentProfileCount() - getProfileCount(CondBOp->getRHS());
1476 uint64_t RHSCount = TrueCount - LHSCount;
1477
1478 ConditionalEvaluation eval(*this);
1479 {
1480 ApplyDebugLocation DL(*this, Cond);
1481 EmitBranchOnBoolExpr(CondBOp->getLHS(), TrueBlock, LHSFalse, LHSCount);
1482 EmitBlock(LHSFalse);
1483 }
1484
1485 incrementProfileCounter(CondBOp);
1486 setCurrentProfileCount(getProfileCount(CondBOp->getRHS()));
1487
1488 // Any temporaries created here are conditional.
1489 eval.begin(*this);
1490 EmitBranchOnBoolExpr(CondBOp->getRHS(), TrueBlock, FalseBlock, RHSCount);
1491
1492 eval.end(*this);
1493
1494 return;
1495 }
1496 }
1497
1498 if (const UnaryOperator *CondUOp = dyn_cast<UnaryOperator>(Cond)) {
1499 // br(!x, t, f) -> br(x, f, t)
1500 if (CondUOp->getOpcode() == UO_LNot) {
1501 // Negate the count.
1502 uint64_t FalseCount = getCurrentProfileCount() - TrueCount;
1503 // Negate the condition and swap the destination blocks.
1504 return EmitBranchOnBoolExpr(CondUOp->getSubExpr(), FalseBlock, TrueBlock,
1505 FalseCount);
1506 }
1507 }
1508
1509 if (const ConditionalOperator *CondOp = dyn_cast<ConditionalOperator>(Cond)) {
1510 // br(c ? x : y, t, f) -> br(c, br(x, t, f), br(y, t, f))
1511 llvm::BasicBlock *LHSBlock = createBasicBlock("cond.true");
1512 llvm::BasicBlock *RHSBlock = createBasicBlock("cond.false");
1513
1514 ConditionalEvaluation cond(*this);
1515 EmitBranchOnBoolExpr(CondOp->getCond(), LHSBlock, RHSBlock,
1516 getProfileCount(CondOp));
1517
1518 // When computing PGO branch weights, we only know the overall count for
1519 // the true block. This code is essentially doing tail duplication of the
1520 // naive code-gen, introducing new edges for which counts are not
1521 // available. Divide the counts proportionally between the LHS and RHS of
1522 // the conditional operator.
1523 uint64_t LHSScaledTrueCount = 0;
1524 if (TrueCount) {
1525 double LHSRatio =
1526 getProfileCount(CondOp) / (double)getCurrentProfileCount();
1527 LHSScaledTrueCount = TrueCount * LHSRatio;
1528 }
1529
1530 cond.begin(*this);
1531 EmitBlock(LHSBlock);
1532 incrementProfileCounter(CondOp);
1533 {
1534 ApplyDebugLocation DL(*this, Cond);
1535 EmitBranchOnBoolExpr(CondOp->getLHS(), TrueBlock, FalseBlock,
1536 LHSScaledTrueCount);
1537 }
1538 cond.end(*this);
1539
1540 cond.begin(*this);
1541 EmitBlock(RHSBlock);
1542 EmitBranchOnBoolExpr(CondOp->getRHS(), TrueBlock, FalseBlock,
1543 TrueCount - LHSScaledTrueCount);
1544 cond.end(*this);
1545
1546 return;
1547 }
1548
1549 if (const CXXThrowExpr *Throw = dyn_cast<CXXThrowExpr>(Cond)) {
1550 // Conditional operator handling can give us a throw expression as a
1551 // condition for a case like:
1552 // br(c ? throw x : y, t, f) -> br(c, br(throw x, t, f), br(y, t, f)
1553 // Fold this to:
1554 // br(c, throw x, br(y, t, f))
1555 EmitCXXThrowExpr(Throw, /*KeepInsertionPoint*/false);
1556 return;
1557 }
1558
1559 // If the branch has a condition wrapped by __builtin_unpredictable,
1560 // create metadata that specifies that the branch is unpredictable.
1561 // Don't bother if not optimizing because that metadata would not be used.
1562 llvm::MDNode *Unpredictable = nullptr;
1563 auto *Call = dyn_cast<CallExpr>(Cond->IgnoreImpCasts());
1564 if (Call && CGM.getCodeGenOpts().OptimizationLevel != 0) {
1565 auto *FD = dyn_cast_or_null<FunctionDecl>(Call->getCalleeDecl());
1566 if (FD && FD->getBuiltinID() == Builtin::BI__builtin_unpredictable) {
1567 llvm::MDBuilder MDHelper(getLLVMContext());
1568 Unpredictable = MDHelper.createUnpredictable();
1569 }
1570 }
1571
1572 // Create branch weights based on the number of times we get here and the
1573 // number of times the condition should be true.
1574 uint64_t CurrentCount = std::max(getCurrentProfileCount(), TrueCount);
1575 llvm::MDNode *Weights =
1576 createProfileWeights(TrueCount, CurrentCount - TrueCount);
1577
1578 // Emit the code with the fully general case.
1579 llvm::Value *CondV;
1580 {
1581 ApplyDebugLocation DL(*this, Cond);
1582 CondV = EvaluateExprAsBool(Cond);
1583 }
1584 Builder.CreateCondBr(CondV, TrueBlock, FalseBlock, Weights, Unpredictable);
1585}

1593/// emitNonZeroVLAInit - Emit the "zero" initialization of a
1594/// variable-length array whose elements have a non-zero bit-pattern.
1595///
1596/// \param baseType the inner-most element type of the array
1597/// \param src - a char* pointing to the bit-pattern for a single
1598/// base element of the array
1599/// \param sizeInChars - the total size of the VLA, in chars
1600static void emitNonZeroVLAInit(CodeGenFunction &CGF, QualType baseType,
1601 Address dest, Address src,
1602 llvm::Value *sizeInChars) {
1603 CGBuilderTy &Builder = CGF.Builder;
1604
1605 CharUnits baseSize = CGF.getContext().getTypeSizeInChars(baseType);
1606 llvm::Value *baseSizeInChars
1607 = llvm::ConstantInt::get(CGF.IntPtrTy, baseSize.getQuantity());
1608
1609 Address begin =
1610 Builder.CreateElementBitCast(dest, CGF.Int8Ty, "vla.begin");
1611 llvm::Value *end =
1612 Builder.CreateInBoundsGEP(begin.getPointer(), sizeInChars, "vla.end");
1613
1614 llvm::BasicBlock *originBB = CGF.Builder.GetInsertBlock();
1615 llvm::BasicBlock *loopBB = CGF.createBasicBlock("vla-init.loop");
1616 llvm::BasicBlock *contBB = CGF.createBasicBlock("vla-init.cont");
1617
1618 // Make a loop over the VLA. C99 guarantees that the VLA element
1619 // count must be nonzero.
1620 CGF.EmitBlock(loopBB);
1621
1622 llvm::PHINode *cur = Builder.CreatePHI(begin.getType(), 2, "vla.cur");
1623 cur->addIncoming(begin.getPointer(), originBB);
1624
1625 CharUnits curAlign =
1626 dest.getAlignment().alignmentOfArrayElement(baseSize);
1627
1628 // memcpy the individual element bit-pattern.
1629 Builder.CreateMemCpy(Address(cur, curAlign), src, baseSizeInChars,
1630 /*volatile*/ false);
1631
1632 // Go to the next element.
1633 llvm::Value *next =
1634 Builder.CreateInBoundsGEP(CGF.Int8Ty, cur, baseSizeInChars, "vla.next");
1635
1636 // Leave if that's the end of the VLA.
1637 llvm::Value *done = Builder.CreateICmpEQ(next, end, "vla-init.isdone");
1638 Builder.CreateCondBr(done, contBB, loopBB);
1639 cur->addIncoming(next, loopBB);
1640
1641 CGF.EmitBlock(contBB);
1642}

1644void
1645CodeGenFunction::EmitNullInitialization(Address DestPtr, QualType Ty) {
1646 // Ignore empty classes in C++.
1647 if (getLangOpts().CPlusPlus) {
1648 if (const RecordType *RT = Ty->getAs<RecordType>()) {
1649 if (cast<CXXRecordDecl>(RT->getDecl())->isEmpty())
1650 return;
1651 }
1652 }
1653
1654 // Cast the dest ptr to the appropriate i8 pointer type.
1655 if (DestPtr.getElementType() != Int8Ty)
1656 DestPtr = Builder.CreateElementBitCast(DestPtr, Int8Ty);
1657
1658 // Get size and alignment info for this aggregate.
1659 CharUnits size = getContext().getTypeSizeInChars(Ty);
1660
1661 llvm::Value *SizeVal;
1662 const VariableArrayType *vla;
1663
1664 // Don't bother emitting a zero-byte memset.
1665 if (size.isZero()) {
1666 // But note that getTypeInfo returns 0 for a VLA.
1667 if (const VariableArrayType *vlaType =
1668 dyn_cast_or_null<VariableArrayType>(
1669 getContext().getAsArrayType(Ty))) {
1670 auto VlaSize = getVLASize(vlaType);
1671 SizeVal = VlaSize.NumElts;
1672 CharUnits eltSize = getContext().getTypeSizeInChars(VlaSize.Type);
1673 if (!eltSize.isOne())
1674 SizeVal = Builder.CreateNUWMul(SizeVal, CGM.getSize(eltSize));
1675 vla = vlaType;
1676 } else {
1677 return;
1678 }
1679 } else {
1680 SizeVal = CGM.getSize(size);
1681 vla = nullptr;
1682 }
1683
1684 // If the type contains a pointer to data member we can't memset it to zero.
1685 // Instead, create a null constant and copy it to the destination.
1686 // TODO: there are other patterns besides zero that we can usefully memset,
1687 // like -1, which happens to be the pattern used by member-pointers.
1688 if (!CGM.getTypes().isZeroInitializable(Ty)) {
1689 // For a VLA, emit a single element, then splat that over the VLA.
1690 if (vla) Ty = getContext().getBaseElementType(vla);
1691
1692 llvm::Constant *NullConstant = CGM.EmitNullConstant(Ty);
1693
1694 llvm::GlobalVariable *NullVariable =
1695 new llvm::GlobalVariable(CGM.getModule(), NullConstant->getType(),
1696 /*isConstant=*/true,
1697 llvm::GlobalVariable::PrivateLinkage,
1698 NullConstant, Twine());
1699 CharUnits NullAlign = DestPtr.getAlignment();
1700 NullVariable->setAlignment(NullAlign.getQuantity());
1701 Address SrcPtr(Builder.CreatePointerBitCastOrAddrSpaceCast(NullVariable, CGM.Int8PtrTy),
1702 NullAlign);
1703
1704 if (vla) return emitNonZeroVLAInit(*this, Ty, DestPtr, SrcPtr, SizeVal);
1705
1706 // Get and call the appropriate llvm.memcpy overload.
1707 Builder.CreateMemCpy(DestPtr, SrcPtr, SizeVal, false);
1708 return;
1709 }
1710
1711 // Otherwise, just memset the whole thing to zero. This is legal
1712 // because in LLVM, all default initializers (other than the ones we just
1713 // handled above) are guaranteed to have a bit pattern of all zeros.
1714 Builder.CreateMemSet(DestPtr, Builder.getInt8(0), SizeVal, false);
1715}

1748/// Computes the length of an array in elements, as well as the base
1749/// element type and a properly-typed first element pointer.
1750llvm::Value *CodeGenFunction::emitArrayLength(const ArrayType *origArrayType,
1751 QualType &baseType,
1752 Address &addr) {
1753 const ArrayType *arrayType = origArrayType;
1754
1755 // If it's a VLA, we have to load the stored size. Note that
1756 // this is the size of the VLA in bytes, not its size in elements.
1757 llvm::Value *numVLAElements = nullptr;
1758 if (isa<VariableArrayType>(arrayType)) {
1759 numVLAElements = getVLASize(cast<VariableArrayType>(arrayType)).NumElts;
1760
1761 // Walk into all VLAs. This doesn't require changes to addr,
1762 // which has type T* where T is the first non-VLA element type.
1763 do {
1764 QualType elementType = arrayType->getElementType();
1765 arrayType = getContext().getAsArrayType(elementType);
1766
1767 // If we only have VLA components, 'addr' requires no adjustment.
1768 if (!arrayType) {
1769 baseType = elementType;
1770 return numVLAElements;
1771 }
1772 } while (isa<VariableArrayType>(arrayType));
1773
1774 // We get out here only if we find a constant array type
1775 // inside the VLA.
1776 }
1777
1778 // We have some number of constant-length arrays, so addr should
1779 // have LLVM type [M x [N x [...]]]*. Build a GEP that walks
1780 // down to the first element of addr.
1781 SmallVector<llvm::Value*, 8> gepIndices;
1782
1783 // GEP down to the array type.
1784 llvm::ConstantInt *zero = Builder.getInt32(0);
1785 gepIndices.push_back(zero);
1786
1787 uint64_t countFromCLAs = 1;
1788 QualType eltType;
1789
1790 llvm::ArrayType *llvmArrayType =
1791 dyn_cast<llvm::ArrayType>(addr.getElementType());
1792 while (llvmArrayType) {
1793 assert(isa<ConstantArrayType>(arrayType));
1794 assert(cast<ConstantArrayType>(arrayType)->getSize().getZExtValue()
1795 == llvmArrayType->getNumElements());
1796
1797 gepIndices.push_back(zero);
1798 countFromCLAs *= llvmArrayType->getNumElements();
1799 eltType = arrayType->getElementType();
1800
1801 llvmArrayType =
1802 dyn_cast<llvm::ArrayType>(llvmArrayType->getElementType());
1803 arrayType = getContext().getAsArrayType(arrayType->getElementType());
1804 assert((!llvmArrayType || arrayType) &&
1805 "LLVM and Clang types are out-of-synch");
1806 }
1807
1808 if (arrayType) {
1809 // From this point onwards, the Clang array type has been emitted
1810 // as some other type (probably a packed struct). Compute the array
1811 // size, and just emit the 'begin' expression as a bitcast.
1812 while (arrayType) {
1813 countFromCLAs *=
1814 cast<ConstantArrayType>(arrayType)->getSize().getZExtValue();
1815 eltType = arrayType->getElementType();
1816 arrayType = getContext().getAsArrayType(eltType);
1817 }
1818
1819 llvm::Type *baseType = ConvertType(eltType);
1820 addr = Builder.CreateElementBitCast(addr, baseType, "array.begin");
1821 } else {
1822 // Create the actual GEP.
1823 addr = Address(Builder.CreateInBoundsGEP(addr.getPointer(),
1824 gepIndices, "array.begin"),
1825 addr.getAlignment());
1826 }
1827
1828 baseType = eltType;
1829
1830 llvm::Value *numElements
1831 = llvm::ConstantInt::get(SizeTy, countFromCLAs);
1832
1833 // If we had any VLA dimensions, factor them in.
1834 if (numVLAElements)
1835 numElements = Builder.CreateNUWMul(numVLAElements, numElements);
1836
1837 return numElements;
1838}

1885void CodeGenFunction::EmitVariablyModifiedType(QualType type) {
1886 assert(type->isVariablyModifiedType() &&
1887 "Must pass variably modified type to EmitVLASizes!");
1888
1889 EnsureInsertPoint();
1890
1891 // We're going to walk down into the type and look for VLA
1892 // expressions.
1893 do {
1894 assert(type->isVariablyModifiedType());
1895
1896 const Type *ty = type.getTypePtr();
1897 switch (ty->getTypeClass()) {
1898
1899#define TYPE(Class, Base)
1900#define ABSTRACT_TYPE(Class, Base)
1901#define NON_CANONICAL_TYPE(Class, Base)
1902#define DEPENDENT_TYPE(Class, Base) case Type::Class:
1903#define NON_CANONICAL_UNLESS_DEPENDENT_TYPE(Class, Base)
1904#include "clang/AST/TypeNodes.def"
1905 llvm_unreachable("unexpected dependent type!");
1906
1907 // These types are never variably-modified.
1908 case Type::Builtin:
1909 case Type::Complex:
1910 case Type::Vector:
1911 case Type::ExtVector:
1912 case Type::Record:
1913 case Type::Enum:
1914 case Type::Elaborated:
1915 case Type::TemplateSpecialization:
1916 case Type::ObjCTypeParam:
1917 case Type::ObjCObject:
1918 case Type::ObjCInterface:
1919 case Type::ObjCObjectPointer:
1920 llvm_unreachable("type class is never variably-modified!");
1921
1922 case Type::Adjusted:
1923 type = cast<AdjustedType>(ty)->getAdjustedType();
1924 break;
1925
1926 case Type::Decayed:
1927 type = cast<DecayedType>(ty)->getPointeeType();
1928 break;
1929
1930 case Type::Pointer:
1931 type = cast<PointerType>(ty)->getPointeeType();
1932 break;
1933
1934 case Type::BlockPointer:
1935 type = cast<BlockPointerType>(ty)->getPointeeType();
1936 break;
1937
1938 case Type::LValueReference:
1939 case Type::RValueReference:
1940 type = cast<ReferenceType>(ty)->getPointeeType();
1941 break;
1942
1943 case Type::MemberPointer:
1944 type = cast<MemberPointerType>(ty)->getPointeeType();
1945 break;
1946
1947 case Type::ConstantArray:
1948 case Type::IncompleteArray:
1949 // Losing element qualification here is fine.
1950 type = cast<ArrayType>(ty)->getElementType();
1951 break;
1952
1953 case Type::VariableArray: {
1954 // Losing element qualification here is fine.
1955 const VariableArrayType *vat = cast<VariableArrayType>(ty);
1956
1957 // Unknown size indication requires no size computation.
1958 // Otherwise, evaluate and record it.
1959 if (const Expr *size = vat->getSizeExpr()) {
1960 // It's possible that we might have emitted this already,
1961 // e.g. with a typedef and a pointer to it.
1962 llvm::Value *&entry = VLASizeMap[size];
1963 if (!entry) {
1964 llvm::Value *Size = EmitScalarExpr(size);
1965
1966 // C11 6.7.6.2p5:
1967 // If the size is an expression that is not an integer constant
1968 // expression [...] each time it is evaluated it shall have a value
1969 // greater than zero.
1970 if (SanOpts.has(SanitizerKind::VLABound) &&
1971 size->getType()->isSignedIntegerType()) {
1972 SanitizerScope SanScope(this);
1973 llvm::Value *Zero = llvm::Constant::getNullValue(Size->getType());
1974 llvm::Constant *StaticArgs[] = {
1975 EmitCheckSourceLocation(size->getBeginLoc()),
1976 EmitCheckTypeDescriptor(size->getType())};
1977 EmitCheck(std::make_pair(Builder.CreateICmpSGT(Size, Zero),
1978 SanitizerKind::VLABound),
1979 SanitizerHandler::VLABoundNotPositive, StaticArgs, Size);
1980 }
1981
1982 // Always zexting here would be wrong if it weren't
1983 // undefined behavior to have a negative bound.
1984 entry = Builder.CreateIntCast(Size, SizeTy, /*signed*/ false);
1985 }
1986 }
1987 type = vat->getElementType();
1988 break;
1989 }
1990
1991 case Type::FunctionProto:
1992 case Type::FunctionNoProto:
1993 type = cast<FunctionType>(ty)->getReturnType();
1994 break;
1995
1996 case Type::Paren:
1997 case Type::TypeOf:
1998 case Type::UnaryTransform:
1999 case Type::Attributed:
2000 case Type::SubstTemplateTypeParm:
2001 case Type::PackExpansion:
2002 case Type::MacroQualified:
2003 // Keep walking after single level desugaring.
2004 type = type.getSingleStepDesugaredType(getContext());
2005 break;
2006
2007 case Type::Typedef:
2008 case Type::Decltype:
2009 case Type::Auto:
2010 case Type::DeducedTemplateSpecialization:
2011 // Stop walking: nothing to do.
2012 return;
2013
2014 case Type::TypeOfExpr:
2015 // Stop walking: emit typeof expression.
2016 EmitIgnoredExpr(cast<TypeOfExprType>(ty)->getUnderlyingExpr());
2017 return;
2018
2019 case Type::Atomic:
2020 type = cast<AtomicType>(ty)->getValueType();
2021 break;
2022
2023 case Type::Pipe:
2024 type = cast<PipeType>(ty)->getElementType();
2025 break;
2026 }
2027 } while (type->isVariablyModifiedType());
2028}

2193static bool hasRequiredFeatures(const SmallVectorImpl<StringRef> &ReqFeatures,
2194 CodeGenModule &CGM, const FunctionDecl *FD,
2195 std::string &FirstMissing) {
2196 // If there aren't any required features listed then go ahead and return.
2197 if (ReqFeatures.empty())
2198 return false;
2199
2200 // Now build up the set of caller features and verify that all the required
2201 // features are there.
2202 llvm::StringMap<bool> CallerFeatureMap;
2203 CGM.getFunctionFeatureMap(CallerFeatureMap, GlobalDecl().getWithDecl(FD));
2204
2205 // If we have at least one of the features in the feature list return
2206 // true, otherwise return false.
2207 return std::all_of(
2208 ReqFeatures.begin(), ReqFeatures.end(), [&](StringRef Feature) {
2209 SmallVector<StringRef, 1> OrFeatures;
2210 Feature.split(OrFeatures, '|');
2211 return llvm::any_of(OrFeatures, [&](StringRef Feature) {
2212 if (!CallerFeatureMap.lookup(Feature)) {
2213 FirstMissing = Feature.str();
2214 return false;
2215 }
2216 return true;
2217 });
2218 });

2221// Emits an error if we don't have a valid set of target features for the
2222// called function.
2223void CodeGenFunction::checkTargetFeatures(const CallExpr *E,
2224 const FunctionDecl *TargetDecl) {
2225 // Early exit if this is an indirect call.
2226 if (!TargetDecl)
2227 return;
2228
2229 // Get the current enclosing function if it exists. If it doesn't
2230 // we can't check the target features anyhow.
2231 const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(CurFuncDecl);
2232 if (!FD)
2233 return;
2234
2235 // Grab the required features for the call. For a builtin this is listed in
2236 // the td file with the default cpu, for an always_inline function this is any
2237 // listed cpu and any listed features.
2238 unsigned BuiltinID = TargetDecl->getBuiltinID();
2239 std::string MissingFeature;
2240 if (BuiltinID) {
2241 SmallVector<StringRef, 1> ReqFeatures;
2242 const char *FeatureList =
2243 CGM.getContext().BuiltinInfo.getRequiredFeatures(BuiltinID);
2244 // Return if the builtin doesn't have any required features.
2245 if (!FeatureList || StringRef(FeatureList) == "")
2246 return;
2247 StringRef(FeatureList).split(ReqFeatures, ',');
2248 if (!hasRequiredFeatures(ReqFeatures, CGM, FD, MissingFeature))
2249 CGM.getDiags().Report(E->getBeginLoc(), diag::err_builtin_needs_feature)
2250 << TargetDecl->getDeclName()
2251 << CGM.getContext().BuiltinInfo.getRequiredFeatures(BuiltinID);
2252
2253 } else if (TargetDecl->hasAttr<TargetAttr>() ||
2254 TargetDecl->hasAttr<CPUSpecificAttr>()) {
2255 // Get the required features for the callee.
2256
2257 const TargetAttr *TD = TargetDecl->getAttr<TargetAttr>();
2258 TargetAttr::ParsedTargetAttr ParsedAttr = CGM.filterFunctionTargetAttrs(TD);
2259
2260 SmallVector<StringRef, 1> ReqFeatures;
2261 llvm::StringMap<bool> CalleeFeatureMap;
2262 CGM.getFunctionFeatureMap(CalleeFeatureMap, TargetDecl);
2263
2264 for (const auto &F : ParsedAttr.Features) {
2265 if (F[0] == '+' && CalleeFeatureMap.lookup(F.substr(1)))
2266 ReqFeatures.push_back(StringRef(F).substr(1));
2267 }
2268
2269 for (const auto &F : CalleeFeatureMap) {
2270 // Only positive features are "required".
2271 if (F.getValue())
2272 ReqFeatures.push_back(F.getKey());
2273 }
2274 if (!hasRequiredFeatures(ReqFeatures, CGM, FD, MissingFeature))
2275 CGM.getDiags().Report(E->getBeginLoc(), diag::err_function_needs_feature)
2276 << FD->getDeclName() << TargetDecl->getDeclName() << MissingFeature;
2277 }
2278}

2327void CodeGenFunction::EmitMultiVersionResolver(
2328 llvm::Function *Resolver, ArrayRef<MultiVersionResolverOption> Options) {
2329 assert((getContext().getTargetInfo().getTriple().getArch() ==
2330 llvm::Triple::x86 ||
2331 getContext().getTargetInfo().getTriple().getArch() ==
2332 llvm::Triple::x86_64) &&
2333 "Only implemented for x86 targets");
2334
2335 bool SupportsIFunc = getContext().getTargetInfo().supportsIFunc();
2336
2337 // Main function's basic block.
2338 llvm::BasicBlock *CurBlock = createBasicBlock("resolver_entry", Resolver);
2339 Builder.SetInsertPoint(CurBlock);
2340 EmitX86CpuInit();
2341
2342 for (const MultiVersionResolverOption &RO : Options) {
2343 Builder.SetInsertPoint(CurBlock);
2344 llvm::Value *Condition = FormResolverCondition(RO);
2345
2346 // The 'default' or 'generic' case.
2347 if (!Condition) {
2348 assert(&RO == Options.end() - 1 &&
2349 "Default or Generic case must be last");
2350 CreateMultiVersionResolverReturn(CGM, Resolver, Builder, RO.Function,
2351 SupportsIFunc);
2352 return;
2353 }
2354
2355 llvm::BasicBlock *RetBlock = createBasicBlock("resolver_return", Resolver);
2356 CGBuilderTy RetBuilder(*this, RetBlock);
2357 CreateMultiVersionResolverReturn(CGM, Resolver, RetBuilder, RO.Function,
2358 SupportsIFunc);
2359 CurBlock = createBasicBlock("resolver_else", Resolver);
2360 Builder.CreateCondBr(Condition, RetBlock, CurBlock);
2361 }
2362
2363 // If no generic/default, emit an unreachable.
2364 Builder.SetInsertPoint(CurBlock);
2365 llvm::CallInst *TrapCall = EmitTrapCall(llvm::Intrinsic::trap);
2366 TrapCall->setDoesNotReturn();
2367 TrapCall->setDoesNotThrow();
2368 Builder.CreateUnreachable();
2369 Builder.ClearInsertionPoint();
2370}

2372// Loc - where the diagnostic will point, where in the source code this
2373// alignment has failed.
2374// SecondaryLoc - if present (will be present if sufficiently different from
2375// Loc), the diagnostic will additionally point a "Note:" to this location.
2376// It should be the location where the __attribute__((assume_aligned))
2377// was written e.g.
2378void CodeGenFunction::EmitAlignmentAssumptionCheck(
2379 llvm::Value *Ptr, QualType Ty, SourceLocation Loc,
2380 SourceLocation SecondaryLoc, llvm::Value *Alignment,
2381 llvm::Value *OffsetValue, llvm::Value *TheCheck,
2382 llvm::Instruction *Assumption) {
2383 assert(Assumption && isa<llvm::CallInst>(Assumption) &&
2384 cast<llvm::CallInst>(Assumption)->getCalledValue() ==
2385 llvm::Intrinsic::getDeclaration(
2386 Builder.GetInsertBlock()->getParent()->getParent(),
2387 llvm::Intrinsic::assume) &&
2388 "Assumption should be a call to llvm.assume().");
2389 assert(&(Builder.GetInsertBlock()->back()) == Assumption &&
2390 "Assumption should be the last instruction of the basic block, "
2391 "since the basic block is still being generated.");
2392
2393 if (!SanOpts.has(SanitizerKind::Alignment))
2394 return;
2395
2396 // Don't check pointers to volatile data. The behavior here is implementation-
2397 // defined.
2398 if (Ty->getPointeeType().isVolatileQualified())
2399 return;
2400
2401 // We need to temorairly remove the assumption so we can insert the
2402 // sanitizer check before it, else the check will be dropped by optimizations.
2403 Assumption->removeFromParent();
2404
2405 {
2406 SanitizerScope SanScope(this);
2407
2408 if (!OffsetValue)
2409 OffsetValue = Builder.getInt1(0); // no offset.
2410
2411 llvm::Constant *StaticData[] = {EmitCheckSourceLocation(Loc),
2412 EmitCheckSourceLocation(SecondaryLoc),
2413 EmitCheckTypeDescriptor(Ty)};
2414 llvm::Value *DynamicData[] = {EmitCheckValue(Ptr),
2415 EmitCheckValue(Alignment),
2416 EmitCheckValue(OffsetValue)};
