22void CGCXXABI::ErrorUnsupportedABI(CodeGenFunction &CGF, StringRef S) {
23 DiagnosticsEngine &Diags = CGF.CGM.getDiags();
24 unsigned DiagID = Diags.getCustomDiagID(DiagnosticsEngine::Error,
25 "cannot yet compile %0 in this ABI");
26 Diags.Report(CGF.getContext().getFullLoc(CGF.CurCodeDecl->getLocation()),
27 DiagID)
28 << S;
29}

40CGCallee CGCXXABI::EmitLoadOfMemberFunctionPointer(
41 CodeGenFunction &CGF, const Expr *E, Address This,
42 llvm::Value *&ThisPtrForCall,
43 llvm::Value *MemPtr, const MemberPointerType *MPT) {
44 ErrorUnsupportedABI(CGF, "calls through member pointers");
45
46 ThisPtrForCall = This.getPointer();
47 const FunctionProtoType *FPT =
48 MPT->getPointeeType()->getAs<FunctionProtoType>();
49 const CXXRecordDecl *RD =
50 cast<CXXRecordDecl>(MPT->getClass()->getAs<RecordType>()->getDecl());
51 llvm::FunctionType *FTy = CGM.getTypes().GetFunctionType(
52 CGM.getTypes().arrangeCXXMethodType(RD, FPT, /*FD=*/nullptr));
53 unsigned AS = CGM.getTargetCodeGenInfo().getDefaultAS();
54 llvm::Constant *FnPtr = llvm::Constant::getNullValue(FTy->getPointerTo(AS));
55 return CGCallee::forDirect(FnPtr, FPT);
56}

123void CGCXXABI::buildThisParam(CodeGenFunction &CGF, FunctionArgList &params) {
124 const CXXMethodDecl *MD = cast<CXXMethodDecl>(CGF.CurGD.getDecl());
125
126 // FIXME: I'm not entirely sure I like using a fake decl just for code
127 // generation. Maybe we can come up with a better way?
128 auto *ThisDecl = ImplicitParamDecl::Create(
129 CGM.getContext(), nullptr, MD->getLocation(),
130 &CGM.getContext().Idents.get("this"), MD->getThisType(),
131 ImplicitParamDecl::CXXThis);
132 params.push_back(ThisDecl);
133 CGF.CXXABIThisDecl = ThisDecl;
134
135 // Compute the presumed alignment of 'this', which basically comes
136 // down to whether we know it's a complete object or not.
137 auto &Layout = CGF.getContext().getASTRecordLayout(MD->getParent());
138 if (MD->getParent()->getNumVBases() == 0 || // avoid vcall in common case
139 MD->getParent()->hasAttr<FinalAttr>() ||
140 !isThisCompleteObject(CGF.CurGD)) {
141 CGF.CXXABIThisAlignment = Layout.getAlignment();
142 } else {
143 CGF.CXXABIThisAlignment = Layout.getNonVirtualAlignment();
144 }
145}

184bool CGCXXABI::requiresArrayCookie(const CXXDeleteExpr *expr,
185 QualType elementType) {
186 // If the class's usual deallocation function takes two arguments,
187 // it needs a cookie.
188 if (expr->doesUsualArrayDeleteWantSize())
189 return true;
190
191 return elementType.isDestructedType();
192}

194bool CGCXXABI::requiresArrayCookie(const CXXNewExpr *expr) {
195 // If the class's usual deallocation function takes two arguments,
196 // it needs a cookie.
197 if (expr->doesUsualArrayDeleteWantSize())
198 return true;
199
200 return expr->getAllocatedType().isDestructedType();
201}

203void CGCXXABI::ReadArrayCookie(CodeGenFunction &CGF, Address ptr,
204 const CXXDeleteExpr *expr, QualType eltTy,
205 llvm::Value *&numElements,
206 llvm::Value *&allocPtr, CharUnits &cookieSize) {
207 // Derive a char* in the same address space as the pointer.
208 ptr = CGF.Builder.CreateElementBitCast(ptr, CGF.Int8Ty);
209
210 // If we don't need an array cookie, bail out early.
211 if (!requiresArrayCookie(expr, eltTy)) {
212 allocPtr = ptr.getPointer();
213 numElements = nullptr;
214 cookieSize = CharUnits::Zero();
215 return;
216 }
217
218 cookieSize = getArrayCookieSizeImpl(eltTy);
219 Address allocAddr =
220 CGF.Builder.CreateConstInBoundsByteGEP(ptr, -cookieSize);
221 allocPtr = allocAddr.getPointer();
222 numElements = readArrayCookieImpl(CGF, allocAddr, cookieSize);
223}

225llvm::Value *CGCXXABI::readArrayCookieImpl(CodeGenFunction &CGF,
226 Address ptr,
227 CharUnits cookieSize) {
228 ErrorUnsupportedABI(CGF, "reading a new[] cookie");
229 return llvm::ConstantInt::get(CGF.SizeTy, 0);
230}

232/// Returns the adjustment, in bytes, required for the given
233/// member-pointer operation. Returns null if no adjustment is
234/// required.
235llvm::Constant *CGCXXABI::getMemberPointerAdjustment(const CastExpr *E) {
236 assert(E->getCastKind() == CK_DerivedToBaseMemberPointer ||
237 E->getCastKind() == CK_BaseToDerivedMemberPointer);
238
239 QualType derivedType;
240 if (E->getCastKind() == CK_DerivedToBaseMemberPointer)
241 derivedType = E->getSubExpr()->getType();
242 else
243 derivedType = E->getType();
244
245 const CXXRecordDecl *derivedClass =
246 derivedType->castAs<MemberPointerType>()->getClass()->getAsCXXRecordDecl();
247
248 return CGM.GetNonVirtualBaseClassOffset(derivedClass,
249 E->path_begin(),
250 E->path_end());
251}

253CharUnits CGCXXABI::getMemberPointerPathAdjustment(const APValue &MP) {
254 // TODO: Store base specifiers in APValue member pointer paths so we can
255 // easily reuse CGM.GetNonVirtualBaseClassOffset().
256 const ValueDecl *MPD = MP.getMemberPointerDecl();
257 CharUnits ThisAdjustment = CharUnits::Zero();
258 ArrayRef<const CXXRecordDecl*> Path = MP.getMemberPointerPath();
259 bool DerivedMember = MP.isMemberPointerToDerivedMember();
260 const CXXRecordDecl *RD = cast<CXXRecordDecl>(MPD->getDeclContext());
261 for (unsigned I = 0, N = Path.size(); I != N; ++I) {
262 const CXXRecordDecl *Base = RD;
263 const CXXRecordDecl *Derived = Path[I];
264 if (DerivedMember)
265 std::swap(Base, Derived);
266 ThisAdjustment +=
267 getContext().getASTRecordLayout(Derived).getBaseClassOffset(Base);
268 RD = Path[I];
269 }
270 if (DerivedMember)
271 ThisAdjustment = -ThisAdjustment;
272 return ThisAdjustment;
273}
