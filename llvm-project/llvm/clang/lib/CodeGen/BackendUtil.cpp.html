233// Check if ASan should use GC-friendly instrumentation for globals.
234// First of all, there is no point if -fdata-sections is off (expect for MachO,
235// where this is not a factor). Also, on ELF this feature requires an assembler
236// extension that only works with -integrated-as at the moment.
237static bool asanUseGlobalsGC(const Triple &T, const CodeGenOptions &CGOpts) {
238 if (!CGOpts.SanitizeAddressGlobalsDeadStripping)
239 return false;
240 switch (T.getObjectFormat()) {
241 case Triple::MachO:
242 case Triple::COFF:
243 return true;
244 case Triple::ELF:
245 return CGOpts.DataSections && !CGOpts.DisableIntegratedAS;
246 default:
247 return false;
248 }
249}

339static TargetLibraryInfoImpl *createTLII(llvm::Triple &TargetTriple,
340 const CodeGenOptions &CodeGenOpts) {
341 TargetLibraryInfoImpl *TLII = new TargetLibraryInfoImpl(TargetTriple);
342 if (!CodeGenOpts.SimplifyLibCalls)
343 TLII->disableAllFunctions();
344 else {
345 // Disable individual libc/libm calls in TargetLibraryInfo.
346 LibFunc F;
347 for (auto &FuncName : CodeGenOpts.getNoBuiltinFuncs())
348 if (TLII->getLibFunc(FuncName, F))
349 TLII->setUnavailable(F);
350 }
351
352 switch (CodeGenOpts.getVecLib()) {
353 case CodeGenOptions::Accelerate:
354 TLII->addVectorizableFunctionsFromVecLib(TargetLibraryInfoImpl::Accelerate);
355 break;
356 case CodeGenOptions::MASSV:
357 TLII->addVectorizableFunctionsFromVecLib(TargetLibraryInfoImpl::MASSV);
358 break;
359 case CodeGenOptions::SVML:
360 TLII->addVectorizableFunctionsFromVecLib(TargetLibraryInfoImpl::SVML);
361 break;
362 default:
363 break;
364 }
365 return TLII;
366}

421static void initTargetOptions(llvm::TargetOptions &Options,
422 const CodeGenOptions &CodeGenOpts,
423 const clang::TargetOptions &TargetOpts,
424 const LangOptions &LangOpts,
425 const HeaderSearchOptions &HSOpts) {
426 Options.ThreadModel =
427 llvm::StringSwitch<llvm::ThreadModel::Model>(CodeGenOpts.ThreadModel)
428 .Case("posix", llvm::ThreadModel::POSIX)
429 .Case("single", llvm::ThreadModel::Single);
430
431 // Set float ABI type.
432 assert((CodeGenOpts.FloatABI == "soft" || CodeGenOpts.FloatABI == "softfp" ||
433 CodeGenOpts.FloatABI == "hard" || CodeGenOpts.FloatABI.empty()) &&
434 "Invalid Floating Point ABI!");
435 Options.FloatABIType =
436 llvm::StringSwitch<llvm::FloatABI::ABIType>(CodeGenOpts.FloatABI)
437 .Case("soft", llvm::FloatABI::Soft)
438 .Case("softfp", llvm::FloatABI::Soft)
439 .Case("hard", llvm::FloatABI::Hard)
440 .Default(llvm::FloatABI::Default);
441
442 // Set FP fusion mode.
443 switch (LangOpts.getDefaultFPContractMode()) {
444 case LangOptions::FPC_Off:
445 // Preserve any contraction performed by the front-end. (Strict performs
446 // splitting of the muladd intrinsic in the backend.)
447 Options.AllowFPOpFusion = llvm::FPOpFusion::Standard;
448 break;
449 case LangOptions::FPC_On:
450 Options.AllowFPOpFusion = llvm::FPOpFusion::Standard;
451 break;
452 case LangOptions::FPC_Fast:
453 Options.AllowFPOpFusion = llvm::FPOpFusion::Fast;
454 break;
455 }
456
457 Options.UseInitArray = CodeGenOpts.UseInitArray;
458 Options.DisableIntegratedAS = CodeGenOpts.DisableIntegratedAS;
459 Options.CompressDebugSections = CodeGenOpts.getCompressDebugSections();
460 Options.RelaxELFRelocations = CodeGenOpts.RelaxELFRelocations;
461
462 // Set EABI version.
463 Options.EABIVersion = TargetOpts.EABIVersion;
464
465 if (LangOpts.SjLjExceptions)
466 Options.ExceptionModel = llvm::ExceptionHandling::SjLj;
467 if (LangOpts.SEHExceptions)
468 Options.ExceptionModel = llvm::ExceptionHandling::WinEH;
469 if (LangOpts.DWARFExceptions)
470 Options.ExceptionModel = llvm::ExceptionHandling::DwarfCFI;
471
472 Options.NoInfsFPMath = CodeGenOpts.NoInfsFPMath;
473 Options.NoNaNsFPMath = CodeGenOpts.NoNaNsFPMath;
474 Options.NoZerosInBSS = CodeGenOpts.NoZeroInitializedInBSS;
475 Options.UnsafeFPMath = CodeGenOpts.UnsafeFPMath;
476 Options.StackAlignmentOverride = CodeGenOpts.StackAlignment;
477 Options.FunctionSections = CodeGenOpts.FunctionSections;
478 Options.DataSections = CodeGenOpts.DataSections;
479 Options.UniqueSectionNames = CodeGenOpts.UniqueSectionNames;
480 Options.EmulatedTLS = CodeGenOpts.EmulatedTLS;
481 Options.ExplicitEmulatedTLS = CodeGenOpts.ExplicitEmulatedTLS;
482 Options.DebuggerTuning = CodeGenOpts.getDebuggerTuning();
483 Options.EmitStackSizeSection = CodeGenOpts.StackSizeSection;
484 Options.EmitAddrsig = CodeGenOpts.Addrsig;
485
486 if (CodeGenOpts.getSplitDwarfMode() != CodeGenOptions::NoFission)
487 Options.MCOptions.SplitDwarfFile = CodeGenOpts.SplitDwarfFile;
488 Options.MCOptions.MCRelaxAll = CodeGenOpts.RelaxAll;
489 Options.MCOptions.MCSaveTempLabels = CodeGenOpts.SaveTempLabels;
490 Options.MCOptions.MCUseDwarfDirectory = !CodeGenOpts.NoDwarfDirectoryAsm;
491 Options.MCOptions.MCNoExecStack = CodeGenOpts.NoExecStack;
492 Options.MCOptions.MCIncrementalLinkerCompatible =
493 CodeGenOpts.IncrementalLinkerCompatible;
494 Options.MCOptions.MCPIECopyRelocations = CodeGenOpts.PIECopyRelocations;
495 Options.MCOptions.MCFatalWarnings = CodeGenOpts.FatalWarnings;
496 Options.MCOptions.AsmVerbose = CodeGenOpts.AsmVerbose;
497 Options.MCOptions.PreserveAsmComments = CodeGenOpts.PreserveAsmComments;
498 Options.MCOptions.ABIName = TargetOpts.ABI;
499 for (const auto &Entry : HSOpts.UserEntries)
500 if (!Entry.IsFramework &&
501 (Entry.Group == frontend::IncludeDirGroup::Quoted ||
502 Entry.Group == frontend::IncludeDirGroup::Angled ||
503 Entry.Group == frontend::IncludeDirGroup::System))
504 Options.MCOptions.IASSearchPaths.push_back(
505 Entry.IgnoreSysRoot ? Entry.Path : HSOpts.Sysroot + Entry.Path);
506}

542void EmitAssemblyHelper::CreatePasses(legacy::PassManager &MPM,
543 legacy::FunctionPassManager &FPM) {
544 // Handle disabling of all LLVM passes, where we want to preserve the
545 // internal module before any optimization.
546 if (CodeGenOpts.DisableLLVMPasses)
547 return;
548
549 // Figure out TargetLibraryInfo. This needs to be added to MPM and FPM
550 // manually (and not via PMBuilder), since some passes (eg. InstrProfiling)
551 // are inserted before PMBuilder ones - they'd get the default-constructed
552 // TLI with an unknown target otherwise.
553 Triple TargetTriple(TheModule->getTargetTriple());
554 std::unique_ptr<TargetLibraryInfoImpl> TLII(
555 createTLII(TargetTriple, CodeGenOpts));
556
557 PassManagerBuilderWrapper PMBuilder(TargetTriple, CodeGenOpts, LangOpts);
558
559 // At O0 and O1 we only run the always inliner which is more efficient. At
560 // higher optimization levels we run the normal inliner.
561 if (CodeGenOpts.OptimizationLevel <= 1) {
562 bool InsertLifetimeIntrinsics = (CodeGenOpts.OptimizationLevel != 0 &&
563 !CodeGenOpts.DisableLifetimeMarkers);
564 PMBuilder.Inliner = createAlwaysInlinerLegacyPass(InsertLifetimeIntrinsics);
565 if (!llvm::MCTargetOptions::cheriUsesCapabilityTable())
566 PMBuilder.addExtension(PassManagerBuilder::EP_EarlyAsPossible,
567 addCHERICapDirectCallsPass);
568 } else {
569 // We do not want to inline hot callsites for SamplePGO module-summary build
570 // because profile annotation will happen again in ThinLTO backend, and we
571 // want the IR of the hot path to match the profile.
572 PMBuilder.Inliner = createFunctionInliningPass(
573 CodeGenOpts.OptimizationLevel, CodeGenOpts.OptimizeSize,
574 (!CodeGenOpts.SampleProfileFile.empty() &&
575 CodeGenOpts.PrepareForThinLTO));
576 }
577
578 PMBuilder.OptLevel = CodeGenOpts.OptimizationLevel;
579 PMBuilder.SizeLevel = CodeGenOpts.OptimizeSize;
580 PMBuilder.SLPVectorize = CodeGenOpts.VectorizeSLP;
581 PMBuilder.LoopVectorize = CodeGenOpts.VectorizeLoop;

821void EmitAssemblyHelper::EmitAssembly(BackendAction Action,
822 std::unique_ptr<raw_pwrite_stream> OS) {
823 TimeRegion Region(FrontendTimesIsEnabled ? &CodeGenerationTime : nullptr);
824
825 setCommandLineOpts(CodeGenOpts);
826
827 bool UsesCodeGen = (Action != Backend_EmitNothing &&
828 Action != Backend_EmitBC &&
829 Action != Backend_EmitLL);
830 CreateTargetMachine(UsesCodeGen);
831
832 if (UsesCodeGen && !TM)
833 return;
834 if (TM)
835 TheModule->setDataLayout(TM->createDataLayout());
836
837 legacy::PassManager PerModulePasses;
838 PerModulePasses.add(
839 createTargetTransformInfoWrapperPass(getTargetIRAnalysis()));
840
841 legacy::FunctionPassManager PerFunctionPasses(TheModule);
842 PerFunctionPasses.add(
843 createTargetTransformInfoWrapperPass(getTargetIRAnalysis()));
844
845 CreatePasses(PerModulePasses, PerFunctionPasses);
846
847 legacy::PassManager CodeGenPasses;
848 CodeGenPasses.add(
849 createTargetTransformInfoWrapperPass(getTargetIRAnalysis()));
850
851 std::unique_ptr<llvm::ToolOutputFile> ThinLinkOS, DwoOS;
852
853 switch (Action) {
854 case Backend_EmitNothing:
855 break;
856
857 case Backend_EmitBC:
858 if (CodeGenOpts.PrepareForThinLTO && !CodeGenOpts.DisableLLVMPasses) {
859 if (!CodeGenOpts.ThinLinkBitcodeFile.empty()) {
860 ThinLinkOS = openOutputFile(CodeGenOpts.ThinLinkBitcodeFile);
861 if (!ThinLinkOS)
862 return;
863 }
864 TheModule->addModuleFlag(Module::Error, "EnableSplitLTOUnit",
865 CodeGenOpts.EnableSplitLTOUnit);
866 PerModulePasses.add(createWriteThinLTOBitcodePass(
867 *OS, ThinLinkOS ? &ThinLinkOS->os() : nullptr));
868 } else {
869 // Emit a module summary by default for Regular LTO except for ld64
870 // targets
871 bool EmitLTOSummary =
872 (CodeGenOpts.PrepareForLTO &&
873 !CodeGenOpts.DisableLLVMPasses &&
874 llvm::Triple(TheModule->getTargetTriple()).getVendor() !=
875 llvm::Triple::Apple);
876 if (EmitLTOSummary) {
877 if (!TheModule->getModuleFlag("ThinLTO"))
878 TheModule->addModuleFlag(Module::Error, "ThinLTO", uint32_t(0));
879 TheModule->addModuleFlag(Module::Error, "EnableSplitLTOUnit",
880 CodeGenOpts.EnableSplitLTOUnit);
881 }
882
883 PerModulePasses.add(createBitcodeWriterPass(
884 *OS, CodeGenOpts.EmitLLVMUseLists, EmitLTOSummary));
885 }
886 break;
887
888 case Backend_EmitLL:
889 PerModulePasses.add(
890 createPrintModulePass(*OS, "", CodeGenOpts.EmitLLVMUseLists));
891 break;
892
893 default:
894 if (!CodeGenOpts.SplitDwarfFile.empty() &&
895 (CodeGenOpts.getSplitDwarfMode() == CodeGenOptions::SplitFileFission)) {
896 DwoOS = openOutputFile(CodeGenOpts.SplitDwarfFile);
897 if (!DwoOS)
898 return;
899 }
900 if (!AddEmitPasses(CodeGenPasses, Action, *OS,
901 DwoOS ? &DwoOS->os() : nullptr))
902 return;
903 }
904
905 // Before executing passes, print the final values of the LLVM options.
906 cl::PrintOptionValues();
907
908 // Run passes. For now we do all passes at once, but eventually we
909 // would like to have the option of streaming code generation.
910
911 {
912 PrettyStackTraceString CrashInfo("Per-function optimization");
913
914 PerFunctionPasses.doInitialization();
915 for (Function &F : *TheModule)
916 if (!F.isDeclaration())
917 PerFunctionPasses.run(F);
918 PerFunctionPasses.doFinalization();
919 }
920
921 {
922 PrettyStackTraceString CrashInfo("Per-module optimization passes");
923 PerModulePasses.run(*TheModule);
924 }
925
926 {
927 PrettyStackTraceString CrashInfo("Code generation");
928 CodeGenPasses.run(*TheModule);
929 }
930
931 if (ThinLinkOS)
932 ThinLinkOS->keep();
933 if (DwoOS)
934 DwoOS->keep();
935}

953 case 1:
954 return PassBuilder::Os;
955
956 case 2:
957 return PassBuilder::Oz;
958 }
959
960 case 3:
961 return PassBuilder::O3;
962 }

1013/// A clean version of `EmitAssembly` that uses the new pass manager.
1014///
1015/// Not all features are currently supported in this system, but where
1016/// necessary it falls back to the legacy pass manager to at least provide
1017/// basic functionality.
1018///
1019/// This API is planned to have its functionality finished and then to replace
1020/// `EmitAssembly` at some point in the future when the default switches.
1021void EmitAssemblyHelper::EmitAssemblyWithNewPassManager(
1022 BackendAction Action, std::unique_ptr<raw_pwrite_stream> OS) {
1023 TimeRegion Region(FrontendTimesIsEnabled ? &CodeGenerationTime : nullptr);
1024 setCommandLineOpts(CodeGenOpts);
1025
1026 bool RequiresCodeGen = (Action != Backend_EmitNothing &&
1027 Action != Backend_EmitBC &&
1028 Action != Backend_EmitLL);
1029 CreateTargetMachine(RequiresCodeGen);
1030
1031 if (RequiresCodeGen && !TM)
1032 return;
1033 if (TM)
1034 TheModule->setDataLayout(TM->createDataLayout());
1035
1036 Optional<PGOOptions> PGOOpt;
1037
1038 if (CodeGenOpts.hasProfileIRInstr())
1039 // -fprofile-generate.
1040 PGOOpt = PGOOptions(CodeGenOpts.InstrProfileOutput.empty()
1041 ? DefaultProfileGenName
1042 : CodeGenOpts.InstrProfileOutput,
1043 "", "", PGOOptions::IRInstr, PGOOptions::NoCSAction,
1044 CodeGenOpts.DebugInfoForProfiling);
1045 else if (CodeGenOpts.hasProfileIRUse()) {
1046 // -fprofile-use.
1047 auto CSAction = CodeGenOpts.hasProfileCSIRUse() ? PGOOptions::CSIRUse
1048 : PGOOptions::NoCSAction;
1049 PGOOpt = PGOOptions(CodeGenOpts.ProfileInstrumentUsePath, "",
1050 CodeGenOpts.ProfileRemappingFile, PGOOptions::IRUse,
1051 CSAction, CodeGenOpts.DebugInfoForProfiling);
1052 } else if (!CodeGenOpts.SampleProfileFile.empty())
1053 // -fprofile-sample-use
1054 PGOOpt =
1055 PGOOptions(CodeGenOpts.SampleProfileFile, "",
1056 CodeGenOpts.ProfileRemappingFile, PGOOptions::SampleUse,
1057 PGOOptions::NoCSAction, CodeGenOpts.DebugInfoForProfiling);
1058 else if (CodeGenOpts.DebugInfoForProfiling)
1059 // -fdebug-info-for-profiling
1060 PGOOpt = PGOOptions("", "", "", PGOOptions::NoAction,
1061 PGOOptions::NoCSAction, true);
1062
1063 // Check to see if we want to generate a CS profile.
1064 if (CodeGenOpts.hasProfileCSIRInstr()) {
1065 assert(!CodeGenOpts.hasProfileCSIRUse() &&
1066 "Cannot have both CSProfileUse pass and CSProfileGen pass at "
1067 "the same time");
1068 if (PGOOpt.hasValue()) {
1069 assert(PGOOpt->Action != PGOOptions::IRInstr &&
1070 PGOOpt->Action != PGOOptions::SampleUse &&
1071 "Cannot run CSProfileGen pass with ProfileGen or SampleUse "
1072 " pass");
1073 PGOOpt->CSProfileGenFile = CodeGenOpts.InstrProfileOutput.empty()
1074 ? DefaultProfileGenName
1075 : CodeGenOpts.InstrProfileOutput;
1076 PGOOpt->CSAction = PGOOptions::CSIRInstr;
1077 } else
1078 PGOOpt = PGOOptions("",
1079 CodeGenOpts.InstrProfileOutput.empty()
1080 ? DefaultProfileGenName
1081 : CodeGenOpts.InstrProfileOutput,
1082 "", PGOOptions::NoAction, PGOOptions::CSIRInstr,
1083 CodeGenOpts.DebugInfoForProfiling);
1084 }
1085
1086 PipelineTuningOptions PTO;
1087 PTO.LoopUnrolling = CodeGenOpts.UnrollLoops;
1088 // For historical reasons, loop interleaving is set to mirror setting for loop
1089 // unrolling.
1090 PTO.LoopInterleaving = CodeGenOpts.UnrollLoops;
1091 PTO.LoopVectorization = CodeGenOpts.VectorizeLoop;
1092 PTO.SLPVectorization = CodeGenOpts.VectorizeSLP;
1093
1094 PassBuilder PB(TM.get(), PTO, PGOOpt);
1095
1096 // Attempt to load pass plugins and register their callbacks with PB.
1097 for (auto &PluginFN : CodeGenOpts.PassPlugins) {
1098 auto PassPlugin = PassPlugin::Load(PluginFN);
1099 if (PassPlugin) {
1100 PassPlugin->registerPassBuilderCallbacks(PB);
1101 } else {
1102 Diags.Report(diag::err_fe_unable_to_load_plugin)
1103 << PluginFN << toString(PassPlugin.takeError());
1104 }
1105 }
1106
1107 LoopAnalysisManager LAM(CodeGenOpts.DebugPassManager);
1108 FunctionAnalysisManager FAM(CodeGenOpts.DebugPassManager);
1109 CGSCCAnalysisManager CGAM(CodeGenOpts.DebugPassManager);
1110 ModuleAnalysisManager MAM(CodeGenOpts.DebugPassManager);
1111
1112 // Register the AA manager first so that our version is the one used.
1113 FAM.registerPass([&] { return PB.buildDefaultAAPipeline(); });
1114
1115 // Register the target library analysis directly and give it a customized
1116 // preset TLI.
1117 Triple TargetTriple(TheModule->getTargetTriple());
1118 std::unique_ptr<TargetLibraryInfoImpl> TLII(
1119 createTLII(TargetTriple, CodeGenOpts));
1120 FAM.registerPass([&] { return TargetLibraryAnalysis(*TLII); });
1121 MAM.registerPass([&] { return TargetLibraryAnalysis(*TLII); });
1122
1123 // Register all the basic analyses with the managers.
1124 PB.registerModuleAnalyses(MAM);
1125 PB.registerCGSCCAnalyses(CGAM);
1126 PB.registerFunctionAnalyses(FAM);
1127 PB.registerLoopAnalyses(LAM);
1128 PB.crossRegisterProxies(LAM, FAM, CGAM, MAM);
1129
1130 ModulePassManager MPM(CodeGenOpts.DebugPassManager);
1131
1132 if (!CodeGenOpts.DisableLLVMPasses) {
1133 bool IsThinLTO = CodeGenOpts.PrepareForThinLTO;
1134 bool IsLTO = CodeGenOpts.PrepareForLTO;
1135
1136 if (CodeGenOpts.OptimizationLevel == 0) {
1137 if (Optional<GCOVOptions> Options = getGCOVOptions(CodeGenOpts))
1138 MPM.addPass(GCOVProfilerPass(*Options));
1139 if (Optional<InstrProfOptions> Options =
1140 getInstrProfOptions(CodeGenOpts, LangOpts))
1141 MPM.addPass(InstrProfiling(*Options, false));
1142
1143 // Build a minimal pipeline based on the semantics required by Clang,
1144 // which is just that always inlining occurs.
1145 MPM.addPass(AlwaysInlinerPass());
1146
1147 // At -O0 we directly run necessary sanitizer passes.
1148 if (LangOpts.Sanitize.has(SanitizerKind::LocalBounds))
1149 MPM.addPass(createModuleToFunctionPassAdaptor(BoundsCheckingPass()));
1150
1151 // Lastly, add semantically necessary passes for LTO.
1152 if (IsLTO || IsThinLTO) {
1153 MPM.addPass(CanonicalizeAliasesPass());
1154 MPM.addPass(NameAnonGlobalPass());
1155 }
1156 } else {
1157 // Map our optimization levels into one of the distinct levels used to
1158 // configure the pipeline.
1159 PassBuilder::OptimizationLevel Level = mapToLevel(CodeGenOpts);
1160
1161 // Register callbacks to schedule sanitizer passes at the appropriate part of
1162 // the pipeline.
1163 // FIXME: either handle asan/the remaining sanitizers or error out
1164 if (LangOpts.Sanitize.has(SanitizerKind::LocalBounds))
1165 PB.registerScalarOptimizerLateEPCallback(
1166 [](FunctionPassManager &FPM, PassBuilder::OptimizationLevel Level) {
1167 FPM.addPass(BoundsCheckingPass());
1168 });
1169 if (LangOpts.Sanitize.has(SanitizerKind::Memory))
1170 PB.registerOptimizerLastEPCallback(
1171 [](FunctionPassManager &FPM, PassBuilder::OptimizationLevel Level) {
1172 FPM.addPass(MemorySanitizerPass({}));
1173 });
1174 if (LangOpts.Sanitize.has(SanitizerKind::Thread))
1175 PB.registerOptimizerLastEPCallback(
1176 [](FunctionPassManager &FPM, PassBuilder::OptimizationLevel Level) {
1177 FPM.addPass(ThreadSanitizerPass());
1178 });
1179 if (LangOpts.Sanitize.has(SanitizerKind::Address)) {
1180 PB.registerPipelineStartEPCallback([&](ModulePassManager &MPM) {
1181 MPM.addPass(
1182 RequireAnalysisPass<ASanGlobalsMetadataAnalysis, Module>());
1183 });
1184 bool Recover = CodeGenOpts.SanitizeRecover.has(SanitizerKind::Address);
1185 bool UseAfterScope = CodeGenOpts.SanitizeAddressUseAfterScope;
1186 PB.registerOptimizerLastEPCallback(
1187 [Recover, UseAfterScope](FunctionPassManager &FPM,
1188 PassBuilder::OptimizationLevel Level) {
1189 FPM.addPass(AddressSanitizerPass(
1190 /*CompileKernel=*/false, Recover, UseAfterScope));
1191 });
1192 bool ModuleUseAfterScope = asanUseGlobalsGC(TargetTriple, CodeGenOpts);
1193 bool UseOdrIndicator = CodeGenOpts.SanitizeAddressUseOdrIndicator;
1194 PB.registerPipelineStartEPCallback(
1195 [Recover, ModuleUseAfterScope,
1196 UseOdrIndicator](ModulePassManager &MPM) {
1197 MPM.addPass(ModuleAddressSanitizerPass(
1198 /*CompileKernel=*/false, Recover, ModuleUseAfterScope,
1199 UseOdrIndicator));
1200 });
1201 }
1202 if (LangOpts.Sanitize.has(SanitizerKind::HWAddress)) {
1203 bool Recover =
1204 CodeGenOpts.SanitizeRecover.has(SanitizerKind::HWAddress);
1205 PB.registerOptimizerLastEPCallback(
1206 [Recover](FunctionPassManager &FPM,
1207 PassBuilder::OptimizationLevel Level) {
1208 FPM.addPass(HWAddressSanitizerPass(
1209 /*CompileKernel=*/false, Recover));
1210 });
1211 }
1212 if (LangOpts.Sanitize.has(SanitizerKind::KernelHWAddress)) {
1213 PB.registerOptimizerLastEPCallback(
1214 [](FunctionPassManager &FPM, PassBuilder::OptimizationLevel Level) {
1215 FPM.addPass(HWAddressSanitizerPass(
1216 /*CompileKernel=*/true, /*Recover=*/true));
1217 });
1218 }
1219 if (Optional<GCOVOptions> Options = getGCOVOptions(CodeGenOpts))
1220 PB.registerPipelineStartEPCallback([Options](ModulePassManager &MPM) {
1221 MPM.addPass(GCOVProfilerPass(*Options));
1222 });
1223 if (Optional<InstrProfOptions> Options =
1224 getInstrProfOptions(CodeGenOpts, LangOpts))
1225 PB.registerPipelineStartEPCallback([Options](ModulePassManager &MPM) {
1226 MPM.addPass(InstrProfiling(*Options, false));
1227 });
1228
1229 if (IsThinLTO) {
1230 MPM = PB.buildThinLTOPreLinkDefaultPipeline(
1231 Level, CodeGenOpts.DebugPassManager);
1232 MPM.addPass(CanonicalizeAliasesPass());
1233 MPM.addPass(NameAnonGlobalPass());
1234 } else if (IsLTO) {
1235 MPM = PB.buildLTOPreLinkDefaultPipeline(Level,
1236 CodeGenOpts.DebugPassManager);
1237 MPM.addPass(CanonicalizeAliasesPass());
1238 MPM.addPass(NameAnonGlobalPass());
1239 } else {
1240 MPM = PB.buildPerModuleDefaultPipeline(Level,
1241 CodeGenOpts.DebugPassManager);
1242 }
1243 }
1244
1245 if (CodeGenOpts.OptimizationLevel == 0)
1246 addSanitizersAtO0(MPM, TargetTriple, LangOpts, CodeGenOpts);
1247 }
1248
1249 // FIXME: We still use the legacy pass manager to do code generation. We
1250 // create that pass manager here and use it as needed below.
1251 legacy::PassManager CodeGenPasses;
1252 bool NeedCodeGen = false;
1253 std::unique_ptr<llvm::ToolOutputFile> ThinLinkOS, DwoOS;
1254
1255 // Append any output we need to the pass manager.
1256 switch (Action) {
1257 case Backend_EmitNothing:
1258 break;
1259
1260 case Backend_EmitBC:
1261 if (CodeGenOpts.PrepareForThinLTO && !CodeGenOpts.DisableLLVMPasses) {
1262 if (!CodeGenOpts.ThinLinkBitcodeFile.empty()) {
1263 ThinLinkOS = openOutputFile(CodeGenOpts.ThinLinkBitcodeFile);
1264 if (!ThinLinkOS)
1265 return;
1266 }
1267 TheModule->addModuleFlag(Module::Error, "EnableSplitLTOUnit",
1268 CodeGenOpts.EnableSplitLTOUnit);
1269 MPM.addPass(ThinLTOBitcodeWriterPass(*OS, ThinLinkOS ? &ThinLinkOS->os()
1270 : nullptr));
1271 } else {
1272 // Emit a module summary by default for Regular LTO except for ld64
1273 // targets
1274 bool EmitLTOSummary =
1275 (CodeGenOpts.PrepareForLTO &&
1276 !CodeGenOpts.DisableLLVMPasses &&
1277 llvm::Triple(TheModule->getTargetTriple()).getVendor() !=
1278 llvm::Triple::Apple);
1279 if (EmitLTOSummary) {
1280 if (!TheModule->getModuleFlag("ThinLTO"))
1281 TheModule->addModuleFlag(Module::Error, "ThinLTO", uint32_t(0));
1282 TheModule->addModuleFlag(Module::Error, "EnableSplitLTOUnit",
1283 CodeGenOpts.EnableSplitLTOUnit);
1284 }
1285 MPM.addPass(
1286 BitcodeWriterPass(*OS, CodeGenOpts.EmitLLVMUseLists, EmitLTOSummary));
1287 }
1288 break;
1289
1290 case Backend_EmitLL:
1291 MPM.addPass(PrintModulePass(*OS, "", CodeGenOpts.EmitLLVMUseLists));
1292 break;
1293
1294 case Backend_EmitAssembly:
1295 case Backend_EmitMCNull:
1296 case Backend_EmitObj:
1297 NeedCodeGen = true;
1298 CodeGenPasses.add(
1299 createTargetTransformInfoWrapperPass(getTargetIRAnalysis()));
1300 if (!CodeGenOpts.SplitDwarfFile.empty()) {
1301 DwoOS = openOutputFile(CodeGenOpts.SplitDwarfFile);
1302 if (!DwoOS)
1303 return;
1304 }
1305 if (!AddEmitPasses(CodeGenPasses, Action, *OS,
1306 DwoOS ? &DwoOS->os() : nullptr))
1307 // FIXME: Should we handle this error differently?
1308 return;
1309 break;
1310 }
1311
1312 // Before executing passes, print the final values of the LLVM options.
1313 cl::PrintOptionValues();
1314
1315 // Now that we have all of the passes ready, run them.
1316 {
1317 PrettyStackTraceString CrashInfo("Optimizer");
1318 MPM.run(*TheModule, MAM);
1319 }
1320
1321 // Now if needed, run the legacy PM for codegen.
1322 if (NeedCodeGen) {
1323 PrettyStackTraceString CrashInfo("Code generation");
1324 CodeGenPasses.run(*TheModule);
1325 }
1326
1327 if (ThinLinkOS)
1328 ThinLinkOS->keep();
1329 if (DwoOS)
1330 DwoOS->keep();
1331}

1333Expected<BitcodeModule> clang::FindThinLTOModule(MemoryBufferRef MBRef) {
1334 Expected<std::vector<BitcodeModule>> BMsOrErr = getBitcodeModuleList(MBRef);
1335 if (!BMsOrErr)
1336 return BMsOrErr.takeError();
1337
1338 // The bitcode file may contain multiple modules, we want the one that is
1339 // marked as being the ThinLTO module.
1340 if (const BitcodeModule *Bm = FindThinLTOModule(*BMsOrErr))
1341 return *Bm;
1342
1343 return make_error<StringError>("Could not find module summary",
1344 inconvertibleErrorCode());

1356static void runThinLTOBackend(ModuleSummaryIndex *CombinedIndex, Module *M,
1357 const HeaderSearchOptions &HeaderOpts,
1358 const CodeGenOptions &CGOpts,
1359 const clang::TargetOptions &TOpts,
1360 const LangOptions &LOpts,
1361 std::unique_ptr<raw_pwrite_stream> OS,
1362 std::string SampleProfile,
1363 std::string ProfileRemapping,
1364 BackendAction Action) {
1365 StringMap<DenseMap<GlobalValue::GUID, GlobalValueSummary *>>
1366 ModuleToDefinedGVSummaries;
1367 CombinedIndex->collectDefinedGVSummariesPerModule(ModuleToDefinedGVSummaries);
1368
1369 setCommandLineOpts(CGOpts);
1370
1371 // We can simply import the values mentioned in the combined index, since
1372 // we should only invoke this using the individual indexes written out
1373 // via a WriteIndexesThinBackend.
1374 FunctionImporter::ImportMapTy ImportList;
1375 for (auto &GlobalList : *CombinedIndex) {
1376 // Ignore entries for undefined references.
1377 if (GlobalList.second.SummaryList.empty())
1378 continue;
1379
1380 auto GUID = GlobalList.first;
1381 for (auto &Summary : GlobalList.second.SummaryList) {
1382 // Skip the summaries for the importing module. These are included to
1383 // e.g. record required linkage changes.
1384 if (Summary->modulePath() == M->getModuleIdentifier())
1385 continue;
1386 // Add an entry to provoke importing by thinBackend.
1387 ImportList[Summary->modulePath()].insert(GUID);
1388 }
1389 }
1390
1391 std::vector<std::unique_ptr<llvm::MemoryBuffer>> OwnedImports;
1392 MapVector<llvm::StringRef, llvm::BitcodeModule> ModuleMap;
1393
1394 for (auto &I : ImportList) {
1395 ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> MBOrErr =
1396 llvm::MemoryBuffer::getFile(I.first());
1397 if (!MBOrErr) {
1398 errs() << "Error loading imported file '" << I.first()
1399 << "': " << MBOrErr.getError().message() << "\n";
1400 return;
1401 }
1402
1403 Expected<BitcodeModule> BMOrErr = FindThinLTOModule(**MBOrErr);
1404 if (!BMOrErr) {
1405 handleAllErrors(BMOrErr.takeError(), [&](ErrorInfoBase &EIB) {
1406 errs() << "Error loading imported file '" << I.first()
1407 << "': " << EIB.message() << '\n';
1408 });
1409 return;
1410 }
1411 ModuleMap.insert({I.first(), *BMOrErr});
1412
1413 OwnedImports.push_back(std::move(*MBOrErr));
1414 }
1415 auto AddStream = [&](size_t Task) {
1416 return llvm::make_unique<lto::NativeObjectStream>(std::move(OS));
1417 };
1418 lto::Config Conf;
1419 if (CGOpts.SaveTempsFilePrefix != "") {
1420 if (Error E = Conf.addSaveTemps(CGOpts.SaveTempsFilePrefix + ".",
1421 /* UseInputModulePath */ false)) {
1422 handleAllErrors(std::move(E), [&](ErrorInfoBase &EIB) {
1423 errs() << "Error setting up ThinLTO save-temps: " << EIB.message()
1424 << '\n';
1425 });
1426 }
1427 }
1428 Conf.CPU = TOpts.CPU;
1429 Conf.CodeModel = getCodeModel(CGOpts);
1430 Conf.MAttrs = TOpts.Features;
1431 Conf.RelocModel = CGOpts.RelocationModel;
1432 Conf.CGOptLevel = getCGOptLevel(CGOpts);
1433 Conf.OptLevel = CGOpts.OptimizationLevel;
1434 initTargetOptions(Conf.Options, CGOpts, TOpts, LOpts, HeaderOpts);
1435 Conf.SampleProfile = std::move(SampleProfile);
1436
1437 // Context sensitive profile.
1438 if (CGOpts.hasProfileCSIRInstr()) {
1439 Conf.RunCSIRInstr = true;
1440 Conf.CSIRProfile = std::move(CGOpts.InstrProfileOutput);
1441 } else if (CGOpts.hasProfileCSIRUse()) {
1442 Conf.RunCSIRInstr = false;
1443 Conf.CSIRProfile = std::move(CGOpts.ProfileInstrumentUsePath);
1444 }
1445
1446 Conf.ProfileRemapping = std::move(ProfileRemapping);
1447 Conf.UseNewPM = CGOpts.ExperimentalNewPassManager;
1448 Conf.DebugPassManager = CGOpts.DebugPassManager;
1449 Conf.RemarksWithHotness = CGOpts.DiagnosticsWithHotness;
1450 Conf.RemarksFilename = CGOpts.OptRecordFile;
1451 Conf.RemarksPasses = CGOpts.OptRecordPasses;
1452 Conf.DwoPath = CGOpts.SplitDwarfFile;
1453 switch (Action) {
1454 case Backend_EmitNothing:
1455 Conf.PreCodeGenModuleHook = [](size_t Task, const Module &Mod) {
1456 return false;
1457 };
1458 break;
1459 case Backend_EmitLL:
1460 Conf.PreCodeGenModuleHook = [&](size_t Task, const Module &Mod) {
1461 M->print(*OS, nullptr, CGOpts.EmitLLVMUseLists);
1462 return false;
1463 };
1464 break;
1465 case Backend_EmitBC:
1466 Conf.PreCodeGenModuleHook = [&](size_t Task, const Module &Mod) {
1467 WriteBitcodeToFile(*M, *OS, CGOpts.EmitLLVMUseLists);
1468 return false;
1469 };
1470 break;
1471 default:
1472 Conf.CGFileType = getCodeGenFileType(Action);
1473 break;
1474 }
1475 if (Error E = thinBackend(
1476 Conf, -1, AddStream, *M, *CombinedIndex, ImportList,
1477 ModuleToDefinedGVSummaries[M->getModuleIdentifier()], ModuleMap)) {
1478 handleAllErrors(std::move(E), [&](ErrorInfoBase &EIB) {
1479 errs() << "Error running ThinLTO backend: " << EIB.message() << '\n';
1480 });
1481 }
1482}

1484void clang::EmitBackendOutput(DiagnosticsEngine &Diags,
1485 const HeaderSearchOptions &HeaderOpts,
1486 const CodeGenOptions &CGOpts,
1487 const clang::TargetOptions &TOpts,
1488 const LangOptions &LOpts,
1489 const llvm::DataLayout &TDesc, Module *M,
1490 BackendAction Action,
1491 std::unique_ptr<raw_pwrite_stream> OS) {
1492
1493 llvm::TimeTraceScope TimeScope("Backend", StringRef(""));
1494
1495 std::unique_ptr<llvm::Module> EmptyModule;
1496 if (!CGOpts.ThinLTOIndexFile.empty()) {
1497 // If we are performing a ThinLTO importing compile, load the function index
1498 // into memory and pass it into runThinLTOBackend, which will run the
1499 // function importer and invoke LTO passes.
1500 Expected<std::unique_ptr<ModuleSummaryIndex>> IndexOrErr =
1501 llvm::getModuleSummaryIndexForFile(CGOpts.ThinLTOIndexFile,
1502 /*IgnoreEmptyThinLTOIndexFile*/true);
1503 if (!IndexOrErr) {
1504 logAllUnhandledErrors(IndexOrErr.takeError(), errs(),
1505 "Error loading index file '" +
1506 CGOpts.ThinLTOIndexFile + "': ");
1507 return;
1508 }
1509 std::unique_ptr<ModuleSummaryIndex> CombinedIndex = std::move(*IndexOrErr);
1510 // A null CombinedIndex means we should skip ThinLTO compilation
1511 // (LLVM will optionally ignore empty index files, returning null instead
1512 // of an error).
1513 if (CombinedIndex) {
1514 if (!CombinedIndex->skipModuleByDistributedBackend()) {
1515 runThinLTOBackend(CombinedIndex.get(), M, HeaderOpts, CGOpts, TOpts,
1516 LOpts, std::move(OS), CGOpts.SampleProfileFile,
1517 CGOpts.ProfileRemappingFile, Action);
1518 return;
1519 }
1520 // Distributed indexing detected that nothing from the module is needed
1521 // for the final linking. So we can skip the compilation. We sill need to
1522 // output an empty object file to make sure that a linker does not fail
1523 // trying to read it. Also for some features, like CFI, we must skip
1524 // the compilation as CombinedIndex does not contain all required
1525 // information.
1526 EmptyModule = llvm::make_unique<llvm::Module>("empty", M->getContext());
1527 EmptyModule->setTargetTriple(M->getTargetTriple());
1528 M = EmptyModule.get();
1529 }
1530 }
1531
1532 EmitAssemblyHelper AsmHelper(Diags, HeaderOpts, CGOpts, TOpts, LOpts, M);
1533
1534 if (CGOpts.ExperimentalNewPassManager)
1535 AsmHelper.EmitAssemblyWithNewPassManager(Action, std::move(OS));
1536 else
1537 AsmHelper.EmitAssembly(Action, std::move(OS));
1538
1539 // Verify clang's TargetInfo DataLayout against the LLVM TargetMachine's
1540 // DataLayout.
1541 if (AsmHelper.TM) {
1542 std::string DLDesc = M->getDataLayout().getStringRepresentation();
1543 if (DLDesc != TDesc.getStringRepresentation()) {
1544 unsigned DiagID = Diags.getCustomDiagID(
1545 DiagnosticsEngine::Error, "backend data layout '%0' does not match "
1546 "expected target description '%1'");
1547 Diags.Report(DiagID) << DLDesc << TDesc.getStringRepresentation();
1548 }
1549 }
1550}

1584// With -fembed-bitcode, save a copy of the llvm IR as data in the
1585// __LLVM,__bitcode section.
1586void clang::EmbedBitcode(llvm::Module *M, const CodeGenOptions &CGOpts,
1587 llvm::MemoryBufferRef Buf) {
1588 if (CGOpts.getEmbedBitcode() == CodeGenOptions::Embed_Off)
1589 return;
1590
1591 // Save llvm.compiler.used and remote it.
1592 SmallVector<Constant*, 2> UsedArray;
1593 SmallPtrSet<GlobalValue*, 4> UsedGlobals;
1594 Type *UsedElementType = Type::getInt8Ty(M->getContext())->getPointerTo(0);
1595 GlobalVariable *Used = collectUsedGlobalVariables(*M, UsedGlobals, true);
1596 for (auto *GV : UsedGlobals) {
1597 if (GV->getName() != "llvm.embedded.module" &&
1598 GV->getName() != "llvm.cmdline")
1599 UsedArray.push_back(
1600 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, UsedElementType));
1601 }
1602 if (Used)
1603 Used->eraseFromParent();
1604
1605 // Embed the bitcode for the llvm module.
1606 std::string Data;
1607 ArrayRef<uint8_t> ModuleData;
1608 Triple T(M->getTargetTriple());
1609 // Create a constant that contains the bitcode.
1610 // In case of embedding a marker, ignore the input Buf and use the empty
1611 // ArrayRef. It is also legal to create a bitcode marker even Buf is empty.
1612 if (CGOpts.getEmbedBitcode() != CodeGenOptions::Embed_Marker) {
1613 if (!isBitcode((const unsigned char *)Buf.getBufferStart(),
1614 (const unsigned char *)Buf.getBufferEnd())) {
1615 // If the input is LLVM Assembly, bitcode is produced by serializing
1616 // the module. Use-lists order need to be perserved in this case.
1617 llvm::raw_string_ostream OS(Data);
1618 llvm::WriteBitcodeToFile(*M, OS, /* ShouldPreserveUseListOrder */ true);
1619 ModuleData =
1620 ArrayRef<uint8_t>((const uint8_t *)OS.str().data(), OS.str().size());
1621 } else
1622 // If the input is LLVM bitcode, write the input byte stream directly.
1623 ModuleData = ArrayRef<uint8_t>((const uint8_t *)Buf.getBufferStart(),
1624 Buf.getBufferSize());
1625 }
1626 llvm::Constant *ModuleConstant =
1627 llvm::ConstantDataArray::get(M->getContext(), ModuleData);
1628 llvm::GlobalVariable *GV = new llvm::GlobalVariable(
1629 *M, ModuleConstant->getType(), true, llvm::GlobalValue::PrivateLinkage,
1630 ModuleConstant);
1631 GV->setSection(getSectionNameForBitcode(T));
1632 UsedArray.push_back(
1633 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, UsedElementType));
1634 if (llvm::GlobalVariable *Old =
1635 M->getGlobalVariable("llvm.embedded.module", true)) {
1636 assert(Old->hasOneUse() &&
1637 "llvm.embedded.module can only be used once in llvm.compiler.used");
1638 GV->takeName(Old);
1639 Old->eraseFromParent();
1640 } else {
1641 GV->setName("llvm.embedded.module");
1642 }
1643
1644 // Skip if only bitcode needs to be embedded.
1645 if (CGOpts.getEmbedBitcode() != CodeGenOptions::Embed_Bitcode) {
1646 // Embed command-line options.
1647 ArrayRef<uint8_t> CmdData(const_cast<uint8_t *>(CGOpts.CmdArgs.data()),
1648 CGOpts.CmdArgs.size());
1649 llvm::Constant *CmdConstant =
1650 llvm::ConstantDataArray::get(M->getContext(), CmdData);
1651 GV = new llvm::GlobalVariable(*M, CmdConstant->getType(), true,
1652 llvm::GlobalValue::PrivateLinkage,
1653 CmdConstant);
1654 GV->setSection(getSectionNameForCommandline(T));
1655 UsedArray.push_back(
1656 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, UsedElementType));
1657 if (llvm::GlobalVariable *Old =
1658 M->getGlobalVariable("llvm.cmdline", true)) {
1659 assert(Old->hasOneUse() &&
1660 "llvm.cmdline can only be used once in llvm.compiler.used");
1661 GV->takeName(Old);
1662 Old->eraseFromParent();
1663 } else {
1664 GV->setName("llvm.cmdline");
1665 }
1666 }
1667
1668 if (UsedArray.empty())
1669 return;
1670
1671 // Recreate llvm.compiler.used.
1672 ArrayType *ATy = ArrayType::get(UsedElementType, UsedArray.size());
1673 auto *NewUsed = new GlobalVariable(
1674 *M, ATy, false, llvm::GlobalValue::AppendingLinkage,
1675 llvm::ConstantArray::get(ATy, UsedArray), "llvm.compiler.used");
1676 NewUsed->setSection("llvm.metadata");
1677}
