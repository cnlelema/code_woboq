106void ConstStructBuilder::
107AppendBytes(CharUnits FieldOffsetInChars, llvm::Constant *InitCst) {
108
109 assert(NextFieldOffsetInChars <= FieldOffsetInChars
110 && "Field offset mismatch!");
111
112 CharUnits FieldAlignment = getAlignment(InitCst);
113
114 // Round up the field offset to the alignment of the field type.
115 CharUnits AlignedNextFieldOffsetInChars =
116 NextFieldOffsetInChars.alignTo(FieldAlignment);
117
118 if (AlignedNextFieldOffsetInChars < FieldOffsetInChars) {
119 // We need to append padding.
120 AppendPadding(FieldOffsetInChars - NextFieldOffsetInChars);
121
122 assert(NextFieldOffsetInChars == FieldOffsetInChars &&
123 "Did not add enough padding!");
124
125 AlignedNextFieldOffsetInChars =
126 NextFieldOffsetInChars.alignTo(FieldAlignment);
127 }
128
129 if (AlignedNextFieldOffsetInChars > FieldOffsetInChars) {
130 assert(!Packed && "Alignment is wrong even with a packed struct!");
131
132 // Convert the struct to a packed struct.
133 ConvertStructToPacked();
134
135 // After we pack the struct, we may need to insert padding.
136 if (NextFieldOffsetInChars < FieldOffsetInChars) {
137 // We need to append padding.
138 AppendPadding(FieldOffsetInChars - NextFieldOffsetInChars);
139
140 assert(NextFieldOffsetInChars == FieldOffsetInChars &&
141 "Did not add enough padding!");
142 }
143 AlignedNextFieldOffsetInChars = NextFieldOffsetInChars;
144 }
145
146 // Add the field.
147 Elements.push_back(InitCst);
148 NextFieldOffsetInChars = AlignedNextFieldOffsetInChars +
149 getSizeInChars(InitCst);
150
151 if (Packed)
152 assert(LLVMStructAlignment == CharUnits::One() &&
153 "Packed struct not byte-aligned!");
154 else
155 LLVMStructAlignment = std::max(LLVMStructAlignment, FieldAlignment);
156}

158void ConstStructBuilder::AppendBitField(const FieldDecl *Field,
159 uint64_t FieldOffset,
160 llvm::ConstantInt *CI) {
161 const ASTContext &Context = CGM.getContext();
162 const uint64_t CharWidth = Context.getCharWidth();
163 uint64_t NextFieldOffsetInBits = Context.toBits(NextFieldOffsetInChars);
164 if (FieldOffset > NextFieldOffsetInBits) {
165 // We need to add padding.
166 CharUnits PadSize = Context.toCharUnitsFromBits(
167 llvm::alignTo(FieldOffset - NextFieldOffsetInBits,
168 Context.getTargetInfo().getCharAlign()));
169
170 AppendPadding(PadSize);
171 }
172
173 uint64_t FieldSize = Field->getBitWidthValue(Context);
174
175 llvm::APInt FieldValue = CI->getValue();
176
177 // Promote the size of FieldValue if necessary
178 // FIXME: This should never occur, but currently it can because initializer
179 // constants are cast to bool, and because clang is not enforcing bitfield
180 // width limits.
181 if (FieldSize > FieldValue.getBitWidth())
182 FieldValue = FieldValue.zext(FieldSize);
183
184 // Truncate the size of FieldValue to the bit field size.
185 if (FieldSize < FieldValue.getBitWidth())
186 FieldValue = FieldValue.trunc(FieldSize);
187
188 NextFieldOffsetInBits = Context.toBits(NextFieldOffsetInChars);
189 if (FieldOffset < NextFieldOffsetInBits) {
190 // Either part of the field or the entire field can go into the previous
191 // byte.
192 assert(!Elements.empty() && "Elements can't be empty!");
193
194 unsigned BitsInPreviousByte = NextFieldOffsetInBits - FieldOffset;
195
196 bool FitsCompletelyInPreviousByte =
197 BitsInPreviousByte >= FieldValue.getBitWidth();
198
199 llvm::APInt Tmp = FieldValue;
200
201 if (!FitsCompletelyInPreviousByte) {
202 unsigned NewFieldWidth = FieldSize - BitsInPreviousByte;
203
204 if (CGM.getDataLayout().isBigEndian()) {
205 Tmp.lshrInPlace(NewFieldWidth);
206 Tmp = Tmp.trunc(BitsInPreviousByte);
207
208 // We want the remaining high bits.
209 FieldValue = FieldValue.trunc(NewFieldWidth);
210 } else {
211 Tmp = Tmp.trunc(BitsInPreviousByte);
212
213 // We want the remaining low bits.
214 FieldValue.lshrInPlace(BitsInPreviousByte);
215 FieldValue = FieldValue.trunc(NewFieldWidth);
216 }
217 }
218
219 Tmp = Tmp.zext(CharWidth);
220 if (CGM.getDataLayout().isBigEndian()) {
221 if (FitsCompletelyInPreviousByte)
222 Tmp = Tmp.shl(BitsInPreviousByte - FieldValue.getBitWidth());
223 } else {
224 Tmp = Tmp.shl(CharWidth - BitsInPreviousByte);
225 }
226
227 // 'or' in the bits that go into the previous byte.
228 llvm::Value *LastElt = Elements.back();
229 if (llvm::ConstantInt *Val = dyn_cast<llvm::ConstantInt>(LastElt))
230 Tmp |= Val->getValue();
231 else {
232 assert(isa<llvm::UndefValue>(LastElt));
233 // If there is an undef field that we're adding to, it can either be a
234 // scalar undef (in which case, we just replace it with our field) or it
235 // is an array. If it is an array, we have to pull one byte off the
236 // array so that the other undef bytes stay around.
237 if (!isa<llvm::IntegerType>(LastElt->getType())) {
238 // The undef padding will be a multibyte array, create a new smaller
239 // padding and then an hole for our i8 to get plopped into.
240 assert(isa<llvm::ArrayType>(LastElt->getType()) &&
241 "Expected array padding of undefs");
242 llvm::ArrayType *AT = cast<llvm::ArrayType>(LastElt->getType());
243 assert(AT->getElementType()->isIntegerTy(CharWidth) &&
244 AT->getNumElements() != 0 &&
245 "Expected non-empty array padding of undefs");
246
247 // Remove the padding array.
248 NextFieldOffsetInChars -= CharUnits::fromQuantity(AT->getNumElements());
249 Elements.pop_back();
250
251 // Add the padding back in two chunks.
252 AppendPadding(CharUnits::fromQuantity(AT->getNumElements()-1));
253 AppendPadding(CharUnits::One());
254 assert(isa<llvm::UndefValue>(Elements.back()) &&
255 Elements.back()->getType()->isIntegerTy(CharWidth) &&
256 "Padding addition didn't work right");
257 }
258 }
259
260 Elements.back() = llvm::ConstantInt::get(CGM.getLLVMContext(), Tmp);
261
262 if (FitsCompletelyInPreviousByte)
263 return;
264 }
265
266 while (FieldValue.getBitWidth() > CharWidth) {
267 llvm::APInt Tmp;
268
269 if (CGM.getDataLayout().isBigEndian()) {
270 // We want the high bits.
271 Tmp =
272 FieldValue.lshr(FieldValue.getBitWidth() - CharWidth).trunc(CharWidth);
273 } else {
274 // We want the low bits.
275 Tmp = FieldValue.trunc(CharWidth);
276
277 FieldValue.lshrInPlace(CharWidth);
278 }
279
280 Elements.push_back(llvm::ConstantInt::get(CGM.getLLVMContext(), Tmp));
281 ++NextFieldOffsetInChars;
282
283 FieldValue = FieldValue.trunc(FieldValue.getBitWidth() - CharWidth);
284 }
285
286 assert(FieldValue.getBitWidth() > 0 &&
287 "Should have at least one bit left!");
288 assert(FieldValue.getBitWidth() <= CharWidth &&
289 "Should not have more than a byte left!");
290
291 if (FieldValue.getBitWidth() < CharWidth) {
292 if (CGM.getDataLayout().isBigEndian()) {
293 unsigned BitWidth = FieldValue.getBitWidth();
294
295 FieldValue = FieldValue.zext(CharWidth) << (CharWidth - BitWidth);
296 } else
297 FieldValue = FieldValue.zext(CharWidth);
298 }
299
300 // Append the last element.
301 Elements.push_back(llvm::ConstantInt::get(CGM.getLLVMContext(),
302 FieldValue));
303 ++NextFieldOffsetInChars;
304}

306void ConstStructBuilder::AppendPadding(CharUnits PadSize) {
307 if (PadSize.isZero())
308 return;
309
310 llvm::Type *Ty = CGM.Int8Ty;
311 if (PadSize > CharUnits::One())
312 Ty = llvm::ArrayType::get(Ty, PadSize.getQuantity());
313
314 llvm::Constant *C = llvm::UndefValue::get(Ty);
315 Elements.push_back(C);
316 assert(getAlignment(C) == CharUnits::One() &&
317 "Padding must have 1 byte alignment!");
318
319 NextFieldOffsetInChars += getSizeInChars(C);
320}
321
322void ConstStructBuilder::AppendTailPadding(CharUnits RecordSize) {
323 assert(NextFieldOffsetInChars <= RecordSize &&
324 "Size mismatch!");
325
326 AppendPadding(RecordSize - NextFieldOffsetInChars);
327}

329void ConstStructBuilder::ConvertStructToPacked() {
330 SmallVector<llvm::Constant *, 16> PackedElements;
331 CharUnits ElementOffsetInChars = CharUnits::Zero();
332
333 for (unsigned i = 0, e = Elements.size(); i != e; ++i) {
334 llvm::Constant *C = Elements[i];
335
336 CharUnits ElementAlign = CharUnits::fromQuantity(
337 CGM.getDataLayout().getABITypeAlignment(C->getType()));
338 CharUnits AlignedElementOffsetInChars =
339 ElementOffsetInChars.alignTo(ElementAlign);
340
341 if (AlignedElementOffsetInChars > ElementOffsetInChars) {
342 // We need some padding.
343 CharUnits NumChars =
344 AlignedElementOffsetInChars - ElementOffsetInChars;
345
346 llvm::Type *Ty = CGM.Int8Ty;
347 if (NumChars > CharUnits::One())
348 Ty = llvm::ArrayType::get(Ty, NumChars.getQuantity());
349
350 llvm::Constant *Padding = llvm::UndefValue::get(Ty);
351 PackedElements.push_back(Padding);
352 ElementOffsetInChars += getSizeInChars(Padding);
353 }
354
355 PackedElements.push_back(C);
356 ElementOffsetInChars += getSizeInChars(C);
357 }
358
359 assert(ElementOffsetInChars == NextFieldOffsetInChars &&
360 "Packing the struct changed its size!");
361
362 Elements.swap(PackedElements);
363 LLVMStructAlignment = CharUnits::One();
364 Packed = true;
365}

367bool ConstStructBuilder::Build(InitListExpr *ILE) {
368 RecordDecl *RD = ILE->getType()->getAs<RecordType>()->getDecl();
369 const ASTRecordLayout &Layout = CGM.getContext().getASTRecordLayout(RD);
370
371 unsigned FieldNo = 0;
372 unsigned ElementNo = 0;
373
374 // Bail out if we have base classes. We could support these, but they only
375 // arise in C++1z where we will have already constant folded most interesting
376 // cases. FIXME: There are still a few more cases we can handle this way.
377 if (auto *CXXRD = dyn_cast<CXXRecordDecl>(RD))
378 if (CXXRD->getNumBases())
379 return false;
380
381 for (RecordDecl::field_iterator Field = RD->field_begin(),
382 FieldEnd = RD->field_end(); Field != FieldEnd; ++Field, ++FieldNo) {
383 // If this is a union, skip all the fields that aren't being initialized.
384 if (RD->isUnion() && ILE->getInitializedFieldInUnion() != *Field)
385 continue;
386
387 // Don't emit anonymous bitfields, they just affect layout.
388 if (Field->isUnnamedBitfield())
389 continue;
390
391 // Get the initializer. A struct can include fields without initializers,
392 // we just use explicit null values for them.
393 llvm::Constant *EltInit;
394 if (ElementNo < ILE->getNumInits())
395 EltInit = Emitter.tryEmitPrivateForMemory(ILE->getInit(ElementNo++),
396 Field->getType());
397 else
398 EltInit = Emitter.emitNullForMemory(Field->getType());
399
400 if (!EltInit)
401 return false;
402
403 if (!Field->isBitField()) {
404 // Handle non-bitfield members.
405 AppendField(*Field, Layout.getFieldOffset(FieldNo), EltInit);
406 } else {
407 // Otherwise we have a bitfield.
408 if (auto *CI = dyn_cast<llvm::ConstantInt>(EltInit)) {
409 AppendBitField(*Field, Layout.getFieldOffset(FieldNo), CI);
410 } else {
411 // We are trying to initialize a bitfield with a non-trivial constant,
412 // this must require run-time code.
413 return false;
414 }
415 }
416 }
417
418 return true;
419}

435bool ConstStructBuilder::Build(const APValue &Val, const RecordDecl *RD,
436 bool IsPrimaryBase,
437 const CXXRecordDecl *VTableClass,
438 CharUnits Offset) {
439 const ASTRecordLayout &Layout = CGM.getContext().getASTRecordLayout(RD);
440
441 if (const CXXRecordDecl *CD = dyn_cast<CXXRecordDecl>(RD)) {
442 // Add a vtable pointer, if we need one and it hasn't already been added.
443 if (CD->isDynamicClass() && !IsPrimaryBase) {
444 llvm::Constant *VTableAddressPoint =
445 CGM.getCXXABI().getVTableAddressPointForConstExpr(
446 BaseSubobject(CD, Offset), VTableClass);
447 AppendBytes(Offset, VTableAddressPoint);
448 }
449
450 // Accumulate and sort bases, in order to visit them in address order, which
451 // may not be the same as declaration order.
452 SmallVector<BaseInfo, 8> Bases;
453 Bases.reserve(CD->getNumBases());
454 unsigned BaseNo = 0;
455 for (CXXRecordDecl::base_class_const_iterator Base = CD->bases_begin(),
456 BaseEnd = CD->bases_end(); Base != BaseEnd; ++Base, ++BaseNo) {
457 assert(!Base->isVirtual() && "should not have virtual bases here");
458 const CXXRecordDecl *BD = Base->getType()->getAsCXXRecordDecl();
459 CharUnits BaseOffset = Layout.getBaseClassOffset(BD);
460 Bases.push_back(BaseInfo(BD, BaseOffset, BaseNo));
461 }
462 llvm::stable_sort(Bases);
463
464 for (unsigned I = 0, N = Bases.size(); I != N; ++I) {
465 BaseInfo &Base = Bases[I];
466
467 bool IsPrimaryBase = Layout.getPrimaryBase() == Base.Decl;
468 Build(Val.getStructBase(Base.Index), Base.Decl, IsPrimaryBase,
469 VTableClass, Offset + Base.Offset);
470 }
471 }
472
473 unsigned FieldNo = 0;
474 uint64_t OffsetBits = CGM.getContext().toBits(Offset);
475
476 for (RecordDecl::field_iterator Field = RD->field_begin(),
477 FieldEnd = RD->field_end(); Field != FieldEnd; ++Field, ++FieldNo) {
478 // If this is a union, skip all the fields that aren't being initialized.
479 if (RD->isUnion() && !declaresSameEntity(Val.getUnionField(), *Field))
480 continue;
481
482 // Don't emit anonymous bitfields, they just affect layout.
483 if (Field->isUnnamedBitfield())
484 continue;
485
486 // Emit the value of the initializer.
487 const APValue &FieldValue =
488 RD->isUnion() ? Val.getUnionValue() : Val.getStructField(FieldNo);
489 llvm::Constant *EltInit =
490 Emitter.tryEmitPrivateForMemory(FieldValue, Field->getType());
491 if (!EltInit)
492 return false;
493
494 if (!Field->isBitField()) {
495 // Handle non-bitfield members.
496 AppendField(*Field, Layout.getFieldOffset(FieldNo) + OffsetBits, EltInit);
497 } else {
498 // Otherwise we have a bitfield.
499 AppendBitField(*Field, Layout.getFieldOffset(FieldNo) + OffsetBits,
500 cast<llvm::ConstantInt>(EltInit));
501 }
502 }
503
504 return true;
505}

507llvm::Constant *ConstStructBuilder::Finalize(QualType Ty) {
508 RecordDecl *RD = Ty->getAs<RecordType>()->getDecl();
509 const ASTRecordLayout &Layout = CGM.getContext().getASTRecordLayout(RD);
510
511 CharUnits LayoutSizeInChars = Layout.getSize();
512
513 if (NextFieldOffsetInChars > LayoutSizeInChars) {
514 // If the struct is bigger than the size of the record type,
515 // we must have a flexible array member at the end.
516 assert(RD->hasFlexibleArrayMember() &&
517 "Must have flexible array member if struct is bigger than type!");
518
519 // No tail padding is necessary.
520 } else {
521 // Append tail padding if necessary.
522 CharUnits LLVMSizeInChars =
523 NextFieldOffsetInChars.alignTo(LLVMStructAlignment);
524
525 if (LLVMSizeInChars != LayoutSizeInChars)
526 AppendTailPadding(LayoutSizeInChars);
527
528 LLVMSizeInChars = NextFieldOffsetInChars.alignTo(LLVMStructAlignment);
529
530 // Check if we need to convert the struct to a packed struct.
531 if (NextFieldOffsetInChars <= LayoutSizeInChars &&
532 LLVMSizeInChars > LayoutSizeInChars) {
533 assert(!Packed && "Size mismatch!");
534
535 ConvertStructToPacked();
536 assert(NextFieldOffsetInChars <= LayoutSizeInChars &&
537 "Converting to packed did not help!");
538 }
539
540 LLVMSizeInChars = NextFieldOffsetInChars.alignTo(LLVMStructAlignment);
541
542 assert(LayoutSizeInChars == LLVMSizeInChars &&
543 "Tail padding mismatch!");
544 }
545
546 // Pick the type to use. If the type is layout identical to the ConvertType
547 // type then use it, otherwise use whatever the builder produced for us.
548 llvm::StructType *STy =
549 llvm::ConstantStruct::getTypeForElements(CGM.getLLVMContext(),
550 Elements, Packed);
551 llvm::Type *ValTy = CGM.getTypes().ConvertType(Ty);
552 if (llvm::StructType *ValSTy = dyn_cast<llvm::StructType>(ValTy)) {
553 if (ValSTy->isLayoutIdentical(STy))
554 STy = ValSTy;
555 }
556
557 llvm::Constant *Result = llvm::ConstantStruct::get(STy, Elements);
558
559 assert(NextFieldOffsetInChars.alignTo(getAlignment(Result)) ==
560 getSizeInChars(Result) &&
561 "Size mismatch!");
562
563 return Result;
564}

606static ConstantAddress tryEmitGlobalCompoundLiteral(CodeGenModule &CGM,
607 CodeGenFunction *CGF,
608 const CompoundLiteralExpr *E) {
609 CharUnits Align = CGM.getContext().getTypeAlignInChars(E->getType());
610 if (llvm::GlobalVariable *Addr =
611 CGM.getAddrOfConstantCompoundLiteralIfEmitted(E))
612 return ConstantAddress(Addr, Align);
613
614 LangAS addressSpace = E->getType().getAddressSpace();
615
616 ConstantEmitter emitter(CGM, CGF);
617 llvm::Constant *C = emitter.tryEmitForInitializer(E->getInitializer(),
618 addressSpace, E->getType());
619 if (!C) {
620 assert(!E->isFileScope() &&
621 "file-scope compound literal did not have constant initializer!");
622 return ConstantAddress::invalid();
623 }
624
625 auto GV = new llvm::GlobalVariable(CGM.getModule(), C->getType(),
626 CGM.isTypeConstant(E->getType(), true),
627 llvm::GlobalValue::InternalLinkage,
628 C, ".compoundliteral", nullptr,
629 llvm::GlobalVariable::NotThreadLocal,
630 CGM.getTargetAddressSpace(addressSpace));
631 emitter.finalize(GV);
632 GV->setAlignment(Align.getQuantity());
633 CGM.setAddrOfConstantCompoundLiteral(E, GV);
634 return ConstantAddress(GV, Align);
635}

637static llvm::Constant *
638EmitArrayConstant(CodeGenModule &CGM, const ConstantArrayType *DestType,
639 llvm::Type *CommonElementType, unsigned ArrayBound,
640 SmallVectorImpl<llvm::Constant *> &Elements,
641 llvm::Constant *Filler) {
642 // Figure out how long the initial prefix of non-zero elements is.
643 unsigned NonzeroLength = ArrayBound;
644 if (Elements.size() < NonzeroLength && Filler->isNullValue())
645 NonzeroLength = Elements.size();
646 if (NonzeroLength == Elements.size()) {
647 while (NonzeroLength > 0 && Elements[NonzeroLength - 1]->isNullValue())
648 --NonzeroLength;
649 }
650
651 if (NonzeroLength == 0) {
652 return llvm::ConstantAggregateZero::get(
653 CGM.getTypes().ConvertType(QualType(DestType, 0)));
654 }
655
656 // Add a zeroinitializer array filler if we have lots of trailing zeroes.
657 unsigned TrailingZeroes = ArrayBound - NonzeroLength;
658 if (TrailingZeroes >= 8) {
659 assert(Elements.size() >= NonzeroLength &&
660 "missing initializer for non-zero element");
661
662 // If all the elements had the same type up to the trailing zeroes, emit a
663 // struct of two arrays (the nonzero data and the zeroinitializer).
664 if (CommonElementType && NonzeroLength >= 8) {
665 llvm::Constant *Initial = llvm::ConstantArray::get(
666 llvm::ArrayType::get(CommonElementType, NonzeroLength),
667 makeArrayRef(Elements).take_front(NonzeroLength));
668 Elements.resize(2);
669 Elements[0] = Initial;
670 } else {
671 Elements.resize(NonzeroLength + 1);
672 }
673
674 auto *FillerType =
675 CommonElementType
676 ? CommonElementType
677 : CGM.getTypes().ConvertType(DestType->getElementType());
678 FillerType = llvm::ArrayType::get(FillerType, TrailingZeroes);
679 Elements.back() = llvm::ConstantAggregateZero::get(FillerType);
680 CommonElementType = nullptr;
681 } else if (Elements.size() != ArrayBound) {
682 // Otherwise pad to the right size with the filler if necessary.
683 Elements.resize(ArrayBound, Filler);
684 if (Filler->getType() != CommonElementType)
685 CommonElementType = nullptr;
686 }
687
688 // If all elements have the same type, just emit an array constant.
689 if (CommonElementType)
690 return llvm::ConstantArray::get(
691 llvm::ArrayType::get(CommonElementType, ArrayBound), Elements);
692
693 // We have mixed types. Use a packed struct.
694 llvm::SmallVector<llvm::Type *, 16> Types;
695 Types.reserve(Elements.size());
696 for (llvm::Constant *Elt : Elements)

754 llvm::Constant *VisitCastExpr(CastExpr *E, QualType destType) {
755 if (const auto *ECE = dyn_cast<ExplicitCastExpr>(E))
756 CGM.EmitExplicitCastExprType(ECE, Emitter.CGF);
757 Expr *subExpr = E->getSubExpr();
758
759 switch (E->getCastKind()) {
760 case CK_ToUnion: {
761 // GCC cast to union extension
762 assert(E->getType()->isUnionType() &&
763 "Destination type is not union type!");
764
765 auto field = E->getTargetUnionField();
766
767 auto C = Emitter.tryEmitPrivateForMemory(subExpr, field->getType());
768 if (!C) return nullptr;
769
770 auto destTy = ConvertType(destType);
771 if (C->getType() == destTy) return C;
772
773 // Build a struct with the union sub-element as the first member,
774 // and padded to the appropriate size.
775 SmallVector<llvm::Constant*, 2> Elts;
776 SmallVector<llvm::Type*, 2> Types;
777 Elts.push_back(C);
778 Types.push_back(C->getType());
779 unsigned CurSize = CGM.getDataLayout().getTypeAllocSize(C->getType());
780 unsigned TotalSize = CGM.getDataLayout().getTypeAllocSize(destTy);
781
782 assert(CurSize <= TotalSize && "Union size mismatch!");
783 if (unsigned NumPadBytes = TotalSize - CurSize) {
784 llvm::Type *Ty = CGM.Int8Ty;
785 if (NumPadBytes > 1)
786 Ty = llvm::ArrayType::get(Ty, NumPadBytes);

788 Elts.push_back(llvm::UndefValue::get(Ty));

910 llvm::Constant *EmitArrayInitialization(InitListExpr *ILE, QualType T) {
911 auto *CAT = CGM.getContext().getAsConstantArrayType(ILE->getType());
912 assert(CAT && "can't emit array init for non-constant-bound array");
913 unsigned NumInitElements = ILE->getNumInits();
914 unsigned NumElements = CAT->getSize().getZExtValue();
915
916 // Initialising an array requires us to automatically
917 // initialise any elements that have not been initialised explicitly
918 unsigned NumInitableElts = std::min(NumInitElements, NumElements);
919
920 QualType EltType = CAT->getElementType();
921
922 // Initialize remaining array elements.
923 llvm::Constant *fillC = nullptr;
924 if (Expr *filler = ILE->getArrayFiller()) {
925 fillC = Emitter.tryEmitAbstractForMemory(filler, EltType);
926 if (!fillC)
927 return nullptr;
928 }
929
930 // Copy initializer elements.
931 SmallVector<llvm::Constant*, 16> Elts;
932 if (fillC && fillC->isNullValue())
933 Elts.reserve(NumInitableElts + 1);
934 else
935 Elts.reserve(NumElements);
936
937 llvm::Type *CommonElementType = nullptr;
938 for (unsigned i = 0; i < NumInitableElts; ++i) {
939 Expr *Init = ILE->getInit(i);
940 llvm::Constant *C = Emitter.tryEmitPrivateForMemory(Init, EltType);
941 if (!C)
942 return nullptr;
943 if (i == 0)
944 CommonElementType = C->getType();
945 else if (C->getType() != CommonElementType)
946 CommonElementType = nullptr;
947 Elts.push_back(C);
948 }
949
950 return EmitArrayConstant(CGM, CAT, CommonElementType, NumElements, Elts,
951 fillC);
952 }

976 llvm::Constant *EmitDesignatedInitUpdater(llvm::Constant *Base,
977 InitListExpr *Updater,
978 QualType destType) {
979 if (auto destAT = CGM.getContext().getAsArrayType(destType)) {
980 llvm::ArrayType *AType = cast<llvm::ArrayType>(ConvertType(destType));
981 llvm::Type *ElemType = AType->getElementType();
982
983 unsigned NumInitElements = Updater->getNumInits();
984 unsigned NumElements = AType->getNumElements();
985
986 std::vector<llvm::Constant *> Elts;
987 Elts.reserve(NumElements);
988
989 QualType destElemType = destAT->getElementType();
990
991 if (auto DataArray = dyn_cast<llvm::ConstantDataArray>(Base))
992 for (unsigned i = 0; i != NumElements; ++i)
993 Elts.push_back(DataArray->getElementAsConstant(i));
994 else if (auto Array = dyn_cast<llvm::ConstantArray>(Base))
995 for (unsigned i = 0; i != NumElements; ++i)
996 Elts.push_back(Array->getOperand(i));
997 else
998 return nullptr; // FIXME: other array types not implemented
999
1000 llvm::Constant *fillC = nullptr;
1001 if (Expr *filler = Updater->getArrayFiller())
1002 if (!isa<NoInitExpr>(filler))
1003 fillC = Emitter.tryEmitAbstractForMemory(filler, destElemType);
1004 bool RewriteType = (fillC && fillC->getType() != ElemType);
1005
1006 for (unsigned i = 0; i != NumElements; ++i) {
1007 Expr *Init = nullptr;
1008 if (i < NumInitElements)
1009 Init = Updater->getInit(i);
1010
1011 if (!Init && fillC)
1012 Elts[i] = fillC;
1013 else if (!Init || isa<NoInitExpr>(Init))
1014 ; // Do nothing.
1015 else if (InitListExpr *ChildILE = dyn_cast<InitListExpr>(Init))
1016 Elts[i] = EmitDesignatedInitUpdater(Elts[i], ChildILE, destElemType);
1017 else
1018 Elts[i] = Emitter.tryEmitPrivateForMemory(Init, destElemType);
1019
1020 if (!Elts[i])
1021 return nullptr;
1022 RewriteType |= (Elts[i]->getType() != ElemType);
1023 }
1024
1025 if (RewriteType) {
1026 std::vector<llvm::Type *> Types;
1027 Types.reserve(NumElements);
1028 for (unsigned i = 0; i != NumElements; ++i)
1029 Types.push_back(Elts[i]->getType());
1030 llvm::StructType *SType = llvm::StructType::get(AType->getContext(),
1031 Types, true);
1032 return llvm::ConstantStruct::get(SType, Elts);
1033 }
1034
1035 return llvm::ConstantArray::get(AType, Elts);
1036 }
1037
1038 if (destType->isRecordType())
1039 return ConstStructBuilder::BuildStruct(Emitter, this, Base, Updater,
1040 destType);
1041
1042 return nullptr;
1043 }

1115bool ConstStructBuilder::Build(ConstExprEmitter *ExprEmitter,
1116 llvm::Constant *Base,
1117 InitListExpr *Updater) {
1118 assert(Base && "base expression should not be empty");
1119
1120 QualType ExprType = Updater->getType();
1121 RecordDecl *RD = ExprType->getAs<RecordType>()->getDecl();
1122 const ASTRecordLayout &Layout = CGM.getContext().getASTRecordLayout(RD);
1123 const llvm::StructLayout *BaseLayout = CGM.getDataLayout().getStructLayout(
1124 cast<llvm::StructType>(Base->getType()));
1125 unsigned FieldNo = -1;
1126 unsigned ElementNo = 0;
1127
1128 // Bail out if we have base classes. We could support these, but they only
1129 // arise in C++1z where we will have already constant folded most interesting
1130 // cases. FIXME: There are still a few more cases we can handle this way.
1131 if (auto *CXXRD = dyn_cast<CXXRecordDecl>(RD))
1132 if (CXXRD->getNumBases())
1133 return false;
1134
1135 for (FieldDecl *Field : RD->fields()) {
1136 ++FieldNo;
1137
1138 if (RD->isUnion() && Updater->getInitializedFieldInUnion() != Field)
1139 continue;
1140
1141 // Skip anonymous bitfields.
1142 if (Field->isUnnamedBitfield())
1143 continue;
1144
1145 llvm::Constant *EltInit = Base->getAggregateElement(ElementNo);
1146
1147 // Bail out if the type of the ConstantStruct does not have the same layout
1148 // as the type of the InitListExpr.
1149 if (CGM.getTypes().ConvertType(Field->getType()) != EltInit->getType() ||
1150 Layout.getFieldOffset(ElementNo) !=
1151 BaseLayout->getElementOffsetInBits(ElementNo))
1152 return false;
1153
1154 // Get the initializer. If we encounter an empty field or a NoInitExpr,
1155 // we use values from the base expression.
1156 Expr *Init = nullptr;
1157 if (ElementNo < Updater->getNumInits())
1158 Init = Updater->getInit(ElementNo);
1159
1160 if (!Init || isa<NoInitExpr>(Init))
1161 ; // Do nothing.
1162 else if (InitListExpr *ChildILE = dyn_cast<InitListExpr>(Init))
1163 EltInit = ExprEmitter->EmitDesignatedInitUpdater(EltInit, ChildILE,
1164 Field->getType());
1165 else
1166 EltInit = Emitter.tryEmitPrivateForMemory(Init, Field->getType());
1167
1168 ++ElementNo;
1169
1170 if (!EltInit)
1171 return false;
1172
1173 if (!Field->isBitField())
1174 AppendField(Field, Layout.getFieldOffset(FieldNo), EltInit);
1175 else if (llvm::ConstantInt *CI = dyn_cast<llvm::ConstantInt>(EltInit))
1176 AppendBitField(Field, Layout.getFieldOffset(FieldNo), CI);
1177 else
1178 // Initializing a bitfield with a non-trivial constant?
1179 return false;
1180 }
1181
1182 return true;
1183}

1337 void replaceInInitializer(llvm::Constant *init) {
1338 // Remember the type of the top-most initializer.
1339 BaseValueTy = init->getType();
1340
1341 // Initialize the stack.
1342 Indices.push_back(0);

1350 assert(Indices.size() == 1 && "didn't pop all indices");
1351
1352 // Do the replacement; this basically invalidates 'init'.
1353 assert(Locations.size() == PlaceholderAddresses.size() &&
1354 "missed a placeholder?");
1355
1356 // We're iterating over a hashtable, so this would be a source of
1357 // non-determinism in compiler output *except* that we're just
1358 // messing around with llvm::Constant structures, which never itself
1359 // does anything that should be visible in compiler output.
1360 for (auto &entry : Locations) {
1361 assert(entry.first->getParent() == nullptr && "not a placeholder!");
1362 entry.first->replaceAllUsesWith(entry.second);
1363 entry.first->eraseFromParent();
1364 }
1365 }
1366
1367 private:
1368 void findLocations(llvm::Constant *init) {
1369 // Recurse into aggregates.
1370 if (auto agg = dyn_cast<llvm::ConstantAggregate>(init)) {
1371 for (unsigned i = 0, e = agg->getNumOperands(); i != e; ++i) {
1372 Indices.push_back(i);
1373 IndexValues.push_back(nullptr);
1374
1375 findLocations(agg->getOperand(i));
1376
1377 IndexValues.pop_back();
1378 Indices.pop_back();
1379 }
1380 return;
1381 }
1382
1383 // Otherwise, check for registered constants.
1384 while (true) {
1385 auto it = PlaceholderAddresses.find(init);
1386 if (it != PlaceholderAddresses.end()) {
1387 setLocation(it->second);
1388 break;
1389 }
1390
1391 // Look through bitcasts or other expressions.
1392 if (auto expr = dyn_cast<llvm::ConstantExpr>(init)) {
1393 init = expr->getOperand(0);
1394 } else {
1395 break;
1396 }
1397 }
1398 }
1399
1400 void setLocation(llvm::GlobalVariable *placeholder) {
1401 assert(Locations.find(placeholder) == Locations.end() &&
1402 "already found location for placeholder!");
1403
1404 // Lazily fill in IndexValues with the values from Indices.
1405 // We do this in reverse because we should always have a strict
1406 // prefix of indices from the start.
1407 assert(Indices.size() == IndexValues.size());
1408 for (size_t i = Indices.size() - 1; i != size_t(-1); --i) {
1409 if (IndexValues[i]) {
1410#ifndef NDEBUG
1411 for (size_t j = 0; j != i + 1; ++j) {
1412 assert(IndexValues[j] &&
1413 isa<llvm::ConstantInt>(IndexValues[j]) &&
1414 cast<llvm::ConstantInt>(IndexValues[j])->getZExtValue()
1415 == Indices[j]);
1416 }
1417#endif
1418 break;
1419 }
1420
1421 IndexValues[i] = llvm::ConstantInt::get(CGM.Int32Ty, Indices[i]);
1422 }
1423
1424 // Form a GEP and then bitcast to the placeholder type so that the
1425 // replacement will succeed.
1426 llvm::Constant *location =
1427 llvm::ConstantExpr::getInBoundsGetElementPtr(BaseValueTy,
1428 Base, IndexValues);
1429 location = llvm::ConstantExpr::getBitCast(location,
1430 placeholder->getType());
1431
1432 Locations.insert({placeholder, location});
1433 }
1434 };
1435}

1467llvm::Constant *ConstantEmitter::tryEmitPrivateForVarInit(const VarDecl &D) {
1468 // Make a quick check if variable can be default NULL initialized
1469 // and avoid going through rest of code which may do, for c++11,
1470 // initialization of memory to all NULLs.
1471 if (!D.hasLocalStorage()) {
1472 QualType Ty = CGM.getContext().getBaseElementType(D.getType());
1473 if (Ty->isRecordType())
1474 if (const CXXConstructExpr *E =
1475 dyn_cast_or_null<CXXConstructExpr>(D.getInit())) {
1476 const CXXConstructorDecl *CD = E->getConstructor();
1477 if (CD->isTrivial() && CD->isDefaultConstructor())
1478 return CGM.EmitNullConstant(D.getType());
1479 }
1480 InConstantContext = true;
1481 }
1482
1483 QualType destType = D.getType();
1484
1485 // Try to emit the initializer. Note that this can allow some things that
1486 // are not allowed by tryEmitPrivateForMemory alone.
1487 if (auto value = D.evaluateValue()) {
1488 return tryEmitPrivateForMemory(*value, destType);
1489 }
1490
1491 // FIXME: Implement C++11 [basic.start.init]p2: if the initializer of a
1492 // reference is a constant expression, and the reference binds to a temporary,
1493 // then constant initialization is performed. ConstExprEmitter will
1494 // incorrectly emit a prvalue constant in this case, and the calling code
1495 // interprets that as the (pointer) value of the reference, rather than the
1496 // desired value of the referee.
1497 if (destType->isReferenceType())
1498 return nullptr;
1499
1500 const Expr *E = D.getInit();
1501 assert(E && "No initializer to emit");
1502
1503 auto nonMemoryDestType = getNonMemoryType(CGM, destType);
1504 auto C =
1505 ConstExprEmitter(*this).Visit(const_cast<Expr*>(E), nonMemoryDestType);
1506 return (C ? emitForMemory(C, destType) : nullptr);
1507}

1538llvm::Constant *ConstantEmitter::emitForMemory(CodeGenModule &CGM,
1539 llvm::Constant *C,
1540 QualType destType) {
1541 // For an _Atomic-qualified constant, we may need to add tail padding.
1542 if (auto AT = destType->getAs<AtomicType>()) {
1543 QualType destValueType = AT->getValueType();
1544 C = emitForMemory(CGM, C, destValueType);
1545
1546 uint64_t innerSize = CGM.getContext().getTypeSize(destValueType);
1547 uint64_t outerSize = CGM.getContext().getTypeSize(destType);
1548 if (innerSize == outerSize)
1549 return C;
1550
1551 assert(innerSize < outerSize && "emitted over-large constant for atomic");
1552 llvm::Constant *elts[] = {
1553 C,
1554 llvm::ConstantAggregateZero::get(
1555 llvm::ArrayType::get(CGM.Int8Ty, (outerSize - innerSize) / 8))
1556 };
1557 return llvm::ConstantStruct::getAnon(elts);
1558 }
1559
1560 // Zero-extend bool.
1561 if (C->getType()->isIntegerTy(1)) {
1562 llvm::Type *boolTy = CGM.getTypes().ConvertTypeForMem(destType);
1563 return llvm::ConstantExpr::getZExt(C, boolTy);
1564 }
1565
1566 return C;
1567}

1569llvm::Constant *ConstantEmitter::tryEmitPrivate(const Expr *E,
1570 QualType destType) {
1571 Expr::EvalResult Result;
1572
1573 bool Success = false;
1574
1575 if (destType->isReferenceType())
1576 Success = E->EvaluateAsLValue(Result, CGM.getContext());
1577 else
1578 Success = E->EvaluateAsRValue(Result, CGM.getContext(), InConstantContext);
1579
1580 llvm::Constant *C;
1581 if (Success && !Result.HasSideEffects)
1582 C = tryEmitPrivate(Result.Val, destType);
1583 else
1584 C = ConstExprEmitter(*this).Visit(const_cast<Expr*>(E), destType);
1585
1586 return C;
1587}

1673llvm::Constant *ConstantLValueEmitter::tryEmit() {
1674 const APValue::LValueBase &base = Value.getLValueBase();
1675
1676 // The destination type should be a pointer or reference
1677 // type, but it might also be a cast thereof.
1678 //
1679 // FIXME: the chain of casts required should be reflected in the APValue.
1680 // We need this in order to correctly handle things like a ptrtoint of a
1681 // non-zero null pointer and addrspace casts that aren't trivially
1682 // represented in LLVM IR.
1683 auto destTy = CGM.getTypes().ConvertTypeForMem(DestType);
1684 assert(isa<llvm::IntegerType>(destTy) || isa<llvm::PointerType>(destTy));
1685
1686 // If there's no base at all, this is a null or absolute pointer,
1687 // possibly cast back to an integer type.
1688 if (!base) {
1689 return tryEmitAbsolute(destTy);
1690 }
1691
1692 // Otherwise, try to emit the base.
1693 ConstantLValue result = tryEmitBase(base);
1694
1695 // If that failed, we're done.
1696 llvm::Constant *value = result.Value;
1697 if (!value) return nullptr;
1698
1699 // Apply the offset if necessary and not already done.
1700 if (!result.HasOffsetApplied) {
1701 value = applyOffset(value);
1702 }
1703
1704 // Convert to the appropriate type; this could be an lvalue for
1705 // an integer. FIXME: performAddrSpaceCast
1706 if (isa<llvm::PointerType>(destTy))
1707 return llvm::ConstantExpr::getPointerCast(value, destTy);
1708
1709 return llvm::ConstantExpr::getPtrToInt(value, destTy);
1710}

1712/// Try to emit an absolute l-value, such as a null pointer or an integer
1713/// bitcast to pointer type.
1714llvm::Constant *
1715ConstantLValueEmitter::tryEmitAbsolute(llvm::Type *destTy) {
1716 // If we're producing a pointer, this is easy.
1717 auto destPtrTy = cast<llvm::PointerType>(destTy);
1718 if (Value.isNullPointer()) {
1719 // FIXME: integer offsets from non-zero null pointers.
1720 return CGM.getNullPointer(destPtrTy, DestType);
1721 }
1722
1723 // Convert the integer to a pointer-sized integer before converting it
1724 // to a pointer.
1725 // FIXME: signedness depends on the original integer type.
1726 auto intptrTy = CGM.getDataLayout().getIntPtrType(destPtrTy);
1727 llvm::Constant *C;
1728 C = llvm::ConstantExpr::getIntegerCast(getOffset(), intptrTy,
1729 /*isSigned*/ false);
1730
1731 // FIXME: This may or may not be a valid pointer on CHERI
1732 // See https://github.com/CTSRD-CHERI/llvm/issues/268
1733 C = llvm::ConstantExpr::getIntToPtr(C, destPtrTy);
1734 return C;
1735}

1737ConstantLValue
1738ConstantLValueEmitter::tryEmitBase(const APValue::LValueBase &base) {
1739 // Handle values.
1740 if (const ValueDecl *D = base.dyn_cast<const ValueDecl*>()) {
1741 if (D->hasAttr<WeakRefAttr>())
1742 return CGM.GetWeakRefReference(D).getPointer();
1743
1744 if (auto FD = dyn_cast<FunctionDecl>(D))
1745 return CGM.GetAddrOfFunction(FD);
1746
1747 if (auto VD = dyn_cast<VarDecl>(D)) {
1748 // We can never refer to a variable with local storage.
1749 if (!VD->hasLocalStorage()) {
1750 if (VD->isFileVarDecl() || VD->hasExternalStorage())
1751 return CGM.GetAddrOfGlobalVar(VD);
1752
1753 if (VD->isLocalVarDecl()) {
1754 return CGM.getOrCreateStaticVarDecl(
1755 *VD, CGM.getLLVMLinkageVarDefinition(VD, /*isConstant=*/false));
1756 }
1757 }
1758 }
1759
1760 return nullptr;
1761 }
1762
1763 // Handle typeid(T).
1764 if (TypeInfoLValue TI = base.dyn_cast<TypeInfoLValue>()) {
1765 unsigned GlobalAS = CGM.getDataLayout().getGlobalsAddressSpace();
1766 llvm::Type *StdTypeInfoPtrTy =
1767 CGM.getTypes().ConvertType(base.getTypeInfoType())->getPointerTo(GlobalAS);
1768 llvm::Constant *TypeInfo =
1769 CGM.GetAddrOfRTTIDescriptor(QualType(TI.getType(), 0));
1770 if (TypeInfo->getType() != StdTypeInfoPtrTy)
1771 TypeInfo = llvm::ConstantExpr::getBitCast(TypeInfo, StdTypeInfoPtrTy);
1772 return TypeInfo;
1773 }
1774
1775 // Otherwise, it must be an expression.
1776 return Visit(base.get<const Expr*>());
1777}

1886llvm::Constant *ConstantEmitter::tryEmitPrivate(const APValue &Value,
1887 QualType DestType) {
1888 switch (Value.getKind()) {
1889 case APValue::None:
1890 case APValue::Indeterminate:
1891 // Out-of-lifetime and indeterminate values can be modeled as 'undef'.
1892 return llvm::UndefValue::get(CGM.getTypes().ConvertType(DestType));
1893 case APValue::LValue:
1894 return ConstantLValueEmitter(*this, Value, DestType).tryEmit();
1895 case APValue::Int: {
1896 // For __uintcap_t and enums whose underlying type is __[u]intcap_t, we
1897 // get an APValue::Int but we actually need to emit a i8 addrspace(200)*
1898 // and not i64 here.
1899 // Use ConvertType() here and not ConvertTypeForMem since that turns
1900 // _Bool into i8/i32 instead of i1
1901 llvm::Type *TargetTy = CGM.getTypes().ConvertType(DestType);
1902 auto AsInt = llvm::ConstantInt::get(CGM.getLLVMContext(), Value.getInt());
1903
1904 if (DestType->isIntCapType()) {
1905 // If we use an inttoptr inside a function the backend will generate a
1906 // CFromPtr $ddc which probably gives a valid capability with vaddr = AsInt
1907 // We really want a null-derived value here instead!
1908 // This previously broke cases such as QReadWriteLock which writes 0x2
1909 // to indicate write locked state but since it was not equal to
1910 // (__intptr_t)0x2 it assumed it was a valid QReadWriteLockerPrivate and
1911 // then crashed trying to read memory at address 0x2
1912 //
1913 // See https://github.com/CTSRD-CHERI/llvm/issues/268
1914
1915 // Would be nice use CGF->setPointerOffset() but we can't since that
1916 // is not a llvm::Constant
1917 // if (CGF)
1918 // return CGF->setCapabilityIntegerValue(
1919 // llvm::ConstantPointerNull::get(cast<llvm::PointerType>(TargetTy)),
1920 // AsInt);
1921
1922 // Note: CodeGenFunction needs to convert this into a
1923 // CGF.getCapabilityIntegerValue
1924 return llvm::ConstantExpr::getIntToPtr(AsInt, TargetTy);
1925 }
1926 assert(!DestType->isCHERICapabilityType(CGM.getContext()));
1927#if !defined(NDEBUG) || defined(LLVM_ENABLE_DUMP)
1928 if (LLVM_UNLIKELY(AsInt->getType() != TargetTy)) {
1929 llvm::errs() << __func__ << ": AsInt->getType(): ";
1930 AsInt->getType()->dump();
1931 llvm::errs() << __func__ << ": TargetTy: ";
1932 TargetTy->dump();
1933 llvm::errs() << __func__ << ": DestType: ";
1934 DestType.dump();
1935 assert(AsInt->getType() == TargetTy);
1936 }
1937#endif
1938 return AsInt;
1939 }
1940 case APValue::FixedPoint:
1941 return llvm::ConstantInt::get(CGM.getLLVMContext(),
1942 Value.getFixedPoint().getValue());
1943 case APValue::ComplexInt: {
1944 llvm::Constant *Complex[2];
1945
1946 Complex[0] = llvm::ConstantInt::get(CGM.getLLVMContext(),
1947 Value.getComplexIntReal());
1948 Complex[1] = llvm::ConstantInt::get(CGM.getLLVMContext(),
1949 Value.getComplexIntImag());
1950
1951 // FIXME: the target may want to specify that this is packed.
1952 llvm::StructType *STy =
1953 llvm::StructType::get(Complex[0]->getType(), Complex[1]->getType());
1954 return llvm::ConstantStruct::get(STy, Complex);
1955 }
1956 case APValue::Float: {
1957 const llvm::APFloat &Init = Value.getFloat();
1958 if (&Init.getSemantics() == &llvm::APFloat::IEEEhalf() &&
1959 !CGM.getContext().getLangOpts().NativeHalfType &&
1960 CGM.getContext().getTargetInfo().useFP16ConversionIntrinsics())
1961 return llvm::ConstantInt::get(CGM.getLLVMContext(),
1962 Init.bitcastToAPInt());
1963 else
1964 return llvm::ConstantFP::get(CGM.getLLVMContext(), Init);
1965 }
1966 case APValue::ComplexFloat: {
1967 llvm::Constant *Complex[2];
1968
1969 Complex[0] = llvm::ConstantFP::get(CGM.getLLVMContext(),
1970 Value.getComplexFloatReal());
1971 Complex[1] = llvm::ConstantFP::get(CGM.getLLVMContext(),
1972 Value.getComplexFloatImag());
1973
1974 // FIXME: the target may want to specify that this is packed.
1975 llvm::StructType *STy =
1976 llvm::StructType::get(Complex[0]->getType(), Complex[1]->getType());
1977 return llvm::ConstantStruct::get(STy, Complex);
1978 }
1979 case APValue::Vector: {
1980 unsigned NumElts = Value.getVectorLength();
1981 SmallVector<llvm::Constant *, 4> Inits(NumElts);
1982
1983 for (unsigned I = 0; I != NumElts; ++I) {
1984 const APValue &Elt = Value.getVectorElt(I);
1985 if (Elt.isInt())
1986 Inits[I] = llvm::ConstantInt::get(CGM.getLLVMContext(), Elt.getInt());
1987 else if (Elt.isFloat())
1988 Inits[I] = llvm::ConstantFP::get(CGM.getLLVMContext(), Elt.getFloat());
1989 else
1990 llvm_unreachable("unsupported vector element type");
1991 }
1992 return llvm::ConstantVector::get(Inits);
1993 }
1994 case APValue::AddrLabelDiff: {
1995 const AddrLabelExpr *LHSExpr = Value.getAddrLabelDiffLHS();
1996 const AddrLabelExpr *RHSExpr = Value.getAddrLabelDiffRHS();
1997 llvm::Constant *LHS = tryEmitPrivate(LHSExpr, LHSExpr->getType());
1998 llvm::Constant *RHS = tryEmitPrivate(RHSExpr, RHSExpr->getType());
1999 if (!LHS || !RHS) return nullptr;
2000
2001 // Compute difference
2002 llvm::Type *ResultType = CGM.getTypes().ConvertType(DestType);
2003 LHS = llvm::ConstantExpr::getPtrToInt(LHS, CGM.IntPtrTy);
2004 RHS = llvm::ConstantExpr::getPtrToInt(RHS, CGM.IntPtrTy);
2005 llvm::Constant *AddrLabelDiff = llvm::ConstantExpr::getSub(LHS, RHS);
2006
2007 // LLVM is a bit sensitive about the exact format of the
2008 // address-of-label difference; make sure to truncate after
2009 // the subtraction.
2010 return llvm::ConstantExpr::getTruncOrBitCast(AddrLabelDiff, ResultType);
2011 }
2012 case APValue::Struct:
2013 case APValue::Union:
2014 return ConstStructBuilder::BuildStruct(*this, Value, DestType);
2015 case APValue::Array: {
2016 const ConstantArrayType *CAT =
2017 CGM.getContext().getAsConstantArrayType(DestType);
2018 unsigned NumElements = Value.getArraySize();
2019 unsigned NumInitElts = Value.getArrayInitializedElts();
2020
2021 // Emit array filler, if there is one.
2022 llvm::Constant *Filler = nullptr;
2023 if (Value.hasArrayFiller()) {
2024 Filler = tryEmitAbstractForMemory(Value.getArrayFiller(),
2025 CAT->getElementType());
2026 if (!Filler)
2027 return nullptr;
2028 }
2029
2030 // Emit initializer elements.
2031 SmallVector<llvm::Constant*, 16> Elts;
2032 if (Filler && Filler->isNullValue())
2033 Elts.reserve(NumInitElts + 1);
2034 else
2035 Elts.reserve(NumElements);
2036
2037 llvm::Type *CommonElementType = nullptr;
2038 for (unsigned I = 0; I < NumInitElts; ++I) {
2039 llvm::Constant *C = tryEmitPrivateForMemory(
2040 Value.getArrayInitializedElt(I), CAT->getElementType());
2041 if (!C) return nullptr;
2042
2043 if (I == 0)
2044 CommonElementType = C->getType();
2045 else if (C->getType() != CommonElementType)
2046 CommonElementType = nullptr;
2047 Elts.push_back(C);
2048 }
2049
2050 // This means that the array type is probably "IncompleteType" or some
2051 // type that is not ConstantArray.
2052 if (CAT == nullptr && CommonElementType == nullptr && !NumInitElts) {
2053 const ArrayType *AT = CGM.getContext().getAsArrayType(DestType);
2054 CommonElementType = CGM.getTypes().ConvertType(AT->getElementType());
2055 llvm::ArrayType *AType = llvm::ArrayType::get(CommonElementType,
2056 NumElements);
2057 return llvm::ConstantAggregateZero::get(AType);
2058 }
2059
2060 return EmitArrayConstant(CGM, CAT, CommonElementType, NumElements, Elts,
2061 Filler);
2062 }
2063 case APValue::MemberPointer:
2064 return CGM.getCXXABI().EmitMemberPointer(Value, DestType, nullptr);
2065 }
2066 llvm_unreachable("Unknown APValue kind");
2067}

2108static llvm::Constant *EmitNullConstant(CodeGenModule &CGM,
2109 const RecordDecl *record,
2110 bool asCompleteObject) {
2111 const CGRecordLayout &layout = CGM.getTypes().getCGRecordLayout(record);
2112 llvm::StructType *structure =
2113 (asCompleteObject ? layout.getLLVMType()
2114 : layout.getBaseSubobjectLLVMType());
2115
2116 unsigned numElements = structure->getNumElements();
2117 std::vector<llvm::Constant *> elements(numElements);
2118
2119 auto CXXR = dyn_cast<CXXRecordDecl>(record);
2120 // Fill in all the bases.
2121 if (CXXR) {
2122 for (const auto &I : CXXR->bases()) {
2123 if (I.isVirtual()) {
2124 // Ignore virtual bases; if we're laying out for a complete
2125 // object, we'll lay these out later.
2126 continue;
2127 }
2128
2129 const CXXRecordDecl *base =
2130 cast<CXXRecordDecl>(I.getType()->castAs<RecordType>()->getDecl());
2131
2132 // Ignore empty bases.
2133 if (base->isEmpty() ||
2134 CGM.getContext().getASTRecordLayout(base).getNonVirtualSize()
2135 .isZero())
2136 continue;
2137
2138 unsigned fieldIndex = layout.getNonVirtualBaseLLVMFieldNo(base);
2139 llvm::Type *baseType = structure->getElementType(fieldIndex);
2140 elements[fieldIndex] = EmitNullConstantForBase(CGM, baseType, base);
2141 }
2142 }
2143
2144 // Fill in all the fields.
2145 for (const auto *Field : record->fields()) {
2146 // Fill in non-bitfields. (Bitfields always use a zero pattern, which we
2147 // will fill in later.)
2148 if (!Field->isBitField()) {
2149 unsigned fieldIndex = layout.getLLVMFieldNo(Field);
2150 elements[fieldIndex] = CGM.EmitNullConstant(Field->getType());
2151 }
2152
2153 // For unions, stop after the first named field.
2154 if (record->isUnion()) {
2155 if (Field->getIdentifier())
2156 break;
2157 if (const auto *FieldRD = Field->getType()->getAsRecordDecl())
2158 if (FieldRD->findFirstNamedDataMember())
2159 break;
2160 }
2161 }
2162
2163 // Fill in the virtual bases, if we're working with the complete object.
2164 if (CXXR && asCompleteObject) {
2165 for (const auto &I : CXXR->vbases()) {
2166 const CXXRecordDecl *base =
2167 cast<CXXRecordDecl>(I.getType()->castAs<RecordType>()->getDecl());
2168
2169 // Ignore empty bases.
2170 if (base->isEmpty())
2171 continue;
2172
2173 unsigned fieldIndex = layout.getVirtualBaseIndex(base);
2174
2175 // We might have already laid this field out.
2176 if (elements[fieldIndex]) continue;
2177
2178 llvm::Type *baseType = structure->getElementType(fieldIndex);
2179 elements[fieldIndex] = EmitNullConstantForBase(CGM, baseType, base);
2180 }
2181 }
2182
2183 // Now go through all other fields and zero them out.
2184 for (unsigned i = 0; i != numElements; ++i) {
2185 if (!elements[i])
2186 elements[i] = llvm::Constant::getNullValue(structure->getElementType(i));
2187 }
2188
2189 return llvm::ConstantStruct::get(structure, elements);
2190}
