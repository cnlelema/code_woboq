212/// EmitAggLoadOfLValue - Given an expression with aggregate type that
213/// represents a value lvalue, this method emits the address of the lvalue,
214/// then loads the result into DestPtr.
215void AggExprEmitter::EmitAggLoadOfLValue(const Expr *E) {
216 LValue LV = CGF.EmitLValue(E);
217
218 // If the type of the l-value is atomic, then do an atomic load.
219 if (LV.getType()->isAtomicType() || CGF.LValueIsSuitableForInlineAtomic(LV)) {
220 CGF.EmitAtomicLoad(LV, E->getExprLoc(), Dest);
221 return;
222 }
223
224 EmitFinalDestCopy(E->getType(), LV);
225}

227/// True if the given aggregate type requires special GC API calls.
228bool AggExprEmitter::TypeRequiresGCollection(QualType T) {
229 // Only record types have members that might require garbage collection.
230 const RecordType *RecordTy = T->getAs<RecordType>();
231 if (!RecordTy) return false;
232
233 // Don't mess with non-trivial C++ types.
234 RecordDecl *Record = RecordTy->getDecl();
235 if (isa<CXXRecordDecl>(Record) &&
236 (cast<CXXRecordDecl>(Record)->hasNonTrivialCopyConstructor() ||
237 !cast<CXXRecordDecl>(Record)->hasTrivialDestructor()))
238 return false;
239
240 // Check whether the type has an object member.
241 return Record->hasObjectMember();
242}

244void AggExprEmitter::withReturnValueSlot(
245 const Expr *E, llvm::function_ref<RValue(ReturnValueSlot)> EmitCall) {
246 QualType RetTy = E->getType();
247 bool RequiresDestruction =
248 Dest.isIgnored() &&
249 RetTy.isDestructedType() == QualType::DK_nontrivial_c_struct;
250
251 // If it makes no observable difference, save a memcpy + temporary.
252 //
253 // We need to always provide our own temporary if destruction is required.
254 // Otherwise, EmitCall will emit its own, notice that it's "unused", and end
255 // its lifetime before we have the chance to emit a proper destructor call.
256 bool UseTemp = Dest.isPotentiallyAliased() || Dest.requiresGCollection() ||
257 (RequiresDestruction && !Dest.getAddress().isValid());
258
259 Address RetAddr = Address::invalid();
260 Address RetAllocaAddr = Address::invalid();
261
262 EHScopeStack::stable_iterator LifetimeEndBlock;
263 llvm::Value *LifetimeSizePtr = nullptr;
264 llvm::IntrinsicInst *LifetimeStartInst = nullptr;
265 if (!UseTemp) {
266 RetAddr = Dest.getAddress();
267 } else {
268 RetAddr = CGF.CreateMemTemp(RetTy, "tmp", &RetAllocaAddr);
269 uint64_t Size =
270 CGF.CGM.getDataLayout().getTypeAllocSize(CGF.ConvertTypeForMem(RetTy));
271 LifetimeSizePtr = CGF.EmitLifetimeStart(Size, RetAllocaAddr.getPointer());
272 if (LifetimeSizePtr) {
273 LifetimeStartInst =
274 cast<llvm::IntrinsicInst>(std::prev(Builder.GetInsertPoint()));
275 assert(LifetimeStartInst->getIntrinsicID() ==
276 llvm::Intrinsic::lifetime_start &&
277 "Last insertion wasn't a lifetime.start?");
278
279 CGF.pushFullExprCleanup<CodeGenFunction::CallLifetimeEnd>(
280 NormalEHLifetimeMarker, RetAllocaAddr, LifetimeSizePtr);
281 LifetimeEndBlock = CGF.EHStack.stable_begin();
282 }
283 }
284
285 RValue Src =
286 EmitCall(ReturnValueSlot(RetAddr, Dest.isVolatile(), IsResultUnused));
287
288 if (RequiresDestruction)
289 CGF.pushDestroy(RetTy.isDestructedType(), Src.getAggregateAddress(), RetTy);
290
291 if (!UseTemp)
292 return;
293
294 assert(Dest.getPointer() != Src.getAggregatePointer());
295 EmitFinalDestCopy(E->getType(), Src);
296
297 if (!RequiresDestruction && LifetimeStartInst) {
298 // If there's no dtor to run, the copy was the last use of our temporary.
299 // Since we're not guaranteed to be in an ExprWithCleanups, clean up
300 // eagerly.
301 CGF.DeactivateCleanupBlock(LifetimeEndBlock, LifetimeStartInst);
302 CGF.EmitLifetimeEnd(LifetimeSizePtr, RetAllocaAddr.getPointer());
303 }
304}

306/// EmitFinalDestCopy - Perform the final copy to DestPtr, if desired.
307void AggExprEmitter::EmitFinalDestCopy(QualType type, RValue src) {
308 assert(src.isAggregate() && "value must be aggregate value!");
309 LValue srcLV = CGF.MakeAddrLValue(src.getAggregateAddress(), type);
310 EmitFinalDestCopy(type, srcLV, EVK_RValue);

313/// EmitFinalDestCopy - Perform the final copy to DestPtr, if desired.
314void AggExprEmitter::EmitFinalDestCopy(QualType type, const LValue &src,
315 ExprValueKind SrcValueKind) {
316 // If Dest is ignored, then we're evaluating an aggregate expression
317 // in a context that doesn't care about the result. Note that loads
318 // from volatile l-values force the existence of a non-ignored
319 // destination.
320 if (Dest.isIgnored())
321 return;
322
323 // Copy non-trivial C structs here.
324 LValue DstLV = CGF.MakeAddrLValue(
325 Dest.getAddress(), Dest.isVolatile() ? type.withVolatile() : type);
326
327 if (SrcValueKind == EVK_RValue) {
328 if (type.isNonTrivialToPrimitiveDestructiveMove() == QualType::PCK_Struct) {
329 if (Dest.isPotentiallyAliased())
330 CGF.callCStructMoveAssignmentOperator(DstLV, src);
331 else
332 CGF.callCStructMoveConstructor(DstLV, src);
333 return;
334 }
335 } else {
336 if (type.isNonTrivialToPrimitiveCopy() == QualType::PCK_Struct) {
337 if (Dest.isPotentiallyAliased())
338 CGF.callCStructCopyAssignmentOperator(DstLV, src);
339 else
340 CGF.callCStructCopyConstructor(DstLV, src);
341 return;
342 }
343 }
344
345 AggValueSlot srcAgg =
346 AggValueSlot::forLValue(src, AggValueSlot::IsDestructed,
347 needsGC(type), AggValueSlot::IsAliased,
348 AggValueSlot::MayOverlap);
349 EmitCopy(type, Dest, srcAgg);
350}

352/// Perform a copy from the source into the destination.
353///
354/// \param type - the type of the aggregate being copied; qualifiers are
355/// ignored
356void AggExprEmitter::EmitCopy(QualType type, const AggValueSlot &dest,
357 const AggValueSlot &src) {
358 if (dest.requiresGCollection()) {
359 CharUnits sz = dest.getPreferredSize(CGF.getContext(), type);
360 llvm::Value *size = llvm::ConstantInt::get(CGF.SizeTy, sz.getQuantity());
361 CGF.CGM.getObjCRuntime().EmitGCMemmoveCollectable(CGF,
362 dest.getAddress(),
363 src.getAddress(),
364 size);
365 return;
366 }
367
368 // If the result of the assignment is used, copy the LHS there also.
369 // It's volatile if either side is. Use the minimum alignment of
370 // the two sides.
371 LValue DestLV = CGF.MakeAddrLValue(dest.getAddress(), type);
372 LValue SrcLV = CGF.MakeAddrLValue(src.getAddress(), type);
373 CGF.EmitAggregateCopy(DestLV, SrcLV, type, dest.mayOverlap(),
374 dest.isVolatile() || src.isVolatile());
375}

377/// Emit the initializer for a std::initializer_list initialized with a
378/// real initializer list.
379void
380AggExprEmitter::VisitCXXStdInitializerListExpr(CXXStdInitializerListExpr *E) {
381 // Emit an array containing the elements. The array is externally destructed
382 // if the std::initializer_list object is.
383 ASTContext &Ctx = CGF.getContext();
384 LValue Array = CGF.EmitLValue(E->getSubExpr());
385 assert(Array.isSimple() && "initializer_list array not a simple lvalue");
386 Address ArrayPtr = Array.getAddress();
387
388 const ConstantArrayType *ArrayType =
389 Ctx.getAsConstantArrayType(E->getSubExpr()->getType());
390 assert(ArrayType && "std::initializer_list constructed from non-array");
391
392 // FIXME: Perform the checks on the field types in SemaInit.
393 RecordDecl *Record = E->getType()->castAs<RecordType>()->getDecl();
394 RecordDecl::field_iterator Field = Record->field_begin();
395 if (Field == Record->field_end()) {
396 CGF.ErrorUnsupported(E, "weird std::initializer_list");
397 return;
398 }
399
400 // Start pointer.
401 if (!Field->getType()->isPointerType() ||
402 !Ctx.hasSameType(Field->getType()->getPointeeType(),
403 ArrayType->getElementType())) {
404 CGF.ErrorUnsupported(E, "weird std::initializer_list");
405 return;
406 }
407
408 AggValueSlot Dest = EnsureSlot(E->getType());
409 LValue DestLV = CGF.MakeAddrLValue(Dest.getAddress(), E->getType());
410 LValue Start = CGF.EmitLValueForFieldInitialization(DestLV, *Field);
411 llvm::Value *Zero = llvm::ConstantInt::get(CGF.PtrDiffTy, 0);
412 llvm::Value *IdxStart[] = { Zero, Zero };
413 llvm::Value *ArrayStart =
414 Builder.CreateInBoundsGEP(ArrayPtr.getPointer(), IdxStart, "arraystart");
415 CGF.EmitStoreThroughLValue(RValue::get(ArrayStart), Start);
416 ++Field;
417
418 if (Field == Record->field_end()) {
419 CGF.ErrorUnsupported(E, "weird std::initializer_list");
420 return;
421 }
422
423 llvm::Value *Size = Builder.getInt(ArrayType->getSize());
424 LValue EndOrLength = CGF.EmitLValueForFieldInitialization(DestLV, *Field);
425 if (Field->getType()->isPointerType() &&
426 Ctx.hasSameType(Field->getType()->getPointeeType(),
427 ArrayType->getElementType())) {
428 // End pointer.
429 llvm::Value *IdxEnd[] = { Zero, Size };
430 llvm::Value *ArrayEnd =
431 Builder.CreateInBoundsGEP(ArrayPtr.getPointer(), IdxEnd, "arrayend");
432 CGF.EmitStoreThroughLValue(RValue::get(ArrayEnd), EndOrLength);
433 } else if (Ctx.hasSameType(Field->getType(), Ctx.getSizeType())) {
434 // Length.
435 CGF.EmitStoreThroughLValue(RValue::get(Size), EndOrLength);
436 } else {
437 CGF.ErrorUnsupported(E, "weird std::initializer_list");
438 return;
439 }
440}

442/// Determine if E is a trivial array filler, that is, one that is
443/// equivalent to zero-initialization.
444static bool isTrivialFiller(Expr *E) {
445 if (!E)
446 return true;
447
448 if (isa<ImplicitValueInitExpr>(E))
449 return true;
450
451 if (auto *ILE = dyn_cast<InitListExpr>(E)) {
452 if (ILE->getNumInits())
453 return false;
454 return isTrivialFiller(ILE->getArrayFiller());
455 }
456
457 if (auto *Cons = dyn_cast_or_null<CXXConstructExpr>(E))
458 return Cons->getConstructor()->isDefaultConstructor() &&
459 Cons->getConstructor()->isTrivial();
460
461 // FIXME: Are there other cases where we can avoid emitting an initializer?
462 return false;
463}

465/// Emit initialization of an array from an initializer list.
466void AggExprEmitter::EmitArrayInit(Address DestPtr, llvm::ArrayType *AType,
467 QualType ArrayQTy, InitListExpr *E) {
468 uint64_t NumInitElements = E->getNumInits();
469
470 uint64_t NumArrayElements = AType->getNumElements();
471 assert(NumInitElements <= NumArrayElements);
472
473 QualType elementType =
474 CGF.getContext().getAsArrayType(ArrayQTy)->getElementType();
475
476 // DestPtr is an array*. Construct an elementType* by drilling
477 // down a level.
478 llvm::Value *zero = llvm::ConstantInt::get(CGF.SizeTy, 0);
479 llvm::Value *indices[] = { zero, zero };
480 llvm::Value *begin =
481 Builder.CreateInBoundsGEP(DestPtr.getPointer(), indices, "arrayinit.begin");
482
483 CharUnits elementSize = CGF.getContext().getTypeSizeInChars(elementType);
484 CharUnits elementAlign =
485 DestPtr.getAlignment().alignmentOfArrayElement(elementSize);
486
487 // Consider initializing the array by copying from a global. For this to be
488 // more efficient than per-element initialization, the size of the elements
489 // with explicit initializers should be large enough.
490 if (NumInitElements * elementSize.getQuantity() > 16 &&
491 elementType.isTriviallyCopyableType(CGF.getContext())) {
492 CodeGen::CodeGenModule &CGM = CGF.CGM;
493 ConstantEmitter Emitter(CGM);
494 LangAS AS = ArrayQTy.getAddressSpace();
495 if (llvm::Constant *C = Emitter.tryEmitForInitializer(E, AS, ArrayQTy)) {
496 auto GV = new llvm::GlobalVariable(
497 CGM.getModule(), C->getType(),
498 CGM.isTypeConstant(ArrayQTy, /* ExcludeCtorDtor= */ true),
499 llvm::GlobalValue::PrivateLinkage, C, "constinit",
500 /* InsertBefore= */ nullptr, llvm::GlobalVariable::NotThreadLocal,
501 CGM.getContext().getTargetAddressSpace(AS));
502 Emitter.finalize(GV);
503 CharUnits Align = CGM.getContext().getTypeAlignInChars(ArrayQTy);
504 GV->setAlignment(Align.getQuantity());
505 EmitFinalDestCopy(ArrayQTy, CGF.MakeAddrLValue(GV, ArrayQTy, Align));
506 return;
507 }
508 }
509
510 // Exception safety requires us to destroy all the
511 // already-constructed members if an initializer throws.
512 // For that, we'll need an EH cleanup.
513 QualType::DestructionKind dtorKind = elementType.isDestructedType();
514 Address endOfInit = Address::invalid();
515 EHScopeStack::stable_iterator cleanup;
516 llvm::Instruction *cleanupDominator = nullptr;
517 if (CGF.needsEHCleanup(dtorKind)) {
518 // In principle we could tell the cleanup where we are more
519 // directly, but the control flow can get so varied here that it
520 // would actually be quite complex. Therefore we go through an
521 // alloca.
522 endOfInit = CGF.CreateTempAlloca(begin->getType(), CGF.getPointerAlign(),
523 "arrayinit.endOfInit");
524 cleanupDominator = Builder.CreateStore(begin, endOfInit);
525 CGF.pushIrregularPartialArrayCleanup(begin, endOfInit, elementType,
526 elementAlign,
527 CGF.getDestroyer(dtorKind));
528 cleanup = CGF.EHStack.stable_begin();
529
530 // Otherwise, remember that we didn't need a cleanup.
531 } else {
532 dtorKind = QualType::DK_none;
533 }
534
535 llvm::Value *one = llvm::ConstantInt::get(CGF.SizeTy, 1);
536
537 // The 'current element to initialize'. The invariants on this
538 // variable are complicated. Essentially, after each iteration of
539 // the loop, it points to the last initialized element, except
540 // that it points to the beginning of the array before any
541 // elements have been initialized.
542 llvm::Value *element = begin;
543
544 // Emit the explicit initializers.
545 for (uint64_t i = 0; i != NumInitElements; ++i) {
546 // Advance to the next element.
547 if (i > 0) {
548 element = Builder.CreateInBoundsGEP(element, one, "arrayinit.element");
549
550 // Tell the cleanup that it needs to destroy up to this
551 // element. TODO: some of these stores can be trivially
552 // observed to be unnecessary.
553 if (endOfInit.isValid()) Builder.CreateStore(element, endOfInit);
554 }
555
556 LValue elementLV =
557 CGF.MakeAddrLValue(Address(element, elementAlign), elementType);
558 EmitInitializationToLValue(E->getInit(i), elementLV);
559 }
560
561 // Check whether there's a non-trivial array-fill expression.
562 Expr *filler = E->getArrayFiller();
563 bool hasTrivialFiller = isTrivialFiller(filler);
564
565 // Any remaining elements need to be zero-initialized, possibly
566 // using the filler expression. We can skip this if the we're
567 // emitting to zeroed memory.
568 if (NumInitElements != NumArrayElements &&
569 !(Dest.isZeroed() && hasTrivialFiller &&
570 CGF.getTypes().isZeroInitializable(elementType))) {
571
572 // Use an actual loop. This is basically
573 // do { *array++ = filler; } while (array != end);
574
575 // Advance to the start of the rest of the array.
576 if (NumInitElements) {
577 element = Builder.CreateInBoundsGEP(element, one, "arrayinit.start");
578 if (endOfInit.isValid()) Builder.CreateStore(element, endOfInit);
579 }
580
581 // Compute the end of the array.
582 llvm::Value *end = Builder.CreateInBoundsGEP(begin,
583 llvm::ConstantInt::get(CGF.SizeTy, NumArrayElements),
584 "arrayinit.end");
585
586 llvm::BasicBlock *entryBB = Builder.GetInsertBlock();
587 llvm::BasicBlock *bodyBB = CGF.createBasicBlock("arrayinit.body");
588
589 // Jump into the body.
590 CGF.EmitBlock(bodyBB);
591 llvm::PHINode *currentElement =
592 Builder.CreatePHI(element->getType(), 2, "arrayinit.cur");
593 currentElement->addIncoming(element, entryBB);
594
595 // Emit the actual filler expression.
596 {
597 // C++1z [class.temporary]p5:
598 // when a default constructor is called to initialize an element of
599 // an array with no corresponding initializer [...] the destruction of
600 // every temporary created in a default argument is sequenced before
601 // the construction of the next array element, if any
602 CodeGenFunction::RunCleanupsScope CleanupsScope(CGF);
603 LValue elementLV =
604 CGF.MakeAddrLValue(Address(currentElement, elementAlign), elementType);
605 if (filler)
606 EmitInitializationToLValue(filler, elementLV);
607 else
608 EmitNullInitializationToLValue(elementLV);
609 }
610
611 // Move on to the next element.
612 llvm::Value *nextElement =
613 Builder.CreateInBoundsGEP(currentElement, one, "arrayinit.next");
614
615 // Tell the EH cleanup that we finished with the last element.
616 if (endOfInit.isValid()) Builder.CreateStore(nextElement, endOfInit);
617
618 // Leave the loop if we're done.
619 llvm::Value *done = Builder.CreateICmpEQ(nextElement, end,
620 "arrayinit.done");
621 llvm::BasicBlock *endBB = CGF.createBasicBlock("arrayinit.end");
622 Builder.CreateCondBr(done, endBB, bodyBB);
623 currentElement->addIncoming(nextElement, Builder.GetInsertBlock());
624
625 CGF.EmitBlock(endBB);
626 }
627
628 // Leave the partial-array cleanup if we entered one.
629 if (dtorKind) CGF.DeactivateCleanupBlock(cleanup, cleanupDominator);
630}

677void AggExprEmitter::VisitCastExpr(CastExpr *E) {
678 if (const auto *ECE = dyn_cast<ExplicitCastExpr>(E))
679 CGF.CGM.EmitExplicitCastExprType(ECE, &CGF);
680 switch (E->getCastKind()) {
681 case CK_Dynamic: {
682 // FIXME: Can this actually happen? We have no test coverage for it.
683 assert(isa<CXXDynamicCastExpr>(E) && "CK_Dynamic without a dynamic_cast?");
684 LValue LV = CGF.EmitCheckedLValue(E->getSubExpr(),
685 CodeGenFunction::TCK_Load);
686 // FIXME: Do we also need to handle property references here?
687 if (LV.isSimple())
688 CGF.EmitDynamicCast(LV.getAddress(), cast<CXXDynamicCastExpr>(E));
689 else
690 CGF.CGM.ErrorUnsupported(E, "non-simple lvalue dynamic_cast");
691
692 if (!Dest.isIgnored())
693 CGF.CGM.ErrorUnsupported(E, "lvalue dynamic_cast with a destination");
694 break;
695 }
696
697 case CK_ToUnion: {
698 // Evaluate even if the destination is ignored.
699 if (Dest.isIgnored()) {
700 CGF.EmitAnyExpr(E->getSubExpr(), AggValueSlot::ignored(),
701 /*ignoreResult=*/true);
702 break;
703 }
704
705 // GCC union extension
706 QualType Ty = E->getSubExpr()->getType();
707 Address CastPtr =
708 Builder.CreateElementBitCast(Dest.getAddress(), CGF.ConvertType(Ty));
709 EmitInitializationToLValue(E->getSubExpr(),
710 CGF.MakeAddrLValue(CastPtr, Ty));
711 break;
712 }
713
714 case CK_DerivedToBase:
715 case CK_BaseToDerived:
716 case CK_UncheckedDerivedToBase: {
717 llvm_unreachable("cannot perform hierarchy conversion in EmitAggExpr: "
718 "should have been unpacked before we got here");
719 }
720
721 case CK_NonAtomicToAtomic:
722 case CK_AtomicToNonAtomic: {
723 bool isToAtomic = (E->getCastKind() == CK_NonAtomicToAtomic);
724
725 // Determine the atomic and value types.
726 QualType atomicType = E->getSubExpr()->getType();
727 QualType valueType = E->getType();
728 if (isToAtomic) std::swap(atomicType, valueType);
729
730 assert(atomicType->isAtomicType());
731 assert(CGF.getContext().hasSameUnqualifiedType(valueType,
732 atomicType->castAs<AtomicType>()->getValueType()));
733
734 // Just recurse normally if we're ignoring the result or the
735 // atomic type doesn't change representation.
736 if (Dest.isIgnored() || !CGF.CGM.isPaddedAtomicType(atomicType)) {
737 return Visit(E->getSubExpr());
738 }
739
740 CastKind peepholeTarget =
741 (isToAtomic ? CK_AtomicToNonAtomic : CK_NonAtomicToAtomic);
742
743 // These two cases are reverses of each other; try to peephole them.
744 if (Expr *op = findPeephole(E->getSubExpr(), peepholeTarget)) {
745 assert(CGF.getContext().hasSameUnqualifiedType(op->getType(),
746 E->getType()) &&
747 "peephole significantly changed types?");
748 return Visit(op);
749 }
750
751 // If we're converting an r-value of non-atomic type to an r-value
752 // of atomic type, just emit directly into the relevant sub-object.
753 if (isToAtomic) {
754 AggValueSlot valueDest = Dest;
755 if (!valueDest.isIgnored() && CGF.CGM.isPaddedAtomicType(atomicType)) {
756 // Zero-initialize. (Strictly speaking, we only need to initialize
757 // the padding at the end, but this is simpler.)
758 if (!Dest.isZeroed())
759 CGF.EmitNullInitialization(Dest.getAddress(), atomicType);
760
761 // Build a GEP to refer to the subobject.
762 Address valueAddr =
763 CGF.Builder.CreateStructGEP(valueDest.getAddress(), 0);
764 valueDest = AggValueSlot::forAddr(valueAddr,
765 valueDest.getQualifiers(),
766 valueDest.isExternallyDestructed(),
767 valueDest.requiresGCollection(),
768 valueDest.isPotentiallyAliased(),
769 AggValueSlot::DoesNotOverlap,
770 AggValueSlot::IsZeroed);
771 }
772
773 CGF.EmitAggExpr(E->getSubExpr(), valueDest);
774 return;
775 }
776
777 // Otherwise, we're converting an atomic type to a non-atomic type.
778 // Make an atomic temporary, emit into that, and then copy the value out.
779 AggValueSlot atomicSlot =
780 CGF.CreateAggTemp(atomicType, "atomic-to-nonatomic.temp");
781 CGF.EmitAggExpr(E->getSubExpr(), atomicSlot);
782
783 Address valueAddr = Builder.CreateStructGEP(atomicSlot.getAddress(), 0);
784 RValue rvalue = RValue::getAggregate(valueAddr, atomicSlot.isVolatile());
785 return EmitFinalDestCopy(valueType, rvalue);
786 }
787 case CK_AddressSpaceConversion:
788 return Visit(E->getSubExpr());
789
790 case CK_LValueToRValue:
791 // If we're loading from a volatile type, force the destination
792 // into existence.
793 if (E->getSubExpr()->getType().isVolatileQualified()) {
794 EnsureDest(E->getType());
795 return Visit(E->getSubExpr());
796 }
797
798 LLVM_FALLTHROUGH;
799
800
801 case CK_NoOp:
802 case CK_UserDefinedConversion:
803 case CK_ConstructorConversion:
804 assert(CGF.getContext().hasSameUnqualifiedType(E->getSubExpr()->getType(),
805 E->getType()) &&
806 "Implicit cast types must be compatible");
807 Visit(E->getSubExpr());
808 break;
809
810 case CK_LValueBitCast:
811 llvm_unreachable("should not be emitting lvalue bitcast as rvalue");
812
813 case CK_Dependent:
814 case CK_BitCast:
815 case CK_ArrayToPointerDecay:
816 case CK_FunctionToPointerDecay:
817 case CK_NullToPointer:
818 case CK_NullToMemberPointer:
819 case CK_BaseToDerivedMemberPointer:
820 case CK_DerivedToBaseMemberPointer:
821 case CK_MemberPointerToBoolean:
822 case CK_ReinterpretMemberPointer:
823 case CK_IntegralToPointer:
824 case CK_PointerToIntegral:
825 case CK_PointerToBoolean:
826 case CK_ToVoid:
827 case CK_VectorSplat:
828 case CK_IntegralCast:
829 case CK_BooleanToSignedIntegral:
830 case CK_IntegralToBoolean:
831 case CK_IntegralToFloating:
832 case CK_FloatingToIntegral:
833 case CK_FloatingToBoolean:
834 case CK_FloatingCast:
835 case CK_CPointerToObjCPointerCast:
836 case CK_BlockPointerToObjCPointerCast:
837 case CK_AnyPointerToBlockPointerCast:
838 case CK_ObjCObjectLValueCast:
839 case CK_FloatingRealToComplex:
840 case CK_FloatingComplexToReal:
841 case CK_FloatingComplexToBoolean:
842 case CK_FloatingComplexCast:
843 case CK_FloatingComplexToIntegralComplex:
844 case CK_IntegralRealToComplex:
845 case CK_IntegralComplexToReal:
846 case CK_IntegralComplexToBoolean:
847 case CK_IntegralComplexCast:
848 case CK_IntegralComplexToFloatingComplex:
849 case CK_ARCProduceObject:
850 case CK_ARCConsumeObject:

854 case CK_BuiltinFnToFnPtr:

897enum CompareKind {
898 CK_Less,
899 CK_Greater,
900 CK_Equal,
901};

903static llvm::Value *EmitCompare(CGBuilderTy &Builder, CodeGenFunction &CGF,
904 const BinaryOperator *E, llvm::Value *LHS,
905 llvm::Value *RHS, CompareKind Kind,
906 const char *NameSuffix = "") {
907 QualType ArgTy = E->getLHS()->getType();
908 if (const ComplexType *CT = ArgTy->getAs<ComplexType>())
909 ArgTy = CT->getElementType();
910
911 if (const auto *MPT = ArgTy->getAs<MemberPointerType>()) {
912 assert(Kind == CK_Equal &&
913 "member pointers may only be compared for equality");
914 return CGF.CGM.getCXXABI().EmitMemberPointerComparison(
915 CGF, LHS, RHS, MPT, /*IsInequality*/ false);
916 }
917
918 // Compute the comparison instructions for the specified comparison kind.
919 struct CmpInstInfo {
920 const char *Name;
921 llvm::CmpInst::Predicate FCmp;
922 llvm::CmpInst::Predicate SCmp;
923 llvm::CmpInst::Predicate UCmp;
924 };
925 CmpInstInfo InstInfo = [&]() -> CmpInstInfo {
926 using FI = llvm::FCmpInst;
927 using II = llvm::ICmpInst;
928 switch (Kind) {
929 case CK_Less:
930 return {"cmp.lt", FI::FCMP_OLT, II::ICMP_SLT, II::ICMP_ULT};
931 case CK_Greater:
932 return {"cmp.gt", FI::FCMP_OGT, II::ICMP_SGT, II::ICMP_UGT};
933 case CK_Equal:
934 return {"cmp.eq", FI::FCMP_OEQ, II::ICMP_EQ, II::ICMP_EQ};
935 }
936 llvm_unreachable("Unrecognised CompareKind enum");
937 }();
938
939 if (ArgTy->hasFloatingRepresentation())
940 return Builder.CreateFCmp(InstInfo.FCmp, LHS, RHS,
941 llvm::Twine(InstInfo.Name) + NameSuffix);
942 if (ArgTy->isIntegralOrEnumerationType() || ArgTy->isPointerType()) {
943 auto Inst =
944 ArgTy->hasSignedIntegerRepresentation() ? InstInfo.SCmp : InstInfo.UCmp;
945 return Builder.CreateICmp(Inst, LHS, RHS,
946 llvm::Twine(InstInfo.Name) + NameSuffix);
947 }
948
949 llvm_unreachable("unsupported aggregate binary expression should have "
950 "already been handled");
951}

953void AggExprEmitter::VisitBinCmp(const BinaryOperator *E) {
954 using llvm::BasicBlock;
955 using llvm::PHINode;
956 using llvm::Value;
957 assert(CGF.getContext().hasSameType(E->getLHS()->getType(),
958 E->getRHS()->getType()));
959 const ComparisonCategoryInfo &CmpInfo =
960 CGF.getContext().CompCategories.getInfoForType(E->getType());
961 assert(CmpInfo.Record->isTriviallyCopyable() &&
962 "cannot copy non-trivially copyable aggregate");
963
964 QualType ArgTy = E->getLHS()->getType();
965
966 // TODO: Handle comparing these types.
967 if (ArgTy->isVectorType())
968 return CGF.ErrorUnsupported(
969 E, "aggregate three-way comparison with vector arguments");
970 if (!ArgTy->isIntegralOrEnumerationType() && !ArgTy->isRealFloatingType() &&
971 !ArgTy->isNullPtrType() && !ArgTy->isPointerType() &&
972 !ArgTy->isMemberPointerType() && !ArgTy->isAnyComplexType()) {
973 return CGF.ErrorUnsupported(E, "aggregate three-way comparison");
974 }
975 bool IsComplex = ArgTy->isAnyComplexType();
976
977 // Evaluate the operands to the expression and extract their values.
978 auto EmitOperand = [&](Expr *E) -> std::pair<Value *, Value *> {
979 RValue RV = CGF.EmitAnyExpr(E);
980 if (RV.isScalar())
981 return {RV.getScalarVal(), nullptr};
982 if (RV.isAggregate())
983 return {RV.getAggregatePointer(), nullptr};
984 assert(RV.isComplex());
985 return RV.getComplexVal();
986 };
987 auto LHSValues = EmitOperand(E->getLHS()),
988 RHSValues = EmitOperand(E->getRHS());
989
990 auto EmitCmp = [&](CompareKind K) {
991 Value *Cmp = EmitCompare(Builder, CGF, E, LHSValues.first, RHSValues.first,
992 K, IsComplex ? ".r" : "");
993 if (!IsComplex)
994 return Cmp;
995 assert(K == CompareKind::CK_Equal);
996 Value *CmpImag = EmitCompare(Builder, CGF, E, LHSValues.second,
997 RHSValues.second, K, ".i");
998 return Builder.CreateAnd(Cmp, CmpImag, "and.eq");
999 };
1000 auto EmitCmpRes = [&](const ComparisonCategoryInfo::ValueInfo *VInfo) {
1001 return Builder.getInt(VInfo->getIntValue());
1002 };
1003
1004 Value *Select;
1005 if (ArgTy->isNullPtrType()) {
1006 Select = EmitCmpRes(CmpInfo.getEqualOrEquiv());
1007 } else if (CmpInfo.isEquality()) {
1008 Select = Builder.CreateSelect(
1009 EmitCmp(CK_Equal), EmitCmpRes(CmpInfo.getEqualOrEquiv()),
1010 EmitCmpRes(CmpInfo.getNonequalOrNonequiv()), "sel.eq");
1011 } else if (!CmpInfo.isPartial()) {
1012 Value *SelectOne =
1013 Builder.CreateSelect(EmitCmp(CK_Less), EmitCmpRes(CmpInfo.getLess()),
1014 EmitCmpRes(CmpInfo.getGreater()), "sel.lt");
1015 Select = Builder.CreateSelect(EmitCmp(CK_Equal),
1016 EmitCmpRes(CmpInfo.getEqualOrEquiv()),
1017 SelectOne, "sel.eq");
1018 } else {
1019 Value *SelectEq = Builder.CreateSelect(
1020 EmitCmp(CK_Equal), EmitCmpRes(CmpInfo.getEqualOrEquiv()),
1021 EmitCmpRes(CmpInfo.getUnordered()), "sel.eq");
1022 Value *SelectGT = Builder.CreateSelect(EmitCmp(CK_Greater),
1023 EmitCmpRes(CmpInfo.getGreater()),
1024 SelectEq, "sel.gt");
1025 Select = Builder.CreateSelect(
1026 EmitCmp(CK_Less), EmitCmpRes(CmpInfo.getLess()), SelectGT, "sel.lt");
1027 }
1028 // Create the return value in the destination slot.
1029 EnsureDest(E->getType());
1030 LValue DestLV = CGF.MakeAddrLValue(Dest.getAddress(), E->getType());
1031
1032 // Emit the address of the first (and only) field in the comparison category
1033 // type, and initialize it from the constant integer value selected above.
1034 LValue FieldLV = CGF.EmitLValueForFieldInitialization(
1035 DestLV, *CmpInfo.Record->field_begin());
1036 CGF.EmitStoreThroughLValue(RValue::get(Select), FieldLV, /*IsInit*/ true);
1037
1038 // All done! The result is in the Dest slot.
1039}

1054/// Is the value of the given expression possibly a reference to or
1055/// into a __block variable?
1056static bool isBlockVarRef(const Expr *E) {
1057 // Make sure we look through parens.
1058 E = E->IgnoreParens();
1059
1060 // Check for a direct reference to a __block variable.
1061 if (const DeclRefExpr *DRE = dyn_cast<DeclRefExpr>(E)) {
1062 const VarDecl *var = dyn_cast<VarDecl>(DRE->getDecl());
1063 return (var && var->hasAttr<BlocksAttr>());
1064 }
1065
1066 // More complicated stuff.
1067
1068 // Binary operators.
1069 if (const BinaryOperator *op = dyn_cast<BinaryOperator>(E)) {
1070 // For an assignment or pointer-to-member operation, just care
1071 // about the LHS.
1072 if (op->isAssignmentOp() || op->isPtrMemOp())
1073 return isBlockVarRef(op->getLHS());
1074
1075 // For a comma, just care about the RHS.
1076 if (op->getOpcode() == BO_Comma)
1077 return isBlockVarRef(op->getRHS());
1078
1079 // FIXME: pointer arithmetic?
1080 return false;
1081
1082 // Check both sides of a conditional operator.
1083 } else if (const AbstractConditionalOperator *op
1084 = dyn_cast<AbstractConditionalOperator>(E)) {
1085 return isBlockVarRef(op->getTrueExpr())
1086 || isBlockVarRef(op->getFalseExpr());
1087
1088 // OVEs are required to support BinaryConditionalOperators.
1089 } else if (const OpaqueValueExpr *op
1090 = dyn_cast<OpaqueValueExpr>(E)) {
1091 if (const Expr *src = op->getSourceExpr())
1092 return isBlockVarRef(src);
1093
1094 // Casts are necessary to get things like (*(int*)&var) = foo().
1095 // We don't really care about the kind of cast here, except
1096 // we don't want to look through l2r casts, because it's okay
1097 // to get the *value* in a __block variable.
1098 } else if (const CastExpr *cast = dyn_cast<CastExpr>(E)) {
1099 if (cast->getCastKind() == CK_LValueToRValue)
1100 return false;
1101 return isBlockVarRef(cast->getSubExpr());
1102
1103 // Handle unary operators. Again, just aggressively look through
1104 // it, ignoring the operation.
1105 } else if (const UnaryOperator *uop = dyn_cast<UnaryOperator>(E)) {
1106 return isBlockVarRef(uop->getSubExpr());
1107
1108 // Look into the base of a field access.
1109 } else if (const MemberExpr *mem = dyn_cast<MemberExpr>(E)) {
1110 return isBlockVarRef(mem->getBase());
1111
1112 // Look into the base of a subscript.
1113 } else if (const ArraySubscriptExpr *sub = dyn_cast<ArraySubscriptExpr>(E)) {
1114 return isBlockVarRef(sub->getBase());
1115 }
1116
1117 return false;
1118}

1120void AggExprEmitter::VisitBinAssign(const BinaryOperator *E) {
1121 // For an assignment to work, the value on the right has
1122 // to be compatible with the value on the left.
1123 assert(CGF.getContext().hasSameUnqualifiedType(E->getLHS()->getType(),
1124 E->getRHS()->getType())
1125 && "Invalid assignment");
1126
1127 // If the LHS might be a __block variable, and the RHS can
1128 // potentially cause a block copy, we need to evaluate the RHS first
1129 // so that the assignment goes the right place.
1130 // This is pretty semantically fragile.
1131 if (isBlockVarRef(E->getLHS()) &&
1132 E->getRHS()->HasSideEffects(CGF.getContext())) {
1133 // Ensure that we have a destination, and evaluate the RHS into that.
1134 EnsureDest(E->getRHS()->getType());
1135 Visit(E->getRHS());
1136
1137 // Now emit the LHS and copy into it.
1138 LValue LHS = CGF.EmitCheckedLValue(E->getLHS(), CodeGenFunction::TCK_Store);
1139
1140 // That copy is an atomic copy if the LHS is atomic.
1141 if (LHS.getType()->isAtomicType() ||
1142 CGF.LValueIsSuitableForInlineAtomic(LHS)) {
1143 CGF.EmitAtomicStore(Dest.asRValue(), LHS, /*isInit*/ false);
1144 return;
1145 }
1146
1147 EmitCopy(E->getLHS()->getType(),
1148 AggValueSlot::forLValue(LHS, AggValueSlot::IsDestructed,
1149 needsGC(E->getLHS()->getType()),
1150 AggValueSlot::IsAliased,
1151 AggValueSlot::MayOverlap),
1152 Dest);
1153 return;
1154 }
1155
1156 LValue LHS = CGF.EmitLValue(E->getLHS());
1157
1158 // If we have an atomic type, evaluate into the destination and then
1159 // do an atomic copy.
1160 if (LHS.getType()->isAtomicType() ||
1161 CGF.LValueIsSuitableForInlineAtomic(LHS)) {
1162 EnsureDest(E->getRHS()->getType());
1163 Visit(E->getRHS());
1164 CGF.EmitAtomicStore(Dest.asRValue(), LHS, /*isInit*/ false);
1165 return;
1166 }
1167
1168 // Codegen the RHS so that it stores directly into the LHS.
1169 AggValueSlot LHSSlot =
1170 AggValueSlot::forLValue(LHS, AggValueSlot::IsDestructed,
1171 needsGC(E->getLHS()->getType()),
1172 AggValueSlot::IsAliased,
1173 AggValueSlot::MayOverlap);
1174 // A non-volatile aggregate destination might have volatile member.
1175 if (!LHSSlot.isVolatile() &&
1176 CGF.hasVolatileMember(E->getLHS()->getType()))
1177 LHSSlot.setVolatile(true);
1178
1179 CGF.EmitAggExpr(E->getRHS(), LHSSlot);
1180
1181 // Copy into the destination if the assignment isn't ignored.
1182 EmitFinalDestCopy(E->getType(), LHS);
1183}

1185void AggExprEmitter::
1186VisitAbstractConditionalOperator(const AbstractConditionalOperator *E) {
1187 llvm::BasicBlock *LHSBlock = CGF.createBasicBlock("cond.true");
1188 llvm::BasicBlock *RHSBlock = CGF.createBasicBlock("cond.false");
1189 llvm::BasicBlock *ContBlock = CGF.createBasicBlock("cond.end");
1190
1191 // Bind the common expression if necessary.
1192 CodeGenFunction::OpaqueValueMapping binding(CGF, E);
1193
1194 CodeGenFunction::ConditionalEvaluation eval(CGF);
1195 CGF.EmitBranchOnBoolExpr(E->getCond(), LHSBlock, RHSBlock,
1196 CGF.getProfileCount(E));
1197
1198 // Save whether the destination's lifetime is externally managed.
1199 bool isExternallyDestructed = Dest.isExternallyDestructed();
1200
1201 eval.begin(CGF);
1202 CGF.EmitBlock(LHSBlock);
1203 CGF.incrementProfileCounter(E);
1204 Visit(E->getTrueExpr());
1205 eval.end(CGF);
1206
1207 assert(CGF.HaveInsertPoint() && "expression evaluation ended with no IP!");
1208 CGF.Builder.CreateBr(ContBlock);
1209
1210 // If the result of an agg expression is unused, then the emission
1211 // of the LHS might need to create a destination slot. That's fine
1212 // with us, and we can safely emit the RHS into the same slot, but
1213 // we shouldn't claim that it's already being destructed.
1214 Dest.setExternallyDestructed(isExternallyDestructed);
1215
1216 eval.begin(CGF);
1217 CGF.EmitBlock(RHSBlock);
1218 Visit(E->getFalseExpr());
1219 eval.end(CGF);
1220
1221 CGF.EmitBlock(ContBlock);
1222}

1228void AggExprEmitter::VisitVAArgExpr(VAArgExpr *VE) {
1229 Address ArgValue = Address::invalid();

1271void
1272AggExprEmitter::VisitLambdaExpr(LambdaExpr *E) {
1273 AggValueSlot Slot = EnsureSlot(E->getType());
1274 LValue SlotLV = CGF.MakeAddrLValue(Slot.getAddress(), E->getType());
1275
1276 // We'll need to enter cleanup scopes in case any of the element
1277 // initializers throws an exception.
1278 SmallVector<EHScopeStack::stable_iterator, 16> Cleanups;
1279 llvm::Instruction *CleanupDominator = nullptr;
1280
1281 CXXRecordDecl::field_iterator CurField = E->getLambdaClass()->field_begin();
1282 for (LambdaExpr::const_capture_init_iterator i = E->capture_init_begin(),
1283 e = E->capture_init_end();

1286 LValue LV = CGF.EmitLValueForFieldInitialization(SlotLV, *CurField);
1287 if (CurField->hasCapturedVLAType()) {
1288 CGF.EmitLambdaVLACapture(CurField->getCapturedVLAType(), LV);
1289 continue;
1290 }
1291
1292 EmitInitializationToLValue(*i, LV);
1293
1294 // Push a destructor if necessary.
1295 if (QualType::DestructionKind DtorKind =
1296 CurField->getType().isDestructedType()) {
1297 assert(LV.isSimple());
1298 if (CGF.needsEHCleanup(DtorKind)) {
1299 if (!CleanupDominator)
1300 CleanupDominator = CGF.Builder.CreateAlignedLoad(
1301 CGF.Int8Ty,
1302 llvm::Constant::getNullValue(CGF.Int8PtrTy),
1303 CharUnits::One()); // placeholder
1304
1305 CGF.pushDestroy(EHCleanup, LV.getAddress(), CurField->getType(),
1306 CGF.getDestroyer(DtorKind), false);
1307 Cleanups.push_back(CGF.EHStack.stable_begin());
1308 }
1309 }
1310 }
1311
1312 // Deactivate all the partial cleanups in reverse order, which
1313 // generally means popping them.
1314 for (unsigned i = Cleanups.size(); i != 0; --i)
1315 CGF.DeactivateCleanupBlock(Cleanups[i-1], CleanupDominator);
1316
1317 // Destroy the placeholder if we made one.
1318 if (CleanupDominator)
1319 CleanupDominator->eraseFromParent();
1320}

1340/// isSimpleZero - If emitting this value will obviously just cause a store of
1341/// zero to memory, return true. This can return false if uncertain, so it just
1342/// handles simple cases.
1343static bool isSimpleZero(const Expr *E, CodeGenFunction &CGF) {
1344 E = E->IgnoreParens();
1345
1346 // 0
1347 if (const IntegerLiteral *IL = dyn_cast<IntegerLiteral>(E))
1348 return IL->getValue() == 0;
1349 // +0.0
1350 if (const FloatingLiteral *FL = dyn_cast<FloatingLiteral>(E))
1351 return FL->getValue().isPosZero();
1352 // int()
1353 if ((isa<ImplicitValueInitExpr>(E) || isa<CXXScalarValueInitExpr>(E)) &&
1354 CGF.getTypes().isZeroInitializable(E->getType()))
1355 return true;
1356 // (int*)0 - Null pointer expressions.
1357 if (const CastExpr *ICE = dyn_cast<CastExpr>(E))
1358 return ICE->getCastKind() == CK_NullToPointer &&
1359 CGF.getTypes().isPointerZeroInitializable(E->getType());
1360 // '\0'
1361 if (const CharacterLiteral *CL = dyn_cast<CharacterLiteral>(E))
1362 return CL->getValue() == 0;
1363
1364 // Otherwise, hard case: conservatively return false.
1365 return false;
1366}

1369void
1370AggExprEmitter::EmitInitializationToLValue(Expr *E, LValue LV) {
1371 QualType type = LV.getType();
1372 // FIXME: Ignore result?
1373 // FIXME: Are initializers affected by volatile?
1374 if (Dest.isZeroed() && isSimpleZero(E, CGF)) {
1375 // Storing "i32 0" to a zero'd memory location is a noop.
1376 return;
1377 } else if (isa<ImplicitValueInitExpr>(E) || isa<CXXScalarValueInitExpr>(E)) {
1378 return EmitNullInitializationToLValue(LV);
1379 } else if (isa<NoInitExpr>(E)) {
1380 // Do nothing.
1381 return;
1382 } else if (type->isReferenceType()) {
1383 RValue RV = CGF.EmitReferenceBindingToExpr(E);
1384 return CGF.EmitStoreThroughLValue(RV, LV);
1385 }
1386
1387 switch (CGF.getEvaluationKind(type)) {
1388 case TEK_Complex:
1389 CGF.EmitComplexExprIntoLValue(E, LV, /*isInit*/ true);
1390 return;
1391 case TEK_Aggregate:
1392 CGF.EmitAggExpr(E, AggValueSlot::forLValue(LV,
1393 AggValueSlot::IsDestructed,
1394 AggValueSlot::DoesNotNeedGCBarriers,
1395 AggValueSlot::IsNotAliased,
1396 AggValueSlot::MayOverlap,
1397 Dest.isZeroed()));
1398 return;
1399 case TEK_Scalar:
1400 if (LV.isSimple()) {
1401 CGF.EmitScalarInit(E, /*D=*/nullptr, LV, /*Captured=*/false);
1402 } else {
1403 CGF.EmitStoreThroughLValue(RValue::get(CGF.EmitScalarExpr(E)), LV);
1404 }
1405 return;
1406 }
1407 llvm_unreachable("bad evaluation kind");
1408}

1410void AggExprEmitter::EmitNullInitializationToLValue(LValue lv) {
1411 QualType type = lv.getType();
1412
1413 // If the destination slot is already zeroed out before the aggregate is
1414 // copied into it, we don't have to emit any zeros here.
1415 if (Dest.isZeroed() && CGF.getTypes().isZeroInitializable(type))
1416 return;
1417
1418 if (CGF.hasScalarEvaluationKind(type)) {
1419 // For non-aggregates, we can store the appropriate null constant.
1420 llvm::Value *null = CGF.CGM.EmitNullConstant(type);
1421 // Note that the following is not equivalent to
1422 // EmitStoreThroughBitfieldLValue for ARC types.
1423 if (lv.isBitField()) {
1424 CGF.EmitStoreThroughBitfieldLValue(RValue::get(null), lv);
1425 } else {
1426 assert(lv.isSimple());
1427 CGF.EmitStoreOfScalar(null, lv, /* isInitialization */ true);
1428 }
1429 } else {
1430 // There's a potential optimization opportunity in combining
1431 // memsets; that would be easy for arrays, but relatively
1432 // difficult for structures with the current code.
1433 CGF.EmitNullInitialization(lv.getAddress(), lv.getType());
1434 }

1615void AggExprEmitter::VisitArrayInitLoopExpr(const ArrayInitLoopExpr *E,
1616 llvm::Value *outerBegin) {
1617 // Emit the common subexpression.
1618 CodeGenFunction::OpaqueValueMapping binding(CGF, E->getCommonExpr());
1619
1620 Address destPtr = EnsureSlot(E->getType()).getAddress();
1621 uint64_t numElements = E->getArraySize().getZExtValue();
1622
1623 if (!numElements)
1624 return;
1625
1626 // destPtr is an array*. Construct an elementType* by drilling down a level.
1627 llvm::Value *zero = llvm::ConstantInt::get(CGF.SizeTy, 0);
1628 llvm::Value *indices[] = {zero, zero};
1629 llvm::Value *begin = Builder.CreateInBoundsGEP(destPtr.getPointer(), indices,
1630 "arrayinit.begin");
1631
1632 // Prepare to special-case multidimensional array initialization: we avoid
1633 // emitting multiple destructor loops in that case.
1634 if (!outerBegin)
1635 outerBegin = begin;
1636 ArrayInitLoopExpr *InnerLoop = dyn_cast<ArrayInitLoopExpr>(E->getSubExpr());
1637
1638 QualType elementType =
1639 CGF.getContext().getAsArrayType(E->getType())->getElementType();
1640 CharUnits elementSize = CGF.getContext().getTypeSizeInChars(elementType);
1641 CharUnits elementAlign =
1642 destPtr.getAlignment().alignmentOfArrayElement(elementSize);
1643
1644 llvm::BasicBlock *entryBB = Builder.GetInsertBlock();
1645 llvm::BasicBlock *bodyBB = CGF.createBasicBlock("arrayinit.body");
1646
1647 // Jump into the body.
1648 CGF.EmitBlock(bodyBB);
1649 llvm::PHINode *index =
1650 Builder.CreatePHI(zero->getType(), 2, "arrayinit.index");
1651 index->addIncoming(zero, entryBB);
1652 llvm::Value *element = Builder.CreateInBoundsGEP(begin, index);
1653
1654 // Prepare for a cleanup.
1655 QualType::DestructionKind dtorKind = elementType.isDestructedType();
1656 EHScopeStack::stable_iterator cleanup;
1657 if (CGF.needsEHCleanup(dtorKind) && !InnerLoop) {
1658 if (outerBegin->getType() != element->getType())
1659 outerBegin = Builder.CreateBitCast(outerBegin, element->getType());
1660 CGF.pushRegularPartialArrayCleanup(outerBegin, element, elementType,
1661 elementAlign,
1662 CGF.getDestroyer(dtorKind));
1663 cleanup = CGF.EHStack.stable_begin();
1664 } else {
1665 dtorKind = QualType::DK_none;
1666 }
1667
1668 // Emit the actual filler expression.
1669 {
1670 // Temporaries created in an array initialization loop are destroyed
1671 // at the end of each iteration.
1672 CodeGenFunction::RunCleanupsScope CleanupsScope(CGF);
1673 CodeGenFunction::ArrayInitLoopExprScope Scope(CGF, index);
1674 LValue elementLV =
1675 CGF.MakeAddrLValue(Address(element, elementAlign), elementType);
1676
1677 if (InnerLoop) {
1678 // If the subexpression is an ArrayInitLoopExpr, share its cleanup.
1679 auto elementSlot = AggValueSlot::forLValue(
1680 elementLV, AggValueSlot::IsDestructed,
1681 AggValueSlot::DoesNotNeedGCBarriers,
1682 AggValueSlot::IsNotAliased,
1683 AggValueSlot::DoesNotOverlap);
1684 AggExprEmitter(CGF, elementSlot, false)
1685 .VisitArrayInitLoopExpr(InnerLoop, outerBegin);
1686 } else
1687 EmitInitializationToLValue(E->getSubExpr(), elementLV);
1688 }
1689
1690 // Move on to the next element.
1691 llvm::Value *nextIndex = Builder.CreateNUWAdd(
1692 index, llvm::ConstantInt::get(CGF.SizeTy, 1), "arrayinit.next");
1693 index->addIncoming(nextIndex, Builder.GetInsertBlock());
1694
1695 // Leave the loop if we're done.
1696 llvm::Value *done = Builder.CreateICmpEQ(
1697 nextIndex, llvm::ConstantInt::get(CGF.SizeTy, numElements),
1698 "arrayinit.done");
1699 llvm::BasicBlock *endBB = CGF.createBasicBlock("arrayinit.end");
1700 Builder.CreateCondBr(done, endBB, bodyBB);
1701
1702 CGF.EmitBlock(endBB);
1703
1704 // Leave the partial-array cleanup if we entered one.
1705 if (dtorKind)
1706 CGF.DeactivateCleanupBlock(cleanup, index);
1707}

1721/// GetNumNonZeroBytesInInit - Get an approximate count of the number of
1722/// non-zero bytes that will be stored when outputting the initializer for the
1723/// specified initializer expression.
1724static CharUnits GetNumNonZeroBytesInInit(const Expr *E, CodeGenFunction &CGF) {
1725 E = E->IgnoreParens();
1726
1727 // 0 and 0.0 won't require any non-zero stores!
1728 if (isSimpleZero(E, CGF)) return CharUnits::Zero();
1729
1730 // If this is an initlist expr, sum up the size of sizes of the (present)
1731 // elements. If this is something weird, assume the whole thing is non-zero.
1732 const InitListExpr *ILE = dyn_cast<InitListExpr>(E);
1733 while (ILE && ILE->isTransparent())
1734 ILE = dyn_cast<InitListExpr>(ILE->getInit(0));
1735 if (!ILE || !CGF.getTypes().isZeroInitializable(ILE->getType()))
1736 return CGF.getContext().getTypeSizeInChars(E->getType());
1737
1738 // InitListExprs for structs have to be handled carefully. If there are
1739 // reference members, we need to consider the size of the reference, not the
1740 // referencee. InitListExprs for unions and arrays can't have references.
1741 if (const RecordType *RT = E->getType()->getAs<RecordType>()) {
1742 if (!RT->isUnionType()) {
1743 RecordDecl *SD = E->getType()->getAs<RecordType>()->getDecl();
1744 CharUnits NumNonZeroBytes = CharUnits::Zero();
1745
1746 unsigned ILEElement = 0;
1747 if (auto *CXXRD = dyn_cast<CXXRecordDecl>(SD))
1748 while (ILEElement != CXXRD->getNumBases())
1749 NumNonZeroBytes +=
1750 GetNumNonZeroBytesInInit(ILE->getInit(ILEElement++), CGF);
1751 for (const auto *Field : SD->fields()) {
1752 // We're done once we hit the flexible array member or run out of
1753 // InitListExpr elements.
1754 if (Field->getType()->isIncompleteArrayType() ||
1755 ILEElement == ILE->getNumInits())
1756 break;
1757 if (Field->isUnnamedBitfield())
1758 continue;
1759
1760 const Expr *E = ILE->getInit(ILEElement++);
1761
1762 // Reference values are always non-null and have the width of a pointer.
1763 if (Field->getType()->isReferenceType())
1764 NumNonZeroBytes += CGF.getContext().toCharUnitsFromBits(
1765 CGF.getTarget().getPointerWidth(0));
1766 else
1767 NumNonZeroBytes += GetNumNonZeroBytesInInit(E, CGF);
1768 }
1769
1770 return NumNonZeroBytes;
1771 }
1772 }
1773
1774
1775 CharUnits NumNonZeroBytes = CharUnits::Zero();
1776 for (unsigned i = 0, e = ILE->getNumInits(); i != e; ++i)
1777 NumNonZeroBytes += GetNumNonZeroBytesInInit(ILE->getInit(i), CGF);
1778 return NumNonZeroBytes;
1779}

1781/// CheckAggExprForMemSetUse - If the initializer is large and has a lot of
1782/// zeros in it, emit a memset and avoid storing the individual zeros.
1783///
1784static void CheckAggExprForMemSetUse(AggValueSlot &Slot, const Expr *E,
1785 CodeGenFunction &CGF) {
1786 // If the slot is already known to be zeroed, nothing to do. Don't mess with
1787 // volatile stores.
1788 if (Slot.isZeroed() || Slot.isVolatile() || !Slot.getAddress().isValid())
1789 return;
1790
1791 // C++ objects with a user-declared constructor don't need zero'ing.
1792 if (CGF.getLangOpts().CPlusPlus)
1793 if (const RecordType *RT = CGF.getContext()
1794 .getBaseElementType(E->getType())->getAs<RecordType>()) {
1795 const CXXRecordDecl *RD = cast<CXXRecordDecl>(RT->getDecl());
1796 if (RD->hasUserDeclaredConstructor())
1797 return;
1798 }
1799
1800 // If the type is 16-bytes or smaller, prefer individual stores over memset.
1801 CharUnits Size = Slot.getPreferredSize(CGF.getContext(), E->getType());
1802 if (Size <= CharUnits::fromQuantity(16))
1803 return;
1804
1805 // Check to see if over 3/4 of the initializer are known to be zero. If so,
1806 // we prefer to emit memset + individual stores for the rest.
1807 CharUnits NumNonZeroBytes = GetNumNonZeroBytesInInit(E, CGF);
1808 if (NumNonZeroBytes*4 > Size)
1809 return;
1810
1811 // Okay, it seems like a good idea to use an initial memset, emit the call.
1812 llvm::Constant *SizeVal = CGF.Builder.getInt64(Size.getQuantity());
1813
1814 Address Loc = Slot.getAddress();
1815 Loc = CGF.Builder.CreateElementBitCast(Loc, CGF.Int8Ty);
1816 CGF.Builder.CreateMemSet(Loc, CGF.Builder.getInt8(0), SizeVal, false);
1817

1820}

1852AggValueSlot::Overlap_t CodeGenFunction::overlapForBaseInit(
1853 const CXXRecordDecl *RD, const CXXRecordDecl *BaseRD, bool IsVirtual) {
1854 // Virtual bases are initialized first, in address order, so there's never
1855 // any overlap during their initialization.
1856 //
1857 // FIXME: Under P0840, this is no longer true: the tail padding of a vbase
1858 // of a field could be reused by a vbase of a containing class.
1859 if (IsVirtual)
1860 return AggValueSlot::DoesNotOverlap;
1861
1862 // If the base class is laid out entirely within the nvsize of the derived
1863 // class, its tail padding cannot yet be initialized, so we can issue
1864 // stores at the full width of the base class.
1865 const ASTRecordLayout &Layout = getContext().getASTRecordLayout(RD);
1866 if (Layout.getBaseClassOffset(BaseRD) +
1867 getContext().getASTRecordLayout(BaseRD).getSize() <=
1868 Layout.getNonVirtualSize())
1869 return AggValueSlot::DoesNotOverlap;
1870
1871 // The tail padding may contain values we need to preserve.
1872 return AggValueSlot::MayOverlap;
1873}

1875void CodeGenFunction::EmitAggregateCopy(LValue Dest, LValue Src, QualType Ty,
1876 AggValueSlot::Overlap_t MayOverlap,
1877 bool isVolatile) {
1878 assert(!Ty->isAnyComplexType() && "Shouldn't happen for complex");
1879
1880 Address DestPtr = Dest.getAddress();
1881 Address SrcPtr = Src.getAddress();
1882
1883 if (getLangOpts().CPlusPlus) {
1884 if (const RecordType *RT = Ty->getAs<RecordType>()) {
1885 CXXRecordDecl *Record = cast<CXXRecordDecl>(RT->getDecl());
1886 assert((Record->hasTrivialCopyConstructor() ||
1887 Record->hasTrivialCopyAssignment() ||
1888 Record->hasTrivialMoveConstructor() ||
1889 Record->hasTrivialMoveAssignment() ||
1890 Record->isUnion()) &&
1891 "Trying to aggregate-copy a type without a trivial copy/move "
1892 "constructor or assignment operator");
1893 // Ignore empty classes in C++.
1894 if (Record->isEmpty())
1895 return;
1896 }
1897 }
1898
1899 // Aggregate assignment turns into llvm.memcpy. This is almost valid per
1900 // C99 6.5.16.1p3, which states "If the value being stored in an object is
1901 // read from another object that overlaps in anyway the storage of the first
1902 // object, then the overlap shall be exact and the two objects shall have
1903 // qualified or unqualified versions of a compatible type."
1904 //
1905 // memcpy is not defined if the source and destination pointers are exactly
1906 // equal, but other compilers do this optimization, and almost every memcpy
1907 // implementation handles this case safely. If there is a libc that does not
1908 // safely handle this, we can add a target hook.
1909
1910 // Get data size info for this aggregate. Don't copy the tail padding if this
1911 // might be a potentially-overlapping subobject, since the tail padding might
1912 // be occupied by a different object. Otherwise, copying it is fine.
1913 std::pair<CharUnits, CharUnits> TypeInfo;
1914 if (MayOverlap)
1915 TypeInfo = getContext().getTypeInfoDataSizeInChars(Ty);
1916 else
1917 TypeInfo = getContext().getTypeInfoInChars(Ty);
1918
1919 llvm::Value *SizeVal = nullptr;
1920 if (TypeInfo.first.isZero()) {
1921 // But note that getTypeInfo returns 0 for a VLA.
1922 if (auto *VAT = dyn_cast_or_null<VariableArrayType>(
1923 getContext().getAsArrayType(Ty))) {
1924 QualType BaseEltTy;
1925 SizeVal = emitArrayLength(VAT, BaseEltTy, DestPtr);
1926 TypeInfo = getContext().getTypeInfoInChars(BaseEltTy);
1927 assert(!TypeInfo.first.isZero());
1928 SizeVal = Builder.CreateNUWMul(
1929 SizeVal,
1930 llvm::ConstantInt::get(SizeTy, TypeInfo.first.getQuantity()));
1931 }
1932 }
1933 if (!SizeVal) {
1934 SizeVal = llvm::ConstantInt::get(SizeTy, TypeInfo.first.getQuantity());
1935 }
1936
1937 // FIXME: If we have a volatile struct, the optimizer can remove what might
1938 // appear to be `extra' memory ops:
1939 //
1940 // volatile struct { int i; } a, b;
1941 //
1942 // int main() {
1943 // a = b;
1944 // a = b;
1945 // }
1946 //
1947 // we need to use a different call here. We use isVolatile to indicate when
1948 // either the source or the destination is volatile.
1949
1950 DestPtr = Builder.CreateElementBitCast(DestPtr, Int8Ty);
1951 SrcPtr = Builder.CreateElementBitCast(SrcPtr, Int8Ty);
1952
1953 // Don't do any of the memmove_collectable tests if GC isn't set.
1954 if (CGM.getLangOpts().getGC() == LangOptions::NonGC) {
1955 // fall through
1956 } else if (const RecordType *RecordTy = Ty->getAs<RecordType>()) {
1957 RecordDecl *Record = RecordTy->getDecl();
1958 if (Record->hasObjectMember()) {
1959 CGM.getObjCRuntime().EmitGCMemmoveCollectable(*this, DestPtr, SrcPtr,
1960 SizeVal);
1961 return;
1962 }
1963 } else if (Ty->isArrayType()) {
1964 QualType BaseType = getContext().getBaseElementType(Ty);
1965 if (const RecordType *RecordTy = BaseType->getAs<RecordType>()) {
1966 if (RecordTy->getDecl()->hasObjectMember()) {
1967 CGM.getObjCRuntime().EmitGCMemmoveCollectable(*this, DestPtr, SrcPtr,
1968 SizeVal);
1969 return;
1970 }
1971 }
1972 }
1973
1974 auto Inst = Builder.CreateMemCpy(DestPtr, SrcPtr, SizeVal, isVolatile);
1975
1976 // Determine the metadata to describe the position of any padding in this
1977 // memcpy, as well as the TBAA tags for the members of the struct, in case
1978 // the optimizer wishes to expand it in to scalar memory operations.
1979 if (llvm::MDNode *TBAAStructTag = CGM.getTBAAStructInfo(Ty))
1980 Inst->setMetadata(llvm::LLVMContext::MD_tbaa_struct, TBAAStructTag);
1981
1982 if (CGM.getCodeGenOpts().NewStructPathTBAA) {
1983 TBAAAccessInfo TBAAInfo = CGM.mergeTBAAInfoForMemoryTransfer(
1984 Dest.getTBAAInfo(), Src.getTBAAInfo());
1985 CGM.DecorateInstructionWithTBAA(Inst, TBAAInfo);
1986 }
