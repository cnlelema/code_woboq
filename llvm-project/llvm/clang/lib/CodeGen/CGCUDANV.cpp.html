77 /// Helper function that generates a constant string and returns a pointer to
78 /// the start of the string. The result of this function can be used anywhere
79 /// where the C code specifies const char*.
80 llvm::Constant *makeConstantString(const std::string &Str,
81 const std::string &Name = "",
82 const std::string &SectionName = "",
83 unsigned Alignment = 0) {
84 llvm::Constant *Zeros[] = {llvm::ConstantInt::get(SizeTy, 0),
85 llvm::ConstantInt::get(SizeTy, 0)};
86 auto ConstStr = CGM.GetAddrOfConstantCString(Str, Name.c_str());
87 llvm::GlobalVariable *GV =
88 cast<llvm::GlobalVariable>(ConstStr.getPointer());
89 if (!SectionName.empty()) {
90 GV->setSection(SectionName);
91 // Mark the address as used which make sure that this section isn't
92 // merged and we will really have it in the object file.
93 GV->setUnnamedAddr(llvm::GlobalValue::UnnamedAddr::None);
94 }
95 if (Alignment)
96 GV->setAlignment(Alignment);
97
98 return llvm::ConstantExpr::getGetElementPtr(ConstStr.getElementType(),
99 ConstStr.getPointer(), Zeros);
100 }

102 /// Helper function that generates an empty dummy function returning void.
103 llvm::Function *makeDummyFunction(llvm::FunctionType *FnTy) {
104 assert(FnTy->getReturnType()->isVoidTy() &&
105 "Can only generate dummy functions returning void!");
106 llvm::Function *DummyFunc = llvm::Function::Create(
107 FnTy, llvm::GlobalValue::InternalLinkage, "dummy", &TheModule);
108
109 llvm::BasicBlock *DummyBlock =
110 llvm::BasicBlock::Create(Context, "", DummyFunc);
111 CGBuilderTy FuncBuilder(CGM, Context);
112 FuncBuilder.SetInsertPoint(DummyBlock);
113 FuncBuilder.CreateRetVoid();
114
115 return DummyFunc;
116 }

140 if (CGM.getLangOpts().HIP)
141 return ((Twine("hip") + Twine(FuncName)).str());
142 return ((Twine("cuda") + Twine(FuncName)).str());

151CGNVCUDARuntime::CGNVCUDARuntime(CodeGenModule &CGM)
152 : CGCUDARuntime(CGM), Context(CGM.getLLVMContext()),
153 TheModule(CGM.getModule()),
154 RelocatableDeviceCode(CGM.getLangOpts().GPURelocatableDeviceCode),
155 DeviceMC(CGM.getContext().createMangleContext(
156 CGM.getContext().getAuxTargetInfo())) {
157 CodeGen::CodeGenTypes &Types = CGM.getTypes();
158 ASTContext &Ctx = CGM.getContext();
159
160 IntTy = CGM.IntTy;
161 SizeTy = CGM.SizeTy;
162 VoidTy = CGM.VoidTy;
163
164 unsigned DefaultAS = CGM.getTargetCodeGenInfo().getDefaultAS();
165 CharPtrTy = llvm::PointerType::get(Types.ConvertType(Ctx.CharTy), DefaultAS);
166 VoidPtrTy = cast<llvm::PointerType>(Types.ConvertType(Ctx.VoidPtrTy));
167 VoidPtrPtrTy = VoidPtrTy->getPointerTo(DefaultAS);
168}

170llvm::FunctionCallee CGNVCUDARuntime::getSetupArgumentFn() const {
171 // cudaError_t cudaSetupArgument(void *, size_t, size_t)
172 llvm::Type *Params[] = {VoidPtrTy, SizeTy, SizeTy};

174 llvm::FunctionType::get(IntTy, Params, false),
175 addPrefixToName("SetupArgument"));

206std::string CGNVCUDARuntime::getDeviceSideName(const Decl *D) {
207 auto *ND = cast<const NamedDecl>(D);
208 std::string DeviceSideName;
209 if (DeviceMC->shouldMangleDeclName(ND)) {
210 SmallString<256> Buffer;
211 llvm::raw_svector_ostream Out(Buffer);
212 DeviceMC->mangleName(ND, Out);
213 DeviceSideName = Out.str();
214 } else
215 DeviceSideName = ND->getIdentifier()->getName();
216 return DeviceSideName;
217}

219void CGNVCUDARuntime::emitDeviceStub(CodeGenFunction &CGF,
220 FunctionArgList &Args) {
221 assert(getDeviceSideName(CGF.CurFuncDecl) == CGF.CurFn->getName() ||
222 getDeviceSideName(CGF.CurFuncDecl) + ".stub" == CGF.CurFn->getName() ||
223 CGF.CGM.getContext().getTargetInfo().getCXXABI() !=
224 CGF.CGM.getContext().getAuxTargetInfo()->getCXXABI());
225
226 EmittedKernels.push_back({CGF.CurFn, CGF.CurFuncDecl});
227 if (CudaFeatureEnabled(CGM.getTarget().getSDKVersion(),
228 CudaFeature::CUDA_USES_NEW_LAUNCH))
229 emitDeviceStubBodyNew(CGF, Args);
230 else
231 emitDeviceStubBodyLegacy(CGF, Args);
232}

234// CUDA 9.0+ uses new way to launch kernels. Parameters are packed in a local
235// array and kernels are launched using cudaLaunchKernel().
236void CGNVCUDARuntime::emitDeviceStubBodyNew(CodeGenFunction &CGF,
237 FunctionArgList &Args) {
238 // Build the shadow stack entry at the very start of the function.
239
240 // Calculate amount of space we will need for all arguments. If we have no
241 // args, allocate a single pointer so we still have a valid pointer to the
242 // argument array that we can pass to runtime, even if it will be unused.
243 Address KernelArgs = CGF.CreateTempAlloca(
244 VoidPtrTy, CharUnits::fromQuantity(16), "kernel_args",
245 llvm::ConstantInt::get(SizeTy, std::max<size_t>(1, Args.size())));
246 // Store pointers to the arguments in a locally allocated launch_args.
247 for (unsigned i = 0; i < Args.size(); ++i) {
248 llvm::Value* VarPtr = CGF.GetAddrOfLocalVar(Args[i]).getPointer();
249 llvm::Value *VoidVarPtr = CGF.Builder.CreatePointerCast(VarPtr, VoidPtrTy);
250 CGF.Builder.CreateDefaultAlignedStore(
251 VoidVarPtr, CGF.Builder.CreateConstGEP1_32(KernelArgs.getPointer(), i));
252 }
253
254 llvm::BasicBlock *EndBlock = CGF.createBasicBlock("setup.end");
255
256 // Lookup cudaLaunchKernel function.
257 // cudaError_t cudaLaunchKernel(const void *func, dim3 gridDim, dim3 blockDim,
258 // void **args, size_t sharedMem,
259 // cudaStream_t stream);
260 TranslationUnitDecl *TUDecl = CGM.getContext().getTranslationUnitDecl();
261 DeclContext *DC = TranslationUnitDecl::castToDeclContext(TUDecl);
262 IdentifierInfo &cudaLaunchKernelII =
263 CGM.getContext().Idents.get("cudaLaunchKernel");
264 FunctionDecl *cudaLaunchKernelFD = nullptr;
265 for (const auto &Result : DC->lookup(&cudaLaunchKernelII)) {
266 if (FunctionDecl *FD = dyn_cast<FunctionDecl>(Result))
267 cudaLaunchKernelFD = FD;
268 }
269
270 if (cudaLaunchKernelFD == nullptr) {
271 CGM.Error(CGF.CurFuncDecl->getLocation(),
272 "Can't find declaration for cudaLaunchKernel()");
273 return;
274 }
275 // Create temporary dim3 grid_dim, block_dim.
276 ParmVarDecl *GridDimParam = cudaLaunchKernelFD->getParamDecl(1);
277 QualType Dim3Ty = GridDimParam->getType();
278 Address GridDim =
279 CGF.CreateMemTemp(Dim3Ty, CharUnits::fromQuantity(8), "grid_dim");
280 Address BlockDim =
281 CGF.CreateMemTemp(Dim3Ty, CharUnits::fromQuantity(8), "block_dim");
282 Address ShmemSize =
283 CGF.CreateTempAlloca(SizeTy, CGM.getSizeAlign(), "shmem_size");
284 Address Stream =
285 CGF.CreateTempAlloca(VoidPtrTy, CGM.getPointerAlign(), "stream");
286 llvm::FunctionCallee cudaPopConfigFn = CGM.CreateRuntimeFunction(
287 llvm::FunctionType::get(IntTy,
288 {/*gridDim=*/GridDim.getType(),
289 /*blockDim=*/BlockDim.getType(),
290 /*ShmemSize=*/ShmemSize.getType(),
291 /*Stream=*/Stream.getType()},
292 /*isVarArg=*/false),
293 "__cudaPopCallConfiguration");
294
295 CGF.EmitRuntimeCallOrInvoke(cudaPopConfigFn,
296 {GridDim.getPointer(), BlockDim.getPointer(),
297 ShmemSize.getPointer(), Stream.getPointer()});
298
299 // Emit the call to cudaLaunch
300 llvm::Value *Kernel = CGF.Builder.CreatePointerCast(CGF.CurFn, VoidPtrTy);
301 CallArgList LaunchKernelArgs;
302 LaunchKernelArgs.add(RValue::get(Kernel),
303 cudaLaunchKernelFD->getParamDecl(0)->getType());
304 LaunchKernelArgs.add(RValue::getAggregate(GridDim), Dim3Ty);
305 LaunchKernelArgs.add(RValue::getAggregate(BlockDim), Dim3Ty);
306 LaunchKernelArgs.add(RValue::get(KernelArgs.getPointer()),
307 cudaLaunchKernelFD->getParamDecl(3)->getType());
308 LaunchKernelArgs.add(RValue::get(CGF.Builder.CreateLoad(ShmemSize)),
309 cudaLaunchKernelFD->getParamDecl(4)->getType());
310 LaunchKernelArgs.add(RValue::get(CGF.Builder.CreateLoad(Stream)),
311 cudaLaunchKernelFD->getParamDecl(5)->getType());
312
313 QualType QT = cudaLaunchKernelFD->getType();
314 QualType CQT = QT.getCanonicalType();
315 llvm::Type *Ty = CGM.getTypes().ConvertType(CQT);
316 llvm::FunctionType *FTy = dyn_cast<llvm::FunctionType>(Ty);
317
318 const CGFunctionInfo &FI =
319 CGM.getTypes().arrangeFunctionDeclaration(cudaLaunchKernelFD);
320 llvm::FunctionCallee cudaLaunchKernelFn =
321 CGM.CreateRuntimeFunction(FTy, "cudaLaunchKernel");
322 CGF.EmitCall(FI, CGCallee::forDirect(cudaLaunchKernelFn), ReturnValueSlot(),
323 LaunchKernelArgs);
324 CGF.EmitBranch(EndBlock);
325
326 CGF.EmitBlock(EndBlock);
327}

329void CGNVCUDARuntime::emitDeviceStubBodyLegacy(CodeGenFunction &CGF,
330 FunctionArgList &Args) {
331 // Emit a call to cudaSetupArgument for each arg in Args.
332 llvm::FunctionCallee cudaSetupArgFn = getSetupArgumentFn();
333 llvm::BasicBlock *EndBlock = CGF.createBasicBlock("setup.end");
334 CharUnits Offset = CharUnits::Zero();
335 for (const VarDecl *A : Args) {
336 CharUnits TyWidth, TyAlign;
337 std::tie(TyWidth, TyAlign) =
338 CGM.getContext().getTypeInfoInChars(A->getType());
339 Offset = Offset.alignTo(TyAlign);
340 llvm::Value *Args[] = {
341 CGF.Builder.CreatePointerCast(CGF.GetAddrOfLocalVar(A).getPointer(),
342 VoidPtrTy),
343 llvm::ConstantInt::get(SizeTy, TyWidth.getQuantity()),
344 llvm::ConstantInt::get(SizeTy, Offset.getQuantity()),
345 };
346 llvm::CallBase *CB = CGF.EmitRuntimeCallOrInvoke(cudaSetupArgFn, Args);
347 llvm::Constant *Zero = llvm::ConstantInt::get(IntTy, 0);
348 llvm::Value *CBZero = CGF.Builder.CreateICmpEQ(CB, Zero);
349 llvm::BasicBlock *NextBlock = CGF.createBasicBlock("setup.next");
350 CGF.Builder.CreateCondBr(CBZero, NextBlock, EndBlock);
351 CGF.EmitBlock(NextBlock);
352 Offset += TyWidth;
353 }
354
355 // Emit the call to cudaLaunch
356 llvm::FunctionCallee cudaLaunchFn = getLaunchFn();
357 llvm::Value *Arg = CGF.Builder.CreatePointerCast(CGF.CurFn, CharPtrTy);
358 CGF.EmitRuntimeCallOrInvoke(cudaLaunchFn, Arg);
359 CGF.EmitBranch(EndBlock);
360
361 CGF.EmitBlock(EndBlock);
362}

364/// Creates a function that sets up state on the host side for CUDA objects that
365/// have a presence on both the host and device sides. Specifically, registers
366/// the host side of kernel functions and device global variables with the CUDA
367/// runtime.
368/// \code
369/// void __cuda_register_globals(void** GpuBinaryHandle) {
370/// __cudaRegisterFunction(GpuBinaryHandle,Kernel0,...);
371/// ...
372/// __cudaRegisterFunction(GpuBinaryHandle,KernelM,...);
373/// __cudaRegisterVar(GpuBinaryHandle, GlobalVar0, ...);
374/// ...
375/// __cudaRegisterVar(GpuBinaryHandle, GlobalVarN, ...);
376/// }
377/// \endcode
378llvm::Function *CGNVCUDARuntime::makeRegisterGlobalsFn() {
379 // No need to register anything
380 if (EmittedKernels.empty() && DeviceVars.empty())
381 return nullptr;
382
383 llvm::Function *RegisterKernelsFunc = llvm::Function::Create(
384 getRegisterGlobalsFnTy(), llvm::GlobalValue::InternalLinkage,
385 addUnderscoredPrefixToName("_register_globals"), &TheModule);
386 llvm::BasicBlock *EntryBB =
387 llvm::BasicBlock::Create(Context, "entry", RegisterKernelsFunc);
388 CGBuilderTy Builder(CGM, Context);
389 Builder.SetInsertPoint(EntryBB);
390
391 unsigned DefaultAS = CGM.getTargetCodeGenInfo().getDefaultAS();
392 // void __cudaRegisterFunction(void **, const char *, char *, const char *,
393 // int, uint3*, uint3*, dim3*, dim3*, int*)
394 llvm::Type *RegisterFuncParams[] = {
395 VoidPtrPtrTy, CharPtrTy, CharPtrTy, CharPtrTy, IntTy,
396 VoidPtrTy, VoidPtrTy, VoidPtrTy, VoidPtrTy, IntTy->getPointerTo(DefaultAS)};
397 llvm::FunctionCallee RegisterFunc = CGM.CreateRuntimeFunction(
398 llvm::FunctionType::get(IntTy, RegisterFuncParams, false),
399 addUnderscoredPrefixToName("RegisterFunction"));
400
401 // Extract GpuBinaryHandle passed as the first argument passed to
402 // __cuda_register_globals() and generate __cudaRegisterFunction() call for
403 // each emitted kernel.
404 llvm::Argument &GpuBinaryHandlePtr = *RegisterKernelsFunc->arg_begin();
405 for (auto &&I : EmittedKernels) {
406 llvm::Constant *KernelName = makeConstantString(getDeviceSideName(I.D));
407 llvm::Constant *NullPtr = llvm::ConstantPointerNull::get(VoidPtrTy);
408 llvm::Value *Args[] = {
409 &GpuBinaryHandlePtr,
410 Builder.CreateBitCast(I.Kernel, VoidPtrTy),
411 KernelName,
412 KernelName,
413 llvm::ConstantInt::get(IntTy, -1),
414 NullPtr,
415 NullPtr,
416 NullPtr,
417 NullPtr,
418 llvm::ConstantPointerNull::get(IntTy->getPointerTo(DefaultAS))};
419 Builder.CreateCall(RegisterFunc, Args);
420 }
421
422 // void __cudaRegisterVar(void **, char *, char *, const char *,
423 // int, int, int, int)
424 llvm::Type *RegisterVarParams[] = {VoidPtrPtrTy, CharPtrTy, CharPtrTy,
425 CharPtrTy, IntTy, IntTy,
426 IntTy, IntTy};
427 llvm::FunctionCallee RegisterVar = CGM.CreateRuntimeFunction(
428 llvm::FunctionType::get(IntTy, RegisterVarParams, false),
429 addUnderscoredPrefixToName("RegisterVar"));
430 for (auto &&Info : DeviceVars) {
431 llvm::GlobalVariable *Var = Info.Var;
432 unsigned Flags = Info.Flag;
433 llvm::Constant *VarName = makeConstantString(getDeviceSideName(Info.D));
434 uint64_t VarSize =
435 CGM.getDataLayout().getTypeAllocSize(Var->getValueType());
436 llvm::Value *Args[] = {
437 &GpuBinaryHandlePtr,
438 Builder.CreateBitCast(Var, VoidPtrTy),
439 VarName,
440 VarName,
441 llvm::ConstantInt::get(IntTy, (Flags & ExternDeviceVar) ? 1 : 0),
442 llvm::ConstantInt::get(IntTy, VarSize),
443 llvm::ConstantInt::get(IntTy, (Flags & ConstantDeviceVar) ? 1 : 0),
444 llvm::ConstantInt::get(IntTy, 0)};
445 Builder.CreateCall(RegisterVar, Args);
446 }
447
448 Builder.CreateRetVoid();
449 return RegisterKernelsFunc;
450}

452/// Creates a global constructor function for the module:
453///
454/// For CUDA:
455/// \code
456/// void __cuda_module_ctor(void*) {
457/// Handle = __cudaRegisterFatBinary(GpuBinaryBlob);
458/// __cuda_register_globals(Handle);
459/// }
460/// \endcode
461///
462/// For HIP:
463/// \code
464/// void __hip_module_ctor(void*) {
465/// if (__hip_gpubin_handle == 0) {
466/// __hip_gpubin_handle = __hipRegisterFatBinary(GpuBinaryBlob);
467/// __hip_register_globals(__hip_gpubin_handle);
468/// }
469/// }
470/// \endcode
471llvm::Function *CGNVCUDARuntime::makeModuleCtorFunction() {
472 bool IsHIP = CGM.getLangOpts().HIP;
473 bool IsCUDA = CGM.getLangOpts().CUDA;
474 // No need to generate ctors/dtors if there is no GPU binary.
475 StringRef CudaGpuBinaryFileName = CGM.getCodeGenOpts().CudaGpuBinaryFileName;
476 if (CudaGpuBinaryFileName.empty() && !IsHIP)
477 return nullptr;
478 if ((IsHIP || (IsCUDA && !RelocatableDeviceCode)) && EmittedKernels.empty() &&
479 DeviceVars.empty())
480 return nullptr;
481
482 // void __{cuda|hip}_register_globals(void* handle);
483 llvm::Function *RegisterGlobalsFunc = makeRegisterGlobalsFn();
484 // We always need a function to pass in as callback. Create a dummy
485 // implementation if we don't need to register anything.
486 if (RelocatableDeviceCode && !RegisterGlobalsFunc)
487 RegisterGlobalsFunc = makeDummyFunction(getRegisterGlobalsFnTy());
488
489 // void ** __{cuda|hip}RegisterFatBinary(void *);
490 llvm::FunctionCallee RegisterFatbinFunc = CGM.CreateRuntimeFunction(
491 llvm::FunctionType::get(VoidPtrPtrTy, VoidPtrTy, false),
492 addUnderscoredPrefixToName("RegisterFatBinary"));
493 // struct { int magic, int version, void * gpu_binary, void * dont_care };
494 llvm::StructType *FatbinWrapperTy =
495 llvm::StructType::get(IntTy, IntTy, VoidPtrTy, VoidPtrTy);
496
497 // Register GPU binary with the CUDA runtime, store returned handle in a
498 // global variable and save a reference in GpuBinaryHandle to be cleaned up
499 // in destructor on exit. Then associate all known kernels with the GPU binary
500 // handle so CUDA runtime can figure out what to call on the GPU side.
501 std::unique_ptr<llvm::MemoryBuffer> CudaGpuBinary = nullptr;
502 if (!CudaGpuBinaryFileName.empty()) {
503 llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> CudaGpuBinaryOrErr =
504 llvm::MemoryBuffer::getFileOrSTDIN(CudaGpuBinaryFileName);
505 if (std::error_code EC = CudaGpuBinaryOrErr.getError()) {
506 CGM.getDiags().Report(diag::err_cannot_open_file)
507 << CudaGpuBinaryFileName << EC.message();
508 return nullptr;
509 }
510 CudaGpuBinary = std::move(CudaGpuBinaryOrErr.get());
511 }
512
513 llvm::Function *ModuleCtorFunc = llvm::Function::Create(
514 llvm::FunctionType::get(VoidTy, VoidPtrTy, false),
515 llvm::GlobalValue::InternalLinkage,
516 addUnderscoredPrefixToName("_module_ctor"), &TheModule);
517 llvm::BasicBlock *CtorEntryBB =
518 llvm::BasicBlock::Create(Context, "entry", ModuleCtorFunc);
519 CGBuilderTy CtorBuilder(CGM, Context);
520
521 CtorBuilder.SetInsertPoint(CtorEntryBB);
522
523 const char *FatbinConstantName;
524 const char *FatbinSectionName;
525 const char *ModuleIDSectionName;
526 StringRef ModuleIDPrefix;
527 llvm::Constant *FatBinStr;
528 unsigned FatMagic;
529 if (IsHIP) {
530 FatbinConstantName = ".hip_fatbin";
531 FatbinSectionName = ".hipFatBinSegment";
532
533 ModuleIDSectionName = "__hip_module_id";
534 ModuleIDPrefix = "__hip_";
535
536 if (CudaGpuBinary) {
537 // If fatbin is available from early finalization, create a string
538 // literal containing the fat binary loaded from the given file.
539 FatBinStr = makeConstantString(CudaGpuBinary->getBuffer(), "",
540 FatbinConstantName, 8);
541 } else {
542 // If fatbin is not available, create an external symbol
543 // __hip_fatbin in section .hip_fatbin. The external symbol is supposed
544 // to contain the fat binary but will be populated somewhere else,
545 // e.g. by lld through link script.
546 FatBinStr = new llvm::GlobalVariable(
547 CGM.getModule(), CGM.Int8Ty,
548 /*isConstant=*/true, llvm::GlobalValue::ExternalLinkage, nullptr,
549 "__hip_fatbin", nullptr,
550 llvm::GlobalVariable::NotThreadLocal);
551 cast<llvm::GlobalVariable>(FatBinStr)->setSection(FatbinConstantName);
552 }
553
554 FatMagic = HIPFatMagic;
555 } else {
556 if (RelocatableDeviceCode)
557 FatbinConstantName = CGM.getTriple().isMacOSX()
558 ? "__NV_CUDA,__nv_relfatbin"
559 : "__nv_relfatbin";
560 else
561 FatbinConstantName =
562 CGM.getTriple().isMacOSX() ? "__NV_CUDA,__nv_fatbin" : ".nv_fatbin";
563 // NVIDIA's cuobjdump looks for fatbins in this section.
564 FatbinSectionName =
565 CGM.getTriple().isMacOSX() ? "__NV_CUDA,__fatbin" : ".nvFatBinSegment";
566
567 ModuleIDSectionName = CGM.getTriple().isMacOSX()
568 ? "__NV_CUDA,__nv_module_id"
569 : "__nv_module_id";
570 ModuleIDPrefix = "__nv_";
571
572 // For CUDA, create a string literal containing the fat binary loaded from
573 // the given file.
574 FatBinStr = makeConstantString(CudaGpuBinary->getBuffer(), "",
575 FatbinConstantName, 8);
576 FatMagic = CudaFatMagic;
577 }
578
579 // Create initialized wrapper structure that points to the loaded GPU binary
580 ConstantInitBuilder Builder(CGM);
581 auto Values = Builder.beginStruct(FatbinWrapperTy);
582 // Fatbin wrapper magic.
583 Values.addInt(IntTy, FatMagic);
584 // Fatbin version.
585 Values.addInt(IntTy, 1);
586 // Data.
587 Values.add(FatBinStr);
588 // Unused in fatbin v1.
589 Values.add(llvm::ConstantPointerNull::get(VoidPtrTy));
590 llvm::GlobalVariable *FatbinWrapper = Values.finishAndCreateGlobal(
591 addUnderscoredPrefixToName("_fatbin_wrapper"), CGM.getPointerAlign(),
592 /*constant*/ true);
593 FatbinWrapper->setSection(FatbinSectionName);
594
595 // There is only one HIP fat binary per linked module, however there are
596 // multiple constructor functions. Make sure the fat binary is registered
597 // only once. The constructor functions are executed by the dynamic loader
598 // before the program gains control. The dynamic loader cannot execute the
599 // constructor functions concurrently since doing that would not guarantee
600 // thread safety of the loaded program. Therefore we can assume sequential
601 // execution of constructor functions here.
602 if (IsHIP) {
603 auto Linkage = CudaGpuBinary ? llvm::GlobalValue::InternalLinkage :
604 llvm::GlobalValue::LinkOnceAnyLinkage;
605 llvm::BasicBlock *IfBlock =
606 llvm::BasicBlock::Create(Context, "if", ModuleCtorFunc);
607 llvm::BasicBlock *ExitBlock =
608 llvm::BasicBlock::Create(Context, "exit", ModuleCtorFunc);
609 // The name, size, and initialization pattern of this variable is part
610 // of HIP ABI.
611 GpuBinaryHandle = new llvm::GlobalVariable(
612 TheModule, VoidPtrPtrTy, /*isConstant=*/false,
613 Linkage,
614 /*Initializer=*/llvm::ConstantPointerNull::get(VoidPtrPtrTy),
615 "__hip_gpubin_handle");
616 GpuBinaryHandle->setAlignment(CGM.getPointerAlign().getQuantity());
617 // Prevent the weak symbol in different shared libraries being merged.
618 if (Linkage != llvm::GlobalValue::InternalLinkage)
619 GpuBinaryHandle->setVisibility(llvm::GlobalValue::HiddenVisibility);
620 Address GpuBinaryAddr(
621 GpuBinaryHandle,
622 CharUnits::fromQuantity(GpuBinaryHandle->getAlignment()));
623 {
624 auto HandleValue = CtorBuilder.CreateLoad(GpuBinaryAddr);
625 llvm::Constant *Zero =
626 llvm::Constant::getNullValue(HandleValue->getType());
627 llvm::Value *EQZero = CtorBuilder.CreateICmpEQ(HandleValue, Zero);
628 CtorBuilder.CreateCondBr(EQZero, IfBlock, ExitBlock);
629 }
630 {
631 CtorBuilder.SetInsertPoint(IfBlock);
632 // GpuBinaryHandle = __hipRegisterFatBinary(&FatbinWrapper);
633 llvm::CallInst *RegisterFatbinCall = CtorBuilder.CreateCall(
634 RegisterFatbinFunc,
635 CtorBuilder.CreateBitCast(FatbinWrapper, VoidPtrTy));
636 CtorBuilder.CreateStore(RegisterFatbinCall, GpuBinaryAddr);
637 CtorBuilder.CreateBr(ExitBlock);
638 }
639 {
640 CtorBuilder.SetInsertPoint(ExitBlock);
641 // Call __hip_register_globals(GpuBinaryHandle);
642 if (RegisterGlobalsFunc) {
643 auto HandleValue = CtorBuilder.CreateLoad(GpuBinaryAddr);
644 CtorBuilder.CreateCall(RegisterGlobalsFunc, HandleValue);
645 }
646 }
647 } else if (!RelocatableDeviceCode) {
648 // Register binary with CUDA runtime. This is substantially different in
649 // default mode vs. separate compilation!
650 // GpuBinaryHandle = __cudaRegisterFatBinary(&FatbinWrapper);
651 llvm::CallInst *RegisterFatbinCall = CtorBuilder.CreateCall(
652 RegisterFatbinFunc,
653 CtorBuilder.CreateBitCast(FatbinWrapper, VoidPtrTy));
654 GpuBinaryHandle = new llvm::GlobalVariable(
655 TheModule, VoidPtrPtrTy, false, llvm::GlobalValue::InternalLinkage,
656 llvm::ConstantPointerNull::get(VoidPtrPtrTy), "__cuda_gpubin_handle");
657 GpuBinaryHandle->setAlignment(CGM.getPointerAlign().getQuantity());
658 CtorBuilder.CreateAlignedStore(RegisterFatbinCall, GpuBinaryHandle,
659 CGM.getPointerAlign());
660
661 // Call __cuda_register_globals(GpuBinaryHandle);
662 if (RegisterGlobalsFunc)
663 CtorBuilder.CreateCall(RegisterGlobalsFunc, RegisterFatbinCall);
664
665 // Call __cudaRegisterFatBinaryEnd(Handle) if this CUDA version needs it.
666 if (CudaFeatureEnabled(CGM.getTarget().getSDKVersion(),
667 CudaFeature::CUDA_USES_FATBIN_REGISTER_END)) {
668 // void __cudaRegisterFatBinaryEnd(void **);
669 llvm::FunctionCallee RegisterFatbinEndFunc = CGM.CreateRuntimeFunction(
670 llvm::FunctionType::get(VoidTy, VoidPtrPtrTy, false),
671 "__cudaRegisterFatBinaryEnd");
672 CtorBuilder.CreateCall(RegisterFatbinEndFunc, RegisterFatbinCall);
673 }
674 } else {
675 // Generate a unique module ID.
676 SmallString<64> ModuleID;
677 llvm::raw_svector_ostream OS(ModuleID);
678 OS << ModuleIDPrefix << llvm::format("%" PRIx64, FatbinWrapper->getGUID());
679 llvm::Constant *ModuleIDConstant =
680 makeConstantString(ModuleID.str(), "", ModuleIDSectionName, 32);
681
682 // Create an alias for the FatbinWrapper that nvcc will look for.
683 llvm::GlobalAlias::create(llvm::GlobalValue::ExternalLinkage,
684 Twine("__fatbinwrap") + ModuleID, FatbinWrapper);
685
686 // void __cudaRegisterLinkedBinary%ModuleID%(void (*)(void *), void *,
687 // void *, void (*)(void **))
688 SmallString<128> RegisterLinkedBinaryName("__cudaRegisterLinkedBinary");
689 RegisterLinkedBinaryName += ModuleID;
690 llvm::FunctionCallee RegisterLinkedBinaryFunc = CGM.CreateRuntimeFunction(
691 getRegisterLinkedBinaryFnTy(), RegisterLinkedBinaryName);
692
693 assert(RegisterGlobalsFunc && "Expecting at least dummy function!");
694 llvm::Value *Args[] = {RegisterGlobalsFunc,
695 CtorBuilder.CreateBitCast(FatbinWrapper, VoidPtrTy),
696 ModuleIDConstant,
697 makeDummyFunction(getCallbackFnTy())};
698 CtorBuilder.CreateCall(RegisterLinkedBinaryFunc, Args);
699 }
700
701 // Create destructor and register it with atexit() the way NVCC does it. Doing
702 // it during regular destructor phase worked in CUDA before 9.2 but results in
703 // double-free in 9.2.
704 if (llvm::Function *CleanupFn = makeModuleDtorFunction()) {
705 // extern "C" int atexit(void (*f)(void));
706 llvm::FunctionType *AtExitTy =
707 llvm::FunctionType::get(IntTy, CleanupFn->getType(), false);
708 llvm::FunctionCallee AtExitFunc =
709 CGM.CreateRuntimeFunction(AtExitTy, "atexit", llvm::AttributeList(),
710 /*Local=*/true);
711 CtorBuilder.CreateCall(AtExitFunc, CleanupFn);
712 }
713
714 CtorBuilder.CreateRetVoid();
715 return ModuleCtorFunc;
716}

718/// Creates a global destructor function that unregisters the GPU code blob
719/// registered by constructor.
720///
721/// For CUDA:
722/// \code
723/// void __cuda_module_dtor(void*) {
724/// __cudaUnregisterFatBinary(Handle);
725/// }
726/// \endcode
727///
728/// For HIP:
729/// \code
730/// void __hip_module_dtor(void*) {
731/// if (__hip_gpubin_handle) {
732/// __hipUnregisterFatBinary(__hip_gpubin_handle);
733/// __hip_gpubin_handle = 0;
734/// }
735/// }
736/// \endcode
737llvm::Function *CGNVCUDARuntime::makeModuleDtorFunction() {
738 // No need for destructor if we don't have a handle to unregister.
739 if (!GpuBinaryHandle)
740 return nullptr;
741
742 // void __cudaUnregisterFatBinary(void ** handle);
743 llvm::FunctionCallee UnregisterFatbinFunc = CGM.CreateRuntimeFunction(
744 llvm::FunctionType::get(VoidTy, VoidPtrPtrTy, false),
745 addUnderscoredPrefixToName("UnregisterFatBinary"));
746
747 llvm::Function *ModuleDtorFunc = llvm::Function::Create(
748 llvm::FunctionType::get(VoidTy, VoidPtrTy, false),
749 llvm::GlobalValue::InternalLinkage,
750 addUnderscoredPrefixToName("_module_dtor"), &TheModule);
751
752 llvm::BasicBlock *DtorEntryBB =
753 llvm::BasicBlock::Create(Context, "entry", ModuleDtorFunc);
754 CGBuilderTy DtorBuilder(CGM, Context);
755 DtorBuilder.SetInsertPoint(DtorEntryBB);
756
757 Address GpuBinaryAddr(GpuBinaryHandle, CharUnits::fromQuantity(
758 GpuBinaryHandle->getAlignment()));
759 auto HandleValue = DtorBuilder.CreateLoad(GpuBinaryAddr);
760 // There is only one HIP fat binary per linked module, however there are
761 // multiple destructor functions. Make sure the fat binary is unregistered
762 // only once.
763 if (CGM.getLangOpts().HIP) {
764 llvm::BasicBlock *IfBlock =
765 llvm::BasicBlock::Create(Context, "if", ModuleDtorFunc);
766 llvm::BasicBlock *ExitBlock =
767 llvm::BasicBlock::Create(Context, "exit", ModuleDtorFunc);
768 llvm::Constant *Zero = llvm::Constant::getNullValue(HandleValue->getType());
769 llvm::Value *NEZero = DtorBuilder.CreateICmpNE(HandleValue, Zero);
770 DtorBuilder.CreateCondBr(NEZero, IfBlock, ExitBlock);
771
772 DtorBuilder.SetInsertPoint(IfBlock);
773 DtorBuilder.CreateCall(UnregisterFatbinFunc, HandleValue);
774 DtorBuilder.CreateStore(Zero, GpuBinaryAddr);
775 DtorBuilder.CreateBr(ExitBlock);
776
777 DtorBuilder.SetInsertPoint(ExitBlock);
778 } else {
779 DtorBuilder.CreateCall(UnregisterFatbinFunc, HandleValue);
780 }
781 DtorBuilder.CreateRetVoid();
782 return ModuleDtorFunc;
783}
