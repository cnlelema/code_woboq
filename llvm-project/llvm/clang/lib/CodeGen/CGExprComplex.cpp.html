31/// Return the complex type that we are meant to emit.
32static const ComplexType *getComplexType(QualType type) {
33 type = type.getCanonicalType();
34 if (const ComplexType *comp = dyn_cast<ComplexType>(type)) {
35 return comp;
36 } else {
37 return cast<ComplexType>(cast<AtomicType>(type)->getValueType());
38 }
39}

346/// EmitLoadOfLValue - Given an RValue reference for a complex, emit code to
347/// load the real and imaginary pieces, returning them as Real/Imag.
348ComplexPairTy ComplexExprEmitter::EmitLoadOfLValue(LValue lvalue,
349 SourceLocation loc) {
350 assert(lvalue.isSimple() && "non-simple complex l-value?");
351 if (lvalue.getType()->isAtomicType())
352 return CGF.EmitAtomicLoad(lvalue, loc).getComplexVal();
353
354 Address SrcPtr = lvalue.getAddress();
355 bool isVolatile = lvalue.isVolatileQualified();
356
357 llvm::Value *Real = nullptr, *Imag = nullptr;
358
359 if (!IgnoreReal || isVolatile) {
360 Address RealP = CGF.emitAddrOfRealComponent(SrcPtr, lvalue.getType());
361 Real = Builder.CreateLoad(RealP, isVolatile, SrcPtr.getName() + ".real");
362 }
363
364 if (!IgnoreImag || isVolatile) {
365 Address ImagP = CGF.emitAddrOfImagComponent(SrcPtr, lvalue.getType());
366 Imag = Builder.CreateLoad(ImagP, isVolatile, SrcPtr.getName() + ".imag");
367 }
368
369 return ComplexPairTy(Real, Imag);
370}

372/// EmitStoreOfComplex - Store the specified real/imag parts into the
373/// specified value pointer.
374void ComplexExprEmitter::EmitStoreOfComplex(ComplexPairTy Val, LValue lvalue,
375 bool isInit) {
376 if (lvalue.getType()->isAtomicType() ||
377 (!isInit && CGF.LValueIsSuitableForInlineAtomic(lvalue)))
378 return CGF.EmitAtomicStore(RValue::getComplex(Val), lvalue, isInit);
379
380 Address Ptr = lvalue.getAddress();
381 Address RealPtr = CGF.emitAddrOfRealComponent(Ptr, lvalue.getType());
382 Address ImagPtr = CGF.emitAddrOfImagComponent(Ptr, lvalue.getType());
383
384 Builder.CreateStore(Val.first, RealPtr, lvalue.isVolatileQualified());
385 Builder.CreateStore(Val.second, ImagPtr, lvalue.isVolatileQualified());
386}

424/// Emit a cast from complex value Val to DestType.
425ComplexPairTy ComplexExprEmitter::EmitComplexToComplexCast(ComplexPairTy Val,
426 QualType SrcType,
427 QualType DestType,
428 SourceLocation Loc) {
429 // Get the src/dest element type.
430 SrcType = SrcType->castAs<ComplexType>()->getElementType();
431 DestType = DestType->castAs<ComplexType>()->getElementType();
432
433 // C99 6.3.1.6: When a value of complex type is converted to another
434 // complex type, both the real and imaginary parts follow the conversion
435 // rules for the corresponding real types.
436 Val.first = CGF.EmitScalarConversion(Val.first, SrcType, DestType, Loc);
437 Val.second = CGF.EmitScalarConversion(Val.second, SrcType, DestType, Loc);
438 return Val;
439}
440
441ComplexPairTy ComplexExprEmitter::EmitScalarToComplexCast(llvm::Value *Val,
442 QualType SrcType,
443 QualType DestType,
444 SourceLocation Loc) {
445 // Convert the input element to the element type of the complex.
446 DestType = DestType->castAs<ComplexType>()->getElementType();
447 Val = CGF.EmitScalarConversion(Val, SrcType, DestType, Loc);
448
449 // Return (realval, 0).
450 return ComplexPairTy(Val, llvm::Constant::getNullValue(Val->getType()));
451}

453ComplexPairTy ComplexExprEmitter::EmitCast(CastKind CK, Expr *Op,
454 QualType DestTy) {
455 switch (CK) {
456 case CK_Dependent: llvm_unreachable("dependent cast kind in IR gen!");
457
458 // Atomic to non-atomic casts may be more than a no-op for some platforms and
459 // for some types.
460 case CK_AtomicToNonAtomic:
461 case CK_NonAtomicToAtomic:
462 case CK_NoOp:
463 case CK_LValueToRValue:
464 case CK_UserDefinedConversion:
465 return Visit(Op);
466
467 case CK_LValueBitCast: {
468 LValue origLV = CGF.EmitLValue(Op);
469 Address V = origLV.getAddress();
470 V = Builder.CreateElementBitCast(V, CGF.ConvertType(DestTy));
471 return EmitLoadOfLValue(CGF.MakeAddrLValue(V, DestTy), Op->getExprLoc());
472 }
473
474 case CK_BitCast:
475 case CK_BaseToDerived:
476 case CK_DerivedToBase:
477 case CK_UncheckedDerivedToBase:
478 case CK_Dynamic:
479 case CK_ToUnion:
480 case CK_ArrayToPointerDecay:
481 case CK_FunctionToPointerDecay:
482 case CK_NullToPointer:
483 case CK_NullToMemberPointer:
484 case CK_BaseToDerivedMemberPointer:
485 case CK_DerivedToBaseMemberPointer:
486 case CK_MemberPointerToBoolean:
487 case CK_ReinterpretMemberPointer:
488 case CK_ConstructorConversion:
489 case CK_IntegralToPointer:
490 case CK_PointerToIntegral:
491 case CK_PointerToBoolean:
492 case CK_ToVoid:
493 case CK_VectorSplat:
494 case CK_IntegralCast:
495 case CK_BooleanToSignedIntegral:
496 case CK_IntegralToBoolean:
497 case CK_IntegralToFloating:
498 case CK_FloatingToIntegral:
499 case CK_FloatingToBoolean:
500 case CK_FloatingCast:
501 case CK_CPointerToObjCPointerCast:
502 case CK_BlockPointerToObjCPointerCast:
503 case CK_AnyPointerToBlockPointerCast:
504 case CK_ObjCObjectLValueCast:
505 case CK_FloatingComplexToReal:
506 case CK_FloatingComplexToBoolean:
507 case CK_IntegralComplexToReal:
508 case CK_IntegralComplexToBoolean:
509 case CK_ARCProduceObject:
510 case CK_ARCConsumeObject:
511 case CK_ARCReclaimReturnedObject:
512 case CK_ARCExtendBlockObject:
513 case CK_CopyAndAutoreleaseBlockObject:
514 case CK_BuiltinFnToFnPtr:
515 case CK_ZeroToOCLOpaqueType:
516 case CK_AddressSpaceConversion:
517 case CK_CHERICapabilityToPointer:
518 case CK_PointerToCHERICapability:
519 case CK_CHERICapabilityToOffset:
520 case CK_CHERICapabilityToAddress:
521 case CK_IntToOCLSampler:
522 case CK_FixedPointCast:
523 case CK_FixedPointToBoolean:
524 case CK_FixedPointToIntegral:
525 case CK_IntegralToFixedPoint:
526 llvm_unreachable("invalid cast kind for complex value");
527
528 case CK_FloatingRealToComplex:
529 case CK_IntegralRealToComplex:
530 return EmitScalarToComplexCast(CGF.EmitScalarExpr(Op), Op->getType(),
531 DestTy, Op->getExprLoc());
532
533 case CK_FloatingComplexCast:
534 case CK_FloatingComplexToIntegralComplex:
535 case CK_IntegralComplexCast:
536 case CK_IntegralComplexToFloatingComplex:
537 return EmitComplexToComplexCast(Visit(Op), Op->getType(), DestTy,
538 Op->getExprLoc());
539 }
540
541 llvm_unreachable("unknown cast resulting in complex value");
542}

544ComplexPairTy ComplexExprEmitter::VisitUnaryMinus(const UnaryOperator *E) {
545 TestAndClearIgnoreReal();
546 TestAndClearIgnoreImag();
547 ComplexPairTy Op = Visit(E->getSubExpr());
548
549 llvm::Value *ResR, *ResI;
550 if (Op.first->getType()->isFloatingPointTy()) {
551 ResR = Builder.CreateFNeg(Op.first, "neg.r");
552 ResI = Builder.CreateFNeg(Op.second, "neg.i");
553 } else {
554 ResR = Builder.CreateNeg(Op.first, "neg.r");
555 ResI = Builder.CreateNeg(Op.second, "neg.i");
556 }
557 return ComplexPairTy(ResR, ResI);
558}
559
560ComplexPairTy ComplexExprEmitter::VisitUnaryNot(const UnaryOperator *E) {
561 TestAndClearIgnoreReal();
562 TestAndClearIgnoreImag();
563 // ~(a+ib) = a + i*-b
564 ComplexPairTy Op = Visit(E->getSubExpr());
565 llvm::Value *ResI;
566 if (Op.second->getType()->isFloatingPointTy())
567 ResI = Builder.CreateFNeg(Op.second, "conj.i");
568 else
569 ResI = Builder.CreateNeg(Op.second, "conj.i");
570
571 return ComplexPairTy(Op.first, ResI);
572}

574ComplexPairTy ComplexExprEmitter::EmitBinAdd(const BinOpInfo &Op) {
575 llvm::Value *ResR, *ResI;
576
577 if (Op.LHS.first->getType()->isFloatingPointTy()) {
578 ResR = Builder.CreateFAdd(Op.LHS.first, Op.RHS.first, "add.r");
579 if (Op.LHS.second && Op.RHS.second)
580 ResI = Builder.CreateFAdd(Op.LHS.second, Op.RHS.second, "add.i");
581 else
582 ResI = Op.LHS.second ? Op.LHS.second : Op.RHS.second;
583 assert(ResI && "Only one operand may be real!");
584 } else {
585 ResR = Builder.CreateAdd(Op.LHS.first, Op.RHS.first, "add.r");
586 assert(Op.LHS.second && Op.RHS.second &&
587 "Both operands of integer complex operators must be complex!");
588 ResI = Builder.CreateAdd(Op.LHS.second, Op.RHS.second, "add.i");
589 }
590 return ComplexPairTy(ResR, ResI);
591}
592
593ComplexPairTy ComplexExprEmitter::EmitBinSub(const BinOpInfo &Op) {
594 llvm::Value *ResR, *ResI;
595 if (Op.LHS.first->getType()->isFloatingPointTy()) {
596 ResR = Builder.CreateFSub(Op.LHS.first, Op.RHS.first, "sub.r");
597 if (Op.LHS.second && Op.RHS.second)
598 ResI = Builder.CreateFSub(Op.LHS.second, Op.RHS.second, "sub.i");
599 else
600 ResI = Op.LHS.second ? Op.LHS.second
601 : Builder.CreateFNeg(Op.RHS.second, "sub.i");
602 assert(ResI && "Only one operand may be real!");
603 } else {
604 ResR = Builder.CreateSub(Op.LHS.first, Op.RHS.first, "sub.r");
605 assert(Op.LHS.second && Op.RHS.second &&
606 "Both operands of integer complex operators must be complex!");
607 ResI = Builder.CreateSub(Op.LHS.second, Op.RHS.second, "sub.i");
608 }
609 return ComplexPairTy(ResR, ResI);
610}

612/// Emit a libcall for a binary operation on complex types.
613ComplexPairTy ComplexExprEmitter::EmitComplexBinOpLibCall(StringRef LibCallName,
614 const BinOpInfo &Op) {
615 CallArgList Args;
616 Args.add(RValue::get(Op.LHS.first),
617 Op.Ty->castAs<ComplexType>()->getElementType());
618 Args.add(RValue::get(Op.LHS.second),
619 Op.Ty->castAs<ComplexType>()->getElementType());
620 Args.add(RValue::get(Op.RHS.first),
621 Op.Ty->castAs<ComplexType>()->getElementType());
622 Args.add(RValue::get(Op.RHS.second),
623 Op.Ty->castAs<ComplexType>()->getElementType());
624
625 // We *must* use the full CG function call building logic here because the
626 // complex type has special ABI handling. We also should not forget about
627 // special calling convention which may be used for compiler builtins.
628
629 // We create a function qualified type to state that this call does not have
630 // any exceptions.
631 FunctionProtoType::ExtProtoInfo EPI;
632 EPI = EPI.withExceptionSpec(
633 FunctionProtoType::ExceptionSpecInfo(EST_BasicNoexcept));
634 SmallVector<QualType, 4> ArgsQTys(
635 4, Op.Ty->castAs<ComplexType>()->getElementType());
636 QualType FQTy = CGF.getContext().getFunctionType(Op.Ty, ArgsQTys, EPI);
637 const CGFunctionInfo &FuncInfo = CGF.CGM.getTypes().arrangeFreeFunctionCall(
638 Args, cast<FunctionType>(FQTy.getTypePtr()), false);
639
640 llvm::FunctionType *FTy = CGF.CGM.getTypes().GetFunctionType(FuncInfo);
641 llvm::FunctionCallee Func = CGF.CGM.CreateRuntimeFunction(
642 FTy, LibCallName, llvm::AttributeList(), true);
643 CGCallee Callee = CGCallee::forDirect(Func, FQTy->getAs<FunctionProtoType>());
644
645 llvm::CallBase *Call;
646 RValue Res = CGF.EmitCall(FuncInfo, Callee, ReturnValueSlot(), Args, &Call);
647 Call->setCallingConv(CGF.CGM.getRuntimeCC());
648 return Res.getComplexVal();
649}

652/// multiply.
653static StringRef getComplexMultiplyLibCallName(llvm::Type *Ty) {
654 switch (Ty->getTypeID()) {
655 default:
656 llvm_unreachable("Unsupported floating point type!");
657 case llvm::Type::HalfTyID:
658 return "__mulhc3";
659 case llvm::Type::FloatTyID:
660 return "__mulsc3";
661 case llvm::Type::DoubleTyID:
662 return "__muldc3";
663 case llvm::Type::PPC_FP128TyID:
664 return "__multc3";
665 case llvm::Type::X86_FP80TyID:
666 return "__mulxc3";
667 case llvm::Type::FP128TyID:
668 return "__multc3";
669 }

672// See C11 Annex G.5.1 for the semantics of multiplicative operators on complex
673// typed values.
674ComplexPairTy ComplexExprEmitter::EmitBinMul(const BinOpInfo &Op) {
675 using llvm::Value;
676 Value *ResR, *ResI;
677 llvm::MDBuilder MDHelper(CGF.getLLVMContext());
678
679 if (Op.LHS.first->getType()->isFloatingPointTy()) {
680 // The general formulation is:
681 // (a + ib) * (c + id) = (a * c - b * d) + i(a * d + b * c)
682 //
683 // But we can fold away components which would be zero due to a real
684 // operand according to C11 Annex G.5.1p2.
685 // FIXME: C11 also provides for imaginary types which would allow folding
686 // still more of this within the type system.
687
688 if (Op.LHS.second && Op.RHS.second) {
689 // If both operands are complex, emit the core math directly, and then
690 // test for NaNs. If we find NaNs in the result, we delegate to a libcall
691 // to carefully re-compute the correct infinity representation if
692 // possible. The expectation is that the presence of NaNs here is
693 // *extremely* rare, and so the cost of the libcall is almost irrelevant.
694 // This is good, because the libcall re-computes the core multiplication
695 // exactly the same as we do here and re-tests for NaNs in order to be
696 // a generic complex*complex libcall.
697
698 // First compute the four products.
699 Value *AC = Builder.CreateFMul(Op.LHS.first, Op.RHS.first, "mul_ac");
700 Value *BD = Builder.CreateFMul(Op.LHS.second, Op.RHS.second, "mul_bd");
701 Value *AD = Builder.CreateFMul(Op.LHS.first, Op.RHS.second, "mul_ad");
702 Value *BC = Builder.CreateFMul(Op.LHS.second, Op.RHS.first, "mul_bc");
703
704 // The real part is the difference of the first two, the imaginary part is
705 // the sum of the second.
706 ResR = Builder.CreateFSub(AC, BD, "mul_r");
707 ResI = Builder.CreateFAdd(AD, BC, "mul_i");
708
709 // Emit the test for the real part becoming NaN and create a branch to
710 // handle it. We test for NaN by comparing the number to itself.
711 Value *IsRNaN = Builder.CreateFCmpUNO(ResR, ResR, "isnan_cmp");
712 llvm::BasicBlock *ContBB = CGF.createBasicBlock("complex_mul_cont");
713 llvm::BasicBlock *INaNBB = CGF.createBasicBlock("complex_mul_imag_nan");
714 llvm::Instruction *Branch = Builder.CreateCondBr(IsRNaN, INaNBB, ContBB);
715 llvm::BasicBlock *OrigBB = Branch->getParent();
716
717 // Give hint that we very much don't expect to see NaNs.
718 // Value chosen to match UR_NONTAKEN_WEIGHT, see BranchProbabilityInfo.cpp
719 llvm::MDNode *BrWeight = MDHelper.createBranchWeights(1, (1U << 20) - 1);
720 Branch->setMetadata(llvm::LLVMContext::MD_prof, BrWeight);
721
722 // Now test the imaginary part and create its branch.
723 CGF.EmitBlock(INaNBB);
724 Value *IsINaN = Builder.CreateFCmpUNO(ResI, ResI, "isnan_cmp");
725 llvm::BasicBlock *LibCallBB = CGF.createBasicBlock("complex_mul_libcall");
726 Branch = Builder.CreateCondBr(IsINaN, LibCallBB, ContBB);
727 Branch->setMetadata(llvm::LLVMContext::MD_prof, BrWeight);
728
729 // Now emit the libcall on this slowest of the slow paths.
730 CGF.EmitBlock(LibCallBB);
731 Value *LibCallR, *LibCallI;
732 std::tie(LibCallR, LibCallI) = EmitComplexBinOpLibCall(
733 getComplexMultiplyLibCallName(Op.LHS.first->getType()), Op);
734 Builder.CreateBr(ContBB);
735
736 // Finally continue execution by phi-ing together the different
737 // computation paths.
738 CGF.EmitBlock(ContBB);
739 llvm::PHINode *RealPHI = Builder.CreatePHI(ResR->getType(), 3, "real_mul_phi");
740 RealPHI->addIncoming(ResR, OrigBB);
741 RealPHI->addIncoming(ResR, INaNBB);
742 RealPHI->addIncoming(LibCallR, LibCallBB);
743 llvm::PHINode *ImagPHI = Builder.CreatePHI(ResI->getType(), 3, "imag_mul_phi");
744 ImagPHI->addIncoming(ResI, OrigBB);
745 ImagPHI->addIncoming(ResI, INaNBB);
746 ImagPHI->addIncoming(LibCallI, LibCallBB);
747 return ComplexPairTy(RealPHI, ImagPHI);
748 }
749 assert((Op.LHS.second || Op.RHS.second) &&
750 "At least one operand must be complex!");
751
752 // If either of the operands is a real rather than a complex, the
753 // imaginary component is ignored when computing the real component of the
754 // result.
755 ResR = Builder.CreateFMul(Op.LHS.first, Op.RHS.first, "mul.rl");
756
757 ResI = Op.LHS.second
758 ? Builder.CreateFMul(Op.LHS.second, Op.RHS.first, "mul.il")
759 : Builder.CreateFMul(Op.LHS.first, Op.RHS.second, "mul.ir");
760 } else {
761 assert(Op.LHS.second && Op.RHS.second &&
762 "Both operands of integer complex operators must be complex!");
763 Value *ResRl = Builder.CreateMul(Op.LHS.first, Op.RHS.first, "mul.rl");
764 Value *ResRr = Builder.CreateMul(Op.LHS.second, Op.RHS.second, "mul.rr");
765 ResR = Builder.CreateSub(ResRl, ResRr, "mul.r");
766
767 Value *ResIl = Builder.CreateMul(Op.LHS.second, Op.RHS.first, "mul.il");
768 Value *ResIr = Builder.CreateMul(Op.LHS.first, Op.RHS.second, "mul.ir");
769 ResI = Builder.CreateAdd(ResIl, ResIr, "mul.i");
770 }
771 return ComplexPairTy(ResR, ResI);
772}

774// See C11 Annex G.5.1 for the semantics of multiplicative operators on complex
775// typed values.
776ComplexPairTy ComplexExprEmitter::EmitBinDiv(const BinOpInfo &Op) {
777 llvm::Value *LHSr = Op.LHS.first, *LHSi = Op.LHS.second;
778 llvm::Value *RHSr = Op.RHS.first, *RHSi = Op.RHS.second;
779
780 llvm::Value *DSTr, *DSTi;
781 if (LHSr->getType()->isFloatingPointTy()) {
782 // If we have a complex operand on the RHS and FastMath is not allowed, we
783 // delegate to a libcall to handle all of the complexities and minimize
784 // underflow/overflow cases. When FastMath is allowed we construct the
785 // divide inline using the same algorithm as for integer operands.
786 //
787 // FIXME: We would be able to avoid the libcall in many places if we
788 // supported imaginary types in addition to complex types.
789 if (RHSi && !CGF.getLangOpts().FastMath) {
790 BinOpInfo LibCallOp = Op;
791 // If LHS was a real, supply a null imaginary part.
792 if (!LHSi)
793 LibCallOp.LHS.second = llvm::Constant::getNullValue(LHSr->getType());
794
795 switch (LHSr->getType()->getTypeID()) {
796 default:
797 llvm_unreachable("Unsupported floating point type!");
798 case llvm::Type::HalfTyID:
799 return EmitComplexBinOpLibCall("__divhc3", LibCallOp);
800 case llvm::Type::FloatTyID:
801 return EmitComplexBinOpLibCall("__divsc3", LibCallOp);
802 case llvm::Type::DoubleTyID:
803 return EmitComplexBinOpLibCall("__divdc3", LibCallOp);
804 case llvm::Type::PPC_FP128TyID:
805 return EmitComplexBinOpLibCall("__divtc3", LibCallOp);
806 case llvm::Type::X86_FP80TyID:
807 return EmitComplexBinOpLibCall("__divxc3", LibCallOp);
808 case llvm::Type::FP128TyID:
809 return EmitComplexBinOpLibCall("__divtc3", LibCallOp);
810 }
811 } else if (RHSi) {
812 if (!LHSi)
813 LHSi = llvm::Constant::getNullValue(RHSi->getType());
814
815 // (a+ib) / (c+id) = ((ac+bd)/(cc+dd)) + i((bc-ad)/(cc+dd))
816 llvm::Value *AC = Builder.CreateFMul(LHSr, RHSr); // a*c
817 llvm::Value *BD = Builder.CreateFMul(LHSi, RHSi); // b*d
818 llvm::Value *ACpBD = Builder.CreateFAdd(AC, BD); // ac+bd
819
820 llvm::Value *CC = Builder.CreateFMul(RHSr, RHSr); // c*c
821 llvm::Value *DD = Builder.CreateFMul(RHSi, RHSi); // d*d
822 llvm::Value *CCpDD = Builder.CreateFAdd(CC, DD); // cc+dd
823
824 llvm::Value *BC = Builder.CreateFMul(LHSi, RHSr); // b*c
825 llvm::Value *AD = Builder.CreateFMul(LHSr, RHSi); // a*d
826 llvm::Value *BCmAD = Builder.CreateFSub(BC, AD); // bc-ad
827
828 DSTr = Builder.CreateFDiv(ACpBD, CCpDD);
829 DSTi = Builder.CreateFDiv(BCmAD, CCpDD);
830 } else {
831 assert(LHSi && "Can have at most one non-complex operand!");
832
833 DSTr = Builder.CreateFDiv(LHSr, RHSr);
834 DSTi = Builder.CreateFDiv(LHSi, RHSr);
835 }
836 } else {
837 assert(Op.LHS.second && Op.RHS.second &&
838 "Both operands of integer complex operators must be complex!");
839 // (a+ib) / (c+id) = ((ac+bd)/(cc+dd)) + i((bc-ad)/(cc+dd))
840 llvm::Value *Tmp1 = Builder.CreateMul(LHSr, RHSr); // a*c
841 llvm::Value *Tmp2 = Builder.CreateMul(LHSi, RHSi); // b*d
842 llvm::Value *Tmp3 = Builder.CreateAdd(Tmp1, Tmp2); // ac+bd
843
844 llvm::Value *Tmp4 = Builder.CreateMul(RHSr, RHSr); // c*c
845 llvm::Value *Tmp5 = Builder.CreateMul(RHSi, RHSi); // d*d
846 llvm::Value *Tmp6 = Builder.CreateAdd(Tmp4, Tmp5); // cc+dd
847
848 llvm::Value *Tmp7 = Builder.CreateMul(LHSi, RHSr); // b*c
849 llvm::Value *Tmp8 = Builder.CreateMul(LHSr, RHSi); // a*d
850 llvm::Value *Tmp9 = Builder.CreateSub(Tmp7, Tmp8); // bc-ad
851
852 if (Op.Ty->castAs<ComplexType>()->getElementType()->isUnsignedIntegerType()) {
853 DSTr = Builder.CreateUDiv(Tmp3, Tmp6);
854 DSTi = Builder.CreateUDiv(Tmp9, Tmp6);
855 } else {
856 DSTr = Builder.CreateSDiv(Tmp3, Tmp6);
857 DSTi = Builder.CreateSDiv(Tmp9, Tmp6);
858 }
859 }
860
861 return ComplexPairTy(DSTr, DSTi);
862}

864ComplexExprEmitter::BinOpInfo
865ComplexExprEmitter::EmitBinOps(const BinaryOperator *E) {
866 TestAndClearIgnoreReal();
867 TestAndClearIgnoreImag();
868 BinOpInfo Ops;
869 if (E->getLHS()->getType()->isRealFloatingType())
870 Ops.LHS = ComplexPairTy(CGF.EmitScalarExpr(E->getLHS()), nullptr);
871 else
872 Ops.LHS = Visit(E->getLHS());
873 if (E->getRHS()->getType()->isRealFloatingType())
874 Ops.RHS = ComplexPairTy(CGF.EmitScalarExpr(E->getRHS()), nullptr);
875 else
876 Ops.RHS = Visit(E->getRHS());
877
878 Ops.Ty = E->getType();
879 return Ops;

883LValue ComplexExprEmitter::
884EmitCompoundAssignLValue(const CompoundAssignOperator *E,
885 ComplexPairTy (ComplexExprEmitter::*Func)(const BinOpInfo&),
886 RValue &Val) {
887 TestAndClearIgnoreReal();
888 TestAndClearIgnoreImag();
889 QualType LHSTy = E->getLHS()->getType();
890 if (const AtomicType *AT = LHSTy->getAs<AtomicType>())
891 LHSTy = AT->getValueType();
892
893 BinOpInfo OpInfo;
894
895 // Load the RHS and LHS operands.
896 // __block variables need to have the rhs evaluated first, plus this should
897 // improve codegen a little.
898 OpInfo.Ty = E->getComputationResultType();
899 QualType ComplexElementTy = cast<ComplexType>(OpInfo.Ty)->getElementType();
900
901 // The RHS should have been converted to the computation type.
902 if (E->getRHS()->getType()->isRealFloatingType()) {
903 assert(
904 CGF.getContext()
905 .hasSameUnqualifiedType(ComplexElementTy, E->getRHS()->getType()));
906 OpInfo.RHS = ComplexPairTy(CGF.EmitScalarExpr(E->getRHS()), nullptr);
907 } else {
908 assert(CGF.getContext()
909 .hasSameUnqualifiedType(OpInfo.Ty, E->getRHS()->getType()));
910 OpInfo.RHS = Visit(E->getRHS());
911 }
912
913 LValue LHS = CGF.EmitLValue(E->getLHS());
914
915 // Load from the l-value and convert it.
916 SourceLocation Loc = E->getExprLoc();
917 if (LHSTy->isAnyComplexType()) {
918 ComplexPairTy LHSVal = EmitLoadOfLValue(LHS, Loc);
919 OpInfo.LHS = EmitComplexToComplexCast(LHSVal, LHSTy, OpInfo.Ty, Loc);
920 } else {
921 llvm::Value *LHSVal = CGF.EmitLoadOfScalar(LHS, Loc);
922 // For floating point real operands we can directly pass the scalar form
923 // to the binary operator emission and potentially get more efficient code.
924 if (LHSTy->isRealFloatingType()) {
925 if (!CGF.getContext().hasSameUnqualifiedType(ComplexElementTy, LHSTy))
926 LHSVal = CGF.EmitScalarConversion(LHSVal, LHSTy, ComplexElementTy, Loc);
927 OpInfo.LHS = ComplexPairTy(LHSVal, nullptr);
928 } else {
929 OpInfo.LHS = EmitScalarToComplexCast(LHSVal, LHSTy, OpInfo.Ty, Loc);
930 }
931 }
932
933 // Expand the binary operator.
934 ComplexPairTy Result = (this->*Func)(OpInfo);
935
936 // Truncate the result and store it into the LHS lvalue.
937 if (LHSTy->isAnyComplexType()) {
938 ComplexPairTy ResVal =
939 EmitComplexToComplexCast(Result, OpInfo.Ty, LHSTy, Loc);
940 EmitStoreOfComplex(ResVal, LHS, /*isInit*/ false);
941 Val = RValue::getComplex(ResVal);
942 } else {
943 llvm::Value *ResVal =
944 CGF.EmitComplexToScalarConversion(Result, OpInfo.Ty, LHSTy, Loc);
945 CGF.EmitStoreOfScalar(ResVal, LHS, /*isInit*/ false);
946 Val = RValue::get(ResVal);
947 }
948
949 return LHS;
950}

952// Compound assignments.
953ComplexPairTy ComplexExprEmitter::
954EmitCompoundAssign(const CompoundAssignOperator *E,
955 ComplexPairTy (ComplexExprEmitter::*Func)(const BinOpInfo&)){
956 RValue Val;
957 LValue LV = EmitCompoundAssignLValue(E, Func, Val);
958
959 // The result of an assignment in C is the assigned r-value.
960 if (!CGF.getLangOpts().CPlusPlus)
961 return Val.getComplexVal();
962
963 // If the lvalue is non-volatile, return the computed value of the assignment.
964 if (!LV.isVolatileQualified())
965 return Val.getComplexVal();
966
967 return EmitLoadOfLValue(LV, E->getExprLoc());
968}

970LValue ComplexExprEmitter::EmitBinAssignLValue(const BinaryOperator *E,
971 ComplexPairTy &Val) {
972 assert(CGF.getContext().hasSameUnqualifiedType(E->getLHS()->getType(),
973 E->getRHS()->getType()) &&
974 "Invalid assignment");
975 TestAndClearIgnoreReal();
976 TestAndClearIgnoreImag();
977
978 // Emit the RHS. __block variables need the RHS evaluated first.
979 Val = Visit(E->getRHS());
980
981 // Compute the address to store into.
982 LValue LHS = CGF.EmitLValue(E->getLHS());
983
984 // Store the result value into the LHS lvalue.
985 EmitStoreOfComplex(Val, LHS, /*isInit*/ false);
986
987 return LHS;
988}

1010ComplexPairTy ComplexExprEmitter::
1011VisitAbstractConditionalOperator(const AbstractConditionalOperator *E) {
1012 TestAndClearIgnoreReal();
1013 TestAndClearIgnoreImag();
1014 llvm::BasicBlock *LHSBlock = CGF.createBasicBlock("cond.true");
1015 llvm::BasicBlock *RHSBlock = CGF.createBasicBlock("cond.false");
1016 llvm::BasicBlock *ContBlock = CGF.createBasicBlock("cond.end");
1017
1018 // Bind the common expression if necessary.
1019 CodeGenFunction::OpaqueValueMapping binding(CGF, E);
1020
1021
1022 CodeGenFunction::ConditionalEvaluation eval(CGF);
1023 CGF.EmitBranchOnBoolExpr(E->getCond(), LHSBlock, RHSBlock,
1024 CGF.getProfileCount(E));
1025
1026 eval.begin(CGF);
1027 CGF.EmitBlock(LHSBlock);
1028 CGF.incrementProfileCounter(E);
1029 ComplexPairTy LHS = Visit(E->getTrueExpr());
1030 LHSBlock = Builder.GetInsertBlock();
1031 CGF.EmitBranch(ContBlock);
1032 eval.end(CGF);
1033
1034 eval.begin(CGF);
1035 CGF.EmitBlock(RHSBlock);
1036 ComplexPairTy RHS = Visit(E->getFalseExpr());
1037 RHSBlock = Builder.GetInsertBlock();
1038 CGF.EmitBlock(ContBlock);
1039 eval.end(CGF);
1040
1041 // Create a PHI node for the real part.
1042 llvm::PHINode *RealPN = Builder.CreatePHI(LHS.first->getType(), 2, "cond.r");
1043 RealPN->addIncoming(LHS.first, LHSBlock);
1044 RealPN->addIncoming(RHS.first, RHSBlock);
1045
1046 // Create a PHI node for the imaginary part.
1047 llvm::PHINode *ImagPN = Builder.CreatePHI(LHS.first->getType(), 2, "cond.i");
1048 ImagPN->addIncoming(LHS.second, LHSBlock);
1049 ImagPN->addIncoming(RHS.second, RHSBlock);
1050
1051 return ComplexPairTy(RealPN, ImagPN);
1052}

1062 Ignore = TestAndClearIgnoreImag();
1063 (void)Ignore;
