35static MemberCallInfo
36commonEmitCXXMemberOrOperatorCall(CodeGenFunction &CGF, const CXXMethodDecl *MD,
37 llvm::Value *This, llvm::Value *ImplicitParam,
38 QualType ImplicitParamTy, const CallExpr *CE,
39 CallArgList &Args, CallArgList *RtlArgs) {
40 assert(CE == nullptr || isa<CXXMemberCallExpr>(CE) ||
41 isa<CXXOperatorCallExpr>(CE));
42 assert(MD->isInstance() &&
43 "Trying to emit a member or operator call expr on a static method!");
44
45 // Push the this ptr.
46 const CXXRecordDecl *RD =
47 CGF.CGM.getCXXABI().getThisArgumentTypeForMethod(MD);
48 Args.add(RValue::get(This), CGF.getTypes().DeriveThisType(RD, MD));
49
50 // If there is an implicit parameter (e.g. VTT), emit it.
51 if (ImplicitParam) {
52 Args.add(RValue::get(ImplicitParam), ImplicitParamTy);
53 }
54
55 const FunctionProtoType *FPT = MD->getType()->castAs<FunctionProtoType>();
56 RequiredArgs required = RequiredArgs::forPrototypePlus(FPT, Args.size());
57 unsigned PrefixSize = Args.size() - 1;
58
59 // And the rest of the call args.
60 if (RtlArgs) {
61 // Special case: if the caller emitted the arguments right-to-left already
62 // (prior to emitting the *this argument), we're done. This happens for
63 // assignment operators.
64 Args.addFrom(*RtlArgs);
65 } else if (CE) {
66 // Special case: skip first argument of CXXOperatorCall (it is "this").
67 unsigned ArgsToSkip = isa<CXXOperatorCallExpr>(CE) ? 1 : 0;
68 CGF.EmitCallArgs(Args, FPT, drop_begin(CE->arguments(), ArgsToSkip),
69 CE->getDirectCallee());
70 } else {
71 assert(
72 FPT->getNumParams() == 0 &&
73 "No CallExpr specified for function with non-zero number of arguments");
74 }
75 return {required, PrefixSize};
76}

78RValue CodeGenFunction::EmitCXXMemberOrOperatorCall(
79 const CXXMethodDecl *MD, const CGCallee &Callee,
80 ReturnValueSlot ReturnValue,
81 llvm::Value *This, llvm::Value *ImplicitParam, QualType ImplicitParamTy,
82 const CallExpr *CE, CallArgList *RtlArgs) {
83 const FunctionProtoType *FPT = MD->getType()->castAs<FunctionProtoType>();
84 CallArgList Args;
85 MemberCallInfo CallInfo = commonEmitCXXMemberOrOperatorCall(
86 *this, MD, This, ImplicitParam, ImplicitParamTy, CE, Args, RtlArgs);
87 auto &FnInfo = CGM.getTypes().arrangeCXXMethodCall(
88 Args, FPT, CallInfo.ReqArgs, CallInfo.PrefixSize);
89 return EmitCall(FnInfo, Callee, ReturnValue, Args, nullptr,
90 CE ? CE->getExprLoc() : SourceLocation());

104RValue CodeGenFunction::EmitCXXPseudoDestructorExpr(
105 const CXXPseudoDestructorExpr *E) {
106 QualType DestroyedType = E->getDestroyedType();
107 if (DestroyedType.hasStrongOrWeakObjCLifetime()) {
108 // Automatic Reference Counting:
109 // If the pseudo-expression names a retainable object with weak or
110 // strong lifetime, the object shall be released.
111 Expr *BaseExpr = E->getBase();
112 Address BaseValue = Address::invalid();
113 Qualifiers BaseQuals;
114
115 // If this is s.x, emit s as an lvalue. If it is s->x, emit s as a scalar.
116 if (E->isArrow()) {
117 BaseValue = EmitPointerWithAlignment(BaseExpr);
118 const PointerType *PTy = BaseExpr->getType()->getAs<PointerType>();
119 BaseQuals = PTy->getPointeeType().getQualifiers();
120 } else {
121 LValue BaseLV = EmitLValue(BaseExpr);
122 BaseValue = BaseLV.getAddress();
123 QualType BaseTy = BaseExpr->getType();
124 BaseQuals = BaseTy.getQualifiers();
125 }
126
127 switch (DestroyedType.getObjCLifetime()) {
128 case Qualifiers::OCL_None:
129 case Qualifiers::OCL_ExplicitNone:
130 case Qualifiers::OCL_Autoreleasing:
131 break;
132
133 case Qualifiers::OCL_Strong:
134 EmitARCRelease(Builder.CreateLoad(BaseValue,
135 DestroyedType.isVolatileQualified()),
136 ARCPreciseLifetime);
137 break;
138
139 case Qualifiers::OCL_Weak:
140 EmitARCDestroyWeak(BaseValue);
141 break;
142 }
143 } else {
144 // C++ [expr.pseudo]p1:
145 // The result shall only be used as the operand for the function call
146 // operator (), and the result of such a call has type void. The only
147 // effect is the evaluation of the postfix-expression before the dot or
148 // arrow.
149 EmitIgnoredExpr(E->getBase());
150 }
151
152 return RValue::get(nullptr);
153}

155static CXXRecordDecl *getCXXRecord(const Expr *E) {
156 QualType T = E->getType();
157 if (const PointerType *PTy = T->getAs<PointerType>())
158 T = PTy->getPointeeType();
159 const RecordType *Ty = T->castAs<RecordType>();
160 return cast<CXXRecordDecl>(Ty->getDecl());
161}

163// Note: This function also emit constructor calls to support a MSVC
164// extensions allowing explicit constructor function call.
165RValue CodeGenFunction::EmitCXXMemberCallExpr(const CXXMemberCallExpr *CE,
166 ReturnValueSlot ReturnValue) {
167 const Expr *callee = CE->getCallee()->IgnoreParens();
168
169 if (isa<BinaryOperator>(callee))
170 return EmitCXXMemberPointerCallExpr(CE, ReturnValue);
171
172 const MemberExpr *ME = cast<MemberExpr>(callee);
173 const CXXMethodDecl *MD = cast<CXXMethodDecl>(ME->getMemberDecl());
174
175 if (MD->isStatic()) {
176 // The method is static, emit it as we would a regular call.
177 CGCallee callee =
178 CGCallee::forDirect(CGM.GetAddrOfFunction(MD), GlobalDecl(MD));
179 return EmitCall(getContext().getPointerType(MD->getType()), callee, CE,
180 ReturnValue);
181 }
182
183 bool HasQualifier = ME->hasQualifier();
184 NestedNameSpecifier *Qualifier = HasQualifier ? ME->getQualifier() : nullptr;
185 bool IsArrow = ME->isArrow();
186 const Expr *Base = ME->getBase();
187
188 return EmitCXXMemberOrOperatorMemberCallExpr(
189 CE, MD, ReturnValue, HasQualifier, Qualifier, IsArrow, Base);
190}

192RValue CodeGenFunction::EmitCXXMemberOrOperatorMemberCallExpr(
193 const CallExpr *CE, const CXXMethodDecl *MD, ReturnValueSlot ReturnValue,
194 bool HasQualifier, NestedNameSpecifier *Qualifier, bool IsArrow,
195 const Expr *Base) {
196 assert(isa<CXXMemberCallExpr>(CE) || isa<CXXOperatorCallExpr>(CE));
197
198 // Compute the object pointer.
199 bool CanUseVirtualCall = MD->isVirtual() && !HasQualifier;
200
201 const CXXMethodDecl *DevirtualizedMethod = nullptr;
202 if (CanUseVirtualCall &&
203 MD->getDevirtualizedMethod(Base, getLangOpts().AppleKext)) {
204 const CXXRecordDecl *BestDynamicDecl = Base->getBestDynamicClassType();
205 DevirtualizedMethod = MD->getCorrespondingMethodInClass(BestDynamicDecl);
206 assert(DevirtualizedMethod);
207 const CXXRecordDecl *DevirtualizedClass = DevirtualizedMethod->getParent();
208 const Expr *Inner = Base->ignoreParenBaseCasts();
209 if (DevirtualizedMethod->getReturnType().getCanonicalType() !=
210 MD->getReturnType().getCanonicalType())
211 // If the return types are not the same, this might be a case where more
212 // code needs to run to compensate for it. For example, the derived
213 // method might return a type that inherits form from the return
214 // type of MD and has a prefix.
215 // For now we just avoid devirtualizing these covariant cases.
216 DevirtualizedMethod = nullptr;
217 else if (getCXXRecord(Inner) == DevirtualizedClass)
218 // If the class of the Inner expression is where the dynamic method
219 // is defined, build the this pointer from it.
220 Base = Inner;
221 else if (getCXXRecord(Base) != DevirtualizedClass) {
222 // If the method is defined in a class that is not the best dynamic
223 // one or the one of the full expression, we would have to build
224 // a derived-to-base cast to compute the correct this pointer, but
225 // we don't have support for that yet, so do a virtual call.
226 DevirtualizedMethod = nullptr;
227 }
228 }
229
230 // C++17 demands that we evaluate the RHS of a (possibly-compound) assignment
231 // operator before the LHS.
232 CallArgList RtlArgStorage;
233 CallArgList *RtlArgs = nullptr;
234 if (auto *OCE = dyn_cast<CXXOperatorCallExpr>(CE)) {
235 if (OCE->isAssignmentOp()) {
236 RtlArgs = &RtlArgStorage;
237 EmitCallArgs(*RtlArgs, MD->getType()->castAs<FunctionProtoType>(),
238 drop_begin(CE->arguments(), 1), CE->getDirectCallee(),
239 /*ParamsToSkip*/0, EvaluationOrder::ForceRightToLeft);
240 }
241 }
242
243 LValue This;
244 if (IsArrow) {
245 LValueBaseInfo BaseInfo;
246 TBAAAccessInfo TBAAInfo;
247 Address ThisValue = EmitPointerWithAlignment(Base, &BaseInfo, &TBAAInfo);
248 This = MakeAddrLValue(ThisValue, Base->getType(), BaseInfo, TBAAInfo);
249 } else {
250 This = EmitLValue(Base);
251 }
252
253 if (const CXXConstructorDecl *Ctor = dyn_cast<CXXConstructorDecl>(MD)) {
254 // This is the MSVC p->Ctor::Ctor(...) extension. We assume that's
255 // constructing a new complete object of type Ctor.
256 assert(!RtlArgs);
257 assert(ReturnValue.isNull() && "Constructor shouldn't have return value");
258 CallArgList Args;
259 commonEmitCXXMemberOrOperatorCall(
260 *this, Ctor, This.getPointer(), /*ImplicitParam=*/nullptr,
261 /*ImplicitParamTy=*/QualType(), CE, Args, nullptr);
262
263 EmitCXXConstructorCall(Ctor, Ctor_Complete, /*ForVirtualBase=*/false,
264 /*Delegating=*/false, This.getAddress(), Args,
265 AggValueSlot::DoesNotOverlap, CE->getExprLoc(),
266 /*NewPointerIsChecked=*/false);
267 return RValue::get(nullptr);
268 }
269
270 if (MD->isTrivial() || (MD->isDefaulted() && MD->getParent()->isUnion())) {
271 if (isa<CXXDestructorDecl>(MD)) return RValue::get(nullptr);
272 if (!MD->getParent()->mayInsertExtraPadding()) {
273 if (MD->isCopyAssignmentOperator() || MD->isMoveAssignmentOperator()) {
274 // We don't like to generate the trivial copy/move assignment operator
275 // when it isn't necessary; just produce the proper effect here.
276 LValue RHS = isa<CXXOperatorCallExpr>(CE)
277 ? MakeNaturalAlignAddrLValue(
278 (*RtlArgs)[0].getRValue(*this).getScalarVal(),
279 (*(CE->arg_begin() + 1))->getType(),
280 (*RtlArgs)[0].getRValue(*this).getAlignment())
281 : EmitLValue(*CE->arg_begin());
282 EmitAggregateAssign(This, RHS, CE->getType());
283 return RValue::get(This.getPointer(),
284 This.getAlignment().getQuantity());
285 }
286 llvm_unreachable("unknown trivial member function");
287 }
288 }
289
290 // Compute the function type we're calling.
291 const CXXMethodDecl *CalleeDecl =
292 DevirtualizedMethod ? DevirtualizedMethod : MD;
293 const CGFunctionInfo *FInfo = nullptr;
294 if (const auto *Dtor = dyn_cast<CXXDestructorDecl>(CalleeDecl))
295 FInfo = &CGM.getTypes().arrangeCXXStructorDeclaration(
296 GlobalDecl(Dtor, Dtor_Complete));
297 else
298 FInfo = &CGM.getTypes().arrangeCXXMethodDeclaration(CalleeDecl);
299
300 llvm::FunctionType *Ty = CGM.getTypes().GetFunctionType(*FInfo);
301
302 // C++11 [class.mfct.non-static]p2:
303 // If a non-static member function of a class X is called for an object that
304 // is not of type X, or of a type derived from X, the behavior is undefined.
305 SourceLocation CallLoc;
306 ASTContext &C = getContext();
307 if (CE)
308 CallLoc = CE->getExprLoc();
309
310 SanitizerSet SkippedChecks;
311 if (const auto *CMCE = dyn_cast<CXXMemberCallExpr>(CE)) {
312 auto *IOA = CMCE->getImplicitObjectArgument();
313 bool IsImplicitObjectCXXThis = IsWrappedCXXThis(IOA);
314 if (IsImplicitObjectCXXThis)
315 SkippedChecks.set(SanitizerKind::Alignment, true);
316 if (IsImplicitObjectCXXThis || isa<DeclRefExpr>(IOA))
317 SkippedChecks.set(SanitizerKind::Null, true);
318 }
319 EmitTypeCheck(CodeGenFunction::TCK_MemberCall, CallLoc, This.getPointer(),
320 C.getRecordType(CalleeDecl->getParent()),
321 /*Alignment=*/CharUnits::Zero(), SkippedChecks);
322
323 // C++ [class.virtual]p12:
324 // Explicit qualification with the scope operator (5.1) suppresses the
325 // virtual call mechanism.
326 //
327 // We also don't emit a virtual call if the base expression has a record type
328 // because then we know what the type is.
329 bool UseVirtualCall = CanUseVirtualCall && !DevirtualizedMethod;
330
331 if (const CXXDestructorDecl *Dtor = dyn_cast<CXXDestructorDecl>(CalleeDecl)) {
332 assert(CE->arg_begin() == CE->arg_end() &&
333 "Destructor shouldn't have explicit parameters");
334 assert(ReturnValue.isNull() && "Destructor shouldn't have return value");
335 if (UseVirtualCall) {
336 CGM.getCXXABI().EmitVirtualDestructorCall(
337 *this, Dtor, Dtor_Complete, This.getAddress(),
338 cast<CXXMemberCallExpr>(CE));
339 } else {
340 GlobalDecl GD(Dtor, Dtor_Complete);
341 CGCallee Callee;
342 if (getLangOpts().AppleKext && Dtor->isVirtual() && HasQualifier)
343 Callee = BuildAppleKextVirtualCall(Dtor, Qualifier, Ty);
344 else if (!DevirtualizedMethod)
345 Callee =
346 CGCallee::forDirect(CGM.getAddrOfCXXStructor(GD, FInfo, Ty), GD);
347 else {
348 Callee = CGCallee::forDirect(CGM.GetAddrOfFunction(GD, Ty), GD);
349 }
350
351 EmitCXXDestructorCall(GD, Callee, This.getPointer(),
352 /*ImplicitParam=*/nullptr,
353 /*ImplicitParamTy=*/QualType(), nullptr);
354 }
355 return RValue::get(nullptr);
356 }
357
358 // FIXME: Uses of 'MD' past this point need to be audited. We may need to use
359 // 'CalleeDecl' instead.
360
361 CGCallee Callee;
362 if (UseVirtualCall) {
363 Callee = CGCallee::forVirtual(CE, MD, This.getAddress(), Ty);
364 } else {
365 if (SanOpts.has(SanitizerKind::CFINVCall) &&
366 MD->getParent()->isDynamicClass()) {
367 llvm::Value *VTable;
368 const CXXRecordDecl *RD;
369 std::tie(VTable, RD) =
370 CGM.getCXXABI().LoadVTablePtr(*this, This.getAddress(),
371 MD->getParent());
372 EmitVTablePtrCheckForCall(RD, VTable, CFITCK_NVCall, CE->getBeginLoc());
373 }
374
375 if (getLangOpts().AppleKext && MD->isVirtual() && HasQualifier)
376 Callee = BuildAppleKextVirtualCall(MD, Qualifier, Ty);
377 else if (!DevirtualizedMethod)
378 Callee =
379 CGCallee::forDirect(CGM.GetAddrOfFunction(MD, Ty), GlobalDecl(MD));
380 else {
381 Callee =
382 CGCallee::forDirect(CGM.GetAddrOfFunction(DevirtualizedMethod, Ty),
383 GlobalDecl(DevirtualizedMethod));
384 }
385 }
386
387 if (MD->isVirtual()) {
388 Address NewThisAddr =
389 CGM.getCXXABI().adjustThisArgumentForVirtualFunctionCall(
390 *this, CalleeDecl, This.getAddress(), UseVirtualCall);
391 This.setAddress(NewThisAddr);
392 }
393
394 return EmitCXXMemberOrOperatorCall(
395 CalleeDecl, Callee, ReturnValue, This.getPointer(),
396 /*ImplicitParam=*/nullptr, QualType(), CE, RtlArgs);
397}

399RValue
400CodeGenFunction::EmitCXXMemberPointerCallExpr(const CXXMemberCallExpr *E,
401 ReturnValueSlot ReturnValue) {
402 const BinaryOperator *BO =
403 cast<BinaryOperator>(E->getCallee()->IgnoreParens());
404 const Expr *BaseExpr = BO->getLHS();
405 const Expr *MemFnExpr = BO->getRHS();
406
407 const MemberPointerType *MPT =
408 MemFnExpr->getType()->castAs<MemberPointerType>();
409
410 const FunctionProtoType *FPT =
411 MPT->getPointeeType()->castAs<FunctionProtoType>();
412 const CXXRecordDecl *RD =
413 cast<CXXRecordDecl>(MPT->getClass()->getAs<RecordType>()->getDecl());
414
415 // Emit the 'this' pointer.
416 Address This = Address::invalid();
417 if (BO->getOpcode() == BO_PtrMemI)
418 This = EmitPointerWithAlignment(BaseExpr);
419 else
420 This = EmitLValue(BaseExpr).getAddress();
421
422 EmitTypeCheck(TCK_MemberCall, E->getExprLoc(), This.getPointer(),
423 QualType(MPT->getClass(), 0));
424
425 // Get the member function pointer.
426 llvm::Value *MemFnPtr = EmitScalarExpr(MemFnExpr);
427
428 // Ask the ABI to load the callee. Note that This is modified.
429 llvm::Value *ThisPtrForCall = nullptr;
430 CGCallee Callee =
431 CGM.getCXXABI().EmitLoadOfMemberFunctionPointer(*this, BO, This,
432 ThisPtrForCall, MemFnPtr, MPT);
433
434 CallArgList Args;
435
436 QualType ThisType =
437 getContext().getPointerType(getContext().getTagDeclType(RD));
438
439 // Push the this ptr.
440 Args.add(RValue::get(ThisPtrForCall), ThisType);
441
442 RequiredArgs required = RequiredArgs::forPrototypePlus(FPT, 1);
443
444 // And the rest of the call args
445 EmitCallArgs(Args, FPT, E->arguments());
446 return EmitCall(CGM.getTypes().arrangeCXXMethodCall(Args, FPT, required,
447 /*PrefixSize=*/0),
448 Callee, ReturnValue, Args, nullptr, E->getExprLoc());
449}

467static void EmitNullBaseClassInitialization(CodeGenFunction &CGF,
468 Address DestPtr,
469 const CXXRecordDecl *Base) {
470 if (Base->isEmpty())
471 return;
472
473 DestPtr = CGF.Builder.CreateElementBitCast(DestPtr, CGF.Int8Ty);
474
475 const ASTRecordLayout &Layout = CGF.getContext().getASTRecordLayout(Base);
476 CharUnits NVSize = Layout.getNonVirtualSize();
477
478 // We cannot simply zero-initialize the entire base sub-object if vbptrs are
479 // present, they are initialized by the most derived class before calling the
480 // constructor.
481 SmallVector<std::pair<CharUnits, CharUnits>, 1> Stores;
482 Stores.emplace_back(CharUnits::Zero(), NVSize);
483
484 // Each store is split by the existence of a vbptr.
485 CharUnits VBPtrWidth = CGF.getPointerSize();
486 std::vector<CharUnits> VBPtrOffsets =
487 CGF.CGM.getCXXABI().getVBPtrOffsets(Base);
488 for (CharUnits VBPtrOffset : VBPtrOffsets) {
489 // Stop before we hit any virtual base pointers located in virtual bases.
490 if (VBPtrOffset >= NVSize)
491 break;
492 std::pair<CharUnits, CharUnits> LastStore = Stores.pop_back_val();
493 CharUnits LastStoreOffset = LastStore.first;
494 CharUnits LastStoreSize = LastStore.second;
495
496 CharUnits SplitBeforeOffset = LastStoreOffset;
497 CharUnits SplitBeforeSize = VBPtrOffset - SplitBeforeOffset;
498 assert(!SplitBeforeSize.isNegative() && "negative store size!");
499 if (!SplitBeforeSize.isZero())
500 Stores.emplace_back(SplitBeforeOffset, SplitBeforeSize);
501
502 CharUnits SplitAfterOffset = VBPtrOffset + VBPtrWidth;
503 CharUnits SplitAfterSize = LastStoreSize - SplitAfterOffset;
504 assert(!SplitAfterSize.isNegative() && "negative store size!");
505 if (!SplitAfterSize.isZero())
506 Stores.emplace_back(SplitAfterOffset, SplitAfterSize);
507 }
508
509 // If the type contains a pointer to data member we can't memset it to zero.
510 // Instead, create a null constant and copy it to the destination.
511 // TODO: there are other patterns besides zero that we can usefully memset,
512 // like -1, which happens to be the pattern used by member-pointers.
513 // TODO: isZeroInitializable can be over-conservative in the case where a
514 // virtual base contains a member pointer.
515 llvm::Constant *NullConstantForBase = CGF.CGM.EmitNullConstantForBase(Base);
516 if (!NullConstantForBase->isNullValue()) {
517 llvm::GlobalVariable *NullVariable = new llvm::GlobalVariable(
518 CGF.CGM.getModule(), NullConstantForBase->getType(),
519 /*isConstant=*/true, llvm::GlobalVariable::PrivateLinkage,
520 NullConstantForBase, Twine());
521
522 CharUnits Align = std::max(Layout.getNonVirtualAlignment(),
523 DestPtr.getAlignment());
524 NullVariable->setAlignment(Align.getQuantity());
525
526 Address SrcPtr = Address(CGF.EmitCastToVoidPtr(NullVariable), Align);
527
528 // Get and call the appropriate llvm.memcpy overload.
529 for (std::pair<CharUnits, CharUnits> Store : Stores) {
530 CharUnits StoreOffset = Store.first;
531 CharUnits StoreSize = Store.second;
532 llvm::Value *StoreSizeVal = CGF.CGM.getSize(StoreSize);
533 CGF.Builder.CreateMemCpy(
534 CGF.Builder.CreateConstInBoundsByteGEP(DestPtr, StoreOffset),
535 CGF.Builder.CreateConstInBoundsByteGEP(SrcPtr, StoreOffset),
536 StoreSizeVal);
537 }
538
539 // Otherwise, just memset the whole thing to zero. This is legal
540 // because in LLVM, all default initializers (other than the ones we just
541 // handled above) are guaranteed to have a bit pattern of all zeros.
542 } else {
543 for (std::pair<CharUnits, CharUnits> Store : Stores) {
544 CharUnits StoreOffset = Store.first;
545 CharUnits StoreSize = Store.second;
546 llvm::Value *StoreSizeVal = CGF.CGM.getSize(StoreSize);
547 CGF.Builder.CreateMemSet(
548 CGF.Builder.CreateConstInBoundsByteGEP(DestPtr, StoreOffset),
549 CGF.Builder.getInt8(0), StoreSizeVal);
550 }
551 }
552}

554void
555CodeGenFunction::EmitCXXConstructExpr(const CXXConstructExpr *E,
556 AggValueSlot Dest) {
557 assert(!Dest.isIgnored() && "Must have a destination!");
558 const CXXConstructorDecl *CD = E->getConstructor();
559
560 // If we require zero initialization before (or instead of) calling the
561 // constructor, as can be the case with a non-user-provided default
562 // constructor, emit the zero initialization now, unless destination is
563 // already zeroed.
564 if (E->requiresZeroInitialization() && !Dest.isZeroed()) {
565 switch (E->getConstructionKind()) {
566 case CXXConstructExpr::CK_Delegating:
567 case CXXConstructExpr::CK_Complete:
568 EmitNullInitialization(Dest.getAddress(), E->getType());
569 break;
570 case CXXConstructExpr::CK_VirtualBase:
571 case CXXConstructExpr::CK_NonVirtualBase:
572 EmitNullBaseClassInitialization(*this, Dest.getAddress(),
573 CD->getParent());
574 break;
575 }
576 }
577
578 // If this is a call to a trivial default constructor, do nothing.
579 if (CD->isTrivial() && CD->isDefaultConstructor())
580 return;
581
582 // Elide the constructor if we're constructing from a temporary.
583 // The temporary check is required because Sema sets this on NRVO
584 // returns.
585 if (getLangOpts().ElideConstructors && E->isElidable()) {
586 assert(getContext().hasSameUnqualifiedType(E->getType(),
587 E->getArg(0)->getType()));
588 if (E->getArg(0)->isTemporaryObject(getContext(), CD->getParent())) {
589 EmitAggExpr(E->getArg(0), Dest);
590 return;
591 }
592 }
593
594 if (const ArrayType *arrayType
595 = getContext().getAsArrayType(E->getType())) {
596 EmitCXXAggrConstructorCall(CD, arrayType, Dest.getAddress(), E,
597 Dest.isSanitizerChecked());
598 } else {
599 CXXCtorType Type = Ctor_Complete;
600 bool ForVirtualBase = false;
601 bool Delegating = false;
602
603 switch (E->getConstructionKind()) {
604 case CXXConstructExpr::CK_Delegating:
605 // We should be emitting a constructor; GlobalDecl will assert this
606 Type = CurGD.getCtorType();
607 Delegating = true;
608 break;
609
610 case CXXConstructExpr::CK_Complete:
611 Type = Ctor_Complete;
612 break;
613
614 case CXXConstructExpr::CK_VirtualBase:
615 ForVirtualBase = true;
616 LLVM_FALLTHROUGH;
617
618 case CXXConstructExpr::CK_NonVirtualBase:
619 Type = Ctor_Base;
620 }
621
622 // Call the constructor.
623 EmitCXXConstructorCall(CD, Type, ForVirtualBase, Delegating, Dest, E);
624 }

662static llvm::Value *EmitCXXNewAllocSize(CodeGenFunction &CGF,
663 const CXXNewExpr *e,
664 unsigned minElements,
665 llvm::Value *&numElements,
666 llvm::Value *&sizeWithoutCookie) {
667 QualType type = e->getAllocatedType();
668
669 if (!e->isArray()) {
670 CharUnits typeSize = CGF.getContext().getTypeSizeInChars(type);
671 sizeWithoutCookie
672 = llvm::ConstantInt::get(CGF.SizeTy, typeSize.getQuantity());
673 return sizeWithoutCookie;
674 }
675
676 // The width of size_t.
677 unsigned sizeWidth = CGF.SizeTy->getBitWidth();
678
679 // Figure out the cookie size.
680 llvm::APInt cookieSize(sizeWidth,
681 CalculateCookiePadding(CGF, e).getQuantity());
682
683 // Emit the array size expression.
684 // We multiply the size of all dimensions for NumElements.
685 // e.g for 'int[2][3]', ElemType is 'int' and NumElements is 6.
686 numElements = ConstantEmitter(CGF).tryEmitAbstract(
687 *e->getArraySize(), (*e->getArraySize())->getType());
688 if (!numElements)
689 numElements = CGF.EmitScalarExpr(*e->getArraySize());
690 assert(isa<llvm::IntegerType>(numElements->getType()));
691
692 // The number of elements can be have an arbitrary integer type;
693 // essentially, we need to multiply it by a constant factor, add a
694 // cookie size, and verify that the result is representable as a
695 // size_t. That's just a gloss, though, and it's wrong in one
696 // important way: if the count is negative, it's an error even if
697 // the cookie size would bring the total size >= 0.
698 bool isSigned
699 = (*e->getArraySize())->getType()->isSignedIntegerOrEnumerationType();
700 llvm::IntegerType *numElementsType
701 = cast<llvm::IntegerType>(numElements->getType());
702 unsigned numElementsWidth = numElementsType->getBitWidth();
703
704 // Compute the constant factor.
705 llvm::APInt arraySizeMultiplier(sizeWidth, 1);
706 while (const ConstantArrayType *CAT
707 = CGF.getContext().getAsConstantArrayType(type)) {
708 type = CAT->getElementType();
709 arraySizeMultiplier *= CAT->getSize();
710 }
711
712 CharUnits typeSize = CGF.getContext().getTypeSizeInChars(type);
713 llvm::APInt typeSizeMultiplier(sizeWidth, typeSize.getQuantity());
714 typeSizeMultiplier *= arraySizeMultiplier;
715
716 // This will be a size_t.
717 llvm::Value *size;
718
719 // If someone is doing 'new int[42]' there is no need to do a dynamic check.
720 // Don't bloat the -O0 code.
721 if (llvm::ConstantInt *numElementsC =
722 dyn_cast<llvm::ConstantInt>(numElements)) {
723 const llvm::APInt &count = numElementsC->getValue();
724
725 bool hasAnyOverflow = false;
726
727 // If 'count' was a negative number, it's an overflow.
728 if (isSigned && count.isNegative())
729 hasAnyOverflow = true;
730
731 // We want to do all this arithmetic in size_t. If numElements is
732 // wider than that, check whether it's already too big, and if so,
733 // overflow.
734 else if (numElementsWidth > sizeWidth &&
735 numElementsWidth - sizeWidth > count.countLeadingZeros())
736 hasAnyOverflow = true;
737
738 // Okay, compute a count at the right width.
739 llvm::APInt adjustedCount = count.zextOrTrunc(sizeWidth);
740
741 // If there is a brace-initializer, we cannot allocate fewer elements than
742 // there are initializers. If we do, that's treated like an overflow.
743 if (adjustedCount.ult(minElements))
744 hasAnyOverflow = true;
745
746 // Scale numElements by that. This might overflow, but we don't
747 // care because it only overflows if allocationSize does, too, and
748 // if that overflows then we shouldn't use this.
749 numElements = llvm::ConstantInt::get(CGF.SizeTy,
750 adjustedCount * arraySizeMultiplier);
751
752 // Compute the size before cookie, and track whether it overflowed.
753 bool overflow;
754 llvm::APInt allocationSize
755 = adjustedCount.umul_ov(typeSizeMultiplier, overflow);
756 hasAnyOverflow |= overflow;
757
758 // Add in the cookie, and check whether it's overflowed.
759 if (cookieSize != 0) {
760 // Save the current size without a cookie. This shouldn't be
761 // used if there was overflow.
762 sizeWithoutCookie = llvm::ConstantInt::get(CGF.SizeTy, allocationSize);
763
764 allocationSize = allocationSize.uadd_ov(cookieSize, overflow);
765 hasAnyOverflow |= overflow;
766 }
767
768 // On overflow, produce a -1 so operator new will fail.
769 if (hasAnyOverflow) {
770 size = llvm::Constant::getAllOnesValue(CGF.SizeTy);
771 } else {
772 size = llvm::ConstantInt::get(CGF.SizeTy, allocationSize);
773 }
774
775 // Otherwise, we might need to use the overflow intrinsics.
776 } else {
777 // There are up to five conditions we need to test for:
778 // 1) if isSigned, we need to check whether numElements is negative;
779 // 2) if numElementsWidth > sizeWidth, we need to check whether
780 // numElements is larger than something representable in size_t;
781 // 3) if minElements > 0, we need to check whether numElements is smaller
782 // than that.
783 // 4) we need to compute
784 // sizeWithoutCookie := numElements * typeSizeMultiplier
785 // and check whether it overflows; and
786 // 5) if we need a cookie, we need to compute
787 // size := sizeWithoutCookie + cookieSize
788 // and check whether it overflows.
789
790 llvm::Value *hasOverflow = nullptr;
791
792 // If numElementsWidth > sizeWidth, then one way or another, we're
793 // going to have to do a comparison for (2), and this happens to
794 // take care of (1), too.
795 if (numElementsWidth > sizeWidth) {
796 llvm::APInt threshold(numElementsWidth, 1);
797 threshold <<= sizeWidth;
798
799 llvm::Value *thresholdV
800 = llvm::ConstantInt::get(numElementsType, threshold);
801
802 hasOverflow = CGF.Builder.CreateICmpUGE(numElements, thresholdV);
803 numElements = CGF.Builder.CreateTrunc(numElements, CGF.SizeTy);
804
805 // Otherwise, if we're signed, we want to sext up to size_t.
806 } else if (isSigned) {
807 if (numElementsWidth < sizeWidth)
808 numElements = CGF.Builder.CreateSExt(numElements, CGF.SizeTy);
809
810 // If there's a non-1 type size multiplier, then we can do the
811 // signedness check at the same time as we do the multiply
812 // because a negative number times anything will cause an
813 // unsigned overflow. Otherwise, we have to do it here. But at least
814 // in this case, we can subsume the >= minElements check.
815 if (typeSizeMultiplier == 1)
816 hasOverflow = CGF.Builder.CreateICmpSLT(numElements,
817 llvm::ConstantInt::get(CGF.SizeTy, minElements));
818
819 // Otherwise, zext up to size_t if necessary.
820 } else if (numElementsWidth < sizeWidth) {
821 numElements = CGF.Builder.CreateZExt(numElements, CGF.SizeTy);
822 }
823
824 assert(numElements->getType() == CGF.SizeTy);
825
826 if (minElements) {
827 // Don't allow allocation of fewer elements than we have initializers.
828 if (!hasOverflow) {
829 hasOverflow = CGF.Builder.CreateICmpULT(numElements,
830 llvm::ConstantInt::get(CGF.SizeTy, minElements));
831 } else if (numElementsWidth > sizeWidth) {
832 // The other existing overflow subsumes this check.
833 // We do an unsigned comparison, since any signed value < -1 is
834 // taken care of either above or below.
835 hasOverflow = CGF.Builder.CreateOr(hasOverflow,
836 CGF.Builder.CreateICmpULT(numElements,
837 llvm::ConstantInt::get(CGF.SizeTy, minElements)));
838 }
839 }
840
841 size = numElements;
842
843 // Multiply by the type size if necessary. This multiplier
844 // includes all the factors for nested arrays.
845 //
846 // This step also causes numElements to be scaled up by the
847 // nested-array factor if necessary. Overflow on this computation
848 // can be ignored because the result shouldn't be used if
849 // allocation fails.
850 if (typeSizeMultiplier != 1) {
851 llvm::Function *umul_with_overflow
852 = CGF.CGM.getIntrinsic(llvm::Intrinsic::umul_with_overflow, CGF.SizeTy);
853
854 llvm::Value *tsmV =
855 llvm::ConstantInt::get(CGF.SizeTy, typeSizeMultiplier);
856 llvm::Value *result =
857 CGF.Builder.CreateCall(umul_with_overflow, {size, tsmV});
858
859 llvm::Value *overflowed = CGF.Builder.CreateExtractValue(result, 1);
860 if (hasOverflow)
861 hasOverflow = CGF.Builder.CreateOr(hasOverflow, overflowed);
862 else
863 hasOverflow = overflowed;
864
865 size = CGF.Builder.CreateExtractValue(result, 0);
866
867 // Also scale up numElements by the array size multiplier.
868 if (arraySizeMultiplier != 1) {
869 // If the base element type size is 1, then we can re-use the
870 // multiply we just did.
871 if (typeSize.isOne()) {
872 assert(arraySizeMultiplier == typeSizeMultiplier);
873 numElements = size;
874
875 // Otherwise we need a separate multiply.
876 } else {
877 llvm::Value *asmV =
878 llvm::ConstantInt::get(CGF.SizeTy, arraySizeMultiplier);
879 numElements = CGF.Builder.CreateMul(numElements, asmV);
880 }
881 }
882 } else {
883 // numElements doesn't need to be scaled.
884 assert(arraySizeMultiplier == 1);
885 }
886
887 // Add in the cookie size if necessary.
888 if (cookieSize != 0) {
889 sizeWithoutCookie = size;
890
891 llvm::Function *uadd_with_overflow
892 = CGF.CGM.getIntrinsic(llvm::Intrinsic::uadd_with_overflow, CGF.SizeTy);
893
894 llvm::Value *cookieSizeV = llvm::ConstantInt::get(CGF.SizeTy, cookieSize);
895 llvm::Value *result =
896 CGF.Builder.CreateCall(uadd_with_overflow, {size, cookieSizeV});
897
898 llvm::Value *overflowed = CGF.Builder.CreateExtractValue(result, 1);
899 if (hasOverflow)
900 hasOverflow = CGF.Builder.CreateOr(hasOverflow, overflowed);
901 else
902 hasOverflow = overflowed;
903
904 size = CGF.Builder.CreateExtractValue(result, 0);
905 }
906
907 // If we had any possibility of dynamic overflow, make a select to
908 // overwrite 'size' with an all-ones value, which should cause
909 // operator new to throw.
910 if (hasOverflow)
911 size = CGF.Builder.CreateSelect(hasOverflow,
912 llvm::Constant::getAllOnesValue(CGF.SizeTy),
913 size);
914 }
915
916 if (cookieSize == 0)
917 sizeWithoutCookie = size;
918 else
919 assert(sizeWithoutCookie && "didn't set sizeWithoutCookie?");
920
921 return size;
922}

952void CodeGenFunction::EmitNewArrayInitializer(
953 const CXXNewExpr *E, QualType ElementType, llvm::Type *ElementTy,
954 Address BeginPtr, llvm::Value *NumElements,
955 llvm::Value *AllocSizeWithoutCookie) {
956 // If we have a type with trivial initialization and no initializer,
957 // there's nothing to do.
958 if (!E->hasInitializer())
959 return;
960
961 Address CurPtr = BeginPtr;
962
963 unsigned InitListElements = 0;
964
965 const Expr *Init = E->getInitializer();
966 Address EndOfInit = Address::invalid();
967 QualType::DestructionKind DtorKind = ElementType.isDestructedType();
968 EHScopeStack::stable_iterator Cleanup;
969 llvm::Instruction *CleanupDominator = nullptr;
970
971 CharUnits ElementSize = getContext().getTypeSizeInChars(ElementType);
972 CharUnits ElementAlign =
973 BeginPtr.getAlignment().alignmentOfArrayElement(ElementSize);
974
975 // Attempt to perform zero-initialization using memset.
976 auto TryMemsetInitialization = [&]() -> bool {
977 // FIXME: If the type is a pointer-to-data-member under the Itanium ABI,
978 // we can initialize with a memset to -1.
979 if (!CGM.getTypes().isZeroInitializable(ElementType))
980 return false;
981
982 // Optimization: since zero initialization will just set the memory
983 // to all zeroes, generate a single memset to do it in one shot.
984
985 // Subtract out the size of any elements we've already initialized.
986 auto *RemainingSize = AllocSizeWithoutCookie;
987 if (InitListElements) {
988 // We know this can't overflow; we check this when doing the allocation.
989 auto *InitializedSize = llvm::ConstantInt::get(
990 RemainingSize->getType(),
991 getContext().getTypeSizeInChars(ElementType).getQuantity() *
992 InitListElements);
993 RemainingSize = Builder.CreateSub(RemainingSize, InitializedSize);
994 }
995
996 // Create the memset.
997 Builder.CreateMemSet(CurPtr, Builder.getInt8(0), RemainingSize, false);
998 return true;
999 };
1000
1001 // If the initializer is an initializer list, first do the explicit elements.
1002 if (const InitListExpr *ILE = dyn_cast<InitListExpr>(Init)) {
1003 // Initializing from a (braced) string literal is a special case; the init
1004 // list element does not initialize a (single) array element.
1005 if (ILE->isStringLiteralInit()) {
1006 // Initialize the initial portion of length equal to that of the string
1007 // literal. The allocation must be for at least this much; we emitted a
1008 // check for that earlier.
1009 AggValueSlot Slot =
1010 AggValueSlot::forAddr(CurPtr, ElementType.getQualifiers(),
1011 AggValueSlot::IsDestructed,
1012 AggValueSlot::DoesNotNeedGCBarriers,
1013 AggValueSlot::IsNotAliased,
1014 AggValueSlot::DoesNotOverlap,
1015 AggValueSlot::IsNotZeroed,
1016 AggValueSlot::IsSanitizerChecked);
1017 EmitAggExpr(ILE->getInit(0), Slot);
1018
1019 // Move past these elements.
1020 InitListElements =
1021 cast<ConstantArrayType>(ILE->getType()->getAsArrayTypeUnsafe())
1022 ->getSize().getZExtValue();
1023 CurPtr =
1024 Address(Builder.CreateInBoundsGEP(CurPtr.getPointer(),
1025 Builder.getSize(InitListElements),
1026 "string.init.end"),
1027 CurPtr.getAlignment().alignmentAtOffset(InitListElements *
1028 ElementSize));
1029
1030 // Zero out the rest, if any remain.
1031 llvm::ConstantInt *ConstNum = dyn_cast<llvm::ConstantInt>(NumElements);
1032 if (!ConstNum || !ConstNum->equalsInt(InitListElements)) {
1033 bool OK = TryMemsetInitialization();
1034 (void)OK;
1035 assert(OK && "couldn't memset character type?");
1036 }
1037 return;
1038 }
1039
1040 InitListElements = ILE->getNumInits();
1041
1042 // If this is a multi-dimensional array new, we will initialize multiple
1043 // elements with each init list element.
1044 QualType AllocType = E->getAllocatedType();
1045 if (const ConstantArrayType *CAT = dyn_cast_or_null<ConstantArrayType>(
1046 AllocType->getAsArrayTypeUnsafe())) {
1047 ElementTy = ConvertTypeForMem(AllocType);
1048 CurPtr = Builder.CreateElementBitCast(CurPtr, ElementTy);
1049 InitListElements *= getContext().getConstantArrayElementCount(CAT);
1050 }
1051
1052 // Enter a partial-destruction Cleanup if necessary.
1053 if (needsEHCleanup(DtorKind)) {
1054 // In principle we could tell the Cleanup where we are more
1055 // directly, but the control flow can get so varied here that it
1056 // would actually be quite complex. Therefore we go through an
1057 // alloca.
1058 EndOfInit = CreateTempAlloca(BeginPtr.getType(), getPointerAlign(),
1059 "array.init.end");
1060 CleanupDominator = Builder.CreateStore(BeginPtr.getPointer(), EndOfInit);
1061 pushIrregularPartialArrayCleanup(BeginPtr.getPointer(), EndOfInit,
1062 ElementType, ElementAlign,
1063 getDestroyer(DtorKind));
1064 Cleanup = EHStack.stable_begin();
1065 }
1066
1067 CharUnits StartAlign = CurPtr.getAlignment();
1068 for (unsigned i = 0, e = ILE->getNumInits(); i != e; ++i) {
1069 // Tell the cleanup that it needs to destroy up to this
1070 // element. TODO: some of these stores can be trivially
1071 // observed to be unnecessary.
1072 if (EndOfInit.isValid()) {
1073 auto FinishedPtr =
1074 Builder.CreateBitCast(CurPtr.getPointer(), BeginPtr.getType());
1075 Builder.CreateStore(FinishedPtr, EndOfInit);
1076 }
1077 // FIXME: If the last initializer is an incomplete initializer list for
1078 // an array, and we have an array filler, we can fold together the two
1079 // initialization loops.
1080 StoreAnyExprIntoOneUnit(*this, ILE->getInit(i),
1081 ILE->getInit(i)->getType(), CurPtr,
1082 AggValueSlot::DoesNotOverlap);
1083 CurPtr = Address(Builder.CreateInBoundsGEP(CurPtr.getPointer(),
1084 Builder.getSize(1),
1085 "array.exp.next"),
1086 StartAlign.alignmentAtOffset((i + 1) * ElementSize));
1087 }
1088
1089 // The remaining elements are filled with the array filler expression.
1090 Init = ILE->getArrayFiller();
1091
1092 // Extract the initializer for the individual array elements by pulling
1093 // out the array filler from all the nested initializer lists. This avoids
1094 // generating a nested loop for the initialization.
1095 while (Init && Init->getType()->isConstantArrayType()) {
1096 auto *SubILE = dyn_cast<InitListExpr>(Init);
1097 if (!SubILE)
1098 break;
1099 assert(SubILE->getNumInits() == 0 && "explicit inits in array filler?");
1100 Init = SubILE->getArrayFiller();
1101 }
1102
1103 // Switch back to initializing one base element at a time.
1104 CurPtr = Builder.CreateBitCast(CurPtr, BeginPtr.getType());
1105 }
1106
1107 // If all elements have already been initialized, skip any further
1108 // initialization.
1109 llvm::ConstantInt *ConstNum = dyn_cast<llvm::ConstantInt>(NumElements);
1110 if (ConstNum && ConstNum->getZExtValue() <= InitListElements) {
1111 // If there was a Cleanup, deactivate it.
1112 if (CleanupDominator)
1113 DeactivateCleanupBlock(Cleanup, CleanupDominator);
1114 return;
1115 }
1116
1117 assert(Init && "have trailing elements to initialize but no initializer");
1118
1119 // If this is a constructor call, try to optimize it out, and failing that
1120 // emit a single loop to initialize all remaining elements.
1121 if (const CXXConstructExpr *CCE = dyn_cast<CXXConstructExpr>(Init)) {
1122 CXXConstructorDecl *Ctor = CCE->getConstructor();
1123 if (Ctor->isTrivial()) {
1124 // If new expression did not specify value-initialization, then there
1125 // is no initialization.
1126 if (!CCE->requiresZeroInitialization() || Ctor->getParent()->isEmpty())
1127 return;
1128
1129 if (TryMemsetInitialization())
1130 return;
1131 }
1132
1133 // Store the new Cleanup position for irregular Cleanups.
1134 //
1135 // FIXME: Share this cleanup with the constructor call emission rather than
1136 // having it create a cleanup of its own.
1137 if (EndOfInit.isValid())
1138 Builder.CreateStore(CurPtr.getPointer(), EndOfInit);
1139
1140 // Emit a constructor call loop to initialize the remaining elements.
1141 if (InitListElements)
1142 NumElements = Builder.CreateSub(
1143 NumElements,
1144 llvm::ConstantInt::get(NumElements->getType(), InitListElements));
1145 EmitCXXAggrConstructorCall(Ctor, NumElements, CurPtr, CCE,
1146 /*NewPointerIsChecked*/true,
1147 CCE->requiresZeroInitialization());
1148 return;
1149 }
1150
1151 // If this is value-initialization, we can usually use memset.
1152 ImplicitValueInitExpr IVIE(ElementType);
1153 if (isa<ImplicitValueInitExpr>(Init)) {
1154 if (TryMemsetInitialization())
1155 return;
1156
1157 // Switch to an ImplicitValueInitExpr for the element type. This handles
1158 // only one case: multidimensional array new of pointers to members. In
1159 // all other cases, we already have an initializer for the array element.
1160 Init = &IVIE;
1161 }
1162
1163 // At this point we should have found an initializer for the individual
1164 // elements of the array.
1165 assert(getContext().hasSameUnqualifiedType(ElementType, Init->getType()) &&
1166 "got wrong type of element to initialize");
1167
1168 // If we have an empty initializer list, we can usually use memset.
1169 if (auto *ILE = dyn_cast<InitListExpr>(Init))
1170 if (ILE->getNumInits() == 0 && TryMemsetInitialization())
1171 return;
1172
1173 // If we have a struct whose every field is value-initialized, we can
1174 // usually use memset.
1175 if (auto *ILE = dyn_cast<InitListExpr>(Init)) {
1176 if (const RecordType *RType = ILE->getType()->getAs<RecordType>()) {
1177 if (RType->getDecl()->isStruct()) {
1178 unsigned NumElements = 0;
1179 if (auto *CXXRD = dyn_cast<CXXRecordDecl>(RType->getDecl()))
1180 NumElements = CXXRD->getNumBases();
1181 for (auto *Field : RType->getDecl()->fields())
1182 if (!Field->isUnnamedBitfield())
1183 ++NumElements;
1184 // FIXME: Recurse into nested InitListExprs.
1185 if (ILE->getNumInits() == NumElements)
1186 for (unsigned i = 0, e = ILE->getNumInits(); i != e; ++i)
1187 if (!isa<ImplicitValueInitExpr>(ILE->getInit(i)))
1188 --NumElements;
1189 if (ILE->getNumInits() == NumElements && TryMemsetInitialization())
1190 return;
1191 }
1192 }
1193 }
1194
1195 // Create the loop blocks.
1196 llvm::BasicBlock *EntryBB = Builder.GetInsertBlock();
1197 llvm::BasicBlock *LoopBB = createBasicBlock("new.loop");
1198 llvm::BasicBlock *ContBB = createBasicBlock("new.loop.end");
1199
1200 // Find the end of the array, hoisted out of the loop.
1201 llvm::Value *EndPtr =
1202 Builder.CreateInBoundsGEP(BeginPtr.getPointer(), NumElements, "array.end");
1203
1204 // If the number of elements isn't constant, we have to now check if there is
1205 // anything left to initialize.
1206 if (!ConstNum) {
1207 llvm::Value *IsEmpty =
1208 Builder.CreateICmpEQ(CurPtr.getPointer(), EndPtr, "array.isempty");
1209 Builder.CreateCondBr(IsEmpty, ContBB, LoopBB);
1210 }
1211
1212 // Enter the loop.
1213 EmitBlock(LoopBB);
1214
1215 // Set up the current-element phi.
1216 llvm::PHINode *CurPtrPhi =
1217 Builder.CreatePHI(CurPtr.getType(), 2, "array.cur");
1218 CurPtrPhi->addIncoming(CurPtr.getPointer(), EntryBB);
1219
1220 CurPtr = Address(CurPtrPhi, ElementAlign);
1221
1222 // Store the new Cleanup position for irregular Cleanups.
1223 if (EndOfInit.isValid())
1224 Builder.CreateStore(CurPtr.getPointer(), EndOfInit);
1225
1226 // Enter a partial-destruction Cleanup if necessary.
1227 if (!CleanupDominator && needsEHCleanup(DtorKind)) {
1228 pushRegularPartialArrayCleanup(BeginPtr.getPointer(), CurPtr.getPointer(),
1229 ElementType, ElementAlign,
1230 getDestroyer(DtorKind));
1231 Cleanup = EHStack.stable_begin();
1232 CleanupDominator = Builder.CreateUnreachable();
1233 }
1234
1235 // Emit the initializer into this element.
1236 StoreAnyExprIntoOneUnit(*this, Init, Init->getType(), CurPtr,
1237 AggValueSlot::DoesNotOverlap);
1238
1239 // Leave the Cleanup if we entered one.
1240 if (CleanupDominator) {
1241 DeactivateCleanupBlock(Cleanup, CleanupDominator);
1242 CleanupDominator->eraseFromParent();
1243 }
1244
1245 // Advance to the next element by adjusting the pointer type as necessary.
1246 llvm::Value *NextPtr =
1247 Builder.CreateConstInBoundsGEP1_32(ElementTy, CurPtr.getPointer(), 1,
1248 "array.next");
1249
1250 // Check whether we've gotten to the end of the array and, if so,
1251 // exit the loop.
1252 llvm::Value *IsEnd = Builder.CreateICmpEQ(NextPtr, EndPtr, "array.atend");
1253 Builder.CreateCondBr(IsEnd, ContBB, LoopBB);
1254 CurPtrPhi->addIncoming(NextPtr, Builder.GetInsertBlock());
1255
1256 EmitBlock(ContBB);
1257}

1272/// Emit a call to an operator new or operator delete function, as implicitly
1273/// created by new-expressions and delete-expressions.
1274static RValue EmitNewDeleteCall(CodeGenFunction &CGF,
1275 const FunctionDecl *CalleeDecl,
1276 const FunctionProtoType *CalleeType,
1277 const CallArgList &Args) {
1278 llvm::CallBase *CallOrInvoke;
1279 llvm::Constant *CalleePtr = CGF.CGM.GetAddrOfFunction(CalleeDecl);
1280 CGCallee Callee = CGCallee::forDirect(CalleePtr, GlobalDecl(CalleeDecl));
1281 RValue RV =
1282 CGF.EmitCall(CGF.CGM.getTypes().arrangeFreeFunctionCall(
1283 Args, CalleeType, /*chainCall=*/false),
1284 Callee, ReturnValueSlot(), Args, &CallOrInvoke);
1285
1286 /// C++1y [expr.new]p10:
1287 /// [In a new-expression,] an implementation is allowed to omit a call
1288 /// to a replaceable global allocation function.
1289 ///
1290 /// We model such elidable calls with the 'builtin' attribute.
1291 llvm::Function *Fn = dyn_cast<llvm::Function>(CalleePtr);
1292 if (CalleeDecl->isReplaceableGlobalAllocationFunction() &&
1293 Fn && Fn->hasFnAttribute(llvm::Attribute::NoBuiltin)) {
1294 CallOrInvoke->addAttribute(llvm::AttributeList::FunctionIndex,
1295 llvm::Attribute::Builtin);
1296 }
1297
1298 return RV;
1299}

1318namespace {
1319/// The parameters to pass to a usual operator delete.
1320struct UsualDeleteParams {
1321 bool DestroyingDelete = false;
1322 bool Size = false;
1323 bool Alignment = false;
1324};
1325}
1326
1327static UsualDeleteParams getUsualDeleteParams(const FunctionDecl *FD) {
1328 UsualDeleteParams Params;
1329
1330 const FunctionProtoType *FPT = FD->getType()->castAs<FunctionProtoType>();
1331 auto AI = FPT->param_type_begin(), AE = FPT->param_type_end();
1332
1333 // The first argument is always a void*.
1334 ++AI;
1335
1336 // The next parameter may be a std::destroying_delete_t.
1337 if (FD->isDestroyingOperatorDelete()) {
1338 Params.DestroyingDelete = true;
1339 assert(AI != AE);
1340 ++AI;
1341 }
1342
1343 // Figure out what other parameters we should be implicitly passing.
1344 if (AI != AE && (*AI)->isIntegerType()) {
1345 Params.Size = true;
1346 ++AI;
1347 }
1348
1349 if (AI != AE && (*AI)->isAlignValT()) {
1350 Params.Alignment = true;
1351 ++AI;
1352 }
1353
1354 assert(AI == AE && "unexpected usual deallocation function parameter");
1355 return Params;
1356}

1358namespace {
1359 /// A cleanup to call the given 'operator delete' function upon abnormal
1360 /// exit from a new expression. Templated on a traits type that deals with
1361 /// ensuring that the arguments dominate the cleanup if necessary.
1362 template<typename Traits>
1363 class CallDeleteDuringNew final : public EHScopeStack::Cleanup {
1364 /// Type used to hold llvm::Value*s.
1365 typedef typename Traits::ValueTy ValueTy;
1366 /// Type used to hold RValues.
1367 typedef typename Traits::RValueTy RValueTy;
1368 struct PlacementArg {
1369 RValueTy ArgValue;
1370 QualType ArgType;
1371 };
1372
1373 unsigned NumPlacementArgs : 31;
1374 unsigned PassAlignmentToPlacementDelete : 1;
1375 const FunctionDecl *OperatorDelete;
1376 ValueTy Ptr;
1377 ValueTy AllocSize;
1378 CharUnits AllocAlign;
1379
1380 PlacementArg *getPlacementArgs() {
1381 return reinterpret_cast<PlacementArg *>(this + 1);
1382 }
1383
1384 public:
1385 static size_t getExtraSize(size_t NumPlacementArgs) {
1386 return NumPlacementArgs * sizeof(PlacementArg);
1387 }
1388
1389 CallDeleteDuringNew(size_t NumPlacementArgs,
1390 const FunctionDecl *OperatorDelete, ValueTy Ptr,
1391 ValueTy AllocSize, bool PassAlignmentToPlacementDelete,
1392 CharUnits AllocAlign)
1393 : NumPlacementArgs(NumPlacementArgs),
1394 PassAlignmentToPlacementDelete(PassAlignmentToPlacementDelete),
1395 OperatorDelete(OperatorDelete), Ptr(Ptr), AllocSize(AllocSize),
1396 AllocAlign(AllocAlign) {}
1397
1398 void setPlacementArg(unsigned I, RValueTy Arg, QualType Type) {
1399 assert(I < NumPlacementArgs && "index out of range");
1400 getPlacementArgs()[I] = {Arg, Type};
1401 }
1402
1403 void Emit(CodeGenFunction &CGF, Flags flags) override {
1404 const FunctionProtoType *FPT =
1405 OperatorDelete->getType()->getAs<FunctionProtoType>();
1406 CallArgList DeleteArgs;
1407
1408 // The first argument is always a void* (or C* for a destroying operator
1409 // delete for class type C).
1410 DeleteArgs.add(Traits::get(CGF, Ptr), FPT->getParamType(0));
1411
1412 // Figure out what other parameters we should be implicitly passing.
1413 UsualDeleteParams Params;
1414 if (NumPlacementArgs) {
1415 // A placement deallocation function is implicitly passed an alignment
1416 // if the placement allocation function was, but is never passed a size.
1417 Params.Alignment = PassAlignmentToPlacementDelete;
1418 } else {
1419 // For a non-placement new-expression, 'operator delete' can take a
1420 // size and/or an alignment if it has the right parameters.
1421 Params = getUsualDeleteParams(OperatorDelete);
1422 }
1423
1424 assert(!Params.DestroyingDelete &&
1425 "should not call destroying delete in a new-expression");
1426
1427 // The second argument can be a std::size_t (for non-placement delete).
1428 if (Params.Size)
1429 DeleteArgs.add(Traits::get(CGF, AllocSize),
1430 CGF.getContext().getSizeType());
1431
1432 // The next (second or third) argument can be a std::align_val_t, which
1433 // is an enum whose underlying type is std::size_t.
1434 // FIXME: Use the right type as the parameter type. Note that in a call
1435 // to operator delete(size_t, ...), we may not have it available.
1436 if (Params.Alignment)
1437 DeleteArgs.add(RValue::get(llvm::ConstantInt::get(
1438 CGF.SizeTy, AllocAlign.getQuantity())),
1439 CGF.getContext().getSizeType());
1440
1441 // Pass the rest of the arguments, which must match exactly.
1442 for (unsigned I = 0; I != NumPlacementArgs; ++I) {
1443 auto Arg = getPlacementArgs()[I];
1444 DeleteArgs.add(Traits::get(CGF, Arg.ArgValue), Arg.ArgType);
1445 }
1446
1447 // Call 'operator delete'.
1448 EmitNewDeleteCall(CGF, OperatorDelete, FPT, DeleteArgs);
1449 }
1450 };
1451}

1453/// Enter a cleanup to call 'operator delete' if the initializer in a
1454/// new-expression throws.
1455static void EnterNewDeleteCleanup(CodeGenFunction &CGF,
1456 const CXXNewExpr *E,
1457 Address NewPtr,
1458 llvm::Value *AllocSize,
1459 CharUnits AllocAlign,
1460 const CallArgList &NewArgs) {
1461 unsigned NumNonPlacementArgs = E->passAlignment() ? 2 : 1;
1462
1463 // If we're not inside a conditional branch, then the cleanup will
1464 // dominate and we can do the easier (and more efficient) thing.
1465 if (!CGF.isInConditionalBranch()) {
1466 struct DirectCleanupTraits {
1467 typedef llvm::Value *ValueTy;
1468 typedef RValue RValueTy;
1469 static RValue get(CodeGenFunction &, ValueTy V) { return RValue::get(V); }
1470 static RValue get(CodeGenFunction &, RValueTy V) { return V; }
1471 };
1472
1473 typedef CallDeleteDuringNew<DirectCleanupTraits> DirectCleanup;
1474
1475 DirectCleanup *Cleanup = CGF.EHStack
1476 .pushCleanupWithExtra<DirectCleanup>(EHCleanup,
1477 E->getNumPlacementArgs(),
1478 E->getOperatorDelete(),
1479 NewPtr.getPointer(),
1480 AllocSize,
1481 E->passAlignment(),
1482 AllocAlign);
1483 for (unsigned I = 0, N = E->getNumPlacementArgs(); I != N; ++I) {
1484 auto &Arg = NewArgs[I + NumNonPlacementArgs];
1485 Cleanup->setPlacementArg(I, Arg.getRValue(CGF), Arg.Ty);
1486 }
1487
1488 return;
1489 }
1490
1491 // Otherwise, we need to save all this stuff.
1492 DominatingValue<RValue>::saved_type SavedNewPtr =
1493 DominatingValue<RValue>::save(
1494 CGF, RValue::get(NewPtr.getPointer(),
1495 NewPtr.getAlignment().getQuantity()));
1496 DominatingValue<RValue>::saved_type SavedAllocSize =
1497 DominatingValue<RValue>::save(CGF, RValue::get(AllocSize));
1498
1499 struct ConditionalCleanupTraits {
1500 typedef DominatingValue<RValue>::saved_type ValueTy;
1501 typedef DominatingValue<RValue>::saved_type RValueTy;
1502 static RValue get(CodeGenFunction &CGF, ValueTy V) {
1503 return V.restore(CGF);
1504 }
1505 };
1506 typedef CallDeleteDuringNew<ConditionalCleanupTraits> ConditionalCleanup;
1507
1508 ConditionalCleanup *Cleanup = CGF.EHStack
1509 .pushCleanupWithExtra<ConditionalCleanup>(EHCleanup,
1510 E->getNumPlacementArgs(),
1511 E->getOperatorDelete(),
1512 SavedNewPtr,
1513 SavedAllocSize,
1514 E->passAlignment(),
1515 AllocAlign);
1516 for (unsigned I = 0, N = E->getNumPlacementArgs(); I != N; ++I) {
1517 auto &Arg = NewArgs[I + NumNonPlacementArgs];
1518 Cleanup->setPlacementArg(
1519 I, DominatingValue<RValue>::save(CGF, Arg.getRValue(CGF)), Arg.Ty);
1520 }
1521
1522 CGF.initFullExprCleanup();
1523}

1525llvm::Value *CodeGenFunction::EmitCXXNewExpr(const CXXNewExpr *E) {
1526 // The element type being allocated.
1527 QualType allocType = getContext().getBaseElementType(E->getAllocatedType());
1528
1529 // 1. Build a call to the allocation function.
1530 FunctionDecl *allocator = E->getOperatorNew();
1531
1532 // If there is a brace-initializer, cannot allocate fewer elements than inits.
1533 unsigned minElements = 0;
1534 if (E->isArray() && E->hasInitializer()) {
1535 const InitListExpr *ILE = dyn_cast<InitListExpr>(E->getInitializer());
1536 if (ILE && ILE->isStringLiteralInit())
1537 minElements =
1538 cast<ConstantArrayType>(ILE->getType()->getAsArrayTypeUnsafe())
1539 ->getSize().getZExtValue();
1540 else if (ILE)
1541 minElements = ILE->getNumInits();
1542 }
1543
1544 llvm::Value *numElements = nullptr;
1545 llvm::Value *allocSizeWithoutCookie = nullptr;
1546 llvm::Value *allocSize =
1547 EmitCXXNewAllocSize(*this, E, minElements, numElements,
1548 allocSizeWithoutCookie);
1549 CharUnits allocAlign = getContext().getTypeAlignInChars(allocType);
1550
1551 // Emit the allocation call. If the allocator is a global placement
1552 // operator, just "inline" it directly.
1553 Address allocation = Address::invalid();
1554 CallArgList allocatorArgs;
1555
1556 const bool ReservedGlobalPlacement = allocator->isReservedGlobalPlacementOperator();
1557 if (ReservedGlobalPlacement) {
1558 assert(E->getNumPlacementArgs() == 1);
1559 const Expr *arg = *E->placement_arguments().begin();
1560
1561 LValueBaseInfo BaseInfo;
1562 allocation = EmitPointerWithAlignment(arg, &BaseInfo);
1563
1564 // If we are setting bounds aggressively, also insert a csetbounds
1565 // after calls to new(addr) Foo(); since it seems like clang
1566 // ignores the contents of a global declaration
1567 // for void* operator new(std::size_t count, void* p)
1568 if (getTarget().areAllPointersCapabilities() &&
1569 getLangOpts().getCheriBounds() >= LangOptions::CBM_Aggressive) {
1570 allocation = Address(
1571 setPointerBounds(allocation.getPointer(), allocSize, E->getExprLoc(),
1572 "new.with.bounds",
1573 E->isArray() ? "non-allocating placement new"
1574 : "non-allocating placement new[]",
1575 /*IsSubObject=*/false,
1576 "for type " + allocType.getAsString(),
1577 allocation.getAlignment().getQuantity()),
1578 allocation.getAlignment());
1579 }
1580
1581 // The pointer expression will, in many cases, be an opaque void*.
1582 // In these cases, discard the computed alignment and use the
1583 // formal alignment of the allocated type.
1584 if (BaseInfo.getAlignmentSource() != AlignmentSource::Decl)
1585 allocation = Address(allocation.getPointer(), allocAlign);
1586
1587 // Set up allocatorArgs for the call to operator delete if it's not
1588 // the reserved global operator.
1589 if (E->getOperatorDelete() &&
1590 !E->getOperatorDelete()->isReservedGlobalPlacementOperator()) {
1591 allocatorArgs.add(RValue::get(allocSize), getContext().getSizeType());
1592 allocatorArgs.add(RValue::get(allocation.getPointer(),
1593 allocation.getAlignment().getQuantity()),
1594 arg->getType());
1595 }
1596
1597 } else {
1598 const FunctionProtoType *allocatorType =
1599 allocator->getType()->castAs<FunctionProtoType>();
1600 unsigned ParamsToSkip = 0;
1601
1602 // The allocation size is the first argument.
1603 QualType sizeType = getContext().getSizeType();
1604 allocatorArgs.add(RValue::get(allocSize), sizeType);
1605 ++ParamsToSkip;
1606
1607 if (allocSize != allocSizeWithoutCookie) {
1608 CharUnits cookieAlign = getSizeAlign(); // FIXME: Ask the ABI.
1609 allocAlign = std::max(allocAlign, cookieAlign);
1610 }
1611
1612 // The allocation alignment may be passed as the second argument.
1613 if (E->passAlignment()) {
1614 QualType AlignValT = sizeType;
1615 if (allocatorType->getNumParams() > 1) {
1616 AlignValT = allocatorType->getParamType(1);
1617 assert(getContext().hasSameUnqualifiedType(
1618 AlignValT->castAs<EnumType>()->getDecl()->getIntegerType(),
1619 sizeType) &&
1620 "wrong type for alignment parameter");
1621 ++ParamsToSkip;
1622 } else {
1623 // Corner case, passing alignment to 'operator new(size_t, ...)'.
1624 assert(allocator->isVariadic() && "can't pass alignment to allocator");
1625 }
1626 allocatorArgs.add(
1627 RValue::get(llvm::ConstantInt::get(SizeTy, allocAlign.getQuantity())),
1628 AlignValT);
1629 }
1630
1631 // FIXME: Why do we not pass a CalleeDecl here?
1632 EmitCallArgs(allocatorArgs, allocatorType, E->placement_arguments(),
1633 /*AC*/AbstractCallee(), /*ParamsToSkip*/ParamsToSkip);
1634
1635 RValue RV =
1636 EmitNewDeleteCall(*this, allocator, allocatorType, allocatorArgs);
1637
1638 // If this was a call to a global replaceable allocation function that does
1639 // not take an alignment argument, the allocator is known to produce
1640 // storage that's suitably aligned for any object that fits, up to a known
1641 // threshold. Otherwise assume it's suitably aligned for the allocated type.
1642 CharUnits allocationAlign = allocAlign;
1643 if (!E->passAlignment() &&
1644 allocator->isReplaceableGlobalAllocationFunction()) {
1645 unsigned AllocatorAlign = llvm::PowerOf2Floor(std::min<uint64_t>(
1646 Target.getNewAlign(), getContext().getTypeSize(allocType)));
1647 allocationAlign = std::max(
1648 allocationAlign, getContext().toCharUnitsFromBits(AllocatorAlign));
1649 }
1650
1651 allocation = Address(RV.getScalarVal(), allocationAlign);
1652 }
1653
1654 // Emit a null check on the allocation result if the allocation
1655 // function is allowed to return null (because it has a non-throwing
1656 // exception spec or is the reserved placement new) and we have an
1657 // interesting initializer will be running sanitizers on the initialization.
1658 bool nullCheck = E->shouldNullCheckAllocation() &&
1659 (!allocType.isPODType(getContext()) || E->hasInitializer() ||
1660 sanitizePerformTypeCheck());
1661
1662 llvm::BasicBlock *nullCheckBB = nullptr;
1663 llvm::BasicBlock *contBB = nullptr;
1664
1665 // The null-check means that the initializer is conditionally
1666 // evaluated.
1667 ConditionalEvaluation conditional(*this);
1668
1669 if (nullCheck) {
1670 conditional.begin(*this);
1671
1672 nullCheckBB = Builder.GetInsertBlock();
1673 llvm::BasicBlock *notNullBB = createBasicBlock("new.notnull");
1674 contBB = createBasicBlock("new.cont");
1675
1676 llvm::Value *isNull =
1677 Builder.CreateIsNull(allocation.getPointer(), "new.isnull");
1678 Builder.CreateCondBr(isNull, contBB, notNullBB);
1679 EmitBlock(notNullBB);
1680 }
1681
1682 // If there's an operator delete, enter a cleanup to call it if an
1683 // exception is thrown.
1684 EHScopeStack::stable_iterator operatorDeleteCleanup;
1685 llvm::Instruction *cleanupDominator = nullptr;
1686 if (E->getOperatorDelete() &&
1687 !E->getOperatorDelete()->isReservedGlobalPlacementOperator()) {
1688 EnterNewDeleteCleanup(*this, E, allocation, allocSize, allocAlign,
1689 allocatorArgs);
1690 operatorDeleteCleanup = EHStack.stable_begin();
1691 cleanupDominator = Builder.CreateUnreachable();
1692 }
1693
1694 assert((allocSize == allocSizeWithoutCookie) ==
1695 CalculateCookiePadding(*this, E).isZero());
1696 if (allocSize != allocSizeWithoutCookie) {
1697 assert(E->isArray());
1698 allocation = CGM.getCXXABI().InitializeArrayCookie(*this, allocation,
1699 numElements,
1700 E, allocType);
1701 }
1702
1703 llvm::Type *elementTy = ConvertTypeForMem(allocType);
1704 Address result = Builder.CreateElementBitCast(allocation, elementTy);
1705
1706 // Passing pointer through launder.invariant.group to avoid propagation of
1707 // vptrs information which may be included in previous type.
1708 // To not break LTO with different optimizations levels, we do it regardless
1709 // of optimization level.
1710 if (CGM.getCodeGenOpts().StrictVTablePointers &&
1711 allocator->isReservedGlobalPlacementOperator())
1712 result = Address(Builder.CreateLaunderInvariantGroup(result.getPointer()),
1713 result.getAlignment());
1714
1715 // Emit sanitizer checks for pointer value now, so that in the case of an
1716 // array it was checked only once and not at each constructor call. We may
1717 // have already checked that the pointer is non-null.
1718 // FIXME: If we have an array cookie and a potentially-throwing allocator,
1719 // we'll null check the wrong pointer here.
1720 SanitizerSet SkippedChecks;
1721 SkippedChecks.set(SanitizerKind::Null, nullCheck);
1722 EmitTypeCheck(CodeGenFunction::TCK_ConstructorCall,
1723 E->getAllocatedTypeSourceInfo()->getTypeLoc().getBeginLoc(),
1724 result.getPointer(), allocType, result.getAlignment(),
1725 SkippedChecks, numElements);
1726
1727 EmitNewInitializer(*this, E, allocType, elementTy, result, numElements,
1728 allocSizeWithoutCookie);
1729 if (E->isArray()) {
1730 // NewPtr is a pointer to the base element type. If we're
1731 // allocating an array of arrays, we'll need to cast back to the
1732 // array pointer type.
1733 llvm::Type *resultType = ConvertTypeForMem(E->getType());
1734 if (result.getType() != resultType)
1735 result = Builder.CreateBitCast(result, resultType);
1736 }
1737
1738 // Deactivate the 'operator delete' cleanup if we finished
1739 // initialization.
1740 if (operatorDeleteCleanup.isValid()) {
1741 DeactivateCleanupBlock(operatorDeleteCleanup, cleanupDominator);
1742 cleanupDominator->eraseFromParent();
1743 }
1744
1745 llvm::Value *resultPtr = result.getPointer();
1746 if (nullCheck) {
1747 conditional.end(*this);
1748
1749 llvm::BasicBlock *notNullBB = Builder.GetInsertBlock();
1750 EmitBlock(contBB);
1751
1752 llvm::PHINode *PHI = Builder.CreatePHI(resultPtr->getType(), 2);
1753 PHI->addIncoming(resultPtr, notNullBB);
1754 PHI->addIncoming(llvm::Constant::getNullValue(resultPtr->getType()),
1755 nullCheckBB);
1756
1757 resultPtr = PHI;
1758 }
1759
1760 // Don't log the non-allocating placement new since it will probably
1761 // not be setting bounds (unless cheri-bounds>=aggresive, but in that case
1762 // we will already have logged the bounds)
1763 if (llvm::cheri::ShouldCollectCSetBoundsStats && !ReservedGlobalPlacement) {
1764 // TODO: placement new with pointer might not be heap!
1765 auto Kind = llvm::cheri::SetBoundsPointerSource::Heap;
1766 uint64_t KnownAlignment = allocation.getAlignment().getQuantity();
1767 llvm::Optional<uint64_t> AllocSizeConstant =
1768 llvm::cheri::inferConstantValue(allocSize);
1769 llvm::Optional<uint64_t> MultipleOf;
1770 std::string BoundsSource;
1771 llvm::raw_string_ostream DS(BoundsSource);
1772 // allocator->getNameForDiagnostic(DS, getContext().getPrintingPolicy(),
1773 // true);
1774 DS << "operator new";
1775 if (E->isArray()) {
1776 DS << "[]";
1777 if (!AllocSizeConstant)
1778 MultipleOf = getContext().getTypeSizeInChars(allocType).getQuantity();
1779 }
1780 DS << " for " << E->getType().getAsString(getContext().getPrintingPolicy());
1781 DS.flush();
1782 // TODO: if it is an array get multiple of
1783 llvm::cheri::CSetBoundsStats->add(
1784 KnownAlignment, AllocSizeConstant, BoundsSource, Kind,
1785 "allocating type " + allocType.getAsString(),
1786 E->getSourceRange().printToString(getContext().getSourceManager()),
1787 MultipleOf);
1788 }
1789
1790 return resultPtr;
1791}
1792
1793void CodeGenFunction::EmitDeleteCall(const FunctionDecl *DeleteFD,
1794 llvm::Value *Ptr, QualType DeleteTy,
1795 llvm::Value *NumElements,
1796 CharUnits CookieSize) {
1797 assert((!NumElements && CookieSize.isZero()) ||
1798 DeleteFD->getOverloadedOperator() == OO_Array_Delete);
1799
1800 const FunctionProtoType *DeleteFTy =
1801 DeleteFD->getType()->getAs<FunctionProtoType>();
1802
1803 CallArgList DeleteArgs;
1804
1805 auto Params = getUsualDeleteParams(DeleteFD);
1806 auto ParamTypeIt = DeleteFTy->param_type_begin();
1807
1808 // Pass the pointer itself.
1809 QualType ArgTy = *ParamTypeIt++;
1810 llvm::Value *DeletePtr = Builder.CreateBitCast(Ptr, ConvertType(ArgTy));
1811 DeleteArgs.add(RValue::get(DeletePtr), ArgTy);
1812
1813 // Pass the std::destroying_delete tag if present.
1814 if (Params.DestroyingDelete) {
1815 QualType DDTag = *ParamTypeIt++;
1816 // Just pass an 'undef'. We expect the tag type to be an empty struct.
1817 auto *V = llvm::UndefValue::get(getTypes().ConvertType(DDTag));
1818 DeleteArgs.add(RValue::get(V), DDTag);
1819 }
1820
1821 // Pass the size if the delete function has a size_t parameter.
1822 if (Params.Size) {
1823 QualType SizeType = *ParamTypeIt++;
1824 CharUnits DeleteTypeSize = getContext().getTypeSizeInChars(DeleteTy);
1825 llvm::Value *Size = llvm::ConstantInt::get(ConvertType(SizeType),
1826 DeleteTypeSize.getQuantity());
1827
1828 // For array new, multiply by the number of elements.
1829 if (NumElements)
1830 Size = Builder.CreateMul(Size, NumElements);
1831
1832 // If there is a cookie, add the cookie size.
1833 if (!CookieSize.isZero())
1834 Size = Builder.CreateAdd(
1835 Size, llvm::ConstantInt::get(SizeTy, CookieSize.getQuantity()));
1836
1837 DeleteArgs.add(RValue::get(Size), SizeType);
1838 }
1839
1840 // Pass the alignment if the delete function has an align_val_t parameter.
1841 if (Params.Alignment) {
1842 QualType AlignValType = *ParamTypeIt++;
1843 CharUnits DeleteTypeAlign = getContext().toCharUnitsFromBits(
1844 getContext().getTypeAlignIfKnown(DeleteTy));

1898/// Emit the code for deleting a single object.
1899static void EmitObjectDelete(CodeGenFunction &CGF,
1900 const CXXDeleteExpr *DE,
1901 Address Ptr,
1902 QualType ElementType) {
1903 // C++11 [expr.delete]p3:
1904 // If the static type of the object to be deleted is different from its
1905 // dynamic type, the static type shall be a base class of the dynamic type
1906 // of the object to be deleted and the static type shall have a virtual
1907 // destructor or the behavior is undefined.
1908 CGF.EmitTypeCheck(CodeGenFunction::TCK_MemberCall,
1909 DE->getExprLoc(), Ptr.getPointer(),
1910 ElementType);
1911
1912 const FunctionDecl *OperatorDelete = DE->getOperatorDelete();
1913 assert(!OperatorDelete->isDestroyingOperatorDelete());
1914
1915 // Find the destructor for the type, if applicable. If the
1916 // destructor is virtual, we'll just emit the vcall and return.
1917 const CXXDestructorDecl *Dtor = nullptr;
1918 if (const RecordType *RT = ElementType->getAs<RecordType>()) {
1919 CXXRecordDecl *RD = cast<CXXRecordDecl>(RT->getDecl());
1920 if (RD->hasDefinition() && !RD->hasTrivialDestructor()) {
1921 Dtor = RD->getDestructor();
1922
1923 if (Dtor->isVirtual()) {
1924 CGF.CGM.getCXXABI().emitVirtualObjectDelete(CGF, DE, Ptr, ElementType,
1925 Dtor);
1926 return;
1927 }
1928 }
1929 }
1930
1931 // Make sure that we call delete even if the dtor throws.
1932 // This doesn't have to a conditional cleanup because we're going
1933 // to pop it off in a second.
1934 CGF.EHStack.pushCleanup<CallObjectDelete>(NormalAndEHCleanup,
1935 Ptr.getPointer(),
1936 OperatorDelete, ElementType);
1937
1938 if (Dtor)
1939 CGF.EmitCXXDestructorCall(Dtor, Dtor_Complete,
1940 /*ForVirtualBase=*/false,
1941 /*Delegating=*/false,
1942 Ptr);
1943 else if (auto Lifetime = ElementType.getObjCLifetime()) {
1944 switch (Lifetime) {
1945 case Qualifiers::OCL_None:
1946 case Qualifiers::OCL_ExplicitNone:
1947 case Qualifiers::OCL_Autoreleasing:
1948 break;
1949
1950 case Qualifiers::OCL_Strong:
1951 CGF.EmitARCDestroyStrong(Ptr, ARCPreciseLifetime);
1952 break;
1953
1954 case Qualifiers::OCL_Weak:
1955 CGF.EmitARCDestroyWeak(Ptr);
1956 break;
1957 }
1958 }
1959
1960 CGF.PopCleanupBlock();
1961}

2032void CodeGenFunction::EmitCXXDeleteExpr(const CXXDeleteExpr *E) {
2033 const Expr *Arg = E->getArgument();
2034 Address Ptr = EmitPointerWithAlignment(Arg);
2035
2036 // Null check the pointer.
2037 llvm::BasicBlock *DeleteNotNull = createBasicBlock("delete.notnull");
2038 llvm::BasicBlock *DeleteEnd = createBasicBlock("delete.end");
2039
2040 llvm::Value *IsNull = Builder.CreateIsNull(Ptr.getPointer(), "isnull");
2041
2042 Builder.CreateCondBr(IsNull, DeleteEnd, DeleteNotNull);
2043 EmitBlock(DeleteNotNull);
2044
2045 QualType DeleteTy = E->getDestroyedType();
2046
2047 // A destroying operator delete overrides the entire operation of the
2048 // delete expression.
2049 if (E->getOperatorDelete()->isDestroyingOperatorDelete()) {
2050 EmitDestroyingObjectDelete(*this, E, Ptr, DeleteTy);
2051 EmitBlock(DeleteEnd);
2052 return;
2053 }
2054
2055 // We might be deleting a pointer to array. If so, GEP down to the
2056 // first non-array element.
2057 // (this assumes that A(*)[3][7] is converted to [3 x [7 x %A]]*)
2058 if (DeleteTy->isConstantArrayType()) {
2059 llvm::Value *Zero = Builder.getInt32(0);
2060 SmallVector<llvm::Value*,8> GEP;
2061
2062 GEP.push_back(Zero); // point at the outermost array
2063
2064 // For each layer of array type we're pointing at:
2065 while (const ConstantArrayType *Arr
2066 = getContext().getAsConstantArrayType(DeleteTy)) {
2067 // 1. Unpeel the array type.
2068 DeleteTy = Arr->getElementType();
2069
2070 // 2. GEP to the first element of the array.
2071 GEP.push_back(Zero);
2072 }
2073
2074 Ptr = Address(Builder.CreateInBoundsGEP(Ptr.getPointer(), GEP, "del.first"),
2075 Ptr.getAlignment());
2076 }
2077
2078 assert(ConvertTypeForMem(DeleteTy) == Ptr.getElementType());
2079
2080 if (E->isArrayForm()) {
2081 EmitArrayDelete(*this, E, Ptr, DeleteTy);
2082 } else {
2083 EmitObjectDelete(*this, E, Ptr, DeleteTy);
2084 }
2085
2086 EmitBlock(DeleteEnd);
2087}

2089static bool isGLValueFromPointerDeref(const Expr *E) {
2090 E = E->IgnoreParens();
2091
2092 if (const auto *CE = dyn_cast<CastExpr>(E)) {
2093 if (!CE->getSubExpr()->isGLValue())
2094 return false;
2095 return isGLValueFromPointerDeref(CE->getSubExpr());
2096 }
2097
2098 if (const auto *OVE = dyn_cast<OpaqueValueExpr>(E))
2099 return isGLValueFromPointerDeref(OVE->getSourceExpr());
2100
2101 if (const auto *BO = dyn_cast<BinaryOperator>(E))
2102 if (BO->getOpcode() == BO_Comma)
2103 return isGLValueFromPointerDeref(BO->getRHS());
2104
2105 if (const auto *ACO = dyn_cast<AbstractConditionalOperator>(E))
2106 return isGLValueFromPointerDeref(ACO->getTrueExpr()) ||
2107 isGLValueFromPointerDeref(ACO->getFalseExpr());
2108
2109 // C++11 [expr.sub]p1:
2110 // The expression E1[E2] is identical (by definition) to *((E1)+(E2))
2111 if (isa<ArraySubscriptExpr>(E))
2112 return true;
2113
2114 if (const auto *UO = dyn_cast<UnaryOperator>(E))
2115 if (UO->getOpcode() == UO_Deref)
2116 return true;
2117
2118 return false;

2150 CGF.Builder.CreateCondBr(IsNull, BadTypeidBlock, EndBlock);

2154 CGF.EmitBlock(EndBlock);

2229 // pointer to or object of the constructor or destructor’s own class or one
2230 // of its bases, the dynamic_cast results in undefined behavior.

2232 SrcRecordTy);

2234 if (DCE->isAlwaysNull())
2235 if (llvm::Value *T = EmitDynamicCastToNull(*this, DestTy))
2236 return T;

2241 // If the value of v is a null pointer value in the pointer case, the result
