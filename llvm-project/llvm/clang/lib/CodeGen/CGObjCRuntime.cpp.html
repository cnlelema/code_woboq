29uint64_t CGObjCRuntime::ComputeIvarBaseOffset(CodeGen::CodeGenModule &CGM,
30 const ObjCInterfaceDecl *OID,
31 const ObjCIvarDecl *Ivar) {
32 return CGM.getContext().lookupFieldBitOffset(OID, nullptr, Ivar) /
33 CGM.getContext().getCharWidth();
34}

36uint64_t CGObjCRuntime::ComputeIvarBaseOffset(CodeGen::CodeGenModule &CGM,
37 const ObjCImplementationDecl *OID,
38 const ObjCIvarDecl *Ivar) {
39 return CGM.getContext().lookupFieldBitOffset(OID->getClassInterface(), OID,

52LValue CGObjCRuntime::EmitValueForIvarAtOffset(CodeGen::CodeGenFunction &CGF,
53 const ObjCInterfaceDecl *OID,
54 llvm::Value *BaseValue,
55 const ObjCIvarDecl *Ivar,
56 unsigned CVRQualifiers,
57 llvm::Value *Offset) {
58 // Compute (type*) ( (char *) BaseValue + Offset)
59 QualType InterfaceTy{OID->getTypeForDecl(), 0};
60 QualType ObjectPtrTy =
61 CGF.CGM.getContext().getObjCObjectPointerType(InterfaceTy);
62 QualType IvarTy =
63 Ivar->getUsageType(ObjectPtrTy).withCVRQualifiers(CVRQualifiers);
64 llvm::Type *LTy = CGF.CGM.getTypes().ConvertTypeForMem(IvarTy);
65 llvm::Value *V = CGF.Builder.CreateBitCast(BaseValue, CGF.Int8PtrTy);
66 V = CGF.Builder.CreateInBoundsGEP(V, Offset, "add.ptr");
67
68 if (!Ivar->isBitField()) {
69 V = CGF.Builder.CreateBitCast(V, llvm::PointerType::get(LTy,
70 CGF.CGM.getTargetCodeGenInfo().getDefaultAS()));
71 LValue LV = CGF.MakeNaturalAlignAddrLValue(V, IvarTy);
72 return LV;
73 }
74
75 // We need to compute an access strategy for this bit-field. We are given the
76 // offset to the first byte in the bit-field, the sub-byte offset is taken
77 // from the original layout. We reuse the normal bit-field access strategy by
78 // treating this as an access to a struct where the bit-field is in byte 0,
79 // and adjust the containing type size as appropriate.
80 //
81 // FIXME: Note that currently we make a very conservative estimate of the
82 // alignment of the bit-field, because (a) it is not clear what guarantees the
83 // runtime makes us, and (b) we don't have a way to specify that the struct is
84 // at an alignment plus offset.
85 //
86 // Note, there is a subtle invariant here: we can only call this routine on
87 // non-synthesized ivars but we may be called for synthesized ivars. However,
88 // a synthesized ivar can never be a bit-field, so this is safe.
89 uint64_t FieldBitOffset =
90 CGF.CGM.getContext().lookupFieldBitOffset(OID, nullptr, Ivar);
91 uint64_t BitOffset = FieldBitOffset % CGF.CGM.getContext().getCharWidth();
92 uint64_t AlignmentBits = CGF.CGM.getTarget().getCharAlign();
93 uint64_t BitFieldSize = Ivar->getBitWidthValue(CGF.getContext());
94 CharUnits StorageSize = CGF.CGM.getContext().toCharUnitsFromBits(
95 llvm::alignTo(BitOffset + BitFieldSize, AlignmentBits));
96 CharUnits Alignment = CGF.CGM.getContext().toCharUnitsFromBits(AlignmentBits);
97
98 // Allocate a new CGBitFieldInfo object to describe this access.
99 //
100 // FIXME: This is incredibly wasteful, these should be uniqued or part of some
101 // layout object. However, this is blocked on other cleanups to the
102 // Objective-C code, so for now we just live with allocating a bunch of these
103 // objects.
104 CGBitFieldInfo *Info = new (CGF.CGM.getContext()) CGBitFieldInfo(
105 CGBitFieldInfo::MakeInfo(CGF.CGM.getTypes(), Ivar, BitOffset, BitFieldSize,
106 CGF.CGM.getContext().toBits(StorageSize),
107 CharUnits::fromQuantity(0)));
108
109 Address Addr(V, Alignment);
110 Addr = CGF.Builder.CreateElementBitCast(Addr,
111 llvm::Type::getIntNTy(CGF.getLLVMContext(),
112 Info->StorageSize));
113 return LValue::MakeBitfield(Addr, *Info, IvarTy,
114 LValueBaseInfo(AlignmentSource::Decl),
115 TBAAAccessInfo());
116}

119 struct CatchHandler {
120 const VarDecl *Variable;
121 const Stmt *Body;
122 llvm::BasicBlock *Block;
123 llvm::Constant *TypeInfo;
124 /// Flags used to differentiate cleanups and catchalls in Windows SEH

128 struct CallObjCEndCatch final : EHScopeStack::Cleanup {
129 CallObjCEndCatch(bool MightThrow, llvm::FunctionCallee Fn)
130 : MightThrow(MightThrow), Fn(Fn) {}
131 bool MightThrow;
132 llvm::FunctionCallee Fn;
133
134 void Emit(CodeGenFunction &CGF, Flags flags) override {
135 if (MightThrow)
136 CGF.EmitRuntimeCallOrInvoke(Fn);
137 else
138 CGF.EmitNounwindRuntimeCall(Fn);
139 }
140 };

143void CGObjCRuntime::EmitTryCatchStmt(CodeGenFunction &CGF,
144 const ObjCAtTryStmt &S,
145 llvm::FunctionCallee beginCatchFn,
146 llvm::FunctionCallee endCatchFn,
147 llvm::FunctionCallee exceptionRethrowFn) {
148 // Jump destination for falling out of catch bodies.
149 CodeGenFunction::JumpDest Cont;
150 if (S.getNumCatchStmts())
151 Cont = CGF.getJumpDestInCurrentScope("eh.cont");
152
153 bool useFunclets = EHPersonality::get(CGF).usesFuncletPads();
154
155 CodeGenFunction::FinallyInfo FinallyInfo;
156 if (!useFunclets)
157 if (const ObjCAtFinallyStmt *Finally = S.getFinallyStmt())
158 FinallyInfo.enter(CGF, Finally->getFinallyBody(),
159 beginCatchFn, endCatchFn, exceptionRethrowFn);
160
161 SmallVector<CatchHandler, 8> Handlers;
162
163
164 // Enter the catch, if there is one.
165 if (S.getNumCatchStmts()) {
166 for (unsigned I = 0, N = S.getNumCatchStmts(); I != N; ++I) {
167 const ObjCAtCatchStmt *CatchStmt = S.getCatchStmt(I);
168 const VarDecl *CatchDecl = CatchStmt->getCatchParamDecl();
169
170 Handlers.push_back(CatchHandler());
171 CatchHandler &Handler = Handlers.back();
172 Handler.Variable = CatchDecl;
173 Handler.Body = CatchStmt->getCatchBody();
174 Handler.Block = CGF.createBasicBlock("catch");
175 Handler.Flags = 0;
176
177 // @catch(...) always matches.
178 if (!CatchDecl) {
179 auto catchAll = getCatchAllTypeInfo();
180 Handler.TypeInfo = catchAll.RTTI;
181 Handler.Flags = catchAll.Flags;
182 // Don't consider any other catches.
183 break;
184 }
185 auto EHType = GetEHType(CatchDecl->getType());
186 if (!EHType) {
187 // CGObjCGNU::GetEHType() returns null to indicated a catch-all
188 // only happens if CGM.getLangOpts().ObjCRuntime.isNonFragile()
189 // which probably explains why we were only seeing this test
190 // fail on some systems (Linux without GNUStep?)
191 Handler.TypeInfo = nullptr; // catch-all
192 // Don't consider any other catches.
193 break;
194 }
195 unsigned AS = CGF.CGM.getTargetCodeGenInfo().getDefaultAS();
196 Handler.TypeInfo =
197 llvm::ConstantExpr::getPointerBitCastOrAddrSpaceCast(EHType,
198 CGM.Int8Ty->getPointerTo(AS));
199 }
200
201 EHCatchScope *Catch = CGF.EHStack.pushCatch(Handlers.size());
202 for (unsigned I = 0, E = Handlers.size(); I != E; ++I)
203 Catch->setHandler(I, { Handlers[I].TypeInfo, Handlers[I].Flags }, Handlers[I].Block);
204 }
205
206 if (useFunclets)
207 if (const ObjCAtFinallyStmt *Finally = S.getFinallyStmt()) {
208 CodeGenFunction HelperCGF(CGM, /*suppressNewContext=*/true);
209 if (!CGF.CurSEHParent)
210 CGF.CurSEHParent = cast<NamedDecl>(CGF.CurFuncDecl);
211 // Outline the finally block.
212 const Stmt *FinallyBlock = Finally->getFinallyBody();
213 HelperCGF.startOutlinedSEHHelper(CGF, /*isFilter*/false, FinallyBlock);
214
215 // Emit the original filter expression, convert to i32, and return.
216 HelperCGF.EmitStmt(FinallyBlock);
217
218 HelperCGF.FinishFunction(FinallyBlock->getEndLoc());
219
220 llvm::Function *FinallyFunc = HelperCGF.CurFn;
221
222
223 // Push a cleanup for __finally blocks.
224 CGF.pushSEHCleanup(NormalAndEHCleanup, FinallyFunc);
225 }
226
227
228 // Emit the try body.
229 CGF.EmitStmt(S.getTryBody());
230
231 // Leave the try.
232 if (S.getNumCatchStmts())
233 CGF.popCatchScope();
234
235 // Remember where we were.
236 CGBuilderTy::InsertPoint SavedIP = CGF.Builder.saveAndClearIP();
237
238 // Emit the handlers.
239 for (unsigned I = 0, E = Handlers.size(); I != E; ++I) {
240 CatchHandler &Handler = Handlers[I];
241
242 CGF.EmitBlock(Handler.Block);
243 llvm::CatchPadInst *CPI = nullptr;
244 SaveAndRestore<llvm::Instruction *> RestoreCurrentFuncletPad(CGF.CurrentFuncletPad);
245 if (useFunclets)
246 if ((CPI = dyn_cast_or_null<llvm::CatchPadInst>(Handler.Block->getFirstNonPHI()))) {
247 CGF.CurrentFuncletPad = CPI;
248 CPI->setOperand(2, CGF.getExceptionSlot().getPointer());
249 }
250 llvm::Value *RawExn = CGF.getExceptionFromSlot();
251
252 // Enter the catch.
253 llvm::Value *Exn = RawExn;
254 if (beginCatchFn)
255 Exn = CGF.EmitNounwindRuntimeCall(beginCatchFn, RawExn, "exn.adjusted");
256
257 CodeGenFunction::LexicalScope cleanups(CGF, Handler.Body->getSourceRange());
258
259 if (endCatchFn) {
260 // Add a cleanup to leave the catch.
261 bool EndCatchMightThrow = (Handler.Variable == nullptr);
262
263 CGF.EHStack.pushCleanup<CallObjCEndCatch>(NormalAndEHCleanup,
264 EndCatchMightThrow,
265 endCatchFn);
266 }
267
268 // Bind the catch parameter if it exists.
269 if (const VarDecl *CatchParam = Handler.Variable) {
270 llvm::Type *CatchType = CGF.ConvertType(CatchParam->getType());
271 llvm::Value *CastExn = CGF.Builder.CreateBitCast(Exn, CatchType);
272
273 CGF.EmitAutoVarDecl(*CatchParam);
274 EmitInitOfCatchParam(CGF, CastExn, CatchParam);
275 }
276 if (CPI)
277 CGF.EHStack.pushCleanup<CatchRetScope>(NormalCleanup, CPI);
278
279 CGF.ObjCEHValueStack.push_back(Exn);
280 CGF.EmitStmt(Handler.Body);
281 CGF.ObjCEHValueStack.pop_back();
282
283 // Leave any cleanups associated with the catch.
284 cleanups.ForceCleanup();
285
286 CGF.EmitBranchThroughCleanup(Cont);
287 }
288
289 // Go back to the try-statement fallthrough.
290 CGF.Builder.restoreIP(SavedIP);
291
292 // Pop out of the finally.
293 if (!useFunclets && S.getFinallyStmt())
294 FinallyInfo.exit(CGF);
295
296 if (Cont.isValid())
297 CGF.EmitBlock(Cont.getBlock());
298}

300void CGObjCRuntime::EmitInitOfCatchParam(CodeGenFunction &CGF,
301 llvm::Value *exn,
302 const VarDecl *paramDecl) {
303
304 Address paramAddr = CGF.GetAddrOfLocalVar(paramDecl);
305
306 switch (paramDecl->getType().getQualifiers().getObjCLifetime()) {
307 case Qualifiers::OCL_Strong:
308 exn = CGF.EmitARCRetainNonBlock(exn);
309 LLVM_FALLTHROUGH;
310
311 case Qualifiers::OCL_None:
312 case Qualifiers::OCL_ExplicitNone:
313 case Qualifiers::OCL_Autoreleasing:
314 CGF.Builder.CreateStore(exn, paramAddr);
315 return;
316
317 case Qualifiers::OCL_Weak:
318 CGF.EmitARCInitWeak(paramAddr, exn);
319 return;
320 }
321 llvm_unreachable("invalid ownership qualifier");
322}

337void CGObjCRuntime::EmitAtSynchronizedStmt(CodeGenFunction &CGF,
338 const ObjCAtSynchronizedStmt &S,
339 llvm::FunctionCallee syncEnterFn,
340 llvm::FunctionCallee syncExitFn) {
341 CodeGenFunction::RunCleanupsScope cleanups(CGF);
342
343 // Evaluate the lock operand. This is guaranteed to dominate the
344 // ARC release and lock-release cleanups.
345 const Expr *lockExpr = S.getSynchExpr();
346 llvm::Value *lock;
347 if (CGF.getLangOpts().ObjCAutoRefCount) {
348 lock = CGF.EmitARCRetainScalarExpr(lockExpr);
349 lock = CGF.EmitObjCConsumeObject(lockExpr->getType(), lock);
350 } else {
351 lock = CGF.EmitScalarExpr(lockExpr);
352 }
353 lock = CGF.Builder.CreateBitCast(lock, CGF.VoidPtrTy);
354
355 // Acquire the lock.
356 CGF.Builder.CreateCall(syncEnterFn, lock)->setDoesNotThrow();
357
358 // Register an all-paths cleanup to release the lock.
359 CGF.EHStack.pushCleanup<CallSyncExit>(NormalAndEHCleanup, syncExitFn, lock);
360
361 // Emit the body of the statement.
362 CGF.EmitStmt(S.getSynchBody());
363}

365/// Compute the pointer-to-function type to which a message send
366/// should be casted in order to correctly call the given method
367/// with the given arguments.
368///
369/// \param method - may be null
370/// \param resultType - the result type to use if there's no method
371/// \param callArgs - the actual arguments, including implicit ones
372CGObjCRuntime::MessageSendInfo
373CGObjCRuntime::getMessageSendInfo(const ObjCMethodDecl *method,
374 QualType resultType,
375 CallArgList &callArgs) {
376 // If there's a method, use information from that.
377 if (method) {
378 const CGFunctionInfo &signature =
379 CGM.getTypes().arrangeObjCMessageSendSignature(method, callArgs[0].Ty);
380
381 llvm::PointerType *signatureType =
382 CGM.getTypes().GetFunctionType(signature)->getPointerTo(
383 CGM.getTargetCodeGenInfo().getDefaultAS());
384
385 const CGFunctionInfo &signatureForCall =
386 CGM.getTypes().arrangeCall(signature, callArgs);
387
388 return MessageSendInfo(signatureForCall, signatureType);
389 }
390
391 // There's no method; just use a default CC.
392 const CGFunctionInfo &argsInfo =
393 CGM.getTypes().arrangeUnprototypedObjCMessageSend(resultType, callArgs);
394
395 // Derive the signature to call from that.
396 llvm::PointerType *signatureType =
397 CGM.getTypes().GetFunctionType(argsInfo)->getPointerTo(
398 CGM.getTargetCodeGenInfo().getDefaultAS());
399 return MessageSendInfo(argsInfo, signatureType);
400}
