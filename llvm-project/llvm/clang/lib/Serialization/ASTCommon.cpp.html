252unsigned serialization::ComputeHash(Selector Sel) {
253 unsigned N = Sel.getNumArgs();
254 if (N == 0)
255 ++N;
256 unsigned R = 5381;
257 for (unsigned I = 0; I != N; ++I)
258 if (IdentifierInfo *II = Sel.getIdentifierInfoForSlot(I))
259 R = llvm::djbHash(II->getName(), R);
260 return R;
261}

263const DeclContext *
264serialization::getDefinitiveDeclContext(const DeclContext *DC) {
265 switch (DC->getDeclKind()) {
266 // These entities may have multiple definitions.
267 case Decl::TranslationUnit:
268 case Decl::ExternCContext:
269 case Decl::Namespace:
270 case Decl::LinkageSpec:
271 case Decl::Export:
272 return nullptr;
273
274 // C/C++ tag types can only be defined in one place.
275 case Decl::Enum:
276 case Decl::Record:
277 if (const TagDecl *Def = cast<TagDecl>(DC)->getDefinition())
278 return Def;
279 return nullptr;
280
281 // FIXME: These can be defined in one place... except special member
282 // functions and out-of-line definitions.
283 case Decl::CXXRecord:
284 case Decl::ClassTemplateSpecialization:
285 case Decl::ClassTemplatePartialSpecialization:
286 return nullptr;
287
288 // Each function, method, and block declaration is its own DeclContext.
289 case Decl::Function:
290 case Decl::CXXMethod:
291 case Decl::CXXConstructor:
292 case Decl::CXXDestructor:
293 case Decl::CXXConversion:
294 case Decl::ObjCMethod:
295 case Decl::Block:
296 case Decl::Captured:
297 // Objective C categories, category implementations, and class
298 // implementations can only be defined in one place.
299 case Decl::ObjCCategory:
300 case Decl::ObjCCategoryImpl:
301 case Decl::ObjCImplementation:
302 return DC;
303
304 case Decl::ObjCProtocol:
305 if (const ObjCProtocolDecl *Def
306 = cast<ObjCProtocolDecl>(DC)->getDefinition())
307 return Def;
308 return nullptr;
309
310 // FIXME: These are defined in one place, but properties in class extensions
311 // end up being back-patched into the main interface. See
312 // Sema::HandlePropertyInClassExtension for the offending code.
313 case Decl::ObjCInterface:
314 return nullptr;
315
316 default:
317 llvm_unreachable("Unhandled DeclContext in AST reader");
318 }
319
320 llvm_unreachable("Unhandled decl kind");
321}

324 switch (static_cast<Decl::Kind>(Kind)) {
325 case Decl::TranslationUnit:
326 case Decl::ExternCContext:
327 // Special case of a "merged" declaration.
328 return true;
329
330 case Decl::Namespace:
331 case Decl::NamespaceAlias:
332 case Decl::Typedef:
333 case Decl::TypeAlias:
334 case Decl::Enum:
335 case Decl::Record:
336 case Decl::CXXRecord:
337 case Decl::ClassTemplateSpecialization:
338 case Decl::ClassTemplatePartialSpecialization:
339 case Decl::VarTemplateSpecialization:
340 case Decl::VarTemplatePartialSpecialization:
341 case Decl::Function:
342 case Decl::CXXDeductionGuide:
343 case Decl::CXXMethod:
344 case Decl::CXXConstructor:
345 case Decl::CXXDestructor:
346 case Decl::CXXConversion:
347 case Decl::UsingShadow:
348 case Decl::ConstructorUsingShadow:
349 case Decl::Var:
350 case Decl::FunctionTemplate:
351 case Decl::ClassTemplate:
352 case Decl::VarTemplate:
353 case Decl::TypeAliasTemplate:
354 case Decl::ObjCProtocol:
355 case Decl::ObjCInterface:
356 case Decl::Empty:
357 return true;
358
359 // Never redeclarable.
360 case Decl::UsingDirective:
361 case Decl::Label:
362 case Decl::UnresolvedUsingTypename:
363 case Decl::TemplateTypeParm:
364 case Decl::EnumConstant:
365 case Decl::UnresolvedUsingValue:
366 case Decl::IndirectField:
367 case Decl::Field:
368 case Decl::MSProperty:
369 case Decl::ObjCIvar:
370 case Decl::ObjCAtDefsField:
371 case Decl::NonTypeTemplateParm:
372 case Decl::TemplateTemplateParm:
373 case Decl::Using:
374 case Decl::UsingPack:
375 case Decl::ObjCMethod:
376 case Decl::ObjCCategory:
377 case Decl::ObjCCategoryImpl:
378 case Decl::ObjCImplementation:
379 case Decl::ObjCProperty:
380 case Decl::ObjCCompatibleAlias:
381 case Decl::LinkageSpec:
382 case Decl::Export:
383 case Decl::ObjCPropertyImpl:
384 case Decl::PragmaComment:
385 case Decl::PragmaDetectMismatch:
386 case Decl::FileScopeAsm:
387 case Decl::AccessSpec:
388 case Decl::Friend:
389 case Decl::FriendTemplate:
390 case Decl::StaticAssert:
391 case Decl::Block:
392 case Decl::Captured:
393 case Decl::ClassScopeFunctionSpecialization:
394 case Decl::Import:
395 case Decl::OMPThreadPrivate:
396 case Decl::OMPAllocate:
397 case Decl::OMPRequires:
398 case Decl::OMPCapturedExpr:
399 case Decl::OMPDeclareReduction:
400 case Decl::OMPDeclareMapper:
401 case Decl::BuiltinTemplate:
402 case Decl::Decomposition:
403 case Decl::Binding:

406 // These indirectly derive from Redeclarable<T> but are not actually

408 case Decl::ImplicitParam:
409 case Decl::ParmVar:
410 case Decl::ObjCTypeParam:
411 return false;
412 }
413
414 llvm_unreachable("Unhandled declaration kind");
415}

417bool serialization::needsAnonymousDeclarationNumber(const NamedDecl *D) {
418 // Friend declarations in dependent contexts aren't anonymous in the usual
419 // sense, but they cannot be found by name lookup in their semantic context
420 // (or indeed in any context), so we treat them as anonymous.
421 //
422 // This doesn't apply to friend tag decls; Sema makes those available to name
423 // lookup in the surrounding context.
424 if (D->getFriendObjectKind() &&
425 D->getLexicalDeclContext()->isDependentContext() && !isa<TagDecl>(D)) {
426 // For function templates and class templates, the template is numbered and
427 // not its pattern.
428 if (auto *FD = dyn_cast<FunctionDecl>(D))
429 return !FD->getDescribedFunctionTemplate();
430 if (auto *RD = dyn_cast<CXXRecordDecl>(D))
431 return !RD->getDescribedClassTemplate();
432 return true;
433 }
434
435 // At block scope, we number everything that we need to deduplicate, since we
436 // can't just use name matching to keep things lined up.
437 // FIXME: This is only necessary for an inline function or a template or
438 // similar.
439 if (D->getLexicalDeclContext()->isFunctionOrMethod()) {
440 if (auto *VD = dyn_cast<VarDecl>(D))
441 return VD->isStaticLocal();
442 // FIXME: What about CapturedDecls (and declarations nested within them)?
443 return isa<TagDecl>(D) || isa<BlockDecl>(D);
444 }
445
446 // Otherwise, we only care about anonymous class members / block-scope decls.
447 // FIXME: We need to handle lambdas and blocks within inline / templated
448 // variables too.
449 if (D->getDeclName() || !isa<CXXRecordDecl>(D->getLexicalDeclContext()))
450 return false;
451 return isa<TagDecl>(D) || isa<FieldDecl>(D);
452}
