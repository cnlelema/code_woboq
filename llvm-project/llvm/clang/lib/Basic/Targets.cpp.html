51/// DefineStd - Define a macro name and standard variants. For example if
52/// MacroName is "unix", then this will define "__unix", "__unix__", and "unix"
53/// when in GNU mode.
54void DefineStd(MacroBuilder &Builder, StringRef MacroName,
55 const LangOptions &Opts) {
56 assert(MacroName[0] != '_' && "Identifier should be in the user's namespace");
57
58 // If in GNU mode (e.g. -std=gnu99 but not -std=c99) define the raw identifier
59 // in the user's namespace.
60 if (Opts.GNUMode)
61 Builder.defineMacro(MacroName);
62
63 // Define __unix.
64 Builder.defineMacro("__" + MacroName);
65
66 // Define __unix__.
67 Builder.defineMacro("__" + MacroName + "__");
68}

70void defineCPUMacros(MacroBuilder &Builder, StringRef CPUName, bool Tuning) {
71 Builder.defineMacro("__" + CPUName);
72 Builder.defineMacro("__" + CPUName + "__");
73 if (Tuning)
74 Builder.defineMacro("__tune_" + CPUName + "__");
75}

77void addCygMingDefines(const LangOptions &Opts, MacroBuilder &Builder) {
78 // Mingw and cygwin define __declspec(a) to __attribute__((a)). Clang
79 // supports __declspec natively under -fms-extensions, but we define a no-op
80 // __declspec macro anyway for pre-processor compatibility.
81 if (Opts.MicrosoftExt)
82 Builder.defineMacro("__declspec", "__declspec");
83 else
84 Builder.defineMacro("__declspec(a)", "__attribute__((a))");
85
86 if (!Opts.MicrosoftExt) {
87 // Provide macros for all the calling convention keywords. Provide both
88 // single and double underscore prefixed variants. These are available on
89 // x64 as well as x86, even though they have no effect.
90 const char *CCs[] = {"cdecl", "stdcall", "fastcall", "thiscall", "pascal"};
91 for (const char *CC : CCs) {
92 std::string GCCSpelling = "__attribute__((__";
93 GCCSpelling += CC;
94 GCCSpelling += "__))";
95 Builder.defineMacro(Twine("_") + CC, GCCSpelling);
96 Builder.defineMacro(Twine("__") + CC, GCCSpelling);
97 }
98 }
99}

101void addMinGWDefines(const llvm::Triple &Triple, const LangOptions &Opts,
102 MacroBuilder &Builder) {
103 DefineStd(Builder, "WIN32", Opts);
104 DefineStd(Builder, "WINNT", Opts);
105 if (Triple.isArch64Bit()) {
106 DefineStd(Builder, "WIN64", Opts);
107 Builder.defineMacro("__MINGW64__");
108 }
109 Builder.defineMacro("__MSVCRT__");
110 Builder.defineMacro("__MINGW32__");
111 addCygMingDefines(Opts, Builder);
112}

598 return new WebAssemblyOSTargetInfo<WebAssembly64TargetInfo>(Triple, Opts);
599 default:
600 return nullptr;
601 }
602
603 case llvm::Triple::renderscript32:
604 return new LinuxTargetInfo<RenderScript32TargetInfo>(Triple, Opts);
605 case llvm::Triple::renderscript64:
606 return new LinuxTargetInfo<RenderScript64TargetInfo>(Triple, Opts);
607 }
608}
609} // namespace targets
610} // namespace clang

612using namespace clang::targets;
613/// CreateTargetInfo - Return the target info object for the specified target
614/// options.
615TargetInfo *
616TargetInfo::CreateTargetInfo(DiagnosticsEngine &Diags,
617 const std::shared_ptr<TargetOptions> &Opts) {
618 llvm::Triple Triple(Opts->Triple);
619
620 // Construct the target
621 std::unique_ptr<TargetInfo> Target(AllocateTarget(Triple, *Opts));
622 if (!Target) {
623 Diags.Report(diag::err_target_unknown_triple) << Triple.str();
624 return nullptr;
625 }
626 Target->TargetOpts = Opts;
627
628 // Set the target CPU if specified.
629 if (!Opts->CPU.empty() && !Target->setCPU(Opts->CPU)) {
630 Diags.Report(diag::err_target_unknown_cpu) << Opts->CPU;
631 SmallVector<StringRef, 32> ValidList;
632 Target->fillValidCPUList(ValidList);
633 if (!ValidList.empty())
634 Diags.Report(diag::note_valid_options) << llvm::join(ValidList, ", ");
635 return nullptr;
636 }
637
638 // Set the target ABI if specified.
639 if (!Opts->ABI.empty() && !Target->setABI(Opts->ABI)) {
640 Diags.Report(diag::err_target_unknown_abi) << Opts->ABI;
641 return nullptr;
642 }
643
644 // Set the fp math unit.
645 if (!Opts->FPMath.empty() && !Target->setFPMath(Opts->FPMath)) {
646 Diags.Report(diag::err_target_unknown_fpmath) << Opts->FPMath;
647 return nullptr;
648 }
649
650 // Compute the default target features, we need the target to handle this
651 // because features may have dependencies on one another.
652 llvm::StringMap<bool> Features;
653 if (!Target->initFeatureMap(Features, Diags, Opts->CPU,
654 Opts->FeaturesAsWritten))
655 return nullptr;
656
657 // Add the features to the compile options.
658 Opts->Features.clear();
659 for (const auto &F : Features)
660 Opts->Features.push_back((F.getValue() ? "+" : "-") + F.getKey().str());
661 // Sort here, so we handle the features in a predictable order. (This matters
662 // when we're dealing with features that overlap.)
663 llvm::sort(Opts->Features);
664
665 if (!Target->handleTargetFeatures(Opts->Features, Diags))
666 return nullptr;
667
668 Target->setSupportedOpenCLOpts();
669 Target->setOpenCLExtensionOpts();
670 Target->setMaxAtomicWidth();
671
672 if (!Target->validateTarget(Diags))
673 return nullptr;
674
675 Target->CheckFixedPointBits();
676
677 return Target.release();
678}
