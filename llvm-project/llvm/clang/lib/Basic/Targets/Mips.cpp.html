33bool MipsTargetInfo::processorSupportsGPR64() const {
34 return llvm::StringSwitch<bool>(CPU)
35 .Case("cheri128", true)
36 .Case("cheri256", true)
37 .Case("cheri64", true)
38 .Case("beri", true)
39 .Case("mips3", true)
40 .Case("mips4", true)
41 .Case("mips5", true)
42 .Case("mips64", true)
43 .Case("mips64r2", true)
44 .Case("mips64r3", true)
45 .Case("mips64r5", true)
46 .Case("mips64r6", true)
47 .Case("octeon", true)
48 .Default(false);
49 return false;
50}

52static constexpr llvm::StringLiteral ValidCPUNames[] = {
53 {"cheri128"}, {"cheri256"}, {"cheri64"}, {"beri"},
54 {"mips1"}, {"mips2"}, {"mips3"}, {"mips4"}, {"mips5"},
55 {"mips32"}, {"mips32r2"}, {"mips32r3"}, {"mips32r5"}, {"mips32r6"},
56 {"mips64"}, {"mips64r2"}, {"mips64r3"}, {"mips64r5"}, {"mips64r6"},
57 {"octeon"}, {"p5600"}};
58
59bool MipsTargetInfo::isValidCPUName(StringRef Name) const {
60 return llvm::find(ValidCPUNames, Name) != std::end(ValidCPUNames);
61}
62
63void MipsTargetInfo::fillValidCPUList(
64 SmallVectorImpl<StringRef> &Values) const {
65 Values.append(std::begin(ValidCPUNames), std::end(ValidCPUNames));
66}

68unsigned MipsTargetInfo::getISARev() const {
69 return llvm::StringSwitch<unsigned>(getCPU())
70 .Cases("mips32", "mips64", 1)
71 .Cases("mips32r2", "mips64r2", 2)
72 .Cases("mips32r3", "mips64r3", 3)
73 .Cases("mips32r5", "mips64r5", 5)
74 .Cases("mips32r6", "mips64r6", 6)
75 .Default(0);
76}

78void MipsTargetInfo::getTargetDefines(const LangOptions &Opts,
79 MacroBuilder &Builder) const {
80 if (BigEndian) {
81 DefineStd(Builder, "MIPSEB", Opts);
82 Builder.defineMacro("_MIPSEB");
83 } else {
84 DefineStd(Builder, "MIPSEL", Opts);
85 Builder.defineMacro("_MIPSEL");
86 }
87
88 Builder.defineMacro("__mips__");
89 Builder.defineMacro("_mips");
90 if (Opts.GNUMode)
91 Builder.defineMacro("mips");
92
93 if (ABI == "o32") {
94 Builder.defineMacro("__mips", "32");
95 Builder.defineMacro("_MIPS_ISA", "_MIPS_ISA_MIPS32");
96 } else {
97 Builder.defineMacro("__mips", "64");
98 Builder.defineMacro("__mips64");
99 Builder.defineMacro("__mips64__");
100 Builder.defineMacro("_MIPS_ISA", "_MIPS_ISA_MIPS64");
101 }
102
103 const std::string ISARev = std::to_string(getISARev());
104
105 if (!ISARev.empty())
106 Builder.defineMacro("__mips_isa_rev", ISARev);
107
108 if (ABI == "o32") {
109 Builder.defineMacro("__mips_o32");
110 Builder.defineMacro("_ABIO32", "1");
111 Builder.defineMacro("_MIPS_SIM", "_ABIO32");
112 } else if (ABI == "n32") {
113 Builder.defineMacro("__mips_n32");
114 Builder.defineMacro("_ABIN32", "2");
115 Builder.defineMacro("_MIPS_SIM", "_ABIN32");
116 } else if (ABI == "n64") {
117 Builder.defineMacro("__mips_n64");
118 Builder.defineMacro("_ABI64", "3");
119 Builder.defineMacro("_MIPS_SIM", "_ABI64");
120 } else
121 llvm_unreachable("Invalid ABI.");
122
123 if (!IsNoABICalls) {
124 Builder.defineMacro("__mips_abicalls");
125 if (CanUseBSDABICalls)
126 Builder.defineMacro("__ABICALLS__");
127 }
128
129 Builder.defineMacro("__REGISTER_PREFIX__", "");
130
131 switch (FloatABI) {
132 case HardFloat:
133 Builder.defineMacro("__mips_hard_float", Twine(1));
134 break;
135 case SoftFloat:
136 Builder.defineMacro("__mips_soft_float", Twine(1));
137 break;
138 }
139
140 if (IsSingleFloat)
141 Builder.defineMacro("__mips_single_float", Twine(1));
142
143 switch (FPMode) {
144 case FPXX:
145 Builder.defineMacro("__mips_fpr", Twine(0));
146 break;
147 case FP32:
148 Builder.defineMacro("__mips_fpr", Twine(32));
149 break;
150 case FP64:
151 Builder.defineMacro("__mips_fpr", Twine(64));
152 break;
153}
154
155 if (FPMode == FP64 || IsSingleFloat)
156 Builder.defineMacro("_MIPS_FPSET", Twine(32));
157 else
158 Builder.defineMacro("_MIPS_FPSET", Twine(16));
159
160 if (IsMips16)
161 Builder.defineMacro("__mips16", Twine(1));
162
163 if (IsMicromips)
164 Builder.defineMacro("__mips_micromips", Twine(1));
165
166 if (IsNan2008)
167 Builder.defineMacro("__mips_nan2008", Twine(1));
168
169 if (IsAbs2008)
170 Builder.defineMacro("__mips_abs2008", Twine(1));
171
172 switch (DspRev) {
173 default:
174 break;
175 case DSP1:
176 Builder.defineMacro("__mips_dsp_rev", Twine(1));
177 Builder.defineMacro("__mips_dsp", Twine(1));
178 break;
179 case DSP2:
180 Builder.defineMacro("__mips_dsp_rev", Twine(2));
181 Builder.defineMacro("__mips_dspr2", Twine(1));
182 Builder.defineMacro("__mips_dsp", Twine(1));
183 break;
184 }
185
186 if (HasMSA)
187 Builder.defineMacro("__mips_msa", Twine(1));
188
189 if (IsCHERI) {
190 if (CapabilityABI) {
191 Builder.defineMacro("__CHERI_SANDBOX__", Twine(4)); // TODO: remove this?
192 auto CapTableABI = llvm::MCTargetOptions::cheriCapabilityTableABI();
193 if (CapTableABI != llvm::CheriCapabilityTableABI::Legacy) {
194 Builder.defineMacro("__CHERI_CAPABILITY_TABLE__",
195 Twine(((int)CapTableABI) + 1));
196 }
197 auto CapTlsABI = llvm::MCTargetOptions::cheriCapabilityTlsABI();
198 if (CapTlsABI != llvm::CheriCapabilityTlsABI::Legacy) {
199 Builder.defineMacro("__CHERI_CAPABILITY_TLS__",
200 Twine((int)CapTlsABI));
201 }
202 }
203
204 // Macros for use with the set and get permissions builtins.
205 Builder.defineMacro("__CHERI_CAP_PERMISSION_GLOBAL__", Twine(1<<0));
206 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_EXECUTE__",
207 Twine(1<<1));
208 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_LOAD__", Twine(1<<2));
209 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_STORE__", Twine(1<<3));
210 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_LOAD_CAPABILITY__",
211 Twine(1<<4));
212 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_STORE_CAPABILITY__",
213 Twine(1<<5));
214 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_STORE_LOCAL__",
215 Twine(1<<6));
216 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_SEAL__", Twine(1<<7));
217 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_CCALL__", Twine(1<<8));
218 Builder.defineMacro("__CHERI_CAP_PERMISSION_PERMIT_UNSEAL__", Twine(1<<9));
219 Builder.defineMacro("__CHERI_CAP_PERMISSION_ACCESS_SYSTEM_REGISTERS__", Twine(1<<10));
220
221 Builder.defineMacro("_MIPS_SZCAP", Twine(getCHERICapabilityWidth()));
222 if (getCHERICapabilityWidth() == 128)
223 Builder.defineMacro("_MIPS_CAP_ALIGN_MASK", "0xfffffffffffffff0");
224 else
225 Builder.defineMacro("_MIPS_CAP_ALIGN_MASK", "0xffffffffffffffe0");
226 }
227 if (DisableMadd4)
228 Builder.defineMacro("__mips_no_madd4", Twine(1));
229
230 Builder.defineMacro("_MIPS_SZPTR", Twine(getPointerWidth(0)));
231 Builder.defineMacro("_MIPS_SZINT", Twine(getIntWidth()));
232 Builder.defineMacro("_MIPS_SZLONG", Twine(getLongWidth()));
233
234 Builder.defineMacro("_MIPS_ARCH", "\"" + CPU + "\"");
235 Builder.defineMacro("_MIPS_ARCH_" + StringRef(CPU).upper());
236
237 // These shouldn't be defined for MIPS-I but there's no need to check
238 // for that since MIPS-I isn't supported.
239 Builder.defineMacro("__GCC_HAVE_SYNC_COMPARE_AND_SWAP_1");
240 Builder.defineMacro("__GCC_HAVE_SYNC_COMPARE_AND_SWAP_2");
241 Builder.defineMacro("__GCC_HAVE_SYNC_COMPARE_AND_SWAP_4");
242
243 // 32-bit MIPS processors don't have the necessary lld/scd instructions
244 // found in 64-bit processors. In the case of O32 on a 64-bit processor,
245 // the instructions exist but using them violates the ABI since they
246 // require 64-bit GPRs and O32 only supports 32-bit GPRs.
247 if (IsCHERI || ABI == "n32" || ABI == "n64")
248 Builder.defineMacro("__GCC_HAVE_SYNC_COMPARE_AND_SWAP_8");
249
250
251 if (getTriple().getOS() == llvm::Triple::UnknownOS &&
252 getTriple().isOSBinFormatELF())
253 Builder.defineMacro("__ELF__");
254}

256bool MipsTargetInfo::hasFeature(StringRef Feature) const {
257 return llvm::StringSwitch<bool>(Feature)
258 .Case("mips", true)
259 .Case("fp64", FPMode == FP64)
260 .Default(false);
261}

263ArrayRef<Builtin::Info> MipsTargetInfo::getTargetBuiltins() const {
264 return llvm::makeArrayRef(BuiltinInfo, clang::Mips::LastTSBuiltin -
265 Builtin::FirstTSBuiltin);
266}

268unsigned MipsTargetInfo::getUnwindWordWidth() const {
269 return llvm::StringSwitch<unsigned>(ABI)
270 .Case("o32", 32)
271 .Case("n32", 64)
272 .Case("n64", 64)
273 .Default(getPointerWidth(0));
274}

276bool MipsTargetInfo::validateTarget(DiagnosticsEngine &Diags) const {
277 // microMIPS64R6 backend was removed.
278 if (getTriple().isMIPS64() && IsMicromips && (ABI == "n32" || ABI == "n64")) {
279 Diags.Report(diag::err_target_unsupported_cpu_for_micromips) << CPU;
280 return false;
281 }
282 // FIXME: It's valid to use O32 on a 64-bit CPU but the backend can't handle
283 // this yet. It's better to fail here than on the backend assertion.
284 if (processorSupportsGPR64() && ABI == "o32") {
285 Diags.Report(diag::err_target_unsupported_abi) << ABI << CPU;
286 return false;
287 }
288
289 // 64-bit ABI's require 64-bit CPU's.
290 if (!processorSupportsGPR64() && (ABI == "n32" || ABI == "n64")) {
291 Diags.Report(diag::err_target_unsupported_abi) << ABI << CPU;
292 return false;
293 }
294
295 // FIXME: It's valid to use O32 on a mips64/mips64el triple but the backend
296 // can't handle this yet. It's better to fail here than on the
297 // backend assertion.
298 if (getTriple().isMIPS64() && ABI == "o32") {
299 Diags.Report(diag::err_target_unsupported_abi_for_triple)
300 << ABI << getTriple().str();
301 return false;
302 }
303
304 // FIXME: It's valid to use N32/N64 on a mips/mipsel triple but the backend
305 // can't handle this yet. It's better to fail here than on the
306 // backend assertion.
307 if (getTriple().isMIPS32() && (ABI == "n32" || ABI == "n64")) {
308 Diags.Report(diag::err_target_unsupported_abi_for_triple)
309 << ABI << getTriple().str();
310 return false;
311 }
312
313 // -fpxx is valid only for the o32 ABI
314 if (FPMode == FPXX && (ABI == "n32" || ABI == "n64")) {
315 Diags.Report(diag::err_unsupported_abi_for_opt) << "-mfpxx" << "o32";
316 return false;
317 }
318
319 // -mfp32 and n32/n64 ABIs are incompatible
320 if (FPMode != FP64 && FPMode != FPXX && !IsSingleFloat &&
321 (ABI == "n32" || ABI == "n64")) {
322 Diags.Report(diag::err_opt_not_valid_with_opt) << "-mfpxx" << CPU;
323 return false;
324 }
325 // Mips revision 6 and -mfp32 are incompatible
326 if (FPMode != FP64 && FPMode != FPXX && (CPU == "mips32r6" ||
327 CPU == "mips64r6")) {
328 Diags.Report(diag::err_opt_not_valid_with_opt) << "-mfp32" << CPU;
329 return false;
330 }
331 // Option -mfp64 permitted on Mips32 iff revision 2 or higher is present
332 if (FPMode == FP64 && (CPU == "mips1" || CPU == "mips2" ||
333 getISARev() < 2) && ABI == "o32") {
334 Diags.Report(diag::err_mips_fp64_req) << "-mfp64";
335 return false;
336 }
337
338 return true;
339}
