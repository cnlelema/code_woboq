32 void setDataLayout() {
33 StringRef Layout;
34 // XXXAR: why do we need this here? can't we use the LLVM one?
35 if (ABI == "o32") {
36 Layout = "m:m-p:32:32-i8:8:32-i16:16:32-i64:64-n32-S64";
37 if (IsCHERI)
38 llvm::report_fatal_error("Cannot use CHERI with " + ABI + " ABI");
39 } else if (ABI == "n32") {
40 Layout = "m:e-p:32:32-i8:8:32-i16:16:32-i64:64-n32:64-S128";
41 if (IsCHERI)
42 llvm::report_fatal_error("Cannot use CHERI with " + ABI + " ABI");
43 } else if (ABI == "n64") {
44 if (IsCHERI) {
45 if (CapSize == 64) {
46 Layout = "m:e-pf200:64:64:64:32-i8:8:32-i16:16:32-i64:64-n32:64-S128";
47 } else if (CapSize == 128) {
48 Layout = "m:e-pf200:128:128:128:64-i8:8:32-i16:16:32-i64:64-n32:64-S128";
49 } else {
50 assert(CapSize == 256);
51 Layout = "m:e-pf200:256:256:256:64-i8:8:32-i16:16:32-i64:64-n32:64-S256";
52 }
53 } else
54 Layout = "m:e-i8:8:32-i16:16:32-i64:64-n32:64-S128";
55 } else
56 llvm_unreachable("Invalid ABI");
57
58 StringRef PurecapOptions = "";
59 // Only set globals address space to 200 for cap-table mode
60 if (CapabilityABI) {
61 if (llvm::MCTargetOptions::cheriUsesCapabilityTable()) {
62 PurecapOptions = "-A200-P200-G200";
63 } else {
64 // We also want program AS 200, but the legacy ABI still needs every
65 // llvm::Function* to be in AS0
66 IgnoreProgramASForFunctions = true;
67 PurecapOptions = "-A200-P200";
68 }
69 }
70
71 if (BigEndian)
72 resetDataLayout(
73 ("E-" + Layout + (CapabilityABI ? PurecapOptions : "")).str());
74 else
75 resetDataLayout(
76 ("e-" + Layout + (CapabilityABI ? PurecapOptions : "")).str());
77
78 if (IsCHERI) {
79 assert(DataLayout->getStackAlignment() >= ((unsigned)CapSize / 8));
80 }
81 }

104public:
105 MipsTargetInfo(const llvm::Triple &Triple, const TargetOptions &Opts)
106 : TargetInfo(Triple), IsMips16(false), IsMicromips(false),
107 IsNan2008(false), IsAbs2008(false), IsSingleFloat(false),
108 IsNoABICalls(false), CanUseBSDABICalls(false), FloatABI(HardFloat),
109 DspRev(NoDSP), HasMSA(false), DisableMadd4(false),
110 UseIndirectJumpHazard(false), FPMode(FPXX) {
111 TheCXXABI.set(TargetCXXABI::GenericMIPS);
112
113 if (Triple.isMIPS32())
114 setABI("o32");
115 else if (Triple.getEnvironment() == llvm::Triple::GNUABIN32)
116 setABI("n32");
117 else
118 setABI("n64");
119
120 CPU = ABI == "o32" ? "mips32r2" : "mips64r2";
121 // If we have a CHERI triple, or an explicit CHERI128 CPU, then assume
122 // CHERI128.
123 CapSize = llvm::StringSwitch<int>(Opts.CPU)
124 .Cases("cheri", "cheri128", 128) // If we have a CHERI CPU, default to assuming CHERI128.
125 .Case("cheri256", 256)
126 .Case("cheri64", 64)
127 .Default(-1);
128 const int TripleCapSize = [](llvm::Triple::SubArchType SubArch) {
129 switch (SubArch) {
130 case llvm::Triple::MipsSubArch_cheri64:
131 return 64;
132 case llvm::Triple::MipsSubArch_cheri128:
133 return 128;
134 case llvm::Triple::MipsSubArch_cheri256:
135 return 256;
136 default:
137 return -1;
138 }
139 }(getTriple().getSubArch());
140 if (CapSize == -1) {
141 CapSize = TripleCapSize;
142 } else if (TripleCapSize != -1 && CapSize != TripleCapSize) {
143 llvm::report_fatal_error("CPU flag " + Opts.CPU +
144 " is incompatible with triple " +
145 getTriple().str());
146 }
147 if (CapSize > 0 || getTriple().getArch() == llvm::Triple::cheri ||
148 getTriple().getEnvironment() == llvm::Triple::CheriPurecap) {
149 IsCHERI = true;
150 }
151 if (IsCHERI) {
152 switch (CapSize) {
153 default:
154 case 128: CPU = "cheri128"; break;
155 case 64: CPU = "cheri64"; break;
156 case 256: CPU = "cheri256"; break;
157 }
158 if (CapSize > 0) {
159 SuitableAlign = CapSize;
160 DefaultAlignForAttributeAligned =
161 std::max(DefaultAlignForAttributeAligned, (unsigned short)CapSize);
162 }
163 if (Triple.isMIPS32())
164 llvm::report_fatal_error("Cannot use CHERI with MIPS32 triple");
165 }
166 CanUseBSDABICalls = Triple.isOSFreeBSD() ||
167 Triple.isOSOpenBSD();
168 }

185 bool setABI(const std::string &Name) override {
186 if (Name == "o32") {
187 setO32ABITypes();
188 ABI = Name;
189 return true;
190 }
191
192 if (Name == "n32") {
193 setN32ABITypes();
194 ABI = Name;
195 return true;
196 }
197 if (Name == "n64") {
198 setN64ABITypes();
199 ABI = Name;
200 return true;
201 }
202 if (Name == "purecap") {
203 setCapabilityABITypes();
204 CapabilityABI = true;
205 ABI = "n64";
206 return true;
207 }
208 return false;

211 void setO32ABITypes() {
212 Int64Type = SignedLongLong;
213 IntMaxType = Int64Type;
214 LongDoubleFormat = &llvm::APFloat::IEEEdouble();
215 LongDoubleWidth = LongDoubleAlign = 64;
216 LongWidth = LongAlign = 32;
217 MaxAtomicPromoteWidth = MaxAtomicInlineWidth = 32;
218 PointerWidth = PointerAlign = 32;
219 PtrDiffType = SignedInt;
220 SizeType = UnsignedInt;
221 SuitableAlign = 64;
222 }
223
224 void setN32N64ABITypes() {
225 LongDoubleWidth = LongDoubleAlign = 128;
226 LongDoubleFormat = &llvm::APFloat::IEEEquad();
227 if (getTriple().isOSFreeBSD()) {
228 LongDoubleWidth = LongDoubleAlign = 64;
229 LongDoubleFormat = &llvm::APFloat::IEEEdouble();
230 }
231 MaxAtomicPromoteWidth = MaxAtomicInlineWidth = 64;
232 if (IsCHERI && CapSize > 0)
233 MaxAtomicInlineWidth = std::max(MaxAtomicInlineWidth, (unsigned short)CapSize);
234 SuitableAlign = std::max(CapSize, 128);
235 }

238 setN32N64ABITypes();

292 return TargetInfo::initFeatureMap(Features, Diags, CPU, FeaturesVec);

343 bool validateAsmConstraint(const char *&Name,
344 TargetInfo::ConstraintInfo &Info) const override {
345 switch (*Name) {
346 default:
347 return false;
348 case 'r': // CPU registers.
349 case 'd': // Equivalent to "r" unless generating MIPS16 code.
350 case 'y': // Equivalent to "r", backward compatibility only.
351 case 'f': // floating-point registers.
352 case 'c': // $25 for indirect jumps
353 case 'l': // lo register
354 case 'x': // hilo register pair
355 Info.setAllowsRegister();
356 return true;
357 case 'I': // Signed 16-bit constant
358 case 'J': // Integer 0
359 case 'K': // Unsigned 16-bit constant
360 case 'L': // Signed 32-bit constant, lower 16-bit zeros (for lui)
361 case 'M': // Constants not loadable via lui, addiu, or ori
362 case 'N': // Constant -1 to -65535
363 case 'O': // A signed 15-bit constant
364 case 'P': // A constant between 1 go 65535
365 return true;
366 case 'R': // An address that can be used in a non-macro load or store
367 Info.setAllowsMemory();
368 return true;
369 case 'C': // Capability register
370 Info.setAllowsRegister();
371 return IsCHERI;
372 case 'Z':
373 if (Name[1] == 'C') { // An address usable by ll, and sc.
374 Info.setAllowsMemory();
375 Name++; // Skip over 'Z'.
376 return true;
377 }
378 return false;
379 }
380 }

382 std::string convertConstraint(const char *&Constraint) const override {
383 std::string R;
384 switch (*Constraint) {
385 case 'Z': // Two-character constraint; add "^" hint for later parsing.
386 if (Constraint[1] == 'C') {
387 R = std::string("^") + std::string(Constraint, 2);
388 Constraint++;
389 return R;
390 }
391 break;
392 }
393 return TargetInfo::convertConstraint(Constraint);
394 }

402 const char *getClobbers() const override {
403 // In GCC, $1 is not widely used in generated code (it's used only in a few
404 // specific situations), so there is no real need for users to add it to
405 // the clobbers list if they want to use it in their inline assembly code.
406 //
407 // In LLVM, $1 is treated as a normal GPR and is always allocatable during
408 // code generation, so using it in inline assembly without adding it to the
409 // clobbers list can cause conflicts between the inline assembly code and
410 // the surrounding generated code.
411 //
412 // Another problem is that LLVM is allowed to choose $1 for inline assembly
413 // operands, which will conflict with the ".set at" assembler option (which
414 // we use only for inline assembly, in order to maintain compatibility with
415 // GCC) and will also conflict with the user's usage of $1.
416 //
417 // The easiest way to avoid these conflicts and keep $1 as an allocatable
418 // register for generated code is to automatically clobber $1 for all inline
419 // assembly code.
420 //
421 // FIXME: We should automatically clobber $1 only for inline assembly code
422 // which actually uses it. This would allow LLVM to use $1 for inline
423 // assembly operands if the user's assembly code doesn't use it.
424 return "~{$1}";
425 }

427 bool handleTargetFeatures(std::vector<std::string> &Features,
428 DiagnosticsEngine &Diags) override {
429 IsMips16 = false;
430 IsMicromips = false;
431 IsNan2008 = isIEEE754_2008Default();
432 IsAbs2008 = isIEEE754_2008Default();
433 IsSingleFloat = false;
434 FloatABI = HardFloat;
435 DspRev = NoDSP;
436 FPMode = isFP64Default() ? FP64 : FPXX;
437 bool CapSizeFeatureFound = false;
438
439 for (const auto &Feature : Features) {
440 if (Feature == "+single-float")
441 IsSingleFloat = true;
442 else if (Feature == "+soft-float")
443 FloatABI = SoftFloat;
444 else if (Feature == "+mips16")
445 IsMips16 = true;
446 else if (Feature == "+micromips")
447 IsMicromips = true;
448 else if (Feature == "+dsp")
449 DspRev = std::max(DspRev, DSP1);
450 else if (Feature == "+dspr2")
451 DspRev = std::max(DspRev, DSP2);
452 else if (Feature == "+msa")
453 HasMSA = true;
454 else if (Feature == "+nomadd4")
455 DisableMadd4 = true;
456 else if (Feature == "+fp64")
457 FPMode = FP64;
458 else if (Feature == "-fp64")
459 FPMode = FP32;
460 else if (Feature == "+fpxx")
461 FPMode = FPXX;
462 else if (Feature == "+nan2008")
463 IsNan2008 = true;
464 else if (Feature == "+chericap")
465 IsCHERI = true;
466 else if (Feature == "+cheri64") {
467 CapSizeFeatureFound = true;
468 CapSize = 64;
469 IsCHERI = true;
470 } else if (Feature == "+cheri128") {
471 CapSizeFeatureFound = true;
472 CapSize = 128;
473 IsCHERI = true;
474 } else if (Feature == "+cheri256") {
475 CapSizeFeatureFound = true;
476 CapSize = 256;
477 IsCHERI = true;
478 } else if (Feature == "-nan2008")
479 IsNan2008 = false;
480 else if (Feature == "+abs2008")
481 IsAbs2008 = true;
482 else if (Feature == "-abs2008")
483 IsAbs2008 = false;
484 else if (Feature == "+noabicalls")
485 IsNoABICalls = true;
486 else if (Feature == "+use-indirect-jump-hazard")
487 UseIndirectJumpHazard = true;
488 }
489
490 if (IsCHERI) {
491 // If we have an implicit size, assume cheri128
492 if (!CapSizeFeatureFound) {
493 Features.push_back("+cheri128");
494 CapSize = 128;
495 }
496 assert(CapSize > 0);
497 SuitableAlign = std::max(SuitableAlign, (unsigned short)CapSize);
498 DefaultAlignForAttributeAligned =
499 std::max(DefaultAlignForAttributeAligned, (unsigned short)CapSize);
500 }
501
502 setDataLayout();
503
504 return true;
505 }
