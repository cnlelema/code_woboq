28TargetInfo::TargetInfo(const llvm::Triple &T) : TargetOpts(), Triple(T) {
29 // Set defaults. Defaults are set for a 32-bit RISC platform, like PPC or
30 // SPARC. These should be overridden by concrete targets as needed.
31 BigEndian = !T.isLittleEndian();
32 TLSSupported = true;
33 VLASupported = true;
34 NoAsmVariants = false;
35 HasLegalHalfType = false;
36 HasFloat128 = false;
37 HasFloat16 = false;
38 PointerWidth = PointerAlign = 32;
39 BoolWidth = BoolAlign = 8;
40 IntWidth = IntAlign = 32;
41 LongWidth = LongAlign = 32;
42 LongLongWidth = LongLongAlign = 64;
43
44 // Fixed point default bit widths
45 ShortAccumWidth = ShortAccumAlign = 16;
46 AccumWidth = AccumAlign = 32;
47 LongAccumWidth = LongAccumAlign = 64;
48 ShortFractWidth = ShortFractAlign = 8;
49 FractWidth = FractAlign = 16;
50 LongFractWidth = LongFractAlign = 32;
51
52 // Fixed point default integral and fractional bit sizes
53 // We give the _Accum 1 fewer fractional bits than their corresponding _Fract
54 // types by default to have the same number of fractional bits between _Accum
55 // and _Fract types.
56 PaddingOnUnsignedFixedPoint = false;
57 ShortAccumScale = 7;
58 AccumScale = 15;
59 LongAccumScale = 31;
60
61 SuitableAlign = 64;
62 DefaultAlignForAttributeAligned = 128;
63 MinGlobalAlign = 0;
64 // From the glibc documentation, on GNU systems, malloc guarantees 16-byte
65 // alignment on 64-bit systems and 8-byte alignment on 32-bit systems. See
66 // https://www.gnu.org/software/libc/manual/html_node/Malloc-Examples.html.
67 // This alignment guarantee also applies to Windows and Android.
68 if (T.isGNUEnvironment() || T.isWindowsMSVCEnvironment() || T.isAndroid())
69 NewAlign = Triple.isArch64Bit() ? 128 : Triple.isArch32Bit() ? 64 : 0;
70 else
71 NewAlign = 0; // Infer from basic type alignment.
72 HalfWidth = 16;
73 HalfAlign = 16;
74 FloatWidth = 32;
75 FloatAlign = 32;
76 DoubleWidth = 64;
77 DoubleAlign = 64;
78 LongDoubleWidth = 64;
79 LongDoubleAlign = 64;
80 Float128Align = 128;
81 LargeArrayMinWidth = 0;
82 LargeArrayAlign = 0;
83 MaxAtomicPromoteWidth = MaxAtomicInlineWidth = 0;
84 MaxVectorAlign = 0;
85 MaxTLSAlign = 0;
86 SimdDefaultAlign = 0;
87 SizeType = UnsignedLong;
88 PtrDiffType = SignedLong;
89 IntMaxType = SignedLongLong;
90 IntPtrType = SignedLong;
91 WCharType = SignedInt;
92 WIntType = SignedInt;
93 Char16Type = UnsignedShort;
94 Char32Type = UnsignedInt;
95 Int64Type = SignedLongLong;
96 SigAtomicType = SignedInt;
97 ProcessIDType = SignedInt;
98 UseSignedCharForObjCBool = true;
99 UseBitFieldTypeAlignment = true;
100 UseZeroLengthBitfieldAlignment = false;
101 UseExplicitBitFieldAlignment = true;
102 ZeroLengthBitfieldBoundary = 0;
103 HalfFormat = &llvm::APFloat::IEEEhalf();
104 FloatFormat = &llvm::APFloat::IEEEsingle();
105 DoubleFormat = &llvm::APFloat::IEEEdouble();
106 LongDoubleFormat = &llvm::APFloat::IEEEdouble();
107 Float128Format = &llvm::APFloat::IEEEquad();
108 MCountName = "mcount";
109 RegParmMax = 0;
110 SSERegParmMax = 0;
111 HasAlignMac68kSupport = false;
112 HasBuiltinMSVaList = false;
113 IsRenderScriptTarget = false;
114
115 // Default to no types using fpret.
116 RealTypeUsesObjCFPRet = 0;
117
118 // Default to not using fp2ret for __Complex long double
119 ComplexLongDoubleUsesFP2Ret = false;
120
121 // Set the C++ ABI based on the triple.
122 TheCXXABI.set(Triple.isKnownWindowsMSVCEnvironment()
123 ? TargetCXXABI::Microsoft
124 : TargetCXXABI::GenericItanium);
125
126 // Default to an empty address space map.
127 AddrSpaceMap = &DefaultAddrSpaceMap;
128 UseAddrSpaceMapMangling = false;
129
130 // Default to an unknown platform name.
131 PlatformName = "unknown";
132 PlatformMinVersion = VersionTuple();
133}

268TargetInfo::RealType TargetInfo::getRealTypeByWidth(unsigned BitWidth) const {
269 if (getFloatWidth() == BitWidth)
270 return Float;
271 if (getDoubleWidth() == BitWidth)
272 return Double;
273
274 switch (BitWidth) {
275 case 96:
276 if (&getLongDoubleFormat() == &llvm::APFloat::x87DoubleExtended())
277 return LongDouble;
278 break;
279 case 128:
280 if (&getLongDoubleFormat() == &llvm::APFloat::PPCDoubleDouble() ||
281 &getLongDoubleFormat() == &llvm::APFloat::IEEEquad())
282 return LongDouble;
283 if (hasFloat128Type())
284 return Float128;
285 break;
286 }
287
288 return NoFloat;
289}

333/// adjust - Set forced language options.
334/// Apply changes to the target information with respect to certain
335/// language options which change the target configuration and adjust
336/// the language based on the target options where applicable.
337void TargetInfo::adjust(LangOptions &Opts) {
338 if (Opts.NoBitFieldTypeAlign)
339 UseBitFieldTypeAlignment = false;
340
341 switch (Opts.WCharSize) {
342 default: llvm_unreachable("invalid wchar_t width");
343 case 0: break;
344 case 1: WCharType = Opts.WCharIsSigned ? SignedChar : UnsignedChar; break;
345 case 2: WCharType = Opts.WCharIsSigned ? SignedShort : UnsignedShort; break;
346 case 4: WCharType = Opts.WCharIsSigned ? SignedInt : UnsignedInt; break;
347 }
348
349 if (Opts.AlignDouble) {
350 DoubleAlign = LongLongAlign = 64;
351 LongDoubleAlign = 64;
352 }
353
354 if (Opts.OpenCL) {
355 // OpenCL C requires specific widths for types, irrespective of
356 // what these normally are for the target.
357 // We also define long long and long double here, although the
358 // OpenCL standard only mentions these as "reserved".
359 IntWidth = IntAlign = 32;
360 LongWidth = LongAlign = 64;
361 LongLongWidth = LongLongAlign = 128;
362 HalfWidth = HalfAlign = 16;
363 FloatWidth = FloatAlign = 32;
364
365 // Embedded 32-bit targets (OpenCL EP) might have double C type
366 // defined as float. Let's not override this as it might lead
367 // to generating illegal code that uses 64bit doubles.
368 if (DoubleWidth != FloatWidth) {
369 DoubleWidth = DoubleAlign = 64;
370 DoubleFormat = &llvm::APFloat::IEEEdouble();
371 }
372 LongDoubleWidth = LongDoubleAlign = 128;
373
374 unsigned MaxPointerWidth = getMaxPointerRange();
375 assert(MaxPointerWidth == 32 || MaxPointerWidth == 64);
376 bool Is32BitArch = MaxPointerWidth == 32;
377 SizeType = Is32BitArch ? UnsignedInt : UnsignedLong;
378 PtrDiffType = Is32BitArch ? SignedInt : SignedLong;
379 IntPtrType = Is32BitArch ? SignedInt : SignedLong;
380
381 IntMaxType = SignedLongLong;
382 Int64Type = SignedLong;
383
384 HalfFormat = &llvm::APFloat::IEEEhalf();
385 FloatFormat = &llvm::APFloat::IEEEsingle();
386 LongDoubleFormat = &llvm::APFloat::IEEEquad();
387 }
388
389 if (Opts.NewAlignOverride)
390 NewAlign = Opts.NewAlignOverride * getCharWidth();
391
392 // Each unsigned fixed point type has the same number of fractional bits as
393 // its corresponding signed type.
394 PaddingOnUnsignedFixedPoint |= Opts.PaddingOnUnsignedFixedPoint;
395 CheckFixedPointBits();
396}

450/// isValidGCCRegisterName - Returns whether the passed in string
451/// is a valid register name according to GCC. This is used by Sema for
452/// inline asm statements.
453bool TargetInfo::isValidGCCRegisterName(StringRef Name) const {
454 if (Name.empty())
455 return false;
456
457 // Get rid of any register prefix.
458 Name = removeGCCRegisterPrefix(Name);
459 if (Name.empty())
460 return false;
461
462 ArrayRef<const char *> Names = getGCCRegNames();
463
464 // If we have a number it maps to an entry in the register name array.
465 if (isDigit(Name[0])) {
466 unsigned n;
467 if (!Name.getAsInteger(0, n))
468 return n < Names.size();
469 }
470
471 // Check register names.
472 if (llvm::is_contained(Names, Name))
473 return true;
474
475 // Check any additional names that we have.
476 for (const AddlRegName &ARN : getGCCAddlRegNames())
477 for (const char *AN : ARN.Names) {
478 if (!AN)
479 break;
480 // Make sure the register that the additional name is for is within
481 // the bounds of the register names from above.
482 if (AN == Name && ARN.RegNum < Names.size())
483 return true;
484 }
485
486 // Now check aliases.
487 for (const GCCRegAlias &GRA : getGCCRegAliases())
488 for (const char *A : GRA.Aliases) {
489 if (!A)
490 break;
491 if (A == Name)
492 return true;
493 }
494
495 return false;
496}
497
498StringRef TargetInfo::getNormalizedGCCRegisterName(StringRef Name,
499 bool ReturnCanonical) const {
500 assert(isValidGCCRegisterName(Name) && "Invalid register passed in");
501
502 // Get rid of any register prefix.
503 Name = removeGCCRegisterPrefix(Name);
504
505 ArrayRef<const char *> Names = getGCCRegNames();
506
507 // First, check if we have a number.
508 if (isDigit(Name[0])) {
509 unsigned n;
510 if (!Name.getAsInteger(0, n)) {
511 assert(n < Names.size() && "Out of bounds register number!");
512 return Names[n];
513 }
514 }
515
516 // Check any additional names that we have.
517 for (const AddlRegName &ARN : getGCCAddlRegNames())
518 for (const char *AN : ARN.Names) {
519 if (!AN)
520 break;
521 // Make sure the register that the additional name is for is within
522 // the bounds of the register names from above.
523 if (AN == Name && ARN.RegNum < Names.size())
524 return ReturnCanonical ? Names[ARN.RegNum] : Name;
525 }
526
527 // Now check aliases.
528 for (const GCCRegAlias &RA : getGCCRegAliases())
529 for (const char *A : RA.Aliases) {
530 if (!A)
531 break;
532 if (A == Name)
533 return RA.Register;
534 }
535
536 return Name;
537}

539bool TargetInfo::validateOutputConstraint(ConstraintInfo &Info) const {
540 const char *Name = Info.getConstraintStr().c_str();
541 // An output constraint must start with '=' or '+'
542 if (*Name != '=' && *Name != '+')
543 return false;
544
545 if (*Name == '+')
546 Info.setIsReadWrite();
547
548 Name++;
549 while (*Name) {
550 switch (*Name) {
551 default:
552 if (!validateAsmConstraint(Name, Info)) {
553 // FIXME: We temporarily return false
554 // so we can add more constraints as we hit it.
555 // Eventually, an unknown constraint should just be treated as 'g'.
556 return false;
557 }
558 break;
559 case '&': // early clobber.
560 Info.setEarlyClobber();
561 break;
562 case '%': // commutative.
563 // FIXME: Check that there is a another register after this one.
564 break;
565 case 'r': // general register.
566 Info.setAllowsRegister();
567 break;
568 case 'm': // memory operand.
569 if (areAllPointersCapabilities())
570 break;
571 LLVM_FALLTHROUGH;
572 case 'o': // offsetable memory operand.
573 case 'V': // non-offsetable memory operand.
574 case '<': // autodecrement memory operand.
575 case '>': // autoincrement memory operand.
576 Info.setAllowsMemory();
577 break;
578 case 'g': // general register, memory operand or immediate integer.
579 case 'X': // any operand.
580 Info.setAllowsRegister();
581 Info.setAllowsMemory();
582 break;
583 case ',': // multiple alternative constraint. Pass it.
584 // Handle additional optional '=' or '+' modifiers.
585 if (Name[1] == '=' || Name[1] == '+')
586 Name++;
587 break;
588 case '#': // Ignore as constraint.
589 while (Name[1] && Name[1] != ',')
590 Name++;
591 break;
592 case '?': // Disparage slightly code.
593 case '!': // Disparage severely.
594 case '*': // Ignore for choosing register preferences.
595 case 'i': // Ignore i,n,E,F as output constraints (match from the other
596 // chars)
597 case 'n':
598 case 'E':
599 case 'F':
600 break; // Pass them.
601 }
602
603 Name++;
604 }
605
606 // Early clobber with a read-write constraint which doesn't permit registers
607 // is invalid.
608 if (Info.earlyClobber() && Info.isReadWrite() && !Info.allowsRegister())
609 return false;
610
611 // If a constraint allows neither memory nor register operands it contains
612 // only modifiers. Reject it.
613 return Info.allowsMemory() || Info.allowsRegister();
614}

616bool TargetInfo::resolveSymbolicName(const char *&Name,
617 ArrayRef<ConstraintInfo> OutputConstraints,
618 unsigned &Index) const {
619 assert(*Name == '[' && "Symbolic name did not start with '['");
620 Name++;
621 const char *Start = Name;
622 while (*Name && *Name != ']')
623 Name++;
624
625 if (!*Name) {
626 // Missing ']'
627 return false;
628 }
629
630 std::string SymbolicName(Start, Name - Start);
631
632 for (Index = 0; Index != OutputConstraints.size(); ++Index)
633 if (SymbolicName == OutputConstraints[Index].getName())
634 return true;
635
636 return false;
637}

639bool TargetInfo::validateInputConstraint(
640 MutableArrayRef<ConstraintInfo> OutputConstraints,
641 ConstraintInfo &Info) const {
642 const char *Name = Info.ConstraintStr.c_str();
643
644 if (!*Name)
645 return false;
646
647 while (*Name) {
648 switch (*Name) {
649 default:
650 // Check if we have a matching constraint
651 if (*Name >= '0' && *Name <= '9') {
652 const char *DigitStart = Name;
653 while (Name[1] >= '0' && Name[1] <= '9')
654 Name++;
655 const char *DigitEnd = Name;
656 unsigned i;
657 if (StringRef(DigitStart, DigitEnd - DigitStart + 1)
658 .getAsInteger(10, i))
659 return false;
660
661 // Check if matching constraint is out of bounds.
662 if (i >= OutputConstraints.size()) return false;
663
664 // A number must refer to an output only operand.
665 if (OutputConstraints[i].isReadWrite())
666 return false;
667
668 // If the constraint is already tied, it must be tied to the
669 // same operand referenced to by the number.
670 if (Info.hasTiedOperand() && Info.getTiedOperand() != i)
671 return false;
672
673 // The constraint should have the same info as the respective
674 // output constraint.
675 Info.setTiedOperand(i, OutputConstraints[i]);
676 } else if (!validateAsmConstraint(Name, Info)) {
677 // FIXME: This error return is in place temporarily so we can
678 // add more constraints as we hit it. Eventually, an unknown
679 // constraint should just be treated as 'g'.
680 return false;
681 }
682 break;
683 case '[': {
684 unsigned Index = 0;
685 if (!resolveSymbolicName(Name, OutputConstraints, Index))
686 return false;
687
688 // If the constraint is already tied, it must be tied to the
689 // same operand referenced to by the number.
690 if (Info.hasTiedOperand() && Info.getTiedOperand() != Index)
691 return false;
692
693 // A number must refer to an output only operand.
694 if (OutputConstraints[Index].isReadWrite())
695 return false;
696
697 Info.setTiedOperand(Index, OutputConstraints[Index]);
698 break;
699 }
700 case '%': // commutative
701 // FIXME: Fail if % is used with the last operand.
702 break;
703 case 'i': // immediate integer.
704 break;
705 case 'n': // immediate integer with a known value.
706 Info.setRequiresImmediate();
707 break;
708 case 'I': // Various constant constraints with target-specific meanings.
709 case 'J':
710 case 'K':
711 case 'L':
712 case 'M':
713 case 'N':
714 case 'O':
715 case 'P':
716 if (!validateAsmConstraint(Name, Info))
717 return false;
718 break;
719 case 'r': // general register.
720 Info.setAllowsRegister();
721 break;
722 case 'm': // memory operand.
723 case 'o': // offsettable memory operand.
724 case 'V': // non-offsettable memory operand.
725 case '<': // autodecrement memory operand.
726 case '>': // autoincrement memory operand.
727 Info.setAllowsMemory();
728 break;
729 case 'g': // general register, memory operand or immediate integer.
730 case 'X': // any operand.
731 Info.setAllowsRegister();
732 Info.setAllowsMemory();
733 break;
734 case 'E': // immediate floating point.
735 case 'F': // immediate floating point.
736 case 'p': // address operand.
737 break;
738 case ',': // multiple alternative constraint. Ignore comma.
739 break;
740 case '#': // Ignore as constraint.
741 while (Name[1] && Name[1] != ',')
742 Name++;
743 break;
744 case '?': // Disparage slightly code.
745 case '!': // Disparage severely.
746 case '*': // Ignore for choosing register preferences.
747 break; // Pass them.
748 }
749
750 Name++;
751 }
752
753 return true;
754}

756void TargetInfo::CheckFixedPointBits() const {
757 // Check that the number of fractional and integral bits (and maybe sign) can
758 // fit into the bits given for a fixed point type.
759 assert(ShortAccumScale + getShortAccumIBits() + 1 <= ShortAccumWidth);
760 assert(AccumScale + getAccumIBits() + 1 <= AccumWidth);
761 assert(LongAccumScale + getLongAccumIBits() + 1 <= LongAccumWidth);
762 assert(getUnsignedShortAccumScale() + getUnsignedShortAccumIBits() <=
763 ShortAccumWidth);
764 assert(getUnsignedAccumScale() + getUnsignedAccumIBits() <= AccumWidth);
765 assert(getUnsignedLongAccumScale() + getUnsignedLongAccumIBits() <=
766 LongAccumWidth);
767
768 assert(getShortFractScale() + 1 <= ShortFractWidth);
769 assert(getFractScale() + 1 <= FractWidth);
770 assert(getLongFractScale() + 1 <= LongFractWidth);
771 assert(getUnsignedShortFractScale() <= ShortFractWidth);
772 assert(getUnsignedFractScale() <= FractWidth);
773 assert(getUnsignedLongFractScale() <= LongFractWidth);
774
775 // Each unsigned fract type has either the same number of fractional bits
776 // as, or one more fractional bit than, its corresponding signed fract type.
777 assert(getShortFractScale() == getUnsignedShortFractScale() ||
778 getShortFractScale() == getUnsignedShortFractScale() - 1);
779 assert(getFractScale() == getUnsignedFractScale() ||
780 getFractScale() == getUnsignedFractScale() - 1);
781 assert(getLongFractScale() == getUnsignedLongFractScale() ||
782 getLongFractScale() == getUnsignedLongFractScale() - 1);
783
784 // When arranged in order of increasing rank (see 6.3.1.3a), the number of
785 // fractional bits is nondecreasing for each of the following sets of
786 // fixed-point types:
787 // - signed fract types
788 // - unsigned fract types
789 // - signed accum types
790 // - unsigned accum types.
791 assert(getLongFractScale() >= getFractScale() &&
792 getFractScale() >= getShortFractScale());
793 assert(getUnsignedLongFractScale() >= getUnsignedFractScale() &&
794 getUnsignedFractScale() >= getUnsignedShortFractScale());
795 assert(LongAccumScale >= AccumScale && AccumScale >= ShortAccumScale);
796 assert(getUnsignedLongAccumScale() >= getUnsignedAccumScale() &&
797 getUnsignedAccumScale() >= getUnsignedShortAccumScale());
798
799 // When arranged in order of increasing rank (see 6.3.1.3a), the number of
800 // integral bits is nondecreasing for each of the following sets of
801 // fixed-point types:
802 // - signed accum types
803 // - unsigned accum types
804 assert(getLongAccumIBits() >= getAccumIBits() &&
805 getAccumIBits() >= getShortAccumIBits());
806 assert(getUnsignedLongAccumIBits() >= getUnsignedAccumIBits() &&
807 getUnsignedAccumIBits() >= getUnsignedShortAccumIBits());
808
809 // Each signed accum type has at least as many integral bits as its
810 // corresponding unsigned accum type.
811 assert(getShortAccumIBits() >= getUnsignedShortAccumIBits());
812 assert(getAccumIBits() >= getUnsignedAccumIBits());
813 assert(getLongAccumIBits() >= getUnsignedLongAccumIBits());
814}
