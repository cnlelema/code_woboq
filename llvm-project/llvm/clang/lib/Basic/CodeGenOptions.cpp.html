23bool CodeGenOptions::isNoBuiltinFunc(const char *Name) const {
24 StringRef FuncName(Name);
25 for (unsigned i = 0, e = NoBuiltinFuncs.size(); i != e; ++i)
26 if (FuncName.equals(NoBuiltinFuncs[i]))
27 return true;
28 return false;
29}
