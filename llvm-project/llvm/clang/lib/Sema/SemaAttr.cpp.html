50void Sema::AddAlignmentAttributesForRecord(RecordDecl *RD) {
51 // If there is no pack value, we don't need any attributes.
52 if (!PackStack.CurrentValue)
53 return;
54
55 // Otherwise, check to see if we need a max field alignment attribute.
56 if (unsigned Alignment = PackStack.CurrentValue) {
57 if (Alignment == Sema::kMac68kAlignmentSentinel)
58 RD->addAttr(AlignMac68kAttr::CreateImplicit(Context));
59 else
60 RD->addAttr(MaxFieldAlignmentAttr::CreateImplicit(Context,
61 Alignment * 8));
62 }
63 if (PackIncludeStack.empty())
64 return;
65 // The #pragma pack affected a record in an included file, so Clang should
66 // warn when that pragma was written in a file that included the included
67 // file.
68 for (auto &PackedInclude : llvm::reverse(PackIncludeStack)) {
69 if (PackedInclude.CurrentPragmaLocation != PackStack.CurrentPragmaLocation)
70 break;
71 if (PackedInclude.HasNonDefaultValue)
72 PackedInclude.ShouldWarnOnInclude = true;
73 }
74}

76void Sema::AddMsStructLayoutForRecord(RecordDecl *RD) {
77 if (MSStructPragmaOn)
78 RD->addAttr(MSStructAttr::CreateImplicit(Context));
79
80 // FIXME: We should merge AddAlignmentAttributesForRecord with
81 // AddMsStructLayoutForRecord into AddPragmaAttributesForRecord, which takes
82 // all active pragmas and applies them as attributes to class definitions.
83 if (VtorDispStack.CurrentValue != getLangOpts().VtorDispMode)
84 RD->addAttr(
85 MSVtorDispAttr::CreateImplicit(Context, VtorDispStack.CurrentValue));
86}

88void Sema::ActOnPragmaOptionsAlign(PragmaOptionsAlignKind Kind,
89 SourceLocation PragmaLoc) {
90 PragmaMsStackAction Action = Sema::PSK_Reset;
91 unsigned Alignment = 0;
92 switch (Kind) {
93 // For all targets we support native and natural are the same.
94 //
95 // FIXME: This is not true on Darwin/PPC.
96 case POAK_Native:
97 case POAK_Power:
98 case POAK_Natural:
99 Action = Sema::PSK_Push_Set;
100 Alignment = 0;
101 break;
102
103 // Note that '#pragma options align=packed' is not equivalent to attribute
104 // packed, it has a different precedence relative to attribute aligned.
105 case POAK_Packed:
106 Action = Sema::PSK_Push_Set;
107 Alignment = 1;
108 break;
109
110 case POAK_Mac68k:
111 // Check if the target supports this.
112 if (!this->Context.getTargetInfo().hasAlignMac68kSupport()) {
113 Diag(PragmaLoc, diag::err_pragma_options_align_mac68k_target_unsupported);
114 return;
115 }
116 Action = Sema::PSK_Push_Set;
117 Alignment = Sema::kMac68kAlignmentSentinel;
118 break;
119
120 case POAK_Reset:
121 // Reset just pops the top of the stack, or resets the current alignment to
122 // default.
123 Action = Sema::PSK_Pop;
124 if (PackStack.Stack.empty()) {
125 if (PackStack.CurrentValue) {
126 Action = Sema::PSK_Reset;
127 } else {
128 Diag(PragmaLoc, diag::warn_pragma_options_align_reset_failed)
129 << "stack empty";
130 return;
131 }
132 }
133 break;
134 }
135
136 PackStack.Act(PragmaLoc, Action, StringRef(), Alignment);
137}

140 PragmaClangSectionKind SecKind, StringRef SecName) {
141 PragmaClangSection *CSec;
142 switch (SecKind) {
143 case PragmaClangSectionKind::PCSK_BSS:
144 CSec = &PragmaClangBSSSection;
145 break;
146 case PragmaClangSectionKind::PCSK_Data:
147 CSec = &PragmaClangDataSection;
148 break;
149 case PragmaClangSectionKind::PCSK_Rodata:
150 CSec = &PragmaClangRodataSection;
151 break;
152 case PragmaClangSectionKind::PCSK_Text:
153 CSec = &PragmaClangTextSection;
154 break;
155 default:
156 llvm_unreachable("invalid clang section kind");
157 }
158
159 if (Action == PragmaClangSectionAction::PCSA_Clear) {
160 CSec->Valid = false;
161 return;
162 }
163
164 CSec->Valid = true;
165 CSec->SectionName = SecName;
166 CSec->PragmaLocation = PragmaLoc;
167}

169void Sema::ActOnPragmaPack(SourceLocation PragmaLoc, PragmaMsStackAction Action,
170 StringRef SlotLabel, Expr *alignment) {
171 Expr *Alignment = static_cast<Expr *>(alignment);
172
173 // If specified then alignment must be a "small" power of two.
174 unsigned AlignmentVal = 0;
175 if (Alignment) {
176 llvm::APSInt Val;
177
178 // pack(0) is like pack(), which just works out since that is what
179 // we use 0 for in PackAttr.
180 if (Alignment->isTypeDependent() ||
181 Alignment->isValueDependent() ||
182 !Alignment->isIntegerConstantExpr(Val, Context) ||
183 !(Val == 0 || Val.isPowerOf2()) ||
184 Val.getZExtValue() > 16) {
185 Diag(PragmaLoc, diag::warn_pragma_pack_invalid_alignment);
186 return; // Ignore
187 }
188
189 AlignmentVal = (unsigned) Val.getZExtValue();
190 }
191 if (Action == Sema::PSK_Show) {
192 // Show the current alignment, making sure to show the right value
193 // for the default.
194 // FIXME: This should come from the target.
195 AlignmentVal = PackStack.CurrentValue;
196 if (AlignmentVal == 0)
197 AlignmentVal = 8;
198 if (AlignmentVal == Sema::kMac68kAlignmentSentinel)
199 Diag(PragmaLoc, diag::warn_pragma_pack_show) << "mac68k";
200 else
201 Diag(PragmaLoc, diag::warn_pragma_pack_show) << AlignmentVal;
202 }
203 // MSDN, C/C++ Preprocessor Reference > Pragma Directives > pack:
204 // "#pragma pack(pop, identifier, n) is undefined"
205 if (Action & Sema::PSK_Pop) {
206 if (Alignment && !SlotLabel.empty())
207 Diag(PragmaLoc, diag::warn_pragma_pack_pop_identifier_and_alignment);
208 if (PackStack.Stack.empty())
209 Diag(PragmaLoc, diag::warn_pragma_pop_failed) << "pack" << "stack empty";
210 }
211
212 PackStack.Act(PragmaLoc, Action, SlotLabel, AlignmentVal);
213}

220void Sema::DiagnoseNonDefaultPragmaPack(PragmaPackDiagnoseKind Kind,
221 SourceLocation IncludeLoc) {
222 if (Kind == PragmaPackDiagnoseKind::NonDefaultStateAtInclude) {
223 SourceLocation PrevLocation = PackStack.CurrentPragmaLocation;
224 // Warn about non-default alignment at #includes (without redundant
225 // warnings for the same directive in nested includes).
226 // The warning is delayed until the end of the file to avoid warnings
227 // for files that don't have any records that are affected by the modified
228 // alignment.
229 bool HasNonDefaultValue =
230 PackStack.hasValue() &&
231 (PackIncludeStack.empty() ||
232 PackIncludeStack.back().CurrentPragmaLocation != PrevLocation);
233 PackIncludeStack.push_back(
234 {PackStack.CurrentValue,
235 PackStack.hasValue() ? PrevLocation : SourceLocation(),
236 HasNonDefaultValue, /*ShouldWarnOnInclude*/ false});
237 return;
238 }
239
240 assert(Kind == PragmaPackDiagnoseKind::ChangedStateAtExit && "invalid kind");
241 PackIncludeState PrevPackState = PackIncludeStack.pop_back_val();
242 if (PrevPackState.ShouldWarnOnInclude) {
243 // Emit the delayed non-default alignment at #include warning.
244 Diag(IncludeLoc, diag::warn_pragma_pack_non_default_at_include);
245 Diag(PrevPackState.CurrentPragmaLocation, diag::note_pragma_pack_here);
246 }
247 // Warn about modified alignment after #includes.
248 if (PrevPackState.CurrentValue != PackStack.CurrentValue) {
249 Diag(IncludeLoc, diag::warn_pragma_pack_modified_after_include);
250 Diag(PackStack.CurrentPragmaLocation, diag::note_pragma_pack_here);
251 }
252}

254void Sema::DiagnoseUnterminatedPragmaPack() {
255 if (PackStack.Stack.empty())
256 return;
257 bool IsInnermost = true;
258 for (const auto &StackSlot : llvm::reverse(PackStack.Stack)) {
259 Diag(StackSlot.PragmaPushLocation, diag::warn_pragma_pack_no_pop_eof);
260 // The user might have already reset the alignment, so suggest replacing
261 // the reset with a pop.
262 if (IsInnermost && PackStack.CurrentValue == PackStack.DefaultValue) {
263 DiagnosticBuilder DB = Diag(PackStack.CurrentPragmaLocation,
264 diag::note_pragma_pack_pop_instead_reset);
265 SourceLocation FixItLoc = Lexer::findLocationAfterToken(
266 PackStack.CurrentPragmaLocation, tok::l_paren, SourceMgr, LangOpts,
267 /*SkipTrailing=*/false);
268 if (FixItLoc.isValid())
269 DB << FixItHint::CreateInsertion(FixItLoc, "pop");
270 }
271 IsInnermost = false;
272 }
273}

311template<typename ValueType>
312void Sema::PragmaStack<ValueType>::Act(SourceLocation PragmaLocation,
313 PragmaMsStackAction Action,
314 llvm::StringRef StackSlotLabel,
315 ValueType Value) {
316 if (Action == PSK_Reset) {
317 CurrentValue = DefaultValue;
318 CurrentPragmaLocation = PragmaLocation;
319 return;
320 }
321 if (Action & PSK_Push)
322 Stack.emplace_back(StackSlotLabel, CurrentValue, CurrentPragmaLocation,
323 PragmaLocation);
324 else if (Action & PSK_Pop) {
325 if (!StackSlotLabel.empty()) {
326 // If we've got a label, try to find it and jump there.
327 auto I = llvm::find_if(llvm::reverse(Stack), [&](const Slot &x) {
328 return x.StackSlotLabel == StackSlotLabel;
329 });
330 // If we found the label so pop from there.
331 if (I != Stack.rend()) {
332 CurrentValue = I->Value;
333 CurrentPragmaLocation = I->PragmaLocation;
334 Stack.erase(std::prev(I.base()), Stack.end());
335 }
336 } else if (!Stack.empty()) {
337 // We do not have a label, just pop the last entry.
338 CurrentValue = Stack.back().Value;
339 CurrentPragmaLocation = Stack.back().PragmaLocation;
340 Stack.pop_back();
341 }
342 }
343 if (Action & PSK_Set) {
344 CurrentValue = Value;
345 CurrentPragmaLocation = PragmaLocation;
346 }
347}

349bool Sema::UnifySection(StringRef SectionName,
350 int SectionFlags,
351 DeclaratorDecl *Decl) {
352 auto Section = Context.SectionInfos.find(SectionName);
353 if (Section == Context.SectionInfos.end()) {
354 Context.SectionInfos[SectionName] =
355 ASTContext::SectionInfo(Decl, SourceLocation(), SectionFlags);
356 return false;
357 }
358 // A pre-declared section takes precedence w/o diagnostic.
359 if (Section->second.SectionFlags == SectionFlags ||
360 !(Section->second.SectionFlags & ASTContext::PSF_Implicit))
361 return false;
362 auto OtherDecl = Section->second.Decl;
363 Diag(Decl->getLocation(), diag::err_section_conflict)
364 << Decl << OtherDecl;
365 Diag(OtherDecl->getLocation(), diag::note_declared_at)
366 << OtherDecl->getName();
367 if (auto A = Decl->getAttr<SectionAttr>())
368 if (A->isImplicit())
369 Diag(A->getLocation(), diag::note_pragma_entered_here);
370 if (auto A = OtherDecl->getAttr<SectionAttr>())
371 if (A->isImplicit())
372 Diag(A->getLocation(), diag::note_pragma_entered_here);
373 return true;
374}
375
376bool Sema::UnifySection(StringRef SectionName,
377 int SectionFlags,
378 SourceLocation PragmaSectionLocation) {
379 auto Section = Context.SectionInfos.find(SectionName);
380 if (Section != Context.SectionInfos.end()) {
381 if (Section->second.SectionFlags == SectionFlags)
382 return false;
383 if (!(Section->second.SectionFlags & ASTContext::PSF_Implicit)) {
384 Diag(PragmaSectionLocation, diag::err_section_conflict)
385 << "this" << "a prior #pragma section";
386 Diag(Section->second.PragmaSectionLocation,
387 diag::note_pragma_entered_here);
388 return true;
389 }
390 }
391 Context.SectionInfos[SectionName] =
392 ASTContext::SectionInfo(nullptr, PragmaSectionLocation, SectionFlags);
393 return false;
394}

398 PragmaMsStackAction Action,

400 StringLiteral *SegmentName,

411 if (SegmentName &&
412 !checkSectionName(SegmentName->getBeginLoc(), SegmentName->getString()))
413 return;
414 Stack->Act(PragmaLocation, Action, StackSlotLabel, SegmentName);
415}

432void Sema::ActOnPragmaUnused(const Token &IdTok, Scope *curScope,
433 SourceLocation PragmaLoc) {
434
435 IdentifierInfo *Name = IdTok.getIdentifierInfo();
436 LookupResult Lookup(*this, Name, IdTok.getLocation(), LookupOrdinaryName);
437 LookupParsedName(Lookup, curScope, nullptr, true);
438
439 if (Lookup.empty()) {
440 Diag(PragmaLoc, diag::warn_pragma_unused_undeclared_var)
441 << Name << SourceRange(IdTok.getLocation());
442 return;
443 }
444
445 VarDecl *VD = Lookup.getAsSingle<VarDecl>();
446 if (!VD) {
447 Diag(PragmaLoc, diag::warn_pragma_unused_expected_var_arg)
448 << Name << SourceRange(IdTok.getLocation());
449 return;
450 }
451
452 // Warn if this was used before being marked unused.
453 if (VD->isUsed())
454 Diag(PragmaLoc, diag::warn_used_but_marked_unused) << Name;
455
456 VD->addAttr(UnusedAttr::CreateImplicit(Context, UnusedAttr::GNU_unused,
457 IdTok.getLocation()));
458}

474Optional<attr::SubjectMatchRule>
475getParentAttrMatcherRule(attr::SubjectMatchRule Rule) {
476 using namespace attr;
477 switch (Rule) {
478 default:
479 return None;
480#define ATTR_MATCH_RULE(Value, Spelling, IsAbstract)
481#define ATTR_MATCH_SUB_RULE(Value, Spelling, IsAbstract, Parent, IsNegated) \
482 case Value: \
483 return Parent;
484#include "clang/Basic/AttrSubMatchRulesList.inc"
485 }
486}
487
488bool isNegatedAttrMatcherSubRule(attr::SubjectMatchRule Rule) {
489 using namespace attr;
490 switch (Rule) {
491 default:
492 return false;
493#define ATTR_MATCH_RULE(Value, Spelling, IsAbstract)
494#define ATTR_MATCH_SUB_RULE(Value, Spelling, IsAbstract, Parent, IsNegated) \
495 case Value: \
496 return IsNegated;
497#include "clang/Basic/AttrSubMatchRulesList.inc"
498 }
499}

501CharSourceRange replacementRangeForListElement(const Sema &S,
502 SourceRange Range) {
503 // Make sure that the ',' is removed as well.
504 SourceLocation AfterCommaLoc = Lexer::findLocationAfterToken(
505 Range.getEnd(), tok::comma, S.getSourceManager(), S.getLangOpts(),
506 /*SkipTrailingWhitespaceAndNewLine=*/false);
507 if (AfterCommaLoc.isValid())
508 return CharSourceRange::getCharRange(Range.getBegin(), AfterCommaLoc);
509 else
510 return CharSourceRange::getTokenRange(Range);
511}

513std::string
514attrMatcherRuleListToString(ArrayRef<attr::SubjectMatchRule> Rules) {
515 std::string Result;
516 llvm::raw_string_ostream OS(Result);
517 for (const auto &I : llvm::enumerate(Rules)) {
518 if (I.index())
519 OS << (I.index() == Rules.size() - 1 ? ", and " : ", ");
520 OS << "'" << attr::getSubjectMatchRuleSpelling(I.value()) << "'";
521 }
522 return OS.str();
523}

527void Sema::ActOnPragmaAttributeAttribute(
528 ParsedAttr &Attribute, SourceLocation PragmaLoc,
529 attr::ParsedSubjectMatchRuleSet Rules) {
530 Attribute.setIsPragmaClangAttribute();
531 SmallVector<attr::SubjectMatchRule, 4> SubjectMatchRules;
532 // Gather the subject match rules that are supported by the attribute.
533 SmallVector<std::pair<attr::SubjectMatchRule, bool>, 4>
534 StrictSubjectMatchRuleSet;
535 Attribute.getMatchRules(LangOpts, StrictSubjectMatchRuleSet);
536
537 // Figure out which subject matching rules are valid.
538 if (StrictSubjectMatchRuleSet.empty()) {
539 // Check for contradicting match rules. Contradicting match rules are
540 // either:
541 // - a top-level rule and one of its sub-rules. E.g. variable and
542 // variable(is_parameter).
543 // - a sub-rule and a sibling that's negated. E.g.
544 // variable(is_thread_local) and variable(unless(is_parameter))
545 llvm::SmallDenseMap<int, std::pair<int, SourceRange>, 2>
546 RulesToFirstSpecifiedNegatedSubRule;
547 for (const auto &Rule : Rules) {
548 attr::SubjectMatchRule MatchRule = attr::SubjectMatchRule(Rule.first);
549 Optional<attr::SubjectMatchRule> ParentRule =
550 getParentAttrMatcherRule(MatchRule);
551 if (!ParentRule)
552 continue;
553 auto It = Rules.find(*ParentRule);
554 if (It != Rules.end()) {
555 // A sub-rule contradicts a parent rule.
556 Diag(Rule.second.getBegin(),
557 diag::err_pragma_attribute_matcher_subrule_contradicts_rule)
558 << attr::getSubjectMatchRuleSpelling(MatchRule)
559 << attr::getSubjectMatchRuleSpelling(*ParentRule) << It->second
560 << FixItHint::CreateRemoval(
561 replacementRangeForListElement(*this, Rule.second));
562 // Keep going without removing this rule as it won't change the set of
563 // declarations that receive the attribute.
564 continue;
565 }
566 if (isNegatedAttrMatcherSubRule(MatchRule))
567 RulesToFirstSpecifiedNegatedSubRule.insert(
568 std::make_pair(*ParentRule, Rule));
569 }
570 bool IgnoreNegatedSubRules = false;
571 for (const auto &Rule : Rules) {
572 attr::SubjectMatchRule MatchRule = attr::SubjectMatchRule(Rule.first);
573 Optional<attr::SubjectMatchRule> ParentRule =
574 getParentAttrMatcherRule(MatchRule);
575 if (!ParentRule)
576 continue;
577 auto It = RulesToFirstSpecifiedNegatedSubRule.find(*ParentRule);
578 if (It != RulesToFirstSpecifiedNegatedSubRule.end() &&
579 It->second != Rule) {
580 // Negated sub-rule contradicts another sub-rule.
581 Diag(
582 It->second.second.getBegin(),
583 diag::
584 err_pragma_attribute_matcher_negated_subrule_contradicts_subrule)
585 << attr::getSubjectMatchRuleSpelling(
586 attr::SubjectMatchRule(It->second.first))
587 << attr::getSubjectMatchRuleSpelling(MatchRule) << Rule.second
588 << FixItHint::CreateRemoval(
589 replacementRangeForListElement(*this, It->second.second));
590 // Keep going but ignore all of the negated sub-rules.
591 IgnoreNegatedSubRules = true;
592 RulesToFirstSpecifiedNegatedSubRule.erase(It);
593 }
594 }
595
596 if (!IgnoreNegatedSubRules) {
597 for (const auto &Rule : Rules)
598 SubjectMatchRules.push_back(attr::SubjectMatchRule(Rule.first));
599 } else {
600 for (const auto &Rule : Rules) {
601 if (!isNegatedAttrMatcherSubRule(attr::SubjectMatchRule(Rule.first)))
602 SubjectMatchRules.push_back(attr::SubjectMatchRule(Rule.first));
603 }
604 }
605 Rules.clear();
606 } else {
607 for (const auto &Rule : StrictSubjectMatchRuleSet) {
608 if (Rules.erase(Rule.first)) {
609 // Add the rule to the set of attribute receivers only if it's supported
610 // in the current language mode.
611 if (Rule.second)
612 SubjectMatchRules.push_back(Rule.first);
613 }
614 }
615 }
616
617 if (!Rules.empty()) {
618 auto Diagnostic =
619 Diag(PragmaLoc, diag::err_pragma_attribute_invalid_matchers)
620 << Attribute.getName();
621 SmallVector<attr::SubjectMatchRule, 2> ExtraRules;
622 for (const auto &Rule : Rules) {
623 ExtraRules.push_back(attr::SubjectMatchRule(Rule.first));
624 Diagnostic << FixItHint::CreateRemoval(
625 replacementRangeForListElement(*this, Rule.second));
626 }
627 Diagnostic << attrMatcherRuleListToString(ExtraRules);
628 }
629
630 if (PragmaAttributeStack.empty()) {
631 Diag(PragmaLoc, diag::err_pragma_attr_attr_no_push);
632 return;
633 }
634
635 PragmaAttributeStack.back().Entries.push_back(
636 {PragmaLoc, &Attribute, std::move(SubjectMatchRules), /*IsUsed=*/false});
637}

646void Sema::ActOnPragmaAttributePop(SourceLocation PragmaLoc,
647 const IdentifierInfo *Namespace) {
648 if (PragmaAttributeStack.empty()) {
649 Diag(PragmaLoc, diag::err_pragma_attribute_stack_mismatch) << 1;
650 return;
651 }
652
653 // Dig back through the stack trying to find the most recently pushed group
654 // that in Namespace. Note that this works fine if no namespace is present,
655 // think of push/pops without namespaces as having an implicit "nullptr"
656 // namespace.
657 for (size_t Index = PragmaAttributeStack.size(); Index;) {
658 --Index;
659 if (PragmaAttributeStack[Index].Namespace == Namespace) {
660 for (const PragmaAttributeEntry &Entry :
661 PragmaAttributeStack[Index].Entries) {
662 if (!Entry.IsUsed) {
663 assert(Entry.Attribute && "Expected an attribute");
664 Diag(Entry.Attribute->getLoc(), diag::warn_pragma_attribute_unused)
665 << *Entry.Attribute;
666 Diag(PragmaLoc, diag::note_pragma_attribute_region_ends_here);
667 }
668 }
669 PragmaAttributeStack.erase(PragmaAttributeStack.begin() + Index);
670 return;
671 }
672 }
673
674 if (Namespace)
675 Diag(PragmaLoc, diag::err_pragma_attribute_stack_mismatch)
676 << 0 << Namespace->getName();
677 else
678 Diag(PragmaLoc, diag::err_pragma_attribute_stack_mismatch) << 1;
679}
680
681void Sema::AddPragmaAttributes(Scope *S, Decl *D) {
682 if (PragmaAttributeStack.empty())
683 return;
684 for (auto &Group : PragmaAttributeStack) {
685 for (auto &Entry : Group.Entries) {
686 ParsedAttr *Attribute = Entry.Attribute;
687 assert(Attribute && "Expected an attribute");
688 assert(Attribute->isPragmaClangAttribute() &&
689 "expected #pragma clang attribute");
690
691 // Ensure that the attribute can be applied to the given declaration.
692 bool Applies = false;
693 for (const auto &Rule : Entry.MatchRules) {
694 if (Attribute->appliesToDecl(D, Rule)) {
695 Applies = true;
696 break;
697 }
698 }
699 if (!Applies)
700 continue;
701 Entry.IsUsed = true;
702 PragmaAttributeCurrentTargetDecl = D;
703 ParsedAttributesView Attrs;
704 Attrs.addAtEnd(Attribute);
705 ProcessDeclAttributeList(S, D, Attrs);
706 PragmaAttributeCurrentTargetDecl = nullptr;
707 }
708 }
709}

737void Sema::AddOptnoneAttributeIfNoConflicts(FunctionDecl *FD,
738 SourceLocation Loc) {
739 // Don't add a conflicting attribute. No diagnostic is needed.
740 if (FD->hasAttr<MinSizeAttr>() || FD->hasAttr<AlwaysInlineAttr>())
741 return;
742
743 // Add attributes only if required. Optnone requires noinline as well, but if
744 // either is already present then don't bother adding them.
745 if (!FD->hasAttr<OptimizeNoneAttr>())
746 FD->addAttr(OptimizeNoneAttr::CreateImplicit(Context, Loc));
747 if (!FD->hasAttr<NoInlineAttr>())
748 FD->addAttr(NoInlineAttr::CreateImplicit(Context, Loc));
749}

751typedef std::vector<std::pair<unsigned, SourceLocation> > VisStack;
752enum : unsigned { NoVisibility = ~0U };
753
754void Sema::AddPushedVisibilityAttribute(Decl *D) {
755 if (!VisContext)
756 return;
757
758 NamedDecl *ND = dyn_cast<NamedDecl>(D);
759 if (ND && ND->getExplicitVisibility(NamedDecl::VisibilityForValue))
760 return;
761
762 VisStack *Stack = static_cast<VisStack*>(VisContext);
763 unsigned rawType = Stack->back().first;
764 if (rawType == NoVisibility) return;
765
766 VisibilityAttr::VisibilityType type
767 = (VisibilityAttr::VisibilityType) rawType;
768 SourceLocation loc = Stack->back().second;
769
770 D->addAttr(VisibilityAttr::CreateImplicit(Context, type, loc));
771}

779static void PushPragmaVisibility(Sema &S, unsigned type, SourceLocation loc) {
780 // Put visibility on stack.
781 if (!S.VisContext)
782 S.VisContext = new VisStack;
783
784 VisStack *Stack = static_cast<VisStack*>(S.VisContext);
785 Stack->push_back(std::make_pair(type, loc));
786}

788void Sema::ActOnPragmaVisibility(const IdentifierInfo* VisType,
789 SourceLocation PragmaLoc) {
790 if (VisType) {
791 // Compute visibility to use.
792 VisibilityAttr::VisibilityType T;
793 if (!VisibilityAttr::ConvertStrToVisibilityType(VisType->getName(), T)) {
794 Diag(PragmaLoc, diag::warn_attribute_unknown_visibility) << VisType;
795 return;
796 }
797 PushPragmaVisibility(*this, T, PragmaLoc);
798 } else {
799 PopPragmaVisibility(false, PragmaLoc);
800 }
801}

829void Sema::PushNamespaceVisibilityAttr(const VisibilityAttr *Attr,
830 SourceLocation Loc) {
831 // Visibility calculations will consider the namespace's visibility.
832 // Here we just want to note that we're in a visibility context
833 // which overrides any enclosing #pragma context, but doesn't itself
834 // contribute visibility.
835 PushPragmaVisibility(*this, NoVisibility, Loc);

837
838void Sema::PopPragmaVisibility(bool IsNamespaceEnd, SourceLocation EndLoc) {
839 if (!VisContext) {
840 Diag(EndLoc, diag::err_pragma_pop_visibility_mismatch);
841 return;
842 }
843
844 // Pop visibility from stack
845 VisStack *Stack = static_cast<VisStack*>(VisContext);
846
847 const std::pair<unsigned, SourceLocation> *Back = &Stack->back();
848 bool StartsWithPragma = Back->first != NoVisibility;
849 if (StartsWithPragma && IsNamespaceEnd) {
850 Diag(Back->second, diag::err_pragma_push_visibility_mismatch);
851 Diag(EndLoc, diag::note_surrounding_namespace_ends_here);
852
853 // For better error recovery, eat all pushes inside the namespace.
854 do {
855 Stack->pop_back();
856 Back = &Stack->back();
857 StartsWithPragma = Back->first != NoVisibility;
858 } while (StartsWithPragma);
859 } else if (!StartsWithPragma && !IsNamespaceEnd) {
860 Diag(EndLoc, diag::err_pragma_pop_visibility_mismatch);
861 Diag(Back->second, diag::note_surrounding_namespace_starts_here);
862 return;
863 }
864
865 Stack->pop_back();
866 // To simplify the implementation, never keep around an empty stack.
867 if (Stack->empty())
868 FreeVisContext();
869}
