53PrintingPolicy Sema::getPrintingPolicy(const ASTContext &Context,
54 const Preprocessor &PP) {
55 PrintingPolicy Policy = Context.getPrintingPolicy();
56 // In diagnostics, we print _Bool as bool if the latter is defined as the
57 // former.
58 Policy.Bool = Context.getLangOpts().Bool;
59 if (!Policy.Bool) {
60 if (const MacroInfo *BoolMacro = PP.getMacroInfo(Context.getBoolName())) {
61 Policy.Bool = BoolMacro->isObjectLike() &&
62 BoolMacro->getNumTokens() == 1 &&
63 BoolMacro->getReplacementToken(0).is(tok::kw__Bool);
64 }
65 }
66
67 return Policy;
68}

78class SemaPPCallbacks : public PPCallbacks {
79 Sema *S = nullptr;
80 llvm::SmallVector<SourceLocation, 8> IncludeStack;
81
82public:
83 void set(Sema &S) { this->S = &S; }
84
85 void reset() { S = nullptr; }
86
87 virtual void FileChanged(SourceLocation Loc, FileChangeReason Reason,
88 SrcMgr::CharacteristicKind FileType,
89 FileID PrevFID) override {
90 if (!S)
91 return;
92 switch (Reason) {
93 case EnterFile: {
94 SourceManager &SM = S->getSourceManager();
95 SourceLocation IncludeLoc = SM.getIncludeLoc(SM.getFileID(Loc));
96 if (IncludeLoc.isValid()) {
97 if (llvm::timeTraceProfilerEnabled()) {
98 const FileEntry *FE = SM.getFileEntryForID(SM.getFileID(Loc));
99 llvm::timeTraceProfilerBegin(
100 "Source", FE != nullptr ? FE->getName() : StringRef("<unknown>"));
101 }
102
103 IncludeStack.push_back(IncludeLoc);
104 S->DiagnoseNonDefaultPragmaPack(
105 Sema::PragmaPackDiagnoseKind::NonDefaultStateAtInclude, IncludeLoc);
106 }
107 break;
108 }
109 case ExitFile:
110 if (!IncludeStack.empty()) {
111 if (llvm::timeTraceProfilerEnabled())
112 llvm::timeTraceProfilerEnd();
113
114 S->DiagnoseNonDefaultPragmaPack(
115 Sema::PragmaPackDiagnoseKind::ChangedStateAtExit,
116 IncludeStack.pop_back_val());
117 }
118 break;
119 default:
120 break;
121 }
122 }
123};

195void Sema::Initialize() {
196 if (SemaConsumer *SC = dyn_cast<SemaConsumer>(&Consumer))
197 SC->InitializeSema(*this);
198
199 // Tell the external Sema source about this Sema object.
200 if (ExternalSemaSource *ExternalSema
201 = dyn_cast_or_null<ExternalSemaSource>(Context.getExternalSource()))
202 ExternalSema->InitializeSema(*this);
203
204 // This needs to happen after ExternalSemaSource::InitializeSema(this) or we
205 // will not be able to merge any duplicate __va_list_tag decls correctly.
206 VAListTagName = PP.getIdentifierInfo("__va_list_tag");
207
208 if (!TUScope)
209 return;
210
211 // Initialize predefined 128-bit integer types, if needed.
212 if (Context.getTargetInfo().hasInt128Type()) {
213 // If either of the 128-bit integer types are unavailable to name lookup,
214 // define them now.
215 DeclarationName Int128 = &Context.Idents.get("__int128_t");
216 if (IdResolver.begin(Int128) == IdResolver.end())
217 PushOnScopeChains(Context.getInt128Decl(), TUScope);
218
219 DeclarationName UInt128 = &Context.Idents.get("__uint128_t");
220 if (IdResolver.begin(UInt128) == IdResolver.end())
221 PushOnScopeChains(Context.getUInt128Decl(), TUScope);
222 }
223
224 // If the target supports capabilities, add capability-as-integer builtin
225 // types.
226 if (PP.getTargetInfo().SupportsCapabilities()) {
227 DeclarationName IntCap = &Context.Idents.get("__intcap_t");
228 if (IdResolver.begin(IntCap) == IdResolver.end())
229 PushOnScopeChains(Context.getIntCapDecl(), TUScope);
230
231 DeclarationName UIntCap = &Context.Idents.get("__uintcap_t");
232 if (IdResolver.begin(UIntCap) == IdResolver.end())
233 PushOnScopeChains(Context.getUIntCapDecl(), TUScope);
234 }
235
236
237 // Initialize predefined Objective-C types:
238 if (getLangOpts().ObjC) {
239 // If 'SEL' does not yet refer to any declarations, make it refer to the
240 // predefined 'SEL'.
241 DeclarationName SEL = &Context.Idents.get("SEL");
242 if (IdResolver.begin(SEL) == IdResolver.end())
243 PushOnScopeChains(Context.getObjCSelDecl(), TUScope);
244
245 // If 'id' does not yet refer to any declarations, make it refer to the
246 // predefined 'id'.
247 DeclarationName Id = &Context.Idents.get("id");
248 if (IdResolver.begin(Id) == IdResolver.end())
249 PushOnScopeChains(Context.getObjCIdDecl(), TUScope);
250
251 // Create the built-in typedef for 'Class'.
252 DeclarationName Class = &Context.Idents.get("Class");
253 if (IdResolver.begin(Class) == IdResolver.end())
254 PushOnScopeChains(Context.getObjCClassDecl(), TUScope);
255
256 // Create the built-in forward declaratino for 'Protocol'.
257 DeclarationName Protocol = &Context.Idents.get("Protocol");
258 if (IdResolver.begin(Protocol) == IdResolver.end())
259 PushOnScopeChains(Context.getObjCProtocolDecl(), TUScope);
260 }
261
262 // Create the internal type for the *StringMakeConstantString builtins.
263 DeclarationName ConstantString = &Context.Idents.get("__NSConstantString");
264 if (IdResolver.begin(ConstantString) == IdResolver.end())
265 PushOnScopeChains(Context.getCFConstantStringDecl(), TUScope);
266
267 // Initialize Microsoft "predefined C++ types".
268 if (getLangOpts().MSVCCompat) {
269 if (getLangOpts().CPlusPlus &&
270 IdResolver.begin(&Context.Idents.get("type_info")) == IdResolver.end())
271 PushOnScopeChains(Context.buildImplicitRecord("type_info", TTK_Class),
272 TUScope);
273
274 addImplicitTypedef("size_t", Context.getSizeType());
275 }
276
277 // Initialize predefined OpenCL types and supported extensions and (optional)
278 // core features.
279 if (getLangOpts().OpenCL) {
280 getOpenCLOptions().addSupport(
281 Context.getTargetInfo().getSupportedOpenCLOpts());
282 getOpenCLOptions().enableSupportedCore(getLangOpts());
283 addImplicitTypedef("sampler_t", Context.OCLSamplerTy);
284 addImplicitTypedef("event_t", Context.OCLEventTy);
285 if (getLangOpts().OpenCLCPlusPlus || getLangOpts().OpenCLVersion >= 200) {
286 addImplicitTypedef("clk_event_t", Context.OCLClkEventTy);
287 addImplicitTypedef("queue_t", Context.OCLQueueTy);
288 addImplicitTypedef("reserve_id_t", Context.OCLReserveIDTy);
289 addImplicitTypedef("atomic_int", Context.getAtomicType(Context.IntTy));
290 addImplicitTypedef("atomic_uint",
291 Context.getAtomicType(Context.UnsignedIntTy));
292 auto AtomicLongT = Context.getAtomicType(Context.LongTy);
293 addImplicitTypedef("atomic_long", AtomicLongT);
294 auto AtomicULongT = Context.getAtomicType(Context.UnsignedLongTy);
295 addImplicitTypedef("atomic_ulong", AtomicULongT);
296 addImplicitTypedef("atomic_float",
297 Context.getAtomicType(Context.FloatTy));
298 auto AtomicDoubleT = Context.getAtomicType(Context.DoubleTy);
299 addImplicitTypedef("atomic_double", AtomicDoubleT);
300 // OpenCLC v2.0, s6.13.11.6 requires that atomic_flag is implemented as
301 // 32-bit integer and OpenCLC v2.0, s6.1.1 int is always 32-bit wide.
302 addImplicitTypedef("atomic_flag", Context.getAtomicType(Context.IntTy));
303 auto AtomicIntPtrT = Context.getAtomicType(Context.getIntPtrType());
304 addImplicitTypedef("atomic_intptr_t", AtomicIntPtrT);
305 auto AtomicUIntPtrT = Context.getAtomicType(Context.getUIntPtrType());
306 addImplicitTypedef("atomic_uintptr_t", AtomicUIntPtrT);
307 auto AtomicSizeT = Context.getAtomicType(Context.getSizeType());
308 addImplicitTypedef("atomic_size_t", AtomicSizeT);
309 auto AtomicPtrDiffT = Context.getAtomicType(Context.getPointerDiffType());
310 addImplicitTypedef("atomic_ptrdiff_t", AtomicPtrDiffT);
311
312 // OpenCL v2.0 s6.13.11.6:
313 // - The atomic_long and atomic_ulong types are supported if the
314 // cl_khr_int64_base_atomics and cl_khr_int64_extended_atomics
315 // extensions are supported.
316 // - The atomic_double type is only supported if double precision
317 // is supported and the cl_khr_int64_base_atomics and
318 // cl_khr_int64_extended_atomics extensions are supported.
319 // - If the device address space is 64-bits, the data types
320 // atomic_intptr_t, atomic_uintptr_t, atomic_size_t and
321 // atomic_ptrdiff_t are supported if the cl_khr_int64_base_atomics and
322 // cl_khr_int64_extended_atomics extensions are supported.
323 std::vector<QualType> Atomic64BitTypes;
324 Atomic64BitTypes.push_back(AtomicLongT);
325 Atomic64BitTypes.push_back(AtomicULongT);
326 Atomic64BitTypes.push_back(AtomicDoubleT);
327 if (Context.getTypeSize(AtomicSizeT) == 64) {
328 Atomic64BitTypes.push_back(AtomicSizeT);
329 Atomic64BitTypes.push_back(AtomicIntPtrT);
330 Atomic64BitTypes.push_back(AtomicUIntPtrT);
331 Atomic64BitTypes.push_back(AtomicPtrDiffT);
332 }
333 for (auto &I : Atomic64BitTypes)
334 setOpenCLExtensionForType(I,
335 "cl_khr_int64_base_atomics cl_khr_int64_extended_atomics");
336
337 setOpenCLExtensionForType(AtomicDoubleT, "cl_khr_fp64");
338 }
339
340 setOpenCLExtensionForType(Context.DoubleTy, "cl_khr_fp64");
341
342#define GENERIC_IMAGE_TYPE_EXT(Type, Id, Ext) \
343 setOpenCLExtensionForType(Context.Id, Ext);
344#include "clang/Basic/OpenCLImageTypes.def"
345#define EXT_OPAQUE_TYPE(ExtType, Id, Ext) \
346 addImplicitTypedef(#ExtType, Context.Id##Ty); \
347 setOpenCLExtensionForType(Context.Id##Ty, #Ext);
348#include "clang/Basic/OpenCLExtensionTypes.def"
349 };
350
351 if (Context.getTargetInfo().hasBuiltinMSVaList()) {
352 DeclarationName MSVaList = &Context.Idents.get("__builtin_ms_va_list");
353 if (IdResolver.begin(MSVaList) == IdResolver.end())
354 PushOnScopeChains(Context.getBuiltinMSVaListDecl(), TUScope);
355 }
356
357 DeclarationName BuiltinVaList = &Context.Idents.get("__builtin_va_list");
358 if (IdResolver.begin(BuiltinVaList) == IdResolver.end())
359 PushOnScopeChains(Context.getBuiltinVaListDecl(), TUScope);
360}

362Sema::~Sema() {
363 if (VisContext) FreeVisContext();
364
365 // Kill all the active scopes.
366 for (sema::FunctionScopeInfo *FSI : FunctionScopes)
367 delete FSI;
368
369 // Tell the SemaConsumer to forget about us; we're going out of scope.
370 if (SemaConsumer *SC = dyn_cast<SemaConsumer>(&Consumer))
371 SC->ForgetSema();
372
373 // Detach from the external Sema source.
374 if (ExternalSemaSource *ExternalSema
375 = dyn_cast_or_null<ExternalSemaSource>(Context.getExternalSource()))
376 ExternalSema->ForgetSema();
377
378 // If Sema's ExternalSource is the multiplexer - we own it.
379 if (isMultiplexExternalSource)
380 delete ExternalSource;
381
382 threadSafety::threadSafetyCleanup(ThreadSafetyDeclCache);
383
384 // Destroys data sharing attributes stack for OpenMP
385 DestroyDataSharingAttributesStack();
386
387 // Detach from the PP callback handler which outlives Sema since it's owned
388 // by the preprocessor.
389 SemaPPCallbackHandler->reset();
390
391 assert(DelayedTypos.empty() && "Uncorrected typos!");
392}

394/// makeUnavailableInSystemHeader - There is an error in the current
395/// context. If we're still in a system header, and we can plausibly
396/// make the relevant declaration unavailable instead of erroring, do
397/// so and return true.
398bool Sema::makeUnavailableInSystemHeader(SourceLocation loc,
399 UnavailableAttr::ImplicitReason reason) {
400 // If we're not in a function, it's an error.
401 FunctionDecl *fn = dyn_cast<FunctionDecl>(CurContext);
402 if (!fn) return false;
403
404 // If we're in template instantiation, it's an error.
405 if (inTemplateInstantiation())
406 return false;
407
408 // If that function's not in a system header, it's an error.
409 if (!Context.getSourceManager().isInSystemHeader(loc))
410 return false;
411
412 // If the function is already unavailable, it's not an error.
413 if (fn->hasAttr<UnavailableAttr>()) return true;
414
415 fn->addAttr(UnavailableAttr::CreateImplicit(Context, "", reason, loc));
416 return true;
417}

453void Sema::diagnoseNullableToNonnullConversion(QualType DstType,
454 QualType SrcType,
455 SourceLocation Loc) {
456 Optional<NullabilityKind> ExprNullability = SrcType->getNullability(Context);

458 return;

467void Sema::diagnoseZeroToNullptrConversion(CastKind Kind, const Expr* E) {
468 if (Diags.isIgnored(diag::warn_zero_as_null_pointer_constant,
469 E->getBeginLoc()))
470 return;
471 // nullptr only exists from C++11 on, so don't warn on its absence earlier.
472 if (!getLangOpts().CPlusPlus11)
473 return;
474
475 if (Kind != CK_NullToPointer && Kind != CK_NullToMemberPointer)
476 return;
477 if (E->IgnoreParenImpCasts()->getType()->isNullPtrType())
478 return;
479
480 // If it is a macro from system header, and if the macro name is not "NULL",
481 // do not warn.
482 SourceLocation MaybeMacroLoc = E->getBeginLoc();
483 if (Diags.getSuppressSystemWarnings() &&
484 SourceMgr.isInSystemMacro(MaybeMacroLoc) &&
485 !findMacroSpelling(MaybeMacroLoc, "NULL"))
486 return;
487
488 Diag(E->getBeginLoc(), diag::warn_zero_as_null_pointer_constant)
489 << FixItHint::CreateReplacement(E->getSourceRange(), "nullptr");
490}

492/// ImpCastExprToType - If Expr is not of type 'Type', insert an implicit cast.
493/// If there is already an implicit cast, merge into the existing one.
494/// The result is of the given category.
495ExprResult Sema::ImpCastExprToType(Expr *E, QualType Ty,
496 CastKind Kind, ExprValueKind VK,
497 const CXXCastPath *BasePath,
498 CheckedConversionKind CCK) {
499#ifndef NDEBUG
500 if (VK == VK_RValue && !E->isRValue()) {
501 switch (Kind) {
502 default:
503 llvm_unreachable("can't implicitly cast lvalue to rvalue with this cast "
504 "kind");
505 case CK_LValueToRValue:
506 case CK_ArrayToPointerDecay:
507 case CK_FunctionToPointerDecay:
508 case CK_ToVoid:
509 case CK_NonAtomicToAtomic:
510 break;
511 }
512 }
513 assert((VK == VK_RValue || !E->isRValue()) && "can't cast rvalue to lvalue");
514#endif
515
516 diagnoseNullableToNonnullConversion(Ty, E->getType(), E->getBeginLoc());
517 diagnoseZeroToNullptrConversion(Kind, E);
518
519 QualType ExprTy = Context.getCanonicalType(E->getType());
520 QualType TypeTy = Context.getCanonicalType(Ty);
521
522 if (ExprTy.getQualifiers().hasOutput() &&
523 !TypeTy.getQualifiers().hasOutput())
524 return ExprError(Diag(E->getExprLoc(), diag::err_typecheck_read_output)
525 << E->getSourceRange());
526
527 if (ExprTy == TypeTy)
528 return E;
529
530 // C++1z [conv.array]: The temporary materialization conversion is applied.
531 // We also use this to fuel C++ DR1213, which applies to C++11 onwards.
532 if (Kind == CK_ArrayToPointerDecay && getLangOpts().CPlusPlus &&
533 E->getValueKind() == VK_RValue) {
534 // The temporary is an lvalue in C++98 and an xvalue otherwise.
535 ExprResult Materialized = CreateMaterializeTemporaryExpr(
536 E->getType(), E, !getLangOpts().CPlusPlus11);
537 if (Materialized.isInvalid())
538 return ExprError();
539 E = Materialized.get();
540 }
541
542 if (ImplicitCastExpr *ImpCast = dyn_cast<ImplicitCastExpr>(E)) {
543 if (ImpCast->getCastKind() == Kind && (!BasePath || BasePath->empty())) {
544 ImpCast->setType(Ty);
545 ImpCast->setValueKind(VK);
546 return E;
547 }
548 }
549
550 // Only allow implicit casts from pointers to CHERI capabilities
551 // for the following cases:
552 //
553 // - String literal
554 // - Null literal
555 // - address-of (&) expressions and types are compatible
556 // - array/function to pointer decay and types are compatible
557 if (const PointerType *EPTy = dyn_cast<PointerType>(ExprTy)) {
558 if (const PointerType *TPTy = dyn_cast<PointerType>(TypeTy)) {
559 if (!EPTy->isCHERICapability() && TPTy->isCHERICapability()) {
560 if (!ImpCastPointerToCHERICapability(ExprTy, TypeTy, E))
561 return ExprError();
562 }
563 }
564 }
565
566 return ImplicitCastExpr::Create(Context, Ty, Kind, E, BasePath, VK);
567}

569bool Sema::ImpCastPointerToCHERICapability(QualType FromTy, QualType ToTy, Expr *&From, bool Diagnose) {
570
571 assert(FromTy->isPointerType() && ToTy->isPointerType() && ToTy->getAs<PointerType>()->isCHERICapability() &&
572 "Both argument types should be PointerType");
573
574 bool StrLit = dyn_cast<StringLiteral>(From->IgnoreImpCasts()) != nullptr;
575 bool NullConstant = From->isNullPointerConstant(Context, Expr::NPC_ValueDependentIsNotNull);
576 bool AddrOf = false;
577 bool Decayed = false;
578 bool CompatTypes = false;
579
580 // address-of (&)
581 if (UnaryOperator *UnOp = dyn_cast<UnaryOperator>(From)) {
582 if (UnOp->getOpcode() == UO_AddrOf) {
583 AddrOf = true;
584 CompatTypes = CheckCHERIAssignCompatible(ToTy, FromTy, From);
585 }
586 }
587 // Function-to-pointer and array-to-pointer decay
588 if (ImplicitCastExpr* Imp = dyn_cast<ImplicitCastExpr>(From)) {
589 if (Imp->getCastKind() == CK_ArrayToPointerDecay
590 || Imp->getCastKind() == CK_FunctionToPointerDecay) {
591 Decayed = true;
592 CompatTypes = CheckCHERIAssignCompatible(ToTy, FromTy, From);
593 }
594 }
595
596 // The type check for address-of (&) could have failed above but may
597 // still succeed below when we look at pointee types
598 if (!(StrLit || NullConstant || Decayed || (AddrOf && CompatTypes))) {
599 // Check if pointee types are assign compatible
600 QualType TPointee = ToTy->getAs<PointerType>()->getPointeeType();
601 QualType EPointee = FromTy->getAs<PointerType>()->getPointeeType();
602 if (CheckCHERIAssignCompatible(TPointee, EPointee, From, false)) {
603 if (Diagnose)
604 Diag(From->getExprLoc(), diag::warn_typecheck_convert_ptr_to_cap)
605 << From->getType() << ToTy
606 << FixItHint::CreateInsertion(From->getExprLoc(), "(__cheri_tocap "
607 + ToTy.getAsString() + ")");
608 CompatTypes = true;
609 } else {
610 // Check for integer pointee types with differences only in sign
611 // (if they were exactly the same integer type then the above if-test would have succeeded)
612 if (TPointee->isIntegerType() && EPointee->isIntegerType()) {
613 if (TPointee->isSignedIntegerType())
614 TPointee = Context.getCorrespondingUnsignedType(TPointee);
615 if (EPointee->isSignedIntegerType())
616 EPointee = Context.getCorrespondingUnsignedType(EPointee);
617
618 if (TPointee == EPointee) {
619 // Output a warning, as unsigned types are the same but the only difference is sign
620 if (Diagnose)
621 Diag(From->getExprLoc(), diag::warn_typecheck_convert_ptr_to_cap_sign_difference) << FromTy << ToTy;
622 CompatTypes = true;
623 }
624 }
625 }
626 if (!(StrLit || NullConstant || Decayed || AddrOf || CompatTypes)) {
627 if (Diagnose) {
628 Diag(From->getExprLoc(), diag::err_typecheck_convert_ptr_to_cap)
629 << From->getType() << ToTy << false;
630 }
631 return false;
632 } else if (!CompatTypes) {
633 if (Diagnose) {
634 Diag(From->getExprLoc(), diag::err_typecheck_convert_ptr_to_cap_unrelated_type)
635 << From->getType() << ToTy << false;
636 }
637 return false;
638 }
639 }
640 return true;
641}

661/// Used to prune the decls of Sema's UnusedFileScopedDecls vector.
662static bool ShouldRemoveFromUnused(Sema *SemaRef, const DeclaratorDecl *D) {
663 if (D->getMostRecentDecl()->isUsed())
664 return true;
665
666 if (D->isExternallyVisible())
667 return true;
668
669 if (const FunctionDecl *FD = dyn_cast<FunctionDecl>(D)) {
670 // If this is a function template and none of its specializations is used,
671 // we should warn.
672 if (FunctionTemplateDecl *Template = FD->getDescribedFunctionTemplate())
673 for (const auto *Spec : Template->specializations())
674 if (ShouldRemoveFromUnused(SemaRef, Spec))
675 return true;
676
677 // UnusedFileScopedDecls stores the first declaration.
678 // The declaration may have become definition so check again.
679 const FunctionDecl *DeclToCheck;
680 if (FD->hasBody(DeclToCheck))
681 return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
682
683 // Later redecls may add new information resulting in not having to warn,
684 // so check again.
685 DeclToCheck = FD->getMostRecentDecl();
686 if (DeclToCheck != FD)
687 return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
688 }
689
690 if (const VarDecl *VD = dyn_cast<VarDecl>(D)) {
691 // If a variable usable in constant expressions is referenced,
692 // don't warn if it isn't used: if the value of a variable is required
693 // for the computation of a constant expression, it doesn't make sense to
694 // warn even if the variable isn't odr-used. (isReferenced doesn't
695 // precisely reflect that, but it's a decent approximation.)
696 if (VD->isReferenced() &&
697 VD->mightBeUsableInConstantExpressions(SemaRef->Context))
698 return true;
699
700 if (VarTemplateDecl *Template = VD->getDescribedVarTemplate())
701 // If this is a variable template and none of its specializations is used,
702 // we should warn.
703 for (const auto *Spec : Template->specializations())
704 if (ShouldRemoveFromUnused(SemaRef, Spec))
705 return true;
706
707 // UnusedFileScopedDecls stores the first declaration.
708 // The declaration may have become definition so check again.
709 const VarDecl *DeclToCheck = VD->getDefinition();
710 if (DeclToCheck)
711 return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
712
713 // Later redecls may add new information resulting in not having to warn,
714 // so check again.
715 DeclToCheck = VD->getMostRecentDecl();
716 if (DeclToCheck != VD)
717 return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
718 }
719
720 return false;
721}

725 return FD->isExternC();

729/// Determine whether ND is an external-linkage function or variable whose
730/// type has no linkage.

732 // Note: it's not quite enough to check whether VD has UniqueExternalLinkage,

740/// Obtains a sorted list of functions and variables that are undefined but
741/// ODR-used.
742void Sema::getUndefinedButUsed(
743 SmallVectorImpl<std::pair<NamedDecl *, SourceLocation> > &Undefined) {
744 for (const auto &UndefinedUse : UndefinedButUsed) {
745 NamedDecl *ND = UndefinedUse.first;
746
747 // Ignore attributes that have become invalid.
748 if (ND->isInvalidDecl()) continue;
749
750 // __attribute__((weakref)) is basically a definition.
751 if (ND->hasAttr<WeakRefAttr>()) continue;
752
753 if (isa<CXXDeductionGuideDecl>(ND))
754 continue;
755
756 if (ND->hasAttr<DLLImportAttr>() || ND->hasAttr<DLLExportAttr>()) {
757 // An exported function will always be emitted when defined, so even if
758 // the function is inline, it doesn't have to be emitted in this TU. An
759 // imported function implies that it has been exported somewhere else.
760 continue;
761 }
762
763 if (FunctionDecl *FD = dyn_cast<FunctionDecl>(ND)) {
764 if (FD->isDefined())
765 continue;
766 if (FD->isExternallyVisible() &&
767 !isExternalWithNoLinkageType(FD) &&
768 !FD->getMostRecentDecl()->isInlined() &&
769 !FD->hasAttr<ExcludeFromExplicitInstantiationAttr>())
770 continue;
771 if (FD->getBuiltinID())
772 continue;
773 } else {
774 auto *VD = cast<VarDecl>(ND);
775 if (VD->hasDefinition() != VarDecl::DeclarationOnly)
776 continue;
777 if (VD->isExternallyVisible() &&
778 !isExternalWithNoLinkageType(VD) &&
779 !VD->getMostRecentDecl()->isInline() &&
780 !VD->hasAttr<ExcludeFromExplicitInstantiationAttr>())
781 continue;
782
783 // Skip VarDecls that lack formal definitions but which we know are in
784 // fact defined somewhere.
785 if (VD->isKnownToBeDefined())
786 continue;
787 }
788
789 Undefined.push_back(std::make_pair(ND, UndefinedUse.second));
790 }
791}

793/// checkUndefinedButUsed - Check for undefined objects with internal linkage
794/// or that are inline.
795static void checkUndefinedButUsed(Sema &S) {
796 if (S.UndefinedButUsed.empty()) return;
797
798 // Collect all the still-undefined entities with internal linkage.
799 SmallVector<std::pair<NamedDecl *, SourceLocation>, 16> Undefined;
800 S.getUndefinedButUsed(Undefined);
801 if (Undefined.empty()) return;
802
803 for (auto Undef : Undefined) {
804 ValueDecl *VD = cast<ValueDecl>(Undef.first);
805 SourceLocation UseLoc = Undef.second;
806
807 if (S.isExternalWithNoLinkageType(VD)) {
808 // C++ [basic.link]p8:
809 // A type without linkage shall not be used as the type of a variable
810 // or function with external linkage unless
811 // -- the entity has C language linkage
812 // -- the entity is not odr-used or is defined in the same TU
813 //
814 // As an extension, accept this in cases where the type is externally
815 // visible, since the function or variable actually can be defined in
816 // another translation unit in that case.
817 S.Diag(VD->getLocation(), isExternallyVisible(VD->getType()->getLinkage())
818 ? diag::ext_undefined_internal_type
819 : diag::err_undefined_internal_type)
820 << isa<VarDecl>(VD) << VD;
821 } else if (!VD->isExternallyVisible()) {
822 // FIXME: We can promote this to an error. The function or variable can't
823 // be defined anywhere else, so the program must necessarily violate the
824 // one definition rule.
825 S.Diag(VD->getLocation(), diag::warn_undefined_internal)
826 << isa<VarDecl>(VD) << VD;
827 } else if (auto *FD = dyn_cast<FunctionDecl>(VD)) {
828 (void)FD;
829 assert(FD->getMostRecentDecl()->isInlined() &&
830 "used object requires definition but isn't inline or internal?");
831 // FIXME: This is ill-formed; we should reject.
832 S.Diag(VD->getLocation(), diag::warn_undefined_inline) << VD;
833 } else {
834 assert(cast<VarDecl>(VD)->getMostRecentDecl()->isInline() &&
835 "used var requires definition but isn't inline or internal?");
836 S.Diag(VD->getLocation(), diag::err_undefined_inline_var) << VD;
837 }
838 if (UseLoc.isValid())
839 S.Diag(UseLoc, diag::note_used_here);
840 }
841
842 S.UndefinedButUsed.clear();
843}

856typedef llvm::DenseMap<const CXXRecordDecl*, bool> RecordCompleteMap;
857
858/// Returns true, if all methods and nested classes of the given
859/// CXXRecordDecl are defined in this translation unit.
860///
861/// Should only be called from ActOnEndOfTranslationUnit so that all
862/// definitions are actually read.
863static bool MethodsAndNestedClassesComplete(const CXXRecordDecl *RD,
864 RecordCompleteMap &MNCComplete) {
865 RecordCompleteMap::iterator Cache = MNCComplete.find(RD);
866 if (Cache != MNCComplete.end())
867 return Cache->second;
868 if (!RD->isCompleteDefinition())
869 return false;
870 bool Complete = true;
871 for (DeclContext::decl_iterator I = RD->decls_begin(),
872 E = RD->decls_end();
873 I != E && Complete; ++I) {
874 if (const CXXMethodDecl *M = dyn_cast<CXXMethodDecl>(*I))
875 Complete = M->isDefined() || M->isDefaulted() ||
876 (M->isPure() && !isa<CXXDestructorDecl>(M));
877 else if (const FunctionTemplateDecl *F = dyn_cast<FunctionTemplateDecl>(*I))
878 // If the template function is marked as late template parsed at this
879 // point, it has not been instantiated and therefore we have not
880 // performed semantic analysis on it yet, so we cannot know if the type
881 // can be considered complete.
882 Complete = !F->getTemplatedDecl()->isLateTemplateParsed() &&
883 F->getTemplatedDecl()->isDefined();
884 else if (const CXXRecordDecl *R = dyn_cast<CXXRecordDecl>(*I)) {
885 if (R->isInjectedClassName())
886 continue;
887 if (R->hasDefinition())
888 Complete = MethodsAndNestedClassesComplete(R->getDefinition(),
889 MNCComplete);
890 else
891 Complete = false;
892 }
893 }
894 MNCComplete[RD] = Complete;
895 return Complete;
896}

898/// Returns true, if the given CXXRecordDecl is fully defined in this
899/// translation unit, i.e. all methods are defined or pure virtual and all
900/// friends, friend functions and nested classes are fully defined in this
901/// translation unit.
902///
903/// Should only be called from ActOnEndOfTranslationUnit so that all
904/// definitions are actually read.
905static bool IsRecordFullyDefined(const CXXRecordDecl *RD,
906 RecordCompleteMap &RecordsComplete,
907 RecordCompleteMap &MNCComplete) {
908 RecordCompleteMap::iterator Cache = RecordsComplete.find(RD);
909 if (Cache != RecordsComplete.end())
910 return Cache->second;
911 bool Complete = MethodsAndNestedClassesComplete(RD, MNCComplete);
912 for (CXXRecordDecl::friend_iterator I = RD->friend_begin(),
913 E = RD->friend_end();
914 I != E && Complete; ++I) {
915 // Check if friend classes and methods are complete.
916 if (TypeSourceInfo *TSI = (*I)->getFriendType()) {
917 // Friend classes are available as the TypeSourceInfo of the FriendDecl.
918 if (CXXRecordDecl *FriendD = TSI->getType()->getAsCXXRecordDecl())
919 Complete = MethodsAndNestedClassesComplete(FriendD, MNCComplete);
920 else
921 Complete = false;
922 } else {
923 // Friend functions are available through the NamedDecl of FriendDecl.
924 if (const FunctionDecl *FD =
925 dyn_cast<FunctionDecl>((*I)->getFriendDecl()))
926 Complete = FD->isDefined();
927 else
928 // This is a template friend, give up.
929 Complete = false;
930 }
931 }
932 RecordsComplete[RD] = Complete;
933 return Complete;
934}

964void Sema::ActOnEndOfTranslationUnitFragment(TUFragmentKind Kind) {
965 // No explicit actions are required at the end of the global module fragment.
966 if (Kind == TUFragmentKind::Global)
967 return;
968
969 // Transfer late parsed template instantiations over to the pending template
970 // instantiation list. During normal compilation, the late template parser
971 // will be installed and instantiating these templates will succeed.
972 //
973 // If we are building a TU prefix for serialization, it is also safe to
974 // transfer these over, even though they are not parsed. The end of the TU
975 // should be outside of any eager template instantiation scope, so when this
976 // AST is deserialized, these templates will not be parsed until the end of
977 // the combined TU.
978 PendingInstantiations.insert(PendingInstantiations.end(),
979 LateParsedInstantiations.begin(),
980 LateParsedInstantiations.end());
981 LateParsedInstantiations.clear();
982
983 // If DefinedUsedVTables ends up marking any virtual member functions it
984 // might lead to more pending template instantiations, which we then need
985 // to instantiate.
986 DefineUsedVTables();
987
988 // C++: Perform implicit template instantiations.
989 //
990 // FIXME: When we perform these implicit instantiations, we do not
991 // carefully keep track of the point of instantiation (C++ [temp.point]).
992 // This means that name lookup that occurs within the template
993 // instantiation will always happen at the end of the translation unit,
994 // so it will find some names that are not required to be found. This is
995 // valid, but we could do better by diagnosing if an instantiation uses a
996 // name that was not visible at its first point of instantiation.
997 if (ExternalSource) {
998 // Load pending instantiations from the external source.
999 SmallVector<PendingImplicitInstantiation, 4> Pending;
1000 ExternalSource->ReadPendingInstantiations(Pending);
1001 for (auto PII : Pending)
1002 if (auto Func = dyn_cast<FunctionDecl>(PII.first))
1003 Func->setInstantiationIsPending(true);
1004 PendingInstantiations.insert(PendingInstantiations.begin(),
1005 Pending.begin(), Pending.end());
1006 }
1007
1008 {
1009 llvm::TimeTraceScope TimeScope("PerformPendingInstantiations",
1010 StringRef(""));
1011 PerformPendingInstantiations();
1012 }
1013
1014 assert(LateParsedInstantiations.empty() &&
1015 "end of TU template instantiation should not create more "
1016 "late-parsed templates");
1017}

1019/// ActOnEndOfTranslationUnit - This is called at the very end of the
1020/// translation unit when EOF is reached and all but the top-level scope is
1021/// popped.
1022void Sema::ActOnEndOfTranslationUnit() {
1023 assert(DelayedDiagnostics.getCurrentPool() == nullptr
1024 && "reached end of translation unit with a pool attached?");
1025
1026 // If code completion is enabled, don't perform any end-of-translation-unit
1027 // work.
1028 if (PP.isCodeCompletionEnabled())
1029 return;
1030
1031 // Complete translation units and modules define vtables and perform implicit
1032 // instantiations. PCH files do not.
1033 if (TUKind != TU_Prefix) {
1034 DiagnoseUseOfUnimplementedSelectors();
1035
1036 ActOnEndOfTranslationUnitFragment(
1037 !ModuleScopes.empty() && ModuleScopes.back().Module->Kind ==
1038 Module::PrivateModuleFragment
1039 ? TUFragmentKind::Private
1040 : TUFragmentKind::Normal);
1041
1042 if (LateTemplateParserCleanup)
1043 LateTemplateParserCleanup(OpaqueParser);
1044
1045 CheckDelayedMemberExceptionSpecs();
1046 } else {
1047 // If we are building a TU prefix for serialization, it is safe to transfer
1048 // these over, even though they are not parsed. The end of the TU should be
1049 // outside of any eager template instantiation scope, so when this AST is
1050 // deserialized, these templates will not be parsed until the end of the
1051 // combined TU.
1052 PendingInstantiations.insert(PendingInstantiations.end(),
1053 LateParsedInstantiations.begin(),
1054 LateParsedInstantiations.end());
1055 LateParsedInstantiations.clear();
1056 }
1057
1058 DiagnoseUnterminatedPragmaPack();
1059 DiagnoseUnterminatedPragmaAttribute();
1060
1061 // All delayed member exception specs should be checked or we end up accepting
1062 // incompatible declarations.
1063 assert(DelayedOverridingExceptionSpecChecks.empty());
1064 assert(DelayedEquivalentExceptionSpecChecks.empty());
1065
1066 // All dllexport classes should have been processed already.
1067 assert(DelayedDllExportClasses.empty());
1068
1069 // Remove file scoped decls that turned out to be used.
1070 UnusedFileScopedDecls.erase(
1071 std::remove_if(UnusedFileScopedDecls.begin(nullptr, true),
1072 UnusedFileScopedDecls.end(),
1073 [this](const DeclaratorDecl *DD) {
1074 return ShouldRemoveFromUnused(this, DD);
1075 }),
1076 UnusedFileScopedDecls.end());
1077
1078 if (TUKind == TU_Prefix) {
1079 // Translation unit prefixes don't need any of the checking below.
1080 if (!PP.isIncrementalProcessingEnabled())
1081 TUScope = nullptr;
1082 return;
1083 }
1084
1085 // Check for #pragma weak identifiers that were never declared
1086 LoadExternalWeakUndeclaredIdentifiers();
1087 for (auto WeakID : WeakUndeclaredIdentifiers) {
1088 if (WeakID.second.getUsed())
1089 continue;
1090
1091 Decl *PrevDecl = LookupSingleName(TUScope, WeakID.first, SourceLocation(),
1092 LookupOrdinaryName);
1093 if (PrevDecl != nullptr &&
1094 !(isa<FunctionDecl>(PrevDecl) || isa<VarDecl>(PrevDecl)))
1095 Diag(WeakID.second.getLocation(), diag::warn_attribute_wrong_decl_type)
1096 << "'weak'" << ExpectedVariableOrFunction;
1097 else
1098 Diag(WeakID.second.getLocation(), diag::warn_weak_identifier_undeclared)
1099 << WeakID.first;
1100 }
1101
1102 if (LangOpts.CPlusPlus11 &&
1103 !Diags.isIgnored(diag::warn_delegating_ctor_cycle, SourceLocation()))
1104 CheckDelegatingCtorCycles();
1105
1106 if (!Diags.hasErrorOccurred()) {
1107 if (ExternalSource)
1108 ExternalSource->ReadUndefinedButUsed(UndefinedButUsed);
1109 checkUndefinedButUsed(*this);
1110 }
1111
1112 // A global-module-fragment is only permitted within a module unit.
1113 bool DiagnosedMissingModuleDeclaration = false;
1114 if (!ModuleScopes.empty() &&
1115 ModuleScopes.back().Module->Kind == Module::GlobalModuleFragment &&
1116 !ModuleScopes.back().ImplicitGlobalModuleFragment) {
1117 Diag(ModuleScopes.back().BeginLoc,
1118 diag::err_module_declaration_missing_after_global_module_introducer);
1119 DiagnosedMissingModuleDeclaration = true;
1120 }
1121
1122 if (TUKind == TU_Module) {
1123 // If we are building a module interface unit, we need to have seen the
1124 // module declaration by now.
1125 if (getLangOpts().getCompilingModule() ==
1126 LangOptions::CMK_ModuleInterface &&
1127 (ModuleScopes.empty() ||
1128 !ModuleScopes.back().Module->isModulePurview()) &&
1129 !DiagnosedMissingModuleDeclaration) {
1130 // FIXME: Make a better guess as to where to put the module declaration.
1131 Diag(getSourceManager().getLocForStartOfFile(
1132 getSourceManager().getMainFileID()),
1133 diag::err_module_declaration_missing);
1134 }
1135
1136 // If we are building a module, resolve all of the exported declarations
1137 // now.
1138 if (Module *CurrentModule = PP.getCurrentModule()) {
1139 ModuleMap &ModMap = PP.getHeaderSearchInfo().getModuleMap();
1140
1141 SmallVector<Module *, 2> Stack;
1142 Stack.push_back(CurrentModule);
1143 while (!Stack.empty()) {
1144 Module *Mod = Stack.pop_back_val();
1145
1146 // Resolve the exported declarations and conflicts.
1147 // FIXME: Actually complain, once we figure out how to teach the
1148 // diagnostic client to deal with complaints in the module map at this
1149 // point.
1150 ModMap.resolveExports(Mod, /*Complain=*/false);
1151 ModMap.resolveUses(Mod, /*Complain=*/false);
1152 ModMap.resolveConflicts(Mod, /*Complain=*/false);
1153
1154 // Queue the submodules, so their exports will also be resolved.
1155 Stack.append(Mod->submodule_begin(), Mod->submodule_end());
1156 }
1157 }
1158
1159 // Warnings emitted in ActOnEndOfTranslationUnit() should be emitted for
1160 // modules when they are built, not every time they are used.
1161 emitAndClearUnusedLocalTypedefWarnings();
1162 }
1163
1164 // C99 6.9.2p2:
1165 // A declaration of an identifier for an object that has file
1166 // scope without an initializer, and without a storage-class
1167 // specifier or with the storage-class specifier static,
1168 // constitutes a tentative definition. If a translation unit
1169 // contains one or more tentative definitions for an identifier,
1170 // and the translation unit contains no external definition for
1171 // that identifier, then the behavior is exactly as if the
1172 // translation unit contains a file scope declaration of that
1173 // identifier, with the composite type as of the end of the
1174 // translation unit, with an initializer equal to 0.
1175 llvm::SmallSet<VarDecl *, 32> Seen;
1176 for (TentativeDefinitionsType::iterator
1177 T = TentativeDefinitions.begin(ExternalSource),
1178 TEnd = TentativeDefinitions.end();
1179 T != TEnd; ++T) {
1180 VarDecl *VD = (*T)->getActingDefinition();
1181
1182 // If the tentative definition was completed, getActingDefinition() returns
1183 // null. If we've already seen this variable before, insert()'s second
1184 // return value is false.
1185 if (!VD || VD->isInvalidDecl() || !Seen.insert(VD).second)
1186 continue;
1187
1188 if (const IncompleteArrayType *ArrayT
1189 = Context.getAsIncompleteArrayType(VD->getType())) {
1190 // Set the length of the array to 1 (C99 6.9.2p5).
1191 Diag(VD->getLocation(), diag::warn_tentative_incomplete_array);
1192 llvm::APInt One(Context.getTypeSize(Context.getSizeType()), true);
1193 QualType T = Context.getConstantArrayType(ArrayT->getElementType(),
1194 One, ArrayType::Normal, 0);
1195 VD->setType(T);
1196 } else if (RequireCompleteType(VD->getLocation(), VD->getType(),
1197 diag::err_tentative_def_incomplete_type))
1198 VD->setInvalidDecl();
1199
1200 // No initialization is performed for a tentative definition.
1201 CheckCompleteVariableDeclaration(VD);
1202
1203 // Notify the consumer that we've completed a tentative definition.
1204 if (!VD->isInvalidDecl())
1205 Consumer.CompleteTentativeDefinition(VD);
1206 }
1207
1208 // If there were errors, disable 'unused' warnings since they will mostly be
1209 // noise. Don't warn for a use from a module: either we should warn on all
1210 // file-scope declarations in modules or not at all, but whether the
1211 // declaration is used is immaterial.
1212 if (!Diags.hasErrorOccurred() && TUKind != TU_Module) {
1213 // Output warning for unused file scoped decls.
1214 for (UnusedFileScopedDeclsType::iterator
1215 I = UnusedFileScopedDecls.begin(ExternalSource),
1216 E = UnusedFileScopedDecls.end(); I != E; ++I) {
1217 if (ShouldRemoveFromUnused(this, *I))
1218 continue;
1219
1220 if (const FunctionDecl *FD = dyn_cast<FunctionDecl>(*I)) {
1221 const FunctionDecl *DiagD;
1222 if (!FD->hasBody(DiagD))
1223 DiagD = FD;
1224 if (DiagD->isDeleted())
1225 continue; // Deleted functions are supposed to be unused.
1226 if (DiagD->isReferenced()) {
1227 if (isa<CXXMethodDecl>(DiagD))
1228 Diag(DiagD->getLocation(), diag::warn_unneeded_member_function)
1229 << DiagD->getDeclName();
1230 else {
1231 if (FD->getStorageClass() == SC_Static &&
1232 !FD->isInlineSpecified() &&
1233 !SourceMgr.isInMainFile(
1234 SourceMgr.getExpansionLoc(FD->getLocation())))
1235 Diag(DiagD->getLocation(),
1236 diag::warn_unneeded_static_internal_decl)
1237 << DiagD->getDeclName();
1238 else
1239 Diag(DiagD->getLocation(), diag::warn_unneeded_internal_decl)
1240 << /*function*/0 << DiagD->getDeclName();
1241 }
1242 } else {
1243 if (FD->getDescribedFunctionTemplate())
1244 Diag(DiagD->getLocation(), diag::warn_unused_template)
1245 << /*function*/0 << DiagD->getDeclName();
1246 else
1247 Diag(DiagD->getLocation(),
1248 isa<CXXMethodDecl>(DiagD) ? diag::warn_unused_member_function
1249 : diag::warn_unused_function)
1250 << DiagD->getDeclName();
1251 }
1252 } else {
1253 const VarDecl *DiagD = cast<VarDecl>(*I)->getDefinition();
1254 if (!DiagD)
1255 DiagD = cast<VarDecl>(*I);
1256 if (DiagD->isReferenced()) {
1257 Diag(DiagD->getLocation(), diag::warn_unneeded_internal_decl)
1258 << /*variable*/1 << DiagD->getDeclName();
1259 } else if (DiagD->getType().isConstQualified()) {
1260 const SourceManager &SM = SourceMgr;
1261 if (SM.getMainFileID() != SM.getFileID(DiagD->getLocation()) ||
1262 !PP.getLangOpts().IsHeaderFile)
1263 Diag(DiagD->getLocation(), diag::warn_unused_const_variable)
1264 << DiagD->getDeclName();
1265 } else {
1266 if (DiagD->getDescribedVarTemplate())
1267 Diag(DiagD->getLocation(), diag::warn_unused_template)
1268 << /*variable*/1 << DiagD->getDeclName();
1269 else
1270 Diag(DiagD->getLocation(), diag::warn_unused_variable)
1271 << DiagD->getDeclName();
1272 }
1273 }
1274 }
1275
1276 emitAndClearUnusedLocalTypedefWarnings();
1277 }
1278
1279 if (!Diags.isIgnored(diag::warn_unused_private_field, SourceLocation())) {
1280 // FIXME: Load additional unused private field candidates from the external
1281 // source.
1282 RecordCompleteMap RecordsComplete;
1283 RecordCompleteMap MNCComplete;
1284 for (NamedDeclSetType::iterator I = UnusedPrivateFields.begin(),
1285 E = UnusedPrivateFields.end(); I != E; ++I) {
1286 const NamedDecl *D = *I;
1287 const CXXRecordDecl *RD = dyn_cast<CXXRecordDecl>(D->getDeclContext());
1288 if (RD && !RD->isUnion() &&
1289 IsRecordFullyDefined(RD, RecordsComplete, MNCComplete)) {
1290 Diag(D->getLocation(), diag::warn_unused_private_field)
1291 << D->getDeclName();
1292 }
1293 }
1294 }
1295
1296 if (!Diags.isIgnored(diag::warn_mismatched_delete_new, SourceLocation())) {
1297 if (ExternalSource)
1298 ExternalSource->ReadMismatchingDeleteExpressions(DeleteExprs);
1299 for (const auto &DeletedFieldInfo : DeleteExprs) {
1300 for (const auto &DeleteExprLoc : DeletedFieldInfo.second) {
1301 AnalyzeDeleteExprMismatch(DeletedFieldInfo.first, DeleteExprLoc.first,
1302 DeleteExprLoc.second);
1303 }
1304 }
1305 }
1306
1307 // Check we've noticed that we're no longer parsing the initializer for every
1308 // variable. If we miss cases, then at best we have a performance issue and
1309 // at worst a rejects-valid bug.
1310 assert(ParsingInitForAutoVars.empty() &&
1311 "Didn't unmark var as having its initializer parsed");
1312
1313 if (!PP.isIncrementalProcessingEnabled())
1314 TUScope = nullptr;
1315}

1361void Sema::EmitCurrentDiagnostic(unsigned DiagID) {
1362 // FIXME: It doesn't make sense to me that DiagID is an incoming argument here
1363 // and yet we also use the current diag ID on the DiagnosticsEngine. This has
1364 // been made more painfully obvious by the refactor that introduced this
1365 // function, but it is possible that the incoming argument can be
1366 // eliminated. If it truly cannot be (for example, there is some reentrancy
1367 // issue I am not seeing yet), then there should at least be a clarifying
1368 // comment somewhere.
1369 if (Optional<TemplateDeductionInfo*> Info = isSFINAEContext()) {
1370 switch (DiagnosticIDs::getDiagnosticSFINAEResponse(
1371 Diags.getCurrentDiagID())) {
1372 case DiagnosticIDs::SFINAE_Report:
1373 // We'll report the diagnostic below.
1374 break;
1375
1376 case DiagnosticIDs::SFINAE_SubstitutionFailure:
1377 // Count this failure so that we know that template argument deduction
1378 // has failed.
1379 ++NumSFINAEErrors;
1380
1381 // Make a copy of this suppressed diagnostic and store it with the
1382 // template-deduction information.
1383 if (*Info && !(*Info)->hasSFINAEDiagnostic()) {
1384 Diagnostic DiagInfo(&Diags);
1385 (*Info)->addSFINAEDiagnostic(DiagInfo.getLocation(),
1386 PartialDiagnostic(DiagInfo, Context.getDiagAllocator()));
1387 }
1388
1389 Diags.setLastDiagnosticIgnored();
1390 Diags.Clear();
1391 return;
1392
1393 case DiagnosticIDs::SFINAE_AccessControl: {
1394 // Per C++ Core Issue 1170, access control is part of SFINAE.
1395 // Additionally, the AccessCheckingSFINAE flag can be used to temporarily
1396 // make access control a part of SFINAE for the purposes of checking
1397 // type traits.
1398 if (!AccessCheckingSFINAE && !getLangOpts().CPlusPlus11)
1399 break;
1400
1401 SourceLocation Loc = Diags.getCurrentDiagLoc();
1402
1403 // Suppress this diagnostic.
1404 ++NumSFINAEErrors;
1405
1406 // Make a copy of this suppressed diagnostic and store it with the
1407 // template-deduction information.
1408 if (*Info && !(*Info)->hasSFINAEDiagnostic()) {
1409 Diagnostic DiagInfo(&Diags);
1410 (*Info)->addSFINAEDiagnostic(DiagInfo.getLocation(),
1411 PartialDiagnostic(DiagInfo, Context.getDiagAllocator()));
1412 }
1413
1414 Diags.setLastDiagnosticIgnored();
1415 Diags.Clear();
1416
1417 // Now the diagnostic state is clear, produce a C++98 compatibility
1418 // warning.
1419 Diag(Loc, diag::warn_cxx98_compat_sfinae_access_control);
1420
1421 // The last diagnostic which Sema produced was ignored. Suppress any
1422 // notes attached to it.
1423 Diags.setLastDiagnosticIgnored();
1424 return;
1425 }
1426
1427 case DiagnosticIDs::SFINAE_Suppress:
1428 // Make a copy of this suppressed diagnostic and store it with the
1429 // template-deduction information;
1430 if (*Info) {
1431 Diagnostic DiagInfo(&Diags);
1432 (*Info)->addSuppressedDiagnostic(DiagInfo.getLocation(),
1433 PartialDiagnostic(DiagInfo, Context.getDiagAllocator()));
1434 }
1435
1436 // Suppress this diagnostic.
1437 Diags.setLastDiagnosticIgnored();
1438 Diags.Clear();
1439 return;
1440 }
1441 }
1442
1443 // Copy the diagnostic printing policy over the ASTContext printing policy.
1444 // TODO: Stop doing that. See: https://reviews.llvm.org/D45093#1090292
1445 Context.setPrintingPolicy(getPrintingPolicy());
1446
1447 // Emit the diagnostic.
1448 if (!Diags.EmitCurrentDiagnostic())
1449 return;
1450
1451 // If this is not a note, and we're in a template instantiation
1452 // that is different from the last template instantiation where
1453 // we emitted an error, print a template instantiation
1454 // backtrace.
1455 if (!DiagnosticIDs::isBuiltinNote(DiagID))
1456 PrintContextStack();
1457}

1481// Emit any deferred diagnostics for FD and erase them from the map in which
1482// they're stored.
1483static void emitDeferredDiags(Sema &S, FunctionDecl *FD) {
1484 auto It = S.DeviceDeferredDiags.find(FD);
1485 if (It == S.DeviceDeferredDiags.end())
1486 return;
1487 bool HasWarningOrError = false;
1488 for (PartialDiagnosticAt &PDAt : It->second) {
1489 const SourceLocation &Loc = PDAt.first;
1490 const PartialDiagnostic &PD = PDAt.second;
1491 HasWarningOrError |= S.getDiagnostics().getDiagnosticLevel(
1492 PD.getDiagID(), Loc) >= DiagnosticsEngine::Warning;
1493 DiagnosticBuilder Builder(S.Diags.Report(Loc, PD.getDiagID()));
1494 Builder.setForceEmit();
1495 PD.Emit(Builder);
1496 }
1497 S.DeviceDeferredDiags.erase(It);
1498
1499 // FIXME: Should this be called after every warning/error emitted in the loop
1500 // above, instead of just once per function? That would be consistent with
1501 // how we handle immediate errors, but it also seems like a bit much.
1502 if (HasWarningOrError)
1503 emitCallStackNotes(S, FD);
1504}

1506// In CUDA, there are some constructs which may appear in semantically-valid
1507// code, but trigger errors if we ever generate code for the function in which
1508// they appear. Essentially every construct you're not allowed to use on the
1509// device falls into this category, because you are allowed to use these
1510// constructs in a __host__ __device__ function, but only if that function is
1511// never codegen'ed on the device.
1512//
1513// To handle semantic checking for these constructs, we keep track of the set of
1514// functions we know will be emitted, either because we could tell a priori that
1515// they would be emitted, or because they were transitively called by a
1516// known-emitted function.
1517//
1518// We also keep a partial call graph of which not-known-emitted functions call
1519// which other not-known-emitted functions.
1520//
1521// When we see something which is illegal if the current function is emitted
1522// (usually by way of CUDADiagIfDeviceCode, CUDADiagIfHostCode, or
1523// CheckCUDACall), we first check if the current function is known-emitted. If
1524// so, we immediately output the diagnostic.
1525//
1526// Otherwise, we "defer" the diagnostic. It sits in Sema::DeviceDeferredDiags
1527// until we discover that the function is known-emitted, at which point we take
1528// it out of this map and emit the diagnostic.
1529
1530Sema::DeviceDiagBuilder::DeviceDiagBuilder(Kind K, SourceLocation Loc,
1531 unsigned DiagID, FunctionDecl *Fn,
1532 Sema &S)
1533 : S(S), Loc(Loc), DiagID(DiagID), Fn(Fn),
1534 ShowCallStack(K == K_ImmediateWithCallStack || K == K_Deferred) {
1535 switch (K) {
1536 case K_Nop:
1537 break;
1538 case K_Immediate:
1539 case K_ImmediateWithCallStack:
1540 ImmediateDiag.emplace(S.Diag(Loc, DiagID));
1541 break;
1542 case K_Deferred:
1543 assert(Fn && "Must have a function to attach the deferred diag to.");
1544 auto &Diags = S.DeviceDeferredDiags[Fn];
1545 PartialDiagId.emplace(Diags.size());
1546 Diags.emplace_back(Loc, S.PDiag(DiagID));
1547 break;
1548 }
1549}

1551Sema::DeviceDiagBuilder::DeviceDiagBuilder(DeviceDiagBuilder &&D)
1552 : S(D.S), Loc(D.Loc), DiagID(D.DiagID), Fn(D.Fn),
1553 ShowCallStack(D.ShowCallStack), ImmediateDiag(D.ImmediateDiag),
1554 PartialDiagId(D.PartialDiagId) {
1555 // Clean the previous diagnostics.
1556 D.ShowCallStack = false;
1557 D.ImmediateDiag.reset();

1576// Indicate that this function (and thus everything it transtively calls) will
1577// be codegen'ed, and emit any deferred diagnostics on this function and its
1578// (transitive) callees.
1579void Sema::markKnownEmitted(
1580 Sema &S, FunctionDecl *OrigCaller, FunctionDecl *OrigCallee,
1581 SourceLocation OrigLoc,
1582 const llvm::function_ref<bool(Sema &, FunctionDecl *)> IsKnownEmitted) {
1583 // Nothing to do if we already know that FD is emitted.
1584 if (IsKnownEmitted(S, OrigCallee)) {
1585 assert(!S.DeviceCallGraph.count(OrigCallee));
1586 return;
1587 }
1588
1589 // We've just discovered that OrigCallee is known-emitted. Walk our call
1590 // graph to see what else we can now discover also must be emitted.
1591
1592 struct CallInfo {
1593 FunctionDecl *Caller;
1594 FunctionDecl *Callee;
1595 SourceLocation Loc;
1596 };
1597 llvm::SmallVector<CallInfo, 4> Worklist = {{OrigCaller, OrigCallee, OrigLoc}};
1598 llvm::SmallSet<CanonicalDeclPtr<FunctionDecl>, 4> Seen;
1599 Seen.insert(OrigCallee);
1600 while (!Worklist.empty()) {
1601 CallInfo C = Worklist.pop_back_val();
1602 assert(!IsKnownEmitted(S, C.Callee) &&
1603 "Worklist should not contain known-emitted functions.");
1604 S.DeviceKnownEmittedFns[C.Callee] = {C.Caller, C.Loc};
1605 emitDeferredDiags(S, C.Callee);
1606
1607 // If this is a template instantiation, explore its callgraph as well:
1608 // Non-dependent calls are part of the template's callgraph, while dependent
1609 // calls are part of to the instantiation's call graph.
1610 if (auto *Templ = C.Callee->getPrimaryTemplate()) {
1611 FunctionDecl *TemplFD = Templ->getAsFunction();
1612 if (!Seen.count(TemplFD) && !S.DeviceKnownEmittedFns.count(TemplFD)) {
1613 Seen.insert(TemplFD);
1614 Worklist.push_back(
1615 {/* Caller = */ C.Caller, /* Callee = */ TemplFD, C.Loc});
1616 }
1617 }
1618
1619 // Add all functions called by Callee to our worklist.
1620 auto CGIt = S.DeviceCallGraph.find(C.Callee);
1621 if (CGIt == S.DeviceCallGraph.end())
1622 continue;
1623
1624 for (std::pair<CanonicalDeclPtr<FunctionDecl>, SourceLocation> FDLoc :
1625 CGIt->second) {
1626 FunctionDecl *NewCallee = FDLoc.first;
1627 SourceLocation CallLoc = FDLoc.second;
1628 if (Seen.count(NewCallee) || IsKnownEmitted(S, NewCallee))
1629 continue;
1630 Seen.insert(NewCallee);
1631 Worklist.push_back(
1632 {/* Caller = */ C.Callee, /* Callee = */ NewCallee, CallLoc});
1633 }
1634
1635 // C.Callee is now known-emitted, so we no longer need to maintain its list
1636 // of callees in DeviceCallGraph.
1637 S.DeviceCallGraph.erase(CGIt);
1638 }
1639}

1734// Check that the type of the VarDecl has an accessible copy constructor and
1735// resolve its destructor's exception specification.
1736static void checkEscapingByref(VarDecl *VD, Sema &S) {
1737 QualType T = VD->getType();
1738 EnterExpressionEvaluationContext scope(
1739 S, Sema::ExpressionEvaluationContext::PotentiallyEvaluated);
1740 SourceLocation Loc = VD->getLocation();
1741 Expr *VarRef =
1742 new (S.Context) DeclRefExpr(S.Context, VD, false, T, VK_LValue, Loc);
1743 ExprResult Result = S.PerformMoveOrCopyInitialization(
1744 InitializedEntity::InitializeBlock(Loc, T, false), VD, VD->getType(),
1745 VarRef, /*AllowNRVO=*/true);
1746 if (!Result.isInvalid()) {
1747 Result = S.MaybeCreateExprWithCleanups(Result);
1748 Expr *Init = Result.getAs<Expr>();
1749 S.Context.setBlockVarCopyInit(VD, Init, S.canThrow(Init));
1750 }
1751
1752 // The destructor's exception specification is needed when IRGen generates
1753 // block copy/destroy functions. Resolve it here.
1754 if (const CXXRecordDecl *RD = T->getAsCXXRecordDecl())
1755 if (CXXDestructorDecl *DD = RD->getDestructor()) {
1756 auto *FPT = DD->getType()->getAs<FunctionProtoType>();
1757 S.ResolveExceptionSpec(Loc, FPT);
1758 }
1759}

1761static void markEscapingByrefs(const FunctionScopeInfo &FSI, Sema &S) {
1762 // Set the EscapingByref flag of __block variables captured by
1763 // escaping blocks.
1764 for (const BlockDecl *BD : FSI.Blocks) {
1765 if (BD->doesNotEscape())
1766 continue;
1767 for (const BlockDecl::Capture &BC : BD->captures()) {
1768 VarDecl *VD = BC.getVariable();
1769 if (VD->hasAttr<BlocksAttr>())
1770 VD->setEscapingByref();
1771 }
1772 }
1773
1774 for (VarDecl *VD : FSI.ByrefBlockVars) {
1775 // __block variables might require us to capture a copy-initializer.
1776 if (!VD->isEscapingByref())
1777 continue;
1778 // It's currently invalid to ever have a __block variable with an
1779 // array type; should we diagnose that here?
1780 // Regardless, we don't want to ignore array nesting when
1781 // constructing this copy.
1782 if (VD->getType()->isStructureOrClassType())
1783 checkEscapingByref(VD, S);
1784 }

1787/// Pop a function (or block or lambda or captured region) scope from the stack.
1788///
1789/// \param WP The warning policy to use for CFG-based warnings, or null if such
1790/// warnings should not be produced.
1791/// \param D The declaration corresponding to this function scope, if producing
1792/// CFG-based warnings.
1793/// \param BlockType The type of the block expression, if D is a BlockDecl.
1794Sema::PoppedFunctionScopePtr
1795Sema::PopFunctionScopeInfo(const AnalysisBasedWarnings::Policy *WP,
1796 const Decl *D, QualType BlockType) {
1797 assert(!FunctionScopes.empty() && "mismatched push/pop!");
1798
1799 markEscapingByrefs(*FunctionScopes.back(), *this);
1800
1801 PoppedFunctionScopePtr Scope(FunctionScopes.pop_back_val(),
1802 PoppedFunctionScopeDeleter(this));
1803
1804 if (LangOpts.OpenMP)
1805 popOpenMPFunctionRegion(Scope.get());
1806

1960/// Figure out if an expression could be turned into a call.
1961///
1962/// Use this when trying to recover from an error where the programmer may have
1963/// written just the name of a function instead of actually calling it.
1964///
1965/// \param E - The expression to examine.
1966/// \param ZeroArgCallReturnTy - If the expression can be turned into a call
1967/// with no arguments, this parameter is set to the type returned by such a
1968/// call; otherwise, it is set to an empty QualType.
1969/// \param OverloadSet - If the expression is an overloaded function
1970/// name, this parameter is populated with the decls of the various overloads.
1971bool Sema::tryExprAsCall(Expr &E, QualType &ZeroArgCallReturnTy,
1972 UnresolvedSetImpl &OverloadSet) {
1973 ZeroArgCallReturnTy = QualType();
1974 OverloadSet.clear();
1975
1976 const OverloadExpr *Overloads = nullptr;
1977 bool IsMemExpr = false;
1978 if (E.getType() == Context.OverloadTy) {
1979 OverloadExpr::FindResult FR = OverloadExpr::find(const_cast<Expr*>(&E));
1980
1981 // Ignore overloads that are pointer-to-member constants.
1982 if (FR.HasFormOfMemberPointer)
1983 return false;
1984
1985 Overloads = FR.Expression;
1986 } else if (E.getType() == Context.BoundMemberTy) {
1987 Overloads = dyn_cast<UnresolvedMemberExpr>(E.IgnoreParens());
1988 IsMemExpr = true;
1989 }
1990
1991 bool Ambiguous = false;
1992 bool IsMV = false;
1993
1994 if (Overloads) {
1995 for (OverloadExpr::decls_iterator it = Overloads->decls_begin(),
1996 DeclsEnd = Overloads->decls_end(); it != DeclsEnd; ++it) {
1997 OverloadSet.addDecl(*it);
1998
1999 // Check whether the function is a non-template, non-member which takes no
2000 // arguments.
2001 if (IsMemExpr)
2002 continue;
2003 if (const FunctionDecl *OverloadDecl
2004 = dyn_cast<FunctionDecl>((*it)->getUnderlyingDecl())) {
2005 if (OverloadDecl->getMinRequiredArguments() == 0) {
2006 if (!ZeroArgCallReturnTy.isNull() && !Ambiguous &&
2007 (!IsMV || !(OverloadDecl->isCPUDispatchMultiVersion() ||
2008 OverloadDecl->isCPUSpecificMultiVersion()))) {
2009 ZeroArgCallReturnTy = QualType();
2010 Ambiguous = true;
2011 } else {
2012 ZeroArgCallReturnTy = OverloadDecl->getReturnType();
2013 IsMV = OverloadDecl->isCPUDispatchMultiVersion() ||
2014 OverloadDecl->isCPUSpecificMultiVersion();
2015 }
2016 }
2017 }
2018 }
2019
2020 // If it's not a member, use better machinery to try to resolve the call
2021 if (!IsMemExpr)
2022 return !ZeroArgCallReturnTy.isNull();
2023 }
2024
2025 // Attempt to call the member with no arguments - this will correctly handle
2026 // member templates with defaults/deduction of template arguments, overloads
2027 // with default arguments, etc.
2028 if (IsMemExpr && !E.isTypeDependent()) {
2029 bool Suppress = getDiagnostics().getSuppressAllDiagnostics();
2030 getDiagnostics().setSuppressAllDiagnostics(true);
2031 ExprResult R = BuildCallToMemberFunction(nullptr, &E, SourceLocation(),
2032 None, SourceLocation());
2033 getDiagnostics().setSuppressAllDiagnostics(Suppress);
2034 if (R.isUsable()) {
2035 ZeroArgCallReturnTy = R.get()->getType();
2036 return true;
2037 }
2038 return false;
2039 }
2040
2041 if (const DeclRefExpr *DeclRef = dyn_cast<DeclRefExpr>(E.IgnoreParens())) {
2042 if (const FunctionDecl *Fun = dyn_cast<FunctionDecl>(DeclRef->getDecl())) {
2043 if (Fun->getMinRequiredArguments() == 0)
2044 ZeroArgCallReturnTy = Fun->getReturnType();
2045 return true;
2046 }
2047 }
2048
2049 // We don't have an expression that's convenient to get a FunctionDecl from,
2050 // but we can at least check if the type is "function of 0 arguments".
2051 QualType ExprTy = E.getType();
2052 const FunctionType *FunTy = nullptr;
2053 QualType PointeeTy = ExprTy->getPointeeType();
2054 if (!PointeeTy.isNull())
2055 FunTy = PointeeTy->getAs<FunctionType>();
2056 if (!FunTy)
2057 FunTy = ExprTy->getAs<FunctionType>();
2058
2059 if (const FunctionProtoType *FPT =
2060 dyn_cast_or_null<FunctionProtoType>(FunTy)) {
2061 if (FPT->getNumParams() == 0)
2062 ZeroArgCallReturnTy = FunTy->getReturnType();
2063 return true;
2064 }
2065 return false;
2066}

2068/// Give notes for a set of overloads.
2069///
2070/// A companion to tryExprAsCall. In cases when the name that the programmer
2071/// wrote was an overloaded function, we may be able to make some guesses about
2072/// plausible overloads based on their return types; such guesses can be handed
2073/// off to this method to be emitted as notes.
2074///
2075/// \param Overloads - The overloads to note.
2076/// \param FinalNoteLoc - If we've suppressed printing some overloads due to
2077/// -fshow-overloads=best, this is the location to attach to the note about too
2078/// many candidates. Typically this will be the location of the original
2079/// ill-formed expression.
2080static void noteOverloads(Sema &S, const UnresolvedSetImpl &Overloads,
2081 const SourceLocation FinalNoteLoc) {
2082 int ShownOverloads = 0;
2083 int SuppressedOverloads = 0;
2084 for (UnresolvedSetImpl::iterator It = Overloads.begin(),
2085 DeclsEnd = Overloads.end(); It != DeclsEnd; ++It) {
2086 // FIXME: Magic number for max shown overloads stolen from
2087 // OverloadCandidateSet::NoteCandidates.
2088 if (ShownOverloads >= 4 && S.Diags.getShowOverloads() == Ovl_Best) {
2089 ++SuppressedOverloads;
2090 continue;
2091 }
2092
2093 NamedDecl *Fn = (*It)->getUnderlyingDecl();
2094 // Don't print overloads for non-default multiversioned functions.
2095 if (const auto *FD = Fn->getAsFunction()) {
2096 if (FD->isMultiVersion() && FD->hasAttr<TargetAttr>() &&
2097 !FD->getAttr<TargetAttr>()->isDefaultVersion())
2098 continue;
2099 }
2100 S.Diag(Fn->getLocation(), diag::note_possible_target_of_call);
2101 ++ShownOverloads;
2102 }
2103
2104 if (SuppressedOverloads)
2105 S.Diag(FinalNoteLoc, diag::note_ovl_too_many_candidates)
2106 << SuppressedOverloads;
2107}

2109static void notePlausibleOverloads(Sema &S, SourceLocation Loc,
2110 const UnresolvedSetImpl &Overloads,
2111 bool (*IsPlausibleResult)(QualType)) {
2112 if (!IsPlausibleResult)
2113 return noteOverloads(S, Overloads, Loc);
2114
2115 UnresolvedSet<2> PlausibleOverloads;
2116 for (OverloadExpr::decls_iterator It = Overloads.begin(),
2117 DeclsEnd = Overloads.end(); It != DeclsEnd; ++It) {
2118 const FunctionDecl *OverloadDecl = cast<FunctionDecl>(*It);
2119 QualType OverloadResultTy = OverloadDecl->getReturnType();
2120 if (IsPlausibleResult(OverloadResultTy))
2121 PlausibleOverloads.addDecl(It.getDecl());
2122 }
2123 noteOverloads(S, PlausibleOverloads, Loc);
2124}

2153bool Sema::tryToRecoverWithCall(ExprResult &E, const PartialDiagnostic &PD,
2154 bool ForceComplain,
2155 bool (*IsPlausibleResult)(QualType)) {
2156 SourceLocation Loc = E.get()->getExprLoc();
2157 SourceRange Range = E.get()->getSourceRange();
2158
2159 QualType ZeroArgCallTy;
2160 UnresolvedSet<4> Overloads;
2161 if (tryExprAsCall(*E.get(), ZeroArgCallTy, Overloads) &&
2162 !ZeroArgCallTy.isNull() &&
2163 (!IsPlausibleResult || IsPlausibleResult(ZeroArgCallTy))) {
2164 // At this point, we know E is potentially callable with 0
2165 // arguments and that it returns something of a reasonable type,
2166 // so we can emit a fixit and carry on pretending that E was
2167 // actually a CallExpr.
2168 SourceLocation ParenInsertionLoc = getLocForEndOfToken(Range.getEnd());
2169 bool IsMV = IsCPUDispatchCPUSpecificMultiVersion(E.get());
2170 Diag(Loc, PD) << /*zero-arg*/ 1 << IsMV << Range
2171 << (IsCallableWithAppend(E.get())
2172 ? FixItHint::CreateInsertion(ParenInsertionLoc, "()")
2173 : FixItHint());
2174 if (!IsMV)
2175 notePlausibleOverloads(*this, Loc, Overloads, IsPlausibleResult);
2176
2177 // FIXME: Try this before emitting the fixit, and suppress diagnostics
2178 // while doing so.
2179 E = BuildCallExpr(nullptr, E.get(), Range.getEnd(), None,
2180 Range.getEnd().getLocWithOffset(1));
2181 return true;
2182 }
2183
2184 if (!ForceComplain) return false;
2185
2186 bool IsMV = IsCPUDispatchCPUSpecificMultiVersion(E.get());
2187 Diag(Loc, PD) << /*not zero-arg*/ 0 << IsMV << Range;
2188 if (!IsMV)
2189 notePlausibleOverloads(*this, Loc, Overloads, IsPlausibleResult);
2190 E = ExprError();
2191 return true;
2192}
