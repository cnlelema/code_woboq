27/// Examines the FunctionScopeInfo stack to determine the nearest
28/// enclosing lambda (to the current lambda) that is 'capture-ready' for
29/// the variable referenced in the current lambda (i.e. \p VarToCapture).
30/// If successful, returns the index into Sema's FunctionScopeInfo stack
31/// of the capture-ready lambda's LambdaScopeInfo.
32///
33/// Climbs down the stack of lambdas (deepest nested lambda - i.e. current
34/// lambda - is on top) to determine the index of the nearest enclosing/outer
35/// lambda that is ready to capture the \p VarToCapture being referenced in
36/// the current lambda.
37/// As we climb down the stack, we want the index of the first such lambda -
38/// that is the lambda with the highest index that is 'capture-ready'.
39///
40/// A lambda 'L' is capture-ready for 'V' (var or this) if:
41/// - its enclosing context is non-dependent
42/// - and if the chain of lambdas between L and the lambda in which
43/// V is potentially used (i.e. the lambda at the top of the scope info
44/// stack), can all capture or have already captured V.
45/// If \p VarToCapture is 'null' then we are trying to capture 'this'.
46///
47/// Note that a lambda that is deemed 'capture-ready' still needs to be checked
48/// for whether it is 'capture-capable' (see
49/// getStackIndexOfNearestEnclosingCaptureCapableLambda), before it can truly
50/// capture.
51///
52/// \param FunctionScopes - Sema's stack of nested FunctionScopeInfo's (which a
53/// LambdaScopeInfo inherits from). The current/deepest/innermost lambda
54/// is at the top of the stack and has the highest index.
55/// \param VarToCapture - the variable to capture. If NULL, capture 'this'.
56///
57/// \returns An Optional<unsigned> Index that if evaluates to 'true' contains
58/// the index (into Sema's FunctionScopeInfo stack) of the innermost lambda
59/// which is capture-ready. If the return value evaluates to 'false' then
60/// no lambda is capture-ready for \p VarToCapture.
61
62static inline Optional<unsigned>
63getStackIndexOfNearestEnclosingCaptureReadyLambda(
64 ArrayRef<const clang::sema::FunctionScopeInfo *> FunctionScopes,
65 VarDecl *VarToCapture) {
66 // Label failure to capture.
67 const Optional<unsigned> NoLambdaIsCaptureReady;
68
69 // Ignore all inner captured regions.
70 unsigned CurScopeIndex = FunctionScopes.size() - 1;
71 while (CurScopeIndex > 0 && isa<clang::sema::CapturedRegionScopeInfo>(
72 FunctionScopes[CurScopeIndex]))
73 --CurScopeIndex;
74 assert(
75 isa<clang::sema::LambdaScopeInfo>(FunctionScopes[CurScopeIndex]) &&
76 "The function on the top of sema's function-info stack must be a lambda");
77
78 // If VarToCapture is null, we are attempting to capture 'this'.
79 const bool IsCapturingThis = !VarToCapture;
80 const bool IsCapturingVariable = !IsCapturingThis;
81
82 // Start with the current lambda at the top of the stack (highest index).
83 DeclContext *EnclosingDC =
84 cast<sema::LambdaScopeInfo>(FunctionScopes[CurScopeIndex])->CallOperator;
85
86 do {
87 const clang::sema::LambdaScopeInfo *LSI =
88 cast<sema::LambdaScopeInfo>(FunctionScopes[CurScopeIndex]);
89 // IF we have climbed down to an intervening enclosing lambda that contains
90 // the variable declaration - it obviously can/must not capture the
91 // variable.
92 // Since its enclosing DC is dependent, all the lambdas between it and the
93 // innermost nested lambda are dependent (otherwise we wouldn't have
94 // arrived here) - so we don't yet have a lambda that can capture the
95 // variable.
96 if (IsCapturingVariable &&
97 VarToCapture->getDeclContext()->Equals(EnclosingDC))
98 return NoLambdaIsCaptureReady;
99
100 // For an enclosing lambda to be capture ready for an entity, all
101 // intervening lambda's have to be able to capture that entity. If even
102 // one of the intervening lambda's is not capable of capturing the entity
103 // then no enclosing lambda can ever capture that entity.
104 // For e.g.
105 // const int x = 10;
106 // [=](auto a) { #1
107 // [](auto b) { #2 <-- an intervening lambda that can never capture 'x'
108 // [=](auto c) { #3
109 // f(x, c); <-- can not lead to x's speculative capture by #1 or #2
110 // }; }; };
111 // If they do not have a default implicit capture, check to see
112 // if the entity has already been explicitly captured.
113 // If even a single dependent enclosing lambda lacks the capability
114 // to ever capture this variable, there is no further enclosing
115 // non-dependent lambda that can capture this variable.
116 if (LSI->ImpCaptureStyle == sema::LambdaScopeInfo::ImpCap_None) {
117 if (IsCapturingVariable && !LSI->isCaptured(VarToCapture))
118 return NoLambdaIsCaptureReady;
119 if (IsCapturingThis && !LSI->isCXXThisCaptured())
120 return NoLambdaIsCaptureReady;
121 }
122 EnclosingDC = getLambdaAwareParentOfDeclContext(EnclosingDC);
123
124 assert(CurScopeIndex);
125 --CurScopeIndex;
126 } while (!EnclosingDC->isTranslationUnit() &&
127 EnclosingDC->isDependentContext() &&
128 isLambdaCallOperator(EnclosingDC));
129
130 assert(CurScopeIndex < (FunctionScopes.size() - 1));
131 // If the enclosingDC is not dependent, then the immediately nested lambda
132 // (one index above) is capture-ready.
133 if (!EnclosingDC->isDependentContext())
134 return CurScopeIndex + 1;
135 return NoLambdaIsCaptureReady;
136}

138/// Examines the FunctionScopeInfo stack to determine the nearest
139/// enclosing lambda (to the current lambda) that is 'capture-capable' for
140/// the variable referenced in the current lambda (i.e. \p VarToCapture).
141/// If successful, returns the index into Sema's FunctionScopeInfo stack
142/// of the capture-capable lambda's LambdaScopeInfo.
143///
144/// Given the current stack of lambdas being processed by Sema and
145/// the variable of interest, to identify the nearest enclosing lambda (to the
146/// current lambda at the top of the stack) that can truly capture
147/// a variable, it has to have the following two properties:
148/// a) 'capture-ready' - be the innermost lambda that is 'capture-ready':
149/// - climb down the stack (i.e. starting from the innermost and examining
150/// each outer lambda step by step) checking if each enclosing
151/// lambda can either implicitly or explicitly capture the variable.
152/// Record the first such lambda that is enclosed in a non-dependent
153/// context. If no such lambda currently exists return failure.
154/// b) 'capture-capable' - make sure the 'capture-ready' lambda can truly
155/// capture the variable by checking all its enclosing lambdas:
156/// - check if all outer lambdas enclosing the 'capture-ready' lambda
157/// identified above in 'a' can also capture the variable (this is done
158/// via tryCaptureVariable for variables and CheckCXXThisCapture for
159/// 'this' by passing in the index of the Lambda identified in step 'a')
160///
161/// \param FunctionScopes - Sema's stack of nested FunctionScopeInfo's (which a
162/// LambdaScopeInfo inherits from). The current/deepest/innermost lambda
163/// is at the top of the stack.
164///
165/// \param VarToCapture - the variable to capture. If NULL, capture 'this'.
166///
167///
168/// \returns An Optional<unsigned> Index that if evaluates to 'true' contains
169/// the index (into Sema's FunctionScopeInfo stack) of the innermost lambda
170/// which is capture-capable. If the return value evaluates to 'false' then
171/// no lambda is capture-capable for \p VarToCapture.
172
173Optional<unsigned> clang::getStackIndexOfNearestEnclosingCaptureCapableLambda(
174 ArrayRef<const sema::FunctionScopeInfo *> FunctionScopes,
175 VarDecl *VarToCapture, Sema &S) {
176
177 const Optional<unsigned> NoLambdaIsCaptureCapable;
178
179 const Optional<unsigned> OptionalStackIndex =
180 getStackIndexOfNearestEnclosingCaptureReadyLambda(FunctionScopes,
181 VarToCapture);
182 if (!OptionalStackIndex)
183 return NoLambdaIsCaptureCapable;
184
185 const unsigned IndexOfCaptureReadyLambda = OptionalStackIndex.getValue();
186 assert(((IndexOfCaptureReadyLambda != (FunctionScopes.size() - 1)) ||
187 S.getCurGenericLambda()) &&
188 "The capture ready lambda for a potential capture can only be the "
189 "current lambda if it is a generic lambda");
190
191 const sema::LambdaScopeInfo *const CaptureReadyLambdaLSI =
192 cast<sema::LambdaScopeInfo>(FunctionScopes[IndexOfCaptureReadyLambda]);
193
194 // If VarToCapture is null, we are attempting to capture 'this'
195 const bool IsCapturingThis = !VarToCapture;
196 const bool IsCapturingVariable = !IsCapturingThis;
197
198 if (IsCapturingVariable) {
199 // Check if the capture-ready lambda can truly capture the variable, by
200 // checking whether all enclosing lambdas of the capture-ready lambda allow
201 // the capture - i.e. make sure it is capture-capable.
202 QualType CaptureType, DeclRefType;
203 const bool CanCaptureVariable =
204 !S.tryCaptureVariable(VarToCapture,
205 /*ExprVarIsUsedInLoc*/ SourceLocation(),
206 clang::Sema::TryCapture_Implicit,
207 /*EllipsisLoc*/ SourceLocation(),
208 /*BuildAndDiagnose*/ false, CaptureType,
209 DeclRefType, &IndexOfCaptureReadyLambda);
210 if (!CanCaptureVariable)
211 return NoLambdaIsCaptureCapable;
212 } else {
213 // Check if the capture-ready lambda can truly capture 'this' by checking
214 // whether all enclosing lambdas of the capture-ready lambda can capture
215 // 'this'.
216 const bool CanCaptureThis =
217 !S.CheckCXXThisCapture(
218 CaptureReadyLambdaLSI->PotentialThisCaptureLocation,
219 /*Explicit*/ false, /*BuildAndDiagnose*/ false,
220 &IndexOfCaptureReadyLambda);
221 if (!CanCaptureThis)
222 return NoLambdaIsCaptureCapable;
223 }
224 return IndexOfCaptureReadyLambda;
225}

241CXXRecordDecl *Sema::createLambdaClosureType(SourceRange IntroducerRange,
242 TypeSourceInfo *Info,
243 bool KnownDependent,
244 LambdaCaptureDefault CaptureDefault) {
245 DeclContext *DC = CurContext;
246 while (!(DC->isFunctionOrMethod() || DC->isRecord() || DC->isFileContext()))
247 DC = DC->getParent();
248 bool IsGenericLambda = getGenericLambdaTemplateParameterList(getCurLambda(),
249 *this);
250 // Start constructing the lambda class.
251 CXXRecordDecl *Class = CXXRecordDecl::CreateLambda(Context, DC, Info,
252 IntroducerRange.getBegin(),
253 KnownDependent,
254 IsGenericLambda,
255 CaptureDefault);
256 DC->addDecl(Class);
257
258 return Class;
259}

261/// Determine whether the given context is or is enclosed in an inline
262/// function.
263static bool isInInlineFunction(const DeclContext *DC) {
264 while (!DC->isFileContext()) {
265 if (const FunctionDecl *FD = dyn_cast<FunctionDecl>(DC))
266 if (FD->isInlined())
267 return true;
268
269 DC = DC->getLexicalParent();
270 }
271
272 return false;
273}

275MangleNumberingContext *
276Sema::getCurrentMangleNumberContext(const DeclContext *DC,
277 Decl *&ManglingContextDecl) {
278 // Compute the context for allocating mangling numbers in the current
279 // expression, if the ABI requires them.
280 ManglingContextDecl = ExprEvalContexts.back().ManglingContextDecl;
281
282 enum ContextKind {
283 Normal,
284 DefaultArgument,
285 DataMember,
286 StaticDataMember,
287 InlineVariable,
288 VariableTemplate
289 } Kind = Normal;
290
291 // Default arguments of member function parameters that appear in a class
292 // definition, as well as the initializers of data members, receive special
293 // treatment. Identify them.
294 if (ManglingContextDecl) {
295 if (ParmVarDecl *Param = dyn_cast<ParmVarDecl>(ManglingContextDecl)) {
296 if (const DeclContext *LexicalDC
297 = Param->getDeclContext()->getLexicalParent())
298 if (LexicalDC->isRecord())
299 Kind = DefaultArgument;
300 } else if (VarDecl *Var = dyn_cast<VarDecl>(ManglingContextDecl)) {
301 if (Var->getDeclContext()->isRecord())
302 Kind = StaticDataMember;
303 else if (Var->getMostRecentDecl()->isInline())
304 Kind = InlineVariable;
305 else if (Var->getDescribedVarTemplate())
306 Kind = VariableTemplate;
307 else if (auto *VTS = dyn_cast<VarTemplateSpecializationDecl>(Var)) {
308 if (!VTS->isExplicitSpecialization())
309 Kind = VariableTemplate;
310 }
311 } else if (isa<FieldDecl>(ManglingContextDecl)) {
312 Kind = DataMember;
313 }
314 }
315
316 // Itanium ABI [5.1.7]:
317 // In the following contexts [...] the one-definition rule requires closure
318 // types in different translation units to "correspond":
319 bool IsInNonspecializedTemplate =
320 inTemplateInstantiation() || CurContext->isDependentContext();
321 switch (Kind) {
322 case Normal: {
323 // -- the bodies of non-exported nonspecialized template functions
324 // -- the bodies of inline functions
325 if ((IsInNonspecializedTemplate &&
326 !(ManglingContextDecl && isa<ParmVarDecl>(ManglingContextDecl))) ||
327 isInInlineFunction(CurContext)) {
328 ManglingContextDecl = nullptr;
329 while (auto *CD = dyn_cast<CapturedDecl>(DC))
330 DC = CD->getParent();
331 return &Context.getManglingNumberContext(DC);
332 }
333
334 ManglingContextDecl = nullptr;
335 return nullptr;
336 }
337
338 case StaticDataMember:
339 // -- the initializers of nonspecialized static members of template classes
340 if (!IsInNonspecializedTemplate) {
341 ManglingContextDecl = nullptr;
342 return nullptr;
343 }
344 // Fall through to get the current context.
345 LLVM_FALLTHROUGH;
346
347 case DataMember:
348 // -- the in-class initializers of class members
349 case DefaultArgument:
350 // -- default arguments appearing in class definitions
351 case InlineVariable:
352 // -- the initializers of inline variables
353 case VariableTemplate:
354 // -- the initializers of templated variables
355 return &ExprEvalContexts.back().getMangleNumberingContext(Context);
356 }
357
358 llvm_unreachable("unexpected context");
359}

370CXXMethodDecl *Sema::startLambdaDefinition(
371 CXXRecordDecl *Class, SourceRange IntroducerRange,
372 TypeSourceInfo *MethodTypeInfo, SourceLocation EndLoc,
373 ArrayRef<ParmVarDecl *> Params, const bool IsConstexprSpecified,
374 Optional<std::pair<unsigned, Decl *>> Mangling) {
375 QualType MethodType = MethodTypeInfo->getType();
376 TemplateParameterList *TemplateParams =
377 getGenericLambdaTemplateParameterList(getCurLambda(), *this);
378 // If a lambda appears in a dependent context or is a generic lambda (has
379 // template parameters) and has an 'auto' return type, deduce it to a
380 // dependent type.
381 if (Class->isDependentContext() || TemplateParams) {
382 const FunctionProtoType *FPT = MethodType->castAs<FunctionProtoType>();
383 QualType Result = FPT->getReturnType();
384 if (Result->isUndeducedType()) {
385 Result = SubstAutoType(Result, Context.DependentTy);
386 MethodType = Context.getFunctionType(Result, FPT->getParamTypes(),
387 FPT->getExtProtoInfo());
388 }
389 }
390
391 // C++11 [expr.prim.lambda]p5:
392 // The closure type for a lambda-expression has a public inline function
393 // call operator (13.5.4) whose parameters and return type are described by
394 // the lambda-expression's parameter-declaration-clause and
395 // trailing-return-type respectively.
396 DeclarationName MethodName
397 = Context.DeclarationNames.getCXXOperatorName(OO_Call);
398 DeclarationNameLoc MethodNameLoc;
399 MethodNameLoc.CXXOperatorName.BeginOpNameLoc
400 = IntroducerRange.getBegin().getRawEncoding();
401 MethodNameLoc.CXXOperatorName.EndOpNameLoc
402 = IntroducerRange.getEnd().getRawEncoding();
403 CXXMethodDecl *Method
404 = CXXMethodDecl::Create(Context, Class, EndLoc,
405 DeclarationNameInfo(MethodName,
406 IntroducerRange.getBegin(),
407 MethodNameLoc),
408 MethodType, MethodTypeInfo,
409 SC_None,
410 /*isInline=*/true,
411 IsConstexprSpecified,
412 EndLoc);
413 Method->setAccess(AS_public);
414
415 // Temporarily set the lexical declaration context to the current
416 // context, so that the Scope stack matches the lexical nesting.
417 Method->setLexicalDeclContext(CurContext);
418 // Create a function template if we have a template parameter list
419 FunctionTemplateDecl *const TemplateMethod = TemplateParams ?
420 FunctionTemplateDecl::Create(Context, Class,
421 Method->getLocation(), MethodName,
422 TemplateParams,
423 Method) : nullptr;
424 if (TemplateMethod) {
425 TemplateMethod->setLexicalDeclContext(CurContext);
426 TemplateMethod->setAccess(AS_public);
427 Method->setDescribedFunctionTemplate(TemplateMethod);
428 }
429
430 // Add parameters.
431 if (!Params.empty()) {
432 Method->setParams(Params);
433 CheckParmsForFunctionDef(Params,
434 /*CheckParameterNames=*/false);
435
436 for (auto P : Method->parameters())
437 P->setOwningFunction(Method);
438 }
439
440 if (Mangling) {
441 Class->setLambdaMangling(Mangling->first, Mangling->second);
442 } else {
443 Decl *ManglingContextDecl;
444 if (MangleNumberingContext *MCtx =
445 getCurrentMangleNumberContext(Class->getDeclContext(),
446 ManglingContextDecl)) {
447 unsigned ManglingNumber = MCtx->getManglingNumber(Method);
448 Class->setLambdaMangling(ManglingNumber, ManglingContextDecl);
449 }
450 }
451
452 return Method;
453}

455void Sema::buildLambdaScope(LambdaScopeInfo *LSI,
456 CXXMethodDecl *CallOperator,
457 SourceRange IntroducerRange,
458 LambdaCaptureDefault CaptureDefault,
459 SourceLocation CaptureDefaultLoc,
460 bool ExplicitParams,
461 bool ExplicitResultType,
462 bool Mutable) {
463 LSI->CallOperator = CallOperator;
464 CXXRecordDecl *LambdaClass = CallOperator->getParent();
465 LSI->Lambda = LambdaClass;
466 if (CaptureDefault == LCD_ByCopy)
467 LSI->ImpCaptureStyle = LambdaScopeInfo::ImpCap_LambdaByval;
468 else if (CaptureDefault == LCD_ByRef)
469 LSI->ImpCaptureStyle = LambdaScopeInfo::ImpCap_LambdaByref;
470 LSI->CaptureDefaultLoc = CaptureDefaultLoc;
471 LSI->IntroducerRange = IntroducerRange;
472 LSI->ExplicitParams = ExplicitParams;
473 LSI->Mutable = Mutable;
474
475 if (ExplicitResultType) {
476 LSI->ReturnType = CallOperator->getReturnType();
477
478 if (!LSI->ReturnType->isDependentType() &&
479 !LSI->ReturnType->isVoidType()) {
480 if (RequireCompleteType(CallOperator->getBeginLoc(), LSI->ReturnType,
481 diag::err_lambda_incomplete_result)) {
482 // Do nothing.
483 }
484 }
485 } else {
486 LSI->HasImplicitReturnType = true;
487 }
488}

511void Sema::addLambdaParameters(
512 ArrayRef<LambdaIntroducer::LambdaCapture> Captures,
513 CXXMethodDecl *CallOperator, Scope *CurScope) {
514 // Introduce our parameters into the function scope
515 for (unsigned p = 0, NumParams = CallOperator->getNumParams();
516 p < NumParams; ++p) {
517 ParmVarDecl *Param = CallOperator->getParamDecl(p);
518
519 // If this has an identifier, add it to the scope stack.
520 if (CurScope && Param->getIdentifier()) {
521 bool Error = false;
522 // Resolution of CWG 2211 in C++17 renders shadowing ill-formed, but we
523 // retroactively apply it.
524 for (const auto &Capture : Captures) {
525 if (Capture.Id == Param->getIdentifier()) {
526 Error = true;
527 Diag(Param->getLocation(), diag::err_parameter_shadow_capture);
528 Diag(Capture.Loc, diag::note_var_explicitly_captured_here)
529 << Capture.Id << true;
530 }
531 }
532 if (!Error)
533 CheckShadow(CurScope, Param);
534
535 PushOnScopeChains(Param, CurScope);
536 }
537 }

540/// If this expression is an enumerator-like expression of some type
541/// T, return the type T; otherwise, return null.
542///
543/// Pointer comparisons on the result here should always work because
544/// it's derived from either the parent of an EnumConstantDecl
545/// (i.e. the definition) or the declaration returned by
546/// EnumType::getDecl() (i.e. the definition).
547static EnumDecl *findEnumForBlockReturn(Expr *E) {
548 // An expression is an enumerator-like expression of type T if,
549 // ignoring parens and parens-like expressions:
550 E = E->IgnoreParens();
551
552 // - it is an enumerator whose enum type is T or
553 if (DeclRefExpr *DRE = dyn_cast<DeclRefExpr>(E)) {
554 if (EnumConstantDecl *D
555 = dyn_cast<EnumConstantDecl>(DRE->getDecl())) {
556 return cast<EnumDecl>(D->getDeclContext());
557 }
558 return nullptr;
559 }
560
561 // - it is a comma expression whose RHS is an enumerator-like
562 // expression of type T or
563 if (BinaryOperator *BO = dyn_cast<BinaryOperator>(E)) {
564 if (BO->getOpcode() == BO_Comma)
565 return findEnumForBlockReturn(BO->getRHS());
566 return nullptr;
567 }
568
569 // - it is a statement-expression whose value expression is an
570 // enumerator-like expression of type T or
571 if (StmtExpr *SE = dyn_cast<StmtExpr>(E)) {
572 if (Expr *last = dyn_cast_or_null<Expr>(SE->getSubStmt()->body_back()))
573 return findEnumForBlockReturn(last);
574 return nullptr;
575 }
576
577 // - it is a ternary conditional operator (not the GNU ?:
578 // extension) whose second and third operands are
579 // enumerator-like expressions of type T or
580 if (ConditionalOperator *CO = dyn_cast<ConditionalOperator>(E)) {
581 if (EnumDecl *ED = findEnumForBlockReturn(CO->getTrueExpr()))
582 if (ED == findEnumForBlockReturn(CO->getFalseExpr()))
583 return ED;
584 return nullptr;
585 }
586
587 // (implicitly:)
588 // - it is an implicit integral conversion applied to an
589 // enumerator-like expression of type T or
590 if (ImplicitCastExpr *ICE = dyn_cast<ImplicitCastExpr>(E)) {
591 // We can sometimes see integral conversions in valid
592 // enumerator-like expressions.
593 if (ICE->getCastKind() == CK_IntegralCast)
594 return findEnumForBlockReturn(ICE->getSubExpr());
595
596 // Otherwise, just rely on the type.
597 }
598
599 // - it is an expression of that formal enum type.
600 if (const EnumType *ET = E->getType()->getAs<EnumType>()) {
601 return ET->getDecl();
602 }
603
604 // Otherwise, nope.
605 return nullptr;
606}

608/// Attempt to find a type T for which the returned expression of the
609/// given statement is an enumerator-like expression of that type.
610static EnumDecl *findEnumForBlockReturn(ReturnStmt *ret) {
611 if (Expr *retValue = ret->getRetValue())
612 return findEnumForBlockReturn(retValue);
613 return nullptr;
614}

616/// Attempt to find a common type T for which all of the returned
617/// expressions in a block are enumerator-like expressions of that
618/// type.
619static EnumDecl *findCommonEnumForBlockReturns(ArrayRef<ReturnStmt*> returns) {
620 ArrayRef<ReturnStmt*>::iterator i = returns.begin(), e = returns.end();
621
622 // Try to find one for the first return.
623 EnumDecl *ED = findEnumForBlockReturn(*i);
624 if (!ED) return nullptr;
625
626 // Check that the rest of the returns have the same enum.
627 for (++i; i != e; ++i) {
628 if (findEnumForBlockReturn(*i) != ED)
629 return nullptr;
630 }
631
632 // Never infer an anonymous enum type.
633 if (!ED->hasNameForLinkage()) return nullptr;
634
635 return ED;
636}

638/// Adjust the given return statements so that they formally return
639/// the given type. It should require, at most, an IntegralCast.
640static void adjustBlockReturnsToEnum(Sema &S, ArrayRef<ReturnStmt*> returns,
641 QualType returnType) {
642 for (ArrayRef<ReturnStmt*>::iterator
643 i = returns.begin(), e = returns.end(); i != e; ++i) {
644 ReturnStmt *ret = *i;
645 Expr *retValue = ret->getRetValue();
646 if (S.Context.hasSameType(retValue->getType(), returnType))
647 continue;
648
649 // Right now we only support integral fixup casts.
650 assert(returnType->isIntegralOrUnscopedEnumerationType());
651 assert(retValue->getType()->isIntegralOrUnscopedEnumerationType());
652
653 ExprWithCleanups *cleanups = dyn_cast<ExprWithCleanups>(retValue);
654
655 Expr *E = (cleanups ? cleanups->getSubExpr() : retValue);
656 E = ImplicitCastExpr::Create(S.Context, returnType, CK_IntegralCast,
657 E, /*base path*/ nullptr, VK_RValue);
658 if (cleanups) {
659 cleanups->setSubExpr(E);
660 } else {
661 ret->setRetValue(E);
662 }
663 }

666void Sema::deduceClosureReturnType(CapturingScopeInfo &CSI) {
667 assert(CSI.HasImplicitReturnType);
668 // If it was ever a placeholder, it had to been deduced to DependentTy.
669 assert(CSI.ReturnType.isNull() || !CSI.ReturnType->isUndeducedType());
670 assert((!isa<LambdaScopeInfo>(CSI) || !getLangOpts().CPlusPlus14) &&
671 "lambda expressions use auto deduction in C++14 onwards");
672
673 // C++ core issue 975:
674 // If a lambda-expression does not include a trailing-return-type,
675 // it is as if the trailing-return-type denotes the following type:
676 // - if there are no return statements in the compound-statement,
677 // or all return statements return either an expression of type
678 // void or no expression or braced-init-list, the type void;
679 // - otherwise, if all return statements return an expression
680 // and the types of the returned expressions after
681 // lvalue-to-rvalue conversion (4.1 [conv.lval]),
682 // array-to-pointer conversion (4.2 [conv.array]), and
683 // function-to-pointer conversion (4.3 [conv.func]) are the
684 // same, that common type;
685 // - otherwise, the program is ill-formed.
686 //
687 // C++ core issue 1048 additionally removes top-level cv-qualifiers
688 // from the types of returned expressions to match the C++14 auto
689 // deduction rules.
690 //
691 // In addition, in blocks in non-C++ modes, if all of the return
692 // statements are enumerator-like expressions of some type T, where
693 // T has a name for linkage, then we infer the return type of the
694 // block to be that type.
695
696 // First case: no return statements, implicit void return type.
697 ASTContext &Ctx = getASTContext();
698 if (CSI.Returns.empty()) {
699 // It's possible there were simply no /valid/ return statements.
700 // In this case, the first one we found may have at least given us a type.
701 if (CSI.ReturnType.isNull())
702 CSI.ReturnType = Ctx.VoidTy;
703 return;
704 }
705
706 // Second case: at least one return statement has dependent type.
707 // Delay type checking until instantiation.
708 assert(!CSI.ReturnType.isNull() && "We should have a tentative return type.");
709 if (CSI.ReturnType->isDependentType())
710 return;
711
712 // Try to apply the enum-fuzz rule.
713 if (!getLangOpts().CPlusPlus) {
714 assert(isa<BlockScopeInfo>(CSI));
715 const EnumDecl *ED = findCommonEnumForBlockReturns(CSI.Returns);
716 if (ED) {
717 CSI.ReturnType = Context.getTypeDeclType(ED);
718 adjustBlockReturnsToEnum(*this, CSI.Returns, CSI.ReturnType);
719 return;
720 }
721 }
722
723 // Third case: only one return statement. Don't bother doing extra work!
724 if (CSI.Returns.size() == 1)
725 return;
726
727 // General case: many return statements.
728 // Check that they all have compatible return types.
729
730 // We require the return types to strictly match here.
731 // Note that we've already done the required promotions as part of
732 // processing the return statement.
733 for (const ReturnStmt *RS : CSI.Returns) {
734 const Expr *RetE = RS->getRetValue();
735
736 QualType ReturnType =
737 (RetE ? RetE->getType() : Context.VoidTy).getUnqualifiedType();
738 if (Context.getCanonicalFunctionResultType(ReturnType) ==
739 Context.getCanonicalFunctionResultType(CSI.ReturnType)) {
740 // Use the return type with the strictest possible nullability annotation.
741 auto RetTyNullability = ReturnType->getNullability(Ctx);
742 auto BlockNullability = CSI.ReturnType->getNullability(Ctx);
743 if (BlockNullability &&
744 (!RetTyNullability ||
745 hasWeakerNullability(*RetTyNullability, *BlockNullability)))
746 CSI.ReturnType = ReturnType;
747 continue;
748 }
749
750 // FIXME: This is a poor diagnostic for ReturnStmts without expressions.
751 // TODO: It's possible that the *first* return is the divergent one.
752 Diag(RS->getBeginLoc(),
753 diag::err_typecheck_missing_return_type_incompatible)
754 << ReturnType << CSI.ReturnType << isa<LambdaScopeInfo>(CSI);
755 // Continue iterating so that we keep emitting diagnostics.
756 }
757}

759QualType Sema::buildLambdaInitCaptureInitialization(
760 SourceLocation Loc, bool ByRef, SourceLocation EllipsisLoc,
761 Optional<unsigned> NumExpansions, IdentifierInfo *Id, bool IsDirectInit,
762 Expr *&Init) {
763 // Create an 'auto' or 'auto&' TypeSourceInfo that we can use to
764 // deduce against.
765 QualType DeductType = Context.getAutoDeductType();
766 TypeLocBuilder TLB;
767 TLB.pushTypeSpec(DeductType).setNameLoc(Loc);
768 if (ByRef) {
769 DeductType = BuildReferenceType(DeductType, true, Loc, Id);
770 assert(!DeductType.isNull() && "can't build reference to auto");
771 TLB.push<ReferenceTypeLoc>(DeductType).setSigilLoc(Loc);
772 }
773 if (EllipsisLoc.isValid()) {
774 if (Init->containsUnexpandedParameterPack()) {
775 Diag(EllipsisLoc, getLangOpts().CPlusPlus2a
776 ? diag::warn_cxx17_compat_init_capture_pack
777 : diag::ext_init_capture_pack);
778 DeductType = Context.getPackExpansionType(DeductType, NumExpansions);
779 TLB.push<PackExpansionTypeLoc>(DeductType).setEllipsisLoc(EllipsisLoc);
780 } else {
781 // Just ignore the ellipsis for now and form a non-pack variable. We'll
782 // diagnose this later when we try to capture it.
783 }
784 }
785 TypeSourceInfo *TSI = TLB.getTypeSourceInfo(Context, DeductType);
786
787 // Deduce the type of the init capture.
788 QualType DeducedType = deduceVarTypeFromInitializer(
789 /*VarDecl*/nullptr, DeclarationName(Id), DeductType, TSI,
790 SourceRange(Loc, Loc), IsDirectInit, Init);
791 if (DeducedType.isNull())
792 return QualType();
793
794 // Are we a non-list direct initialization?
795 ParenListExpr *CXXDirectInit = dyn_cast<ParenListExpr>(Init);
796
797 // Perform initialization analysis and ensure any implicit conversions
798 // (such as lvalue-to-rvalue) are enforced.
799 InitializedEntity Entity =
800 InitializedEntity::InitializeLambdaCapture(Id, DeducedType, Loc);
801 InitializationKind Kind =
802 IsDirectInit
803 ? (CXXDirectInit ? InitializationKind::CreateDirect(
804 Loc, Init->getBeginLoc(), Init->getEndLoc())
805 : InitializationKind::CreateDirectList(Loc))
806 : InitializationKind::CreateCopy(Loc, Init->getBeginLoc());
807
808 MultiExprArg Args = Init;
809 if (CXXDirectInit)
810 Args =
811 MultiExprArg(CXXDirectInit->getExprs(), CXXDirectInit->getNumExprs());
812 QualType DclT;
813 InitializationSequence InitSeq(*this, Entity, Kind, Args);
814 ExprResult Result = InitSeq.Perform(*this, Entity, Kind, Args, &DclT);
815
816 if (Result.isInvalid())
817 return QualType();
818
819 Init = Result.getAs<Expr>();
820 return DeducedType;
821}

856void Sema::ActOnStartOfLambdaDefinition(LambdaIntroducer &Intro,
857 Declarator &ParamInfo,
858 Scope *CurScope) {
859 LambdaScopeInfo *const LSI = getCurLambda();
860 assert(LSI && "LambdaScopeInfo should be on stack!");
861
862 // Determine if we're within a context where we know that the lambda will
863 // be dependent, because there are template parameters in scope.
864 bool KnownDependent;
865 if (LSI->NumExplicitTemplateParams > 0) {
866 auto *TemplateParamScope = CurScope->getTemplateParamParent();
867 assert(TemplateParamScope &&
868 "Lambda with explicit template param list should establish a "
869 "template param scope");
870 assert(TemplateParamScope->getParent());
871 KnownDependent = TemplateParamScope->getParent()
872 ->getTemplateParamParent() != nullptr;
873 } else {
874 KnownDependent = CurScope->getTemplateParamParent() != nullptr;
875 }
876
877 // Determine the signature of the call operator.
878 TypeSourceInfo *MethodTyInfo;
879 bool ExplicitParams = true;
880 bool ExplicitResultType = true;
881 bool ContainsUnexpandedParameterPack = false;
882 SourceLocation EndLoc;
883 SmallVector<ParmVarDecl *, 8> Params;
884 if (ParamInfo.getNumTypeObjects() == 0) {
885 // C++11 [expr.prim.lambda]p4:
886 // If a lambda-expression does not include a lambda-declarator, it is as
887 // if the lambda-declarator were ().
888 FunctionProtoType::ExtProtoInfo EPI(Context.getDefaultCallingConvention(
889 /*IsVariadic=*/false, /*IsCXXMethod=*/true));
890 EPI.HasTrailingReturn = true;
891 EPI.TypeQuals.addConst();
892 // C++1y [expr.prim.lambda]:
893 // The lambda return type is 'auto', which is replaced by the
894 // trailing-return type if provided and/or deduced from 'return'
895 // statements
896 // We don't do this before C++1y, because we don't support deduced return
897 // types there.
898 QualType DefaultTypeForNoTrailingReturn =
899 getLangOpts().CPlusPlus14 ? Context.getAutoDeductType()
900 : Context.DependentTy;
901 QualType MethodTy =
902 Context.getFunctionType(DefaultTypeForNoTrailingReturn, None, EPI);
903 MethodTyInfo = Context.getTrivialTypeSourceInfo(MethodTy);
904 ExplicitParams = false;
905 ExplicitResultType = false;
906 EndLoc = Intro.Range.getEnd();
907 } else {
908 assert(ParamInfo.isFunctionDeclarator() &&
909 "lambda-declarator is a function");
910 DeclaratorChunk::FunctionTypeInfo &FTI = ParamInfo.getFunctionTypeInfo();
911
912 // C++11 [expr.prim.lambda]p5:
913 // This function call operator is declared const (9.3.1) if and only if
914 // the lambda-expression's parameter-declaration-clause is not followed
915 // by mutable. It is neither virtual nor declared volatile. [...]
916 if (!FTI.hasMutableQualifier()) {
917 FTI.getOrCreateMethodQualifiers().SetTypeQual(DeclSpec::TQ_const,
918 SourceLocation());
919 }
920
921 MethodTyInfo = GetTypeForDeclarator(ParamInfo, CurScope);
922 assert(MethodTyInfo && "no type from lambda-declarator");
923 EndLoc = ParamInfo.getSourceRange().getEnd();
924
925 ExplicitResultType = FTI.hasTrailingReturnType();
926
927 if (FTIHasNonVoidParameters(FTI)) {
928 Params.reserve(FTI.NumParams);
929 for (unsigned i = 0, e = FTI.NumParams; i != e; ++i)
930 Params.push_back(cast<ParmVarDecl>(FTI.Params[i].Param));
931 }
932
933 // Check for unexpanded parameter packs in the method type.
934 if (MethodTyInfo->getType()->containsUnexpandedParameterPack())
935 ContainsUnexpandedParameterPack = true;
936 }
937
938 CXXRecordDecl *Class = createLambdaClosureType(Intro.Range, MethodTyInfo,
939 KnownDependent, Intro.Default);
940
941 CXXMethodDecl *Method =
942 startLambdaDefinition(Class, Intro.Range, MethodTyInfo, EndLoc, Params,
943 ParamInfo.getDeclSpec().isConstexprSpecified());
944 if (ExplicitParams)
945 CheckCXXDefaultArguments(Method);
946
947 // This represents the function body for the lambda function, check if we
948 // have to apply optnone due to a pragma.
949 AddRangeBasedOptnone(Method);
950
951 // code_seg attribute on lambda apply to the method.
952 if (Attr *A = getImplicitCodeSegOrSectionAttrForFunction(Method, /*IsDefinition=*/true))
953 Method->addAttr(A);
954
955 // Attributes on the lambda apply to the method.
956 ProcessDeclAttributes(CurScope, Method, ParamInfo);
957
958 // CUDA lambdas get implicit attributes based on the scope in which they're
959 // declared.
960 if (getLangOpts().CUDA)
961 CUDASetLambdaAttrs(Method);
962
963 // Introduce the function call operator as the current declaration context.
964 PushDeclContext(CurScope, Method);
965
966 // Build the lambda scope.
967 buildLambdaScope(LSI, Method, Intro.Range, Intro.Default, Intro.DefaultLoc,
968 ExplicitParams, ExplicitResultType, !Method->isConst());
969
970 // C++11 [expr.prim.lambda]p9:
971 // A lambda-expression whose smallest enclosing scope is a block scope is a
972 // local lambda expression; any other lambda expression shall not have a
973 // capture-default or simple-capture in its lambda-introducer.
974 //
975 // For simple-captures, this is covered by the check below that any named
976 // entity is a variable that can be captured.
977 //
978 // For DR1632, we also allow a capture-default in any context where we can
979 // odr-use 'this' (in particular, in a default initializer for a non-static
980 // data member).
981 if (Intro.Default != LCD_None && !Class->getParent()->isFunctionOrMethod() &&
982 (getCurrentThisType().isNull() ||
983 CheckCXXThisCapture(SourceLocation(), /*Explicit*/true,
984 /*BuildAndDiagnose*/false)))
985 Diag(Intro.DefaultLoc, diag::err_capture_default_non_local);
986
987 // Distinct capture names, for diagnostics.
988 llvm::SmallSet<IdentifierInfo*, 8> CaptureNames;
989
990 // Handle explicit captures.
991 SourceLocation PrevCaptureLoc
992 = Intro.Default == LCD_None? Intro.Range.getBegin() : Intro.DefaultLoc;
993 for (auto C = Intro.Captures.begin(), E = Intro.Captures.end(); C != E;
994 PrevCaptureLoc = C->Loc, ++C) {
995 if (C->Kind == LCK_This || C->Kind == LCK_StarThis) {
996 if (C->Kind == LCK_StarThis)
997 Diag(C->Loc, !getLangOpts().CPlusPlus17
998 ? diag::ext_star_this_lambda_capture_cxx17
999 : diag::warn_cxx14_compat_star_this_lambda_capture);
1000
1001 // C++11 [expr.prim.lambda]p8:
1002 // An identifier or this shall not appear more than once in a
1003 // lambda-capture.
1004 if (LSI->isCXXThisCaptured()) {
1005 Diag(C->Loc, diag::err_capture_more_than_once)
1006 << "'this'" << SourceRange(LSI->getCXXThisCapture().getLocation())
1007 << FixItHint::CreateRemoval(
1008 SourceRange(getLocForEndOfToken(PrevCaptureLoc), C->Loc));
1009 continue;
1010 }
1011
1012 // C++2a [expr.prim.lambda]p8:
1013 // If a lambda-capture includes a capture-default that is =,
1014 // each simple-capture of that lambda-capture shall be of the form
1015 // "&identifier", "this", or "* this". [ Note: The form [&,this] is
1016 // redundant but accepted for compatibility with ISO C++14. --end note ]
1017 if (Intro.Default == LCD_ByCopy && C->Kind != LCK_StarThis)
1018 Diag(C->Loc, !getLangOpts().CPlusPlus2a
1019 ? diag::ext_equals_this_lambda_capture_cxx2a
1020 : diag::warn_cxx17_compat_equals_this_lambda_capture);
1021
1022 // C++11 [expr.prim.lambda]p12:
1023 // If this is captured by a local lambda expression, its nearest
1024 // enclosing function shall be a non-static member function.
1025 QualType ThisCaptureType = getCurrentThisType();
1026 if (ThisCaptureType.isNull()) {
1027 Diag(C->Loc, diag::err_this_capture) << true;
1028 continue;
1029 }
1030
1031 CheckCXXThisCapture(C->Loc, /*Explicit=*/true, /*BuildAndDiagnose*/ true,
1032 /*FunctionScopeIndexToStopAtPtr*/ nullptr,
1033 C->Kind == LCK_StarThis);
1034 if (!LSI->Captures.empty())
1035 LSI->ExplicitCaptureRanges[LSI->Captures.size() - 1] = C->ExplicitRange;
1036 continue;
1037 }
1038
1039 assert(C->Id && "missing identifier for capture");
1040
1041 if (C->Init.isInvalid())
1042 continue;
1043
1044 VarDecl *Var = nullptr;
1045 if (C->Init.isUsable()) {
1046 Diag(C->Loc, getLangOpts().CPlusPlus14
1047 ? diag::warn_cxx11_compat_init_capture
1048 : diag::ext_init_capture);
1049
1050 // If the initializer expression is usable, but the InitCaptureType
1051 // is not, then an error has occurred - so ignore the capture for now.
1052 // for e.g., [n{0}] { }; <-- if no <initializer_list> is included.
1053 // FIXME: we should create the init capture variable and mark it invalid
1054 // in this case.
1055 if (C->InitCaptureType.get().isNull())
1056 continue;
1057
1058 if (C->Init.get()->containsUnexpandedParameterPack() &&
1059 !C->InitCaptureType.get()->getAs<PackExpansionType>())
1060 ContainsUnexpandedParameterPack = true;
1061
1062 unsigned InitStyle;
1063 switch (C->InitKind) {
1064 case LambdaCaptureInitKind::NoInit:
1065 llvm_unreachable("not an init-capture?");
1066 case LambdaCaptureInitKind::CopyInit:
1067 InitStyle = VarDecl::CInit;
1068 break;
1069 case LambdaCaptureInitKind::DirectInit:
1070 InitStyle = VarDecl::CallInit;
1071 break;
1072 case LambdaCaptureInitKind::ListInit:
1073 InitStyle = VarDecl::ListInit;
1074 break;
1075 }
1076 Var = createLambdaInitCaptureVarDecl(C->Loc, C->InitCaptureType.get(),
1077 C->EllipsisLoc, C->Id, InitStyle,
1078 C->Init.get());
1079 // C++1y [expr.prim.lambda]p11:
1080 // An init-capture behaves as if it declares and explicitly
1081 // captures a variable [...] whose declarative region is the
1082 // lambda-expression's compound-statement
1083 if (Var)
1084 PushOnScopeChains(Var, CurScope, false);
1085 } else {
1086 assert(C->InitKind == LambdaCaptureInitKind::NoInit &&
1087 "init capture has valid but null init?");
1088
1089 // C++11 [expr.prim.lambda]p8:
1090 // If a lambda-capture includes a capture-default that is &, the
1091 // identifiers in the lambda-capture shall not be preceded by &.
1092 // If a lambda-capture includes a capture-default that is =, [...]
1093 // each identifier it contains shall be preceded by &.
1094 if (C->Kind == LCK_ByRef && Intro.Default == LCD_ByRef) {
1095 Diag(C->Loc, diag::err_reference_capture_with_reference_default)
1096 << FixItHint::CreateRemoval(
1097 SourceRange(getLocForEndOfToken(PrevCaptureLoc), C->Loc));
1098 continue;
1099 } else if (C->Kind == LCK_ByCopy && Intro.Default == LCD_ByCopy) {
1100 Diag(C->Loc, diag::err_copy_capture_with_copy_default)
1101 << FixItHint::CreateRemoval(
1102 SourceRange(getLocForEndOfToken(PrevCaptureLoc), C->Loc));
1103 continue;
1104 }
1105
1106 // C++11 [expr.prim.lambda]p10:
1107 // The identifiers in a capture-list are looked up using the usual
1108 // rules for unqualified name lookup (3.4.1)
1109 DeclarationNameInfo Name(C->Id, C->Loc);
1110 LookupResult R(*this, Name, LookupOrdinaryName);
1111 LookupName(R, CurScope);
1112 if (R.isAmbiguous())
1113 continue;
1114 if (R.empty()) {
1115 // FIXME: Disable corrections that would add qualification?
1116 CXXScopeSpec ScopeSpec;
1117 DeclFilterCCC<VarDecl> Validator{};
1118 if (DiagnoseEmptyLookup(CurScope, ScopeSpec, R, Validator))
1119 continue;
1120 }
1121
1122 Var = R.getAsSingle<VarDecl>();
1123 if (Var && DiagnoseUseOfDecl(Var, C->Loc))
1124 continue;
1125 }
1126
1127 // C++11 [expr.prim.lambda]p8:
1128 // An identifier or this shall not appear more than once in a
1129 // lambda-capture.
1130 if (!CaptureNames.insert(C->Id).second) {
1131 if (Var && LSI->isCaptured(Var)) {
1132 Diag(C->Loc, diag::err_capture_more_than_once)
1133 << C->Id << SourceRange(LSI->getCapture(Var).getLocation())
1134 << FixItHint::CreateRemoval(
1135 SourceRange(getLocForEndOfToken(PrevCaptureLoc), C->Loc));
1136 } else
1137 // Previous capture captured something different (one or both was
1138 // an init-cpature): no fixit.
1139 Diag(C->Loc, diag::err_capture_more_than_once) << C->Id;
1140 continue;
1141 }
1142
1143 // C++11 [expr.prim.lambda]p10:
1144 // [...] each such lookup shall find a variable with automatic storage
1145 // duration declared in the reaching scope of the local lambda expression.
1146 // Note that the 'reaching scope' check happens in tryCaptureVariable().
1147 if (!Var) {
1148 Diag(C->Loc, diag::err_capture_does_not_name_variable) << C->Id;
1149 continue;
1150 }
1151
1152 // Ignore invalid decls; they'll just confuse the code later.
1153 if (Var->isInvalidDecl())
1154 continue;
1155
1156 if (!Var->hasLocalStorage()) {
1157 Diag(C->Loc, diag::err_capture_non_automatic_variable) << C->Id;
1158 Diag(Var->getLocation(), diag::note_previous_decl) << C->Id;
1159 continue;
1160 }
1161
1162 // C++11 [expr.prim.lambda]p23:
1163 // A capture followed by an ellipsis is a pack expansion (14.5.3).
1164 SourceLocation EllipsisLoc;
1165 if (C->EllipsisLoc.isValid()) {
1166 if (Var->isParameterPack()) {
1167 EllipsisLoc = C->EllipsisLoc;
1168 } else {
1169 Diag(C->EllipsisLoc, diag::err_pack_expansion_without_parameter_packs)
1170 << (C->Init.isUsable() ? C->Init.get()->getSourceRange()
1171 : SourceRange(C->Loc));
1172
1173 // Just ignore the ellipsis.
1174 }
1175 } else if (Var->isParameterPack()) {
1176 ContainsUnexpandedParameterPack = true;
1177 }
1178
1179 if (C->Init.isUsable()) {
1180 addInitCapture(LSI, Var);
1181 } else {
1182 TryCaptureKind Kind = C->Kind == LCK_ByRef ? TryCapture_ExplicitByRef :
1183 TryCapture_ExplicitByVal;
1184 tryCaptureVariable(Var, C->Loc, Kind, EllipsisLoc);
1185 }
1186 if (!LSI->Captures.empty())
1187 LSI->ExplicitCaptureRanges[LSI->Captures.size() - 1] = C->ExplicitRange;
1188 }
1189 finishLambdaExplicitCaptures(LSI);
1190
1191 LSI->ContainsUnexpandedParameterPack = ContainsUnexpandedParameterPack;
1192
1193 // Add lambda parameters into scope.
1194 addLambdaParameters(Intro.Captures, Method, CurScope);
1195
1196 // Enter a new evaluation context to insulate the lambda from any
1197 // cleanups from the enclosing full-expression.
1198 PushExpressionEvaluationContext(
1199 ExpressionEvaluationContext::PotentiallyEvaluated);
1200}

1246 SourceRange IntroducerRange,
1247 CXXRecordDecl *Class,
1248 CXXMethodDecl *CallOperator) {
1249 // This conversion is explicitly disabled if the lambda's function has
1250 // pass_object_size attributes on any of its parameters.
1251 auto HasPassObjectSizeAttr = [](const ParmVarDecl *P) {
1252 return P->hasAttr<PassObjectSizeAttr>();
1253 };
1254 if (llvm::any_of(CallOperator->parameters(), HasPassObjectSizeAttr))
1255 return;
1256
1257 // Add the conversion to function pointer.
1258 QualType InvokerFunctionTy = S.getLambdaConversionFunctionResultType(
1259 CallOperator->getType()->castAs<FunctionProtoType>());
1260 QualType PtrToFunctionTy = S.Context.getPointerType(InvokerFunctionTy);
1261
1262 // Create the type of the conversion function.
1263 FunctionProtoType::ExtProtoInfo ConvExtInfo(
1264 S.Context.getDefaultCallingConvention(
1265 /*IsVariadic=*/false, /*IsCXXMethod=*/true));
1266 // The conversion function is always const and noexcept.
1267 ConvExtInfo.TypeQuals = Qualifiers();
1268 ConvExtInfo.TypeQuals.addConst();
1269 ConvExtInfo.ExceptionSpec.Type = EST_BasicNoexcept;
1270 QualType ConvTy =
1271 S.Context.getFunctionType(PtrToFunctionTy, None, ConvExtInfo);
1272
1273 SourceLocation Loc = IntroducerRange.getBegin();
1274 DeclarationName ConversionName
1275 = S.Context.DeclarationNames.getCXXConversionFunctionName(
1276 S.Context.getCanonicalType(PtrToFunctionTy));
1277 DeclarationNameLoc ConvNameLoc;
1278 // Construct a TypeSourceInfo for the conversion function, and wire
1279 // all the parameters appropriately for the FunctionProtoTypeLoc
1280 // so that everything works during transformation/instantiation of
1281 // generic lambdas.
1282 // The main reason for wiring up the parameters of the conversion
1283 // function with that of the call operator is so that constructs
1284 // like the following work:
1285 // auto L = [](auto b) { <-- 1
1286 // return [](auto a) -> decltype(a) { <-- 2
1287 // return a;
1288 // };
1289 // };
1290 // int (*fp)(int) = L(5);
1291 // Because the trailing return type can contain DeclRefExprs that refer
1292 // to the original call operator's variables, we hijack the call
1293 // operators ParmVarDecls below.
1294 TypeSourceInfo *ConvNamePtrToFunctionTSI =
1295 S.Context.getTrivialTypeSourceInfo(PtrToFunctionTy, Loc);
1296 ConvNameLoc.NamedType.TInfo = ConvNamePtrToFunctionTSI;
1297
1298 // The conversion function is a conversion to a pointer-to-function.
1299 TypeSourceInfo *ConvTSI = S.Context.getTrivialTypeSourceInfo(ConvTy, Loc);
1300 FunctionProtoTypeLoc ConvTL =
1301 ConvTSI->getTypeLoc().getAs<FunctionProtoTypeLoc>();
1302 // Get the result of the conversion function which is a pointer-to-function.
1303 PointerTypeLoc PtrToFunctionTL =
1304 ConvTL.getReturnLoc().getAs<PointerTypeLoc>();
1305 // Do the same for the TypeSourceInfo that is used to name the conversion
1306 // operator.
1307 PointerTypeLoc ConvNamePtrToFunctionTL =
1308 ConvNamePtrToFunctionTSI->getTypeLoc().getAs<PointerTypeLoc>();
1309
1310 // Get the underlying function types that the conversion function will
1311 // be converting to (should match the type of the call operator).
1312 FunctionProtoTypeLoc CallOpConvTL =
1313 PtrToFunctionTL.getPointeeLoc().getAs<FunctionProtoTypeLoc>();
1314 FunctionProtoTypeLoc CallOpConvNameTL =
1315 ConvNamePtrToFunctionTL.getPointeeLoc().getAs<FunctionProtoTypeLoc>();
1316
1317 // Wire up the FunctionProtoTypeLocs with the call operator's parameters.
1318 // These parameter's are essentially used to transform the name and
1319 // the type of the conversion operator. By using the same parameters
1320 // as the call operator's we don't have to fix any back references that
1321 // the trailing return type of the call operator's uses (such as
1322 // decltype(some_type<decltype(a)>::type{} + decltype(a){}) etc.)
1323 // - we can simply use the return type of the call operator, and
1324 // everything should work.
1325 SmallVector<ParmVarDecl *, 4> InvokerParams;
1326 for (unsigned I = 0, N = CallOperator->getNumParams(); I != N; ++I) {
1327 ParmVarDecl *From = CallOperator->getParamDecl(I);
1328
1329 InvokerParams.push_back(ParmVarDecl::Create(
1330 S.Context,
1331 // Temporarily add to the TU. This is set to the invoker below.
1332 S.Context.getTranslationUnitDecl(), From->getBeginLoc(),
1333 From->getLocation(), From->getIdentifier(), From->getType(),
1334 From->getTypeSourceInfo(), From->getStorageClass(),
1335 /*DefaultArg=*/nullptr));
1336 CallOpConvTL.setParam(I, From);
1337 CallOpConvNameTL.setParam(I, From);
1338 }
1339
1340 CXXConversionDecl *Conversion = CXXConversionDecl::Create(
1341 S.Context, Class, Loc,
1342 DeclarationNameInfo(ConversionName, Loc, ConvNameLoc), ConvTy, ConvTSI,
1343 /*isInline=*/true, ExplicitSpecifier(),
1344 /*isConstexpr=*/S.getLangOpts().CPlusPlus17,
1345 CallOperator->getBody()->getEndLoc());
1346 Conversion->setAccess(AS_public);
1347 Conversion->setImplicit(true);
1348
1349 if (Class->isGenericLambda()) {
1350 // Create a template version of the conversion operator, using the template
1351 // parameter list of the function call operator.
1352 FunctionTemplateDecl *TemplateCallOperator =
1353 CallOperator->getDescribedFunctionTemplate();
1354 FunctionTemplateDecl *ConversionTemplate =
1355 FunctionTemplateDecl::Create(S.Context, Class,
1356 Loc, ConversionName,
1357 TemplateCallOperator->getTemplateParameters(),
1358 Conversion);
1359 ConversionTemplate->setAccess(AS_public);
1360 ConversionTemplate->setImplicit(true);
1361 Conversion->setDescribedFunctionTemplate(ConversionTemplate);
1362 Class->addDecl(ConversionTemplate);
1363 } else
1364 Class->addDecl(Conversion);
1365 // Add a non-static member function that will be the result of
1366 // the conversion with a certain unique ID.
1367 DeclarationName InvokerName = &S.Context.Idents.get(
1368 getLambdaStaticInvokerName());
1369 // FIXME: Instead of passing in the CallOperator->getTypeSourceInfo()
1370 // we should get a prebuilt TrivialTypeSourceInfo from Context
1371 // using FunctionTy & Loc and get its TypeLoc as a FunctionProtoTypeLoc
1372 // then rewire the parameters accordingly, by hoisting up the InvokeParams
1373 // loop below and then use its Params to set Invoke->setParams(...) below.
1374 // This would avoid the 'const' qualifier of the calloperator from
1375 // contaminating the type of the invoker, which is currently adjusted
1376 // in SemaTemplateDeduction.cpp:DeduceTemplateArguments. Fixing the
1377 // trailing return type of the invoker would require a visitor to rebuild
1378 // the trailing return type and adjusting all back DeclRefExpr's to refer
1379 // to the new static invoker parameters - not the call operator's.
1380 CXXMethodDecl *Invoke = CXXMethodDecl::Create(
1381 S.Context, Class, Loc, DeclarationNameInfo(InvokerName, Loc),
1382 InvokerFunctionTy, CallOperator->getTypeSourceInfo(), SC_Static,
1383 /*IsInline=*/true,
1384 /*IsConstexpr=*/false, CallOperator->getBody()->getEndLoc());
1385 for (unsigned I = 0, N = CallOperator->getNumParams(); I != N; ++I)
1386 InvokerParams[I]->setOwningFunction(Invoke);
1387 Invoke->setParams(InvokerParams);
1388 Invoke->setAccess(AS_private);
1389 Invoke->setImplicit(true);
1390 if (Class->isGenericLambda()) {
1391 FunctionTemplateDecl *TemplateCallOperator =
1392 CallOperator->getDescribedFunctionTemplate();
1393 FunctionTemplateDecl *StaticInvokerTemplate = FunctionTemplateDecl::Create(
1394 S.Context, Class, Loc, InvokerName,
1395 TemplateCallOperator->getTemplateParameters(),
1396 Invoke);
1397 StaticInvokerTemplate->setAccess(AS_private);
1398 StaticInvokerTemplate->setImplicit(true);
1399 Invoke->setDescribedFunctionTemplate(StaticInvokerTemplate);
1400 Class->addDecl(StaticInvokerTemplate);
1401 } else
1402 Class->addDecl(Invoke);
1403}

1437ExprResult Sema::BuildCaptureInit(const Capture &Cap,
1438 SourceLocation ImplicitCaptureLoc,
1439 bool IsOpenMPMapping) {
1440 // VLA captures don't have a stored initialization expression.
1441 if (Cap.isVLATypeCapture())
1442 return ExprResult();
1443
1444 // An init-capture is initialized directly from its stored initializer.
1445 if (Cap.isInitCapture())
1446 return Cap.getVariable()->getInit();
1447
1448 // For anything else, build an initialization expression. For an implicit
1449 // capture, the capture notionally happens at the capture-default, so use
1450 // that location here.
1451 SourceLocation Loc =
1452 ImplicitCaptureLoc.isValid() ? ImplicitCaptureLoc : Cap.getLocation();
1453
1454 // C++11 [expr.prim.lambda]p21:
1455 // When the lambda-expression is evaluated, the entities that
1456 // are captured by copy are used to direct-initialize each
1457 // corresponding non-static data member of the resulting closure
1458 // object. (For array members, the array elements are
1459 // direct-initialized in increasing subscript order.) These
1460 // initializations are performed in the (unspecified) order in
1461 // which the non-static data members are declared.
1462
1463 // C++ [expr.prim.lambda]p12:
1464 // An entity captured by a lambda-expression is odr-used (3.2) in
1465 // the scope containing the lambda-expression.
1466 ExprResult Init;
1467 IdentifierInfo *Name = nullptr;
1468 if (Cap.isThisCapture()) {
1469 QualType ThisTy = getCurrentThisType();
1470 Expr *This = BuildCXXThisExpr(Loc, ThisTy, ImplicitCaptureLoc.isValid());
1471 if (Cap.isCopyCapture())
1472 Init = CreateBuiltinUnaryOp(Loc, UO_Deref, This);
1473 else
1474 Init = This;
1475 } else {
1476 assert(Cap.isVariableCapture() && "unknown kind of capture");
1477 VarDecl *Var = Cap.getVariable();
1478 Name = Var->getIdentifier();
1479 Init = BuildDeclarationNameExpr(
1480 CXXScopeSpec(), DeclarationNameInfo(Var->getDeclName(), Loc), Var);
1481 }
1482
1483 // In OpenMP, the capture kind doesn't actually describe how to capture:
1484 // variables are "mapped" onto the device in a process that does not formally
1485 // make a copy, even for a "copy capture".
1486 if (IsOpenMPMapping)
1487 return Init;
1488
1489 if (Init.isInvalid())
1490 return ExprError();
1491
1492 Expr *InitExpr = Init.get();
1493 InitializedEntity Entity = InitializedEntity::InitializeLambdaCapture(
1494 Name, Cap.getCaptureType(), Loc);
1495 InitializationKind InitKind =
1496 InitializationKind::CreateDirect(Loc, Loc, Loc);
1497 InitializationSequence InitSeq(*this, Entity, InitKind, InitExpr);
1498 return InitSeq.Perform(*this, Entity, InitKind, InitExpr);
1499}

1524bool Sema::CaptureHasSideEffects(const Capture &From) {
1525 if (From.isInitCapture()) {
1526 Expr *Init = From.getVariable()->getInit();
1527 if (Init && Init->HasSideEffects(Context))
1528 return true;
1529 }
1530
1531 if (!From.isCopyCapture())
1532 return false;
1533
1534 const QualType T = From.isThisCapture()
1535 ? getCurrentThisType()->getPointeeType()
1536 : From.getCaptureType();
1537
1538 if (T.isVolatileQualified())
1539 return true;
1540
1541 const Type *BaseT = T->getBaseElementTypeUnsafe();
1542 if (const CXXRecordDecl *RD = BaseT->getAsCXXRecordDecl())
1543 return !RD->isCompleteDefinition() || !RD->hasTrivialCopyConstructor() ||
1544 !RD->hasTrivialDestructor();
1545
1546 return false;
1547}

1570 const sema::Capture &Capture) {

1572 QualType FieldType = Capture.getCaptureType();
1573
1574 TypeSourceInfo *TSI = nullptr;

1576 auto *Var = Capture.getVariable();

1578 TSI = Capture.getVariable()->getTypeSourceInfo();
1579 }
1580
1581 // FIXME: Should we really be doing this? A null TypeSourceInfo seems more
1582 // appropriate, at least for an implicit capture.
1583 if (!TSI)
1584 TSI = Context.getTrivialTypeSourceInfo(FieldType, Loc);
1585

1587 FieldDecl *Field =
1588 FieldDecl::Create(Context, RD, Loc, Loc, nullptr, FieldType, TSI, nullptr,

1591 // invalid as well.

1594 RD->setInvalidDecl();

1598 FieldType->isIncompleteType(&Def);
1599 if (Def && Def->isInvalidDecl()) {
1600 RD->setInvalidDecl();
1601 Field->setInvalidDecl();
1602 }
1603 }
1604 }
1605 Field->setImplicit(true);
1606 Field->setAccess(AS_private);
1607 RD->addDecl(Field);
1608
1609 if (Capture.isVLATypeCapture())
1610 Field->setCapturedVLAType(Capture.getCapturedVLAType());
1611

1615ExprResult Sema::BuildLambdaExpr(SourceLocation StartLoc, SourceLocation EndLoc,
1616 LambdaScopeInfo *LSI) {
1617 // Collect information from the lambda scope.
1618 SmallVector<LambdaCapture, 4> Captures;
1619 SmallVector<Expr *, 4> CaptureInits;
1620 SourceLocation CaptureDefaultLoc = LSI->CaptureDefaultLoc;
1621 LambdaCaptureDefault CaptureDefault =
1622 mapImplicitCaptureStyle(LSI->ImpCaptureStyle);
1623 CXXRecordDecl *Class;
1624 CXXMethodDecl *CallOperator;
1625 SourceRange IntroducerRange;
1626 bool ExplicitParams;
1627 bool ExplicitResultType;
1628 CleanupInfo LambdaCleanup;
1629 bool ContainsUnexpandedParameterPack;
1630 bool IsGenericLambda;
1631 {
1632 CallOperator = LSI->CallOperator;
1633 Class = LSI->Lambda;
1634 IntroducerRange = LSI->IntroducerRange;
1635 ExplicitParams = LSI->ExplicitParams;
1636 ExplicitResultType = !LSI->HasImplicitReturnType;
1637 LambdaCleanup = LSI->Cleanup;
1638 ContainsUnexpandedParameterPack = LSI->ContainsUnexpandedParameterPack;
1639 IsGenericLambda = Class->isGenericLambda();
1640
1641 CallOperator->setLexicalDeclContext(Class);
1642 Decl *TemplateOrNonTemplateCallOperatorDecl =
1643 CallOperator->getDescribedFunctionTemplate()
1644 ? CallOperator->getDescribedFunctionTemplate()
1645 : cast<Decl>(CallOperator);
1646
1647 TemplateOrNonTemplateCallOperatorDecl->setLexicalDeclContext(Class);
1648 Class->addDecl(TemplateOrNonTemplateCallOperatorDecl);
1649
1650 PopExpressionEvaluationContext();
1651
1652 // True if the current capture has a used capture or default before it.
1653 bool CurHasPreviousCapture = CaptureDefault != LCD_None;
1654 SourceLocation PrevCaptureLoc = CurHasPreviousCapture ?
1655 CaptureDefaultLoc : IntroducerRange.getBegin();
1656
1657 for (unsigned I = 0, N = LSI->Captures.size(); I != N; ++I) {
1658 const Capture &From = LSI->Captures[I];
1659
1660 if (From.isInvalid())
1661 return ExprError();
1662
1663 assert(!From.isBlockCapture() && "Cannot capture __block variables");
1664 bool IsImplicit = I >= LSI->NumExplicitCaptures;
1665 SourceLocation ImplicitCaptureLoc =
1666 IsImplicit ? CaptureDefaultLoc : SourceLocation();
1667
1668 // Use source ranges of explicit captures for fixits where available.
1669 SourceRange CaptureRange = LSI->ExplicitCaptureRanges[I];
1670
1671 // Warn about unused explicit captures.
1672 bool IsCaptureUsed = true;
1673 if (!CurContext->isDependentContext() && !IsImplicit &&
1674 !From.isODRUsed()) {
1675 // Initialized captures that are non-ODR used may not be eliminated.
1676 // FIXME: Where did the IsGenericLambda here come from?
1677 bool NonODRUsedInitCapture =
1678 IsGenericLambda && From.isNonODRUsed() && From.isInitCapture();
1679 if (!NonODRUsedInitCapture) {
1680 bool IsLast = (I + 1) == LSI->NumExplicitCaptures;
1681 SourceRange FixItRange;
1682 if (CaptureRange.isValid()) {
1683 if (!CurHasPreviousCapture && !IsLast) {
1684 // If there are no captures preceding this capture, remove the
1685 // following comma.
1686 FixItRange = SourceRange(CaptureRange.getBegin(),
1687 getLocForEndOfToken(CaptureRange.getEnd()));
1688 } else {
1689 // Otherwise, remove the comma since the last used capture.
1690 FixItRange = SourceRange(getLocForEndOfToken(PrevCaptureLoc),
1691 CaptureRange.getEnd());
1692 }
1693 }
1694
1695 IsCaptureUsed = !DiagnoseUnusedLambdaCapture(FixItRange, From);
1696 }
1697 }
1698
1699 if (CaptureRange.isValid()) {
1700 CurHasPreviousCapture |= IsCaptureUsed;
1701 PrevCaptureLoc = CaptureRange.getEnd();
1702 }
1703
1704 // Map the capture to our AST representation.
1705 LambdaCapture Capture = [&] {
1706 if (From.isThisCapture()) {
1707 // Capturing 'this' implicitly with a default of '[=]' is deprecated,
1708 // because it results in a reference capture. Don't warn prior to
1709 // C++2a; there's nothing that can be done about it before then.
1710 if (getLangOpts().CPlusPlus2a && IsImplicit &&
1711 CaptureDefault == LCD_ByCopy) {
1712 Diag(From.getLocation(), diag::warn_deprecated_this_capture);
1713 Diag(CaptureDefaultLoc, diag::note_deprecated_this_capture)
1714 << FixItHint::CreateInsertion(
1715 getLocForEndOfToken(CaptureDefaultLoc), ", this");
1716 }
1717 return LambdaCapture(From.getLocation(), IsImplicit,
1718 From.isCopyCapture() ? LCK_StarThis : LCK_This);
1719 } else if (From.isVLATypeCapture()) {
1720 return LambdaCapture(From.getLocation(), IsImplicit, LCK_VLAType);
1721 } else {
1722 assert(From.isVariableCapture() && "unknown kind of capture");
1723 VarDecl *Var = From.getVariable();
1724 LambdaCaptureKind Kind =
1725 From.isCopyCapture() ? LCK_ByCopy : LCK_ByRef;
1726 return LambdaCapture(From.getLocation(), IsImplicit, Kind, Var,
1727 From.getEllipsisLoc());
1728 }
1729 }();
1730
1731 // Form the initializer for the capture field.
1732 ExprResult Init = BuildCaptureInit(From, ImplicitCaptureLoc);
1733
1734 // FIXME: Skip this capture if the capture is not used, the initializer
1735 // has no side-effects, the type of the capture is trivial, and the
1736 // lambda is not externally visible.
1737
1738 // Add a FieldDecl for the capture and form its initializer.
1739 BuildCaptureField(Class, From);
1740 Captures.push_back(Capture);
1741 CaptureInits.push_back(Init.get());
1742 }
1743
1744 // C++11 [expr.prim.lambda]p6:
1745 // The closure type for a lambda-expression with no lambda-capture
1746 // has a public non-virtual non-explicit const conversion function
1747 // to pointer to function having the same parameter and return
1748 // types as the closure type's function call operator.
1749 if (Captures.empty() && CaptureDefault == LCD_None)
1750 addFunctionPointerConversion(*this, IntroducerRange, Class,
1751 CallOperator);
1752
1753 // Objective-C++:
1754 // The closure type for a lambda-expression has a public non-virtual
1755 // non-explicit const conversion function to a block pointer having the
1756 // same parameter and return types as the closure type's function call
1757 // operator.
1758 // FIXME: Fix generic lambda to block conversions.
1759 if (getLangOpts().Blocks && getLangOpts().ObjC && !IsGenericLambda)
1760 addBlockPointerConversion(*this, IntroducerRange, Class, CallOperator);
1761
1762 // Finalize the lambda class.
1763 SmallVector<Decl*, 4> Fields(Class->fields());
1764 ActOnFields(nullptr, Class->getLocation(), Class, Fields, SourceLocation(),
1765 SourceLocation(), ParsedAttributesView());
1766 CheckCompletedCXXClass(Class);
1767 }
1768
1769 Cleanup.mergeFrom(LambdaCleanup);
1770
1771 LambdaExpr *Lambda = LambdaExpr::Create(Context, Class, IntroducerRange,
1772 CaptureDefault, CaptureDefaultLoc,
1773 Captures,
1774 ExplicitParams, ExplicitResultType,
1775 CaptureInits, EndLoc,
1776 ContainsUnexpandedParameterPack);
1777 // If the lambda expression's call operator is not explicitly marked constexpr
1778 // and we are not in a dependent context, analyze the call operator to infer
1779 // its constexpr-ness, suppressing diagnostics while doing so.
1780 if (getLangOpts().CPlusPlus17 && !CallOperator->isInvalidDecl() &&
1781 !CallOperator->isConstexpr() &&
1782 !isa<CoroutineBodyStmt>(CallOperator->getBody()) &&
1783 !Class->getDeclContext()->isDependentContext()) {
1784 TentativeAnalysisScope DiagnosticScopeGuard(*this);
1785 CallOperator->setConstexpr(
1786 CheckConstexprFunctionDecl(CallOperator) &&
1787 CheckConstexprFunctionBody(CallOperator, CallOperator->getBody()));
1788 }
1789
1790 // Emit delayed shadowing warnings now that the full capture list is known.
1791 DiagnoseShadowingLambdaDecls(LSI);
1792
1793 if (!CurContext->isDependentContext()) {
1794 switch (ExprEvalContexts.back().Context) {
1795 // C++11 [expr.prim.lambda]p2:
1796 // A lambda-expression shall not appear in an unevaluated operand
1797 // (Clause 5).
1798 case ExpressionEvaluationContext::Unevaluated:
1799 case ExpressionEvaluationContext::UnevaluatedList:
1800 case ExpressionEvaluationContext::UnevaluatedAbstract:
1801 // C++1y [expr.const]p2:
1802 // A conditional-expression e is a core constant expression unless the
1803 // evaluation of e, following the rules of the abstract machine, would
1804 // evaluate [...] a lambda-expression.
1805 //
1806 // This is technically incorrect, there are some constant evaluated contexts
1807 // where this should be allowed. We should probably fix this when DR1607 is
1808 // ratified, it lays out the exact set of conditions where we shouldn't
1809 // allow a lambda-expression.
1810 case ExpressionEvaluationContext::ConstantEvaluated:
1811 // We don't actually diagnose this case immediately, because we
1812 // could be within a context where we might find out later that
1813 // the expression is potentially evaluated (e.g., for typeid).
1814 ExprEvalContexts.back().Lambdas.push_back(Lambda);
1815 break;
1816
1817 case ExpressionEvaluationContext::DiscardedStatement:
1818 case ExpressionEvaluationContext::PotentiallyEvaluated:
1819 case ExpressionEvaluationContext::PotentiallyEvaluatedIfUsed:
1820 break;
1821 }
1822 }
1823
1824 return MaybeBindToTemporary(Lambda);
1825}

1827ExprResult Sema::BuildBlockForLambdaConversion(SourceLocation CurrentLocation,
1828 SourceLocation ConvLocation,
1829 CXXConversionDecl *Conv,
1830 Expr *Src) {
1831 // Make sure that the lambda call operator is marked used.
1832 CXXRecordDecl *Lambda = Conv->getParent();
1833 CXXMethodDecl *CallOperator
1834 = cast<CXXMethodDecl>(
1835 Lambda->lookup(
1836 Context.DeclarationNames.getCXXOperatorName(OO_Call)).front());
1837 CallOperator->setReferenced();
1838 CallOperator->markUsed(Context);
1839
1840 ExprResult Init = PerformCopyInitialization(
1841 InitializedEntity::InitializeLambdaToBlock(ConvLocation, Src->getType(),
1842 /*NRVO=*/false),
1843 CurrentLocation, Src);
1844 if (!Init.isInvalid())
1845 Init = ActOnFinishFullExpr(Init.get(), /*DiscardedValue*/ false);
1846
1847 if (Init.isInvalid())
1848 return ExprError();
1849
1850 // Create the new block to be returned.
1851 BlockDecl *Block = BlockDecl::Create(Context, CurContext, ConvLocation);
1852
1853 // Set the type information.
1854 Block->setSignatureAsWritten(CallOperator->getTypeSourceInfo());
1855 Block->setIsVariadic(CallOperator->isVariadic());
1856 Block->setBlockMissingReturnType(false);
1857
1858 // Add parameters.
1859 SmallVector<ParmVarDecl *, 4> BlockParams;
1860 for (unsigned I = 0, N = CallOperator->getNumParams(); I != N; ++I) {
1861 ParmVarDecl *From = CallOperator->getParamDecl(I);
1862 BlockParams.push_back(ParmVarDecl::Create(
1863 Context, Block, From->getBeginLoc(), From->getLocation(),
1864 From->getIdentifier(), From->getType(), From->getTypeSourceInfo(),
1865 From->getStorageClass(),
1866 /*DefaultArg=*/nullptr));
1867 }
1868 Block->setParams(BlockParams);
1869
1870 Block->setIsConversionFromLambda(true);
1871
1872 // Add capture. The capture uses a fake variable, which doesn't correspond
1873 // to any actual memory location. However, the initializer copy-initializes
1874 // the lambda object.
1875 TypeSourceInfo *CapVarTSI =
1876 Context.getTrivialTypeSourceInfo(Src->getType());
1877 VarDecl *CapVar = VarDecl::Create(Context, Block, ConvLocation,
1878 ConvLocation, nullptr,
1879 Src->getType(), CapVarTSI,
1880 SC_None);
1881 BlockDecl::Capture Capture(/*Variable=*/CapVar, /*ByRef=*/false,
1882 /*Nested=*/false, /*Copy=*/Init.get());
1883 Block->setCaptures(Context, Capture, /*CapturesCXXThis=*/false);
1884
1885 // Add a fake function body to the block. IR generation is responsible
1886 // for filling in the actual body, which cannot be expressed as an AST.
1887 Block->setBody(new (Context) CompoundStmt(ConvLocation));
1888
1889 // Create the block literal expression.
1890 Expr *BuildBlock = new (Context) BlockExpr(Block, Conv->getConversionType());
1891 ExprCleanupObjects.push_back(Block);
1892 Cleanup.setExprNeedsCleanups(true);
1893
1894 return BuildBlock;
1895}
