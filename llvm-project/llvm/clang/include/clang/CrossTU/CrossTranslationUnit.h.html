37enum class index_error_code {
38 unspecified = 1,
39 missing_index_file,
40 invalid_index_format,
41 multiple_definitions,
42 missing_definition,
43 failed_import,
44 failed_to_get_external_ast,
45 failed_to_generate_usr,
46 triple_mismatch,
47 lang_mismatch,
48 lang_dialect_mismatch
49};
50
51class IndexError : public llvm::ErrorInfo<IndexError> {
52public:
53 static char ID;
54 IndexError(index_error_code C) : Code(C), LineNo(0) {}
55 IndexError(index_error_code C, std::string FileName, int LineNo = 0)
56 : Code(C), FileName(std::move(FileName)), LineNo(LineNo) {}
57 IndexError(index_error_code C, std::string FileName, std::string TripleToName,
58 std::string TripleFromName)
59 : Code(C), FileName(std::move(FileName)),
60 TripleToName(std::move(TripleToName)),
61 TripleFromName(std::move(TripleFromName)) {}
62 void log(raw_ostream &OS) const override;
63 std::error_code convertToErrorCode() const override;
64 index_error_code getCode() const { return Code; }
65 int getLineNum() const { return LineNo; }
66 std::string getFileName() const { return FileName; }
67 std::string getTripleToName() const { return TripleToName; }
68 std::string getTripleFromName() const { return TripleFromName; }
69
70private:
71 index_error_code Code;
72 std::string FileName;
73 int LineNo;
74 std::string TripleToName;
75 std::string TripleFromName;
76};

79/// translation unit contains which definition.
80///
81/// The index file format is the following:

84/// \return Returns a map where the USR is the key and the filepath is the value
85/// or an error.

94/// This class is used for tools that requires cross translation
95/// unit capability.
96///
97/// This class can load definitions from external AST files.
98/// The loaded definition will be merged back to the original AST using the
99/// AST Importer.
100/// In order to use this class, an index file is required that describes
101/// the locations of the AST files for each definition.
102///
103/// Note that this class also implements caching.
104class CrossTranslationUnitContext {
105public:
106 CrossTranslationUnitContext(CompilerInstance &CI);
107 ~CrossTranslationUnitContext();
108
109 /// This function loads a function or variable definition from an
110 /// external AST file and merges it into the original AST.
111 ///
112 /// This method should only be used on functions that have no definitions or
113 /// variables that have no initializer in
114 /// the current translation unit. A function definition with the same
115 /// declaration will be looked up in the index file which should be in the
116 /// \p CrossTUDir directory, called \p IndexName. In case the declaration is
117 /// found in the index the corresponding AST file will be loaded and the
118 /// definition will be merged into the original AST using the AST Importer.
119 ///
120 /// \return The declaration with the definition will be returned.
121 /// If no suitable definition is found in the index file or multiple
122 /// definitions found error will be returned.
123 ///
124 /// Note that the AST files should also be in the \p CrossTUDir.
125 llvm::Expected<const FunctionDecl *>
126 getCrossTUDefinition(const FunctionDecl *FD, StringRef CrossTUDir,
127 StringRef IndexName, bool DisplayCTUProgress = false);
128 llvm::Expected<const VarDecl *>
129 getCrossTUDefinition(const VarDecl *VD, StringRef CrossTUDir,
130 StringRef IndexName, bool DisplayCTUProgress = false);
131
132 /// This function loads a definition from an external AST file.
133 ///
134 /// A definition with the same declaration will be looked up in the
135 /// index file which should be in the \p CrossTUDir directory, called
136 /// \p IndexName. In case the declaration is found in the index the
137 /// corresponding AST file will be loaded.
138 ///
139 /// \return Returns a pointer to the ASTUnit that contains the definition of
140 /// the looked up name or an Error.
141 /// The returned pointer is never a nullptr.
142 ///
143 /// Note that the AST files should also be in the \p CrossTUDir.
144 llvm::Expected<ASTUnit *> loadExternalAST(StringRef LookupName,
145 StringRef CrossTUDir,
146 StringRef IndexName,
147 bool DisplayCTUProgress = false);
148
149 /// This function merges a definition from a separate AST Unit into
150 /// the current one which was created by the compiler instance that
151 /// was passed to the constructor.
152 ///
153 /// \return Returns the resulting definition or an error.
154 llvm::Expected<const FunctionDecl *> importDefinition(const FunctionDecl *FD);
155 llvm::Expected<const VarDecl *> importDefinition(const VarDecl *VD);
156
157 /// Get a name to identify a named decl.
158 static std::string getLookupName(const NamedDecl *ND);
159
160 /// Emit diagnostics for the user for potential configuration errors.
161 void emitCrossTUDiagnostics(const IndexError &IE);
162
163private:
164 void lazyInitLookupTable(TranslationUnitDecl *ToTU);
165 ASTImporter &getOrCreateASTImporter(ASTContext &From);
166 template <typename T>
167 llvm::Expected<const T *> getCrossTUDefinitionImpl(const T *D,
168 StringRef CrossTUDir,
169 StringRef IndexName,
170 bool DisplayCTUProgress);
171 template <typename T>
172 const T *findDefInDeclContext(const DeclContext *DC,
173 StringRef LookupName);
174 template <typename T>
175 llvm::Expected<const T *> importDefinitionImpl(const T *D);
176
177 llvm::StringMap<std::unique_ptr<clang::ASTUnit>> FileASTUnitMap;
178 llvm::StringMap<clang::ASTUnit *> NameASTUnitMap;
179 llvm::StringMap<std::string> NameFileMap;
180 llvm::DenseMap<TranslationUnitDecl *, std::unique_ptr<ASTImporter>>
181 ASTUnitImporterMap;
182 CompilerInstance &CI;
183 ASTContext &Context;
184 std::unique_ptr<ASTImporterLookupTable> LookupTable;
185};
