274 /// Tracker for '<' tokens that might have been intended to be treated as an
275 /// angle bracket instead of a less-than comparison.
276 ///
277 /// This happens when the user intends to form a template-id, but typoes the
278 /// template-name or forgets a 'template' keyword for a dependent template
279 /// name.
280 ///
281 /// We track these locations from the point where we see a '<' with a
282 /// name-like expression on its left until we see a '>' or '>>' that might
283 /// match it.
284 struct AngleBracketTracker {
285 /// Flags used to rank candidate template names when there is more than one
286 /// '<' in a scope.
287 enum Priority : unsigned short {
288 /// A non-dependent name that is a potential typo for a template name.
289 PotentialTypo = 0x0,
290 /// A dependent name that might instantiate to a template-name.
291 DependentName = 0x2,
292
293 /// A space appears before the '<' token.
294 SpaceBeforeLess = 0x0,
295 /// No space before the '<' token
296 NoSpaceBeforeLess = 0x1,
297
298 LLVM_MARK_AS_BITMASK_ENUM(/*LargestValue*/ DependentName)
299 };

301 struct Loc {
302 Expr *TemplateName;
303 SourceLocation LessLoc;
304 AngleBracketTracker::Priority Priority;
305 unsigned short ParenCount, BracketCount, BraceCount;
306
307 bool isActive(Parser &P) const {
308 return P.ParenCount == ParenCount && P.BracketCount == BracketCount &&
309 P.BraceCount == BraceCount;
310 }
311
312 bool isActiveOrNested(Parser &P) const {
313 return isActive(P) || P.ParenCount > ParenCount ||
314 P.BracketCount > BracketCount || P.BraceCount > BraceCount;
315 }
316 };

318 SmallVector<Loc, 8> Locs;

320 /// Add an expression that might have been intended to be a template name.
321 /// In the case of ambiguity, we arbitrarily select the innermost such
322 /// expression, for example in 'foo < bar < baz', 'bar' is the current
323 /// candidate. No attempt is made to track that 'foo' is also a candidate
324 /// for the case where we see a second suspicious '>' token.
325 void add(Parser &P, Expr *TemplateName, SourceLocation LessLoc,
326 Priority Prio) {
327 if (!Locs.empty() && Locs.back().isActive(P)) {
328 if (Locs.back().Priority <= Prio) {
329 Locs.back().TemplateName = TemplateName;
330 Locs.back().LessLoc = LessLoc;
331 Locs.back().Priority = Prio;
332 }
333 } else {
334 Locs.push_back({TemplateName, LessLoc, Prio,
335 P.ParenCount, P.BracketCount, P.BraceCount});
336 }
337 }
338
339 /// Mark the current potential missing template location as having been
340 /// handled (this happens if we pass a "corresponding" '>' or '>>' token
341 /// or leave a bracket scope).
342 void clear(Parser &P) {
343 while (!Locs.empty() && Locs.back().isActiveOrNested(P))
344 Locs.pop_back();
345 }
346
347 /// Get the current enclosing expression that might hve been intended to be
348 /// a template name.
349 Loc *getCurrent(Parser &P) {
350 if (!Locs.empty() && Locs.back().isActive(P))
351 return &Locs.back();
352 return nullptr;
353 }
354 };

445 /// ConsumeToken - Consume the current 'peek token' and lex the next one.
446 /// This does not work with special tokens: string literals, code completion,
447 /// annotation tokens and balanced tokens must be handled using the specific
448 /// consume methods.
449 /// Returns the location of the consumed token.
450 SourceLocation ConsumeToken() {
451 assert(!isTokenSpecial() &&
452 "Should consume special tokens with Consume*Token");
453 PrevTokLocation = Tok.getLocation();
454 PP.Lex(Tok);
455 return PrevTokLocation;
456 }
457
458 bool TryConsumeToken(tok::TokenKind Expected) {
459 if (Tok.isNot(Expected))
460 return false;
461 assert(!isTokenSpecial() &&
462 "Should consume special tokens with Consume*Token");
463 PrevTokLocation = Tok.getLocation();
464 PP.Lex(Tok);
465 return true;
466 }
467
468 bool TryConsumeToken(tok::TokenKind Expected, SourceLocation &Loc) {
469 if (!TryConsumeToken(Expected))
470 return false;
471 Loc = PrevTokLocation;
472 return true;
473 }

475 /// ConsumeAnyToken - Dispatch to the right Consume* method based on the
476 /// current token type. This should only be used in cases where the type of
477 /// the token really isn't known, e.g. in error recovery.
478 SourceLocation ConsumeAnyToken(bool ConsumeCodeCompletionTok = false) {
479 if (isTokenParen())
480 return ConsumeParen();
481 if (isTokenBracket())
482 return ConsumeBracket();
483 if (isTokenBrace())
484 return ConsumeBrace();
485 if (isTokenStringLiteral())
486 return ConsumeStringToken();
487 if (Tok.is(tok::code_completion))
488 return ConsumeCodeCompletionTok ? ConsumeCodeCompletionToken()
489 : handleUnexpectedCodeCompletionToken();
490 if (Tok.isAnnotation())
491 return ConsumeAnnotationToken();
492 return ConsumeToken();
493 }

554 /// ConsumeParen - This consume method keeps the paren count up-to-date.
555 ///
556 SourceLocation ConsumeParen() {
557 assert(isTokenParen() && "wrong consume method");
558 if (Tok.getKind() == tok::l_paren)
559 ++ParenCount;
560 else if (ParenCount) {
561 AngleBrackets.clear(*this);
562 --ParenCount; // Don't let unbalanced )'s drive the count negative.
563 }
564 PrevTokLocation = Tok.getLocation();
565 PP.Lex(Tok);
566 return PrevTokLocation;
567 }
568
569 /// ConsumeBracket - This consume method keeps the bracket count up-to-date.
570 ///
571 SourceLocation ConsumeBracket() {
572 assert(isTokenBracket() && "wrong consume method");
573 if (Tok.getKind() == tok::l_square)
574 ++BracketCount;
575 else if (BracketCount) {
576 AngleBrackets.clear(*this);
577 --BracketCount; // Don't let unbalanced ]'s drive the count negative.
578 }
579
580 PrevTokLocation = Tok.getLocation();
581 PP.Lex(Tok);
582 return PrevTokLocation;
583 }
584
585 /// ConsumeBrace - This consume method keeps the brace count up-to-date.
586 ///
587 SourceLocation ConsumeBrace() {
588 assert(isTokenBrace() && "wrong consume method");
589 if (Tok.getKind() == tok::l_brace)
590 ++BraceCount;
591 else if (BraceCount) {
592 AngleBrackets.clear(*this);
593 --BraceCount; // Don't let unbalanced }'s drive the count negative.
594 }
595
596 PrevTokLocation = Tok.getLocation();
597 PP.Lex(Tok);
598 return PrevTokLocation;
599 }

863 /// TentativeParsingAction - An object that is used as a kind of "tentative
864 /// parsing transaction". It gets instantiated to mark the token position and
865 /// after the token consumption is done, Commit() or Revert() is called to
866 /// either "commit the consumed tokens" or revert to the previously marked
867 /// token position. Example:
868 ///
869 /// TentativeParsingAction TPA(*this);
870 /// ConsumeToken();
871 /// ....
872 /// TPA.Revert();
873 ///
874 class TentativeParsingAction {
875 Parser &P;
876 PreferredTypeBuilder PrevPreferredType;
877 Token PrevTok;
878 size_t PrevTentativelyDeclaredIdentifierCount;
879 unsigned short PrevParenCount, PrevBracketCount, PrevBraceCount;
880 bool isActive;
881
882 public:
883 explicit TentativeParsingAction(Parser& p) : P(p) {
884 PrevPreferredType = P.PreferredType;
885 PrevTok = P.Tok;
886 PrevTentativelyDeclaredIdentifierCount =
887 P.TentativelyDeclaredIdentifiers.size();
888 PrevParenCount = P.ParenCount;
889 PrevBracketCount = P.BracketCount;
890 PrevBraceCount = P.BraceCount;
891 P.PP.EnableBacktrackAtThisPos();
892 isActive = true;
893 }
894 void Commit() {
895 assert(isActive && "Parsing action was finished!");
896 P.TentativelyDeclaredIdentifiers.resize(
897 PrevTentativelyDeclaredIdentifierCount);
898 P.PP.CommitBacktrackedTokens();
899 isActive = false;
900 }
901 void Revert() {
902 assert(isActive && "Parsing action was finished!");
903 P.PP.Backtrack();
904 P.PreferredType = PrevPreferredType;
905 P.Tok = PrevTok;
906 P.TentativelyDeclaredIdentifiers.resize(
907 PrevTentativelyDeclaredIdentifierCount);
908 P.ParenCount = PrevParenCount;
909 P.BracketCount = PrevBracketCount;
910 P.BraceCount = PrevBraceCount;
911 isActive = false;
912 }
913 ~TentativeParsingAction() {
914 assert(!isActive && "Forgot to call Commit or Revert!");
915 }
916 };
