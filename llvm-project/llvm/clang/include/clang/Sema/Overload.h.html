221 /// StandardConversionSequence - represents a standard conversion
222 /// sequence (C++ 13.3.3.1.1). A standard conversion sequence
223 /// contains between zero and three conversions. If a particular
224 /// conversion is not needed, it will be set to the identity conversion
225 /// (ICK_Identity). Note that the three conversions are
226 /// specified as separate members (rather than in an array) so that
227 /// we can keep the size of a standard conversion sequence to a
228 /// single word.
229 class StandardConversionSequence {
230 public:
231 /// First -- The first conversion can be an lvalue-to-rvalue
232 /// conversion, array-to-pointer conversion, or
233 /// function-to-pointer conversion.
234 ImplicitConversionKind First : 8;
235
236 /// Second - The second conversion can be an integral promotion,
237 /// floating point promotion, integral conversion, floating point
238 /// conversion, floating-integral conversion, pointer conversion,
239 /// pointer-to-member conversion, or boolean conversion.
240 ImplicitConversionKind Second : 8;
241
242 /// Third - The third conversion can be a qualification conversion
243 /// or a function conversion.
244 ImplicitConversionKind Third : 8;
245
246 /// Whether this is the deprecated conversion of a
247 /// string literal to a pointer to non-const character data
248 /// (C++ 4.2p2).
249 unsigned DeprecatedStringLiteralToCharPtr : 1;
250
251 /// Whether the qualification conversion involves a change in the
252 /// Objective-C lifetime (for automatic reference counting).
253 unsigned QualificationIncludesObjCLifetime : 1;
254
255 /// IncompatibleObjC - Whether this is an Objective-C conversion
256 /// that we should warn about (if we actually use it).
257 unsigned IncompatibleObjC : 1;
258
259 /// ReferenceBinding - True when this is a reference binding
260 /// (C++ [over.ics.ref]).
261 unsigned ReferenceBinding : 1;
262
263 /// DirectBinding - True when this is a reference binding that is a
264 /// direct binding (C++ [dcl.init.ref]).
265 unsigned DirectBinding : 1;
266
267 /// Whether this is an lvalue reference binding (otherwise, it's
268 /// an rvalue reference binding).
269 unsigned IsLvalueReference : 1;
270
271 /// Whether we're binding to a function lvalue.
272 unsigned BindsToFunctionLvalue : 1;
273
274 /// Whether we're binding to an rvalue.
275 unsigned BindsToRvalue : 1;
276
277 /// Whether this binds an implicit object argument to a
278 /// non-static member function without a ref-qualifier.
279 unsigned BindsImplicitObjectArgumentWithoutRefQualifier : 1;
280
281 /// Whether this binds a reference to an object with a different
282 /// Objective-C lifetime qualifier.
283 unsigned ObjCLifetimeConversionBinding : 1;
284
285 /// \brief e.g. conversions from pointer -> capability without an explicit __cheri_tocap
286 unsigned IncompatibleCHERIConversion : 1; // XXXAR: would be nice if we had a ctor to initialize this
287
288 bool isInvalidCHERICapabilityConversion() const {
289 return IncompatibleCHERIConversion;
290 }
291 void setInvalidCHERIConversion(bool IsInvalid = true) {
292 assert(!IncompatibleCHERIConversion); // should have been initialized to false (see above)
293 IncompatibleCHERIConversion = IsInvalid;
294 }
295
296 /// FromType - The type that this conversion is converting
297 /// from. This is an opaque pointer that can be translated into a
298 /// QualType.
299 void *FromTypePtr;
300
301 /// ToType - The types that this conversion is converting to in
302 /// each step. This is an opaque pointer that can be translated
303 /// into a QualType.
304 void *ToTypePtrs[3];
305
306 /// CopyConstructor - The copy constructor that is used to perform
307 /// this conversion, when the conversion is actually just the
308 /// initialization of an object via copy constructor. Such
309 /// conversions are either identity conversions or derived-to-base
310 /// conversions.
311 CXXConstructorDecl *CopyConstructor;
312 DeclAccessPair FoundCopyConstructor;
313
314 void setFromType(QualType T) { FromTypePtr = T.getAsOpaquePtr(); }
315
316 void setToType(unsigned Idx, QualType T) {
317 assert(Idx < 3 && "To type index is out of range");
318 ToTypePtrs[Idx] = T.getAsOpaquePtr();
319 }
320
321 void setAllToTypes(QualType T) {
322 ToTypePtrs[0] = T.getAsOpaquePtr();
323 ToTypePtrs[1] = ToTypePtrs[0];
324 ToTypePtrs[2] = ToTypePtrs[0];
325 }
326
327 QualType getFromType() const {
328 return QualType::getFromOpaquePtr(FromTypePtr);
329 }
330
331 QualType getToType(unsigned Idx) const {
332 assert(Idx < 3 && "To type index is out of range");
333 return QualType::getFromOpaquePtr(ToTypePtrs[Idx]);
334 }
335
336 void setAsIdentityConversion();
337
338 bool isIdentityConversion() const {
339 return Second == ICK_Identity && Third == ICK_Identity;
340 }
341
342 ImplicitConversionRank getRank() const;
343 NarrowingKind
344 getNarrowingKind(ASTContext &Context, const Expr *Converted,
345 APValue &ConstantValue, QualType &ConstantType,
346 bool IgnoreFloatToIntegralConversion = false) const;
347 bool isPointerConversionToBool() const;
348 bool isPointerConversionToVoidPointer(ASTContext& Context) const;
349 void dump() const;
350 };

352 /// UserDefinedConversionSequence - Represents a user-defined
353 /// conversion sequence (C++ 13.3.3.1.2).
354 struct UserDefinedConversionSequence {
355 /// Represents the standard conversion that occurs before
356 /// the actual user-defined conversion.
357 ///
358 /// C++11 13.3.3.1.2p1:
359 /// If the user-defined conversion is specified by a constructor
360 /// (12.3.1), the initial standard conversion sequence converts
361 /// the source type to the type required by the argument of the
362 /// constructor. If the user-defined conversion is specified by
363 /// a conversion function (12.3.2), the initial standard
364 /// conversion sequence converts the source type to the implicit
365 /// object parameter of the conversion function.
366 StandardConversionSequence Before;
367
368 /// EllipsisConversion - When this is true, it means user-defined
369 /// conversion sequence starts with a ... (ellipsis) conversion, instead of
370 /// a standard conversion. In this case, 'Before' field must be ignored.
371 // FIXME. I much rather put this as the first field. But there seems to be
372 // a gcc code gen. bug which causes a crash in a test. Putting it here seems
373 // to work around the crash.
374 bool EllipsisConversion : 1;
375
376 /// HadMultipleCandidates - When this is true, it means that the
377 /// conversion function was resolved from an overloaded set having
378 /// size greater than 1.
379 bool HadMultipleCandidates : 1;
380
381 /// After - Represents the standard conversion that occurs after
382 /// the actual user-defined conversion.
383 StandardConversionSequence After;
384
385 /// ConversionFunction - The function that will perform the
386 /// user-defined conversion. Null if the conversion is an
387 /// aggregate initialization from an initializer list.
388 FunctionDecl* ConversionFunction;
389
390 /// The declaration that we found via name lookup, which might be
391 /// the same as \c ConversionFunction or it might be a using declaration
392 /// that refers to \c ConversionFunction.
393 DeclAccessPair FoundConversionFunction;
394
395 void dump() const;
396 };

493 /// ImplicitConversionSequence - Represents an implicit conversion
494 /// sequence, which may be a standard conversion sequence
495 /// (C++ 13.3.3.1.1), user-defined conversion sequence (C++ 13.3.3.1.2),
496 /// or an ellipsis conversion sequence (C++ 13.3.3.1.3).
497 class ImplicitConversionSequence {
498 public:
499 /// Kind - The kind of implicit conversion sequence. BadConversion
500 /// specifies that there is no conversion from the source type to
501 /// the target type. AmbiguousConversion represents the unique
502 /// ambiguous conversion (C++0x [over.best.ics]p10).
503 enum Kind {
504 StandardConversion = 0,
505 UserDefinedConversion,
506 AmbiguousConversion,
507 EllipsisConversion,
508 BadConversion
509 };
510
511 enum SetKindAction {
512 MemsetToZero,
513 KeepState,
514 };
515
516 private:
517 enum {
518 Uninitialized = BadConversion + 1
519 };
520
521 /// ConversionKind - The kind of implicit conversion sequence.
522 unsigned ConversionKind : 30;
523
524 /// Whether the target is really a std::initializer_list, and the
525 /// sequence only represents the worst element conversion.
526 unsigned StdInitializerListElement : 1;
527
528 void setKind(Kind K) {
529 destruct();
530 ConversionKind = K;
531 }
532
533 void destruct() {
534 if (ConversionKind == AmbiguousConversion) Ambiguous.destruct();
535 }
536
537 public:
538 union {
539 /// When ConversionKind == StandardConversion, provides the
540 /// details of the standard conversion sequence.
541 StandardConversionSequence Standard;
542
543 /// When ConversionKind == UserDefinedConversion, provides the
544 /// details of the user-defined conversion sequence.
545 UserDefinedConversionSequence UserDefined;
546
547 /// When ConversionKind == AmbiguousConversion, provides the
548 /// details of the ambiguous conversion.
549 AmbiguousConversionSequence Ambiguous;
550
551 /// When ConversionKind == BadConversion, provides the details
552 /// of the bad conversion.
553 BadConversionSequence Bad;
554 };
555
556 ImplicitConversionSequence()
557 : ConversionKind(Uninitialized), StdInitializerListElement(false) {
558 Standard.setAsIdentityConversion();
559 }
560
561 ImplicitConversionSequence(const ImplicitConversionSequence &Other)
562 : ConversionKind(Other.ConversionKind),
563 StdInitializerListElement(Other.StdInitializerListElement) {
564 switch (ConversionKind) {
565 case Uninitialized: break;
566 case StandardConversion: Standard = Other.Standard; break;
567 case UserDefinedConversion: UserDefined = Other.UserDefined; break;
568 case AmbiguousConversion: Ambiguous.copyFrom(Other.Ambiguous); break;
569 case EllipsisConversion: break;
570 case BadConversion: Bad = Other.Bad; break;
571 }
572 }
573
574 ImplicitConversionSequence &
575 operator=(const ImplicitConversionSequence &Other) {
576 destruct();
577 new (this) ImplicitConversionSequence(Other);
578 return *this;
579 }
580
581 ~ImplicitConversionSequence() {
582 destruct();
583 }
584
585 Kind getKind() const {
586 assert(isInitialized() && "querying uninitialized conversion");
587 return Kind(ConversionKind);
588 }
589
590 /// Return a ranking of the implicit conversion sequence
591 /// kind, where smaller ranks represent better conversion
592 /// sequences.
593 ///
594 /// In particular, this routine gives user-defined conversion
595 /// sequences and ambiguous conversion sequences the same rank,
596 /// per C++ [over.best.ics]p10.
597 unsigned getKindRank() const {
598 switch (getKind()) {
599 case StandardConversion:
600 return 0;
601
602 case UserDefinedConversion:
603 case AmbiguousConversion:
604 return 1;
605
606 case EllipsisConversion:
607 return 2;
608
609 case BadConversion:
610 return 3;
611 }
612
613 llvm_unreachable("Invalid ImplicitConversionSequence::Kind!");
614 }
615
616 bool isBad() const { return getKind() == BadConversion; }
617 bool isStandard() const { return getKind() == StandardConversion; }
618 bool isEllipsis() const { return getKind() == EllipsisConversion; }
619 bool isAmbiguous() const { return getKind() == AmbiguousConversion; }
620 bool isUserDefined() const { return getKind() == UserDefinedConversion; }
621 bool isFailure() const { return isBad() || isAmbiguous(); }
622
623 /// Determines whether this conversion sequence has been
624 /// initialized. Most operations should never need to query
625 /// uninitialized conversions and should assert as above.
626 bool isInitialized() const { return ConversionKind != Uninitialized; }
627
628 /// Sets this sequence as a bad conversion for an explicit argument.
629 void setBad(BadConversionSequence::FailureKind Failure,
630 Expr *FromExpr, QualType ToType) {
631 setKind(BadConversion);
632 Bad.init(Failure, FromExpr, ToType);
633 }
634
635 /// Sets this sequence as a bad conversion for an implicit argument.
636 void setBad(BadConversionSequence::FailureKind Failure,
637 QualType FromType, QualType ToType) {
638 setKind(BadConversion);
639 Bad.init(Failure, FromType, ToType);
640 }
641
642 void setStandard(SetKindAction Action) {
643 setKind(StandardConversion);
644 if (Action == MemsetToZero)
645 memset(&Standard, 0, sizeof(Standard));
646 }
647 void setStandard(const StandardConversionSequence& NewSeq) {
648 setKind(StandardConversion);
649 Standard = NewSeq;
650 }
651 void setEllipsis() { setKind(EllipsisConversion); }
652 void setUserDefined() { setKind(UserDefinedConversion); }
653
654 void setAmbiguous() {
655 if (ConversionKind == AmbiguousConversion) return;
656 ConversionKind = AmbiguousConversion;
657 Ambiguous.construct();
658 }
659
660 void setAsIdentityConversion(QualType T) {
661 setStandard(MemsetToZero); // XXXAR: not sure this is correct
662 Standard.setAsIdentityConversion();
663 Standard.setFromType(T);
664 Standard.setAllToTypes(T);
665 }

693 enum OverloadFailureKind {
694 ovl_fail_too_many_arguments,
695 ovl_fail_too_few_arguments,
696 ovl_fail_bad_conversion,
697 ovl_fail_bad_deduction,
698
699 /// This conversion candidate was not considered because it
700 /// duplicates the work of a trivial or derived-to-base
701 /// conversion.
702 ovl_fail_trivial_conversion,
703
704 /// This conversion candidate was not considered because it is
705 /// an illegal instantiation of a constructor temploid: it is
706 /// callable with one argument, we only have one argument, and
707 /// its first parameter type is exactly the type of the class.
708 ///
709 /// Defining such a constructor directly is illegal, and
710 /// template-argument deduction is supposed to ignore such
711 /// instantiations, but we can still get one with the right
712 /// kind of implicit instantiation.
713 ovl_fail_illegal_constructor,
714
715 /// This conversion candidate is not viable because its result
716 /// type is not implicitly convertible to the desired type.
717 ovl_fail_bad_final_conversion,
718
719 /// This conversion function template specialization candidate is not
720 /// viable because the final conversion was not an exact match.
721 ovl_fail_final_conversion_not_exact,
722
723 /// (CUDA) This candidate was not viable because the callee
724 /// was not accessible from the caller's target (i.e. host->device,
725 /// global->host, device->host).
726 ovl_fail_bad_target,
727
728 /// This candidate function was not viable because an enable_if
729 /// attribute disabled it.
730 ovl_fail_enable_if,
731
732 /// This candidate constructor or conversion fonction
733 /// is used implicitly but the explicit(bool) specifier
734 /// was resolved to true
735 ovl_fail_explicit_resolved,
736
737 /// This candidate was not viable because its address could not be taken.
738 ovl_fail_addr_not_available,
739
740 /// This candidate was not viable because its OpenCL extension is disabled.
741 ovl_fail_ext_disabled,
742
743 /// This inherited constructor is not viable because it would slice the
744 /// argument.
745 ovl_fail_inhctor_slice,
746
747 /// This candidate was not viable because it is a non-default multiversioned
748 /// function.
749 ovl_non_default_multiversion_function,
750 };

757 /// OverloadCandidate - A single candidate in an overload set (C++ 13.3).
758 struct OverloadCandidate {
759 /// Function - The actual function that this candidate
760 /// represents. When NULL, this is a built-in candidate
761 /// (C++ [over.oper]) or a surrogate for a conversion to a
762 /// function pointer or reference (C++ [over.call.object]).
763 FunctionDecl *Function;
764
765 /// FoundDecl - The original declaration that was looked up /
766 /// invented / otherwise found, together with its access.
767 /// Might be a UsingShadowDecl or a FunctionTemplateDecl.

834 bool CanFix = Fix.tryToFixConversion(
835 Conversions[Idx].Bad.FromExpr,
836 Conversions[Idx].Bad.getFromType(),
837 Conversions[Idx].Bad.getToType(), S);

893 // Allocator for ConversionSequenceLists. We store the first few of these
894 // inline to avoid allocation for small sets.
895 llvm::BumpPtrAllocator SlabAllocator;
896
897 SourceLocation Loc;
898 CandidateSetKind Kind;
899
900 constexpr static unsigned NumInlineBytes =
901 24 * sizeof(ImplicitConversionSequence);
902 unsigned NumInlineBytesUsed = 0;
903 llvm::AlignedCharArray<alignof(void *), NumInlineBytes> InlineSpace;
904
905 /// If we have space, allocates from inline storage. Otherwise, allocates
906 /// from the slab allocator.
907 /// FIXME: It would probably be nice to have a SmallBumpPtrAllocator
908 /// instead.
909 /// FIXME: Now that this only allocates ImplicitConversionSequences, do we
910 /// want to un-generalize this?
911 template <typename T>
912 T *slabAllocate(unsigned N) {
913 // It's simpler if this doesn't need to consider alignment.
914 static_assert(alignof(T) == alignof(void *),
915 "Only works for pointer-aligned types.");
916 static_assert(std::is_trivial<T>::value ||
917 std::is_same<ImplicitConversionSequence, T>::value,
918 "Add destruction logic to OverloadCandidateSet::clear().");
919
920 unsigned NBytes = sizeof(T) * N;
921 if (NBytes > NumInlineBytes - NumInlineBytesUsed)
922 return SlabAllocator.Allocate<T>(N);
923 char *FreeSpaceStart = InlineSpace.buffer + NumInlineBytesUsed;
924 assert(uintptr_t(FreeSpaceStart) % alignof(void *) == 0 &&
925 "Misaligned storage!");
926
927 NumInlineBytesUsed += NBytes;
928 return reinterpret_cast<T *>(FreeSpaceStart);
929 }
