35// Note that \p TextGenerator is allowed to fail, e.g. when trying to access a
36// matched node that was not bound. Allowing this to fail simplifies error
37// handling for interactive tools like clang-query.
38using TextGenerator = std::function<Expected<std::string>(
39 const ast_matchers::MatchFinder::MatchResult &)>;
40
41/// Wraps a string as a TextGenerator.
42inline TextGenerator text(std::string M) {
43 return [M](const ast_matchers::MatchFinder::MatchResult &)
44 -> Expected<std::string> { return M; };
45}

47// Description of a source-code edit, expressed in terms of an AST node.
48// Includes: an ID for the (bound) node, a selector for source related to the
49// node, a replacement and, optionally, an explanation for the edit.
50//
51// * Target: the source code impacted by the rule. This identifies an AST node,
52// or part thereof (\c Part), whose source range indicates the extent of the
53// replacement applied by the replacement term. By default, the extent is the
54// node matched by the pattern term (\c NodePart::Node). Target's are typed
55// (\c Kind), which guides the determination of the node extent.
56//
57// * Replacement: a function that produces a replacement string for the target,
58// based on the match result.
59//
60// * Note: (optional) a note specifically for this edit, potentially referencing
61// elements of the match. This will be displayed to the user, where possible;
62// for example, in clang-tidy diagnostics. Use of notes should be rare --
63// explanations of the entire rewrite should be set in the rule
64// (`RewriteRule::Explanation`) instead. Notes serve the rare cases wherein
65// edit-specific diagnostics are required.
66//
67// `ASTEdit` should be built using the `change` convenience functions. For
68// example,
69// \code
70// change(name(fun), text("Frodo"))
71// \endcode
72// Or, if we use Stencil for the TextGenerator:
73// \code
74// using stencil::cat;
75// change(statement(thenNode), cat("{", thenNode, "}"))
76// change(callArgs(call), cat(x, ",", y))
77// \endcode
78// Or, if you are changing the node corresponding to the rule's matcher, you can
79// use the single-argument override of \c change:
80// \code
81// change(cat("different_expr"))
82// \endcode
83struct ASTEdit {
84 RangeSelector TargetRange;
85 TextGenerator Replacement;
86 TextGenerator Note;
87};

89/// Description of a source-code transformation.
90//
91// A *rewrite rule* describes a transformation of source code. A simple rule
92// contains each of the following components:
93//
94// * Matcher: the pattern term, expressed as clang matchers (with Transformer
95// extensions).
96//
97// * Edits: a set of Edits to the source code, described with ASTEdits.
98//
99// * Explanation: explanation of the rewrite. This will be displayed to the
100// user, where possible; for example, in clang-tidy diagnostics.
101//
102// However, rules can also consist of (sub)rules, where the first that matches
103// is applied and the rest are ignored. So, the above components are gathered
104// as a `Case` and a rule is a list of cases.
105//
106// Rule cases have an additional, implicit, component: the parameters. These are
107// portions of the pattern which are left unspecified, yet bound in the pattern
108// so that we can reference them in the edits.
109//
110// The \c Transformer class can be used to apply the rewrite rule and obtain the
111// corresponding replacements.
112struct RewriteRule {
113 struct Case {
114 ast_matchers::internal::DynTypedMatcher Matcher;
115 SmallVector<ASTEdit, 1> Edits;
116 TextGenerator Explanation;
117 };
118 // We expect RewriteRules will most commonly include only one case.
119 SmallVector<Case, 1> Cases;
120
121 // ID used as the default target of each match. The node described by the
122 // matcher is should always be bound to this id.
123 static constexpr llvm::StringLiteral RootID = "___root___";
124};
125
126/// Convenience function for constructing a simple \c RewriteRule.
127RewriteRule makeRule(ast_matchers::internal::DynTypedMatcher M,
128 SmallVector<ASTEdit, 1> Edits,
129 TextGenerator Explanation = nullptr);
130
131/// Convenience overload of \c makeRule for common case of only one edit.
132inline RewriteRule makeRule(ast_matchers::internal::DynTypedMatcher M,
133 ASTEdit Edit,
134 TextGenerator Explanation = nullptr) {
135 SmallVector<ASTEdit, 1> Edits;
136 Edits.emplace_back(std::move(Edit));
137 return makeRule(std::move(M), std::move(Edits), std::move(Explanation));
138}

140/// Applies the first rule whose pattern matches; other rules are ignored.
141///
142/// N.B. All of the rules must use the same kind of matcher (that is, share a
143/// base class in the AST hierarchy). However, this constraint is caused by an
144/// implementation detail and should be lifted in the future.
145//
146// `applyFirst` is like an `anyOf` matcher with an edit action attached to each
147// of its cases. Anywhere you'd use `anyOf(m1.bind("id1"), m2.bind("id2"))` and
148// then dispatch on those ids in your code for control flow, `applyFirst` lifts
149// that behavior to the rule level. So, you can write `applyFirst({makeRule(m1,
150// action1), makeRule(m2, action2), ...});`
151//
152// For example, consider a type `T` with a deterministic serialization function,
153// `serialize()`. For performance reasons, we would like to make it
154// non-deterministic. Therefore, we want to drop the expectation that
155// `a.serialize() = b.serialize() iff a = b` (although we'll maintain
156// `deserialize(a.serialize()) = a`).
157//
158// We have three cases to consider (for some equality function, `eq`):
159// ```
160// eq(a.serialize(), b.serialize()) --> eq(a,b)
161// eq(a, b.serialize()) --> eq(deserialize(a), b)
162// eq(a.serialize(), b) --> eq(a, deserialize(b))
163// ```
164//
165// `applyFirst` allows us to specify each independently:
166// ```
167// auto eq_fun = functionDecl(...);
168// auto method_call = cxxMemberCallExpr(...);
169//
170// auto two_calls = callExpr(callee(eq_fun), hasArgument(0, method_call),
171// hasArgument(1, method_call));
172// auto left_call =
173// callExpr(callee(eq_fun), callExpr(hasArgument(0, method_call)));
174// auto right_call =
175// callExpr(callee(eq_fun), callExpr(hasArgument(1, method_call)));
176//
177// RewriteRule R = applyFirst({makeRule(two_calls, two_calls_action),
178// makeRule(left_call, left_call_action),
179// makeRule(right_call, right_call_action)});
180// ```
181RewriteRule applyFirst(ArrayRef<RewriteRule> Rules);
182
183/// Replaces a portion of the source text with \p Replacement.
184ASTEdit change(RangeSelector Target, TextGenerator Replacement);
185
186/// Replaces the entirety of a RewriteRule's match with \p Replacement. For
187/// example, to replace a function call, one could write:
188/// \code
189/// makeRule(callExpr(callee(functionDecl(hasName("foo")))),
190/// change(text("bar()")))
191/// \endcode
192inline ASTEdit change(TextGenerator Replacement) {
193 return change(node(RewriteRule::RootID), std::move(Replacement));
194}
195
196/// Inserts \p Replacement before \p S, leaving the source selected by \S
197/// unchanged.
198inline ASTEdit insertBefore(RangeSelector S, TextGenerator Replacement) {
199 return change(before(std::move(S)), std::move(Replacement));
200}
201
202/// Inserts \p Replacement after \p S, leaving the source selected by \S
203/// unchanged.
204inline ASTEdit insertAfter(RangeSelector S, TextGenerator Replacement) {
205 return change(after(std::move(S)), std::move(Replacement));
206}
207
208/// Removes the source selected by \p S.
209inline ASTEdit remove(RangeSelector S) {
210 return change(std::move(S), text(""));
211}

213/// The following three functions are a low-level part of the RewriteRule
214/// API. We expose them for use in implementing the fixtures that interpret

231/// A source "transformation," represented by a character range in the source to
232/// be replaced and a corresponding replacement string.
233struct Transformation {
234 CharSourceRange Range;
235 std::string Replacement;
236};
237
238/// Attempts to translate `Edits`, which are in terms of AST nodes bound in the
239/// match `Result`, into Transformations, which are in terms of the source code
240/// text.
241///
242/// Returns an empty vector if any of the edits apply to portions of the source
243/// that are ineligible for rewriting (certain interactions with macros, for
244/// example). Fails if any invariants are violated relating to bound nodes in
245/// the match. However, it does not fail in the case of conflicting edits --
246/// conflict handling is left to clients. We recommend use of the \c
247/// AtomicChange or \c Replacements classes for assistance in detecting such
248/// conflicts.
249Expected<SmallVector<Transformation, 1>>
250translateEdits(const ast_matchers::MatchFinder::MatchResult &Result,
251 llvm::ArrayRef<ASTEdit> Edits);
252} // namespace detail

254/// Handles the matcher and callback registration for a single rewrite rule, as
255/// defined by the arguments of the constructor.
256class Transformer : public ast_matchers::MatchFinder::MatchCallback {
257public:
258 using ChangeConsumer =
259 std::function<void(Expected<clang::tooling::AtomicChange> Change)>;
260
261 /// \param Consumer Receives each rewrite or error. Will not necessarily be
262 /// called for each match; for example, if the rewrite is not applicable
263 /// because of macros, but doesn't fail. Note that clients are responsible
264 /// for handling the case that independent \c AtomicChanges conflict with each
265 /// other.
266 Transformer(RewriteRule Rule, ChangeConsumer Consumer)
267 : Rule(std::move(Rule)), Consumer(std::move(Consumer)) {}
268
269 /// N.B. Passes `this` pointer to `MatchFinder`. So, this object should not
270 /// be moved after this call.
271 void registerMatchers(ast_matchers::MatchFinder *MatchFinder);
272
273 /// Not called directly by users -- called by the framework, via base class
274 /// pointer.
275 void run(const ast_matchers::MatchFinder::MatchResult &Result) override;
276
277private:
278 RewriteRule Rule;
279 /// Receives each successful rewrites as an \c AtomicChange.
280 ChangeConsumer Consumer;
281};
