8// Record tokens that a preprocessor emits and define operations to map between
9// the tokens written in a file and tokens produced by the preprocessor.
10//
11// When running the compiler, there are two token streams we are interested in:
12// - "spelled" tokens directly correspond to a substring written in some
13// source file.
14// - "expanded" tokens represent the result of preprocessing, parses consumes
15// this token stream to produce the AST.
16//
17// Expanded tokens correspond directly to locations found in the AST, allowing
18// to find subranges of the token stream covered by various AST nodes. Spelled
19// tokens correspond directly to the source code written by the user.
20//
21// To allow composing these two use-cases, we also define operations that map
22// between expanded and spelled tokens that produced them (macro calls,
23// directives, etc).
24//
25//===----------------------------------------------------------------------===//

49/// A half-open character range inside a particular file, the start offset is
50/// included and the end offset is excluded from the range.
51struct FileRange {
52 /// EXPECTS: File.isValid() && Begin <= End.
53 FileRange(FileID File, unsigned BeginOffset, unsigned EndOffset);
54 /// EXPECTS: BeginLoc.isValid() && BeginLoc.isFileID().
55 FileRange(const SourceManager &SM, SourceLocation BeginLoc, unsigned Length);
56 /// EXPECTS: BeginLoc.isValid() && BeginLoc.isFileID(), Begin <= End and files
57 /// are the same.
58 FileRange(const SourceManager &SM, SourceLocation BeginLoc,
59 SourceLocation EndLoc);
60
61 FileID file() const { return File; }
62 /// Start is a start offset (inclusive) in the corresponding file.
63 unsigned beginOffset() const { return Begin; }
64 /// End offset (exclusive) in the corresponding file.
65 unsigned endOffset() const { return End; }
66
67 unsigned length() const { return End - Begin; }
68
69 /// Gets the substring that this FileRange refers to.
70 llvm::StringRef text(const SourceManager &SM) const;
71
72 friend bool operator==(const FileRange &L, const FileRange &R) {
73 return std::tie(L.File, L.Begin, L.End) == std::tie(R.File, R.Begin, R.End);
74 }
75 friend bool operator!=(const FileRange &L, const FileRange &R) {
76 return !(L == R);
77 }
78
79private:
80 FileID File;
81 unsigned Begin;
82 unsigned End;
83};

85/// For debugging purposes.
86llvm::raw_ostream &operator<<(llvm::raw_ostream &OS, const FileRange &R);

88/// A token coming directly from a file or from a macro invocation. Has just
89/// enough information to locate the token in the source code.
90/// Can represent both expanded and spelled tokens.
91class Token {
92public:
93 Token(SourceLocation Location, unsigned Length, tok::TokenKind Kind)
94 : Location(Location), Length(Length), Kind(Kind) {}
95 /// EXPECTS: clang::Token is not an annotation token.
96 explicit Token(const clang::Token &T);
97
98 tok::TokenKind kind() const { return Kind; }
99 /// Location of the first character of a token.
100 SourceLocation location() const { return Location; }
101 /// Location right after the last character of a token.
102 SourceLocation endLocation() const {
103 return Location.getLocWithOffset(Length);
104 }
105 unsigned length() const { return Length; }
106
107 /// Get the substring covered by the token. Note that will include all
108 /// digraphs, newline continuations, etc. E.g. tokens for 'int' and
109 /// in\
110 /// t
111 /// both have the same kind tok::kw_int, but results of text() are different.
112 llvm::StringRef text(const SourceManager &SM) const;
113
114 /// Gets a range of this token.
115 /// EXPECTS: token comes from a file, not from a macro expansion.
116 FileRange range(const SourceManager &SM) const;
117
118 /// Given two tokens inside the same file, returns a file range that starts at
119 /// \p First and ends at \p Last.
120 /// EXPECTS: First and Last are file tokens from the same file, Last starts
121 /// after First.
122 static FileRange range(const SourceManager &SM, const syntax::Token &First,
123 const syntax::Token &Last);
124
125 std::string dumpForTests(const SourceManager &SM) const;
126 /// For debugging purposes.
127 std::string str() const;
128
129private:
130 SourceLocation Location;
131 unsigned Length;
132 tok::TokenKind Kind;
133};
134/// For debugging purposes. Equivalent to a call to Token::str().
135llvm::raw_ostream &operator<<(llvm::raw_ostream &OS, const Token &T);

137/// A list of tokens obtained by preprocessing a text buffer and operations to
138/// map between the expanded and spelled tokens, i.e. TokenBuffer has
139/// information about two token streams:
140/// 1. Expanded tokens: tokens produced by the preprocessor after all macro
141/// replacements,
142/// 2. Spelled tokens: corresponding directly to the source code of a file
143/// before any macro replacements occurred.
144/// Here's an example to illustrate a difference between those two:
145/// #define FOO 10
146/// int a = FOO;
147///
148/// Spelled tokens are {'#','define','FOO','10','int','a','=','FOO',';'}.
149/// Expanded tokens are {'int','a','=','10',';','eof'}.
150///
151/// Note that the expanded token stream has a tok::eof token at the end, the
152/// spelled tokens never store a 'eof' token.
153///
154/// The full list expanded tokens can be obtained with expandedTokens(). Spelled
155/// tokens for each of the files can be obtained via spelledTokens(FileID).
156///
157/// To map between the expanded and spelled tokens use findSpelledByExpanded().
158///
159/// To build a token buffer use the TokenCollector class. You can also compute
160/// the spelled tokens of a file using the tokenize() helper.
161///
162/// FIXME: allow to map from spelled to expanded tokens when use-case shows up.
163class TokenBuffer {
164public:
165 TokenBuffer(const SourceManager &SourceMgr) : SourceMgr(&SourceMgr) {}
166 /// All tokens produced by the preprocessor after all macro replacements,
167 /// directives, etc. Source locations found in the clang AST will always
168 /// point to one of these tokens.
169 /// FIXME: figure out how to handle token splitting, e.g. '>>' can be split
170 /// into two '>' tokens by the parser. However, TokenBuffer currently
171 /// keeps it as a single '>>' token.
172 llvm::ArrayRef<syntax::Token> expandedTokens() const {
173 return ExpandedTokens;
174 }
175
176 /// Find the subrange of spelled tokens that produced the corresponding \p
177 /// Expanded tokens.
178 ///
179 /// EXPECTS: \p Expanded is a subrange of expandedTokens().
180 ///
181 /// Will fail if the expanded tokens do not correspond to a
182 /// sequence of spelled tokens. E.g. for the following example:
183 ///
184 /// #define FIRST f1 f2 f3
185 /// #define SECOND s1 s2 s3
186 ///
187 /// a FIRST b SECOND c // expanded tokens are: a f1 f2 f3 b s1 s2 s3 c
188 ///
189 /// the results would be:
190 /// expanded => spelled
191 /// ------------------------
192 /// a => a
193 /// s1 s2 s3 => SECOND
194 /// a f1 f2 f3 => a FIRST
195 /// a f1 => can't map
196 /// s1 s2 => can't map
197 ///
198 /// If \p Expanded is empty, the returned value is llvm::None.
199 /// Complexity is logarithmic.
200 llvm::Optional<llvm::ArrayRef<syntax::Token>>
201 spelledForExpanded(llvm::ArrayRef<syntax::Token> Expanded) const;
202
203 /// Lexed tokens of a file before preprocessing. E.g. for the following input
204 /// #define DECL(name) int name = 10
205 /// DECL(a);
206 /// spelledTokens() returns {"#", "define", "DECL", "(", "name", ")", "eof"}.
207 /// FIXME: we do not yet store tokens of directives, like #include, #define,
208 /// #pragma, etc.
209 llvm::ArrayRef<syntax::Token> spelledTokens(FileID FID) const;
210
211 std::string dumpForTests() const;
212
213private:
214 /// Describes a mapping between a continuous subrange of spelled tokens and
215 /// expanded tokens. Represents macro expansions, preprocessor directives,
216 /// conditionally disabled pp regions, etc.
217 /// #define FOO 1+2
218 /// #define BAR(a) a + 1
219 /// FOO // invocation #1, tokens = {'1','+','2'}, macroTokens = {'FOO'}.
220 /// BAR(1) // invocation #2, tokens = {'a', '+', '1'},
221 /// macroTokens = {'BAR', '(', '1', ')'}.
222 struct Mapping {
223 // Positions in the corresponding spelled token stream. The corresponding
224 // range is never empty.
225 unsigned BeginSpelled = 0;
226 unsigned EndSpelled = 0;
227 // Positions in the expanded token stream. The corresponding range can be
228 // empty.
229 unsigned BeginExpanded = 0;
230 unsigned EndExpanded = 0;
231
232 /// For debugging purposes.
233 std::string str() const;
234 };
235 /// Spelled tokens of the file with information about the subranges.
236 struct MarkedFile {
237 /// Lexed, but not preprocessed, tokens of the file. These map directly to
238 /// text in the corresponding files and include tokens of all preprocessor
239 /// directives.
240 /// FIXME: spelled tokens don't change across FileID that map to the same
241 /// FileEntry. We could consider deduplicating them to save memory.
242 std::vector<syntax::Token> SpelledTokens;
243 /// A sorted list to convert between the spelled and expanded token streams.
244 std::vector<Mapping> Mappings;
245 /// The first expanded token produced for this FileID.
246 unsigned BeginExpanded = 0;
247 unsigned EndExpanded = 0;
248 };
249
250 friend class TokenCollector;
251
252 /// Maps a single expanded token to its spelled counterpart or a mapping that
253 /// produced it.
254 std::pair<const syntax::Token *, const Mapping *>
255 spelledForExpandedToken(const syntax::Token *Expanded) const;
256
257 /// Token stream produced after preprocessing, conceputally this captures the
258 /// same stream as 'clang -E' (excluding the preprocessor directives like
259 /// #file, etc.).
260 std::vector<syntax::Token> ExpandedTokens;
261 llvm::DenseMap<FileID, MarkedFile> Files;
262 // The value is never null, pointer instead of reference to avoid disabling
263 // implicit assignment operator.
264 const SourceManager *SourceMgr;
265};

267/// Lex the text buffer, corresponding to \p FID, in raw mode and record the
268/// resulting spelled tokens. Does minimal post-processing on raw identifiers,
269/// setting the appropriate token kind (instead of the raw_identifier reported
270/// by lexer in raw mode). This is a very low-level function, most users should
271/// prefer to use TokenCollector. Lexing in raw mode produces wildly different
272/// results from what one might expect when running a C++ frontend, e.g.
273/// preprocessor does not run at all.
274/// The result will *not* have a 'eof' token at the end.
275std::vector<syntax::Token> tokenize(FileID FID, const SourceManager &SM,
276 const LangOptions &LO);

278/// Collects tokens for the main file while running the frontend action. An
279/// instance of this object should be created on
280/// FrontendAction::BeginSourceFile() and the results should be consumed after
281/// FrontendAction::Execute() finishes.
282class TokenCollector {
283public:
284 /// Adds the hooks to collect the tokens. Should be called before the
285 /// preprocessing starts, i.e. as a part of BeginSourceFile() or
286 /// CreateASTConsumer().
287 TokenCollector(Preprocessor &P);
288
289 /// Finalizes token collection. Should be called after preprocessing is
290 /// finished, i.e. after running Execute().
291 LLVM_NODISCARD TokenBuffer consume() &&;
292
293private:
294 class Builder;
295 std::vector<syntax::Token> Expanded;
296 Preprocessor &PP;
297};
