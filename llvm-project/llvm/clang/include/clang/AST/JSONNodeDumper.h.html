29class NodeStreamer {
30 bool FirstChild = true;
31 bool TopLevel = true;
32 llvm::SmallVector<std::function<void(bool IsLastChild)>, 32> Pending;
33
34protected:
35 llvm::json::OStream JOS;
36
37public:
38 /// Add a child of the current node. Calls DoAddChild without arguments
39 template <typename Fn> void AddChild(Fn DoAddChild) {
40 return AddChild("", DoAddChild);
41 }
42
43 /// Add a child of the current node with an optional label.
44 /// Calls DoAddChild without arguments.
45 template <typename Fn> void AddChild(StringRef Label, Fn DoAddChild) {
46 // If we're at the top level, there's nothing interesting to do; just
47 // run the dumper.
48 if (TopLevel) {
49 TopLevel = false;
50 JOS.objectBegin();
51
52 DoAddChild();
53
54 while (!Pending.empty()) {
55 Pending.back()(true);
56 Pending.pop_back();
57 }
58
59 JOS.objectEnd();
60 TopLevel = true;
61 return;
62 }
63
64 // We need to capture an owning-string in the lambda because the lambda
65 // is invoked in a deferred manner.
66 std::string LabelStr = !Label.empty() ? Label : "inner";
67 bool WasFirstChild = FirstChild;
68 auto DumpWithIndent = [=](bool IsLastChild) {
69 if (WasFirstChild) {
70 JOS.attributeBegin(LabelStr);
71 JOS.arrayBegin();
72 }
73
74 FirstChild = true;
75 unsigned Depth = Pending.size();
76 JOS.objectBegin();
77
78 DoAddChild();
79
80 // If any children are left, they're the last at their nesting level.
81 // Dump those ones out now.
82 while (Depth < Pending.size()) {
83 Pending.back()(true);
84 this->Pending.pop_back();
85 }
86
87 JOS.objectEnd();
88
89 if (IsLastChild) {
90 JOS.arrayEnd();
91 JOS.attributeEnd();
92 }
93 };
94
95 if (FirstChild) {
96 Pending.push_back(std::move(DumpWithIndent));
97 } else {
98 Pending.back()(false);
99 Pending.back() = std::move(DumpWithIndent);
100 }
101 FirstChild = false;
102 }
103
104 NodeStreamer(raw_ostream &OS) : JOS(OS, 2) {}
105};

291class JSONDumper : public ASTNodeTraverser<JSONDumper, JSONNodeDumper> {

294 template <typename SpecializationDecl>
295 void writeTemplateDeclSpecialization(const SpecializationDecl *SD,
296 bool DumpExplicitInst,
297 bool DumpRefOnly) {
298 bool DumpedAny = false;
299 for (const auto *RedeclWithBadType : SD->redecls()) {
300 // FIXME: The redecls() range sometimes has elements of a less-specific
301 // type. (In particular, ClassTemplateSpecializationDecl::redecls() gives
302 // us TagDecls, and should give CXXRecordDecls).
303 const auto *Redecl = dyn_cast<SpecializationDecl>(RedeclWithBadType);
304 if (!Redecl) {
305 // Found the injected-class-name for a class template. This will be
306 // dumped as part of its surrounding class so we don't need to dump it
307 // here.
308 assert(isa<CXXRecordDecl>(RedeclWithBadType) &&
309 "expected an injected-class-name");
310 continue;
311 }
312
313 switch (Redecl->getTemplateSpecializationKind()) {
314 case TSK_ExplicitInstantiationDeclaration:
315 case TSK_ExplicitInstantiationDefinition:
316 if (!DumpExplicitInst)
317 break;
318 LLVM_FALLTHROUGH;
319 case TSK_Undeclared:
320 case TSK_ImplicitInstantiation:
321 if (DumpRefOnly)
322 NodeDumper.JOS.value(NodeDumper.createBareDeclRef(Redecl));
323 else
324 // FIXME: this isn't quite right -- we want to call Visit() rather
325 // than NodeDumper.Visit() but that causes issues because it attempts
326 // to create a new array of child objects due to calling AddChild(),
327 // which messes up the JSON creation.
328 NodeDumper.JOS.object([this, Redecl] { NodeDumper.Visit(Redecl); });
329 DumpedAny = true;
330 break;
331 case TSK_ExplicitSpecialization:
332 break;
333 }
334 }
335
336 // Ensure we dump at least one decl for each specialization.
337 if (!DumpedAny)
338 NodeDumper.JOS.value(NodeDumper.createBareDeclRef(SD));
339 }
340
341 template <typename TemplateDecl>
342 void writeTemplateDecl(const TemplateDecl *TD, bool DumpExplicitInst) {
343 if (const TemplateParameterList *TPL = TD->getTemplateParameters()) {
344 NodeDumper.JOS.attributeArray("templateParams", [this, TPL] {
345 for (const auto &TP : *TPL) {
346 NodeDumper.JOS.object([this, TP] { NodeDumper.Visit(TP); });
347 }
348 });
349 }
350
351 Visit(TD->getTemplatedDecl());
352
353 auto spec_range = TD->specializations();
354 if (!llvm::empty(spec_range)) {
355 NodeDumper.JOS.attributeArray(
356 "specializations", [this, spec_range, TD, DumpExplicitInst] {
357 for (const auto *Child : spec_range)
358 writeTemplateDeclSpecialization(Child, DumpExplicitInst,
359 !TD->isCanonicalDecl());
360 });
361 }
362 }

364public:

368 : NodeDumper(OS, SrcMgr, Ctx, PrintPolicy, Traits) {}

370 JSONNodeDumper &doGetNodeDelegate() { return NodeDumper; }
