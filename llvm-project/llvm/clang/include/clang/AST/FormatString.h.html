31/// Class representing optional flags with location and representation
32/// information.
33class OptionalFlag {
34public:
35 OptionalFlag(const char *Representation)
36 : representation(Representation), flag(false) {}
37 bool isSet() const { return flag; }
38 void set() { flag = true; }
39 void clear() { flag = false; }
40 void setPosition(const char *position) {
41 assert(position);
42 flag = true;
43 this->position = position;
44 }
45 const char *getPosition() const {
46 assert(position);
47 return position;
48 }
49 const char *toString() const { return representation; }
50
51 // Overloaded operators for bool like qualities
52 explicit operator bool() const { return flag; }
53 OptionalFlag& operator=(const bool &rhs) {
54 flag = rhs;
55 return *this; // Return a reference to myself.
56 }
57private:
58 const char *representation;
59 const char *position;
60 bool flag;
61};

63/// Represents the length modifier in a format string in scanf/printf.
64class LengthModifier {
65public:
66 enum Kind {
67 None,
68 AsChar, // 'hh'
69 AsShort, // 'h'
70 AsShortLong, // 'hl' (OpenCL float/int vector element)
71 AsLong, // 'l'
72 AsLongLong, // 'll'
73 AsQuad, // 'q' (BSD, deprecated, for 64-bit integer types)
74 AsIntMax, // 'j'
75 AsIntPtr, // 'P'
76 AsSizeT, // 'z'
77 AsPtrDiff, // 't'
78 AsInt32, // 'I32' (MSVCRT, like __int32)
79 AsInt3264, // 'I' (MSVCRT, like __int3264 from MIDL)
80 AsInt64, // 'I64' (MSVCRT, like __int64)
81 AsLongDouble, // 'L'
82 AsAllocate, // for '%as', GNU extension to C90 scanf
83 AsMAllocate, // for '%ms', GNU extension to scanf
84 AsWide, // 'w' (MSVCRT, like l but only for c, C, s, S, or Z
85 AsWideChar = AsLong // for '%ls', only makes sense for printf
86 };
87
88 LengthModifier()
89 : Position(nullptr), kind(None) {}
90 LengthModifier(const char *pos, Kind k)
91 : Position(pos), kind(k) {}
92
93 const char *getStart() const {
94 return Position;
95 }
96
97 unsigned getLength() const {
98 switch (kind) {
99 default:
100 return 1;
101 case AsLongLong:
102 case AsChar:
103 return 2;
104 case AsInt32:
105 case AsInt64:
106 return 3;
107 case None:
108 return 0;
109 }
110 }
111
112 Kind getKind() const { return kind; }
113 void setKind(Kind k) { kind = k; }
114
115 const char *toString() const;
116
117private:
118 const char *Position;
119 Kind kind;
120};

122class ConversionSpecifier {
123public:
124 enum Kind {
125 InvalidSpecifier = 0,
126 // C99 conversion specifiers.
127 cArg,
128 dArg,
129 DArg, // Apple extension
130 iArg,
131 IntArgBeg = dArg,
132 IntArgEnd = iArg,
133
134 oArg,
135 OArg, // Apple extension
136 uArg,
137 UArg, // Apple extension
138 xArg,
139 XArg,
140 UIntArgBeg = oArg,
141 UIntArgEnd = XArg,
142
143 fArg,
144 FArg,
145 eArg,
146 EArg,
147 gArg,
148 GArg,
149 aArg,
150 AArg,
151 DoubleArgBeg = fArg,
152 DoubleArgEnd = AArg,
153
154 sArg,
155 pArg,
156 nArg,
157 PercentArg,
158 CArg,
159 SArg,
160
161 // Apple extension: P specifies to os_log that the data being pointed to is
162 // to be copied by os_log. The precision indicates the number of bytes to
163 // copy.
164 PArg,
165
166 // ** Printf-specific **
167
168 ZArg, // MS extension
169
170 // Objective-C specific specifiers.
171 ObjCObjArg, // '@'
172 ObjCBeg = ObjCObjArg,
173 ObjCEnd = ObjCObjArg,
174
175 // FreeBSD kernel specific specifiers.
176 FreeBSDbArg,
177 FreeBSDDArg,
178 FreeBSDrArg,
179 FreeBSDyArg,
180
181 // GlibC specific specifiers.
182 PrintErrno, // 'm'
183
184 PrintfConvBeg = ObjCObjArg,
185 PrintfConvEnd = PrintErrno,
186
187 // ** Scanf-specific **
188 ScanListArg, // '['
189 ScanfConvBeg = ScanListArg,
190 ScanfConvEnd = ScanListArg
191 };
192
193 ConversionSpecifier(bool isPrintf = true)
194 : IsPrintf(isPrintf), Position(nullptr), EndScanList(nullptr),
195 kind(InvalidSpecifier) {}
196
197 ConversionSpecifier(bool isPrintf, const char *pos, Kind k)
198 : IsPrintf(isPrintf), Position(pos), EndScanList(nullptr), kind(k) {}
199
200 const char *getStart() const {
201 return Position;
202 }
203
204 StringRef getCharacters() const {
205 return StringRef(getStart(), getLength());
206 }
207
208 bool consumesDataArgument() const {
209 switch (kind) {
210 case PrintErrno:
211 assert(IsPrintf);
212 return false;
213 case PercentArg:
214 return false;
215 case InvalidSpecifier:
216 return false;
217 default:
218 return true;
219 }
220 }
221
222 Kind getKind() const { return kind; }
223 void setKind(Kind k) { kind = k; }
224 unsigned getLength() const {
225 return EndScanList ? EndScanList - Position : 1;
226 }
227 void setEndScanList(const char *pos) { EndScanList = pos; }
228
229 bool isIntArg() const { return (kind >= IntArgBeg && kind <= IntArgEnd) ||
230 kind == FreeBSDrArg || kind == FreeBSDyArg; }
231 bool isUIntArg() const { return kind >= UIntArgBeg && kind <= UIntArgEnd; }
232 bool isAnyIntArg() const { return kind >= IntArgBeg && kind <= UIntArgEnd; }
233 bool isDoubleArg() const {
234 return kind >= DoubleArgBeg && kind <= DoubleArgEnd;
235 }
236
237 const char *toString() const;
238
239 bool isPrintfKind() const { return IsPrintf; }
240
241 Optional<ConversionSpecifier> getStandardSpecifier() const;
242
243protected:
244 bool IsPrintf;
245 const char *Position;
246 const char *EndScanList;
247 Kind kind;
248};

250class ArgType {
251public:
252 enum Kind { UnknownTy, InvalidTy, SpecificTy, ObjCPointerTy, CPointerTy,
253 AnyCharTy, CStrTy, WCStrTy, WIntTy };
254
255 enum MatchKind { NoMatch = 0, Match = 1, NoMatchPedantic };
256
257private:
258 const Kind K;
259 QualType T;
260 const char *Name = nullptr;
261 bool Ptr = false;
262
263 /// The TypeKind identifies certain well-known types like size_t and
264 /// ptrdiff_t.
265 enum class TypeKind { DontCare, SizeT, PtrdiffT };
266 TypeKind TK = TypeKind::DontCare;
267
268public:
269 ArgType(Kind K = UnknownTy, const char *N = nullptr) : K(K), Name(N) {}
270 ArgType(QualType T, const char *N = nullptr) : K(SpecificTy), T(T), Name(N) {}
271 ArgType(CanQualType T) : K(SpecificTy), T(T) {}
272
273 static ArgType Invalid() { return ArgType(InvalidTy); }
274 bool isValid() const { return K != InvalidTy; }
275
276 bool isSizeT() const { return TK == TypeKind::SizeT; }
277
278 bool isPtrdiffT() const { return TK == TypeKind::PtrdiffT; }
279
280 /// Create an ArgType which corresponds to the type pointer to A.
281 static ArgType PtrTo(const ArgType& A) {
282 assert(A.K >= InvalidTy && "ArgType cannot be pointer to invalid/unknown");
283 ArgType Res = A;
284 Res.Ptr = true;
285 return Res;
286 }
287
288 /// Create an ArgType which corresponds to the size_t/ssize_t type.
289 static ArgType makeSizeT(const ArgType &A) {
290 ArgType Res = A;
291 Res.TK = TypeKind::SizeT;
292 return Res;
293 }
294
295 /// Create an ArgType which corresponds to the ptrdiff_t/unsigned ptrdiff_t
296 /// type.
297 static ArgType makePtrdiffT(const ArgType &A) {
298 ArgType Res = A;
299 Res.TK = TypeKind::PtrdiffT;
300 return Res;
301 }
302
303 MatchKind matchesType(ASTContext &C, QualType argTy) const;
304
305 QualType getRepresentativeType(ASTContext &C) const;
306
307 ArgType makeVectorType(ASTContext &C, unsigned NumElts) const;
308
309 std::string getRepresentativeTypeName(ASTContext &C) const;
310};

312class OptionalAmount {
313public:
314 enum HowSpecified { NotSpecified, Constant, Arg, Invalid };
315
316 OptionalAmount(HowSpecified howSpecified,
317 unsigned amount,
318 const char *amountStart,
319 unsigned amountLength,
320 bool usesPositionalArg)
321 : start(amountStart), length(amountLength), hs(howSpecified), amt(amount),
322 UsesPositionalArg(usesPositionalArg), UsesDotPrefix(0) {}
323
324 OptionalAmount(bool valid = true)
325 : start(nullptr),length(0), hs(valid ? NotSpecified : Invalid), amt(0),
326 UsesPositionalArg(0), UsesDotPrefix(0) {}
327
328 explicit OptionalAmount(unsigned Amount)
329 : start(nullptr), length(0), hs(Constant), amt(Amount),
330 UsesPositionalArg(false), UsesDotPrefix(false) {}
331
332 bool isInvalid() const {
333 return hs == Invalid;
334 }
335
336 HowSpecified getHowSpecified() const { return hs; }
337 void setHowSpecified(HowSpecified h) { hs = h; }
338
339 bool hasDataArgument() const { return hs == Arg; }
340
341 unsigned getArgIndex() const {
342 assert(hasDataArgument());
343 return amt;
344 }
345
346 unsigned getConstantAmount() const {
347 assert(hs == Constant);
348 return amt;
349 }
350
351 const char *getStart() const {
352 // We include the . character if it is given.
353 return start - UsesDotPrefix;
354 }
355
356 unsigned getConstantLength() const {
357 assert(hs == Constant);
358 return length + UsesDotPrefix;
359 }
360
361 ArgType getArgType(ASTContext &Ctx) const;
362
363 void toString(raw_ostream &os) const;
364
365 bool usesPositionalArg() const { return (bool) UsesPositionalArg; }
366 unsigned getPositionalArgIndex() const {
367 assert(hasDataArgument());
368 return amt + 1;
369 }
370
371 bool usesDotPrefix() const { return UsesDotPrefix; }
372 void setUsesDotPrefix() { UsesDotPrefix = true; }
373
374private:
375 const char *start;
376 unsigned length;
377 HowSpecified hs;
378 unsigned amt;
379 bool UsesPositionalArg : 1;
380 bool UsesDotPrefix;
381};
