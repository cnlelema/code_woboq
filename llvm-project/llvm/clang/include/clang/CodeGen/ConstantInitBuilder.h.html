32/// A convenience builder class for complex constant initializers,
33/// especially for anonymous global structures used by various language
34/// runtimes.
35///
36/// The basic usage pattern is expected to be something like:
37/// ConstantInitBuilder builder(CGM);
38/// auto toplevel = builder.beginStruct();
39/// toplevel.addInt(CGM.SizeTy, widgets.size());
40/// auto widgetArray = builder.beginArray();
41/// for (auto &widget : widgets) {
42/// auto widgetDesc = widgetArray.beginStruct();
43/// widgetDesc.addInt(CGM.SizeTy, widget.getPower());
44/// widgetDesc.add(CGM.GetAddrOfConstantString(widget.getName()));
45/// widgetDesc.add(CGM.GetAddrOfGlobal(widget.getInitializerDecl()));
46/// widgetDesc.finishAndAddTo(widgetArray);
47/// }
48/// widgetArray.finishAndAddTo(toplevel);
49/// auto global = toplevel.finishAndCreateGlobal("WIDGET_LIST", Align,
50/// /*constant*/ true);
51class ConstantInitBuilderBase {
52 struct SelfReference {
53 llvm::GlobalVariable *Dummy;
54 llvm::SmallVector<llvm::Constant*, 4> Indices;
55
56 SelfReference(llvm::GlobalVariable *dummy) : Dummy(dummy) {}
57 };
58 CodeGenModule &CGM;
59 llvm::SmallVector<llvm::Constant*, 16> Buffer;
60 std::vector<SelfReference> SelfReferences;
61 bool Frozen = false;
62
63 friend class ConstantInitFuture;
64 friend class ConstantAggregateBuilderBase;
65 template <class, class>
66 friend class ConstantAggregateBuilderTemplateBase;
67
68protected:
69 explicit ConstantInitBuilderBase(CodeGenModule &CGM) : CGM(CGM) {}
70
71 ~ConstantInitBuilderBase() {
72 assert(Buffer.empty() && "didn't claim all values out of buffer");
73 assert(SelfReferences.empty() && "didn't apply all self-references");
74 }
75
76private:
77 llvm::GlobalVariable *createGlobal(llvm::Constant *initializer,
78 const llvm::Twine &name,
79 CharUnits alignment,
80 bool constant = false,
81 llvm::GlobalValue::LinkageTypes linkage
82 = llvm::GlobalValue::InternalLinkage,
83 unsigned addressSpace = -1);
84
85 ConstantInitFuture createFuture(llvm::Constant *initializer);
86
87 void setGlobalInitializer(llvm::GlobalVariable *GV,
88 llvm::Constant *initializer);
89
90 void resolveSelfReferences(llvm::GlobalVariable *GV);
91
92 void abandon(size_t newEnd);
93};

95/// A concrete base class for struct and array aggregate
96/// initializer builders.
97class ConstantAggregateBuilderBase {
98protected:
99 ConstantInitBuilderBase &Builder;
100 ConstantAggregateBuilderBase *Parent;
101 size_t Begin;
102 mutable size_t CachedOffsetEnd = 0;
103 bool Finished = false;
104 bool Frozen = false;
105 bool Packed = false;
106 mutable CharUnits CachedOffsetFromGlobal;
107
108 llvm::SmallVectorImpl<llvm::Constant*> &getBuffer() {
109 return Builder.Buffer;
110 }
111
112 const llvm::SmallVectorImpl<llvm::Constant*> &getBuffer() const {
113 return Builder.Buffer;
114 }
115
116 ConstantAggregateBuilderBase(ConstantInitBuilderBase &builder,
117 ConstantAggregateBuilderBase *parent)
118 : Builder(builder), Parent(parent), Begin(builder.Buffer.size()) {
119 if (parent) {
120 assert(!parent->Frozen && "parent already has child builder active");
121 parent->Frozen = true;
122 } else {
123 assert(!builder.Frozen && "builder already has child builder active");
124 builder.Frozen = true;
125 }
126 }
127
128 ~ConstantAggregateBuilderBase() {
129 assert(Finished && "didn't finish aggregate builder");
130 }
131
132 void markFinished() {
133 assert(!Frozen && "child builder still active");
134 assert(!Finished && "builder already finished");
135 Finished = true;
136 if (Parent) {
137 assert(Parent->Frozen &&
138 "parent not frozen while child builder active");
139 Parent->Frozen = false;
140 } else {
141 assert(Builder.Frozen &&
142 "builder not frozen while child builder active");
143 Builder.Frozen = false;
144 }
145 }
146
147public:
148 // Not copyable.
149 ConstantAggregateBuilderBase(const ConstantAggregateBuilderBase &) = delete;
150 ConstantAggregateBuilderBase &operator=(const ConstantAggregateBuilderBase &)
151 = delete;
152
153 // Movable, mostly to allow returning. But we have to write this out
154 // properly to satisfy the assert in the destructor.
155 ConstantAggregateBuilderBase(ConstantAggregateBuilderBase &&other)
156 : Builder(other.Builder), Parent(other.Parent), Begin(other.Begin),
157 CachedOffsetEnd(other.CachedOffsetEnd),
158 Finished(other.Finished), Frozen(other.Frozen), Packed(other.Packed),
159 CachedOffsetFromGlobal(other.CachedOffsetFromGlobal) {
160 other.Finished = true;
161 }
162 ConstantAggregateBuilderBase &operator=(ConstantAggregateBuilderBase &&other)
163 = delete;
164
165 /// Return the number of elements that have been added to
166 /// this struct or array.
167 size_t size() const {
168 assert(!this->Finished && "cannot query after finishing builder");
169 assert(!this->Frozen && "cannot query while sub-builder is active");
170 assert(this->Begin <= this->getBuffer().size());
171 return this->getBuffer().size() - this->Begin;
172 }
173
174 /// Return true if no elements have yet been added to this struct or array.
175 bool empty() const {
176 return size() == 0;
177 }
178
179 /// Abandon this builder completely.
180 void abandon() {
181 markFinished();
182 Builder.abandon(Begin);
183 }
184
185 /// Add a new value to this initializer.
186 void add(llvm::Constant *value) {
187 assert(value && "adding null value to constant initializer");
188 assert(!Finished && "cannot add more values after finishing builder");
189 assert(!Frozen && "cannot add values while subbuilder is active");
190 Builder.Buffer.push_back(value);
191 }
192
193 /// Add an integer value of type size_t.
194 void addSize(CharUnits size);
195
196 /// Add an integer value of a specific type.
197 void addInt(llvm::IntegerType *intTy, uint64_t value,
198 bool isSigned = false) {
199 add(llvm::ConstantInt::get(intTy, value, isSigned));
200 }
201
202 /// Add a null pointer of a specific type.
203 void addNullPointer(llvm::PointerType *ptrTy) {
204 add(llvm::ConstantPointerNull::get(ptrTy));
205 }
206
207 /// Add a bitcast of a value to a specific type.
208 void addBitCast(llvm::Constant *value, llvm::Type *type) {
209 add(llvm::ConstantExpr::getBitCast(value, type));
210 }
211
212 void addPointerdBitCastOrAddrSpaceCast(llvm::Constant *value, llvm::Type *type) {
213 add(llvm::ConstantExpr::getPointerBitCastOrAddrSpaceCast(value, type));
214 }
215
216 /// Add a bunch of new values to this initializer.
217 void addAll(llvm::ArrayRef<llvm::Constant *> values) {
218 assert(!Finished && "cannot add more values after finishing builder");
219 assert(!Frozen && "cannot add values while subbuilder is active");
220 Builder.Buffer.append(values.begin(), values.end());
221 }
222
223 /// Add a relative offset to the given target address, i.e. the
224 /// static difference between the target address and the address
225 /// of the relative offset. The target must be known to be defined
226 /// in the current linkage unit. The offset will have the given
227 /// integer type, which must be no wider than intptr_t. Some
228 /// targets may not fully support this operation.
229 void addRelativeOffset(llvm::IntegerType *type, llvm::Constant *target) {
230 add(getRelativeOffset(type, target));
231 }
232
233 /// Add a relative offset to the target address, plus a small
234 /// constant offset. This is primarily useful when the relative
235 /// offset is known to be a multiple of (say) four and therefore
236 /// the tag can be used to express an extra two bits of information.
237 void addTaggedRelativeOffset(llvm::IntegerType *type,
238 llvm::Constant *address,
239 unsigned tag) {
240 llvm::Constant *offset = getRelativeOffset(type, address);
241 if (tag) {
242 offset = llvm::ConstantExpr::getAdd(offset,
243 llvm::ConstantInt::get(type, tag));
244 }
245 add(offset);
246 }
247
248 /// Return the offset from the start of the initializer to the
249 /// next position, assuming no padding is required prior to it.
250 ///
251 /// This operation will not succeed if any unsized placeholders are
252 /// currently in place in the initializer.
253 CharUnits getNextOffsetFromGlobal() const {
254 assert(!Finished && "cannot add more values after finishing builder");
255 assert(!Frozen && "cannot add values while subbuilder is active");
256 return getOffsetFromGlobalTo(Builder.Buffer.size());
257 }
258
259 /// An opaque class to hold the abstract position of a placeholder.
260 class PlaceholderPosition {
261 size_t Index;
262 friend class ConstantAggregateBuilderBase;
263 PlaceholderPosition(size_t index) : Index(index) {}
264 };
265
266 /// Add a placeholder value to the structure. The returned position
267 /// can be used to set the value later; it will not be invalidated by
268 /// any intermediate operations except (1) filling the same position or
269 /// (2) finishing the entire builder.
270 ///
271 /// This is useful for emitting certain kinds of structure which
272 /// contain some sort of summary field, generally a count, before any
273 /// of the data. By emitting a placeholder first, the structure can
274 /// be emitted eagerly.
275 PlaceholderPosition addPlaceholder() {
276 assert(!Finished && "cannot add more values after finishing builder");
277 assert(!Frozen && "cannot add values while subbuilder is active");
278 Builder.Buffer.push_back(nullptr);
279 return Builder.Buffer.size() - 1;
280 }
281
282 /// Add a placeholder, giving the expected type that will be filled in.
283 PlaceholderPosition addPlaceholderWithSize(llvm::Type *expectedType);
284
285 /// Fill a previously-added placeholder.
286 void fillPlaceholderWithInt(PlaceholderPosition position,
287 llvm::IntegerType *type, uint64_t value,
288 bool isSigned = false) {
289 fillPlaceholder(position, llvm::ConstantInt::get(type, value, isSigned));
290 }
291
292 /// Fill a previously-added placeholder.
293 void fillPlaceholder(PlaceholderPosition position, llvm::Constant *value) {
294 assert(!Finished && "cannot change values after finishing builder");
295 assert(!Frozen && "cannot add values while subbuilder is active");
296 llvm::Constant *&slot = Builder.Buffer[position.Index];
297 assert(slot == nullptr && "placeholder already filled");
298 slot = value;
299 }
300
301 /// Produce an address which will eventually point to the next
302 /// position to be filled. This is computed with an indexed
303 /// getelementptr rather than by computing offsets.
304 ///
305 /// The returned pointer will have type T*, where T is the given
306 /// position.
307 llvm::Constant *getAddrOfCurrentPosition(llvm::Type *type);
308
309 llvm::ArrayRef<llvm::Constant*> getGEPIndicesToCurrentPosition(
310 llvm::SmallVectorImpl<llvm::Constant*> &indices) {
311 getGEPIndicesTo(indices, Builder.Buffer.size());
312 return indices;
313 }
314
315protected:
316 llvm::Constant *finishArray(llvm::Type *eltTy);
317 llvm::Constant *finishStruct(llvm::StructType *structTy);
318
319private:
320 void getGEPIndicesTo(llvm::SmallVectorImpl<llvm::Constant*> &indices,
321 size_t position) const;
322
323 llvm::Constant *getRelativeOffset(llvm::IntegerType *offsetType,
324 llvm::Constant *target);
325
326 CharUnits getOffsetFromGlobalTo(size_t index) const;
327};

329template <class Impl, class Traits>
330class ConstantAggregateBuilderTemplateBase
331 : public Traits::AggregateBuilderBase {
332 using super = typename Traits::AggregateBuilderBase;
333public:
334 using InitBuilder = typename Traits::InitBuilder;
335 using ArrayBuilder = typename Traits::ArrayBuilder;
336 using StructBuilder = typename Traits::StructBuilder;
337 using AggregateBuilderBase = typename Traits::AggregateBuilderBase;
338
339protected:
340 ConstantAggregateBuilderTemplateBase(InitBuilder &builder,
341 AggregateBuilderBase *parent)
342 : super(builder, parent) {}
343
344 Impl &asImpl() { return *static_cast<Impl*>(this); }
345
346public:
347 ArrayBuilder beginArray(llvm::Type *eltTy = nullptr) {
348 return ArrayBuilder(static_cast<InitBuilder&>(this->Builder), this, eltTy);
349 }
350
351 StructBuilder beginStruct(llvm::StructType *ty = nullptr) {
352 return StructBuilder(static_cast<InitBuilder&>(this->Builder), this, ty);
353 }
354
355 /// Given that this builder was created by beginning an array or struct
356 /// component on the given parent builder, finish the array/struct
357 /// component and add it to the parent.
358 ///
359 /// It is an intentional choice that the parent is passed in explicitly
360 /// despite it being redundant with information already kept in the
361 /// builder. This aids in readability by making it easier to find the
362 /// places that add components to a builder, as well as "bookending"
363 /// the sub-builder more explicitly.
364 void finishAndAddTo(AggregateBuilderBase &parent) {
365 assert(this->Parent == &parent && "adding to non-parent builder");
366 parent.add(asImpl().finishImpl());
367 }
368
369 /// Given that this builder was created by beginning an array or struct
370 /// directly on a ConstantInitBuilder, finish the array/struct and
371 /// create a global variable with it as the initializer.
372 template <class... As>
373 llvm::GlobalVariable *finishAndCreateGlobal(As &&...args) {
374 assert(!this->Parent && "finishing non-root builder");
375 return this->Builder.createGlobal(asImpl().finishImpl(),
376 std::forward<As>(args)...);
377 }
378
379 /// Given that this builder was created by beginning an array or struct
380 /// directly on a ConstantInitBuilder, finish the array/struct and
381 /// set it as the initializer of the given global variable.
382 void finishAndSetAsInitializer(llvm::GlobalVariable *global) {
383 assert(!this->Parent && "finishing non-root builder");
384 return this->Builder.setGlobalInitializer(global, asImpl().finishImpl());
385 }
386
387 /// Given that this builder was created by beginning an array or struct
388 /// directly on a ConstantInitBuilder, finish the array/struct and
389 /// return a future which can be used to install the initializer in
390 /// a global later.
391 ///
392 /// This is useful for allowing a finished initializer to passed to
393 /// an API which will build the global. However, the "future" preserves
394 /// a dependency on the original builder; it is an error to pass it aside.
395 ConstantInitFuture finishAndCreateFuture() {
396 assert(!this->Parent && "finishing non-root builder");
397 return this->Builder.createFuture(asImpl().finishImpl());
398 }
399};

401template <class Traits>
402class ConstantArrayBuilderTemplateBase
403 : public ConstantAggregateBuilderTemplateBase<typename Traits::ArrayBuilder,
404 Traits> {
405 using super =
406 ConstantAggregateBuilderTemplateBase<typename Traits::ArrayBuilder, Traits>;
407
408public:
409 using InitBuilder = typename Traits::InitBuilder;
410 using AggregateBuilderBase = typename Traits::AggregateBuilderBase;
411
412private:
413 llvm::Type *EltTy;
414
415 template <class, class>
416 friend class ConstantAggregateBuilderTemplateBase;
417
418protected:
419 ConstantArrayBuilderTemplateBase(InitBuilder &builder,
420 AggregateBuilderBase *parent,
421 llvm::Type *eltTy)
422 : super(builder, parent), EltTy(eltTy) {}
423
424private:
425 /// Form an array constant from the values that have been added to this
426 /// builder.
427 llvm::Constant *finishImpl() {
428 return AggregateBuilderBase::finishArray(EltTy);
429 }
430};

432/// A template class designed to allow other frontends to
433/// easily customize the builder classes used by ConstantInitBuilder,
434/// and thus to extend the API to work with the abstractions they
435/// prefer. This would probably not be necessary if C++ just
436/// supported extension methods.
437template <class Traits>
438class ConstantStructBuilderTemplateBase
439 : public ConstantAggregateBuilderTemplateBase<typename Traits::StructBuilder,
440 Traits> {
441 using super =
442 ConstantAggregateBuilderTemplateBase<typename Traits::StructBuilder,Traits>;
443
444public:
445 using InitBuilder = typename Traits::InitBuilder;
446 using AggregateBuilderBase = typename Traits::AggregateBuilderBase;
447
448private:
449 llvm::StructType *StructTy;
450
451 template <class, class>
452 friend class ConstantAggregateBuilderTemplateBase;
453
454protected:
455 ConstantStructBuilderTemplateBase(InitBuilder &builder,
456 AggregateBuilderBase *parent,
457 llvm::StructType *structTy)
458 : super(builder, parent), StructTy(structTy) {
459 if (structTy) this->Packed = structTy->isPacked();
460 }
461
462public:
463 void setPacked(bool packed) {
464 this->Packed = packed;
465 }
466
467 /// Use the given type for the struct if its element count is correct.
468 /// Don't add more elements after calling this.
469 void suggestType(llvm::StructType *structTy) {
470 if (this->size() == structTy->getNumElements()) {
471 StructTy = structTy;
472 }
473 }
474
475private:
476 /// Form an array constant from the values that have been added to this
477 /// builder.
478 llvm::Constant *finishImpl() {
479 return AggregateBuilderBase::finishStruct(StructTy);
480 }
481};

483/// A template class designed to allow other frontends to
484/// easily customize the builder classes used by ConstantInitBuilder,
485/// and thus to extend the API to work with the abstractions they
486/// prefer. This would probably not be necessary if C++ just
487/// supported extension methods.
488template <class Traits>
489class ConstantInitBuilderTemplateBase : public ConstantInitBuilderBase {
490protected:
491 ConstantInitBuilderTemplateBase(CodeGenModule &CGM)
492 : ConstantInitBuilderBase(CGM) {}
493
494public:
495 using InitBuilder = typename Traits::InitBuilder;
496 using ArrayBuilder = typename Traits::ArrayBuilder;
497 using StructBuilder = typename Traits::StructBuilder;
498
499 ArrayBuilder beginArray(llvm::Type *eltTy = nullptr) {
500 return ArrayBuilder(static_cast<InitBuilder&>(*this), nullptr, eltTy);
501 }
502
503 StructBuilder beginStruct(llvm::StructType *structTy = nullptr) {
504 return StructBuilder(static_cast<InitBuilder&>(*this), nullptr, structTy);
505 }
506};

512struct ConstantInitBuilderTraits {
