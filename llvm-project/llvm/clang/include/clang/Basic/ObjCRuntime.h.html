24namespace clang {

26/// The basic abstraction for the target Objective-C runtime.
27class ObjCRuntime {
28public:
29 /// The basic Objective-C runtimes that we know about.
30 enum Kind {
31 /// 'macosx' is the Apple-provided NeXT-derived runtime on Mac OS
32 /// X platforms that use the non-fragile ABI; the version is a
33 /// release of that OS.
34 MacOSX,
35
36 /// 'macosx-fragile' is the Apple-provided NeXT-derived runtime on
37 /// Mac OS X platforms that use the fragile ABI; the version is a
38 /// release of that OS.
39 FragileMacOSX,
40
41 /// 'ios' is the Apple-provided NeXT-derived runtime on iOS or the iOS
42 /// simulator; it is always non-fragile. The version is a release
43 /// version of iOS.
44 iOS,
45
46 /// 'watchos' is a variant of iOS for Apple's watchOS. The version
47 /// is a release version of watchOS.
48 WatchOS,
49
50 /// 'gcc' is the Objective-C runtime shipped with GCC, implementing a
51 /// fragile Objective-C ABI
52 GCC,
53
54 /// 'gnustep' is the modern non-fragile GNUstep runtime.
55 GNUstep,
56
57 /// 'objfw' is the Objective-C runtime included in ObjFW
58 ObjFW
59 };
60
61private:
62 Kind TheKind = MacOSX;
63 VersionTuple Version;
64
65public:
66 /// A bogus initialization of the runtime.
67 ObjCRuntime() = default;
68 ObjCRuntime(Kind kind, const VersionTuple &version)
69 : TheKind(kind), Version(version) {}
70
71 void set(Kind kind, VersionTuple version) {
72 TheKind = kind;
73 Version = version;
74 }
75
76 Kind getKind() const { return TheKind; }
77 const VersionTuple &getVersion() const { return Version; }
78
79 /// Does this runtime follow the set of implied behaviors for a
80 /// "non-fragile" ABI?
81 bool isNonFragile() const {
82 switch (getKind()) {
83 case FragileMacOSX: return false;
84 case GCC: return false;
85 case MacOSX: return true;
86 case GNUstep: return true;
87 case ObjFW: return true;
88 case iOS: return true;
89 case WatchOS: return true;
90 }
91 llvm_unreachable("bad kind");
92 }
93
94 /// The inverse of isNonFragile(): does this runtime follow the set of
95 /// implied behaviors for a "fragile" ABI?
96 bool isFragile() const { return !isNonFragile(); }
97
98 /// The default dispatch mechanism to use for the specified architecture
99 bool isLegacyDispatchDefaultForArch(llvm::Triple::ArchType Arch) {
100 // The GNUstep runtime uses a newer dispatch method by default from
101 // version 1.6 onwards
102 if (getKind() == GNUstep && getVersion() >= VersionTuple(1, 6)) {
103 if (Arch == llvm::Triple::arm ||
104 Arch == llvm::Triple::x86 ||
105 Arch == llvm::Triple::x86_64)
106 return false;
107 if (getVersion() >= VersionTuple(1, 9) &&
108 (Arch == llvm::Triple::cheri))
109 return false;
110 if (getVersion() >= VersionTuple(1, 7) &&
111 (Arch == llvm::Triple::mips64))
112 return false;
113 }
114 else if ((getKind() == MacOSX) && isNonFragile() &&
115 (getVersion() >= VersionTuple(10, 0)) &&
116 (getVersion() < VersionTuple(10, 6)))
117 return Arch != llvm::Triple::x86_64;
118 // Except for deployment target of 10.5 or less,
119 // Mac runtimes use legacy dispatch everywhere now.
120 return true;
121 }
122
123 /// Is this runtime basically of the GNU family of runtimes?
124 bool isGNUFamily() const {
125 switch (getKind()) {
126 case FragileMacOSX:
127 case MacOSX:
128 case iOS:
129 case WatchOS:
130 return false;
131 case GCC:
132 case GNUstep:
133 case ObjFW:
134 return true;
135 }
136 llvm_unreachable("bad kind");
137 }
138
139 /// Is this runtime basically of the NeXT family of runtimes?
140 bool isNeXTFamily() const {
141 // For now, this is just the inverse of isGNUFamily(), but that's
142 // not inherently true.
143 return !isGNUFamily();
144 }
145
146 /// Does this runtime allow ARC at all?
147 bool allowsARC() const {
148 switch (getKind()) {
149 case FragileMacOSX:
150 // No stub library for the fragile runtime.
151 return getVersion() >= VersionTuple(10, 7);
152 case MacOSX: return true;
153 case iOS: return true;
154 case WatchOS: return true;
155 case GCC: return false;
156 case GNUstep: return true;
157 case ObjFW: return true;
158 }
159 llvm_unreachable("bad kind");
160 }
161
162 /// Does this runtime natively provide the ARC entrypoints?
163 ///
164 /// ARC cannot be directly supported on a platform that does not provide
165 /// these entrypoints, although it may be supportable via a stub
166 /// library.
167 bool hasNativeARC() const {
168 switch (getKind()) {
169 case FragileMacOSX: return getVersion() >= VersionTuple(10, 7);
170 case MacOSX: return getVersion() >= VersionTuple(10, 7);
171 case iOS: return getVersion() >= VersionTuple(5);
172 case WatchOS: return true;
173
174 case GCC: return false;
175 case GNUstep: return getVersion() >= VersionTuple(1, 6);
176 case ObjFW: return true;
177 }
178 llvm_unreachable("bad kind");
179 }
180
181 /// Does this runtime provide ARC entrypoints that are likely to be faster
182 /// than an ordinary message send of the appropriate selector?
183 ///
184 /// The ARC entrypoints are guaranteed to be equivalent to just sending the
185 /// corresponding message. If the entrypoint is implemented naively as just a
186 /// message send, using it is a trade-off: it sacrifices a few cycles of
187 /// overhead to save a small amount of code. However, it's possible for
188 /// runtimes to detect and special-case classes that use "standard"
189 /// retain/release behavior; if that's dynamically a large proportion of all
190 /// retained objects, using the entrypoint will also be faster than using a
191 /// message send.
192 ///
193 /// When this method returns true, Clang will turn non-super message sends of
194 /// certain selectors into calls to the correspond entrypoint:
195 /// retain => objc_retain
196 /// release => objc_release
197 /// autorelease => objc_autorelease
198 bool shouldUseARCFunctionsForRetainRelease() const {
199 switch (getKind()) {
200 case FragileMacOSX:
201 return false;
202 case MacOSX:
203 return getVersion() >= VersionTuple(10, 10);
204 case iOS:
205 return getVersion() >= VersionTuple(8);
206 case WatchOS:
207 return true;
208 case GCC:
209 return false;
210 case GNUstep:
211 return false;
212 case ObjFW:
213 return false;
214 }
215 llvm_unreachable("bad kind");
216 }
217
218 /// Does this runtime provide entrypoints that are likely to be faster
219 /// than an ordinary message send of the "alloc" selector?
220 ///
221 /// The "alloc" entrypoint is guaranteed to be equivalent to just sending the
222 /// corresponding message. If the entrypoint is implemented naively as just a
223 /// message send, using it is a trade-off: it sacrifices a few cycles of
224 /// overhead to save a small amount of code. However, it's possible for
225 /// runtimes to detect and special-case classes that use "standard"
226 /// alloc behavior; if that's dynamically a large proportion of all
227 /// objects, using the entrypoint will also be faster than using a message
228 /// send.
229 ///
230 /// When this method returns true, Clang will turn non-super message sends of
231 /// certain selectors into calls to the corresponding entrypoint:
232 /// alloc => objc_alloc
233 /// allocWithZone:nil => objc_allocWithZone
234 bool shouldUseRuntimeFunctionsForAlloc() const {
235 switch (getKind()) {
236 case FragileMacOSX:
237 return false;
238 case MacOSX:
239 return getVersion() >= VersionTuple(10, 10);
240 case iOS:
241 return getVersion() >= VersionTuple(8);
242 case WatchOS:
243 return true;
244
245 case GCC:
246 return false;
247 case GNUstep:
248 return false;
249 case ObjFW:
250 return false;
251 }
252 llvm_unreachable("bad kind");
253 }
254
255 /// Does this runtime provide the objc_alloc_init entrypoint? This can apply
256 /// the same optimization as objc_alloc, but also sends an -init message,
257 /// reducing code size on the caller.
258 bool shouldUseRuntimeFunctionForCombinedAllocInit() const {
259 switch (getKind()) {
260 case MacOSX:
261 return getVersion() >= VersionTuple(10, 14, 4);
262 case iOS:
263 return getVersion() >= VersionTuple(12, 2);
264 case WatchOS:
265 return getVersion() >= VersionTuple(5, 2);
266 default:
267 return false;
268 }
269 }
270
271 /// Does this runtime supports optimized setter entrypoints?
272 bool hasOptimizedSetter() const {
273 switch (getKind()) {
274 case MacOSX:
275 return getVersion() >= VersionTuple(10, 8);
276 case iOS:
277 return (getVersion() >= VersionTuple(6));
278 case WatchOS:
279 return true;
280 case GNUstep:
281 return getVersion() >= VersionTuple(1, 7);
282 default:
283 return false;
284 }
285 }
286
287 /// Does this runtime allow the use of __weak?
288 bool allowsWeak() const {
289 return hasNativeWeak();
290 }
291
292 /// Does this runtime natively provide ARC-compliant 'weak'
293 /// entrypoints?
294 bool hasNativeWeak() const {
295 // Right now, this is always equivalent to whether the runtime
296 // natively supports ARC decision.
297 return hasNativeARC();
298 }
299
300 /// Does this runtime directly support the subscripting methods?
301 ///
302 /// This is really a property of the library, not the runtime.
303 bool hasSubscripting() const {
304 switch (getKind()) {
305 case FragileMacOSX: return false;
306 case MacOSX: return getVersion() >= VersionTuple(10, 11);
307 case iOS: return getVersion() >= VersionTuple(9);
308 case WatchOS: return true;
309
310 // This is really a lie, because some implementations and versions
311 // of the runtime do not support ARC. Probably -fgnu-runtime
312 // should imply a "maximal" runtime or something?
313 case GCC: return true;
314 case GNUstep: return true;
315 case ObjFW: return true;
316 }
317 llvm_unreachable("bad kind");
318 }
319
320 /// Does this runtime allow sizeof or alignof on object types?
321 bool allowsSizeofAlignof() const {
322 return isFragile();
323 }
324
325 /// Does this runtime allow pointer arithmetic on objects?
326 ///
327 /// This covers +, -, ++, --, and (if isSubscriptPointerArithmetic()
328 /// yields true) [].
329 bool allowsPointerArithmetic() const {
330 switch (getKind()) {
331 case FragileMacOSX:
332 case GCC:
333 return true;
334 case MacOSX:
335 case iOS:
336 case WatchOS:
337 case GNUstep:
338 case ObjFW:
339 return false;
340 }
341 llvm_unreachable("bad kind");
342 }
343
344 /// Is subscripting pointer arithmetic?
345 bool isSubscriptPointerArithmetic() const {
346 return allowsPointerArithmetic();
347 }
348
349 /// Does this runtime provide an objc_terminate function?
350 ///
351 /// This is used in handlers for exceptions during the unwind process;
352 /// without it, abort() must be used in pure ObjC files.
353 bool hasTerminate() const {
354 switch (getKind()) {
355 case FragileMacOSX: return getVersion() >= VersionTuple(10, 8);
356 case MacOSX: return getVersion() >= VersionTuple(10, 8);
357 case iOS: return getVersion() >= VersionTuple(5);
358 case WatchOS: return true;
359 case GCC: return false;
360 case GNUstep: return false;
361 case ObjFW: return false;
362 }
363 llvm_unreachable("bad kind");
364 }
365
366 /// Does this runtime support weakly importing classes?
367 bool hasWeakClassImport() const {
368 switch (getKind()) {
369 case MacOSX: return true;
370 case iOS: return true;
371 case WatchOS: return true;
372 case FragileMacOSX: return false;
373 case GCC: return true;
374 case GNUstep: return true;
375 case ObjFW: return true;
376 }
377 llvm_unreachable("bad kind");
378 }
379
380 /// Does this runtime use zero-cost exceptions?
381 bool hasUnwindExceptions() const {
382 switch (getKind()) {
383 case MacOSX: return true;
384 case iOS: return true;
385 case WatchOS: return true;
386 case FragileMacOSX: return false;
387 case GCC: return true;
388 case GNUstep: return true;
389 case ObjFW: return true;
390 }
391 llvm_unreachable("bad kind");
392 }
393
394 bool hasAtomicCopyHelper() const {
395 switch (getKind()) {
396 case FragileMacOSX:
397 case MacOSX:
398 case iOS:
399 case WatchOS:
400 return true;
401 case GNUstep:
402 return getVersion() >= VersionTuple(1, 7);
403 default: return false;
404 }
405 }
406
407 /// Is objc_unsafeClaimAutoreleasedReturnValue available?
408 bool hasARCUnsafeClaimAutoreleasedReturnValue() const {
409 switch (getKind()) {
410 case MacOSX:
411 case FragileMacOSX:
412 return getVersion() >= VersionTuple(10, 11);
413 case iOS:
414 return getVersion() >= VersionTuple(9);
415 case WatchOS:
416 return getVersion() >= VersionTuple(2);
417 case GNUstep:
418 return false;
419 default:
420 return false;
421 }
422 }
423
424 /// Are the empty collection symbols available?
425 bool hasEmptyCollections() const {
426 switch (getKind()) {
427 default:
428 return false;
429 case MacOSX:
430 return getVersion() >= VersionTuple(10, 11);
431 case iOS:
432 return getVersion() >= VersionTuple(9);
433 case WatchOS:
434 return getVersion() >= VersionTuple(2);
435 }
436 }
437
438 /// Returns true if this Objective-C runtime supports Objective-C class
439 /// stubs.
440 bool allowsClassStubs() const {
441 switch (getKind()) {
442 case FragileMacOSX:
443 case GCC:
444 case GNUstep:
445 case ObjFW:
446 return false;
447 case MacOSX:
448 case iOS:
449 case WatchOS:
450 return true;
451 }
452 llvm_unreachable("bad kind");
453 }
454
455 /// Try to parse an Objective-C runtime specification from the given
456 /// string.
457 ///
458 /// \return true on error.
459 bool tryParse(StringRef input);
460
461 std::string getAsString() const;
462
463 friend bool operator==(const ObjCRuntime &left, const ObjCRuntime &right) {
464 return left.getKind() == right.getKind() &&
465 left.getVersion() == right.getVersion();
466 }
467
468 friend bool operator!=(const ObjCRuntime &left, const ObjCRuntime &right) {
469 return !(left == right);
470 }
471};
