29/// Bitfields of CodeGenOptions, split out from CodeGenOptions to ensure
30/// that this large collection of bitfields is a trivial class type.
31class CodeGenOptionsBase {
32public:
33#define CODEGENOPT(Name, Bits, Default) unsigned Name : Bits;
34#define ENUM_CODEGENOPT(Name, Type, Bits, Default)
35#include "clang/Basic/CodeGenOptions.def"
36
37protected:
38#define CODEGENOPT(Name, Bits, Default)
39#define ENUM_CODEGENOPT(Name, Type, Bits, Default) unsigned Name : Bits;
40#include "clang/Basic/CodeGenOptions.def"
41};
42
43/// CodeGenOptions - Track various options which control how the code
44/// is optimized and passed to the backend.
45class CodeGenOptions : public CodeGenOptionsBase {
46public:
47 enum InliningMethod {
48 NormalInlining, // Use the standard function inlining pass.
49 OnlyHintInlining, // Inline only (implicitly) hinted functions.
50 OnlyAlwaysInlining // Only run the always inlining pass.
51 };
52
53 enum VectorLibrary {
54 NoLibrary, // Don't use any vector library.
55 Accelerate, // Use the Accelerate framework.
56 MASSV, // IBM MASS vector library.
57 SVML // Intel short vector math library.
58 };
59
60
61 enum ObjCDispatchMethodKind {
62 Legacy = 0,
63 NonLegacy = 1,
64 Mixed = 2
65 };
66
67 enum TLSModel {
68 GeneralDynamicTLSModel,
69 LocalDynamicTLSModel,
70 InitialExecTLSModel,
71 LocalExecTLSModel
72 };
73
74 enum DwarfFissionKind { NoFission, SplitFileFission, SingleFileFission };
75
76 /// Clang versions with different platform ABI conformance.
77 enum class ClangABI {
78 /// Attempt to be ABI-compatible with code generated by Clang 3.8.x
79 /// (SVN r257626). This causes <1 x long long> to be passed in an
80 /// integer register instead of an SSE register on x64_64.
81 Ver3_8,
82
83 /// Attempt to be ABI-compatible with code generated by Clang 4.0.x
84 /// (SVN r291814). This causes move operations to be ignored when
85 /// determining whether a class type can be passed or returned directly.
86 Ver4,
87
88 /// Conform to the underlying platform's C and C++ ABIs as closely
89 /// as we can.
90 Latest
91 };
92
93 enum StructReturnConventionKind {
94 SRCK_Default, // No special option was passed.
95 SRCK_OnStack, // Small structs on the stack (-fpcc-struct-return).
96 SRCK_InRegs // Small structs in registers (-freg-struct-return).
97 };
98
99 enum ProfileInstrKind {
100 ProfileNone, // Profile instrumentation is turned off.
101 ProfileClangInstr, // Clang instrumentation to generate execution counts
102 // to use with PGO.
103 ProfileIRInstr, // IR level PGO instrumentation in LLVM.
104 ProfileCSIRInstr, // IR level PGO context sensitive instrumentation in LLVM.
105 };
106
107 enum EmbedBitcodeKind {
108 Embed_Off, // No embedded bitcode.
109 Embed_All, // Embed both bitcode and commandline in the output.
110 Embed_Bitcode, // Embed just the bitcode in the output.
111 Embed_Marker // Embed a marker as a placeholder for bitcode.
112 };
113
114 enum SignReturnAddressScope {
115 None, // No signing for any function
116 NonLeaf, // Sign the return address of functions that spill LR
117 All // Sign the return address of all functions
118 };
119
120 enum SignReturnAddressKeyValue { AKey, BKey };
121
122 /// The code model to use (-mcmodel).
123 std::string CodeModel;
124
125 /// The filename with path we use for coverage data files. The runtime
126 /// allows further manipulation with the GCOV_PREFIX and GCOV_PREFIX_STRIP
127 /// environment variables.
128 std::string CoverageDataFile;
129
130 /// The filename with path we use for coverage notes files.
131 std::string CoverageNotesFile;
132
133 /// Regexes separated by a semi-colon to filter the files to instrument.
134 std::string ProfileFilterFiles;
135
136 /// Regexes separated by a semi-colon to filter the files to not instrument.
137 std::string ProfileExcludeFiles;
138
139 /// The version string to put into coverage files.
140 char CoverageVersion[4];
141
142 /// Enable additional debugging information.
143 std::string DebugPass;
144
145 /// The string to embed in debug information as the current working directory.
146 std::string DebugCompilationDir;
147
148 /// The string to embed in the debug information for the compile unit, if
149 /// non-empty.
150 std::string DwarfDebugFlags;
151
152 /// The string containing the commandline for the llvm.commandline metadata,
153 /// if non-empty.
154 std::string RecordCommandLine;
155
156 std::map<std::string, std::string> DebugPrefixMap;
157
158 /// The ABI to use for passing floating point arguments.
159 std::string FloatABI;
160
161 /// The floating-point denormal mode to use.
162 std::string FPDenormalMode;
163
164 /// The float precision limit to use, if non-empty.
165 std::string LimitFloatPrecision;
166
167 struct BitcodeFileToLink {
168 /// The filename of the bitcode file to link in.
169 std::string Filename;
170 /// If true, we set attributes functions in the bitcode library according to
171 /// our CodeGenOptions, much as we set attrs on functions that we generate
172 /// ourselves.
173 bool PropagateAttrs = false;
174 /// If true, we use LLVM module internalizer.
175 bool Internalize = false;
176 /// Bitwise combination of llvm::Linker::Flags, passed to the LLVM linker.
177 unsigned LinkFlags = 0;
178 };
179
180 /// The files specified here are linked in to the module before optimizations.
181 std::vector<BitcodeFileToLink> LinkBitcodeFiles;
182
183 /// The user provided name for the "main file", if non-empty. This is useful
184 /// in situations where the input file name does not match the original input
185 /// file, for example with -save-temps.
186 std::string MainFileName;
187
188 /// The name for the split debug info file that we'll break out. This is used
189 /// in the backend for setting the name in the skeleton cu.
190 std::string SplitDwarfFile;
191
192 /// The name of the relocation model to use.
193 llvm::Reloc::Model RelocationModel;
194
195 /// The thread model to use
196 std::string ThreadModel;
197
198 /// If not an empty string, trap intrinsics are lowered to calls to this
199 /// function instead of to trap instructions.
200 std::string TrapFuncName;
201
202 /// A list of dependent libraries.
203 std::vector<std::string> DependentLibraries;
204
205 /// A list of linker options to embed in the object file.
206 std::vector<std::string> LinkerOptions;
207
208 /// Name of the profile file to use as output for -fprofile-instr-generate,
209 /// -fprofile-generate, and -fcs-profile-generate.
210 std::string InstrProfileOutput;
211
212 /// Name of the profile file to use with -fprofile-sample-use.
213 std::string SampleProfileFile;
214
215 /// Name of the profile file to use as input for -fprofile-instr-use
216 std::string ProfileInstrumentUsePath;
217
218 /// Name of the profile remapping file to apply to the profile data supplied
219 /// by -fprofile-sample-use or -fprofile-instr-use.
220 std::string ProfileRemappingFile;
221
222 /// Name of the function summary index file to use for ThinLTO function
223 /// importing.
224 std::string ThinLTOIndexFile;
225
226 /// Name of a file that can optionally be written with minimized bitcode
227 /// to be used as input for the ThinLTO thin link step, which only needs
228 /// the summary and module symbol table (and not, e.g. any debug metadata).
229 std::string ThinLinkBitcodeFile;
230
231 /// Prefix to use for -save-temps output.
232 std::string SaveTempsFilePrefix;
233
234 /// The name of the file to which the backend should save CHERI-related
235 /// statistics (number of ptr->int casts, csetbounds info, etc..) as JSON.
236 /// Note: This file can be a shared between multiple compiler instances
237 /// since we will use a lock file global
238 std::string CHERIStatsFile;
239
240 /// Name of file passed with -fcuda-include-gpubinary option to forward to
241 /// CUDA runtime back-end for incorporating them into host-side object file.
242 std::string CudaGpuBinaryFileName;
243
244 /// The name of the file to which the backend should save YAML optimization
245 /// records.
246 std::string OptRecordFile;
247
248 /// The regex that filters the passes that should be saved to the optimization
249 /// records.
250 std::string OptRecordPasses;
251
252 /// The name of the partition that symbols are assigned to, specified with
253 /// -fsymbol-partition (see https://lld.llvm.org/Partitions.html).
254 std::string SymbolPartition;
255
256 /// Regular expression to select optimizations for which we should enable
257 /// optimization remarks. Transformation passes whose name matches this
258 /// expression (and support this feature), will emit a diagnostic
259 /// whenever they perform a transformation. This is enabled by the
260 /// -Rpass=regexp flag.
261 std::shared_ptr<llvm::Regex> OptimizationRemarkPattern;
262
263 /// Regular expression to select optimizations for which we should enable
264 /// missed optimization remarks. Transformation passes whose name matches this
265 /// expression (and support this feature), will emit a diagnostic
266 /// whenever they tried but failed to perform a transformation. This is
267 /// enabled by the -Rpass-missed=regexp flag.
268 std::shared_ptr<llvm::Regex> OptimizationRemarkMissedPattern;
269
270 /// Regular expression to select optimizations for which we should enable
271 /// optimization analyses. Transformation passes whose name matches this
272 /// expression (and support this feature), will emit a diagnostic
273 /// whenever they want to explain why they decided to apply or not apply
274 /// a given transformation. This is enabled by the -Rpass-analysis=regexp
275 /// flag.
276 std::shared_ptr<llvm::Regex> OptimizationRemarkAnalysisPattern;
277
278 /// Set of files defining the rules for the symbol rewriting.
279 std::vector<std::string> RewriteMapFiles;
280
281 /// Set of sanitizer checks that are non-fatal (i.e. execution should be
282 /// continued when possible).
283 SanitizerSet SanitizeRecover;
284
285 /// Set of sanitizer checks that trap rather than diagnose.
286 SanitizerSet SanitizeTrap;
287
288 /// List of backend command-line options for -fembed-bitcode.
289 std::vector<uint8_t> CmdArgs;
290
291 /// A list of all -fno-builtin-* function names (e.g., memset).
292 std::vector<std::string> NoBuiltinFuncs;
293
294 std::vector<std::string> Reciprocals;
295
296 /// The preferred width for auto-vectorization transforms. This is intended to
297 /// override default transforms based on the width of the architected vector
298 /// registers.
299 std::string PreferVectorWidth;
300
301 /// Set of XRay instrumentation kinds to emit.
302 XRayInstrSet XRayInstrumentationBundle;
303
304 std::vector<std::string> DefaultFunctionAttrs;
305
306 /// List of dynamic shared object files to be loaded as pass plugins.
307 std::vector<std::string> PassPlugins;
308
309public:
310 // Define accessors/mutators for code generation options of enumeration type.
311#define CODEGENOPT(Name, Bits, Default)
312#define ENUM_CODEGENOPT(Name, Type, Bits, Default) \
313 Type get##Name() const { return static_cast<Type>(Name); } \
314 void set##Name(Type Value) { Name = static_cast<unsigned>(Value); }
315#include "clang/Basic/CodeGenOptions.def"
316
317 CodeGenOptions();
318
319 /// Is this a libc/libm function that is no longer recognized as a
320 /// builtin because a -fno-builtin-* option has been specified?
321 bool isNoBuiltinFunc(const char *Name) const;
322
323 const std::vector<std::string> &getNoBuiltinFuncs() const {
324 return NoBuiltinFuncs;
325 }
326
327 /// Check if Clang profile instrumenation is on.
328 bool hasProfileClangInstr() const {
329 return getProfileInstr() == ProfileClangInstr;
330 }
331
332 /// Check if IR level profile instrumentation is on.
333 bool hasProfileIRInstr() const {
334 return getProfileInstr() == ProfileIRInstr;
335 }
336
337 /// Check if CS IR level profile instrumentation is on.
338 bool hasProfileCSIRInstr() const {
339 return getProfileInstr() == ProfileCSIRInstr;
340 }
341
342 /// Check if Clang profile use is on.
343 bool hasProfileClangUse() const {
344 return getProfileUse() == ProfileClangInstr;
345 }
346
347 /// Check if IR level profile use is on.
348 bool hasProfileIRUse() const {
349 return getProfileUse() == ProfileIRInstr ||
350 getProfileUse() == ProfileCSIRInstr;
351 }
352
353 /// Check if CSIR profile use is on.
354 bool hasProfileCSIRUse() const { return getProfileUse() == ProfileCSIRInstr; }
355};
