17namespace clang {

19inline raw_ostream &Indent(raw_ostream &Out, const unsigned int Space,
20 bool IsDot) {
21 for (unsigned int I = 0; I < Space * 2; ++I)
22 Out << (IsDot ? "&nbsp;" : " ");
23 return Out;
24}
25
26inline std::string JsonFormat(StringRef RawSR, bool AddQuotes) {
27 if (RawSR.empty())
28 return "null";
29
30 // Trim special characters.
31 std::string Str = RawSR.trim().str();
32 size_t Pos = 0;
33
34 // Escape double quotes.
35 while (true) {
36 Pos = Str.find('\"', Pos);
37 if (Pos == std::string::npos)
38 break;
39
40 // Prevent bad conversions.
41 size_t TempPos = (Pos != 0) ? Pos - 1 : 0;
42
43 // See whether the current double quote is escaped.
44 if (TempPos != Str.find("\\\"", TempPos)) {
45 Str.insert(Pos, "\\");
46 ++Pos; // As we insert the escape-character move plus one.
47 }
48
49 ++Pos;
50 }
51
52 // Remove new-lines.
53 Str.erase(std::remove(Str.begin(), Str.end(), '\n'), Str.end());
54
55 if (!AddQuotes)
56 return Str;
57
58 return '\"' + Str + '\"';
59}
