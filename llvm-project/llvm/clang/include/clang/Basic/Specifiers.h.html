112 /// The categorization of expression values, currently following the
113 /// C++11 scheme.
114 enum ExprValueKind {
115 /// An r-value expression (a pr-value in the C++11 taxonomy)
116 /// produces a temporary value.
117 VK_RValue,
118
119 /// An l-value expression is a reference to an object with
120 /// independent storage.
121 VK_LValue,
122
123 /// An x-value expression is a reference to an object with

125 /// efficiently cannibalized for its resources.

165 /// Describes the kind of template specialization that a
166 /// particular template specialization declaration represents.
167 enum TemplateSpecializationKind {
168 /// This template specialization was formed from a template-id but
169 /// has not yet been declared, defined, or instantiated.
170 TSK_Undeclared = 0,
171 /// This template specialization was implicitly instantiated from a
172 /// template. (C++ [temp.inst]).
173 TSK_ImplicitInstantiation,
174 /// This template specialization was declared or defined by an
175 /// explicit specialization (C++ [temp.expl.spec]) or partial
176 /// specialization (C++ [temp.class.spec]).
177 TSK_ExplicitSpecialization,
178 /// This template specialization was instantiated from a template
179 /// due to an explicit instantiation declaration request
180 /// (C++11 [temp.explicit]).
181 TSK_ExplicitInstantiationDeclaration,
182 /// This template specialization was instantiated from a template
183 /// due to an explicit instantiation definition request
184 /// (C++ [temp.explicit]).
185 TSK_ExplicitInstantiationDefinition
186 };
187
188 /// Determine whether this template specialization kind refers
189 /// to an instantiation of an entity (as opposed to a non-template or
190 /// an explicit specialization).
191 inline bool isTemplateInstantiation(TemplateSpecializationKind Kind) {
192 return Kind != TSK_Undeclared && Kind != TSK_ExplicitSpecialization;
193 }
194
195 /// True if this template specialization kind is an explicit
196 /// specialization, explicit instantiation declaration, or explicit
197 /// instantiation definition.
198 inline bool isTemplateExplicitInstantiationOrSpecialization(
199 TemplateSpecializationKind Kind) {
200 switch (Kind) {
201 case TSK_ExplicitSpecialization:
202 case TSK_ExplicitInstantiationDeclaration:
203 case TSK_ExplicitInstantiationDefinition:
204 return true;
205
206 case TSK_Undeclared:
207 case TSK_ImplicitInstantiation:
208 return false;
209 }
210 llvm_unreachable("bad template specialization kind");
211 }

281 /// Checks whether the given calling convention supports variadic
282 /// calls. Unprototyped calls also use the variadic call rules.
283 inline bool supportsVariadicCall(CallingConv CC) {
284 switch (CC) {
285 case CC_X86StdCall:
286 case CC_X86FastCall:
287 case CC_X86ThisCall:
288 case CC_X86RegCall:
289 case CC_X86Pascal:
290 case CC_X86VectorCall:
291 case CC_SpirFunction:
292 case CC_OpenCLKernel:
293 case CC_Swift:
294 return false;
295 default:
296 return true;
297 }
298 }

309 /// Describes the nullability of a particular type.
310 enum class NullabilityKind : uint8_t {
311 /// Values of this type can never be null.
312 NonNull = 0,
313 /// Values of this type can be null.
314 Nullable,
315 /// Whether values of this type can be null is (explicitly)
316 /// unspecified. This captures a (fairly rare) case where we
317 /// can't conclude anything about the nullability of the type even
318 /// though it has been considered.
319 Unspecified
320 };
321
322 /// Return true if \p L has a weaker nullability annotation than \p R. The
323 /// ordering is: Unspecified < Nullable < NonNull.
324 inline bool hasWeakerNullability(NullabilityKind L, NullabilityKind R) {
325 return uint8_t(L) > uint8_t(R);
326 }

332 /// Kinds of parameter ABI.
333 enum class ParameterABI {
334 /// This parameter uses ordinary ABI rules for its type.
335 Ordinary,
336
337 /// This parameter (which must have pointer type) is a Swift
338 /// indirect result parameter.
339 SwiftIndirectResult,
340
341 /// This parameter (which must have pointer-to-pointer type) uses
342 /// the special Swift error-result ABI treatment. There can be at
343 /// most one parameter on a given function that uses this treatment.
344 SwiftErrorResult,
345
346 /// This parameter (which must have pointer type) uses the special
347 /// Swift context-pointer ABI treatment. There can be at
348 /// most one parameter on a given function that uses this treatment.
349 SwiftContext
350 };
