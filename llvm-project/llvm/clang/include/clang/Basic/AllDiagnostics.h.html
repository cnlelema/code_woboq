29namespace clang {
30template <size_t SizeOfStr, typename FieldType>
31class StringSizerHelper {
32 static_assert(SizeOfStr <= FieldType(~0U), "Field too small!");
33public:
34 enum { Size = SizeOfStr };
35};
36} // end namespace clang
37
38#define STR_SIZE(str, fieldTy) clang::StringSizerHelper<sizeof(str)-1, \
39 fieldTy>::Size
