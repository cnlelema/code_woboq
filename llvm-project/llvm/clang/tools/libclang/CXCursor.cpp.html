97CXCursor cxcursor::MakeCXCursor(const Decl *D, CXTranslationUnit TU,
98 SourceRange RegionOfInterest,
99 bool FirstInDeclGroup) {
100 assert(D && TU && "Invalid arguments!");
101
102 CXCursorKind K = getCursorKindForDecl(D);
103
104 if (K == CXCursor_ObjCClassMethodDecl ||
105 K == CXCursor_ObjCInstanceMethodDecl) {
106 int SelectorIdIndex = -1;
107 // Check if cursor points to a selector id.
108 if (RegionOfInterest.isValid() &&
109 RegionOfInterest.getBegin() == RegionOfInterest.getEnd()) {
110 SmallVector<SourceLocation, 16> SelLocs;
111 cast<ObjCMethodDecl>(D)->getSelectorLocs(SelLocs);
112 SmallVectorImpl<SourceLocation>::iterator I =
113 llvm::find(SelLocs, RegionOfInterest.getBegin());
114 if (I != SelLocs.end())
115 SelectorIdIndex = I - SelLocs.begin();
116 }
117 CXCursor C = { K, SelectorIdIndex,
118 { D, (void*)(intptr_t) (FirstInDeclGroup ? 1 : 0), TU }};
119 return C;
120 }
121
122 CXCursor C = { K, 0, { D, (void*)(intptr_t) (FirstInDeclGroup ? 1 : 0), TU }};
123 return C;
124}

1095CXCursor cxcursor::getTypeRefCursor(CXCursor cursor) {
1096 if (cursor.kind != CXCursor_CallExpr)
1097 return cursor;
1098
1099 if (cursor.xdata == 0)
1100 return cursor;
1101
1102 const Expr *E = getCursorExpr(cursor);
1103 TypeSourceInfo *Type = nullptr;
1104 if (const CXXUnresolvedConstructExpr *
1105 UnCtor = dyn_cast<CXXUnresolvedConstructExpr>(E)) {
1106 Type = UnCtor->getTypeSourceInfo();
1107 } else if (const CXXTemporaryObjectExpr *Tmp =
1108 dyn_cast<CXXTemporaryObjectExpr>(E)){
1109 Type = Tmp->getTypeSourceInfo();
1110 }
1111
1112 if (!Type)
1113 return cursor;
1114
1115 CXTranslationUnit TU = getCursorTU(cursor);
1116 QualType Ty = Type->getType();
1117 TypeLoc TL = Type->getTypeLoc();
1118 SourceLocation Loc = TL.getBeginLoc();
1119
1120 if (const ElaboratedType *ElabT = Ty->getAs<ElaboratedType>()) {
1121 Ty = ElabT->getNamedType();
1122 ElaboratedTypeLoc ElabTL = TL.castAs<ElaboratedTypeLoc>();
1123 Loc = ElabTL.getNamedTypeLoc().getBeginLoc();
1124 }
1125
1126 if (const TypedefType *Typedef = Ty->getAs<TypedefType>())
1127 return MakeCursorTypeRef(Typedef->getDecl(), Loc, TU);
1128 if (const TagType *Tag = Ty->getAs<TagType>())
1129 return MakeCursorTypeRef(Tag->getDecl(), Loc, TU);
1130 if (const TemplateTypeParmType *TemplP = Ty->getAs<TemplateTypeParmType>())
1131 return MakeCursorTypeRef(TemplP->getDecl(), Loc, TU);
1132
1133 return cursor;
1134}

1237enum CXGetTemplateArgumentStatus {
1238 /** The operation completed successfully */
1239 CXGetTemplateArgumentStatus_Success = 0,
1240
1241 /** The specified cursor did not represent a FunctionDecl. */
1242 CXGetTemplateArgumentStatus_CursorNotFunctionDecl = -1,
1243
1244 /** The specified cursor was not castable to a FunctionDecl. */
1245 CXGetTemplateArgumentStatus_BadFunctionDeclCast = -2,
1246
1247 /** A NULL FunctionTemplateSpecializationInfo was retrieved. */

1254static int clang_Cursor_getTemplateArgument(
1255 CXCursor C, unsigned I, TemplateArgument *TA) {
1256 if (clang_getCursorKind(C) != CXCursor_FunctionDecl) {
1257 return CXGetTemplateArgumentStatus_CursorNotFunctionDecl;
1258 }
1259
1260 const FunctionDecl *FD = llvm::dyn_cast_or_null<clang::FunctionDecl>(
1261 getCursorDecl(C));
1262 if (!FD) {
1263 return CXGetTemplateArgumentStatus_BadFunctionDeclCast;
1264 }
1265
1266 const FunctionTemplateSpecializationInfo* SpecInfo =
1267 FD->getTemplateSpecializationInfo();
1268 if (!SpecInfo) {
1269 return CXGetTemplateArgumentStatus_NullTemplSpecInfo;
1270 }
1271
1272 if (I >= SpecInfo->TemplateArguments->size()) {
1273 return CXGetTemplateArgumentStatus_InvalidIndex;
1274 }
1275
1276 *TA = SpecInfo->TemplateArguments->get(I);
1277 return 0;
1278}

1280enum CXTemplateArgumentKind clang_Cursor_getTemplateArgumentKind(CXCursor C,
1281 unsigned I) {
1282 TemplateArgument TA;
1283 if (clang_Cursor_getTemplateArgument(C, I, &TA)) {
1284 return CXTemplateArgumentKind_Invalid;
1285 }
1286
1287 switch (TA.getKind()) {
1288 case TemplateArgument::Null: return CXTemplateArgumentKind_Null;
1289 case TemplateArgument::Type: return CXTemplateArgumentKind_Type;
1290 case TemplateArgument::Declaration:
1291 return CXTemplateArgumentKind_Declaration;
1292 case TemplateArgument::NullPtr: return CXTemplateArgumentKind_NullPtr;
1293 case TemplateArgument::Integral: return CXTemplateArgumentKind_Integral;
1294 case TemplateArgument::Template: return CXTemplateArgumentKind_Template;
1295 case TemplateArgument::TemplateExpansion:
1296 return CXTemplateArgumentKind_TemplateExpansion;
1297 case TemplateArgument::Expression: return CXTemplateArgumentKind_Expression;
1298 case TemplateArgument::Pack: return CXTemplateArgumentKind_Pack;
1299 }
1300
1301 return CXTemplateArgumentKind_Invalid;
1302}

1470void clang_getOverriddenCursors(CXCursor cursor,
1471 CXCursor **overridden,
1472 unsigned *num_overridden) {
1473 if (overridden)
1474 *overridden = nullptr;
1475 if (num_overridden)
1476 *num_overridden = 0;
1477
1478 CXTranslationUnit TU = cxcursor::getCursorTU(cursor);
1479
1480 if (!overridden || !num_overridden || !TU)
1481 return;
1482
1483 if (!clang_isDeclaration(cursor.kind))
1484 return;
1485
1486 OverridenCursorsPool &pool =
1487 *static_cast<OverridenCursorsPool*>(TU->OverridenCursorsPool);
1488
1489 OverridenCursorsPool::CursorVec *Vec = nullptr;
1490
1491 if (!pool.AvailableCursors.empty()) {
1492 Vec = pool.AvailableCursors.back();
1493 pool.AvailableCursors.pop_back();
1494 }
1495 else {
1496 Vec = new OverridenCursorsPool::CursorVec();
1497 pool.AllCursors.push_back(Vec);
1498 }
1499
1500 // Clear out the vector, but don't free the memory contents. This
1501 // reduces malloc() traffic.
1502 Vec->clear();
1503
1504 // Use the first entry to contain a back reference to the vector.
1505 // This is a complete hack.
1506 CXCursor backRefCursor = MakeCXCursorInvalid(CXCursor_InvalidFile, TU);
1507 backRefCursor.data[0] = Vec;
1508 assert(cxcursor::getCursorTU(backRefCursor) == TU);
1509 Vec->push_back(backRefCursor);
1510
1511 // Get the overridden cursors.
1512 cxcursor::getOverriddenCursors(cursor, *Vec);
1513
1514 // Did we get any overridden cursors? If not, return Vec to the pool
1515 // of available cursor vectors.
1516 if (Vec->size() == 1) {
1517 pool.AvailableCursors.push_back(Vec);
1518 return;
1519 }
1520
1521 // Now tell the caller about the overridden cursors.
1522 assert(Vec->size() > 1);
1523 *overridden = &((*Vec)[1]);
1524 *num_overridden = Vec->size() - 1;
1525}

1527void clang_disposeOverriddenCursors(CXCursor *overridden) {
1528 if (!overridden)
1529 return;
1530
1531 // Use pointer arithmetic to get back the first faux entry
1532 // which has a back-reference to the TU and the vector.
1533 --overridden;
1534 OverridenCursorsPool::CursorVec *Vec =
1535 static_cast<OverridenCursorsPool::CursorVec *>(
1536 const_cast<void *>(overridden->data[0]));
1537 CXTranslationUnit TU = getCursorTU(*overridden);
1538
1539 assert(Vec && TU);
1540
1541 OverridenCursorsPool &pool =
1542 *static_cast<OverridenCursorsPool*>(TU->OverridenCursorsPool);
1543
1544 pool.AvailableCursors.push_back(Vec);
1545}

1547int clang_Cursor_isDynamicCall(CXCursor C) {
1548 const Expr *E = nullptr;
1549 if (clang_isExpression(C.kind))
1550 E = getCursorExpr(C);
1551 if (!E)
1552 return 0;
1553
1554 if (const ObjCMessageExpr *MsgE = dyn_cast<ObjCMessageExpr>(E)) {
1555 if (MsgE->getReceiverKind() != ObjCMessageExpr::Instance)
1556 return false;
1557 if (auto *RecE = dyn_cast<ObjCMessageExpr>(
1558 MsgE->getInstanceReceiver()->IgnoreParenCasts())) {
1559 if (RecE->getMethodFamily() == OMF_alloc)
1560 return false;
1561 }
1562 return true;
1563 }
1564
1565 if (auto *PropRefE = dyn_cast<ObjCPropertyRefExpr>(E)) {
1566 return !PropRefE->isSuperReceiver();
1567 }
1568
1569 const MemberExpr *ME = nullptr;
1570 if (isa<MemberExpr>(E))
1571 ME = cast<MemberExpr>(E);
1572 else if (const CallExpr *CE = dyn_cast<CallExpr>(E))
1573 ME = dyn_cast_or_null<MemberExpr>(CE->getCallee());
1574
1575 if (ME) {
1576 if (const CXXMethodDecl *
1577 MD = dyn_cast_or_null<CXXMethodDecl>(ME->getMemberDecl()))
1578 return MD->isVirtual() &&
1579 ME->performsVirtualDispatch(
1580 cxcursor::getCursorContext(C).getLangOpts());
1581 }
1582
1583 return 0;
1584}

1586CXType clang_Cursor_getReceiverType(CXCursor C) {
1587 CXTranslationUnit TU = cxcursor::getCursorTU(C);
1588 const Expr *E = nullptr;
1589 if (clang_isExpression(C.kind))
1590 E = getCursorExpr(C);
1591
1592 if (const ObjCMessageExpr *MsgE = dyn_cast_or_null<ObjCMessageExpr>(E))
1593 return cxtype::MakeCXType(MsgE->getReceiverType(), TU);
1594
1595 if (auto *PropRefE = dyn_cast<ObjCPropertyRefExpr>(E)) {
1596 return cxtype::MakeCXType(
1597 PropRefE->getReceiverType(cxcursor::getCursorContext(C)), TU);
1598 }
1599
1600 const MemberExpr *ME = nullptr;
1601 if (isa<MemberExpr>(E))
1602 ME = cast<MemberExpr>(E);
1603 else if (const CallExpr *CE = dyn_cast<CallExpr>(E))
1604 ME = dyn_cast_or_null<MemberExpr>(CE->getCallee());
1605
1606 if (ME) {
1607 if (dyn_cast_or_null<CXXMethodDecl>(ME->getMemberDecl())) {
1608 auto receiverTy = ME->getBase()->IgnoreImpCasts()->getType();
1609 return cxtype::MakeCXType(receiverTy, TU);
1610 }
1611 }
1612
1613 return cxtype::MakeCXType(QualType(), TU);
