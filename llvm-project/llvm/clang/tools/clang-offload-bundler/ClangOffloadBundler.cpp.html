111/// Obtain the offload kind and real machine triple out of the target
112/// information specified by the user.
113static void getOffloadKindAndTriple(StringRef Target, StringRef &OffloadKind,
114 StringRef &Triple) {
115 auto KindTriplePair = Target.split('-');
116 OffloadKind = KindTriplePair.first;
117 Triple = KindTriplePair.second;
118}
119static StringRef getTriple(StringRef Target) {
120 StringRef OffloadKind;
121 StringRef Triple;
122 getOffloadKindAndTriple(Target, OffloadKind, Triple);
123 return Triple;
124}
125static bool hasHostKind(StringRef Target) {
126 StringRef OffloadKind;
127 StringRef Triple;
128 getOffloadKindAndTriple(Target, OffloadKind, Triple);
129 return OffloadKind == "host";

132/// Generic file handler interface.
133class FileHandler {
134public:
135 FileHandler() {}
136
137 virtual ~FileHandler() {}
138
139 /// Update the file handler with information from the header of the bundled
140 /// file
141 virtual void ReadHeader(MemoryBuffer &Input) = 0;
142
143 /// Read the marker of the next bundled to be read in the file. The triple of
144 /// the target associated with that bundle is returned. An empty string is
145 /// returned if there are no more bundles to be read.
146 virtual StringRef ReadBundleStart(MemoryBuffer &Input) = 0;
147
148 /// Read the marker that closes the current bundle.
149 virtual void ReadBundleEnd(MemoryBuffer &Input) = 0;
150
151 /// Read the current bundle and write the result into the stream \a OS.
152 virtual void ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) = 0;
153

155 /// gathered from \a Inputs.

157 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) = 0;
158
159 /// Write the marker that initiates a bundle for the triple \a TargetTriple to
160 /// \a OS.
161 virtual void WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) = 0;
162
163 /// Write the marker that closes a bundle for the triple \a TargetTriple to \a

165
166 virtual bool WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) = 0;
167
168 /// Write the bundle from \a Input into \a OS.
169 virtual void WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) = 0;

172/// Handler for binary files. The bundled file will have the following format
173/// (all integers are stored in little-endian format):
174///
175/// "OFFLOAD_BUNDLER_MAGIC_STR" (ASCII encoding of the string)
176///
177/// NumberOfOffloadBundles (8-byte integer)
178///
179/// OffsetOfBundle1 (8-byte integer)
180/// SizeOfBundle1 (8-byte integer)
181/// NumberOfBytesInTripleOfBundle1 (8-byte integer)
182/// TripleOfBundle1 (byte length defined before)
183///
184/// ...
185///
186/// OffsetOfBundleN (8-byte integer)
187/// SizeOfBundleN (8-byte integer)
188/// NumberOfBytesInTripleOfBundleN (8-byte integer)
189/// TripleOfBundleN (byte length defined before)
190///
191/// Bundle1
192/// ...
193/// BundleN
194
195/// Read 8-byte integers from a buffer in little-endian format.
196static uint64_t Read8byteIntegerFromBuffer(StringRef Buffer, size_t pos) {
197 uint64_t Res = 0;
198 const char *Data = Buffer.data();
199
200 for (unsigned i = 0; i < 8; ++i) {
201 Res <<= 8;
202 uint64_t Char = (uint64_t)Data[pos + 7 - i];
203 Res |= 0xffu & Char;
204 }
205 return Res;
206}
207
208/// Write 8-byte integers to a buffer in little-endian format.
209static void Write8byteIntegerToBuffer(raw_fd_ostream &OS, uint64_t Val) {
210 for (unsigned i = 0; i < 8; ++i) {
211 char Char = (char)(Val & 0xffu);
212 OS.write(&Char, 1);
213 Val >>= 8;
214 }
215}

217class BinaryFileHandler final : public FileHandler {
218 /// Information about the bundles extracted from the header.
219 struct BundleInfo final {
220 /// Size of the bundle.
221 uint64_t Size = 0u;
222 /// Offset at which the bundle starts in the bundled file.
223 uint64_t Offset = 0u;
224
225 BundleInfo() {}
226 BundleInfo(uint64_t Size, uint64_t Offset) : Size(Size), Offset(Offset) {}
227 };
228
229 /// Map between a triple and the corresponding bundle information.
230 StringMap<BundleInfo> BundlesInfo;
231
232 /// Iterator for the bundle information that is being read.
233 StringMap<BundleInfo>::iterator CurBundleInfo;
234
235public:
236 BinaryFileHandler() : FileHandler() {}
237
238 ~BinaryFileHandler() final {}
239
240 void ReadHeader(MemoryBuffer &Input) final {
241 StringRef FC = Input.getBuffer();
242
243 // Initialize the current bundle with the end of the container.
244 CurBundleInfo = BundlesInfo.end();
245
246 // Check if buffer is smaller than magic string.
247 size_t ReadChars = sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
248 if (ReadChars > FC.size())
249 return;
250
251 // Check if no magic was found.
252 StringRef Magic(FC.data(), sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1);
253 if (!Magic.equals(OFFLOAD_BUNDLER_MAGIC_STR))
254 return;
255
256 // Read number of bundles.
257 if (ReadChars + 8 > FC.size())
258 return;
259
260 uint64_t NumberOfBundles = Read8byteIntegerFromBuffer(FC, ReadChars);
261 ReadChars += 8;
262
263 // Read bundle offsets, sizes and triples.
264 for (uint64_t i = 0; i < NumberOfBundles; ++i) {
265
266 // Read offset.
267 if (ReadChars + 8 > FC.size())
268 return;
269
270 uint64_t Offset = Read8byteIntegerFromBuffer(FC, ReadChars);
271 ReadChars += 8;
272
273 // Read size.
274 if (ReadChars + 8 > FC.size())
275 return;
276
277 uint64_t Size = Read8byteIntegerFromBuffer(FC, ReadChars);
278 ReadChars += 8;
279
280 // Read triple size.
281 if (ReadChars + 8 > FC.size())
282 return;
283
284 uint64_t TripleSize = Read8byteIntegerFromBuffer(FC, ReadChars);
285 ReadChars += 8;
286
287 // Read triple.
288 if (ReadChars + TripleSize > FC.size())
289 return;
290
291 StringRef Triple(&FC.data()[ReadChars], TripleSize);
292 ReadChars += TripleSize;
293
294 // Check if the offset and size make sense.
295 if (!Offset || Offset + Size > FC.size())
296 return;
297
298 assert(BundlesInfo.find(Triple) == BundlesInfo.end() &&
299 "Triple is duplicated??");
300 BundlesInfo[Triple] = BundleInfo(Size, Offset);
301 }
302 // Set the iterator to where we will start to read.
303 CurBundleInfo = BundlesInfo.begin();
304 }
305
306 StringRef ReadBundleStart(MemoryBuffer &Input) final {
307 if (CurBundleInfo == BundlesInfo.end())
308 return StringRef();
309
310 return CurBundleInfo->first();
311 }
312
313 void ReadBundleEnd(MemoryBuffer &Input) final {
314 assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
315 ++CurBundleInfo;
316 }
317
318 void ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
319 assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
320 StringRef FC = Input.getBuffer();
321 OS.write(FC.data() + CurBundleInfo->second.Offset,
322 CurBundleInfo->second.Size);
323 }
324
325 void WriteHeader(raw_fd_ostream &OS,
326 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
327 // Compute size of the header.
328 uint64_t HeaderSize = 0;
329
330 HeaderSize += sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
331 HeaderSize += 8; // Number of Bundles
332
333 for (auto &T : TargetNames) {
334 HeaderSize += 3 * 8; // Bundle offset, Size of bundle and size of triple.
335 HeaderSize += T.size(); // The triple.
336 }
337
338 // Write to the buffer the header.
339 OS << OFFLOAD_BUNDLER_MAGIC_STR;
340
341 Write8byteIntegerToBuffer(OS, TargetNames.size());
342
343 unsigned Idx = 0;
344 for (auto &T : TargetNames) {
345 MemoryBuffer &MB = *Inputs[Idx++].get();
346 // Bundle offset.
347 Write8byteIntegerToBuffer(OS, HeaderSize);
348 // Size of the bundle (adds to the next bundle's offset)
349 Write8byteIntegerToBuffer(OS, MB.getBufferSize());
350 HeaderSize += MB.getBufferSize();
351 // Size of the triple
352 Write8byteIntegerToBuffer(OS, T.size());
353 // Triple
354 OS << T;
355 }
356 }
357
358 void WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {}
359
360 bool WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
361 return false;
362 }
363
364 void WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
365 OS.write(Input.getBufferStart(), Input.getBufferSize());
366 }
367};

369/// Handler for object files. The bundles are organized by sections with a
370/// designated name.
371///
372/// In order to bundle we create an IR file with the content of each section and
373/// use incremental linking to produce the resulting object. We also add section
374/// with a single byte to state the name of the component the main object file
375/// (the one we are bundling into) refers to.
376///
377/// To unbundle, we use just copy the contents of the designated section. If the
378/// requested bundle refer to the main object file, we just copy it with no
379/// changes.
380class ObjectFileHandler final : public FileHandler {
381
382 /// The object file we are currently dealing with.
383 std::unique_ptr<ObjectFile> Obj;
384
385 /// Return the input file contents.
386 StringRef getInputFileContents() const { return Obj->getData(); }
387
388 /// Return true if the provided section is an offload section and return the
389 /// triple by reference.
390 static bool IsOffloadSection(SectionRef CurSection,
391 StringRef &OffloadTriple) {
392 StringRef SectionName;
393 CurSection.getName(SectionName);
394
395 if (SectionName.empty())
396 return false;
397
398 // If it does not start with the reserved suffix, just skip this section.
399 if (!SectionName.startswith(OFFLOAD_BUNDLER_MAGIC_STR))
400 return false;
401
402 // Return the triple that is right after the reserved prefix.
403 OffloadTriple = SectionName.substr(sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1);
404 return true;
405 }
406
407 /// Total number of inputs.
408 unsigned NumberOfInputs = 0;
409
410 /// Total number of processed inputs, i.e, inputs that were already
411 /// read from the buffers.
412 unsigned NumberOfProcessedInputs = 0;
413
414 /// LLVM context used to create the auxiliary modules.
415 LLVMContext VMContext;
416
417 /// LLVM module used to create an object with all the bundle
418 /// components.
419 std::unique_ptr<Module> AuxModule;
420
421 /// The current triple we are working with.
422 StringRef CurrentTriple;
423
424 /// The name of the main input file.
425 StringRef MainInputFileName;
426
427 /// Iterator of the current and next section.
428 section_iterator CurrentSection;
429 section_iterator NextSection;
430
431public:
432 ObjectFileHandler(std::unique_ptr<ObjectFile> ObjIn)
433 : FileHandler(), Obj(std::move(ObjIn)),
434 CurrentSection(Obj->section_begin()),
435 NextSection(Obj->section_begin()) {}
436
437 ~ObjectFileHandler() final {}
438
439 void ReadHeader(MemoryBuffer &Input) final {}
440
441 StringRef ReadBundleStart(MemoryBuffer &Input) final {
442 while (NextSection != Obj->section_end()) {
443 CurrentSection = NextSection;
444 ++NextSection;
445
446 StringRef OffloadTriple;
447 // Check if the current section name starts with the reserved prefix. If
448 // so, return the triple.
449 if (IsOffloadSection(*CurrentSection, OffloadTriple))
450 return OffloadTriple;
451 }
452 return StringRef();
453 }
454
455 void ReadBundleEnd(MemoryBuffer &Input) final {}
456
457 void ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
458 // If the current section has size one, that means that the content we are
459 // interested in is the file itself. Otherwise it is the content of the
460 // section.
461 //
462 // TODO: Instead of copying the input file as is, deactivate the section
463 // that is no longer needed.
464
465 Expected<StringRef> Content = CurrentSection->getContents();
466 if (!Content) {
467 consumeError(Content.takeError());
468 return;
469 }
470
471 if (Content->size() < 2)
472 OS.write(Input.getBufferStart(), Input.getBufferSize());
473 else
474 OS.write(Content->data(), Content->size());
475 }
476
477 void WriteHeader(raw_fd_ostream &OS,
478 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
479 assert(HostInputIndex != ~0u && "Host input index not defined.");
480
481 // Record number of inputs.
482 NumberOfInputs = Inputs.size();
483
484 // Create an LLVM module to have the content we need to bundle.
485 auto *M = new Module("clang-offload-bundle", VMContext);
486 M->setTargetTriple(getTriple(TargetNames[HostInputIndex]));
487 AuxModule.reset(M);
488 }
489
490 void WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {
491 ++NumberOfProcessedInputs;
492
493 // Record the triple we are using, that will be used to name the section we
494 // will create.
495 CurrentTriple = TargetTriple;
496 }
497
498 bool WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
499 assert(NumberOfProcessedInputs <= NumberOfInputs &&
500 "Processing more inputs that actually exist!");
501 assert(HostInputIndex != ~0u && "Host input index not defined.");
502
503 // If this is not the last output, we don't have to do anything.
504 if (NumberOfProcessedInputs != NumberOfInputs)
505 return false;
506
507 // Create the bitcode file name to write the resulting code to. Keep it if
508 // save-temps is active.
509 SmallString<128> BitcodeFileName;
510 if (sys::fs::createTemporaryFile("clang-offload-bundler", "bc",
511 BitcodeFileName)) {
512 errs() << "error: unable to create temporary file.\n";
513 return true;
514 }
515
516 // Dump the contents of the temporary file if that was requested.
517 if (DumpTemporaryFiles) {
518 errs() << ";\n; Object file bundler IR file.\n;\n";
519 AuxModule.get()->print(errs(), nullptr,
520 /*ShouldPreserveUseListOrder=*/false,
521 /*IsForDebug=*/true);
522 errs() << '\n';
523 }
524
525 // Find clang in order to create the bundle binary.
526 StringRef Dir = sys::path::parent_path(BundlerExecutable);
527
528 auto ClangBinary = sys::findProgramByName("clang", Dir);
529 if (ClangBinary.getError()) {
530 // Remove bitcode file.
531 sys::fs::remove(BitcodeFileName);
532
533 errs() << "error: unable to find 'clang' in path.\n";
534 return true;
535 }
536
537 // Do the incremental linking. We write to the output file directly. So, we
538 // close it and use the name to pass down to clang.
539 OS.close();
540 SmallString<128> TargetName = getTriple(TargetNames[HostInputIndex]);
541 std::vector<StringRef> ClangArgs = {"clang",
542 "-r",
543 "-target",
544 TargetName.c_str(),
545 "-o",
546 OutputFileNames.front().c_str(),
547 InputFileNames[HostInputIndex].c_str(),
548 BitcodeFileName.c_str(),
549 "-nostdlib"};
550
551 // If the user asked for the commands to be printed out, we do that instead
552 // of executing it.
553 if (PrintExternalCommands) {
554 errs() << "\"" << ClangBinary.get() << "\"";
555 for (StringRef Arg : ClangArgs)
556 errs() << " \"" << Arg << "\"";
557 errs() << "\n";
558 } else {
559 // Write the bitcode contents to the temporary file.
560 {
561 std::error_code EC;
562 raw_fd_ostream BitcodeFile(BitcodeFileName, EC, sys::fs::F_None);
563 if (EC) {
564 errs() << "error: unable to open temporary file.\n";
565 return true;
566 }
567 WriteBitcodeToFile(*AuxModule, BitcodeFile);
568 }
569
570 bool Failed = sys::ExecuteAndWait(ClangBinary.get(), ClangArgs);
571
572 // Remove bitcode file.
573 sys::fs::remove(BitcodeFileName);
574
575 if (Failed) {
576 errs() << "error: incremental linking by external tool failed.\n";
577 return true;
578 }
579 }
580
581 return false;
582 }
583
584 void WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
585 Module *M = AuxModule.get();
586
587 // Create the new section name, it will consist of the reserved prefix
588 // concatenated with the triple.
589 std::string SectionName = OFFLOAD_BUNDLER_MAGIC_STR;
590 SectionName += CurrentTriple;
591
592 // Create the constant with the content of the section. For the input we are
593 // bundling into (the host input), this is just a place-holder, so a single
594 // byte is sufficient.
595 assert(HostInputIndex != ~0u && "Host input index undefined??");
596 Constant *Content;
597 if (NumberOfProcessedInputs == HostInputIndex + 1) {
598 uint8_t Byte[] = {0};
599 Content = ConstantDataArray::get(VMContext, Byte);
600 } else
601 Content = ConstantDataArray::get(
602 VMContext, ArrayRef<uint8_t>(reinterpret_cast<const uint8_t *>(
603 Input.getBufferStart()),
604 Input.getBufferSize()));
605
606 // Create the global in the desired section. We don't want these globals in
607 // the symbol table, so we mark them private.
608 auto *GV = new GlobalVariable(*M, Content->getType(), /*IsConstant=*/true,
609 GlobalVariable::PrivateLinkage, Content);
610 GV->setSection(SectionName);
611 }
612};

614/// Handler for text files. The bundled file will have the following format.
615///
616/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__START__ triple"
617/// Bundle 1
618/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__END__ triple"
619/// ...
620/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__START__ triple"
621/// Bundle N
622/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__END__ triple"
623class TextFileHandler final : public FileHandler {
624 /// String that begins a line comment.
625 StringRef Comment;
626
627 /// String that initiates a bundle.
628 std::string BundleStartString;
629
630 /// String that closes a bundle.
631 std::string BundleEndString;
632
633 /// Number of chars read from input.
634 size_t ReadChars = 0u;
635
636protected:
637 void ReadHeader(MemoryBuffer &Input) final {}
638
639 StringRef ReadBundleStart(MemoryBuffer &Input) final {
640 StringRef FC = Input.getBuffer();
641
642 // Find start of the bundle.
643 ReadChars = FC.find(BundleStartString, ReadChars);
644 if (ReadChars == FC.npos)
645 return StringRef();
646
647 // Get position of the triple.
648 size_t TripleStart = ReadChars = ReadChars + BundleStartString.size();
649
650 // Get position that closes the triple.
651 size_t TripleEnd = ReadChars = FC.find("\n", ReadChars);
652 if (TripleEnd == FC.npos)
653 return StringRef();
654
655 // Next time we read after the new line.
656 ++ReadChars;
657
658 return StringRef(&FC.data()[TripleStart], TripleEnd - TripleStart);
659 }
660
661 void ReadBundleEnd(MemoryBuffer &Input) final {
662 StringRef FC = Input.getBuffer();
663
664 // Read up to the next new line.
665 assert(FC[ReadChars] == '\n' && "The bundle should end with a new line.");
666
667 size_t TripleEnd = ReadChars = FC.find("\n", ReadChars + 1);
668 if (TripleEnd == FC.npos)
669 return;
670
671 // Next time we read after the new line.
672 ++ReadChars;
673 }
674
675 void ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
676 StringRef FC = Input.getBuffer();
677 size_t BundleStart = ReadChars;
678
679 // Find end of the bundle.
680 size_t BundleEnd = ReadChars = FC.find(BundleEndString, ReadChars);
681
682 StringRef Bundle(&FC.data()[BundleStart], BundleEnd - BundleStart);
683 OS << Bundle;
684 }
685
686 void WriteHeader(raw_fd_ostream &OS,
687 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {}
688
689 void WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {
690 OS << BundleStartString << TargetTriple << "\n";
691 }
692
693 bool WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
694 OS << BundleEndString << TargetTriple << "\n";
695 return false;
696 }
697
698 void WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
699 OS << Input.getBuffer();
700 }
701
702public:
703 TextFileHandler(StringRef Comment)
704 : FileHandler(), Comment(Comment), ReadChars(0) {
705 BundleStartString =
706 "\n" + Comment.str() + " " OFFLOAD_BUNDLER_MAGIC_STR "__START__ ";
707 BundleEndString =
708 "\n" + Comment.str() + " " OFFLOAD_BUNDLER_MAGIC_STR "__END__ ";
709 }
710};

712/// Return an appropriate object file handler. We use the specific object
713/// handler if we know how to deal with that format, otherwise we use a default
714/// binary file handler.
715static FileHandler *CreateObjectFileHandler(MemoryBuffer &FirstInput) {
716 // Check if the input file format is one that we know how to deal with.
717 Expected<std::unique_ptr<Binary>> BinaryOrErr = createBinary(FirstInput);
718
719 // Failed to open the input as a known binary. Use the default binary handler.
720 if (!BinaryOrErr) {
721 // We don't really care about the error (we just consume it), if we could
722 // not get a valid device binary object we use the default binary handler.
723 consumeError(BinaryOrErr.takeError());
724 return new BinaryFileHandler();
725 }
726
727 // We only support regular object files. If this is not an object file,
728 // default to the binary handler. The handler will be owned by the client of
729 // this function.
730 std::unique_ptr<ObjectFile> Obj(
731 dyn_cast<ObjectFile>(BinaryOrErr.get().release()));
732
733 if (!Obj)
734 return new BinaryFileHandler();
735
736 return new ObjectFileHandler(std::move(Obj));
737}
738
739/// Return an appropriate handler given the input files and options.
740static FileHandler *CreateFileHandler(MemoryBuffer &FirstInput) {
741 if (FilesType == "i")
742 return new TextFileHandler(/*Comment=*/"//");
743 if (FilesType == "ii")
744 return new TextFileHandler(/*Comment=*/"//");
745 if (FilesType == "ll")
746 return new TextFileHandler(/*Comment=*/";");
747 if (FilesType == "bc")
748 return new BinaryFileHandler();
749 if (FilesType == "s")
750 return new TextFileHandler(/*Comment=*/"#");
751 if (FilesType == "o")
752 return CreateObjectFileHandler(FirstInput);
753 if (FilesType == "gch")
754 return new BinaryFileHandler();
755 if (FilesType == "ast")
756 return new BinaryFileHandler();
757
758 errs() << "error: invalid file type specified.\n";
759 return nullptr;
760}

762/// Bundle the files. Return true if an error was found.
763static bool BundleFiles() {
764 std::error_code EC;
765
766 // Create output file.
767 raw_fd_ostream OutputFile(OutputFileNames.front(), EC, sys::fs::F_None);
768
769 if (EC) {
770 errs() << "error: Can't open file " << OutputFileNames.front() << ".\n";
771 return true;
772 }
773
774 // Open input files.
775 std::vector<std::unique_ptr<MemoryBuffer>> InputBuffers(
776 InputFileNames.size());
777
778 unsigned Idx = 0;
779 for (auto &I : InputFileNames) {
780 ErrorOr<std::unique_ptr<MemoryBuffer>> CodeOrErr =
781 MemoryBuffer::getFileOrSTDIN(I);
782 if (std::error_code EC = CodeOrErr.getError()) {
783 errs() << "error: Can't open file " << I << ": " << EC.message() << "\n";
784 return true;
785 }
786 InputBuffers[Idx++] = std::move(CodeOrErr.get());
787 }
788
789 // Get the file handler. We use the host buffer as reference.
790 assert(HostInputIndex != ~0u && "Host input index undefined??");
791 std::unique_ptr<FileHandler> FH;
792 FH.reset(CreateFileHandler(*InputBuffers[HostInputIndex].get()));
793
794 // Quit if we don't have a handler.
795 if (!FH.get())
796 return true;
797
798 // Write header.
799 FH.get()->WriteHeader(OutputFile, InputBuffers);
800
801 // Write all bundles along with the start/end markers. If an error was found
802 // writing the end of the bundle component, abort the bundle writing.
803 auto Input = InputBuffers.begin();
804 for (auto &Triple : TargetNames) {
805 FH.get()->WriteBundleStart(OutputFile, Triple);
806 FH.get()->WriteBundle(OutputFile, *Input->get());
807 if (FH.get()->WriteBundleEnd(OutputFile, Triple))
808 return true;
809 ++Input;
810 }
811 return false;
812}
813
814// Unbundle the files. Return true if an error was found.
815static bool UnbundleFiles() {
816 // Open Input file.
817 ErrorOr<std::unique_ptr<MemoryBuffer>> CodeOrErr =
818 MemoryBuffer::getFileOrSTDIN(InputFileNames.front());
819 if (std::error_code EC = CodeOrErr.getError()) {
820 errs() << "error: Can't open file " << InputFileNames.front() << ": "
821 << EC.message() << "\n";
822 return true;
823 }
824
825 MemoryBuffer &Input = *CodeOrErr.get();
826
827 // Select the right files handler.
828 std::unique_ptr<FileHandler> FH;
829 FH.reset(CreateFileHandler(Input));
830
831 // Quit if we don't have a handler.
832 if (!FH.get())
833 return true;
834
835 // Read the header of the bundled file.
836 FH.get()->ReadHeader(Input);
837
838 // Create a work list that consist of the map triple/output file.
839 StringMap<StringRef> Worklist;
840 auto Output = OutputFileNames.begin();
841 for (auto &Triple : TargetNames) {
842 Worklist[Triple] = *Output;
843 ++Output;
844 }
845
846 // Read all the bundles that are in the work list. If we find no bundles we
847 // assume the file is meant for the host target.
848 bool FoundHostBundle = false;
849 while (!Worklist.empty()) {
850 StringRef CurTriple = FH.get()->ReadBundleStart(Input);
851
852 // We don't have more bundles.
853 if (CurTriple.empty())
854 break;
855
856 auto Output = Worklist.find(CurTriple);
857 // The file may have more bundles for other targets, that we don't care
858 // about. Therefore, move on to the next triple
859 if (Output == Worklist.end()) {
860 continue;
861 }
862
863 // Check if the output file can be opened and copy the bundle to it.
864 std::error_code EC;
865 raw_fd_ostream OutputFile(Output->second, EC, sys::fs::F_None);
866 if (EC) {
867 errs() << "error: Can't open file " << Output->second << ": "
868 << EC.message() << "\n";
869 return true;
870 }
871 FH.get()->ReadBundle(OutputFile, Input);
872 FH.get()->ReadBundleEnd(Input);
873 Worklist.erase(Output);
874
875 // Record if we found the host bundle.
876 if (hasHostKind(CurTriple))
877 FoundHostBundle = true;
878 }
879
880 // If no bundles were found, assume the input file is the host bundle and
881 // create empty files for the remaining targets.
882 if (Worklist.size() == TargetNames.size()) {
883 for (auto &E : Worklist) {
884 std::error_code EC;
885 raw_fd_ostream OutputFile(E.second, EC, sys::fs::F_None);
886 if (EC) {
887 errs() << "error: Can't open file " << E.second << ": " << EC.message()
888 << "\n";
889 return true;
890 }
891
892 // If this entry has a host kind, copy the input file to the output file.
893 if (hasHostKind(E.first()))
894 OutputFile.write(Input.getBufferStart(), Input.getBufferSize());
895 }
896 return false;
897 }
898
899 // If we found elements, we emit an error if none of those were for the host.
900 if (!FoundHostBundle) {
901 errs() << "error: Can't find bundle for the host target\n";
902 return true;
903 }
904
905 // If we still have any elements in the worklist, create empty files for them.
906 for (auto &E : Worklist) {
907 std::error_code EC;
908 raw_fd_ostream OutputFile(E.second, EC, sys::fs::F_None);
909 if (EC) {
910 errs() << "error: Can't open file " << E.second << ": " << EC.message()
911 << "\n";
912 return true;
913 }
914 }
915
916 return false;
917}

923int main(int argc, const char **argv) {
924 sys::PrintStackTraceOnErrorSignal(argv[0]);
925
926 cl::HideUnrelatedOptions(ClangOffloadBundlerCategory);
927 cl::SetVersionPrinter(PrintVersion);
928 cl::ParseCommandLineOptions(
929 argc, argv,
930 "A tool to bundle several input files of the specified type <type> \n"
931 "referring to the same source file but different targets into a single \n"
932 "one. The resulting file can also be unbundled into different files by \n"
933 "this tool if -unbundle is provided.\n");
934
935 if (Help) {
936 cl::PrintHelpMessage();
937 return 0;
938 }
939
940 bool Error = false;
941 if (Unbundle) {
942 if (InputFileNames.size() != 1) {
943 Error = true;
944 errs() << "error: only one input file supported in unbundling mode.\n";
945 }
946 if (OutputFileNames.size() != TargetNames.size()) {
947 Error = true;
948 errs() << "error: number of output files and targets should match in "
949 "unbundling mode.\n";
950 }
951 } else {
952 if (OutputFileNames.size() != 1) {
953 Error = true;
954 errs() << "error: only one output file supported in bundling mode.\n";
955 }
956 if (InputFileNames.size() != TargetNames.size()) {
957 Error = true;
958 errs() << "error: number of input files and targets should match in "
959 "bundling mode.\n";
960 }
961 }
962
963 // Verify that the offload kinds and triples are known. We also check that we
964 // have exactly one host target.
965 unsigned Index = 0u;
966 unsigned HostTargetNum = 0u;
967 for (StringRef Target : TargetNames) {
968 StringRef Kind;
969 StringRef Triple;
970 getOffloadKindAndTriple(Target, Kind, Triple);
971
972 bool KindIsValid = !Kind.empty();
973 KindIsValid = KindIsValid && StringSwitch<bool>(Kind)
974 .Case("host", true)
975 .Case("openmp", true)
976 .Case("hip", true)
977 .Default(false);
978
979 bool TripleIsValid = !Triple.empty();
980 llvm::Triple T(Triple);
981 TripleIsValid &= T.getArch() != Triple::UnknownArch;
982
983 if (!KindIsValid || !TripleIsValid) {
984 Error = true;
985 errs() << "error: invalid target '" << Target << "'";
986
987 if (!KindIsValid)
988 errs() << ", unknown offloading kind '" << Kind << "'";
989 if (!TripleIsValid)
990 errs() << ", unknown target triple '" << Triple << "'";
991 errs() << ".\n";
992 }
993
994 if (KindIsValid && Kind == "host") {
995 ++HostTargetNum;
996 // Save the index of the input that refers to the host.
997 HostInputIndex = Index;
998 }
999
1000 ++Index;
1001 }
1002
1003 if (HostTargetNum != 1) {
1004 Error = true;
1005 errs() << "error: expecting exactly one host target but got "
1006 << HostTargetNum << ".\n";
1007 }
1008
1009 if (Error)
1010 return 1;
1011
1012 // Save the current executable directory as it will be useful to find other
1013 // tools.
1014 BundlerExecutable = sys::fs::getMainExecutable(argv[0], &BundlerExecutable);
1015
1016 return Unbundle ? UnbundleFiles() : BundleFiles();
1017}
