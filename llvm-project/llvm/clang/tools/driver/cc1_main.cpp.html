57static void LLVMErrorHandler(void *UserData, const std::string &Message,
58 bool GenCrashDiag) {
59 DiagnosticsEngine &Diags = *static_cast<DiagnosticsEngine*>(UserData);
60
61 Diags.Report(diag::err_fe_error_backend) << Message;
62
63 // Run the interrupt handlers to make sure any special cleanups get done, in
64 // particular that we remove files registered with RemoveFileOnSignal.
65 llvm::sys::RunInterruptHandlers();
66
67 // We cannot recover from llvm errors. When reporting a fatal error, exit
68 // with status 70 to generate crash diagnostics. For BSD systems this is
69 // defined as an internal software error. Otherwise, exit with status 1.
70 exit(GenCrashDiag ? 70 : 1);
71}

81static size_t getCurrentStackAllocation() {
82 // If we can't compute the current stack usage, allow for 512K of command
83 // line arguments and environment.
84 size_t Usage = 512 * 1024;
85 if (FILE *StatFile = fopen("/proc/self/stat", "r")) {
86 // We assume that the stack extends from its current address to the end of
87 // the environment space. In reality, there is another string literal (the
88 // program name) after the environment, but this is close enough (we only
89 // need to be within 100K or so).
90 unsigned long StackPtr, EnvEnd;
91 // Disable silly GCC -Wformat warning that complains about length
92 // modifiers on ignored format specifiers. We want to retain these
93 // for documentation purposes even though they have no effect.
94#if defined(__GNUC__) && !defined(__clang__)
95#pragma GCC diagnostic push
96#pragma GCC diagnostic ignored "-Wformat"
97#endif
98 if (fscanf(StatFile,
99 "%*d %*s %*c %*d %*d %*d %*d %*d %*u %*lu %*lu %*lu %*lu %*lu "
100 "%*lu %*ld %*ld %*ld %*ld %*ld %*ld %*llu %*lu %*ld %*lu %*lu "
101 "%*lu %*lu %lu %*lu %*lu %*lu %*lu %*lu %*llu %*lu %*lu %*d %*d "
102 "%*u %*u %*llu %*lu %*ld %*lu %*lu %*lu %*lu %*lu %*lu %lu %*d",
103 &StackPtr, &EnvEnd) == 2) {
104#if defined(__GNUC__) && !defined(__clang__)
105#pragma GCC diagnostic pop
106#endif
107 Usage = StackPtr < EnvEnd ? EnvEnd - StackPtr : StackPtr - EnvEnd;
108 }
109 fclose(StatFile);
110 }
111 return Usage;
112}

114#include <alloca.h>

116LLVM_ATTRIBUTE_NOINLINE
117static void ensureStackAddressSpace() {
118 // Linux kernels prior to 4.1 will sometimes locate the heap of a PIE binary
119 // relatively close to the stack (they are only guaranteed to be 128MiB
120 // apart). This results in crashes if we happen to heap-allocate more than
121 // 128MiB before we reach our stack high-water mark.
122 //
123 // To avoid these crashes, ensure that we have sufficient virtual memory
124 // pages allocated before we start running.
125 size_t Curr = getCurrentStackAllocation();
126 const int kTargetStack = DesiredStackSize - 256 * 1024;
127 if (Curr < kTargetStack) {
128 volatile char *volatile Alloc =
129 static_cast<volatile char *>(alloca(kTargetStack - Curr));
130 Alloc[0] = 0;
131 Alloc[kTargetStack - Curr - 1] = 0;
132 }
133}
134#else
135static void ensureStackAddressSpace() {}
136#endif
137
138/// Attempt to ensure that we have at least 8MiB of usable stack space.
139static void ensureSufficientStack() {
140 struct rlimit rlim;
141 if (getrlimit(RLIMIT_STACK, &rlim) != 0)
142 return;
143
144 // Increase the soft stack limit to our desired level, if necessary and
145 // possible.
146 if (rlim.rlim_cur != RLIM_INFINITY &&
147 rlim.rlim_cur < rlim_t(DesiredStackSize)) {
148 // Try to allocate sufficient stack.
149 if (rlim.rlim_max == RLIM_INFINITY ||
150 rlim.rlim_max >= rlim_t(DesiredStackSize))
151 rlim.rlim_cur = DesiredStackSize;
152 else if (rlim.rlim_cur == rlim.rlim_max)
153 return;
154 else
155 rlim.rlim_cur = rlim.rlim_max;
156
157 if (setrlimit(RLIMIT_STACK, &rlim) != 0 ||
158 rlim.rlim_cur != DesiredStackSize)
159 return;
160 }
161
162 // We should now have a stack of size at least DesiredStackSize. Ensure
163 // that we can actually use that much, if necessary.
164 ensureStackAddressSpace();
165}
166#else
167static void ensureSufficientStack() {}
168#endif

170int cc1_main(ArrayRef<const char *> Argv, const char *Argv0, void *MainAddr) {
171 ensureSufficientStack();
172
173 std::unique_ptr<CompilerInstance> Clang(new CompilerInstance());
174 IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
175
176 // Register the support for object-file-wrapped Clang modules.
177 auto PCHOps = Clang->getPCHContainerOperations();
178 PCHOps->registerWriter(llvm::make_unique<ObjectFilePCHContainerWriter>());
179 PCHOps->registerReader(llvm::make_unique<ObjectFilePCHContainerReader>());
180
181 // Initialize targets first, so that --version shows registered targets.
182 llvm::InitializeAllTargets();
183 llvm::InitializeAllTargetMCs();
184 llvm::InitializeAllAsmPrinters();
185 llvm::InitializeAllAsmParsers();
186
187#ifdef LINK_POLLY_INTO_TOOLS
188 llvm::PassRegistry &Registry = *llvm::PassRegistry::getPassRegistry();
189 polly::initializePollyPasses(Registry);
190#endif
191
192 // Buffer diagnostics from argument parsing so that we can output them using a
193 // well formed diagnostic object.
194 IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions();
195 TextDiagnosticBuffer *DiagsBuffer = new TextDiagnosticBuffer;
196 DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagsBuffer);
197 bool Success = CompilerInvocation::CreateFromArgs(
198 Clang->getInvocation(), Argv.begin(), Argv.end(), Diags);
199
200 if (Clang->getFrontendOpts().TimeTrace)
201 llvm::timeTraceProfilerInitialize();
202
203 // Infer the builtin include path if unspecified.
204 if (Clang->getHeaderSearchOpts().UseBuiltinIncludes &&
205 Clang->getHeaderSearchOpts().ResourceDir.empty())
206 Clang->getHeaderSearchOpts().ResourceDir =
207 CompilerInvocation::GetResourcesPath(Argv0, MainAddr);
208
209 // Create the actual diagnostics engine.
210 Clang->createDiagnostics();
211 if (!Clang->hasDiagnostics())
212 return 1;
213
214 // Set an error handler, so that any LLVM backend diagnostics go through our
215 // error handler.
216 llvm::install_fatal_error_handler(LLVMErrorHandler,
217 static_cast<void*>(&Clang->getDiagnostics()));
218
219 DiagsBuffer->FlushDiagnostics(Clang->getDiagnostics());
220 if (!Success)
221 return 1;
222
223 // Execute the frontend actions.
224 {
225 llvm::TimeTraceScope TimeScope("ExecuteCompiler", StringRef(""));
226 Success = ExecuteCompilerInvocation(Clang.get());
227 }
228
229 // Dump the CHERI CSetBounds stats now
230 if (llvm::cheri::ShouldCollectCSetBoundsStats) {
231 StringRef StatsOutput = llvm::cheri::CSetBoundsStatistics::outputFile();
232 if (StatsOutput.empty())
233 StatsOutput = Clang->getCodeGenOpts().CHERIStatsFile;
234 auto StatsFile = llvm::cheri::StatsOutputFile::open(
235 StatsOutput,
236 [&Clang](StringRef StatsFile, const std::error_code &EC) {
237 Clang->getDiagnostics().Report(
238 diag::warn_fe_unable_to_open_stats_file)
239 << StatsFile << EC.message();
240 },
241 [&Clang](StringRef StatsFile, const std::error_code &EC) {
242 Clang->getDiagnostics().Report(
243 diag::warn_fe_unable_to_lock_stats_file)
244 << StatsFile << EC.message();
245 });
246 if (StatsFile) {
247 StringRef MainFile = Clang->getCodeGenOpts().MainFileName;
248 if (MainFile.empty()) {
249 const SourceManager &SM = Clang->getSourceManager();
250 SourceLocation MainFileLoc =
251 SM.getLocForStartOfFile(SM.getMainFileID());
252 MainFile = SM.getFilename(MainFileLoc);
253 }
254 llvm::cheri::CSetBoundsStats->print(*StatsFile, MainFile);
255 }
256 }
257
258 // If any timers were active but haven't been destroyed yet, print their
259 // results now. This happens in -disable-free mode.
260 llvm::TimerGroup::printAll(llvm::errs());
261
262 if (llvm::timeTraceProfilerEnabled()) {
263 SmallString<128> Path(Clang->getFrontendOpts().OutputFile);
264 llvm::sys::path::replace_extension(Path, "json");
265 auto profilerOutput =
266 Clang->createOutputFile(Path.str(),
267 /*Binary=*/false,
268 /*RemoveFileOnSignal=*/false, "",
269 /*Extension=*/"json",
270 /*useTemporary=*/false);
271
272 llvm::timeTraceProfilerWrite(*profilerOutput);
273 // FIXME(ibiryukov): make profilerOutput flush in destructor instead.
274 profilerOutput->flush();
275 llvm::timeTraceProfilerCleanup();
276
277 llvm::errs() << "Time trace json-file dumped to " << Path.str() << "\n";
278 llvm::errs()
279 << "Use chrome://tracing or Speedscope App "
280 "(https://www.speedscope.app) for flamegraph visualization\n";
281 }
282
283 // Our error handler depends on the Diagnostics object, which we're
284 // potentially about to delete. Uninstall the handler now so that any
285 // later errors use the default handling behavior instead.
286 llvm::remove_fatal_error_handler();
287
288 // When running with -disable-free, don't do any destruction or shutdown.
289 if (Clang->getFrontendOpts().DisableFree) {
290 llvm::BuryPointer(std::move(Clang));
291 return !Success;
292 }
293
294 return !Success;
295}
