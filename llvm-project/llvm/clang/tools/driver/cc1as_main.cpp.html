65/// Helper class for representing a single invocation of the assembler.
66struct AssemblerInvocation {
67 /// @name Target Options
68 /// @{
69
70 /// The name of the target triple to assemble for.
71 std::string Triple;
72
73 /// If given, the name of the target CPU to determine which instructions
74 /// are legal.
75 std::string CPU;
76
77 /// The list of target specific features to enable or disable -- this should
78 /// be a list of strings starting with '+' or '-'.
79 std::vector<std::string> Features;
80
81 /// The list of symbol definitions.
82 std::vector<std::string> SymbolDefs;
83
84 /// @}
85 /// @name Language Options
86 /// @{
87
88 std::vector<std::string> IncludePaths;
89 unsigned NoInitialTextSection : 1;
90 unsigned SaveTemporaryLabels : 1;
91 unsigned GenDwarfForAssembly : 1;
92 unsigned RelaxELFRelocations : 1;
93 unsigned DwarfVersion;
94 std::string DwarfDebugFlags;
95 std::string DwarfDebugProducer;
96 std::string DebugCompilationDir;
97 std::map<const std::string, const std::string> DebugPrefixMap;
98 llvm::DebugCompressionType CompressDebugSections =
99 llvm::DebugCompressionType::None;
100 std::string MainFileName;
101 std::string SplitDwarfFile;
102
103 /// @}
104 /// @name Frontend Options
105 /// @{
106
107 std::string InputFile;
108 std::vector<std::string> LLVMArgs;
109 std::string OutputPath;
110 enum FileType {
111 FT_Asm, ///< Assembly (.s) output, transliterate mode.
112 FT_Null, ///< No output, for timing purposes.
113 FT_Obj ///< Object file output.
114 };
115 FileType OutputType;
116 unsigned ShowHelp : 1;
117 unsigned ShowVersion : 1;
118
119 /// @}
120 /// @name Transliterate Options
121 /// @{
122
123 unsigned OutputAsmVariant;
124 unsigned ShowEncoding : 1;
125 unsigned ShowInst : 1;
126
127 /// @}
128 /// @name Assembler Options
129 /// @{
130
131 unsigned RelaxAll : 1;
132 unsigned NoExecStack : 1;
133 unsigned FatalWarnings : 1;
134 unsigned IncrementalLinkerCompatible : 1;
135 unsigned EmbedBitcode : 1;
136
137 /// The name of the relocation model to use.
138 std::string RelocationModel;
139
140 /// The ABI targeted by the backend. Specified using -target-abi. Empty
141 /// otherwise.
142 std::string TargetABI;
143
144 /// @}
145
146public:
147 AssemblerInvocation() {
148 Triple = "";
149 NoInitialTextSection = 0;
150 InputFile = "-";
151 OutputPath = "-";
152 OutputType = FT_Asm;
153 OutputAsmVariant = 0;
154 ShowInst = 0;
155 ShowEncoding = 0;
156 RelaxAll = 0;
157 NoExecStack = 0;
158 FatalWarnings = 0;
159 IncrementalLinkerCompatible = 0;
160 DwarfVersion = 0;
161 EmbedBitcode = 0;
162 }
163
164 static bool CreateFromArgs(AssemblerInvocation &Res,
165 ArrayRef<const char *> Argv,
166 DiagnosticsEngine &Diags);
167};
168
169}
170
171bool AssemblerInvocation::CreateFromArgs(AssemblerInvocation &Opts,
172 ArrayRef<const char *> Argv,
173 DiagnosticsEngine &Diags) {
174 bool Success = true;
175
176 // Parse the arguments.
177 std::unique_ptr<OptTable> OptTbl(createDriverOptTable());
178
179 const unsigned IncludedFlagsBitmask = options::CC1AsOption;
180 unsigned MissingArgIndex, MissingArgCount;
181 InputArgList Args = OptTbl->ParseArgs(Argv, MissingArgIndex, MissingArgCount,
182 IncludedFlagsBitmask);
183
184 // Check for missing argument error.
185 if (MissingArgCount) {
186 Diags.Report(diag::err_drv_missing_argument)
187 << Args.getArgString(MissingArgIndex) << MissingArgCount;
188 Success = false;
189 }
190
191 // Issue errors on unknown arguments.
192 for (const Arg *A : Args.filtered(OPT_UNKNOWN)) {
193 auto ArgString = A->getAsString(Args);
194 std::string Nearest;
195 if (OptTbl->findNearest(ArgString, Nearest, IncludedFlagsBitmask) > 1)
196 Diags.Report(diag::err_drv_unknown_argument) << ArgString;
197 else
198 Diags.Report(diag::err_drv_unknown_argument_with_suggestion)
199 << ArgString << Nearest;
200 Success = false;
201 }
202
203 // Construct the invocation.
204
205 // Target Options
206 Opts.Triple = llvm::Triple::normalize(Args.getLastArgValue(OPT_triple));
207 Opts.CPU = Args.getLastArgValue(OPT_target_cpu);
208 Opts.Features = Args.getAllArgValues(OPT_target_feature);
209
210 // Use the default target triple if unspecified.
211 if (Opts.Triple.empty())
212 Opts.Triple = llvm::sys::getDefaultTargetTriple();
213
214 // Language Options
215 Opts.IncludePaths = Args.getAllArgValues(OPT_I);
216 Opts.NoInitialTextSection = Args.hasArg(OPT_n);
217 Opts.SaveTemporaryLabels = Args.hasArg(OPT_msave_temp_labels);
218 // Any DebugInfoKind implies GenDwarfForAssembly.
219 Opts.GenDwarfForAssembly = Args.hasArg(OPT_debug_info_kind_EQ);
220
221 if (const Arg *A = Args.getLastArg(OPT_compress_debug_sections,
222 OPT_compress_debug_sections_EQ)) {
223 if (A->getOption().getID() == OPT_compress_debug_sections) {
224 // TODO: be more clever about the compression type auto-detection
225 Opts.CompressDebugSections = llvm::DebugCompressionType::GNU;
226 } else {
227 Opts.CompressDebugSections =
228 llvm::StringSwitch<llvm::DebugCompressionType>(A->getValue())
229 .Case("none", llvm::DebugCompressionType::None)
230 .Case("zlib", llvm::DebugCompressionType::Z)
231 .Case("zlib-gnu", llvm::DebugCompressionType::GNU)
232 .Default(llvm::DebugCompressionType::None);
233 }
234 }
235
236 Opts.RelaxELFRelocations = Args.hasArg(OPT_mrelax_relocations);
237 Opts.DwarfVersion = getLastArgIntValue(Args, OPT_dwarf_version_EQ, 2, Diags);
238 Opts.DwarfDebugFlags = Args.getLastArgValue(OPT_dwarf_debug_flags);
239 Opts.DwarfDebugProducer = Args.getLastArgValue(OPT_dwarf_debug_producer);
240 Opts.DebugCompilationDir = Args.getLastArgValue(OPT_fdebug_compilation_dir);
241 Opts.MainFileName = Args.getLastArgValue(OPT_main_file_name);
242
243 for (const auto &Arg : Args.getAllArgValues(OPT_fdebug_prefix_map_EQ))
244 Opts.DebugPrefixMap.insert(StringRef(Arg).split('='));
245
246 // Frontend Options
247 if (Args.hasArg(OPT_INPUT)) {
248 bool First = true;
249 for (const Arg *A : Args.filtered(OPT_INPUT)) {
250 if (First) {
251 Opts.InputFile = A->getValue();
252 First = false;
253 } else {
254 Diags.Report(diag::err_drv_unknown_argument) << A->getAsString(Args);
255 Success = false;
256 }
257 }
258 }
259
260 Opts.LLVMArgs = Args.getAllArgValues(OPT_mllvm);
261 Opts.OutputPath = Args.getLastArgValue(OPT_o);
262 Opts.SplitDwarfFile = Args.getLastArgValue(OPT_split_dwarf_file);
263 if (Arg *A = Args.getLastArg(OPT_filetype)) {
264 StringRef Name = A->getValue();
265 unsigned OutputType = StringSwitch<unsigned>(Name)
266 .Case("asm", FT_Asm)
267 .Case("null", FT_Null)
268 .Case("obj", FT_Obj)
269 .Default(~0U);
270 if (OutputType == ~0U) {
271 Diags.Report(diag::err_drv_invalid_value) << A->getAsString(Args) << Name;
272 Success = false;
273 } else
274 Opts.OutputType = FileType(OutputType);
275 }
276 Opts.ShowHelp = Args.hasArg(OPT_help);
277 Opts.ShowVersion = Args.hasArg(OPT_version);
278
279 // Transliterate Options
280 Opts.OutputAsmVariant =
281 getLastArgIntValue(Args, OPT_output_asm_variant, 0, Diags);
282 Opts.ShowEncoding = Args.hasArg(OPT_show_encoding);
283 Opts.ShowInst = Args.hasArg(OPT_show_inst);
284
285 // Assemble Options
286 Opts.RelaxAll = Args.hasArg(OPT_mrelax_all);
287 Opts.NoExecStack = Args.hasArg(OPT_mno_exec_stack);
288 Opts.FatalWarnings = Args.hasArg(OPT_massembler_fatal_warnings);
289 Opts.RelocationModel = Args.getLastArgValue(OPT_mrelocation_model, "pic");
290 Opts.TargetABI = Args.getLastArgValue(OPT_target_abi);
291 Opts.IncrementalLinkerCompatible =
292 Args.hasArg(OPT_mincremental_linker_compatible);
293 Opts.SymbolDefs = Args.getAllArgValues(OPT_defsym);
294
295 // EmbedBitcode Option. If -fembed-bitcode is enabled, set the flag.
296 // EmbedBitcode behaves the same for all embed options for assembly files.
297 if (auto *A = Args.getLastArg(OPT_fembed_bitcode_EQ)) {
298 Opts.EmbedBitcode = llvm::StringSwitch<unsigned>(A->getValue())
299 .Case("all", 1)
300 .Case("bitcode", 1)
301 .Case("marker", 1)
302 .Default(0);
303 }
304
305 // Fixup the triple for CHERI ABI flags:
306 llvm::Triple TT(Opts.Triple);
307 if (TT.isMIPS()) {
308 if (Opts.TargetABI != "purecap" && TT.getEnvironment() == llvm::Triple::CheriPurecap) {
309 // Can't use -mabi=64 with -purecap triple
310 if (Opts.TargetABI.empty())
311 Opts.TargetABI = "purecap";
312 else
313 Diags.Report(diag::err_drv_abi_incompatible_with_triple) << Opts.TargetABI << TT.str();
314 } else if (Opts.TargetABI == "purecap") {
315 if (TT.getEnvironment() == llvm::Triple::UnknownEnvironment) {
316 TT.setEnvironment(llvm::Triple::CheriPurecap);
317 Opts.Triple = llvm::Triple::normalize(TT.str());
318 } else if (TT.getEnvironment() != llvm::Triple::CheriPurecap) {
319 // e.g. explicit gnuabin32 triple with -mabi=purecap
320 Diags.Report(diag::err_drv_abi_incompatible_with_triple) << Opts.TargetABI << TT.str();
321 }
322 }
323 }
324
325 return Success;
326}

328static std::unique_ptr<raw_fd_ostream>
329getOutputStream(StringRef Path, DiagnosticsEngine &Diags, bool Binary) {
330 // Make sure that the Out file gets unlinked from the disk if we get a
331 // SIGINT.
332 if (Path != "-")
333 sys::RemoveFileOnSignal(Path);
334
335 std::error_code EC;
336 auto Out = llvm::make_unique<raw_fd_ostream>(
337 Path, EC, (Binary ? sys::fs::F_None : sys::fs::F_Text));
338 if (EC) {
339 Diags.Report(diag::err_fe_unable_to_open_output) << Path << EC.message();
340 return nullptr;
341 }
342
343 return Out;
344}
345
346static bool ExecuteAssembler(AssemblerInvocation &Opts,
347 DiagnosticsEngine &Diags) {
348 // Get the target specific parser.
349 std::string Error;
350 const Target *TheTarget = TargetRegistry::lookupTarget(Opts.Triple, Error);
351 if (!TheTarget)
352 return Diags.Report(diag::err_target_unknown_triple) << Opts.Triple;
353
354 ErrorOr<std::unique_ptr<MemoryBuffer>> Buffer =
355 MemoryBuffer::getFileOrSTDIN(Opts.InputFile);
356
357 if (std::error_code EC = Buffer.getError()) {
358 Error = EC.message();
359 return Diags.Report(diag::err_fe_error_reading) << Opts.InputFile;
360 }
361
362 SourceMgr SrcMgr;
363
364 // Tell SrcMgr about this buffer, which is what the parser will pick up.
365 unsigned BufferIndex = SrcMgr.AddNewSourceBuffer(std::move(*Buffer), SMLoc());
366
367 // Record the location of the include directories so that the lexer can find
368 // it later.
369 SrcMgr.setIncludeDirs(Opts.IncludePaths);
370
371 std::unique_ptr<MCRegisterInfo> MRI(TheTarget->createMCRegInfo(Opts.Triple));
372 assert(MRI && "Unable to create target register info!");
373
374 std::unique_ptr<MCAsmInfo> MAI(TheTarget->createMCAsmInfo(*MRI, Opts.Triple));
375 assert(MAI && "Unable to create target asm info!");
376
377 // Ensure MCAsmInfo initialization occurs before any use, otherwise sections
378 // may be created with a combination of default and explicit settings.
379 MAI->setCompressDebugSections(Opts.CompressDebugSections);
380
381 MAI->setRelaxELFRelocations(Opts.RelaxELFRelocations);
382
383 bool IsBinary = Opts.OutputType == AssemblerInvocation::FT_Obj;
384 if (Opts.OutputPath.empty())
385 Opts.OutputPath = "-";
386 std::unique_ptr<raw_fd_ostream> FDOS =
387 getOutputStream(Opts.OutputPath, Diags, IsBinary);
388 if (!FDOS)
389 return true;
390 std::unique_ptr<raw_fd_ostream> DwoOS;
391 if (!Opts.SplitDwarfFile.empty())
392 DwoOS = getOutputStream(Opts.SplitDwarfFile, Diags, IsBinary);
393
394 // FIXME: This is not pretty. MCContext has a ptr to MCObjectFileInfo and
395 // MCObjectFileInfo needs a MCContext reference in order to initialize itself.
396 std::unique_ptr<MCObjectFileInfo> MOFI(new MCObjectFileInfo());
397
398 MCContext Ctx(MAI.get(), MRI.get(), MOFI.get(), &SrcMgr);
399
400 bool PIC = false;
401 if (Opts.RelocationModel == "static") {
402 PIC = false;
403 } else if (Opts.RelocationModel == "pic") {
404 PIC = true;
405 } else {
406 assert(Opts.RelocationModel == "dynamic-no-pic" &&
407 "Invalid PIC model!");
408 PIC = false;
409 }
410
411 MOFI->InitMCObjectFileInfo(Triple(Opts.Triple), PIC, Ctx);
412 if (Opts.SaveTemporaryLabels)
413 Ctx.setAllowTemporaryLabels(false);
414 if (Opts.GenDwarfForAssembly)
415 Ctx.setGenDwarfForAssembly(true);
416 if (!Opts.DwarfDebugFlags.empty())
417 Ctx.setDwarfDebugFlags(StringRef(Opts.DwarfDebugFlags));
418 if (!Opts.DwarfDebugProducer.empty())
419 Ctx.setDwarfDebugProducer(StringRef(Opts.DwarfDebugProducer));
420 if (!Opts.DebugCompilationDir.empty())
421 Ctx.setCompilationDir(Opts.DebugCompilationDir);
422 else {
423 // If no compilation dir is set, try to use the current directory.
424 SmallString<128> CWD;
425 if (!sys::fs::current_path(CWD))
426 Ctx.setCompilationDir(CWD);
427 }
428 if (!Opts.DebugPrefixMap.empty())
429 for (const auto &KV : Opts.DebugPrefixMap)
430 Ctx.addDebugPrefixMapEntry(KV.first, KV.second);
431 if (!Opts.MainFileName.empty())
432 Ctx.setMainFileName(StringRef(Opts.MainFileName));
433 Ctx.setDwarfVersion(Opts.DwarfVersion);
434 if (Opts.GenDwarfForAssembly)
435 Ctx.setGenDwarfRootFile(Opts.InputFile,
436 SrcMgr.getMemoryBuffer(BufferIndex)->getBuffer());
437
438 // Build up the feature string from the target feature list.
439 std::string FS;
440 if (!Opts.Features.empty()) {
441 FS = Opts.Features[0];
442 for (unsigned i = 1, e = Opts.Features.size(); i != e; ++i)
443 FS += "," + Opts.Features[i];
444 }
445
446 std::unique_ptr<MCStreamer> Str;
447
448 std::unique_ptr<MCInstrInfo> MCII(TheTarget->createMCInstrInfo());
449 std::unique_ptr<MCSubtargetInfo> STI(
450 TheTarget->createMCSubtargetInfo(Opts.Triple, Opts.CPU, FS));
451
452 raw_pwrite_stream *Out = FDOS.get();
453 std::unique_ptr<buffer_ostream> BOS;
454
455 MCTargetOptions MCOptions;
456 MCOptions.ABIName = Opts.TargetABI;
457
458 // FIXME: There is a bit of code duplication with addPassesToEmitFile.
459 if (Opts.OutputType == AssemblerInvocation::FT_Asm) {
460 MCInstPrinter *IP = TheTarget->createMCInstPrinter(
461 llvm::Triple(Opts.Triple), Opts.OutputAsmVariant, *MAI, *MCII, *MRI);
462
463 std::unique_ptr<MCCodeEmitter> CE;
464 if (Opts.ShowEncoding)
465 CE.reset(TheTarget->createMCCodeEmitter(*MCII, *MRI, Ctx));
466 std::unique_ptr<MCAsmBackend> MAB(
467 TheTarget->createMCAsmBackend(*STI, *MRI, MCOptions));
468
469 auto FOut = llvm::make_unique<formatted_raw_ostream>(*Out);
470 Str.reset(TheTarget->createAsmStreamer(
471 Ctx, std::move(FOut), /*asmverbose*/ true,
472 /*useDwarfDirectory*/ true, IP, std::move(CE), std::move(MAB),
473 Opts.ShowInst));
474 } else if (Opts.OutputType == AssemblerInvocation::FT_Null) {
475 Str.reset(createNullStreamer(Ctx));
476 } else {
477 assert(Opts.OutputType == AssemblerInvocation::FT_Obj &&
478 "Invalid file type!");
479 if (!FDOS->supportsSeeking()) {
480 BOS = make_unique<buffer_ostream>(*FDOS);
481 Out = BOS.get();
482 }
483
484 std::unique_ptr<MCCodeEmitter> CE(
485 TheTarget->createMCCodeEmitter(*MCII, *MRI, Ctx));
486 std::unique_ptr<MCAsmBackend> MAB(
487 TheTarget->createMCAsmBackend(*STI, *MRI, MCOptions));
488 std::unique_ptr<MCObjectWriter> OW =
489 DwoOS ? MAB->createDwoObjectWriter(*Out, *DwoOS)
490 : MAB->createObjectWriter(*Out);
491
492 Triple T(Opts.Triple);
493 Str.reset(TheTarget->createMCObjectStreamer(
494 T, Ctx, std::move(MAB), std::move(OW), std::move(CE), *STI,
495 Opts.RelaxAll, Opts.IncrementalLinkerCompatible,
496 /*DWARFMustBeAtTheEnd*/ true));
497 Str.get()->InitSections(Opts.NoExecStack);
498 }
499
500 // When -fembed-bitcode is passed to clang_as, a 1-byte marker
501 // is emitted in __LLVM,__asm section if the object file is MachO format.
502 if (Opts.EmbedBitcode && Ctx.getObjectFileInfo()->getObjectFileType() ==
503 MCObjectFileInfo::IsMachO) {
504 MCSection *AsmLabel = Ctx.getMachOSection(
505 "__LLVM", "__asm", MachO::S_REGULAR, 4, SectionKind::getReadOnly());
506 Str.get()->SwitchSection(AsmLabel);
507 Str.get()->EmitZeros(1);
508 }
509
510 // Assembly to object compilation should leverage assembly info.
511 Str->setUseAssemblerInfoForParsing(true);
512
513 bool Failed = false;
514
515 std::unique_ptr<MCAsmParser> Parser(
516 createMCAsmParser(SrcMgr, Ctx, *Str.get(), *MAI));
517
518 // FIXME: init MCTargetOptions from sanitizer flags here.
519 std::unique_ptr<MCTargetAsmParser> TAP(
520 TheTarget->createMCAsmParser(*STI, *Parser, *MCII, MCOptions));
521 if (!TAP)
522 Failed = Diags.Report(diag::err_target_unknown_triple) << Opts.Triple;
523
524 // Set values for symbols, if any.
525 for (auto &S : Opts.SymbolDefs) {
526 auto Pair = StringRef(S).split('=');
527 auto Sym = Pair.first;
528 auto Val = Pair.second;
529 int64_t Value;
530 // We have already error checked this in the driver.
531 Val.getAsInteger(0, Value);
532 Ctx.setSymbolValue(Parser->getStreamer(), Sym, Value);
533 }
534
535 if (!Failed) {
536 Parser->setTargetParser(*TAP.get());
537 Failed = Parser->Run(Opts.NoInitialTextSection);
538 }
539
540 // Close Streamer first.
541 // It might have a reference to the output stream.
542 Str.reset();
543 // Close the output stream early.
544 BOS.reset();
545 FDOS.reset();
546
547 // Delete output file if there were errors.
548 if (Failed) {
549 if (Opts.OutputPath != "-")
550 sys::fs::remove(Opts.OutputPath);
551 if (!Opts.SplitDwarfFile.empty() && Opts.SplitDwarfFile != "-")
552 sys::fs::remove(Opts.SplitDwarfFile);
553 }
554
555 return Failed;
556}

558static void LLVMErrorHandler(void *UserData, const std::string &Message,
559 bool GenCrashDiag) {
560 DiagnosticsEngine &Diags = *static_cast<DiagnosticsEngine*>(UserData);
561
562 Diags.Report(diag::err_fe_error_backend) << Message;
563
564 // We cannot recover from llvm errors.
565 exit(1);
566}

568int cc1as_main(ArrayRef<const char *> Argv, const char *Argv0, void *MainAddr) {
569 // Initialize targets and assembly printers/parsers.
570 InitializeAllTargetInfos();
571 InitializeAllTargetMCs();
572 InitializeAllAsmParsers();
573
574 // Construct our diagnostic client.
575 IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions();
576 TextDiagnosticPrinter *DiagClient
577 = new TextDiagnosticPrinter(errs(), &*DiagOpts);
578 DiagClient->setPrefix("clang -cc1as");
579 IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
580 DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
581
582 // Set an error handler, so that any LLVM backend diagnostics go through our
583 // error handler.
584 ScopedFatalErrorHandler FatalErrorHandler
585 (LLVMErrorHandler, static_cast<void*>(&Diags));
586
587 // Parse the arguments.
588 AssemblerInvocation Asm;
589 if (!AssemblerInvocation::CreateFromArgs(Asm, Argv, Diags))
590 return 1;
591
592 if (Asm.ShowHelp) {
593 std::unique_ptr<OptTable> Opts(driver::createDriverOptTable());
594 Opts->PrintHelp(llvm::outs(), "clang -cc1as [options] file...",
595 "Clang Integrated Assembler",
596 /*Include=*/driver::options::CC1AsOption, /*Exclude=*/0,
597 /*ShowAllAliases=*/false);
598 return 0;
599 }
600
601 // Honor -version.
602 //
603 // FIXME: Use a better -version message?
604 if (Asm.ShowVersion) {
605 llvm::cl::PrintVersionMessage();
606 return 0;
607 }
608
609 // Honor -mllvm.
610 //
611 // FIXME: Remove this, one day.
612 if (!Asm.LLVMArgs.empty()) {
613 unsigned NumArgs = Asm.LLVMArgs.size();
614 auto Args = llvm::make_unique<const char*[]>(NumArgs + 2);
615 Args[0] = "clang (LLVM option parsing)";
616 for (unsigned i = 0; i != NumArgs; ++i)
617 Args[i + 1] = Asm.LLVMArgs[i].c_str();
618 Args[NumArgs + 1] = nullptr;
619 llvm::cl::ParseCommandLineOptions(NumArgs + 1, Args.get());
620 }
621
622 // Execute the invocation, unless there were parsing errors.
623 bool Failed = Diags.hasErrorOccurred() || ExecuteAssembler(Asm, Diags);
624
625 // If any timers were active but haven't been destroyed yet, print their
626 // results now.
627 TimerGroup::printAll(errs());
628
