24void createVirtualFileIfNeeded(ASTUnit *ToAST, StringRef FileName,
25 std::unique_ptr<llvm::MemoryBuffer> &&Buffer) {
26 assert(ToAST);
27 ASTContext &ToCtx = ToAST->getASTContext();
28 auto *OFS = static_cast<llvm::vfs::OverlayFileSystem *>(
29 &ToCtx.getSourceManager().getFileManager().getVirtualFileSystem());
30 auto *MFS = static_cast<llvm::vfs::InMemoryFileSystem *>(
31 OFS->overlays_begin()->get());
32 MFS->addFile(FileName, 0, std::move(Buffer));
33}
34
35void createVirtualFileIfNeeded(ASTUnit *ToAST, StringRef FileName,
36 StringRef Code) {
37 return createVirtualFileIfNeeded(ToAST, FileName,
38 llvm::MemoryBuffer::getMemBuffer(Code));
39}

41ASTImporterTestBase::TU::TU(StringRef Code, StringRef FileName, ArgVector Args,
42 ImporterConstructor C)
43 : Code(Code), FileName(FileName),
44 Unit(tooling::buildASTFromCodeWithArgs(this->Code, Args, this->FileName)),
45 TUDecl(Unit->getASTContext().getTranslationUnitDecl()), Creator(C) {
46 Unit->enableSourceFileDiagnostics();
47
48 // If the test doesn't need a specific ASTImporter, we just create a
49 // normal ASTImporter with it.
50 if (!Creator)
51 Creator = [](ASTContext &ToContext, FileManager &ToFileManager,
52 ASTContext &FromContext, FileManager &FromFileManager,
53 bool MinimalImport, ASTImporterLookupTable *LookupTable) {
54 return new ASTImporter(ToContext, ToFileManager, FromContext,
55 FromFileManager, MinimalImport, LookupTable);
56 };
57}
58
59ASTImporterTestBase::TU::~TU() {}

61void ASTImporterTestBase::TU::lazyInitImporter(
62 ASTImporterLookupTable &LookupTable, ASTUnit *ToAST) {
63 assert(ToAST);
64 if (!Importer)
65 Importer.reset(Creator(ToAST->getASTContext(), ToAST->getFileManager(),
66 Unit->getASTContext(), Unit->getFileManager(), false,
67 &LookupTable));
68 assert(&ToAST->getASTContext() == &Importer->getToContext());
69 createVirtualFileIfNeeded(ToAST, FileName, Code);
70}

72Decl *ASTImporterTestBase::TU::import(ASTImporterLookupTable &LookupTable,
73 ASTUnit *ToAST, Decl *FromDecl) {
74 lazyInitImporter(LookupTable, ToAST);
75 if (auto ImportedOrErr = Importer->Import(FromDecl))
76 return *ImportedOrErr;
77 else {
78 llvm::consumeError(ImportedOrErr.takeError());
79 return nullptr;
80 }
81}

83QualType ASTImporterTestBase::TU::import(ASTImporterLookupTable &LookupTable,
84 ASTUnit *ToAST, QualType FromType) {
85 lazyInitImporter(LookupTable, ToAST);
86 if (auto ImportedOrErr = Importer->Import(FromType))
87 return *ImportedOrErr;
88 else {
89 llvm::consumeError(ImportedOrErr.takeError());
90 return QualType{};
91 }

94void ASTImporterTestBase::lazyInitLookupTable(TranslationUnitDecl *ToTU) {
95 assert(ToTU);
96 if (!LookupTablePtr)
97 LookupTablePtr = llvm::make_unique<ASTImporterLookupTable>(*ToTU);

100void ASTImporterTestBase::lazyInitToAST(Language ToLang, StringRef ToSrcCode,
101 StringRef FileName) {
102 if (ToAST)
103 return;
104 ArgVector ToArgs = getArgVectorForLanguage(ToLang);
105 // Source code must be a valid live buffer through the tests lifetime.
106 ToCode = ToSrcCode;
107 // Build the AST from an empty file.
108 ToAST = tooling::buildASTFromCodeWithArgs(ToCode, ToArgs, FileName);
109 ToAST->enableSourceFileDiagnostics();
110 lazyInitLookupTable(ToAST->getASTContext().getTranslationUnitDecl());
111}

113ASTImporterTestBase::TU *ASTImporterTestBase::findFromTU(Decl *From) {
114 // Create a virtual file in the To Ctx which corresponds to the file from
115 // which we want to import the `From` Decl. Without this source locations
116 // will be invalid in the ToCtx.
117 auto It = llvm::find_if(FromTUs, [From](const TU &E) {
118 return E.TUDecl == From->getTranslationUnitDecl();
119 });
120 assert(It != FromTUs.end());
121 return &*It;
122}

124std::tuple<Decl *, Decl *>
125ASTImporterTestBase::getImportedDecl(StringRef FromSrcCode, Language FromLang,
126 StringRef ToSrcCode, Language ToLang,
127 StringRef Identifier) {
128 ArgVector FromArgs = getArgVectorForLanguage(FromLang),
129 ToArgs = getArgVectorForLanguage(ToLang);
130
131 FromTUs.emplace_back(FromSrcCode, InputFileName, FromArgs, Creator);
132 TU &FromTU = FromTUs.back();
133
134 assert(!ToAST);
135 lazyInitToAST(ToLang, ToSrcCode, OutputFileName);
136
137 ASTContext &FromCtx = FromTU.Unit->getASTContext();
138
139 IdentifierInfo *ImportedII = &FromCtx.Idents.get(Identifier);
140 assert(ImportedII && "Declaration with the given identifier "
141 "should be specified in test!");
142 DeclarationName ImportDeclName(ImportedII);
143 SmallVector<NamedDecl *, 1> FoundDecls;
144 FromCtx.getTranslationUnitDecl()->localUncachedLookup(ImportDeclName,
145 FoundDecls);
146
147 assert(FoundDecls.size() == 1);
148
149 Decl *Imported =
150 FromTU.import(*LookupTablePtr, ToAST.get(), FoundDecls.front());
151
152 assert(Imported);
153 return std::make_tuple(*FoundDecls.begin(), Imported);
154}

156TranslationUnitDecl *ASTImporterTestBase::getTuDecl(StringRef SrcCode,
157 Language Lang,
158 StringRef FileName) {
159 assert(llvm::find_if(FromTUs, [FileName](const TU &E) {
160 return E.FileName == FileName;
161 }) == FromTUs.end());
162
163 ArgVector Args = getArgVectorForLanguage(Lang);
164 FromTUs.emplace_back(SrcCode, FileName, Args);
165 TU &Tu = FromTUs.back();
166
167 return Tu.TUDecl;
168}
169
170TranslationUnitDecl *ASTImporterTestBase::getToTuDecl(StringRef ToSrcCode,
171 Language ToLang) {
172 ArgVector ToArgs = getArgVectorForLanguage(ToLang);
173 assert(!ToAST);
174 lazyInitToAST(ToLang, ToSrcCode, OutputFileName);
175 return ToAST->getASTContext().getTranslationUnitDecl();
176}
177
178Decl *ASTImporterTestBase::Import(Decl *From, Language ToLang) {
179 lazyInitToAST(ToLang, "", OutputFileName);
180 TU *FromTU = findFromTU(From);
181 assert(LookupTablePtr);
182 return FromTU->import(*LookupTablePtr, ToAST.get(), From);
183}
184
185QualType ASTImporterTestBase::ImportType(QualType FromType, Decl *TUDecl,
186 Language ToLang) {
187 lazyInitToAST(ToLang, "", OutputFileName);
188 TU *FromTU = findFromTU(TUDecl);
189 assert(LookupTablePtr);
190 return FromTU->import(*LookupTablePtr, ToAST.get(), FromType);
191}

193ASTImporterTestBase::~ASTImporterTestBase() {
194 if (!::testing::Test::HasFailure())
195 return;
196
197 for (auto &Tu : FromTUs) {
198 assert(Tu.Unit);
199 llvm::errs() << "FromAST:\n";
200 Tu.Unit->getASTContext().getTranslationUnitDecl()->dump();
201 llvm::errs() << "\n";
202 }
203 if (ToAST) {
204 llvm::errs() << "ToAST:\n";
205 ToAST->getASTContext().getTranslationUnitDecl()->dump();
206 }
207}
