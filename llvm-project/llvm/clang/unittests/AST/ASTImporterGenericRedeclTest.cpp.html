68struct FunctionTemplateSpec {
69 using DeclTy = FunctionDecl;
70 static constexpr auto *Prototype =
71 R"(
72 // Proto of the primary template.
73 template <class T>
74 void X();
75 // Proto of the specialization.
76 template <>
77 void X<int>();
78 )";
79 static constexpr auto *Definition =
80 R"(
81 // Proto of the primary template.
82 template <class T>
83 void X();
84 // Specialization and definition.
85 template <>
86 void X<int>() {}
87 )";
88 BindableMatcher<Decl> getPattern() {
89 return functionDecl(hasName("X"), isExplicitTemplateSpecialization());
90 }

110template <typename TypeParam>
111struct RedeclChain : ASTImporterOptionSpecificTestBase {
112
113 using DeclTy = typename TypeParam::DeclTy;
114 std::string getPrototype() { return TypeParam::Prototype; }
115 std::string getDefinition() { return TypeParam::Definition; }
116 BindableMatcher<Decl> getPattern() const { return TypeParam().getPattern(); }
117
118 void CheckPreviousDecl(Decl *Prev, Decl *Current) {
119 ASSERT_NE(Prev, Current);
120 ASSERT_EQ(&Prev->getASTContext(), &Current->getASTContext());
121 EXPECT_EQ(Prev->getCanonicalDecl(), Current->getCanonicalDecl());
122
123 // Templates.
124 if (auto *PrevT = dyn_cast<TemplateDecl>(Prev)) {
125 EXPECT_EQ(Current->getPreviousDecl(), Prev);
126 auto *CurrentT = cast<TemplateDecl>(Current);
127 ASSERT_TRUE(PrevT->getTemplatedDecl());
128 ASSERT_TRUE(CurrentT->getTemplatedDecl());
129 EXPECT_EQ(CurrentT->getTemplatedDecl()->getPreviousDecl(),
130 PrevT->getTemplatedDecl());
131 return;
132 }
133
134 // Specializations.
135 if (auto *PrevF = dyn_cast<FunctionDecl>(Prev)) {
136 if (PrevF->getTemplatedKind() ==
137 FunctionDecl::TK_FunctionTemplateSpecialization) {
138 // There may be a hidden fwd spec decl before a spec decl.
139 // In that case the previous visible decl can be reached through that
140 // invisible one.
141 EXPECT_THAT(Prev, testing::AnyOf(
142 Current->getPreviousDecl(),
143 Current->getPreviousDecl()->getPreviousDecl()));
144 auto *ToTU = Prev->getTranslationUnitDecl();
145 auto *TemplateD = FirstDeclMatcher<FunctionTemplateDecl>().match(
146 ToTU, functionTemplateDecl());
147 auto *FirstSpecD = *(TemplateD->spec_begin());
148 EXPECT_EQ(FirstSpecD->getCanonicalDecl(), PrevF->getCanonicalDecl());
149 return;
150 }
151 }
152
153 // The rest: Classes, Functions, etc.
154 EXPECT_EQ(Current->getPreviousDecl(), Prev);
155 }

158 TypedTest_PrototypeShouldBeImportedAsAPrototypeWhenThereIsNoDefinition() {
159 Decl *FromTU = getTuDecl(getPrototype(), Lang_CXX);
160 auto *FromD = FirstDeclMatcher<DeclTy>().match(FromTU, getPattern());
161 ASSERT_FALSE(FromD->isThisDeclarationADefinition());
162
163 Decl *ImportedD = Import(FromD, Lang_CXX);
164 Decl *ToTU = ImportedD->getTranslationUnitDecl();
165
166 EXPECT_EQ(DeclCounter<DeclTy>().match(ToTU, getPattern()), 1u);
167 auto *ToD = LastDeclMatcher<DeclTy>().match(ToTU, getPattern());
168 EXPECT_TRUE(ImportedD == ToD);
169 EXPECT_FALSE(ToD->isThisDeclarationADefinition());
170 if (auto *ToT = dyn_cast<TemplateDecl>(ToD)) {
171 EXPECT_TRUE(ToT->getTemplatedDecl());
172 }
173 }
