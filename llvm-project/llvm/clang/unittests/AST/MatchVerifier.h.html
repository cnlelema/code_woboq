31/// \brief Base class for verifying some property of nodes found by a matcher.
32template <typename NodeType>
33class MatchVerifier : public MatchFinder::MatchCallback {
34public:
35 template <typename MatcherType>
36 testing::AssertionResult match(const std::string &Code,
37 const MatcherType &AMatcher) {
38 std::vector<std::string> Args;
39 return match(Code, AMatcher, Args, Lang_CXX);
40 }
41
42 template <typename MatcherType>
43 testing::AssertionResult match(const std::string &Code,
44 const MatcherType &AMatcher,
45 Language L) {
46 std::vector<std::string> Args;
47 return match(Code, AMatcher, Args, L);
48 }
49
50 template <typename MatcherType>
51 testing::AssertionResult match(const std::string &Code,
52 const MatcherType &AMatcher,
53 std::vector<std::string>& Args,
54 Language L);
55
56 template <typename MatcherType>
57 testing::AssertionResult match(const Decl *D, const MatcherType &AMatcher);
58
59protected:
60 void run(const MatchFinder::MatchResult &Result) override;
61 virtual void verify(const MatchFinder::MatchResult &Result,
62 const NodeType &Node) {}
63
64 void setFailure(const Twine &Result) {
65 Verified = false;
66 VerifyResult = Result.str();
67 }
68
69 void setSuccess() {
70 Verified = true;
71 }
72
73private:
74 bool Verified;
75 std::string VerifyResult;
76};

78/// \brief Runs a matcher over some code, and returns the result of the
79/// verifier for the matched node.
80template <typename NodeType> template <typename MatcherType>
81testing::AssertionResult MatchVerifier<NodeType>::match(
82 const std::string &Code, const MatcherType &AMatcher,
83 std::vector<std::string>& Args, Language L) {
84 MatchFinder Finder;
85 Finder.addMatcher(AMatcher.bind(""), this);
86 std::unique_ptr<tooling::FrontendActionFactory> Factory(
87 tooling::newFrontendActionFactory(&Finder));
88
89 StringRef FileName;
90 switch (L) {
91 case Lang_C:
92 Args.push_back("-std=c99");
93 FileName = "input.c";
94 break;
95 case Lang_C89:
96 Args.push_back("-std=c89");
97 FileName = "input.c";
98 break;
99 case Lang_CXX:
100 Args.push_back("-std=c++98");
101 FileName = "input.cc";
102 break;
103 case Lang_CXX11:
104 Args.push_back("-std=c++11");
105 FileName = "input.cc";
106 break;
107 case Lang_CXX14:
108 Args.push_back("-std=c++14");
109 FileName = "input.cc";
110 break;
111 case Lang_CXX2a:
112 Args.push_back("-std=c++2a");
113 FileName = "input.cc";
114 break;
115 case Lang_OpenCL:
116 FileName = "input.cl";
117 break;
118 case Lang_OBJCXX:
119 FileName = "input.mm";
120 break;
121 }
122
123 // Default to failure in case callback is never called
124 setFailure("Could not find match");
125 if (!tooling::runToolOnCodeWithArgs(Factory->create(), Code, Args, FileName))
126 return testing::AssertionFailure() << "Parsing error";
127 if (!Verified)
128 return testing::AssertionFailure() << VerifyResult;
129 return testing::AssertionSuccess();
130}

132/// \brief Runs a matcher over some AST, and returns the result of the
133/// verifier for the matched node.
134template <typename NodeType> template <typename MatcherType>
135testing::AssertionResult MatchVerifier<NodeType>::match(
136 const Decl *D, const MatcherType &AMatcher) {
137 MatchFinder Finder;
138 Finder.addMatcher(AMatcher.bind(""), this);
139
140 setFailure("Could not find match");
141 Finder.match(*D, D->getASTContext());
142
143 if (!Verified)
144 return testing::AssertionFailure() << VerifyResult;
145 return testing::AssertionSuccess();
146}

148template <typename NodeType>
149void MatchVerifier<NodeType>::run(const MatchFinder::MatchResult &Result) {
150 const NodeType *Node = Result.Nodes.getNodeAs<NodeType>("");
151 if (!Node) {
152 setFailure("Matched node has wrong type");
153 } else {
154 // Callback has been called, default to success.
155 setSuccess();
156 verify(Result, *Node);
157 }
158}
159
160template <>
161inline void MatchVerifier<ast_type_traits::DynTypedNode>::run(
162 const MatchFinder::MatchResult &Result) {
163 BoundNodes::IDToNodeMap M = Result.Nodes.getMap();
164 BoundNodes::IDToNodeMap::const_iterator I = M.find("");
165 if (I == M.end()) {
166 setFailure("Node was not bound");
167 } else {
168 // Callback has been called, default to success.
169 setSuccess();
170 verify(Result, I->second);
171 }
172}

174/// \brief Verify whether a node has the correct source location.
175///
176/// By default, Node.getSourceLocation() is checked. This can be changed
177/// by overriding getLocation().
178template <typename NodeType>
179class LocationVerifier : public MatchVerifier<NodeType> {
180public:
181 void expectLocation(unsigned Line, unsigned Column) {
182 ExpectLine = Line;
183 ExpectColumn = Column;
184 }
185
186protected:
187 void verify(const MatchFinder::MatchResult &Result,
188 const NodeType &Node) override {
189 SourceLocation Loc = getLocation(Node);
190 unsigned Line = Result.SourceManager->getSpellingLineNumber(Loc);
191 unsigned Column = Result.SourceManager->getSpellingColumnNumber(Loc);
192 if (Line != ExpectLine || Column != ExpectColumn) {
193 std::string MsgStr;
194 llvm::raw_string_ostream Msg(MsgStr);
195 Msg << "Expected location <" << ExpectLine << ":" << ExpectColumn
196 << ">, found <";
197 Loc.print(Msg, *Result.SourceManager);
198 Msg << '>';
199 this->setFailure(Msg.str());
200 }
201 }
202
203 virtual SourceLocation getLocation(const NodeType &Node) {
204 return Node.getLocation();
205 }
206
207private:
208 unsigned ExpectLine, ExpectColumn;
209};

211/// \brief Verify whether a node has the correct source range.
212///
213/// By default, Node.getSourceRange() is checked. This can be changed
214/// by overriding getRange().
215template <typename NodeType>
216class RangeVerifier : public MatchVerifier<NodeType> {
217public:
218 void expectRange(unsigned BeginLine, unsigned BeginColumn,
219 unsigned EndLine, unsigned EndColumn) {
220 ExpectBeginLine = BeginLine;
221 ExpectBeginColumn = BeginColumn;
222 ExpectEndLine = EndLine;
223 ExpectEndColumn = EndColumn;
224 }
225
226protected:
227 void verify(const MatchFinder::MatchResult &Result,
228 const NodeType &Node) override {
229 SourceRange R = getRange(Node);
230 SourceLocation Begin = R.getBegin();
231 SourceLocation End = R.getEnd();
232 unsigned BeginLine = Result.SourceManager->getSpellingLineNumber(Begin);
233 unsigned BeginColumn = Result.SourceManager->getSpellingColumnNumber(Begin);
234 unsigned EndLine = Result.SourceManager->getSpellingLineNumber(End);
235 unsigned EndColumn = Result.SourceManager->getSpellingColumnNumber(End);
236 if (BeginLine != ExpectBeginLine || BeginColumn != ExpectBeginColumn ||
237 EndLine != ExpectEndLine || EndColumn != ExpectEndColumn) {
238 std::string MsgStr;
239 llvm::raw_string_ostream Msg(MsgStr);
240 Msg << "Expected range <" << ExpectBeginLine << ":" << ExpectBeginColumn
241 << '-' << ExpectEndLine << ":" << ExpectEndColumn << ">, found <";
242 Begin.print(Msg, *Result.SourceManager);
243 Msg << '-';
244 End.print(Msg, *Result.SourceManager);
245 Msg << '>';
246 this->setFailure(Msg.str());
247 }
248 }
249
250 virtual SourceRange getRange(const NodeType &Node) {
251 return Node.getSourceRange();
252 }
253
254private:
255 unsigned ExpectBeginLine, ExpectBeginColumn, ExpectEndLine, ExpectEndColumn;
256};

258/// \brief Verify whether a node's dump contains a given substring.
259class DumpVerifier : public MatchVerifier<ast_type_traits::DynTypedNode> {
260public:
261 void expectSubstring(const std::string &Str) {
262 ExpectSubstring = Str;
263 }
264
265protected:
266 void verify(const MatchFinder::MatchResult &Result,
267 const ast_type_traits::DynTypedNode &Node) override {
268 std::string DumpStr;
269 llvm::raw_string_ostream Dump(DumpStr);
270 Node.dump(Dump, *Result.SourceManager);
271
272 if (Dump.str().find(ExpectSubstring) == std::string::npos) {
273 std::string MsgStr;
274 llvm::raw_string_ostream Msg(MsgStr);
275 Msg << "Expected dump substring <" << ExpectSubstring << ">, found <"
276 << Dump.str() << '>';
277 this->setFailure(Msg.str());
278 }
279 }
280
281private:
282 std::string ExpectSubstring;
283};

285/// \brief Verify whether a node's pretty print matches a given string.
286class PrintVerifier : public MatchVerifier<ast_type_traits::DynTypedNode> {
287public:
288 void expectString(const std::string &Str) {
289 ExpectString = Str;
290 }
291
292protected:
293 void verify(const MatchFinder::MatchResult &Result,
294 const ast_type_traits::DynTypedNode &Node) override {
295 std::string PrintStr;
296 llvm::raw_string_ostream Print(PrintStr);
297 Node.print(Print, Result.Context->getPrintingPolicy());
298
299 if (Print.str() != ExpectString) {
300 std::string MsgStr;
301 llvm::raw_string_ostream Msg(MsgStr);
302 Msg << "Expected pretty print <" << ExpectString << ">, found <"
303 << Print.str() << '>';
304 this->setFailure(Msg.str());
305 }
306 }
307
308private:
309 std::string ExpectString;
310};
