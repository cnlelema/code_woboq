29// In tests, we can't directly match on llvm::Expected since its accessors
30// mutate the object. So, we collapse it to an Optional.
31static llvm::Optional<std::string> toOptional(llvm::Expected<std::string> V) {
32 if (V)
33 return *V;
34 ADD_FAILURE() << "Losing error in conversion to IsSomething: "
35 << llvm::toString(V.takeError());
36 return llvm::None;
37}
38
39// A very simple matcher for llvm::Optional values.
40MATCHER_P(IsSomething, ValueMatcher, "") {
41 if (!arg)
42 return false;
43 return ::testing::ExplainMatchResult(ValueMatcher, *arg, result_listener);
44}

46// Create a valid translation-unit from a statement.
47static std::string wrapSnippet(llvm::Twine StatementCode) {
48 return ("auto stencil_test_snippet = []{" + StatementCode + "};").str();
49}
50
51static DeclarationMatcher wrapMatcher(const StatementMatcher &Matcher) {
52 return varDecl(hasName("stencil_test_snippet"),
53 hasDescendant(compoundStmt(hasAnySubstatement(Matcher))));
54}
55
56struct TestMatch {
57 // The AST unit from which `result` is built. We bundle it because it backs
58 // the result. Users are not expected to access it.
59 std::unique_ptr<ASTUnit> AstUnit;
60 // The result to use in the test. References `ast_unit`.
61 MatchResult Result;
62};
63
64// Matches `Matcher` against the statement `StatementCode` and returns the
65// result. Handles putting the statement inside a function and modifying the
66// matcher correspondingly. `Matcher` should match `StatementCode` exactly --
67// that is, produce exactly one match.
68static llvm::Optional<TestMatch> matchStmt(llvm::Twine StatementCode,
69 StatementMatcher Matcher) {
70 auto AstUnit = buildASTFromCode(wrapSnippet(StatementCode));
71 if (AstUnit == nullptr) {
72 ADD_FAILURE() << "AST construction failed";
73 return llvm::None;
74 }
75 ASTContext &Context = AstUnit->getASTContext();
76 auto Matches = ast_matchers::match(wrapMatcher(Matcher), Context);
77 // We expect a single, exact match for the statement.
78 if (Matches.size() != 1) {
79 ADD_FAILURE() << "Wrong number of matches: " << Matches.size();
80 return llvm::None;
81 }
82 return TestMatch{std::move(AstUnit), MatchResult(Matches[0], &Context)};
83}

85class StencilTest : public ::testing::Test {
86protected:
87 // Verifies that the given stencil fails when evaluated on a valid match
88 // result. Binds a statement to "stmt", a (non-member) ctor-initializer to
89 // "init", an expression to "expr" and a (nameless) declaration to "decl".
90 void testError(const Stencil &Stencil,
91 ::testing::Matcher<std::string> Matcher) {
92 const std::string Snippet = R"cc(
93 struct A {};
94 class F : public A {
95 public:
96 F(int) {}
97 };
98 F(1);
99 )cc";
100 auto StmtMatch = matchStmt(
101 Snippet,
102 stmt(hasDescendant(
103 cxxConstructExpr(
104 hasDeclaration(decl(hasDescendant(cxxCtorInitializer(
105 isBaseInitializer())
106 .bind("init")))
107 .bind("decl")))
108 .bind("expr")))
109 .bind("stmt"));
110 ASSERT_TRUE(StmtMatch);
111 if (auto ResultOrErr = Stencil.eval(StmtMatch->Result)) {
112 ADD_FAILURE() << "Expected failure but succeeded: " << *ResultOrErr;
113 } else {
114 auto Err = llvm::handleErrors(ResultOrErr.takeError(),
115 [&Matcher](const llvm::StringError &Err) {
116 EXPECT_THAT(Err.getMessage(), Matcher);
117 });
118 if (Err) {
119 ADD_FAILURE() << "Unhandled error: " << llvm::toString(std::move(Err));
120 }
121 }
122 }
123
124 // Tests failures caused by references to unbound nodes. `unbound_id` is the
125 // id that will cause the failure.
126 void testUnboundNodeError(const Stencil &Stencil, llvm::StringRef UnboundId) {
127 testError(Stencil, AllOf(HasSubstr(UnboundId), HasSubstr("not bound")));
128 }
129};

131TEST_F(StencilTest, SingleStatement) {
132 StringRef Condition("C"), Then("T"), Else("E");
133 const std::string Snippet = R"cc(
134 if (true)
135 return 1;
136 else
137 return 0;
138 )cc";
139 auto StmtMatch = matchStmt(
140 Snippet, ifStmt(hasCondition(expr().bind(Condition)),
141 hasThen(stmt().bind(Then)), hasElse(stmt().bind(Else))));
142 ASSERT_TRUE(StmtMatch);
143 // Invert the if-then-else.
144 auto Stencil = cat("if (!", node(Condition), ") ", statement(Else), " else ",
145 statement(Then));
146 EXPECT_THAT(toOptional(Stencil.eval(StmtMatch->Result)),
147 IsSomething(Eq("if (!true) return 0; else return 1;")));
148}

169TEST_F(StencilTest, UnboundNode) {
170 const std::string Snippet = R"cc(
171 if (true)
172 return 1;
173 else
174 return 0;
175 )cc";
176 auto StmtMatch = matchStmt(Snippet, ifStmt(hasCondition(stmt().bind("a1")),
177 hasThen(stmt().bind("a2"))));
178 ASSERT_TRUE(StmtMatch);
179 auto Stencil = cat("if(!", node("a1"), ") ", node("UNBOUND"), ";");
180 auto ResultOrErr = Stencil.eval(StmtMatch->Result);
181 EXPECT_TRUE(llvm::errorToBool(ResultOrErr.takeError()))
182 << "Expected unbound node, got " << *ResultOrErr;
183}

185// Tests that a stencil with a single parameter (`Id`) evaluates to the expected
186// string, when `Id` is bound to the expression-statement in `Snippet`.
187void testExpr(StringRef Id, StringRef Snippet, const Stencil &Stencil,
188 StringRef Expected) {
189 auto StmtMatch = matchStmt(Snippet, expr().bind(Id));
190 ASSERT_TRUE(StmtMatch);
191 EXPECT_THAT(toOptional(Stencil.eval(StmtMatch->Result)),
192 IsSomething(Expected));
193}

195TEST_F(StencilTest, SelectionOp) {
196 StringRef Id = "id";
197 testExpr(Id, "3;", cat(node(Id)), "3");
198}
